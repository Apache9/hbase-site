045/**
046 * Used to perform Put operations for a single row.
047 * <p>
048 * To perform a Put, instantiate a Put object with the row to insert to, and
049 * for each column to be inserted, execute {@link #addColumn(byte[], byte[],
050 * byte[]) add} or {@link #addColumn(byte[], byte[], long, byte[]) add} if
051 * setting the timestamp.
052 */
053@InterfaceAudience.Public
054@InterfaceStability.Stable
055public class Put extends Mutation implements HeapSize, Comparable<Row> {
056  /**
057   * Create a Put operation for the specified row.
058   * @param row row key
059   */
060  public Put(byte [] row) {
061    this(row, HConstants.LATEST_TIMESTAMP);
062  }
063
064  /**
065   * Create a Put operation for the specified row, using a given timestamp.
066   *
067   * @param row row key; we make a copy of what we are passed to keep local.
068   * @param ts timestamp
069   */
070  public Put(byte[] row, long ts) {
071    this(row, 0, row.length, ts);
072  }

074  /**

081    this(rowArray, rowOffset, rowLength, HConstants.LATEST_TIMESTAMP);

084  /**
085   * @param row row key; we make a copy of what we are passed to keep local.
086   * @param ts  timestamp
087   */
088  public Put(ByteBuffer row, long ts) {
089    if (ts < 0) {
090      throw new IllegalArgumentException("Timestamp cannot be negative. ts=" + ts);
091    }
092    checkRow(row);
093    this.row = new byte[row.remaining()];
094    row.get(this.row);
095    this.ts = ts;
096  }

098  /**
099   * @param row row key; we make a copy of what we are passed to keep local.
100   */
101  public Put(ByteBuffer row) {
102    this(row, HConstants.LATEST_TIMESTAMP);
103  }

105  /**
106   * We make a copy of the passed in row key to keep local.
107   * @param rowArray
108   * @param rowOffset
109   * @param rowLength
110   * @param ts
111   */
112  public Put(byte [] rowArray, int rowOffset, int rowLength, long ts) {
113    checkRow(rowArray, rowOffset, rowLength);
114    this.row = Bytes.copy(rowArray, rowOffset, rowLength);
115    this.ts = ts;
116    if (ts < 0) {
117      throw new IllegalArgumentException("Timestamp cannot be negative. ts=" + ts);
118    }
119  }

121  /**
122   * Create a Put operation for an immutable row key.
123   *
124   * @param row row key
125   * @param rowIsImmutable whether the input row is immutable.
126   *                       Set to true if the caller can guarantee that
127   *                       the row will not be changed for the Put duration.
128   */
129  public Put(byte [] row, boolean rowIsImmutable) {
130    this(row, HConstants.LATEST_TIMESTAMP, rowIsImmutable);
131  }
132
133  /**
134   * Create a Put operation for an immutable row key, using a given timestamp.
135   *
136   * @param row row key
137   * @param ts timestamp
138   * @param rowIsImmutable whether the input row is immutable.
139   *                       Set to true if the caller can guarantee that
140   *                       the row will not be changed for the Put duration.
141   */
142  public Put(byte[] row, long ts, boolean rowIsImmutable) {
143    // Check and set timestamp
144    if (ts < 0) {
145      throw new IllegalArgumentException("Timestamp cannot be negative. ts=" + ts);
146    }
147    this.ts = ts;
148
149    // Deal with row according to rowIsImmutable
150    checkRow(row);
151    if (rowIsImmutable) {  // Row is immutable
152      this.row = row;  // Do not make a local copy, but point to the provided byte array directly
153    } else {  // Row is not immutable
154      this.row = Bytes.copy(row, 0, row.length);  // Make a local copy
155    }
156  }

158  /**
159   * Copy constructor.  Creates a Put operation cloned from the specified Put.
160   * @param putToCopy put to copy
161   */
162  public Put(Put putToCopy) {
163    this(putToCopy.getRow(), putToCopy.ts);
164    this.familyMap = new TreeMap<byte [], List<Cell>>(Bytes.BYTES_COMPARATOR);
165    for(Map.Entry<byte [], List<Cell>> entry: putToCopy.getFamilyCellMap().entrySet()) {
166      this.familyMap.put(entry.getKey(), new ArrayList<Cell>(entry.getValue()));
167    }
168    this.durability = putToCopy.durability;
169    for (Map.Entry<String, byte[]> entry : putToCopy.getAttributesMap().entrySet()) {
170      this.setAttribute(entry.getKey(), entry.getValue());
171    }
172  }

205  /**
206   * Add the specified column and value, with the specified timestamp as
207   * its version to this Put operation.
208   * @param family family name
209   * @param qualifier column qualifier
210   * @param ts version timestamp
211   * @param value column value
212   * @return this
213   */
214  public Put addColumn(byte [] family, byte [] qualifier, long ts, byte [] value) {
215    if (ts < 0) {
216      throw new IllegalArgumentException("Timestamp cannot be negative. ts=" + ts);
217    }
218    List<Cell> list = getCellList(family);
219    KeyValue kv = createPutKeyValue(family, qualifier, ts, value);
220    list.add(kv);
221    familyMap.put(CellUtil.cloneFamily(kv), list);
222    return this;
223  }
224
225  /**
226   * See {@link #addColumn(byte[], byte[], long, byte[])}. This version expects
227   * that the underlying arrays won't change. It's intended
228   * for usage internal HBase to and for advanced client applications.
229   */
230  public Put addImmutable(byte [] family, byte [] qualifier, long ts, byte [] value) {
231    // Family can not be null, otherwise NullPointerException is thrown when putting the cell into familyMap
232    if (family == null) {
233      throw new IllegalArgumentException("Family cannot be null");
234    }
235
236    // Check timestamp
237    if (ts < 0) {
238      throw new IllegalArgumentException("Timestamp cannot be negative. ts=" + ts);
239    }
240
241    List<Cell> list = getCellList(family);
242    list.add(new IndividualBytesFieldCell(this.row, family, qualifier, ts, KeyValue.Type.Put, value));
243    familyMap.put(family, list);
244    return this;
245  }

318  /**
319   * Add the specified KeyValue to this Put operation.  Operation assumes that
320   * the passed KeyValue is immutable and its backing array will not be modified
321   * for the duration of this Put.
322   * @param kv individual KeyValue
323   * @return this
324   * @throws java.io.IOException e
325   */
326  public Put add(Cell kv) throws IOException{
327    byte [] family = CellUtil.cloneFamily(kv);
328    List<Cell> list = getCellList(family);
329    //Checking that the row of the kv is the same as the put
330    if (!CellUtil.matchingRow(kv, this.row)) {
331      throw new WrongRowIOException("The row in " + kv.toString() +
332        " doesn't match the original one " +  Bytes.toStringBinary(this.row));
333    }
334    list.add(kv);
335    familyMap.put(family, list);
336    return this;
337  }

399  /*
400   * Private method to determine if this object's familyMap contains
401   * the given value assigned to the given family, qualifier and timestamp
402   * respecting the 2 boolean arguments
403   *
404   * @param family
405   * @param qualifier
406   * @param ts
407   * @param value
408   * @param ignoreTS
409   * @param ignoreValue
410   * @return returns true if the given family, qualifier timestamp and value
411   * already has an existing KeyValue object in the family map.
412   */
413  private boolean has(byte[] family, byte[] qualifier, long ts, byte[] value,
414                      boolean ignoreTS, boolean ignoreValue) {
415    List<Cell> list = getCellList(family);
416    if (list.isEmpty()) {
417      return false;
418    }
419    // Boolean analysis of ignoreTS/ignoreValue.
420    // T T => 2
421    // T F => 3 (first is always true)
422    // F T => 2
423    // F F => 1
424    if (!ignoreTS && !ignoreValue) {
425      for (Cell cell : list) {
426        if (CellUtil.matchingFamily(cell, family) &&
427            CellUtil.matchingQualifier(cell, qualifier)  &&
428            CellUtil.matchingValue(cell, value) &&
429            cell.getTimestamp() == ts) {
430          return true;
431        }
432      }
433    } else if (ignoreValue && !ignoreTS) {
434      for (Cell cell : list) {
435        if (CellUtil.matchingFamily(cell, family) && CellUtil.matchingQualifier(cell, qualifier)
436            && cell.getTimestamp() == ts) {
437          return true;
438        }
439      }
440    } else if (!ignoreValue && ignoreTS) {
441      for (Cell cell : list) {
442        if (CellUtil.matchingFamily(cell, family) && CellUtil.matchingQualifier(cell, qualifier)
443            && CellUtil.matchingValue(cell, value)) {
444          return true;
445        }
446      }
447    } else {
448      for (Cell cell : list) {
449        if (CellUtil.matchingFamily(cell, family) &&
450            CellUtil.matchingQualifier(cell, qualifier)) {
451          return true;
452        }
453      }
454    }
455    return false;
456  }

458  /**
459   * Returns a list of all KeyValue objects with matching column family and qualifier.
460   *
461   * @param family column family
462   * @param qualifier column qualifier
463   * @return a list of KeyValue objects with the matching family and qualifier,
464   * returns an empty list if one doesn't exist for the given family.
465   */
466  public List<Cell> get(byte[] family, byte[] qualifier) {
467    List<Cell> filteredList = new ArrayList<Cell>();
468    for (Cell cell: getCellList(family)) {
469      if (CellUtil.matchingQualifier(cell, qualifier)) {
470        filteredList.add(cell);
471      }
472    }
473    return filteredList;
474  }
