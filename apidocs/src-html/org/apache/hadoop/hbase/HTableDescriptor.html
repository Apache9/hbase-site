359  /**
360   * Construct a table descriptor by cloning the descriptor passed as a parameter
361   * but using a different table name.
362   * <p>
363   * Makes a deep copy of the supplied descriptor.
364   * Can make a modifiable descriptor from an UnmodifyableHTableDescriptor.
365   * @param name Table name.
366   * @param desc The descriptor.
367   */
368  public HTableDescriptor(final TableName name, final HTableDescriptor desc) {
369    super();
370    setName(name);
371    setMetaFlags(this.name);
372    for (HColumnDescriptor c: desc.families.values()) {
373      this.families.put(c.getName(), new HColumnDescriptor(c));
374    }
375    for (Map.Entry<Bytes, Bytes> e :
376        desc.values.entrySet()) {
377      setValue(e.getKey(), e.getValue());
378    }
379    for (Map.Entry<String, String> e : desc.configuration.entrySet()) {
380      this.configuration.put(e.getKey(), e.getValue());
381    }
382  }

434  private synchronized Boolean calculateIsMetaRegion() {
435    byte [] value = getValue(IS_META_KEY);
436    return (value != null)? Boolean.valueOf(Bytes.toString(value)): Boolean.FALSE;
437  }

441    byte [] value = getValue(key);

445    return valueIfNull;

535   * Setter for storing metadata as a (key, value) pair in {@link #values} map
536   *
537   * @param key The key.
538   * @param value The value.
539   */
540  public HTableDescriptor setValue(final Bytes key, final Bytes value) {
541    if (key.compareTo(DEFERRED_LOG_FLUSH_KEY) == 0) {
542      boolean isDeferredFlush = Boolean.valueOf(Bytes.toString(value.get()));
543      LOG.warn("HTableDescriptor property:" + DEFERRED_LOG_FLUSH + " is deprecated, " +
544          "use " + DURABILITY + " instead");
545      setDurability(isDeferredFlush ? Durability.ASYNC_WAL : DEFAULT_DURABLITY);
546      return this;
547    }
548    values.put(key, value);
549    return this;
550  }

670  /**
671   * Returns the durability setting for the table.
672   * @return durability setting for the table.
673   */
674  public Durability getDurability() {
675    if (this.durability == null) {
676      byte[] durabilityValue = getValue(DURABILITY_KEY);
677      if (durabilityValue == null) {
678        this.durability = DEFAULT_DURABLITY;
679      } else {
680        try {
681          this.durability = Durability.valueOf(Bytes.toString(durabilityValue));
682        } catch (IllegalArgumentException ex) {

684            + " is not known. Durability:" + Bytes.toString(durabilityValue));
685          this.durability = DEFAULT_DURABLITY;
686        }
687      }
688    }
689    return this.durability;

927  private StringBuilder getValues(boolean printDefaults) {
928    StringBuilder s = new StringBuilder();
929
930    // step 1: set partitioning and pruning
931    Set<Bytes> reservedKeys = new TreeSet<>();
932    Set<Bytes> userKeys = new TreeSet<>();
933    for (Map.Entry<Bytes, Bytes> entry : values.entrySet()) {
934      if (entry.getKey() == null || entry.getKey().get() == null) continue;
935      String key = Bytes.toString(entry.getKey().get());
936      // in this section, print out reserved keywords + coprocessor info
937      if (!RESERVED_KEYWORDS.contains(entry.getKey()) && !key.startsWith("coprocessor$")) {
938        userKeys.add(entry.getKey());
939        continue;
940      }
941      // only print out IS_ROOT/IS_META if true
942      String value = Bytes.toString(entry.getValue().get());
943      if (key.equalsIgnoreCase(IS_ROOT) || key.equalsIgnoreCase(IS_META)) {
944        if (Boolean.valueOf(value) == false) continue;
945      }
946      // see if a reserved key is a default value. may not want to print it out
947      if (printDefaults
948          || !DEFAULT_VALUES.containsKey(key)
949          || !DEFAULT_VALUES.get(key).equalsIgnoreCase(value)) {
950        reservedKeys.add(entry.getKey());
951      }
952    }
953
954    // early exit optimization
955    boolean hasAttributes = !reservedKeys.isEmpty() || !userKeys.isEmpty();
956    if (!hasAttributes && configuration.isEmpty()) return s;
957
958    s.append(", {");
959    // step 2: printing attributes
960    if (hasAttributes) {
961      s.append("TABLE_ATTRIBUTES => {");
962
963      // print all reserved keys first
964      boolean printCommaForAttr = false;
965      for (Bytes k : reservedKeys) {
966        String key = Bytes.toString(k.get());
967        String value = Bytes.toStringBinary(values.get(k).get());
968        if (printCommaForAttr) s.append(", ");
969        printCommaForAttr = true;
970        s.append(key);
971        s.append(" => ");
972        s.append('\'').append(value).append('\'');
973      }
974
975      if (!userKeys.isEmpty()) {
976        // print all non-reserved, advanced config keys as a separate subset
977        if (printCommaForAttr) s.append(", ");
978        printCommaForAttr = true;
979        s.append(HConstants.METADATA).append(" => ");
980        s.append("{");
981        boolean printCommaForCfg = false;
982        for (Bytes k : userKeys) {
983          String key = Bytes.toString(k.get());
984          String value = Bytes.toStringBinary(values.get(k).get());
985          if (printCommaForCfg) s.append(", ");
986          printCommaForCfg = true;
987          s.append('\'').append(key).append('\'');
988          s.append(" => ");
989          s.append('\'').append(value).append('\'');
990        }
991        s.append("}");
992      }
993    }
994
995    // step 3: printing all configuration:
996    if (!configuration.isEmpty()) {
997      if (hasAttributes) {
998        s.append(", ");
999      }
1000      s.append(HConstants.CONFIGURATION).append(" => ");
1001      s.append('{');
1002      boolean printCommaForConfig = false;
1003      for (Map.Entry<String, String> e : configuration.entrySet()) {
1004        if (printCommaForConfig) s.append(", ");
1005        printCommaForConfig = true;
1006        s.append('\'').append(e.getKey()).append('\'');
1007        s.append(" => ");
1008        s.append('\'').append(e.getValue()).append('\'');
1009      }
1010      s.append("}");
1011    }
1012    s.append("}"); // end METHOD
1013    return s;
1014  }

1057  /**
1058   * Compares the descriptor with another descriptor which is passed as a parameter.
1059   * This compares the content of the two descriptors and not the reference.
1060   *
1061   * @return 0 if the contents of the descriptors are exactly matching,
1062   *         1 if there is a mismatch in the contents
1063   */
1064  @Override
1065  public int compareTo(final HTableDescriptor other) {
1066    int result = this.name.compareTo(other.name);
1067    if (result == 0) {
1068      result = families.size() - other.families.size();
1069    }
1070    if (result == 0 && families.size() != other.families.size()) {
1071      result = Integer.valueOf(families.size()).compareTo(
1072          Integer.valueOf(other.families.size()));
1073    }
1074    if (result == 0) {
1075      for (Iterator<HColumnDescriptor> it = families.values().iterator(),
1076          it2 = other.families.values().iterator(); it.hasNext(); ) {
1077        result = it.next().compareTo(it2.next());
1078        if (result != 0) {
1079          break;
1080        }
1081      }
1082    }
1083    if (result == 0) {
1084      // punt on comparison for ordering, just calculate difference
1085      result = this.values.hashCode() - other.values.hashCode();
1086      if (result < 0)
1087        result = -1;
1088      else if (result > 0)
1089        result = 1;
1090    }
1091    if (result == 0) {
1092      result = this.configuration.hashCode() - other.configuration.hashCode();
1093      if (result < 0)
1094        result = -1;
1095      else if (result > 0)
1096        result = 1;
1097    }
1098    return result;
1099  }

1261   * or Endpoint.
1262   * It won't check if the class can be loaded or not.
1263   * Whether a coprocessor is loadable or not will be determined when
1264   * a region is opened.
1265   * @param jarFilePath Path of the jar file. If it's null, the class will be
1266   * loaded from default classloader.
1267   * @param className Full class name.
1268   * @param priority Priority
1269   * @param kvs Arbitrary key-value parameter pairs passed into the coprocessor.
1270   * @throws IOException
1271   */
1272  public HTableDescriptor addCoprocessor(String className, Path jarFilePath,
1273                             int priority, final Map<String, String> kvs)
1274  throws IOException {
1275    checkHasCoprocessor(className);
1276
1277    // Validate parameter kvs and then add key/values to kvString.
1278    StringBuilder kvString = new StringBuilder();
1279    if (kvs != null) {
1280      for (Map.Entry<String, String> e: kvs.entrySet()) {
1281        if (!e.getKey().matches(HConstants.CP_HTD_ATTR_VALUE_PARAM_KEY_PATTERN)) {
1282          throw new IOException("Illegal parameter key = " + e.getKey());
1283        }
1284        if (!e.getValue().matches(HConstants.CP_HTD_ATTR_VALUE_PARAM_VALUE_PATTERN)) {
1285          throw new IOException("Illegal parameter (" + e.getKey() +
1286              ") value = " + e.getValue());
1287        }
1288        if (kvString.length() != 0) {
1289          kvString.append(',');
1290        }
1291        kvString.append(e.getKey());
1292        kvString.append('=');
1293        kvString.append(e.getValue());
1294      }
1295    }
1296
1297    String value = ((jarFilePath == null)? "" : jarFilePath.toString()) +
1298        "|" + className + "|" + Integer.toString(priority) + "|" +
1299        kvString.toString();
1300    return addCoprocessorToMap(value);
1301  }

1330  /**
1331   * Add coprocessor to values Map
1332   * @param specStr The Coprocessor specification all in in one String formatted so matches
1333   * {@link HConstants#CP_HTD_ATTR_VALUE_PATTERN}
1334   * @return Returns <code>this</code>
1335   */
1336  private HTableDescriptor addCoprocessorToMap(final String specStr) {
1337    if (specStr == null) return this;
1338    // generate a coprocessor key
1339    int maxCoprocessorNumber = 0;
1340    Matcher keyMatcher;
1341    for (Map.Entry<Bytes, Bytes> e: this.values.entrySet()) {
1342      keyMatcher = HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(Bytes.toString(e.getKey().get()));
1343      if (!keyMatcher.matches()) {
1344        continue;
1345      }
1346      maxCoprocessorNumber = Math.max(Integer.parseInt(keyMatcher.group(1)), maxCoprocessorNumber);
1347    }
1348    maxCoprocessorNumber++;
1349    String key = "coprocessor$" + Integer.toString(maxCoprocessorNumber);
1350    this.values.put(new Bytes(Bytes.toBytes(key)), new Bytes(Bytes.toBytes(specStr)));
1351    return this;
1352  }

1354  /**
1355   * Check if the table has an attached co-processor represented by the name className
1356   *
1357   * @param classNameToMatch - Class name of the co-processor
1358   * @return true of the table has a co-processor className
1359   */
1360  public boolean hasCoprocessor(String classNameToMatch) {
1361    Matcher keyMatcher;
1362    for (Map.Entry<Bytes, Bytes> e :
1363        this.values.entrySet()) {
1364      keyMatcher =
1365          HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(
1366              Bytes.toString(e.getKey().get()));
1367      if (!keyMatcher.matches()) {
1368        continue;
1369      }
1370      String className = getCoprocessorClassNameFromSpecStr(Bytes.toString(e.getValue().get()));
1371      if (className == null) continue;
1372      if (className.equals(classNameToMatch.trim())) {
1373        return true;
1374      }
1375    }
1376    return false;
1377  }

1379  /**
1380   * Return the list of attached co-processor represented by their name className
1381   *
1382   * @return The list of co-processors classNames
1383   */
1384  public List<String> getCoprocessors() {
1385    List<String> result = new ArrayList<>(this.values.entrySet().size());
1386    Matcher keyMatcher;
1387    for (Map.Entry<Bytes, Bytes> e : this.values.entrySet()) {
1388      keyMatcher = HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(Bytes.toString(e.getKey().get()));
1389      if (!keyMatcher.matches()) {
1390        continue;
1391      }
1392      String className = getCoprocessorClassNameFromSpecStr(Bytes.toString(e.getValue().get()));
1393      if (className == null) continue;
1394      result.add(className); // classname is the 2nd field
1395    }
1396    return result;
1397  }

1399  /**
1400   * @param spec String formatted as per {@link HConstants#CP_HTD_ATTR_VALUE_PATTERN}
1401   * @return Class parsed from passed in <code>spec</code> or null if no match or classpath found
1402   */
1403  private static String getCoprocessorClassNameFromSpecStr(final String spec) {
1404    Matcher matcher = HConstants.CP_HTD_ATTR_VALUE_PATTERN.matcher(spec);
1405    // Classname is the 2nd field
1406    return matcher != null && matcher.matches()? matcher.group(2).trim(): null;
1407  }

1409  /**
1410   * Remove a coprocessor from those set on the table
1411   * @param className Class name of the co-processor
1412   */
1413  public void removeCoprocessor(String className) {
1414    Bytes match = null;
1415    Matcher keyMatcher;
1416    Matcher valueMatcher;
1417    for (Map.Entry<Bytes, Bytes> e : this.values
1418        .entrySet()) {
1419      keyMatcher = HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(Bytes.toString(e
1420          .getKey().get()));
1421      if (!keyMatcher.matches()) {
1422        continue;
1423      }
1424      valueMatcher = HConstants.CP_HTD_ATTR_VALUE_PATTERN.matcher(Bytes
1425          .toString(e.getValue().get()));
1426      if (!valueMatcher.matches()) {
1427        continue;
1428      }
1429      // get className and compare
1430      String clazz = valueMatcher.group(2).trim(); // classname is the 2nd field
1431      // remove the CP if it is present
1432      if (clazz.equals(className.trim())) {
1433        match = e.getKey();
1434        break;
1435      }
1436    }
1437    // if we found a match, remove it
1438    if (match != null)
1439      remove(match);
1440  }

1515  /**
1516   * @param bytes A pb serialized {@link HTableDescriptor} instance with pb magic prefix
1517   * @return An instance of {@link HTableDescriptor} made from <code>bytes</code>
1518   * @throws DeserializationException
1519   * @throws IOException
1520   * @see #toByteArray()
1521   */
1522  public static HTableDescriptor parseFrom(final byte [] bytes)
1523  throws DeserializationException, IOException {
1524    if (!ProtobufUtil.isPBMagicPrefix(bytes)) {
1525      throw new DeserializationException("Expected PB encoded HTableDescriptor");
1526    }
1527    int pblen = ProtobufUtil.lengthOfPBMagic();
1528    TableSchema.Builder builder = TableSchema.newBuilder();
1529    TableSchema ts;
1530    try {
1531      ProtobufUtil.mergeFrom(builder, bytes, pblen, bytes.length - pblen);
1532      ts = builder.build();
1533    } catch (IOException e) {
1534      throw new DeserializationException(e);
1535    }
1536    return ProtobufUtil.convertToHTableDesc(ts);
1537  }
