38   * Utility to help ipc'ing.

40  @InterfaceAudience.Private

43    /**
44     * Write out header, param, and cell block if there is one.
45     * @param dos Stream to write into
46     * @param header to write
47     * @param param to write
48     * @param cellBlock to write
49     * @return Total number of bytes written.
50     * @throws IOException if write action fails
51     */
52    public static int write(final OutputStream dos, final Message header, final Message param,
53        final ByteBuffer cellBlock) throws IOException {
54      // Must calculate total size and write that first so other side can read it all in in one
55      // swoop. This is dictated by how the server is currently written. Server needs to change
56      // if we are to be able to write without the length prefixing.
57      int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(header, param);
58      if (cellBlock != null) {
59        totalSize += cellBlock.remaining();
60      }
61      return write(dos, header, param, cellBlock, totalSize);
62    }
63
64    private static int write(final OutputStream dos, final Message header, final Message param,
65        final ByteBuffer cellBlock, final int totalSize) throws IOException {
66      // I confirmed toBytes does same as DataOutputStream#writeInt.
67      dos.write(Bytes.toBytes(totalSize));
68      // This allocates a buffer that is the size of the message internally.
69      header.writeDelimitedTo(dos);
70      if (param != null) {
71        param.writeDelimitedTo(dos);
72      }
73      if (cellBlock != null) {
74        dos.write(cellBlock.array(), 0, cellBlock.remaining());
75      }
76      dos.flush();
77      return totalSize;
78    }
79
80    /**
81     * @return Size on the wire when the two messages are written with writeDelimitedTo
82     */
83    public static int getTotalSizeWhenWrittenDelimited(Message... messages) {
84      int totalSize = 0;
85      for (Message m : messages) {
86        if (m == null) {
87          continue;
88        }
89        totalSize += m.getSerializedSize();
90        totalSize += CodedOutputStream.computeRawVarint32Size(m.getSerializedSize());
91      }
92      Preconditions.checkArgument(totalSize < Integer.MAX_VALUE);
93      return totalSize;
94    }

103   /**
104    * @param e exception to be wrapped
105    * @return RemoteException made from passed <code>e</code>
106    */
107   public static RemoteException createRemoteException(final ExceptionResponse e) {
108     String innerExceptionClassName = e.getExceptionClassName();
109     boolean doNotRetry = e.getDoNotRetry();
110     return e.hasHostname() ?
111     // If a hostname then add it to the RemoteWithExtrasException
112         new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), e.getHostname(),
113             e.getPort(), doNotRetry)
114         : new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), doNotRetry);
115   }

117   /**
118    * Takes an Exception and the address we were trying to connect to and return an IOException with
119    * the input exception as the cause. The new exception provides the stack trace of the place where
120    * the exception is thrown and some extra diagnostics information. If the exception is
121    * ConnectException or SocketTimeoutException, return a new one of the same type; Otherwise return
122    * an IOException.
123    * @param addr target address
124    * @param exception the relevant exception
125    * @return an exception to throw
126    */
127   public static IOException wrapException(InetSocketAddress addr, Exception exception) {
128     if (exception instanceof ConnectException) {
129       // connection refused; include the host:port in the error
130       return (ConnectException) new ConnectException(
131           "Call to " + addr + " failed on connection exception: " + exception).initCause(exception);
132     } else if (exception instanceof SocketTimeoutException) {
133       return (SocketTimeoutException) new SocketTimeoutException(
134           "Call to " + addr + " failed because " + exception).initCause(exception);
135     } else if (exception instanceof ConnectionClosingException) {
136       return (ConnectionClosingException) new ConnectionClosingException(
137           "Call to " + addr + " failed on local exception: " + exception).initCause(exception);
138     } else {
139       return (IOException) new IOException(
140           "Call to " + addr + " failed on local exception: " + exception).initCause(exception);
141     }
142   }
