27  /**
28   * View and edit the current cluster schema. Use this API making any modification to
29   * namespaces, tables, etc.
30   *
31   * <h2>Implementation Notes</h2>
32   * Nonces are for when operation is non-idempotent to ensure once-only semantic, even
33   * across process failures.
34   */
35  // ClusterSchema is introduced to encapsulate schema modification. Currently the different aspects
36  // are spread about the code base. This effort is about cleanup, shutting down access, and
37  // coalescing common code. In particular, we'd contain filesystem modification. Other
38  // benefits are to make all schema modification work the same way (one way to do an operation only
39  // rather than the current approach where how an operation is done varies with context) and to make
40  // it so clusterschema modification can stand apart from Master to faciliate standalone
41  // testing. It is part of the filesystem refactor project that undoes the dependency on a
42  // layout in HDFS that mimics our model of tables have regions have column families have files.
43  // With this Interface in place, with all modifications going via this route where no filesystem
44  // particulars are exposed, redoing our internals will take less effort.
45  //
46  // Currently ClusterSchema Interface will include namespace and table manipulation. Ideally a
47  // form of this Interface will go all the ways down to the file manipulation level but currently
48  // TBD.
49  //
50  // ClusterSchema is private to the Master; only the Master knows current cluster state and has
51  // means of editing/altering it.
52  //
53  // TODO: Remove Server argument when MasterServices are passed.
54  // TODO: We return Future<ProcedureInfo> in the below from most methods. It may change to return
55  // a ProcedureFuture subsequently.
56  @InterfaceAudience.Private
57  public interface ClusterSchema {
58    /**
59     * Timeout for cluster operations in milliseconds.
60     */
61    public static final String HBASE_MASTER_CLUSTER_SCHEMA_OPERATION_TIMEOUT_KEY =
62        "hbase.master.cluster.schema.operation.timeout";
63    /**
64     * Default operation timeout in milliseconds.
65     */
66    public static final int DEFAULT_HBASE_MASTER_CLUSTER_SCHEMA_OPERATION_TIMEOUT =
67        5 * 60 * 1000;
68  
69    /**
70     * For internals use only. Do not use! Provisionally part of this Interface.
71     * Prefer the high-level APIs available elsewhere in this API.
72     * @return Instance of {@link TableNamespaceManager}
73     */
74    // TODO: Remove from here. Keep internal. This Interface is too high-level to host this accessor.
75    TableNamespaceManager getTableNamespaceManager();
76  
77    /**
78     * Create a new Namespace.
79     * @param namespaceDescriptor descriptor for new Namespace
80     * @param nonceGroup Identifier for the source of the request, a client or process.
81     * @param nonce A unique identifier for this operation from the client or process identified by
82     *    <code>nonceGroup</code> (the source must ensure each operation gets a unique id).
83     * @return procedure id
84     * @throws IOException Throws {@link ClusterSchemaException} and {@link InterruptedIOException}
85     *    as well as {@link IOException}
86     */
87    long createNamespace(NamespaceDescriptor namespaceDescriptor, long nonceGroup, long nonce)
88    throws IOException;
89  
90    /**
91     * Modify an existing Namespace.
92     * @param nonceGroup Identifier for the source of the request, a client or process.
93     * @param nonce A unique identifier for this operation from the client or process identified by
94     *    <code>nonceGroup</code> (the source must ensure each operation gets a unique id).
95     * @return procedure id
96     * @throws IOException Throws {@link ClusterSchemaException} and {@link InterruptedIOException}
97     *    as well as {@link IOException}
98     */
99    long modifyNamespace(NamespaceDescriptor descriptor, long nonceGroup, long nonce)
100   throws IOException;
101 
102   /**
103    * Delete an existing Namespace.
104    * Only empty Namespaces (no tables) can be removed.
105    * @param nonceGroup Identifier for the source of the request, a client or process.
106    * @param nonce A unique identifier for this operation from the client or process identified by
107    *    <code>nonceGroup</code> (the source must ensure each operation gets a unique id).
108    * @return procedure id
109    * @throws IOException Throws {@link ClusterSchemaException} and {@link InterruptedIOException}
110    *    as well as {@link IOException}
111    */
112   long deleteNamespace(String name, long nonceGroup, long nonce)
113   throws IOException;
114 
115   /**
116    * Get a Namespace
117    * @param name Name of the Namespace
118    * @return Namespace descriptor for <code>name</code>
119    * @throws IOException Throws {@link ClusterSchemaException} and {@link InterruptedIOException}
120    *    as well as {@link IOException}
121    */
122   // No Future here because presumption is that the request will go against cached metadata so
123   // return immediately -- no need of running a Procedure.
124   NamespaceDescriptor getNamespace(String name) throws IOException;
125 
126   /**
127    * Get all Namespaces
128    * @return All Namespace descriptors
129    */
130   List<NamespaceDescriptor> getNamespaces() throws IOException;
131 }
