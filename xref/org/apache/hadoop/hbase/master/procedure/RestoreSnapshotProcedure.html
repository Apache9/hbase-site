117   @Override
118   protected Flow executeFromState(final MasterProcedureEnv env, final RestoreSnapshotState state)
119       throws InterruptedException {
120     if (isTraceEnabled()) {
121       LOG.trace(this + " execute state=" + state);
122     }
123
124     // Make sure that the monitor status is set up
125     getMonitorStatus();
126
127     try {
128       switch (state) {
129         case RESTORE_SNAPSHOT_PRE_OPERATION:
130           // Verify if we can restore the table
131           prepareRestore(env);
132           setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_UPDATE_TABLE_DESCRIPTOR);
133           break;
134         case RESTORE_SNAPSHOT_UPDATE_TABLE_DESCRIPTOR:
135           updateTableDescriptor(env);
136           setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_WRITE_FS_LAYOUT);
137           break;
138         case RESTORE_SNAPSHOT_WRITE_FS_LAYOUT:
139           restoreSnapshot(env);
140           setNextState(RestoreSnapshotState.RESTORE_SNAPSHOT_UPDATE_META);
141           break;
142         case RESTORE_SNAPSHOT_UPDATE_META:
143           updateMETA(env);
144           return Flow.NO_MORE_STATE;
145         default:
146           throw new UnsupportedOperationException("unhandled state=" + state);
147       }
148     } catch (IOException e) {
149       if (isRollbackSupported(state)) {
150         setFailure("master-restore-snapshot", e);
151       } else {
152         LOG.warn("Retriable error trying to restore snapshot=" + snapshot.getName() +
153           " to table=" + getTableName() + " (in state=" + state + ")", e);
154       }
155     }
156     return Flow.HAS_MORE_STATE;
157   }

159   @Override
160   protected void rollbackState(final MasterProcedureEnv env, final RestoreSnapshotState state)
161       throws IOException {
162     if (state == RestoreSnapshotState.RESTORE_SNAPSHOT_PRE_OPERATION) {
163       // nothing to rollback
164       return;
165     }
166
167     // The restore snapshot doesn't have a rollback. The execution will succeed, at some point.
168     throw new UnsupportedOperationException("unhandled state=" + state);
169   }

171   @Override
172   protected boolean isRollbackSupported(final RestoreSnapshotState state) {
173     switch (state) {
174       case RESTORE_SNAPSHOT_PRE_OPERATION:
175         return true;
176       default:
177         return false;
178     }

222   @Override
223   public void serializeStateData(final OutputStream stream) throws IOException {
224     super.serializeStateData(stream);
225
226     MasterProcedureProtos.RestoreSnapshotStateData.Builder restoreSnapshotMsg =
227       MasterProcedureProtos.RestoreSnapshotStateData.newBuilder()
228         .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
229         .setSnapshot(this.snapshot)
230         .setModifiedTableSchema(ProtobufUtil.convertToTableSchema(modifiedHTableDescriptor));
231
232     if (regionsToRestore != null) {
233       for (HRegionInfo hri: regionsToRestore) {
234         restoreSnapshotMsg.addRegionInfoForRestore(HRegionInfo.convert(hri));
235       }
236     }
237     if (regionsToRemove != null) {
238       for (HRegionInfo hri: regionsToRemove) {
239         restoreSnapshotMsg.addRegionInfoForRemove(HRegionInfo.convert(hri));
240       }
241     }
242     if (regionsToAdd != null) {
243       for (HRegionInfo hri: regionsToAdd) {
244         restoreSnapshotMsg.addRegionInfoForAdd(HRegionInfo.convert(hri));
245       }
246     }
247     if (!parentsToChildrenPairMap.isEmpty()) {
248       final Iterator<Map.Entry<String, Pair<String, String>>> it =
249         parentsToChildrenPairMap.entrySet().iterator();
250       while (it.hasNext()) {
251         final Map.Entry<String, Pair<String, String>> entry = it.next();
252
253         MasterProcedureProtos.RestoreParentToChildRegionsPair.Builder parentToChildrenPair =
254           MasterProcedureProtos.RestoreParentToChildRegionsPair.newBuilder()
255           .setParentRegionName(entry.getKey())
256           .setChild1RegionName(entry.getValue().getFirst())
257           .setChild2RegionName(entry.getValue().getSecond());
258         restoreSnapshotMsg.addParentToChildRegionsPairList (parentToChildrenPair);
259       }
260     }
261     restoreSnapshotMsg.build().writeDelimitedTo(stream);
262   }

264   @Override
265   public void deserializeStateData(final InputStream stream) throws IOException {
266     super.deserializeStateData(stream);
267
268     MasterProcedureProtos.RestoreSnapshotStateData restoreSnapshotMsg =
269       MasterProcedureProtos.RestoreSnapshotStateData.parseDelimitedFrom(stream);
270     setUser(MasterProcedureUtil.toUserInfo(restoreSnapshotMsg.getUserInfo()));
271     snapshot = restoreSnapshotMsg.getSnapshot();
272     modifiedHTableDescriptor =
273       ProtobufUtil.convertToHTableDesc(restoreSnapshotMsg.getModifiedTableSchema());
274
275     if (restoreSnapshotMsg.getRegionInfoForRestoreCount() == 0) {
276       regionsToRestore = null;
277     } else {
278       regionsToRestore =
279         new ArrayList<HRegionInfo>(restoreSnapshotMsg.getRegionInfoForRestoreCount());
280       for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForRestoreList()) {
281         regionsToRestore.add(HRegionInfo.convert(hri));
282       }
283     }
284     if (restoreSnapshotMsg.getRegionInfoForRemoveCount() == 0) {
285       regionsToRemove = null;
286     } else {
287       regionsToRemove =
288         new ArrayList<HRegionInfo>(restoreSnapshotMsg.getRegionInfoForRemoveCount());
289       for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForRemoveList()) {
290         regionsToRemove.add(HRegionInfo.convert(hri));
291       }
292     }
293     if (restoreSnapshotMsg.getRegionInfoForAddCount() == 0) {
294       regionsToAdd = null;
295     } else {
296       regionsToAdd = new ArrayList<HRegionInfo>(restoreSnapshotMsg.getRegionInfoForAddCount());
297       for (HBaseProtos.RegionInfo hri: restoreSnapshotMsg.getRegionInfoForAddList()) {
298         regionsToAdd.add(HRegionInfo.convert(hri));
299       }
300     }
301     if (restoreSnapshotMsg.getParentToChildRegionsPairListCount() > 0) {
302       for (MasterProcedureProtos.RestoreParentToChildRegionsPair parentToChildrenPair:
303         restoreSnapshotMsg.getParentToChildRegionsPairListList()) {
304         parentsToChildrenPairMap.put(
305           parentToChildrenPair.getParentRegionName(),
306           new Pair<String, String>(
307             parentToChildrenPair.getChild1RegionName(),
308             parentToChildrenPair.getChild2RegionName()));
309       }
310     }

313   /**
314    * Action before any real action of restoring from snapshot.
315    * @param env MasterProcedureEnv
316    * @throws IOException
317    */
318   private void prepareRestore(final MasterProcedureEnv env) throws IOException {
319     final TableName tableName = getTableName();
320     // Checks whether the table exists
321     if (!MetaTableAccessor.tableExists(env.getMasterServices().getConnection(), tableName)) {
322       throw new TableNotFoundException(tableName);
323     }
324
325     // Check whether table is disabled.
326     env.getMasterServices().checkTableModifiable(tableName);
327
328     // Check that we have at least 1 CF
329     if (modifiedHTableDescriptor.getColumnFamilies().length == 0) {
330       throw new DoNotRetryIOException("Table " + getTableName().toString() +
331         " should have at least one column family.");
332     }
333
334     if (!getTableName().isSystemTable()) {
335       // Table already exist. Check and update the region quota for this table namespace.
336       final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
337       SnapshotManifest manifest = SnapshotManifest.open(
338         env.getMasterConfiguration(),
339         mfs.getFileSystem(),
340         SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, mfs.getRootDir()),
341         snapshot);
342       int snapshotRegionCount = manifest.getRegionManifestsMap().size();
343       int tableRegionCount =
344           ProcedureSyncWait.getMasterQuotaManager(env).getRegionCountOfTable(tableName);
345
346       if (snapshotRegionCount > 0 && tableRegionCount != snapshotRegionCount) {
347         ProcedureSyncWait.getMasterQuotaManager(env).checkAndUpdateNamespaceRegionQuota(
348           tableName, snapshotRegionCount);
349       }
350     }
351   }

362   /**
363    * Execute the on-disk Restore
364    * @param env MasterProcedureEnv
365    * @throws IOException
366    **/
367   private void restoreSnapshot(final MasterProcedureEnv env) throws IOException {
368     MasterFileSystem fileSystemManager = env.getMasterServices().getMasterFileSystem();
369     FileSystem fs = fileSystemManager.getFileSystem();
370     Path rootDir = fileSystemManager.getRootDir();
371     final ForeignExceptionDispatcher monitorException = new ForeignExceptionDispatcher();
372
373     LOG.info("Starting restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot));
374     try {
375       Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshot, rootDir);
376       SnapshotManifest manifest = SnapshotManifest.open(
377         env.getMasterServices().getConfiguration(), fs, snapshotDir, snapshot);
378       RestoreSnapshotHelper restoreHelper = new RestoreSnapshotHelper(
379         env.getMasterServices().getConfiguration(),
380         fs,
381         manifest,
382         modifiedHTableDescriptor,
383         rootDir,
384         monitorException,
385         getMonitorStatus());
386
387       RestoreSnapshotHelper.RestoreMetaChanges metaChanges = restoreHelper.restoreHdfsRegions();
388       regionsToRestore = metaChanges.getRegionsToRestore();
389       regionsToRemove = metaChanges.getRegionsToRemove();
390       regionsToAdd = metaChanges.getRegionsToAdd();
391       parentsToChildrenPairMap = metaChanges.getParentToChildrenPairMap();
392     } catch (IOException e) {
393       String msg = "restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot)
394         + " failed in on-disk restore. Try re-running the restore command.";
395       LOG.error(msg, e);
396       monitorException.receive(
397         new ForeignException(env.getMasterServices().getServerName().toString(), e));
398       throw new IOException(msg, e);
399     }
400   }

402   /**
403    * Apply changes to hbase:meta
404    * @param env MasterProcedureEnv
405    * @throws IOException
406    **/
407   private void updateMETA(final MasterProcedureEnv env) throws IOException {
408     try {
409       Connection conn = env.getMasterServices().getConnection();
410
411       // 1. Forces all the RegionStates to be offline
412       //
413       // The AssignmentManager keeps all the region states around
414       // with no possibility to remove them, until the master is restarted.
415       // This means that a region marked as SPLIT before the restore will never be assigned again.
416       // To avoid having all states around all the regions are switched to the OFFLINE state,
417       // which is the same state that the regions will be after a delete table.
418       forceRegionsOffline(env, regionsToAdd);
419       forceRegionsOffline(env, regionsToRestore);
420       forceRegionsOffline(env, regionsToRemove);
421
422       getMonitorStatus().setStatus("Preparing to restore each region");
423
424       // 2. Applies changes to hbase:meta
425       // (2.1). Removes the current set of regions from META
426       //
427       // By removing also the regions to restore (the ones present both in the snapshot
428       // and in the current state) we ensure that no extra fields are present in META
429       // e.g. with a simple add addRegionToMeta() the splitA and splitB attributes
430       // not overwritten/removed, so you end up with old informations
431       // that are not correct after the restore.
432       if (regionsToRemove != null) {
433         MetaTableAccessor.deleteRegions(conn, regionsToRemove);
434       }
435
436       // (2.2). Add the new set of regions to META
437       //
438       // At this point the old regions are no longer present in META.
439       // and the set of regions present in the snapshot will be written to META.
440       // All the information in hbase:meta are coming from the .regioninfo of each region present
441       // in the snapshot folder.
442       if (regionsToAdd != null) {
443         MetaTableAccessor.addRegionsToMeta(
444           conn,
445           regionsToAdd,
446           modifiedHTableDescriptor.getRegionReplication());
447       }
448
449       if (regionsToRestore != null) {
450         MetaTableAccessor.overwriteRegions(
451           conn,
452           regionsToRestore,
453           modifiedHTableDescriptor.getRegionReplication());
454       }
455
456       RestoreSnapshotHelper.RestoreMetaChanges metaChanges =
457         new RestoreSnapshotHelper.RestoreMetaChanges(
458           modifiedHTableDescriptor, parentsToChildrenPairMap);
459       metaChanges.updateMetaParentRegions(conn, regionsToAdd);
460
461       // At this point the restore is complete.
462       LOG.info("Restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot) +
463         " on table=" + getTableName() + " completed!");
464     } catch (IOException e) {
465       final ForeignExceptionDispatcher monitorException = new ForeignExceptionDispatcher();
466       String msg = "restore snapshot=" + ClientSnapshotDescriptionUtils.toString(snapshot)
467           + " failed in meta update. Try re-running the restore command.";
468       LOG.error(msg, e);
469       monitorException.receive(
470         new ForeignException(env.getMasterServices().getServerName().toString(), e));
471       throw new IOException(msg, e);
472     }
473
474     monitorStatus.markComplete("Restore snapshot '"+ snapshot.getName() +"'!");
475     MetricsSnapshot metricsSnapshot = new MetricsSnapshot();
476     metricsSnapshot.addSnapshotRestore(
477       monitorStatus.getCompletionTimestamp() - monitorStatus.getStartTime());
478   }

480   /**
481    * Make sure that region states of the region list is in OFFLINE state.
482    * @param env MasterProcedureEnv
483    * @param hris region info list
484    **/
485   private void forceRegionsOffline(final MasterProcedureEnv env, final List<HRegionInfo> hris) {
486     RegionStates states = env.getMasterServices().getAssignmentManager().getRegionStates();
487     if (hris != null) {
488       for (HRegionInfo hri: hris) {
489         states.regionOffline(hri);
490       }
491     }
492   }
