44  /**
45   * Used by the HMaster on startup to split meta logs and assign the meta table.
46   */
47  @InterfaceAudience.Private
48  public class MasterMetaBootstrap {
49    private static final Log LOG = LogFactory.getLog(MasterMetaBootstrap.class);

51    private final MonitoredTask status;
52    private final HMaster master;

54    private Set<ServerName> previouslyFailedServers;
55    private Set<ServerName> previouslyFailedMetaRSs;

57    public MasterMetaBootstrap(final HMaster master, final MonitoredTask status) {
58      this.master = master;
59      this.status = status;
60    }

62    public void splitMetaLogsBeforeAssignment() throws IOException, KeeperException {
63      // get a list for previously failed RS which need log splitting work
64      // we recover hbase:meta region servers inside master initialization and
65      // handle other failed servers in SSH in order to start up master node ASAP
66      previouslyFailedServers = master.getMasterWalManager().getFailedServersFromLogFolders();
67
68      // log splitting for hbase:meta server
69      ServerName oldMetaServerLocation = master.getMetaTableLocator()
70          .getMetaRegionLocation(master.getZooKeeper());
71      if (oldMetaServerLocation != null && previouslyFailedServers.contains(oldMetaServerLocation)) {
72        splitMetaLogBeforeAssignment(oldMetaServerLocation);
73        // Note: we can't remove oldMetaServerLocation from previousFailedServers list because it
74        // may also host user regions
75      }
76      previouslyFailedMetaRSs = getPreviouselyFailedMetaServersFromZK();
77      // need to use union of previouslyFailedMetaRSs recorded in ZK and previouslyFailedServers
78      // instead of previouslyFailedMetaRSs alone to address the following two situations:
79      // 1) the chained failure situation(recovery failed multiple times in a row).
80      // 2) master get killed right before it could delete the recovering hbase:meta from ZK while the
81      // same server still has non-meta wals to be replayed so that
82      // removeStaleRecoveringRegionsFromZK can't delete the stale hbase:meta region
83      // Passing more servers into splitMetaLog is all right. If a server doesn't have hbase:meta wal,
84      // there is no op for the server.
85      previouslyFailedMetaRSs.addAll(previouslyFailedServers);
86    }
87
88    public void assignMeta() throws InterruptedException, IOException, KeeperException {
89      assignMeta(previouslyFailedMetaRSs, HRegionInfo.DEFAULT_REPLICA_ID);
90    }

92    public void processDeadServers() throws IOException {
93      // Master has recovered hbase:meta region server and we put
94      // other failed region servers in a queue to be handled later by SSH
95      for (ServerName tmpServer : previouslyFailedServers) {
96        master.getServerManager().processDeadServer(tmpServer, true);
97      }
98    }

100   public void assignMetaReplicas()
101       throws IOException, InterruptedException, KeeperException {
102     int numReplicas = master.getConfiguration().getInt(HConstants.META_REPLICAS_NUM,
103            HConstants.DEFAULT_META_REPLICA_NUM);
104     final Set<ServerName> EMPTY_SET = new HashSet<ServerName>();
105     for (int i = 1; i < numReplicas; i++) {
106       assignMeta(EMPTY_SET, i);
107     }
108     unassignExcessMetaReplica(numReplicas);
109   }
110
111   private void splitMetaLogBeforeAssignment(ServerName currentMetaServer) throws IOException {
112     if (RecoveryMode.LOG_REPLAY == master.getMasterWalManager().getLogRecoveryMode()) {
113       // In log replay mode, we mark hbase:meta region as recovering in ZK
114       master.getMasterWalManager().prepareLogReplay(currentMetaServer,
115         Collections.<HRegionInfo>singleton(HRegionInfo.FIRST_META_REGIONINFO));
116     } else {
117       // In recovered.edits mode: create recovered edits file for hbase:meta server
118       master.getMasterWalManager().splitMetaLog(currentMetaServer);
119     }
120   }

122   private void unassignExcessMetaReplica(int numMetaReplicasConfigured) {
123     final ZooKeeperWatcher zooKeeper = master.getZooKeeper();
124     // unassign the unneeded replicas (for e.g., if the previous master was configured
125     // with a replication of 3 and now it is 2, we need to unassign the 1 unneeded replica)
126     try {
127       List<String> metaReplicaZnodes = zooKeeper.getMetaReplicaNodes();
128       for (String metaReplicaZnode : metaReplicaZnodes) {
129         int replicaId = zooKeeper.getMetaReplicaIdFromZnode(metaReplicaZnode);
130         if (replicaId >= numMetaReplicasConfigured) {
131           RegionState r = MetaTableLocator.getMetaRegionState(zooKeeper, replicaId);
132           LOG.info("Closing excess replica of meta region " + r.getRegion());
133           // send a close and wait for a max of 30 seconds
134           ServerManager.closeRegionSilentlyAndWait(master.getClusterConnection(),
135               r.getServerName(), r.getRegion(), 30000);
136           ZKUtil.deleteNode(zooKeeper, zooKeeper.getZNodeForReplica(replicaId));
137         }
138       }
139     } catch (Exception ex) {
140       // ignore the exception since we don't want the master to be wedged due to potential
141       // issues in the cleanup of the extra regions. We can do that cleanup via hbck or manually
142       LOG.warn("Ignoring exception " + ex);
143     }
144   }

146   /**
147    * Check <code>hbase:meta</code> is assigned. If not, assign it.
148    */
149   protected void assignMeta(Set<ServerName> previouslyFailedMetaRSs, int replicaId)
150       throws InterruptedException, IOException, KeeperException {
151     final AssignmentManager assignmentManager = master.getAssignmentManager();
152
153     // Work on meta region
154     int assigned = 0;
155     long timeout = master.getConfiguration().getLong("hbase.catalog.verification.timeout", 1000);
156     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
157       status.setStatus("Assigning hbase:meta region");
158     } else {
159       status.setStatus("Assigning hbase:meta region, replicaId " + replicaId);
160     }
161
162     // Get current meta state from zk.
163     RegionState metaState = MetaTableLocator.getMetaRegionState(master.getZooKeeper(), replicaId);
164     HRegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO,
165         replicaId);
166     RegionStates regionStates = assignmentManager.getRegionStates();
167     regionStates.createRegionState(hri, metaState.getState(),
168         metaState.getServerName(), null);
169
170     if (!metaState.isOpened() || !master.getMetaTableLocator().verifyMetaRegionLocation(
171         master.getClusterConnection(), master.getZooKeeper(), timeout, replicaId)) {
172       ServerName currentMetaServer = metaState.getServerName();
173       if (master.getServerManager().isServerOnline(currentMetaServer)) {
174         if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
175           LOG.info("Meta was in transition on " + currentMetaServer);
176         } else {
177           LOG.info("Meta with replicaId " + replicaId + " was in transition on " +
178                     currentMetaServer);
179         }
180         assignmentManager.processRegionsInTransition(Collections.singletonList(metaState));
181       } else {
182         if (currentMetaServer != null) {
183           if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
184             splitMetaLogBeforeAssignment(currentMetaServer);
185             regionStates.logSplit(HRegionInfo.FIRST_META_REGIONINFO);
186             previouslyFailedMetaRSs.add(currentMetaServer);
187           }
188         }
189         LOG.info("Re-assigning hbase:meta with replicaId, " + replicaId +
190             " it was on " + currentMetaServer);
191         assignmentManager.assignMeta(hri);
192       }
193       assigned++;
194     }
195
196     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
197       // TODO: should we prevent from using state manager before meta was initialized?
198       // tableStateManager.start();
199       master.getTableStateManager()
200         .setTableState(TableName.META_TABLE_NAME, TableState.State.ENABLED);
201     }
202
203     if ((RecoveryMode.LOG_REPLAY == master.getMasterWalManager().getLogRecoveryMode())
204         && (!previouslyFailedMetaRSs.isEmpty())) {
205       // replay WAL edits mode need new hbase:meta RS is assigned firstly
206       status.setStatus("replaying log for Meta Region");
207       master.getMasterWalManager().splitMetaLog(previouslyFailedMetaRSs);
208     }
209
210     assignmentManager.setEnabledTable(TableName.META_TABLE_NAME);
211     master.getTableStateManager().start();
212
213     // Make sure a hbase:meta location is set. We need to enable SSH here since
214     // if the meta region server is died at this time, we need it to be re-assigned
215     // by SSH so that system tables can be assigned.
216     // No need to wait for meta is assigned = 0 when meta is just verified.
217     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableCrashedServerProcessing(assigned != 0);
218     LOG.info("hbase:meta with replicaId " + replicaId + " assigned=" + assigned + ", location="
219       + master.getMetaTableLocator().getMetaRegionLocation(master.getZooKeeper(), replicaId));
220     status.setStatus("META assigned.");
221   }

223   private void enableCrashedServerProcessing(final boolean waitForMeta)
224       throws IOException, InterruptedException {
225     // If crashed server processing is disabled, we enable it and expire those dead but not expired
226     // servers. This is required so that if meta is assigning to a server which dies after
227     // assignMeta starts assignment, ServerCrashProcedure can re-assign it. Otherwise, we will be
228     // stuck here waiting forever if waitForMeta is specified.
229     if (!master.isServerCrashProcessingEnabled()) {
230       master.setServerCrashProcessingEnabled(true);
231       master.getServerManager().processQueuedDeadServers();
232     }
233
234     if (waitForMeta) {
235       master.getMetaTableLocator().waitMetaRegionLocation(master.getZooKeeper());
236     }
237   }

239   /**
240    * This function returns a set of region server names under hbase:meta recovering region ZK node
241    * @return Set of meta server names which were recorded in ZK
242    */
243   private Set<ServerName> getPreviouselyFailedMetaServersFromZK() throws KeeperException {
244     final ZooKeeperWatcher zooKeeper = master.getZooKeeper();
245     Set<ServerName> result = new HashSet<ServerName>();
246     String metaRecoveringZNode = ZKUtil.joinZNode(zooKeeper.recoveringRegionsZNode,
247       HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
248     List<String> regionFailedServers = ZKUtil.listChildrenNoWatch(zooKeeper, metaRecoveringZNode);
249     if (regionFailedServers == null) return result;
250
251     for (String failedServer : regionFailedServers) {
252       ServerName server = ServerName.parseServerName(failedServer);
253       result.add(server);
254     }
255     return result;
256   }
