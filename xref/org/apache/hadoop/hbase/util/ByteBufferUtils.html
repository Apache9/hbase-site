58    /**
59     * Similar to {@link WritableUtils#writeVLong(java.io.DataOutput, long)},
60     * but writes to a {@link ByteBuffer}.
61     */
62    public static void writeVLong(ByteBuffer out, long i) {
63      if (i >= -112 && i <= 127) {
64        out.put((byte) i);
65        return;
66      }
67
68      int len = -112;
69      if (i < 0) {
70        i ^= -1L; // take one's complement
71        len = -120;
72      }
73
74      long tmp = i;
75      while (tmp != 0) {
76        tmp = tmp >> 8;
77        len--;
78      }
79
80      out.put((byte) len);
81
82      len = (len < -120) ? -(len + 120) : -(len + 112);
83
84      for (int idx = len; idx != 0; idx--) {
85        int shiftbits = (idx - 1) * 8;
86        long mask = 0xFFL << shiftbits;
87        out.put((byte) ((i & mask) >> shiftbits));
88      }
89    }

91    /**
92     * Similar to {@link WritableUtils#readVLong(DataInput)} but reads from a
93     * {@link ByteBuffer}.
94     */
95    public static long readVLong(ByteBuffer in) {
96      byte firstByte = in.get();
97      int len = WritableUtils.decodeVIntSize(firstByte);
98      if (len == 1) {
99        return firstByte;
100     }
101     long i = 0;
102     for (int idx = 0; idx < len-1; idx++) {
103       byte b = in.get();
104       i = i << 8;
105       i = i | (b & 0xFF);
106     }
107     return (WritableUtils.isNegativeVInt(firstByte) ? (i ^ -1L) : i);
108   }

111   /**
112    * Put in buffer integer using 7 bit encoding. For each written byte:
113    * 7 bits are used to store value
114    * 1 bit is used to indicate whether there is next bit.
115    * @param value Int to be compressed.
116    * @param out Where to put compressed data
117    * @return Number of bytes written.
118    * @throws IOException on stream error
119    */
120    public static int putCompressedInt(OutputStream out, final int value)
121       throws IOException {
122     int i = 0;
123     int tmpvalue = value;
124     do {
125       byte b = (byte) (tmpvalue & VALUE_MASK);
126       tmpvalue >>>= NEXT_BIT_SHIFT;
127       if (tmpvalue != 0) {
128         b |= (byte) NEXT_BIT_MASK;
129       }
130       out.write(b);
131       i++;
132     } while (tmpvalue != 0);
133     return i;
134   }

175    * Copy data from a buffer to an output stream. Does not update the position
176    * in the buffer.
177    * @param out the stream to write bytes to
178    * @param in the buffer to read bytes from
179    * @param offset the offset in the buffer (from the buffer's array offset)
180    *      to start copying bytes from
181    * @param length the number of bytes to copy
182    */
183   public static void copyBufferToStream(OutputStream out, ByteBuffer in,
184       int offset, int length) throws IOException {
185     if (out instanceof ByteBufferWriter) {
186       ((ByteBufferWriter) out).write(in, offset, length);
187     } else if (in.hasArray()) {
188       out.write(in.array(), in.arrayOffset() + offset, length);
189     } else {
190       for (int i = 0; i < length; ++i) {
191         out.write(toByte(in, offset + i));
192       }
193     }
194   }

215   /**
216    * Check how many bytes are required to store value.
217    * @param value Value which size will be tested.
218    * @return How many bytes are required to store value.
219    */
220   public static int longFitsIn(final long value) {
221     if (value < 0) {
222       return 8;
223     }
224
225     if (value < (1l << 4 * 8)) {
226       // no more than 4 bytes
227       if (value < (1l << 2 * 8)) {
228         if (value < (1l << 1 * 8)) {
229           return 1;
230         }
231         return 2;
232       }
233       if (value < (1l << 3 * 8)) {
234         return 3;
235       }
236       return 4;
237     }
238     // more than 4 bytes
239     if (value < (1l << 6 * 8)) {
240       if (value < (1l << 5 * 8)) {
241         return 5;
242       }
243       return 6;
244     }
245     if (value < (1l << 7 * 8)) {
246       return 7;
247     }
248     return 8;
249   }

256   public static int intFitsIn(final int value) {

261     if (value < (1 << 2 * 8)) {
262       if (value < (1 << 1 * 8)) {
263         return 1;
264       }
265       return 2;
266     }
267     if (value <= (1 << 3 * 8)) {
268       return 3;
269     }
270     return 4;

273   /**
274    * Read integer from stream coded in 7 bits and increment position.
275    * @return the integer that has been read
276    * @throws IOException
277    */
278   public static int readCompressedInt(InputStream input)
279       throws IOException {
280     int result = 0;
281     int i = 0;
282     byte b;
283     do {
284       b = (byte) input.read();
285       result += (b & VALUE_MASK) << (NEXT_BIT_SHIFT * i);
286       i++;
287       if (i > Bytes.SIZEOF_INT + 1) {
288         throw new IllegalStateException(
289             "Corrupted compressed int (too long: " + (i + 1) + " bytes)");
290       }
291     } while (0 != (b & NEXT_BIT_MASK));
292     return result;
293   }

295   /**
296    * Read integer from buffer coded in 7 bits and increment position.
297    * @return Read integer.
298    */
299   public static int readCompressedInt(ByteBuffer buffer) {
300     byte b = buffer.get();
301     if ((b & NEXT_BIT_MASK) != 0) {
302       return (b & VALUE_MASK) + (readCompressedInt(buffer) << NEXT_BIT_SHIFT);
303     }
304     return b & VALUE_MASK;
305   }

315     long tmpLong = 0;
316     for (int i = 0; i < fitInBytes; ++i) {
317       tmpLong |= (in.read() & 0xffl) << (8 * i);

366   /**
367    * Copy one buffer's whole data to another. Write starts at the current position of 'out' buffer.
368    * Note : This will advance the position marker of {@code out} and also change the position maker
369    * for {@code in}.
370    * @param in source buffer
371    * @param out destination buffer
372    */
373   public static void copyFromBufferToBuffer(ByteBuffer in, ByteBuffer out) {
374     if (in.hasArray() && out.hasArray()) {
375       int length = in.remaining();
376       System.arraycopy(in.array(), in.arrayOffset(), out.array(), out.arrayOffset(), length);
377       out.position(out.position() + length);
378       in.position(in.limit());
379     } else if (UNSAFE_AVAIL) {
380       int length = in.remaining();
381       UnsafeAccess.copy(in, in.position(), out, out.position(), length);
382       out.position(out.position() + length);
383       in.position(in.limit());
384     } else {
385       out.put(in);
386     }
387   }

389   /**
390    * Copy from one buffer to another from given offset. This will be absolute positional copying and
391    * won't affect the position of any of the buffers.
392    * @param in
393    * @param out
394    * @param sourceOffset
395    * @param destinationOffset
396    * @param length
397    */
398   public static int copyFromBufferToBuffer(ByteBuffer in, ByteBuffer out, int sourceOffset,
399       int destinationOffset, int length) {
400     if (in.hasArray() && out.hasArray()) {
401       System.arraycopy(in.array(), sourceOffset + in.arrayOffset(), out.array(), out.arrayOffset()
402           + destinationOffset, length);
403     } else if (UNSAFE_AVAIL) {
404       UnsafeAccess.copy(in, sourceOffset, out, destinationOffset, length);
405     } else {
406       int outOldPos = out.position();
407       out.position(destinationOffset);
408       ByteBuffer inDup = in.duplicate();
409       inDup.position(sourceOffset).limit(sourceOffset + length);
410       out.put(inDup);
411       out.position(outOldPos);
412     }
413     return destinationOffset + length;
414   }

443    * Find length of common prefix of two parts in the buffer
444    * @param buffer Where parts are located.
445    * @param offsetLeft Offset of the first part.
446    * @param offsetRight Offset of the second part.
447    * @param limit Maximal length of common prefix.
448    * @return Length of prefix.
449    */
450   public static int findCommonPrefix(ByteBuffer buffer, int offsetLeft,
451       int offsetRight, int limit) {
452     int prefix = 0;
453
454     for (; prefix < limit; ++prefix) {
455       if (buffer.get(offsetLeft + prefix) != buffer.get(offsetRight + prefix)) {
456         break;
457       }
458     }
459
460     return prefix;
461   }

463   /**
464    * Find length of common prefix in two arrays.
465    * @param left Array to be compared.
466    * @param leftOffset Offset in left array.
467    * @param leftLength Length of left array.
468    * @param right Array to be compared.
469    * @param rightOffset Offset in right array.
470    * @param rightLength Length of right array.
471    */
472   public static int findCommonPrefix(
473       byte[] left, int leftOffset, int leftLength,
474       byte[] right, int rightOffset, int rightLength) {
475     int length = Math.min(leftLength, rightLength);
476     int result = 0;
477
478     while (result < length &&
479         left[leftOffset + result] == right[rightOffset + result]) {
480       result++;
481     }
482
483     return result;
484   }

508   /**
509    * Check whether two parts in the same buffer are equal.
510    * @param buffer In which buffer there are parts
511    * @param offsetLeft Beginning of first part.
512    * @param lengthLeft Length of the first part.
513    * @param offsetRight Beginning of the second part.
514    * @param lengthRight Length of the second part.
515    * @return True if equal
516    */
517   public static boolean arePartsEqual(ByteBuffer buffer,
518       int offsetLeft, int lengthLeft,
519       int offsetRight, int lengthRight) {
520     if (lengthLeft != lengthRight) {
521       return false;
522     }
523
524     if (buffer.hasArray()) {
525       return 0 == Bytes.compareTo(
526           buffer.array(), buffer.arrayOffset() + offsetLeft, lengthLeft,
527           buffer.array(), buffer.arrayOffset() + offsetRight, lengthRight);
528     }
529
530     for (int i = 0; i < lengthRight; ++i) {
531       if (buffer.get(offsetLeft + i) != buffer.get(offsetRight + i)) {
532         return false;
533       }
534     }
535     return true;
536   }

582   public static boolean equals(ByteBuffer buf1, int o1, int l1, ByteBuffer buf2, int o2, int l2) {
583     if ((l1 == 0) || (l2 == 0)) {
584       // both 0 length, return true, or else false
585       return l1 == l2;
586     }
587     // Since we're often comparing adjacent sorted data,
588     // it's usual to have equal arrays except for the very last byte
589     // so check that first
590     if (toByte(buf1, o1 + l1 - 1) != toByte(buf2, o2 + l2 - 1)) return false;
591     return compareTo(buf1, o1, l1, buf2, o2, l2) == 0;
592   }

610   public static int compareTo(ByteBuffer buf1, int o1, int l1, ByteBuffer buf2, int o2, int l2) {
611     if (UNSAFE_UNALIGNED) {
612       long offset1Adj, offset2Adj;
613       Object refObj1 = null, refObj2 = null;
614       if (buf1.isDirect()) {
615         offset1Adj = o1 + ((DirectBuffer) buf1).address();
616       } else {
617         offset1Adj = o1 + buf1.arrayOffset() + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET;
618         refObj1 = buf1.array();
619       }
620       if (buf2.isDirect()) {
621         offset2Adj = o2 + ((DirectBuffer) buf2).address();
622       } else {
623         offset2Adj = o2 + buf2.arrayOffset() + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET;
624         refObj2 = buf2.array();
625       }
626       return compareToUnsafe(refObj1, offset1Adj, l1, refObj2, offset2Adj, l2);
627     }
628     int end1 = o1 + l1;
629     int end2 = o2 + l2;
630     for (int i = o1, j = o2; i < end1 && j < end2; i++, j++) {
631       int a = buf1.get(i) & 0xFF;
632       int b = buf2.get(j) & 0xFF;
633       if (a != b) {
634         return a - b;
635       }
636     }
637     return l1 - l2;
638   }

652   public static int compareTo(byte [] buf1, int o1, int l1, ByteBuffer buf2, int o2, int l2) {
653     // This method is nearly same as the compareTo that follows but hard sharing code given
654     // byte array and bytebuffer types and this is a hot code path
655     if (UNSAFE_UNALIGNED) {
656       long offset2Adj;
657       Object refObj2 = null;
658       if (buf2.isDirect()) {
659         offset2Adj = o2 + ((DirectBuffer)buf2).address();
660       } else {
661         offset2Adj = o2 + buf2.arrayOffset() + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET;
662         refObj2 = buf2.array();
663       }
664       return compareToUnsafe(buf1, o1 + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET, l1,
665           refObj2, offset2Adj, l2);
666     }
667     int end1 = o1 + l1;
668     int end2 = o2 + l2;
669     for (int i = o1, j = o2; i < end1 && j < end2; i++, j++) {
670       int a = buf1[i] & 0xFF;
671       int b = buf2.get(i) & 0xFF;
672       if (a != b) {
673         return a - b;
674       }
675     }
676     return l1 - l2;

679   public static int compareTo(ByteBuffer buf1, int o1, int l1, byte[] buf2, int o2, int l2) {
680     if (UNSAFE_UNALIGNED) {
681       long offset1Adj;
682       Object refObj1 = null;
683       if (buf1.isDirect()) {
684         offset1Adj = o1 + ((DirectBuffer) buf1).address();
685       } else {
686         offset1Adj = o1 + buf1.arrayOffset() + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET;
687         refObj1 = buf1.array();
688       }
689       return compareToUnsafe(refObj1, offset1Adj, l1,
690           buf2, o2 + UnsafeAccess.BYTE_ARRAY_BASE_OFFSET, l2);
691     }
692     int end1 = o1 + l1;
693     int end2 = o2 + l2;
694     for (int i = o1, j = o2; i < end1 && j < end2; i++, j++) {
695       int a = buf1.get(i) & 0xFF;
696       int b = buf2[j] & 0xFF;
697       if (a != b) {
698         return a - b;
699       }
700     }

704   static int compareToUnsafe(Object obj1, long o1, int l1, Object obj2, long o2, int l2) {
705     final int minLength = Math.min(l1, l2);
706     final int minWords = minLength / Bytes.SIZEOF_LONG;
707
708     /*
709      * Compare 8 bytes at a time. Benchmarking shows comparing 8 bytes at a time is no slower than
710      * comparing 4 bytes at a time even on 32-bit. On the other hand, it is substantially faster on
711      * 64-bit.
712      */
713     int j = minWords << 3; // Same as minWords * SIZEOF_LONG
714     for (int i = 0; i < j; i += Bytes.SIZEOF_LONG) {
715       long lw = UnsafeAccess.theUnsafe.getLong(obj1, o1 + i);
716       long rw = UnsafeAccess.theUnsafe.getLong(obj2, o2 + i);
717       long diff = lw ^ rw;
718       if (diff != 0) {
719         return lessThanUnsignedLong(lw, rw) ? -1 : 1;
720       }
721     }
722     int offset = j;
723
724     if (minLength - offset >= Bytes.SIZEOF_INT) {
725       int il = UnsafeAccess.theUnsafe.getInt(obj1, o1 + offset);
726       int ir = UnsafeAccess.theUnsafe.getInt(obj2, o2 + offset);
727       if (il != ir) {
728         return lessThanUnsignedInt(il, ir) ? -1 : 1;
729       }
730       offset += Bytes.SIZEOF_INT;
731     }
732     if (minLength - offset >= Bytes.SIZEOF_SHORT) {
733       short sl = UnsafeAccess.theUnsafe.getShort(obj1, o1 + offset);
734       short sr = UnsafeAccess.theUnsafe.getShort(obj2, o2 + offset);
735       if (sl != sr) {
736         return lessThanUnsignedShort(sl, sr) ? -1 : 1;
737       }
738       offset += Bytes.SIZEOF_SHORT;
739     }
740     if (minLength - offset == 1) {
741       int a = (UnsafeAccess.theUnsafe.getByte(obj1, o1 + offset) & 0xff);
742       int b = (UnsafeAccess.theUnsafe.getByte(obj2, o2 + offset) & 0xff);
743       if (a != b) {
744         return a - b;
745       }
746     }
747     return l1 - l2;
748   }

832    * @param length Number of bytes used to store the int value.
833    * @return the int value
834    * @throws IllegalArgumentException
835    *           if there's not enough bytes left in the buffer after the given offset

837   public static int readAsInt(ByteBuffer buf, int offset, final int length) {
838     if (offset + length > buf.limit()) {
839       throw new IllegalArgumentException("offset (" + offset + ") + length (" + length
840           + ") exceed the" + " limit of the buffer: " + buf.limit());
841     }
842     int n = 0;

844       n <<= 8;
845       n ^= toByte(buf, i) & 0xFF;
846     }
847     return n;
