55    @Override
56    public ByteBuffer decodeKeyValues(DataInputStream source,
57        HFileBlockDecodingContext blkDecodingCtx) throws IOException {
58      if (blkDecodingCtx.getClass() != HFileBlockDefaultDecodingContext.class) {
59        throw new IOException(this.getClass().getName() + " only accepts "
60            + HFileBlockDefaultDecodingContext.class.getName() + " as the decoding context.");
61      }
62
63      HFileBlockDefaultDecodingContext decodingCtx =
64          (HFileBlockDefaultDecodingContext) blkDecodingCtx;
65      if (decodingCtx.getHFileContext().isIncludesTags()
66          && decodingCtx.getHFileContext().isCompressTags()) {
67        if (decodingCtx.getTagCompressionContext() != null) {
68          // It will be overhead to create the TagCompressionContext again and again for every block
69          // decoding.
70          decodingCtx.getTagCompressionContext().clear();
71        } else {
72          try {
73            TagCompressionContext tagCompressionContext = new TagCompressionContext(
74                LRUDictionary.class, Byte.MAX_VALUE);
75            decodingCtx.setTagCompressionContext(tagCompressionContext);
76          } catch (Exception e) {
77            throw new IOException("Failed to initialize TagCompressionContext", e);
78          }
79        }
80      }
81      return internalDecodeKeyValues(source, 0, 0, decodingCtx);
82    }

118     /** We need to store a copy of the key. */

124     protected KeyValue.KeyOnlyKeyValue currentKey = new KeyValue.KeyOnlyKeyValue();
125     // A temp pair object which will be reused by ByteBuff#asSubByteBuffer calls. This avoids too

130     public SeekerState(ObjectIntPair<ByteBuffer> tmpPair, boolean includeTags) {
131       this.tmpPair = tmpPair;

135     protected boolean isValid() {
136       return valueOffset != -1;
137     }

139     protected void invalidate() {
140       valueOffset = -1;
141       tagsCompressedLength = 0;
142       currentKey.clear();
143       uncompressTags = true;

147     protected void ensureSpaceForKey() {
148       if (keyLength > keyBuffer.length) {
149         int newKeyBufferLength = Integer.highestOneBit(Math.max(
150             INITIAL_KEY_BUFFER_SIZE, keyLength) - 1) << 1;
151         byte[] newKeyBuffer = new byte[newKeyBufferLength];
152         System.arraycopy(keyBuffer, 0, newKeyBuffer, 0, keyBuffer.length);
153         keyBuffer = newKeyBuffer;
154       }
155     }
156
157     protected void ensureSpaceForTags() {
158       if (tagsLength > tagsBuffer.length) {
159         int newTagsBufferLength = Integer.highestOneBit(Math.max(
160             INITIAL_KEY_BUFFER_SIZE, tagsLength) - 1) << 1;
161         byte[] newTagsBuffer = new byte[newTagsBufferLength];
162         System.arraycopy(tagsBuffer, 0, newTagsBuffer, 0, tagsBuffer.length);
163         tagsBuffer = newTagsBuffer;
164       }
165     }

167     protected void setKey(byte[] keyBuffer, long memTS) {
168       currentKey.setKey(keyBuffer, 0, keyLength);
169       memstoreTS = memTS;
170     }

172     /**
173      * Copy the state from the next one into this instance (the previous state
174      * placeholder). Used to save the previous state when we are advancing the
175      * seeker to the next key/value.
176      */
177     protected void copyFromNext(SeekerState nextState) {
178       if (keyBuffer.length != nextState.keyBuffer.length) {
179         keyBuffer = nextState.keyBuffer.clone();
180       } else if (!isValid()) {
181         // Note: we can only call isValid before we override our state, so this
182         // comes before all the assignments at the end of this method.
183         System.arraycopy(nextState.keyBuffer, 0, keyBuffer, 0,
184              nextState.keyLength);
185       } else {
186         // don't copy the common prefix between this key and the previous one
187         System.arraycopy(nextState.keyBuffer, nextState.lastCommonPrefix,
188             keyBuffer, nextState.lastCommonPrefix, nextState.keyLength
189                 - nextState.lastCommonPrefix);
190       }
191       currentKey.set(nextState.currentKey);
192
193       valueOffset = nextState.valueOffset;
194       keyLength = nextState.keyLength;
195       valueLength = nextState.valueLength;
196       lastCommonPrefix = nextState.lastCommonPrefix;
197       nextKvOffset = nextState.nextKvOffset;
198       memstoreTS = nextState.memstoreTS;
199       currentBuffer = nextState.currentBuffer;
200       tagsOffset = nextState.tagsOffset;
201       tagsLength = nextState.tagsLength;
202       if (nextState.tagCompressionContext != null) {
203         tagCompressionContext = nextState.tagCompressionContext;
204       }
205     }

207     public Cell toCell() {
208       // Buffer backing the value and tags part from the HFileBlock's buffer
209       // When tag compression in use, this will be only the value bytes area.
210       ByteBuffer valAndTagsBuffer;
211       int vOffset;
212       int valAndTagsLength = this.valueLength;
213       int tagsLenSerializationSize = 0;
214       if (this.includeTags && this.tagCompressionContext == null) {
215         // Include the tags part also. This will be the tags bytes + 2 bytes of for storing tags
216         // length
217         tagsLenSerializationSize = this.tagsOffset - (this.valueOffset + this.valueLength);
218         valAndTagsLength += tagsLenSerializationSize + this.tagsLength;
219       }
220       this.currentBuffer.asSubByteBuffer(this.valueOffset, valAndTagsLength, this.tmpPair);
221       valAndTagsBuffer = this.tmpPair.getFirst();
222       vOffset = this.tmpPair.getSecond();// This is the offset to value part in the BB
223       if (valAndTagsBuffer.hasArray()) {
224         return toOnheapCell(valAndTagsBuffer, vOffset, tagsLenSerializationSize);
225       } else {
226         return toOffheapCell(valAndTagsBuffer, vOffset, tagsLenSerializationSize);
227       }
228     }
229
230     private Cell toOnheapCell(ByteBuffer valAndTagsBuffer, int vOffset,
231         int tagsLenSerializationSize) {
232       byte[] tagsArray = HConstants.EMPTY_BYTE_ARRAY;
233       int tOffset = 0;
234       if (this.includeTags) {
235         if (this.tagCompressionContext == null) {
236           tagsArray = valAndTagsBuffer.array();
237           tOffset = valAndTagsBuffer.arrayOffset() + vOffset + this.valueLength
238               + tagsLenSerializationSize;
239         } else {
240           tagsArray = Bytes.copy(tagsBuffer, 0, this.tagsLength);
241           tOffset = 0;
242         }
243       }
244       return new OnheapDecodedCell(Bytes.copy(keyBuffer, 0, this.keyLength),
245           currentKey.getRowLength(), currentKey.getFamilyOffset(), currentKey.getFamilyLength(),
246           currentKey.getQualifierOffset(), currentKey.getQualifierLength(),
247           currentKey.getTimestamp(), currentKey.getTypeByte(), valAndTagsBuffer.array(),
248           valAndTagsBuffer.arrayOffset() + vOffset, this.valueLength, memstoreTS, tagsArray,
249           tOffset, this.tagsLength);
250     }
251
252     private Cell toOffheapCell(ByteBuffer valAndTagsBuffer, int vOffset,
253         int tagsLenSerializationSize) {
254       ByteBuffer tagsBuf =  HConstants.EMPTY_BYTE_BUFFER;
255       int tOffset = 0;
256       if (this.includeTags) {
257         if (this.tagCompressionContext == null) {
258           tagsBuf = valAndTagsBuffer;
259           tOffset = vOffset + this.valueLength + tagsLenSerializationSize;
260         } else {
261           tagsBuf = ByteBuffer.wrap(Bytes.copy(tagsBuffer, 0, this.tagsLength));
262           tOffset = 0;
263         }
264       }
265       return new OffheapDecodedCell(ByteBuffer.wrap(Bytes.copy(keyBuffer, 0, this.keyLength)),
266           currentKey.getRowLength(), currentKey.getFamilyOffset(), currentKey.getFamilyLength(),
267           currentKey.getQualifierOffset(), currentKey.getQualifierLength(),
268           currentKey.getTimestamp(), currentKey.getTypeByte(), valAndTagsBuffer, vOffset,
269           this.valueLength, memstoreTS, tagsBuf, tOffset, this.tagsLength);
270     }

712   protected abstract static class BufferedEncodedSeeker<STATE extends SeekerState>
713       extends AbstractEncodedSeeker {
714     protected ByteBuff currentBuffer;
715     protected TagCompressionContext tagCompressionContext = null;
716     protected  KeyValue.KeyOnlyKeyValue keyOnlyKV = new KeyValue.KeyOnlyKeyValue();
717     // A temp pair object which will be reused by ByteBuff#asSubByteBuffer calls. This avoids too

800     protected void decodeTags() {
801       current.tagsLength = ByteBuff.readCompressedInt(currentBuffer);
802       if (tagCompressionContext != null) {
803         if (current.uncompressTags) {
804           // Tag compression is been used. uncompress it into tagsBuffer
805           current.ensureSpaceForTags();
806           try {
807             current.tagsCompressedLength = tagCompressionContext.uncompressTags(currentBuffer,
808                 current.tagsBuffer, 0, current.tagsLength);
809           } catch (IOException e) {
810             throw new RuntimeException("Exception while uncompressing tags", e);
811           }
812         } else {
813           currentBuffer.skip(current.tagsCompressedLength);
814           current.uncompressTags = true;// Reset this.
815         }
816         current.tagsOffset = -1;
817       } else {
818         // When tag compress is not used, let us not do copying of tags bytes into tagsBuffer.
819         // Just mark the tags Offset so as to create the KV buffer later in getKeyValueBuffer()
820         current.tagsOffset = currentBuffer.position();
821         currentBuffer.skip(current.tagsLength);
822       }
823     }

825     @Override
826     public int seekToKeyInBlock(Cell seekCell, boolean seekBefore) {
827       int rowCommonPrefix = 0;
828       int familyCommonPrefix = 0;
829       int qualCommonPrefix = 0;
830       previous.invalidate();
831       do {
832         int comp;
833         keyOnlyKV.setKey(current.keyBuffer, 0, current.keyLength);
834         if (current.lastCommonPrefix != 0) {
835           // The KV format has row key length also in the byte array. The
836           // common prefix
837           // includes it. So we need to subtract to find out the common prefix
838           // in the
839           // row part alone
840           rowCommonPrefix = Math.min(rowCommonPrefix, current.lastCommonPrefix - 2);
841         }
842         if (current.lastCommonPrefix <= 2) {
843           rowCommonPrefix = 0;
844         }
845         rowCommonPrefix += findCommonPrefixInRowPart(seekCell, keyOnlyKV, rowCommonPrefix);
846         comp = compareCommonRowPrefix(seekCell, keyOnlyKV, rowCommonPrefix);
847         if (comp == 0) {
848           comp = compareTypeBytes(seekCell, keyOnlyKV);
849           if (comp == 0) {
850             // Subtract the fixed row key length and the family key fixed length
851             familyCommonPrefix = Math.max(
852                 0,
853                 Math.min(familyCommonPrefix,
854                     current.lastCommonPrefix - (3 + keyOnlyKV.getRowLength())));
855             familyCommonPrefix += findCommonPrefixInFamilyPart(seekCell, keyOnlyKV,
856                 familyCommonPrefix);
857             comp = compareCommonFamilyPrefix(seekCell, keyOnlyKV, familyCommonPrefix);
858             if (comp == 0) {
859               // subtract the rowkey fixed length and the family key fixed
860               // length
861               qualCommonPrefix = Math.max(
862                   0,
863                   Math.min(
864                       qualCommonPrefix,
865                       current.lastCommonPrefix
866                           - (3 + keyOnlyKV.getRowLength() + keyOnlyKV.getFamilyLength())));
867               qualCommonPrefix += findCommonPrefixInQualifierPart(seekCell, keyOnlyKV,
868                   qualCommonPrefix);
869               comp = compareCommonQualifierPrefix(seekCell, keyOnlyKV, qualCommonPrefix);
870               if (comp == 0) {
871                 comp = CellComparator.compareTimestamps(seekCell, keyOnlyKV);
872                 if (comp == 0) {
873                   // Compare types. Let the delete types sort ahead of puts;
874                   // i.e. types
875                   // of higher numbers sort before those of lesser numbers.
876                   // Maximum
877                   // (255)
878                   // appears ahead of everything, and minimum (0) appears
879                   // after
880                   // everything.
881                   comp = (0xff & keyOnlyKV.getTypeByte()) - (0xff & seekCell.getTypeByte());
882                 }
883               }
884             }
885           }
886         }
887         if (comp == 0) { // exact match
888           if (seekBefore) {
889             if (!previous.isValid()) {
890               // The caller (seekBefore) has to ensure that we are not at the
891               // first key in the block.
892               throw new IllegalStateException("Cannot seekBefore if "
893                   + "positioned at the first key in the block: key="
894                   + Bytes.toStringBinary(seekCell.getRowArray()));
895             }
896             moveToPrevious();
897             return 1;
898           }
899           return 0;
900         }
901
902         if (comp < 0) { // already too large, check previous
903           if (previous.isValid()) {
904             moveToPrevious();
905           } else {
906             return HConstants.INDEX_KEY_MAGIC; // using optimized index key
907           }
908           return 1;
909         }
910
911         // move to next, if more data is available
912         if (currentBuffer.hasRemaining()) {
913           previous.copyFromNext(current);
914           decodeNext();
915           current.setKey(current.keyBuffer, current.memstoreTS);
916         } else {
917           break;
918         }
919       } while (true);
920
921       // we hit the end of the block, not an exact match
922       return 1;
923     }

925     private int compareTypeBytes(Cell key, Cell right) {
926       if (key.getFamilyLength() + key.getQualifierLength() == 0
927           && key.getTypeByte() == Type.Minimum.getCode()) {
928         // left is "bigger", i.e. it appears later in the sorted order
929         return 1;
930       }
931       if (right.getFamilyLength() + right.getQualifierLength() == 0
932           && right.getTypeByte() == Type.Minimum.getCode()) {
933         return -1;
934       }
935       return 0;
936     }

960     private void moveToPrevious() {
961       if (!previous.isValid()) {
962         throw new IllegalStateException(
963             "Can move back only once and not in first key in the block.");
964       }
965
966       STATE tmp = previous;
967       previous = current;
968       current = tmp;
969
970       // move after last key value
971       currentBuffer.position(current.nextKvOffset);
972       // Already decoded the tag bytes. We cache this tags into current state and also the total
973       // compressed length of the tags bytes. For the next time decodeNext() we don't need to decode
974       // the tags again. This might pollute the Data Dictionary what we use for the compression.
975       // When current.uncompressTags is false, we will just reuse the current.tagsBuffer and skip
976       // 'tagsCompressedLength' bytes of source stream.
977       // See in decodeTags()
978       current.tagsBuffer = previous.tagsBuffer;
979       current.tagsCompressedLength = previous.tagsCompressedLength;
980       current.uncompressTags = false;
981       // The current key has to be reset with the previous Cell
982       current.setKey(current.keyBuffer, current.memstoreTS);
983       previous.invalidate();
984     }

1006     int size = 0;
1007     if (encodingCtx.getHFileContext().isIncludesTags()) {
1008       int tagsLength = cell.getTagsLength();
1009       ByteBufferUtils.putCompressedInt(out, tagsLength);
1010       // There are some tags to be written
1011       if (tagsLength > 0) {
1012         TagCompressionContext tagCompressionContext = encodingCtx.getTagCompressionContext();
1013         // When tag compression is enabled, tagCompressionContext will have a not null value. Write
1014         // the tags using Dictionary compression in such a case
1015         if (tagCompressionContext != null) {
1016           // Not passing tagsLength considering that parsing of the tagsLength is not costly
1017           CellUtil.compressTags(out, cell, tagCompressionContext);
1018         } else {
1019           CellUtil.writeTags(out, cell, tagsLength);
1020         }
1021       }
1022       size += tagsLength + KeyValue.TAGS_LENGTH_SIZE;
1023     }
1024     if (encodingCtx.getHFileContext().isIncludesMvcc()) {
1025       // Copy memstore timestamp from the byte buffer to the output stream.
1026       long memstoreTS = cell.getSequenceId();
1027       WritableUtils.writeVLong(out, memstoreTS);
1028       // TODO use a writeVLong which returns the #bytes written so that 2 time parsing can be
1029       // avoided.
1030       size += WritableUtils.getVIntSize(memstoreTS);
1031     }
1032     return size;
1033   }

1035   protected final void afterDecodingKeyValue(DataInputStream source,
1036       ByteBuffer dest, HFileBlockDefaultDecodingContext decodingCtx) throws IOException {
1037     if (decodingCtx.getHFileContext().isIncludesTags()) {
1038       int tagsLength = ByteBufferUtils.readCompressedInt(source);
1039       // Put as unsigned short
1040       dest.put((byte) ((tagsLength >> 8) & 0xff));
1041       dest.put((byte) (tagsLength & 0xff));
1042       if (tagsLength > 0) {
1043         TagCompressionContext tagCompressionContext = decodingCtx.getTagCompressionContext();
1044         // When tag compression is been used in this file, tagCompressionContext will have a not
1045         // null value passed.
1046         if (tagCompressionContext != null) {
1047           tagCompressionContext.uncompressTags(source, dest, tagsLength);
1048         } else {
1049           ByteBufferUtils.copyFromStreamToBuffer(dest, source, tagsLength);
1050         }
1051       }
1052     }
1053     if (decodingCtx.getHFileContext().isIncludesMvcc()) {
1054       long memstoreTS = -1;
1055       try {
1056         // Copy memstore timestamp from the data input stream to the byte
1057         // buffer.
1058         memstoreTS = WritableUtils.readVLong(source);
1059         ByteBufferUtils.writeVLong(dest, memstoreTS);
1060       } catch (IOException ex) {
1061         throw new RuntimeException("Unable to copy memstore timestamp " +
1062             memstoreTS + " after decoding a key/value");
