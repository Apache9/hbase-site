154   public HFileWriterImpl(final Configuration conf, CacheConfig cacheConf, Path path,
155       FSDataOutputStream outputStream,
156       CellComparator comparator, HFileContext fileContext) {
157     this.outputStream = outputStream;
158     this.path = path;
159     this.name = path != null ? path.getName() : outputStream.toString();
160     this.hFileContext = fileContext;
161     DataBlockEncoding encoding = hFileContext.getDataBlockEncoding();
162     if (encoding != DataBlockEncoding.NONE) {
163       this.blockEncoder = new HFileDataBlockEncoderImpl(encoding);
164     } else {
165       this.blockEncoder = NoOpDataBlockEncoder.INSTANCE;
166     }
167     this.comparator = comparator != null ? comparator
168         : CellComparator.COMPARATOR;
169 
170     closeOutputStream = path != null;
171     this.cacheConf = cacheConf;
172     finishInit(conf);
173     if (LOG.isTraceEnabled()) {
174       LOG.trace("Writer" + (path != null ? " for " + path : "") +
175         " initialized with cacheConf: " + cacheConf +
176         " comparator: " + comparator.getClass().getSimpleName() +
177         " fileContext: " + fileContext);
178     }
179   }

212   /**
213    * Checks that the given Cell's key does not violate the key order.
214    *
215    * @param cell Cell whose key to check.
216    * @return true if the key is duplicate
217    * @throws IOException if the key or the key order is wrong
218    */
219   protected boolean checkKey(final Cell cell) throws IOException {
220     boolean isDuplicateKey = false;
221 
222     if (cell == null) {
223       throw new IOException("Key cannot be null or empty");
224     }
225     if (lastCell != null) {
226       int keyComp = comparator.compareKeyIgnoresMvcc(lastCell, cell);
227 
228       if (keyComp > 0) {
229         throw new IOException("Added a key not lexically larger than"
230             + " previous. Current cell = " + cell + ", lastCell = " + lastCell);
231       } else if (keyComp == 0) {
232         isDuplicateKey = true;
233       }
234     }
235     return isDuplicateKey;
236   }

241     if (value == null) {
242       throw new IOException("Value cannot be null");
243     }
244   }

296   /**

298    *
299    * @throws IOException
300    */

302     if (fsBlockWriter.blockSizeWritten() < hFileContext.getBlocksize()) return;

308   /** Clean up the current data block */
309   private void finishBlock() throws IOException {
310     if (!fsBlockWriter.isWriting() || fsBlockWriter.blockSizeWritten() == 0) return;
311 
312     // Update the first data block offset for scanning.
313     if (firstDataBlockOffset == -1) {
314       firstDataBlockOffset = outputStream.getPos();
315     }
316     // Update the last data block offset
317     lastDataBlockOffset = outputStream.getPos();
318     fsBlockWriter.writeHeaderAndData(outputStream);
319     int onDiskSize = fsBlockWriter.getOnDiskSizeWithHeader();
320     Cell indexEntry =
321       getMidpoint(this.comparator, lastCellOfPreviousBlock, firstCellInBlock);
322     dataBlockIndexWriter.addEntry(CellUtil.getCellKeySerializedAsKeyValueKey(indexEntry),
323       lastDataBlockOffset, onDiskSize);
324     totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
325     if (cacheConf.shouldCacheDataOnWrite()) {
326       doCacheOnWrite(lastDataBlockOffset);
327     }
328   }

330   /**
331    * Try to return a Cell that falls between <code>left</code> and
332    * <code>right</code> but that is shorter; i.e. takes up less space. This
333    * trick is used building HFile block index. Its an optimization. It does not
334    * always work. In this case we'll just return the <code>right</code> cell.
335    * 
336    * @param comparator
337    *          Comparator to use.
338    * @param left
339    * @param right
340    * @return A cell that sorts between <code>left</code> and <code>right</code>.
341    */
342   public static Cell getMidpoint(final CellComparator comparator, final Cell left,
343       final Cell right) {
344     // TODO: Redo so only a single pass over the arrays rather than one to
345     // compare and then a
346     // second composing midpoint.
347     if (right == null) {
348       throw new IllegalArgumentException("right cell can not be null");
349     }
350     if (left == null) {
351       return right;
352     }
353     // If Cells from meta table, don't mess around. meta table Cells have schema
354     // (table,startrow,hash) so can't be treated as plain byte arrays. Just skip
355     // out without
356     // trying to do this optimization.
357     if (comparator instanceof MetaCellComparator) {
358       return right;
359     }
360     int diff = comparator.compareRows(left, right);
361     if (diff > 0) {
362       throw new IllegalArgumentException("Left row sorts after right row; left="
363           + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
364     }
365     if (diff < 0) {
366       // Left row is < right row.
367       byte[] midRow = getMinimumMidpointArray(left.getRowArray(), left.getRowOffset(),
368           left.getRowLength(), right.getRowArray(), right.getRowOffset(), right.getRowLength());
369       // If midRow is null, just return 'right'. Can't do optimization.
370       if (midRow == null)
371         return right;
372       return CellUtil.createCell(midRow);
373     }
374     // Rows are same. Compare on families.
375     int lFamOffset = left.getFamilyOffset();
376     int rFamOffset = right.getFamilyOffset();
377     int lFamLength = left.getFamilyLength();
378     int rFamLength = right.getFamilyLength();
379     diff = CellComparator.compareFamilies(left, right);
380     if (diff > 0) {
381       throw new IllegalArgumentException("Left family sorts after right family; left="
382           + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
383     }
384     if (diff < 0) {
385       byte[] midRow = getMinimumMidpointArray(left.getFamilyArray(), lFamOffset,
386           lFamLength, right.getFamilyArray(), rFamOffset,
387           rFamLength);
388       // If midRow is null, just return 'right'. Can't do optimization.
389       if (midRow == null)
390         return right;
391       // Return new Cell where we use right row and then a mid sort family.
392       return CellUtil.createCell(right.getRowArray(), right.getRowOffset(), right.getRowLength(),
393           midRow, 0, midRow.length, HConstants.EMPTY_BYTE_ARRAY, 0,
394           HConstants.EMPTY_BYTE_ARRAY.length);
395     }
396     // Families are same. Compare on qualifiers.
397     diff = CellComparator.compareQualifiers(left, right);
398     if (diff > 0) {
399       throw new IllegalArgumentException("Left qualifier sorts after right qualifier; left="
400           + CellUtil.getCellKeyAsString(left) + ", right=" + CellUtil.getCellKeyAsString(right));
401     }
402     if (diff < 0) {
403       byte[] midRow = getMinimumMidpointArray(left.getQualifierArray(), left.getQualifierOffset(),
404           left.getQualifierLength(), right.getQualifierArray(), right.getQualifierOffset(),
405           right.getQualifierLength());
406       // If midRow is null, just return 'right'. Can't do optimization.
407       if (midRow == null)
408         return right;
409       // Return new Cell where we use right row and family and then a mid sort
410       // qualifier.
411       return CellUtil.createCell(right.getRowArray(), right.getRowOffset(), right.getRowLength(),
412           right.getFamilyArray(), right.getFamilyOffset(), right.getFamilyLength(), midRow, 0,
413           midRow.length);
414     }
415     // No opportunity for optimization. Just return right key.
416     return right;
417   }

419   /**
420    * @param leftArray
421    * @param leftOffset
422    * @param leftLength
423    * @param rightArray
424    * @param rightOffset
425    * @param rightLength
426    * @return Return a new array that is between left and right and minimally
427    *         sized else just return null as indicator that we could not create a
428    *         mid point.
429    */
430   private static byte[] getMinimumMidpointArray(final byte[] leftArray, final int leftOffset,
431       final int leftLength, final byte[] rightArray, final int rightOffset, final int rightLength) {
432     // rows are different
433     int minLength = leftLength < rightLength ? leftLength : rightLength;
434     int diffIdx = 0;
435     while (diffIdx < minLength
436         && leftArray[leftOffset + diffIdx] == rightArray[rightOffset + diffIdx]) {
437       diffIdx++;
438     }
439     byte[] minimumMidpointArray = null;
440     if (diffIdx >= minLength) {
441       // leftKey's row is prefix of rightKey's.
442       minimumMidpointArray = new byte[diffIdx + 1];
443       System.arraycopy(rightArray, rightOffset, minimumMidpointArray, 0, diffIdx + 1);
444     } else {
445       int diffByte = leftArray[leftOffset + diffIdx];
446       if ((0xff & diffByte) < 0xff && (diffByte + 1) < (rightArray[rightOffset + diffIdx] & 0xff)) {
447         minimumMidpointArray = new byte[diffIdx + 1];
448         System.arraycopy(leftArray, leftOffset, minimumMidpointArray, 0, diffIdx);
449         minimumMidpointArray[diffIdx] = (byte) (diffByte + 1);
450       } else {
451         minimumMidpointArray = new byte[diffIdx + 1];
452         System.arraycopy(rightArray, rightOffset, minimumMidpointArray, 0, diffIdx + 1);
453       }
454     }
455     return minimumMidpointArray;
456   }

458   /** Gives inline block writers an opportunity to contribute blocks. */
459   private void writeInlineBlocks(boolean closing) throws IOException {
460     for (InlineBlockWriter ibw : inlineBlockWriters) {
461       while (ibw.shouldWriteBlock(closing)) {
462         long offset = outputStream.getPos();
463         boolean cacheThisBlock = ibw.getCacheOnWrite();
464         ibw.writeInlineBlock(fsBlockWriter.startWriting(
465             ibw.getInlineBlockType()));
466         fsBlockWriter.writeHeaderAndData(outputStream);
467         ibw.blockWritten(offset, fsBlockWriter.getOnDiskSizeWithHeader(),
468             fsBlockWriter.getUncompressedSizeWithoutHeader());
469         totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
470 
471         if (cacheThisBlock) {
472           doCacheOnWrite(offset);
473         }
474       }
475     }
476   }

502   /**
503    * Add a meta block to the end of the file. Call before close(). Metadata
504    * blocks are expensive. Fill one with a bunch of serialized data rather than
505    * do a metadata block per metadata instance. If metadata is small, consider
506    * adding to file info using {@link #appendFileInfo(byte[], byte[])}
507    *
508    * @param metaBlockName
509    *          name of the block
510    * @param content
511    *          will call readFields to get data later (DO NOT REUSE)
512    */
513   @Override
514   public void appendMetaBlock(String metaBlockName, Writable content) {
515     byte[] key = Bytes.toBytes(metaBlockName);
516     int i;
517     for (i = 0; i < metaNames.size(); ++i) {
518       // stop when the current key is greater than our own
519       byte[] cur = metaNames.get(i);
520       if (Bytes.BYTES_RAWCOMPARATOR.compare(cur, 0, cur.length, key, 0,
521           key.length) > 0) {
522         break;
523       }
524     }
525     metaNames.add(i, key);
526     metaData.add(i, content);
527   }

529   @Override
530   public void close() throws IOException {
531     if (outputStream == null) {
532       return;
533     }
534     // Save data block encoder metadata in the file info.
535     blockEncoder.saveMetadata(this);
536     // Write out the end of the data blocks, then write meta data blocks.
537     // followed by fileinfo, data block index and meta block index.
538 
539     finishBlock();
540     writeInlineBlocks(true);
541 
542     FixedFileTrailer trailer = new FixedFileTrailer(getMajorVersion(), getMinorVersion());
543 
544     // Write out the metadata blocks if any.
545     if (!metaNames.isEmpty()) {
546       for (int i = 0; i < metaNames.size(); ++i) {
547         // store the beginning offset
548         long offset = outputStream.getPos();
549         // write the metadata content
550         DataOutputStream dos = fsBlockWriter.startWriting(BlockType.META);
551         metaData.get(i).write(dos);
552 
553         fsBlockWriter.writeHeaderAndData(outputStream);
554         totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
555 
556         // Add the new meta block to the meta index.
557         metaBlockIndexWriter.addEntry(metaNames.get(i), offset,
558             fsBlockWriter.getOnDiskSizeWithHeader());
559       }
560     }
561 
562     // Load-on-open section.
563 
564     // Data block index.
565     //
566     // In version 2, this section of the file starts with the root level data
567     // block index. We call a function that writes intermediate-level blocks
568     // first, then root level, and returns the offset of the root level block
569     // index.
570 
571     long rootIndexOffset = dataBlockIndexWriter.writeIndexBlocks(outputStream);
572     trailer.setLoadOnOpenOffset(rootIndexOffset);
573 
574     // Meta block index.
575     metaBlockIndexWriter.writeSingleLevelIndex(fsBlockWriter.startWriting(
576         BlockType.ROOT_INDEX), "meta");
577     fsBlockWriter.writeHeaderAndData(outputStream);
578     totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
579 
580     if (this.hFileContext.isIncludesMvcc()) {
581       appendFileInfo(MAX_MEMSTORE_TS_KEY, Bytes.toBytes(maxMemstoreTS));
582       appendFileInfo(KEY_VALUE_VERSION, Bytes.toBytes(KEY_VALUE_VER_WITH_MEMSTORE));
583     }
584 
585     // File info
586     writeFileInfo(trailer, fsBlockWriter.startWriting(BlockType.FILE_INFO));
587     fsBlockWriter.writeHeaderAndData(outputStream);
588     totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
589 
590     // Load-on-open data supplied by higher levels, e.g. Bloom filters.
591     for (BlockWritable w : additionalLoadOnOpenData){
592       fsBlockWriter.writeBlock(w, outputStream);
593       totalUncompressedBytes += fsBlockWriter.getUncompressedSizeWithHeader();
594     }
595 
596     // Now finish off the trailer.
597     trailer.setNumDataIndexLevels(dataBlockIndexWriter.getNumLevels());
598     trailer.setUncompressedDataIndexSize(
599         dataBlockIndexWriter.getTotalUncompressedSize());
600     trailer.setFirstDataBlockOffset(firstDataBlockOffset);
601     trailer.setLastDataBlockOffset(lastDataBlockOffset);
602     trailer.setComparatorClass(comparator.getClass());
603     trailer.setDataIndexCount(dataBlockIndexWriter.getNumRootEntries());
604 
605 
606     finishClose(trailer);
607 
608     fsBlockWriter.release();
609   }

626   private void addBloomFilter(final BloomFilterWriter bfw,
627       final BlockType blockType) {
628     if (bfw.getKeyCount() <= 0)
629       return;
630 
631     if (blockType != BlockType.GENERAL_BLOOM_META &&
632         blockType != BlockType.DELETE_FAMILY_BLOOM_META) {
633       throw new RuntimeException("Block Type: " + blockType.toString() +
634           "is not supported");
635     }
636     additionalLoadOnOpenData.add(new BlockWritable() {
637       @Override
638       public BlockType getBlockType() {
639         return blockType;
640       }
641 
642       @Override
643       public void writeToBlock(DataOutput out) throws IOException {
644         bfw.getMetaWriter().write(out);

657   /**
658    * Add key/value to file. Keys must be added in an order that agrees with the
659    * Comparator passed on construction.
660    *
661    * @param cell
662    *          Cell to add. Cannot be empty nor null.
663    * @throws IOException
664    */
665   @Override
666   public void append(final Cell cell) throws IOException {
667     // checkKey uses comparator to check we are writing in order.
668     boolean dupKey = checkKey(cell);
669     if (!dupKey) {
670       checkBlockBoundary();
671     }
672 
673     if (!fsBlockWriter.isWriting()) {
674       newBlock();
675     }
676 
677     fsBlockWriter.write(cell);
678 
679     totalKeyLength += CellUtil.estimatedSerializedSizeOfKey(cell);
680     totalValueLength += cell.getValueLength();
681 
682     // Are we the first key in this block?
683     if (firstCellInBlock == null) {
684       // If cell is big, block will be closed and this firstCellInBlock reference will only last
685       // a short while.
686       firstCellInBlock = cell;
687     }
688 
689     // TODO: What if cell is 10MB and we write infrequently? We hold on to cell here indefinetly?
690     lastCell = cell;
691     entryCount++;
692     this.maxMemstoreTS = Math.max(this.maxMemstoreTS, cell.getSequenceId());
693     int tagsLength = cell.getTagsLength();
694     if (tagsLength > this.maxTagsLength) {
695       this.maxTagsLength = tagsLength;
696     }
697   }

699   protected void finishFileInfo() throws IOException {
700     if (lastCell != null) {
701       // Make a copy. The copy is stuffed into our fileinfo map. Needs a clean
702       // byte buffer. Won't take a tuple.
703       byte [] lastKey = CellUtil.getCellKeySerializedAsKeyValueKey(this.lastCell);
704       fileInfo.append(FileInfo.LASTKEY, lastKey, false);
705     }
706 
707     // Average key length.
708     int avgKeyLen =
709         entryCount == 0 ? 0 : (int) (totalKeyLength / entryCount);
710     fileInfo.append(FileInfo.AVG_KEY_LEN, Bytes.toBytes(avgKeyLen), false);
711     fileInfo.append(FileInfo.CREATE_TIME_TS, Bytes.toBytes(hFileContext.getFileCreateTime()),
712       false);
713 
714     // Average value length.
715     int avgValueLen =
716         entryCount == 0 ? 0 : (int) (totalValueLength / entryCount);
717     fileInfo.append(FileInfo.AVG_VALUE_LEN, Bytes.toBytes(avgValueLen), false);
718     if (hFileContext.getDataBlockEncoding() == DataBlockEncoding.PREFIX_TREE) {
719       // In case of Prefix Tree encoding, we always write tags information into HFiles even if all
720       // KVs are having no tags.
721       fileInfo.append(FileInfo.MAX_TAGS_LEN, Bytes.toBytes(this.maxTagsLength), false);
722     } else if (hFileContext.isIncludesTags()) {
723       // When tags are not being written in this file, MAX_TAGS_LEN is excluded
724       // from the FileInfo
725       fileInfo.append(FileInfo.MAX_TAGS_LEN, Bytes.toBytes(this.maxTagsLength), false);
726       boolean tagsCompressed = (hFileContext.getDataBlockEncoding() != DataBlockEncoding.NONE)
727         && hFileContext.isCompressTags();
728       fileInfo.append(FileInfo.TAGS_COMPRESSED, Bytes.toBytes(tagsCompressed), false);
729     }
730   }
