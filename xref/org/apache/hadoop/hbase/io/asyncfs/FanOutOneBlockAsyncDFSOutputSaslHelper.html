227   private static TransparentCryptoHelper createTransparentCryptoHelper()
228       throws NoSuchMethodException {
229     Method decryptEncryptedDataEncryptionKeyMethod = DFSClient.class
230         .getDeclaredMethod("decryptEncryptedDataEncryptionKey", FileEncryptionInfo.class);
231     decryptEncryptedDataEncryptionKeyMethod.setAccessible(true);
232     return new TransparentCryptoHelper() {
233
234       @Override
235       public Encryptor createEncryptor(Configuration conf, FileEncryptionInfo feInfo,
236           DFSClient client) throws IOException {
237         try {
238           KeyVersion decryptedKey =
239               (KeyVersion) decryptEncryptedDataEncryptionKeyMethod.invoke(client, feInfo);
240           CryptoCodec cryptoCodec = CryptoCodec.getInstance(conf, feInfo.getCipherSuite());
241           Encryptor encryptor = cryptoCodec.createEncryptor();
242           encryptor.init(decryptedKey.getMaterial(), feInfo.getIV());
243           return encryptor;
244         } catch (InvocationTargetException e) {
245           Throwables.propagateIfPossible(e.getTargetException(), IOException.class);
246           throw new RuntimeException(e.getTargetException());
247         } catch (GeneralSecurityException e) {
248           throw new IOException(e);
249         } catch (IllegalAccessException e) {
250           throw new RuntimeException(e);
251         }
252       }
253     };
254   }

273   private static final class SaslClientCallbackHandler implements CallbackHandler {
274
275     private final char[] password;
276     private final String userName;
277
278     /**
279      * Creates a new SaslClientCallbackHandler.
280      * @param userName SASL user name
281      * @Param password SASL password
282      */
283     public SaslClientCallbackHandler(String userName, char[] password) {
284       this.password = password;
285       this.userName = userName;
286     }
287
288     @Override
289     public void handle(Callback[] callbacks) throws IOException, UnsupportedCallbackException {
290       NameCallback nc = null;
291       PasswordCallback pc = null;
292       RealmCallback rc = null;
293       for (Callback callback : callbacks) {
294         if (callback instanceof RealmChoiceCallback) {
295           continue;
296         } else if (callback instanceof NameCallback) {
297           nc = (NameCallback) callback;
298         } else if (callback instanceof PasswordCallback) {
299           pc = (PasswordCallback) callback;
300         } else if (callback instanceof RealmCallback) {
301           rc = (RealmCallback) callback;
302         } else {
303           throw new UnsupportedCallbackException(callback, "Unrecognized SASL client callback");
304         }
305       }
306       if (nc != null) {
307         nc.setName(userName);
308       }
309       if (pc != null) {
310         pc.setPassword(password);
311       }
312       if (rc != null) {
313         rc.setText(rc.getDefaultText());
314       }
315     }
316   }

318   private static final class SaslNegotiateHandler extends ChannelDuplexHandler {
319
320     private final Configuration conf;
321
322     private final Map<String, String> saslProps;
323
324     private final SaslClient saslClient;
325
326     private final int timeoutMs;

328     private final Promise<Void> promise;
329
330     private int step = 0;
331
332     public SaslNegotiateHandler(Configuration conf, String username, char[] password,
333         Map<String, String> saslProps, int timeoutMs, Promise<Void> promise) throws SaslException {
334       this.conf = conf;
335       this.saslProps = saslProps;
336       this.saslClient = Sasl.createSaslClient(new String[] { MECHANISM }, username, PROTOCOL,
337         SERVER_NAME, saslProps, new SaslClientCallbackHandler(username, password));
338       this.timeoutMs = timeoutMs;
339       this.promise = promise;

341
342     private void sendSaslMessage(ChannelHandlerContext ctx, byte[] payload) throws IOException {
343       sendSaslMessage(ctx, payload, null);
344     }
345
346     private List<CipherOption> getCipherOptions() throws IOException {
347       // Negotiate cipher suites if configured. Currently, the only supported
348       // cipher suite is AES/CTR/NoPadding, but the protocol allows multiple
349       // values for future expansion.
350       String cipherSuites = conf.get(DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY);
351       if (StringUtils.isBlank(cipherSuites)) {
352         return null;
353       }
354       if (!cipherSuites.equals(CipherSuite.AES_CTR_NOPADDING.getName())) {
355         throw new IOException(String.format("Invalid cipher suite, %s=%s",
356           DFS_ENCRYPT_DATA_TRANSFER_CIPHER_SUITES_KEY, cipherSuites));
357       }
358       return Arrays.asList(new CipherOption(CipherSuite.AES_CTR_NOPADDING));
359     }
360
361     private void sendSaslMessage(ChannelHandlerContext ctx, byte[] payload,
362         List<CipherOption> options) throws IOException {
363       DataTransferEncryptorMessageProto.Builder builder =
364           DataTransferEncryptorMessageProto.newBuilder();
365       builder.setStatus(DataTransferEncryptorStatus.SUCCESS);
366       if (payload != null) {
367         builder.setPayload(ByteStringer.wrap(payload));
368       }
369       if (options != null) {
370         builder.addAllCipherOption(PB_HELPER.convertCipherOptions(options));
371       }
372       DataTransferEncryptorMessageProto proto = builder.build();
373       int size = proto.getSerializedSize();
374       size += CodedOutputStream.computeRawVarint32Size(size);
375       ByteBuf buf = ctx.alloc().buffer(size);
376       proto.writeDelimitedTo(new ByteBufOutputStream(buf));
377       ctx.write(buf);
378     }
379
380     @Override
381     public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
382       ctx.write(ctx.alloc().buffer(4).writeInt(SASL_TRANSFER_MAGIC_NUMBER));
383       sendSaslMessage(ctx, new byte[0]);
384       ctx.flush();
385       step++;
386     }
387
388     @Override
389     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
390       saslClient.dispose();
391     }
392
393     private void check(DataTransferEncryptorMessageProto proto) throws IOException {
394       if (proto.getStatus() == DataTransferEncryptorStatus.ERROR_UNKNOWN_KEY) {
395         throw new InvalidEncryptionKeyException(proto.getMessage());
396       } else if (proto.getStatus() == DataTransferEncryptorStatus.ERROR) {
397         throw new IOException(proto.getMessage());
398       }
399     }
400
401     private String getNegotiatedQop() {
402       return (String) saslClient.getNegotiatedProperty(Sasl.QOP);
403     }
404
405     private boolean isNegotiatedQopPrivacy() {
406       String qop = getNegotiatedQop();
407       return qop != null && "auth-conf".equalsIgnoreCase(qop);
408     }
409
410     private boolean requestedQopContainsPrivacy() {
411       Set<String> requestedQop =
412           ImmutableSet.copyOf(Arrays.asList(saslProps.get(Sasl.QOP).split(",")));
413       return requestedQop.contains("auth-conf");
414     }
415
416     private void checkSaslComplete() throws IOException {
417       if (!saslClient.isComplete()) {
418         throw new IOException("Failed to complete SASL handshake");
419       }
420       Set<String> requestedQop =
421           ImmutableSet.copyOf(Arrays.asList(saslProps.get(Sasl.QOP).split(",")));
422       String negotiatedQop = getNegotiatedQop();
423       LOG.debug(
424         "Verifying QOP, requested QOP = " + requestedQop + ", negotiated QOP = " + negotiatedQop);
425       if (!requestedQop.contains(negotiatedQop)) {
426         throw new IOException(String.format("SASL handshake completed, but "
427             + "channel does not have acceptable quality of protection, "
428             + "requested = %s, negotiated = %s",
429           requestedQop, negotiatedQop));
430       }
431     }
432
433     private boolean useWrap() {
434       String qop = (String) saslClient.getNegotiatedProperty(Sasl.QOP);
435       return qop != null && !"auth".equalsIgnoreCase(qop);
436     }

438     private CipherOption unwrap(CipherOption option, SaslClient saslClient) throws IOException {
439       byte[] inKey = option.getInKey();
440       if (inKey != null) {
441         inKey = saslClient.unwrap(inKey, 0, inKey.length);
442       }
443       byte[] outKey = option.getOutKey();
444       if (outKey != null) {
445         outKey = saslClient.unwrap(outKey, 0, outKey.length);
446       }
447       return new CipherOption(option.getCipherSuite(), inKey, option.getInIv(), outKey,
448           option.getOutIv());
449     }
450
451     private CipherOption getCipherOption(DataTransferEncryptorMessageProto proto,
452         boolean isNegotiatedQopPrivacy, SaslClient saslClient) throws IOException {
453       List<CipherOption> cipherOptions =
454           PB_HELPER.convertCipherOptionProtos(proto.getCipherOptionList());
455       if (cipherOptions == null || cipherOptions.isEmpty()) {
456         return null;
457       }
458       CipherOption cipherOption = cipherOptions.get(0);
459       return isNegotiatedQopPrivacy ? unwrap(cipherOption, saslClient) : cipherOption;
460     }
461
462     @Override
463     public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
464       if (msg instanceof DataTransferEncryptorMessageProto) {
465         DataTransferEncryptorMessageProto proto = (DataTransferEncryptorMessageProto) msg;
466         check(proto);
467         byte[] challenge = proto.getPayload().toByteArray();
468         byte[] response = saslClient.evaluateChallenge(challenge);
469         switch (step) {
470           case 1: {
471             List<CipherOption> cipherOptions = null;
472             if (requestedQopContainsPrivacy()) {
473               cipherOptions = getCipherOptions();
474             }
475             sendSaslMessage(ctx, response, cipherOptions);
476             ctx.flush();
477             step++;
478             break;
479           }
480           case 2: {
481             assert response == null;
482             checkSaslComplete();
483             CipherOption cipherOption =
484                 getCipherOption(proto, isNegotiatedQopPrivacy(), saslClient);
485             ChannelPipeline p = ctx.pipeline();
486             while (p.first() != null) {
487               p.removeFirst();
488             }
489             if (cipherOption != null) {
490               CryptoCodec codec = CryptoCodec.getInstance(conf, cipherOption.getCipherSuite());
491               p.addLast(new EncryptHandler(codec, cipherOption.getInKey(), cipherOption.getInIv()),
492                 new DecryptHandler(codec, cipherOption.getOutKey(), cipherOption.getOutIv()));
493             } else {
494               if (useWrap()) {
495                 p.addLast(new SaslWrapHandler(saslClient),
496                   new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4),
497                   new SaslUnwrapHandler(saslClient));
498               }
499             }
500             promise.trySuccess(null);
501             break;
502           }
503           default:
504             throw new IllegalArgumentException("Unrecognized negotiation step: " + step);
505         }
506       } else {
507         ctx.fireChannelRead(msg);
508       }
509     }
510
511     @Override
512     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
513       promise.tryFailure(cause);
514     }
515
516     @Override
517     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
518       if (evt instanceof IdleStateEvent && ((IdleStateEvent) evt).state() == READER_IDLE) {
519         promise.tryFailure(new IOException("Timeout(" + timeoutMs + "ms) waiting for response"));
520       } else {
521         super.userEventTriggered(ctx, evt);
522       }
523     }
524   }

548   private static final class SaslWrapHandler extends ChannelOutboundHandlerAdapter {
549
550     private final SaslClient saslClient;
551
552     private CompositeByteBuf cBuf;
553
554     public SaslWrapHandler(SaslClient saslClient) {
555       this.saslClient = saslClient;
556     }
557
558     @Override
559     public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
560       cBuf = new CompositeByteBuf(ctx.alloc(), false, Integer.MAX_VALUE);
561     }
562
563     @Override
564     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
565         throws Exception {
566       if (msg instanceof ByteBuf) {
567         ByteBuf buf = (ByteBuf) msg;
568         cBuf.addComponent(buf);
569         cBuf.writerIndex(cBuf.writerIndex() + buf.readableBytes());
570       } else {
571         ctx.write(msg);
572       }
573     }
574
575     @Override
576     public void flush(ChannelHandlerContext ctx) throws Exception {
577       if (cBuf.isReadable()) {
578         byte[] b = new byte[cBuf.readableBytes()];
579         cBuf.readBytes(b);
580         cBuf.discardReadComponents();
581         byte[] wrapped = saslClient.wrap(b, 0, b.length);
582         ByteBuf buf = ctx.alloc().ioBuffer(4 + wrapped.length);
583         buf.writeInt(wrapped.length);
584         buf.writeBytes(wrapped);
585         ctx.write(buf);
586       }
587       ctx.flush();
588     }
589
590     @Override
591     public void close(ChannelHandlerContext ctx, ChannelPromise promise) throws Exception {
592       cBuf.release();
593       cBuf = null;
594     }
595   }

597   private static final class DecryptHandler extends SimpleChannelInboundHandler<ByteBuf> {
598
599     private final Decryptor decryptor;
600
601     public DecryptHandler(CryptoCodec codec, byte[] key, byte[] iv)
602         throws GeneralSecurityException, IOException {
603       this.decryptor = codec.createDecryptor();
604       this.decryptor.init(key, Arrays.copyOf(iv, iv.length));
605     }
606
607     @Override
608     protected void channelRead0(ChannelHandlerContext ctx, ByteBuf msg) throws Exception {
609       ByteBuf inBuf;
610       boolean release = false;
611       if (msg.nioBufferCount() == 1) {
612         inBuf = msg;
613       } else {
614         inBuf = ctx.alloc().directBuffer(msg.readableBytes());
615         msg.readBytes(inBuf);
616         release = true;
617       }
618       ByteBuffer inBuffer = inBuf.nioBuffer();
619       ByteBuf outBuf = ctx.alloc().directBuffer(inBuf.readableBytes());
620       ByteBuffer outBuffer = outBuf.nioBuffer(0, inBuf.readableBytes());
621       decryptor.decrypt(inBuffer, outBuffer);
622       outBuf.writerIndex(inBuf.readableBytes());
623       if (release) {
624         inBuf.release();
625       }
626       ctx.fireChannelRead(outBuf);
627     }

630   private static final class EncryptHandler extends MessageToByteEncoder<ByteBuf> {
631
632     private final Encryptor encryptor;
633
634     public EncryptHandler(CryptoCodec codec, byte[] key, byte[] iv)
635         throws GeneralSecurityException, IOException {
636       this.encryptor = codec.createEncryptor();
637       this.encryptor.init(key, Arrays.copyOf(iv, iv.length));
638     }
639
640     @Override
641     protected ByteBuf allocateBuffer(ChannelHandlerContext ctx, ByteBuf msg, boolean preferDirect)
642         throws Exception {
643       if (preferDirect) {
644         return ctx.alloc().directBuffer(msg.readableBytes());
645       } else {
646         return ctx.alloc().buffer(msg.readableBytes());
647       }
648     }
649
650     @Override
651     protected void encode(ChannelHandlerContext ctx, ByteBuf msg, ByteBuf out) throws Exception {
652       ByteBuf inBuf;
653       boolean release = false;
654       if (msg.nioBufferCount() == 1) {
655         inBuf = msg;
656       } else {
657         inBuf = ctx.alloc().directBuffer(msg.readableBytes());
658         msg.readBytes(inBuf);
659         release = true;
660       }
661       ByteBuffer inBuffer = inBuf.nioBuffer();
662       ByteBuffer outBuffer = out.nioBuffer(0, inBuf.readableBytes());
663       encryptor.encrypt(inBuffer, outBuffer);
664       out.writerIndex(inBuf.readableBytes());
665       if (release) {
666         inBuf.release();
667       }
668     }

709   static void trySaslNegotiate(Configuration conf, Channel channel, DatanodeInfo dnInfo,
710       int timeoutMs, DFSClient client, Token<BlockTokenIdentifier> accessToken,
711       Promise<Void> saslPromise) throws IOException {
712     SaslDataTransferClient saslClient = client.getSaslDataTransferClient();
713     SaslPropertiesResolver saslPropsResolver = SASL_ADAPTOR.getSaslPropsResolver(saslClient);
714     TrustedChannelResolver trustedChannelResolver =
715         SASL_ADAPTOR.getTrustedChannelResolver(saslClient);
716     AtomicBoolean fallbackToSimpleAuth = SASL_ADAPTOR.getFallbackToSimpleAuth(saslClient);
717     InetAddress addr = ((InetSocketAddress) channel.remoteAddress()).getAddress();
718     if (trustedChannelResolver.isTrusted() || trustedChannelResolver.isTrusted(addr)) {
719       saslPromise.trySuccess(null);
720       return;
721     }
722     DataEncryptionKey encryptionKey = client.newDataEncryptionKey();
723     if (encryptionKey != null) {
724       if (LOG.isDebugEnabled()) {
725         LOG.debug(
726           "SASL client doing encrypted handshake for addr = " + addr + ", datanodeId = " + dnInfo);
727       }
728       doSaslNegotiation(conf, channel, timeoutMs, getUserNameFromEncryptionKey(encryptionKey),
729         encryptionKeyToPassword(encryptionKey.encryptionKey),
730         createSaslPropertiesForEncryption(encryptionKey.encryptionAlgorithm), saslPromise);
731     } else if (!UserGroupInformation.isSecurityEnabled()) {
732       if (LOG.isDebugEnabled()) {
733         LOG.debug("SASL client skipping handshake in unsecured configuration for addr = " + addr
734             + ", datanodeId = " + dnInfo);
735       }
736       saslPromise.trySuccess(null);
737     } else if (dnInfo.getXferPort() < 1024) {
738       if (LOG.isDebugEnabled()) {
739         LOG.debug("SASL client skipping handshake in secured configuration with "
740             + "privileged port for addr = " + addr + ", datanodeId = " + dnInfo);
741       }
742       saslPromise.trySuccess(null);
743     } else if (fallbackToSimpleAuth != null && fallbackToSimpleAuth.get()) {
744       if (LOG.isDebugEnabled()) {
745         LOG.debug("SASL client skipping handshake in secured configuration with "
746             + "unsecured cluster for addr = " + addr + ", datanodeId = " + dnInfo);
747       }
748       saslPromise.trySuccess(null);
749     } else if (saslPropsResolver != null) {
750       if (LOG.isDebugEnabled()) {
751         LOG.debug(
752           "SASL client doing general handshake for addr = " + addr + ", datanodeId = " + dnInfo);
753       }
754       doSaslNegotiation(conf, channel, timeoutMs, buildUsername(accessToken),
755         buildClientPassword(accessToken), saslPropsResolver.getClientProperties(addr), saslPromise);
756     } else {
757       // It's a secured cluster using non-privileged ports, but no SASL. The only way this can
758       // happen is if the DataNode has ignore.secure.ports.for.testing configured, so this is a rare
759       // edge case.
760       if (LOG.isDebugEnabled()) {
761         LOG.debug("SASL client skipping handshake in secured configuration with no SASL "
762             + "protection configured for addr = " + addr + ", datanodeId = " + dnInfo);
763       }
764       saslPromise.trySuccess(null);
765     }
766   }
