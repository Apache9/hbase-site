42    /**
43     * Returns tag value in a new byte array.
44     * Primarily for use client-side. If server-side, use
45     * {@link Tag#getValueArray()} with appropriate {@link Tag#getValueOffset()}
46     * and {@link Tag#getValueLength()} instead to save on allocations.
47     *
48     * @param tag The Tag whose value to be returned
49     * @return tag value in a new byte array.
50     */
51    public static byte[] cloneValue(Tag tag) {
52      int tagLength = tag.getValueLength();
53      byte[] tagArr = new byte[tagLength];
54      if (tag.hasArray()) {
55        Bytes.putBytes(tagArr, 0, tag.getValueArray(), tag.getValueOffset(), tagLength);
56      } else {
57        ByteBufferUtils.copyFromBufferToArray(tagArr, tag.getValueByteBuffer(), tag.getValueOffset(),
58            0, tagLength);
59      }
60      return tagArr;
61    }

63    /**
64     * Creates list of tags from given byte array, expected that it is in the expected tag format.
65     *
66     * @param b The byte array
67     * @param offset The offset in array where tag bytes begin
68     * @param length Total length of all tags bytes
69     * @return List of tags
70     */
71    public static List<Tag> asList(byte[] b, int offset, int length) {
72      List<Tag> tags = new ArrayList<Tag>();
73      int pos = offset;
74      while (pos < offset + length) {
75        int tagLen = Bytes.readAsInt(b, pos, TAG_LENGTH_SIZE);
76        tags.add(new ArrayBackedTag(b, pos, tagLen + TAG_LENGTH_SIZE));
77        pos += TAG_LENGTH_SIZE + tagLen;
78      }
79      return tags;
80    }

82    /**
83     * Creates list of tags from given ByteBuffer, expected that it is in the expected tag format.
84     *
85     * @param b The ByteBuffer
86     * @param offset The offset in ByteBuffer where tag bytes begin
87     * @param length Total length of all tags bytes
88     * @return List of tags
89     */
90    public static List<Tag> asList(ByteBuffer b, int offset, int length) {
91      List<Tag> tags = new ArrayList<Tag>();
92      int pos = offset;
93      while (pos < offset + length) {
94        int tagLen = ByteBufferUtils.readAsInt(b, pos, TAG_LENGTH_SIZE);
95        tags.add(new OffheapTag(b, pos, tagLen + TAG_LENGTH_SIZE));
96        pos += TAG_LENGTH_SIZE + tagLen;
97      }
98      return tags;

101   /**
102    * Write a list of tags into a byte array
103    *
104    * @param tags The list of tags
105    * @return the serialized tag data as bytes
106    */
107   public static byte[] fromList(List<Tag> tags) {
108     if (tags == null || tags.isEmpty()) {
109       return HConstants.EMPTY_BYTE_ARRAY;
110     }
111     int length = 0;
112     for (Tag tag : tags) {
113       length += tag.getValueLength() + Tag.INFRASTRUCTURE_SIZE;
114     }
115     byte[] b = new byte[length];
116     int pos = 0;
117     int tlen;
118     for (Tag tag : tags) {
119       tlen = tag.getValueLength();
120       pos = Bytes.putAsShort(b, pos, tlen + Tag.TYPE_LENGTH_SIZE);
121       pos = Bytes.putByte(b, pos, tag.getType());
122       if (tag.hasArray()) {
123         pos = Bytes.putBytes(b, pos, tag.getValueArray(), tag.getValueOffset(), tlen);
124       } else {
125         ByteBufferUtils.copyFromBufferToArray(b, tag.getValueByteBuffer(), tag.getValueOffset(),
126             pos, tlen);
127         pos += tlen;
128       }
129     }
130     return b;
131   }

169   /**
170    * Matches the value part of given tags
171    * @param t1 Tag to match the value
172    * @param t2 Tag to match the value
173    * @return True if values of both tags are same.
174    */
175   public static boolean matchingValue(Tag t1, Tag t2) {
176     if (t1.hasArray() && t2.hasArray()) {
177       return Bytes.equals(t1.getValueArray(), t1.getValueOffset(), t1.getValueLength(),
178           t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
179     }
180     if (t1.hasArray()) {
181       return ByteBufferUtils.equals(t2.getValueByteBuffer(), t2.getValueOffset(),
182           t2.getValueLength(), t1.getValueArray(), t1.getValueOffset(), t1.getValueLength());
183     }
184     if (t2.hasArray()) {
185       return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(),
186           t1.getValueLength(), t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
187     }
188     return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(), t1.getValueLength(),
189         t2.getValueByteBuffer(), t2.getValueOffset(), t2.getValueLength());
190   }

228   /**
229    * Add to <code>tagsOrNull</code> any Tags <code>cell</code> is carrying or null if none.
230    */
231   public static List<Tag> carryForwardTags(final List<Tag> tagsOrNull, final Cell cell) {
232     Iterator<Tag> itr = CellUtil.tagsIterator(cell);
233     if (itr == EMPTY_TAGS_ITR) {
234       // If no Tags, return early.
235       return tagsOrNull;
236     }
237     List<Tag> tags = tagsOrNull;
238     if (tags == null) {
239       tags = new ArrayList<Tag>();
240     }
241     while (itr.hasNext()) {
242       tags.add(itr.next());
243     }
244     return tags;
245   }

247   /**
248    * @return Carry forward the TTL tag.
249    */
250   public static List<Tag> carryForwardTTLTag(final List<Tag> tagsOrNull, final long ttl) {
251     if (ttl == Long.MAX_VALUE) {
252       return tagsOrNull;
253     }
254     List<Tag> tags = tagsOrNull;
255     // If we are making the array in here, given we are the last thing checked, we'll be only thing
256     // in the array so set its size to '1' (I saw this being done in earlier version of
257     // tag-handling).
258     if (tags == null) {
259       tags = new ArrayList<Tag>(1);
260     }
261     tags.add(new ArrayBackedTag(TagType.TTL_TAG_TYPE, Bytes.toBytes(ttl)));
262     return tags;
263   }
