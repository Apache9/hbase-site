41    /**
42     * Returns tag value in a new byte array.
43     * Primarily for use client-side. If server-side, use
44     * {@link Tag#getValueArray()} with appropriate {@link Tag#getValueOffset()}
45     * and {@link Tag#getValueLength()} instead to save on allocations.
46     *
47     * @param tag The Tag whose value to be returned
48     * @return tag value in a new byte array.
49     */
50    public static byte[] cloneValue(Tag tag) {
51      int tagLength = tag.getValueLength();
52      byte[] tagArr = new byte[tagLength];
53      if (tag.hasArray()) {
54        Bytes.putBytes(tagArr, 0, tag.getValueArray(), tag.getValueOffset(), tagLength);
55      } else {
56        ByteBufferUtils.copyFromBufferToArray(tagArr, tag.getValueByteBuffer(), tag.getValueOffset(),
57            0, tagLength);
58      }
59      return tagArr;
60    }

62    /**
63     * Creates list of tags from given byte array, expected that it is in the expected tag format.
64     *
65     * @param b The byte array
66     * @param offset The offset in array where tag bytes begin
67     * @param length Total length of all tags bytes
68     * @return List of tags
69     */
70    public static List<Tag> asList(byte[] b, int offset, int length) {
71      List<Tag> tags = new ArrayList<Tag>();
72      int pos = offset;
73      while (pos < offset + length) {
74        int tagLen = Bytes.readAsInt(b, pos, TAG_LENGTH_SIZE);
75        tags.add(new ArrayBackedTag(b, pos, tagLen + TAG_LENGTH_SIZE));
76        pos += TAG_LENGTH_SIZE + tagLen;
77      }
78      return tags;
79    }

81    /**
82     * Creates list of tags from given ByteBuffer, expected that it is in the expected tag format.
83     *
84     * @param b The ByteBuffer
85     * @param offset The offset in ByteBuffer where tag bytes begin
86     * @param length Total length of all tags bytes
87     * @return List of tags
88     */
89    public static List<Tag> asList(ByteBuffer b, int offset, int length) {
90      List<Tag> tags = new ArrayList<Tag>();
91      int pos = offset;
92      while (pos < offset + length) {
93        int tagLen = ByteBufferUtils.readAsInt(b, pos, TAG_LENGTH_SIZE);
94        tags.add(new OffheapTag(b, pos, tagLen + TAG_LENGTH_SIZE));
95        pos += TAG_LENGTH_SIZE + tagLen;
96      }
97      return tags;
98    }

100   /**
101    * Write a list of tags into a byte array
102    *
103    * @param tags The list of tags
104    * @return the serialized tag data as bytes
105    */
106   public static byte[] fromList(List<Tag> tags) {
107     if (tags.isEmpty()) {
108       return HConstants.EMPTY_BYTE_ARRAY;
109     }
110     int length = 0;
111     for (Tag tag : tags) {
112       length += tag.getValueLength() + Tag.INFRASTRUCTURE_SIZE;
113     }
114     byte[] b = new byte[length];
115     int pos = 0;
116     int tlen;
117     for (Tag tag : tags) {
118       tlen = tag.getValueLength();
119       pos = Bytes.putAsShort(b, pos, tlen + Tag.TYPE_LENGTH_SIZE);
120       pos = Bytes.putByte(b, pos, tag.getType());
121       if (tag.hasArray()) {
122         pos = Bytes.putBytes(b, pos, tag.getValueArray(), tag.getValueOffset(), tlen);
123       } else {
124         ByteBufferUtils.copyFromBufferToArray(b, tag.getValueByteBuffer(), tag.getValueOffset(),
125             pos, tlen);
126         pos += tlen;
127       }
128     }
129     return b;
130   }

168   /**
169    * Matches the value part of given tags
170    * @param t1 Tag to match the value
171    * @param t2 Tag to match the value
172    * @return True if values of both tags are same.
173    */
174   public static boolean matchingValue(Tag t1, Tag t2) {
175     if (t1.hasArray() && t2.hasArray()) {
176       return Bytes.equals(t1.getValueArray(), t1.getValueOffset(), t1.getValueLength(),
177           t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
178     }
179     if (t1.hasArray()) {
180       return ByteBufferUtils.equals(t2.getValueByteBuffer(), t2.getValueOffset(),
181           t2.getValueLength(), t1.getValueArray(), t1.getValueOffset(), t1.getValueLength());
182     }
183     if (t2.hasArray()) {
184       return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(),
185           t1.getValueLength(), t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
186     }
187     return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(), t1.getValueLength(),
188         t2.getValueByteBuffer(), t2.getValueOffset(), t2.getValueLength());
189   }
