19  package org.apache.hadoop.hbase.regionserver;
20  
21  import java.util.concurrent.ConcurrentMap;
22  import java.util.concurrent.ConcurrentSkipListMap;
23  import java.util.concurrent.atomic.AtomicLong;
24  
25  import org.apache.hadoop.hbase.classification.InterfaceAudience;
26  import org.apache.hadoop.hbase.util.Bytes;
27  
28  /**
29   * RegionServerAccounting keeps record of some basic real time information about
30   * the Region Server. Currently, it only keeps record the global memstore size. 
31   */
32  @InterfaceAudience.Private
33  public class RegionServerAccounting {
34  
35    private final AtomicLong globalMemstoreDataSize = new AtomicLong(0);
36    private final AtomicLong globalMemstoreHeapOverhead = new AtomicLong(0);
37
38    // Store the edits size during replaying WAL. Use this to roll back the  
39    // global memstore size once a region opening failed.
40    private final ConcurrentMap<byte[], MemstoreSize> replayEditsPerRegion =
41      new ConcurrentSkipListMap<byte[], MemstoreSize>(Bytes.BYTES_COMPARATOR);
42
43    /**
44     * @return the global Memstore size in the RegionServer
45     */
46    public long getGlobalMemstoreSize() {
47      return globalMemstoreDataSize.get();
48    }
49
50    public long getGlobalMemstoreHeapOverhead() {
51      return this.globalMemstoreHeapOverhead.get();
52    }
53
54    /**
55     * @param memStoreSize the Memstore size will be added to 
56     *        the global Memstore size 
57     */
58    public void incGlobalMemstoreSize(MemstoreSize memStoreSize) {
59      globalMemstoreDataSize.addAndGet(memStoreSize.getDataSize());
60      globalMemstoreHeapOverhead.addAndGet(memStoreSize.getHeapOverhead());
61    }
62
63    public void decGlobalMemstoreSize(MemstoreSize memStoreSize) {
64      globalMemstoreDataSize.addAndGet(-memStoreSize.getDataSize());
65      globalMemstoreHeapOverhead.addAndGet(-memStoreSize.getHeapOverhead());
66    }
67
68    /***
69     * Add memStoreSize to replayEditsPerRegion.
70     * 
71     * @param regionName region name.
72     * @param memStoreSize the Memstore size will be added to replayEditsPerRegion.
73     */
74    public void addRegionReplayEditsSize(byte[] regionName, MemstoreSize memStoreSize) {
75      MemstoreSize replayEdistsSize = replayEditsPerRegion.get(regionName);
76      // All ops on the same MemstoreSize object is going to be done by single thread, sequentially
77      // only. First calls to this method to increment the per region reply edits size and then call
78      // to either rollbackRegionReplayEditsSize or clearRegionReplayEditsSize as per the result of
79      // the region open operation. No need to handle multi thread issues on one region's entry in
80      // this Map.
81      if (replayEdistsSize == null) {
82        replayEdistsSize = new MemstoreSize();
83        replayEditsPerRegion.put(regionName, replayEdistsSize);
84      }
85      replayEdistsSize.incMemstoreSize(memStoreSize);
86    }
87
88    /**
89     * Roll back the global MemStore size for a specified region when this region
90     * can't be opened.
91     * 
92     * @param regionName the region which could not open.
93     */
94    public void rollbackRegionReplayEditsSize(byte[] regionName) {
95      MemstoreSize replayEditsSize = replayEditsPerRegion.get(regionName);
96      if (replayEditsSize != null) {
97        clearRegionReplayEditsSize(regionName);
98        decGlobalMemstoreSize(replayEditsSize);
99      }
100   }
101
102   /**
103    * Clear a region from replayEditsPerRegion.
104    * 
105    * @param regionName region name.
106    */
107   public void clearRegionReplayEditsSize(byte[] regionName) {
108     replayEditsPerRegion.remove(regionName);
109   }
110 }
