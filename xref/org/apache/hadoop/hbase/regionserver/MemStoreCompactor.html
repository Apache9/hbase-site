32  /**
33   * The ongoing MemStore Compaction manager, dispatches a solo running compaction and interrupts
34   * the compaction if requested. The compaction is interrupted and stopped by CompactingMemStore,
35   * for example when another compaction needs to be started.
36   * Prior to compaction the MemStoreCompactor evaluates
37   * the compacting ratio and aborts the compaction if it is not worthy.
38   * The MemStoreScanner is used to traverse the compaction pipeline. The MemStoreScanner
39   * is included in internal store scanner, where all compaction logic is implemented.
40   * Threads safety: It is assumed that the compaction pipeline is immutable,
41   * therefore no special synchronization is required.
42   */
43  @InterfaceAudience.Private
44  public class MemStoreCompactor {
45
46    public static final long DEEP_OVERHEAD = ClassSize
47        .align(ClassSize.OBJECT
48            + 4 * ClassSize.REFERENCE
49            // compactingMemStore, versionedList, action, isInterrupted (the reference)
50            // "action" is an enum and thus it is a class with static final constants,
51            // so counting only the size of the reference to it and not the size of the internals
52            + Bytes.SIZEOF_INT            // compactionKVMax
53            + ClassSize.ATOMIC_BOOLEAN    // isInterrupted (the internals)
54        );
55
56    // Configuration options for MemStore compaction
57    static final String INDEX_COMPACTION_CONFIG = "index-compaction";
58    static final String DATA_COMPACTION_CONFIG  = "data-compaction";
59
60    // The external setting of the compacting MemStore behaviour
61    // Compaction of the index without the data is the default
62    static final String COMPACTING_MEMSTORE_TYPE_KEY = "hbase.hregion.compacting.memstore.type";
63    static final String COMPACTING_MEMSTORE_TYPE_DEFAULT = INDEX_COMPACTION_CONFIG;
64
65    // The upper bound for the number of segments we store in the pipeline prior to merging.
66    // This constant is subject to further experimentation.
67    private static final int THRESHOLD_PIPELINE_SEGMENTS = 1;
68
69    private static final Log LOG = LogFactory.getLog(MemStoreCompactor.class);
70
71    private CompactingMemStore compactingMemStore;
72
73    // a static version of the segment list from the pipeline
74    private VersionedSegmentsList versionedList;
75
76    // a flag raised when compaction is requested to stop
77    private final AtomicBoolean isInterrupted = new AtomicBoolean(false);
78
79    // the limit to the size of the groups to be later provided to MemStoreSegmentsIterator
80    private final int compactionKVMax;
81
82    /**
83     * Types of actions to be done on the pipeline upon MemStoreCompaction invocation.
84     * Note that every value covers the previous ones, i.e. if MERGE is the action it implies
85     * that the youngest segment is going to be flatten anyway.
86     */
87    private enum Action {
88      NOOP,
89      FLATTEN,  // flatten the youngest segment in the pipeline
90      MERGE,    // merge all the segments in the pipeline into one
91      COMPACT   // copy-compact the data of all the segments in the pipeline
92    }
93
94    private Action action = Action.FLATTEN;
95
96    public MemStoreCompactor(CompactingMemStore compactingMemStore) {
97      this.compactingMemStore = compactingMemStore;
98      this.compactionKVMax = compactingMemStore.getConfiguration()
99          .getInt(HConstants.COMPACTION_KV_MAX, HConstants.COMPACTION_KV_MAX_DEFAULT);
100     initiateAction();
101   }

103   /**----------------------------------------------------------------------
104    * The request to dispatch the compaction asynchronous task.
105    * The method returns true if compaction was successfully dispatched, or false if there
106    * is already an ongoing compaction or no segments to compact.
107    */
108   public boolean start() throws IOException {
109     if (!compactingMemStore.hasImmutableSegments()) { // no compaction on empty pipeline
110       return false;
111     }
112
113     // get a snapshot of the list of the segments from the pipeline,
114     // this local copy of the list is marked with specific version
115     versionedList = compactingMemStore.getImmutableSegments();
116
117     if (LOG.isDebugEnabled()) {
118       LOG.debug("Starting the In-Memory Compaction for store "
119           + compactingMemStore.getStore().getColumnFamilyName());
120     }
121
122     doCompaction();
123     return true;
124   }
125
126   /**----------------------------------------------------------------------
127   * The request to cancel the compaction asynchronous task
128   * The compaction may still happen if the request was sent too late
129   * Non-blocking request
130   */
131   public void stop() {
132       isInterrupted.compareAndSet(false, true);
133   }
134
135   /**----------------------------------------------------------------------
136    * The interface to check whether user requested the index-compaction
137    */
138   public boolean isIndexCompaction() {
139     return (action == Action.MERGE);
140   }

142   /**----------------------------------------------------------------------
143   * Reset the interruption indicator and clear the pointers in order to allow good
144   * garbage collection
145   */
146   private void releaseResources() {
147     isInterrupted.set(false);
148     versionedList = null;
149   }
150
151   /**----------------------------------------------------------------------
152    * Decide what to do with the new and old segments in the compaction pipeline.
153    * Implements basic in-memory compaction policy.
154    */
155   private Action policy() {
156
157     if (isInterrupted.get()) {      // if the entire process is interrupted cancel flattening
158       return Action.NOOP;           // the compaction also doesn't start when interrupted
159     }
160
161     if (action == Action.COMPACT) { // compact according to the user request
162       LOG.debug("In-Memory Compaction Pipeline for store " + compactingMemStore.getFamilyName()
163           + " is going to be compacted, number of"
164           + " cells before compaction is " + versionedList.getNumOfCells());
165       return Action.COMPACT;
166     }
167
168     // compaction shouldn't happen or doesn't worth it
169     // limit the number of the segments in the pipeline
170     int numOfSegments = versionedList.getNumOfSegments();
171     if (numOfSegments > THRESHOLD_PIPELINE_SEGMENTS) {
172       LOG.debug("In-Memory Compaction Pipeline for store " + compactingMemStore.getFamilyName()
173           + " is going to be merged, as there are " + numOfSegments + " segments");
174       return Action.MERGE;          // to avoid too many segments, merge now
175     }
176
177     // if nothing of the above, then just flatten the newly joined segment
178     LOG.debug("The youngest segment in the in-Memory Compaction Pipeline for store "
179         + compactingMemStore.getFamilyName() + " is going to be flattened");
180     return Action.FLATTEN;
181   }
182
183   /**----------------------------------------------------------------------
184   * The worker thread performs the compaction asynchronously.
185   * The solo (per compactor) thread only reads the compaction pipeline.
186   * There is at most one thread per memstore instance.
187   */
188   private void doCompaction() {
189     ImmutableSegment result = null;
190     boolean resultSwapped = false;
191     Action nextStep = null;
192     try {
193       nextStep = policy();
194
195       if (nextStep == Action.NOOP) {
196         return;
197       }
198       if (nextStep == Action.FLATTEN) {
199         // Youngest Segment in the pipeline is with SkipList index, make it flat
200         compactingMemStore.flattenOneSegment(versionedList.getVersion());
201         return;
202       }
203
204       // Create one segment representing all segments in the compaction pipeline,
205       // either by compaction or by merge
206       if (!isInterrupted.get()) {
207         result = createSubstitution();
208       }
209
210       // Substitute the pipeline with one segment
211       if (!isInterrupted.get()) {
212         if (resultSwapped = compactingMemStore.swapCompactedSegments(
213             versionedList, result, (action==Action.MERGE))) {
214           // update the wal so it can be truncated and not get too long
215           compactingMemStore.updateLowestUnflushedSequenceIdInWAL(true); // only if greater
216         }
217       }
218     } catch (IOException e) {
219       LOG.debug("Interrupting the MemStore in-memory compaction for store "
220           + compactingMemStore.getFamilyName());
221       Thread.currentThread().interrupt();
222     } finally {
223       // For the MERGE case, if the result was created, but swap didn't happen,
224       // we DON'T need to close the result segment (meaning its MSLAB)!
225       // Because closing the result segment means closing the chunks of all segments
226       // in the compaction pipeline, which still have ongoing scans.
227       if (nextStep != Action.MERGE) {
228         if ((result != null) && (!resultSwapped)) {
229           result.close();
230         }
231       }
232       releaseResources();
233     }
234
235   }

237   /**----------------------------------------------------------------------
238    * Creation of the ImmutableSegment either by merge or copy-compact of the segments of the
239    * pipeline, based on the Compactor Iterator. The new ImmutableSegment is returned.
240    */
241   private ImmutableSegment createSubstitution() throws IOException {
242
243     ImmutableSegment result = null;
244     MemStoreSegmentsIterator iterator = null;
245
246     switch (action) {
247     case COMPACT:
248       iterator =
249           new MemStoreCompactorSegmentsIterator(versionedList.getStoreSegments(),
250               compactingMemStore.getComparator(),
251               compactionKVMax, compactingMemStore.getStore());
252
253       result = SegmentFactory.instance().createImmutableSegmentByCompaction(
254           compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator,
255           versionedList.getNumOfCells(), ImmutableSegment.Type.ARRAY_MAP_BASED);
256       iterator.close();
257       break;
258     case MERGE:
259       iterator =
260           new MemStoreMergerSegmentsIterator(versionedList.getStoreSegments(),
261               compactingMemStore.getComparator(),
262               compactionKVMax, compactingMemStore.getStore());
263
264       result = SegmentFactory.instance().createImmutableSegmentByMerge(
265           compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator,
266           versionedList.getNumOfCells(), ImmutableSegment.Type.ARRAY_MAP_BASED,
267           versionedList.getStoreSegments());
268       iterator.close();
269       break;
270     default: throw new RuntimeException("Unknown action " + action); // sanity check
271     }
272
273     return result;
274   }

276   /**----------------------------------------------------------------------
277    * Initiate the action according to user config, after its default is Action.MERGE
278    */
279   @VisibleForTesting
280   void initiateAction() {
281     String memStoreType = compactingMemStore.getConfiguration().get(COMPACTING_MEMSTORE_TYPE_KEY,
282         COMPACTING_MEMSTORE_TYPE_DEFAULT);
283
284     switch (memStoreType) {
285     case INDEX_COMPACTION_CONFIG: action = Action.MERGE;
286       break;
287     case DATA_COMPACTION_CONFIG: action = Action.COMPACT;
288       break;
289     default:
290       throw new RuntimeException("Unknown memstore type " + memStoreType); // sanity check
291     }
292   }
