18  package org.apache.hadoop.hbase.regionserver;
19
20  import java.util.Collection;
21  import java.util.HashSet;
22  import java.util.Set;
23
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  import org.apache.hadoop.hbase.HBaseInterfaceAudience;
27  import org.apache.hadoop.hbase.classification.InterfaceAudience;
28
29  /**
30   * A {@link FlushPolicy} that only flushes store larger a given threshold. If no store is large
31   * enough, then all stores will be flushed.
32   */
33  @InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
34  public class FlushAllLargeStoresPolicy extends FlushLargeStoresPolicy{
35
36    private static final Log LOG = LogFactory.getLog(FlushAllLargeStoresPolicy.class);
37
38    @Override
39    protected void configureForRegion(HRegion region) {
40      super.configureForRegion(region);
41      int familyNumber = region.getTableDesc().getFamilies().size();
42      if (familyNumber <= 1) {
43        // No need to parse and set flush size lower bound if only one family
44        // Family number might also be zero in some of our unit test case
45        return;
46      }
47      this.flushSizeLowerBound = getFlushSizeLowerBound(region);
48    }
49
50    @Override
51    public Collection<Store> selectStoresToFlush() {
52      // no need to select stores if only one family
53      if (region.getTableDesc().getFamilies().size() == 1) {
54        return region.stores.values();
55      }
56      // start selection
57      Collection<Store> stores = region.stores.values();
58      Set<Store> specificStoresToFlush = new HashSet<Store>();
59      for (Store store : stores) {
60        if (shouldFlush(store)) {
61          specificStoresToFlush.add(store);
62        }
63      }
64      if (!specificStoresToFlush.isEmpty()) return specificStoresToFlush;
65
66      // Didn't find any CFs which were above the threshold for selection.
67      if (LOG.isDebugEnabled()) {
68        LOG.debug("Since none of the CFs were above the size, flushing all.");
69      }
70      return stores;
71    }
72
73    @Override
74    protected boolean shouldFlush(Store store) {
75      return (super.shouldFlush(store) || region.shouldFlushStore(store));
76    }
77
78  }
