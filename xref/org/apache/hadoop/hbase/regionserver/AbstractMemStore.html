71    protected AbstractMemStore(final Configuration conf, final CellComparator c) {
72      this.conf = conf;
73      this.comparator = c;
74      resetCellSet();
75      this.snapshot = SegmentFactory.instance().createImmutableSegment(conf, c, 0);
76      this.snapshotId = NO_SNAPSHOT_ID;

79    protected void resetCellSet() {
80      // Reset heap to not include any keys
81      this.active = SegmentFactory.instance().createMutableSegment(
82          conf, comparator, DEEP_OVERHEAD);
83      this.timeOfOldestEdit = Long.MAX_VALUE;
84    }

87    * Calculate how the MemStore size has changed.  Includes overhead of the
88    * backing Map.
89    * @param cell
90    * @param notPresent True if the cell was NOT present in the set.
91    * @return change in size

93    static long heapSizeChange(final Cell cell, final boolean notPresent) {
94      return notPresent ? ClassSize.align(ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY
95          + CellUtil.estimatedHeapSizeOf(cell)) : 0;

117   /**
118    * Update or insert the specified Cells.
119    * <p>
120    * For each Cell, insert into MemStore.  This will atomically upsert the
121    * value for that row/family/qualifier.  If a Cell did already exist,
122    * it will then be removed.
123    * <p>
124    * Currently the memstoreTS is kept at 0 so as each insert happens, it will
125    * be immediately visible.  May want to change this so it is atomic across
126    * all Cells.
127    * <p>
128    * This is called under row lock, so Get operations will still see updates
129    * atomically.  Scans will only see each Cell update as atomic.
130    *
131    * @param cells the cells to be updated
132    * @param readpoint readpoint below which we can safely remove duplicate KVs
133    * @return change in memstore size
134    */
135   @Override
136   public long upsert(Iterable<Cell> cells, long readpoint) {
137     long size = 0;
138     for (Cell cell : cells) {
139       size += upsert(cell, readpoint);
140     }
141     return size;
142   }

173   /**
174    * The passed snapshot was successfully persisted; it can be let go.
175    * @param id Id of the snapshot to clean out.
176    * @see MemStore#snapshot(long)
177    */
178   @Override
179   public void clearSnapshot(long id) throws UnexpectedStateException {
180     if (this.snapshotId != id) {
181       throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
182           + id);
183     }
184     // OK. Passed in snapshot is same as current snapshot. If not-empty,
185     // create a new snapshot and let the old one go.
186     Segment oldSnapshot = this.snapshot;
187     if (!this.snapshot.isEmpty()) {
188       this.snapshot = SegmentFactory.instance().createImmutableSegment(
189           getComparator(), 0);
190     }
191     this.snapshotId = NO_SNAPSHOT_ID;
192     oldSnapshot.close();
193   }

244   protected void rollbackInSnapshot(Cell cell) {
245     // If the key is in the snapshot, delete it. We should not update
246     // this.size, because that tracks the size of only the memstore and
247     // not the snapshot. The flush of this snapshot to disk has not
248     // yet started because Store.flush() waits for all rwcc transactions to
249     // commit before starting the flush to disk.
250     snapshot.rollback(cell);
251   }
252 
253   protected void rollbackInActive(Cell cell) {
254     // If the key is in the memstore, delete it. Update this.size.
255     long sz = active.rollback(cell);
256     if (sz != 0) {
257       setOldestEditTimeToNow();
258     }
259   }

271   /**
272    * Inserts the specified Cell into MemStore and deletes any existing
273    * versions of the same row/family/qualifier as the specified Cell.
274    * <p>
275    * First, the specified Cell is inserted into the Memstore.
276    * <p>
277    * If there are any existing Cell in this MemStore with the same row,
278    * family, and qualifier, they are removed.
279    * <p>
280    * Callers must hold the read lock.
281    *
282    * @param cell the cell to be updated
283    * @param readpoint readpoint below which we can safely remove duplicate KVs
284    * @return change in size of MemStore
285    */
286   private long upsert(Cell cell, long readpoint) {
287     // Add the Cell to the MemStore
288     // Use the internalAdd method here since we (a) already have a lock
289     // and (b) cannot safely use the MSLAB here without potentially
290     // hitting OOME - see TestMemStore.testUpsertMSLAB for a
291     // test that triggers the pathological case if we don't avoid MSLAB
292     // here.
293     long addedSize = internalAdd(cell);
294 
295     // Get the Cells for the row/family/qualifier regardless of timestamp.
296     // For this case we want to clean up any other puts
297     Cell firstCell = KeyValueUtil.createFirstOnRow(
298         cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
299         cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
300         cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength());
301     SortedSet<Cell> ss = active.tailSet(firstCell);
302     Iterator<Cell> it = ss.iterator();
303     // versions visible to oldest scanner
304     int versionsVisible = 0;
305     while (it.hasNext()) {
306       Cell cur = it.next();
307 
308       if (cell == cur) {
309         // ignore the one just put in
310         continue;
311       }
312       // check that this is the row and column we are interested in, otherwise bail
313       if (CellUtil.matchingRow(cell, cur) && CellUtil.matchingQualifier(cell, cur)) {
314         // only remove Puts that concurrent scanners cannot possibly see
315         if (cur.getTypeByte() == KeyValue.Type.Put.getCode() &&
316             cur.getSequenceId() <= readpoint) {
317           if (versionsVisible >= 1) {
318             // if we get here we have seen at least one version visible to the oldest scanner,
319             // which means we can prove that no scanner will see this version
320 
321             // false means there was a change, so give us the size.
322             long delta = heapSizeChange(cur, true);
323             addedSize -= delta;
324             active.incSize(-delta);
325             it.remove();
326             setOldestEditTimeToNow();
327           } else {
328             versionsVisible++;
329           }
330         }
331       } else {
332         // past the row or column, done
333         break;
334       }
335     }
336     return addedSize;
337   }

341    * @param b
342    * @return Return lowest of a or b or null if both a and b are null

344   protected Cell getLowest(final Cell a, final Cell b) {
345     if (a == null) {
346       return b;
347     }
348     if (b == null) {
349       return a;
350     }
351     return comparator.compareRows(a, b) <= 0? a: b;
352   }

354   /*
355    * @param key Find row that follows this one.  If null, return first.
356    * @param set Set to look in for a row beyond <code>row</code>.
357    * @return Next row or null if none found.  If one found, will be a new
358    * KeyValue -- can be destroyed by subsequent calls to this method.
359    */
360   protected Cell getNextRow(final Cell key,
361       final NavigableSet<Cell> set) {
362     Cell result = null;
363     SortedSet<Cell> tail = key == null? set: set.tailSet(key);
364     // Iterate until we fall into the next row; i.e. move off current row
365     for (Cell cell: tail) {
366       if (comparator.compareRows(cell, key) <= 0) {
367         continue;
368       }
369       // Note: Not suppressing deletes or expired cells.  Needs to be handled
370       // by higher up functions.
371       result = cell;
372       break;
373     }
374     return result;
375   }

377   /**
378    * Given the specs of a column, update it, first by inserting a new record,
379    * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
380    * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
381    * store will ensure that the insert/delete each are atomic. A scanner/reader will either
382    * get the new value, or the old value and all readers will eventually only see the new
383    * value after the old was removed.
384    */
385   @VisibleForTesting
386   @Override
387   public long updateColumnValue(byte[] row, byte[] family, byte[] qualifier,
388       long newValue, long now) {
389     Cell firstCell = KeyValueUtil.createFirstOnRow(row, family, qualifier);
390     // Is there a Cell in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
391     Cell snc = snapshot.getFirstAfter(firstCell);
392     if(snc != null) {
393       // is there a matching Cell in the snapshot?
394       if (CellUtil.matchingRow(snc, firstCell) && CellUtil.matchingQualifier(snc, firstCell)) {
395         if (snc.getTimestamp() == now) {
396           now += 1;
397         }
398       }
399     }
400     // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
401     // But the timestamp should also be max(now, mostRecentTsInMemstore)
402 
403     // so we cant add the new Cell w/o knowing what's there already, but we also
404     // want to take this chance to delete some cells. So two loops (sad)
405 
406     SortedSet<Cell> ss = getActive().tailSet(firstCell);
407     for (Cell cell : ss) {
408       // if this isnt the row we are interested in, then bail:
409       if (!CellUtil.matchingColumn(cell, family, qualifier)
410           || !CellUtil.matchingRow(cell, firstCell)) {
411         break; // rows dont match, bail.
412       }
413 
414       // if the qualifier matches and it's a put, just RM it out of the active.
415       if (cell.getTypeByte() == KeyValue.Type.Put.getCode() &&
416           cell.getTimestamp() > now && CellUtil.matchingQualifier(firstCell, cell)) {
417         now = cell.getTimestamp();
418       }
419     }
420 
421     // create or update (upsert) a new Cell with
422     // 'now' and a 0 memstoreTS == immediately visible
423     List<Cell> cells = new ArrayList<Cell>(1);
424     cells.add(new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue)));
425     return upsert(cells, 1L);
426   }

432   /**
433    * Internal version of add() that doesn't clone Cells with the
434    * allocator, and doesn't take the lock.
435    *
436    * Callers should ensure they already have the read lock taken
437    */
438   private long internalAdd(final Cell toAdd) {
439     long s = active.add(toAdd);
440     setOldestEditTimeToNow();
441     checkActiveSize();
442     return s;
