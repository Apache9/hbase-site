76  /**
77   * Implementation of {@link WAL} to go against {@link FileSystem}; i.e. keep WALs in HDFS. Only one
78   * WAL is ever being written at a time. When a WAL hits a configured maximum size, it is rolled.
79   * This is done internal to the implementation.
80   * <p>
81   * As data is flushed from the MemStore to other on-disk structures (files sorted by key, hfiles), a
82   * WAL becomes obsolete. We can let go of all the log edits/entries for a given HRegion-sequence id.
83   * A bunch of work in the below is done keeping account of these region sequence ids -- what is
84   * flushed out to hfiles, and what is yet in WAL and in memory only.
85   * <p>
86   * It is only practical to delete entire files. Thus, we delete an entire on-disk file
87   * <code>F</code> when all of the edits in <code>F</code> have a log-sequence-id that's older
88   * (smaller) than the most-recent flush.
89   * <p>
90   * To read an WAL, call
91   * {@link WALFactory#createReader(org.apache.hadoop.fs.FileSystem, org.apache.hadoop.fs.Path)}. *
92   * <h2>Failure Semantic</h2> If an exception on append or sync, roll the WAL because the current WAL
93   * is now a lame duck; any more appends or syncs will fail also with the same original exception. If
94   * we have made successful appends to the WAL and we then are unable to sync them, our current
95   * semantic is to return error to the client that the appends failed but also to abort the current
96   * context, usually the hosting server. We need to replay the WALs. <br>
97   * TODO: Change this semantic. A roll of WAL may be sufficient as long as we have flagged client
98   * that the append failed. <br>
99   * TODO: replication may pick up these last edits though they have been marked as failed append
100  * (Need to keep our own file lengths, not rely on HDFS).
101  */
102 @InterfaceAudience.Private
103 public abstract class AbstractFSWAL<W> implements WAL {

272   /**
273    * A log file has a creation timestamp (in ms) in its file name ({@link #filenum}. This helper
274    * method returns the creation timestamp from a given log file. It extracts the timestamp assuming
275    * the filename is created with the {@link #computeFilename(long filenum)} method.
276    * @return timestamp, as in the log file name.
277    */
278   protected long getFileNumFromFileName(Path fileName) {
279     checkNotNull(fileName, "file name can't be null");
280     if (!ourFiles.accept(fileName)) {
281       throw new IllegalArgumentException(
282           "The log file " + fileName + " doesn't belong to this WAL. (" + toString() + ")");
283     }
284     final String fileNameString = fileName.toString();
285     String chompedPath = fileNameString.substring(prefixPathStr.length(),
286       (fileNameString.length() - walFileSuffix.length()));
287     return Long.parseLong(chompedPath);
288   }

290   private int calculateMaxLogFiles(float memstoreSizeRatio, long logRollSize) {
291     MemoryUsage mu = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
292     return Math.round(mu.getMax() * memstoreSizeRatio * 2 / logRollSize);
293   }

295   // must be power of 2
296   protected final int getPreallocatedEventCount() {
297     // Preallocate objects to use on the ring buffer. The way that appends and syncs work, we will
298     // be stuck and make no progress if the buffer is filled with appends only and there is no
299     // sync. If no sync, then the handlers will be outstanding just waiting on sync completion
300     // before they return.
301     int preallocatedEventCount = this.conf.getInt("hbase.regionserver.wal.disruptor.event.count",
302       1024 * 16);
303     checkArgument(preallocatedEventCount >= 0,
304       "hbase.regionserver.wal.disruptor.event.count must > 0");
305     int floor = Integer.highestOneBit(preallocatedEventCount);
306     if (floor == preallocatedEventCount) {
307       return floor;
308     }
309     // max capacity is 1 << 30
310     if (floor >= 1 << 29) {
311       return 1 << 30;
312     }
313     return floor << 1;
314   }

316   protected AbstractFSWAL(final FileSystem fs, final Path rootDir, final String logDir,
317       final String archiveDir, final Configuration conf, final List<WALActionsListener> listeners,
318       final boolean failIfWALExists, final String prefix, final String suffix)
319       throws FailedLogCloseException, IOException {
320     this.fs = fs;
321     this.walDir = new Path(rootDir, logDir);
322     this.walArchiveDir = new Path(rootDir, archiveDir);
323     this.conf = conf;
324
325     if (!fs.exists(walDir) && !fs.mkdirs(walDir)) {
326       throw new IOException("Unable to mkdir " + walDir);
327     }
328
329     if (!fs.exists(this.walArchiveDir)) {
330       if (!fs.mkdirs(this.walArchiveDir)) {
331         throw new IOException("Unable to mkdir " + this.walArchiveDir);
332       }
333     }
334
335     // If prefix is null||empty then just name it wal
336     this.walFilePrefix = prefix == null || prefix.isEmpty() ? "wal"
337         : URLEncoder.encode(prefix, "UTF8");
338     // we only correctly differentiate suffices when numeric ones start with '.'
339     if (suffix != null && !(suffix.isEmpty()) && !(suffix.startsWith(WAL_FILE_NAME_DELIMITER))) {
340       throw new IllegalArgumentException("WAL suffix must start with '" + WAL_FILE_NAME_DELIMITER
341           + "' but instead was '" + suffix + "'");
342     }
343     // Now that it exists, set the storage policy for the entire directory of wal files related to
344     // this FSHLog instance
345     FSUtils.setStoragePolicy(fs, conf, this.walDir, HConstants.WAL_STORAGE_POLICY,
346       HConstants.DEFAULT_WAL_STORAGE_POLICY);
347     this.walFileSuffix = (suffix == null) ? "" : URLEncoder.encode(suffix, "UTF8");
348     this.prefixPathStr = new Path(walDir, walFilePrefix + WAL_FILE_NAME_DELIMITER).toString();
349
350     this.ourFiles = new PathFilter() {
351       @Override
352       public boolean accept(final Path fileName) {
353         // The path should start with dir/<prefix> and end with our suffix
354         final String fileNameString = fileName.toString();
355         if (!fileNameString.startsWith(prefixPathStr)) {
356           return false;
357         }
358         if (walFileSuffix.isEmpty()) {
359           // in the case of the null suffix, we need to ensure the filename ends with a timestamp.
360           return org.apache.commons.lang.StringUtils
361               .isNumeric(fileNameString.substring(prefixPathStr.length()));
362         } else if (!fileNameString.endsWith(walFileSuffix)) {
363           return false;
364         }
365         return true;
366       }
367     };
368
369     if (failIfWALExists) {
370       final FileStatus[] walFiles = FSUtils.listStatus(fs, walDir, ourFiles);
371       if (null != walFiles && 0 != walFiles.length) {
372         throw new IOException("Target WAL already exists within directory " + walDir);
373       }
374     }
375
376     // Register listeners. TODO: Should this exist anymore? We have CPs?
377     if (listeners != null) {
378       for (WALActionsListener i : listeners) {
379         registerWALActionsListener(i);
380       }
381     }
382     this.coprocessorHost = new WALCoprocessorHost(this, conf);
383
384     // Get size to roll log at. Roll at 95% of HDFS block size so we avoid crossing HDFS blocks
385     // (it costs a little x'ing bocks)
386     final long blocksize = this.conf.getLong("hbase.regionserver.hlog.blocksize",
387       FSUtils.getDefaultBlockSize(this.fs, this.walDir));
388     this.logrollsize = (long) (blocksize
389         * conf.getFloat("hbase.regionserver.logroll.multiplier", 0.95f));
390
391     float memstoreRatio = conf.getFloat(HeapMemorySizeUtil.MEMSTORE_SIZE_KEY, conf.getFloat(
392       HeapMemorySizeUtil.MEMSTORE_SIZE_OLD_KEY, HeapMemorySizeUtil.DEFAULT_MEMSTORE_SIZE));
393     boolean maxLogsDefined = conf.get("hbase.regionserver.maxlogs") != null;
394     if (maxLogsDefined) {
395       LOG.warn("'hbase.regionserver.maxlogs' was deprecated.");
396     }
397     this.maxLogs = conf.getInt("hbase.regionserver.maxlogs",
398       Math.max(32, calculateMaxLogFiles(memstoreRatio, logrollsize)));
399
400     LOG.info("WAL configuration: blocksize=" + StringUtils.byteDesc(blocksize) + ", rollsize="
401         + StringUtils.byteDesc(this.logrollsize) + ", prefix=" + this.walFilePrefix + ", suffix="
402         + walFileSuffix + ", logDir=" + this.walDir + ", archiveDir=" + this.walArchiveDir);
403     this.slowSyncNs =
404         1000000 * conf.getInt("hbase.regionserver.hlog.slowsync.ms", DEFAULT_SLOW_SYNC_TIME_MS);
405     this.walSyncTimeout = conf.getLong("hbase.regionserver.hlog.sync.timeout",
406         DEFAULT_WAL_SYNC_TIMEOUT_MS);
407     int maxHandlersCount = conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT, 200);
408     // Presize our map of SyncFutures by handler objects.
409     this.syncFuturesByHandler = new ConcurrentHashMap<Thread, SyncFuture>(maxHandlersCount);
410   }

457     // sequenceids are too old. It is also used reporting the master our oldest sequenceid for use

464     return this.sequenceIdAccounting.getLowestSequenceId(encodedRegionName, familyName);

495    * retrieve the next path to use for writing. Increments the internal filenum.
496    */
497   private Path getNewPath() throws IOException {
498     this.filenum.set(System.currentTimeMillis());
499     Path newPath = getCurrentFileName();
500     while (fs.exists(newPath)) {
501       this.filenum.incrementAndGet();
502       newPath = getCurrentFileName();
503     }
504     return newPath;
505   }

559   /**
560    * If the number of un-archived WAL files is greater than maximum allowed, check the first
561    * (oldest) WAL file, and returns those regions which should be flushed so that it can be
562    * archived.
563    * @return regions (encodedRegionNames) to flush in order to archive oldest WAL file.
564    */
565   byte[][] findRegionsToForceFlush() throws IOException {
566     byte[][] regions = null;
567     int logCount = getNumRolledLogFiles();
568     if (logCount > this.maxLogs && logCount > 0) {
569       Map.Entry<Path, WalProps> firstWALEntry = this.walFile2Props.firstEntry();
570       regions = this.sequenceIdAccounting
571           .findLower(firstWALEntry.getValue().encodedName2HighestSequenceId);
572     }
573     if (regions != null) {
574       StringBuilder sb = new StringBuilder();
575       for (int i = 0; i < regions.length; i++) {
576         if (i > 0) {
577           sb.append(", ");
578         }
579         sb.append(Bytes.toStringBinary(regions[i]));
580       }
581       LOG.info("Too many WALs; count=" + logCount + ", max=" + this.maxLogs + "; forcing flush of "
582           + regions.length + " regions(s): " + sb.toString());
583     }
584     return regions;
585   }

587   /**
588    * Archive old logs. A WAL is eligible for archiving if all its WALEdits have been flushed.
589    */
590   private void cleanOldLogs() throws IOException {
591     List<Pair<Path, Long>> logsToArchive = null;
592     // For each log file, look at its Map of regions to highest sequence id; if all sequence ids
593     // are older than what is currently in memory, the WAL can be GC'd.
594     for (Map.Entry<Path, WalProps> e : this.walFile2Props.entrySet()) {
595       Path log = e.getKey();
596       Map<byte[], Long> sequenceNums = e.getValue().encodedName2HighestSequenceId;
597       if (this.sequenceIdAccounting.areAllLower(sequenceNums)) {
598         if (logsToArchive == null) {
599           logsToArchive = new ArrayList<>();
600         }
601         logsToArchive.add(Pair.newPair(log, e.getValue().logSize));
602         if (LOG.isTraceEnabled()) {
603           LOG.trace("WAL file ready for archiving " + log);
604         }
605       }
606     }
607     if (logsToArchive != null) {
608       for (Pair<Path, Long> logAndSize : logsToArchive) {
609         this.totalLogSize.addAndGet(-logAndSize.getSecond());
610         archiveLogFile(logAndSize.getFirst());
611         this.walFile2Props.remove(logAndSize.getFirst());
612       }
613     }
614   }

647    * <p>
648    * <ul>
649    * <li>In the case of creating a new WAL, oldPath will be null.</li>
650    * <li>In the case of rolling over from one file to the next, none of the parameters will be null.

654    * </ul>

656    * @param newPath may be null

659    * @throws IOException if there is a problem flushing or closing the underlying FS
660    */

662     TraceScope scope = Trace.startSpan("FSHFile.replaceWriter");
663     try {
664       long oldFileLen = doReplaceWriter(oldPath, newPath, nextWriter);
665       int oldNumEntries = this.numEntries.getAndSet(0);
666       final String newPathString = (null == newPath ? null : FSUtils.getPath(newPath));
667       if (oldPath != null) {

669           new WalProps(this.sequenceIdAccounting.resetHighest(), oldFileLen));

672             + ", filesize=" + StringUtils.byteDesc(oldFileLen) + "; new WAL " + newPathString);
673       } else {
674         LOG.info("New WAL " + newPathString);
675       }
676       return newPath;
677     } finally {
678       scope.close();
679     }

682   protected Span blockOnSync(final SyncFuture syncFuture) throws IOException {
683     // Now we have published the ringbuffer, halt the current thread until we get an answer back.
684     try {
685       syncFuture.get(walSyncTimeout);
686       return syncFuture.getSpan();
687     } catch (TimeoutIOException tioe) {
688       // SyncFuture reuse by thread, if TimeoutIOException happens, ringbuffer
689       // still refer to it, so if this thread use it next time may get a wrong
690       // result.
691       this.syncFuturesByHandler.remove(Thread.currentThread());
692       throw tioe;
693     } catch (InterruptedException ie) {
694       LOG.warn("Interrupted", ie);
695       throw convertInterruptedExceptionToIOException(ie);
696     } catch (ExecutionException e) {
697       throw ensureIOException(e.getCause());
698     }
699   }

712   @Override
713   public byte[][] rollWriter(boolean force) throws FailedLogCloseException, IOException {
714     rollWriterLock.lock();
715     try {
716       // Return if nothing to flush.
717       if (!force && this.writer != null && this.numEntries.get() <= 0) {
718         return null;
719       }
720       byte[][] regionsToFlush = null;
721       if (this.closed) {
722         LOG.debug("WAL closed. Skipping rolling of writer");
723         return regionsToFlush;
724       }
725       if (!closeBarrier.beginOp()) {
726         LOG.debug("WAL closing. Skipping rolling of writer");
727         return regionsToFlush;
728       }
729       TraceScope scope = Trace.startSpan("FSHLog.rollWriter");
730       try {
731         Path oldPath = getOldPath();
732         Path newPath = getNewPath();
733         // Any exception from here on is catastrophic, non-recoverable so we currently abort.
734         W nextWriter = this.createWriterInstance(newPath);
735         tellListenersAboutPreLogRoll(oldPath, newPath);
736         // NewPath could be equal to oldPath if replaceWriter fails.
737         newPath = replaceWriter(oldPath, newPath, nextWriter);
738         tellListenersAboutPostLogRoll(oldPath, newPath);
739         if (LOG.isDebugEnabled()) {
740           LOG.debug("Create new " + getClass().getSimpleName() + " writer with pipeline: "
741               + Arrays.toString(getPipeline()));
742         }
743         // Can we delete any of the old log files?
744         if (getNumRolledLogFiles() > 0) {
745           cleanOldLogs();
746           regionsToFlush = findRegionsToForceFlush();
747         }
748       } finally {
749         closeBarrier.endOp();
750         assert scope == NullScope.INSTANCE || !scope.isDetached();
751         scope.close();
752       }
753       return regionsToFlush;
754     } finally {
755       rollWriterLock.unlock();
756     }
757   }

781       return;
782     }
783     closed = true;
784     try {

789       Thread.currentThread().interrupt();

792     if (!this.listeners.isEmpty()) {
793       for (WALActionsListener i : this.listeners) {

795       }
796     }

880   protected boolean append(W writer, FSWALEntry entry) throws IOException {
881     // TODO: WORK ON MAKING THIS APPEND FASTER. DOING WAY TOO MUCH WORK WITH CPs, PBing, etc.
882     atHeadOfRingBufferEventHandlerAppend();
883     long start = EnvironmentEdgeManager.currentTime();
884     byte[] encodedRegionName = entry.getKey().getEncodedRegionName();
885     long regionSequenceId = WALKey.NO_SEQUENCE_ID;
886     // We are about to append this edit; update the region-scoped sequence number. Do it
887     // here inside this single appending/writing thread. Events are ordered on the ringbuffer
888     // so region sequenceids will also be in order.
889     regionSequenceId = entry.stampRegionSequenceId();
890     // Edits are empty, there is nothing to append. Maybe empty when we are looking for a
891     // region sequence id only, a region edit/sequence id that is not associated with an actual
892     // edit. It has to go through all the rigmarole to be sure we have the right ordering.
893     if (entry.getEdit().isEmpty()) {
894       return false;
895     }
896
897     // Coprocessor hook.
898     if (!coprocessorHost.preWALWrite(entry.getHRegionInfo(), entry.getKey(), entry.getEdit())) {
899       if (entry.getEdit().isReplay()) {
900         // Set replication scope null so that this won't be replicated
901         entry.getKey().serializeReplicationScope(false);
902       }
903     }
904     if (!listeners.isEmpty()) {
905       for (WALActionsListener i : listeners) {
906         i.visitLogEntryBeforeWrite(entry.getKey(), entry.getEdit());
907       }
908     }
909     doAppend(writer, entry);
910     assert highestUnsyncedTxid < entry.getTxid();
911     highestUnsyncedTxid = entry.getTxid();
912     sequenceIdAccounting.update(encodedRegionName, entry.getFamilyNames(), regionSequenceId,
913       entry.isInMemstore());
914     coprocessorHost.postWALWrite(entry.getHRegionInfo(), entry.getKey(), entry.getEdit());
915     // Update metrics.
916     postAppend(entry, EnvironmentEdgeManager.currentTime() - start);
917     numEntries.incrementAndGet();
918     return true;
919   }
