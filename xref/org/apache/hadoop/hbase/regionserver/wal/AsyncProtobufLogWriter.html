46  /**
47   * AsyncWriter for protobuf-based WAL.
48   */
49  @InterfaceAudience.Private
50  public class AsyncProtobufLogWriter extends AbstractProtobufLogWriter
51      implements AsyncFSWALProvider.AsyncWriter {
52  
53    private static final Log LOG = LogFactory.getLog(AsyncProtobufLogWriter.class);
54  
55    private final EventLoop eventLoop;
56  
57    private AsyncFSOutput output;
58  
59    private static final class OutputStreamWrapper extends OutputStream
60        implements ByteBufferWriter {
61
62      private final AsyncFSOutput out;
63
64      private final byte[] oneByteBuf = new byte[1];
65
66      @Override
67      public void write(int b) throws IOException {
68        oneByteBuf[0] = (byte) b;
69        write(oneByteBuf);
70      }
71
72      public OutputStreamWrapper(AsyncFSOutput out) {
73        this.out = out;
74      }
75
76      @Override
77      public void write(ByteBuffer b, int off, int len) throws IOException {
78        ByteBuffer bb = b.duplicate();
79        bb.position(off);
80        bb.limit(off + len);
81        out.write(bb);
82      }
83
84      @Override
85      public void writeInt(int i) throws IOException {
86        out.writeInt(i);
87      }
88
89      @Override
90      public void write(byte[] b, int off, int len) throws IOException {
91        out.write(b, off, len);
92      }
93
94      @Override
95      public void close() throws IOException {
96        out.close();
97      }
98    }
99
100   private OutputStream asyncOutputWrapper;
101
102   public AsyncProtobufLogWriter(EventLoop eventLoop) {
103     this.eventLoop = eventLoop;
104   }
105
106   @Override
107   public void append(Entry entry) {
108     int buffered = output.buffered();
109     entry.setCompressionContext(compressionContext);
110     try {
111       entry.getKey().getBuilder(compressor).setFollowingKvCount(entry.getEdit().size()).build()
112           .writeDelimitedTo(asyncOutputWrapper);
113     } catch (IOException e) {
114       throw new AssertionError("should not happen", e);
115     }
116     try {
117       for (Cell cell : entry.getEdit().getCells()) {
118         cellEncoder.write(cell);
119       }
120     } catch (IOException e) {
121       throw new AssertionError("should not happen", e);
122     }
123     length.addAndGet(output.buffered() - buffered);
124   }
125
126   @Override
127   public CompletableFuture<Long> sync() {
128     return output.flush(false);
129   }
130 
131   @Override
132   public synchronized void close() throws IOException {
133     if (this.output == null) {
134       return;
135     }
136     try {
137       writeWALTrailer();
138       output.close();
139     } catch (Exception e) {
140       LOG.warn("normal close failed, try recover", e);
141       output.recoverAndClose(null);
142     }
143     this.output = null;
144   }
145
146   public AsyncFSOutput getOutput() {
147     return this.output;
148   }
149
150   @Override
151   protected void initOutput(FileSystem fs, Path path, boolean overwritable, int bufferSize,
152       short replication, long blockSize) throws IOException {
153     this.output = AsyncFSOutputHelper.createOutput(fs, path, overwritable, false, replication,
154       blockSize, eventLoop);
155     this.asyncOutputWrapper = new OutputStreamWrapper(output);
156   }
157
158   private long write(Consumer<CompletableFuture<Long>> action) throws IOException {
159     CompletableFuture<Long> future = new CompletableFuture<Long>();
160     eventLoop.execute(() -> action.accept(future));
161     try {
162       return future.get().longValue();
163     } catch (InterruptedException e) {
164       InterruptedIOException ioe = new InterruptedIOException();
165       ioe.initCause(e);
166       throw ioe;
167     } catch (ExecutionException e) {
168       Throwables.propagateIfPossible(e.getCause(), IOException.class);
169       throw new RuntimeException(e.getCause());
170     }
171   }
172
173   @Override
174   protected long writeMagicAndWALHeader(byte[] magic, WALHeader header) throws IOException {
175     return write(future -> {
176       output.write(magic);
177       try {
178         header.writeDelimitedTo(asyncOutputWrapper);
179       } catch (IOException e) {
180         // should not happen
181         throw new AssertionError(e);
182       }
183       output.flush(false).whenComplete((len, error) -> {
184         if (error != null) {
185           future.completeExceptionally(error);
186         } else {
187           future.complete(len);
188         }
189       });
190     });
191   }
192
193   @Override
194   protected long writeWALTrailerAndMagic(WALTrailer trailer, byte[] magic) throws IOException {
195     return write(future -> {
196       try {
197         trailer.writeTo(asyncOutputWrapper);
198       } catch (IOException e) {
199         // should not happen
200         throw new AssertionError(e);
201       }
202       output.writeInt(trailer.getSerializedSize());
203       output.write(magic);
204       output.flush(false).whenComplete((len, error) -> {
205         if (error != null) {
206           future.completeExceptionally(error);
207         } else {
208           future.complete(len);
209         }
210       });
211     });
212   }
213
214   @Override
215   protected OutputStream getOutputStreamForCellEncoder() {
216     return asyncOutputWrapper;
217   }
