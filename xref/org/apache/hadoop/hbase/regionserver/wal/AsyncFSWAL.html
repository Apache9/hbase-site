63  /**
64   * An asynchronous implementation of FSWAL.
65   * <p>
66   * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog. We do not use RingBuffer here
67   * because RingBuffer need an exclusive thread to consume the entries in it, and here we want to run
68   * the append and sync operation inside EventLoop. We can not use EventLoop as the RingBuffer's
69   * executor otherwise the EventLoop can not process any other events such as socket read and write.
70   * <p>
71   * For append, we process it as follow:
72   * <ol>
73   * <li>In the caller thread(typically, in the rpc handler thread):
74   * <ol>
75   * <li>Lock 'waitingConsumePayloads', bump nextTxid, and insert the entry to
76   * 'waitingConsumePayloads'.</li>
77   * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
78   * </li>
79   * </ol>
80   * </li>
81   * <li>In the consumer task(in the EventLoop thread)
82   * <ol>
83   * <li>Poll the entry from 'waitingConsumePayloads' and insert it into 'waitingAppendEntries'</li>
84   * <li>Poll the entry from 'waitingAppendEntries', append it to the AsyncWriter, and insert it into
85   * 'unackedEntries'</li>
86   * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
87   * sync on the AsyncWriter.</li>
88   * <li>In the callback methods(CompletionHandler):
89   * <ul>
90   * <li>If succeeded, poll the entry from 'unackedEntries' and drop it.</li>
91   * <li>If failed, add all the entries in 'unackedEntries' back to 'waitingAppendEntries' and wait
92   * for writing them again.</li>
93   * </ul>
94   * </li>
95   * </ol>
96   * </li>
97   * </ol>
98   * For sync, the processing stages are almost same except that if it is not assigned with a new
99   * 'txid', we just assign the previous 'txid' to it without bumping the 'nextTxid'. And different
100  * from FSHLog, we will open a new writer and rewrite unacked entries to the new writer and sync
101  * again if we hit a sync error.
102  * <p>
103  * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
104  * FSHLog.<br>
105  * For a normal roll request(for example, we have reached the log roll size):
106  * <ol>
107  * <li>In the log roller thread, we add a roll payload to 'waitingConsumePayloads', and then wait on
108  * the rollPromise(see {@link #waitForSafePoint()}).</li>
109  * <li>In the consumer thread, we will stop polling entries from 'waitingConsumePayloads' if we hit
110  * a Payload which contains a roll request.</li>
111  * <li>Append all entries to current writer, issue a sync request if possible.</li>
112  * <li>If sync succeeded, check if we could finish a roll request. There 3 conditions:
113  * <ul>
114  * <li>'rollPromise' is not null which means we have a pending roll request.</li>
115  * <li>'waitingAppendEntries' is empty.</li>
116  * <li>'unackedEntries' is empty.</li>
117  * </ul>
118  * </li>
119  * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
120  * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
121  * <li>Acquire 'waitingConsumePayloads' lock, set 'writerBroken' and 'waitingRoll' to false, cancel
122  * log roller exit checker if any(see the comments in the 'failed' method of the sync
123  * CompletionHandler to see why we need a checker here).</li>
124  * <li>Schedule the consumer task if needed.</li>
125  * <li>Schedule a background task to close the old writer.</li>
126  * </ol>
127  * For a broken writer roll request, the only difference is that we can bypass the wait for safe
128  * point stage. See the comments in the 'failed' method of the sync CompletionHandler for more
129  * details.
130  */

232   private final class LogRollerExitedChecker implements Runnable {
233 
234     private boolean cancelled;
235 
236     private ScheduledFuture<?> future;
237 
238     public synchronized void setFuture(ScheduledFuture<?> future) {
239       this.future = future;
240     }
241 
242     @Override
243     public void run() {
244       if (!logRollerExited) {
245         return;
246       }
247       // rollWriter is called in the log roller thread, and logRollerExited will be set just before
248       // the log rolled exit. So here we can confirm that no one could cancel us if the 'canceled'
249       // check passed. So it is safe to release the lock after checking 'canceled' flag.
250       synchronized (this) {
251         if (cancelled) {
252           return;
253         }
254       }
255       unackedEntries.clear();
256       waitingAppendEntries.clear();
257       IOException error = new IOException("sync failed but log roller exited");
258       for (SyncFuture future; (future = syncFutures.peek()) != null;) {
259         future.done(highestProcessedTxid, error);
260         syncFutures.remove();
261       }
262       synchronized (waitingConsumePayloads) {
263         for (Payload p : waitingConsumePayloads) {
264           if (p.entry != null) {
265             try {
266               p.entry.stampRegionSequenceId();
267             } catch (IOException e) {
268               throw new AssertionError("should not happen", e);
269             }
270           } else if (p.sync != null) {
271             p.sync.done(nextTxid, error);
272           }
273         }
274         waitingConsumePayloads.clear();
275       }
276     }
277 
278     public synchronized void cancel() {
279       future.cancel(false);
280       cancelled = true;
281     }
282   }

303   private void tryFinishRoll() {
304     // 1. a roll is requested
305     // 2. we have written out all entries before the roll point.
306     // 3. all entries have been acked.
307     if (rollPromise != null && waitingAppendEntries.isEmpty() && unackedEntries.isEmpty()) {
308       rollPromise.trySuccess(null);
309       rollPromise = null;
310     }
311   }

313   private void sync(final AsyncWriter writer, final long processedTxid) {
314     fileLengthAtLastSync = writer.getLength();
315     final long startTimeNs = System.nanoTime();
316     writer.sync(new CompletionHandler<Long, Void>() {
317 
318       @Override
319       public void completed(Long result, Void attachment) {
320         highestSyncedTxid.set(processedTxid);
321         int syncCount = finishSync(true);
322         for (Iterator<FSWALEntry> iter = unackedEntries.iterator(); iter.hasNext();) {
323           if (iter.next().getTxid() <= processedTxid) {
324             iter.remove();
325           } else {
326             break;
327           }
328         }
329         postSync(System.nanoTime() - startTimeNs, syncCount);
330         tryFinishRoll();
331         if (!rollWriterLock.tryLock()) {
332           return;
333         }
334         try {
335           if (writer.getLength() >= logrollsize) {
336             requestLogRoll();
337           }
338         } finally {
339           rollWriterLock.unlock();
340         }
341       }
342 
343       @Override
344       public void failed(Throwable exc, Void attachment) {
345         LOG.warn("sync failed", exc);
346         // Here we depends on the implementation of FanOutOneBlockAsyncDFSOutput and netty.
347         // When error occur, FanOutOneBlockAsyncDFSOutput will fail all pending flush requests. It
348         // is execute inside EventLoop. And in DefaultPromise in netty, it will notifyListener
349         // directly if it is already in the EventLoop thread. And in the listener method, it will
350         // call us. So here we know that all failed flush request will call us continuously, and
351         // before the last one finish, no other task can be executed in EventLoop. So here we are
352         // safe to use writerBroken as a guard.
353         // Do not forget to revisit this if we change the implementation of
354         // FanOutOneBlockAsyncDFSOutput!
355         synchronized (waitingConsumePayloads) {
356           if (writerBroken) {
357             return;
358           }
359           // schedule a periodical task to check if log roller is exited. Otherwise the the sync
360           // request maybe blocked forever since we are still waiting for a new writer to write the
361           // pending data and sync it...
362           logRollerExitedChecker = new LogRollerExitedChecker();
363           // we are currently in the EventLoop thread, so it is safe to set the future after
364           // schedule it since the task can not be executed before we release the thread.
365           logRollerExitedChecker.setFuture(eventLoop.scheduleAtFixedRate(logRollerExitedChecker,
366             logRollerExitedCheckIntervalMs, logRollerExitedCheckIntervalMs, TimeUnit.MILLISECONDS));
367           writerBroken = true;
368         }
369         for (Iterator<FSWALEntry> iter = unackedEntries.descendingIterator(); iter.hasNext();) {
370           waitingAppendEntries.addFirst(iter.next());
371         }
372         highestUnsyncedTxid = highestSyncedTxid.get();
373         if (rollPromise != null) {
374           rollPromise.trySuccess(null);
375           rollPromise = null;
376           return;
377         }
378         // request a roll.
379         if (!rollWriterLock.tryLock()) {
380           return;
381         }
382         try {
383           requestLogRoll();
384         } finally {
385           rollWriterLock.unlock();
386         }
387       }
388     }, null);
389   }

397   private int finishSync(boolean addSyncTrace) {
398     long doneTxid = highestSyncedTxid.get();
399     int finished = 0;
400     for (SyncFuture future; (future = syncFutures.peek()) != null;) {
401       if (future.getTxid() <= doneTxid) {
402         future.done(doneTxid, null);
403         syncFutures.remove();
404         finished++;
405         addTimeAnnotation(future, "writer synced");
406       } else {
407         break;
408       }
409     }
410     return finished;
411   }

413   private void consume() {
414     final AsyncWriter writer = this.writer;
415     // maybe a sync request is not queued when we issue a sync, so check here to see if we could
416     // finish some.
417     finishSync(false);
418     long newHighestProcessedTxid = -1L;
419     for (Iterator<FSWALEntry> iter = waitingAppendEntries.iterator(); iter.hasNext();) {
420       FSWALEntry entry = iter.next();
421       boolean appended;
422       try {
423         appended = append(writer, entry);
424       } catch (IOException e) {
425         throw new AssertionError("should not happen", e);
426       }
427       newHighestProcessedTxid = entry.getTxid();
428       iter.remove();
429       if (appended) {
430         unackedEntries.addLast(entry);
431         if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
432           break;
433         }
434       }
435     }
436     // if we have a newer transaction id, update it.
437     // otherwise, use the previous transaction id.
438     if (newHighestProcessedTxid > 0) {
439       highestProcessedTxid = newHighestProcessedTxid;
440     } else {
441       newHighestProcessedTxid = highestProcessedTxid;
442     }
443     if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
444       // sync because buffer size limit.
445       sync(writer, newHighestProcessedTxid);
446     } else if ((!syncFutures.isEmpty() || rollPromise != null)
447         && writer.getLength() > fileLengthAtLastSync) {
448       // first we should have at least one sync request or a roll request
449       // second we should have some unsynced data.
450       sync(writer, newHighestProcessedTxid);
451     } else if (writer.getLength() == fileLengthAtLastSync) {
452       // we haven't written anything out, just advance the highestSyncedSequence since we may only
453       // stamped some region sequence id.
454       highestSyncedTxid.set(newHighestProcessedTxid);
455       finishSync(false);
456       tryFinishRoll();
457     }
458   }

460   private static final Comparator<SyncFuture> SEQ_COMPARATOR = new Comparator<SyncFuture>() {
461 
462     @Override
463     public int compare(SyncFuture o1, SyncFuture o2) {
464       int c = Long.compare(o1.getTxid(), o2.getTxid());
465       return c != 0 ? c : Integer.compare(System.identityHashCode(o1), System.identityHashCode(o2));
466     }

469   private final Runnable consumer = new Runnable() {
470 
471     @Override
472     public void run() {
473       synchronized (waitingConsumePayloads) {
474         assert consumerScheduled;
475         if (writerBroken) {
476           // waiting for reschedule after rollWriter.
477           consumerScheduled = false;
478           return;
479         }
480         if (waitingRoll) {
481           // we may have toWriteEntries if the consume method does not write all pending entries
482           // out, this is usually happen if we have too many toWriteEntries that exceeded the
483           // batchSize limit.
484           if (waitingAppendEntries.isEmpty()) {
485             consumerScheduled = false;
486             return;
487           }
488         } else {
489           for (Payload p; (p = waitingConsumePayloads.pollFirst()) != null;) {
490             if (p.entry != null) {
491               waitingAppendEntries.addLast(p.entry);
492             } else if (p.sync != null) {
493               syncFutures.add(p.sync);
494             } else {
495               rollPromise = p.roll;
496               waitingRoll = true;
497               break;
498             }
499           }
500         }
501       }
502       consume();
503       synchronized (waitingConsumePayloads) {
504         if (waitingRoll) {
505           if (waitingAppendEntries.isEmpty()) {
506             consumerScheduled = false;
507             return;
508           }
509         } else {
510           if (waitingConsumePayloads.isEmpty() && waitingAppendEntries.isEmpty()) {
511             consumerScheduled = false;
512             return;
513           }
514         }
515       }
516       // reschedule if we still have something to write.
517       eventLoop.execute(this);
518     }
519   };
520 
521   private boolean shouldScheduleConsumer() {
522     if (writerBroken || waitingRoll) {
523       return false;
524     }
525     if (consumerScheduled) {
526       return false;
527     }
528     consumerScheduled = true;
529     return true;
530   }

532   @Override
533   public long append(HRegionInfo hri, WALKey key, WALEdit edits, boolean inMemstore)
534       throws IOException {
535     boolean scheduleTask;
536     long txid;
537     synchronized (waitingConsumePayloads) {
538       if (this.closed) {
539         throw new IOException("Cannot append; log is closed");
540       }
541       txid = nextTxid++;
542       FSWALEntry entry = new FSWALEntry(txid, key, edits, hri, inMemstore);
543       scheduleTask = shouldScheduleConsumer();
544       waitingConsumePayloads.add(new Payload(entry));
545     }
546     if (scheduleTask) {
547       eventLoop.execute(consumer);
548     }
549     return txid;
550   }

552   @Override
553   public void sync() throws IOException {
554     TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
555     try {
556       SyncFuture future;
557       boolean scheduleTask;
558       synchronized (waitingConsumePayloads) {
559         scheduleTask = shouldScheduleConsumer();
560         future = getSyncFuture(nextTxid - 1, scope.detach());
561         waitingConsumePayloads.addLast(new Payload(future));
562       }
563       if (scheduleTask) {
564         eventLoop.execute(consumer);
565       }
566       scope = Trace.continueSpan(blockOnSync(future));
567     } finally {
568       assert scope == NullScope.INSTANCE || !scope.isDetached();
569       scope.close();
570     }
571   }
572 
573   @Override
574   public void sync(long txid) throws IOException {
575     if (highestSyncedTxid.get() >= txid) {
576       return;
577     }
578     TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
579     try {
580       SyncFuture future = getSyncFuture(txid, scope.detach());
581       boolean scheduleTask;
582       synchronized (waitingConsumePayloads) {
583         scheduleTask = shouldScheduleConsumer();
584         waitingConsumePayloads.addLast(new Payload(future));
585       }
586       if (scheduleTask) {
587         eventLoop.execute(consumer);
588       }
589       scope = Trace.continueSpan(blockOnSync(future));
590     } finally {
591       assert scope == NullScope.INSTANCE || !scope.isDetached();
592       scope.close();
593     }
594   }

601   @Override
602   protected AsyncWriter createWriterInstance(Path path) throws IOException {
603     boolean overwrite = false;
604     for (int retry = 0;; retry++) {
605       try {
606         return AsyncFSWALProvider.createAsyncWriter(conf, fs, path, overwrite, eventLoop);
607       } catch (RemoteException e) {
608         LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
609         if (shouldRetryCreate(e)) {
610           if (retry >= createMaxRetries) {
611             break;
612           }
613         } else {
614           throw e.unwrapRemoteException();
615         }
616       } catch (NameNodeException e) {
617         throw e;
618       } catch (IOException e) {
619         LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
620         if (retry >= createMaxRetries) {
621           break;
622         }
623         // overwrite the old broken file.
624         overwrite = true;
625         try {
626           Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
627         } catch (InterruptedException ie) {
628           throw new InterruptedIOException();
629         }
630       }
631     }
632     throw new IOException("Failed to create wal log writer " + path + " after retrying "
633         + createMaxRetries + " time(s)");
634   }

636   private void waitForSafePoint() {
637     Future<Void> roll;
638     boolean scheduleTask;
639     synchronized (waitingConsumePayloads) {
640       if (!writerBroken && this.writer != null) {
641         Promise<Void> promise = eventLoop.newPromise();
642         if (consumerScheduled) {
643           scheduleTask = false;
644         } else {
645           scheduleTask = consumerScheduled = true;
646         }
647         waitingConsumePayloads.addLast(new Payload(promise));
