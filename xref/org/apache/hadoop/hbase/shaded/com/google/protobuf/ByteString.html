53  /**
54   * Immutable sequence of bytes.  Substring is supported by sharing the reference
55   * to the immutable underlying bytes.  Concatenation is likewise supported
56   * without copying (long strings) by building a tree of pieces in
57   * {@link RopeByteString}.
58   * <p>
59   * Like {@link String}, the contents of a {@link ByteString} can never be
60   * observed to change, not even in the presence of a data race or incorrect
61   * API usage in the client code.
62   *
63   * @author crazybob@google.com Bob Lee
64   * @author kenton@google.com Kenton Varda
65   * @author carlanton@google.com Carl Haverl
66   * @author martinrb@google.com Martin Buchholz
67   */
68  public abstract class ByteString implements Iterable<Byte>, Serializable {

71     * When two strings to be concatenated have a combined length shorter than
72     * this, we just copy their bytes on {@link #concat(ByteString)}.
73     * The trade-off is copy size versus the overhead of creating tree nodes
74     * in {@link RopeByteString}.
75     */
76    static final int CONCATENATE_BY_COPY_SIZE = 128;

91    /**
92     * An interface to efficiently copy {@code byte[]}.
93     *
94     * <p>One of the noticeable costs of copying a byte[] into a new array using
95     * {@code System.arraycopy} is nullification of a new buffer before the copy. It has been shown
96     * the Hotspot VM is capable to intrisicfy {@code Arrays.copyOfRange} operation to avoid this
97     * expensive nullification and provide substantial performance gain. Unfortunately this does not
98     * hold on Android runtimes and could make the copy slightly slower due to additional code in
99     * the {@code Arrays.copyOfRange}. Thus we provide two different implementation for array copier
100    * for Hotspot and Android runtimes.
101    */
102   private interface ByteArrayCopier {
103     /**
104      * Copies the specified range of the specified array into a new array
105      */
106     byte[] copyFrom(byte[] bytes, int offset, int size);
107   }
108
109   /** Implementation of {@code ByteArrayCopier} which uses {@link System#arraycopy}. */
110   private static final class SystemByteArrayCopier implements ByteArrayCopier {
111     @Override
112     public byte[] copyFrom(byte[] bytes, int offset, int size) {
113       byte[] copy = new byte[size];
114       System.arraycopy(bytes, offset, copy, 0, size);
115       return copy;
116     }
117   }
118
119   /** Implementation of {@code ByteArrayCopier} which uses {@link Arrays#copyOfRange}. */
120   private static final class ArraysByteArrayCopier implements ByteArrayCopier {
121     @Override
122     public byte[] copyFrom(byte[] bytes, int offset, int size) {
123       return Arrays.copyOfRange(bytes, offset, offset + size);
124     }
125   }

153    * #substring(int, int)} first if necessary.

161   /**
162    * Return a {@link ByteString.ByteIterator} over the bytes in the ByteString.
163    * To avoid auto-boxing, you may get the iterator manually and call
164    * {@link ByteIterator#nextByte()}.
165    *
166    * @return the iterator
167    */
168   @Override
169   public final ByteIterator iterator() {
170     return new ByteIterator() {
171       private int position = 0;
172       private final int limit = size();
173
174       @Override
175       public boolean hasNext() {
176         return position < limit;
177       }
178
179       @Override
180       public Byte next() {
181         // Boxing calls Byte.valueOf(byte), which does not instantiate.
182         return nextByte();
183       }
184
185       @Override
186       public byte nextByte() {
187         try {
188           return byteAt(position++);
189         } catch (IndexOutOfBoundsException e) {
190           throw new NoSuchElementException(e.getMessage());
191         }
192       }
193
194       @Override
195       public void remove() {
196         throw new UnsupportedOperationException();
197       }
198     };
199   }

415   /**
416    * Completely reads the given stream's bytes into a
417    * {@code ByteString}, blocking if necessary until all bytes are
418    * read through to the end of the stream.
419    *
420    * <b>Performance notes:</b> The returned {@code ByteString} is an
421    * immutable tree of byte arrays ("chunks") of the stream data.  The
422    * first chunk is small, with subsequent chunks each being double
423    * the size, up to 8K.
424    *
425    * <p>Each byte read from the input stream will be copied twice to ensure
426    * that the resulting ByteString is truly immutable.
427    *
428    * @param streamToDrain The source stream, which is read completely
429    *     but not closed.
430    * @return A new {@code ByteString} which is made up of chunks of
431    *     various sizes, depending on the behavior of the underlying
432    *     stream.
433    * @throws IOException IOException is thrown if there is a problem
434    *     reading the underlying stream.
435    */
436   public static ByteString readFrom(InputStream streamToDrain)
437       throws IOException {
438     return readFrom(streamToDrain, MIN_READ_FROM_CHUNK_SIZE, MAX_READ_FROM_CHUNK_SIZE);
439   }
440
441   /**
442    * Completely reads the given stream's bytes into a
443    * {@code ByteString}, blocking if necessary until all bytes are
444    * read through to the end of the stream.
445    *
446    * <b>Performance notes:</b> The returned {@code ByteString} is an
447    * immutable tree of byte arrays ("chunks") of the stream data.  The
448    * chunkSize parameter sets the size of these byte arrays.
449    *
450    * <p>Each byte read from the input stream will be copied twice to ensure
451    * that the resulting ByteString is truly immutable.
452    *
453    * @param streamToDrain The source stream, which is read completely
454    *     but not closed.
455    * @param chunkSize The size of the chunks in which to read the
456    *     stream.
457    * @return A new {@code ByteString} which is made up of chunks of
458    *     the given size.
459    * @throws IOException IOException is thrown if there is a problem
460    *     reading the underlying stream.
461    */
462   public static ByteString readFrom(InputStream streamToDrain, int chunkSize)
463       throws IOException {
464     return readFrom(streamToDrain, chunkSize, chunkSize);
465   }
466
467   // Helper method that takes the chunk size range as a parameter.
468   public static ByteString readFrom(InputStream streamToDrain, int minChunkSize,
469       int maxChunkSize) throws IOException {
470     Collection<ByteString> results = new ArrayList<ByteString>();
471
472     // copy the inbound bytes into a list of chunks; the chunk size
473     // grows exponentially to support both short and long streams.
474     int chunkSize = minChunkSize;
475     while (true) {
476       ByteString chunk = readChunk(streamToDrain, chunkSize);
477       if (chunk == null) {
478         break;
479       }
480       results.add(chunk);
481       chunkSize = Math.min(chunkSize * 2, maxChunkSize);
482     }
483
484     return ByteString.copyFrom(results);
485   }
486
487   /**
488    * Blocks until a chunk of the given size can be made from the
489    * stream, or EOF is reached.  Calls read() repeatedly in case the
490    * given stream implementation doesn't completely fill the given
491    * buffer in one read() call.
492    *
493    * @return A chunk of the desired size, or else a chunk as large as
494    * was available when end of stream was reached. Returns null if the
495    * given stream had no more data in it.
496    */
497   private static ByteString readChunk(InputStream in, final int chunkSize)
498       throws IOException {
499       final byte[] buf = new byte[chunkSize];
500       int bytesRead = 0;
501       while (bytesRead < chunkSize) {
502         final int count = in.read(buf, bytesRead, chunkSize - bytesRead);
503         if (count == -1) {
504           break;
505         }
506         bytesRead += count;
507       }
508
509       if (bytesRead == 0) {
510         return null;
511       }
512
513       // Always make a copy since InputStream could steal a reference to buf.
514       return ByteString.copyFrom(buf, 0, bytesRead);
515   }

520   /**
521    * Concatenate the given {@code ByteString} to this one. Short concatenations,
522    * of total size smaller than {@link ByteString#CONCATENATE_BY_COPY_SIZE}, are
523    * produced by copying the underlying bytes (as per Rope.java, <a
524    * href="http://www.cs.ubc.ca/local/reading/proceedings/spe91-95/spe/vol25/issue12/spe986.pdf">
525    * BAP95 </a>. In general, the concatenate involves no copying.
526    *
527    * @param other string to concatenate
528    * @return a new {@code ByteString} instance
529    */
530   public final ByteString concat(ByteString other) {
531     if (Integer.MAX_VALUE - size() < other.size()) {
532       throw new IllegalArgumentException("ByteString would be too long: " +
533           size() + "+" + other.size());
534     }
535
536     return RopeByteString.concatenate(this, other);
537   }

539   /**
540    * Concatenates all byte strings in the iterable and returns the result.
541    * This is designed to run in O(list size), not O(total bytes).
542    *
543    * <p>The returned {@code ByteString} is not necessarily a unique object.
544    * If the list is empty, the returned object is the singleton empty
545    * {@code ByteString}.  If the list has only one element, that
546    * {@code ByteString} will be returned without copying.
547    *
548    * @param byteStrings strings to be concatenated
549    * @return new {@code ByteString}
550    */
551   public static ByteString copyFrom(Iterable<ByteString> byteStrings) {
552     // Determine the size;
553     final int size;
554     if (!(byteStrings instanceof Collection)) {
555       int tempSize = 0;
556       for (Iterator<ByteString> iter = byteStrings.iterator(); iter.hasNext();
557           iter.next(), ++tempSize) {
558       }
559       size = tempSize;
560     } else {
561       size = ((Collection<ByteString>) byteStrings).size();
562     }
563
564     if (size == 0) {
565       return EMPTY;
566     }
567
568     return balancedConcat(byteStrings.iterator(), size);
569   }
570
571   // Internal function used by copyFrom(Iterable<ByteString>).
572   // Create a balanced concatenation of the next "length" elements from the
573   // iterable.
574   private static ByteString balancedConcat(Iterator<ByteString> iterator, int length) {
575     assert length >= 1;
576     ByteString result;
577     if (length == 1) {
578       result = iterator.next();
579     } else {
580       int halfLength = length >>> 1;
581       ByteString left = balancedConcat(iterator, halfLength);
582       ByteString right = balancedConcat(iterator, length - halfLength);
583       result = left.concat(right);
584     }
585     return result;
586   }

687   /**
688    * Writes this {@link ByteString} to the provided {@link ByteOutput}. Calling
689    * this method may result in multiple operations on the target {@link ByteOutput}.
690    *
691    * <p>This method may expose internal backing buffers of the {@link ByteString} to the {@link
692    * ByteOutput} in order to avoid additional copying overhead. It would be possible for a malicious

694    *
695    * @param  byteOutput  the output target to receive the bytes
696    * @throws IOException  if an I/O error occurs

939   /**
940    * Outputs to a {@code ByteString} instance. Call {@link #toByteString()} to
941    * create the {@code ByteString} instance.
942    */
943   public static final class Output extends OutputStream {
944     // Implementation note.
945     // The public methods of this class must be synchronized.  ByteStrings
946     // are guaranteed to be immutable.  Without some sort of locking, it could
947     // be possible for one thread to call toByteSring(), while another thread
948     // is still modifying the underlying byte array.
949
950     private static final byte[] EMPTY_BYTE_ARRAY = new byte[0];
951     // argument passed by user, indicating initial capacity.
952     private final int initialCapacity;
953     // ByteStrings to be concatenated to create the result
954     private final ArrayList<ByteString> flushedBuffers;
955     // Total number of bytes in the ByteStrings of flushedBuffers
956     private int flushedBuffersTotalBytes;
957     // Current buffer to which we are writing
958     private byte[] buffer;
959     // Location in buffer[] to which we write the next byte.
960     private int bufferPos;
961
962     /**
963      * Creates a new ByteString output stream with the specified
964      * initial capacity.
965      *
966      * @param initialCapacity  the initial capacity of the output stream.
967      */
968     Output(int initialCapacity) {
969       if (initialCapacity < 0) {
970         throw new IllegalArgumentException("Buffer size < 0");
971       }
972       this.initialCapacity = initialCapacity;
973       this.flushedBuffers = new ArrayList<ByteString>();
974       this.buffer = new byte[initialCapacity];
975     }
976
977     @Override
978     public synchronized void write(int b) {
979       if (bufferPos == buffer.length) {
980         flushFullBuffer(1);
981       }
982       buffer[bufferPos++] = (byte)b;
983     }
984
985     @Override
986     public synchronized void write(byte[] b, int offset, int length)  {
987       if (length <= buffer.length - bufferPos) {
988         // The bytes can fit into the current buffer.
989         System.arraycopy(b, offset, buffer, bufferPos, length);
990         bufferPos += length;
991       } else {
992         // Use up the current buffer
993         int copySize  = buffer.length - bufferPos;
994         System.arraycopy(b, offset, buffer, bufferPos, copySize);
995         offset += copySize;
996         length -= copySize;
997         // Flush the buffer, and get a new buffer at least big enough to cover
998         // what we still need to output
999         flushFullBuffer(length);
1000         System.arraycopy(b, offset, buffer, 0 /* count */, length);
1001         bufferPos = length;
1002       }
1003     }
1004
1005     /**
1006      * Creates a byte string. Its size is the current size of this output
1007      * stream and its output has been copied to it.
1008      *
1009      * @return  the current contents of this output stream, as a byte string.
1010      */
1011     public synchronized ByteString toByteString() {
1012       flushLastBuffer();
1013       return ByteString.copyFrom(flushedBuffers);
1014     }
1015
1016     /**
1017      * Implement java.util.Arrays.copyOf() for jdk 1.5.
1018      */
1019     private byte[] copyArray(byte[] buffer, int length) {
1020       byte[] result = new byte[length];
1021       System.arraycopy(buffer, 0, result, 0, Math.min(buffer.length, length));
1022       return result;
1023     }
1024
1025     /**
1026      * Writes the complete contents of this byte array output stream to
1027      * the specified output stream argument.
1028      *
1029      * @param out the output stream to which to write the data.
1030      * @throws IOException  if an I/O error occurs.
1031      */
1032     public void writeTo(OutputStream out) throws IOException {
1033       ByteString[] cachedFlushBuffers;
1034       byte[] cachedBuffer;
1035       int cachedBufferPos;
1036       synchronized (this) {
1037         // Copy the information we need into local variables so as to hold
1038         // the lock for as short a time as possible.
1039         cachedFlushBuffers =
1040             flushedBuffers.toArray(new ByteString[flushedBuffers.size()]);
1041         cachedBuffer = buffer;
1042         cachedBufferPos = bufferPos;
1043       }
1044       for (ByteString byteString : cachedFlushBuffers) {
1045         byteString.writeTo(out);
1046       }
1047
1048       out.write(copyArray(cachedBuffer, cachedBufferPos));
1049     }
1050
1051     /**
1052      * Returns the current size of the output stream.
1053      *
1054      * @return  the current size of the output stream
1055      */
1056     public synchronized int size() {
1057       return flushedBuffersTotalBytes + bufferPos;
1058     }
1059
1060     /**
1061      * Resets this stream, so that all currently accumulated output in the
1062      * output stream is discarded. The output stream can be used again,
1063      * reusing the already allocated buffer space.
1064      */
1065     public synchronized void reset() {
1066       flushedBuffers.clear();
1067       flushedBuffersTotalBytes = 0;
1068       bufferPos = 0;
1069     }
1070
1071     @Override
1072     public String toString() {
1073       return String.format("<ByteString.Output@%s size=%d>",
1074           Integer.toHexString(System.identityHashCode(this)), size());
1075     }
1076
1077     /**
1078      * Internal function used by writers.  The current buffer is full, and the
1079      * writer needs a new buffer whose size is at least the specified minimum
1080      * size.
1081      */
1082     private void flushFullBuffer(int minSize)  {
1083       flushedBuffers.add(new LiteralByteString(buffer));
1084       flushedBuffersTotalBytes += buffer.length;
1085       // We want to increase our total capacity by 50%, but as a minimum,
1086       // the new buffer should also at least be >= minSize and
1087       // >= initial Capacity.
1088       int newSize = Math.max(initialCapacity,
1089           Math.max(minSize, flushedBuffersTotalBytes >>> 1));
1090       buffer = new byte[newSize];
1091       bufferPos = 0;
1092     }
1093
1094     /**
1095      * Internal function used by {@link #toByteString()}. The current buffer may
1096      * or may not be full, but it needs to be flushed.
1097      */
1098     private void flushLastBuffer()  {
1099       if (bufferPos < buffer.length) {
1100         if (bufferPos > 0) {
1101           byte[] bufferCopy = copyArray(buffer, bufferPos);
1102           flushedBuffers.add(new LiteralByteString(bufferCopy));
1103         }
1104         // We reuse this buffer for further writes.
1105       } else {
1106         // Buffer is completely full.  Huzzah.
1107         flushedBuffers.add(new LiteralByteString(buffer));
1108         // 99% of the time, we're not going to use this OutputStream again.
1109         // We set buffer to an empty byte stream so that we're handling this
1110         // case without wasting space.  In the rare case that more writes
1111         // *do* occur, this empty buffer will be flushed and an appropriately
1112         // sized new buffer will be created.
1113         buffer = EMPTY_BYTE_ARRAY;
1114       }
1115       flushedBuffersTotalBytes += bufferPos;
1116       bufferPos = 0;
1117     }
1118   }

1213       if (index < 0) {

1216       throw new ArrayIndexOutOfBoundsException("Index > length: " + index + ", " + size);

1220   /**
1221    * Checks that the given range falls within the bounds of an array
1222    *
1223    * @param startIndex the start index of the range (inclusive)
1224    * @param endIndex the end index of the range (exclusive)
1225    * @param size the size of the array.
1226    * @return the length of the range.
1227    * @throws IndexOutOfBoundsException some or all of the range falls outside of the array.
1228    */
1229   static int checkRange(int startIndex, int endIndex, int size) {
1230     final int length = endIndex - startIndex;
1231     if ((startIndex | endIndex | length | (size - endIndex)) < 0) {
1232       if (startIndex < 0) {
1233         throw new IndexOutOfBoundsException("Beginning index: " + startIndex + " < 0");
1234       }
1235       if (endIndex < startIndex) {
1236         throw new IndexOutOfBoundsException(
1237             "Beginning index larger than ending index: " + startIndex + ", " + endIndex);
1238       }
1239       // endIndex >= size
1240       throw new IndexOutOfBoundsException("End index: " + endIndex + " >= " + size);
1241     }
1242     return length;
1243   }

1387         LiteralByteString otherAsLiteral = (LiteralByteString) other;
1388         // If we know the hash codes and they are not equal, we know the byte
1389         // strings are not equal.
1390         int thisHash = peekCachedHashCode();
1391         int thatHash = otherAsLiteral.peekCachedHashCode();
1392         if (thisHash != 0 && thatHash != 0 && thisHash != thatHash) {
1393           return false;
1394         }
1395
1396         return equalsRange((LiteralByteString) other, 0, size());
1397       } else {
1398         // RopeByteString and NioByteString.
1399         return other.equals(this);
1400       }
1401     }

1404      * Check equality of the substring of given length of this object starting at
1405      * zero with another {@code LiteralByteString} substring starting at offset.
1406      *
1407      * @param other  what to compare a substring in
1408      * @param offset offset into other
1409      * @param length number of bytes to compare
1410      * @return true for equality of substrings, else false.
1411      */
1412     @Override
1413     final boolean equalsRange(ByteString other, int offset, int length) {
1414       if (length > other.size()) {
1415         throw new IllegalArgumentException("Length too large: " + length + size());
1416       }
1417       if (offset + length > other.size()) {
1418         throw new IllegalArgumentException(
1419             "Ran off end of other: " + offset + ", " + length + ", " + other.size());
1420       }
1421
1422       if (other instanceof LiteralByteString) {
1423         LiteralByteString lbsOther = (LiteralByteString) other;
1424         byte[] thisBytes = bytes;
1425         byte[] otherBytes = lbsOther.bytes;
1426         int thisLimit = getOffsetIntoBytes() + length;
1427         for (
1428             int thisIndex = getOffsetIntoBytes(),
1429                 otherIndex = lbsOther.getOffsetIntoBytes() + offset;
1430             (thisIndex < thisLimit); ++thisIndex, ++otherIndex) {
1431           if (thisBytes[thisIndex] != otherBytes[otherIndex]) {
1432             return false;
1433           }
1434         }
1435         return true;
1436       }
1437
1438       return other.substring(offset, offset + length).equals(substring(0, length));
1439     }
