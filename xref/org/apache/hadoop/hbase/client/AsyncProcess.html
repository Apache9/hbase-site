60  /**
61   * This class  allows a continuous flow of requests. It's written to be compatible with a
62   * synchronous caller such as HTable.
63   * <p>
64   * The caller sends a buffer of operation, by calling submit. This class extract from this list
65   * the operations it can send, i.e. the operations that are on region that are not considered
66   * as busy. The process is asynchronous, i.e. it returns immediately when if has finished to
67   * iterate on the list. If, and only if, the maximum number of current task is reached, the call
68   * to submit will block. Alternatively, the caller can call submitAll, in which case all the
69   * operations will be sent. Each call to submit returns a future-like object that can be used
70   * to track operation progress.
71   * </p>
72   * <p>
73   * The class manages internally the retries.
74   * </p>
75   * <p>
76   * The class can be constructed in regular mode, or "global error" mode. In global error mode,
77   * AP tracks errors across all calls (each "future" also has global view of all errors). That
78   * mode is necessary for backward compat with HTable behavior, where multiple submissions are
79   * made and the errors can propagate using any put/flush call, from previous calls.
80   * In "regular" mode, the errors are tracked inside the Future object that is returned.
81   * The results are always tracked inside the Future object and can be retrieved when the call
82   * has finished. Partial results can also be retrieved if some part of multi-request failed.
83   * </p>
84   * <p>
85   * This class is thread safe in regular mode; in global error code, submitting operations and
86   * retrieving errors from different threads may be not thread safe.
87   * Internally, the class is thread safe enough to manage simultaneously new submission and results
88   * arising from older operations.
89   * </p>
90   * <p>
91   * Internally, this class works with {@link Row}, this mean it could be theoretically used for
92   * gets as well.
93   * </p>
94   */

222   public AsyncProcess(ClusterConnection hc, Configuration conf, ExecutorService pool,
223       RpcRetryingCallerFactory rpcCaller, boolean useGlobalErrors,
224       RpcControllerFactory rpcFactory, int rpcTimeout, int operationTimeout) {
225     if (hc == null) {
226       throw new IllegalArgumentException("ClusterConnection cannot be null.");
227     }
228
229     this.connection = hc;
230     this.pool = pool;
231     this.globalErrors = useGlobalErrors ? new BatchErrors() : null;
232
233     this.id = COUNTER.incrementAndGet();
234
235     this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
236         HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
237     // how many times we could try in total, one more than retry number
238     this.numTries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
239         HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER) + 1;
240     this.rpcTimeout = rpcTimeout;
241     this.operationTimeout = operationTimeout;
242     this.primaryCallTimeoutMicroseconds = conf.getInt(PRIMARY_CALL_TIMEOUT_KEY, 10000);
243
244     this.maxTotalConcurrentTasks = conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
245       HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS);
246     this.maxConcurrentTasksPerServer = conf.getInt(HConstants.HBASE_CLIENT_MAX_PERSERVER_TASKS,
247           HConstants.DEFAULT_HBASE_CLIENT_MAX_PERSERVER_TASKS);
248     this.maxConcurrentTasksPerRegion = conf.getInt(HConstants.HBASE_CLIENT_MAX_PERREGION_TASKS,
249           HConstants.DEFAULT_HBASE_CLIENT_MAX_PERREGION_TASKS);
250     this.maxHeapSizePerRequest = conf.getLong(HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE,
251           DEFAULT_HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE);
252     this.maxHeapSizeSubmit = conf.getLong(HBASE_CLIENT_MAX_SUBMIT_HEAPSIZE, DEFAULT_HBASE_CLIENT_MAX_SUBMIT_HEAPSIZE);
253     this.startLogErrorsCnt =
254         conf.getInt(START_LOG_ERRORS_AFTER_COUNT_KEY, DEFAULT_START_LOG_ERRORS_AFTER_COUNT);
255
256     if (this.maxTotalConcurrentTasks <= 0) {
257       throw new IllegalArgumentException("maxTotalConcurrentTasks=" + maxTotalConcurrentTasks);
258     }
259     if (this.maxConcurrentTasksPerServer <= 0) {
260       throw new IllegalArgumentException("maxConcurrentTasksPerServer=" +
261           maxConcurrentTasksPerServer);
262     }
263     if (this.maxConcurrentTasksPerRegion <= 0) {
264       throw new IllegalArgumentException("maxConcurrentTasksPerRegion=" +
265           maxConcurrentTasksPerRegion);
266     }
267     if (this.maxHeapSizePerRequest <= 0) {
268       throw new IllegalArgumentException("maxHeapSizePerServer=" +
269           maxHeapSizePerRequest);
270     }
271
272     if (this.maxHeapSizeSubmit <= 0) {
273       throw new IllegalArgumentException("maxHeapSizeSubmit=" +
274           maxHeapSizeSubmit);
275     }
276     // Server tracker allows us to do faster, and yet useful (hopefully), retries.
277     // However, if we are too useful, we might fail very quickly due to retry count limit.
278     // To avoid this, we are going to cheat for now (see HBASE-7659), and calculate maximum
279     // retry time if normal retries were used. Then we will retry until this time runs out.
280     // If we keep hitting one server, the net effect will be the incremental backoff, and
281     // essentially the same number of retries as planned. If we have to do faster retries,
282     // we will do more retries in aggregate, but the user will be none the wiser.
283     this.serverTrackerTimeout = 0;
284     for (int i = 0; i < this.numTries; ++i) {
285       serverTrackerTimeout += ConnectionUtils.getPauseTime(this.pause, i);
286     }
287
288     this.rpcCallerFactory = rpcCaller;
289     this.rpcFactory = rpcFactory;
290     this.logBatchErrorDetails = conf.getBoolean(LOG_DETAILS_FOR_BATCH_ERROR, false);
291
292     this.thresholdToLogUndoneTaskDetails =
293         conf.getInt(THRESHOLD_TO_LOG_UNDONE_TASK_DETAILS,
294           DEFAULT_THRESHOLD_TO_LOG_UNDONE_TASK_DETAILS);
295   }

331   /**
332    * Extract from the rows list what we can submit. The rows we can not submit are kept in the
333    * list. Does not send requests to replicas (not currently used for anything other
334    * than streaming puts anyway).
335    *
336    * @param pool ExecutorService to use.
337    * @param tableName The table for which this request is needed.
338    * @param callback Batch callback. Only called on success (94 behavior).
339    * @param needResults Whether results are needed, or can be discarded.
340    * @param rows - the submitted row. Modified by the method: we remove the rows we took.
341    * @param atLeastOne true if we should submit at least a subset.
342    */
343   public <CResult> AsyncRequestFuture submit(ExecutorService pool, TableName tableName,
344       RowAccess<? extends Row> rows, boolean atLeastOne, Batch.Callback<CResult> callback,
345       boolean needResults) throws InterruptedIOException {
346     if (rows.isEmpty()) {
347       return NO_REQS_RESULT;
348     }
349
350     Map<ServerName, MultiAction> actionsByServer =
351         new HashMap<ServerName, MultiAction>();
352     List<Action> retainedActions = new ArrayList<Action>(rows.size());
353
354     NonceGenerator ng = this.connection.getNonceGenerator();
355     long nonceGroup = ng.getNonceGroup(); // Currently, nonce group is per entire client.
356
357     // Location errors that happen before we decide what requests to take.
358     List<Exception> locationErrors = null;
359     List<Integer> locationErrorRows = null;
360     RowCheckerHost checker = createRowCheckerHost();
361     boolean firstIter = true;
362     do {
363       // Wait until there is at least one slot for a new task.
364       waitForMaximumCurrentTasks(maxTotalConcurrentTasks - 1, tableName.getNameAsString());
365       int posInList = -1;
366       if (!firstIter) {
367         checker.reset();
368       }
369       Iterator<? extends Row> it = rows.iterator();
370       while (it.hasNext()) {
371         Row r = it.next();
372         HRegionLocation loc;
373         try {
374           if (r == null) {
375             throw new IllegalArgumentException("#" + id + ", row cannot be null");
376           }
377           // Make sure we get 0-s replica.
378           RegionLocations locs = connection.locateRegion(
379               tableName, r.getRow(), true, true, RegionReplicaUtil.DEFAULT_REPLICA_ID);
380           if (locs == null || locs.isEmpty() || locs.getDefaultRegionLocation() == null) {
381             throw new IOException("#" + id + ", no location found, aborting submit for"
382                 + " tableName=" + tableName + " rowkey=" + Bytes.toStringBinary(r.getRow()));
383           }
384           loc = locs.getDefaultRegionLocation();
385         } catch (IOException ex) {
386           locationErrors = new ArrayList<Exception>();
387           locationErrorRows = new ArrayList<Integer>();
388           LOG.error("Failed to get region location ", ex);
389           // This action failed before creating ars. Retain it, but do not add to submit list.
390           // We will then add it to ars in an already-failed state.
391           retainedActions.add(new Action(r, ++posInList));
392           locationErrors.add(ex);
393           locationErrorRows.add(posInList);
394           it.remove();
395           break; // Backward compat: we stop considering actions on location error.
396         }
397         long rowSize = (r instanceof Mutation) ? ((Mutation) r).heapSize() : 0;
398         ReturnCode code = checker.canTakeOperation(loc, rowSize);
399         if (code == ReturnCode.END) {
400           break;
401         }
402         if (code == ReturnCode.INCLUDE) {
403           Action action = new Action(r, ++posInList);
404           setNonce(ng, r, action);
405           retainedActions.add(action);
406           // TODO: replica-get is not supported on this path
407           byte[] regionName = loc.getRegionInfo().getRegionName();
408           addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
409           it.remove();
410         }
411       }
412       firstIter = false;
413     } while (retainedActions.isEmpty() && atLeastOne && (locationErrors == null));
414
415     if (retainedActions.isEmpty()) return NO_REQS_RESULT;
416
417     return submitMultiActions(tableName, retainedActions, nonceGroup, callback, null, needResults,
418         locationErrors, locationErrorRows, actionsByServer, pool);
419   }
420
421   private RowCheckerHost createRowCheckerHost() {
422     return new RowCheckerHost(Arrays.asList(
423         new TaskCountChecker(maxTotalConcurrentTasks,
424           maxConcurrentTasksPerServer,
425           maxConcurrentTasksPerRegion,
426           tasksInProgress,
427           taskCounterPerServer,
428           taskCounterPerRegion)
429         , new RequestSizeChecker(maxHeapSizePerRequest)
430         , new SubmittedSizeChecker(maxHeapSizeSubmit)
431     ));
432   }
433   <CResult> AsyncRequestFuture submitMultiActions(TableName tableName,
434       List<Action> retainedActions, long nonceGroup, Batch.Callback<CResult> callback,
435       Object[] results, boolean needResults, List<Exception> locationErrors,
436       List<Integer> locationErrorRows, Map<ServerName, MultiAction> actionsByServer,
437       ExecutorService pool) {
438     AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(
439       tableName, retainedActions, nonceGroup, pool, callback, results, needResults, null, -1);
440     // Add location errors if any
441     if (locationErrors != null) {
442       for (int i = 0; i < locationErrors.size(); ++i) {
443         int originalIndex = locationErrorRows.get(i);
444         Row row = retainedActions.get(originalIndex).getAction();
445         ars.manageError(originalIndex, row,
446             AsyncRequestFutureImpl.Retry.NO_LOCATION_PROBLEM, locationErrors.get(i), null);
447       }
448     }
449     ars.sendMultiAction(actionsByServer, 1, null, false);
450     return ars;
451   }

461   /**
462    * Helper that is used when grouping the actions per region server.
463    *
464    * @param server - server
465    * @param regionName - regionName
466    * @param action - the action to add to the multiaction
467    * @param actionsByServer the multiaction per server
468    * @param nonceGroup Nonce group.
469    */
470   static void addAction(ServerName server, byte[] regionName, Action action,
471       Map<ServerName, MultiAction> actionsByServer, long nonceGroup) {
472     MultiAction multiAction = actionsByServer.get(server);
473     if (multiAction == null) {
474       multiAction = new MultiAction();
475       actionsByServer.put(server, multiAction);
476     }
477     if (action.hasNonce() && !multiAction.hasNonceGroup()) {
478       multiAction.setNonceGroup(nonceGroup);
479     }
480
481     multiAction.add(regionName, action);
482   }

484   public <CResult> AsyncRequestFuture submitAll(ExecutorService pool, TableName tableName,

488   /**
489    * Submit immediately the list of rows, whatever the server status. Kept for backward
490    * compatibility: it allows to be used with the batch interface that return an array of objects.
491    *
492    * @param pool ExecutorService to use.
493    * @param tableName name of the table for which the submission is made.
494    * @param rows the list of rows.
495    * @param callback the callback.
496    * @param results Optional array to return the results thru; backward compat.
497    * @param rpcTimeout rpc timeout for this batch, set -1 if want to use current setting.
498    */
499   public <CResult> AsyncRequestFuture submitAll(ExecutorService pool, TableName tableName,
500       List<? extends Row> rows, Batch.Callback<CResult> callback, Object[] results,
501       CancellableRegionServerCallable callable, int rpcTimeout) {
502     List<Action> actions = new ArrayList<Action>(rows.size());
503
504     // The position will be used by the processBatch to match the object array returned.
505     int posInList = -1;
506     NonceGenerator ng = this.connection.getNonceGenerator();
507     for (Row r : rows) {
508       posInList++;
509       if (r instanceof Put) {
510         Put put = (Put) r;
511         if (put.isEmpty()) {
512           throw new IllegalArgumentException("No columns to insert for #" + (posInList+1)+ " item");
513         }
514       }
515       Action action = new Action(r, posInList);
516       setNonce(ng, r, action);
517       actions.add(action);
518     }
519     AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(
520         tableName, actions, ng.getNonceGroup(), getPool(pool), callback, results, results != null,
521         callable, rpcTimeout);
522     ars.groupAndSendMultiAction(actions, 1);
523     return ars;
524   }
525
526   private void setNonce(NonceGenerator ng, Row r, Action action) {
527     if (!(r instanceof Append) && !(r instanceof Increment)) return;
528     action.setNonce(ng.newNonce()); // Action handles NO_NONCE, so it's ok if ng is disabled.
529   }

541   /** Wait until the async does not have more than max tasks in progress. */
542   protected void waitForMaximumCurrentTasks(int max, String tableName)
543       throws InterruptedIOException {
544     waitForMaximumCurrentTasks(max, tasksInProgress, id, tableName);
545   }

547   // Break out this method so testable
548   @VisibleForTesting
549   void waitForMaximumCurrentTasks(int max, final AtomicLong tasksInProgress, final long id,
550       String tableName) throws InterruptedIOException {
551     long lastLog = EnvironmentEdgeManager.currentTime();
552     long currentInProgress, oldInProgress = Long.MAX_VALUE;
553     while ((currentInProgress = tasksInProgress.get()) > max) {
554       if (oldInProgress != currentInProgress) { // Wait for in progress to change.
555         long now = EnvironmentEdgeManager.currentTime();
556         if (now > lastLog + 10000) {
557           lastLog = now;
558           LOG.info("#" + id + ", waiting for some tasks to finish. Expected max="
559               + max + ", tasksInProgress=" + currentInProgress +
560               " hasError=" + hasError() + tableName == null ? "" : ", tableName=" + tableName);
561           if (currentInProgress <= thresholdToLogUndoneTaskDetails) {
562             logDetailsOfUndoneTasks(currentInProgress);
563           }
564         }
565       }
566       oldInProgress = currentInProgress;
567       try {
568         synchronized (tasksInProgress) {
569           if (tasksInProgress.get() == oldInProgress) {
570             tasksInProgress.wait(10);
571           }
572         }
573       } catch (InterruptedException e) {
574         throw new InterruptedIOException("#" + id + ", interrupted." +
575             " currentNumberOfTask=" + currentInProgress);
576       }
577     }
578   }

580   void logDetailsOfUndoneTasks(long taskInProgress) {
581     ArrayList<ServerName> servers = new ArrayList<ServerName>();
582     for (Map.Entry<ServerName, AtomicInteger> entry : taskCounterPerServer.entrySet()) {
583       if (entry.getValue().get() > 0) {
584         servers.add(entry.getKey());
585       }
586     }
587     LOG.info("Left over " + taskInProgress + " task(s) are processed on server(s): " + servers);
588     if (taskInProgress <= THRESHOLD_TO_LOG_REGION_DETAILS) {
589       ArrayList<String> regions = new ArrayList<String>();
590       for (Map.Entry<byte[], AtomicInteger> entry : taskCounterPerRegion.entrySet()) {
591         if (entry.getValue().get() > 0) {
592           regions.add(Bytes.toString(entry.getKey()));
593         }
594       }
595       LOG.info("Regions against which left over task(s) are processed: " + regions);
596     }

633    * increment the tasks counters for a given set of regions. MT safe.
634    */
635   protected void incTaskCounters(Collection<byte[]> regions, ServerName sn) {
636     tasksInProgress.incrementAndGet();
637
638     AtomicInteger serverCnt = taskCounterPerServer.get(sn);
639     if (serverCnt == null) {
640       taskCounterPerServer.putIfAbsent(sn, new AtomicInteger());
641       serverCnt = taskCounterPerServer.get(sn);
642     }
643     serverCnt.incrementAndGet();
644
645     for (byte[] regBytes : regions) {
646       AtomicInteger regionCnt = taskCounterPerRegion.get(regBytes);
647       if (regionCnt == null) {
648         regionCnt = new AtomicInteger();
649         AtomicInteger oldCnt = taskCounterPerRegion.putIfAbsent(regBytes, regionCnt);
650         if (oldCnt != null) {
651           regionCnt = oldCnt;
652         }
653       }
654       regionCnt.incrementAndGet();
655     }
656   }

660    */
661   protected void decTaskCounters(Collection<byte[]> regions, ServerName sn) {
662     for (byte[] regBytes : regions) {
663       AtomicInteger regionCnt = taskCounterPerRegion.get(regBytes);
664       regionCnt.decrementAndGet();
665     }
666
667     taskCounterPerServer.get(sn).decrementAndGet();
668     tasksInProgress.decrementAndGet();
669     synchronized (tasksInProgress) {
670       tasksInProgress.notifyAll();
671     }

703   @VisibleForTesting
704   static class RowCheckerHost {
705     private final List<RowChecker> checkers;
706     private boolean isEnd = false;
707     RowCheckerHost(final List<RowChecker> checkers) {
708       this.checkers = checkers;
709     }
710     void reset() throws InterruptedIOException {
711       isEnd = false;
712       InterruptedIOException e = null;
713       for (RowChecker checker : checkers) {
714         try {
715           checker.reset();
716         } catch (InterruptedIOException ex) {
717           e = ex;
718         }
719       }
720       if (e != null) {
721         throw e;
722       }
723     }
724     ReturnCode canTakeOperation(HRegionLocation loc, long rowSize) {
725       if (isEnd) {
726         return ReturnCode.END;
727       }
728       ReturnCode code = ReturnCode.INCLUDE;
729       for (RowChecker checker : checkers) {
730         switch (checker.canTakeOperation(loc, rowSize)) {
731           case END:
732             isEnd = true;
733             code = ReturnCode.END;
734             break;
735           case SKIP:
736             code = ReturnCode.SKIP;
737             break;
738           case INCLUDE:
739           default:
740             break;
741         }
742         if (code == ReturnCode.END) {
743           break;
744         }
745       }
746       for (RowChecker checker : checkers) {
747         checker.notifyFinal(code, loc, rowSize);
748       }
749       return code;
750     }

817   /**
818    * limit the max number of tasks in an AsyncProcess.
819    */
820   @VisibleForTesting
821   static class TaskCountChecker implements RowChecker {
822     private static final long MAX_WAITING_TIME = 1000; //ms
823     private final Set<HRegionInfo> regionsIncluded = new HashSet<>();
824     private final Set<ServerName> serversIncluded = new HashSet<>();
825     private final int maxConcurrentTasksPerRegion;
826     private final int maxTotalConcurrentTasks;
827     private final int maxConcurrentTasksPerServer;
828     private final Map<byte[], AtomicInteger> taskCounterPerRegion;
829     private final Map<ServerName, AtomicInteger> taskCounterPerServer;
830     private final Set<byte[]> busyRegions = new TreeSet<>(Bytes.BYTES_COMPARATOR);
831     private final AtomicLong tasksInProgress;
832     TaskCountChecker(final int maxTotalConcurrentTasks,
833       final int maxConcurrentTasksPerServer,
834       final int maxConcurrentTasksPerRegion,
835       final AtomicLong tasksInProgress,
836       final Map<ServerName, AtomicInteger> taskCounterPerServer,
837       final Map<byte[], AtomicInteger> taskCounterPerRegion) {
838       this.maxTotalConcurrentTasks = maxTotalConcurrentTasks;
839       this.maxConcurrentTasksPerRegion = maxConcurrentTasksPerRegion;
840       this.maxConcurrentTasksPerServer = maxConcurrentTasksPerServer;
841       this.taskCounterPerRegion = taskCounterPerRegion;
842       this.taskCounterPerServer = taskCounterPerServer;
843       this.tasksInProgress = tasksInProgress;
844     }
845     @Override
846     public void reset() throws InterruptedIOException {
847       // prevent the busy-waiting
848       waitForRegion();
849       regionsIncluded.clear();
850       serversIncluded.clear();
851       busyRegions.clear();
852     }
853     private void waitForRegion() throws InterruptedIOException {
854       if (busyRegions.isEmpty()) {
855         return;
856       }
857       EnvironmentEdge ee = EnvironmentEdgeManager.getDelegate();
858       final long start = ee.currentTime();
859       while ((ee.currentTime() - start) <= MAX_WAITING_TIME) {
860         for (byte[] region : busyRegions) {
861           AtomicInteger count = taskCounterPerRegion.get(region);
862           if (count == null || count.get() < maxConcurrentTasksPerRegion) {
863             return;
864           }
865         }
866         try {
867           synchronized (tasksInProgress) {
868             tasksInProgress.wait(10);
869           }
870         } catch (InterruptedException e) {
871           throw new InterruptedIOException("Interrupted." +
872               " tasksInProgress=" + tasksInProgress);
873         }
874       }
875     }
876     /**
877      * 1) check the regions is allowed.
878      * 2) check the concurrent tasks for regions.
879      * 3) check the total concurrent tasks.
880      * 4) check the concurrent tasks for server.
881      * @param loc
882      * @param rowSize
883      * @return
884      */
885     @Override
886     public ReturnCode canTakeOperation(HRegionLocation loc, long rowSize) {
887
888       HRegionInfo regionInfo = loc.getRegionInfo();
889       if (regionsIncluded.contains(regionInfo)) {
890         // We already know what to do with this region.
891         return ReturnCode.INCLUDE;
892       }
893       AtomicInteger regionCnt = taskCounterPerRegion.get(loc.getRegionInfo().getRegionName());
894       if (regionCnt != null && regionCnt.get() >= maxConcurrentTasksPerRegion) {
895         // Too many tasks on this region already.
896         return ReturnCode.SKIP;
897       }
898       int newServers = serversIncluded.size()
899         + (serversIncluded.contains(loc.getServerName()) ? 0 : 1);
900       if ((newServers + tasksInProgress.get()) > maxTotalConcurrentTasks) {
901         // Too many tasks.
902         return ReturnCode.SKIP;
903       }
904       AtomicInteger serverCnt = taskCounterPerServer.get(loc.getServerName());
905       if (serverCnt != null && serverCnt.get() >= maxConcurrentTasksPerServer) {
906         // Too many tasks for this individual server
907         return ReturnCode.SKIP;
908       }
909       return ReturnCode.INCLUDE;
910     }
911
912     @Override
913     public void notifyFinal(ReturnCode code, HRegionLocation loc, long rowSize) {
914       if (code == ReturnCode.INCLUDE) {
915         regionsIncluded.add(loc.getRegionInfo());
916         serversIncluded.add(loc.getServerName());
917       }
918       busyRegions.add(loc.getRegionInfo().getRegionName());
919     }
920   }
