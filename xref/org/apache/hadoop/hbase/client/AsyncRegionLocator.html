18  package org.apache.hadoop.hbase.client;

20  import java.io.Closeable;
21  import java.io.IOException;
22  import java.util.concurrent.CompletableFuture;
23
24  import org.apache.commons.io.IOUtils;
25  import org.apache.hadoop.conf.Configuration;
26  import org.apache.hadoop.hbase.HRegionLocation;
27  import org.apache.hadoop.hbase.ServerName;
28  import org.apache.hadoop.hbase.TableName;
29  import org.apache.hadoop.hbase.classification.InterfaceAudience;
30
31  /**
32   * TODO: reimplement using aync connection when the scan logic is ready. The current implementation
33   * is based on the blocking client.
34   */
35  @InterfaceAudience.Private
36  class AsyncRegionLocator implements Closeable {
37
38    private final ConnectionImplementation conn;
39
40    AsyncRegionLocator(Configuration conf) throws IOException {
41      conn = (ConnectionImplementation) ConnectionFactory.createConnection(conf);
42    }
43
44    CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
45        boolean reload) {
46      CompletableFuture<HRegionLocation> future = new CompletableFuture<>();
47      try {
48        future.complete(conn.getRegionLocation(tableName, row, reload));
49      } catch (IOException e) {
50        future.completeExceptionally(e);
51      }
52      return future;
53    }
54
55    void updateCachedLocations(TableName tableName, byte[] regionName, byte[] row, Object exception,
56        ServerName source) {
57      conn.updateCachedLocations(tableName, regionName, row, exception, source);
58    }
59
60    @Override
61    public void close() {
62      IOUtils.closeQuietly(conn);
63    }
64  }
