24  import org.apache.hadoop.hbase.ServerName;
25  import org.apache.hadoop.hbase.TableName;
26  import org.apache.hadoop.hbase.classification.InterfaceAudience;
27  import org.apache.hadoop.hbase.protobuf.generated.ClientProtos.ClientService;

29  /**
30   * Implementations call a RegionServer and implement {@link #call(int)}.
31   * Passed to a {@link RpcRetryingCaller} so we retry on fail.
32   * TODO: this class is actually tied to one region, because most of the paths make use of
33   *       the regioninfo part of location when building requests. The only reason it works for
34   *       multi-region requests (e.g. batch) is that they happen to not use the region parts.
35   *       This could be done cleaner (e.g. having a generic parameter and 2 derived classes,
36   *       RegionCallable and actual RegionServerCallable with ServerName.
37   * @param <T> the class that the ServerCallable handles
38   */
39  @InterfaceAudience.Private
40  public abstract class RegionServerCallable<T> extends AbstractRegionServerCallable<T> implements
41      RetryingCallable<T> {
42
43    private ClientService.BlockingInterface stub;
44
45    /**
46     * @param connection Connection to use.
47     * @param tableName Table name to which <code>row</code> belongs.
48     * @param row The row we want in <code>tableName</code>.
49     */
50    public RegionServerCallable(Connection connection, TableName tableName, byte [] row) {
51      super(connection, tableName, row);
52    }
53
54    void setClientByServiceName(ServerName service) throws IOException {
55      this.setStub(getConnection().getClient(service));
56    }
57
58    /**
59     * @return Client Rpc protobuf communication stub
60     */
61    protected ClientService.BlockingInterface getStub() {
62      return this.stub;
63    }
64
65    /**
66     * Set the client protobuf communication stub
67     * @param stub to set
68     */
69    void setStub(final ClientService.BlockingInterface stub) {
70      this.stub = stub;
71    }
72  }
