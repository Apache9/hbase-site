34  /**
35   * Implementations make an rpc call against a RegionService via a protobuf Service.
36   * Implement rpcCall(). Be sure to make use of the RpcController that this instance is carrying
37   * via {@link #getRpcController()}.
38   *
39   * <p>TODO: this class is actually tied to one region, because most of the paths make use of
40   *       the regioninfo part of location when building requests. The only reason it works for
41   *       multi-region requests (e.g. batch) is that they happen to not use the region parts.
42   *       This could be done cleaner (e.g. having a generic parameter and 2 derived classes,
43   *       RegionCallable and actual RegionServerCallable with ServerName.
44   *
45   * @param <T> the class that the ServerCallable handles
46   */
47  @InterfaceAudience.Private
48  public abstract class RegionServerCallable<T> extends AbstractRegionServerCallable<T> {
49    private ClientService.BlockingInterface stub;
50
51    /* This is 99% of the time a PayloadCarryingRpcController but this RegionServerCallable is
52     * also used doing Coprocessor Endpoints and in this case, it is a ServerRpcControllable which is
53     * not a PayloadCarryingRpcController. Too hard to untangle it all at this stage since
54     * downstreamers are using RegionServerCallable invoking CPEPs so just do ugly instanceof
55     * checks in the below.
56     */
57    private final RpcController rpcController;
58
59    /**
60     * @param connection Connection to use.
61     * @param tableName Table name to which <code>row</code> belongs.
62     * @param row The row we want in <code>tableName</code>.
63     */
64    public RegionServerCallable(Connection connection, RpcControllerFactory rpcControllerFactory,
65        TableName tableName, byte [] row) {
66      this(connection, rpcControllerFactory.newController(), tableName, row);
67    }
68
69    public RegionServerCallable(Connection connection, RpcController rpcController,
70        TableName tableName, byte [] row) {
71      super(connection, tableName, row);
72      this.rpcController = rpcController;
73    }
74
75    void setClientByServiceName(ServerName service) throws IOException {
76      this.setStub(getConnection().getClient(service));
77    }
78
79    /**
80     * @return Client Rpc protobuf communication stub
81     */
82    protected ClientService.BlockingInterface getStub() {
83      return this.stub;
84    }
85
86    /**
87     * Set the client protobuf communication stub
88     * @param stub to set
89     */
90    void setStub(final ClientService.BlockingInterface stub) {
91      this.stub = stub;
92    }
93
94    /**
95     * Override that changes call Exception from {@link Exception} to {@link IOException}. It also
96     * does setup of an rpcController and calls through to the unimplemented
97     * rpcCall() method. If rpcController is an instance of PayloadCarryingRpcController,
98     * we will set a timeout on it.
99     */
100   @Override
101   public T call(int callTimeout) throws IOException {
102     try {
103       if (this.rpcController != null) {
104         // Do a reset to clear previous states, such as CellScanner.
105         this.rpcController.reset();
106         if (this.rpcController instanceof HBaseRpcController) {
107           HBaseRpcController pcrc = (HBaseRpcController)this.rpcController;
108           // If it is an instance of PayloadCarryingRpcController, we can set priority on the
109           // controller based off the tableName. RpcController may be null in tests when mocking so allow
110           // for null controller.
111           pcrc.setPriority(tableName);
112           pcrc.setCallTimeout(callTimeout);
113         }
114       }
115       return rpcCall();
116     } catch (Exception e) {
117       throw ProtobufUtil.handleRemoteException(e);
118     }
119   }
120
121   /**
122    * Run the RPC call. Implement this method. To get at the rpcController that has been created
123    * and configured to make this rpc call, use getRpcController(). We are trying to contain
124    * rpcController references so we don't pollute codebase with protobuf references; keep the
125    * protobuf references contained and only present in a few classes rather than all about the
126    * code base.
127    * @throws Exception
128    */
129   protected abstract T rpcCall() throws Exception;
130
131   protected RpcController getRpcController() {
132     return this.rpcController;
133   }
134
135   /**
136    * Get the RpcController CellScanner.
137    * If the RpcController is a PayloadCarryingRpcController, which it is in all cases except
138    * when we are processing Coprocessor Endpoint, then this method returns a reference to the
139    * CellScanner that the PayloadCarryingRpcController is carrying. Do it up here in this Callable
140    * so we don't have to scatter ugly instanceof tests around the codebase. Will fail if called in
141    * a Coproccessor Endpoint context. Should never happen.
142    */
143   protected CellScanner getRpcControllerCellScanner() {
144     return ((HBaseRpcController)this.rpcController).cellScanner();
145   }
146
147   protected void setRpcControllerCellScanner(CellScanner cellScanner) {
148     ((HBaseRpcController)this.rpcController).setCellScanner(cellScanner);
149   }
150 }
