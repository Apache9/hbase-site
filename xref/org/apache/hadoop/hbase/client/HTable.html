78  /**
79   * An implementation of {@link Table}. Used to communicate with a single HBase table.
80   * Lightweight. Get as needed and just close when done.
81   * Instances of this class SHOULD NOT be constructed directly.
82   * Obtain an instance via {@link Connection}. See {@link ConnectionFactory}
83   * class comment for an example of how.
84   *
85   * <p>This class is NOT thread safe for reads nor writes.
86   * In the case of writes (Put, Delete), the underlying write buffer can
87   * be corrupted if multiple threads contend over a single HTable instance.
88   * In the case of reads, some fields used by a Scan are shared among all threads.
89   *
90   * <p>HTable is no longer a client API. Use {@link Table} instead. It is marked
91   * InterfaceAudience.Private indicating that this is an HBase-internal class as defined in
92   * <a href="https://hadoop.apache.org/docs/current/hadoop-project-dist/hadoop-common/InterfaceClassification.html">Hadoop
93   * Interface Classification</a>
94   * There are no guarantees for backwards source / binary compatibility and methods or class can
95   * change or go away without deprecation.
96   *
97   * @see Table
98   * @see Admin
99   * @see Connection
100  * @see ConnectionFactory
101  */

103 @InterfaceStability.Stable

128   // Marked Private @since 1.0
129   @InterfaceAudience.Private
130   public static ThreadPoolExecutor getDefaultExecutor(Configuration conf) {
131     int maxThreads = conf.getInt("hbase.htable.threads.max", Integer.MAX_VALUE);
132     if (maxThreads == 0) {
133       maxThreads = 1; // is there a better default?
134     }
135     int corePoolSize = conf.getInt("hbase.htable.threads.coresize", 1);
136     long keepAliveTime = conf.getLong("hbase.htable.threads.keepalivetime", 60);
137
138     // Using the "direct handoff" approach, new threads will only be created
139     // if it is necessary and will grow unbounded. This could be bad but in HCM
140     // we only create as many Runnables as there are region servers. It means
141     // it also scales when new region servers are added.
142     ThreadPoolExecutor pool = new ThreadPoolExecutor(corePoolSize, maxThreads, keepAliveTime,
143       TimeUnit.SECONDS, new SynchronousQueue<Runnable>(), Threads.newDaemonThreadFactory("htable"));
144     pool.allowCoreThreadTimeOut(true);
145     return pool;
146   }

148   /**
149    * Creates an object to access a HBase table.
150    * Used by HBase internally.  DO NOT USE. See {@link ConnectionFactory} class comment for how to
151    * get a {@link Table} instance (use {@link Table} instead of {@link HTable}).
152    * @param tableName Name of the table.
153    * @param connection Connection to be used.
154    * @param pool ExecutorService to be used.
155    * @throws IOException if a remote or network exception occurs
156    */
157   @InterfaceAudience.Private
158   protected HTable(TableName tableName, final ClusterConnection connection,
159       final ConnectionConfiguration tableConfig,
160       final RpcRetryingCallerFactory rpcCallerFactory,
161       final RpcControllerFactory rpcControllerFactory,
162       final ExecutorService pool) throws IOException {
163     if (connection == null || connection.isClosed()) {
164       throw new IllegalArgumentException("Connection is null or closed.");
165     }
166     this.tableName = tableName;
167     this.cleanupConnectionOnClose = false;
168     this.connection = connection;
169     this.configuration = connection.getConfiguration();
170     this.connConfiguration = tableConfig;
171     this.pool = pool;
172     if (pool == null) {
173       this.pool = getDefaultExecutor(this.configuration);
174       this.cleanupPoolOnClose = true;
175     } else {
176       this.cleanupPoolOnClose = false;
177     }
178
179     this.rpcCallerFactory = rpcCallerFactory;
180     this.rpcControllerFactory = rpcControllerFactory;
181
182     this.finishSetup();
183   }

213   /**
214    * setup this HTable's parameter based on the passed configuration
215    */
216   private void finishSetup() throws IOException {
217     if (connConfiguration == null) {
218       connConfiguration = new ConnectionConfiguration(configuration);
219     }
220
221     this.operationTimeout = tableName.isSystemTable() ?
222         connConfiguration.getMetaOperationTimeout() : connConfiguration.getOperationTimeout();
223     this.readRpcTimeout = configuration.getInt(HConstants.HBASE_RPC_READ_TIMEOUT_KEY,
224         configuration.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
225             HConstants.DEFAULT_HBASE_RPC_TIMEOUT));
226     this.writeRpcTimeout = configuration.getInt(HConstants.HBASE_RPC_WRITE_TIMEOUT_KEY,
227         configuration.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
228             HConstants.DEFAULT_HBASE_RPC_TIMEOUT));
229     this.scannerCaching = connConfiguration.getScannerCaching();
230     this.scannerMaxResultSize = connConfiguration.getScannerMaxResultSize();
231     if (this.rpcCallerFactory == null) {
232       this.rpcCallerFactory = connection.getNewRpcRetryingCallerFactory(configuration);
233     }
234     if (this.rpcControllerFactory == null) {
235       this.rpcControllerFactory = RpcControllerFactory.instantiate(configuration);
236     }
237
238     // puts need to track errors globally due to how the APIs currently work.
239     multiAp = this.connection.getAsyncProcess();
240     this.locator = new HRegionLocator(getName(), connection);
241   }

296   /**
297    * Get the corresponding start keys and regions for an arbitrary range of
298    * keys.
299    * <p>
300    * @param startKey Starting row in range, inclusive
301    * @param endKey Ending row in range
302    * @param includeEndKey true if endRow is inclusive, false if exclusive
303    * @param reload true to reload information or false to use cached information
304    * @return A pair of list of start keys and list of HRegionLocations that
305    *         contain the specified range
306    * @throws IOException if a remote or network exception occurs
307    */
308   private Pair<List<byte[]>, List<HRegionLocation>> getKeysAndRegionsInRange(
309       final byte[] startKey, final byte[] endKey, final boolean includeEndKey,
310       final boolean reload) throws IOException {
311     final boolean endKeyIsEndOfTable = Bytes.equals(endKey,HConstants.EMPTY_END_ROW);
312     if ((Bytes.compareTo(startKey, endKey) > 0) && !endKeyIsEndOfTable) {
313       throw new IllegalArgumentException(
314         "Invalid range: " + Bytes.toStringBinary(startKey) +
315         " > " + Bytes.toStringBinary(endKey));
316     }
317     List<byte[]> keysInRange = new ArrayList<byte[]>();
318     List<HRegionLocation> regionsInRange = new ArrayList<HRegionLocation>();
319     byte[] currentKey = startKey;
320     do {
321       HRegionLocation regionLocation = getRegionLocator().getRegionLocation(currentKey, reload);
322       keysInRange.add(currentKey);
323       regionsInRange.add(regionLocation);
324       currentKey = regionLocation.getRegionInfo().getEndKey();
325     } while (!Bytes.equals(currentKey, HConstants.EMPTY_END_ROW)
326         && (endKeyIsEndOfTable || Bytes.compareTo(currentKey, endKey) < 0
327             || (includeEndKey && Bytes.compareTo(currentKey, endKey) == 0)));
328     return new Pair<List<byte[]>, List<HRegionLocation>>(keysInRange,
329         regionsInRange);
330   }

332   /**
333    * The underlying {@link HTable} must not be closed.
334    * {@link Table#getScanner(Scan)} has other usage details.
335    */
336   @Override
337   public ResultScanner getScanner(final Scan scan) throws IOException {
338     if (scan.getBatch() > 0 && scan.isSmall()) {
339       throw new IllegalArgumentException("Small scan should not be used with batching");
340     }
341
342     if (scan.getCaching() <= 0) {
343       scan.setCaching(scannerCaching);
344     }
345     if (scan.getMaxResultSize() <= 0) {
346       scan.setMaxResultSize(scannerMaxResultSize);
347     }
348
349     Boolean async = scan.isAsyncPrefetch();
350     if (async == null) {
351       async = connConfiguration.isClientScannerAsyncPrefetch();
352     }
353
354     if (scan.isReversed()) {
355       if (scan.isSmall()) {
356         return new ClientSmallReversedScanner(getConfiguration(), scan, getName(),
357             this.connection, this.rpcCallerFactory, this.rpcControllerFactory,
358             pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
359       } else {
360         return new ReversedClientScanner(getConfiguration(), scan, getName(),
361             this.connection, this.rpcCallerFactory, this.rpcControllerFactory,
362             pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
363       }
364     }
365
366     if (scan.isSmall()) {
367       return new ClientSmallScanner(getConfiguration(), scan, getName(),
368           this.connection, this.rpcCallerFactory, this.rpcControllerFactory,
369           pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
370     } else {
371       if (async) {
372         return new ClientAsyncPrefetchScanner(getConfiguration(), scan, getName(), this.connection,
373             this.rpcCallerFactory, this.rpcControllerFactory,
374             pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
375       } else {
376         return new ClientSimpleScanner(getConfiguration(), scan, getName(), this.connection,
377             this.rpcCallerFactory, this.rpcControllerFactory,
378             pool, connConfiguration.getReplicaCallTimeoutMicroSecondScan());
379       }
380     }
381   }

411     return get(get, get.isCheckExistenceOnly());
412   }

414   private Result get(Get get, final boolean checkExistenceOnly) throws IOException {
415     // if we are changing settings to the get, clone it.
416     if (get.isCheckExistenceOnly() != checkExistenceOnly || get.getConsistency() == null) {
417       get = ReflectionUtils.newInstance(get.getClass(), get);
418       get.setCheckExistenceOnly(checkExistenceOnly);
419       if (get.getConsistency() == null){
420         get.setConsistency(defaultConsistency);
421       }
422     }
423
424     if (get.getConsistency() == Consistency.STRONG) {
425       final Get configuredGet = get;
426       ClientServiceCallable<Result> callable = new ClientServiceCallable<Result>(this.connection, getName(),
427           get.getRow(), this.rpcControllerFactory.newController()) {
428         @Override
429         protected Result rpcCall() throws Exception {
430           ClientProtos.GetRequest request = RequestConverter.buildGetRequest(
431               getLocation().getRegionInfo().getRegionName(), configuredGet);
432           ClientProtos.GetResponse response = doGet(request);
433           return response == null? null:
434             ProtobufUtil.toResult(response.getResult(), getRpcControllerCellScanner());
435         }
436       };
437       return rpcCallerFactory.<Result>newCaller(readRpcTimeout).callWithRetries(callable,
438           this.operationTimeout);
439     }
440
441     // Call that takes into account the replica
442     RpcRetryingCallerWithReadReplicas callable = new RpcRetryingCallerWithReadReplicas(
443         rpcControllerFactory, tableName, this.connection, get, pool,
444         connConfiguration.getRetriesNumber(),
445         operationTimeout, readRpcTimeout,
446         connConfiguration.getPrimaryCallTimeoutMicroSecond());
447     return callable.call(operationTimeout);
448   }

450   /**
451    * {@inheritDoc}
452    */
453   @Override
454   public Result[] get(List<Get> gets) throws IOException {
455     if (gets.size() == 1) {
456       return new Result[]{get(gets.get(0))};
457     }
458     try {
459       Object[] r1 = new Object[gets.size()];
460       batch((List<? extends Row>)gets, r1, readRpcTimeout);
461       // Translate.
462       Result [] results = new Result[r1.length];
463       int i = 0;
464       for (Object obj: r1) {
465         // Batch ensures if there is a failure we get an exception instead
466         results[i++] = (Result)obj;
467       }
468       return results;
469     } catch (InterruptedException e) {
470       throw (InterruptedIOException)new InterruptedIOException().initCause(e);
471     }
472   }

520     CancellableRegionServerCallable<SingleResponse> callable =

529         return ResponseConverter.getResult(request, response, getRpcControllerCellScanner());
530       }
531     };
532     List<Delete> rows = Collections.singletonList(delete);
533     AsyncRequestFuture ars = multiAp.submitAll(pool, tableName, rows,
534         null, null, callable, writeRpcTimeout);
535     ars.waitUntilDone();
536     if (ars.hasError()) {
537       throw ars.getErrors();
538     }

541   /**
542    * {@inheritDoc}
543    */
544   @Override
545   public void delete(final List<Delete> deletes)
546   throws IOException {
547     Object[] results = new Object[deletes.size()];
548     try {
549       batch(deletes, results, writeRpcTimeout);
550     } catch (InterruptedException e) {
551       throw (InterruptedIOException)new InterruptedIOException().initCause(e);
552     } finally {
553       // mutate list so that it is empty for complete success, or contains only failed records
554       // results are returned in the same order as the requests in list walk the list backwards,
555       // so we can remove from list without impacting the indexes of earlier members
556       for (int i = results.length - 1; i>=0; i--) {
557         // if result is not null, it succeeded
558         if (results[i] instanceof Result) {
559           deletes.remove(i);
560         }
561       }
562     }
563   }

585   /**
586    * {@inheritDoc}
587    */
588   @Override
589   public void mutateRow(final RowMutations rm) throws IOException {
590     CancellableRegionServerCallable<MultiResponse> callable =
591       new CancellableRegionServerCallable<MultiResponse>(this.connection, getName(), rm.getRow(),
592           rpcControllerFactory.newController(), writeRpcTimeout, new RetryingTimeTracker().start()){
593       @Override
594       protected MultiResponse rpcCall() throws Exception {
595         RegionAction.Builder regionMutationBuilder = RequestConverter.buildRegionAction(
596             getLocation().getRegionInfo().getRegionName(), rm);
597         regionMutationBuilder.setAtomic(true);
598         MultiRequest request =
599             MultiRequest.newBuilder().addRegionAction(regionMutationBuilder.build()).build();
600         ClientProtos.MultiResponse response = doMulti(request);
601         ClientProtos.RegionActionResult res = response.getRegionActionResultList().get(0);
602         if (res.hasException()) {
603           Throwable ex = ProtobufUtil.toException(res.getException());
604           if (ex instanceof IOException) {
605             throw (IOException) ex;
606           }
607           throw new IOException("Failed to mutate row: " + Bytes.toStringBinary(rm.getRow()), ex);
608         }
609         return ResponseConverter.getResults(request, response, getRpcControllerCellScanner());
610       }
611     };
612     AsyncRequestFuture ars = multiAp.submitAll(pool, tableName, rm.getMutations(),
613         null, null, callable, writeRpcTimeout);
614     ars.waitUntilDone();
615     if (ars.hasError()) {
616       throw ars.getErrors();
617     }
618   }

1171       byte[] startKey, byte[] endKey, final R responsePrototype, final Callback<R> callback)
1172       throws ServiceException, Throwable {

1175       startKey = HConstants.EMPTY_START_ROW;
1176     }
1177     if (endKey == null) {
1178       endKey = HConstants.EMPTY_END_ROW;
1179     }
1180     // get regions covered by the row range
1181     Pair<List<byte[]>, List<HRegionLocation>> keysAndRegions =
1182         getKeysAndRegionsInRange(startKey, endKey, true);
1183     List<byte[]> keys = keysAndRegions.getFirst();
1184     List<HRegionLocation> regions = keysAndRegions.getSecond();
1185
1186     // check if we have any calls to make
1187     if (keys.isEmpty()) {
1188       LOG.info("No regions were selected by key range start=" + Bytes.toStringBinary(startKey) +
1189           ", end=" + Bytes.toStringBinary(endKey));
1190       return;
1191     }
1192
1193     List<RegionCoprocessorServiceExec> execs =
1194         new ArrayList<RegionCoprocessorServiceExec>(keys.size());
1195     final Map<byte[], RegionCoprocessorServiceExec> execsByRow =
1196         new TreeMap<byte[], RegionCoprocessorServiceExec>(Bytes.BYTES_COMPARATOR);
1197     for (int i = 0; i < keys.size(); i++) {
1198       final byte[] rowKey = keys.get(i);
1199       final byte[] region = regions.get(i).getRegionInfo().getRegionName();
1200       RegionCoprocessorServiceExec exec =
1201           new RegionCoprocessorServiceExec(region, rowKey, methodDescriptor, request);
1202       execs.add(exec);
1203       execsByRow.put(rowKey, exec);
1204     }
1205
1206     // tracking for any possible deserialization errors on success callback
1207     // TODO: it would be better to be able to reuse AsyncProcess.BatchErrors here
1208     final List<Throwable> callbackErrorExceptions = new ArrayList<Throwable>();
1209     final List<Row> callbackErrorActions = new ArrayList<Row>();
1210     final List<String> callbackErrorServers = new ArrayList<String>();
1211     Object[] results = new Object[execs.size()];
1212
1213     AsyncProcess asyncProcess =
1214         new AsyncProcess(connection, configuration, pool,
1215             RpcRetryingCallerFactory.instantiate(configuration, connection.getStatisticsTracker()),
1216             true, RpcControllerFactory.instantiate(configuration), readRpcTimeout,
1217             operationTimeout);
1218
1219     AsyncRequestFuture future = asyncProcess.submitAll(null, tableName, execs,
1220         new Callback<ClientProtos.CoprocessorServiceResult>() {
1221           @Override
1222           public void update(byte[] region, byte[] row,
1223                               ClientProtos.CoprocessorServiceResult serviceResult) {

1225               LOG.trace("Received result for endpoint " + methodDescriptor.getFullName() +
1226                   ": region=" + Bytes.toStringBinary(region) +
1227                   ", row=" + Bytes.toStringBinary(row) +
1228                   ", value=" + serviceResult.getValue().getValue());
1229             }
1230             try {
1231               Message.Builder builder = responsePrototype.newBuilderForType();
1232               org.apache.hadoop.hbase.protobuf.ProtobufUtil.mergeFrom(builder,
1233                   serviceResult.getValue().getValue().toByteArray());
1234               callback.update(region, row, (R) builder.build());
1235             } catch (IOException e) {

1237                   e);

1248       throw future.getErrors();
1249     } else if (!callbackErrorExceptions.isEmpty()) {
1250       throw new RetriesExhaustedWithDetailsException(callbackErrorExceptions, callbackErrorActions,
1251           callbackErrorServers);
