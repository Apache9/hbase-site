60  /**
61   * The context, and return value, for a single submit/submitAll call.
62   * Note on how this class (one AP submit) works. Initially, all requests are split into groups
63   * by server; request is sent to each server in parallel; the RPC calls are not async so a
64   * thread per server is used. Every time some actions fail, regions/locations might have
65   * changed, so we re-group them by server and region again and send these groups in parallel
66   * too. The result, in case of retries, is a "tree" of threads, with parent exiting after
67   * scheduling children. This is why lots of code doesn't require any synchronization.
68   */
69  @InterfaceAudience.Private
70  class AsyncRequestFutureImpl<CResult> implements AsyncRequestFuture {

76    /**
77     * Runnable (that can be submitted to thread pool) that waits for when it's time
78     * to issue replica calls, finds region replicas, groups the requests by replica and
79     * issues the calls (on separate threads, via sendMultiAction).
80     * This is done on a separate thread because we don't want to wait on user thread for
81     * our asynchronous call, and usually we have to wait before making replica calls.
82     */
83    private final class ReplicaCallIssuingRunnable implements Runnable {
84      private final long startTime;
85      private final List<Action<Row>> initialActions;
86
87      public ReplicaCallIssuingRunnable(List<Action<Row>> initialActions, long startTime) {
88        this.initialActions = initialActions;
89        this.startTime = startTime;
90      }
91
92      @Override
93      public void run() {
94        boolean done = false;
95        if (asyncProcess.primaryCallTimeoutMicroseconds > 0) {
96          try {
97            done = waitUntilDone(startTime * 1000L + asyncProcess.primaryCallTimeoutMicroseconds);
98          } catch (InterruptedException ex) {
99            LOG.error("Replica thread was interrupted - no replica calls: " + ex.getMessage());
100           return;
101         }
102       }
103       if (done) return; // Done within primary timeout
104       Map<ServerName, MultiAction<Row>> actionsByServer =
105           new HashMap<ServerName, MultiAction<Row>>();
106       List<Action<Row>> unknownLocActions = new ArrayList<Action<Row>>();
107       if (replicaGetIndices == null) {
108         for (int i = 0; i < results.length; ++i) {
109           addReplicaActions(i, actionsByServer, unknownLocActions);
110         }
111       } else {
112         for (int replicaGetIndice : replicaGetIndices) {
113           addReplicaActions(replicaGetIndice, actionsByServer, unknownLocActions);
114         }
115       }
116       if (!actionsByServer.isEmpty()) {
117         sendMultiAction(actionsByServer, 1, null, unknownLocActions.isEmpty());
118       }
119       if (!unknownLocActions.isEmpty()) {
120         actionsByServer = new HashMap<ServerName, MultiAction<Row>>();
121         for (Action<Row> action : unknownLocActions) {
122           addReplicaActionsAgain(action, actionsByServer);
123         }
124         // Some actions may have completely failed, they are handled inside addAgain.
125         if (!actionsByServer.isEmpty()) {
126           sendMultiAction(actionsByServer, 1, null, true);
127         }
128       }
129     }
130
131     /**
132      * Add replica actions to action map by server.
133      * @param index Index of the original action.
134      * @param actionsByServer The map by server to add it to.
135      */
136     private void addReplicaActions(int index, Map<ServerName, MultiAction<Row>> actionsByServer,
137                                    List<Action<Row>> unknownReplicaActions) {
138       if (results[index] != null) return; // opportunistic. Never goes from non-null to null.
139       Action<Row> action = initialActions.get(index);
140       RegionLocations loc = findAllLocationsOrFail(action, true);
141       if (loc == null) return;
142       HRegionLocation[] locs = loc.getRegionLocations();
143       if (locs.length == 1) {
144         LOG.warn("No replicas found for " + action.getAction());
145         return;
146       }
147       synchronized (replicaResultLock) {
148         // Don't run replica calls if the original has finished. We could do it e.g. if
149         // original has already failed before first replica call (unlikely given retries),
150         // but that would require additional synchronization w.r.t. returning to caller.
151         if (results[index] != null) return;
152         // We set the number of calls here. After that any path must call setResult/setError.
153         // True even for replicas that are not found - if we refuse to send we MUST set error.
154         results[index] = new ReplicaResultState(locs.length);
155       }
156       for (int i = 1; i < locs.length; ++i) {
157         Action<Row> replicaAction = new Action<Row>(action, i);
158         if (locs[i] != null) {
159           asyncProcess.addAction(locs[i].getServerName(), locs[i].getRegionInfo().getRegionName(),
160               replicaAction, actionsByServer, nonceGroup);
161         } else {
162           unknownReplicaActions.add(replicaAction);
163         }
164       }
165     }
166
167     private void addReplicaActionsAgain(
168         Action<Row> action, Map<ServerName, MultiAction<Row>> actionsByServer) {
169       if (action.getReplicaId() == RegionReplicaUtil.DEFAULT_REPLICA_ID) {
170         throw new AssertionError("Cannot have default replica here");
171       }
172       HRegionLocation loc = getReplicaLocationOrFail(action);
173       if (loc == null) return;
174       asyncProcess.addAction(loc.getServerName(), loc.getRegionInfo().getRegionName(),
175           action, actionsByServer, nonceGroup);
176     }
177   }

179   /**
180    * Runnable (that can be submitted to thread pool) that submits MultiAction to a
181    * single server. The server call is synchronous, therefore we do it on a thread pool.
182    */
183   private final class SingleServerRequestRunnable implements Runnable {
184     private final MultiAction<Row> multiAction;
185     private final int numAttempt;
186     private final ServerName server;
187     private final Set<CancellableRegionServerCallable> callsInProgress;
188     private Long heapSize = null;
189     private SingleServerRequestRunnable(
190         MultiAction<Row> multiAction, int numAttempt, ServerName server,
191         Set<CancellableRegionServerCallable> callsInProgress) {
192       this.multiAction = multiAction;
193       this.numAttempt = numAttempt;
194       this.server = server;
195       this.callsInProgress = callsInProgress;
196     }
197
198     @VisibleForTesting
199     long heapSize() {
200       if (heapSize != null) {
201         return heapSize;
202       }
203       heapSize = 0L;
204       for (Map.Entry<byte[], List<Action<Row>>> e: this.multiAction.actions.entrySet()) {
205         List<Action<Row>> actions = e.getValue();
206         for (Action<Row> action: actions) {
207           Row row = action.getAction();
208           if (row instanceof Mutation) {
209             heapSize += ((Mutation) row).heapSize();
210           }
211         }
212       }
213       return heapSize;
214     }
215
216     @Override
217     public void run() {
218       AbstractResponse res = null;
219       CancellableRegionServerCallable callable = currentCallable;
220       try {
221         // setup the callable based on the actions, if we don't have one already from the request
222         if (callable == null) {
223           callable = createCallable(server, tableName, multiAction);
224         }
225         RpcRetryingCaller<AbstractResponse> caller = asyncProcess.createCaller(callable,rpcTimeout);
226         try {
227           if (callsInProgress != null) {
228             callsInProgress.add(callable);
229           }
230           res = caller.callWithoutRetries(callable, operationTimeout);
231           if (res == null) {
232             // Cancelled
233             return;
234           }
235         } catch (IOException e) {
236           // The service itself failed . It may be an error coming from the communication
237           //   layer, but, as well, a functional error raised by the server.
238           receiveGlobalFailure(multiAction, server, numAttempt, e);
239           return;
240         } catch (Throwable t) {
241           // This should not happen. Let's log & retry anyway.
242           LOG.error("#" + asyncProcess.id + ", Caught throwable while calling. This is unexpected." +
243               " Retrying. Server is " + server + ", tableName=" + tableName, t);
244           receiveGlobalFailure(multiAction, server, numAttempt, t);
245           return;
246         }
247         if (res.type() == AbstractResponse.ResponseType.MULTI) {
248           // Normal case: we received an answer from the server, and it's not an exception.
249           receiveMultiAction(multiAction, server, (MultiResponse) res, numAttempt);
250         } else {
251           if (results != null) {
252             SingleResponse singleResponse = (SingleResponse) res;
253             results[0] = singleResponse.getEntry();
254           }
255           decActionCounter(1);
256         }
257       } catch (Throwable t) {
258         // Something really bad happened. We are on the send thread that will now die.
259         LOG.error("Internal AsyncProcess #" + asyncProcess.id + " error for "
260             + tableName + " processing for " + server, t);
261         throw new RuntimeException(t);
262       } finally {
263         asyncProcess.decTaskCounters(multiAction.getRegions(), server);
264         if (callsInProgress != null && callable != null && res != null) {
265           callsInProgress.remove(callable);
266         }
267       }
268     }
269   }

320   /** Sync point for calls to multiple replicas for the same user request (Get).
321    * Created and put in the results array (we assume replica calls require results) when

323    * POJO, all fields are public. To modify them, the object itself is locked. */

343   public AsyncRequestFutureImpl(TableName tableName, List<Action<Row>> actions, long nonceGroup,
344       ExecutorService pool, boolean needResults, Object[] results, Batch.Callback<CResult> callback,
345       CancellableRegionServerCallable callable, int operationTimeout, int rpcTimeout,
346       AsyncProcess asyncProcess) {
347     this.pool = pool;
348     this.callback = callback;
349     this.nonceGroup = nonceGroup;
350     this.tableName = tableName;
351     this.actionsInProgress.set(actions.size());
352     if (results != null) {
353       assert needResults;
354       if (results.length != actions.size()) {
355         throw new AssertionError("results.length");
356       }
357       this.results = results;
358       for (int i = 0; i != this.results.length; ++i) {
359         results[i] = null;
360       }
361     } else {
362       this.results = needResults ? new Object[actions.size()] : null;
363     }
364     List<Integer> replicaGetIndices = null;
365     boolean hasAnyReplicaGets = false;
366     if (needResults) {
367       // Check to see if any requests might require replica calls.
368       // We expect that many requests will consist of all or no multi-replica gets; in such
369       // cases we would just use a boolean (hasAnyReplicaGets). If there's a mix, we will
370       // store the list of action indexes for which replica gets are possible, and set
371       // hasAnyReplicaGets to true.
372       boolean hasAnyNonReplicaReqs = false;
373       int posInList = 0;
374       for (Action<Row> action : actions) {
375         boolean isReplicaGet = AsyncProcess.isReplicaGet(action.getAction());
376         if (isReplicaGet) {
377           hasAnyReplicaGets = true;
378           if (hasAnyNonReplicaReqs) { // Mixed case
379             if (replicaGetIndices == null) {
380               replicaGetIndices = new ArrayList<Integer>(actions.size() - 1);
381             }
382             replicaGetIndices.add(posInList);
383           }
384         } else if (!hasAnyNonReplicaReqs) {
385           // The first non-multi-replica request in the action list.
386           hasAnyNonReplicaReqs = true;
387           if (posInList > 0) {
388             // Add all the previous requests to the index lists. We know they are all
389             // replica-gets because this is the first non-multi-replica request in the list.
390             replicaGetIndices = new ArrayList<Integer>(actions.size() - 1);
391             for (int i = 0; i < posInList; ++i) {
392               replicaGetIndices.add(i);
393             }
394           }
395         }
396         ++posInList;
397       }
398     }
399     this.hasAnyReplicaGets = hasAnyReplicaGets;
400     if (replicaGetIndices != null) {
401       this.replicaGetIndices = new int[replicaGetIndices.size()];
402       int i = 0;
403       for (Integer el : replicaGetIndices) {
404         this.replicaGetIndices[i++] = el;
405       }
406     } else {
407       this.replicaGetIndices = null;
408     }
409     this.callsInProgress = !hasAnyReplicaGets ? null :
410         Collections.newSetFromMap(
411             new ConcurrentHashMap<CancellableRegionServerCallable, Boolean>());
412     this.asyncProcess = asyncProcess;
413     this.errorsByServer = createServerErrorTracker();
414     this.errors = (asyncProcess.globalErrors != null)
415         ? asyncProcess.globalErrors : new BatchErrors();
416     this.operationTimeout = operationTimeout;
417     this.rpcTimeout = rpcTimeout;
418     this.currentCallable = callable;
419     if (callable == null) {
420       tracker = new RetryingTimeTracker().start();
421     }
422   }

444   /**
445    * Group a list of actions per region servers, and send them.
446    *
447    * @param currentActions - the list of row to submit
448    * @param numAttempt - the current numAttempt (first attempt is 1)
449    */
450   void groupAndSendMultiAction(List<Action<Row>> currentActions, int numAttempt) {
451     Map<ServerName, MultiAction<Row>> actionsByServer =
452         new HashMap<ServerName, MultiAction<Row>>();
453
454     boolean isReplica = false;
455     List<Action<Row>> unknownReplicaActions = null;
456     for (Action<Row> action : currentActions) {
457       RegionLocations locs = findAllLocationsOrFail(action, true);
458       if (locs == null) continue;
459       boolean isReplicaAction = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
460       if (isReplica && !isReplicaAction) {
461         // This is the property of the current implementation, not a requirement.
462         throw new AssertionError("Replica and non-replica actions in the same retry");
463       }
464       isReplica = isReplicaAction;
465       HRegionLocation loc = locs.getRegionLocation(action.getReplicaId());
466       if (loc == null || loc.getServerName() == null) {
467         if (isReplica) {
468           if (unknownReplicaActions == null) {
469             unknownReplicaActions = new ArrayList<Action<Row>>();
470           }
471           unknownReplicaActions.add(action);
472         } else {
473           // TODO: relies on primary location always being fetched
474           manageLocationError(action, null);
475         }
476       } else {
477         byte[] regionName = loc.getRegionInfo().getRegionName();
478         AsyncProcess.addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
479       }
480     }
481     boolean doStartReplica = (numAttempt == 1 && !isReplica && hasAnyReplicaGets);
482     boolean hasUnknown = unknownReplicaActions != null && !unknownReplicaActions.isEmpty();
483
484     if (!actionsByServer.isEmpty()) {
485       // If this is a first attempt to group and send, no replicas, we need replica thread.
486       sendMultiAction(actionsByServer, numAttempt, (doStartReplica && !hasUnknown)
487           ? currentActions : null, numAttempt > 1 && !hasUnknown);
488     }
489
490     if (hasUnknown) {
491       actionsByServer = new HashMap<ServerName, MultiAction<Row>>();
492       for (Action<Row> action : unknownReplicaActions) {
493         HRegionLocation loc = getReplicaLocationOrFail(action);
494         if (loc == null) continue;
495         byte[] regionName = loc.getRegionInfo().getRegionName();
496         AsyncProcess.addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
497       }
498       if (!actionsByServer.isEmpty()) {
499         sendMultiAction(
500             actionsByServer, numAttempt, doStartReplica ? currentActions : null, true);
501       }
502     }
503   }

505   private HRegionLocation getReplicaLocationOrFail(Action<Row> action) {
506     // We are going to try get location once again. For each action, we'll do it once
507     // from cache, because the previous calls in the loop might populate it.
508     int replicaId = action.getReplicaId();
509     RegionLocations locs = findAllLocationsOrFail(action, true);
510     if (locs == null) return null; // manageError already called
511     HRegionLocation loc = locs.getRegionLocation(replicaId);
512     if (loc == null || loc.getServerName() == null) {
513       locs = findAllLocationsOrFail(action, false);
514       if (locs == null) return null; // manageError already called
515       loc = locs.getRegionLocation(replicaId);
516     }
517     if (loc == null || loc.getServerName() == null) {
518       manageLocationError(action, null);
519       return null;
520     }
521     return loc;
522   }
523
524   private void manageLocationError(Action<Row> action, Exception ex) {
525     String msg = "Cannot get replica " + action.getReplicaId()
526         + " location for " + action.getAction();
527     LOG.error(msg);
528     if (ex == null) {
529       ex = new IOException(msg);
530     }
531     manageError(action.getOriginalIndex(), action.getAction(),
532         Retry.NO_LOCATION_PROBLEM, ex, null);
533   }
534
535   private RegionLocations findAllLocationsOrFail(Action<Row> action, boolean useCache) {
536     if (action.getAction() == null) throw new IllegalArgumentException("#" + asyncProcess.id +
537         ", row cannot be null");
538     RegionLocations loc = null;
539     try {
540       loc = asyncProcess.connection.locateRegion(
541           tableName, action.getAction().getRow(), useCache, true, action.getReplicaId());
542     } catch (IOException ex) {
543       manageLocationError(action, ex);
544     }
545     return loc;
546   }

548   /**
549    * Send a multi action structure to the servers, after a delay depending on the attempt
550    * number. Asynchronous.
551    *
552    * @param actionsByServer the actions structured by regions
553    * @param numAttempt the attempt number.
554    * @param actionsForReplicaThread original actions for replica thread; null on non-first call.
555    */
556   void sendMultiAction(Map<ServerName, MultiAction<Row>> actionsByServer,
557                                int numAttempt, List<Action<Row>> actionsForReplicaThread, boolean reuseThread) {
558     // Run the last item on the same thread if we are already on a send thread.
559     // We hope most of the time it will be the only item, so we can cut down on threads.
560     int actionsRemaining = actionsByServer.size();
561     // This iteration is by server (the HRegionLocation comparator is by server portion only).
562     for (Map.Entry<ServerName, MultiAction<Row>> e : actionsByServer.entrySet()) {
563       ServerName server = e.getKey();
564       MultiAction<Row> multiAction = e.getValue();
565       Collection<? extends Runnable> runnables = getNewMultiActionRunnable(server, multiAction,
566           numAttempt);
567       // make sure we correctly count the number of runnables before we try to reuse the send
568       // thread, in case we had to split the request into different runnables because of backoff
569       if (runnables.size() > actionsRemaining) {
570         actionsRemaining = runnables.size();
571       }
572
573       // run all the runnables
574       for (Runnable runnable : runnables) {
575         if ((--actionsRemaining == 0) && reuseThread) {
576           runnable.run();
577         } else {
578           try {
579             pool.submit(runnable);
580           } catch (Throwable t) {
581             if (t instanceof RejectedExecutionException) {
582               // This should never happen. But as the pool is provided by the end user,
583               // let's secure this a little.
584               LOG.warn("#" + asyncProcess.id + ", the task was rejected by the pool. This is unexpected." +
585                   " Server is " + server.getServerName(), t);
586             } else {
587               // see #HBASE-14359 for more details
588               LOG.warn("Caught unexpected exception/error: ", t);
589             }
590             asyncProcess.decTaskCounters(multiAction.getRegions(), server);
591             // We're likely to fail again, but this will increment the attempt counter,
592             // so it will finish.
593             receiveGlobalFailure(multiAction, server, numAttempt, t);
594           }
595         }
596       }
597     }
598
599     if (actionsForReplicaThread != null) {
600       startWaitingForReplicaCalls(actionsForReplicaThread);
601     }
602   }

604   private Collection<? extends Runnable> getNewMultiActionRunnable(ServerName server,
605                                                                    MultiAction<Row> multiAction,
606                                                                    int numAttempt) {
607     // no stats to manage, just do the standard action
608     if (asyncProcess.connection.getStatisticsTracker() == null) {
609       if (asyncProcess.connection.getConnectionMetrics() != null) {
610         asyncProcess.connection.getConnectionMetrics().incrNormalRunners();
611       }
612       asyncProcess.incTaskCounters(multiAction.getRegions(), server);
613       SingleServerRequestRunnable runnable = addSingleServerRequestHeapSize(server,
614           new SingleServerRequestRunnable(multiAction, numAttempt, server, callsInProgress));
615       return Collections.singletonList(Trace.wrap("AsyncProcess.sendMultiAction", runnable));
616     }
617
618     // group the actions by the amount of delay
619     Map<Long, DelayingRunner> actions = new HashMap<Long, DelayingRunner>(multiAction
620         .size());
621
622     // split up the actions
623     for (Map.Entry<byte[], List<Action<Row>>> e : multiAction.actions.entrySet()) {
624       Long backoff = getBackoff(server, e.getKey());
625       DelayingRunner runner = actions.get(backoff);
626       if (runner == null) {
627         actions.put(backoff, new DelayingRunner(backoff, e));
628       } else {
629         runner.add(e);
630       }
631     }
632
633     List<Runnable> toReturn = new ArrayList<Runnable>(actions.size());
634     for (DelayingRunner runner : actions.values()) {
635       asyncProcess.incTaskCounters(runner.getActions().getRegions(), server);
636       String traceText = "AsyncProcess.sendMultiAction";
637       Runnable runnable = addSingleServerRequestHeapSize(server,
638           new SingleServerRequestRunnable(runner.getActions(), numAttempt, server, callsInProgress));
639       // use a delay runner only if we need to sleep for some time
640       if (runner.getSleepTime() > 0) {
641         runner.setRunner(runnable);
642         traceText = "AsyncProcess.clientBackoff.sendMultiAction";
643         runnable = runner;
644         if (asyncProcess.connection.getConnectionMetrics() != null) {
645           asyncProcess.connection.getConnectionMetrics().incrDelayRunners();
646           asyncProcess.connection.getConnectionMetrics().updateDelayInterval(runner.getSleepTime());
647         }
648       } else {
649         if (asyncProcess.connection.getConnectionMetrics() != null) {
650           asyncProcess.connection.getConnectionMetrics().incrNormalRunners();
651         }
652       }
653       runnable = Trace.wrap(traceText, runnable);
654       toReturn.add(runnable);
655
656     }
657     return toReturn;
658   }

694   /**
695    * Check that we can retry acts accordingly: logs, set the error status.
696    *
697    * @param originalIndex the position in the list sent
698    * @param row           the row
699    * @param canRetry      if false, we won't retry whatever the settings.
700    * @param throwable     the throwable, if any (can be null)
701    * @param server        the location, if any (can be null)
702    * @return true if the action can be retried, false otherwise.
703    */
704   Retry manageError(int originalIndex, Row row, Retry canRetry,
705                                         Throwable throwable, ServerName server) {
706     if (canRetry == Retry.YES
707         && throwable != null && throwable instanceof DoNotRetryIOException) {
708       canRetry = Retry.NO_NOT_RETRIABLE;
709     }
710
711     if (canRetry != Retry.YES) {
712       // Batch.Callback<Res> was not called on failure in 0.94. We keep this.
713       setError(originalIndex, row, throwable, server);
714     } else if (isActionComplete(originalIndex, row)) {
715       canRetry = Retry.NO_OTHER_SUCCEEDED;
716     }
717     return canRetry;
718   }

720   /**
721    * Resubmit all the actions from this multiaction after a failure.
722    *
723    * @param rsActions  the actions still to do from the initial list
724    * @param server   the destination
725    * @param numAttempt the number of attempts so far
726    * @param t the throwable (if any) that caused the resubmit
727    */
728   private void receiveGlobalFailure(
729       MultiAction<Row> rsActions, ServerName server, int numAttempt, Throwable t) {
730     errorsByServer.reportServerError(server);
731     Retry canRetry = errorsByServer.canTryMore(numAttempt)
732         ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED;
733
734     if (tableName == null && ClientExceptionsUtil.isMetaClearingException(t)) {
735       // tableName is null when we made a cross-table RPC call.
736       asyncProcess.connection.clearCaches(server);
737     }
738     int failed = 0, stopped = 0;
739     List<Action<Row>> toReplay = new ArrayList<Action<Row>>();
740     for (Map.Entry<byte[], List<Action<Row>>> e : rsActions.actions.entrySet()) {
741       byte[] regionName = e.getKey();
742       byte[] row = e.getValue().iterator().next().getAction().getRow();
743       // Do not use the exception for updating cache because it might be coming from
744       // any of the regions in the MultiAction.
745       try {
746         if (tableName != null) {
747           asyncProcess.connection.updateCachedLocations(tableName, regionName, row,
748               ClientExceptionsUtil.isMetaClearingException(t) ? null : t, server);
749         }
750       } catch (Throwable ex) {
751         // That should never happen, but if it did, we want to make sure
752         // we still process errors
753         LOG.error("Couldn't update cached region locations: " + ex);
754       }
755       for (Action<Row> action : e.getValue()) {
756         Retry retry = manageError(
757             action.getOriginalIndex(), action.getAction(), canRetry, t, server);
758         if (retry == Retry.YES) {
759           toReplay.add(action);
760         } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
761           ++stopped;
762         } else {
763           ++failed;
764         }
765       }
766     }
767
768     if (toReplay.isEmpty()) {
769       logNoResubmit(server, numAttempt, rsActions.size(), t, failed, stopped);
770     } else {
771       resubmit(server, toReplay, numAttempt, rsActions.size(), t);
772     }
773   }

775   /**
776    * Log as much info as possible, and, if there is something to replay,
777    * submit it again after a back off sleep.
778    */
779   private void resubmit(ServerName oldServer, List<Action<Row>> toReplay,
780                         int numAttempt, int failureCount, Throwable throwable) {
781     // We have something to replay. We're going to sleep a little before.
782
783     // We have two contradicting needs here:
784     //  1) We want to get the new location after having slept, as it may change.
785     //  2) We want to take into account the location when calculating the sleep time.
786     //  3) If all this is just because the response needed to be chunked try again FAST.
787     // It should be possible to have some heuristics to take the right decision. Short term,
788     //  we go for one.
789     boolean retryImmediately = throwable instanceof RetryImmediatelyException;
790     int nextAttemptNumber = retryImmediately ? numAttempt : numAttempt + 1;
791     long backOffTime = retryImmediately ? 0 :
792         errorsByServer.calculateBackoffTime(oldServer, asyncProcess.pause);
793     if (numAttempt > asyncProcess.startLogErrorsCnt) {
794       // We use this value to have some logs when we have multiple failures, but not too many
795       //  logs, as errors are to be expected when a region moves, splits and so on
796       LOG.info(createLog(numAttempt, failureCount, toReplay.size(),
797           oldServer, throwable, backOffTime, true, null, -1, -1));
798     }
799
800     try {
801       if (backOffTime > 0) {
802         Thread.sleep(backOffTime);
803       }
804     } catch (InterruptedException e) {
805       LOG.warn("#" + asyncProcess.id + ", not sent: " + toReplay.size() + " operations, " + oldServer, e);
806       Thread.currentThread().interrupt();
807       return;
808     }
809
810     groupAndSendMultiAction(toReplay, nextAttemptNumber);
811   }

813   private void logNoResubmit(ServerName oldServer, int numAttempt,
814                              int failureCount, Throwable throwable, int failed, int stopped) {
815     if (failureCount != 0 || numAttempt > asyncProcess.startLogErrorsCnt + 1) {
816       String timeStr = new Date(errorsByServer.getStartTrackingTime()).toString();
817       String logMessage = createLog(numAttempt, failureCount, 0, oldServer,
818           throwable, -1, false, timeStr, failed, stopped);
819       if (failed != 0) {
820         // Only log final failures as warning
821         LOG.warn(logMessage);
822       } else {
823         LOG.info(logMessage);
824       }
825     }
826   }

828   /**
829    * Called when we receive the result of a server query.
830    *
831    * @param multiAction    - the multiAction we sent
832    * @param server       - the location. It's used as a server name.
833    * @param responses      - the response, if any
834    * @param numAttempt     - the attempt
835    */
836   private void receiveMultiAction(MultiAction<Row> multiAction,
837                                   ServerName server, MultiResponse responses, int numAttempt) {
838     assert responses != null;
839
840     // Success or partial success
841     // Analyze detailed results. We can still have individual failures to be redo.
842     // two specific throwables are managed:
843     //  - DoNotRetryIOException: we continue to retry for other actions
844     //  - RegionMovedException: we update the cache with the new region location
845
846     List<Action<Row>> toReplay = new ArrayList<Action<Row>>();
847     Throwable throwable = null;
848     int failureCount = 0;
849     boolean canRetry = true;
850
851     Map<byte[], MultiResponse.RegionResult> results = responses.getResults();
852     updateStats(server, results);
853
854     int failed = 0, stopped = 0;
855     // Go by original action.
856     for (Map.Entry<byte[], List<Action<Row>>> regionEntry : multiAction.actions.entrySet()) {
857       byte[] regionName = regionEntry.getKey();
858       Map<Integer, Object> regionResults = results.get(regionName) == null
859           ?  null : results.get(regionName).result;
860       if (regionResults == null) {
861         if (!responses.getExceptions().containsKey(regionName)) {
862           LOG.error("Server sent us neither results nor exceptions for "
863               + Bytes.toStringBinary(regionName));
864           responses.getExceptions().put(regionName, new RuntimeException("Invalid response"));
865         }
866         continue;
867       }
868       boolean regionFailureRegistered = false;
869       for (Action<Row> sentAction : regionEntry.getValue()) {
870         Object result = regionResults.get(sentAction.getOriginalIndex());
871         // Failure: retry if it's make sense else update the errors lists
872         if (result == null || result instanceof Throwable) {
873           Row row = sentAction.getAction();
874           throwable = ClientExceptionsUtil.findException(result);
875           // Register corresponding failures once per server/once per region.
876           if (!regionFailureRegistered) {
877             regionFailureRegistered = true;
878             try {
879               asyncProcess.connection.updateCachedLocations(
880                   tableName, regionName, row.getRow(), result, server);
881             } catch (Throwable ex) {
882               // That should never happen, but if it did, we want to make sure
883               // we still process errors
884               LOG.error("Couldn't update cached region locations: " + ex);
885             }
886           }
887           if (failureCount == 0) {
888             errorsByServer.reportServerError(server);
889             // We determine canRetry only once for all calls, after reporting server failure.
890             canRetry = errorsByServer.canTryMore(numAttempt);
891           }
892           ++failureCount;
893           Retry retry = manageError(sentAction.getOriginalIndex(), row,
894               canRetry ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED, (Throwable) result, server);
895           if (retry == Retry.YES) {
896             toReplay.add(sentAction);
897           } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
898             ++stopped;
899           } else {
900             ++failed;
901           }
902         } else {
903           if (callback != null) {
904             try {
905               //noinspection unchecked
906               // TODO: would callback expect a replica region name if it gets one?
907               this.callback.update(regionName, sentAction.getAction().getRow(), (CResult) result);
908             } catch (Throwable t) {
909               LOG.error("User callback threw an exception for "
910                   + Bytes.toStringBinary(regionName) + ", ignoring", t);
911             }
912           }
913           setResult(sentAction, result);
914         }
915       }
916     }
917
918     // The failures global to a region. We will use for multiAction we sent previously to find the
919     //   actions to replay.
920     for (Map.Entry<byte[], Throwable> throwableEntry : responses.getExceptions().entrySet()) {
921       throwable = throwableEntry.getValue();
922       byte[] region = throwableEntry.getKey();
923       List<Action<Row>> actions = multiAction.actions.get(region);
924       if (actions == null || actions.isEmpty()) {
925         throw new IllegalStateException("Wrong response for the region: " +
926             HRegionInfo.encodeRegionName(region));
927       }
928
929       if (failureCount == 0) {
930         errorsByServer.reportServerError(server);
931         canRetry = errorsByServer.canTryMore(numAttempt);
932       }
933       if (null == tableName && ClientExceptionsUtil.isMetaClearingException(throwable)) {
934         // For multi-actions, we don't have a table name, but we want to make sure to clear the
935         // cache in case there were location-related exceptions. We don't to clear the cache
936         // for every possible exception that comes through, however.
937         asyncProcess.connection.clearCaches(server);
938       } else {
939         try {
940           asyncProcess.connection.updateCachedLocations(
941               tableName, region, actions.get(0).getAction().getRow(), throwable, server);
942         } catch (Throwable ex) {
943           // That should never happen, but if it did, we want to make sure
944           // we still process errors
945           LOG.error("Couldn't update cached region locations: " + ex);
946         }
947       }
948       failureCount += actions.size();
949
950       for (Action<Row> action : actions) {
951         Row row = action.getAction();
952         Retry retry = manageError(action.getOriginalIndex(), row,
953             canRetry ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED, throwable, server);
954         if (retry == Retry.YES) {
955           toReplay.add(action);
956         } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
957           ++stopped;
958         } else {
959           ++failed;
960         }
961       }
962     }
963     if (toReplay.isEmpty()) {
964       logNoResubmit(server, numAttempt, failureCount, throwable, failed, stopped);
965     } else {
966       resubmit(server, toReplay, numAttempt, failureCount, throwable);
967     }
968   }

1022   /**
1023    * Sets the non-error result from a particular action.
1024    * @param action Action (request) that the server responded to.
1025    * @param result The result.
1026    */
1027   private void setResult(Action<Row> action, Object result) {
1028     if (result == null) {
1029       throw new RuntimeException("Result cannot be null");
1030     }
1031     ReplicaResultState state = null;
1032     boolean isStale = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
1033     int index = action.getOriginalIndex();
1034     if (results == null) {
1035       decActionCounter(index);
1036       return; // Simple case, no replica requests.
1037     }
1038     state = trySetResultSimple(index, action.getAction(), false, result, null, isStale);
1039     if (state == null) {
1040       return; // Simple case, no replica requests.
1041     }
1042     // At this point we know that state is set to replica tracking class.
1043     // It could be that someone else is also looking at it; however, we know there can
1044     // only be one state object, and only one thread can set callCount to 0. Other threads
1045     // will either see state with callCount 0 after locking it; or will not see state at all
1046     // we will replace it with the result.
1047     synchronized (state) {
1048       if (state.callCount == 0) {
1049         return; // someone already set the result
1050       }
1051       state.callCount = 0;
1052     }
1053     synchronized (replicaResultLock) {
1054       if (results[index] != state) {
1055         throw new AssertionError("We set the callCount but someone else replaced the result");
1056       }
1057       results[index] = result;
1058     }
1059
1060     decActionCounter(index);
1061   }

1063   /**
1064    * Sets the error from a particular action.
1065    * @param index Original action index.
1066    * @param row Original request.
1067    * @param throwable The resulting error.
1068    * @param server The source server.
1069    */
1070   private void setError(int index, Row row, Throwable throwable, ServerName server) {
1071     ReplicaResultState state = null;
1072     if (results == null) {
1073       // Note that we currently cannot have replica requests with null results. So it shouldn't
1074       // happen that multiple replica calls will call dAC for same actions with results == null.
1075       // Only one call per action should be present in this case.
1076       errors.add(throwable, row, server);
1077       decActionCounter(index);
1078       return; // Simple case, no replica requests.
1079     }
1080     state = trySetResultSimple(index, row, true, throwable, server, false);
1081     if (state == null) {
1082       return; // Simple case, no replica requests.
1083     }
1084     BatchErrors target = null; // Error will be added to final errors, or temp replica errors.
1085     boolean isActionDone = false;
1086     synchronized (state) {
1087       switch (state.callCount) {
1088         case 0: return; // someone already set the result
1089         case 1: { // All calls failed, we are the last error.
1090           target = errors;
1091           isActionDone = true;
1092           break;
1093         }
1094         default: {
1095           assert state.callCount > 1;
1096           if (state.replicaErrors == null) {
1097             state.replicaErrors = new BatchErrors();
1098           }
1099           target = state.replicaErrors;
1100           break;
1101         }
1102       }
1103       --state.callCount;
1104     }
1105     target.add(throwable, row, server);
1106     if (isActionDone) {
1107       if (state.replicaErrors != null) { // last call, no need to lock
1108         errors.merge(state.replicaErrors);
1109       }
1110       // See setResult for explanations.
1111       synchronized (replicaResultLock) {
1112         if (results[index] != state) {
1113           throw new AssertionError("We set the callCount but someone else replaced the result");
1114         }
1115         results[index] = throwable;
1116       }
1117       decActionCounter(index);
1118     }
1119   }

1121   /**
1122    * Checks if the action is complete; used on error to prevent needless retries.
1123    * Does not synchronize, assuming element index/field accesses are atomic.
1124    * This is an opportunistic optimization check, doesn't have to be strict.
1125    * @param index Original action index.
1126    * @param row Original request.
1127    */
1128   private boolean isActionComplete(int index, Row row) {
1129     if (!AsyncProcess.isReplicaGet(row)) return false;
1130     Object resObj = results[index];
1131     return (resObj != null) && (!(resObj instanceof ReplicaResultState)
1132         || ((ReplicaResultState)resObj).callCount == 0);
1133   }

1135   /**
1136    * Tries to set the result or error for a particular action as if there were no replica calls.
1137    * @return null if successful; replica state if there were in fact replica calls.
1138    */
1139   private ReplicaResultState trySetResultSimple(int index, Row row, boolean isError,
1140                                                 Object result, ServerName server, boolean isFromReplica) {
1141     Object resObj = null;
1142     if (!AsyncProcess.isReplicaGet(row)) {
1143       if (isFromReplica) {
1144         throw new AssertionError("Unexpected stale result for " + row);
1145       }
1146       results[index] = result;
1147     } else {
1148       synchronized (replicaResultLock) {
1149         resObj = results[index];
1150         if (resObj == null) {
1151           if (isFromReplica) {
1152             throw new AssertionError("Unexpected stale result for " + row);
1153           }
1154           results[index] = result;
1155         }
1156       }
1157     }
1158
1159     ReplicaResultState rrs =
1160         (resObj instanceof ReplicaResultState) ? (ReplicaResultState)resObj : null;
1161     if (rrs == null && isError) {
1162       // The resObj is not replica state (null or already set).
1163       errors.add((Throwable)result, row, server);
1164     }
1165
1166     if (resObj == null) {
1167       // resObj is null - no replica calls were made.
1168       decActionCounter(index);
1169       return null;
1170     }
1171     return rrs;
1172   }

1174   private void decActionCounter(int index) {
1175     long actionsRemaining = actionsInProgress.decrementAndGet();
1176     if (actionsRemaining < 0) {
1177       String error = buildDetailedErrorMsg("Incorrect actions in progress", index);
1178       throw new AssertionError(error);
1179     } else if (actionsRemaining == 0) {
1180       synchronized (actionsInProgress) {
1181         actionsInProgress.notifyAll();
1182       }
1183     }
1184   }

1186   private String buildDetailedErrorMsg(String string, int index) {
1187     StringBuilder error = new StringBuilder(128);
1188     error.append(string).append("; called for ").append(index).append(", actionsInProgress ")
1189         .append(actionsInProgress.get()).append("; replica gets: ");

1222   private boolean waitUntilDone(long cutoff) throws InterruptedException {
1223     boolean hasWait = cutoff != Long.MAX_VALUE;
1224     long lastLog = EnvironmentEdgeManager.currentTime();
1225     long currentInProgress;
1226     while (0 != (currentInProgress = actionsInProgress.get())) {
1227       long now = EnvironmentEdgeManager.currentTime();
1228       if (hasWait && (now * 1000L) > cutoff) {
1229         return false;
1230       }
1231       if (!hasWait) { // Only log if wait is infinite.
1232         if (now > lastLog + 10000) {
1233           lastLog = now;
1234           LOG.info("#" + asyncProcess.id + ", waiting for " + currentInProgress
1235               + "  actions to finish on table: " + tableName);
1236           if (currentInProgress <= asyncProcess.thresholdToLogUndoneTaskDetails) {
1237             asyncProcess.logDetailsOfUndoneTasks(currentInProgress);
1238           }
1239         }
1240       }
1241       synchronized (actionsInProgress) {
1242         if (actionsInProgress.get() == 0) break;
1243         if (!hasWait) {
1244           actionsInProgress.wait(10);
1245         } else {
1246           long waitMicroSecond = Math.min(100000L, (cutoff - now * 1000L));
1247           TimeUnit.MICROSECONDS.timedWait(actionsInProgress, waitMicroSecond);
1248         }
1249       }
1250     }
1251     return true;
1252   }
