73    @Override
74    public Set<String> getAllWALs() {
75      Set<String> allWals = new HashSet<String>();
76      ResultScanner allQueues = null;
77      try (Table replicationTable = getOrBlockOnReplicationTable()) {
78        allQueues = replicationTable.getScanner(new Scan());
79        for (Result queue : allQueues) {
80          for (String wal : readWALsFromResult(queue)) {
81            allWals.add(wal);
82          }
83        }
84      } catch (IOException e) {
85        String errMsg = "Failed getting all WAL's in Replication Table";
86        abortable.abort(errMsg, e);
87      } finally {
88        if (allQueues != null) {
89          allQueues.close();
90        }
91      }
92      return allWals;
93    }
