64  /**
65   * This class provides an implementation of the ReplicationQueues interface using an HBase table
66   * "Replication Table". The basic schema of this table will store each individual queue as a
67   * seperate row. The row key will be a unique identifier of the creating server's name and the
68   * queueId. Each queue must have the following two columns:
69   *  COL_OWNER: tracks which server is currently responsible for tracking the queue
70   *  COL_QUEUE_ID: tracks the queue's id as stored in ReplicationSource
71   * They will also have columns mapping [WAL filename : offset]
72   * One key difference from the ReplicationQueuesZkImpl is that when queues are reclaimed we
73   * simply return its HBase row key as its new "queueId"
74   */
75
76  @InterfaceAudience.Private
77  public class ReplicationQueuesHBaseImpl extends ReplicationStateZKBase
78      implements ReplicationQueues {

82    /** Name of the HBase Table used for tracking replication*/
83    public static final TableName REPLICATION_TABLE_NAME =
84      TableName.valueOf(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR, "replication");

86    // Column family and column names for the Replication Table
87    private static final byte[] CF = Bytes.toBytes("r");
88    private static final byte[] COL_OWNER = Bytes.toBytes("o");
89    private static final byte[] COL_OWNER_HISTORY = Bytes.toBytes("h");

91    // The value used to delimit the queueId and server name inside of a queue's row key. Currently a
92    // hyphen, because it is guaranteed that queueId (which is a cluster id) cannot contain hyphens.
93    // See HBASE-11394.
94    private static String ROW_KEY_DELIMITER = "-";
95
96    // Column Descriptor for the Replication Table
97    private static final HColumnDescriptor REPLICATION_COL_DESCRIPTOR =
98      new HColumnDescriptor(CF).setMaxVersions(1)
99        .setInMemory(true)
100       .setScope(HConstants.REPLICATION_SCOPE_LOCAL)
101         // TODO: Figure out which bloom filter to use
102       .setBloomFilterType(BloomType.NONE)
103       .setCacheDataInL1(true);
104
105   // Common byte values used in replication offset tracking
106   private static final byte[] INITIAL_OFFSET_BYTES = Bytes.toBytes(0L);
107   private static final byte[] EMPTY_STRING_BYTES = Bytes.toBytes("");

109   /*
110    * Make sure that HBase table operations for replication have a high number of retries. This is
111    * because the server is aborted if any HBase table operation fails. Each RPC will be attempted
112    * 3600 times before exiting. This provides each operation with 2 hours of retries
113    * before the server is aborted.
114    */
115   private static final int CLIENT_RETRIES = 3600;
116   private static final int RPC_TIMEOUT = 2000;
117   private static final int OPERATION_TIMEOUT = CLIENT_RETRIES * RPC_TIMEOUT;

136   @Override
137   public void init(String serverName) throws ReplicationException {
138     try {
139       this.serverName = serverName;
140       this.serverNameBytes = Bytes.toBytes(serverName);
141       // Modify the connection's config so that the Replication Table it returns has a much higher
142       // number of client retries
143       this.connection = ConnectionFactory.createConnection(modifiedConf);
144       this.admin = connection.getAdmin();
145       replicationTable = createAndGetReplicationTable();
146       replicationTable.setRpcTimeout(RPC_TIMEOUT);
147       replicationTable.setOperationTimeout(OPERATION_TIMEOUT);
148     } catch (IOException e) {
149       throw new ReplicationException(e);
150     }
151   }

153   @Override
154   public void removeQueue(String queueId) {
155
156     try {
157       byte[] rowKey = queueIdToRowKey(queueId);
158       Delete deleteQueue = new Delete(rowKey);
159       safeQueueUpdate(deleteQueue);
160     } catch (IOException | ReplicationException e) {
161       String errMsg = "Failed removing queue queueId=" + queueId;
162       abortable.abort(errMsg, e);
163     }
164   }

166   @Override
167   public void addLog(String queueId, String filename) throws ReplicationException {
168     try {
169       if (!checkQueueExists(queueId)) {
170         // Each queue will have an Owner, OwnerHistory, and a collection of [WAL:offset] key values
171         Put putNewQueue = new Put(Bytes.toBytes(buildQueueRowKey(queueId)));
172         putNewQueue.addColumn(CF, COL_OWNER, serverNameBytes);
173         putNewQueue.addColumn(CF, COL_OWNER_HISTORY, EMPTY_STRING_BYTES);
174         putNewQueue.addColumn(CF, Bytes.toBytes(filename), INITIAL_OFFSET_BYTES);
175         replicationTable.put(putNewQueue);
176       } else {
177         // Otherwise simply add the new log and offset as a new column
178         Put putNewLog = new Put(queueIdToRowKey(queueId));
179         putNewLog.addColumn(CF, Bytes.toBytes(filename), INITIAL_OFFSET_BYTES);
180         safeQueueUpdate(putNewLog);
181       }
182     } catch (IOException | ReplicationException e) {
183       String errMsg = "Failed adding log queueId=" + queueId + " filename=" + filename;
184       abortable.abort(errMsg, e);
185     }
186   }

201   @Override
202   public void setLogPosition(String queueId, String filename, long position) {
203     try {
204       byte[] rowKey = queueIdToRowKey(queueId);
205       // Check that the log exists. addLog() must have been called before setLogPosition().
206       Get checkLogExists = new Get(rowKey);
207       checkLogExists.addColumn(CF, Bytes.toBytes(filename));
208       if (!replicationTable.exists(checkLogExists)) {
209         String errMsg = "Could not set position of non-existent log from queueId=" + queueId +
210           ", filename=" + filename;
211         abortable.abort(errMsg, new ReplicationException(errMsg));
212         return;
213       }
214       // Update the log offset if it exists
215       Put walAndOffset = new Put(rowKey);
216       walAndOffset.addColumn(CF, Bytes.toBytes(filename), Bytes.toBytes(position));
217       safeQueueUpdate(walAndOffset);
218     } catch (IOException | ReplicationException e) {
219       String errMsg = "Failed writing log position queueId=" + queueId + "filename=" +
220         filename + " position=" + position;
221       abortable.abort(errMsg, e);
222     }
223   }

225   @Override
226   public long getLogPosition(String queueId, String filename) throws ReplicationException {
227     try {
228       byte[] rowKey = queueIdToRowKey(queueId);
229       Get getOffset = new Get(rowKey);
230       getOffset.addColumn(CF, Bytes.toBytes(filename));
231       Result result = getResultIfOwner(getOffset);
232       if (result == null || !result.containsColumn(CF, Bytes.toBytes(filename))) {
233         throw new ReplicationException("Could not read empty result while getting log position " +
234             "queueId=" + queueId + ", filename=" + filename);
235       }
236       return Bytes.toLong(result.getValue(CF, Bytes.toBytes(filename)));
237     } catch (IOException e) {
238       throw new ReplicationException("Could not get position in log for queueId=" + queueId +
239           ", filename=" + filename);
240     }
241   }

257   private List<String> getLogsInQueue(byte[] rowKey) {
258     String errMsg = "Could not get logs in queue queueId=" + Bytes.toString(rowKey);
259     try {
260       Get getQueue = new Get(rowKey);
261       Result queue = getResultIfOwner(getQueue);
262       // The returned queue could be null if we have lost ownership of it
263       if (queue == null) {
264         abortable.abort(errMsg, new ReplicationException(errMsg));
265         return null;
266       }
267       return readWALsFromResult(queue);
268     } catch (IOException e) {
269       abortable.abort(errMsg, e);
270       return null;
271     }
272   }

274   @Override
275   public List<String> getAllQueues() {
276     List<String> allQueues = new ArrayList<String>();
277     ResultScanner queueScanner = null;
278     try {
279       queueScanner = this.getQueuesBelongingToServer(serverName);
280       for (Result queue : queueScanner) {
281         String rowKey =  Bytes.toString(queue.getRow());
282         // If the queue does not have a Owner History, then we must be its original owner. So we
283         // want to return its queueId in raw form
284         if (Bytes.toString(queue.getValue(CF, COL_OWNER_HISTORY)).length() == 0) {
285           allQueues.add(getRawQueueIdFromRowKey(rowKey));
286         } else {
287           allQueues.add(rowKey);
288         }
289       }
290       return allQueues;
291     } catch (IOException e) {
292       String errMsg = "Failed getting list of all replication queues";
293       abortable.abort(errMsg, e);
294       return null;
295     } finally {
296       if (queueScanner != null) {
297         queueScanner.close();
298       }
299     }
300   }

302   @Override
303   public Map<String, Set<String>> claimQueues(String regionserver) {
304     Map<String, Set<String>> queues = new HashMap<>();
305     if (isThisOurRegionServer(regionserver)) {
306       return queues;
307     }
308     ResultScanner queuesToClaim = null;
309     try {
310       queuesToClaim = this.getQueuesBelongingToServer(regionserver);
311       for (Result queue : queuesToClaim) {
312         if (attemptToClaimQueue(queue, regionserver)) {
313           String rowKey = Bytes.toString(queue.getRow());
314           ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(rowKey);
315           if (peerExists(replicationQueueInfo.getPeerId())) {
316             Set<String> sortedLogs = new HashSet<String>();
317             List<String> logs = getLogsInQueue(queue.getRow());
318             for (String log : logs) {
319               sortedLogs.add(log);
320             }
321             queues.put(rowKey, sortedLogs);
322             LOG.info(serverName + " has claimed queue " + rowKey + " from " + regionserver);
323           } else {
324             // Delete orphaned queues
325             removeQueue(Bytes.toString(queue.getRow()));
326             LOG.info(serverName + " has deleted abandoned queue " + rowKey + " from " +
327                 regionserver);
328           }
329         }
330       }
331     } catch (IOException | KeeperException e) {
332       String errMsg = "Failed claiming queues for regionserver=" + regionserver;
333       abortable.abort(errMsg, e);
334       queues.clear();
335     } finally {
336       if (queuesToClaim != null) {
337         queuesToClaim.close();
338       }
339     }
340     return queues;
341   }

343   @Override
344   public List<String> getListOfReplicators() {
345     // scan all of the queues and return a list of all unique OWNER values
346     Set<String> peerServers = new HashSet<String>();
347     ResultScanner allQueuesInCluster = null;
348     try {
349       Scan scan = new Scan();
350       scan.addColumn(CF, COL_OWNER);
351       allQueuesInCluster = replicationTable.getScanner(scan);
352       for (Result queue : allQueuesInCluster) {
353         peerServers.add(Bytes.toString(queue.getValue(CF, COL_OWNER)));
354       }
355     } catch (IOException e) {
356       String errMsg = "Failed getting list of replicators";
357       abortable.abort(errMsg, e);
358     } finally {
359       if (allQueuesInCluster != null) {
360         allQueuesInCluster.close();
361       }
362     }
363     return new ArrayList<String>(peerServers);
364   }

395   /**
396    * Gets the Replication Table. Builds and blocks until the table is available if the Replication
397    * Table does not exist.
398    *
399    * @return the Replication Table
400    * @throws IOException if the Replication Table takes too long to build
401    */
402   private Table createAndGetReplicationTable() throws IOException {
403     if (!replicationTableExists()) {
404       createReplicationTable();
405     }
406     int maxRetries = conf.getInt("replication.queues.createtable.retries.number", 100);
407     RetryCounterFactory counterFactory = new RetryCounterFactory(maxRetries, 100);
408     RetryCounter retryCounter = counterFactory.create();
409     while (!replicationTableExists()) {
410       try {
411         retryCounter.sleepUntilNextRetry();
412         if (!retryCounter.shouldRetry()) {
413           throw new IOException("Unable to acquire the Replication Table");
414         }
415       } catch (InterruptedException e) {
416         return null;
417       }
418     }
419     return connection.getTable(REPLICATION_TABLE_NAME);
420   }

495   /**
496    * Attempt to mutate a given queue in the Replication Table with a checkAndPut on the OWNER column
497    * of the queue. Abort the server if this checkAndPut fails: which means we have somehow lost
498    * ownership of the column or an IO Exception has occurred during the transaction.
499    *
500    * @param mutate Mutation to perform on a given queue
501    */
502   private void safeQueueUpdate(RowMutations mutate) throws ReplicationException, IOException{
503     boolean updateSuccess = replicationTable.checkAndMutate(mutate.getRow(), CF, COL_OWNER,
504       CompareFilter.CompareOp.EQUAL, serverNameBytes, mutate);
505     if (!updateSuccess) {
506       throw new ReplicationException("Failed to update Replication Table because we lost queue " +
507         " ownership");
508     }
509   }

511   /**
512    * Returns a queue's row key given either its raw or reclaimed queueId
513    *
514    * @param queueId queueId of the queue
515    * @return byte representation of the queue's row key
516    */
517   private byte[] queueIdToRowKey(String queueId) {
518     // Cluster id's are guaranteed to have no hyphens, so if the passed in queueId has no hyphen
519     // then this is not a reclaimed queue.
520     if (!queueId.contains(ROW_KEY_DELIMITER)) {
521       return Bytes.toBytes(buildQueueRowKey(queueId));
522       // If the queueId contained some hyphen it was reclaimed. In this case, the queueId is the
523       // queue's row key
524     } else {
525       return Bytes.toBytes(queueId);
526     }
527   }

529   /**
530    * Get the QueueIds belonging to the named server from the ReplicationTable
531    *
532    * @param server name of the server
533    * @return a ResultScanner over the QueueIds belonging to the server
534    * @throws IOException
535    */
536   private ResultScanner getQueuesBelongingToServer(String server) throws IOException {
537     Scan scan = new Scan();
538     SingleColumnValueFilter filterMyQueues = new SingleColumnValueFilter(CF, COL_OWNER,
539       CompareFilter.CompareOp.EQUAL, Bytes.toBytes(server));
540     scan.setFilter(filterMyQueues);
541     scan.addColumn(CF, COL_OWNER);
542     scan.addColumn(CF, COL_OWNER_HISTORY);
543     ResultScanner results = replicationTable.getScanner(scan);
544     return results;
545   }

559   /**
560    * Read all of the WAL's from a queue into a list
561    *
562    * @param queue HBase query result containing the queue
563    * @return a list of all the WAL filenames
564    */
565   private List<String> readWALsFromResult(Result queue) {
566     List<String> wals = new ArrayList<>();
567     Map<byte[], byte[]> familyMap = queue.getFamilyMap(CF);
568     for(byte[] cQualifier : familyMap.keySet()) {
569       // Ignore the meta data fields of the queue
570       if (Arrays.equals(cQualifier, COL_OWNER) || Arrays.equals(cQualifier, COL_OWNER_HISTORY)) {
571         continue;
572       }
573       wals.add(Bytes.toString(cQualifier));
574     }
575     return wals;
576   }

578   /**
579    * Attempt to claim the given queue with a checkAndPut on the OWNER column. We check that the
580    * recently killed server is still the OWNER before we claim it.
581    *
582    * @param queue The queue that we are trying to claim
583    * @param originalServer The server that originally owned the queue
584    * @return Whether we successfully claimed the queue
585    * @throws IOException
586    */
587   private boolean attemptToClaimQueue (Result queue, String originalServer) throws IOException{
588     Put putQueueNameAndHistory = new Put(queue.getRow());
589     putQueueNameAndHistory.addColumn(CF, COL_OWNER, Bytes.toBytes(serverName));
590     String newOwnerHistory = buildClaimedQueueHistory(Bytes.toString(queue.getValue(CF,
591       COL_OWNER_HISTORY)), originalServer);
592     putQueueNameAndHistory.addColumn(CF, COL_OWNER_HISTORY, Bytes.toBytes(newOwnerHistory));
593     RowMutations claimAndRenameQueue = new RowMutations(queue.getRow());
594     claimAndRenameQueue.add(putQueueNameAndHistory);
595     // Attempt to claim ownership for this queue by checking if the current OWNER is the original
596     // server. If it is not then another RS has already claimed it. If it is we set ourselves as the
597     // new owner and update the queue's history
598     boolean success = replicationTable.checkAndMutate(queue.getRow(), CF, COL_OWNER,
599       CompareFilter.CompareOp.EQUAL, Bytes.toBytes(originalServer), claimAndRenameQueue);
600     return success;
601   }

603   /**
604    * Creates a "|" delimited record of the queue's past region server owners.
605    *
606    * @param originalHistory the queue's original owner history
607    * @param oldServer the name of the server that used to own the queue
608    * @return the queue's new owner history
609    */
610   private String buildClaimedQueueHistory(String originalHistory, String oldServer) {
611     return originalHistory + "|" + oldServer;

614   /**
615    * Attempts to run a Get on some queue. Will only return a non-null result if we currently own
616    * the queue.
617    *
618    * @param get The get that we want to query
619    * @return The result of the get if this server is the owner of the queue. Else it returns null
620    * @throws IOException
621    */
622   private Result getResultIfOwner(Get get) throws IOException {
623     Scan scan = new Scan(get);
624     // Check if the Get currently contains all columns or only specific columns
625     if (scan.getFamilyMap().size() > 0) {
626       // Add the OWNER column if the scan is already only over specific columns
627       scan.addColumn(CF, COL_OWNER);
628     }
629     scan.setMaxResultSize(1);
