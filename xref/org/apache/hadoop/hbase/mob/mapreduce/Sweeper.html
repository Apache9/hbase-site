43  /**
44   * The sweep tool. It deletes the mob files that are not used and merges the small mob files to
45   * bigger ones. Each run of this sweep tool only handles one column family. The runs on
46   * the same column family are mutually exclusive. And the major compaction and sweep tool on the
47   * same column family are mutually exclusive too.
48   */
49  @InterfaceAudience.Public
50  @InterfaceStability.Evolving
51  public class Sweeper extends Configured implements Tool {
52  
53    /**
54     * Sweeps the mob files on one column family. It deletes the unused mob files and merges
55     * the small mob files into bigger ones.
56     * @param tableName The current table name in string format.
57     * @param familyName The column family name.
58     * @return 0 if success, 2 if job aborted with an exception, 3 if unable to start due to
59     *   other compaction,4 if mr job was unsuccessful
60     * @throws IOException
61     * @throws InterruptedException
62     * @throws ClassNotFoundException
63     * @throws KeeperException
64     * @throws ServiceException
65     */
66    int sweepFamily(String tableName, String familyName) throws IOException, InterruptedException,
67        ClassNotFoundException, KeeperException, ServiceException {
68      Configuration conf = getConf();
69      // make sure the target HBase exists.
70      HBaseAdmin.checkHBaseAvailable(conf);
71      Connection connection = ConnectionFactory.createConnection(getConf());
72      Admin admin = connection.getAdmin();
73      try {
74        FileSystem fs = FileSystem.get(conf);
75        TableName tn = TableName.valueOf(tableName);
76        HTableDescriptor htd = admin.getTableDescriptor(tn);
77        HColumnDescriptor family = htd.getFamily(Bytes.toBytes(familyName));
78        if (family == null || !family.isMobEnabled()) {
79            throw new IOException("Column family " + familyName + " is not a MOB column family");
80        }
81        SweepJob job = new SweepJob(conf, fs);
82        // Run the sweeping
83        return job.sweep(tn, family);
84      } catch (Exception e) {
85        System.err.println("Job aborted due to exception " + e);
86        return 2; // job failed
87      } finally {
88        try {
89          admin.close();
90        } catch (IOException e) {
91          System.out.println("Failed to close the HBaseAdmin: " + e.getMessage());
92        }
93        try {
94          connection.close();
95        } catch (IOException e) {
96          System.out.println("Failed to close the connection: " + e.getMessage());
97        }
98      }
99    }
100 
101   public static void main(String[] args) throws Exception {
102     Configuration conf = HBaseConfiguration.create();
103     int ret = ToolRunner.run(conf, new Sweeper(), args);
104     System.exit(ret);
105   }

107   private void printUsage() {
108     System.err.println("Usage:\n" + "--------------------------\n" + Sweeper.class.getName()
109         + " tableName familyName");
110     System.err.println(" tableName        The table name");
111     System.err.println(" familyName       The column family name");
112   }
113 
114   /**
115    * Main method for the tool.
116    * @return 0 if success, 1 for bad args. 2 if job aborted with an exception,
117    *   3 if unable to start due to other compaction, 4 if mr job was unsuccessful
118    */
119   public int run(String[] args) throws Exception {
120     if (args.length != 2) {
121       printUsage();
122       return 1;
123     }
124     String table = args[0];
125     String family = args[1];
126     return sweepFamily(table, family);
127   }
