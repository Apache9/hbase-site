46  /**
47   * The cleaner to delete the expired MOB files.
48   */
49  @InterfaceAudience.Private
50  public class ExpiredMobFileCleaner extends Configured implements Tool {
51  
52    private static final Log LOG = LogFactory.getLog(ExpiredMobFileCleaner.class);
53    /**
54     * Cleans the MOB files when they're expired and their min versions are 0.
55     * If the latest timestamp of Cells in a MOB file is older than the TTL in the column family,
56     * it's regarded as expired. This cleaner deletes them.
57     * At a time T0, the cells in a mob file M0 are expired. If a user starts a scan before T0, those
58     * mob cells are visible, this scan still runs after T0. At that time T1, this mob file M0
59     * is expired, meanwhile a cleaner starts, the M0 is archived and can be read in the archive
60     * directory.
61     * @param tableName The current table name.
62     * @param family The current family.
63     * @throws ServiceException
64     * @throws IOException
65     */
66    public void cleanExpiredMobFiles(String tableName, HColumnDescriptor family)
67        throws ServiceException, IOException {
68      Configuration conf = getConf();
69      TableName tn = TableName.valueOf(tableName);
70      FileSystem fs = FileSystem.get(conf);
71      LOG.info("Cleaning the expired MOB files of " + family.getNameAsString() + " in " + tableName);
72      // disable the block cache.
73      Configuration copyOfConf = new Configuration(conf);
74      copyOfConf.setFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY, 0f);
75      CacheConfig cacheConfig = new CacheConfig(copyOfConf);
76      MobUtils.cleanExpiredMobFiles(fs, conf, tn, family, cacheConfig,
77          EnvironmentEdgeManager.currentTime());
78    }
79  
80    public static void main(String[] args) throws Exception {
81      Configuration conf = HBaseConfiguration.create();
82      ToolRunner.run(conf, new ExpiredMobFileCleaner(), args);
83    }
84  
85    private void printUsage() {
86      System.err.println("Usage:\n" + "--------------------------\n"
87          + ExpiredMobFileCleaner.class.getName() + " tableName familyName");
88      System.err.println(" tableName        The table name");
89      System.err.println(" familyName       The column family name");
90    }
91  
92    public int run(String[] args) throws Exception {
93      if (args.length != 2) {
94        printUsage();
95        return 1;
96      }
97      String tableName = args[0];
98      String familyName = args[1];
99      TableName tn = TableName.valueOf(tableName);
100     HBaseAdmin.checkHBaseAvailable(getConf());
101     Connection connection = ConnectionFactory.createConnection(getConf());
102     Admin admin = connection.getAdmin();
103     try {
104       HTableDescriptor htd = admin.getTableDescriptor(tn);
105       HColumnDescriptor family = htd.getFamily(Bytes.toBytes(familyName));
106       if (family == null || !family.isMobEnabled()) {
107         throw new IOException("Column family " + familyName + " is not a MOB column family");
108       }
109       if (family.getMinVersions() > 0) {
110         throw new IOException(
111             "The minVersions of the column family is not 0, could not be handled by this cleaner");
112       }
113       cleanExpiredMobFiles(tableName, family);
114       return 0;
115     } finally {
116       try {
117         admin.close();
118       } catch (IOException e) {
119         LOG.error("Failed to close the HBaseAdmin.", e);
120       }
121       try {
122         connection.close();
123       } catch (IOException e) {
124         LOG.error("Failed to close the connection.", e);
125       }
126     }
127   }
