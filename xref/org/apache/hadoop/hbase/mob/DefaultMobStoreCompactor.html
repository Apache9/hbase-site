53  /**
54   * Compact passed set of files in the mob-enabled column family.
55   */
56  @InterfaceAudience.Private
57  public class DefaultMobStoreCompactor extends DefaultCompactor {

59    private static final Log LOG = LogFactory.getLog(DefaultMobStoreCompactor.class);
60    private long mobSizeThreshold;
61    private HMobStore mobStore;

63    private final InternalScannerFactory scannerFactory = new InternalScannerFactory() {
64  
65      @Override
66      public ScanType getScanType(CompactionRequest request) {
67        return request.isRetainDeleteMarkers() ? ScanType.COMPACT_RETAIN_DELETES
68            : ScanType.COMPACT_DROP_DELETES;
69      }
70  
71      @Override
72      public InternalScanner createScanner(List<StoreFileScanner> scanners,
73          ScanType scanType, FileDetails fd, long smallestReadPoint) throws IOException {
74        Scan scan = new Scan();
75        scan.setMaxVersions(store.getFamily().getMaxVersions());
76        if (scanType == ScanType.COMPACT_DROP_DELETES) {
77          // In major compaction, we need to write the delete markers to del files, so we have to
78          // retain the them in scanning.
79          scanType = ScanType.COMPACT_RETAIN_DELETES;
80          return new MobCompactionStoreScanner(store, store.getScanInfo(), scan, scanners,
81              scanType, smallestReadPoint, fd.earliestPutTs, true);
82        } else {
83          return new MobCompactionStoreScanner(store, store.getScanInfo(), scan, scanners,
84              scanType, smallestReadPoint, fd.earliestPutTs, false);
85        }
86      }
87    };

89    private final CellSinkFactory<Writer> writerFactory = new CellSinkFactory<Writer>() {
90  
91      @Override
92      public Writer createWriter(InternalScanner scanner,
93          org.apache.hadoop.hbase.regionserver.compactions.Compactor.FileDetails fd,
94          boolean shouldDropBehind) throws IOException {
95        // make this writer with tags always because of possible new cells with tags.
96        return store.createWriterInTmp(fd.maxKeyCount, compactionCompression, true, true, true,
97          shouldDropBehind);
98      }

101   public DefaultMobStoreCompactor(Configuration conf, Store store) {
102     super(conf, store);
103     // The mob cells reside in the mob-enabled column family which is held by HMobStore.
104     // During the compaction, the compactor reads the cells from the mob files and
105     // probably creates new mob files. All of these operations are included in HMobStore,
106     // so we need to cast the Store to HMobStore.
107     if (!(store instanceof HMobStore)) {
108       throw new IllegalArgumentException("The store " + store + " is not a HMobStore");
109     }
110     mobStore = (HMobStore) store;
111     mobSizeThreshold = store.getFamily().getMobThreshold();
112   }
113 
114   @Override
115   public List<Path> compact(CompactionRequest request, ThroughputController throughputController,
116       User user) throws IOException {
117     return compact(request, scannerFactory, writerFactory, throughputController, user);
118   }

122   /**
123    * Performs compaction on a column family with the mob flag enabled.
124    * This is for when the mob threshold size has changed or if the mob
125    * column family mode has been toggled via an alter table statement.
126    * Compacts the files by the following rules.
127    * 1. If the cell has a mob reference tag, the cell's value is the path of the mob file.
128    * <ol>
129    * <li>
130    * If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
131    * directly copy the (with mob tag) cell into the new store file.
132    * </li>
133    * <li>
134    * Otherwise, retrieve the mob cell from the mob file, and writes a copy of the cell into
135    * the new store file.
136    * </li>
137    * </ol>
138    * 2. If the cell doesn't have a reference tag.
139    * <ol>
140    * <li>
141    * If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
142    * write this cell to a mob file, and write the path of this mob file to the store file.
143    * </li>
144    * <li>
145    * Otherwise, directly write this cell into the store file.
146    * </li>
147    * </ol>
148    * In the mob compaction, the {@link MobCompactionStoreScanner} is used as a scanner
149    * which could output the normal cells and delete markers together when required.
150    * After the major compaction on the normal hfiles, we have a guarantee that we have purged all
151    * deleted or old version mob refs, and the delete markers are written to a del file with the
152    * suffix _del. Because of this, it is safe to use the del file in the mob compaction.
153    * The mob compaction doesn't take place in the normal hfiles, it occurs directly in the
154    * mob files. When the small mob files are merged into bigger ones, the del file is added into
155    * the scanner to filter the deleted cells.
156    * @param fd File details
157    * @param scanner Where to read from.
158    * @param writer Where to write to.
159    * @param smallestReadPoint Smallest read point.
160    * @param cleanSeqId When true, remove seqId(used to be mvcc) value which is <= smallestReadPoint
161    * @param throughputController The compaction throughput controller.
162    * @param major Is a major compaction.
163    * @return Whether compaction ended; false if it was interrupted for any reason.
164    */
165   @Override
166   protected boolean performCompaction(FileDetails fd, InternalScanner scanner, CellSink writer,
167       long smallestReadPoint, boolean cleanSeqId,
168       ThroughputController throughputController,  boolean major) throws IOException {
169     if (!(scanner instanceof MobCompactionStoreScanner)) {
170       throw new IllegalArgumentException(
171         "The scanner should be an instance of MobCompactionStoreScanner");
172     }
173     MobCompactionStoreScanner compactionScanner = (MobCompactionStoreScanner) scanner;
174     int bytesWritten = 0;
175     // Since scanner.next() can return 'false' but still be delivering data,
176     // we have to use a do/while loop.
177     List<Cell> cells = new ArrayList<Cell>();
178     // Limit to "hbase.hstore.compaction.kv.max" (default 10) to avoid OOME
179     int closeCheckInterval = HStore.getCloseCheckInterval();
180     boolean hasMore;
181     Path path = MobUtils.getMobFamilyPath(conf, store.getTableName(), store.getColumnFamilyName());
182     byte[] fileName = null;
183     Writer mobFileWriter = null, delFileWriter = null;
184     long mobCells = 0, deleteMarkersCount = 0;
185     Tag tableNameTag = new ArrayBackedTag(TagType.MOB_TABLE_NAME_TAG_TYPE,
186         store.getTableName().getName());
187     long cellsCountCompactedToMob = 0, cellsCountCompactedFromMob = 0;
188     long cellsSizeCompactedToMob = 0, cellsSizeCompactedFromMob = 0;
189     try {
190       try {
191         // If the mob file writer could not be created, directly write the cell to the store file.
192         mobFileWriter = mobStore.createWriterInTmp(new Date(fd.latestPutTs), fd.maxKeyCount,
193             store.getFamily().getCompression(), store.getRegionInfo().getStartKey());
194         fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
195       } catch (IOException e) {
196         LOG.error("Failed to create mob writer, "
197                + "we will continue the compaction by writing MOB cells directly in store files", e);
198       }
199       delFileWriter = mobStore.createDelFileWriterInTmp(new Date(fd.latestPutTs), fd.maxKeyCount,
200           store.getFamily().getCompression(), store.getRegionInfo().getStartKey());
201       ScannerContext scannerContext =
202               ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
203       do {
204         hasMore = compactionScanner.next(cells, scannerContext);
205         for (Cell c : cells) {
206           if (compactionScanner.isOutputDeleteMarkers() && CellUtil.isDelete(c)) {
207             delFileWriter.append(c);
208             deleteMarkersCount++;
209           } else if (mobFileWriter == null || c.getTypeByte() != KeyValue.Type.Put.getCode()) {
210             // If the mob file writer is null or the kv type is not put, directly write the cell
211             // to the store file.
212             writer.append(c);
213           } else if (MobUtils.isMobReferenceCell(c)) {
214             if (MobUtils.hasValidMobRefCellValue(c)) {
215               int size = MobUtils.getMobValueLength(c);
216               if (size > mobSizeThreshold) {
217                 // If the value size is larger than the threshold, it's regarded as a mob. Since
218                 // its value is already in the mob file, directly write this cell to the store file
219                 writer.append(c);
220               } else {
221                 // If the value is not larger than the threshold, it's not regarded a mob. Retrieve
222                 // the mob cell from the mob file, and write it back to the store file.
223                 Cell mobCell = mobStore.resolve(c, false);
224                 if (mobCell.getValueLength() != 0) {
225                   // put the mob data back to the store file
226                   CellUtil.setSequenceId(mobCell, c.getSequenceId());
227                   writer.append(mobCell);
228                   cellsCountCompactedFromMob++;
229                   cellsSizeCompactedFromMob += mobCell.getValueLength();
230                 } else {
231                   // If the value of a file is empty, there might be issues when retrieving,
232                   // directly write the cell to the store file, and leave it to be handled by the
233                   // next compaction.
234                   writer.append(c);
235                 }
236               }
237             } else {
238               LOG.warn("The value format of the KeyValue " + c
239                   + " is wrong, its length is less than " + Bytes.SIZEOF_INT);
240               writer.append(c);
241             }
242           } else if (c.getValueLength() <= mobSizeThreshold) {
243             //If value size of a cell is not larger than the threshold, directly write to store file
244             writer.append(c);
245           } else {
246             // If the value size of a cell is larger than the threshold, it's regarded as a mob,
247             // write this cell to a mob file, and write the path to the store file.
248             mobCells++;
249             // append the original keyValue in the mob file.
250             mobFileWriter.append(c);
251             KeyValue reference = MobUtils.createMobRefKeyValue(c, fileName, tableNameTag);
252             // write the cell whose value is the path of a mob file to the store file.
253             writer.append(reference);
254             cellsCountCompactedToMob++;
255             cellsSizeCompactedToMob += c.getValueLength();
256           }
257           ++progress.currentCompactedKVs;
258           // check periodically to see if a system stop is requested
259           if (closeCheckInterval > 0) {
260             bytesWritten += KeyValueUtil.length(c);
261             if (bytesWritten > closeCheckInterval) {
262               bytesWritten = 0;
263               if (!store.areWritesEnabled()) {
264                 progress.cancel();
265                 return false;
266               }
267             }
268           }
269         }
270         cells.clear();
271       } while (hasMore);
272     } finally {
273       if (mobFileWriter != null) {
274         mobFileWriter.appendMetadata(fd.maxSeqId, major, mobCells);
275         mobFileWriter.close();
276       }
277       if (delFileWriter != null) {
278         delFileWriter.appendMetadata(fd.maxSeqId, major, deleteMarkersCount);
279         delFileWriter.close();
280       }
281     }
282     if (mobFileWriter != null) {
283       if (mobCells > 0) {
284         // If the mob file is not empty, commit it.
285         mobStore.commitFile(mobFileWriter.getPath(), path);
286       } else {
287         try {
288           // If the mob file is empty, delete it instead of committing.
289           store.getFileSystem().delete(mobFileWriter.getPath(), true);
290         } catch (IOException e) {
291           LOG.error("Failed to delete the temp mob file", e);
292         }
293       }
294     }
295     if (delFileWriter != null) {
296       if (deleteMarkersCount > 0) {
297         // If the del file is not empty, commit it.
298         // If the commit fails, the compaction is re-performed again.
299         mobStore.commitFile(delFileWriter.getPath(), path);
300       } else {
301         try {
302           // If the del file is empty, delete it instead of committing.
303           store.getFileSystem().delete(delFileWriter.getPath(), true);
304         } catch (IOException e) {
305           LOG.error("Failed to delete the temp del file", e);
306         }
307       }
308     }
309     mobStore.updateCellsCountCompactedFromMob(cellsCountCompactedFromMob);
310     mobStore.updateCellsCountCompactedToMob(cellsCountCompactedToMob);
311     mobStore.updateCellsSizeCompactedFromMob(cellsSizeCompactedFromMob);
312     mobStore.updateCellsSizeCompactedToMob(cellsSizeCompactedToMob);
313     progress.complete();
314     return true;
315   }
