30  /**
31   * Provides a unified view of all the underlying ByteBuffers and will look as if a bigger
32   * sequential buffer. This class provides similar APIs as in {@link ByteBuffer} to put/get int,
33   * short, long etc and doing operations like mark, reset, slice etc. This has to be used when
34   * data is split across multiple byte buffers and we don't want copy them to single buffer
35   * for reading from it.
36   */
37  @InterfaceAudience.Private
38  public class MultiByteBuff extends ByteBuff {
39  
40    private final ByteBuffer[] items;
41    // Pointer to the current item in the MBB
42    private ByteBuffer curItem = null;
43    // Index of the current item in the MBB
44    private int curItemIndex = 0;
45  
46    private int limit = 0;
47    private int limitedItemIndex;
48    private int markedItemIndex = -1;
49    private final int[] itemBeginPos;
50  
51    public MultiByteBuff(ByteBuffer... items) {
52      assert items != null;
53      assert items.length > 0;
54      this.items = items;
55      this.curItem = this.items[this.curItemIndex];
56      // See below optimization in getInt(int) where we check whether the given index land in current
57      // item. For this we need to check whether the passed index is less than the next item begin
58      // offset. To handle this effectively for the last item buffer, we add an extra item into this
59      // array.
60      itemBeginPos = new int[items.length + 1];
61      int offset = 0;
62      for (int i = 0; i < items.length; i++) {
63        ByteBuffer item = items[i];
64        item.rewind();
65        itemBeginPos[i] = offset;
66        int l = item.limit() - item.position();
67        offset += l;
68      }
69      this.limit = offset;
70      this.itemBeginPos[items.length] = offset + 1;
71      this.limitedItemIndex = this.items.length - 1;
72    }
73  
74    private MultiByteBuff(ByteBuffer[] items, int[] itemBeginPos, int limit, int limitedIndex,
75        int curItemIndex, int markedIndex) {
76      this.items = items;
77      this.curItemIndex = curItemIndex;
78      this.curItem = this.items[this.curItemIndex];
79      this.itemBeginPos = itemBeginPos;
80      this.limit = limit;
81      this.limitedItemIndex = limitedIndex;
82      this.markedItemIndex = markedIndex;
83    }

134   @Override
135   public byte getByteAfterPosition(int offset) {
136     // Mostly the index specified will land within this current item. Short circuit for that
137     int index = offset + this.position();
138     int itemIndex = getItemIndexFromCurItemIndex(index);
139     return ByteBufferUtils.toByte(this.items[itemIndex], index - this.itemBeginPos[itemIndex]);

142   /*
143    * Returns in which sub ByteBuffer, the given element index will be available.
144    */
145   private int getItemIndex(int elemIndex) {
146     int index = 1;
147     while (elemIndex >= this.itemBeginPos[index]) {
148       index++;
149       if (index == this.itemBeginPos.length) {
150         throw new IndexOutOfBoundsException();
151       }
152     }
153     return index - 1;
154   }
155 
156   /*
157    * Returns in which sub ByteBuffer, the given element index will be available. In this case we are
158    * sure that the item will be after MBB's current position
159    */
160   private int getItemIndexFromCurItemIndex(int elemIndex) {
161     int index = this.curItemIndex;
162     while (elemIndex >= this.itemBeginPos[index]) {
163       index++;
164       if (index == this.itemBeginPos.length) {
165         throw new IndexOutOfBoundsException();
166       }
167     }
168     return index - 1;
169   }

171   /**
172    * Fetches the int at the given index. Does not change position of the underlying ByteBuffers
173    * @param index
174    * @return the int value at the given index
175    */
176   public int getInt(int index) {
177     // Mostly the index specified will land within this current item. Short circuit for that
178     int itemIndex;
179     if (this.itemBeginPos[this.curItemIndex] <= index
180         && this.itemBeginPos[this.curItemIndex + 1] > index) {
181       itemIndex = this.curItemIndex;
182     } else {
183       itemIndex = getItemIndex(index);
184     }
185     return getInt(index, itemIndex);
186   }

201   /**
202    * Fetches the short at the given index. Does not change position of the underlying ByteBuffers
203    * @param index
204    * @return the short value at the given index
205    */
206   public short getShort(int index) {
207     // Mostly the index specified will land within this current item. Short circuit for that
208     int itemIndex;
209     if (this.itemBeginPos[this.curItemIndex] <= index
210         && this.itemBeginPos[this.curItemIndex + 1] > index) {
211       itemIndex = this.curItemIndex;
212     } else {
213       itemIndex = getItemIndex(index);
214     }
215     ByteBuffer item = items[itemIndex];
216     int offsetInItem = index - this.itemBeginPos[itemIndex];
217     if (item.limit() - offsetInItem >= Bytes.SIZEOF_SHORT) {
218       return ByteBufferUtils.toShort(item, offsetInItem);
219     }
220     if (items.length - 1 == itemIndex) {
221       // means cur item is the last one and we wont be able to read a int. Throw exception
222       throw new BufferUnderflowException();
223     }
224     ByteBuffer nextItem = items[itemIndex + 1];
225     // Get available one byte from this item and remaining one from next
226     short n = 0;
227     n ^= ByteBufferUtils.toByte(item, offsetInItem) & 0xFF;
228     n <<= 8;
229     n ^= ByteBufferUtils.toByte(nextItem, 0) & 0xFF;
230     return n;
231   }

246   private int getInt(int index, int itemIndex) {
247     ByteBuffer item = items[itemIndex];
248     int offsetInItem = index - this.itemBeginPos[itemIndex];
249     int remainingLen = item.limit() - offsetInItem;
250     if (remainingLen >= Bytes.SIZEOF_INT) {
251       return ByteBufferUtils.toInt(item, offsetInItem);
252     }
253     if (items.length - 1 == itemIndex) {
254       // means cur item is the last one and we wont be able to read a int. Throw exception
255       throw new BufferUnderflowException();
256     }
257     ByteBuffer nextItem = items[itemIndex + 1];
258     // Get available bytes from this item and remaining from next
259     int l = 0;
260     for (int i = offsetInItem; i < item.capacity(); i++) {
261       l <<= 8;
262       l ^= ByteBufferUtils.toByte(item, i) & 0xFF;
263     }
264     for (int i = 0; i < Bytes.SIZEOF_INT - remainingLen; i++) {
265       l <<= 8;
266       l ^= ByteBufferUtils.toByte(nextItem, i) & 0xFF;
267     }
268     return l;
269   }

380   /**
381    * Sets this MBB's position to the given value.
382    * @param position
383    * @return this object
384    */
385   @Override
386   public MultiByteBuff position(int position) {
387     // Short circuit for positioning within the cur item. Mostly that is the case.
388     if (this.itemBeginPos[this.curItemIndex] <= position
389         && this.itemBeginPos[this.curItemIndex + 1] > position) {
390       this.curItem.position(position - this.itemBeginPos[this.curItemIndex]);
391       return this;
392     }
393     int itemIndex = getItemIndex(position);
394     // All items from 0 - curItem-1 set position at end.
395     for (int i = 0; i < itemIndex; i++) {
396       this.items[i].position(this.items[i].limit());
397     }
398     // All items after curItem set position at begin
399     for (int i = itemIndex + 1; i < this.items.length; i++) {
400       this.items[i].position(0);
401     }
402     this.curItem = this.items[itemIndex];
403     this.curItem.position(position - this.itemBeginPos[itemIndex]);
404     this.curItemIndex = itemIndex;
405     return this;
406   }

408   /**
409    * Rewinds this MBB and the position is set to 0
410    * @return this object
411    */
412   @Override
413   public MultiByteBuff rewind() {
414     for (int i = 0; i < this.items.length; i++) {
415       this.items[i].rewind();
416     }
417     this.curItemIndex = 0;
418     this.curItem = this.items[this.curItemIndex];
419     this.markedItemIndex = -1;
420     return this;
421   }

434   /**
435    * Similar to {@link ByteBuffer}.reset(), ensures that this MBB
436    * is reset back to last marked position.
437    * @return This MBB
438    */
439   @Override
440   public MultiByteBuff reset() {
441     // when the buffer is moved to the next one.. the reset should happen on the previous marked
442     // item and the new one should be taken as the base
443     if (this.markedItemIndex < 0) throw new InvalidMarkException();
444     ByteBuffer markedItem = this.items[this.markedItemIndex];
445     markedItem.reset();
446     this.curItem = markedItem;
447     // All items after the marked position upto the current item should be reset to 0
448     for (int i = this.curItemIndex; i > this.markedItemIndex; i--) {
449       this.items[i].position(0);
450     }
451     this.curItemIndex = this.markedItemIndex;
452     return this;
453   }

478   /**
479    * A relative method that returns byte at the current position.  Increments the
480    * current position by the size of a byte.
481    * @return the byte at the current position
482    */
483   @Override
484   public byte get() {
485     if (this.curItem.remaining() == 0) {
486       if (items.length - 1 == this.curItemIndex) {
487         // means cur item is the last one and we wont be able to read a long. Throw exception
488         throw new BufferUnderflowException();
489       }
490       this.curItemIndex++;
491       this.curItem = this.items[this.curItemIndex];
492     }
493     return this.curItem.get();
494   }

496   /**
497    * Returns the short value at the current position. Also advances the position by the size
498    * of short
499    *
500    * @return the short value at the current position
501    */
502   @Override
503   public short getShort() {
504     int remaining = this.curItem.remaining();
505     if (remaining >= Bytes.SIZEOF_SHORT) {
506       return this.curItem.getShort();
507     }
508     if (remaining == 0) {
509       if (items.length - 1 == this.curItemIndex) {
510         // means cur item is the last one and we wont be able to read a long. Throw exception
511         throw new BufferUnderflowException();
512       }
513       this.curItemIndex++;
514       this.curItem = this.items[this.curItemIndex];
515       return this.curItem.getShort();
516     }
517     short n = 0;
518     n ^= get() & 0xFF;
519     n <<= 8;
520     n ^= get() & 0xFF;
521     return n;
522   }

586    * @param dst
587    */

589   public void get(byte[] dst) {
590     get(dst, 0, dst.length);
591   }
592 
593   /**
594    * Copies the specified number of bytes from this MBB's current position to the byte[]'s offset.
595    * Also advances the position of the MBB by the given length.
596    * @param dst
597    * @param offset within the current array
598    * @param length upto which the bytes to be copied
599    */
600   @Override
601   public void get(byte[] dst, int offset, int length) {
602     while (length > 0) {
603       int toRead = Math.min(length, this.curItem.remaining());
604       ByteBufferUtils.copyFromBufferToArray(dst, this.curItem, this.curItem.position(), offset,
605           toRead);
606       this.curItem.position(this.curItem.position() + toRead);
607       length -= toRead;
608       if (length == 0)
609         break;
610       this.curItemIndex++;
611       this.curItem = this.items[this.curItemIndex];
612       offset += toRead;
613     }
614   }

616   @Override
617   public void get(int sourceOffset, byte[] dst, int offset, int length) {
618     int itemIndex = getItemIndex(sourceOffset);
619     ByteBuffer item = this.items[itemIndex];
620     sourceOffset = sourceOffset - this.itemBeginPos[itemIndex];
621     while (length > 0) {
622       int toRead = Math.min((item.limit() - sourceOffset), length);
623       ByteBufferUtils.copyFromBufferToArray(dst, item, sourceOffset, offset,
624           toRead);
625       length -= toRead;
626       if (length == 0)
627         break;
628       itemIndex++;
629       item = this.items[itemIndex];
630       offset += toRead;
631       sourceOffset = 0;
632     }
633   }

635   /**
636    * Marks the limit of this MBB.
637    * @param limit
638    * @return This MBB
639    */
640   @Override
641   public MultiByteBuff limit(int limit) {
642     this.limit = limit;
643     // Normally the limit will try to limit within the last BB item
644     int limitedIndexBegin = this.itemBeginPos[this.limitedItemIndex];
645     if (limit >= limitedIndexBegin && limit < this.itemBeginPos[this.limitedItemIndex + 1]) {
646       this.items[this.limitedItemIndex].limit(limit - limitedIndexBegin);
647       return this;
648     }
649     int itemIndex = getItemIndex(limit);
650     int beginOffset = this.itemBeginPos[itemIndex];
651     int offsetInItem = limit - beginOffset;
652     ByteBuffer item = items[itemIndex];
653     item.limit(offsetInItem);
654     for (int i = this.limitedItemIndex; i < itemIndex; i++) {
655       this.items[i].limit(this.items[i].capacity());
656     }
657     this.limitedItemIndex = itemIndex;
658     for (int i = itemIndex + 1; i < this.items.length; i++) {
659       this.items[i].limit(this.items[i].position());
660     }
661     return this;
662   }

673   /**
674    * Returns an MBB which is a sliced version of this MBB. The position, limit and mark
675    * of the new MBB will be independent than that of the original MBB.
676    * The content of the new MBB will start at this MBB's current position
677    * @return a sliced MBB
678    */
679   @Override
680   public MultiByteBuff slice() {
681     ByteBuffer[] copy = new ByteBuffer[this.limitedItemIndex - this.curItemIndex + 1];
682     for (int i = curItemIndex, j = 0; i <= this.limitedItemIndex; i++, j++) {
683       copy[j] = this.items[i].slice();
684     }
685     return new MultiByteBuff(copy);
686   }

750     offset = offset - this.itemBeginPos[destItemIndex];

753     srcOffset = srcOffset - this.itemBeginPos[srcItemIndex];
754     int toRead, toWrite, toMove;
755     while (length > 0) {
756       toWrite = destItem.limit() - offset;
757       toRead = srcItem.limit() - srcOffset;
758       toMove = Math.min(length, Math.min(toRead, toWrite));
759       ByteBufferUtils.copyFromBufferToBuffer(srcItem, destItem, srcOffset, offset, toMove);
760       length -= toMove;
761       if (length == 0) break;
762       if (toRead < toWrite) {
763         srcItem = getItemByteBuffer(src, ++srcItemIndex);
764         srcOffset = 0;
765         offset += toMove;
766       } else if (toRead > toWrite) {
767         destItem = this.items[++destItemIndex];
768         offset = 0;
769         srcOffset += toMove;
770       } else {
771         // toRead = toWrite case
772         srcItem = getItemByteBuffer(src, ++srcItemIndex);
773         srcOffset = 0;
774         destItem = this.items[++destItemIndex];
775         offset = 0;
776       }
777     }

1018       pair.setSecond(offset);
1019       return;
1020     }
1021     byte[] dst = new byte[length];
