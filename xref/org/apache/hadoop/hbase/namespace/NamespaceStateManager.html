37  /**
38   * NamespaceStateManager manages state (in terms of quota) of all the namespaces. It contains
39   * a cache which is updated based on the hooks in the NamespaceAuditor class.
40   */
41  @InterfaceAudience.Private
42  class NamespaceStateManager {
43  
44    private static final Log LOG = LogFactory.getLog(NamespaceStateManager.class);
45    private ConcurrentMap<String, NamespaceTableAndRegionInfo> nsStateCache;
46    private MasterServices master;
47    private volatile boolean initialized = false;
48  
49    public NamespaceStateManager(MasterServices masterServices) {
50      nsStateCache = new ConcurrentHashMap<String, NamespaceTableAndRegionInfo>();
51      master = masterServices;
52    }
53  
54    /**
55     * Starts the NamespaceStateManager. The boot strap of cache
56     * is done in the post master start hook of the NamespaceAuditor
57     * class.
58     *
59     * @throws IOException Signals that an I/O exception has occurred.
60     */
61    public void start() throws IOException {
62      LOG.info("Namespace State Manager started.");
63      initialize();
64    }
65  
66    /**
67     * Gets an instance of NamespaceTableAndRegionInfo associated with namespace.
68     * @param The name of the namespace
69     * @return An instance of NamespaceTableAndRegionInfo.
70     */
71    public NamespaceTableAndRegionInfo getState(String name) {
72      return nsStateCache.get(name);
73    }

75    /**
76     * Check if adding a region violates namespace quota, if not update namespace cache.
77     *
78     * @param TableName
79     * @param regionName
80     * @param incr
81     * @return true, if region can be added to table.
82     * @throws IOException Signals that an I/O exception has occurred.
83     */
84    synchronized boolean checkAndUpdateNamespaceRegionCount(TableName name,
85        byte[] regionName, int incr) throws IOException {
86      String namespace = name.getNamespaceAsString();
87      NamespaceDescriptor nspdesc = getNamespaceDescriptor(namespace);
88      if (nspdesc != null) {
89        NamespaceTableAndRegionInfo currentStatus;
90        currentStatus = getState(namespace);
91        if (incr > 0 &&
92            currentStatus.getRegionCount() >= TableNamespaceManager.getMaxRegions(nspdesc)) {
93          LOG.warn("The region " + Bytes.toStringBinary(regionName)
94              + " cannot be created. The region count  will exceed quota on the namespace. "
95              + "This may be transient, please retry later if there are any ongoing split"
96              + " operations in the namespace.");
97          return false;
98        }
99        NamespaceTableAndRegionInfo nsInfo = nsStateCache.get(namespace);
100       if (nsInfo != null) {
101         nsInfo.incRegionCountForTable(name, incr);
102       } else {
103         LOG.warn("Namespace state found null for namespace : " + namespace);
104       }
105     }
106     return true;
107   }
108   
109   /**
110    * Check and update region count for an existing table. To handle scenarios like restore snapshot
111    * @param TableName name of the table for region count needs to be checked and updated
112    * @param incr count of regions
113    * @throws QuotaExceededException if quota exceeds for the number of regions allowed in a
114    *           namespace
115    * @throws IOException Signals that an I/O exception has occurred.
116    */
117   synchronized void checkAndUpdateNamespaceRegionCount(TableName name, int incr) 
118       throws IOException {
119     String namespace = name.getNamespaceAsString();
120     NamespaceDescriptor nspdesc = getNamespaceDescriptor(namespace);
121     if (nspdesc != null) {
122       NamespaceTableAndRegionInfo currentStatus = getState(namespace);
123       int regionCountOfTable = currentStatus.getRegionCountOfTable(name);
124       if ((currentStatus.getRegionCount() - regionCountOfTable + incr) > TableNamespaceManager
125           .getMaxRegions(nspdesc)) {
126         throw new QuotaExceededException("The table " + name.getNameAsString()
127             + " region count cannot be updated as it would exceed maximum number "
128             + "of regions allowed in the namespace.  The total number of regions permitted is "
129             + TableNamespaceManager.getMaxRegions(nspdesc));
130       }
131       currentStatus.removeTable(name);
132       currentStatus.addTable(name, incr);
133     }
134   }

136   private NamespaceDescriptor getNamespaceDescriptor(String namespaceAsString) {
137     try {
138       return this.master.getNamespaceDescriptor(namespaceAsString);
139     } catch (IOException e) {
140       LOG.error("Error while fetching namespace descriptor for namespace : " + namespaceAsString);
141       return null;
142     }
143   }
144 
145   synchronized void checkAndUpdateNamespaceTableCount(TableName table, int numRegions)
146       throws IOException {
147     String namespace = table.getNamespaceAsString();
148     NamespaceDescriptor nspdesc = getNamespaceDescriptor(namespace);
149     if (nspdesc != null) {
150       NamespaceTableAndRegionInfo currentStatus;
151       currentStatus = getState(nspdesc.getName());
152       if ((currentStatus.getTables().size()) >= TableNamespaceManager.getMaxTables(nspdesc)) {
153         throw new QuotaExceededException("The table " + table.getNameAsString()
154             + "cannot be created as it would exceed maximum number of tables allowed "
155             + " in the namespace.  The total number of tables permitted is "
156             + TableNamespaceManager.getMaxTables(nspdesc));
157       }
158       if ((currentStatus.getRegionCount() + numRegions) > TableNamespaceManager
159           .getMaxRegions(nspdesc)) {
160         throw new QuotaExceededException("The table " + table.getNameAsString()
161             + " is not allowed to have " + numRegions
162             + " regions. The total number of regions permitted is only "
163             + TableNamespaceManager.getMaxRegions(nspdesc)
164             + ", while current region count is " + currentStatus.getRegionCount()
165             + ". This may be transient, please retry later if there are any"
166             + " ongoing split operations in the namespace.");
167       }
168     } else {
169       throw new IOException("Namespace Descriptor found null for " + namespace
170           + " This is unexpected.");
171     }
172     addTable(table, numRegions);
173   }
174 
175   NamespaceTableAndRegionInfo addNamespace(String namespace) {
176     if (!nsStateCache.containsKey(namespace)) {
177       NamespaceTableAndRegionInfo a1 = new NamespaceTableAndRegionInfo(namespace);
178       nsStateCache.put(namespace, a1);
179     }
180     return nsStateCache.get(namespace);
181   }
182 
183   /**
184    * Delete the namespace state.
185    *
186    * @param An instance of NamespaceTableAndRegionInfo
187    */
188   void deleteNamespace(String namespace) {
189     this.nsStateCache.remove(namespace);
190   }
191 
192   private void addTable(TableName tableName, int regionCount) throws IOException {
193     NamespaceTableAndRegionInfo info =
194         nsStateCache.get(tableName.getNamespaceAsString());
195     if(info != null) {
196       info.addTable(tableName, regionCount);
197     } else {
198       throw new IOException("Bad state : Namespace quota information not found for namespace : "
199           + tableName.getNamespaceAsString());
200     }
201   }
202 
203   synchronized void removeTable(TableName tableName) {
204     NamespaceTableAndRegionInfo info =
205         nsStateCache.get(tableName.getNamespaceAsString());
206     if (info != null) {
207       info.removeTable(tableName);
208     }
209   }

211   /**
212    * Initialize namespace state cache by scanning meta table.
213    */
214   private void initialize() throws IOException {
215     List<NamespaceDescriptor> namespaces = this.master.listNamespaceDescriptors();
216     for (NamespaceDescriptor namespace : namespaces) {
217       addNamespace(namespace.getName());
218       List<TableName> tables = this.master.listTableNamesByNamespace(namespace.getName());
219       for (TableName table : tables) {
220         if (table.isSystemTable()) {
221           continue;
222         }
223         List<HRegionInfo> regions =
224             MetaTableAccessor.getTableRegions(this.master.getConnection(), table, true);
225         addTable(table, regions.size());
226       }
227     }
228     LOG.info("Finished updating state of " + nsStateCache.size() + " namespaces. ");
229     initialized = true;
230   }
231 
232   boolean isInitialized() {
233     return initialized;
234   }
235 
236   public synchronized void removeRegionFromTable(HRegionInfo hri) throws IOException {
237     String namespace = hri.getTable().getNamespaceAsString();
238     NamespaceTableAndRegionInfo nsInfo = nsStateCache.get(namespace);
239     if (nsInfo != null) {
240       nsInfo.decrementRegionCountForTable(hri.getTable(), 1);
241     } else {
242       throw new IOException("Namespace state found null for namespace : " + namespace);
243     }
244   }
