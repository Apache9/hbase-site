054/**
055 * This class tests operations in MasterRpcServices which require ADMIN access.
056 * It doesn't test all operations which require ADMIN access, only those which get vetted within
057 * MasterRpcServices at the point of entry itself (unlike old approach of using
058 * hooks in AccessController).
059 *
060 * Sidenote:
061 * There is one big difference between how security tests for AccessController hooks work, and how
062 * the tests in this class for security in MasterRpcServices work.
063 * The difference arises because of the way AC & MasterRpcServices get the user.
064 *
065 * In AccessController, it first checks if there is an active rpc user in ObserverContext. If not,
066 * it uses UserProvider for current user. This *might* make sense in the context of coprocessors,
067 * because they can be called outside the context of RPCs.
068 * But in the context of MasterRpcServices, only one way makes sense - RPCServer.getRequestUser().
069 *
070 * In AC tests, when we do FooUser.runAs on AccessController instance directly, it bypasses
071 * the rpc framework completely, but works because UserProvider provides the correct user, i.e.
072 * FooUser in this case.
073 *
074 * But this doesn't work for the tests here, so we go around by doing complete RPCs.
075 */
076@Category({SecurityTests.class, MediumTests.class})
077public class TestAdminOnlyOperations {
078  @ClassRule
079  public static final HBaseClassTestRule CLASS_RULE =
080      HBaseClassTestRule.forClass(TestAdminOnlyOperations.class);
081
082  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
083  private static Configuration conf;
084

087  // user without admin permissions
088  private static User USER_NON_ADMIN;

090  private static final String GROUP_ADMIN = "admin_group";
091  private static User USER_GROUP_ADMIN;
092
093  // Dummy service to test execService calls. Needs to be public so can be loaded as Coprocessor.
094  public static class DummyCpService implements MasterCoprocessor, RegionServerCoprocessor {
095    public DummyCpService() {}
096

099      return Collections.singleton(mock(TestRpcServiceProtos.TestProtobufRpcProto.class));

101  }

103  private static void enableSecurity(Configuration conf) throws IOException {
104    conf.set("hadoop.security.authorization", "false");
105    conf.set("hadoop.security.authentication", "simple");
106    conf.set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY, AccessController.class.getName() +
107      "," + DummyCpService.class.getName());
108    conf.set(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY, AccessController.class.getName());
109    conf.set(CoprocessorHost.REGIONSERVER_COPROCESSOR_CONF_KEY, AccessController.class.getName() +
110      "," + DummyCpService.class.getName());
111    conf.set(User.HBASE_SECURITY_AUTHORIZATION_CONF_KEY, "true");
112    SecureTestUtil.configureSuperuser(conf);
113  }

115  @BeforeClass
116  public static void setup() throws Exception {
117    conf = TEST_UTIL.getConfiguration();
118
119    // Enable security
120    enableSecurity(conf);
121    TEST_UTIL.startMiniCluster();
122
123    // Wait for the ACL table to become available
124    TEST_UTIL.waitUntilAllRegionsAssigned(AccessControlLists.ACL_TABLE_NAME);
125
126    // Create users
127    USER_ADMIN = User.createUserForTesting(conf, "admin", new String[0]);
128    USER_NON_ADMIN = User.createUserForTesting(conf, "non_admin", new String[0]);
129    USER_GROUP_ADMIN =
130        User.createUserForTesting(conf, "user_group_admin", new String[] { GROUP_ADMIN });
131
132    // Assign permissions to users and groups
133    SecureTestUtil.grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(), Permission.Action.ADMIN);
134    SecureTestUtil.grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
135    // No permissions to USER_NON_ADMIN
136  }
137
138  interface Action {
139    void run(Admin admin) throws Exception;
140  }
141
142  private void verifyAllowed(User user, Action action) throws Exception {
143    user.runAs((PrivilegedExceptionAction<?>) () -> {
144      try (Connection conn = ConnectionFactory.createConnection(conf);
145          Admin admin = conn.getAdmin()) {
146        action.run(admin);
147      } catch (IOException e) {
148        fail(e.toString());
149      }
150      return null;
151    });
152  }
153
154  private void verifyDenied(User user, Action action) throws Exception {
155    user.runAs((PrivilegedExceptionAction<?>) () -> {
156      boolean accessDenied = false;
157      try (Connection conn = ConnectionFactory.createConnection(conf);
158          Admin admin = conn.getAdmin()) {
159        action.run(admin);
160      } catch (AccessDeniedException e) {
161        accessDenied = true;
162      }
163      assertTrue("Expected access to be denied", accessDenied);
164      return null;
165    });
166  }
167
168  private void verifiedDeniedServiceException(User user, Action action) throws Exception {
169    user.runAs((PrivilegedExceptionAction<?>) () -> {
170      boolean accessDenied = false;
171      try (Connection conn = ConnectionFactory.createConnection(conf);
172          Admin admin = conn.getAdmin()) {
173        action.run(admin);
174      } catch (ServiceException e) {
175        // For MasterRpcServices.execService.
176        if (e.getCause() instanceof AccessDeniedException) {
177          accessDenied = true;
178        }
179      }
180      assertTrue("Expected access to be denied", accessDenied);
181      return null;
182    });
183
184  }
185
186  private void verifyAdminCheckForAction(Action action) throws Exception {
187    verifyAllowed(USER_ADMIN, action);
188    verifyAllowed(USER_GROUP_ADMIN, action);
189    verifyDenied(USER_NON_ADMIN, action);
190  }

221  @Test
222  public void testExecService() throws Exception {
223    Action action = (admin) -> {
224      TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface service =
225          TestRpcServiceProtos.TestProtobufRpcProto.newBlockingStub(admin.coprocessorService());
226      service.ping(null, TestProtos.EmptyRequestProto.getDefaultInstance());
227    };
228
229    verifyAllowed(USER_ADMIN, action);
230    verifyAllowed(USER_GROUP_ADMIN, action);
231    // This is same as above verifyAccessDenied
232    verifiedDeniedServiceException(USER_NON_ADMIN, action);
233  }
