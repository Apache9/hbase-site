181   * Create a protocol buffer MutateRequest for a client increment
182   *
183   * @param regionName
184   * @param row
185   * @param family
186   * @param qualifier
187   * @param amount
188   * @param durability
189   * @return a mutate request
190   */
191  public static MutateRequest buildIncrementRequest(
192      final byte[] regionName, final byte[] row, final byte[] family, final byte[] qualifier,
193      final long amount, final Durability durability, long nonceGroup, long nonce) {
194    MutateRequest.Builder builder = MutateRequest.newBuilder();
195    RegionSpecifier region = buildRegionSpecifier(
196      RegionSpecifierType.REGION_NAME, regionName);
197    builder.setRegion(region);
198
199    MutationProto.Builder mutateBuilder = MutationProto.newBuilder();
200    mutateBuilder.setRow(UnsafeByteOperations.unsafeWrap(row));
201    mutateBuilder.setMutateType(MutationType.INCREMENT);
202    mutateBuilder.setDurability(ProtobufUtil.toDurability(durability));
203    ColumnValue.Builder columnBuilder = ColumnValue.newBuilder();
204    columnBuilder.setFamily(UnsafeByteOperations.unsafeWrap(family));
205    QualifierValue.Builder valueBuilder = QualifierValue.newBuilder();
206    valueBuilder.setValue(UnsafeByteOperations.unsafeWrap(Bytes.toBytes(amount)));
207    valueBuilder.setQualifier(UnsafeByteOperations
208        .unsafeWrap(qualifier == null ? HConstants.EMPTY_BYTE_ARRAY : qualifier));
209    valueBuilder.setTimestamp(HConstants.LATEST_TIMESTAMP);
210    columnBuilder.addQualifierValue(valueBuilder.build());
211    mutateBuilder.addColumnValue(columnBuilder.build());
212    if (nonce != HConstants.NO_NONCE) {
213      mutateBuilder.setNonce(nonce);
214    }
215    builder.setMutation(mutateBuilder.build());
216    if (nonceGroup != HConstants.NO_NONCE) {
217      builder.setNonceGroup(nonceGroup);
218    }
219    return builder.build();
220  }

292  public static ClientProtos.MultiRequest buildMutateRequest(
293      final byte[] regionName, final byte[] row, final byte[] family,
294      final byte [] qualifier, final ByteArrayComparable comparator,
295      final CompareType compareType, final RowMutations rowMutations) throws IOException {
296    RegionAction.Builder builder =
297        getRegionActionBuilderWithRegion(RegionAction.newBuilder(), regionName);
298    builder.setAtomic(true);
299    ClientProtos.Action.Builder actionBuilder = ClientProtos.Action.newBuilder();
300    MutationProto.Builder mutationBuilder = MutationProto.newBuilder();
301    Condition condition = buildCondition(
302        row, family, qualifier, comparator, compareType);
303    for (Mutation mutation: rowMutations.getMutations()) {
304      MutationType mutateType = null;
305      if (mutation instanceof Put) {
306        mutateType = MutationType.PUT;
307      } else if (mutation instanceof Delete) {
308        mutateType = MutationType.DELETE;
309      } else {
310        throw new DoNotRetryIOException("RowMutations supports only put and delete, not " +
311            mutation.getClass().getName());
312      }
313      mutationBuilder.clear();
314      MutationProto mp = ProtobufUtil.toMutation(mutateType, mutation, mutationBuilder);
315      actionBuilder.clear();
316      actionBuilder.setMutation(mp);
317      builder.addAction(actionBuilder.build());
318    }
319    ClientProtos.MultiRequest request =
320        ClientProtos.MultiRequest.newBuilder().addRegionAction(builder.build())
321            .setCondition(condition).build();
322    return request;
323  }

627  /**
628   * Create a protocol buffer multi request for a list of actions. Propagates Actions original
629   * index. The passed in multiRequestBuilder will be populated with region actions.
630   * @param regionName The region name of the actions.
631   * @param actions The actions that are grouped by the same region name.
632   * @param multiRequestBuilder The multiRequestBuilder to be populated with region actions.
633   * @param regionActionBuilder regionActionBuilder to be used to build region action.
634   * @param actionBuilder actionBuilder to be used to build action.
635   * @param mutationBuilder mutationBuilder to be used to build mutation.
636   * @param nonceGroup nonceGroup to be applied.
637   * @param rowMutationsIndexMap Map of created RegionAction to the original index for a
638   *          RowMutations within the original list of actions
639   * @throws IOException
640   */
641  public static void buildRegionActions(final byte[] regionName,
642      final List<Action> actions, final MultiRequest.Builder multiRequestBuilder,
643      final RegionAction.Builder regionActionBuilder,
644      final ClientProtos.Action.Builder actionBuilder,
645      final MutationProto.Builder mutationBuilder,
646      long nonceGroup, final Map<Integer, Integer> rowMutationsIndexMap) throws IOException {
647    regionActionBuilder.clear();
648    RegionAction.Builder builder = getRegionActionBuilderWithRegion(
649      regionActionBuilder, regionName);
650    ClientProtos.CoprocessorServiceCall.Builder cpBuilder = null;
651    boolean hasNonce = false;
652    List<Action> rowMutationsList = new ArrayList<>();
653
654    for (Action action: actions) {
655      Row row = action.getAction();
656      actionBuilder.clear();
657      actionBuilder.setIndex(action.getOriginalIndex());
658      mutationBuilder.clear();
659      if (row instanceof Get) {
660        Get g = (Get)row;
661        builder.addAction(actionBuilder.setGet(ProtobufUtil.toGet(g)));
662      } else if (row instanceof Put) {
663        builder.addAction(actionBuilder.
664          setMutation(ProtobufUtil.toMutation(MutationType.PUT, (Put)row, mutationBuilder)));
665      } else if (row instanceof Delete) {
666        builder.addAction(actionBuilder.
667          setMutation(ProtobufUtil.toMutation(MutationType.DELETE, (Delete)row, mutationBuilder)));
668      } else if (row instanceof Append) {
669        builder.addAction(actionBuilder.setMutation(ProtobufUtil.toMutation(
670            MutationType.APPEND, (Append)row, mutationBuilder, action.getNonce())));
671        hasNonce = true;
672      } else if (row instanceof Increment) {
673        builder.addAction(actionBuilder.setMutation(ProtobufUtil.toMutation(
674            MutationType.INCREMENT, (Increment)row, mutationBuilder, action.getNonce())));
675        hasNonce = true;
676      } else if (row instanceof RegionCoprocessorServiceExec) {
677        RegionCoprocessorServiceExec exec = (RegionCoprocessorServiceExec) row;
678        // DUMB COPY!!! FIX!!! Done to copy from c.g.p.ByteString to shaded ByteString.
679        org.apache.hbase.thirdparty.com.google.protobuf.ByteString value =
680         org.apache.hbase.thirdparty.com.google.protobuf.UnsafeByteOperations.unsafeWrap(
681             exec.getRequest().toByteArray());
682        if (cpBuilder == null) {
683          cpBuilder = ClientProtos.CoprocessorServiceCall.newBuilder();
684        } else {
685          cpBuilder.clear();
686        }
687        builder.addAction(actionBuilder.setServiceCall(
688            cpBuilder.setRow(UnsafeByteOperations.unsafeWrap(exec.getRow()))
689              .setServiceName(exec.getMethod().getService().getFullName())
690              .setMethodName(exec.getMethod().getName())
691              .setRequest(value)));
692      } else if (row instanceof RowMutations) {
693        rowMutationsList.add(action);
694      } else {
695        throw new DoNotRetryIOException("Multi doesn't support " + row.getClass().getName());
696      }
697    }
698    if (!multiRequestBuilder.hasNonceGroup() && hasNonce) {
699      multiRequestBuilder.setNonceGroup(nonceGroup);
700    }
701    multiRequestBuilder.addRegionAction(builder.build());
702
703    // Process RowMutations here. We can not process it in the big loop above because
704    // it will corrupt the sequence order maintained in cells.
705    // RowMutations is a set of Puts and/or Deletes all to be applied atomically
706    // on the one row. We do separate RegionAction for each RowMutations.
707    // We maintain a map to keep track of this RegionAction and the original Action index.
708    for (Action action : rowMutationsList) {
709      RowMutations rms = (RowMutations) action.getAction();
710      RegionAction.Builder rowMutationsRegionActionBuilder =
711          RequestConverter.buildRegionAction(regionName, rms);
712      rowMutationsRegionActionBuilder.setAtomic(true);
713      // Put it in the multiRequestBuilder
714      multiRequestBuilder.addRegionAction(rowMutationsRegionActionBuilder.build());
715      // This rowMutations region action is at (multiRequestBuilder.getRegionActionCount() - 1)
716      // in the overall multiRequest.
717      rowMutationsIndexMap.put(multiRequestBuilder.getRegionActionCount() - 1,
718        action.getOriginalIndex());
719    }
720  }

722  /**
723   * Create a protocol buffer multirequest with NO data for a list of actions (data is carried
724   * otherwise than via protobuf).  This means it just notes attributes, whether to write the
725   * WAL, etc., and the presence in protobuf serves as place holder for the data which is
726   * coming along otherwise.  Note that Get is different.  It does not contain 'data' and is always
727   * carried by protobuf.  We return references to the data by adding them to the passed in
728   * <code>data</code> param.
729   * <p> Propagates Actions original index.
730   * <p> The passed in multiRequestBuilder will be populated with region actions.
731   * @param regionName The region name of the actions.
732   * @param actions The actions that are grouped by the same region name.
733   * @param cells Place to stuff references to actual data.
734   * @param multiRequestBuilder The multiRequestBuilder to be populated with region actions.
735   * @param regionActionBuilder regionActionBuilder to be used to build region action.
736   * @param actionBuilder actionBuilder to be used to build action.
737   * @param mutationBuilder mutationBuilder to be used to build mutation.
738   * @param nonceGroup nonceGroup to be applied.
739   * @param rowMutationsIndexMap Map of created RegionAction to the original index for a
740   *          RowMutations within the original list of actions
741   * @throws IOException
742   */
743  public static void buildNoDataRegionActions(final byte[] regionName,
744      final Iterable<Action> actions, final List<CellScannable> cells,
745      final MultiRequest.Builder multiRequestBuilder,
746      final RegionAction.Builder regionActionBuilder,
747      final ClientProtos.Action.Builder actionBuilder,
748      final MutationProto.Builder mutationBuilder,
749      long nonceGroup, final Map<Integer, Integer> rowMutationsIndexMap) throws IOException {
750    regionActionBuilder.clear();
751    RegionAction.Builder builder = getRegionActionBuilderWithRegion(
752      regionActionBuilder, regionName);
753    ClientProtos.CoprocessorServiceCall.Builder cpBuilder = null;
754    RegionAction.Builder rowMutationsRegionActionBuilder = null;
755    boolean hasNonce = false;
756    List<Action> rowMutationsList = new ArrayList<>();
757
758    for (Action action: actions) {
759      Row row = action.getAction();
760      actionBuilder.clear();
761      actionBuilder.setIndex(action.getOriginalIndex());
762      mutationBuilder.clear();
763      if (row instanceof Get) {
764        Get g = (Get)row;
765        builder.addAction(actionBuilder.setGet(ProtobufUtil.toGet(g)));
766      } else if (row instanceof Put) {
767        Put p = (Put)row;
768        cells.add(p);
769        builder.addAction(actionBuilder.
770          setMutation(ProtobufUtil.toMutationNoData(MutationType.PUT, p, mutationBuilder)));
771      } else if (row instanceof Delete) {
772        Delete d = (Delete)row;
773        int size = d.size();
774        // Note that a legitimate Delete may have a size of zero; i.e. a Delete that has nothing
775        // in it but the row to delete.  In this case, the current implementation does not make
776        // a KeyValue to represent a delete-of-all-the-row until we serialize... For such cases
777        // where the size returned is zero, we will send the Delete fully pb'd rather than have
778        // metadata only in the pb and then send the kv along the side in cells.
779        if (size > 0) {
780          cells.add(d);
781          builder.addAction(actionBuilder.
782            setMutation(ProtobufUtil.toMutationNoData(MutationType.DELETE, d, mutationBuilder)));
783        } else {
784          builder.addAction(actionBuilder.
785            setMutation(ProtobufUtil.toMutation(MutationType.DELETE, d, mutationBuilder)));
786        }
787      } else if (row instanceof Append) {
788        Append a = (Append)row;
789        cells.add(a);
790        builder.addAction(actionBuilder.setMutation(ProtobufUtil.toMutationNoData(
791          MutationType.APPEND, a, mutationBuilder, action.getNonce())));
792        hasNonce = true;
793      } else if (row instanceof Increment) {
794        Increment i = (Increment)row;
795        cells.add(i);
796        builder.addAction(actionBuilder.setMutation(ProtobufUtil.toMutationNoData(
797          MutationType.INCREMENT, i, mutationBuilder, action.getNonce())));
798        hasNonce = true;
799      } else if (row instanceof RegionCoprocessorServiceExec) {
800        RegionCoprocessorServiceExec exec = (RegionCoprocessorServiceExec) row;
801        // DUMB COPY!!! FIX!!! Done to copy from c.g.p.ByteString to shaded ByteString.
802        org.apache.hbase.thirdparty.com.google.protobuf.ByteString value =
803         org.apache.hbase.thirdparty.com.google.protobuf.UnsafeByteOperations.unsafeWrap(
804             exec.getRequest().toByteArray());
805        if (cpBuilder == null) {
806          cpBuilder = ClientProtos.CoprocessorServiceCall.newBuilder();
807        } else {
808          cpBuilder.clear();
809        }
810        builder.addAction(actionBuilder.setServiceCall(
811            cpBuilder.setRow(UnsafeByteOperations.unsafeWrap(exec.getRow()))
812              .setServiceName(exec.getMethod().getService().getFullName())
813              .setMethodName(exec.getMethod().getName())
814              .setRequest(value)));
815      } else if (row instanceof RowMutations) {
816        rowMutationsList.add(action);
817      } else {
818        throw new DoNotRetryIOException("Multi doesn't support " + row.getClass().getName());
819      }
820    }
821    if (!multiRequestBuilder.hasNonceGroup() && hasNonce) {
822      multiRequestBuilder.setNonceGroup(nonceGroup);
823    }
824    multiRequestBuilder.addRegionAction(builder.build());
825
826    // Process RowMutations here. We can not process it in the big loop above because
827    // it will corrupt the sequence order maintained in cells.
828    // RowMutations is a set of Puts and/or Deletes all to be applied atomically
829    // on the one row. We do separate RegionAction for each RowMutations.
830    // We maintain a map to keep track of this RegionAction and the original Action index.
831    for (Action action : rowMutationsList) {
832      RowMutations rms = (RowMutations) action.getAction();
833      if (rowMutationsRegionActionBuilder == null) {
834        rowMutationsRegionActionBuilder = ClientProtos.RegionAction.newBuilder();
835      } else {
836        rowMutationsRegionActionBuilder.clear();
837      }
838      rowMutationsRegionActionBuilder.setRegion(
839        RequestConverter.buildRegionSpecifier(RegionSpecifierType.REGION_NAME, regionName));
840      rowMutationsRegionActionBuilder = RequestConverter.buildNoDataRegionAction(regionName, rms,
841        cells, rowMutationsRegionActionBuilder, actionBuilder, mutationBuilder);
842      rowMutationsRegionActionBuilder.setAtomic(true);
843      // Put it in the multiRequestBuilder
844      multiRequestBuilder.addRegionAction(rowMutationsRegionActionBuilder.build());
845      // This rowMutations region action is at (multiRequestBuilder.getRegionActionCount() - 1)
846      // in the overall multiRequest.
847      rowMutationsIndexMap.put(multiRequestBuilder.getRegionActionCount() - 1,
848        action.getOriginalIndex());
849    }
850  }
