050/**
051 * Procedure that describes the unassignment of a single region.
052 * There can only be one RegionTransitionProcedure -- i.e. an assign or an unassign -- per region
053 * running at a time, since each procedure takes a lock on the region.
054 *
055 * <p>The Unassign starts by placing a "close region" request in the Remote Dispatcher
056 * queue, and the procedure will then go into a "waiting state" (suspend).
057 * The Remote Dispatcher will batch the various requests for that server and
058 * they will be sent to the RS for execution.
059 * The RS will complete the open operation by calling master.reportRegionStateTransition().
060 * The AM will intercept the transition report, and notify this procedure.
061 * The procedure will wakeup and finish the unassign by publishing its new state on meta.
062 * <p>If we are unable to contact the remote regionserver whether because of ConnectException
063 * or socket timeout, we will call expire on the server we were trying to contact. We will remain
064 * in suspended state waiting for a wake up from the ServerCrashProcedure that is processing the
065 * failed server. The basic idea is that if we notice a crashed server, then we have a
066 * responsibility; i.e. we should not let go of the region until we are sure the server that was
067 * hosting has had its crash processed. If we let go of the region before then, an assign might
068 * run before the logs have been split which would make for data loss.
069 *
070 * <p>TODO: Rather than this tricky coordination between SCP and this Procedure, instead, work on
071 * returning a SCP as our subprocedure; probably needs work on the framework to do this,
072 * especially if the SCP already created.
073 */
074@InterfaceAudience.Private
075public class UnassignProcedure extends RegionTransitionProcedure {
076  private static final Logger LOG = LoggerFactory.getLogger(UnassignProcedure.class);
077
078  /**
079   * Where to send the unassign RPC.
080   */
081  protected volatile ServerName hostingServer;
082  /**
083   * The Server we will subsequently assign the region too (can be null).
084   */
085  protected volatile ServerName destinationServer;
086
087  // TODO: should this be in a reassign procedure?
088  //       ...and keep unassign for 'disable' case?
089  private boolean force;
090
091  /**
092   * Whether deleting the region from in-memory states after unassigning the region.
093   */
094  private boolean removeAfterUnassigning;
095
096  public UnassignProcedure() {
097    // Required by the Procedure framework to create the procedure on replay
098    super();
099  }
100
101  public UnassignProcedure(final RegionInfo regionInfo, final ServerName hostingServer,
102      final boolean force, final boolean removeAfterUnassigning) {
103    this(regionInfo, hostingServer, null, force, removeAfterUnassigning);
104  }
105
106  public UnassignProcedure(final RegionInfo regionInfo,
107      final ServerName hostingServer, final ServerName destinationServer, final boolean force) {
108    this(regionInfo, hostingServer, destinationServer, force, false);
109  }
110
111  public UnassignProcedure(final RegionInfo regionInfo, final ServerName hostingServer,
112      final ServerName destinationServer, final boolean force,
113      final boolean removeAfterUnassigning) {
114    super(regionInfo);
115    this.hostingServer = hostingServer;
116    this.destinationServer = destinationServer;
117    this.force = force;
118    this.removeAfterUnassigning = removeAfterUnassigning;
119
120    // we don't need REGION_TRANSITION_QUEUE, we jump directly to sending the request
121    setTransitionState(RegionTransitionState.REGION_TRANSITION_DISPATCH);
122  }
123
124  @Override
125  public TableOperationType getTableOperationType() {
126    return TableOperationType.REGION_UNASSIGN;
127  }
128
129  @Override
130  protected boolean isRollbackSupported(final RegionTransitionState state) {
131    switch (state) {
132      case REGION_TRANSITION_QUEUE:
133      case REGION_TRANSITION_DISPATCH:
134        return true;
135      default:
136        return false;
137    }
138  }
139
140  @Override
141  protected void serializeStateData(ProcedureStateSerializer serializer)
142      throws IOException {
143    UnassignRegionStateData.Builder state = UnassignRegionStateData.newBuilder()
144        .setTransitionState(getTransitionState())
145        .setHostingServer(ProtobufUtil.toServerName(this.hostingServer))
146        .setRegionInfo(ProtobufUtil.toRegionInfo(getRegionInfo()));
147    if (this.destinationServer != null) {
148      state.setDestinationServer(ProtobufUtil.toServerName(destinationServer));
149    }
150    if (force) {
151      state.setForce(true);
152    }
153    if (removeAfterUnassigning) {
154      state.setRemoveAfterUnassigning(true);
155    }
156    serializer.serialize(state.build());
157  }
158
159  @Override
160  protected void deserializeStateData(ProcedureStateSerializer serializer)
161      throws IOException {
162    final UnassignRegionStateData state =
163        serializer.deserialize(UnassignRegionStateData.class);
164    setTransitionState(state.getTransitionState());
165    setRegionInfo(ProtobufUtil.toRegionInfo(state.getRegionInfo()));
166    this.hostingServer = ProtobufUtil.toServerName(state.getHostingServer());
167    force = state.getForce();
168    if (state.hasDestinationServer()) {
169      this.destinationServer = ProtobufUtil.toServerName(state.getDestinationServer());
170    }
171    removeAfterUnassigning = state.getRemoveAfterUnassigning();
172  }
173
174  @Override
175  protected boolean startTransition(final MasterProcedureEnv env, final RegionStateNode regionNode) {
176    // nothing to do here. we skip the step in the constructor
177    // by jumping to REGION_TRANSITION_DISPATCH
178    throw new UnsupportedOperationException();
179  }
180
181  @Override
182  protected boolean updateTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
183        throws IOException {
184    // if the region is already closed or offline we can't do much...
185    if (regionNode.isInState(State.CLOSED, State.OFFLINE)) {
186      LOG.info("Not unassigned " + this + "; " + regionNode.toShortString());
187      return false;
188    }
189
190    // if we haven't started the operation yet, we can abort
191    if (aborted.get() && regionNode.isInState(State.OPEN)) {
192      setAbortFailure(getClass().getSimpleName(), "abort requested");
193      return false;
194    }
195
196    // Mark the region as CLOSING.
197    env.getAssignmentManager().markRegionAsClosing(regionNode);
198
199    // Add the close region operation the the server dispatch queue.
200    if (!addToRemoteDispatcher(env, regionNode.getRegionLocation())) {
201      // If addToRemoteDispatcher fails, it calls the callback #remoteCallFailed.
202    }
203
204    // Return true to keep the procedure running.
205    return true;
206  }
207
208  @Override
209  protected void finishTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
210      throws IOException {
211    AssignmentManager am = env.getAssignmentManager();
212    RegionInfo regionInfo = getRegionInfo();
213
214    if (!removeAfterUnassigning) {
215      am.markRegionAsClosed(regionNode);
216    } else {
217      // Remove from in-memory states
218      am.getRegionStates().deleteRegion(regionInfo);
219      env.getMasterServices().getServerManager().removeRegion(regionInfo);
220      FavoredNodesManager fnm = env.getMasterServices().getFavoredNodesManager();
221      if (fnm != null) {
222        fnm.deleteFavoredNodesForRegions(Lists.newArrayList(regionInfo));
223      }
224    }
225  }
226
227  @Override
228  public RemoteOperation remoteCallBuild(final MasterProcedureEnv env, final ServerName serverName) {
229    assert serverName.equals(getRegionState(env).getRegionLocation());
230    return new RegionCloseOperation(this, getRegionInfo(), this.destinationServer);
231  }
232
233  @Override
234  protected void reportTransition(final MasterProcedureEnv env, final RegionStateNode regionNode,
235      final TransitionCode code, final long seqId) throws UnexpectedStateException {
236    switch (code) {
237      case CLOSED:
238        setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
239        break;
240      default:
241        throw new UnexpectedStateException(String.format(
242          "Received report unexpected transition state=%s for region=%s server=%s, expected CLOSED.",
243          code, regionNode.getRegionInfo(), regionNode.getRegionLocation()));
244    }
245  }
246
247  @Override
248  protected boolean remoteCallFailed(final MasterProcedureEnv env, final RegionStateNode regionNode,
249      final IOException exception) {
250    // TODO: Is there on-going rpc to cleanup?
251    if (exception instanceof ServerCrashException) {
252      // This exception comes from ServerCrashProcedure after log splitting.
253      // SCP found this region as a RIT. Its call into here says it is ok to let this procedure go
254      // on to a complete close now. This will release lock on this region so subsequent action on
255      // region can succeed; e.g. the assign that follows this unassign when a move (w/o wait on SCP
256      // the assign could run w/o logs being split so data loss).
257      try {
258        reportTransition(env, regionNode, TransitionCode.CLOSED, HConstants.NO_SEQNUM);
259      } catch (UnexpectedStateException e) {
260        // Should never happen.
261        throw new RuntimeException(e);
262      }
263    } else if (exception instanceof RegionServerAbortedException ||
264        exception instanceof RegionServerStoppedException ||
265        exception instanceof ServerNotRunningYetException) {
266      // TODO
267      // RS is aborting, we cannot offline the region since the region may need to do WAL
268      // recovery. Until we see the RS expiration, we should retry.
269      // TODO: This should be suspend like the below where we call expire on server?
270      LOG.info("Ignoring; waiting on ServerCrashProcedure", exception);
271    } else if (exception instanceof NotServingRegionException) {
272      LOG.info("IS THIS OK? ANY LOGS TO REPLAY; ACTING AS THOUGH ALL GOOD " + regionNode,
273        exception);
274      setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
275    } else {
276      LOG.warn("Expiring server " + this + "; " + regionNode.toShortString() +
277        ", exception=" + exception);
278      env.getMasterServices().getServerManager().expireServer(regionNode.getRegionLocation());
279      // Return false so this procedure stays in suspended state. It will be woken up by a
280      // ServerCrashProcedure when it notices this RIT.
281      // TODO: Add a SCP as a new subprocedure that we now come to depend on.
282      return false;
283    }
284    return true;
285  }
286
287  @Override
288  public void toStringClassDetails(StringBuilder sb) {
289    super.toStringClassDetails(sb);
290    sb.append(", server=").append(this.hostingServer);
291  }
292
293  @Override
294  public ServerName getServer(final MasterProcedureEnv env) {
295    return this.hostingServer;
296  }
297
298  @Override
299  protected ProcedureMetrics getProcedureMetrics(MasterProcedureEnv env) {
300    return env.getAssignmentManager().getAssignmentManagerMetrics().getUnassignProcMetrics();
301  }
