074  private static final class LocateRequest {
075
076    public final byte[] row;
077
078    public final RegionLocateType locateType;
079
080    public LocateRequest(byte[] row, RegionLocateType locateType) {
081      this.row = row;
082      this.locateType = locateType;
083    }
084
085    @Override
086    public int hashCode() {
087      return Bytes.hashCode(row) ^ locateType.hashCode();
088    }
089
090    @Override
091    public boolean equals(Object obj) {
092      if (obj == null || obj.getClass() != LocateRequest.class) {
093        return false;
094      }
095      LocateRequest that = (LocateRequest) obj;
096      return locateType.equals(that.locateType) && Bytes.equals(row, that.row);
097    }
098  }

100  private static final class TableCache {
101
102    public final ConcurrentNavigableMap<byte[], HRegionLocation> cache =
103      new ConcurrentSkipListMap<>(BYTES_COMPARATOR);
104
105    public final Set<LocateRequest> pendingRequests = new HashSet<>();
106
107    public final Map<LocateRequest, CompletableFuture<HRegionLocation>> allRequests =
108      new LinkedHashMap<>();
109
110    public boolean hasQuota(int max) {
111      return pendingRequests.size() < max;
112    }
113
114    public boolean isPending(LocateRequest req) {
115      return pendingRequests.contains(req);
116    }
117
118    public void send(LocateRequest req) {
119      pendingRequests.add(req);
120    }
121
122    public Optional<LocateRequest> getCandidate() {
123      return allRequests.keySet().stream().filter(r -> !isPending(r)).findFirst();
124    }
125
126    public void clearCompletedRequests(Optional<HRegionLocation> location) {
127      for (Iterator<Map.Entry<LocateRequest, CompletableFuture<HRegionLocation>>> iter =
128        allRequests.entrySet().iterator(); iter.hasNext();) {
129        Map.Entry<LocateRequest, CompletableFuture<HRegionLocation>> entry = iter.next();
130        if (tryComplete(entry.getKey(), entry.getValue(), location)) {
131          iter.remove();
132        }
133      }
134    }
135
136    private boolean tryComplete(LocateRequest req, CompletableFuture<HRegionLocation> future,
137        Optional<HRegionLocation> location) {
138      if (future.isDone()) {
139        return true;
140      }
141      if (!location.isPresent()) {
142        return false;
143      }
144      HRegionLocation loc = location.get();
145      boolean completed;
146      if (req.locateType.equals(RegionLocateType.BEFORE)) {
147        // for locating the row before current row, the common case is to find the previous region
148        // in reverse scan, so we check the endKey first. In general, the condition should be
149        // startKey < req.row and endKey >= req.row. Here we split it to endKey == req.row ||
150        // (endKey > req.row && startKey < req.row). The two conditions are equal since startKey <
151        // endKey.
152        int c = Bytes.compareTo(loc.getRegion().getEndKey(), req.row);
153        completed =
154          c == 0 || (c > 0 && Bytes.compareTo(loc.getRegion().getStartKey(), req.row) < 0);
155      } else {
156        completed = loc.getRegion().containsRow(req.row);
157      }
158      if (completed) {
159        future.complete(loc);
160        return true;
161      } else {
162        return false;
163      }
164    }
165  }

167  AsyncNonMetaRegionLocator(AsyncConnectionImpl conn) {
168    this.conn = conn;
169    this.maxConcurrentLocateRequestPerTable = conn.getConfiguration().getInt(
170      MAX_CONCURRENT_LOCATE_REQUEST_PER_TABLE, DEFAULT_MAX_CONCURRENT_LOCATE_REQUEST_PER_TABLE);
171  }

173  private TableCache getTableCache(TableName tableName) {
174    return computeIfAbsent(cache, tableName, TableCache::new);
175  }
176
177  private void removeFromCache(HRegionLocation loc) {
178    TableCache tableCache = cache.get(loc.getRegion().getTable());
179    if (tableCache == null) {
180      return;
181    }
182    tableCache.cache.computeIfPresent(loc.getRegion().getStartKey(), (k, oldLoc) -> {
183      if (oldLoc.getSeqNum() > loc.getSeqNum() ||
184        !oldLoc.getServerName().equals(loc.getServerName())) {
185        return oldLoc;
186      }
187      return null;
188    });
189  }
190
191  // return whether we add this loc to cache
192  private boolean addToCache(TableCache tableCache, HRegionLocation loc) {
193    if (LOG.isTraceEnabled()) {
194      LOG.trace("Try adding " + loc + " to cache");
195    }
196    byte[] startKey = loc.getRegion().getStartKey();
197    HRegionLocation oldLoc = tableCache.cache.putIfAbsent(startKey, loc);
198    if (oldLoc == null) {
199      return true;
200    }
201    if (oldLoc.getSeqNum() > loc.getSeqNum() ||
202      oldLoc.getServerName().equals(loc.getServerName())) {
203      if (LOG.isTraceEnabled()) {
204        LOG.trace("Will not add " + loc + " to cache because the old value " + oldLoc +
205          " is newer than us or has the same server name");
206      }
207      return false;
208    }
209    return loc == tableCache.cache.compute(startKey, (k, oldValue) -> {
210      if (oldValue == null || oldValue.getSeqNum() <= loc.getSeqNum()) {
211        return loc;
212      }
213      if (LOG.isTraceEnabled()) {
214        LOG.trace("Will not add " + loc + " to cache because the old value " + oldValue +
215          " is newer than us or has the same server name." +
216          " Maybe it is updated before we replace it");
217      }
218      return oldValue;
219    });
220  }
221
222  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "UPM_UNCALLED_PRIVATE_METHOD",
223      justification = "Called by lambda expression")
224  private void addToCache(HRegionLocation loc) {
225    addToCache(getTableCache(loc.getRegion().getTable()), loc);
226    if (LOG.isTraceEnabled()) {
227      LOG.trace("Try adding " + loc + " to cache");
228    }
229  }

231  private void complete(TableName tableName, LocateRequest req, HRegionLocation loc,
232      Throwable error) {
233    if (error != null) {
234      LOG.warn("Failed to locate region in '" + tableName + "', row='" +
235        Bytes.toStringBinary(req.row) + "', locateType=" + req.locateType, error);
236    }
237    Optional<LocateRequest> toSend = Optional.empty();
238    TableCache tableCache = getTableCache(tableName);
239    if (loc != null) {
240      if (!addToCache(tableCache, loc)) {
241        // someone is ahead of us.
242        synchronized (tableCache) {
243          tableCache.pendingRequests.remove(req);
244          tableCache.clearCompletedRequests(Optional.empty());
245          // Remove a complete locate request in a synchronized block, so the table cache must have
246          // quota to send a candidate request.
247          toSend = tableCache.getCandidate();
248          toSend.ifPresent(r -> tableCache.send(r));
249        }
250        toSend.ifPresent(r -> locateInMeta(tableName, r));
251        return;
252      }
253    }
254    synchronized (tableCache) {
255      tableCache.pendingRequests.remove(req);
256      if (error instanceof DoNotRetryIOException) {
257        CompletableFuture<?> future = tableCache.allRequests.remove(req);
258        if (future != null) {
259          future.completeExceptionally(error);
260        }
261      }
262      tableCache.clearCompletedRequests(Optional.ofNullable(loc));
263      // Remove a complete locate request in a synchronized block, so the table cache must have
264      // quota to send a candidate request.
265      toSend = tableCache.getCandidate();
266      toSend.ifPresent(r -> tableCache.send(r));
267    }
268    toSend.ifPresent(r -> locateInMeta(tableName, r));
269  }

271  // return whether we should stop the scan
272  private boolean onScanNext(TableName tableName, LocateRequest req, Result result) {
273    RegionLocations locs = MetaTableAccessor.getRegionLocations(result);
274    LOG.debug("The fetched location of '{}', row='{}', locateType={} is {}", tableName,
275      Bytes.toStringBinary(req.row), req.locateType, locs);
276
277    if (locs == null || locs.getDefaultRegionLocation() == null) {
278      complete(tableName, req, null,
279        new IOException(String.format("No location found for '%s', row='%s', locateType=%s",
280          tableName, Bytes.toStringBinary(req.row), req.locateType)));
281      return true;
282    }
283    HRegionLocation loc = locs.getDefaultRegionLocation();
284    RegionInfo info = loc.getRegion();
285    if (info == null) {
286      complete(tableName, req, null,
287        new IOException(String.format("HRegionInfo is null for '%s', row='%s', locateType=%s",
288          tableName, Bytes.toStringBinary(req.row), req.locateType)));
289      return true;
290    }
291    if (info.isSplitParent()) {
292      return false;
293    }
294    if (loc.getServerName() == null) {
295      complete(tableName, req, null,
296        new IOException(
297            String.format("No server address listed for region '%s', row='%s', locateType=%s",
298              info.getRegionNameAsString(), Bytes.toStringBinary(req.row), req.locateType)));
299      return true;
300    }
301    complete(tableName, req, loc, null);
302    return true;
303  }

305  private HRegionLocation locateRowInCache(TableCache tableCache, TableName tableName, byte[] row) {
306    Map.Entry<byte[], HRegionLocation> entry = tableCache.cache.floorEntry(row);
307    if (entry == null) {
308      return null;
309    }
310    HRegionLocation loc = entry.getValue();
311    byte[] endKey = loc.getRegion().getEndKey();
312    if (isEmptyStopRow(endKey) || Bytes.compareTo(row, endKey) < 0) {
313      if (LOG.isTraceEnabled()) {
314        LOG.trace("Found " + loc + " in cache for '" + tableName + "', row='" +
315          Bytes.toStringBinary(row) + "', locateType=" + RegionLocateType.CURRENT);
316      }
317      return loc;
318    } else {
319      return null;
320    }
321  }
322
323  private HRegionLocation locateRowBeforeInCache(TableCache tableCache, TableName tableName,
324      byte[] row) {
325    Map.Entry<byte[], HRegionLocation> entry =
326      isEmptyStopRow(row) ? tableCache.cache.lastEntry() : tableCache.cache.lowerEntry(row);
327    if (entry == null) {
328      return null;
329    }
330    HRegionLocation loc = entry.getValue();
331    if (isEmptyStopRow(loc.getRegion().getEndKey()) ||
332      Bytes.compareTo(loc.getRegion().getEndKey(), row) >= 0) {
333      if (LOG.isTraceEnabled()) {
334        LOG.trace("Found " + loc + " in cache for '" + tableName + "', row='" +
335          Bytes.toStringBinary(row) + "', locateType=" + RegionLocateType.BEFORE);
336      }
337      return loc;
338    } else {
339      return null;
340    }
341  }

343  private void locateInMeta(TableName tableName, LocateRequest req) {
344    if (LOG.isTraceEnabled()) {
345      LOG.trace("Try locate '" + tableName + "', row='" + Bytes.toStringBinary(req.row) +
346        "', locateType=" + req.locateType + " in meta");
347    }
348    byte[] metaStartKey;
349    if (req.locateType.equals(RegionLocateType.BEFORE)) {
350      if (isEmptyStopRow(req.row)) {
351        byte[] binaryTableName = tableName.getName();
352        metaStartKey = Arrays.copyOf(binaryTableName, binaryTableName.length + 1);
353      } else {
354        metaStartKey = createRegionName(tableName, req.row, ZEROES, false);
355      }
356    } else {
357      metaStartKey = createRegionName(tableName, req.row, NINES, false);
358    }
359    byte[] metaStopKey =
360      RegionInfo.createRegionName(tableName, HConstants.EMPTY_START_ROW, "", false);
361    conn.getTable(META_TABLE_NAME)
362      .scan(new Scan().withStartRow(metaStartKey).withStopRow(metaStopKey, true)
363        .addFamily(HConstants.CATALOG_FAMILY).setReversed(true).setCaching(5)
364        .setReadType(ReadType.PREAD), new AdvancedScanResultConsumer() {
365
366          private boolean completeNormally = false;
367
368          private boolean tableNotFound = true;
369
370          @Override
371          public void onError(Throwable error) {
372            complete(tableName, req, null, error);
373          }
374
375          @Override
376          public void onComplete() {
377            if (tableNotFound) {
378              complete(tableName, req, null, new TableNotFoundException(tableName));
379            } else if (!completeNormally) {
380              complete(tableName, req, null, new IOException(
381                "Unable to find region for " + Bytes.toStringBinary(req.row) + " in " + tableName));
382            }
383          }
384
385          @Override
386          public void onNext(Result[] results, ScanController controller) {
387            for (Result result : results) {
388              tableNotFound = false;
389              if (onScanNext(tableName, req, result)) {
390                completeNormally = true;
391                controller.terminate();
392                return;
393              }
394            }
395          }
396        });
397  }

399  private HRegionLocation locateInCache(TableCache tableCache, TableName tableName, byte[] row,
400      RegionLocateType locateType) {
401    return locateType.equals(RegionLocateType.BEFORE)
402      ? locateRowBeforeInCache(tableCache, tableName, row)
403      : locateRowInCache(tableCache, tableName, row);
404  }

406  // locateToPrevious is true means we will use the start key of a region to locate the region
407  // placed before it. Used for reverse scan. See the comment of
408  // AsyncRegionLocator.getPreviousRegionLocation.
409  private CompletableFuture<HRegionLocation> getRegionLocationInternal(TableName tableName,
410      byte[] row, RegionLocateType locateType, boolean reload) {
411    // AFTER should be convert to CURRENT before calling this method
412    assert !locateType.equals(RegionLocateType.AFTER);
413    TableCache tableCache = getTableCache(tableName);
414    if (!reload) {
415      HRegionLocation loc = locateInCache(tableCache, tableName, row, locateType);
416      if (loc != null) {
417        return CompletableFuture.completedFuture(loc);
418      }
419    }
420    CompletableFuture<HRegionLocation> future;
421    LocateRequest req;
422    boolean sendRequest = false;
423    synchronized (tableCache) {
424      // check again
425      if (!reload) {
426        HRegionLocation loc = locateInCache(tableCache, tableName, row, locateType);
427        if (loc != null) {
428          return CompletableFuture.completedFuture(loc);
429        }
430      }
431      req = new LocateRequest(row, locateType);
432      future = tableCache.allRequests.get(req);
433      if (future == null) {
434        future = new CompletableFuture<>();
435        tableCache.allRequests.put(req, future);
436        if (tableCache.hasQuota(maxConcurrentLocateRequestPerTable) && !tableCache.isPending(req)) {
437          tableCache.send(req);
438          sendRequest = true;
439        }
440      }
441    }
442    if (sendRequest) {
443      locateInMeta(tableName, req);
444    }
445    return future;
446  }
447
448  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
449      RegionLocateType locateType, boolean reload) {
450    if (locateType.equals(RegionLocateType.BEFORE)) {
451      return getRegionLocationInternal(tableName, row, locateType, reload);
452    } else {
453      // as we know the exact row after us, so we can just create the new row, and use the same
454      // algorithm to locate it.
455      if (locateType.equals(RegionLocateType.AFTER)) {
456        row = createClosestRowAfter(row);
457      }
458      return getRegionLocationInternal(tableName, row, RegionLocateType.CURRENT, reload);
459    }
460  }

462  void updateCachedLocation(HRegionLocation loc, Throwable exception) {
463    AsyncRegionLocator.updateCachedLocation(loc, exception, l -> {
464      TableCache tableCache = cache.get(l.getRegion().getTable());
465      if (tableCache == null) {
466        return null;
467      }
468      return tableCache.cache.get(l.getRegion().getStartKey());
469    }, this::addToCache, this::removeFromCache);
470  }
471
472  void clearCache(TableName tableName) {
473    TableCache tableCache = cache.remove(tableName);
474    if (tableCache == null) {
475      return;
476    }
477    synchronized (tableCache) {
478      if (!tableCache.allRequests.isEmpty()) {
479        IOException error = new IOException("Cache cleared");
480        tableCache.allRequests.values().forEach(f -> f.completeExceptionally(error));
481      }
482    }
483  }
