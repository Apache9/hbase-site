062/**
063 * This client class is for invoking the aggregate functions deployed on the
064 * Region Server side via the AggregateService. This class will implement the
065 * supporting functionality for summing/processing the individual results
066 * obtained from the AggregateService for each region.
067 * <p>
068 * This will serve as the client side handler for invoking the aggregate
069 * functions.
070 * For all aggregate functions,
071 * <ul>
072 * <li>start row &lt; end row is an essential condition (if they are not
073 * {@link HConstants#EMPTY_BYTE_ARRAY})
074 * <li>Column family can't be null. In case where multiple families are
075 * provided, an IOException will be thrown. An optional column qualifier can
076 * also be defined.</li>
077 * <li>For methods to find maximum, minimum, sum, rowcount, it returns the
078 * parameter type. For average and std, it returns a double value. For row
079 * count, it returns a long value.</li>
080 * </ul>
081 * <p>Call {@link #close()} when done.
082 */
083@InterfaceAudience.Public
084public class AggregationClient implements Closeable {

176  /**
177   * It gives the maximum value of a column for a given column family for the
178   * given range. In case qualifier is null, a max of all values for the given
179   * family is returned.
180   * @param table
181   * @param ci
182   * @param scan
183   * @return max val &lt;&gt;
184   * @throws Throwable
185   *           The caller is supposed to handle the exception as they are thrown
186   *           &amp; propagated to it.
187   */
188  public <R, S, P extends Message, Q extends Message, T extends Message>
189  R max(final Table table, final ColumnInterpreter<R, S, P, Q, T> ci,
190      final Scan scan) throws Throwable {
191    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
192    class MaxCallBack implements Batch.Callback<R> {
193      R max = null;
194
195      R getMax() {
196        return max;
197      }
198
199      @Override
200      public synchronized void update(byte[] region, byte[] row, R result) {
201        max = (max == null || (result != null && ci.compare(max, result) < 0)) ? result : max;
202      }
203    }
204    MaxCallBack aMaxCallBack = new MaxCallBack();
205    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
206        new Batch.Call<AggregateService, R>() {
207          @Override
208          public R call(AggregateService instance) throws IOException {
209            RpcController controller = new AggregationClientRpcController();
210            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
211                new CoprocessorRpcUtils.BlockingRpcCallback<>();
212            instance.getMax(controller, requestArg, rpcCallback);
213            AggregateResponse response = rpcCallback.get();
214            if (controller.failed()) {
215              throw new IOException(controller.errorText());
216            }
217            if (response.getFirstPartCount() > 0) {
218              ByteString b = response.getFirstPart(0);
219              Q q = getParsedGenericInstance(ci.getClass(), 3, b);
220              return ci.getCellValueFromProto(q);
221            }
222            return null;
223          }
224        }, aMaxCallBack);
225    return aMaxCallBack.getMax();
226  }

458  /**
459   * It computes average while fetching sum and row count from all the
460   * corresponding regions. Approach is to compute a global sum of region level
461   * sum and rowcount and then compute the average.
462   * @param table
463   * @param scan
464   * @throws Throwable
465   */
466  private <R, S, P extends Message, Q extends Message, T extends Message>
467  Pair<S, Long> getAvgArgs(final Table table,
468      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
469    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
470    class AvgCallBack implements Batch.Callback<Pair<S, Long>> {
471      S sum = null;
472      Long rowCount = 0l;
473
474      public synchronized Pair<S, Long> getAvgArgs() {
475        return new Pair<>(sum, rowCount);
476      }
477
478      @Override
479      public synchronized void update(byte[] region, byte[] row, Pair<S, Long> result) {
480        sum = ci.add(sum, result.getFirst());
481        rowCount += result.getSecond();
482      }
483    }
484    AvgCallBack avgCallBack = new AvgCallBack();
485    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
486        new Batch.Call<AggregateService, Pair<S, Long>>() {
487          @Override
488          public Pair<S, Long> call(AggregateService instance) throws IOException {
489            RpcController controller = new AggregationClientRpcController();
490            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
491                new CoprocessorRpcUtils.BlockingRpcCallback<>();
492            instance.getAvg(controller, requestArg, rpcCallback);
493            AggregateResponse response = rpcCallback.get();
494            if (controller.failed()) {
495              throw new IOException(controller.errorText());
496            }
497            Pair<S, Long> pair = new Pair<>(null, 0L);
498            if (response.getFirstPartCount() == 0) {
499              return pair;
500            }
501            ByteString b = response.getFirstPart(0);
502            T t = getParsedGenericInstance(ci.getClass(), 4, b);
503            S s = ci.getPromotedValueFromProto(t);
504            pair.setFirst(s);
505            ByteBuffer bb = ByteBuffer.allocate(8).put(
506                getBytesFromResponse(response.getSecondPart()));
507            bb.rewind();
508            pair.setSecond(bb.getLong());
509            return pair;
510          }
511        }, avgCallBack);
512    return avgCallBack.getAvgArgs();
513  }

552  /**
553   * It computes a global standard deviation for a given column and its value.
554   * Standard deviation is square root of (average of squares -
555   * average*average). From individual regions, it obtains sum, square sum and
556   * number of rows. With these, the above values are computed to get the global
557   * std.
558   * @param table
559   * @param scan
560   * @return standard deviations
561   * @throws Throwable
562   */
563  private <R, S, P extends Message, Q extends Message, T extends Message>
564  Pair<List<S>, Long> getStdArgs(final Table table,
565      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
566    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
567    class StdCallback implements Batch.Callback<Pair<List<S>, Long>> {
568      long rowCountVal = 0l;
569      S sumVal = null, sumSqVal = null;
570
571      public synchronized Pair<List<S>, Long> getStdParams() {
572        List<S> l = new ArrayList<>(2);
573        l.add(sumVal);
574        l.add(sumSqVal);
575        Pair<List<S>, Long> p = new Pair<>(l, rowCountVal);
576        return p;
577      }
578
579      @Override
580      public synchronized void update(byte[] region, byte[] row, Pair<List<S>, Long> result) {
581        if (result.getFirst().size() > 0) {
582          sumVal = ci.add(sumVal, result.getFirst().get(0));
583          sumSqVal = ci.add(sumSqVal, result.getFirst().get(1));
584          rowCountVal += result.getSecond();
585        }
586      }
587    }
588    StdCallback stdCallback = new StdCallback();
589    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
590        new Batch.Call<AggregateService, Pair<List<S>, Long>>() {
591          @Override
592          public Pair<List<S>, Long> call(AggregateService instance) throws IOException {
593            RpcController controller = new AggregationClientRpcController();
594            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
595                new CoprocessorRpcUtils.BlockingRpcCallback<>();
596            instance.getStd(controller, requestArg, rpcCallback);
597            AggregateResponse response = rpcCallback.get();
598            if (controller.failed()) {
599              throw new IOException(controller.errorText());
600            }
601            Pair<List<S>, Long> pair = new Pair<>(new ArrayList<>(), 0L);
602            if (response.getFirstPartCount() == 0) {
603              return pair;
604            }
605            List<S> list = new ArrayList<>();
606            for (int i = 0; i < response.getFirstPartCount(); i++) {
607              ByteString b = response.getFirstPart(i);
608              T t = getParsedGenericInstance(ci.getClass(), 4, b);
609              S s = ci.getPromotedValueFromProto(t);
610              list.add(s);
611            }
612            pair.setFirst(list);
613            ByteBuffer bb = ByteBuffer.allocate(8).put(
614                getBytesFromResponse(response.getSecondPart()));
615            bb.rewind();
616            pair.setSecond(bb.getLong());
617            return pair;
618          }
619        }, stdCallback);
620    return stdCallback.getStdParams();
621  }

643  /**
644   * This is the client side interface/handle for calling the std method for a
645   * given cf-cq combination. It was necessary to add one more call stack as its
646   * return type should be a decimal value, irrespective of what
647   * columninterpreter says. So, this methods collects the necessary parameters
648   * to compute the std and returns the double value.
649   * @param table
650   * @param ci
651   * @param scan
652   * @return &lt;R, S&gt;
653   * @throws Throwable
654   */
655  public <R, S, P extends Message, Q extends Message, T extends Message> double std(
656      final Table table, ColumnInterpreter<R, S, P, Q, T> ci, Scan scan) throws Throwable {
657    Pair<List<S>, Long> p = getStdArgs(table, ci, scan);
658    double res = 0d;
659    double avg = ci.divideForAvg(p.getFirst().get(0), p.getSecond());
660    double avgOfSumSq = ci.divideForAvg(p.getFirst().get(1), p.getSecond());
661    res = avgOfSumSq - (avg) * (avg); // variance
662    res = Math.pow(res, 0.5);
663    return res;
664  }

666  /**
667   * It helps locate the region with median for a given column whose weight
668   * is specified in an optional column.
669   * From individual regions, it obtains sum of values and sum of weights.
670   * @param table
671   * @param ci
672   * @param scan
673   * @return pair whose first element is a map between start row of the region
674   *  and (sum of values, sum of weights) for the region, the second element is
675   *  (sum of values, sum of weights) for all the regions chosen
676   * @throws Throwable
677   */
678  private <R, S, P extends Message, Q extends Message, T extends Message>
679  Pair<NavigableMap<byte[], List<S>>, List<S>>
680  getMedianArgs(final Table table,
681      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
682    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
683    final NavigableMap<byte[], List<S>> map = new TreeMap<>(Bytes.BYTES_COMPARATOR);
684    class StdCallback implements Batch.Callback<List<S>> {
685      S sumVal = null, sumWeights = null;
686
687      public synchronized Pair<NavigableMap<byte[], List<S>>, List<S>> getMedianParams() {
688        List<S> l = new ArrayList<>(2);
689        l.add(sumVal);
690        l.add(sumWeights);
691        Pair<NavigableMap<byte[], List<S>>, List<S>> p = new Pair<>(map, l);
692        return p;
693      }
694
695      @Override
696      public synchronized void update(byte[] region, byte[] row, List<S> result) {
697        map.put(row, result);
698        sumVal = ci.add(sumVal, result.get(0));
699        sumWeights = ci.add(sumWeights, result.get(1));
700      }
701    }
702    StdCallback stdCallback = new StdCallback();
703    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
704        new Batch.Call<AggregateService, List<S>>() {
705          @Override
706          public List<S> call(AggregateService instance) throws IOException {
707            RpcController controller = new AggregationClientRpcController();
708            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
709                new CoprocessorRpcUtils.BlockingRpcCallback<>();
710            instance.getMedian(controller, requestArg, rpcCallback);
711            AggregateResponse response = rpcCallback.get();
712            if (controller.failed()) {
713              throw new IOException(controller.errorText());
714            }
715
716            List<S> list = new ArrayList<>();
717            for (int i = 0; i < response.getFirstPartCount(); i++) {
718              ByteString b = response.getFirstPart(i);
719              T t = getParsedGenericInstance(ci.getClass(), 4, b);
720              S s = ci.getPromotedValueFromProto(t);
721              list.add(s);
722            }
723            return list;
724          }
725
726        }, stdCallback);
727    return stdCallback.getMedianParams();
728  }

748  /**
749   * This is the client side interface/handler for calling the median method for a
750   * given cf-cq combination. This method collects the necessary parameters
751   * to compute the median and returns the median.
752   * @param table
753   * @param ci
754   * @param scan
755   * @return R the median
756   * @throws Throwable
757   */
758  public <R, S, P extends Message, Q extends Message, T extends Message>
759  R median(final Table table, ColumnInterpreter<R, S, P, Q, T> ci,
760      Scan scan) throws Throwable {
761    Pair<NavigableMap<byte[], List<S>>, List<S>> p = getMedianArgs(table, ci, scan);
762    byte[] startRow = null;
763    byte[] colFamily = scan.getFamilies()[0];
764    NavigableSet<byte[]> quals = scan.getFamilyMap().get(colFamily);
765    NavigableMap<byte[], List<S>> map = p.getFirst();
766    S sumVal = p.getSecond().get(0);
767    S sumWeights = p.getSecond().get(1);
768    double halfSumVal = ci.divideForAvg(sumVal, 2L);
769    double movingSumVal = 0;
770    boolean weighted = false;
771    if (quals.size() > 1) {
772      weighted = true;
773      halfSumVal = ci.divideForAvg(sumWeights, 2L);
774    }
775
776    for (Map.Entry<byte[], List<S>> entry : map.entrySet()) {
777      S s = weighted ? entry.getValue().get(1) : entry.getValue().get(0);
778      double newSumVal = movingSumVal + ci.divideForAvg(s, 1L);
779      if (newSumVal > halfSumVal) break;  // we found the region with the median
780      movingSumVal = newSumVal;
781      startRow = entry.getKey();
782    }
783    // scan the region with median and find it
784    Scan scan2 = new Scan(scan);
785    // inherit stop row from method parameter
786    if (startRow != null) scan2.setStartRow(startRow);
787    ResultScanner scanner = null;
788    try {
789      int cacheSize = scan2.getCaching();
790      if (!scan2.getCacheBlocks() || scan2.getCaching() < 2) {
791        scan2.setCacheBlocks(true);
792        cacheSize = 5;
793        scan2.setCaching(cacheSize);
794      }
795      scanner = table.getScanner(scan2);
796      Result[] results = null;
797      byte[] qualifier = quals.pollFirst();
798      // qualifier for the weight column
799      byte[] weightQualifier = weighted ? quals.pollLast() : qualifier;
800      R value = null;
801      do {
802        results = scanner.next(cacheSize);
803        if (results != null && results.length > 0) {
804          for (int i = 0; i < results.length; i++) {
805            Result r = results[i];
806            // retrieve weight
807            Cell kv = r.getColumnLatestCell(colFamily, weightQualifier);
808            R newValue = ci.getValue(colFamily, weightQualifier, kv);
809            S s = ci.castToReturnType(newValue);
810            double newSumVal = movingSumVal + ci.divideForAvg(s, 1L);
811            // see if we have moved past the median
812            if (newSumVal > halfSumVal) {
813              return value;
814            }
815            movingSumVal = newSumVal;
816            kv = r.getColumnLatestCell(colFamily, qualifier);
817            value = ci.getValue(colFamily, qualifier, kv);
818            }
819          }
820      } while (results != null && results.length > 0);
821    } finally {
822      if (scanner != null) {
823        scanner.close();
824      }
825    }
826    return null;
827  }
