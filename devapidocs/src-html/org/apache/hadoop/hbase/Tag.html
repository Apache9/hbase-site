020package org.apache.hadoop.hbase;
021
022import java.util.ArrayList;
023import java.util.List;
024
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026import org.apache.hadoop.hbase.classification.InterfaceStability;
027import org.apache.hadoop.hbase.util.Bytes;
028/**
029 * Tags are part of cells and helps to add metadata about the KVs.
030 * Metadata could be ACLs per cells, visibility labels, etc.
031 */
032@InterfaceAudience.Private
033@InterfaceStability.Evolving
034public class Tag {
035  public final static int TYPE_LENGTH_SIZE = Bytes.SIZEOF_BYTE;
036  public final static int TAG_LENGTH_SIZE = Bytes.SIZEOF_SHORT;
037  public final static int INFRASTRUCTURE_SIZE = TYPE_LENGTH_SIZE + TAG_LENGTH_SIZE;
038  public static final int MAX_TAG_LENGTH = (2 * Short.MAX_VALUE) + 1 - TAG_LENGTH_SIZE;
039
040  private final byte type;
041  private final byte[] bytes;
042  private int offset = 0;
043  private int length = 0;
044
045  /**
046   * The special tag will write the length of each tag and that will be
047   * followed by the type and then the actual tag.
048   * So every time the length part is parsed we need to add + 1 byte to it to
049   * get the type and then get the actual tag.
050   */
051  public Tag(byte tagType, String tag) {
052    this(tagType, Bytes.toBytes(tag));
053  }
054
055  /**
056   * Format for a tag :
057   * {@code <length of tag - 2 bytes><type code - 1 byte><tag>} tag length is serialized
058   * using 2 bytes only but as this will be unsigned, we can have max tag length of
059   * (Short.MAX_SIZE * 2) +1. It includes 1 byte type length and actual tag bytes length.
060   */
061  public Tag(byte tagType, byte[] tag) {
062    int tagLength = tag.length + TYPE_LENGTH_SIZE;
063    if (tagLength > MAX_TAG_LENGTH) {
064      throw new IllegalArgumentException(
065          "Invalid tag data being passed. Its length can not exceed " + MAX_TAG_LENGTH);
066    }
067    length = TAG_LENGTH_SIZE + tagLength;
068    bytes = new byte[length];
069    int pos = Bytes.putAsShort(bytes, 0, tagLength);
070    pos = Bytes.putByte(bytes, pos, tagType);
071    Bytes.putBytes(bytes, pos, tag, 0, tag.length);
072    this.type = tagType;
073  }
074
075  /**
076   * Creates a Tag from the specified byte array and offset. Presumes
077   * <code>bytes</code> content starting at <code>offset</code> is formatted as
078   * a Tag blob.
079   * The bytes to include the tag type, tag length and actual tag bytes.
080   * @param offset offset to start of Tag
081   */
082  public Tag(byte[] bytes, int offset) {
083    this(bytes, offset, getLength(bytes, offset));
084  }
085
086  private static int getLength(byte[] bytes, int offset) {
087    return TAG_LENGTH_SIZE + Bytes.readAsInt(bytes, offset, TAG_LENGTH_SIZE);
088  }
089
090  /**
091   * Creates a Tag from the specified byte array, starting at offset, and for length
092   * <code>length</code>. Presumes <code>bytes</code> content starting at <code>offset</code> is
093   * formatted as a Tag blob.
094   */
095  public Tag(byte[] bytes, int offset, int length) {
096    if (length > MAX_TAG_LENGTH) {
097      throw new IllegalArgumentException(
098          "Invalid tag data being passed. Its length can not exceed " + MAX_TAG_LENGTH);
099    }
100    this.bytes = bytes;
101    this.offset = offset;
102    this.length = length;
103    this.type = bytes[offset + TAG_LENGTH_SIZE];
104  }
105
106  /**
107   * @return The byte array backing this Tag.
108   */
109  public byte[] getBuffer() {
110    return this.bytes;
111  }
112
113  /**
114   * @return the tag type
115   */
116  public byte getType() {
117    return this.type;
118  }
119
120  /**
121   * @return Length of actual tag bytes within the backed buffer
122   */
123  public int getTagLength() {
124    return this.length - INFRASTRUCTURE_SIZE;
125  }
126
127  /**
128   * @return Offset of actual tag bytes within the backed buffer
129   */
130  public int getTagOffset() {
131    return this.offset + INFRASTRUCTURE_SIZE;
132  }
133
134  /**
135   * Returns tag value in a new byte array.
136   * Primarily for use client-side. If server-side, use
137   * {@link #getBuffer()} with appropriate {@link #getTagOffset()} and {@link #getTagLength()}
138   * instead to save on allocations.
139   * @return tag value in a new byte array.
140   */
141  public byte[] getValue() {
142    int tagLength = getTagLength();
143    byte[] tag = new byte[tagLength];
144    Bytes.putBytes(tag, 0, bytes, getTagOffset(), tagLength);
145    return tag;
146  }
147
148  /**
149   * Creates the list of tags from the byte array b. Expected that b is in the
150   * expected tag format
151   * @param b
152   * @param offset
153   * @param length
154   * @return List of tags
155   */
156  public static List<Tag> asList(byte[] b, int offset, int length) {
157    List<Tag> tags = new ArrayList<Tag>();
158    int pos = offset;
159    while (pos < offset + length) {
160      int tagLen = Bytes.readAsInt(b, pos, TAG_LENGTH_SIZE);
161      tags.add(new Tag(b, pos, tagLen + TAG_LENGTH_SIZE));
162      pos += TAG_LENGTH_SIZE + tagLen;
163    }
164    return tags;
165  }
166
167  /**
168   * Write a list of tags into a byte array
169   * @param tags
170   * @return the serialized tag data as bytes
171   */
172  public static byte[] fromList(List<Tag> tags) {
173    int length = 0;
174    for (Tag tag: tags) {
175      length += tag.length;
176    }
177    byte[] b = new byte[length];
178    int pos = 0;
179    for (Tag tag: tags) {
180      System.arraycopy(tag.bytes, tag.offset, b, pos, tag.length);
181      pos += tag.length;
182    }
183    return b;
184  }
185
186  /**
187   * Retrieve the first tag from the tags byte array matching the passed in tag type
188   * @param b
189   * @param offset
190   * @param length
191   * @param type
192   * @return null if there is no tag of the passed in tag type
193   */
194  public static Tag getTag(byte[] b, int offset, int length, byte type) {
195    int pos = offset;
196    while (pos < offset + length) {
197      int tagLen = Bytes.readAsInt(b, pos, TAG_LENGTH_SIZE);
198      if(b[pos + TAG_LENGTH_SIZE] == type) {
199        return new Tag(b, pos, tagLen + TAG_LENGTH_SIZE);
200      }
201      pos += TAG_LENGTH_SIZE + tagLen;
202    }
203    return null;
204  }
205
206  /**
207   * Returns the total length of the entire tag entity
208   */
209  int getLength() {
210    return this.length;
211  }
212
213  /**
214   * Returns the offset of the entire tag entity
215   */
216  int getOffset() {
217    return this.offset;
218  }
219}
