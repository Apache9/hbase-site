020package org.apache.hadoop.hbase;
021
022import java.nio.ByteBuffer;
023import java.util.List;
024
025import org.apache.hadoop.hbase.util.ByteBufferUtils;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.yetus.audience.InterfaceStability;
029
030/**
031 * Tags are part of cells and helps to add metadata about them.
032 * Metadata could be ACLs, visibility labels, etc.
033 * <p>
034 * Each Tag is having a type (one byte) and value part. The max value length for a Tag is 65533.
035 * <p>
036 * See {@link TagType} for reserved tag types.
037 */
038@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
039@InterfaceStability.Evolving
040public interface Tag {
041
042  public final static int TYPE_LENGTH_SIZE = Bytes.SIZEOF_BYTE;
043  public final static int TAG_LENGTH_SIZE = Bytes.SIZEOF_SHORT;
044  public final static int INFRASTRUCTURE_SIZE = TYPE_LENGTH_SIZE + TAG_LENGTH_SIZE;
045  public static final int MAX_TAG_LENGTH = (2 * Short.MAX_VALUE) + 1 - TAG_LENGTH_SIZE;
046
047  /**
048   * Custom tags if created are suggested to be above this range. So that
049   * it does not overlap with internal tag types
050   */
051  public static final byte CUSTOM_TAG_TYPE_RANGE = (byte)64;
052  /**
053   * @return the tag type
054   */
055  byte getType();
056
057  /**
058   * @return Offset of tag value within the backed buffer
059   */
060  int getValueOffset();
061
062  /**
063   * @return Length of tag value within the backed buffer
064   */
065  int getValueLength();
066
067  /**
068   * Tells whether or not this Tag is backed by a byte array.
069   * @return true when this Tag is backed by byte array
070   */
071  boolean hasArray();
072
073  /**
074   * @return The array containing the value bytes.
075   * @throws UnsupportedOperationException
076   *           when {@link #hasArray()} return false. Use {@link #getValueByteBuffer()} in such
077   *           situation
078   */
079  byte[] getValueArray();
080
081  /**
082   * @return The {@link java.nio.ByteBuffer} containing the value bytes.
083   */
084  ByteBuffer getValueByteBuffer();
085
086  /**
087   * Returns tag value in a new byte array. Primarily for use client-side. If server-side, use
088   * {@link Tag#getValueArray()} with appropriate {@link Tag#getValueOffset()} and
089   * {@link Tag#getValueLength()} instead to save on allocations.
090   * @param tag The Tag whose value to be returned
091   * @return tag value in a new byte array.
092   */
093  public static byte[] cloneValue(Tag tag) {
094    int tagLength = tag.getValueLength();
095    byte[] tagArr = new byte[tagLength];
096    if (tag.hasArray()) {
097      Bytes.putBytes(tagArr, 0, tag.getValueArray(), tag.getValueOffset(), tagLength);
098    } else {
099      ByteBufferUtils.copyFromBufferToArray(tagArr, tag.getValueByteBuffer(), tag.getValueOffset(),
100        0, tagLength);
101    }
102    return tagArr;
103  }
104
105  /**
106   * Converts the value bytes of the given tag into a String value
107   * @param tag The Tag
108   * @return value as String
109   */
110  public static String getValueAsString(Tag tag) {
111    if (tag.hasArray()) {
112      return Bytes.toString(tag.getValueArray(), tag.getValueOffset(), tag.getValueLength());
113    }
114    return Bytes.toString(cloneValue(tag));
115  }
116
117  /**
118   * Matches the value part of given tags
119   * @param t1 Tag to match the value
120   * @param t2 Tag to match the value
121   * @return True if values of both tags are same.
122   */
123  public static boolean matchingValue(Tag t1, Tag t2) {
124    if (t1.hasArray() && t2.hasArray()) {
125      return Bytes.equals(t1.getValueArray(), t1.getValueOffset(), t1.getValueLength(),
126        t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
127    }
128    if (t1.hasArray()) {
129      return ByteBufferUtils.equals(t2.getValueByteBuffer(), t2.getValueOffset(),
130        t2.getValueLength(), t1.getValueArray(), t1.getValueOffset(), t1.getValueLength());
131    }
132    if (t2.hasArray()) {
133      return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(),
134        t1.getValueLength(), t2.getValueArray(), t2.getValueOffset(), t2.getValueLength());
135    }
136    return ByteBufferUtils.equals(t1.getValueByteBuffer(), t1.getValueOffset(), t1.getValueLength(),
137      t2.getValueByteBuffer(), t2.getValueOffset(), t2.getValueLength());
138  }
139
140  /**
141   * Copies the tag's value bytes to the given byte array
142   * @param tag The Tag
143   * @param out The byte array where to copy the Tag value.
144   * @param offset The offset within 'out' array where to copy the Tag value.
145   */
146  public static void copyValueTo(Tag tag, byte[] out, int offset) {
147    if (tag.hasArray()) {
148      Bytes.putBytes(out, offset, tag.getValueArray(), tag.getValueOffset(), tag.getValueLength());
149    } else {
150      ByteBufferUtils.copyFromBufferToArray(out, tag.getValueByteBuffer(), tag.getValueOffset(),
151        offset, tag.getValueLength());
152    }
153  }
154
155  /**
156   * Write a list of tags into a byte array
157   * @param tags The list of tags
158   * @return the serialized tag data as bytes
159   */
160  // TODO : Remove this when we move to RawCellBuilder
161  public static byte[] fromList(List<Tag> tags) {
162    if (tags == null || tags.isEmpty()) {
163      return HConstants.EMPTY_BYTE_ARRAY;
164    }
165    int length = 0;
166    for (Tag tag : tags) {
167      length += tag.getValueLength() + Tag.INFRASTRUCTURE_SIZE;
168    }
169    byte[] b = new byte[length];
170    int pos = 0;
171    int tlen;
172    for (Tag tag : tags) {
173      tlen = tag.getValueLength();
174      pos = Bytes.putAsShort(b, pos, tlen + Tag.TYPE_LENGTH_SIZE);
175      pos = Bytes.putByte(b, pos, tag.getType());
176      if (tag.hasArray()) {
177        pos = Bytes.putBytes(b, pos, tag.getValueArray(), tag.getValueOffset(), tlen);
178      } else {
179        ByteBufferUtils.copyFromBufferToArray(b, tag.getValueByteBuffer(), tag.getValueOffset(),
180          pos, tlen);
181        pos += tlen;
182      }
183    }
184    return b;
185  }
186
187  /**
188   * Converts the value bytes of the given tag into a long value
189   * @param tag The Tag
190   * @return value as long
191   */
192  public static long getValueAsLong(Tag tag) {
193    if (tag.hasArray()) {
194      return Bytes.toLong(tag.getValueArray(), tag.getValueOffset(), tag.getValueLength());
195    }
196    return ByteBufferUtils.toLong(tag.getValueByteBuffer(), tag.getValueOffset());
197  }
198
199  /**
200   * Converts the value bytes of the given tag into a byte value
201   * @param tag The Tag
202   * @return value as byte
203   */
204  public static byte getValueAsByte(Tag tag) {
205    if (tag.hasArray()) {
206      return tag.getValueArray()[tag.getValueOffset()];
207    }
208    return ByteBufferUtils.toByte(tag.getValueByteBuffer(), tag.getValueOffset());
209  }
210}
