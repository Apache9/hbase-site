083  public DispatchMergingRegionsProcedure(
084      final MasterProcedureEnv env,
085      final TableName tableName,
086      final HRegionInfo [] regionsToMerge,
087      final boolean forcible) {
088    this.traceEnabled = isTraceEnabled();
089    this.assignmentManager = getAssignmentManager(env);
090    this.tableName = tableName;
091    // For now, we only merge 2 regions.  It could be extended to more than 2 regions in
092    // the future.
093    assert(regionsToMerge.length == 2);
094    this.regionsToMerge = regionsToMerge;
095    this.forcible = forcible;
096
097    this.user = env.getRequestUser();
098    this.setOwner(this.user.getShortName());
099
100    this.timeout = -1;
101    this.regionsToMergeListFullName = getRegionsToMergeListFullNameString();
102    this.regionsToMergeListEncodedName = getRegionsToMergeListEncodedNameString();
103  }

105  @Override
106  protected Flow executeFromState(
107      final MasterProcedureEnv env,
108      final DispatchMergingRegionsState state) throws InterruptedException {
109    if (isTraceEnabled()) {
110      LOG.trace(this + " execute state=" + state);
111    }
112
113    try {
114      switch (state) {
115      case DISPATCH_MERGING_REGIONS_PREPARE:
116        prepareMergeRegion(env);
117        setNextState(DispatchMergingRegionsState.DISPATCH_MERGING_REGIONS_PRE_OPERATION);
118        break;
119      case DISPATCH_MERGING_REGIONS_PRE_OPERATION:
120        //Unused for now - reserve to add preMerge coprocessor in the future
121        setNextState(DispatchMergingRegionsState.DISPATCH_MERGING_REGIONS_MOVE_REGION_TO_SAME_RS);
122        break;
123      case DISPATCH_MERGING_REGIONS_MOVE_REGION_TO_SAME_RS:
124        if (MoveRegionsToSameRS(env)) {
125          setNextState(DispatchMergingRegionsState.DISPATCH_MERGING_REGIONS_DO_MERGE_IN_RS);
126        } else {
127          LOG.info("Cancel merging regions " + getRegionsToMergeListFullNameString()
128            + ", because can't move them to the same RS");
129          setNextState(DispatchMergingRegionsState.DISPATCH_MERGING_REGIONS_POST_OPERATION);
130        }
131        break;
132      case DISPATCH_MERGING_REGIONS_DO_MERGE_IN_RS:
133        doMergeInRS(env);
134        setNextState(DispatchMergingRegionsState.DISPATCH_MERGING_REGIONS_POST_OPERATION);
135        break;
136      case DISPATCH_MERGING_REGIONS_POST_OPERATION:
137        //Unused for now - reserve to add postCompletedMerge coprocessor in the future
138        return Flow.NO_MORE_STATE;
139      default:
140        throw new UnsupportedOperationException(this + " unhandled state=" + state);
141      }
142    } catch (IOException e) {
143      LOG.warn("Error trying to merge regions " + getRegionsToMergeListFullNameString() +
144        " in the table " + tableName + " (in state=" + state + ")", e);
145
146      setFailure("master-merge-regions", e);
147    }
148    return Flow.HAS_MORE_STATE;
149  }

151  @Override
152  protected void rollbackState(
153      final MasterProcedureEnv env,
154      final DispatchMergingRegionsState state) throws IOException, InterruptedException {
155    if (isTraceEnabled()) {
156      LOG.trace(this + " rollback state=" + state);
157    }
158
159    try {
160      switch (state) {
161      case DISPATCH_MERGING_REGIONS_POST_OPERATION:
162        break; // nothing to rollback
163      case DISPATCH_MERGING_REGIONS_DO_MERGE_IN_RS:
164        String msg = this + " We are in the " + state + " state."
165            + " It is complicated to rollback the merge operation that region server is working on."
166            + " Rollback is not supported and we should let the merge operation to complete";
167        LOG.warn(msg);
168        break;
169      case DISPATCH_MERGING_REGIONS_MOVE_REGION_TO_SAME_RS:
170        break; // nothing to rollback
171      case DISPATCH_MERGING_REGIONS_PRE_OPERATION:
172        break; // nothing to rollback
173      case DISPATCH_MERGING_REGIONS_PREPARE:
174        break; // nothing to rollback
175      default:
176        throw new UnsupportedOperationException(this + " unhandled state=" + state);
177      }
178    } catch (Exception e) {
179      // This will be retried. Unless there is a bug in the code,
180      // this should be just a "temporary error" (e.g. network down)
181      LOG.warn("Failed rollback attempt step " + state + " for merging the regions "
182          + getRegionsToMergeListFullNameString() + " in table " + tableName, e);
183      throw e;
184    }
185  }

219    super.serializeStateData(stream);
220
221    MasterProcedureProtos.DispatchMergingRegionsStateData.Builder dispatchMergingRegionsMsg =
222        MasterProcedureProtos.DispatchMergingRegionsStateData.newBuilder()
223        .setUserInfo(MasterProcedureUtil.toProtoUserInfo(user))
224        .setTableName(ProtobufUtil.toProtoTableName(tableName))
225        .setForcible(forcible);
226    for (HRegionInfo hri: regionsToMerge) {

232  @Override
233  public void deserializeStateData(final InputStream stream) throws IOException {
234    super.deserializeStateData(stream);
235
236    MasterProcedureProtos.DispatchMergingRegionsStateData dispatchMergingRegionsMsg =
237        MasterProcedureProtos.DispatchMergingRegionsStateData.parseDelimitedFrom(stream);
238    user = MasterProcedureUtil.toUserInfo(dispatchMergingRegionsMsg.getUserInfo());
239    tableName = ProtobufUtil.toTableName(dispatchMergingRegionsMsg.getTableName());
240
241    assert(dispatchMergingRegionsMsg.getRegionInfoCount() == 2);
242    regionsToMerge = new HRegionInfo[dispatchMergingRegionsMsg.getRegionInfoCount()];
243    for (int i = 0; i < regionsToMerge.length; i++) {
244      regionsToMerge[i] = HRegionInfo.convert(dispatchMergingRegionsMsg.getRegionInfo(i));
245    }
246  }

281  /**
282   * Prepare merge and do some check
283   * @param env MasterProcedureEnv
284   * @throws IOException
285   */
286  private void prepareMergeRegion(final MasterProcedureEnv env) throws IOException {
287    // Note: the following logic assumes that we only have 2 regions to merge.  In the future,
288    // if we want to extend to more than 2 regions, the code needs to modify a little bit.
289    //
290    CatalogJanitor catalogJanitor = env.getMasterServices().getCatalogJanitor();
291    boolean regionAHasMergeQualifier = !catalogJanitor.cleanMergeQualifier(regionsToMerge[0]);
292    if (regionAHasMergeQualifier
293        || !catalogJanitor.cleanMergeQualifier(regionsToMerge[1])) {
294      String msg = "Skip merging regions " + regionsToMerge[0].getRegionNameAsString()
295          + ", " + regionsToMerge[1].getRegionNameAsString() + ", because region "
296          + (regionAHasMergeQualifier ? regionsToMerge[0].getEncodedName() : regionsToMerge[1]
297              .getEncodedName()) + " has merge qualifier";
298      LOG.info(msg);
299      throw new MergeRegionException(msg);
300    }
301
302      RegionStates regionStates = getAssignmentManager(env).getRegionStates();
303      RegionState regionStateA = regionStates.getRegionState(regionsToMerge[0].getEncodedName());
304      RegionState regionStateB = regionStates.getRegionState(regionsToMerge[1].getEncodedName());
305      if (regionStateA == null || regionStateB == null) {
306        throw new UnknownRegionException(
307          regionStateA == null ?
308              regionsToMerge[0].getEncodedName() : regionsToMerge[1].getEncodedName());
309      }
310
311      if (!regionStateA.isOpened() || !regionStateB.isOpened()) {
312        throw new MergeRegionException(
313          "Unable to merge regions not online " + regionStateA + ", " + regionStateB);
314      }
315
316      if (regionsToMerge[0].getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID ||
317          regionsToMerge[1].getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID) {
318        throw new MergeRegionException("Can't merge non-default replicas");
319      }
320
321      if (!forcible && !HRegionInfo.areAdjacent(regionsToMerge[0], regionsToMerge[1])) {
322        throw new MergeRegionException(
323          "Unable to merge not adjacent regions "
324            + regionsToMerge[0].getRegionNameAsString() + ", "
325            + regionsToMerge[1].getRegionNameAsString()
326            + " where forcible = " + forcible);
327      }
328  }

330  /**
331   * Move all regions to the same region server
332   * @param env MasterProcedureEnv
333   * @return whether target regions hosted by the same RS
334   * @throws IOException
335   */
336  private boolean MoveRegionsToSameRS(final MasterProcedureEnv env) throws IOException {
337    // Make sure regions are on the same regionserver before send merge
338    // regions request to region server.
339    //
340    boolean onSameRS = isRegionsOnTheSameServer(env);
341    if (!onSameRS) {
342      // Note: the following logic assumes that we only have 2 regions to merge.  In the future,
343      // if we want to extend to more than 2 regions, the code needs to modify a little bit.
344      //
345      RegionStates regionStates = getAssignmentManager(env).getRegionStates();
346      ServerName regionLocation2 = regionStates.getRegionServerOfRegion(regionsToMerge[1]);
347
348      RegionLoad loadOfRegionA = getRegionLoad(env, regionLocation, regionsToMerge[0]);
349      RegionLoad loadOfRegionB = getRegionLoad(env, regionLocation2, regionsToMerge[1]);
350      if (loadOfRegionA != null && loadOfRegionB != null
351          && loadOfRegionA.getRequestsCount() < loadOfRegionB.getRequestsCount()) {
352        // switch regionsToMerge[0] and regionsToMerge[1]
353        HRegionInfo tmpRegion = this.regionsToMerge[0];
354        this.regionsToMerge[0] = this.regionsToMerge[1];
355        this.regionsToMerge[1] = tmpRegion;
356        ServerName tmpLocation = regionLocation;
357        regionLocation = regionLocation2;
358        regionLocation2 = tmpLocation;
359      }
360
361      long startTime = EnvironmentEdgeManager.currentTime();
362
363      RegionPlan regionPlan = new RegionPlan(regionsToMerge[1], regionLocation2, regionLocation);
364      LOG.info("Moving regions to same server for merge: " + regionPlan.toString());
365      getAssignmentManager(env).balance(regionPlan);
366      do {
367        try {
368          Thread.sleep(20);
369          // Make sure check RIT first, then get region location, otherwise
370          // we would make a wrong result if region is online between getting
371          // region location and checking RIT
372          boolean isRIT = regionStates.isRegionInTransition(regionsToMerge[1]);
373          regionLocation2 = regionStates.getRegionServerOfRegion(regionsToMerge[1]);
374          onSameRS = regionLocation.equals(regionLocation2);
375          if (onSameRS || !isRIT) {
376            // Regions are on the same RS, or regionsToMerge[1] is not in
377            // RegionInTransition any more
378            break;
379          }
380        } catch (InterruptedException e) {
381          InterruptedIOException iioe = new InterruptedIOException();
382          iioe.initCause(e);
383          throw iioe;
384        }
385      } while ((EnvironmentEdgeManager.currentTime() - startTime) <= getTimeout(env));
386    }
387    return onSameRS;
388  }

390  /**
391   * Do the real merge operation in the region server that hosts regions
392   * @param env MasterProcedureEnv
393   * @throws IOException
394   */
395  private void doMergeInRS(final MasterProcedureEnv env) throws IOException {
396    long duration = 0;
397    long startTime = EnvironmentEdgeManager.currentTime();
398    do {
399      try {
400        if (getServerName(env) == null) {
401          // The merge probably already happen. Check
402          RegionState regionState = getAssignmentManager(env).getRegionStates().getRegionState(
403            regionsToMerge[0].getEncodedName());
404          if (regionState.isMerging() || regionState.isMerged()) {
405            LOG.info("Merge regions " +  getRegionsToMergeListEncodedNameString() +
406              " is in progress or completed.  No need to send a new request.");
407          } else {
408            LOG.warn("Cannot sending merge to hosting server of the regions " +
409              getRegionsToMergeListEncodedNameString() + " as the server is unknown");
410          }
411          return;
412        }
413        // TODO: the following RPC call is not idempotent.  Multiple calls (eg. after master
414        // failover, re-execute this step) could result in some exception thrown that does not
415        // paint the correct picture.  This behavior is on-par with old releases.  Improvement
416        // could happen in the future.
417        env.getMasterServices().getServerManager().sendRegionsMerge(
418          getServerName(env),
419          regionsToMerge[0],
420          regionsToMerge[1],
421          forcible,
422          user);
423        LOG.info("Sent merge to server " + getServerName(env) + " for region " +
424            getRegionsToMergeListEncodedNameString() + ", focible=" + forcible);
425        return;
426      } catch (RegionOpeningException roe) {
427        // Do a retry since region should be online on RS immediately
428        LOG.warn("Failed mergering regions in " + getServerName(env) + ", retrying...", roe);
429      } catch (Exception ie) {
430        LOG.warn("Failed sending merge to " + getServerName(env) + " for regions " +
431            getRegionsToMergeListEncodedNameString() + ", focible=" + forcible, ie);
432        return;
433      }
434    } while ((duration = EnvironmentEdgeManager.currentTime() - startTime) <= getTimeout(env));
435
436    // If we reaches here, it means that we get timed out.
437    String msg = "Failed sending merge to " + getServerName(env) + " after " + duration + "ms";
438    LOG.warn(msg);
439    throw new IOException(msg);
440  }

442  private RegionLoad getRegionLoad(
443      final MasterProcedureEnv env,
444      final ServerName sn,
445      final HRegionInfo hri) {
446    ServerManager serverManager =  env.getMasterServices().getServerManager();
447    ServerLoad load = serverManager.getLoad(sn);
448    if (load != null) {
449      Map<byte[], RegionLoad> regionsLoad = load.getRegionsLoad();
450      if (regionsLoad != null) {
451        return regionsLoad.get(hri.getRegionName());
452      }
453    }
454    return null;
455  }

457  /**
458   * The procedure could be restarted from a different machine. If the variable is null, we need to
459   * retrieve it.
460   * @param env MasterProcedureEnv
461   * @return whether target regions hosted by the same RS
462   */
463  private boolean isRegionsOnTheSameServer(final MasterProcedureEnv env) throws IOException{
464    Boolean onSameRS = true;
465    int i = 0;
466    RegionStates regionStates = getAssignmentManager(env).getRegionStates();
467    regionLocation = regionStates.getRegionServerOfRegion(regionsToMerge[i]);
468    if (regionLocation != null) {
469      for(i = 1; i < regionsToMerge.length; i++) {
470        ServerName regionLocation2 = regionStates.getRegionServerOfRegion(regionsToMerge[i]);
471        if (regionLocation2 != null) {
472          if (onSameRS) {
473            onSameRS = regionLocation.equals(regionLocation2);
474          }
475        } else {
476          // At least one region is not online, merge will fail, no need to continue.
477          break;
478        }
479      }
480      if (i == regionsToMerge.length) {
481        // Finish checking all regions, return the result;
482        return onSameRS;
483      }
484    }
485
486    // If reaching here, at least one region is not online.
487    String msg = "Skip merging regions " + getRegionsToMergeListFullNameString() +
488        ", because region " + regionsToMerge[i].getEncodedName() + " is not online now.";
489    LOG.warn(msg);
490    throw new IOException(msg);
491  }
