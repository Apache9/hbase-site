045/**
046 * ProcedureScheduler for the Master Procedures.
047 * This ProcedureScheduler tries to provide to the ProcedureExecutor procedures
048 * that can be executed without having to wait on a lock.
049 * Most of the master operations can be executed concurrently, if they
050 * are operating on different tables (e.g. two create table procedures can be performed
051 * at the same time) or against two different servers; say two servers that crashed at
052 * about the same time.
053 *
054 * <p>Each procedure should implement an Interface providing information for this queue.
055 * For example table related procedures should implement TableProcedureInterface.
056 * Each procedure will be pushed in its own queue, and based on the operation type
057 * we may make smarter decisions: e.g. we can abort all the operations preceding
058 * a delete table, or similar.
059 *
060 * <h4>Concurrency control</h4>
061 * Concurrent access to member variables (tableRunQueue, serverRunQueue, locking, tableMap,
062 * serverBuckets) is controlled by schedLock(). This mainly includes:<br>
063 * <ul>
064 *   <li>
065 *     {@link #push(Procedure, boolean, boolean)}: A push will add a Queue back to run-queue
066 *     when:
067 *     <ol>
068 *       <li>Queue was empty before push (so must have been out of run-queue)</li>
069 *       <li>Child procedure is added (which means parent procedure holds exclusive lock, and it
070 *           must have moved Queue out of run-queue)</li>
071 *     </ol>
072 *   </li>
073 *   <li>
074 *     {@link #poll(long)}: A poll will remove a Queue from run-queue when:
075 *     <ol>
076 *       <li>Queue becomes empty after poll</li>
077 *       <li>Exclusive lock is requested by polled procedure and lock is available (returns the
078 *           procedure)</li>
079 *       <li>Exclusive lock is requested but lock is not available (returns null)</li>
080 *       <li>Polled procedure is child of parent holding exclusive lock and the next procedure is
081 *           not a child</li>
082 *     </ol>
083 *   </li>
084 *   <li>
085 *     Namespace/table/region locks: Queue is added back to run-queue when lock being released is:
086 *     <ol>
087 *       <li>Exclusive lock</li>
088 *       <li>Last shared lock (in case queue was removed because next procedure in queue required
089 *           exclusive lock)</li>
090 *     </ol>
091 *   </li>
092 * </ul>
093 */
094@InterfaceAudience.Private

115  @Override

120  @Override
121  protected void enqueue(final Procedure proc, final boolean addFront) {
122    if (isTableProcedure(proc)) {
123      doAdd(tableRunQueue, getTableQueue(getTableName(proc)), proc, addFront);
124    } else if (isServerProcedure(proc)) {
125      doAdd(serverRunQueue, getServerQueue(getServerName(proc)), proc, addFront);
126    } else if (isPeerProcedure(proc)) {
127      doAdd(peerRunQueue, getPeerQueue(getPeerId(proc)), proc, addFront);
128    } else {
129      // TODO: at the moment we only have Table and Server procedures
130      // if you are implementing a non-table/non-server procedure, you have two options: create
131      // a group for all the non-table/non-server procedures or try to find a key for your
132      // non-table/non-server procedures and implement something similar to the TableRunQueue.
133      throw new UnsupportedOperationException(
134        "RQs for non-table/non-server procedures are not implemented yet: " + proc);
135    }
136  }

138  private <T extends Comparable<T>> void doAdd(final FairQueue<T> fairq,
139      final Queue<T> queue, final Procedure<?> proc, final boolean addFront) {
140    queue.add(proc, addFront);
141    if (!queue.getLockStatus().hasExclusiveLock() || queue.getLockStatus().isLockOwner(proc.getProcId())) {
142      // if the queue was not remove for an xlock execution
143      // or the proc is the lock owner, put the queue back into execution
144      addToRunQueue(fairq, queue);
145    } else if (queue.getLockStatus().hasParentLock(proc)) {
146      assert addFront : "expected to add a child in the front";
147      // our (proc) parent has the xlock,
148      // so the queue is not in the fairq (run-queue)
149      // add it back to let the child run (inherit the lock)
150      addToRunQueue(fairq, queue);
151    }
152  }

160  @Override
161  protected Procedure dequeue() {
162    // For now, let server handling have precedence over table handling; presumption is that it
163    // is more important handling crashed servers than it is running the
164    // enabling/disabling tables, etc.
165    Procedure<?> pollResult = doPoll(serverRunQueue);
166    if (pollResult == null) {
167      pollResult = doPoll(peerRunQueue);
168    }
169    if (pollResult == null) {
170      pollResult = doPoll(tableRunQueue);
171    }
172    return pollResult;
173  }
174
175  private <T extends Comparable<T>> Procedure<?> doPoll(final FairQueue<T> fairq) {
176    final Queue<T> rq = fairq.poll();
177    if (rq == null || !rq.isAvailable()) {
178      return null;
179    }
180
181    final Procedure<?> pollResult = rq.peek();
182    if (pollResult == null) {
183      return null;
184    }
185    final boolean xlockReq = rq.requireExclusiveLock(pollResult);
186    if (xlockReq && rq.getLockStatus().isLocked() && !rq.getLockStatus().hasLockAccess(pollResult)) {
187      // someone is already holding the lock (e.g. shared lock). avoid a yield
188      removeFromRunQueue(fairq, rq);
189      return null;
190    }
191
192    rq.poll();
193    if (rq.isEmpty() || xlockReq) {
194      removeFromRunQueue(fairq, rq);
195    } else if (rq.getLockStatus().hasParentLock(pollResult)) {
196      // if the rq is in the fairq because of runnable child
197      // check if the next procedure is still a child.
198      // if not, remove the rq from the fairq and go back to the xlock state
199      Procedure<?> nextProc = rq.peek();
200      if (nextProc != null && !Procedure.haveSameParent(nextProc, pollResult)) {
201        removeFromRunQueue(fairq, rq);
202      }
203    }
204
205    return pollResult;
206  }

239  private void clearQueue() {
240    // Remove Servers
241    for (int i = 0; i < serverBuckets.length; ++i) {
242      clear(serverBuckets[i], serverRunQueue, SERVER_QUEUE_KEY_COMPARATOR);
243      serverBuckets[i] = null;
244    }
245
246    // Remove Tables
247    clear(tableMap, tableRunQueue, TABLE_QUEUE_KEY_COMPARATOR);
248    tableMap = null;
249
250    assert size() == 0 : "expected queue size to be 0, got " + size();
251  }
252
253  private <T extends Comparable<T>, TNode extends Queue<T>> void clear(TNode treeMap,
254      final FairQueue<T> fairq, final AvlKeyComparator<TNode> comparator) {
255    while (treeMap != null) {
256      Queue<T> node = AvlTree.getFirst(treeMap);
257      treeMap = AvlTree.remove(treeMap, node.getKey(), comparator);
258      if (fairq != null) removeFromRunQueue(fairq, node);
259    }
260  }
261
262  @Override
263  protected int queueSize() {
264    int count = 0;
265
266    // Server queues
267    final AvlTreeIterator<ServerQueue> serverIter = new AvlTreeIterator<>();
268    for (int i = 0; i < serverBuckets.length; ++i) {
269      serverIter.seekFirst(serverBuckets[i]);
270      while (serverIter.hasNext()) {
271        count += serverIter.next().size();
272      }
273    }
274
275    // Table queues
276    final AvlTreeIterator<TableQueue> tableIter = new AvlTreeIterator<>(tableMap);
277    while (tableIter.hasNext()) {
278      count += tableIter.next().size();
279    }
280
281    // Peer queues
282    final AvlTreeIterator<PeerQueue> peerIter = new AvlTreeIterator<>(peerMap);
283    while (peerIter.hasNext()) {
284      count += peerIter.next().size();
285    }
286
287    return count;
288  }

290  @Override
291  public void completionCleanup(final Procedure proc) {
292    if (proc instanceof TableProcedureInterface) {
293      TableProcedureInterface iProcTable = (TableProcedureInterface) proc;
294      boolean tableDeleted;
295      if (proc.hasException()) {
296        Exception procEx = proc.getException().unwrapRemoteException();
297        if (iProcTable.getTableOperationType() == TableOperationType.CREATE) {
298          // create failed because the table already exist
299          tableDeleted = !(procEx instanceof TableExistsException);
300        } else {
301          // the operation failed because the table does not exist
302          tableDeleted = (procEx instanceof TableNotFoundException);
303        }
304      } else {
305        // the table was deleted
306        tableDeleted = (iProcTable.getTableOperationType() == TableOperationType.DELETE);
307      }
308      if (tableDeleted) {
309        markTableAsDeleted(iProcTable.getTableName(), proc);
310        return;
311      }
312    } else if (proc instanceof PeerProcedureInterface) {
313      PeerProcedureInterface iProcPeer = (PeerProcedureInterface) proc;
314      tryCleanupPeerQueue(iProcPeer.getPeerId(), proc);
315    } else {
316      // No cleanup for ServerProcedureInterface types, yet.
317      return;
318    }
319  }

321  private static <T extends Comparable<T>> void addToRunQueue(FairQueue<T> fairq, Queue<T> queue) {
322    if (!AvlIterableList.isLinked(queue) && !queue.isEmpty()) {
323      fairq.add(queue);
324    }
325  }
326
327  private static <T extends Comparable<T>> void removeFromRunQueue(
328      FairQueue<T> fairq, Queue<T> queue) {
329    if (AvlIterableList.isLinked(queue)) {
330      fairq.remove(queue);
331    }
332  }

337  private TableQueue getTableQueue(TableName tableName) {
338    TableQueue node = AvlTree.get(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
339    if (node != null) return node;
340
341    node = new TableQueue(tableName, MasterProcedureUtil.getTablePriority(tableName),
342        locking.getTableLock(tableName), locking.getNamespaceLock(tableName.getNamespaceAsString()));
343    tableMap = AvlTree.insert(tableMap, node);
344    return node;
345  }
346
347  private void removeTableQueue(TableName tableName) {
348    tableMap = AvlTree.remove(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
349    locking.removeTableLock(tableName);
350  }

363  private ServerQueue getServerQueue(ServerName serverName) {
364    final int index = getBucketIndex(serverBuckets, serverName.hashCode());
365    ServerQueue node = AvlTree.get(serverBuckets[index], serverName, SERVER_QUEUE_KEY_COMPARATOR);
366    if (node != null) return node;
367
368    node = new ServerQueue(serverName, locking.getServerLock(serverName));
369    serverBuckets[index] = AvlTree.insert(serverBuckets[index], node);
370    return node;
371  }
372
373  private static int getBucketIndex(Object[] buckets, int hashCode) {
374    return Math.abs(hashCode) % buckets.length;
375  }

403  private void tryCleanupPeerQueue(String peerId, Procedure<?> procedure) {
404    schedLock();
405    try {
406      PeerQueue queue = AvlTree.get(peerMap, peerId, PEER_QUEUE_KEY_COMPARATOR);
407      if (queue == null) {
408        return;
409      }
410
411      final LockAndQueue lock = locking.getPeerLock(peerId);
412      if (queue.isEmpty() && lock.tryExclusiveLock(procedure)) {
413        removeFromRunQueue(peerRunQueue, queue);
414        removePeerQueue(peerId);
415      }
416    } finally {
417      schedUnlock();
418    }
419  }

432  /**
433   * Get lock info for a resource of specified type and name and log details
434   */
435  private void logLockedResource(LockedResourceType resourceType, String resourceName) {
436    if (!LOG.isDebugEnabled()) {
437      return;
438    }
439
440    LockedResource lockedResource = getLockResource(resourceType, resourceName);
441    if (lockedResource != null) {
442      String msg = resourceType.toString() + " '" + resourceName + "', shared lock count=" +
443          lockedResource.getSharedLockCount();
444
445      Procedure<?> proc = lockedResource.getExclusiveLockOwnerProcedure();
446      if (proc != null) {
447        msg += ", exclusively locked by procId=" + proc.getProcId();
448      }
449      LOG.debug(msg);
450    }
451  }

453  /**
454   * Suspend the procedure if the specified table is already locked.
455   * Other operations in the table-queue will be executed after the lock is released.
456   * @param procedure the procedure trying to acquire the lock
457   * @param table Table to lock
458   * @return true if the procedure has to wait for the table to be available
459   */
460  public boolean waitTableExclusiveLock(final Procedure procedure, final TableName table) {
461    schedLock();
462    try {
463      final String namespace = table.getNamespaceAsString();
464      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
465      final LockAndQueue tableLock = locking.getTableLock(table);
466      if (!namespaceLock.trySharedLock()) {
467        waitProcedure(namespaceLock, procedure);
468        logLockedResource(LockedResourceType.NAMESPACE, namespace);
469        return true;
470      }
471      if (!tableLock.tryExclusiveLock(procedure)) {
472        namespaceLock.releaseSharedLock();
473        waitProcedure(tableLock, procedure);
474        logLockedResource(LockedResourceType.TABLE, table.getNameAsString());
475        return true;
476      }
477      removeFromRunQueue(tableRunQueue, getTableQueue(table));
478      return false;
479    } finally {
480      schedUnlock();
481    }
482  }
483
484  /**
485   * Wake the procedures waiting for the specified table
486   * @param procedure the procedure releasing the lock
487   * @param table the name of the table that has the exclusive lock
488   */
489  public void wakeTableExclusiveLock(final Procedure procedure, final TableName table) {
490    schedLock();
491    try {
492      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
493      final LockAndQueue tableLock = locking.getTableLock(table);
494      int waitingCount = 0;
495
496      if (!tableLock.hasParentLock(procedure)) {
497        tableLock.releaseExclusiveLock(procedure);
498        waitingCount += wakeWaitingProcedures(tableLock);
499      }
500      if (namespaceLock.releaseSharedLock()) {
501        waitingCount += wakeWaitingProcedures(namespaceLock);
502      }
503      addToRunQueue(tableRunQueue, getTableQueue(table));
504      wakePollIfNeeded(waitingCount);
505    } finally {
506      schedUnlock();
507    }
508  }

510  /**
511   * Suspend the procedure if the specified table is already locked.
512   * other "read" operations in the table-queue may be executed concurrently,
513   * @param procedure the procedure trying to acquire the lock
514   * @param table Table to lock
515   * @return true if the procedure has to wait for the table to be available
516   */
517  public boolean waitTableSharedLock(final Procedure procedure, final TableName table) {
518    return waitTableQueueSharedLock(procedure, table) == null;
519  }
520
521  private TableQueue waitTableQueueSharedLock(final Procedure<?> procedure, final TableName table) {
522    schedLock();
523    try {
524      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
525      final LockAndQueue tableLock = locking.getTableLock(table);
526      if (!namespaceLock.trySharedLock()) {
527        waitProcedure(namespaceLock, procedure);
528        return null;
529      }
530
531      if (!tableLock.trySharedLock()) {
532        namespaceLock.releaseSharedLock();
533        waitProcedure(tableLock, procedure);
534        return null;
535      }
536
537      return getTableQueue(table);
538    } finally {
539      schedUnlock();
540    }
541  }
542
543  /**
544   * Wake the procedures waiting for the specified table
545   * @param procedure the procedure releasing the lock
546   * @param table the name of the table that has the shared lock
547   */
548  public void wakeTableSharedLock(final Procedure procedure, final TableName table) {
549    schedLock();
550    try {
551      final LockAndQueue namespaceLock = locking.getNamespaceLock(table.getNamespaceAsString());
552      final LockAndQueue tableLock = locking.getTableLock(table);
553      int waitingCount = 0;
554      if (tableLock.releaseSharedLock()) {
555        addToRunQueue(tableRunQueue, getTableQueue(table));
556        waitingCount += wakeWaitingProcedures(tableLock);
557      }
558      if (namespaceLock.releaseSharedLock()) {
559        waitingCount += wakeWaitingProcedures(namespaceLock);
560      }
561      wakePollIfNeeded(waitingCount);
562    } finally {
563      schedUnlock();
564    }
565  }

567  /**
568   * Tries to remove the queue and the table-lock of the specified table.
569   * If there are new operations pending (e.g. a new create),
570   * the remove will not be performed.
571   * @param table the name of the table that should be marked as deleted
572   * @param procedure the procedure that is removing the table
573   * @return true if deletion succeeded, false otherwise meaning that there are
574   *     other new operations pending for that table (e.g. a new create).
575   */
576  @VisibleForTesting
577  boolean markTableAsDeleted(final TableName table, final Procedure<?> procedure) {
578    schedLock();
579    try {
580      final TableQueue queue = getTableQueue(table);
581      final LockAndQueue tableLock = locking.getTableLock(table);
582      if (queue == null) return true;
583
584      if (queue.isEmpty() && tableLock.tryExclusiveLock(procedure)) {
585        // remove the table from the run-queue and the map
586        if (AvlIterableList.isLinked(queue)) {
587          tableRunQueue.remove(queue);
588        }
589        removeTableQueue(table);
590      } else {
591        // TODO: If there are no create, we can drop all the other ops
592        return false;
593      }
594    } finally {
595      schedUnlock();
596    }
597    return true;
598  }

603  /**
604   * Suspend the procedure if the specified region is already locked.
605   * @param procedure the procedure trying to acquire the lock on the region
606   * @param regionInfo the region we are trying to lock
607   * @return true if the procedure has to wait for the regions to be available

609  public boolean waitRegion(final Procedure procedure, final RegionInfo regionInfo) {
610    return waitRegions(procedure, regionInfo.getTable(), regionInfo);
611  }
612
613  /**
614   * Suspend the procedure if the specified set of regions are already locked.
615   * @param procedure the procedure trying to acquire the lock on the regions
616   * @param table the table name of the regions we are trying to lock
617   * @param regionInfo the list of regions we are trying to lock
618   * @return true if the procedure has to wait for the regions to be available
619   */
620  public boolean waitRegions(final Procedure procedure, final TableName table,
621      final RegionInfo... regionInfo) {
622    Arrays.sort(regionInfo, RegionInfo.COMPARATOR);
623    schedLock();
624    try {
625      // If there is parent procedure, it would have already taken xlock, so no need to take
626      // shared lock here. Otherwise, take shared lock.
627      if (!procedure.hasParent()
628          && waitTableQueueSharedLock(procedure, table) == null) {
629          return true;
630      }
631
632      // acquire region xlocks or wait
633      boolean hasLock = true;
634      final LockAndQueue[] regionLocks = new LockAndQueue[regionInfo.length];
635      for (int i = 0; i < regionInfo.length; ++i) {
636        LOG.info("{} checking lock on {}", procedure, regionInfo[i].getEncodedName());
637        assert table != null;
638        assert regionInfo[i] != null;
639        assert regionInfo[i].getTable() != null;
640        assert regionInfo[i].getTable().equals(table): regionInfo[i] + " " + procedure;
641        assert i == 0 || regionInfo[i] != regionInfo[i - 1] : "duplicate region: " + regionInfo[i];
642
643        regionLocks[i] = locking.getRegionLock(regionInfo[i].getEncodedName());
644        if (!regionLocks[i].tryExclusiveLock(procedure)) {
645          waitProcedure(regionLocks[i], procedure);
646          hasLock = false;
647          while (i-- > 0) {
648            regionLocks[i].releaseExclusiveLock(procedure);
649          }
650          break;
651        }
652      }
653
654      if (!hasLock && !procedure.hasParent()) {
655        wakeTableSharedLock(procedure, table);
656      }
657      return !hasLock;
658    } finally {
659      schedUnlock();
660    }

664   * Wake the procedures waiting for the specified region
665   * @param procedure the procedure that was holding the region
666   * @param regionInfo the region the procedure was holding
667   */

669    wakeRegions(procedure, regionInfo.getTable(), regionInfo);
670  }

672  /**
673   * Wake the procedures waiting for the specified regions
674   * @param procedure the procedure that was holding the regions
675   * @param regionInfo the list of regions the procedure was holding
676   */
677  public void wakeRegions(final Procedure procedure,final TableName table,
678      final RegionInfo... regionInfo) {
679    Arrays.sort(regionInfo, RegionInfo.COMPARATOR);
680    schedLock();
681    try {
682      int numProcs = 0;
683      final Procedure[] nextProcs = new Procedure[regionInfo.length];
684      for (int i = 0; i < regionInfo.length; ++i) {
685        assert regionInfo[i].getTable().equals(table);
686        assert i == 0 || regionInfo[i] != regionInfo[i - 1] : "duplicate region: " + regionInfo[i];
687
688        LockAndQueue regionLock = locking.getRegionLock(regionInfo[i].getEncodedName());
689        if (regionLock.releaseExclusiveLock(procedure)) {
690          if (!regionLock.isEmpty()) {
691            // release one procedure at the time since regions has an xlock
692            nextProcs[numProcs++] = regionLock.removeFirst();
693          } else {
694            locking.removeRegionLock(regionInfo[i].getEncodedName());
695          }
696        }
697      }
698
699      // awake procedures if any
700      for (int i = numProcs - 1; i >= 0; --i) {
701        wakeProcedure(nextProcs[i]);
702      }
703      wakePollIfNeeded(numProcs);
704      if (!procedure.hasParent()) {
705        // release the table shared-lock.
706        // (if we have a parent, it is holding an xlock so we didn't take the shared-lock)
707        wakeTableSharedLock(procedure, table);
708      }
709    } finally {
710      schedUnlock();
711    }
712  }

715  //  Namespace Locking Helpers
716  // ============================================================================
717  /**
718   * Suspend the procedure if the specified namespace is already locked.
719   * @see #wakeNamespaceExclusiveLock(Procedure,String)
720   * @param procedure the procedure trying to acquire the lock
721   * @param namespace Namespace to lock
722   * @return true if the procedure has to wait for the namespace to be available
723   */
724  public boolean waitNamespaceExclusiveLock(final Procedure procedure, final String namespace) {
725    schedLock();
726    try {
727      final LockAndQueue systemNamespaceTableLock =
728          locking.getTableLock(TableName.NAMESPACE_TABLE_NAME);
729      if (!systemNamespaceTableLock.trySharedLock()) {
730        waitProcedure(systemNamespaceTableLock, procedure);
731        logLockedResource(LockedResourceType.TABLE,
732            TableName.NAMESPACE_TABLE_NAME.getNameAsString());
733        return true;
734      }
735
736      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
737      if (!namespaceLock.tryExclusiveLock(procedure)) {
738        systemNamespaceTableLock.releaseSharedLock();
739        waitProcedure(namespaceLock, procedure);
740        logLockedResource(LockedResourceType.NAMESPACE, namespace);
741        return true;
742      }
743      return false;
744    } finally {
745      schedUnlock();
746    }
747  }

749  /**
750   * Wake the procedures waiting for the specified namespace
751   * @see #waitNamespaceExclusiveLock(Procedure,String)
752   * @param procedure the procedure releasing the lock
753   * @param namespace the namespace that has the exclusive lock
754   */
755  public void wakeNamespaceExclusiveLock(final Procedure procedure, final String namespace) {
756    schedLock();
757    try {
758      final LockAndQueue namespaceLock = locking.getNamespaceLock(namespace);
759      final LockAndQueue systemNamespaceTableLock =
760          locking.getTableLock(TableName.NAMESPACE_TABLE_NAME);
761      namespaceLock.releaseExclusiveLock(procedure);
762      int waitingCount = 0;
763      if (systemNamespaceTableLock.releaseSharedLock()) {
764        addToRunQueue(tableRunQueue, getTableQueue(TableName.NAMESPACE_TABLE_NAME));
765        waitingCount += wakeWaitingProcedures(systemNamespaceTableLock);
766      }
767      waitingCount += wakeWaitingProcedures(namespaceLock);
768      wakePollIfNeeded(waitingCount);
769    } finally {
770      schedUnlock();
771    }
772  }
