051/**
052 * ProcedureRunnableSet for the Master Procedures.
053 * This RunnableSet tries to provide to the ProcedureExecutor procedures
054 * that can be executed without having to wait on a lock.
055 * Most of the master operations can be executed concurrently, if they
056 * are operating on different tables (e.g. two create table can be performed
057 * at the same, time assuming table A and table B) or against two different servers; say
058 * two servers that crashed at about the same time.
059 *
060 * <p>Each procedure should implement an interface providing information for this queue.
061 * for example table related procedures should implement TableProcedureInterface.
062 * each procedure will be pushed in its own queue, and based on the operation type
063 * we may take smarter decision. e.g. we can abort all the operations preceding
064 * a delete table, or similar.
065 */
066@InterfaceAudience.Private
067@InterfaceStability.Evolving
068public class MasterProcedureScheduler implements ProcedureRunnableSet {
069  private static final Log LOG = LogFactory.getLog(MasterProcedureScheduler.class);

125  private void doAdd(final Procedure proc, final boolean addFront, final boolean notify) {
126    schedLock.lock();
127    try {
128      if (isTableProcedure(proc)) {
129        doAdd(tableRunQueue, getTableQueue(getTableName(proc)), proc, addFront);
130      } else if (isServerProcedure(proc)) {
131        doAdd(serverRunQueue, getServerQueue(getServerName(proc)), proc, addFront);
132      } else {
133        // TODO: at the moment we only have Table and Server procedures
134        // if you are implementing a non-table/non-server procedure, you have two options: create
135        // a group for all the non-table/non-server procedures or try to find a key for your
136        // non-table/non-server procedures and implement something similar to the TableRunQueue.
137        throw new UnsupportedOperationException(
138          "RQs for non-table/non-server procedures are not implemented yet: " + proc);
139      }
140      if (notify) {
141        schedWaitCond.signal();
142      }
143    } finally {
144      schedLock.unlock();
145    }
146  }

148  private <T extends Comparable<T>> void doAdd(final FairQueue<T> fairq,
149      final Queue<T> queue, final Procedure proc, final boolean addFront) {
150    if (proc.isSuspended()) return;
151
152    queue.add(proc, addFront);
153    if (!(queue.isSuspended() ||
154        (queue.hasExclusiveLock() && !queue.isLockOwner(proc.getProcId())))) {
155      // the queue is not suspended or removed from the fairq (run-queue)
156      // because someone has an xlock on it.
157      // so, if the queue is not-linked we should add it
158      if (queue.size() == 1 && !AvlIterableList.isLinked(queue)) {
159        fairq.add(queue);
160      }
161    } else if (queue.hasParentLock(proc)) {
162      assert addFront : "expected to add a child in the front";
163      assert !queue.isSuspended() : "unexpected suspended state for the queue";
164      // our (proc) parent has the xlock,
165      // so the queue is not in the fairq (run-queue)
166      // add it back to let the child run (inherit the lock)
167      if (!AvlIterableList.isLinked(queue)) {
168        fairq.add(queue);
169      }
170    }
171  }

178  @edu.umd.cs.findbugs.annotations.SuppressWarnings("WA_AWAIT_NOT_IN_LOOP")
179  protected Procedure poll(long waitNsec) {
180    Procedure pollResult = null;
181    schedLock.lock();
182    try {
183      if (!hasRunnables()) {
184        if (waitNsec < 0) {
185          schedWaitCond.await();
186        } else {
187          schedWaitCond.awaitNanos(waitNsec);
188        }
189        if (!hasRunnables()) {
190          return null;
191        }
192      }
193
194      // For now, let server handling have precedence over table handling; presumption is that it
195      // is more important handling crashed servers than it is running the
196      // enabling/disabling tables, etc.
197      pollResult = doPoll(serverRunQueue);
198      if (pollResult == null) {
199        pollResult = doPoll(tableRunQueue);
200      }
201
202      // update metrics
203      pollCalls++;
204      nullPollCalls += (pollResult == null) ? 1 : 0;
205    } catch (InterruptedException e) {
206      Thread.currentThread().interrupt();
207    } finally {
208      schedLock.unlock();
209    }
210    return pollResult;
211  }
212
213  private boolean hasRunnables() {
214    return tableRunQueue.hasRunnables() || serverRunQueue.hasRunnables();
215  }

217  private <T extends Comparable<T>> Procedure doPoll(final FairQueue<T> fairq) {
218    final Queue<T> rq = fairq.poll();
219    if (rq == null || !rq.isAvailable()) {
220      return null;
221    }
222
223    assert !rq.isSuspended() : "rq=" + rq + " is suspended";
224    final Procedure pollResult = rq.peek();
225    final boolean xlockReq = rq.requireExclusiveLock(pollResult);
226    if (xlockReq && rq.isLocked() && !rq.hasLockAccess(pollResult)) {
227      // someone is already holding the lock (e.g. shared lock). avoid a yield
228      return null;
229    }
230
231    rq.poll();
232    if (rq.isEmpty() || xlockReq) {
233      removeFromRunQueue(fairq, rq);
234    } else if (rq.hasParentLock(pollResult)) {
235      // if the rq is in the fairq because of runnable child
236      // check if the next procedure is still a child.
237      // if not, remove the rq from the fairq and go back to the xlock state
238      Procedure nextProc = rq.peek();
239      if (nextProc != null && !Procedure.haveSameParent(nextProc, pollResult)) {
240        removeFromRunQueue(fairq, rq);
241      }
242    }
243
244    return pollResult;
245  }

247  @Override
248  public void clear() {
249    // NOTE: USED ONLY FOR TESTING
250    schedLock.lock();
251    try {
252      // Remove Servers
253      for (int i = 0; i < serverBuckets.length; ++i) {
254        clear(serverBuckets[i], serverRunQueue, SERVER_QUEUE_KEY_COMPARATOR);
255        serverBuckets[i] = null;
256      }
257
258      // Remove Tables
259      clear(tableMap, tableRunQueue, TABLE_QUEUE_KEY_COMPARATOR);
260      tableMap = null;
261
262      assert size() == 0 : "expected queue size to be 0, got " + size();
263    } finally {
264      schedLock.unlock();
265    }
266  }

268  private <T extends Comparable<T>, TNode extends Queue<T>> void clear(TNode treeMap,
269      final FairQueue<T> fairq, final AvlKeyComparator<TNode> comparator) {
270    while (treeMap != null) {
271      Queue<T> node = AvlTree.getFirst(treeMap);
272      assert !node.isSuspended() : "can't clear suspended " + node.getKey();
273      treeMap = AvlTree.remove(treeMap, node.getKey(), comparator);
274      removeFromRunQueue(fairq, node);
275    }
276  }

278  private void wakePollIfNeeded(final int waitingCount) {
279    if (waitingCount > 1) {
280      schedWaitCond.signalAll();
281    } else if (waitingCount > 0) {
282      schedWaitCond.signal();
283    }

296  @Override
297  public int size() {
298    schedLock.lock();
299    try {
300      int count = 0;
301
302      // Server queues
303      final AvlTreeIterator<ServerQueue> serverIter = new AvlTreeIterator<ServerQueue>();
304      for (int i = 0; i < serverBuckets.length; ++i) {
305        serverIter.seekFirst(serverBuckets[i]);
306        while (serverIter.hasNext()) {
307          count += serverIter.next().size();
308        }
309      }
310
311      // Table queues
312      final AvlTreeIterator<TableQueue> tableIter = new AvlTreeIterator<TableQueue>(tableMap);
313      while (tableIter.hasNext()) {
314        count += tableIter.next().size();
315      }
316      return count;
317    } finally {
318      schedLock.unlock();
319    }
320  }

322  @Override
323  public void completionCleanup(final Procedure proc) {
324    if (proc instanceof TableProcedureInterface) {
325      TableProcedureInterface iProcTable = (TableProcedureInterface)proc;
326      boolean tableDeleted;
327      if (proc.hasException()) {
328        Exception procEx = proc.getException().unwrapRemoteException();
329        if (iProcTable.getTableOperationType() == TableOperationType.CREATE) {
330          // create failed because the table already exist
331          tableDeleted = !(procEx instanceof TableExistsException);
332        } else {
333          // the operation failed because the table does not exist
334          tableDeleted = (procEx instanceof TableNotFoundException);
335        }
336      } else {
337        // the table was deleted
338        tableDeleted = (iProcTable.getTableOperationType() == TableOperationType.DELETE);
339      }
340      if (tableDeleted) {
341        markTableAsDeleted(iProcTable.getTableName(), proc);
342        return;
343      }
344    } else {
345      // No cleanup for ServerProcedureInterface types, yet.
346      return;
347    }
348  }

350  private <T extends Comparable<T>> void addToRunQueue(FairQueue<T> fairq, Queue<T> queue) {
351    if (AvlIterableList.isLinked(queue)) return;
352    if (!queue.isEmpty())  {
353      fairq.add(queue);
354    }
355  }

357  private <T extends Comparable<T>> void removeFromRunQueue(FairQueue<T> fairq, Queue<T> queue) {
358    if (!AvlIterableList.isLinked(queue)) return;
359    fairq.remove(queue);
360  }

390   * @param suspendQueue true if the entire queue of the procedure should be suspended
391   * @return true if the procedure has to wait for the event to be ready, false otherwise.

395    return waitEvent(event, /* lockEvent= */false, procedure, suspendQueue);
396  }

398  private boolean waitEvent(final ProcedureEvent event, final boolean lockEvent,
399      final Procedure procedure, final boolean suspendQueue) {
400    synchronized (event) {
401      if (event.isReady()) {
402        if (lockEvent) {
403          event.setReady(false);
404        }
405        return false;
406      }
407
408      if (!suspendQueue) {
409        suspendProcedure(event, procedure);
410      } else if (isTableProcedure(procedure)) {
411        waitTableEvent(event, procedure);
412      } else if (isServerProcedure(procedure)) {
413        waitServerEvent(event, procedure);
414      } else {
415        // TODO: at the moment we only have Table and Server procedures
416        // if you are implementing a non-table/non-server procedure, you have two options: create
417        // a group for all the non-table/non-server procedures or try to find a key for your
418        // non-table/non-server procedures and implement something similar to the TableRunQueue.
419        throw new UnsupportedOperationException(
420          "RQs for non-table/non-server procedures are not implemented yet: " + procedure);
421      }
422    }
423    return true;
424  }

426  private void waitTableEvent(final ProcedureEvent event, final Procedure procedure) {
427    final TableName tableName = getTableName(procedure);
428    final boolean isDebugEnabled = LOG.isDebugEnabled();
429
430    schedLock.lock();
431    try {
432      TableQueue queue = getTableQueue(tableName);
433      queue.addFront(procedure);
434      if (queue.isSuspended()) return;
435
436      if (isDebugEnabled) {
437        LOG.debug("Suspend table queue " + tableName);
438      }
439      queue.setSuspended(true);
440      removeFromRunQueue(tableRunQueue, queue);
441      event.suspendTableQueue(queue);
442    } finally {
443      schedLock.unlock();
444    }
445  }
446
447  private void waitServerEvent(final ProcedureEvent event, final Procedure procedure) {
448    final ServerName serverName = getServerName(procedure);
449    final boolean isDebugEnabled = LOG.isDebugEnabled();
450
451    schedLock.lock();
452    try {
453      // TODO: This will change once we have the new AM
454      ServerQueue queue = getServerQueue(serverName);
455      queue.addFront(procedure);
456      if (queue.isSuspended()) return;
457
458      if (isDebugEnabled) {
459        LOG.debug("Suspend server queue " + serverName);
460      }
461      queue.setSuspended(true);
462      removeFromRunQueue(serverRunQueue, queue);
463      event.suspendServerQueue(queue);
464    } finally {
465      schedLock.unlock();
466    }
467  }

484  /**
485   * Wake every procedure waiting for the specified event
486   * (By design each event has only one "wake" caller)
487   * @param event the event to wait
488   */
489  public void wakeEvent(final ProcedureEvent event) {
490    final boolean isTraceEnabled = LOG.isTraceEnabled();
491    synchronized (event) {
492      event.setReady(true);
493      if (isTraceEnabled) {
494        LOG.trace("Wake event " + event);
495      }
496
497      schedLock.lock();
498      try {
499        final int waitingCount = popEventWaitingObjects(event);
500        wakePollIfNeeded(waitingCount);
501      } finally {
502        schedLock.unlock();
503      }
504    }
505  }
506
507  /**
508   * Wake every procedure waiting for the specified events.
509   * (By design each event has only one "wake" caller)
510   * @param events the list of events to wake
511   * @param count the number of events in the array to wake
512   */
513  public void wakeEvents(final ProcedureEvent[] events, final int count) {
514    final boolean isTraceEnabled = LOG.isTraceEnabled();
515    schedLock.lock();
516    try {
517      int waitingCount = 0;
518      for (int i = 0; i < count; ++i) {
519        final ProcedureEvent event = events[i];
520        synchronized (event) {
521          event.setReady(true);
522          if (isTraceEnabled) {
523            LOG.trace("Wake event " + event);
524          }
525          waitingCount += popEventWaitingObjects(event);
526        }
527      }
528      wakePollIfNeeded(waitingCount);
529    } finally {
530      schedLock.unlock();
531    }
532  }
533
534  private int popEventWaitingObjects(final ProcedureEvent event) {
535    int count = 0;
536    while (event.hasWaitingTables()) {
537      final Queue<TableName> queue = event.popWaitingTable();
538      queue.setSuspended(false);
539      addToRunQueue(tableRunQueue, queue);
540      count += queue.size();
541    }
542    // TODO: This will change once we have the new AM
543    while (event.hasWaitingServers()) {
544      final Queue<ServerName> queue = event.popWaitingServer();
545      queue.setSuspended(false);
546      addToRunQueue(serverRunQueue, queue);
547      count += queue.size();
548    }
549
550    while (event.hasWaitingProcedures()) {
551      wakeProcedure(event.popWaitingProcedure(false));
552      count++;
553    }
554    return count;
555  }

557  private void suspendProcedure(final BaseProcedureEvent event, final Procedure procedure) {
558    procedure.suspend();
559    event.suspendProcedure(procedure);
560  }

562  private void wakeProcedure(final Procedure procedure) {
563    procedure.resume();
564    doAdd(procedure, /* addFront= */ true, /* notify= */false);
565  }

567  private static abstract class BaseProcedureEvent {
568    private ArrayDeque<Procedure> waitingProcedures = null;
569
570    protected void suspendProcedure(final Procedure proc) {
571      if (waitingProcedures == null) {
572        waitingProcedures = new ArrayDeque<Procedure>();
573      }
574      waitingProcedures.addLast(proc);
575    }
576
577    protected boolean hasWaitingProcedures() {
578      return waitingProcedures != null;
579    }
580
581    protected Procedure popWaitingProcedure(final boolean popFront) {
582      // it will be nice to use IterableList on a procedure and avoid allocations...
583      Procedure proc = popFront ? waitingProcedures.removeFirst() : waitingProcedures.removeLast();
584      if (waitingProcedures.isEmpty()) {
585        waitingProcedures = null;
586      }
587      return proc;
588    }
589  }

591  public static class ProcedureEvent extends BaseProcedureEvent {
592    private final String description;
593
594    private Queue<ServerName> waitingServers = null;
595    private Queue<TableName> waitingTables = null;
596    private boolean ready = false;
597
598    protected ProcedureEvent() {
599      this(null);
600    }
601
602    public ProcedureEvent(final String description) {
603      this.description = description;
604    }
605
606    public synchronized boolean isReady() {
607      return ready;
608    }
609
610    private synchronized void setReady(boolean isReady) {
611      this.ready = isReady;
612    }
613
614    private void suspendTableQueue(Queue<TableName> queue) {
615      waitingTables = AvlIterableList.append(waitingTables, queue);
616    }
617
618    private void suspendServerQueue(Queue<ServerName> queue) {
619      waitingServers = AvlIterableList.append(waitingServers, queue);
620    }
621
622    private boolean hasWaitingTables() {
623      return waitingTables != null;
624    }
625
626    private Queue<TableName> popWaitingTable() {
627      Queue<TableName> node = waitingTables;
628      waitingTables = AvlIterableList.remove(waitingTables, node);
629      return node;
630    }

632    private boolean hasWaitingServers() {
633      return waitingServers != null;
634    }
635
636    private Queue<ServerName> popWaitingServer() {
637      Queue<ServerName> node = waitingServers;
638      waitingServers = AvlIterableList.remove(waitingServers, node);
639      return node;
640    }
641
642    protected String getDescription() {
643      if (description == null) {
644        // you should override this method if you are using the default constructor
645        throw new UnsupportedOperationException();
646      }
647      return description;
648    }

650    @Override

652      return String.format("%s(%s)", getClass().getSimpleName(), getDescription());
653    }

668  private TableQueue getTableQueue(TableName tableName) {
669    TableQueue node = AvlTree.get(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
670    if (node != null) return node;
671
672    NamespaceQueue nsQueue = getNamespaceQueue(tableName.getNamespaceAsString());
673    node = new TableQueue(tableName, nsQueue, getTablePriority(tableName));
674    tableMap = AvlTree.insert(tableMap, node);
675    return node;
676  }
677
678  private void removeTableQueue(TableName tableName) {
679    tableMap = AvlTree.remove(tableMap, tableName, TABLE_QUEUE_KEY_COMPARATOR);
680  }

682  private int getTablePriority(TableName tableName) {
683    if (tableName.equals(TableName.META_TABLE_NAME)) {
684      return metaTablePriority;
685    } else if (tableName.isSystemTable()) {
686      return sysTablePriority;
687    }
688    return userTablePriority;
689  }

724    final int index = getBucketIndex(serverBuckets, serverName.hashCode());
725    ServerQueue node = AvlTree.get(serverBuckets[index], serverName, SERVER_QUEUE_KEY_COMPARATOR);
726    if (node != null) return node;

729    serverBuckets[index] = AvlTree.insert(serverBuckets[index], node);
730    return (ServerQueue)node;

822  public static class TableQueue extends QueueImpl<TableName> {
823    private final NamespaceQueue namespaceQueue;
824
825    private HashMap<HRegionInfo, RegionEvent> regionEventMap;
826    private TableLock tableLock = null;
827
828    public TableQueue(TableName tableName, NamespaceQueue namespaceQueue, int priority) {
829      super(tableName, priority);
830      this.namespaceQueue = namespaceQueue;
831    }
832
833    public NamespaceQueue getNamespaceQueue() {
834      return namespaceQueue;
835    }

837    @Override
838    public synchronized boolean isAvailable() {
839      // if there are no items in the queue, or the namespace is locked.
840      // we can't execute operation on this table
841      if (isEmpty() || namespaceQueue.hasExclusiveLock()) {
842        return false;
843      }
844
845      if (hasExclusiveLock()) {
846        // if we have an exclusive lock already taken
847        // only child of the lock owner can be executed
848        final Procedure nextProc = peek();
849        return nextProc != null && hasLockAccess(nextProc);
850      }
851
852      // no xlock
853      return true;
854    }

856    public synchronized RegionEvent getRegionEvent(final HRegionInfo regionInfo) {
857      if (regionEventMap == null) {
858        regionEventMap = new HashMap<HRegionInfo, RegionEvent>();
859      }
860      RegionEvent event = regionEventMap.get(regionInfo);
861      if (event == null) {
862        event = new RegionEvent(regionInfo);
863        regionEventMap.put(regionInfo, event);
864      }
865      return event;
866    }
867
868    public synchronized void removeRegionEvent(final RegionEvent event) {
869      regionEventMap.remove(event.getRegionInfo());
870      if (regionEventMap.isEmpty()) {
871        regionEventMap = null;
872      }

875    // TODO: We can abort pending/in-progress operation if the new call is
876    //       something like drop table. We can Override addBack(),
877    //       check the type and abort all the in-flight procedurs.
878    private boolean canAbortPendingOperations(Procedure proc) {
879      TableProcedureInterface tpi = (TableProcedureInterface)proc;
880      switch (tpi.getTableOperationType()) {
881        case DELETE:
882          return true;
883        default:
884          return false;
885      }
886    }

888    public boolean requireExclusiveLock(Procedure proc) {
889      TableProcedureInterface tpi = (TableProcedureInterface)proc;
890      switch (tpi.getTableOperationType()) {
891        case CREATE:
892        case DELETE:
893        case DISABLE:
894        case ENABLE:
895          return true;
896        case EDIT:
897          // we allow concurrent edit on the NS table
898          return !tpi.getTableName().equals(TableName.NAMESPACE_TABLE_NAME);
899        case READ:
900          return false;
901        // region operations are using the shared-lock on the table
902        // and then they will grab an xlock on the region.
903        case SPLIT:
904        case MERGE:
905        case ASSIGN:
906        case UNASSIGN:
907          return false;
908        default:
909          break;
910      }
911      throw new UnsupportedOperationException("unexpected type " + tpi.getTableOperationType());
912    }

914    private synchronized boolean tryZkSharedLock(final TableLockManager lockManager,
915        final String purpose) {
916      // Since we only have one lock resource.  We should only acquire zk lock if the znode
917      // does not exist.
918      //
919      if (isSingleSharedLock()) {
920        // Take zk-read-lock
921        TableName tableName = getKey();
922        tableLock = lockManager.readLock(tableName, purpose);
923        try {
924          tableLock.acquire();
925        } catch (IOException e) {
926          LOG.error("failed acquire read lock on " + tableName, e);
927          tableLock = null;
928          return false;
929        }
930      }
931      return true;
932    }
933
934    private synchronized void releaseZkSharedLock(final TableLockManager lockManager) {
935      if (isSingleSharedLock()) {
936        releaseTableLock(lockManager, true);
937      }
938    }
939
940    private synchronized boolean tryZkExclusiveLock(final TableLockManager lockManager,
941        final String purpose) {
942      // Take zk-write-lock
943      TableName tableName = getKey();
944      tableLock = lockManager.writeLock(tableName, purpose);
945      try {
946        tableLock.acquire();
947      } catch (IOException e) {
948        LOG.error("failed acquire write lock on " + tableName, e);
949        tableLock = null;
950        return false;
951      }
952      return true;
953    }
954
955    private synchronized void releaseZkExclusiveLock(final TableLockManager lockManager) {
956      releaseTableLock(lockManager, true);
957    }
958
959    private void releaseTableLock(final TableLockManager lockManager, boolean reset) {
960      for (int i = 0; i < 3; ++i) {
961        try {
962          tableLock.release();
963          if (reset) {
964            tableLock = null;
965          }
966          break;
967        } catch (IOException e) {
968          LOG.warn("Could not release the table write-lock", e);
969        }
970      }
971    }
972  }

1025  /**
1026   * Try to acquire the exclusive lock on the specified table.
1027   * other operations in the table-queue will be executed after the lock is released.
1028   * @param procedure the procedure trying to acquire the lock
1029   * @param table Table to lock
1030   * @return true if we were able to acquire the lock on the table, otherwise false.
1031   */
1032  public boolean tryAcquireTableExclusiveLock(final Procedure procedure, final TableName table) {
1033    schedLock.lock();
1034    TableQueue queue = getTableQueue(table);
1035    if (!queue.getNamespaceQueue().trySharedLock()) {
1036      schedLock.unlock();
1037      return false;
1038    }
1039
1040    if (!queue.tryExclusiveLock(procedure)) {
1041      queue.getNamespaceQueue().releaseSharedLock();
1042      schedLock.unlock();
1043      return false;
1044    }
1045
1046    removeFromRunQueue(tableRunQueue, queue);
1047    boolean hasParentLock = queue.hasParentLock(procedure);
1048    schedLock.unlock();
1049
1050    boolean hasXLock = true;
1051    if (!hasParentLock) {
1052      // Zk lock is expensive...
1053      hasXLock = queue.tryZkExclusiveLock(lockManager, procedure.toString());
1054      if (!hasXLock) {
1055        schedLock.lock();
1056        if (!hasParentLock) queue.releaseExclusiveLock();
1057        queue.getNamespaceQueue().releaseSharedLock();
1058        addToRunQueue(tableRunQueue, queue);
1059        schedLock.unlock();
1060      }
1061    }
1062    return hasXLock;
1063  }

1066   * Release the exclusive lock taken with tryAcquireTableWrite()
1067   * @param procedure the procedure releasing the lock
1068   * @param table the name of the table that has the exclusive lock

1081    queue.getNamespaceQueue().releaseSharedLock();

1083    schedLock.unlock();

1088   * other "read" operations in the table-queue may be executed concurrently,

1091   * @return true if we were able to acquire the lock on the table, otherwise false.

1097  private TableQueue tryAcquireTableQueueSharedLock(final Procedure procedure,
1098      final TableName table) {
1099    schedLock.lock();
1100    TableQueue queue = getTableQueue(table);
1101    if (!queue.getNamespaceQueue().trySharedLock()) {
1102      return null;
1103    }
1104
1105    if (!queue.trySharedLock()) {
1106      queue.getNamespaceQueue().releaseSharedLock();
1107      schedLock.unlock();
1108      return null;
1109    }
1110
1111    // TODO: Zk lock is expensive and it would be perf bottleneck.  Long term solution is
1112    // to remove it.
1113    if (!queue.tryZkSharedLock(lockManager, procedure.toString())) {
1114      queue.releaseSharedLock();
1115      queue.getNamespaceQueue().releaseSharedLock();
1116      schedLock.unlock();
1117      return null;
1118    }
1119
1120    schedLock.unlock();
1121
1122    return queue;
1123  }

1142  /**
1143   * Tries to remove the queue and the table-lock of the specified table.
1144   * If there are new operations pending (e.g. a new create),
1145   * the remove will not be performed.
1146   * @param table the name of the table that should be marked as deleted
1147   * @param procedure the procedure that is removing the table
1148   * @return true if deletion succeeded, false otherwise meaning that there are
1149   *     other new operations pending for that table (e.g. a new create).
1150   */
1151  @VisibleForTesting
1152  protected boolean markTableAsDeleted(final TableName table, final Procedure procedure) {
1153    final ReentrantLock l = schedLock;
1154    l.lock();
1155    try {
1156      TableQueue queue = getTableQueue(table);
1157      if (queue == null) return true;
1158
1159      if (queue.isEmpty() && queue.tryExclusiveLock(procedure)) {
1160        // remove the table from the run-queue and the map
1161        if (AvlIterableList.isLinked(queue)) {
1162          tableRunQueue.remove(queue);
1163        }
1164
1165        // Remove the table lock
1166        try {
1167          lockManager.tableDeleted(table);
1168        } catch (IOException e) {
1169          LOG.warn("Received exception from TableLockManager.tableDeleted:", e); //not critical
1170        }
1171
1172        removeTableQueue(table);
1173      } else {
1174        // TODO: If there are no create, we can drop all the other ops
1175        return false;
1176      }
1177    } finally {
1178      l.unlock();
1179    }
1180    return true;
1181  }

1196  /**
1197   * Suspend the procedure if the specified set of regions are already locked.
1198   * @param procedure the procedure trying to acquire the lock on the regions
1199   * @param table the table name of the regions we are trying to lock
1200   * @param regionInfo the list of regions we are trying to lock
1201   * @return true if the procedure has to wait for the regions to be available
1202   */
1203  public boolean waitRegions(final Procedure procedure, final TableName table,
1204      final HRegionInfo... regionInfo) {
1205    Arrays.sort(regionInfo);
1206
1207    final TableQueue queue;
1208    if (procedure.hasParent()) {
1209      // the assumption is that the parent procedure have already the table xlock
1210      queue = getTableQueueWithLock(table);
1211    } else {
1212      // acquire the table shared-lock
1213      queue = tryAcquireTableQueueSharedLock(procedure, table);
1214      if (queue == null) return true;
1215    }
1216
1217    // acquire region xlocks or wait
1218    boolean hasLock = true;
1219    final RegionEvent[] event = new RegionEvent[regionInfo.length];
1220    synchronized (queue) {
1221      for (int i = 0; i < regionInfo.length; ++i) {
1222        assert regionInfo[i].getTable().equals(table);
1223        assert i == 0 || regionInfo[i] != regionInfo[i-1] : "duplicate region: " + regionInfo[i];
1224
1225        event[i] = queue.getRegionEvent(regionInfo[i]);
1226        if (!event[i].tryExclusiveLock(procedure.getProcId())) {
1227          suspendProcedure(event[i], procedure);
1228          hasLock = false;
1229          while (i-- > 0) {
1230            event[i].releaseExclusiveLock();
1231          }
1232          break;
1233        }
1234      }
1235    }
1236
1237    if (!hasLock && !procedure.hasParent()) {
1238      releaseTableSharedLock(procedure, table);
1239    }
1240    return !hasLock;
1241  }

1252  /**
1253   * Wake the procedures waiting for the specified regions
1254   * @param procedure the procedure that was holding the regions
1255   * @param regionInfo the list of regions the procedure was holding
1256   */
1257  public void wakeRegions(final Procedure procedure,final TableName table,
1258      final HRegionInfo... regionInfo) {
1259    Arrays.sort(regionInfo);
1260
1261    final TableQueue queue = getTableQueueWithLock(table);
1262
1263    int numProcs = 0;
1264    final Procedure[] nextProcs = new Procedure[regionInfo.length];
1265    synchronized (queue) {
1266      HRegionInfo prevRegion = null;
1267      for (int i = 0; i < regionInfo.length; ++i) {
1268        assert regionInfo[i].getTable().equals(table);
1269        assert i == 0 || regionInfo[i] != regionInfo[i-1] : "duplicate region: " + regionInfo[i];
1270
1271        RegionEvent event = queue.getRegionEvent(regionInfo[i]);
1272        event.releaseExclusiveLock();
1273        if (event.hasWaitingProcedures()) {
1274          // release one procedure at the time since regions has an xlock
1275          nextProcs[numProcs++] = event.popWaitingProcedure(true);
1276        } else {
1277          queue.removeRegionEvent(event);
1278        }
1279      }
1280    }
1281
1282    // awake procedures if any
1283    schedLock.lock();
1284    try {
1285      for (int i = numProcs - 1; i >= 0; --i) {
1286        wakeProcedure(nextProcs[i]);
1287      }
1288
1289      wakePollIfNeeded(numProcs);
1290
1291      if (!procedure.hasParent()) {
1292        // release the table shared-lock.
1293        // (if we have a parent, it is holding an xlock so we didn't take the shared-lock)
1294        releaseTableSharedLock(procedure, table);
1295      }
1296    } finally {
1297      schedLock.unlock();
1298    }
1299  }

1434    private int sharedLock = 0;
1435
1436    private final TKey key;
1437    private final int priority;

1457     * True if the queue is not in the run-queue and it is owned by an event.
1458     */

1461    }

1465      this.suspended = isSuspended;
1466      return true;
1467    }

1469    // ======================================================================
1470    //  Read/Write Locking helpers
1471    // ======================================================================
1472    public synchronized boolean isLocked() {
1473      return hasExclusiveLock() || sharedLock > 0;
1474    }
1475
1476    public synchronized boolean hasExclusiveLock() {
1477      return this.exclusiveLockProcIdOwner != Long.MIN_VALUE;
1478    }
1479
1480    public synchronized boolean trySharedLock() {
1481      if (hasExclusiveLock()) return false;
1482      sharedLock++;
1483      return true;
1484    }
1485
1486    public synchronized void releaseSharedLock() {
1487      sharedLock--;
1488    }
1489
1490    protected synchronized boolean isSingleSharedLock() {
1491      return sharedLock == 1;
1492    }
1493
1494    public synchronized boolean isLockOwner(long procId) {
1495      return exclusiveLockProcIdOwner == procId;
1496    }
1497
1498    public synchronized boolean hasParentLock(final Procedure proc) {
1499      return proc.hasParent() && isLockOwner(proc.getParentProcId());
1500    }
1501
1502    public synchronized boolean hasLockAccess(final Procedure proc) {
1503      return isLockOwner(proc.getProcId()) || hasParentLock(proc);
1504    }
1505
1506    public synchronized boolean tryExclusiveLock(final Procedure proc) {
1507      if (isLocked()) return hasLockAccess(proc);
1508      exclusiveLockProcIdOwner = proc.getProcId();
1509      return true;
1510    }
1511
1512    public synchronized void releaseExclusiveLock() {
1513      exclusiveLockProcIdOwner = Long.MIN_VALUE;
1514    }
1515
1516    // This should go away when we have the new AM and its events
1517    // and we move xlock to the lock-event-queue.
1518    public synchronized boolean isAvailable() {
1519      return !hasExclusiveLock() && !isEmpty();
1520    }

1590  private static class FairQueue<T extends Comparable<T>> {
1591    private final int quantum;
1592
1593    private Queue<T> currentQueue = null;
1594    private Queue<T> queueHead = null;
1595    private int currentQuantum = 0;
1596    private int size = 0;
1597
1598    public FairQueue() {
1599      this(1);
1600    }
1601
1602    public FairQueue(int quantum) {
1603      this.quantum = quantum;
1604    }
1605
1606    public boolean hasRunnables() {
1607      return size > 0;
1608    }
1609
1610    public void add(Queue<T> queue) {
1611      queueHead = AvlIterableList.append(queueHead, queue);
1612      if (currentQueue == null) setNextQueue(queueHead);
1613      size++;
1614    }
1615
1616    public void remove(Queue<T> queue) {
1617      Queue<T> nextQueue = AvlIterableList.readNext(queue);
1618      queueHead = AvlIterableList.remove(queueHead, queue);
1619      if (currentQueue == queue) {
1620        setNextQueue(queueHead != null ? nextQueue : null);
1621      }
1622      size--;
1623    }
1624
1625    public Queue<T> poll() {
1626      if (currentQuantum == 0) {
1627        if (!nextQueue()) {
1628          return null; // nothing here
1629        }
1630        currentQuantum = calculateQuantum(currentQueue) - 1;
1631      } else {
1632        currentQuantum--;
1633      }
1634
1635      // This should go away when we have the new AM and its events
1636      if (!currentQueue.isAvailable()) {
1637        Queue<T> lastQueue = currentQueue;
1638        do {
1639          if (!nextQueue())
1640            return null;
1641        } while (currentQueue != lastQueue && !currentQueue.isAvailable());
1642
1643        currentQuantum = calculateQuantum(currentQueue) - 1;
1644      }
1645      return currentQueue;
1646    }
1647
1648    private boolean nextQueue() {
1649      if (currentQueue == null) return false;
1650      currentQueue = AvlIterableList.readNext(currentQueue);
1651      return currentQueue != null;
1652    }
1653
1654    private void setNextQueue(Queue<T> queue) {
1655      currentQueue = queue;
1656      if (queue != null) {
1657        currentQuantum = calculateQuantum(currentQueue);
1658      } else {
1659        currentQuantum = 0;
1660      }
1661    }
1662
1663    private int calculateQuantum(final Queue queue) {
1664      return Math.max(1, queue.getPriority() * quantum); // TODO
1665    }
1666  }
