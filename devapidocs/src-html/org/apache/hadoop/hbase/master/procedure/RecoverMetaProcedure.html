046/**
047 * This procedure recovers meta from prior shutdown/ crash of a server, and brings meta online by
048 * assigning meta region/s. Any place where meta is accessed and requires meta to be online, need to
049 * submit this procedure instead of duplicating steps to recover meta in the code.
050 */
051public class RecoverMetaProcedure
052    extends StateMachineProcedure<MasterProcedureEnv, MasterProcedureProtos.RecoverMetaState>
053    implements TableProcedureInterface {
054  private static final Logger LOG = LoggerFactory.getLogger(RecoverMetaProcedure.class);
055
056  private ServerName failedMetaServer;
057  private boolean shouldSplitWal;
058  private int replicaId;
059
060  private final ProcedurePrepareLatch syncLatch;
061  private HMaster master;
062
063  /**
064   * Call this constructor to queue up a {@link RecoverMetaProcedure} in response to meta
065   * carrying server crash
066   * @param failedMetaServer failed/ crashed region server that was carrying meta
067   * @param shouldSplitLog split log file of meta region
068   */
069  public RecoverMetaProcedure(final ServerName failedMetaServer, final boolean shouldSplitLog) {
070    this(failedMetaServer, shouldSplitLog, null);
071  }
072
073  /**
074   * Constructor with latch, for blocking/ sync usage
075   */
076  public RecoverMetaProcedure(final ServerName failedMetaServer, final boolean shouldSplitLog,
077                              final ProcedurePrepareLatch latch) {
078    this.failedMetaServer = failedMetaServer;
079    this.shouldSplitWal = shouldSplitLog;
080    this.replicaId = RegionInfo.DEFAULT_REPLICA_ID;
081    this.syncLatch = latch;
082  }
083
084  /**
085   * This constructor is also used when deserializing from a procedure store; we'll construct one
086   * of these then call #deserializeStateData(InputStream). Do not use directly.
087   */
088  public RecoverMetaProcedure() {
089    this(null, false);
090  }
091
092  @Override
093  protected Flow executeFromState(MasterProcedureEnv env,
094      MasterProcedureProtos.RecoverMetaState state)
095      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
096    prepare(env);
097
098    if (!isRunRequired()) {
099      LOG.info(this + "; Meta already initialized. Skipping run");
100      return Flow.NO_MORE_STATE;
101    }
102
103    try {
104      switch (state) {
105        case RECOVER_META_SPLIT_LOGS:
106          LOG.info("Start " + this);
107          if (shouldSplitWal) {
108            // TODO: Matteo. We BLOCK here but most important thing to be doing at this moment.
109            if (failedMetaServer != null) {
110              master.getMasterWalManager().splitMetaLog(failedMetaServer);
111            } else {
112              ServerName serverName =
113                  master.getMetaTableLocator().getMetaRegionLocation(master.getZooKeeper());
114              Set<ServerName> previouslyFailedServers =
115                  master.getMasterWalManager().getFailedServersFromLogFolders();
116              if (serverName != null && previouslyFailedServers.contains(serverName)) {
117                master.getMasterWalManager().splitMetaLog(serverName);
118              }
119            }
120          }
121          setNextState(RecoverMetaState.RECOVER_META_ASSIGN_REGIONS);
122          break;
123
124        case RECOVER_META_ASSIGN_REGIONS:
125          RegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(
126              RegionInfoBuilder.FIRST_META_REGIONINFO, this.replicaId);
127
128          AssignProcedure metaAssignProcedure;
129          if (failedMetaServer != null) {
130            LOG.info(this + "; Assigning meta with new plan. previous meta server=" +
131                failedMetaServer);
132            metaAssignProcedure = master.getAssignmentManager().createAssignProcedure(hri);
133          } else {
134            // get server carrying meta from zk
135            ServerName metaServer =
136                MetaTableLocator.getMetaRegionState(master.getZooKeeper()).getServerName();
137            LOG.info(this + "; Retaining meta assignment to server=" + metaServer);
138            metaAssignProcedure =
139                master.getAssignmentManager().createAssignProcedure(hri, metaServer);
140          }
141
142          addChildProcedure(metaAssignProcedure);
143          return Flow.NO_MORE_STATE;
144
145        default:
146          throw new UnsupportedOperationException("unhandled state=" + state);
147      }
148    } catch (IOException|KeeperException e) {
149      LOG.warn(this + "; Failed state=" + state + ", retry " + this + "; cycles=" +
150          getCycles(), e);
151    }
152    return Flow.HAS_MORE_STATE;
153  }
154
155  @Override
156  protected void rollbackState(MasterProcedureEnv env,
157      MasterProcedureProtos.RecoverMetaState recoverMetaState)
158      throws IOException, InterruptedException {
159    // Can't rollback
160    throw new UnsupportedOperationException("unhandled state=" + recoverMetaState);
161  }
162
163  @Override
164  protected MasterProcedureProtos.RecoverMetaState getState(int stateId) {
165    return RecoverMetaState.forNumber(stateId);
166  }
167
168  @Override
169  protected int getStateId(MasterProcedureProtos.RecoverMetaState recoverMetaState) {
170    return recoverMetaState.getNumber();
171  }
172
173  @Override
174  protected MasterProcedureProtos.RecoverMetaState getInitialState() {
175    return RecoverMetaState.RECOVER_META_SPLIT_LOGS;
176  }
177
178  @Override
179  protected void toStringClassDetails(StringBuilder sb) {
180    sb.append(getClass().getSimpleName());
181    sb.append(" failedMetaServer=");
182    sb.append(failedMetaServer);
183    sb.append(", splitWal=");
184    sb.append(shouldSplitWal);
185  }
186
187  @Override
188  protected void serializeStateData(ProcedureStateSerializer serializer)
189      throws IOException {
190    super.serializeStateData(serializer);
191    MasterProcedureProtos.RecoverMetaStateData.Builder state =
192        MasterProcedureProtos.RecoverMetaStateData.newBuilder().setShouldSplitWal(shouldSplitWal);
193    if (failedMetaServer != null) {
194      state.setFailedMetaServer(ProtobufUtil.toServerName(failedMetaServer));
195    }
196    state.setReplicaId(replicaId);
197    serializer.serialize(state.build());
198  }
199
200  @Override
201  protected void deserializeStateData(ProcedureStateSerializer serializer)
202      throws IOException {
203    super.deserializeStateData(serializer);
204    MasterProcedureProtos.RecoverMetaStateData state =
205        serializer.deserialize(MasterProcedureProtos.RecoverMetaStateData.class);
206    this.shouldSplitWal = state.hasShouldSplitWal() && state.getShouldSplitWal();
207    this.failedMetaServer = state.hasFailedMetaServer() ?
208        ProtobufUtil.toServerName(state.getFailedMetaServer()) : null;
209    this.replicaId = state.hasReplicaId() ? state.getReplicaId() : RegionInfo.DEFAULT_REPLICA_ID;
210  }
211
212  @Override
213  protected LockState acquireLock(MasterProcedureEnv env) {
214    if (env.getProcedureScheduler().waitTableExclusiveLock(this, TableName.META_TABLE_NAME)) {
215      return LockState.LOCK_EVENT_WAIT;
216    }
217    return LockState.LOCK_ACQUIRED;
218  }
219
220  @Override
221  protected void releaseLock(MasterProcedureEnv env) {
222    env.getProcedureScheduler().wakeTableExclusiveLock(this, TableName.META_TABLE_NAME);
223  }
224
225  @Override
226  protected void completionCleanup(MasterProcedureEnv env) {
227    ProcedurePrepareLatch.releaseLatch(syncLatch, this);
228  }
229
230  @Override
231  public TableName getTableName() {
232    return TableName.META_TABLE_NAME;
233  }
234
235  @Override
236  public TableOperationType getTableOperationType() {
237    return TableOperationType.ENABLE;
238  }
239
240  /**
241   * @return true if failedMetaServer is not null (meta carrying server crashed) or meta is
242   * already initialized
243   */
244  private boolean isRunRequired() {
245    return failedMetaServer != null || !master.getAssignmentManager().isMetaInitialized();
246  }
247
248  /**
249   * Prepare for execution
250   */
251  private void prepare(MasterProcedureEnv env) {
252    if (master == null) {
253      master = (HMaster) env.getMasterServices();
254      Preconditions.checkArgument(master != null);
255    }
256  }
