019package org.apache.hadoop.hbase.master.procedure;
020
021import java.io.IOException;
022
023import org.apache.hadoop.fs.Path;
024import org.apache.hadoop.hbase.DoNotRetryIOException;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.procedure2.NoNodeDispatchException;
027import org.apache.hadoop.hbase.procedure2.NoServerDispatchException;
028import org.apache.hadoop.hbase.procedure2.NullTargetServerDispatchException;
029import org.apache.hadoop.hbase.procedure2.Procedure;
030import org.apache.hadoop.hbase.procedure2.ProcedureEvent;
031import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
032import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;
033import org.apache.hadoop.hbase.procedure2.ProcedureYieldException;
034import org.apache.hadoop.hbase.procedure2.RemoteProcedureDispatcher;
035import org.apache.hadoop.hbase.procedure2.RemoteProcedureException;
036import org.apache.hadoop.hbase.regionserver.SplitWALCallable;
037import org.apache.hadoop.hbase.wal.AbstractFSWALProvider;
038import org.apache.yetus.audience.InterfaceAudience;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
043import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos;
044/**
045 * A remote procedure which is used to send split WAL request to region server.
046 * it will return null if the task is succeed or return a DoNotRetryIOException
047 * {@link SplitWALProcedure} will help handle the situation that encounter
048 * DoNotRetryIOException. Otherwise it will retry until succeed.
049 */
050@InterfaceAudience.Private
051public class SplitWALRemoteProcedure extends Procedure<MasterProcedureEnv>
052    implements RemoteProcedureDispatcher.RemoteProcedure<MasterProcedureEnv, ServerName>,
053    ServerProcedureInterface {
054  private static final Logger LOG = LoggerFactory.getLogger(SplitWALRemoteProcedure.class);
055  private String walPath;
056  private ServerName worker;
057  private ServerName crashedServer;
058  private boolean dispatched;
059  private ProcedureEvent<?> event;
060  private boolean success = false;
061
062  public SplitWALRemoteProcedure() {
063  }
064
065  public SplitWALRemoteProcedure(ServerName worker, ServerName crashedServer, String wal) {
066    this.worker = worker;
067    this.crashedServer = crashedServer;
068    this.walPath = wal;
069  }
070
071  @Override
072  protected Procedure<MasterProcedureEnv>[] execute(MasterProcedureEnv env)
073      throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {
074    if (dispatched) {
075      if (success) {
076        return null;
077      }
078      dispatched = false;
079    }
080    try {
081      env.getRemoteDispatcher().addOperationToNode(worker, this);
082    } catch (NoNodeDispatchException | NullTargetServerDispatchException
083        | NoServerDispatchException e) {
084      // When send to a wrong target server, it need construct a new SplitWALRemoteProcedure.
085      // Thus return null for this procedure and let SplitWALProcedure to handle this.
086      LOG.warn("dispatch WAL {} to {} failed, will retry on another server", walPath, worker, e);
087      return null;
088    }
089    dispatched = true;
090    event = new ProcedureEvent<>(this);
091    event.suspendIfNotReady(this);
092    throw new ProcedureSuspendedException();
093  }
094
095  @Override
096  protected void rollback(MasterProcedureEnv env) throws IOException, InterruptedException {
097    throw new UnsupportedOperationException();
098  }
099
100  @Override
101  protected boolean abort(MasterProcedureEnv env) {
102    return false;
103  }
104
105  @Override
106  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
107    MasterProcedureProtos.SplitWALRemoteData.Builder builder =
108        MasterProcedureProtos.SplitWALRemoteData.newBuilder();
109    builder.setWalPath(walPath).setWorker(ProtobufUtil.toServerName(worker))
110        .setCrashedServer(ProtobufUtil.toServerName(crashedServer));
111    serializer.serialize(builder.build());
112  }
113
114  @Override
115  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
116    MasterProcedureProtos.SplitWALRemoteData data =
117        serializer.deserialize(MasterProcedureProtos.SplitWALRemoteData.class);
118    walPath = data.getWalPath();
119    worker = ProtobufUtil.toServerName(data.getWorker());
120    crashedServer = ProtobufUtil.toServerName(data.getCrashedServer());
121  }
122
123  @Override
124  public RemoteProcedureDispatcher.RemoteOperation remoteCallBuild(MasterProcedureEnv env,
125      ServerName serverName) {
126    return new RSProcedureDispatcher.ServerOperation(this, getProcId(), SplitWALCallable.class,
127        MasterProcedureProtos.SplitWALParameter.newBuilder().setWalPath(walPath).build()
128            .toByteArray());
129  }
130
131  @Override
132  public void remoteCallFailed(MasterProcedureEnv env, ServerName serverName,
133      IOException exception) {
134    complete(env, exception);
135  }
136
137  @Override
138  public void remoteOperationCompleted(MasterProcedureEnv env) {
139    complete(env, null);
140  }
141
142  private void complete(MasterProcedureEnv env, Throwable error) {
143    if (event == null) {
144      LOG.warn("procedure event for {} is null, maybe the procedure is created when recovery",
145        getProcId());
146      return;
147    }
148    if (error == null) {
149      LOG.info("split WAL {} on {} succeeded", walPath, worker);
150      try {
151        env.getMasterServices().getSplitWALManager().deleteSplitWAL(walPath);
152      } catch (IOException e){
153        LOG.warn("remove WAL {} failed, ignore...", walPath, e);
154      }
155      success = true;
156    } else {
157      if (error instanceof DoNotRetryIOException) {
158        LOG.warn("WAL split task of {} send to a wrong server {}, will retry on another server",
159          walPath, worker, error);
160        success = true;
161      } else {
162        LOG.warn("split WAL {} failed, retry...", walPath, error);
163        success = false;
164      }
165
166    }
167    event.wake(env.getProcedureScheduler());
168    event = null;
169  }
170
171  @Override
172  public void remoteOperationFailed(MasterProcedureEnv env, RemoteProcedureException error) {
173    complete(env, error);
174  }
175
176  public String getWAL() {
177    return this.walPath;
178  }
179
180  @Override
181  public ServerName getServerName() {
182    // return the crashed server is to use the queue of root ServerCrashProcedure
183    return this.crashedServer;
184  }
185
186  @Override
187  public boolean hasMetaTableRegion() {
188    return AbstractFSWALProvider.isMetaFile(new Path(walPath));
189  }
190
191  @Override
192  public ServerOperationType getServerOperationType() {
193    return ServerOperationType.SPLIT_WAL_REMOTE;
194  }
195}
