041/**
042 * ProcedureRunnableSet for the Master Procedures.
043 * This RunnableSet tries to provide to the ProcedureExecutor procedures
044 * that can be executed without having to wait on a lock.
045 * Most of the master operations can be executed concurrently, if they
046 * are operating on different tables (e.g. two create table can be performed
047 * at the same, time assuming table A and table B) or against two different servers; say
048 * two servers that crashed at about the same time.
049 *
050 * <p>Each procedure should implement an interface providing information for this queue.
051 * for example table related procedures should implement TableProcedureInterface.
052 * each procedure will be pushed in its own queue, and based on the operation type
053 * we may take smarter decision. e.g. we can abort all the operations preceding
054 * a delete table, or similar.
055 */
056@InterfaceAudience.Private
057@InterfaceStability.Evolving
058public class MasterProcedureScheduler implements ProcedureRunnableSet {
059  private static final Log LOG = LogFactory.getLog(MasterProcedureScheduler.class);
060

062  private final ReentrantLock schedLock = new ReentrantLock();
063  private final Condition schedWaitCond = schedLock.newCondition();

070  private Queue<String> namespaceMap = null;

105  private void doAdd(final Procedure proc, final boolean addFront) {
106    schedLock.lock();
107    try {
108      if (isTableProcedure(proc)) {
109        doAdd(tableRunQueue, getTableQueue(getTableName(proc)), proc, addFront);
110      } else if (isServerProcedure(proc)) {
111        doAdd(serverRunQueue, getServerQueue(getServerName(proc)), proc, addFront);
112      } else {
113        // TODO: at the moment we only have Table and Server procedures
114        // if you are implementing a non-table/non-server procedure, you have two options: create
115        // a group for all the non-table/non-server procedures or try to find a key for your
116        // non-table/non-server procedures and implement something similar to the TableRunQueue.
117        throw new UnsupportedOperationException(
118          "RQs for non-table/non-server procedures are not implemented yet");
119      }
120      schedWaitCond.signal();
121    } finally {
122      schedLock.unlock();
123    }
124  }
125
126  private <T extends Comparable<T>> void doAdd(final FairQueue<T> fairq,
127      final Queue<T> queue, final Procedure proc, final boolean addFront) {
128    queue.add(proc, addFront);
129    if (!(queue.isSuspended() || queue.hasExclusiveLock())) {
130      if (queue.size() == 1 && !IterableList.isLinked(queue)) {
131        fairq.add(queue);
132      }
133      queueSize++;
134    }
135  }

142  @edu.umd.cs.findbugs.annotations.SuppressWarnings("WA_AWAIT_NOT_IN_LOOP")
143  Procedure poll(long waitNsec) {
144    Procedure pollResult = null;
145    schedLock.lock();
146    try {
147      if (queueSize == 0) {
148        if (waitNsec < 0) {
149          schedWaitCond.await();
150        } else {
151          schedWaitCond.awaitNanos(waitNsec);
152        }
153        if (queueSize == 0) {
154          return null;
155        }
156      }
157
158      // For now, let server handling have precedence over table handling; presumption is that it
159      // is more important handling crashed servers than it is running the
160      // enabling/disabling tables, etc.
161      pollResult = doPoll(serverRunQueue);
162      if (pollResult == null) {
163        pollResult = doPoll(tableRunQueue);
164      }
165
166      // update metrics
167      pollCalls++;
168      nullPollCalls += (pollResult == null) ? 1 : 0;
169    } catch (InterruptedException e) {
170      Thread.currentThread().interrupt();
171    } finally {
172      schedLock.unlock();
173    }
174    return pollResult;
175  }
176
177  private <T extends Comparable<T>> Procedure doPoll(final FairQueue<T> fairq) {
178    Queue<T> rq = fairq.poll();
179    if (rq == null || !rq.isAvailable()) {
180      return null;
181    }
182
183    assert !rq.isSuspended() : "rq=" + rq + " is suspended";
184    Procedure pollResult = rq.poll();
185    this.queueSize--;
186    if (rq.isEmpty() || rq.requireExclusiveLock(pollResult)) {
187      removeFromRunQueue(fairq, rq);
188    }
189    return pollResult;
190  }

192  @Override
193  public void clear() {
194    // NOTE: USED ONLY FOR TESTING
195    schedLock.lock();
196    try {
197      // Remove Servers
198      for (int i = 0; i < serverBuckets.length; ++i) {
199        clear((ServerQueue)serverBuckets[i], serverRunQueue);
200        serverBuckets[i] = null;
201      }
202
203      // Remove Tables
204      clear(tableMap, tableRunQueue);
205      tableMap = null;
206
207      assert queueSize == 0 : "expected queue size to be 0, got " + queueSize;
208    } finally {
209      schedLock.unlock();
210    }
211  }
212
213  private <T extends Comparable<T>> void clear(Queue<T> treeMap, FairQueue<T> fairq) {
214    while (treeMap != null) {
215      Queue<T> node = AvlTree.getFirst(treeMap);
216      assert !node.isSuspended() : "can't clear suspended " + node.getKey();
217      treeMap = AvlTree.remove(treeMap, node.getKey());
218      removeFromRunQueue(fairq, node);
219    }
220  }

242  @Override
243  public void completionCleanup(Procedure proc) {
244    if (proc instanceof TableProcedureInterface) {
245      TableProcedureInterface iProcTable = (TableProcedureInterface)proc;
246      boolean tableDeleted;
247      if (proc.hasException()) {
248        IOException procEx =  proc.getException().unwrapRemoteException();
249        if (iProcTable.getTableOperationType() == TableOperationType.CREATE) {
250          // create failed because the table already exist
251          tableDeleted = !(procEx instanceof TableExistsException);
252        } else {
253          // the operation failed because the table does not exist
254          tableDeleted = (procEx instanceof TableNotFoundException);
255        }
256      } else {
257        // the table was deleted
258        tableDeleted = (iProcTable.getTableOperationType() == TableOperationType.DELETE);
259      }
260      if (tableDeleted) {
261        markTableAsDeleted(iProcTable.getTableName());
262        return;
263      }
264    } else {
265      // No cleanup for ServerProcedureInterface types, yet.
266      return;
267    }
268  }

270  private <T extends Comparable<T>> void addToRunQueue(FairQueue<T> fairq, Queue<T> queue) {
271    if (IterableList.isLinked(queue)) return;
272    if (!queue.isEmpty())  {
273      fairq.add(queue);
274      queueSize += queue.size();
275    }
276  }
277
278  private <T extends Comparable<T>> void removeFromRunQueue(FairQueue<T> fairq, Queue<T> queue) {
279    if (!IterableList.isLinked(queue)) return;
280    fairq.remove(queue);
281    queueSize -= queue.size();
282  }

302  public boolean waitEvent(ProcedureEvent event, Procedure procedure, boolean suspendQueue) {
303    synchronized (event) {
304      if (event.isReady()) {
305        return false;
306      }
307
308      // TODO: Suspend single procedure not implemented yet, fallback to suspending the queue
309      if (!suspendQueue) suspendQueue = true;
310
311      if (isTableProcedure(procedure)) {
312        waitTableEvent(event, procedure, suspendQueue);
313      } else if (isServerProcedure(procedure)) {
314        waitServerEvent(event, procedure, suspendQueue);
315      } else {
316        // TODO: at the moment we only have Table and Server procedures
317        // if you are implementing a non-table/non-server procedure, you have two options: create
318        // a group for all the non-table/non-server procedures or try to find a key for your
319        // non-table/non-server procedures and implement something similar to the TableRunQueue.
320        throw new UnsupportedOperationException(
321          "RQs for non-table/non-server procedures are not implemented yet");
322      }
323    }
324    return true;
325  }
326
327  private void waitTableEvent(ProcedureEvent event, Procedure procedure, boolean suspendQueue) {
328    final TableName tableName = getTableName(procedure);
329    final boolean isDebugEnabled = LOG.isDebugEnabled();
330
331    schedLock.lock();
332    try {
333      TableQueue queue = getTableQueue(tableName);
334      if (queue.isSuspended()) return;
335
336      // TODO: if !suspendQueue
337
338      if (isDebugEnabled) {
339        LOG.debug("Suspend table queue " + tableName);
340      }
341      queue.setSuspended(true);
342      removeFromRunQueue(tableRunQueue, queue);
343      event.suspendTableQueue(queue);
344    } finally {
345      schedLock.unlock();
346    }
347  }
348
349  private void waitServerEvent(ProcedureEvent event, Procedure procedure, boolean suspendQueue) {
350    final ServerName serverName = getServerName(procedure);
351    final boolean isDebugEnabled = LOG.isDebugEnabled();
352
353    schedLock.lock();
354    try {
355      // TODO: This will change once we have the new AM
356      ServerQueue queue = getServerQueue(serverName);
357      if (queue.isSuspended()) return;
358
359      // TODO: if !suspendQueue
360
361      if (isDebugEnabled) {
362        LOG.debug("Suspend server queue " + serverName);
363      }
364      queue.setSuspended(true);
365      removeFromRunQueue(serverRunQueue, queue);
366      event.suspendServerQueue(queue);
367    } finally {
368      schedLock.unlock();
369    }
370  }

372  public void suspend(ProcedureEvent event) {
373    final boolean isDebugEnabled = LOG.isDebugEnabled();
374    synchronized (event) {
375      event.setReady(false);
376      if (isDebugEnabled) {
377        LOG.debug("Suspend event " + event);
378      }
379    }
380  }
381
382  public void wake(ProcedureEvent event) {
383    final boolean isDebugEnabled = LOG.isDebugEnabled();
384    synchronized (event) {
385      event.setReady(true);
386      if (isDebugEnabled) {
387        LOG.debug("Wake event " + event);
388      }
389
390      schedLock.lock();
391      try {
392        while (event.hasWaitingTables()) {
393          Queue<TableName> queue = event.popWaitingTable();
394          addToRunQueue(tableRunQueue, queue);
395        }
396        // TODO: This will change once we have the new AM
397        while (event.hasWaitingServers()) {
398          Queue<ServerName> queue = event.popWaitingServer();
399          addToRunQueue(serverRunQueue, queue);
400        }
401
402        if (queueSize > 1) {
403          schedWaitCond.signalAll();
404        } else if (queueSize > 0) {
405          schedWaitCond.signal();
406        }
407      } finally {
408        schedLock.unlock();
409      }
410    }
411  }

413  public static class ProcedureEvent {
414    private final String description;
415
416    private Queue<ServerName> waitingServers = null;
417    private Queue<TableName> waitingTables = null;
418    private boolean ready = false;
419
420    public ProcedureEvent(String description) {
421      this.description = description;
422    }
423
424    public synchronized boolean isReady() {
425      return ready;
426    }
427
428    private synchronized void setReady(boolean isReady) {
429      this.ready = isReady;
430    }
431
432    private void suspendTableQueue(Queue<TableName> queue) {
433      waitingTables = IterableList.append(waitingTables, queue);
434    }
435
436    private void suspendServerQueue(Queue<ServerName> queue) {
437      waitingServers = IterableList.append(waitingServers, queue);
438    }
439
440    private boolean hasWaitingTables() {
441      return waitingTables != null;
442    }
443
444    private Queue<TableName> popWaitingTable() {
445      Queue<TableName> node = waitingTables;
446      waitingTables = IterableList.remove(waitingTables, node);
447      node.setSuspended(false);
448      return node;
449    }
450
451    private boolean hasWaitingServers() {
452      return waitingServers != null;
453    }
454
455    private Queue<ServerName> popWaitingServer() {
456      Queue<ServerName> node = waitingServers;
457      waitingServers = IterableList.remove(waitingServers, node);
458      node.setSuspended(false);
459      return node;
460    }
461
462    @Override
463    public String toString() {
464      return String.format("ProcedureEvent(%s)", description);
465    }
466  }

481    Queue<TableName> node = AvlTree.get(tableMap, tableName);

485    node = new TableQueue(tableName, nsQueue, getTablePriority(tableName));
486    tableMap = AvlTree.insert(tableMap, node);
487    return (TableQueue)node;
488  }

535  private ServerQueue getServerQueue(ServerName serverName) {
536    int index = getBucketIndex(serverBuckets, serverName.hashCode());
537    Queue<ServerName> root = getTreeRoot(serverBuckets, index);
538    Queue<ServerName> node = AvlTree.get(root, serverName);
539    if (node != null) return (ServerQueue)node;
540
541    node = new ServerQueue(serverName);
542    serverBuckets[index] = AvlTree.insert(root, node);
543    return (ServerQueue)node;
544  }

579        case CRASH_HANDLER:

588  public static class TableQueue extends QueueImpl<TableName> {
589    private final NamespaceQueue namespaceQueue;
590
591    private TableLock tableLock = null;
592
593    public TableQueue(TableName tableName, NamespaceQueue namespaceQueue, int priority) {
594      super(tableName, priority);
595      this.namespaceQueue = namespaceQueue;
596    }
597
598    public NamespaceQueue getNamespaceQueue() {
599      return namespaceQueue;
600    }
601
602    @Override
603    public synchronized boolean isAvailable() {
604      return super.isAvailable() && !namespaceQueue.hasExclusiveLock();
605    }
606
607    // TODO: We can abort pending/in-progress operation if the new call is
608    //       something like drop table. We can Override addBack(),
609    //       check the type and abort all the in-flight procedurs.
610    private boolean canAbortPendingOperations(Procedure proc) {
611      TableProcedureInterface tpi = (TableProcedureInterface)proc;
612      switch (tpi.getTableOperationType()) {
613        case DELETE:
614          return true;
615        default:
616          return false;
617      }
618    }
619
620    public boolean requireExclusiveLock(Procedure proc) {
621      TableProcedureInterface tpi = (TableProcedureInterface)proc;
622      switch (tpi.getTableOperationType()) {
623        case CREATE:
624        case DELETE:
625        case DISABLE:
626        case ENABLE:
627          return true;
628        case EDIT:
629          // we allow concurrent edit on the NS table
630          return !tpi.getTableName().equals(TableName.NAMESPACE_TABLE_NAME);
631        case READ:
632          return false;
633        default:
634          break;
635      }
636      throw new UnsupportedOperationException("unexpected type " + tpi.getTableOperationType());
637    }
638
639    private synchronized boolean tryZkSharedLock(final TableLockManager lockManager,
640        final String purpose) {
641      // Take zk-read-lock
642      TableName tableName = getKey();
643      tableLock = lockManager.readLock(tableName, purpose);
644      try {
645        tableLock.acquire();
646      } catch (IOException e) {
647        LOG.error("failed acquire read lock on " + tableName, e);
648        tableLock = null;
649        return false;
650      }
651      return true;
652    }
653
654    private synchronized void releaseZkSharedLock(final TableLockManager lockManager) {
655      releaseTableLock(lockManager, isSingleSharedLock());
656    }
657
658    private synchronized boolean tryZkExclusiveLock(final TableLockManager lockManager,
659        final String purpose) {
660      // Take zk-write-lock
661      TableName tableName = getKey();
662      tableLock = lockManager.writeLock(tableName, purpose);
663      try {
664        tableLock.acquire();
665      } catch (IOException e) {
666        LOG.error("failed acquire write lock on " + tableName, e);
667        tableLock = null;
668        return false;
669      }
670      return true;
671    }
672
673    private synchronized void releaseZkExclusiveLock(final TableLockManager lockManager) {
674      releaseTableLock(lockManager, true);
675    }
676
677    private void releaseTableLock(final TableLockManager lockManager, boolean reset) {
678      for (int i = 0; i < 3; ++i) {
679        try {
680          tableLock.release();
681          if (reset) {
682            tableLock = null;
683          }
684          break;
685        } catch (IOException e) {
686          LOG.warn("Could not release the table write-lock", e);
687        }
688      }
689    }
690  }

731  /**
732   * Try to acquire the exclusive lock on the specified table.
733   * other operations in the table-queue will be executed after the lock is released.
734   * @param table Table to lock
735   * @param purpose Human readable reason for locking the table
736   * @return true if we were able to acquire the lock on the table, otherwise false.
737   */
738  public boolean tryAcquireTableExclusiveLock(final TableName table, final String purpose) {
739    schedLock.lock();
740    TableQueue queue = getTableQueue(table);
741    if (!queue.getNamespaceQueue().trySharedLock()) {
742      return false;
743    }
744
745    if (!queue.tryExclusiveLock()) {
746      queue.getNamespaceQueue().releaseSharedLock();
747      schedLock.unlock();
748      return false;
749    }
750
751    removeFromRunQueue(tableRunQueue, queue);
752    schedLock.unlock();
753
754    // Zk lock is expensive...
755    boolean hasXLock = queue.tryZkExclusiveLock(lockManager, purpose);
756    if (!hasXLock) {
757      schedLock.lock();
758      queue.releaseExclusiveLock();
759      queue.getNamespaceQueue().releaseSharedLock();
760      addToRunQueue(tableRunQueue, queue);
761      schedLock.unlock();
762    }
763    return hasXLock;
764  }

768   * @param table the name of the table that has the exclusive lock

785  /**
786   * Try to acquire the shared lock on the specified table.
787   * other "read" operations in the table-queue may be executed concurrently,
788   * @param table Table to lock
789   * @param purpose Human readable reason for locking the table
790   * @return true if we were able to acquire the lock on the table, otherwise false.
791   */
792  public boolean tryAcquireTableSharedLock(final TableName table, final String purpose) {
793    schedLock.lock();
794    TableQueue queue = getTableQueue(table);
795    if (!queue.getNamespaceQueue().trySharedLock()) {
796      return false;
797    }
798
799    if (!queue.trySharedLock()) {
800      queue.getNamespaceQueue().releaseSharedLock();
801      schedLock.unlock();
802      return false;
803    }
804
805    schedLock.unlock();
806
807    // Zk lock is expensive...
808    boolean hasXLock = queue.tryZkSharedLock(lockManager, purpose);
809    if (!hasXLock) {
810      schedLock.lock();
811      queue.releaseSharedLock();
812      queue.getNamespaceQueue().releaseSharedLock();
813      schedLock.unlock();
814    }
815    return hasXLock;
816  }

836  /**
837   * Tries to remove the queue and the table-lock of the specified table.
838   * If there are new operations pending (e.g. a new create),
839   * the remove will not be performed.
840   * @param table the name of the table that should be marked as deleted
841   * @return true if deletion succeeded, false otherwise meaning that there are
842   *     other new operations pending for that table (e.g. a new create).
843   */
844  protected boolean markTableAsDeleted(final TableName table) {
845    final ReentrantLock l = schedLock;
846    l.lock();
847    try {
848      TableQueue queue = getTableQueue(table);
849      if (queue == null) return true;
850
851      if (queue.isEmpty() && queue.acquireDeleteLock()) {
852        // remove the table from the run-queue and the map
853        if (IterableList.isLinked(queue)) {
854          tableRunQueue.remove(queue);
855        }
856
857        // Remove the table lock
858        try {
859          lockManager.tableDeleted(table);
860        } catch (IOException e) {
861          LOG.warn("Received exception from TableLockManager.tableDeleted:", e); //not critical
862        }
863
864        removeTableQueue(table);
865      } else {
866        // TODO: If there are no create, we can drop all the other ops
867        return false;
868      }
869    } finally {
870      l.unlock();
871    }
872    return true;
873  }

888      if (!tableQueue.trySharedLock()) return false;
889
890      NamespaceQueue nsQueue = getNamespaceQueue(nsName);
891      boolean hasLock = nsQueue.tryExclusiveLock();
892      if (!hasLock) {

894      }
895      return hasLock;
896    } finally {
897      schedLock.unlock();
898    }

991  private static abstract class Queue<TKey extends Comparable<TKey>> implements QueueInterface {
992    private Queue<TKey> avlRight = null;
993    private Queue<TKey> avlLeft = null;

995
996    private Queue<TKey> iterNext = null;
997    private Queue<TKey> iterPrev = null;
998    private boolean suspended = false;
999
1000    private boolean exclusiveLock = false;
1001    private int sharedLock = 0;
1002
1003    private final TKey key;
1004    private final int priority;
1005
1006    public Queue(TKey key) {
1007      this(key, 1);
1008    }
1009
1010    public Queue(TKey key, int priority) {
1011      this.key = key;
1012      this.priority = priority;
1013    }
1014
1015    protected TKey getKey() {
1016      return key;
1017    }
1018
1019    protected int getPriority() {
1020      return priority;
1021    }
1022
1023    /**
1024     * True if the queue is not in the run-queue and it is owned by an event.
1025     */
1026    public boolean isSuspended() {
1027      return suspended;
1028    }
1029
1030    protected boolean setSuspended(boolean isSuspended) {
1031      if (this.suspended == isSuspended) return false;
1032      this.suspended = isSuspended;
1033      return true;
1034    }

1036    // ======================================================================
1037    //  Read/Write Locking helpers
1038    // ======================================================================
1039    public synchronized boolean isLocked() {
1040      return hasExclusiveLock() || sharedLock > 0;
1041    }
1042
1043    public synchronized boolean hasExclusiveLock() {
1044      return this.exclusiveLock;
1045    }
1046
1047    public synchronized boolean trySharedLock() {
1048      if (hasExclusiveLock()) return false;
1049      sharedLock++;
1050      return true;
1051    }
1052
1053    public synchronized void releaseSharedLock() {
1054      sharedLock--;
1055    }
1056
1057    protected synchronized boolean isSingleSharedLock() {
1058      return sharedLock == 1;
1059    }
1060
1061    public synchronized boolean tryExclusiveLock() {
1062      if (isLocked()) return false;
1063      exclusiveLock = true;
1064      return true;
1065    }
1066
1067    public synchronized void releaseExclusiveLock() {
1068      exclusiveLock = false;
1069    }
1070
1071    public synchronized boolean acquireDeleteLock() {
1072      return tryExclusiveLock();
1073    }
1074
1075    // This should go away when we have the new AM and its events
1076    // and we move xlock to the lock-event-queue.
1077    public synchronized boolean isAvailable() {
1078      return !exclusiveLock && !isEmpty();
1079    }

1143  private static class FairQueue<T extends Comparable<T>> {
1144    private final int quantum;
1145
1146    private Queue<T> currentQueue = null;
1147    private Queue<T> queueHead = null;
1148    private int currentQuantum = 0;
1149
1150    public FairQueue() {
1151      this(1);
1152    }
1153
1154    public FairQueue(int quantum) {
1155      this.quantum = quantum;
1156    }
1157
1158    public void add(Queue<T> queue) {
1159      queueHead = IterableList.append(queueHead, queue);
1160      if (currentQueue == null) setNextQueue(queueHead);
1161    }
1162
1163    public void remove(Queue<T> queue) {
1164      Queue<T> nextQueue = queue.iterNext;
1165      queueHead = IterableList.remove(queueHead, queue);
1166      if (currentQueue == queue) {
1167        setNextQueue(queueHead != null ? nextQueue : null);
1168      }
1169    }
1170
1171    public Queue<T> poll() {
1172      if (currentQuantum == 0) {
1173        if (!nextQueue()) {
1174          return null; // nothing here
1175        }
1176        currentQuantum = calculateQuantum(currentQueue) - 1;
1177      } else {
1178        currentQuantum--;
1179      }
1180
1181      // This should go away when we have the new AM and its events
1182      if (!currentQueue.isAvailable()) {
1183        Queue<T> lastQueue = currentQueue;
1184        do {
1185          if (!nextQueue())
1186            return null;
1187        } while (currentQueue != lastQueue && !currentQueue.isAvailable());
1188
1189        currentQuantum = calculateQuantum(currentQueue) - 1;
1190      }
1191      return currentQueue;
1192    }
1193
1194    private boolean nextQueue() {
1195      if (currentQueue == null) return false;
1196      currentQueue = currentQueue.iterNext;
1197      return currentQueue != null;
1198    }
1199
1200    private void setNextQueue(Queue<T> queue) {
1201      currentQueue = queue;
1202      if (queue != null) {
1203        currentQuantum = calculateQuantum(currentQueue);
1204      } else {
1205        currentQuantum = 0;
1206      }
1207    }
1208
1209    private int calculateQuantum(final Queue queue) {
1210      return Math.max(1, queue.getPriority() * quantum); // TODO
1211    }
1212  }

1214  private static class AvlTree {
1215    public static <T extends Comparable<T>> Queue<T> get(Queue<T> root, T key) {
1216      while (root != null) {
1217        int cmp = root.compareKey(key);
1218        if (cmp > 0) {
1219          root = root.avlLeft;
1220        } else if (cmp < 0) {
1221          root = root.avlRight;
1222        } else {
1223          return root;
1224        }
1225      }
1226      return null;
1227    }
1228
1229    public static <T extends Comparable<T>> Queue<T> getFirst(Queue<T> root) {
1230      if (root != null) {
1231        while (root.avlLeft != null) {
1232          root = root.avlLeft;
1233        }
1234      }
1235      return root;
1236    }
1237
1238    public static <T extends Comparable<T>> Queue<T> getLast(Queue<T> root) {
1239      if (root != null) {
1240        while (root.avlRight != null) {
1241          root = root.avlRight;
1242        }
1243      }
1244      return root;
1245    }
1246
1247    public static <T extends Comparable<T>> Queue<T> insert(Queue<T> root, Queue<T> node) {
1248      if (root == null) return node;
1249      if (node.compareTo(root) < 0) {
1250        root.avlLeft = insert(root.avlLeft, node);
1251      } else {
1252        root.avlRight = insert(root.avlRight, node);
1253      }
1254      return balance(root);
1255    }
1256
1257    private static <T extends Comparable<T>> Queue<T> removeMin(Queue<T> p) {
1258      if (p.avlLeft == null)
1259        return p.avlRight;
1260      p.avlLeft = removeMin(p.avlLeft);
1261      return balance(p);
1262    }
1263
1264    public static <T extends Comparable<T>> Queue<T> remove(Queue<T> root, T key) {
1265      if (root == null) return null;
1266
1267      int cmp = root.compareKey(key);
1268      if (cmp == 0) {
1269        Queue<T> q = root.avlLeft;
1270        Queue<T> r = root.avlRight;
1271        if (r == null) return q;
1272        Queue<T> min = getFirst(r);
1273        min.avlRight = removeMin(r);
1274        min.avlLeft = q;
1275        return balance(min);
1276      } else if (cmp > 0) {
1277        root.avlLeft = remove(root.avlLeft, key);
1278      } else /* if (cmp < 0) */ {
1279        root.avlRight = remove(root.avlRight, key);
1280      }
1281      return balance(root);
1282    }
1283
1284    private static <T extends Comparable<T>> Queue<T> balance(Queue<T> p) {
1285      fixHeight(p);
1286      int balance = balanceFactor(p);
1287      if (balance == 2) {
1288        if (balanceFactor(p.avlRight) < 0) {
1289          p.avlRight = rotateRight(p.avlRight);
1290        }
1291        return rotateLeft(p);
1292      } else if (balance == -2) {
1293        if (balanceFactor(p.avlLeft) > 0) {
1294          p.avlLeft = rotateLeft(p.avlLeft);
1295        }
1296        return rotateRight(p);
1297      }
1298      return p;
1299    }
1300
1301    private static <T extends Comparable<T>> Queue<T> rotateRight(Queue<T> p) {
1302      Queue<T> q = p.avlLeft;
1303      p.avlLeft = q.avlRight;
1304      q.avlRight = p;
1305      fixHeight(p);
1306      fixHeight(q);
1307      return q;
1308    }
1309
1310    private static <T extends Comparable<T>> Queue<T> rotateLeft(Queue<T> q) {
1311      Queue<T> p = q.avlRight;
1312      q.avlRight = p.avlLeft;
1313      p.avlLeft = q;
1314      fixHeight(q);
1315      fixHeight(p);
1316      return p;
1317    }
1318
1319    private static <T extends Comparable<T>> void fixHeight(Queue<T> node) {
1320      int heightLeft = height(node.avlLeft);
1321      int heightRight = height(node.avlRight);
1322      node.avlHeight = 1 + Math.max(heightLeft, heightRight);
1323    }
1324
1325    private static <T extends Comparable<T>> int height(Queue<T> node) {
1326      return node != null ? node.avlHeight : 0;
1327    }
1328
1329    private static <T extends Comparable<T>> int balanceFactor(Queue<T> node) {
1330      return height(node.avlRight) - height(node.avlLeft);
1331    }
1332  }

1334  private static class IterableList {
1335    public static <T extends Comparable<T>> Queue<T> prepend(Queue<T> head, Queue<T> node) {
1336      assert !isLinked(node) : node + " is already linked";
1337      if (head != null) {
1338        Queue<T> tail = head.iterPrev;
1339        tail.iterNext = node;
1340        head.iterPrev = node;
1341        node.iterNext = head;
1342        node.iterPrev = tail;
1343      } else {
1344        node.iterNext = node;
1345        node.iterPrev = node;
1346      }
1347      return node;
1348    }
1349
1350    public static <T extends Comparable<T>> Queue<T> append(Queue<T> head, Queue<T> node) {
1351      assert !isLinked(node) : node + " is already linked";
1352      if (head != null) {
1353        Queue<T> tail = head.iterPrev;
1354        tail.iterNext = node;
1355        node.iterNext = head;
1356        node.iterPrev = tail;
1357        head.iterPrev = node;
1358        return head;
1359      }
1360      node.iterNext = node;
1361      node.iterPrev = node;
1362      return node;
1363    }
1364
1365    public static <T extends Comparable<T>> Queue<T> appendList(Queue<T> head, Queue<T> otherHead) {
1366      if (head == null) return otherHead;
1367      if (otherHead == null) return head;
1368
1369      Queue<T> tail = head.iterPrev;
1370      Queue<T> otherTail = otherHead.iterPrev;
1371      tail.iterNext = otherHead;
1372      otherHead.iterPrev = tail;
1373      otherTail.iterNext = head;
1374      head.iterPrev = otherTail;
1375      return head;
1376    }
1377
1378    private static <T extends Comparable<T>> Queue<T> remove(Queue<T> head, Queue<T> node) {
1379      assert isLinked(node) : node + " is not linked";
1380      if (node != node.iterNext) {
1381        node.iterPrev.iterNext = node.iterNext;
1382        node.iterNext.iterPrev = node.iterPrev;
1383        head = (head == node) ? node.iterNext : head;
1384      } else {
1385        head = null;
1386      }
1387      node.iterNext = null;
1388      node.iterPrev = null;
1389      return head;
1390    }
1391
1392    private static <T extends Comparable<T>> boolean isLinked(Queue<T> node) {
1393      return node.iterPrev != null && node.iterNext != null;
1394    }
