056/**
057 * Handle crashed server. This is a port to ProcedureV2 of what used to be euphemistically called
058 * ServerShutdownHandler.
059 *
060 * <p>The procedure flow varies dependent on whether meta is assigned, if we are
061 * doing distributed log replay versus distributed log splitting, and if we are to split logs at
062 * all.
063 *
064 * <p>This procedure asks that all crashed servers get processed equally; we yield after the
065 * completion of each successful flow step. We do this so that we do not 'deadlock' waiting on
066 * a region assignment so we can replay edits which could happen if a region moved there are edits
067 * on two servers for replay.
068 *
069 * <p>TODO: ASSIGN and WAIT_ON_ASSIGN (at least) are not idempotent. Revisit when assign is pv2.
070 * TODO: We do not have special handling for system tables.
071 */
072public class ServerCrashProcedure
073extends StateMachineProcedure<MasterProcedureEnv, ServerCrashState>
074implements ServerProcedureInterface {
075  private static final Log LOG = LogFactory.getLog(ServerCrashProcedure.class);

164  private void throwProcedureYieldException(final String msg) throws ProcedureYieldException {
165    String logMsg = msg + "; cycle=" + this.cycles + ", running for " +
166        StringUtils.formatTimeDiff(System.currentTimeMillis(), getStartTime());
167    // The procedure executor logs ProcedureYieldException at trace level. For now, log these
168    // yields for server crash processing at DEBUG. Revisit when stable.
169    if (LOG.isDebugEnabled()) LOG.debug(logMsg);
170    throw new ProcedureYieldException(logMsg);
171  }

173  @Override
174  protected Flow executeFromState(MasterProcedureEnv env, ServerCrashState state)
175  throws ProcedureYieldException {
176    if (LOG.isTraceEnabled()) {
177      LOG.trace(state);
178    }
179    // Keep running count of cycles
180    if (state.ordinal() != this.previousState) {
181      this.previousState = state.ordinal();
182      this.cycles = 0;
183    } else {
184      this.cycles++;
185    }
186    MasterServices services = env.getMasterServices();
187    // Is master fully online? If not, yield. No processing of servers unless master is up
188    if (!services.getAssignmentManager().isFailoverCleanupDone()) {
189      throwProcedureYieldException("Waiting on master failover to complete");
190    }
191    // HBASE-14802
192    // If we have not yet notified that we are processing a dead server, we should do now.
193    if (!notifiedDeadServer) {
194      services.getServerManager().getDeadServers().notifyServer(serverName);
195      notifiedDeadServer = true;
196    }
197
198    try {
199      switch (state) {
200      case SERVER_CRASH_START:
201        LOG.info("Start processing crashed " + this.serverName);
202        start(env);
203        // If carrying meta, process it first. Else, get list of regions on crashed server.
204        if (this.carryingMeta) setNextState(ServerCrashState.SERVER_CRASH_PROCESS_META);
205        else setNextState(ServerCrashState.SERVER_CRASH_GET_REGIONS);
206        break;
207
208      case SERVER_CRASH_GET_REGIONS:
209        // If hbase:meta is not assigned, yield.
210        if (!isMetaAssignedQuickTest(env)) {
211          throwProcedureYieldException("Waiting on hbase:meta assignment");
212        }
213        this.regionsOnCrashedServer =
214          services.getAssignmentManager().getRegionStates().getServerRegions(this.serverName);
215        // Where to go next? Depends on whether we should split logs at all or if we should do
216        // distributed log splitting (DLS) vs distributed log replay (DLR).
217        if (!this.shouldSplitWal) {
218          setNextState(ServerCrashState.SERVER_CRASH_ASSIGN);
219        } else if (this.distributedLogReplay) {
220          setNextState(ServerCrashState.SERVER_CRASH_PREPARE_LOG_REPLAY);
221        } else {
222          setNextState(ServerCrashState.SERVER_CRASH_SPLIT_LOGS);
223        }
224        break;
225
226      case SERVER_CRASH_PROCESS_META:
227        // If we fail processing hbase:meta, yield.
228        if (!processMeta(env)) {
229          throwProcedureYieldException("Waiting on regions-in-transition to clear");
230        }
231        setNextState(ServerCrashState.SERVER_CRASH_GET_REGIONS);
232        break;
233
234      case SERVER_CRASH_PREPARE_LOG_REPLAY:
235        prepareLogReplay(env, this.regionsOnCrashedServer);
236        setNextState(ServerCrashState.SERVER_CRASH_ASSIGN);
237        break;
238
239      case SERVER_CRASH_SPLIT_LOGS:
240        splitLogs(env);
241        // If DLR, go to FINISH. Otherwise, if DLS, go to SERVER_CRASH_CALC_REGIONS_TO_ASSIGN
242        if (this.distributedLogReplay) setNextState(ServerCrashState.SERVER_CRASH_FINISH);
243        else setNextState(ServerCrashState.SERVER_CRASH_ASSIGN);
244        break;
245
246      case SERVER_CRASH_ASSIGN:
247        List<HRegionInfo> regionsToAssign = calcRegionsToAssign(env);
248
249        // Assign may not be idempotent. SSH used to requeue the SSH if we got an IOE assigning
250        // which is what we are mimicing here but it looks prone to double assignment if assign
251        // fails midway. TODO: Test.
252
253        // If no regions to assign, skip assign and skip to the finish.
254        boolean regions = regionsToAssign != null && !regionsToAssign.isEmpty();
255        if (regions) {
256          this.regionsAssigned = regionsToAssign;
257          if (!assign(env, regionsToAssign)) {
258            throwProcedureYieldException("Failed assign; will retry");
259          }
260        }
261        if (this.shouldSplitWal && distributedLogReplay) {
262          // Take this route even if there are apparently no regions assigned. This may be our
263          // second time through here; i.e. we assigned and crashed just about here. On second
264          // time through, there will be no regions because we assigned them in the previous step.
265          // Even though no regions, we need to go through here to clean up the DLR zk markers.
266          setNextState(ServerCrashState.SERVER_CRASH_WAIT_ON_ASSIGN);
267        } else {
268          setNextState(ServerCrashState.SERVER_CRASH_FINISH);
269        }
270        break;
271
272      case SERVER_CRASH_WAIT_ON_ASSIGN:
273        // TODO: The list of regionsAssigned may be more than we actually assigned. See down in
274        // AM #1629 around 'if (regionStates.wasRegionOnDeadServer(encodedName)) {' where where we
275        // will skip assigning a region because it is/was on a dead server. Should never happen!
276        // It was on this server. Worst comes to worst, we'll still wait here till other server is
277        // processed.
278
279        // If the wait on assign failed, yield -- if we have regions to assign.
280        if (this.regionsAssigned != null && !this.regionsAssigned.isEmpty()) {
281          if (!waitOnAssign(env, this.regionsAssigned)) {
282            throwProcedureYieldException("Waiting on region assign");
283          }
284        }
285        setNextState(ServerCrashState.SERVER_CRASH_SPLIT_LOGS);
286        break;
287
288      case SERVER_CRASH_FINISH:
289        LOG.info("Finished processing of crashed " + serverName);
290        services.getServerManager().getDeadServers().finish(serverName);
291        return Flow.NO_MORE_STATE;
292
293      default:
294          throw new UnsupportedOperationException("unhandled state=" + state);
295      }
296    } catch (IOException e) {
297      LOG.warn("Failed serverName=" + this.serverName + ", state=" + state + "; retry", e);
298    } catch (InterruptedException e) {
299      // TODO: Make executor allow IEs coming up out of execute.
300      LOG.warn("Interrupted serverName=" + this.serverName + ", state=" + state + "; retry", e);
301      Thread.currentThread().interrupt();
302    }
303    return Flow.HAS_MORE_STATE;
304  }

311  private void start(final MasterProcedureEnv env) throws IOException {
312    MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
313    // Set recovery mode late. This is what the old ServerShutdownHandler used do.
314    mfs.setLogRecoveryMode();
315    this.distributedLogReplay = mfs.getLogRecoveryMode() == RecoveryMode.LOG_REPLAY;

318  /**
319   * @param env
320   * @return False if we fail to assign and split logs on meta ('process').
321   * @throws IOException
322   * @throws InterruptedException
323   */
324  private boolean processMeta(final MasterProcedureEnv env)
325  throws IOException {
326    if (LOG.isDebugEnabled()) LOG.debug("Processing hbase:meta that was on " + this.serverName);
327    MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
328    AssignmentManager am = env.getMasterServices().getAssignmentManager();
329    HRegionInfo metaHRI = HRegionInfo.FIRST_META_REGIONINFO;
330    if (this.shouldSplitWal) {
331      if (this.distributedLogReplay) {
332        prepareLogReplay(env, META_REGION_SET);
333      } else {
334        // TODO: Matteo. We BLOCK here but most important thing to be doing at this moment.
335        mfs.splitMetaLog(serverName);
336        am.getRegionStates().logSplit(metaHRI);
337      }
338    }
339
340    // Assign meta if still carrying it. Check again: region may be assigned because of RIT timeout
341    boolean processed = true;
342    if (am.isCarryingMeta(serverName)) {
343      // TODO: May block here if hard time figuring state of meta.
344      am.regionOffline(HRegionInfo.FIRST_META_REGIONINFO);
345      verifyAndAssignMetaWithRetries(env);
346      if (this.shouldSplitWal && distributedLogReplay) {
347        int timeout = env.getMasterConfiguration().getInt(KEY_WAIT_ON_RIT, DEFAULT_WAIT_ON_RIT);
348        if (!waitOnRegionToClearRegionsInTransition(am, metaHRI, timeout)) {
349          processed = false;
350        } else {
351          // TODO: Matteo. We BLOCK here but most important thing to be doing at this moment.
352          mfs.splitMetaLog(serverName);
353        }
354      }
355    }
356    return processed;
357  }

359  /**
360   * @return True if region cleared RIT, else false if we timed out waiting.
361   * @throws InterruptedIOException
362   */
363  private boolean waitOnRegionToClearRegionsInTransition(AssignmentManager am,
364      final HRegionInfo hri, final int timeout)
365  throws InterruptedIOException {
366    try {
367      if (!am.waitOnRegionToClearRegionsInTransition(hri, timeout)) {
368        // Wait here is to avoid log replay hits current dead server and incur a RPC timeout
369        // when replay happens before region assignment completes.
370        LOG.warn("Region " + hri.getEncodedName() + " didn't complete assignment in time");
371        return false;
372      }
373    } catch (InterruptedException ie) {
374      throw new InterruptedIOException("Caught " + ie +
375        " during waitOnRegionToClearRegionsInTransition for " + hri);
376    }
377    return true;
378  }

380  private void prepareLogReplay(final MasterProcedureEnv env, final Set<HRegionInfo> regions)
381  throws IOException {
382    if (LOG.isDebugEnabled()) {
383      LOG.debug("Mark " + size(this.regionsOnCrashedServer) + " regions-in-recovery from " +
384        this.serverName);
385    }
386    MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
387    AssignmentManager am = env.getMasterServices().getAssignmentManager();
388    mfs.prepareLogReplay(this.serverName, regions);
389    am.getRegionStates().logSplit(this.serverName);
390  }

408  /**
409   * Figure out what we need to assign. Should be idempotent.
410   * @param env
411   * @return List of calculated regions to assign; may be empty or null.
412   * @throws IOException
413   */
414  private List<HRegionInfo> calcRegionsToAssign(final MasterProcedureEnv env)
415  throws IOException {
416    AssignmentManager am = env.getMasterServices().getAssignmentManager();
417    List<HRegionInfo> regionsToAssignAggregator = new ArrayList<HRegionInfo>();
418    int replicaCount = env.getMasterConfiguration().getInt(HConstants.META_REPLICAS_NUM,
419      HConstants.DEFAULT_META_REPLICA_NUM);
420    for (int i = 1; i < replicaCount; i++) {
421      HRegionInfo metaHri =
422          RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO, i);
423      if (am.isCarryingMetaReplica(this.serverName, metaHri)) {
424        if (LOG.isDebugEnabled()) {
425          LOG.debug("Reassigning meta replica" + metaHri + " that was on " + this.serverName);
426        }
427        regionsToAssignAggregator.add(metaHri);
428      }
429    }
430    // Clean out anything in regions in transition.
431    List<HRegionInfo> regionsInTransition = am.cleanOutCrashedServerReferences(serverName);
432    if (LOG.isDebugEnabled()) {
433      LOG.debug("Reassigning " + size(this.regionsOnCrashedServer) +
434        " region(s) that " + (serverName == null? "null": serverName)  +
435        " was carrying (and " + regionsInTransition.size() +
436        " regions(s) that were opening on this server)");
437    }
438    regionsToAssignAggregator.addAll(regionsInTransition);
439
440    // Iterate regions that were on this server and figure which of these we need to reassign
441    if (this.regionsOnCrashedServer != null && !this.regionsOnCrashedServer.isEmpty()) {
442      RegionStates regionStates = am.getRegionStates();
443      for (HRegionInfo hri: this.regionsOnCrashedServer) {
444        if (regionsInTransition.contains(hri)) continue;
445        String encodedName = hri.getEncodedName();
446        Lock lock = am.acquireRegionLock(encodedName);
447        try {
448          RegionState rit = regionStates.getRegionTransitionState(hri);
449          if (processDeadRegion(hri, am)) {
450            ServerName addressFromAM = regionStates.getRegionServerOfRegion(hri);
451            if (addressFromAM != null && !addressFromAM.equals(this.serverName)) {
452              // If this region is in transition on the dead server, it must be
453              // opening or pending_open, which should have been covered by
454              // AM#cleanOutCrashedServerReferences
455              LOG.info("Skip assigning " + hri.getRegionNameAsString()
456                + " because opened on " + addressFromAM.getServerName());
457              continue;
458            }
459            if (rit != null) {
460              if (rit.getServerName() != null && !rit.isOnServer(this.serverName)) {
461                // Skip regions that are in transition on other server
462                LOG.info("Skip assigning region in transition on other server" + rit);
463                continue;
464              }
465              LOG.info("Reassigning region " + rit + " and clearing zknode if exists");
466              regionStates.updateRegionState(hri, RegionState.State.OFFLINE);
467            } else if (regionStates.isRegionInState(
468                hri, RegionState.State.SPLITTING_NEW, RegionState.State.MERGING_NEW)) {
469              regionStates.updateRegionState(hri, RegionState.State.OFFLINE);
470            }
471            regionsToAssignAggregator.add(hri);
472          // TODO: The below else if is different in branch-1 from master branch.
473          } else if (rit != null) {
474            if ((rit.isClosing() || rit.isFailedClose() || rit.isOffline())
475                && am.getTableStateManager().isTableState(hri.getTable(),
476                TableState.State.DISABLED, TableState.State.DISABLING) ||
477                am.getReplicasToClose().contains(hri)) {
478              // If the table was partially disabled and the RS went down, we should clear the
479              // RIT and remove the node for the region.
480              // The rit that we use may be stale in case the table was in DISABLING state
481              // but though we did assign we will not be clearing the znode in CLOSING state.
482              // Doing this will have no harm. See HBASE-5927
483              regionStates.updateRegionState(hri, RegionState.State.OFFLINE);
484              am.offlineDisabledRegion(hri);
485            } else {
486              LOG.warn("THIS SHOULD NOT HAPPEN: unexpected region in transition "
487                + rit + " not to be assigned by SSH of server " + serverName);
488            }
489          }
490        } finally {
491          lock.unlock();
492        }
493      }
494    }
495    return regionsToAssignAggregator;
496  }

498  private boolean assign(final MasterProcedureEnv env, final List<HRegionInfo> hris)
499  throws InterruptedIOException {
500    AssignmentManager am = env.getMasterServices().getAssignmentManager();
501    try {
502      am.assign(hris);
503    } catch (InterruptedException ie) {
504      LOG.error("Caught " + ie + " during round-robin assignment");
505      throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
506    } catch (IOException ioe) {
507      LOG.info("Caught " + ioe + " during region assignment, will retry");
508      return false;
509    }
510    return true;
511  }
512
513  private boolean waitOnAssign(final MasterProcedureEnv env, final List<HRegionInfo> hris)
514  throws InterruptedIOException {
515    int timeout = env.getMasterConfiguration().getInt(KEY_WAIT_ON_RIT, DEFAULT_WAIT_ON_RIT);
516    for (HRegionInfo hri: hris) {
517      // TODO: Blocks here.
518      if (!waitOnRegionToClearRegionsInTransition(env.getMasterServices().getAssignmentManager(),
519          hri, timeout)) {
520        return false;
521      }
522    }
523    return true;
524  }

576  @Override
577  public void serializeStateData(final OutputStream stream) throws IOException {
578    super.serializeStateData(stream);
579
580    MasterProcedureProtos.ServerCrashStateData.Builder state =
581      MasterProcedureProtos.ServerCrashStateData.newBuilder().
582      setServerName(ProtobufUtil.toServerName(this.serverName)).
583      setDistributedLogReplay(this.distributedLogReplay).
584      setCarryingMeta(this.carryingMeta).
585      setShouldSplitWal(this.shouldSplitWal);
586    if (this.regionsOnCrashedServer != null && !this.regionsOnCrashedServer.isEmpty()) {
587      for (HRegionInfo hri: this.regionsOnCrashedServer) {
588        state.addRegionsOnCrashedServer(HRegionInfo.convert(hri));
589      }
590    }
591    if (this.regionsAssigned != null && !this.regionsAssigned.isEmpty()) {
592      for (HRegionInfo hri: this.regionsAssigned) {
593        state.addRegionsAssigned(HRegionInfo.convert(hri));
594      }
595    }
596    state.build().writeDelimitedTo(stream);
597  }

600  public void deserializeStateData(final InputStream stream) throws IOException {
601    super.deserializeStateData(stream);
602
603    MasterProcedureProtos.ServerCrashStateData state =
604      MasterProcedureProtos.ServerCrashStateData.parseDelimitedFrom(stream);
605    this.serverName = ProtobufUtil.toServerName(state.getServerName());
606    this.distributedLogReplay = state.hasDistributedLogReplay()?
607      state.getDistributedLogReplay(): false;
608    this.carryingMeta = state.hasCarryingMeta()? state.getCarryingMeta(): false;
609    // shouldSplitWAL has a default over in pb so this invocation will always work.
610    this.shouldSplitWal = state.getShouldSplitWal();
611    int size = state.getRegionsOnCrashedServerCount();
612    if (size > 0) {
613      this.regionsOnCrashedServer = new HashSet<HRegionInfo>(size);
614      for (RegionInfo ri: state.getRegionsOnCrashedServerList()) {
615        this.regionsOnCrashedServer.add(HRegionInfo.convert(ri));
616      }
617    }
618    size = state.getRegionsAssignedCount();
619    if (size > 0) {
620      this.regionsAssigned = new ArrayList<HRegionInfo>(size);
621      for (RegionInfo ri: state.getRegionsOnCrashedServerList()) {
622        this.regionsAssigned.add(HRegionInfo.convert(ri));
623      }
624    }
625  }

627  /**
628   * Process a dead region from a dead RS. Checks if the region is disabled or
629   * disabling or if the region has a partially completed split.
630   * @param hri
631   * @param assignmentManager
632   * @return Returns true if specified region should be assigned, false if not.
633   * @throws IOException
634   */
635  private static boolean processDeadRegion(HRegionInfo hri, AssignmentManager assignmentManager)
636  throws IOException {
637    boolean tablePresent = assignmentManager.getTableStateManager().isTablePresent(hri.getTable());
638    if (!tablePresent) {
639      LOG.info("The table " + hri.getTable() + " was deleted.  Hence not proceeding.");
640      return false;
641    }
642    // If table is not disabled but the region is offlined,
643    boolean disabled = assignmentManager.getTableStateManager().isTableState(hri.getTable(),
644      TableState.State.DISABLED);
645    if (disabled){
646      LOG.info("The table " + hri.getTable() + " was disabled.  Hence not proceeding.");
647      return false;
648    }
649    if (hri.isOffline() && hri.isSplit()) {
650      // HBASE-7721: Split parent and daughters are inserted into hbase:meta as an atomic operation.
651      // If the meta scanner saw the parent split, then it should see the daughters as assigned
652      // to the dead server. We don't have to do anything.
653      return false;
654    }
655    boolean disabling = assignmentManager.getTableStateManager().isTableState(hri.getTable(),
656        TableState.State.DISABLING);
657    if (disabling) {
658      LOG.info("The table " + hri.getTable() + " is disabled.  Hence not assigning region" +
659        hri.getEncodedName());
660      return false;
661    }
662    return true;
663  }

665  /**
666   * If hbase:meta is not assigned already, assign.
667   * @throws IOException
668   */
669  private void verifyAndAssignMetaWithRetries(final MasterProcedureEnv env) throws IOException {
670    MasterServices services = env.getMasterServices();
671    int iTimes = services.getConfiguration().getInt(KEY_RETRIES_ON_META, DEFAULT_RETRIES_ON_META);
672    // Just reuse same time as we have for short wait on meta. Adding another config is overkill.
673    long waitTime =
674      services.getConfiguration().getLong(KEY_SHORT_WAIT_ON_META, DEFAULT_SHORT_WAIT_ON_META);
675    int iFlag = 0;
676    while (true) {
677      try {
678        verifyAndAssignMeta(env);
679        break;
680      } catch (KeeperException e) {
681        services.abort("In server shutdown processing, assigning meta", e);
682        throw new IOException("Aborting", e);
683      } catch (Exception e) {
684        if (iFlag >= iTimes) {
685          services.abort("verifyAndAssignMeta failed after" + iTimes + " retries, aborting", e);
686          throw new IOException("Aborting", e);
687        }
688        try {
689          Thread.sleep(waitTime);
690        } catch (InterruptedException e1) {
691          LOG.warn("Interrupted when is the thread sleep", e1);
692          Thread.currentThread().interrupt();
693          throw (InterruptedIOException)new InterruptedIOException().initCause(e1);
694        }
695        iFlag++;
696      }
697    }
698  }

700  /**
701   * If hbase:meta is not assigned already, assign.
702   * @throws InterruptedException
703   * @throws IOException
704   * @throws KeeperException
705   */
706  private void verifyAndAssignMeta(final MasterProcedureEnv env)
707      throws InterruptedException, IOException, KeeperException {
708    MasterServices services = env.getMasterServices();
709    if (!isMetaAssignedQuickTest(env)) {
710      services.getAssignmentManager().assignMeta(HRegionInfo.FIRST_META_REGIONINFO);
711    } else if (serverName.equals(services.getMetaTableLocator().
712        getMetaRegionLocation(services.getZooKeeper()))) {
713      throw new IOException("hbase:meta is onlined on the dead server " + this.serverName);
714    } else {
715      LOG.info("Skip assigning hbase:meta because it is online at "
716          + services.getMetaTableLocator().getMetaRegionLocation(services.getZooKeeper()));
717    }
718  }

720  /**
721   * A quick test that hbase:meta is assigned; blocks for short time only.
722   * @return True if hbase:meta location is available and verified as good.
723   * @throws InterruptedException
724   * @throws IOException
725   */
726  private boolean isMetaAssignedQuickTest(final MasterProcedureEnv env)
727  throws InterruptedException, IOException {
728    ZooKeeperWatcher zkw = env.getMasterServices().getZooKeeper();
729    MetaTableLocator mtl = env.getMasterServices().getMetaTableLocator();
730    boolean metaAssigned = false;
731    // Is hbase:meta location available yet?
732    if (mtl.isLocationAvailable(zkw)) {
733      ClusterConnection connection = env.getMasterServices().getConnection();
734      // Is hbase:meta location good yet?
735      long timeout =
736        env.getMasterConfiguration().getLong(KEY_SHORT_WAIT_ON_META, DEFAULT_SHORT_WAIT_ON_META);
737      if (mtl.verifyMetaRegionLocation(connection, zkw, timeout)) {
738        metaAssigned = true;
739      }
740    }
741    return metaAssigned;
742  }
