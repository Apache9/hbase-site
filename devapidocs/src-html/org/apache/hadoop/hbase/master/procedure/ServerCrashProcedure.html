047/**
048 * Handle crashed server. This is a port to ProcedureV2 of what used to be euphemistically called
049 * ServerShutdownHandler.
050 *
051 * <p>The procedure flow varies dependent on whether meta is assigned and if we are to split logs.
052 *
053 * <p>We come in here after ServerManager has noticed a server has expired. Procedures
054 * queued on the rpc should have been notified about fail and should be concurrently
055 * getting themselves ready to assign elsewhere.
056 */
057@InterfaceAudience.Private
058public class ServerCrashProcedure
059    extends StateMachineProcedure<MasterProcedureEnv, ServerCrashState>
060    implements ServerProcedureInterface {
061  private static final Logger LOG = LoggerFactory.getLogger(ServerCrashProcedure.class);
062
063  /**
064   * Name of the crashed server to process.
065   */
066  private ServerName serverName;
067
068  /**
069   * Whether DeadServer knows that we are processing it.
070   */
071  private boolean notifiedDeadServer = false;
072
073  /**
074   * Regions that were on the crashed server.
075   */
076  private List<RegionInfo> regionsOnCrashedServer;
077
078  private boolean carryingMeta = false;
079  private boolean shouldSplitWal;
080
081  /**
082   * Call this constructor queuing up a Procedure.
083   * @param serverName Name of the crashed server.
084   * @param shouldSplitWal True if we should split WALs as part of crashed server processing.
085   * @param carryingMeta True if carrying hbase:meta table region.
086   */
087  public ServerCrashProcedure(final MasterProcedureEnv env, final ServerName serverName,
088      final boolean shouldSplitWal, final boolean carryingMeta) {
089    this.serverName = serverName;
090    this.shouldSplitWal = shouldSplitWal;
091    this.carryingMeta = carryingMeta;
092    this.setOwner(env.getRequestUser());
093  }

102  public boolean isInRecoverMetaState() {
103    return getCurrentState() == ServerCrashState.SERVER_CRASH_PROCESS_META;
104  }

106  @Override
107  protected Flow executeFromState(MasterProcedureEnv env, ServerCrashState state)
108      throws ProcedureSuspendedException, ProcedureYieldException {
109    final MasterServices services = env.getMasterServices();
110    // HBASE-14802
111    // If we have not yet notified that we are processing a dead server, we should do now.
112    if (!notifiedDeadServer) {
113      services.getServerManager().getDeadServers().notifyServer(serverName);
114      notifiedDeadServer = true;
115    }
116
117    switch (state) {
118      case SERVER_CRASH_START:
119      case SERVER_CRASH_SPLIT_META_LOGS:
120      case SERVER_CRASH_ASSIGN_META:
121        break;
122      default:
123        // If hbase:meta is not assigned, yield.
124        if (env.getAssignmentManager().waitMetaLoaded(this)) {
125          throw new ProcedureSuspendedException();
126        }
127    }
128    try {
129      switch (state) {
130        case SERVER_CRASH_START:
131          LOG.info("Start " + this);
132          // If carrying meta, process it first. Else, get list of regions on crashed server.
133          if (this.carryingMeta) {
134            setNextState(ServerCrashState.SERVER_CRASH_SPLIT_META_LOGS);
135          } else {
136            setNextState(ServerCrashState.SERVER_CRASH_GET_REGIONS);
137          }
138          break;
139        case SERVER_CRASH_SPLIT_META_LOGS:
140          splitMetaLogs(env);
141          setNextState(ServerCrashState.SERVER_CRASH_ASSIGN_META);
142          break;
143        case SERVER_CRASH_ASSIGN_META:
144          assignRegions(env, Arrays.asList(RegionInfoBuilder.FIRST_META_REGIONINFO));
145          setNextState(ServerCrashState.SERVER_CRASH_GET_REGIONS);
146          break;
147        case SERVER_CRASH_GET_REGIONS:
148          this.regionsOnCrashedServer =
149            services.getAssignmentManager().getRegionsOnServer(serverName);
150          // Where to go next? Depends on whether we should split logs at all or
151          // if we should do distributed log splitting.
152          if (!this.shouldSplitWal) {
153            setNextState(ServerCrashState.SERVER_CRASH_ASSIGN);
154          } else {
155            setNextState(ServerCrashState.SERVER_CRASH_SPLIT_LOGS);
156          }
157          break;
158        case SERVER_CRASH_SPLIT_LOGS:
159          splitLogs(env);
160          setNextState(ServerCrashState.SERVER_CRASH_ASSIGN);
161          break;
162        case SERVER_CRASH_ASSIGN:
163          // If no regions to assign, skip assign and skip to the finish.
164          // Filter out meta regions. Those are handled elsewhere in this procedure.
165          // Filter changes this.regionsOnCrashedServer.
166          if (filterDefaultMetaRegions()) {
167            if (LOG.isTraceEnabled()) {
168              LOG
169                .trace("Assigning regions " + RegionInfo.getShortNameToLog(regionsOnCrashedServer) +
170                  ", " + this + "; cycles=" + getCycles());
171            }
172            assignRegions(env, regionsOnCrashedServer);
173          }
174          setNextState(ServerCrashState.SERVER_CRASH_FINISH);
175          break;
176        case SERVER_CRASH_HANDLE_RIT2:
177          // Noop. Left in place because we used to call handleRIT here for a second time
178          // but no longer necessary since HBASE-20634.
179          setNextState(ServerCrashState.SERVER_CRASH_FINISH);
180          break;
181        case SERVER_CRASH_FINISH:
182          services.getAssignmentManager().getRegionStates().removeServer(serverName);
183          services.getServerManager().getDeadServers().finish(serverName);
184          return Flow.NO_MORE_STATE;
185        default:
186          throw new UnsupportedOperationException("unhandled state=" + state);
187      }
188    } catch (IOException e) {
189      LOG.warn("Failed state=" + state + ", retry " + this + "; cycles=" + getCycles(), e);
190    }
191    return Flow.HAS_MORE_STATE;
192  }
193
194  private boolean filterDefaultMetaRegions() {
195    if (regionsOnCrashedServer == null) {
196      return false;
197    }
198    regionsOnCrashedServer.removeIf(this::isDefaultMetaRegion);
199    return !regionsOnCrashedServer.isEmpty();
200  }
201
202  private boolean isDefaultMetaRegion(RegionInfo hri) {
203    return hri.isMetaRegion() && RegionReplicaUtil.isDefaultReplica(hri);
204  }
205
206  private void splitMetaLogs(MasterProcedureEnv env) throws IOException {
207    LOG.debug("Splitting meta WALs {}", this);
208    MasterWalManager mwm = env.getMasterServices().getMasterWalManager();
209    AssignmentManager am = env.getMasterServices().getAssignmentManager();
210    am.getRegionStates().metaLogSplitting(serverName);
211    mwm.splitMetaLog(serverName);
212    am.getRegionStates().metaLogSplit(serverName);
213    LOG.debug("Done splitting meta WALs {}", this);
214  }
215
216  private void splitLogs(final MasterProcedureEnv env) throws IOException {
217    LOG.debug("Splitting WALs {}", this);
218    MasterWalManager mwm = env.getMasterServices().getMasterWalManager();
219    AssignmentManager am = env.getMasterServices().getAssignmentManager();
220    // TODO: For Matteo. Below BLOCKs!!!! Redo so can relinquish executor while it is running.
221    // PROBLEM!!! WE BLOCK HERE.
222    am.getRegionStates().logSplitting(this.serverName);
223    mwm.splitLog(this.serverName);
224    if (!carryingMeta) {
225      mwm.archiveMetaLog(this.serverName);
226    }
227    am.getRegionStates().logSplit(this.serverName);
228    LOG.debug("Done splitting WALs {}", this);
229  }

283  @Override
284  protected void serializeStateData(ProcedureStateSerializer serializer)
285      throws IOException {
286    super.serializeStateData(serializer);
287
288    MasterProcedureProtos.ServerCrashStateData.Builder state =
289      MasterProcedureProtos.ServerCrashStateData.newBuilder().
290      setServerName(ProtobufUtil.toServerName(this.serverName)).
291      setCarryingMeta(this.carryingMeta).
292      setShouldSplitWal(this.shouldSplitWal);
293    if (this.regionsOnCrashedServer != null && !this.regionsOnCrashedServer.isEmpty()) {
294      for (RegionInfo hri: this.regionsOnCrashedServer) {
295        state.addRegionsOnCrashedServer(ProtobufUtil.toRegionInfo(hri));
296      }
297    }
298    serializer.serialize(state.build());
299  }
300
301  @Override
302  protected void deserializeStateData(ProcedureStateSerializer serializer)
303      throws IOException {
304    super.deserializeStateData(serializer);
305
306    MasterProcedureProtos.ServerCrashStateData state =
307        serializer.deserialize(MasterProcedureProtos.ServerCrashStateData.class);
308    this.serverName = ProtobufUtil.toServerName(state.getServerName());
309    this.carryingMeta = state.hasCarryingMeta()? state.getCarryingMeta(): false;
310    // shouldSplitWAL has a default over in pb so this invocation will always work.
311    this.shouldSplitWal = state.getShouldSplitWal();
312    int size = state.getRegionsOnCrashedServerCount();
313    if (size > 0) {
314      this.regionsOnCrashedServer = new ArrayList<>(size);
315      for (org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos.RegionInfo ri: state.getRegionsOnCrashedServerList()) {
316        this.regionsOnCrashedServer.add(ProtobufUtil.toRegionInfo(ri));
317      }
318    }
319  }

344  /**
345   * Assign the regions on the crashed RS to other Rses.
346   * <p/>
347   * In this method we will go through all the RegionStateNodes of the give regions to find out
348   * whether there is already an TRSP for the region, if so we interrupt it and let it retry on
349   * other server, otherwise we will schedule a TRSP to bring the region online.
350   * <p/>
351   * We will also check whether the table for a region is enabled, if not, we will skip assigning
352   * it.
353   */
354  private void assignRegions(MasterProcedureEnv env, List<RegionInfo> regions) throws IOException {
355    AssignmentManager am = env.getMasterServices().getAssignmentManager();
356    for (RegionInfo region : regions) {
357      RegionStateNode regionNode = am.getRegionStates().getOrCreateRegionStateNode(region);
358      regionNode.lock();
359      try {
360        if (regionNode.getProcedure() != null) {
361          LOG.info("{} found RIT {}; {}", this, regionNode.getProcedure(), regionNode);
362          regionNode.getProcedure().serverCrashed(env, regionNode, getServerName());
363        } else {
364          if (env.getMasterServices().getTableStateManager().isTableState(regionNode.getTable(),
365            TableState.State.DISABLING, TableState.State.DISABLED)) {
366            continue;
367          }
368          TransitRegionStateProcedure proc = TransitRegionStateProcedure.assign(env, region, null);
369          regionNode.setProcedure(proc);
370          addChildProcedure(proc);
371        }
372      } finally {
373        regionNode.unlock();
374      }
375    }
376  }
