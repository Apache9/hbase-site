019package org.apache.hadoop.hbase.master.procedure;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.MetaTableAccessor;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.hadoop.hbase.TableNotFoundException;
026import org.apache.yetus.audience.InterfaceAudience;
027import org.apache.hadoop.hbase.procedure2.StateMachineProcedure;
028import org.apache.hadoop.hbase.security.User;
029
030/**
031 * Base class for all the Table procedures that want to use a StateMachineProcedure.
032 * It provides helpers like basic locking, sync latch, and toStringClassDetails().
033 */
034@InterfaceAudience.Private
035public abstract class AbstractStateMachineTableProcedure<TState>
036    extends StateMachineProcedure<MasterProcedureEnv, TState>
037    implements TableProcedureInterface {
038
039  // used for compatibility with old clients
040  private final ProcedurePrepareLatch syncLatch;
041
042  private User user;
043
044  protected AbstractStateMachineTableProcedure() {
045    // Required by the Procedure framework to create the procedure on replay
046    syncLatch = null;
047  }
048
049  protected AbstractStateMachineTableProcedure(final MasterProcedureEnv env) {
050    this(env, null);
051  }
052
053  /**
054   * @param env Uses this to set Procedure Owner at least.
055   */
056  protected AbstractStateMachineTableProcedure(final MasterProcedureEnv env,
057      final ProcedurePrepareLatch latch) {
058    if (env != null) {
059      this.user = env.getRequestUser();
060      this.setOwner(user);
061    }
062    // used for compatibility with clients without procedures
063    // they need a sync TableExistsException, TableNotFoundException, TableNotDisabledException, ...
064    this.syncLatch = latch;
065  }
066
067  @Override
068  public abstract TableName getTableName();
069
070  @Override
071  public abstract TableOperationType getTableOperationType();
072
073  @Override
074  public void toStringClassDetails(final StringBuilder sb) {
075    sb.append(getClass().getSimpleName());
076    sb.append(" table=");
077    sb.append(getTableName());
078  }
079
080  @Override
081  protected LockState acquireLock(final MasterProcedureEnv env) {
082    if (env.waitInitialized(this)) return LockState.LOCK_EVENT_WAIT;
083    if (env.getProcedureScheduler().waitTableExclusiveLock(this, getTableName())) {
084      return LockState.LOCK_EVENT_WAIT;
085    }
086    return LockState.LOCK_ACQUIRED;
087  }
088
089  @Override
090  protected void releaseLock(final MasterProcedureEnv env) {
091    env.getProcedureScheduler().wakeTableExclusiveLock(this, getTableName());
092  }
093
094  protected User getUser() {
095    return user;
096  }
097
098  protected void setUser(final User user) {
099    this.user = user;
100  }
101
102  protected void releaseSyncLatch() {
103    ProcedurePrepareLatch.releaseLatch(syncLatch, this);
104  }
105
106  /**
107   * Check whether a table is modifiable - exists and either offline or online with config set
108   * @param env MasterProcedureEnv
109   * @throws IOException
110   */
111  protected void checkTableModifiable(final MasterProcedureEnv env) throws IOException {
112    // Checks whether the table exists
113    if (!MetaTableAccessor.tableExists(env.getMasterServices().getConnection(), getTableName())) {
114      throw new TableNotFoundException(getTableName());
115    }
116  }
117}
