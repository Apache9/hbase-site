091  public SplitTableRegionProcedure(final MasterProcedureEnv env,
092      final HRegionInfo regionToSplit, final byte[] splitRow) throws IOException {
093    super(env);
094
095    checkSplitRow(regionToSplit, splitRow);
096
097    this.traceEnabled = null;
098    this.parentHRI = regionToSplit;
099
100    final TableName table = regionToSplit.getTable();
101    final long rid = getDaughterRegionIdTimestamp(regionToSplit);
102    this.daughter_1_HRI = new HRegionInfo(table, regionToSplit.getStartKey(), splitRow, false, rid);
103    this.daughter_2_HRI = new HRegionInfo(table, splitRow, regionToSplit.getEndKey(), false, rid);
104  }
105
106  private static void checkSplitRow(final HRegionInfo regionToSplit, final byte[] splitRow)
107      throws IOException {
108    if (splitRow == null || splitRow.length == 0) {
109      throw new DoNotRetryIOException("Split row cannot be null");
110    }
111
112    if (Bytes.equals(regionToSplit.getStartKey(), splitRow)) {
113      throw new DoNotRetryIOException(
114        "Split row is equal to startkey: " + Bytes.toStringBinary(splitRow));
115    }
116
117    if (!regionToSplit.containsRow(splitRow)) {
118      throw new DoNotRetryIOException(
119        "Split row is not inside region key range splitKey:" + Bytes.toStringBinary(splitRow) +
120        " region: " + regionToSplit);
121    }
122  }

124  /**
125   * Calculate daughter regionid to use.
126   * @param hri Parent {@link HRegionInfo}
127   * @return Daughter region id (timestamp) to use.
128   */
129  private static long getDaughterRegionIdTimestamp(final HRegionInfo hri) {
130    long rid = EnvironmentEdgeManager.currentTime();
131    // Regionid is timestamp.  Can't be less than that of parent else will insert
132    // at wrong location in hbase:meta (See HBASE-710).
133    if (rid < hri.getRegionId()) {
134      LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
135        " but current time here is " + rid);
136      rid = hri.getRegionId() + 1;
137    }
138    return rid;
139  }

141  @Override
142  protected Flow executeFromState(final MasterProcedureEnv env, final SplitTableRegionState state)
143      throws InterruptedException {
144    if (isTraceEnabled()) {
145      LOG.trace(this + " execute state=" + state);
146    }
147
148    try {
149      switch (state) {
150      case SPLIT_TABLE_REGION_PREPARE:
151        if (prepareSplitRegion(env)) {
152          setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION);
153          break;
154        } else {
155          assert isFailed() : "split region should have an exception here";
156          return Flow.NO_MORE_STATE;
157        }
158      case SPLIT_TABLE_REGION_PRE_OPERATION:
159        preSplitRegion(env);
160        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_SET_SPLITTING_TABLE_STATE);
161        break;
162      case SPLIT_TABLE_REGION_SET_SPLITTING_TABLE_STATE:
163        setRegionStateToSplitting(env);
164        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CLOSED_PARENT_REGION);
165        break;
166      case SPLIT_TABLE_REGION_CLOSED_PARENT_REGION:
167        closeParentRegionForSplit(env);
168        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS);
169        break;
170      case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
171        createDaughterRegions(env);
172        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_PONR);
173        break;
174      case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_PONR:
175        preSplitRegionBeforePONR(env);
176        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_UPDATE_META);
177        break;
178      case SPLIT_TABLE_REGION_UPDATE_META:
179        // This is the point of no return.  Adding subsequent edits to .META. as we
180        // do below when we do the daughter opens adding each to .META. can fail in
181        // various interesting ways the most interesting of which is a timeout
182        // BUT the edits all go through (See HBASE-3872).  IF we reach the PONR
183        // then subsequent failures need to crash out this region server; the
184        // server shutdown processing should be able to fix-up the incomplete split.
185        // The offlined parent will have the daughters as extra columns.  If
186        // we leave the daughter regions in place and do not remove them when we
187        // crash out, then they will have their references to the parent in place
188        // still and the server shutdown fixup of .META. will point to these
189        // regions.
190        // We should add PONR JournalEntry before offlineParentInMeta,so even if
191        // OfflineParentInMeta timeout,this will cause regionserver exit,and then
192        // master ServerShutdownHandler will fix daughter & avoid data loss. (See
193        // HBase-4562).
194        updateMetaForDaughterRegions(env);
195        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_PONR);
196        break;
197      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_PONR:
198        preSplitRegionAfterPONR(env);
199        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS);
200        break;
201      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
202        openDaughterRegions(env);
203        setNextState(SplitTableRegionState.SPLIT_TABLE_REGION_POST_OPERATION);
204        break;
205      case SPLIT_TABLE_REGION_POST_OPERATION:
206        postSplitRegion(env);
207        return Flow.NO_MORE_STATE;
208      default:
209        throw new UnsupportedOperationException(this + " unhandled state=" + state);
210      }
211    } catch (IOException e) {
212      String msg = "Error trying to split region " + parentHRI.getEncodedName() + " in the table "
213          + getTableName() + " (in state=" + state + ")";
214      if (!isRollbackSupported(state)) {
215        // We reach a state that cannot be rolled back. We just need to keep retry.
216        LOG.warn(msg, e);
217      } else {
218        LOG.error(msg, e);
219        setFailure("master-split-region", e);
220      }
221    }
222    return Flow.HAS_MORE_STATE;
223  }

225  @Override
226  protected void rollbackState(final MasterProcedureEnv env, final SplitTableRegionState state)
227      throws IOException, InterruptedException {
228    if (isTraceEnabled()) {
229      LOG.trace(this + " rollback state=" + state);
230    }
231
232    try {
233      switch (state) {
234      case SPLIT_TABLE_REGION_POST_OPERATION:
235      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
236      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_PONR:
237      case SPLIT_TABLE_REGION_UPDATE_META:
238        // PONR
239        throw new UnsupportedOperationException(this + " unhandled state=" + state);
240      case SPLIT_TABLE_REGION_PRE_OPERATION_BEFORE_PONR:
241        break;
242      case SPLIT_TABLE_REGION_CREATE_DAUGHTER_REGIONS:
243        // Doing nothing, as re-open parent region would clean up daughter region directories.
244        break;
245      case SPLIT_TABLE_REGION_CLOSED_PARENT_REGION:
246        openParentRegion(env);
247        break;
248      case SPLIT_TABLE_REGION_SET_SPLITTING_TABLE_STATE:
249        setRegionStateToRevertSplitting(env);
250        break;
251      case SPLIT_TABLE_REGION_PRE_OPERATION:
252        preSplitRegionRollback(env);
253        break;
254      case SPLIT_TABLE_REGION_PREPARE:
255        break; // nothing to do
256      default:
257        throw new UnsupportedOperationException(this + " unhandled state=" + state);
258      }
259    } catch (IOException e) {
260      // This will be retried. Unless there is a bug in the code,
261      // this should be just a "temporary error" (e.g. network down)
262      LOG.warn("Failed rollback attempt step " + state + " for splitting the region "
263        + parentHRI.getEncodedName() + " in table " + getTableName(), e);
264      throw e;
265    }
266  }

268  /*
269   * Check whether we are in the state that can be rollback
270   */
271  @Override
272  protected boolean isRollbackSupported(final SplitTableRegionState state) {
273    switch (state) {
274      case SPLIT_TABLE_REGION_POST_OPERATION:
275      case SPLIT_TABLE_REGION_OPEN_CHILD_REGIONS:
276      case SPLIT_TABLE_REGION_PRE_OPERATION_AFTER_PONR:
277      case SPLIT_TABLE_REGION_UPDATE_META:
278        // It is not safe to rollback if we reach to these states.
279        return false;
280      default:
281        break;
282    }
283    return true;
284  }

301  @Override
302  public void serializeStateData(final OutputStream stream) throws IOException {
303    super.serializeStateData(stream);
304
305    final MasterProcedureProtos.SplitTableRegionStateData.Builder splitTableRegionMsg =
306        MasterProcedureProtos.SplitTableRegionStateData.newBuilder()
307        .setUserInfo(MasterProcedureUtil.toProtoUserInfo(getUser()))
308        .setParentRegionInfo(HRegionInfo.convert(parentHRI))
309        .addChildRegionInfo(HRegionInfo.convert(daughter_1_HRI))

314  @Override
315  public void deserializeStateData(final InputStream stream) throws IOException {
316    super.deserializeStateData(stream);
317
318    final MasterProcedureProtos.SplitTableRegionStateData splitTableRegionsMsg =
319        MasterProcedureProtos.SplitTableRegionStateData.parseDelimitedFrom(stream);
320    setUser(MasterProcedureUtil.toUserInfo(splitTableRegionsMsg.getUserInfo()));
321    parentHRI = HRegionInfo.convert(splitTableRegionsMsg.getParentRegionInfo());
322    if (splitTableRegionsMsg.getChildRegionInfoCount() == 0) {
323      daughter_1_HRI = daughter_2_HRI = null;
324    } else {
325      assert(splitTableRegionsMsg.getChildRegionInfoCount() == 2);
326      daughter_1_HRI = HRegionInfo.convert(splitTableRegionsMsg.getChildRegionInfoList().get(0));
327      daughter_2_HRI = HRegionInfo.convert(splitTableRegionsMsg.getChildRegionInfoList().get(1));
328    }
329  }

378   * @param env MasterProcedureEnv
379   * @throws IOException
380   */
381  @VisibleForTesting
382  public boolean prepareSplitRegion(final MasterProcedureEnv env) throws IOException {
383    // Check whether the region is splittable
384    final RegionState state = getParentRegionState(env);
385    if (state.isClosing() || state.isClosed() ||
386        state.isSplittingOrSplitOnServer(state.getServerName())) {
387      setFailure(
388        "master-split-region",
389        new IOException("Split region " + parentHRI + " failed due to region is not splittable"));
390      return false;
391    }
392    return true;
393  }

425   * @param env MasterProcedureEnv
426   * @throws IOException
427   */
428  @VisibleForTesting
429  public void setRegionStateToSplitting(final MasterProcedureEnv env) throws IOException {
430    RegionStateTransition.Builder transition = RegionStateTransition.newBuilder();
431    transition.setTransitionCode(TransitionCode.READY_TO_SPLIT);
432    transition.addRegionInfo(HRegionInfo.convert(parentHRI));
433    transition.addRegionInfo(HRegionInfo.convert(daughter_1_HRI));
434    transition.addRegionInfo(HRegionInfo.convert(daughter_2_HRI));
435    if (env.getMasterServices().getAssignmentManager().onRegionTransition(
436      getParentRegionState(env).getServerName(), transition.build()) != null) {
437      throw new IOException("Failed to update region state to SPLITTING for "
438          + parentHRI.getRegionNameAsString());
439    }
440  }

492   * Create daughter regions
493   * @param env MasterProcedureEnv
494   * @throws IOException
495   */
496  @VisibleForTesting
497  public void createDaughterRegions(final MasterProcedureEnv env) throws IOException {
498    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
499    final Path tabledir = FSUtils.getTableDir(mfs.getRootDir(), parentHRI.getTable());
500    final FileSystem fs = mfs.getFileSystem();
501    HRegionFileSystem regionFs = HRegionFileSystem.openRegionFromFileSystem(
502      env.getMasterConfiguration(), fs, tabledir, parentHRI, false);
503    regionFs.createSplitsDir();

505    Pair<Integer, Integer> expectedReferences = splitStoreFiles(env, regionFs);
506
507    assertReferenceFileCount(
508      fs, expectedReferences.getFirst(), regionFs.getSplitsDir(daughter_1_HRI));
509    //Move the files from the temporary .splits to the final /table/region directory
510    regionFs.commitDaughterRegion(daughter_1_HRI);
511    assertReferenceFileCount(
512      fs,
513      expectedReferences.getFirst(),
514      new Path(tabledir, daughter_1_HRI.getEncodedName()));
515
516    assertReferenceFileCount(
517      fs, expectedReferences.getSecond(), regionFs.getSplitsDir(daughter_2_HRI));
518    regionFs.commitDaughterRegion(daughter_2_HRI);
519    assertReferenceFileCount(
520      fs,
521      expectedReferences.getSecond(),
522      new Path(tabledir, daughter_2_HRI.getEncodedName()));
523  }

525  /**
526   * Create Split directory
527   * @param env MasterProcedureEnv
528   * @throws IOException
529   */
530  private Pair<Integer, Integer> splitStoreFiles(final MasterProcedureEnv env,
531      final HRegionFileSystem regionFs) throws IOException {
532    final MasterFileSystem mfs = env.getMasterServices().getMasterFileSystem();
533    final Configuration conf = env.getMasterConfiguration();
534
535    // The following code sets up a thread pool executor with as many slots as
536    // there's files to split. It then fires up everything, waits for
537    // completion and finally checks for any exception
538    //
539    // Note: splitStoreFiles creates daughter region dirs under the parent splits dir
540    // Nothing to unroll here if failure -- re-run createSplitsDir will
541    // clean this up.
542    int nbFiles = 0;
543    for (String family: regionFs.getFamilies()) {
544      Collection<StoreFileInfo> storeFiles = regionFs.getStoreFiles(family);
545      if (storeFiles != null) {
546        nbFiles += storeFiles.size();
547      }
548    }
549    if (nbFiles == 0) {
550      // no file needs to be splitted.
551      return new Pair<Integer, Integer>(0,0);
552    }
553    // Default max #threads to use is the smaller of table's configured number of blocking store
554    // files or the available number of logical cores.
555    int defMaxThreads = Math.min(
556      conf.getInt(HStore.BLOCKING_STOREFILES_KEY, HStore.DEFAULT_BLOCKING_STOREFILE_COUNT),
557      Runtime.getRuntime().availableProcessors());
558    // Max #threads is the smaller of the number of storefiles or the default max determined above.
559    int maxThreads = Math.min(
560      conf.getInt(HConstants.REGION_SPLIT_THREADS_MAX, defMaxThreads), nbFiles);
561    LOG.info("Preparing to split " + nbFiles + " storefiles for region " + parentHRI +
562            " using " + maxThreads + " threads");
563    ThreadPoolExecutor threadPool = (ThreadPoolExecutor) Executors.newFixedThreadPool(
564      maxThreads, Threads.getNamedThreadFactory("StoreFileSplitter-%1$d"));
565    List<Future<Pair<Path,Path>>> futures = new ArrayList<Future<Pair<Path,Path>>> (nbFiles);
566
567    // Split each store file.
568    final HTableDescriptor htd = env.getMasterServices().getTableDescriptors().get(getTableName());
569    for (String family: regionFs.getFamilies()) {
570      final HColumnDescriptor hcd = htd.getFamily(family.getBytes());
571      final Collection<StoreFileInfo> storeFiles = regionFs.getStoreFiles(family);
572      if (storeFiles != null && storeFiles.size() > 0) {
573        final CacheConfig cacheConf = new CacheConfig(conf, hcd);
574        for (StoreFileInfo storeFileInfo: storeFiles) {
575          StoreFileSplitter sfs = new StoreFileSplitter(regionFs, family.getBytes(),
576            new StoreFile(mfs.getFileSystem(), storeFileInfo, conf,
577              cacheConf, hcd.getBloomFilterType()));
578          futures.add(threadPool.submit(sfs));
579        }
580      }
581    }
582    // Shutdown the pool
583    threadPool.shutdown();
584
585    // Wait for all the tasks to finish
586    long fileSplitTimeout = conf.getLong("hbase.master.fileSplitTimeout", 30000);
587    try {
588      boolean stillRunning = !threadPool.awaitTermination(fileSplitTimeout, TimeUnit.MILLISECONDS);
589      if (stillRunning) {
590        threadPool.shutdownNow();
591        // wait for the thread to shutdown completely.
592        while (!threadPool.isTerminated()) {
593          Thread.sleep(50);
594        }
595        throw new IOException("Took too long to split the" +
596            " files and create the references, aborting split");
597      }
598    } catch (InterruptedException e) {
599      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
600    }
601
602    int daughterA = 0;
603    int daughterB = 0;
604    // Look for any exception
605    for (Future<Pair<Path, Path>> future : futures) {
606      try {
607        Pair<Path, Path> p = future.get();
608        daughterA += p.getFirst() != null ? 1 : 0;
609        daughterB += p.getSecond() != null ? 1 : 0;
610      } catch (InterruptedException e) {
611        throw (InterruptedIOException) new InterruptedIOException().initCause(e);
612      } catch (ExecutionException e) {
613        throw new IOException(e);
614      }
615    }
616
617    if (LOG.isDebugEnabled()) {
618      LOG.debug("Split storefiles for region " + parentHRI + " Daughter A: " + daughterA
619          + " storefiles, Daughter B: " + daughterB + " storefiles.");
620    }
621    return new Pair<Integer, Integer>(daughterA, daughterB);
622  }

624  private void assertReferenceFileCount(
625      final FileSystem fs,
626      final int expectedReferenceFileCount,
627      final Path dir)
628      throws IOException {
629    if (expectedReferenceFileCount != 0 &&
630        expectedReferenceFileCount != FSUtils.getRegionReferenceFileCount(fs, dir)) {
631      throw new IOException("Failing split. Expected reference file count isn't equal.");
632    }
633  }

635  private Pair<Path, Path> splitStoreFile(final HRegionFileSystem regionFs,
636      final byte[] family, final StoreFile sf) throws IOException {
637    if (LOG.isDebugEnabled()) {
638      LOG.debug("Splitting started for store file: " + sf.getPath() + " for region: " + parentHRI);
639    }
640
641    final byte[] splitRow = getSplitRow();
642    final String familyName = Bytes.toString(family);
643    final Path path_first =
644        regionFs.splitStoreFile(this.daughter_1_HRI, familyName, sf, splitRow, false, null);
645    final Path path_second =
646        regionFs.splitStoreFile(this.daughter_2_HRI, familyName, sf, splitRow, true, null);
647    if (LOG.isDebugEnabled()) {
648      LOG.debug("Splitting complete for store file: " + sf.getPath() + " for region: " + parentHRI);
649    }
650    return new Pair<Path,Path>(path_first, path_second);
651  }

653  /**
654   * Utility class used to do the file splitting / reference writing
655   * in parallel instead of sequentially.
656   */
657  private class StoreFileSplitter implements Callable<Pair<Path,Path>> {
658    private final HRegionFileSystem regionFs;
659    private final byte[] family;
660    private final StoreFile sf;
661
662    /**
663     * Constructor that takes what it needs to split
664     * @param regionFs the file system
665     * @param family Family that contains the store file
666     * @param sf which file
667     */
668    public StoreFileSplitter(
669        final HRegionFileSystem regionFs,
670        final byte[] family,
671        final StoreFile sf) {
672      this.regionFs = regionFs;
673      this.sf = sf;
674      this.family = family;
675    }
676
677    public Pair<Path,Path> call() throws IOException {
678      return splitStoreFile(regionFs, family, sf);
679    }
680  }

682  /**
683   * Post split region actions before the Point-of-No-Return step
684   * @param env MasterProcedureEnv
685   **/
686  private void preSplitRegionBeforePONR(final MasterProcedureEnv env)
687    throws IOException, InterruptedException {
688    final List<Mutation> metaEntries = new ArrayList<Mutation>();
689    final MasterCoprocessorHost cpHost = env.getMasterCoprocessorHost();
690    if (cpHost != null) {
691      if (cpHost.preSplitBeforePONRAction(getSplitRow(), metaEntries, getUser())) {
692        throw new IOException("Coprocessor bypassing region " +
693            parentHRI.getRegionNameAsString() + " split.");
694      }
695      try {
696        for (Mutation p : metaEntries) {
697          HRegionInfo.parseRegionName(p.getRow());
698        }
699      } catch (IOException e) {
700        LOG.error("Row key of mutation from coprossor is not parsable as region name."
701            + "Mutations from coprocessor should only for hbase:meta table.");
702        throw e;
703      }
704    }
705  }

707  /**
708   * Add daughter regions to META
709   * @param env MasterProcedureEnv
710   * @throws IOException
711   */
712  private void updateMetaForDaughterRegions(final MasterProcedureEnv env) throws IOException {
713    RegionStateTransition.Builder transition = RegionStateTransition.newBuilder();
714    transition.setTransitionCode(TransitionCode.SPLIT_PONR);
715    transition.addRegionInfo(HRegionInfo.convert(parentHRI));
716    transition.addRegionInfo(HRegionInfo.convert(daughter_1_HRI));
717    transition.addRegionInfo(HRegionInfo.convert(daughter_2_HRI));
718    if (env.getMasterServices().getAssignmentManager().onRegionTransition(
719      getParentRegionState(env).getServerName(), transition.build()) != null) {
720      throw new IOException("Failed to update meta to add daughter regions in split region "
721          + parentHRI.getRegionNameAsString());
722    }
723  }

760  /**
761   * Get parent region state
762   * @param env MasterProcedureEnv
763   * @return parent region state
764   */
765  private RegionState getParentRegionState(final MasterProcedureEnv env) {
766    RegionStates regionStates = env.getMasterServices().getAssignmentManager().getRegionStates();
767    RegionState state = regionStates.getRegionState(parentHRI);
768    if (state == null) {
769      LOG.warn("Split but not in region states: " + parentHRI);
770      state = regionStates.createRegionState(parentHRI);
771    }
772    return state;
