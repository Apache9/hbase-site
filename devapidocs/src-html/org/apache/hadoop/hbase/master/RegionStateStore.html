050/**
051 * A helper to persist region state in meta. We may change this class
052 * to StateStore later if we also use it to store other states in meta
053 */
054@InterfaceAudience.Private
055public class RegionStateStore {
056  private static final Log LOG = LogFactory.getLog(RegionStateStore.class);
057
058  /** The delimiter for meta columns for replicaIds &gt; 0 */
059  protected static final char META_REPLICA_ID_DELIMITER = '_';
060
061  private volatile Region metaRegion;
062  private volatile boolean initialized;
063  private MultiHConnection multiHConnection;
064  private final MasterServices server;
065
066  /**
067   * Returns the {@link ServerName} from catalog table {@link Result}
068   * where the region is transitioning. It should be the same as
069   * {@link MetaTableAccessor#getServerName(Result,int)} if the server is at OPEN state.
070   * @param r Result to pull the transitioning server name from
071   * @return A ServerName instance or {@link MetaTableAccessor#getServerName(Result,int)}
072   * if necessary fields not found or empty.
073   */
074  static ServerName getRegionServer(final Result r, int replicaId) {
075    Cell cell = r.getColumnLatestCell(HConstants.CATALOG_FAMILY, getServerNameColumn(replicaId));
076    if (cell == null || cell.getValueLength() == 0) {
077      RegionLocations locations = MetaTableAccessor.getRegionLocations(r);
078      if (locations != null) {
079        HRegionLocation location = locations.getRegionLocation(replicaId);
080        if (location != null) {
081          return location.getServerName();
082        }
083      }
084      return null;
085    }
086    return ServerName.parseServerName(Bytes.toString(cell.getValueArray(),
087      cell.getValueOffset(), cell.getValueLength()));
088  }
089
090  private static byte[] getServerNameColumn(int replicaId) {
091    return replicaId == 0
092        ? HConstants.SERVERNAME_QUALIFIER
093        : Bytes.toBytes(HConstants.SERVERNAME_QUALIFIER_STR + META_REPLICA_ID_DELIMITER
094          + String.format(HRegionInfo.REPLICA_ID_FORMAT, replicaId));
095  }
096
097  /**
098   * Pull the region state from a catalog table {@link Result}.
099   * @param r Result to pull the region state from
100   * @return the region state, or OPEN if there's no value written.
101   */
102  static State getRegionState(final Result r, int replicaId) {
103    Cell cell = r.getColumnLatestCell(HConstants.CATALOG_FAMILY, getStateColumn(replicaId));
104    if (cell == null || cell.getValueLength() == 0) return State.OPEN;
105    return State.valueOf(Bytes.toString(cell.getValueArray(),
106      cell.getValueOffset(), cell.getValueLength()));
107  }
108
109  private static byte[] getStateColumn(int replicaId) {
110    return replicaId == 0
111        ? HConstants.STATE_QUALIFIER
112        : Bytes.toBytes(HConstants.STATE_QUALIFIER_STR + META_REPLICA_ID_DELIMITER
113          + String.format(HRegionInfo.REPLICA_ID_FORMAT, replicaId));
114  }
115
116  /**
117   * Check if we should persist a state change in meta. Generally it's
118   * better to persist all state changes. However, we should not do that
119   * if the region is not in meta at all. Based on the state and the
120   * previous state, we can identify if a user region has an entry
121   * in meta. For example, merged regions are deleted from meta;
122   * New merging parents, or splitting daughters are
123   * not created in meta yet.
124   */
125  private boolean shouldPersistStateChange(
126      HRegionInfo hri, RegionState state, RegionState oldState) {
127    return !hri.isMetaRegion() && !RegionStates.isOneOfStates(
128      state, State.MERGING_NEW, State.SPLITTING_NEW, State.MERGED)
129      && !(RegionStates.isOneOfStates(state, State.OFFLINE)
130        && RegionStates.isOneOfStates(oldState, State.MERGING_NEW,
131          State.SPLITTING_NEW, State.MERGED));
132  }
133
134  RegionStateStore(final MasterServices server) {
135    this.server = server;
136    initialized = false;
137  }
138
139  void start() throws IOException {
140    if (server instanceof RegionServerServices) {
141      metaRegion = ((RegionServerServices)server).getFromOnlineRegions(
142        HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
143    }
144    // When meta is not colocated on master
145    if (metaRegion == null) {
146      Configuration conf = server.getConfiguration();
147      // Config to determine the no of HConnections to META.
148      // A single Connection should be sufficient in most cases. Only if
149      // you are doing lot of writes (>1M) to META,
150      // increasing this value might improve the write throughput.
151      multiHConnection =
152          new MultiHConnection(conf, conf.getInt("hbase.regionstatestore.meta.connection", 1));
153    }
154    initialized = true;
155  }
156
157  void stop() {
158    initialized = false;
159    if (multiHConnection != null) {
160      multiHConnection.close();
161    }
162  }
163
164  void updateRegionState(long openSeqNum,
165      RegionState newState, RegionState oldState) {
166    try {
167      HRegionInfo hri = newState.getRegion();
168
169      // Update meta before checking for initialization. Meta state stored in zk.
170      if (hri.isMetaRegion()) {
171        // persist meta state in MetaTableLocator (which in turn is zk storage currently)
172        try {
173          MetaTableLocator.setMetaLocation(server.getZooKeeper(),
174            newState.getServerName(), hri.getReplicaId(), newState.getState());
175          return; // Done
176        } catch (KeeperException e) {
177          throw new IOException("Failed to update meta ZNode", e);
178        }
179      }
180
181      if (!initialized
182          || !shouldPersistStateChange(hri, newState, oldState)) {
183        return;
184      }
185
186      ServerName oldServer = oldState != null ? oldState.getServerName() : null;
187      ServerName serverName = newState.getServerName();
188      State state = newState.getState();
189
190      int replicaId = hri.getReplicaId();
191      Put metaPut = new Put(MetaTableAccessor.getMetaKeyForRegion(hri));
192      StringBuilder info = new StringBuilder("Updating hbase:meta row ");
193      info.append(hri.getRegionNameAsString()).append(" with state=").append(state);
194      if (serverName != null && !serverName.equals(oldServer)) {
195        metaPut.addImmutable(HConstants.CATALOG_FAMILY, getServerNameColumn(replicaId),
196          Bytes.toBytes(serverName.getServerName()));
197        info.append(", sn=").append(serverName);
198      }
199      if (openSeqNum >= 0) {
200        Preconditions.checkArgument(state == State.OPEN
201          && serverName != null, "Open region should be on a server");
202        MetaTableAccessor.addLocation(metaPut, serverName, openSeqNum, -1, replicaId);
203        info.append(", openSeqNum=").append(openSeqNum);
204        info.append(", server=").append(serverName);
205      }
206      metaPut.addImmutable(HConstants.CATALOG_FAMILY, getStateColumn(replicaId),
207        Bytes.toBytes(state.name()));
208      LOG.info(info);
209      HTableDescriptor descriptor = server.getTableDescriptors().get(hri.getTable());
210      boolean serial = false;
211      if (descriptor != null) {
212        serial = server.getTableDescriptors().get(hri.getTable()).hasSerialReplicationScope();
213      }
214      boolean shouldPutBarrier = serial && state == State.OPEN;
215      // Persist the state change to meta
216      if (metaRegion != null) {
217        try {
218          // Assume meta is pinned to master.
219          // At least, that's what we want.
220          metaRegion.put(metaPut);
221          if (shouldPutBarrier) {
222            Put barrierPut = MetaTableAccessor.makeBarrierPut(hri.getEncodedNameAsBytes(),
223                openSeqNum, hri.getTable().getName());
224            metaRegion.put(barrierPut);
225          }
226          return; // Done here
227        } catch (Throwable t) {
228          // In unit tests, meta could be moved away by intention
229          // So, the shortcut is gone. We won't try to establish the
230          // shortcut any more because we prefer meta to be pinned
231          // to the master
232          synchronized (this) {
233            if (metaRegion != null) {
234              LOG.info("Meta region shortcut failed", t);
235              if (multiHConnection == null) {
236                multiHConnection = new MultiHConnection(server.getConfiguration(), 1);
237              }
238              metaRegion = null;
239            }
240          }
241        }
242      }
243      // Called when meta is not on master
244      List<Put> list = shouldPutBarrier ?
245          Arrays.asList(metaPut, MetaTableAccessor.makeBarrierPut(hri.getEncodedNameAsBytes(),
246              openSeqNum, hri.getTable().getName())) : Arrays.asList(metaPut);
247      multiHConnection.processBatchCallback(list, TableName.META_TABLE_NAME, null, null);
248
249    } catch (IOException ioe) {
250      LOG.error("Failed to persist region state " + newState, ioe);
251      server.abort("Failed to update region location", ioe);
252    }
253  }
254
255  void splitRegion(HRegionInfo p,
256      HRegionInfo a, HRegionInfo b, ServerName sn, int regionReplication) throws IOException {
257    MetaTableAccessor.splitRegion(server.getConnection(), p, a, b, sn, regionReplication,
258        server.getTableDescriptors().get(p.getTable()).hasSerialReplicationScope());
259  }
260
261  void mergeRegions(HRegionInfo p,
262      HRegionInfo a, HRegionInfo b, ServerName sn, int regionReplication) throws IOException {
263    MetaTableAccessor.mergeRegions(server.getConnection(), p, a, b, sn, regionReplication,
264        EnvironmentEdgeManager.currentTime(),
265        server.getTableDescriptors().get(p.getTable()).hasSerialReplicationScope());
266  }
