018package org.apache.hadoop.hbase.master.replication;
019
020import java.io.IOException;
021import org.apache.hadoop.hbase.master.procedure.MasterProcedureEnv;
022import org.apache.hadoop.hbase.master.procedure.PeerProcedureInterface;
023import org.apache.hadoop.hbase.master.procedure.ProcedurePrepareLatch;
024import org.apache.hadoop.hbase.procedure2.ProcedureStateSerializer;
025import org.apache.hadoop.hbase.procedure2.ProcedureSuspendedException;
026import org.apache.hadoop.hbase.procedure2.ProcedureYieldException;
027import org.apache.hadoop.hbase.procedure2.StateMachineProcedure;
028import org.apache.hadoop.hbase.replication.ReplicationException;
029import org.apache.yetus.audience.InterfaceAudience;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.ModifyPeerStateData;
034import org.apache.hadoop.hbase.shaded.protobuf.generated.MasterProcedureProtos.PeerModificationState;
035
036/**
037 * The base class for all replication peer related procedure.
038 */
039@InterfaceAudience.Private
040public abstract class ModifyPeerProcedure
041    extends StateMachineProcedure<MasterProcedureEnv, PeerModificationState>
042    implements PeerProcedureInterface {
043
044  private static final Logger LOG = LoggerFactory.getLogger(ModifyPeerProcedure.class);
045
046  protected String peerId;
047
048  private volatile boolean locked;
049
050  // used to keep compatible with old client where we can only returns after updateStorage.
051  protected ProcedurePrepareLatch latch;
052
053  protected ModifyPeerProcedure() {
054  }
055
056  protected ModifyPeerProcedure(String peerId) {
057    this.peerId = peerId;
058    this.latch = ProcedurePrepareLatch.createLatch(2, 0);
059  }
060
061  public ProcedurePrepareLatch getLatch() {
062    return latch;
063  }
064
065  @Override
066  public String getPeerId() {
067    return peerId;
068  }
069
070  /**
071   * Called before we start the actual processing. The implementation should call the pre CP hook,
072   * and also the pre-check for the peer modification.
073   * <p>
074   * If an IOException is thrown then we will give up and mark the procedure as failed directly. If
075   * all checks passes then the procedure can not be rolled back any more.
076   */
077  protected abstract void prePeerModification(MasterProcedureEnv env)
078      throws IOException, ReplicationException;
079
080  protected abstract void updatePeerStorage(MasterProcedureEnv env) throws ReplicationException;
081
082  /**
083   * Called before we finish the procedure. The implementation can do some logging work, and also
084   * call the coprocessor hook if any.
085   * <p>
086   * Notice that, since we have already done the actual work, throwing {@code IOException} here will
087   * not fail this procedure, we will just ignore it and finish the procedure as suceeded. If
088   * {@code ReplicationException} is thrown we will retry since this usually means we fails to
089   * update the peer storage.
090   */
091  protected abstract void postPeerModification(MasterProcedureEnv env)
092      throws IOException, ReplicationException;
093
094  private void releaseLatch() {
095    ProcedurePrepareLatch.releaseLatch(latch, this);
096  }
097
098  @Override
099  protected Flow executeFromState(MasterProcedureEnv env, PeerModificationState state)
100      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
101    switch (state) {
102      case PRE_PEER_MODIFICATION:
103        try {
104          prePeerModification(env);
105        } catch (IOException e) {
106          LOG.warn("{} failed to call pre CP hook or the pre check is failed for peer {}, " +
107            "mark the procedure as failure and give up", getClass().getName(), peerId, e);
108          setFailure("master-" + getPeerOperationType().name().toLowerCase() + "-peer", e);
109          releaseLatch();
110          return Flow.NO_MORE_STATE;
111        } catch (ReplicationException e) {
112          LOG.warn("{} failed to call prePeerModification for peer {}, retry", getClass().getName(),
113            peerId, e);
114          throw new ProcedureYieldException();
115        }
116        setNextState(PeerModificationState.UPDATE_PEER_STORAGE);
117        return Flow.HAS_MORE_STATE;
118      case UPDATE_PEER_STORAGE:
119        try {
120          updatePeerStorage(env);
121        } catch (ReplicationException e) {
122          LOG.warn("{} update peer storage for peer {} failed, retry", getClass().getName(), peerId,
123            e);
124          throw new ProcedureYieldException();
125        }
126        setNextState(PeerModificationState.REFRESH_PEER_ON_RS);
127        return Flow.HAS_MORE_STATE;
128      case REFRESH_PEER_ON_RS:
129        addChildProcedure(env.getMasterServices().getServerManager().getOnlineServersList().stream()
130            .map(sn -> new RefreshPeerProcedure(peerId, getPeerOperationType(), sn))
131            .toArray(RefreshPeerProcedure[]::new));
132        setNextState(PeerModificationState.POST_PEER_MODIFICATION);
133        return Flow.HAS_MORE_STATE;
134      case POST_PEER_MODIFICATION:
135        try {
136          postPeerModification(env);
137        } catch (ReplicationException e) {
138          LOG.warn("{} failed to call postPeerModification for peer {}, retry",
139            getClass().getName(), peerId, e);
140          throw new ProcedureYieldException();
141        } catch (IOException e) {
142          LOG.warn("{} failed to call post CP hook for peer {}, " +
143            "ignore since the procedure has already done", getClass().getName(), peerId, e);
144        }
145        releaseLatch();
146        return Flow.NO_MORE_STATE;
147      default:
148        throw new UnsupportedOperationException("unhandled state=" + state);
149    }
150  }
151
152  @Override
153  protected LockState acquireLock(MasterProcedureEnv env) {
154    if (env.getProcedureScheduler().waitPeerExclusiveLock(this, peerId)) {
155      return  LockState.LOCK_EVENT_WAIT;
156    }
157    locked = true;
158    return LockState.LOCK_ACQUIRED;
159  }
160
161  @Override
162  protected void releaseLock(MasterProcedureEnv env) {
163    locked = false;
164    env.getProcedureScheduler().wakePeerExclusiveLock(this, peerId);
165  }
166
167  @Override
168  protected boolean holdLock(MasterProcedureEnv env) {
169    return true;
170  }
171
172  @Override
173  protected boolean hasLock(MasterProcedureEnv env) {
174    return locked;
175  }
176
177  @Override
178  protected void rollbackState(MasterProcedureEnv env, PeerModificationState state)
179      throws IOException, InterruptedException {
180    if (state == PeerModificationState.PRE_PEER_MODIFICATION) {
181      // actually the peer related operations has no rollback, but if we haven't done any
182      // modifications on the peer storage yet, we can just return.
183      return;
184    }
185    throw new UnsupportedOperationException();
186  }
187
188  @Override
189  protected PeerModificationState getState(int stateId) {
190    return PeerModificationState.forNumber(stateId);
191  }
192
193  @Override
194  protected int getStateId(PeerModificationState state) {
195    return state.getNumber();
196  }
197
198  @Override
199  protected PeerModificationState getInitialState() {
200    return PeerModificationState.PRE_PEER_MODIFICATION;
201  }
202
203  @Override
204  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
205    super.serializeStateData(serializer);
206    serializer.serialize(ModifyPeerStateData.newBuilder().setPeerId(peerId).build());
207  }
208
209  @Override
210  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
211    super.deserializeStateData(serializer);
212    peerId = serializer.deserialize(ModifyPeerStateData.class).getPeerId();
213  }
214}
