044/**
045 * Manages and performs all replication admin operations.
046 * Used to add/remove a replication peer.
047 */
048@InterfaceAudience.Private
049public class ReplicationManager {
050
051  private final Configuration conf;
052  private final ZKWatcher zkw;
053  private final ReplicationQueuesClient replicationQueuesClient;
054  private final ReplicationPeers replicationPeers;
055
056  public ReplicationManager(Configuration conf, ZKWatcher zkw, Abortable abortable)
057      throws IOException {
058    this.conf = conf;
059    this.zkw = zkw;
060    try {
061      this.replicationQueuesClient = ReplicationFactory
062          .getReplicationQueuesClient(new ReplicationQueuesClientArguments(conf, abortable, zkw));
063      this.replicationQueuesClient.init();
064      this.replicationPeers = ReplicationFactory.getReplicationPeers(zkw, conf,
065        this.replicationQueuesClient, abortable);
066      this.replicationPeers.init();
067    } catch (Exception e) {
068      throw new IOException("Failed to construct ReplicationManager", e);
069    }
070  }
071
072  public void addReplicationPeer(String peerId, ReplicationPeerConfig peerConfig, boolean enabled)
073      throws ReplicationException, IOException {
074    checkNamespacesAndTableCfsConfigConflict(peerConfig.getNamespaces(),
075      peerConfig.getTableCFsMap());
076    checkConfiguredWALEntryFilters(peerConfig);
077    replicationPeers.registerPeer(peerId, peerConfig, enabled);
078    replicationPeers.peerConnected(peerId);
079  }
080
081  public void removeReplicationPeer(String peerId) throws ReplicationException {
082    replicationPeers.peerDisconnected(peerId);
083    replicationPeers.unregisterPeer(peerId);
084  }
085
086  public void enableReplicationPeer(String peerId) throws ReplicationException {
087    this.replicationPeers.enablePeer(peerId);
088  }
089
090  public void disableReplicationPeer(String peerId) throws ReplicationException {
091    this.replicationPeers.disablePeer(peerId);
092  }
093
094  public ReplicationPeerConfig getPeerConfig(String peerId) throws ReplicationException,
095      ReplicationPeerNotFoundException {
096    ReplicationPeerConfig peerConfig = replicationPeers.getReplicationPeerConfig(peerId);
097    if (peerConfig == null) {
098      throw new ReplicationPeerNotFoundException(peerId);
099    }
100    return peerConfig;
101  }
102
103  public void updatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
104      throws ReplicationException, IOException {
105    checkNamespacesAndTableCfsConfigConflict(peerConfig.getNamespaces(),
106      peerConfig.getTableCFsMap());
107    checkConfiguredWALEntryFilters(peerConfig);
108    this.replicationPeers.updatePeerConfig(peerId, peerConfig);
109  }
110
111  public List<ReplicationPeerDescription> listReplicationPeers(Pattern pattern)
112      throws ReplicationException {
113    List<ReplicationPeerDescription> peers = new ArrayList<>();
114    List<String> peerIds = replicationPeers.getAllPeerIds();
115    for (String peerId : peerIds) {
116      if (pattern == null || (pattern != null && pattern.matcher(peerId).matches())) {
117        peers.add(new ReplicationPeerDescription(peerId, replicationPeers
118            .getStatusOfPeerFromBackingStore(peerId), replicationPeers
119            .getReplicationPeerConfig(peerId)));
120      }
121    }
122    return peers;
123  }
124
125  /**
126   * Set a namespace in the peer config means that all tables in this namespace
127   * will be replicated to the peer cluster.
128   *
129   * 1. If you already have set a namespace in the peer config, then you can't set any table
130   *    of this namespace to the peer config.
131   * 2. If you already have set a table in the peer config, then you can't set this table's
132   *    namespace to the peer config.
133   *
134   * @param namespaces
135   * @param tableCfs
136   * @throws ReplicationException
137   */
138  private void checkNamespacesAndTableCfsConfigConflict(Set<String> namespaces,
139      Map<TableName, ? extends Collection<String>> tableCfs) throws ReplicationException {
140    if (namespaces == null || namespaces.isEmpty()) {
141      return;
142    }
143    if (tableCfs == null || tableCfs.isEmpty()) {
144      return;
145    }
146    for (Map.Entry<TableName, ? extends Collection<String>> entry : tableCfs.entrySet()) {
147      TableName table = entry.getKey();
148      if (namespaces.contains(table.getNamespaceAsString())) {
149        throw new ReplicationException(
150            "Table-cfs config conflict with namespaces config in peer");
151      }
152    }
153
154
155  }
156
157  private void checkConfiguredWALEntryFilters(ReplicationPeerConfig peerConfig)
158      throws IOException {
159    String filterCSV = peerConfig.getConfiguration().
160        get(BaseReplicationEndpoint.REPLICATION_WALENTRYFILTER_CONFIG_KEY);
161    if (filterCSV != null && !filterCSV.isEmpty()){
162      String [] filters = filterCSV.split(",");
163      for (String filter : filters) {
164        try {
165          Class clazz = Class.forName(filter);
166          Object o = clazz.newInstance();
167        } catch (Exception e) {
168          throw new DoNotRetryIOException("Configured WALEntryFilter " + filter +
169              " could not be created. Failing add/update " + "peer operation.", e);
170        }
171      }
172    }
173  }
