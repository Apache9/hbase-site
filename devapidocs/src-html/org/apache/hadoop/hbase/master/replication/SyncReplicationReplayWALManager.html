018package org.apache.hadoop.hbase.master.replication;
019
020import static org.apache.hadoop.hbase.replication.ReplicationUtils.REMOTE_WAL_REPLAY_SUFFIX;
021import static org.apache.hadoop.hbase.replication.ReplicationUtils.getPeerRemoteWALDir;
022import static org.apache.hadoop.hbase.replication.ReplicationUtils.getPeerReplayWALDir;
023import static org.apache.hadoop.hbase.replication.ReplicationUtils.getPeerSnapshotWALDir;
024
025import java.io.IOException;
026import java.util.ArrayList;
027import java.util.HashMap;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Map;
031import java.util.Optional;
032import java.util.Set;
033
034import org.apache.hadoop.fs.FileStatus;
035import org.apache.hadoop.fs.FileSystem;
036import org.apache.hadoop.fs.Path;
037import org.apache.hadoop.hbase.ServerName;
038import org.apache.hadoop.hbase.master.MasterServices;
039import org.apache.hadoop.hbase.replication.ReplicationException;
040import org.apache.hadoop.hbase.replication.ReplicationUtils;

048@InterfaceAudience.Private
049public class SyncReplicationReplayWALManager {
050
051  private static final Logger LOG =
052      LoggerFactory.getLogger(SyncReplicationReplayWALManager.class);
053
054  private final MasterServices services;
055
056  private final FileSystem fs;
057
058  private final Path walRootDir;
059
060  private final Path remoteWALDir;
061
062  private final ZKSyncReplicationReplayWALWorkerStorage workerStorage;
063
064  private final Map<String, Set<ServerName>> workers = new HashMap<>();
065
066  private final Object workerLock = new Object();
067
068  public SyncReplicationReplayWALManager(MasterServices services)
069      throws IOException, ReplicationException {
070    this.services = services;
071    this.fs = services.getMasterFileSystem().getWALFileSystem();
072    this.walRootDir = services.getMasterFileSystem().getWALRootDir();
073    this.remoteWALDir = new Path(this.walRootDir, ReplicationUtils.REMOTE_WAL_DIR_NAME);
074    this.workerStorage = new ZKSyncReplicationReplayWALWorkerStorage(services.getZooKeeper(),
075        services.getConfiguration());
076    checkReplayingWALDir();
077  }
078
079  private void checkReplayingWALDir() throws IOException, ReplicationException {
080    FileStatus[] files = fs.listStatus(remoteWALDir);
081    for (FileStatus file : files) {
082      String name = file.getPath().getName();
083      if (name.endsWith(REMOTE_WAL_REPLAY_SUFFIX)) {
084        String peerId = name.substring(0, name.length() - REMOTE_WAL_REPLAY_SUFFIX.length());
085        workers.put(peerId, workerStorage.getPeerWorkers(peerId));
086      }
087    }
088  }
089
090  public void registerPeer(String peerId) throws ReplicationException {
091    workers.put(peerId, new HashSet<>());
092    workerStorage.addPeer(peerId);
093  }
094
095  public void unregisterPeer(String peerId) throws ReplicationException {
096    workers.remove(peerId);
097    workerStorage.removePeer(peerId);
098  }
099
100  public ServerName getPeerWorker(String peerId) throws ReplicationException {
101    Optional<ServerName> worker = Optional.empty();
102    ServerName workerServer = null;
103    synchronized (workerLock) {
104      worker = services.getServerManager().getOnlineServers().keySet().stream()
105          .filter(server -> !workers.get(peerId).contains(server)).findFirst();
106      if (worker.isPresent()) {
107        workerServer = worker.get();
108        workers.get(peerId).add(workerServer);
109      }
110    }
111    if (workerServer != null) {
112      workerStorage.addPeerWorker(peerId, workerServer);
113    }
114    return workerServer;
115  }
116
117  public void removePeerWorker(String peerId, ServerName worker) throws ReplicationException {
118    synchronized (workerLock) {
119      workers.get(peerId).remove(worker);
120    }
121    workerStorage.removePeerWorker(peerId, worker);
122  }
123  public void createPeerRemoteWALDir(String peerId) throws IOException {
124    Path peerRemoteWALDir = getPeerRemoteWALDir(remoteWALDir, peerId);
125    if (!fs.exists(peerRemoteWALDir) && !fs.mkdirs(peerRemoteWALDir)) {
126      throw new IOException("Unable to mkdir " + peerRemoteWALDir);
127    }
128  }
129
130  private void rename(Path src, Path dst, String peerId) throws IOException {
131    if (fs.exists(src)) {
132      deleteDir(dst, peerId);
133      if (!fs.rename(src, dst)) {
134        throw new IOException(
135            "Failed to rename dir from " + src + " to " + dst + " for peer id=" + peerId);
136      }
137      LOG.info("Renamed dir from {} to {} for peer id={}", src, dst, peerId);
138    } else if (!fs.exists(dst)) {
139      throw new IOException(
140          "Want to rename from " + src + " to " + dst + ", but they both do not exist");
141    }
142  }
143
144  public void renameToPeerReplayWALDir(String peerId) throws IOException {
145    rename(getPeerRemoteWALDir(remoteWALDir, peerId), getPeerReplayWALDir(remoteWALDir, peerId),
146        peerId);
147  }
148
149  public void renameToPeerSnapshotWALDir(String peerId) throws IOException {
150    rename(getPeerReplayWALDir(remoteWALDir, peerId), getPeerSnapshotWALDir(remoteWALDir, peerId),
151        peerId);
152  }
153
154  public List<Path> getReplayWALsAndCleanUpUnusedFiles(String peerId) throws IOException {
155    Path peerReplayWALDir = getPeerReplayWALDir(remoteWALDir, peerId);
156    for (FileStatus status : fs.listStatus(peerReplayWALDir,
157      p -> p.getName().endsWith(ReplicationUtils.RENAME_WAL_SUFFIX))) {
158      Path src = status.getPath();
159      String srcName = src.getName();
160      String dstName =
161          srcName.substring(0, srcName.length() - ReplicationUtils.RENAME_WAL_SUFFIX.length());
162      FSUtils.renameFile(fs, src, new Path(src.getParent(), dstName));
163    }
164    List<Path> wals = new ArrayList<>();
165    for (FileStatus status : fs.listStatus(peerReplayWALDir)) {
166      Path path = status.getPath();
167      if (path.getName().endsWith(ReplicationUtils.SYNC_WAL_SUFFIX)) {
168        wals.add(path);
169      } else {
170        if (!fs.delete(path, true)) {
171          LOG.warn("Can not delete unused file: " + path);
172        }
173      }
174    }
175    return wals;
176  }
177
178  public void snapshotPeerReplayWALDir(String peerId) throws IOException {
179    Path peerReplayWALDir = getPeerReplayWALDir(remoteWALDir, peerId);
180    if (fs.exists(peerReplayWALDir) && !fs.delete(peerReplayWALDir, true)) {
181      throw new IOException(
182          "Failed to remove replay wals dir " + peerReplayWALDir + " for peer id=" + peerId);
183    }
184  }
185
186  private void deleteDir(Path dir, String peerId) throws IOException {
187    if (!fs.delete(dir, true) && fs.exists(dir)) {
188      throw new IOException("Failed to remove dir " + dir + " for peer id=" + peerId);
189    }
190  }
191
192  public void removePeerRemoteWALs(String peerId) throws IOException {
193    deleteDir(getPeerRemoteWALDir(remoteWALDir, peerId), peerId);
194    deleteDir(getPeerReplayWALDir(remoteWALDir, peerId), peerId);
195    deleteDir(getPeerSnapshotWALDir(remoteWALDir, peerId), peerId);
196  }
197
198  public String removeWALRootPath(Path path) {
199    String pathStr = path.toString();
200    // remove the "/" too.
201    return pathStr.substring(walRootDir.toString().length() + 1);
202  }
203
204  public void finishReplayWAL(String wal) throws IOException {
205    Path walPath = new Path(walRootDir, wal);
206    fs.truncate(walPath, 0);
207  }
208
209  public boolean isReplayWALFinished(String wal) throws IOException {
210    Path walPath = new Path(walRootDir, wal);
211    return fs.getFileStatus(walPath).getLen() == 0;
212  }
213
214  @VisibleForTesting
215  public Path getRemoteWALDir() {
216    return remoteWALDir;
217  }
