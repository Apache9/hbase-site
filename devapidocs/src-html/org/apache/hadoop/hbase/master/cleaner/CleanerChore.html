041/**
042 * Abstract Cleaner that uses a chain of delegates to clean a directory of files
043 * @param <T> Cleaner delegate class that is dynamically loaded from configuration
044 */
045public abstract class CleanerChore<T extends FileCleanerDelegate> extends ScheduledChore {
046
047  private static final Log LOG = LogFactory.getLog(CleanerChore.class.getName());
048
049  private final FileSystem fs;
050  private final Path oldFileDir;
051  private final Configuration conf;
052  protected List<T> cleanersChain;
053  protected Map<String, Object> params;
054  private AtomicBoolean enabled = new AtomicBoolean(true);
055
056  public CleanerChore(String name, final int sleepPeriod, final Stoppable s, Configuration conf,
057                      FileSystem fs, Path oldFileDir, String confKey) {
058    this(name, sleepPeriod, s, conf, fs, oldFileDir, confKey, null);
059  }
060
061  /**
062   * @param name name of the chore being run
063   * @param sleepPeriod the period of time to sleep between each run
064   * @param s the stopper
065   * @param conf configuration to use
066   * @param fs handle to the FS
067   * @param oldFileDir the path to the archived files
068   * @param confKey configuration key for the classes to instantiate
069   * @param params members could be used in cleaner
070   */
071  public CleanerChore(String name, final int sleepPeriod, final Stoppable s, Configuration conf,
072      FileSystem fs, Path oldFileDir, String confKey, Map<String, Object> params) {
073    super(name, s, sleepPeriod);
074    this.fs = fs;
075    this.oldFileDir = oldFileDir;
076    this.conf = conf;
077    this.params = params;
078    initCleanerChain(confKey);
079  }
080
081
082  /**
083   * Validate the file to see if it even belongs in the directory. If it is valid, then the file
084   * will go through the cleaner delegates, but otherwise the file is just deleted.
085   * @param file full {@link Path} of the file to be checked
086   * @return <tt>true</tt> if the file is valid, <tt>false</tt> otherwise
087   */
088  protected abstract boolean validate(Path file);
089
090  /**
091   * Instantiate and initialize all the file cleaners set in the configuration
092   * @param confKey key to get the file cleaner classes from the configuration
093   */
094  private void initCleanerChain(String confKey) {
095    this.cleanersChain = new LinkedList<>();
096    String[] logCleaners = conf.getStrings(confKey);
097    if (logCleaners != null) {
098      for (String className : logCleaners) {
099        T logCleaner = newFileCleaner(className, conf);
100        if (logCleaner != null) {
101          LOG.debug("initialize cleaner=" + className);
102          this.cleanersChain.add(logCleaner);
103        }
104      }
105    }
106  }
107
108  /**
109   * A utility method to create new instances of LogCleanerDelegate based on the class name of the
110   * LogCleanerDelegate.
111   * @param className fully qualified class name of the LogCleanerDelegate
112   * @param conf
113   * @return the new instance
114   */
115  private T newFileCleaner(String className, Configuration conf) {
116    try {
117      Class<? extends FileCleanerDelegate> c = Class.forName(className).asSubclass(
118        FileCleanerDelegate.class);
119      @SuppressWarnings("unchecked")
120      T cleaner = (T) c.newInstance();
121      cleaner.setConf(conf);
122      cleaner.init(this.params);
123      return cleaner;
124    } catch (Exception e) {
125      LOG.warn("Can NOT create CleanerDelegate: " + className, e);
126      // skipping if can't instantiate
127      return null;
128    }
129  }
130
131  @Override
132  protected void chore() {
133    if (getEnabled()) {
134      runCleaner();
135    } else {
136      LOG.debug("Cleaner chore disabled! Not cleaning.");
137    }
138  }
139
140  public Boolean runCleaner() {
141    try {
142      FileStatus[] files = FSUtils.listStatus(this.fs, this.oldFileDir);
143      checkAndDeleteEntries(files);
144    } catch (IOException e) {
145      e = e instanceof RemoteException ?
146              ((RemoteException)e).unwrapRemoteException() : e;
147      LOG.warn("Error while cleaning the logs", e);
148      return false;
149    }
150    return true;
151  }
152
153  /**
154   * Loop over the given directory entries, and check whether they can be deleted.
155   * If an entry is itself a directory it will be recursively checked and deleted itself iff
156   * all subentries are deleted (and no new subentries are added in the mean time)
157   *
158   * @param entries directory entries to check
159   * @return true if all entries were successfully deleted
160   */
161  private boolean checkAndDeleteEntries(FileStatus[] entries) {
162    if (entries == null) {
163      return true;
164    }
165    boolean allEntriesDeleted = true;
166    List<FileStatus> files = Lists.newArrayListWithCapacity(entries.length);
167    for (FileStatus child : entries) {
168      Path path = child.getPath();
169      if (child.isDirectory()) {
170        // for each subdirectory delete it and all entries if possible
171        if (!checkAndDeleteDirectory(path)) {
172          allEntriesDeleted = false;
173        }
174      } else {
175        // collect all files to attempt to delete in one batch
176        files.add(child);
177      }
178    }
179    if (!checkAndDeleteFiles(files)) {
180      allEntriesDeleted = false;
181    }
182    return allEntriesDeleted;
183  }
184  
185  /**
186   * Attempt to delete a directory and all files under that directory. Each child file is passed
187   * through the delegates to see if it can be deleted. If the directory has no children when the
188   * cleaners have finished it is deleted.
189   * <p>
190   * If new children files are added between checks of the directory, the directory will <b>not</b>
191   * be deleted.
192   * @param dir directory to check
193   * @return <tt>true</tt> if the directory was deleted, <tt>false</tt> otherwise.
194   */
195  @VisibleForTesting boolean checkAndDeleteDirectory(Path dir) {
196    if (LOG.isTraceEnabled()) {
197      LOG.trace("Checking directory: " + dir);
198    }
199
200    try {
201      FileStatus[] children = FSUtils.listStatus(fs, dir);
202      boolean allChildrenDeleted = checkAndDeleteEntries(children);
203  
204      // if the directory still has children, we can't delete it, so we are done
205      if (!allChildrenDeleted) return false;
206    } catch (IOException e) {
207      e = e instanceof RemoteException ?
208              ((RemoteException)e).unwrapRemoteException() : e;
209      LOG.warn("Error while listing directory: " + dir, e);
210      // couldn't list directory, so don't try to delete, and don't return success
211      return false;
212    }
213
214    // otherwise, all the children (that we know about) have been deleted, so we should try to
215    // delete this directory. However, don't do so recursively so we don't delete files that have
216    // been added since we last checked.
217    try {
218      return fs.delete(dir, false);
219    } catch (IOException e) {
220      if (LOG.isTraceEnabled()) {
221        LOG.trace("Couldn't delete directory: " + dir, e);
222      }
223      // couldn't delete w/o exception, so we can't return success.
224      return false;
225    }
226  }
227
228  /**
229   * Run the given files through each of the cleaners to see if it should be deleted, deleting it if
230   * necessary.
231   * @param files List of FileStatus for the files to check (and possibly delete)
232   * @return true iff successfully deleted all files
233   */
234  private boolean checkAndDeleteFiles(List<FileStatus> files) {
235    // first check to see if the path is valid
236    List<FileStatus> validFiles = Lists.newArrayListWithCapacity(files.size());
237    List<FileStatus> invalidFiles = Lists.newArrayList();
238    for (FileStatus file : files) {
239      if (validate(file.getPath())) {
240        validFiles.add(file);
241      } else {
242        LOG.warn("Found a wrongly formatted file: " + file.getPath() + " - will delete it.");
243        invalidFiles.add(file);
244      }
245    }
246
247    Iterable<FileStatus> deletableValidFiles = validFiles;
248    // check each of the cleaners for the valid files
249    for (T cleaner : cleanersChain) {
250      if (cleaner.isStopped() || this.getStopper().isStopped()) {
251        LOG.warn("A file cleaner" + this.getName() + " is stopped, won't delete any more files in:"
252            + this.oldFileDir);
253        return false;
254      }
255
256      Iterable<FileStatus> filteredFiles = cleaner.getDeletableFiles(deletableValidFiles);
257      
258      // trace which cleaner is holding on to each file
259      if (LOG.isTraceEnabled()) {
260        ImmutableSet<FileStatus> filteredFileSet = ImmutableSet.copyOf(filteredFiles);
261        for (FileStatus file : deletableValidFiles) {
262          if (!filteredFileSet.contains(file)) {
263            LOG.trace(file.getPath() + " is not deletable according to:" + cleaner);
264          }
265        }
266      }
267      
268      deletableValidFiles = filteredFiles;
269    }
270    
271    Iterable<FileStatus> filesToDelete = Iterables.concat(invalidFiles, deletableValidFiles);
272    int deletedFileCount = 0;
273    for (FileStatus file : filesToDelete) {
274      Path filePath = file.getPath();
275      if (LOG.isDebugEnabled()) {
276        LOG.debug("Removing: " + filePath + " from archive");
277      }
278      try {
279        boolean success = this.fs.delete(filePath, false);
280        if (success) {
281          deletedFileCount++;
282        } else {
283          LOG.warn("Attempted to delete:" + filePath
284              + ", but couldn't. Run cleaner chain and attempt to delete on next pass.");
285        }
286      } catch (IOException e) {
287        e = e instanceof RemoteException ?
288                  ((RemoteException)e).unwrapRemoteException() : e;
289        LOG.warn("Error while deleting: " + filePath, e);
290      }
291    }
292
293    return deletedFileCount == files.size();
294  }
295
296  @Override
297  public void cleanup() {
298    for (T lc : this.cleanersChain) {
299      try {
300        lc.stop("Exiting");
301      } catch (Throwable t) {
302        LOG.warn("Stopping", t);
303      }
304    }
305  }
306
307  /**
308   * @param enabled
309   */
310  public boolean setEnabled(final boolean enabled) {
311    return this.enabled.getAndSet(enabled);
312  }
313
314  public boolean getEnabled() {
315    return this.enabled.get();
316  }
