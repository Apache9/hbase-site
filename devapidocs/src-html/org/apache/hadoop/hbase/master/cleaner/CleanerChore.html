046/**
047 * Abstract Cleaner that uses a chain of delegates to clean a directory of files
048 * @param <T> Cleaner delegate class that is dynamically loaded from configuration
049 */
050public abstract class CleanerChore<T extends FileCleanerDelegate> extends ScheduledChore {
051
052  private static final Log LOG = LogFactory.getLog(CleanerChore.class.getName());
053
054  protected final FileSystem fs;
055  private final Path oldFileDir;
056  private final Configuration conf;
057  protected List<T> cleanersChain;
058  protected Map<String, Object> params;
059  private AtomicBoolean enabled = new AtomicBoolean(true);
060
061  public CleanerChore(String name, final int sleepPeriod, final Stoppable s, Configuration conf,
062                      FileSystem fs, Path oldFileDir, String confKey) {
063    this(name, sleepPeriod, s, conf, fs, oldFileDir, confKey, null);
064  }
065
066  /**
067   * @param name name of the chore being run
068   * @param sleepPeriod the period of time to sleep between each run
069   * @param s the stopper
070   * @param conf configuration to use
071   * @param fs handle to the FS
072   * @param oldFileDir the path to the archived files
073   * @param confKey configuration key for the classes to instantiate
074   * @param params members could be used in cleaner
075   */
076  public CleanerChore(String name, final int sleepPeriod, final Stoppable s, Configuration conf,
077      FileSystem fs, Path oldFileDir, String confKey, Map<String, Object> params) {
078    super(name, s, sleepPeriod);
079    this.fs = fs;
080    this.oldFileDir = oldFileDir;
081    this.conf = conf;
082    this.params = params;
083    initCleanerChain(confKey);
084  }
085
086
087  /**
088   * Validate the file to see if it even belongs in the directory. If it is valid, then the file
089   * will go through the cleaner delegates, but otherwise the file is just deleted.
090   * @param file full {@link Path} of the file to be checked
091   * @return <tt>true</tt> if the file is valid, <tt>false</tt> otherwise
092   */
093  protected abstract boolean validate(Path file);
094
095  /**
096   * Instantiate and initialize all the file cleaners set in the configuration
097   * @param confKey key to get the file cleaner classes from the configuration
098   */
099  private void initCleanerChain(String confKey) {
100    this.cleanersChain = new LinkedList<>();
101    String[] logCleaners = conf.getStrings(confKey);
102    if (logCleaners != null) {
103      for (String className : logCleaners) {
104        T logCleaner = newFileCleaner(className, conf);
105        if (logCleaner != null) {
106          LOG.debug("initialize cleaner=" + className);
107          this.cleanersChain.add(logCleaner);
108        }
109      }
110    }
111  }
112
113  /**
114   * A utility method to create new instances of LogCleanerDelegate based on the class name of the
115   * LogCleanerDelegate.
116   * @param className fully qualified class name of the LogCleanerDelegate
117   * @param conf
118   * @return the new instance
119   */
120  private T newFileCleaner(String className, Configuration conf) {
121    try {
122      Class<? extends FileCleanerDelegate> c = Class.forName(className).asSubclass(
123        FileCleanerDelegate.class);
124      @SuppressWarnings("unchecked")
125      T cleaner = (T) c.newInstance();
126      cleaner.setConf(conf);
127      cleaner.init(this.params);
128      return cleaner;
129    } catch (Exception e) {
130      LOG.warn("Can NOT create CleanerDelegate: " + className, e);
131      // skipping if can't instantiate
132      return null;
133    }
134  }
135
136  @Override
137  protected void chore() {
138    if (getEnabled()) {
139      runCleaner();
140    } else {
141      LOG.debug("Cleaner chore disabled! Not cleaning.");
142    }
143  }
144
145  public Boolean runCleaner() {
146    try {
147      FileStatus[] files = FSUtils.listStatus(this.fs, this.oldFileDir);
148      checkAndDeleteEntries(files);
149    } catch (IOException e) {
150      e = e instanceof RemoteException ?
151              ((RemoteException)e).unwrapRemoteException() : e;
152      LOG.warn("Error while cleaning the logs", e);
153      return false;
154    }
155    return true;
156  }
157
158  /**
159   * Sort the given list in (descending) order of the space each element takes
160   * @param dirs the list to sort, element in it should be directory (not file)
161   */
162  private void sortByConsumedSpace(List<FileStatus> dirs) {
163    if (dirs == null || dirs.size() < 2) {
164      // no need to sort for empty or single directory
165      return;
166    }
167    Collections.sort(dirs, new Comparator<FileStatus>() {
168      HashMap<FileStatus, Long> directorySpaces = new HashMap<FileStatus, Long>();
169
170      @Override
171      public int compare(FileStatus f1, FileStatus f2) {
172        long f1ConsumedSpace = getSpace(f1);
173        long f2ConsumedSpace = getSpace(f2);
174        return (f1ConsumedSpace > f2ConsumedSpace) ? -1
175            : (f1ConsumedSpace < f2ConsumedSpace ? 1 : 0);
176      }
177
178      private long getSpace(FileStatus f) {
179        Long cached = directorySpaces.get(f);
180        if (cached != null) {
181          return cached;
182        }
183        try {
184          long space =
185              f.isDirectory() ? fs.getContentSummary(f.getPath()).getSpaceConsumed() : f.getLen();
186          directorySpaces.put(f, space);
187          return space;
188        } catch (IOException e) {
189          if (LOG.isTraceEnabled()) {
190            LOG.trace("failed to get space consumed by path " + f.getPath(), e);
191          }
192          return -1;
193        }
194      }
195    });
196  }
197
198  /**
199   * Loop over the given directory entries, and check whether they can be deleted.
200   * If an entry is itself a directory it will be recursively checked and deleted itself iff
201   * all subentries are deleted (and no new subentries are added in the mean time)
202   *
203   * @param entries directory entries to check
204   * @return true if all entries were successfully deleted
205   */
206  private boolean checkAndDeleteEntries(FileStatus[] entries) {
207    if (entries == null) {
208      return true;
209    }
210    boolean allEntriesDeleted = true;
211    List<FileStatus> files = Lists.newArrayListWithCapacity(entries.length);
212    List<FileStatus> dirs = new ArrayList<>();
213    for (FileStatus child : entries) {
214      if (child.isDirectory()) {
215        dirs.add(child);
216      } else {
217        // collect all files to attempt to delete in one batch
218        files.add(child);
219      }
220    }
221    if (dirs.size() > 0) {
222      sortByConsumedSpace(dirs);
223      LOG.debug("Prepared to delete files in directories: " + dirs);
224      for (FileStatus child : dirs) {
225        Path path = child.getPath();
226        // for each subdirectory delete it and all entries if possible
227        if (!checkAndDeleteDirectory(path)) {
228          allEntriesDeleted = false;
229        }
230      }
231    }
232    if (!checkAndDeleteFiles(files)) {
233      allEntriesDeleted = false;
234    }
235    return allEntriesDeleted;
236  }
237  
238  /**
239   * Attempt to delete a directory and all files under that directory. Each child file is passed
240   * through the delegates to see if it can be deleted. If the directory has no children when the
241   * cleaners have finished it is deleted.
242   * <p>
243   * If new children files are added between checks of the directory, the directory will <b>not</b>
244   * be deleted.
245   * @param dir directory to check
246   * @return <tt>true</tt> if the directory was deleted, <tt>false</tt> otherwise.
247   */
248  @VisibleForTesting boolean checkAndDeleteDirectory(Path dir) {
249    if (LOG.isTraceEnabled()) {
250      LOG.trace("Checking directory: " + dir);
251    }
252
253    try {
254      FileStatus[] children = FSUtils.listStatus(fs, dir);
255      boolean allChildrenDeleted = checkAndDeleteEntries(children);
256  
257      // if the directory still has children, we can't delete it, so we are done
258      if (!allChildrenDeleted) return false;
259    } catch (IOException e) {
260      e = e instanceof RemoteException ?
261              ((RemoteException)e).unwrapRemoteException() : e;
262      LOG.warn("Error while listing directory: " + dir, e);
263      // couldn't list directory, so don't try to delete, and don't return success
264      return false;
265    }
266
267    // otherwise, all the children (that we know about) have been deleted, so we should try to
268    // delete this directory. However, don't do so recursively so we don't delete files that have
269    // been added since we last checked.
270    try {
271      return fs.delete(dir, false);
272    } catch (IOException e) {
273      if (LOG.isTraceEnabled()) {
274        LOG.trace("Couldn't delete directory: " + dir, e);
275      }
276      // couldn't delete w/o exception, so we can't return success.
277      return false;
278    }
279  }
280
281  /**
282   * Run the given files through each of the cleaners to see if it should be deleted, deleting it if
283   * necessary.
284   * @param files List of FileStatus for the files to check (and possibly delete)
285   * @return true iff successfully deleted all files
286   */
287  private boolean checkAndDeleteFiles(List<FileStatus> files) {
288    // first check to see if the path is valid
289    List<FileStatus> validFiles = Lists.newArrayListWithCapacity(files.size());
290    List<FileStatus> invalidFiles = Lists.newArrayList();
291    for (FileStatus file : files) {
292      if (validate(file.getPath())) {
293        validFiles.add(file);
294      } else {
295        LOG.warn("Found a wrongly formatted file: " + file.getPath() + " - will delete it.");
296        invalidFiles.add(file);
297      }
298    }
299
300    Iterable<FileStatus> deletableValidFiles = validFiles;
301    // check each of the cleaners for the valid files
302    for (T cleaner : cleanersChain) {
303      if (cleaner.isStopped() || this.getStopper().isStopped()) {
304        LOG.warn("A file cleaner" + this.getName() + " is stopped, won't delete any more files in:"
305            + this.oldFileDir);
306        return false;
307      }
308
309      Iterable<FileStatus> filteredFiles = cleaner.getDeletableFiles(deletableValidFiles);
310      
311      // trace which cleaner is holding on to each file
312      if (LOG.isTraceEnabled()) {
313        ImmutableSet<FileStatus> filteredFileSet = ImmutableSet.copyOf(filteredFiles);
314        for (FileStatus file : deletableValidFiles) {
315          if (!filteredFileSet.contains(file)) {
316            LOG.trace(file.getPath() + " is not deletable according to:" + cleaner);
317          }
318        }
319      }
320      
321      deletableValidFiles = filteredFiles;
322    }
323    
324    Iterable<FileStatus> filesToDelete = Iterables.concat(invalidFiles, deletableValidFiles);
325    return deleteFiles(filesToDelete) == files.size();
326  }
327
328  /**
329   * Delete the given files
330   * @param filesToDelete files to delete
331   * @return number of deleted files
332   */
333  protected int deleteFiles(Iterable<FileStatus> filesToDelete) {
334    int deletedFileCount = 0;
335    for (FileStatus file : filesToDelete) {
336      Path filePath = file.getPath();
337      if (LOG.isDebugEnabled()) {
338        LOG.debug("Removing: " + filePath + " from archive");
339      }
340      try {
341        boolean success = this.fs.delete(filePath, false);
342        if (success) {
343          deletedFileCount++;
344        } else {
345          LOG.warn("Attempted to delete:" + filePath
346              + ", but couldn't. Run cleaner chain and attempt to delete on next pass.");
347        }
348      } catch (IOException e) {
349        e = e instanceof RemoteException ?
350                  ((RemoteException)e).unwrapRemoteException() : e;
351        LOG.warn("Error while deleting: " + filePath, e);
352      }
353    }
354    return deletedFileCount;
355  }
356
357  @Override
358  public void cleanup() {
359    for (T lc : this.cleanersChain) {
360      try {
361        lc.stop("Exiting");
362      } catch (Throwable t) {
363        LOG.warn("Stopping", t);
364      }
365    }
366  }
367
368  /**
369   * @param enabled
370   */
371  public boolean setEnabled(final boolean enabled) {
372    return this.enabled.getAndSet(enabled);
373  }
374
375  public boolean getEnabled() {
376    return this.enabled.get();
377  }
