018package org.apache.hadoop.hbase.master.cleaner;
019
020import java.io.IOException;
021import java.util.Collections;
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Map;
026import java.util.Set;
027
028import org.apache.hadoop.hbase.HConstants;
029import org.apache.hadoop.hbase.MetaTableAccessor;
030import org.apache.hadoop.hbase.ScheduledChore;
031import org.apache.hadoop.hbase.Stoppable;
032import org.apache.hadoop.hbase.TableName;
033import org.apache.yetus.audience.InterfaceAudience;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036import org.apache.hadoop.hbase.client.Admin;
037import org.apache.hadoop.hbase.client.ColumnFamilyDescriptor;
038import org.apache.hadoop.hbase.client.Delete;
039import org.apache.hadoop.hbase.client.Table;
040import org.apache.hadoop.hbase.client.TableDescriptor;
041import org.apache.hadoop.hbase.master.MasterServices;
042import org.apache.hadoop.hbase.replication.ReplicationPeerDescription;
043import org.apache.hadoop.hbase.util.Bytes;
044
045/**
046 * This chore is to clean up the useless data in hbase:meta which is used by serial replication.
047 */
048@InterfaceAudience.Private
049public class ReplicationMetaCleaner extends ScheduledChore {
050
051  private static final Logger LOG = LoggerFactory.getLogger(ReplicationMetaCleaner.class);
052
053  private final Admin admin;
054  private final MasterServices master;
055
056  public ReplicationMetaCleaner(MasterServices master, Stoppable stoppable, int period)
057      throws IOException {
058    super("ReplicationMetaCleaner", stoppable, period);
059    this.master = master;
060    admin = master.getConnection().getAdmin();
061  }
062
063  @Override
064  protected void chore() {
065    try {
066      Map<String, TableDescriptor> tables = master.getTableDescriptors().getAllDescriptors();
067      Map<String, Set<String>> serialTables = new HashMap<>();
068      for (Map.Entry<String, TableDescriptor> entry : tables.entrySet()) {
069        boolean hasSerialScope = false;
070        for (ColumnFamilyDescriptor column : entry.getValue().getColumnFamilies()) {
071          if (column.getScope() == HConstants.REPLICATION_SCOPE_SERIAL) {
072            hasSerialScope = true;
073            break;
074          }
075        }
076        if (hasSerialScope) {
077          serialTables.put(entry.getValue().getTableName().getNameAsString(), new HashSet<>());
078        }
079      }
080      if (serialTables.isEmpty()){
081        return;
082      }
083
084      List<ReplicationPeerDescription> peers = admin.listReplicationPeers();
085      for (ReplicationPeerDescription peerDesc : peers) {
086        Map<TableName, List<String>> tableCFsMap = peerDesc.getPeerConfig().getTableCFsMap();
087        if (tableCFsMap ==null) {
088          continue;
089        }
090
091        for (Map.Entry<TableName, List<String>> map : tableCFsMap.entrySet()) {
092          if (serialTables.containsKey(map.getKey().getNameAsString())) {
093            serialTables.get(map.getKey().getNameAsString()).add(peerDesc.getPeerId());
094            break;
095          }
096        }
097      }
098
099      Map<String, List<Long>> barrierMap = MetaTableAccessor.getAllBarriers(master.getConnection());
100      for (Map.Entry<String, List<Long>> entry : barrierMap.entrySet()) {
101        String encodedName = entry.getKey();
102        byte[] encodedBytes = Bytes.toBytes(encodedName);
103        boolean canClearRegion = false;
104        Map<String, Long> posMap = MetaTableAccessor.getReplicationPositionForAllPeer(
105            master.getConnection(), encodedBytes);
106        if (posMap.isEmpty()) {
107          continue;
108        }
109
110        String tableName = MetaTableAccessor.getSerialReplicationTableName(
111            master.getConnection(), encodedBytes);
112        Set<String> confPeers = serialTables.get(tableName);
113        if (confPeers == null) {
114          // This table doesn't exist or all cf's scope is not serial any more, we can clear meta.
115          canClearRegion = true;
116        } else {
117          if (!allPeersHavePosition(confPeers, posMap)) {
118            continue;
119          }
120
121          String daughterValue = MetaTableAccessor
122              .getSerialReplicationDaughterRegion(master.getConnection(), encodedBytes);
123          if (daughterValue != null) {
124            //this region is merged or split
125            boolean allDaughterStart = true;
126            String[] daughterRegions = daughterValue.split(",");
127            for (String daughter : daughterRegions) {
128              byte[] region = Bytes.toBytes(daughter);
129              if (!MetaTableAccessor.getReplicationBarriers(
130                  master.getConnection(), region).isEmpty() &&
131                  !allPeersHavePosition(confPeers,
132                      MetaTableAccessor
133                          .getReplicationPositionForAllPeer(master.getConnection(), region))) {
134                allDaughterStart = false;
135                break;
136              }
137            }
138            if (allDaughterStart) {
139              canClearRegion = true;
140            }
141          }
142        }
143        if (canClearRegion) {
144          Delete delete = new Delete(encodedBytes);
145          delete.addFamily(HConstants.REPLICATION_POSITION_FAMILY);
146          delete.addFamily(HConstants.REPLICATION_BARRIER_FAMILY);
147          delete.addFamily(HConstants.REPLICATION_META_FAMILY);
148          try (Table metaTable = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
149            metaTable.delete(delete);
150          }
151        } else {
152
153          // Barriers whose seq is larger than min pos of all peers, and the last barrier whose seq
154          // is smaller than min pos should be kept. All other barriers can be deleted.
155
156          long minPos = Long.MAX_VALUE;
157          for (Map.Entry<String, Long> pos : posMap.entrySet()) {
158            minPos = Math.min(minPos, pos.getValue());
159          }
160          List<Long> barriers = entry.getValue();
161          int index = Collections.binarySearch(barriers, minPos);
162          if (index < 0) {
163            index = -index - 1;
164          }
165          Delete delete = new Delete(encodedBytes);
166          for (int i = 0; i < index - 1; i++) {
167            delete.addColumn(HConstants.REPLICATION_BARRIER_FAMILY, Bytes.toBytes(barriers.get(i)));
168          }
169          try (Table metaTable = master.getConnection().getTable(TableName.META_TABLE_NAME)) {
170            metaTable.delete(delete);
171          }
172        }
173
174      }
175
176    } catch (IOException e) {
177      LOG.error("Exception during cleaning up.", e);
178    }
179
180  }
181
182  private boolean allPeersHavePosition(Set<String> peers, Map<String, Long> posMap)
183      throws IOException {
184    for(String peer:peers){
185      if (!posMap.containsKey(peer)){
186        return false;
187      }
188    }
189    return true;
190  }
191}
