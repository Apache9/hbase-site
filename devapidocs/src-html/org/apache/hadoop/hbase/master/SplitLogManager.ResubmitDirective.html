068/**
069 * Distributes the task of log splitting to the available region servers.
070 * Coordination happens via coordination engine. For every log file that has to be split a
071 * task is created. SplitLogWorkers race to grab a task.
072 *
073 * <p>SplitLogManager monitors the tasks that it creates using the
074 * timeoutMonitor thread. If a task's progress is slow then
075 * {@link SplitLogManagerCoordination#checkTasks} will take away the
076 * task from the owner {@link org.apache.hadoop.hbase.regionserver.SplitLogWorker}
077 * and the task will be up for grabs again. When the task is done then it is
078 * deleted by SplitLogManager.
079 *
080 * <p>Clients call {@link #splitLogDistributed(Path)} to split a region server's
081 * log files. The caller thread waits in this method until all the log files
082 * have been split.
083 *
084 * <p>All the coordination calls made by this class are asynchronous. This is mainly
085 * to help reduce response time seen by the callers.
086 *
087 * <p>There is race in this design between the SplitLogManager and the
088 * SplitLogWorker. SplitLogManager might re-queue a task that has in reality
089 * already been completed by a SplitLogWorker. We rely on the idempotency of
090 * the log splitting task for correctness.
091 *
092 * <p>It is also assumed that every log splitting task is unique and once
093 * completed (either with success or with error) it will be not be submitted
094 * again. If a task is resubmitted then there is a risk that old "delete task"
095 * can delete the re-submission.
096 */
097@InterfaceAudience.Private

127  /**
128   * Its OK to construct this object even when region-servers are not online. It does lookup the
129   * orphan tasks in coordination engine but it doesn't block waiting for them to be done.
130   * @param master the master services
131   * @param conf the HBase configuration

134  public SplitLogManager(MasterServices master, Configuration conf)
135      throws IOException {
136    this.server = master;
137    this.conf = conf;
138    this.choreService = new ChoreService(master.getServerName() + "_splitLogManager_");
139    if (server.getCoordinatedStateManager() != null) {
140      SplitLogManagerCoordination coordination = getSplitLogManagerCoordination();
141      Set<String> failedDeletions = Collections.synchronizedSet(new HashSet<String>());
142      SplitLogManagerDetails details = new SplitLogManagerDetails(tasks, master, failedDeletions);
143      coordination.setDetails(details);
144      coordination.init();
145      // Determine recovery mode
146    }
147    this.unassignedTimeout =
148        conf.getInt("hbase.splitlog.manager.unassigned.timeout", DEFAULT_UNASSIGNED_TIMEOUT);
149    this.timeoutMonitor =
150        new TimeoutMonitor(conf.getInt("hbase.splitlog.manager.timeoutmonitor.period", 1000),
151            master);
152    choreService.scheduleChore(timeoutMonitor);
153  }

164  /**
165   * Get a list of paths that need to be split given a set of server-specific directories and
166   * optionally  a filter.
167   *
168   * See {@link AbstractFSWALProvider#getServerNameFromWALDirectoryName} for more info on directory
169   * layout.
170   *
171   * Should be package-private, but is needed by
172   * {@link org.apache.hadoop.hbase.wal.WALSplitter#split(Path, Path, Path, FileSystem,
173   *     Configuration, org.apache.hadoop.hbase.wal.WALFactory)} for tests.
174   */
175  @VisibleForTesting
176  public static FileStatus[] getFileList(final Configuration conf, final List<Path> logDirs,
177      final PathFilter filter)
178      throws IOException {
179    List<FileStatus> fileStatus = new ArrayList<>();
180    for (Path logDir : logDirs) {
181      final FileSystem fs = logDir.getFileSystem(conf);
182      if (!fs.exists(logDir)) {
183        LOG.warn(logDir + " doesn't exist. Nothing to do!");
184        continue;
185      }
186      FileStatus[] logfiles = FSUtils.listStatus(fs, logDir, filter);
187      if (logfiles == null || logfiles.length == 0) {
188        LOG.info(logDir + " is empty dir, no logs to split");
189      } else {
190        Collections.addAll(fileStatus, logfiles);
191      }
192    }
193    FileStatus[] a = new FileStatus[fileStatus.size()];
194    return fileStatus.toArray(a);
195  }

209  /**
210   * The caller will block until all the log files of the given region server have been processed -
211   * successfully split or an error is encountered - by an available worker region server. This
212   * method must only be called after the region servers have been brought online.
213   * @param logDirs List of log dirs to split
214   * @throws IOException If there was an error while splitting any log file
215   * @return cumulative size of the logfiles split
216   */
217  public long splitLogDistributed(final List<Path> logDirs) throws IOException {
218    if (logDirs.isEmpty()) {
219      return 0;
220    }
221    Set<ServerName> serverNames = new HashSet<>();
222    for (Path logDir : logDirs) {
223      try {
224        ServerName serverName = AbstractFSWALProvider.getServerNameFromWALDirectoryName(logDir);
225        if (serverName != null) {
226          serverNames.add(serverName);
227        }
228      } catch (IllegalArgumentException e) {
229        // ignore invalid format error.
230        LOG.warn("Cannot parse server name from " + logDir);
231      }
232    }
233    return splitLogDistributed(serverNames, logDirs, null);
234  }

236  /**
237   * The caller will block until all the hbase:meta log files of the given region server have been
238   * processed - successfully split or an error is encountered - by an available worker region
239   * server. This method must only be called after the region servers have been brought online.
240   * @param logDirs List of log dirs to split
241   * @param filter the Path filter to select specific files for considering
242   * @throws IOException If there was an error while splitting any log file
243   * @return cumulative size of the logfiles split
244   */
245  public long splitLogDistributed(final Set<ServerName> serverNames, final List<Path> logDirs,
246      PathFilter filter) throws IOException {
247    MonitoredTask status = TaskMonitor.get().createStatus("Doing distributed log split in " +
248      logDirs + " for serverName=" + serverNames);
249    FileStatus[] logfiles = getFileList(logDirs, filter);
250    status.setStatus("Checking directory contents...");
251    SplitLogCounters.tot_mgr_log_split_batch_start.incrementAndGet();
252    LOG.info("Started splitting " + logfiles.length + " logs in " + logDirs +
253      " for " + serverNames);
254    long t = EnvironmentEdgeManager.currentTime();
255    long totalSize = 0;
256    TaskBatch batch = new TaskBatch();
257    Boolean isMetaRecovery = (filter == null) ? null : false;
258    for (FileStatus lf : logfiles) {
259      // TODO If the log file is still being written to - which is most likely
260      // the case for the last log file - then its length will show up here
261      // as zero. The size of such a file can only be retrieved after
262      // recover-lease is done. totalSize will be under in most cases and the
263      // metrics that it drives will also be under-reported.
264      totalSize += lf.getLen();
265      String pathToLog = FSUtils.removeWALRootPath(lf.getPath(), conf);
266      if (!enqueueSplitTask(pathToLog, batch)) {
267        throw new IOException("duplicate log split scheduled for " + lf.getPath());
268      }
269    }
270    waitForSplittingCompletion(batch, status);
271    // remove recovering regions
272    if (filter == MasterWalManager.META_FILTER /* reference comparison */) {
273      // we split meta regions and user regions separately therefore logfiles are either all for
274      // meta or user regions but won't for both( we could have mixed situations in tests)
275      isMetaRecovery = true;
276    }
277    removeRecoveringRegions(serverNames, isMetaRecovery);
278
279    if (batch.done != batch.installed) {
280      batch.isDead = true;
281      SplitLogCounters.tot_mgr_log_split_batch_err.incrementAndGet();
282      LOG.warn("error while splitting logs in " + logDirs + " installed = " + batch.installed
283          + " but only " + batch.done + " done");
284      String msg = "error or interrupted while splitting logs in " + logDirs + " Task = " + batch;
285      status.abort(msg);
286      throw new IOException(msg);
287    }
288    for (Path logDir : logDirs) {
289      status.setStatus("Cleaning up log directory...");
290      final FileSystem fs = logDir.getFileSystem(conf);
291      try {
292        if (fs.exists(logDir) && !fs.delete(logDir, false)) {
293          LOG.warn("Unable to delete log src dir. Ignoring. " + logDir);
294        }
295      } catch (IOException ioe) {
296        FileStatus[] files = fs.listStatus(logDir);
297        if (files != null && files.length > 0) {
298          LOG.warn("Returning success without actually splitting and "
299              + "deleting all the log files in path " + logDir + ": "
300              + Arrays.toString(files), ioe);
301        } else {
302          LOG.warn("Unable to delete log src dir. Ignoring. " + logDir, ioe);
303        }
304      }
305      SplitLogCounters.tot_mgr_log_split_batch_success.incrementAndGet();
306    }
307    String msg =
308        "finished splitting (more than or equal to) " + totalSize + " bytes in " + batch.installed
309            + " log files in " + logDirs + " in "
310            + (EnvironmentEdgeManager.currentTime() - t) + "ms";
311    status.markComplete(msg);
312    LOG.info(msg);
313    return totalSize;
314  }

316  /**
317   * Add a task entry to coordination if it is not already there.
318   * @param taskname the path of the log to be split
319   * @param batch the batch this task belongs to
320   * @return true if a new entry is created, false if it is already there.
321   */
322  boolean enqueueSplitTask(String taskname, TaskBatch batch) {
323    lastTaskCreateTime = EnvironmentEdgeManager.currentTime();
324    String task = getSplitLogManagerCoordination().prepareTask(taskname);
325    Task oldtask = createTaskIfAbsent(task, batch);
326    if (oldtask == null) {
327      // publish the task in the coordination engine
328      getSplitLogManagerCoordination().submitTask(task);
329      return true;
330    }
331    return false;
332  }

334  private void waitForSplittingCompletion(TaskBatch batch, MonitoredTask status) {
335    synchronized (batch) {
336      while ((batch.done + batch.error) != batch.installed) {
337        try {
338          status.setStatus("Waiting for distributed tasks to finish. " + " scheduled="
339              + batch.installed + " done=" + batch.done + " error=" + batch.error);
340          int remaining = batch.installed - (batch.done + batch.error);
341          int actual = activeTasks(batch);
342          if (remaining != actual) {
343            LOG.warn("Expected " + remaining + " active tasks, but actually there are " + actual);
344          }
345          int remainingTasks = getSplitLogManagerCoordination().remainingTasksInCoordination();
346          if (remainingTasks >= 0 && actual > remainingTasks) {
347            LOG.warn("Expected at least" + actual + " tasks remaining, but actually there are "
348                + remainingTasks);
349          }
350          if (remainingTasks == 0 || actual == 0) {
351            LOG.warn("No more task remaining, splitting "
352                + "should have completed. Remaining tasks is " + remainingTasks
353                + ", active tasks in map " + actual);
354            if (remainingTasks == 0 && actual == 0) {
355              return;
356            }
357          }
358          batch.wait(100);
359          if (server.isStopped()) {
360            LOG.warn("Stopped while waiting for log splits to be completed");
361            return;
362          }
363        } catch (InterruptedException e) {
364          LOG.warn("Interrupted while waiting for log splits to be completed");
365          Thread.currentThread().interrupt();
366          return;
367        }
368      }
369    }
370  }

377  private int activeTasks(final TaskBatch batch) {
378    int count = 0;
379    for (Task t : tasks.values()) {
380      if (t.batch == batch && t.status == TerminationStatus.IN_PROGRESS) {
381        count++;
382      }
383    }
384    return count;

388  /**
389   * It removes recovering regions under /hbase/recovering-regions/[encoded region name] so that the
390   * region server hosting the region can allow reads to the recovered region
391   * @param serverNames servers which are just recovered
392   * @param isMetaRecovery whether current recovery is for the meta region on {@code serverNames}
393   */
394  private void removeRecoveringRegions(final Set<ServerName> serverNames, Boolean isMetaRecovery) {
395    if (!isLogReplaying()) {
396      // the function is only used in WALEdit direct replay mode
397      return;
398    }
399    if (serverNames == null || serverNames.isEmpty()) return;
400
401    Set<String> recoveredServerNameSet = new HashSet<>();
402    for (ServerName tmpServerName : serverNames) {
403      recoveredServerNameSet.add(tmpServerName.getServerName());
404    }
405
406    this.recoveringRegionLock.lock();
407    try {
408      getSplitLogManagerCoordination().removeRecoveringRegions(
409        recoveredServerNameSet, isMetaRecovery);
410    } catch (IOException e) {
411      LOG.warn("removeRecoveringRegions got exception. Will retry", e);
412      if (serverNames != null && !serverNames.isEmpty()) {
413        this.failedRecoveringRegionDeletions.add(new Pair<>(serverNames, isMetaRecovery));
414      }
415    } finally {
416      this.recoveringRegionLock.unlock();
417    }
418  }

421   * It removes stale recovering regions under /hbase/recovering-regions/[encoded region name]
422   * during master initialization phase.
423   * @param failedServers A set of known failed servers
424   * @throws IOException
425   */
426  void removeStaleRecoveringRegions(final Set<ServerName> failedServers) throws IOException,
427      InterruptedIOException {
428    Set<String> knownFailedServers = new HashSet<>();
429    if (failedServers != null) {
430      for (ServerName tmpServerName : failedServers) {
431        knownFailedServers.add(tmpServerName.getServerName());
432      }
433    }
434
435    this.recoveringRegionLock.lock();
436    try {
437      getSplitLogManagerCoordination().removeStaleRecoveringRegions(knownFailedServers);
438    } finally {
439      this.recoveringRegionLock.unlock();
440    }
441  }

443  /**
444   * @param path
445   * @param batch
446   * @return null on success, existing task on error
447   */
448  private Task createTaskIfAbsent(String path, TaskBatch batch) {
449    Task oldtask;
450    // batch.installed is only changed via this function and
451    // a single thread touches batch.installed.
452    Task newtask = new Task();
453    newtask.batch = batch;
454    oldtask = tasks.putIfAbsent(path, newtask);
455    if (oldtask == null) {
456      batch.installed++;
457      return null;
458    }
459    // new task was not used.
460    synchronized (oldtask) {
461      if (oldtask.isOrphan()) {
462        if (oldtask.status == SUCCESS) {
463          // The task is already done. Do not install the batch for this
464          // task because it might be too late for setDone() to update
465          // batch.done. There is no need for the batch creator to wait for
466          // this task to complete.
467          return (null);
468        }
469        if (oldtask.status == IN_PROGRESS) {
470          oldtask.batch = batch;
471          batch.installed++;
472          LOG.debug("Previously orphan task " + path + " is now being waited upon");
473          return null;
474        }
475        while (oldtask.status == FAILURE) {
476          LOG.debug("wait for status of task " + path + " to change to DELETED");
477          SplitLogCounters.tot_mgr_wait_for_zk_delete.incrementAndGet();
478          try {
479            oldtask.wait();
480          } catch (InterruptedException e) {
481            Thread.currentThread().interrupt();
482            LOG.warn("Interrupted when waiting for znode delete callback");
483            // fall through to return failure
484            break;
485          }
486        }
487        if (oldtask.status != DELETED) {
488          LOG.warn("Failure because previously failed task"
489              + " state still present. Waiting for znode delete callback" + " path=" + path);
490          return oldtask;
491        }
492        // reinsert the newTask and it must succeed this time
493        Task t = tasks.putIfAbsent(path, newtask);
494        if (t == null) {
495          batch.installed++;
496          return null;
497        }
498        LOG.fatal("Logic error. Deleted task still present in tasks map");
499        assert false : "Deleted task still present in tasks map";
500        return t;
501      }
502      LOG.warn("Failure because two threads can't wait for the same task; path=" + path);
503      return oldtask;
504    }
505  }

516  void handleDeadWorker(ServerName workerName) {
517    // resubmit the tasks on the TimeoutMonitor thread. Makes it easier
518    // to reason about concurrency. Makes it easier to retry.
519    synchronized (deadWorkersLock) {
520      if (deadWorkers == null) {
521        deadWorkers = new HashSet<>(100);
522      }
523      deadWorkers.add(workerName);
524    }
525    LOG.info("dead splitlog worker " + workerName);
526  }
527
528  void handleDeadWorkers(Set<ServerName> serverNames) {
529    synchronized (deadWorkersLock) {
530      if (deadWorkers == null) {
531        deadWorkers = new HashSet<>(100);
532      }
533      deadWorkers.addAll(serverNames);
534    }
535    LOG.info("dead splitlog workers " + serverNames);
536  }

605  /**
606   * in memory state of an active task.
607   */
608  @InterfaceAudience.Private
609  public static class Task {
610    public volatile long last_update;
611    public volatile int last_version;
612    public volatile ServerName cur_worker_name;
613    public volatile TaskBatch batch;
614    public volatile TerminationStatus status;
615    public volatile AtomicInteger incarnation = new AtomicInteger(0);
616    public final AtomicInteger unforcedResubmits = new AtomicInteger();
617    public volatile boolean resubmitThresholdReached;
618
619    @Override
620    public String toString() {
621      return ("last_update = " + last_update + " last_version = " + last_version
622          + " cur_worker_name = " + cur_worker_name + " status = " + status + " incarnation = "
623          + incarnation + " resubmits = " + unforcedResubmits.get() + " batch = " + batch);
624    }
625
626    public Task() {
627      last_version = -1;
628      status = IN_PROGRESS;
629      setUnassigned();
630    }
631
632    public boolean isOrphan() {
633      return (batch == null || batch.isDead);
634    }
635
636    public boolean isUnassigned() {
637      return (cur_worker_name == null);

656  /**
657   * Periodically checks all active tasks and resubmits the ones that have timed out
658   */
659  private class TimeoutMonitor extends ScheduledChore {
660    private long lastLog = 0;
661
662    public TimeoutMonitor(final int period, Stoppable stopper) {
663      super("SplitLogManager Timeout Monitor", stopper, period);
664    }
665
666    @Override
667    protected void chore() {
668      if (server.getCoordinatedStateManager() == null) return;
669
670      int resubmitted = 0;
671      int unassigned = 0;
672      int tot = 0;
673      boolean found_assigned_task = false;
674      Set<ServerName> localDeadWorkers;
675
676      synchronized (deadWorkersLock) {
677        localDeadWorkers = deadWorkers;
678        deadWorkers = null;
679      }
680
681      for (Map.Entry<String, Task> e : tasks.entrySet()) {
682        String path = e.getKey();
683        Task task = e.getValue();
684        ServerName cur_worker = task.cur_worker_name;
685        tot++;
686        // don't easily resubmit a task which hasn't been picked up yet. It
687        // might be a long while before a SplitLogWorker is free to pick up a
688        // task. This is because a SplitLogWorker picks up a task one at a
689        // time. If we want progress when there are no region servers then we
690        // will have to run a SplitLogWorker thread in the Master.
691        if (task.isUnassigned()) {
692          unassigned++;
693          continue;
694        }
695        found_assigned_task = true;
696        if (localDeadWorkers != null && localDeadWorkers.contains(cur_worker)) {
697          SplitLogCounters.tot_mgr_resubmit_dead_server_task.incrementAndGet();
698          if (getSplitLogManagerCoordination().resubmitTask(path, task, FORCE)) {
699            resubmitted++;
700          } else {
701            handleDeadWorker(cur_worker);
702            LOG.warn("Failed to resubmit task " + path + " owned by dead " + cur_worker
703                + ", will retry.");
704          }
705        } else if (getSplitLogManagerCoordination().resubmitTask(path, task, CHECK)) {
706          resubmitted++;
707        }
708      }
709      if (tot > 0) {
710        long now = EnvironmentEdgeManager.currentTime();
711        if (now > lastLog + 5000) {
712          lastLog = now;
713          LOG.info("total=" + tot + ", unassigned=" + unassigned + ", tasks=" + tasks);
714        }
715      }
716      if (resubmitted > 0) {
717        LOG.info("resubmitted " + resubmitted + " out of " + tot + " tasks");
718      }
719      // If there are pending tasks and all of them have been unassigned for
720      // some time then put up a RESCAN node to ping the workers.
721      // ZKSplitlog.DEFAULT_UNASSIGNED_TIMEOUT is of the order of minutes
722      // because a. it is very unlikely that every worker had a
723      // transient error when trying to grab the task b. if there are no
724      // workers then all tasks wills stay unassigned indefinitely and the
725      // manager will be indefinitely creating RESCAN nodes. TODO may be the
726      // master should spawn both a manager and a worker thread to guarantee
727      // that there is always one worker in the system
728      if (tot > 0
729          && !found_assigned_task
730          && ((EnvironmentEdgeManager.currentTime() - lastTaskCreateTime) > unassignedTimeout)) {
731        for (Map.Entry<String, Task> e : tasks.entrySet()) {
732          String key = e.getKey();
733          Task task = e.getValue();
734          // we have to do task.isUnassigned() check again because tasks might
735          // have been asynchronously assigned. There is no locking required
736          // for these checks ... it is OK even if tryGetDataSetWatch() is
737          // called unnecessarily for a taskpath
738          if (task.isUnassigned() && (task.status != FAILURE)) {
739            // We just touch the znode to make sure its still there
740            getSplitLogManagerCoordination().checkTaskStillAvailable(key);
741          }
742        }
743        getSplitLogManagerCoordination().checkTasks();
744        SplitLogCounters.tot_mgr_resubmit_unassigned.incrementAndGet();
745        LOG.debug("resubmitting unassigned task(s) after timeout");
746      }
747      Set<String> failedDeletions =
748        getSplitLogManagerCoordination().getDetails().getFailedDeletions();
749      // Retry previously failed deletes
750      if (failedDeletions.size() > 0) {
751        List<String> tmpPaths = new ArrayList<>(failedDeletions);
752        for (String tmpPath : tmpPaths) {
753          // deleteNode is an async call
754          getSplitLogManagerCoordination().deleteTask(tmpPath);
755        }
756        failedDeletions.removeAll(tmpPaths);
757      }
758
759      // Garbage collect left-over
760      long timeInterval =
761          EnvironmentEdgeManager.currentTime()
762              - getSplitLogManagerCoordination().getLastRecoveryTime();
763      if (!failedRecoveringRegionDeletions.isEmpty()
764          || (tot == 0 && tasks.isEmpty() && (timeInterval > checkRecoveringTimeThreshold))) {
765        // inside the function there have more checks before GC anything
766        if (!failedRecoveringRegionDeletions.isEmpty()) {
767          List<Pair<Set<ServerName>, Boolean>> previouslyFailedDeletions =
768              new ArrayList<>(failedRecoveringRegionDeletions);
769          failedRecoveringRegionDeletions.removeAll(previouslyFailedDeletions);
770          for (Pair<Set<ServerName>, Boolean> failedDeletion : previouslyFailedDeletions) {
771            removeRecoveringRegions(failedDeletion.getFirst(), failedDeletion.getSecond());
772          }
773        } else {
774          removeRecoveringRegions(null, null);
775        }
776      }
777    }
778  }
