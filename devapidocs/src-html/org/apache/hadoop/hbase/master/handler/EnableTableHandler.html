053/**
054 * Handler to run enable of a table.
055 */
056@InterfaceAudience.Private
057public class EnableTableHandler extends EventHandler {
058  private static final Log LOG = LogFactory.getLog(EnableTableHandler.class);
059  private final TableName tableName;
060  private final AssignmentManager assignmentManager;
061  private final TableLockManager tableLockManager;
062  private boolean skipTableStateCheck = false;
063  private TableLock tableLock;
064  private MasterServices services;
065
066  public EnableTableHandler(Server server, TableName tableName,
067      AssignmentManager assignmentManager, TableLockManager tableLockManager,
068      boolean skipTableStateCheck) {
069    super(server, EventType.C_M_ENABLE_TABLE);
070    this.tableName = tableName;
071    this.assignmentManager = assignmentManager;
072    this.tableLockManager = tableLockManager;
073    this.skipTableStateCheck = skipTableStateCheck;
074  }
075
076  public EnableTableHandler(MasterServices services, TableName tableName,
077      AssignmentManager assignmentManager,
078      TableLockManager tableLockManager, boolean skipTableStateCheck) {
079    this((Server)services, tableName, assignmentManager, tableLockManager,
080        skipTableStateCheck);
081    this.services = services;
082  }
083
084  public EnableTableHandler prepare()
085      throws TableNotFoundException, TableNotDisabledException, IOException {
086    //acquire the table write lock, blocking
087    this.tableLock = this.tableLockManager.writeLock(tableName,
088        EventType.C_M_ENABLE_TABLE.toString());
089    this.tableLock.acquire();
090
091    boolean success = false;
092    try {
093      // Check if table exists
094      if (!MetaTableAccessor.tableExists(this.server.getConnection(), tableName)) {
095        throw new TableNotFoundException(tableName);
096      }
097
098      // There could be multiple client requests trying to disable or enable
099      // the table at the same time. Ensure only the first request is honored
100      // After that, no other requests can be accepted until the table reaches
101      // DISABLED or ENABLED.
102      if (!skipTableStateCheck) {
103        TableState.State state = this.assignmentManager
104                .getTableStateManager().setTableStateIfInStates(
105                this.tableName, TableState.State.ENABLING,
106                TableState.State.DISABLED);
107        if (state!=null) {
108          LOG.info("Table " + tableName + " isn't disabled;is "+state.name()+"; skipping enable");
109          throw new TableNotDisabledException(this.tableName+" state is "+state.name());
110        }
111      }
112      success = true;
113    } finally {
114      if (!success) {
115        releaseTableLock();
116      }
117    }
118    return this;
119  }
120
121  @Override
122  public String toString() {
123    String name = "UnknownServerName";
124    if(server != null && server.getServerName() != null) {
125      name = server.getServerName().toString();
126    }
127    return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
128        tableName;
129  }
130
131  @Override
132  public void process() {
133    try {
134      LOG.info("Attempting to enable the table " + this.tableName);
135      MasterCoprocessorHost cpHost = ((HMaster) this.server)
136          .getMasterCoprocessorHost();
137      if (cpHost != null) {
138        cpHost.preEnableTableAction(this.tableName);
139      }
140      handleEnableTable();
141      if (cpHost != null) {
142        cpHost.postCompletedEnableTableAction(this.tableName);
143      }
144    } catch (IOException | InterruptedException e) {
145      LOG.error("Error trying to enable the table " + this.tableName, e);
146    } finally {
147      releaseTableLock();
148    }
149  }
150
151  private void releaseTableLock() {
152    if (this.tableLock != null) {
153      try {
154        this.tableLock.release();
155      } catch (IOException ex) {
156        LOG.warn("Could not release the table lock", ex);
157      }
158    }
159  }
160
161  private void handleEnableTable() throws IOException,
162      InterruptedException {
163    // I could check table is disabling and if so, not enable but require
164    // that user first finish disabling but that might be obnoxious.
165
166    this.assignmentManager.getTableStateManager().setTableState(this.tableName,
167      TableState.State.ENABLING);
168    boolean done = false;
169    ServerManager serverManager = ((HMaster)this.server).getServerManager();
170    // Get the regions of this table. We're done when all listed
171    // tables are onlined.
172    List<Pair<HRegionInfo, ServerName>> tableRegionsAndLocations;
173    if (TableName.META_TABLE_NAME.equals(tableName)) {
174      tableRegionsAndLocations = new MetaTableLocator().getMetaRegionsAndLocations(
175        server.getZooKeeper());
176    } else {
177      tableRegionsAndLocations = MetaTableAccessor.getTableRegionsAndLocations(
178        server.getConnection(), tableName, true);
179    }
180
181    int countOfRegionsInTable = tableRegionsAndLocations.size();
182    Map<HRegionInfo, ServerName> regionsToAssign =
183        regionsToAssignWithServerName(tableRegionsAndLocations);
184    if (services != null) {
185      // need to potentially create some regions for the replicas
186      List<HRegionInfo> unrecordedReplicas = AssignmentManager.replicaRegionsNotRecordedInMeta(
187          new HashSet<HRegionInfo>(regionsToAssign.keySet()), services);
188      Map<ServerName, List<HRegionInfo>> srvToUnassignedRegs =
189            this.assignmentManager.getBalancer().roundRobinAssignment(unrecordedReplicas,
190                serverManager.getOnlineServersList());
191      if (srvToUnassignedRegs != null) {
192        for (Map.Entry<ServerName, List<HRegionInfo>> entry : srvToUnassignedRegs.entrySet()) {
193          for (HRegionInfo h : entry.getValue()) {
194            regionsToAssign.put(h, entry.getKey());
195          }
196        }
197      }
198    }
199    int regionsCount = regionsToAssign.size();
200    if (regionsCount == 0) {
201      done = true;
202    }
203    LOG.info("Table '" + this.tableName + "' has " + countOfRegionsInTable
204      + " regions, of which " + regionsCount + " are offline.");
205    List<ServerName> onlineServers = serverManager.createDestinationServersList();
206    Map<ServerName, List<HRegionInfo>> bulkPlan =
207        this.assignmentManager.getBalancer().retainAssignment(regionsToAssign, onlineServers);
208    if (bulkPlan != null) {
209      LOG.info("Bulk assigning " + regionsCount + " region(s) across " + bulkPlan.size()
210          + " server(s), retainAssignment=true");
211
212      BulkAssigner ba =
213          new GeneralBulkAssigner(this.server, bulkPlan, this.assignmentManager, true);
214      try {
215        if (ba.bulkAssign()) {
216          done = true;
217        }
218      } catch (InterruptedException e) {
219        LOG.warn("Enable operation was interrupted when enabling table '"
220            + this.tableName + "'");
221        // Preserve the interrupt.
222        Thread.currentThread().interrupt();
223      }
224    } else {
225      LOG.info("Balancer was unable to find suitable servers for table " + tableName
226          + ", leaving unassigned");
227      done = true;
228    }
229    if (done) {
230      // Flip the table to enabled.
231      this.assignmentManager.getTableStateManager().setTableState(
232        this.tableName, TableState.State.ENABLED);
233      LOG.info("Table '" + this.tableName
234      + "' was successfully enabled. Status: done=" + done);
235    } else {
236      LOG.warn("Table '" + this.tableName
237      + "' wasn't successfully enabled. Status: done=" + done);
238    }
239  }
240
241  /**
242   * @param regionsInMeta
243   * @return List of regions neither in transition nor assigned.
244   * @throws IOException
245   */
246  private Map<HRegionInfo, ServerName> regionsToAssignWithServerName(
247      final List<Pair<HRegionInfo, ServerName>> regionsInMeta) throws IOException {
248    Map<HRegionInfo, ServerName> regionsToAssign =
249        new HashMap<HRegionInfo, ServerName>(regionsInMeta.size());
250    RegionStates regionStates = this.assignmentManager.getRegionStates();
251    for (Pair<HRegionInfo, ServerName> regionLocation : regionsInMeta) {
252      HRegionInfo hri = regionLocation.getFirst();
253      ServerName sn = regionLocation.getSecond();
254      if (regionStates.isRegionOffline(hri)) {
255        regionsToAssign.put(hri, sn);
256      } else {
257        if (LOG.isDebugEnabled()) {
258          LOG.debug("Skipping assign for the region " + hri + " during enable table "
259              + hri.getTable() + " because its already in tranition or assigned.");
260        }
261      }
262    }
263    return regionsToAssign;
264  }
