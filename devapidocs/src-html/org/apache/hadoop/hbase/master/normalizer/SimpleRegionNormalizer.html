042/**
043 * Simple implementation of region normalizer.
044 *
045 * Logic in use:
046 *
047 *  <ol>
048 *  <li> get all regions of a given table
049 *  <li> get avg size S of each region (by total size of store files reported in RegionLoad)
050 *  <li> If biggest region is bigger than S * 2, it is kindly requested to split,
051 *    and normalization stops
052 *  <li> Otherwise, two smallest region R1 and its smallest neighbor R2 are kindly requested
053 *    to merge, if R1 + R1 &lt;  S, and normalization stops
054 *  <li> Otherwise, no action is performed
055 * </ol>
056 * <p>
057 * Region sizes are coarse and approximate on the order of megabytes. Additionally,
058 * "empty" regions (less than 1MB, with the previous note) are not merged away. This
059 * is by design to prevent normalization from undoing the pre-splitting of a table.
060 */
061@InterfaceAudience.Private
062public class SimpleRegionNormalizer implements RegionNormalizer {
063
064  private static final Log LOG = LogFactory.getLog(SimpleRegionNormalizer.class);
065  private static final int MIN_REGION_COUNT = 3;
066  private MasterServices masterServices;
067  private MasterRpcServices masterRpcServices;
068  private static long[] skippedCount = new long[NormalizationPlan.PlanType.values().length];
069
070  /**
071   * Set the master service.
072   * @param masterServices inject instance of MasterServices
073   */
074  @Override
075  public void setMasterServices(MasterServices masterServices) {
076    this.masterServices = masterServices;
077  }
078
079  @Override
080  public void setMasterRpcServices(MasterRpcServices masterRpcServices) {
081    this.masterRpcServices = masterRpcServices;
082  }
083
084  @Override
085  public void planSkipped(HRegionInfo hri, PlanType type) {
086    skippedCount[type.ordinal()]++;
087  }
088
089  @Override
090  public long getSkippedCount(NormalizationPlan.PlanType type) {
091    return skippedCount[type.ordinal()];
092  }
093
094  // Comparator that gives higher priority to region Split plan
095  private Comparator<NormalizationPlan> planComparator =
096      new Comparator<NormalizationPlan>() {
097    @Override
098    public int compare(NormalizationPlan plan, NormalizationPlan plan2) {
099      if (plan instanceof SplitNormalizationPlan) {
100        return -1;
101      }
102      if (plan2 instanceof SplitNormalizationPlan) {
103        return 1;
104      }
105      return 0;
106    }
107  };
108
109  /**
110   * Computes next most "urgent" normalization action on the table.
111   * Action may be either a split, or a merge, or no action.
112   *
113   * @param table table to normalize
114   * @return normalization plan to execute
115   */
116  @Override
117  public List<NormalizationPlan> computePlanForTable(TableName table) throws HBaseIOException {
118    if (table == null || table.isSystemTable()) {
119      LOG.debug("Normalization of system table " + table + " isn't allowed");
120      return null;
121    }
122
123    List<NormalizationPlan> plans = new ArrayList<NormalizationPlan>();
124    List<HRegionInfo> tableRegions = masterServices.getAssignmentManager().getRegionStates().
125      getRegionsOfTable(table);
126
127    //TODO: should we make min number of regions a config param?
128    if (tableRegions == null || tableRegions.size() < MIN_REGION_COUNT) {
129      int nrRegions = tableRegions == null ? 0 : tableRegions.size();
130      LOG.debug("Table " + table + " has " + nrRegions + " regions, required min number"
131        + " of regions for normalizer to run is " + MIN_REGION_COUNT + ", not running normalizer");
132      return null;
133    }
134
135    LOG.debug("Computing normalization plan for table: " + table +
136      ", number of regions: " + tableRegions.size());
137
138    long totalSizeMb = 0;
139
140    for (int i = 0; i < tableRegions.size(); i++) {
141      HRegionInfo hri = tableRegions.get(i);
142      long regionSize = getRegionSize(hri);
143      totalSizeMb += regionSize;
144    }
145
146    double avgRegionSize = totalSizeMb / (double) tableRegions.size();
147
148    LOG.debug("Table " + table + ", total aggregated regions size: " + totalSizeMb);
149    LOG.debug("Table " + table + ", average region size: " + avgRegionSize);
150
151    int candidateIdx = 0;
152    boolean splitEnabled = true, mergeEnabled = true;
153    try {
154      splitEnabled = masterRpcServices.isSplitOrMergeEnabled(null,
155        RequestConverter.buildIsSplitOrMergeEnabledRequest(MasterSwitchType.SPLIT)).getEnabled();
156    } catch (ServiceException se) {
157      LOG.debug("Unable to determine whether split is enabled", se);
158    }
159    try {
160      mergeEnabled = masterRpcServices.isSplitOrMergeEnabled(null,
161        RequestConverter.buildIsSplitOrMergeEnabledRequest(MasterSwitchType.MERGE)).getEnabled();
162    } catch (ServiceException se) {
163      LOG.debug("Unable to determine whether split is enabled", se);
164    }
165    while (candidateIdx < tableRegions.size()) {
166      HRegionInfo hri = tableRegions.get(candidateIdx);
167      long regionSize = getRegionSize(hri);
168      // if the region is > 2 times larger than average, we split it, split
169      // is more high priority normalization action than merge.
170      if (regionSize > 2 * avgRegionSize) {
171        if (splitEnabled) {
172          LOG.info("Table " + table + ", large region " + hri.getRegionNameAsString() + " has size "
173              + regionSize + ", more than twice avg size, splitting");
174          plans.add(new SplitNormalizationPlan(hri, null));
175        }
176      } else {
177        if (candidateIdx == tableRegions.size()-1) {
178          break;
179        }
180        if (mergeEnabled) {
181          HRegionInfo hri2 = tableRegions.get(candidateIdx+1);
182          long regionSize2 = getRegionSize(hri2);
183          if (regionSize + regionSize2 < avgRegionSize) {
184            LOG.info("Table " + table + ", small region size: " + regionSize
185              + " plus its neighbor size: " + regionSize2
186              + ", less than the avg size " + avgRegionSize + ", merging them");
187            plans.add(new MergeNormalizationPlan(hri, hri2));
188            candidateIdx++;
189          }
190        }
191      }
192      candidateIdx++;
193    }
194    if (plans.isEmpty()) {
195      LOG.debug("No normalization needed, regions look good for table: " + table);
196      return null;
197    }
198    Collections.sort(plans, planComparator);
199    return plans;
200  }
201
202  private long getRegionSize(HRegionInfo hri) {
203    ServerName sn = masterServices.getAssignmentManager().getRegionStates().
204      getRegionServerOfRegion(hri);
205    RegionLoad regionLoad = masterServices.getServerManager().getLoad(sn).
206      getRegionsLoad().get(hri.getRegionName());
207    return regionLoad.getStorefileSizeMB();
208  }
