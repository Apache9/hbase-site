178/**
179 * HMaster is the "master server" for HBase. An HBase cluster has one active
180 * master.  If many masters are started, all compete.  Whichever wins goes on to
181 * run the cluster.  All others park themselves in their constructor until
182 * master or cluster shutdown or until the active master loses its lease in
183 * zookeeper.  Thereafter, all running master jostle to take over master role.
184 *
185 * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
186 * this case it will tell all regionservers to go down and then wait on them
187 * all reporting in that they are down.  This master will then shut itself down.
188 *
189 * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
190 *
191 * @see org.apache.zookeeper.Watcher
192 */
193@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)

199  /**
200   * Protection against zombie master. Started once Master accepts active responsibility and
201   * starts taking over responsibilities. Allows a finite time window before giving up ownership.
202   */
203  private static class InitializationMonitor extends HasThread {
204    /** The amount of time in milliseconds to sleep before checking initialization status. */
205    public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
206    public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
207
208    /**
209     * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
210     * true, do nothing otherwise.
211     */
212    public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
213    public static final boolean HALT_DEFAULT = false;
214
215    private final HMaster master;
216    private final long timeout;
217    private final boolean haltOnTimeout;
218
219    /** Creates a Thread that monitors the {@link #isInitialized()} state. */
220    InitializationMonitor(HMaster master) {
221      super("MasterInitializationMonitor");
222      this.master = master;
223      this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
224      this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
225      this.setDaemon(true);
226    }
227
228    @Override
229    public void run() {
230      try {
231        while (!master.isStopped() && master.isActiveMaster()) {
232          Thread.sleep(timeout);
233          if (master.isInitialized()) {
234            LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
235          } else {
236            LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
237                + " consider submitting a bug report including a thread dump of this process.");
238            if (haltOnTimeout) {
239              LOG.error("Zombie Master exiting. Thread dump to stdout");
240              Threads.printThreadInfo(System.out, "Zombie HMaster");
241              System.exit(-1);
242            }
243          }
244        }
245      } catch (InterruptedException ie) {
246        LOG.trace("InitMonitor thread interrupted. Existing.");
247      }
248    }
249  }

398  /**
399   * Initializes the HMaster. The steps are as follows:
400   * <p>
401   * <ol>
402   * <li>Initialize the local HRegionServer
403   * <li>Start the ActiveMasterManager.
404   * </ol>
405   * <p>
406   * Remaining steps of initialization occur in
407   * #finishActiveMasterInitialization(MonitoredTask) after
408   * the master becomes the active one.
409   */
410  public HMaster(final Configuration conf, CoordinatedStateManager csm)
411      throws IOException, KeeperException {
412    super(conf, csm);
413    this.rsFatals = new MemoryBoundedLogMessageBuffer(
414      conf.getLong("hbase.master.buffer.for.rs.fatals", 1*1024*1024));
415
416    LOG.info("hbase.rootdir=" + getRootDir() +
417      ", hbase.cluster.distributed=" + this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
418
419    // Disable usage of meta replicas in the master
420    this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
421
422    Replication.decorateMasterConfiguration(this.conf);
423
424    // Hack! Maps DFSClient => Master for logs.  HDFS made this
425    // config param for task trackers, but we can piggyback off of it.
426    if (this.conf.get("mapreduce.task.attempt.id") == null) {
427      this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
428    }
429
430    // should we check the compression codec type at master side, default true, HBASE-6370
431    this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
432
433    // should we check encryption settings at master side, default true
434    this.masterCheckEncryption = conf.getBoolean("hbase.master.check.encryption", true);
435
436    this.metricsMaster = new MetricsMaster(new MetricsMasterWrapperImpl(this));
437
438    // preload table descriptor at startup
439    this.preLoadTableDescriptors = conf.getBoolean("hbase.master.preload.tabledescriptors", true);
440
441    this.maxBlancingTime = getMaxBalancingTime();
442    this.maxRitPercent = conf.getDouble(HConstants.HBASE_MASTER_BALANCER_MAX_RIT_PERCENT,
443      HConstants.DEFAULT_HBASE_MASTER_BALANCER_MAX_RIT_PERCENT);
444
445    // Do we publish the status?
446
447    boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
448        HConstants.STATUS_PUBLISHED_DEFAULT);
449    Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
450        conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
451            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
452            ClusterStatusPublisher.Publisher.class);
453
454    if (shouldPublish) {
455      if (publisherClass == null) {
456        LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
457            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
458            " is not set - not publishing status");
459      } else {
460        clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
461        getChoreService().scheduleChore(clusterStatusPublisherChore);
462      }
463    }
464
465    // Some unit tests don't need a cluster, so no zookeeper at all
466    if (!conf.getBoolean("hbase.testing.nocluster", false)) {
467      setInitLatch(new CountDownLatch(1));
468      activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName, this);
469      int infoPort = putUpJettyServer();
470      startActiveMasterManager(infoPort);
471    } else {
472      activeMasterManager = null;
473    }
474  }

476  // return the actual infoPort, -1 means disable info server.
477  private int putUpJettyServer() throws IOException {
478    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
479      return -1;
480    }
481    int infoPort = conf.getInt("hbase.master.info.port.orig",
482      HConstants.DEFAULT_MASTER_INFOPORT);
483    // -1 is for disabling info server, so no redirecting
484    if (infoPort < 0 || infoServer == null) {
485      return -1;
486    }
487    String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
488    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
489      String msg =
490          "Failed to start redirecting jetty server. Address " + addr
491              + " does not belong to this host. Correct configuration parameter: "
492              + "hbase.master.info.bindAddress";
493      LOG.error(msg);
494      throw new IOException(msg);
495    }
496
497    RedirectServlet.regionServerInfoPort = infoServer.getPort();
498    if(RedirectServlet.regionServerInfoPort == infoPort) {
499      return infoPort;
500    }
501    masterJettyServer = new org.mortbay.jetty.Server();
502    Connector connector = new SelectChannelConnector();
503    connector.setHost(addr);
504    connector.setPort(infoPort);
505    masterJettyServer.addConnector(connector);
506    masterJettyServer.setStopAtShutdown(true);
507    Context context = new Context(masterJettyServer, "/", Context.NO_SESSIONS);
508    context.addServlet(RedirectServlet.class, "/*");
509    try {
510      masterJettyServer.start();
511    } catch (Exception e) {
512      throw new IOException("Failed to start redirecting jetty server", e);
513    }
514    return connector.getLocalPort();
515  }

661  /**
662   * Finish initialization of HMaster after becoming the primary master.
663   *
664   * <ol>
665   * <li>Initialize master components - file system manager, server manager,
666   *     assignment manager, region server tracker, etc</li>
667   * <li>Start necessary service threads - balancer, catalog janior,
668   *     executor services, etc</li>
669   * <li>Set cluster as UP in ZooKeeper</li>
670   * <li>Wait for RegionServers to check-in</li>
671   * <li>Split logs and perform data recovery, if necessary</li>
672   * <li>Ensure assignment of meta/namespace regions<li>
673   * <li>Handle either fresh cluster start or master failover</li>
674   * </ol>
675   */
676  private void finishActiveMasterInitialization(MonitoredTask status)
677      throws IOException, InterruptedException, KeeperException, CoordinatedStateException {
678
679    isActiveMaster = true;
680    Thread zombieDetector = new Thread(new InitializationMonitor(this),
681        "ActiveMasterInitializationMonitor-" + System.currentTimeMillis());
682    zombieDetector.start();
683
684    /*
685     * We are active master now... go initialize components we need to run.
686     * Note, there may be dross in zk from previous runs; it'll get addressed
687     * below after we determine if cluster startup or failover.
688     */
689
690    status.setStatus("Initializing Master file system");
691
692    this.masterActiveTime = System.currentTimeMillis();
693    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
694    this.fileSystemManager = new MasterFileSystem(this);
695    this.walManager = new MasterWalManager(this);
696
697    // enable table descriptors cache
698    this.tableDescriptors.setCacheOn();
699    // set the META's descriptor to the correct replication
700    this.tableDescriptors.get(TableName.META_TABLE_NAME).setRegionReplication(
701        conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM));
702    // warm-up HTDs cache on master initialization
703    if (preLoadTableDescriptors) {
704      status.setStatus("Pre-loading table descriptors");
705      this.tableDescriptors.getAll();
706    }
707
708    // publish cluster ID
709    status.setStatus("Publishing Cluster ID in ZooKeeper");
710    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
711    this.initLatch.countDown();
712
713    this.serverManager = createServerManager(this);
714
715    // Invalidate all write locks held previously
716    this.tableLockManager.reapWriteLocks();
717    this.tableStateManager = new TableStateManager(this);
718
719    status.setStatus("Initializing ZK system trackers");
720    initializeZKBasedSystemTrackers();
721
722    // This is for backwards compatibility
723    // See HBASE-11393
724    status.setStatus("Update TableCFs node in ZNode");
725    TableCFsUpdater tableCFsUpdater = new TableCFsUpdater(zooKeeper,
726            conf, this.clusterConnection);
727    tableCFsUpdater.update();
728
729    // initialize master side coprocessors before we start handling requests
730    status.setStatus("Initializing master coprocessors");
731    this.cpHost = new MasterCoprocessorHost(this, this.conf);
732
733    // start up all service threads.
734    status.setStatus("Initializing master service threads");
735    startServiceThreads();
736
737    // Wake up this server to check in
738    sleeper.skipSleepCycle();
739
740    // Wait for region servers to report in
741    status.setStatus("Wait for region servers to report in");
742    waitForRegionServers(status);
743
744    // get a list for previously failed RS which need log splitting work
745    // we recover hbase:meta region servers inside master initialization and
746    // handle other failed servers in SSH in order to start up master node ASAP
747    MasterMetaBootstrap metaBootstrap = createMetaBootstrap(this, status);
748    metaBootstrap.splitMetaLogsBeforeAssignment();
749
750    this.initializationBeforeMetaAssignment = true;
751
752    // Wait for regionserver to finish initialization.
753    if (BaseLoadBalancer.tablesOnMaster(conf)) {
754      waitForServerOnline();
755    }
756
757    //initialize load balancer
758    this.balancer.setMasterServices(this);
759    this.balancer.setClusterStatus(getClusterStatus());
760    this.balancer.initialize();
761
762    // Check if master is shutting down because of some issue
763    // in initializing the regionserver or the balancer.
764    if (isStopped()) return;
765
766    // Make sure meta assigned before proceeding.
767    status.setStatus("Assigning Meta Region");
768    metaBootstrap.assignMeta();
769
770    // check if master is shutting down because above assignMeta could return even hbase:meta isn't
771    // assigned when master is shutting down
772    if (isStopped()) return;
773
774    // migrating existent table state from zk, so splitters
775    // and recovery process treat states properly.
776    for (Map.Entry<TableName, TableState.State> entry : ZKDataMigrator
777        .queryForTableStates(getZooKeeper()).entrySet()) {
778      LOG.info("Converting state from zk to new states:" + entry);
779      tableStateManager.setTableState(entry.getKey(), entry.getValue());
780    }
781    ZKUtil.deleteChildrenRecursively(getZooKeeper(), getZooKeeper().znodePaths.tableZNode);
782
783    status.setStatus("Submitting log splitting work for previously failed region servers");
784    metaBootstrap.processDeadServers();
785
786    // Fix up assignment manager status
787    status.setStatus("Starting assignment manager");
788    this.assignmentManager.joinCluster();
789
790    // set cluster status again after user regions are assigned
791    this.balancer.setClusterStatus(getClusterStatus());
792
793    // Start balancer and meta catalog janitor after meta and regions have been assigned.
794    status.setStatus("Starting balancer and catalog janitor");
795    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
796    getChoreService().scheduleChore(clusterStatusChore);
797    this.balancerChore = new BalancerChore(this);
798    getChoreService().scheduleChore(balancerChore);
799    this.normalizerChore = new RegionNormalizerChore(this);
800    getChoreService().scheduleChore(normalizerChore);
801    this.catalogJanitorChore = new CatalogJanitor(this);
802    getChoreService().scheduleChore(catalogJanitorChore);
803
804    // Do Metrics periodically
805    periodicDoMetricsChore = new PeriodicDoMetrics(msgInterval, this);
806    getChoreService().scheduleChore(periodicDoMetricsChore);
807
808    status.setStatus("Starting cluster schema service");
809    initClusterSchemaService();
810
811    if (this.cpHost != null) {
812      try {
813        this.cpHost.preMasterInitialization();
814      } catch (IOException e) {
815        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
816      }
817    }
818
819    status.markComplete("Initialization successful");
820    LOG.info("Master has completed initialization");
821    configurationManager.registerObserver(this.balancer);
822
823    // Set master as 'initialized'.
824    setInitialized(true);
825
826    status.setStatus("Assign meta replicas");
827    metaBootstrap.assignMetaReplicas();
828
829    status.setStatus("Starting quota manager");
830    initQuotaManager();
831
832    // clear the dead servers with same host name and port of online server because we are not
833    // removing dead server with same hostname and port of rs which is trying to check in before
834    // master initialization. See HBASE-5916.
835    this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
836
837    // Check and set the znode ACLs if needed in case we are overtaking a non-secure configuration
838    status.setStatus("Checking ZNode ACLs");
839    zooKeeper.checkAndSetZNodeAcls();
840
841    status.setStatus("Initializing MOB Cleaner");
842    initMobCleaner();
843
844    status.setStatus("Calling postStartMaster coprocessors");
845    if (this.cpHost != null) {
846      // don't let cp initialization errors kill the master
847      try {
848        this.cpHost.postStartMaster();
849      } catch (IOException ioe) {
850        LOG.error("Coprocessor postStartMaster() hook failed", ioe);
851      }
852    }
853
854    zombieDetector.interrupt();
855  }

966      conf.getInt("hbase.master.executor.logreplayops.threads", 10));

969   // at a time.  To do concurrency, would need fencing of enable/disable of

1146  /**
1147   * It first sleep to the next balance plan start time. Meanwhile, throttling by the max
1148   * number regions in transition to protect availability.
1149   * @param nextBalanceStartTime The next balance plan start time
1150   * @param maxRegionsInTransition max number of regions in transition
1151   * @param cutoffTime when to exit balancer
1152   */
1153  private void balanceThrottling(long nextBalanceStartTime, int maxRegionsInTransition,
1154      long cutoffTime) {
1155    boolean interrupted = false;
1156
1157    // Sleep to next balance plan start time
1158    // But if there are zero regions in transition, it can skip sleep to speed up.
1159    while (!interrupted && System.currentTimeMillis() < nextBalanceStartTime
1160        && this.assignmentManager.getRegionStates().getRegionsInTransitionCount() != 0) {
1161      try {
1162        Thread.sleep(100);
1163      } catch (InterruptedException ie) {
1164        interrupted = true;
1165      }
1166    }
1167
1168    // Throttling by max number regions in transition
1169    while (!interrupted
1170        && maxRegionsInTransition > 0
1171        && this.assignmentManager.getRegionStates().getRegionsInTransitionCount()
1172        >= maxRegionsInTransition && System.currentTimeMillis() <= cutoffTime) {
1173      try {
1174        // sleep if the number of regions in transition exceeds the limit
1175        Thread.sleep(100);
1176      } catch (InterruptedException ie) {
1177        interrupted = true;
1178      }
1179    }
1180
1181    if (interrupted) Thread.currentThread().interrupt();
1182  }

1188  public boolean balance(boolean force) throws IOException {
1189    // if master not initialized, don't run balancer.
1190    if (!isInitialized()) {
1191      LOG.debug("Master has not been initialized, don't run balancer.");
1192      return false;
1193    }
1194
1195    if (isInMaintenanceMode()) {
1196      LOG.info("Master is in maintenanceMode mode, don't run balancer.");
1197      return false;
1198    }
1199
1200    int maxRegionsInTransition = getMaxRegionsInTransition();
1201    synchronized (this.balancer) {
1202      // If balance not true, don't run balancer.
1203      if (!this.loadBalancerTracker.isBalancerOn()) return false;
1204      // Only allow one balance run at at time.
1205      if (this.assignmentManager.getRegionStates().isRegionsInTransition()) {
1206        Set<RegionState> regionsInTransition =
1207          this.assignmentManager.getRegionStates().getRegionsInTransition();
1208        // if hbase:meta region is in transition, result of assignment cannot be recorded
1209        // ignore the force flag in that case
1210        boolean metaInTransition = assignmentManager.getRegionStates().isMetaRegionInTransition();
1211        String prefix = force && !metaInTransition ? "R" : "Not r";
1212        LOG.debug(prefix + "unning balancer because " + regionsInTransition.size() +
1213          " region(s) in transition: " + org.apache.commons.lang.StringUtils.
1214            abbreviate(regionsInTransition.toString(), 256));
1215        if (!force || metaInTransition) return false;
1216      }
1217      if (this.serverManager.areDeadServersInProgress()) {
1218        LOG.debug("Not running balancer because processing dead regionserver(s): " +
1219          this.serverManager.getDeadServers());
1220        return false;
1221      }
1222
1223      if (this.cpHost != null) {
1224        try {
1225          if (this.cpHost.preBalance()) {
1226            LOG.debug("Coprocessor bypassing balancer request");
1227            return false;
1228          }
1229        } catch (IOException ioe) {
1230          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1231          return false;
1232        }
1233      }
1234
1235      Map<TableName, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1236        this.assignmentManager.getRegionStates().getAssignmentsByTable();
1237
1238      List<RegionPlan> plans = new ArrayList<RegionPlan>();
1239
1240      //Give the balancer the current cluster state.
1241      this.balancer.setClusterStatus(getClusterStatus());
1242      this.balancer.setClusterLoad(
1243              this.assignmentManager.getRegionStates().getAssignmentsByTable(true));
1244
1245      for (Entry<TableName, Map<ServerName, List<HRegionInfo>>> e : assignmentsByTable.entrySet()) {
1246        List<RegionPlan> partialPlans = this.balancer.balanceCluster(e.getKey(), e.getValue());
1247        if (partialPlans != null) plans.addAll(partialPlans);
1248      }
1249
1250      long balanceStartTime = System.currentTimeMillis();
1251      long cutoffTime = balanceStartTime + this.maxBlancingTime;
1252      int rpCount = 0;  // number of RegionPlans balanced so far
1253      if (plans != null && !plans.isEmpty()) {
1254        int balanceInterval = this.maxBlancingTime / plans.size();
1255        LOG.info("Balancer plans size is " + plans.size() + ", the balance interval is "
1256            + balanceInterval + " ms, and the max number regions in transition is "
1257            + maxRegionsInTransition);
1258
1259        for (RegionPlan plan: plans) {
1260          LOG.info("balance " + plan);
1261          //TODO: bulk assign
1262          this.assignmentManager.balance(plan);
1263          rpCount++;
1264
1265          balanceThrottling(balanceStartTime + rpCount * balanceInterval, maxRegionsInTransition,
1266            cutoffTime);
1267
1268          // if performing next balance exceeds cutoff time, exit the loop
1269          if (rpCount < plans.size() && System.currentTimeMillis() > cutoffTime) {
1270            // TODO: After balance, there should not be a cutoff time (keeping it as
1271            // a security net for now)
1272            LOG.debug("No more balancing till next balance run; maxBalanceTime="
1273                + this.maxBlancingTime);
1274            break;
1275          }
1276        }
1277      }
1278
1279      if (this.cpHost != null) {
1280        try {
1281          this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1282        } catch (IOException ioe) {
1283          // balancing already succeeded so don't change the result
1284          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1285        }
1286      }
1287    }
1288    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1289    // Return true indicating a success.
1290    return true;
1291  }

1300   * Perform normalization of cluster (invoked by {@link RegionNormalizerChore}).
1301   *
1302   * @return true if normalization step was performed successfully, false otherwise
1303   *    (specifically, if HMaster hasn't been initialized properly or normalization
1304   *    is globally disabled)
1305   */
1306  public boolean normalizeRegions() throws IOException {
1307    if (!isInitialized()) {
1308      LOG.debug("Master has not been initialized, don't run region normalizer.");
1309      return false;
1310    }
1311
1312    if (isInMaintenanceMode()) {
1313      LOG.info("Master is in maintenance mode, don't run region normalizer.");
1314      return false;
1315    }
1316
1317    if (!this.regionNormalizerTracker.isNormalizerOn()) {
1318      LOG.debug("Region normalization is disabled, don't run region normalizer.");
1319      return false;
1320    }
1321
1322    synchronized (this.normalizer) {
1323      // Don't run the normalizer concurrently
1324      List<TableName> allEnabledTables = new ArrayList<>(
1325        this.tableStateManager.getTablesInStates(TableState.State.ENABLED));
1326
1327      Collections.shuffle(allEnabledTables);
1328
1329      for (TableName table : allEnabledTables) {
1330        if (isInMaintenanceMode()) {
1331          LOG.debug("Master is in maintenance mode, stop running region normalizer.");
1332          return false;
1333        }
1334
1335        HTableDescriptor tblDesc = getTableDescriptors().get(table);
1336        if (table.isSystemTable() || (tblDesc != null &&
1337            !tblDesc.isNormalizationEnabled())) {
1338          LOG.debug("Skipping normalization for table: " + table + ", as it's either system"
1339              + " table or doesn't have auto normalization turned on");
1340          continue;
1341        }
1342        List<NormalizationPlan> plans = this.normalizer.computePlanForTable(table);
1343        if (plans != null) {
1344          for (NormalizationPlan plan : plans) {
1345            plan.execute(clusterConnection.getAdmin());
1346            if (plan.getType() == PlanType.SPLIT) {
1347              splitPlanCount++;
1348            } else if (plan.getType() == PlanType.MERGE) {
1349              mergePlanCount++;
1350            }
1351          }
1352        }
1353      }
1354    }
1355    // If Region did not generate any plans, it means the cluster is already balanced.
1356    // Return true indicating a success.
1357    return true;
1358  }

1489  void move(final byte[] encodedRegionName,
1490      final byte[] destServerName) throws HBaseIOException {
1491    RegionState regionState = assignmentManager.getRegionStates().
1492      getRegionState(Bytes.toString(encodedRegionName));
1493
1494    HRegionInfo hri;
1495    if (regionState != null) {
1496      hri = regionState.getRegion();
1497    } else {
1498      throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1499    }
1500
1501    ServerName dest;
1502    if (destServerName == null || destServerName.length == 0) {
1503      LOG.info("Passed destination servername is null/empty so " +
1504        "choosing a server at random");
1505      final List<ServerName> destServers = this.serverManager.createDestinationServersList(
1506        regionState.getServerName());
1507      dest = balancer.randomAssignment(hri, destServers);
1508      if (dest == null) {
1509        LOG.debug("Unable to determine a plan to assign " + hri);
1510        return;
1511      }
1512    } else {
1513      ServerName candidate = ServerName.valueOf(Bytes.toString(destServerName));
1514      dest = balancer.randomAssignment(hri, Lists.newArrayList(candidate));
1515      if (dest == null) {
1516        LOG.debug("Unable to determine a plan to assign " + hri);
1517        return;
1518      }
1519      if (dest.equals(serverName) && balancer instanceof BaseLoadBalancer
1520          && !((BaseLoadBalancer)balancer).shouldBeOnMaster(hri)) {
1521        // To avoid unnecessary region moving later by balancer. Don't put user
1522        // regions on master. Regions on master could be put on other region
1523        // server intentionally by test however.
1524        LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1525          + " to avoid unnecessary region moving later by load balancer,"
1526          + " because it should not be on master");
1527        return;
1528      }
1529    }
1530
1531    if (dest.equals(regionState.getServerName())) {
1532      LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1533        + " because region already assigned to the same server " + dest + ".");
1534      return;
1535    }
1536
1537    // Now we can do the move
1538    RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1539
1540    try {
1541      checkInitialized();
1542      if (this.cpHost != null) {
1543        if (this.cpHost.preMove(hri, rp.getSource(), rp.getDestination())) {
1544          return;
1545        }
1546      }
1547      // warmup the region on the destination before initiating the move. this call
1548      // is synchronous and takes some time. doing it before the source region gets
1549      // closed
1550      serverManager.sendRegionWarmup(rp.getDestination(), hri);
1551
1552      LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
1553      this.assignmentManager.balance(rp);
1554      if (this.cpHost != null) {
1555        this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
1556      }
1557    } catch (IOException ioe) {
1558      if (ioe instanceof HBaseIOException) {
1559        throw (HBaseIOException)ioe;
1560      }
1561      throw new HBaseIOException(ioe);
1562    }
1563  }

1627  /**
1628   * Checks whether the table conforms to some sane limits, and configured
1629   * values (compression, etc) work. Throws an exception if something is wrong.
1630   * @throws IOException
1631   */
1632  private void sanityCheckTableDescriptor(final HTableDescriptor htd) throws IOException {
1633    final String CONF_KEY = "hbase.table.sanity.checks";
1634    boolean logWarn = false;
1635    if (!conf.getBoolean(CONF_KEY, true)) {
1636      logWarn = true;
1637    }
1638    String tableVal = htd.getConfigurationValue(CONF_KEY);
1639    if (tableVal != null && !Boolean.valueOf(tableVal)) {
1640      logWarn = true;
1641    }
1642
1643    // check max file size
1644    long maxFileSizeLowerLimit = 2 * 1024 * 1024L; // 2M is the default lower limit
1645    long maxFileSize = htd.getMaxFileSize();
1646    if (maxFileSize < 0) {
1647      maxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE, maxFileSizeLowerLimit);
1648    }
1649    if (maxFileSize < conf.getLong("hbase.hregion.max.filesize.limit", maxFileSizeLowerLimit)) {
1650      String message = "MAX_FILESIZE for table descriptor or "
1651          + "\"hbase.hregion.max.filesize\" (" + maxFileSize
1652          + ") is too small, which might cause over splitting into unmanageable "
1653          + "number of regions.";
1654      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1655    }
1656
1657    // check flush size
1658    long flushSizeLowerLimit = 1024 * 1024L; // 1M is the default lower limit
1659    long flushSize = htd.getMemStoreFlushSize();
1660    if (flushSize < 0) {
1661      flushSize = conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSizeLowerLimit);
1662    }
1663    if (flushSize < conf.getLong("hbase.hregion.memstore.flush.size.limit", flushSizeLowerLimit)) {
1664      String message = "MEMSTORE_FLUSHSIZE for table descriptor or "
1665          + "\"hbase.hregion.memstore.flush.size\" ("+flushSize+") is too small, which might cause"
1666          + " very frequent flushing.";
1667      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1668    }
1669
1670    // check that coprocessors and other specified plugin classes can be loaded
1671    try {
1672      checkClassLoading(conf, htd);
1673    } catch (Exception ex) {
1674      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, ex.getMessage(), null);
1675    }
1676
1677    // check compression can be loaded
1678    try {
1679      checkCompression(htd);
1680    } catch (IOException e) {
1681      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1682    }
1683
1684    // check encryption can be loaded
1685    try {
1686      checkEncryption(conf, htd);
1687    } catch (IOException e) {
1688      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1689    }
1690    // Verify compaction policy
1691    try{
1692      checkCompactionPolicy(conf, htd);
1693    } catch(IOException e){
1694      warnOrThrowExceptionForFailure(false, CONF_KEY, e.getMessage(), e);
1695    }
1696    // check that we have at least 1 CF
1697    if (htd.getColumnFamilyCount() == 0) {
1698      String message = "Table should have at least one column family.";
1699      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1700    }
1701
1702    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1703      if (hcd.getTimeToLive() <= 0) {
1704        String message = "TTL for column family " + hcd.getNameAsString() + " must be positive.";
1705        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1706      }
1707
1708      // check blockSize
1709      if (hcd.getBlocksize() < 1024 || hcd.getBlocksize() > 16 * 1024 * 1024) {
1710        String message = "Block size for column family " + hcd.getNameAsString()
1711            + "  must be between 1K and 16MB.";
1712        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1713      }
1714
1715      // check versions
1716      if (hcd.getMinVersions() < 0) {
1717        String message = "Min versions for column family " + hcd.getNameAsString()
1718          + "  must be positive.";
1719        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1720      }
1721      // max versions already being checked
1722
1723      // HBASE-13776 Setting illegal versions for HColumnDescriptor
1724      //  does not throw IllegalArgumentException
1725      // check minVersions <= maxVerions
1726      if (hcd.getMinVersions() > hcd.getMaxVersions()) {
1727        String message = "Min versions for column family " + hcd.getNameAsString()
1728            + " must be less than the Max versions.";
1729        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1730      }
1731
1732      // check replication scope
1733      checkReplicationScope(hcd);
1734
1735      // check data replication factor, it can be 0(default value) when user has not explicitly
1736      // set the value, in this case we use default replication factor set in the file system.
1737      if (hcd.getDFSReplication() < 0) {
1738        String message = "HFile Replication for column family " + hcd.getNameAsString()
1739            + "  must be greater than zero.";
1740        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1741      }
1742
1743      // TODO: should we check coprocessors and encryption ?
1744    }
1745  }

1759  private void checkCompactionPolicy(Configuration conf, HTableDescriptor htd)
1760      throws IOException {
1761    // FIFO compaction has some requirements
1762    // Actually FCP ignores periodic major compactions
1763    String className =
1764        htd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1765    if (className == null) {
1766      className =
1767          conf.get(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
1768            ExploringCompactionPolicy.class.getName());
1769    }
1770
1771    int blockingFileCount = HStore.DEFAULT_BLOCKING_STOREFILE_COUNT;
1772    String sv = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1773    if (sv != null) {
1774      blockingFileCount = Integer.parseInt(sv);
1775    } else {
1776      blockingFileCount = conf.getInt(HStore.BLOCKING_STOREFILES_KEY, blockingFileCount);
1777    }
1778
1779    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1780      String compactionPolicy =
1781          hcd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1782      if (compactionPolicy == null) {
1783        compactionPolicy = className;
1784      }
1785      if (!compactionPolicy.equals(FIFOCompactionPolicy.class.getName())) {
1786        continue;
1787      }
1788      // FIFOCompaction
1789      String message = null;
1790
1791      // 1. Check TTL
1792      if (hcd.getTimeToLive() == HColumnDescriptor.DEFAULT_TTL) {
1793        message = "Default TTL is not supported for FIFO compaction";
1794        throw new IOException(message);
1795      }
1796
1797      // 2. Check min versions
1798      if (hcd.getMinVersions() > 0) {
1799        message = "MIN_VERSION > 0 is not supported for FIFO compaction";
1800        throw new IOException(message);
1801      }
1802
1803      // 3. blocking file count
1804      String sbfc = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1805      if (sbfc != null) {
1806        blockingFileCount = Integer.parseInt(sbfc);
1807      }
1808      if (blockingFileCount < 1000) {
1809        message =
1810            "blocking file count '" + HStore.BLOCKING_STOREFILES_KEY + "' " + blockingFileCount
1811                + " is below recommended minimum of 1000";

1827  private void startActiveMasterManager(int infoPort) throws KeeperException {
1828    String backupZNode = ZKUtil.joinZNode(
1829      zooKeeper.znodePaths.backupMasterAddressesZNode, serverName.toString());
1830    /*
1831    * Add a ZNode for ourselves in the backup master directory since we
1832    * may not become the active master. If so, we want the actual active
1833    * master to know we are backup masters, so that it won't assign
1834    * regions to us if so configured.
1835    *
1836    * If we become the active master later, ActiveMasterManager will delete
1837    * this node explicitly.  If we crash before then, ZooKeeper will delete
1838    * this node for us since it is ephemeral.
1839    */
1840    LOG.info("Adding backup master ZNode " + backupZNode);
1841    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode,
1842        serverName, infoPort)) {
1843      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
1844    }
1845
1846    activeMasterManager.setInfoPort(infoPort);
1847    // Start a thread to try to become the active master, so we won't block here
1848    Threads.setDaemonThreadRunning(new Thread(new Runnable() {
1849      @Override
1850      public void run() {
1851        int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
1852          HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
1853        // If we're a backup master, stall until a primary to writes his address
1854        if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP,
1855          HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
1856          LOG.debug("HMaster started in backup mode. "
1857            + "Stalling until master znode is written.");
1858          // This will only be a minute or so while the cluster starts up,
1859          // so don't worry about setting watches on the parent znode
1860          while (!activeMasterManager.hasActiveMaster()) {
1861            LOG.debug("Waiting for master address ZNode to be written "
1862              + "(Also watching cluster state node)");
1863            Threads.sleep(timeout);
1864          }
1865        }
1866        MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
1867        status.setDescription("Master startup");
1868        try {
1869          if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
1870            finishActiveMasterInitialization(status);
1871          }
1872        } catch (Throwable t) {
1873          status.setStatus("Failed to become active: " + t.getMessage());
1874          LOG.fatal("Failed to become active master", t);
1875          // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
1876          if (t instanceof NoClassDefFoundError &&
1877            t.getMessage()
1878              .contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
1879            // improved error message for this special case
1880            abort("HBase is having a problem with its Hadoop jars.  You may need to "
1881              + "recompile HBase against Hadoop version "
1882              + org.apache.hadoop.util.VersionInfo.getVersion()
1883              + " or change your hadoop jars to start properly", t);
1884          } else {
1885            abort("Unhandled exception. Starting shutdown.", t);
1886          }
1887        } finally {
1888          status.cleanup();
1889        }
1890      }
1891    }, getServerName().toShortString() + ".activeMasterManager"));
1892  }

2225    // Build Set of backup masters from ZK nodes
2226    List<String> backupMasterStrings;
2227    try {
2228      backupMasterStrings = ZKUtil.listChildrenNoWatch(this.zooKeeper,
2229        this.zooKeeper.znodePaths.backupMasterAddressesZNode);
2230    } catch (KeeperException e) {
2231      LOG.warn(this.zooKeeper.prefix("Unable to list backup servers"), e);
2232      backupMasterStrings = null;
2233    }
2234
2235    List<ServerName> backupMasters = null;
2236    if (backupMasterStrings != null && !backupMasterStrings.isEmpty()) {
2237      backupMasters = new ArrayList<ServerName>(backupMasterStrings.size());
2238      for (String s: backupMasterStrings) {
2239        try {
2240          byte [] bytes;
2241          try {
2242            bytes = ZKUtil.getData(this.zooKeeper, ZKUtil.joinZNode(
2243                this.zooKeeper.znodePaths.backupMasterAddressesZNode, s));
2244          } catch (InterruptedException e) {
2245            throw new InterruptedIOException();
2246          }
2247          if (bytes != null) {
2248            ServerName sn;
2249            try {
2250              sn = ProtobufUtil.parseServerNameFrom(bytes);
2251            } catch (DeserializationException e) {
2252              LOG.warn("Failed parse, skipping registering backup server", e);
2253              continue;
2254            }
2255            backupMasters.add(sn);
2256          }
2257        } catch (KeeperException e) {
2258          LOG.warn(this.zooKeeper.prefix("Unable to get information about " +
2259                   "backup servers"), e);
2260        }
2261      }
2262      Collections.sort(backupMasters, new Comparator<ServerName>() {
2263        @Override
2264        public int compare(ServerName s1, ServerName s2) {
2265          return s1.getServerName().compareTo(s2.getServerName());
2266        }});
2267    }

2270      fileSystemManager.getClusterId().toString() : null;
2271    Set<RegionState> regionsInTransition = assignmentManager != null ?
2272      assignmentManager.getRegionStates().getRegionsInTransition() : null;
2273    String[] coprocessors = cpHost != null ? getMasterCoprocessors() : null;

2800  /**
2801   * @return list of table table descriptors after filtering by regex and whether to include system
2802   *    tables, etc.
2803   * @throws IOException
2804   */
2805  private List<HTableDescriptor> getTableDescriptors(final List<HTableDescriptor> htds,
2806      final String namespace, final String regex, final List<TableName> tableNameList,
2807      final boolean includeSysTables)
2808  throws IOException {
2809    if (tableNameList == null || tableNameList.size() == 0) {
2810      // request for all TableDescriptors
2811      Collection<HTableDescriptor> allHtds;
2812      if (namespace != null && namespace.length() > 0) {
2813        // Do a check on the namespace existence. Will fail if does not exist.
2814        this.clusterSchemaService.getNamespace(namespace);
2815        allHtds = tableDescriptors.getByNamespace(namespace).values();
2816      } else {
2817        allHtds = tableDescriptors.getAll().values();
2818      }
2819      for (HTableDescriptor desc: allHtds) {
2820        if (tableStateManager.isTablePresent(desc.getTableName())
2821            && (includeSysTables || !desc.getTableName().isSystemTable())) {
2822          htds.add(desc);
2823        }
2824      }
2825    } else {
2826      for (TableName s: tableNameList) {
2827        if (tableStateManager.isTablePresent(s)) {
2828          HTableDescriptor desc = tableDescriptors.get(s);
2829          if (desc != null) {
2830            htds.add(desc);
2831          }
2832        }
2833      }
2834    }
2835
2836    // Retains only those matched by regular expression.
2837    if (regex != null) filterTablesByRegex(htds, Pattern.compile(regex));
2838    return htds;
2839  }

2841  /**
2842   * Removes the table descriptors that don't match the pattern.
2843   * @param descriptors list of table descriptors to filter
2844   * @param pattern the regex to use
2845   */
2846  private static void filterTablesByRegex(final Collection<HTableDescriptor> descriptors,
2847      final Pattern pattern) {
2848    final String defaultNS = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
2849    Iterator<HTableDescriptor> itr = descriptors.iterator();
2850    while (itr.hasNext()) {
2851      HTableDescriptor htd = itr.next();
2852      String tableName = htd.getTableName().getNameAsString();
2853      boolean matched = pattern.matcher(tableName).matches();
2854      if (!matched && htd.getTableName().getNamespaceAsString().equals(defaultNS)) {
2855        matched = pattern.matcher(defaultNS + TableName.NAMESPACE_DELIM + tableName).matches();
2856      }
2857      if (!matched) {
2858        itr.remove();
2859      }
2860    }
2861  }

2890  public void reportMobCompactionStart(TableName tableName) throws IOException {
2891    IdLock.Entry lockEntry = null;
2892    try {
2893      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2894      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2895      if (compactionsCount == null) {
2896        compactionsCount = new AtomicInteger(0);
2897        mobCompactionStates.put(tableName, compactionsCount);
2898      }
2899      compactionsCount.incrementAndGet();
2900    } finally {
2901      if (lockEntry != null) {
2902        mobCompactionLock.releaseLockEntry(lockEntry);
2903      }
2904    }
2905  }

2907  public void reportMobCompactionEnd(TableName tableName) throws IOException {
2908    IdLock.Entry lockEntry = null;
2909    try {
2910      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2911      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2912      if (compactionsCount != null) {
2913        int count = compactionsCount.decrementAndGet();
2914        // remove the entry if the count is 0.
2915        if (count == 0) {
2916          mobCompactionStates.remove(tableName);
2917        }
2918      }
2919    } finally {
2920      if (lockEntry != null) {
2921        mobCompactionLock.releaseLockEntry(lockEntry);
2922      }
2923    }
2924  }
