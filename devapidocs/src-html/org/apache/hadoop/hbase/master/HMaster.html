175/**
176 * HMaster is the "master server" for HBase. An HBase cluster has one active
177 * master.  If many masters are started, all compete.  Whichever wins goes on to
178 * run the cluster.  All others park themselves in their constructor until
179 * master or cluster shutdown or until the active master loses its lease in
180 * zookeeper.  Thereafter, all running master jostle to take over master role.
181 *
182 * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
183 * this case it will tell all regionservers to go down and then wait on them
184 * all reporting in that they are down.  This master will then shut itself down.
185 *
186 * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
187 *
188 * @see org.apache.zookeeper.Watcher
189 */
190@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
191@SuppressWarnings("deprecation")
192public class HMaster extends HRegionServer implements MasterServices {

195  /**
196   * Protection against zombie master. Started once Master accepts active responsibility and
197   * starts taking over responsibilities. Allows a finite time window before giving up ownership.
198   */
199  private static class InitializationMonitor extends HasThread {
200    /** The amount of time in milliseconds to sleep before checking initialization status. */
201    public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
202    public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
203
204    /**
205     * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
206     * true, do nothing otherwise.
207     */
208    public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
209    public static final boolean HALT_DEFAULT = false;
210
211    private final HMaster master;
212    private final long timeout;
213    private final boolean haltOnTimeout;
214
215    /** Creates a Thread that monitors the {@link #isInitialized()} state. */
216    InitializationMonitor(HMaster master) {
217      super("MasterInitializationMonitor");
218      this.master = master;
219      this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
220      this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
221      this.setDaemon(true);
222    }
223
224    @Override
225    public void run() {
226      try {
227        while (!master.isStopped() && master.isActiveMaster()) {
228          Thread.sleep(timeout);
229          if (master.isInitialized()) {
230            LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
231          } else {
232            LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
233                + " consider submitting a bug report including a thread dump of this process.");
234            if (haltOnTimeout) {
235              LOG.error("Zombie Master exiting. Thread dump to stdout");
236              Threads.printThreadInfo(System.out, "Zombie HMaster");
237              System.exit(-1);
238            }
239          }
240        }
241      } catch (InterruptedException ie) {
242        LOG.trace("InitMonitor thread interrupted. Existing.");
243      }
244    }
245  }

370  /**
371   * Initializes the HMaster. The steps are as follows:
372   * <p>
373   * <ol>
374   * <li>Initialize the local HRegionServer
375   * <li>Start the ActiveMasterManager.
376   * </ol>
377   * <p>
378   * Remaining steps of initialization occur in
379   * #finishActiveMasterInitialization(MonitoredTask) after
380   * the master becomes the active one.

383      throws IOException, KeeperException {
384    super(conf, csm);
385    this.rsFatals = new MemoryBoundedLogMessageBuffer(
386      conf.getLong("hbase.master.buffer.for.rs.fatals", 1*1024*1024));
387
388    LOG.info("hbase.rootdir=" + FSUtils.getRootDir(this.conf) +
389      ", hbase.cluster.distributed=" + this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
390
391    // Disable usage of meta replicas in the master
392    this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
393
394    Replication.decorateMasterConfiguration(this.conf);
395
396    // Hack! Maps DFSClient => Master for logs.  HDFS made this
397    // config param for task trackers, but we can piggyback off of it.
398    if (this.conf.get("mapreduce.task.attempt.id") == null) {
399      this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
400    }
401
402    // should we check the compression codec type at master side, default true, HBASE-6370
403    this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
404
405    // should we check encryption settings at master side, default true
406    this.masterCheckEncryption = conf.getBoolean("hbase.master.check.encryption", true);
407
408    this.metricsMaster = new MetricsMaster(new MetricsMasterWrapperImpl(this));
409
410    // preload table descriptor at startup
411    this.preLoadTableDescriptors = conf.getBoolean("hbase.master.preload.tabledescriptors", true);
412
413    // Do we publish the status?
414
415    boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
416        HConstants.STATUS_PUBLISHED_DEFAULT);
417    Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
418        conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
419            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
420            ClusterStatusPublisher.Publisher.class);
421
422    if (shouldPublish) {
423      if (publisherClass == null) {
424        LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
425            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
426            " is not set - not publishing status");
427      } else {
428        clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
429        getChoreService().scheduleChore(clusterStatusPublisherChore);
430      }
431    }
432
433    // Some unit tests don't need a cluster, so no zookeeper at all
434    if (!conf.getBoolean("hbase.testing.nocluster", false)) {
435      activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName, this);
436      int infoPort = putUpJettyServer();
437      startActiveMasterManager(infoPort);
438    } else {
439      activeMasterManager = null;
440    }
441  }

443  // return the actual infoPort, -1 means disable info server.
444  private int putUpJettyServer() throws IOException {
445    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
446      return -1;
447    }
448    int infoPort = conf.getInt("hbase.master.info.port.orig",
449      HConstants.DEFAULT_MASTER_INFOPORT);
450    // -1 is for disabling info server, so no redirecting
451    if (infoPort < 0 || infoServer == null) {
452      return -1;
453    }
454    String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
455    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
456      String msg =
457          "Failed to start redirecting jetty server. Address " + addr
458              + " does not belong to this host. Correct configuration parameter: "
459              + "hbase.master.info.bindAddress";
460      LOG.error(msg);
461      throw new IOException(msg);
462    }
463
464    RedirectServlet.regionServerInfoPort = infoServer.getPort();
465    if(RedirectServlet.regionServerInfoPort == infoPort) {
466      return infoPort;
467    }
468    masterJettyServer = new org.mortbay.jetty.Server();
469    Connector connector = new SelectChannelConnector();
470    connector.setHost(addr);
471    connector.setPort(infoPort);
472    masterJettyServer.addConnector(connector);
473    masterJettyServer.setStopAtShutdown(true);
474    Context context = new Context(masterJettyServer, "/", Context.NO_SESSIONS);
475    context.addServlet(RedirectServlet.class, "/*");
476    try {
477      masterJettyServer.start();
478    } catch (Exception e) {
479      throw new IOException("Failed to start redirecting jetty server", e);
480    }
481    return connector.getLocalPort();
482  }

626  /**
627   * Finish initialization of HMaster after becoming the primary master.
628   *
629   * <ol>
630   * <li>Initialize master components - file system manager, server manager,
631   *     assignment manager, region server tracker, etc</li>
632   * <li>Start necessary service threads - balancer, catalog janior,
633   *     executor services, etc</li>
634   * <li>Set cluster as UP in ZooKeeper</li>
635   * <li>Wait for RegionServers to check-in</li>
636   * <li>Split logs and perform data recovery, if necessary</li>
637   * <li>Ensure assignment of meta/namespace regions<li>
638   * <li>Handle either fresh cluster start or master failover</li>
639   * </ol>
640   */
641  private void finishActiveMasterInitialization(MonitoredTask status)
642      throws IOException, InterruptedException, KeeperException, CoordinatedStateException {
643
644    isActiveMaster = true;
645    Thread zombieDetector = new Thread(new InitializationMonitor(this));
646    zombieDetector.start();
647
648    /*
649     * We are active master now... go initialize components we need to run.
650     * Note, there may be dross in zk from previous runs; it'll get addressed
651     * below after we determine if cluster startup or failover.
652     */
653
654    status.setStatus("Initializing Master file system");
655
656    this.masterActiveTime = System.currentTimeMillis();
657    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
658    this.fileSystemManager = new MasterFileSystem(this, this);
659
660    // enable table descriptors cache
661    this.tableDescriptors.setCacheOn();
662    // set the META's descriptor to the correct replication
663    this.tableDescriptors.get(TableName.META_TABLE_NAME).setRegionReplication(
664        conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM));
665    // warm-up HTDs cache on master initialization
666    if (preLoadTableDescriptors) {
667      status.setStatus("Pre-loading table descriptors");
668      this.tableDescriptors.getAll();
669    }
670
671    // publish cluster ID
672    status.setStatus("Publishing Cluster ID in ZooKeeper");
673    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
674    this.serverManager = createServerManager(this, this);
675
676    // Invalidate all write locks held previously
677    this.tableLockManager.reapWriteLocks();
678    this.tableStateManager = new TableStateManager(this);
679
680    status.setStatus("Initializing ZK system trackers");
681    initializeZKBasedSystemTrackers();
682
683    // initialize master side coprocessors before we start handling requests
684    status.setStatus("Initializing master coprocessors");
685    this.cpHost = new MasterCoprocessorHost(this, this.conf);
686
687    // start up all service threads.
688    status.setStatus("Initializing master service threads");
689    startServiceThreads();
690
691    // Wake up this server to check in
692    sleeper.skipSleepCycle();
693
694    // Wait for region servers to report in
695    this.serverManager.waitForRegionServers(status);
696    // Check zk for region servers that are up but didn't register
697    for (ServerName sn: this.regionServerTracker.getOnlineServers()) {
698      // The isServerOnline check is opportunistic, correctness is handled inside
699      if (!this.serverManager.isServerOnline(sn)
700          && serverManager.checkAndRecordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD)) {
701        LOG.info("Registered server found up in zk but who has not yet reported in: " + sn);
702      }
703    }
704
705    // get a list for previously failed RS which need log splitting work
706    // we recover hbase:meta region servers inside master initialization and
707    // handle other failed servers in SSH in order to start up master node ASAP
708    Set<ServerName> previouslyFailedServers =
709      this.fileSystemManager.getFailedServersFromLogFolders();
710
711    // log splitting for hbase:meta server
712    ServerName oldMetaServerLocation = metaTableLocator.getMetaRegionLocation(this.getZooKeeper());
713    if (oldMetaServerLocation != null && previouslyFailedServers.contains(oldMetaServerLocation)) {
714      splitMetaLogBeforeAssignment(oldMetaServerLocation);
715      // Note: we can't remove oldMetaServerLocation from previousFailedServers list because it
716      // may also host user regions
717    }
718    Set<ServerName> previouslyFailedMetaRSs = getPreviouselyFailedMetaServersFromZK();
719    // need to use union of previouslyFailedMetaRSs recorded in ZK and previouslyFailedServers
720    // instead of previouslyFailedMetaRSs alone to address the following two situations:
721    // 1) the chained failure situation(recovery failed multiple times in a row).
722    // 2) master get killed right before it could delete the recovering hbase:meta from ZK while the
723    // same server still has non-meta wals to be replayed so that
724    // removeStaleRecoveringRegionsFromZK can't delete the stale hbase:meta region
725    // Passing more servers into splitMetaLog is all right. If a server doesn't have hbase:meta wal,
726    // there is no op for the server.
727    previouslyFailedMetaRSs.addAll(previouslyFailedServers);
728
729    this.initializationBeforeMetaAssignment = true;
730
731    // Wait for regionserver to finish initialization.
732    if (BaseLoadBalancer.tablesOnMaster(conf)) {
733      waitForServerOnline();
734    }
735
736    //initialize load balancer
737    this.balancer.setClusterStatus(getClusterStatus());
738    this.balancer.setMasterServices(this);
739    this.balancer.initialize();
740
741    // Check if master is shutting down because of some issue
742    // in initializing the regionserver or the balancer.
743    if (isStopped()) return;
744
745    // Make sure meta assigned before proceeding.
746    status.setStatus("Assigning Meta Region");
747    assignMeta(status, previouslyFailedMetaRSs, HRegionInfo.DEFAULT_REPLICA_ID);
748    // check if master is shutting down because above assignMeta could return even hbase:meta isn't
749    // assigned when master is shutting down
750    if (isStopped()) return;
751
752    // migrating existent table state from zk, so splitters
753    // and recovery process treat states properly.
754    for (Map.Entry<TableName, TableState.State> entry : ZKDataMigrator
755        .queryForTableStates(getZooKeeper()).entrySet()) {
756      LOG.info("Converting state from zk to new states:" + entry);
757      tableStateManager.setTableState(entry.getKey(), entry.getValue());
758    }
759    ZKUtil.deleteChildrenRecursively(getZooKeeper(), getZooKeeper().tableZNode);
760
761    status.setStatus("Submitting log splitting work for previously failed region servers");
762    // Master has recovered hbase:meta region server and we put
763    // other failed region servers in a queue to be handled later by SSH
764    for (ServerName tmpServer : previouslyFailedServers) {
765      this.serverManager.processDeadServer(tmpServer, true);
766    }
767
768    // Fix up assignment manager status
769    status.setStatus("Starting assignment manager");
770    this.assignmentManager.joinCluster();
771
772    // set cluster status again after user regions are assigned
773    this.balancer.setClusterStatus(getClusterStatus());
774
775    // Start balancer and meta catalog janitor after meta and regions have been assigned.
776    status.setStatus("Starting balancer and catalog janitor");
777    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
778    getChoreService().scheduleChore(clusterStatusChore);
779    this.balancerChore = new BalancerChore(this);
780    getChoreService().scheduleChore(balancerChore);
781    this.normalizerChore = new RegionNormalizerChore(this);
782    getChoreService().scheduleChore(normalizerChore);
783    this.catalogJanitorChore = new CatalogJanitor(this, this);
784    getChoreService().scheduleChore(catalogJanitorChore);
785
786    status.setStatus("Starting cluster schema service");
787    initClusterSchemaService();
788
789    if (this.cpHost != null) {
790      try {
791        this.cpHost.preMasterInitialization();
792      } catch (IOException e) {
793        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
794      }
795    }
796
797    status.markComplete("Initialization successful");
798    LOG.info("Master has completed initialization");
799    configurationManager.registerObserver(this.balancer);
800
801    // Set master as 'initialized'.
802    setInitialized(true);
803
804    // assign the meta replicas
805    Set<ServerName> EMPTY_SET = new HashSet<ServerName>();
806    int numReplicas = conf.getInt(HConstants.META_REPLICAS_NUM,
807           HConstants.DEFAULT_META_REPLICA_NUM);
808    for (int i = 1; i < numReplicas; i++) {
809      assignMeta(status, EMPTY_SET, i);
810    }
811    unassignExcessMetaReplica(zooKeeper, numReplicas);
812
813    status.setStatus("Starting quota manager");
814    initQuotaManager();
815
816    // clear the dead servers with same host name and port of online server because we are not
817    // removing dead server with same hostname and port of rs which is trying to check in before
818    // master initialization. See HBASE-5916.
819    this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
820
821    // Check and set the znode ACLs if needed in case we are overtaking a non-secure configuration
822    status.setStatus("Checking ZNode ACLs");
823    zooKeeper.checkAndSetZNodeAcls();
824
825    status.setStatus("Calling postStartMaster coprocessors");
826
827    this.expiredMobFileCleanerChore = new ExpiredMobFileCleanerChore(this);
828    getChoreService().scheduleChore(expiredMobFileCleanerChore);
829
830    int mobCompactionPeriod = conf.getInt(MobConstants.MOB_COMPACTION_CHORE_PERIOD,
831      MobConstants.DEFAULT_MOB_COMPACTION_CHORE_PERIOD);
832    if (mobCompactionPeriod > 0) {
833      this.mobCompactChore = new MobCompactionChore(this, mobCompactionPeriod);
834      getChoreService().scheduleChore(mobCompactChore);
835    } else {
836      LOG
837        .info("The period is " + mobCompactionPeriod + " seconds, MobCompactionChore is disabled");
838    }
839    this.mobCompactThread = new MasterMobCompactionThread(this);
840
841    if (this.cpHost != null) {
842      // don't let cp initialization errors kill the master
843      try {
844        this.cpHost.postStartMaster();
845      } catch (IOException ioe) {
846        LOG.error("Coprocessor postStartMaster() hook failed", ioe);
847      }
848    }
849
850    zombieDetector.interrupt();
851  }

865  private void unassignExcessMetaReplica(ZooKeeperWatcher zkw, int numMetaReplicasConfigured) {
866    // unassign the unneeded replicas (for e.g., if the previous master was configured
867    // with a replication of 3 and now it is 2, we need to unassign the 1 unneeded replica)
868    try {
869      List<String> metaReplicaZnodes = zooKeeper.getMetaReplicaNodes();
870      for (String metaReplicaZnode : metaReplicaZnodes) {
871        int replicaId = zooKeeper.getMetaReplicaIdFromZnode(metaReplicaZnode);
872        if (replicaId >= numMetaReplicasConfigured) {
873          RegionState r = MetaTableLocator.getMetaRegionState(zkw, replicaId);
874          LOG.info("Closing excess replica of meta region " + r.getRegion());
875          // send a close and wait for a max of 30 seconds
876          ServerManager.closeRegionSilentlyAndWait(getClusterConnection(), r.getServerName(),
877              r.getRegion(), 30000);
878          ZKUtil.deleteNode(zkw, zkw.getZNodeForReplica(replicaId));
879        }
880      }
881    } catch (Exception ex) {
882      // ignore the exception since we don't want the master to be wedged due to potential
883      // issues in the cleanup of the extra regions. We can do that cleanup via hbck or manually
884      LOG.warn("Ignoring exception " + ex);
885    }
886  }

888  /**
889   * Check <code>hbase:meta</code> is assigned. If not, assign it.
890   */
891  void assignMeta(MonitoredTask status, Set<ServerName> previouslyFailedMetaRSs, int replicaId)
892      throws InterruptedException, IOException, KeeperException {
893    // Work on meta region
894    int assigned = 0;
895    long timeout = this.conf.getLong("hbase.catalog.verification.timeout", 1000);
896    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
897      status.setStatus("Assigning hbase:meta region");
898    } else {
899      status.setStatus("Assigning hbase:meta region, replicaId " + replicaId);
900    }
901
902    // Get current meta state from zk.
903    RegionState metaState = MetaTableLocator.getMetaRegionState(getZooKeeper(), replicaId);
904    HRegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO,
905        replicaId);
906    RegionStates regionStates = assignmentManager.getRegionStates();
907    regionStates.createRegionState(hri, metaState.getState(),
908        metaState.getServerName(), null);
909
910    if (!metaState.isOpened() || !metaTableLocator.verifyMetaRegionLocation(
911        this.getClusterConnection(), this.getZooKeeper(), timeout, replicaId)) {
912      ServerName currentMetaServer = metaState.getServerName();
913      if (serverManager.isServerOnline(currentMetaServer)) {
914        if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
915          LOG.info("Meta was in transition on " + currentMetaServer);
916        } else {
917          LOG.info("Meta with replicaId " + replicaId + " was in transition on " +
918                    currentMetaServer);
919        }
920        assignmentManager.processRegionsInTransition(Arrays.asList(metaState));
921      } else {
922        if (currentMetaServer != null) {
923          if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
924            splitMetaLogBeforeAssignment(currentMetaServer);
925            regionStates.logSplit(HRegionInfo.FIRST_META_REGIONINFO);
926            previouslyFailedMetaRSs.add(currentMetaServer);
927          }
928        }
929        LOG.info("Re-assigning hbase:meta with replicaId, " + replicaId +
930            " it was on " + currentMetaServer);
931        assignmentManager.assignMeta(hri);
932      }
933      assigned++;
934    }
935
936    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID)
937      getTableStateManager().setTableState(TableName.META_TABLE_NAME, TableState.State.ENABLED);
938    // TODO: should we prevent from using state manager before meta was initialized?
939    // tableStateManager.start();
940
941    if ((RecoveryMode.LOG_REPLAY == this.getMasterFileSystem().getLogRecoveryMode())
942        && (!previouslyFailedMetaRSs.isEmpty())) {
943      // replay WAL edits mode need new hbase:meta RS is assigned firstly
944      status.setStatus("replaying log for Meta Region");
945      this.fileSystemManager.splitMetaLog(previouslyFailedMetaRSs);
946    }
947
948    this.assignmentManager.setEnabledTable(TableName.META_TABLE_NAME);
949    tableStateManager.start();
950
951    // Make sure a hbase:meta location is set. We need to enable SSH here since
952    // if the meta region server is died at this time, we need it to be re-assigned
953    // by SSH so that system tables can be assigned.
954    // No need to wait for meta is assigned = 0 when meta is just verified.
955    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableCrashedServerProcessing(assigned != 0);
956    LOG.info("hbase:meta with replicaId " + replicaId + " assigned=" + assigned + ", location="
957      + metaTableLocator.getMetaRegionLocation(this.getZooKeeper(), replicaId));
958    status.setStatus("META assigned.");
959  }

992  throws IOException, InterruptedException {

1054   this.service.startExecutorService(ExecutorType.MASTER_OPEN_REGION,
1055      conf.getInt("hbase.master.executor.openregion.threads", 5));
1056   this.service.startExecutorService(ExecutorType.MASTER_CLOSE_REGION,
1057      conf.getInt("hbase.master.executor.closeregion.threads", 5));
1058   this.service.startExecutorService(ExecutorType.MASTER_SERVER_OPERATIONS,
1059      conf.getInt("hbase.master.executor.serverops.threads", 5));
1060   this.service.startExecutorService(ExecutorType.MASTER_META_SERVER_OPERATIONS,
1061      conf.getInt("hbase.master.executor.meta.serverops.threads", 5));
1062   this.service.startExecutorService(ExecutorType.M_LOG_REPLAY_OPS,
1063      conf.getInt("hbase.master.executor.logreplayops.threads", 10));
1064
1065   // We depend on there being only one instance of this executor running
1066   // at a time.  To do concurrency, would need fencing of enable/disable of
1067   // tables.
1068   // Any time changing this maxThreads to > 1, pls see the comment at
1069   // AccessController#postCreateTableHandler
1070   this.service.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
1071   startProcedureExecutor();
1072
1073   // Start log cleaner thread
1074   int cleanerInterval = conf.getInt("hbase.master.cleaner.interval", 60 * 1000);
1075   this.logCleaner =
1076      new LogCleaner(cleanerInterval,
1077         this, conf, getMasterFileSystem().getFileSystem(),
1078         getMasterFileSystem().getOldLogDir());
1079    getChoreService().scheduleChore(logCleaner);
1080
1081   //start the hfile archive cleaner thread
1082    Path archiveDir = HFileArchiveUtil.getArchivePath(conf);
1083    this.hfileCleaner = new HFileCleaner(cleanerInterval, this, conf, getMasterFileSystem()
1084        .getFileSystem(), archiveDir);
1085    getChoreService().scheduleChore(hfileCleaner);
1086    serviceStarted = true;
1087    if (LOG.isTraceEnabled()) {
1088      LOG.trace("Started service threads");
1089    }
1090  }

1226  public boolean balance(boolean force) throws IOException {
1227    // if master not initialized, don't run balancer.
1228    if (!isInitialized()) {
1229      LOG.debug("Master has not been initialized, don't run balancer.");
1230      return false;
1231    }
1232    // Do this call outside of synchronized block.
1233    int maximumBalanceTime = getBalancerCutoffTime();
1234    synchronized (this.balancer) {
1235      // If balance not true, don't run balancer.
1236      if (!this.loadBalancerTracker.isBalancerOn()) return false;
1237      // Only allow one balance run at at time.
1238      if (this.assignmentManager.getRegionStates().isRegionsInTransition()) {
1239        Map<String, RegionState> regionsInTransition =
1240          this.assignmentManager.getRegionStates().getRegionsInTransition();
1241        // if hbase:meta region is in transition, result of assignment cannot be recorded
1242        // ignore the force flag in that case
1243        boolean metaInTransition = assignmentManager.getRegionStates().isMetaRegionInTransition();
1244        String prefix = force && !metaInTransition ? "R" : "Not r";
1245        LOG.debug(prefix + "unning balancer because " + regionsInTransition.size() +
1246          " region(s) in transition: " + org.apache.commons.lang.StringUtils.
1247            abbreviate(regionsInTransition.toString(), 256));
1248        if (!force || metaInTransition) return false;
1249      }
1250      if (this.serverManager.areDeadServersInProgress()) {
1251        LOG.debug("Not running balancer because processing dead regionserver(s): " +
1252          this.serverManager.getDeadServers());
1253        return false;
1254      }
1255
1256      if (this.cpHost != null) {
1257        try {
1258          if (this.cpHost.preBalance()) {
1259            LOG.debug("Coprocessor bypassing balancer request");
1260            return false;
1261          }
1262        } catch (IOException ioe) {
1263          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1264          return false;
1265        }
1266      }
1267
1268      Map<TableName, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1269        this.assignmentManager.getRegionStates().getAssignmentsByTable();
1270
1271      List<RegionPlan> plans = new ArrayList<RegionPlan>();
1272
1273      //Give the balancer the current cluster state.
1274      this.balancer.setClusterStatus(getClusterStatus());
1275      for (Entry<TableName, Map<ServerName, List<HRegionInfo>>> e : assignmentsByTable.entrySet()) {
1276        List<RegionPlan> partialPlans = this.balancer.balanceCluster(e.getKey(), e.getValue());
1277        if (partialPlans != null) plans.addAll(partialPlans);
1278      }
1279
1280      long cutoffTime = System.currentTimeMillis() + maximumBalanceTime;
1281      int rpCount = 0;  // number of RegionPlans balanced so far
1282      long totalRegPlanExecTime = 0;
1283      if (plans != null && !plans.isEmpty()) {
1284        for (RegionPlan plan: plans) {
1285          LOG.info("balance " + plan);
1286          long balStartTime = System.currentTimeMillis();
1287          //TODO: bulk assign
1288          this.assignmentManager.balance(plan);
1289          totalRegPlanExecTime += System.currentTimeMillis()-balStartTime;
1290          rpCount++;
1291          if (rpCount < plans.size() &&
1292              // if performing next balance exceeds cutoff time, exit the loop
1293              (System.currentTimeMillis() + (totalRegPlanExecTime / rpCount)) > cutoffTime) {
1294            //TODO: After balance, there should not be a cutoff time (keeping it as
1295            // a security net for now)
1296            LOG.debug("No more balancing till next balance run; maximumBalanceTime=" +
1297              maximumBalanceTime);
1298            break;
1299          }
1300        }
1301      }
1302      if (this.cpHost != null) {
1303        try {
1304          this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1305        } catch (IOException ioe) {
1306          // balancing already succeeded so don't change the result
1307          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1308        }
1309      }
1310    }
1311    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1312    // Return true indicating a success.
1313    return true;
1314  }

1322   * Perform normalization of cluster (invoked by {@link RegionNormalizerChore}).
1323   *
1324   * @return true if normalization step was performed successfully, false otherwise
1325   *    (specifically, if HMaster hasn't been initialized properly or normalization
1326   *    is globally disabled)
1327   */
1328  public boolean normalizeRegions() throws IOException {
1329    if (!isInitialized()) {
1330      LOG.debug("Master has not been initialized, don't run region normalizer.");
1331      return false;
1332    }
1333
1334    if (!this.regionNormalizerTracker.isNormalizerOn()) {
1335      LOG.debug("Region normalization is disabled, don't run region normalizer.");
1336      return false;
1337    }
1338
1339    synchronized (this.normalizer) {
1340      // Don't run the normalizer concurrently
1341      List<TableName> allEnabledTables = new ArrayList<>(
1342        this.tableStateManager.getTablesInStates(TableState.State.ENABLED));
1343
1344      Collections.shuffle(allEnabledTables);
1345
1346      for (TableName table : allEnabledTables) {
1347        TableDescriptor tblDesc = getTableDescriptors().getDescriptor(table);
1348        if (table.isSystemTable() || (tblDesc != null &&
1349            tblDesc.getHTableDescriptor() != null &&
1350            !tblDesc.getHTableDescriptor().isNormalizationEnabled())) {
1351          LOG.debug("Skipping normalization for table: " + table + ", as it's either system"
1352              + " table or doesn't have auto normalization turned on");
1353          continue;
1354        }
1355        List<NormalizationPlan> plans = this.normalizer.computePlanForTable(table);
1356        if (plans != null) {
1357          for (NormalizationPlan plan : plans) {
1358            plan.execute(clusterConnection.getAdmin());
1359            if (plan.getType() == PlanType.SPLIT) {
1360              splitPlanCount++;
1361            } else if (plan.getType() == PlanType.MERGE) {
1362              mergePlanCount++;
1363            }
1364          }
1365        }
1366      }
1367    }
1368    // If Region did not generate any plans, it means the cluster is already balanced.
1369    // Return true indicating a success.
1370    return true;

1398  void move(final byte[] encodedRegionName,
1399      final byte[] destServerName) throws HBaseIOException {
1400    RegionState regionState = assignmentManager.getRegionStates().
1401      getRegionState(Bytes.toString(encodedRegionName));
1402
1403    HRegionInfo hri;
1404    if (regionState != null) {
1405      hri = regionState.getRegion();
1406    } else {
1407      throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1408    }
1409
1410    ServerName dest;
1411    if (destServerName == null || destServerName.length == 0) {
1412      LOG.info("Passed destination servername is null/empty so " +
1413        "choosing a server at random");
1414      final List<ServerName> destServers = this.serverManager.createDestinationServersList(
1415        regionState.getServerName());
1416      dest = balancer.randomAssignment(hri, destServers);
1417      if (dest == null) {
1418        LOG.debug("Unable to determine a plan to assign " + hri);
1419        return;
1420      }
1421    } else {
1422      ServerName candidate = ServerName.valueOf(Bytes.toString(destServerName));
1423      dest = balancer.randomAssignment(hri, Lists.newArrayList(candidate));
1424      if (dest == null) {
1425        LOG.debug("Unable to determine a plan to assign " + hri);
1426        return;
1427      }
1428      if (dest.equals(serverName) && balancer instanceof BaseLoadBalancer
1429          && !((BaseLoadBalancer)balancer).shouldBeOnMaster(hri)) {
1430        // To avoid unnecessary region moving later by balancer. Don't put user
1431        // regions on master. Regions on master could be put on other region
1432        // server intentionally by test however.
1433        LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1434          + " to avoid unnecessary region moving later by load balancer,"
1435          + " because it should not be on master");
1436        return;
1437      }
1438    }
1439
1440    if (dest.equals(regionState.getServerName())) {
1441      LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1442        + " because region already assigned to the same server " + dest + ".");
1443      return;
1444    }
1445
1446    // Now we can do the move
1447    RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1448
1449    try {
1450      checkInitialized();
1451      if (this.cpHost != null) {
1452        if (this.cpHost.preMove(hri, rp.getSource(), rp.getDestination())) {
1453          return;
1454        }
1455      }
1456      // warmup the region on the destination before initiating the move. this call
1457      // is synchronous and takes some time. doing it before the source region gets
1458      // closed
1459      serverManager.sendRegionWarmup(rp.getDestination(), hri);
1460
1461      LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
1462      this.assignmentManager.balance(rp);
1463      if (this.cpHost != null) {
1464        this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
1465      }
1466    } catch (IOException ioe) {
1467      if (ioe instanceof HBaseIOException) {
1468        throw (HBaseIOException)ioe;
1469      }
1470      throw new HBaseIOException(ioe);
1471    }
1472  }

1512  /**
1513   * Checks whether the table conforms to some sane limits, and configured
1514   * values (compression, etc) work. Throws an exception if something is wrong.
1515   * @throws IOException
1516   */
1517  private void sanityCheckTableDescriptor(final HTableDescriptor htd) throws IOException {
1518    final String CONF_KEY = "hbase.table.sanity.checks";
1519    boolean logWarn = false;
1520    if (!conf.getBoolean(CONF_KEY, true)) {
1521      logWarn = true;
1522    }
1523    String tableVal = htd.getConfigurationValue(CONF_KEY);
1524    if (tableVal != null && !Boolean.valueOf(tableVal)) {
1525      logWarn = true;
1526    }
1527
1528    // check max file size
1529    long maxFileSizeLowerLimit = 2 * 1024 * 1024L; // 2M is the default lower limit
1530    long maxFileSize = htd.getMaxFileSize();
1531    if (maxFileSize < 0) {
1532      maxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE, maxFileSizeLowerLimit);
1533    }
1534    if (maxFileSize < conf.getLong("hbase.hregion.max.filesize.limit", maxFileSizeLowerLimit)) {
1535      String message = "MAX_FILESIZE for table descriptor or "
1536          + "\"hbase.hregion.max.filesize\" (" + maxFileSize
1537          + ") is too small, which might cause over splitting into unmanageable "
1538          + "number of regions.";
1539      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1540    }
1541
1542    // check flush size
1543    long flushSizeLowerLimit = 1024 * 1024L; // 1M is the default lower limit
1544    long flushSize = htd.getMemStoreFlushSize();
1545    if (flushSize < 0) {
1546      flushSize = conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSizeLowerLimit);
1547    }
1548    if (flushSize < conf.getLong("hbase.hregion.memstore.flush.size.limit", flushSizeLowerLimit)) {
1549      String message = "MEMSTORE_FLUSHSIZE for table descriptor or "
1550          + "\"hbase.hregion.memstore.flush.size\" ("+flushSize+") is too small, which might cause"
1551          + " very frequent flushing.";
1552      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1553    }
1554
1555    // check that coprocessors and other specified plugin classes can be loaded
1556    try {
1557      checkClassLoading(conf, htd);
1558    } catch (Exception ex) {
1559      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, ex.getMessage(), null);
1560    }
1561
1562    // check compression can be loaded
1563    try {
1564      checkCompression(htd);
1565    } catch (IOException e) {
1566      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1567    }
1568
1569    // check encryption can be loaded
1570    try {
1571      checkEncryption(conf, htd);
1572    } catch (IOException e) {
1573      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1574    }
1575    // Verify compaction policy
1576    try{
1577      checkCompactionPolicy(conf, htd);
1578    } catch(IOException e){
1579      warnOrThrowExceptionForFailure(false, CONF_KEY, e.getMessage(), e);
1580    }
1581    // check that we have at least 1 CF
1582    if (htd.getColumnFamilies().length == 0) {
1583      String message = "Table should have at least one column family.";
1584      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1585    }
1586
1587    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1588      if (hcd.getTimeToLive() <= 0) {
1589        String message = "TTL for column family " + hcd.getNameAsString() + " must be positive.";
1590        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1591      }
1592
1593      // check blockSize
1594      if (hcd.getBlocksize() < 1024 || hcd.getBlocksize() > 16 * 1024 * 1024) {
1595        String message = "Block size for column family " + hcd.getNameAsString()
1596            + "  must be between 1K and 16MB.";
1597        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1598      }
1599
1600      // check versions
1601      if (hcd.getMinVersions() < 0) {
1602        String message = "Min versions for column family " + hcd.getNameAsString()
1603          + "  must be positive.";
1604        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1605      }
1606      // max versions already being checked
1607
1608      // HBASE-13776 Setting illegal versions for HColumnDescriptor
1609      //  does not throw IllegalArgumentException
1610      // check minVersions <= maxVerions
1611      if (hcd.getMinVersions() > hcd.getMaxVersions()) {
1612        String message = "Min versions for column family " + hcd.getNameAsString()
1613            + " must be less than the Max versions.";
1614        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1615      }
1616
1617      // check replication scope
1618      if (hcd.getScope() < 0) {
1619        String message = "Replication scope for column family "
1620          + hcd.getNameAsString() + "  must be positive.";
1621        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1622      }
1623
1624      // check data replication factor, it can be 0(default value) when user has not explicitly
1625      // set the value, in this case we use default replication factor set in the file system.
1626      if (hcd.getDFSReplication() < 0) {
1627        String message = "HFile Replication for column family " + hcd.getNameAsString()
1628            + "  must be greater than zero.";
1629        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1630      }
1631
1632      // TODO: should we check coprocessors and encryption ?
1633    }
1634  }

1636  private void checkCompactionPolicy(Configuration conf, HTableDescriptor htd)
1637      throws IOException {
1638    // FIFO compaction has some requirements
1639    // Actually FCP ignores periodic major compactions
1640    String className =
1641        htd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1642    if (className == null) {
1643      className =
1644          conf.get(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
1645            ExploringCompactionPolicy.class.getName());
1646    }
1647
1648    int blockingFileCount = HStore.DEFAULT_BLOCKING_STOREFILE_COUNT;
1649    String sv = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1650    if (sv != null) {
1651      blockingFileCount = Integer.parseInt(sv);
1652    } else {
1653      blockingFileCount = conf.getInt(HStore.BLOCKING_STOREFILES_KEY, blockingFileCount);
1654    }
1655
1656    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1657      String compactionPolicy =
1658          hcd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1659      if (compactionPolicy == null) {
1660        compactionPolicy = className;
1661      }
1662      if (!compactionPolicy.equals(FIFOCompactionPolicy.class.getName())) {
1663        continue;
1664      }
1665      // FIFOCompaction
1666      String message = null;
1667
1668      // 1. Check TTL
1669      if (hcd.getTimeToLive() == HColumnDescriptor.DEFAULT_TTL) {
1670        message = "Default TTL is not supported for FIFO compaction";
1671        throw new IOException(message);
1672      }
1673
1674      // 2. Check min versions
1675      if (hcd.getMinVersions() > 0) {
1676        message = "MIN_VERSION > 0 is not supported for FIFO compaction";
1677        throw new IOException(message);
1678      }
1679
1680      // 3. blocking file count
1681      String sbfc = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1682      if (sbfc != null) {
1683        blockingFileCount = Integer.parseInt(sbfc);
1684      }
1685      if (blockingFileCount < 1000) {
1686        message =
1687            "blocking file count '" + HStore.BLOCKING_STOREFILES_KEY + "' " + blockingFileCount
1688                + " is below recommended minimum of 1000";
1689        throw new IOException(message);
1690      }
1691    }
1692  }

1704  private void startActiveMasterManager(int infoPort) throws KeeperException {
1705    String backupZNode = ZKUtil.joinZNode(
1706      zooKeeper.backupMasterAddressesZNode, serverName.toString());
1707    /*
1708    * Add a ZNode for ourselves in the backup master directory since we
1709    * may not become the active master. If so, we want the actual active
1710    * master to know we are backup masters, so that it won't assign
1711    * regions to us if so configured.
1712    *
1713    * If we become the active master later, ActiveMasterManager will delete
1714    * this node explicitly.  If we crash before then, ZooKeeper will delete
1715    * this node for us since it is ephemeral.
1716    */
1717    LOG.info("Adding backup master ZNode " + backupZNode);
1718    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode,
1719        serverName, infoPort)) {
1720      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
1721    }
1722
1723    activeMasterManager.setInfoPort(infoPort);
1724    // Start a thread to try to become the active master, so we won't block here
1725    Threads.setDaemonThreadRunning(new Thread(new Runnable() {
1726      @Override
1727      public void run() {
1728        int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
1729          HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
1730        // If we're a backup master, stall until a primary to writes his address
1731        if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP,
1732          HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
1733          LOG.debug("HMaster started in backup mode. "
1734            + "Stalling until master znode is written.");
1735          // This will only be a minute or so while the cluster starts up,
1736          // so don't worry about setting watches on the parent znode
1737          while (!activeMasterManager.hasActiveMaster()) {
1738            LOG.debug("Waiting for master address ZNode to be written "
1739              + "(Also watching cluster state node)");
1740            Threads.sleep(timeout);
1741          }
1742        }
1743        MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
1744        status.setDescription("Master startup");
1745        try {
1746          if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
1747            finishActiveMasterInitialization(status);
1748          }
1749        } catch (Throwable t) {
1750          status.setStatus("Failed to become active: " + t.getMessage());
1751          LOG.fatal("Failed to become active master", t);
1752          // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
1753          if (t instanceof NoClassDefFoundError &&
1754            t.getMessage()
1755              .contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
1756            // improved error message for this special case
1757            abort("HBase is having a problem with its Hadoop jars.  You may need to "
1758              + "recompile HBase against Hadoop version "
1759              + org.apache.hadoop.util.VersionInfo.getVersion()
1760              + " or change your hadoop jars to start properly", t);
1761          } else {
1762            abort("Unhandled exception. Starting shutdown.", t);
1763          }
1764        } finally {
1765          status.cleanup();
1766        }
1767      }
1768    }, getServerName().toShortString() + ".activeMasterManager"));
1769  }

2649  /**
2650   * @return list of table table descriptors after filtering by regex and whether to include system
2651   *    tables, etc.
2652   * @throws IOException
2653   */
2654  private List<HTableDescriptor> getTableDescriptors(final List<HTableDescriptor> htds,
2655      final String namespace, final String regex, final List<TableName> tableNameList,
2656      final boolean includeSysTables)
2657  throws IOException {
2658    if (tableNameList == null || tableNameList.size() == 0) {
2659      // request for all TableDescriptors
2660      Collection<HTableDescriptor> allHtds;
2661      if (namespace != null && namespace.length() > 0) {
2662        // Do a check on the namespace existence. Will fail if does not exist.
2663        this.clusterSchemaService.getNamespace(namespace);
2664        allHtds = tableDescriptors.getByNamespace(namespace).values();
2665      } else {
2666        allHtds = tableDescriptors.getAll().values();
2667      }
2668      for (HTableDescriptor desc: allHtds) {
2669        if (tableStateManager.isTablePresent(desc.getTableName())
2670            && (includeSysTables || !desc.getTableName().isSystemTable())) {
2671          htds.add(desc);
2672        }
2673      }
2674    } else {
2675      for (TableName s: tableNameList) {
2676        if (tableStateManager.isTablePresent(s)) {
2677          HTableDescriptor desc = tableDescriptors.get(s);
2678          if (desc != null) {
2679            htds.add(desc);
2680          }
2681        }
2682      }
2683    }
2684
2685    // Retains only those matched by regular expression.
2686    if (regex != null) filterTablesByRegex(htds, Pattern.compile(regex));
2687    return htds;
2688  }

2690  /**
2691   * Removes the table descriptors that don't match the pattern.
2692   * @param descriptors list of table descriptors to filter
2693   * @param pattern the regex to use
2694   */
2695  private static void filterTablesByRegex(final Collection<HTableDescriptor> descriptors,
2696      final Pattern pattern) {
2697    final String defaultNS = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
2698    Iterator<HTableDescriptor> itr = descriptors.iterator();
2699    while (itr.hasNext()) {
2700      HTableDescriptor htd = itr.next();
2701      String tableName = htd.getTableName().getNameAsString();
2702      boolean matched = pattern.matcher(tableName).matches();
2703      if (!matched && htd.getTableName().getNamespaceAsString().equals(defaultNS)) {
2704        matched = pattern.matcher(defaultNS + TableName.NAMESPACE_DELIM + tableName).matches();
2705      }
2706      if (!matched) {
2707        itr.remove();
2708      }
2709    }
2710  }

2739  public void reportMobCompactionStart(TableName tableName) throws IOException {
2740    IdLock.Entry lockEntry = null;
2741    try {
2742      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2743      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2744      if (compactionsCount == null) {
2745        compactionsCount = new AtomicInteger(0);
2746        mobCompactionStates.put(tableName, compactionsCount);
2747      }
2748      compactionsCount.incrementAndGet();
2749    } finally {
2750      if (lockEntry != null) {
2751        mobCompactionLock.releaseLockEntry(lockEntry);
2752      }
2753    }
2754  }

2756  public void reportMobCompactionEnd(TableName tableName) throws IOException {
2757    IdLock.Entry lockEntry = null;
2758    try {
2759      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2760      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2761      if (compactionsCount != null) {
2762        int count = compactionsCount.decrementAndGet();
2763        // remove the entry if the count is 0.
2764        if (count == 0) {
2765          mobCompactionStates.remove(tableName);
2766        }
2767      }
2768    } finally {
2769      if (lockEntry != null) {
2770        mobCompactionLock.releaseLockEntry(lockEntry);
2771      }
2772    }
2773  }
