179/**
180 * HMaster is the "master server" for HBase. An HBase cluster has one active
181 * master.  If many masters are started, all compete.  Whichever wins goes on to
182 * run the cluster.  All others park themselves in their constructor until
183 * master or cluster shutdown or until the active master loses its lease in
184 * zookeeper.  Thereafter, all running master jostle to take over master role.
185 *
186 * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
187 * this case it will tell all regionservers to go down and then wait on them
188 * all reporting in that they are down.  This master will then shut itself down.
189 *
190 * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
191 *
192 * @see org.apache.zookeeper.Watcher
193 */
194@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
195@SuppressWarnings("deprecation")

199  /**
200   * Protection against zombie master. Started once Master accepts active responsibility and
201   * starts taking over responsibilities. Allows a finite time window before giving up ownership.
202   */
203  private static class InitializationMonitor extends HasThread {
204    /** The amount of time in milliseconds to sleep before checking initialization status. */
205    public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
206    public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
207
208    /**
209     * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
210     * true, do nothing otherwise.
211     */
212    public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
213    public static final boolean HALT_DEFAULT = false;
214
215    private final HMaster master;
216    private final long timeout;
217    private final boolean haltOnTimeout;
218
219    /** Creates a Thread that monitors the {@link #isInitialized()} state. */
220    InitializationMonitor(HMaster master) {
221      super("MasterInitializationMonitor");
222      this.master = master;
223      this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
224      this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
225      this.setDaemon(true);
226    }
227
228    @Override
229    public void run() {
230      try {
231        while (!master.isStopped() && master.isActiveMaster()) {
232          Thread.sleep(timeout);
233          if (master.isInitialized()) {
234            LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
235          } else {
236            LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
237                + " consider submitting a bug report including a thread dump of this process.");
238            if (haltOnTimeout) {
239              LOG.error("Zombie Master exiting. Thread dump to stdout");
240              Threads.printThreadInfo(System.out, "Zombie HMaster");
241              System.exit(-1);
242            }
243          }
244        }
245      } catch (InterruptedException ie) {
246        LOG.trace("InitMonitor thread interrupted. Existing.");
247      }
248    }
249  }

394  /**
395   * Initializes the HMaster. The steps are as follows:
396   * <p>
397   * <ol>
398   * <li>Initialize the local HRegionServer
399   * <li>Start the ActiveMasterManager.
400   * </ol>
401   * <p>
402   * Remaining steps of initialization occur in
403   * #finishActiveMasterInitialization(MonitoredTask) after
404   * the master becomes the active one.
405   */
406  public HMaster(final Configuration conf, CoordinatedStateManager csm)
407      throws IOException, KeeperException {
408    super(conf, csm);
409    this.rsFatals = new MemoryBoundedLogMessageBuffer(
410      conf.getLong("hbase.master.buffer.for.rs.fatals", 1*1024*1024));
411
412    LOG.info("hbase.rootdir=" + FSUtils.getRootDir(this.conf) +
413      ", hbase.cluster.distributed=" + this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
414
415    // Disable usage of meta replicas in the master
416    this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
417
418    Replication.decorateMasterConfiguration(this.conf);
419
420    // Hack! Maps DFSClient => Master for logs.  HDFS made this
421    // config param for task trackers, but we can piggyback off of it.
422    if (this.conf.get("mapreduce.task.attempt.id") == null) {
423      this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
424    }
425
426    // should we check the compression codec type at master side, default true, HBASE-6370
427    this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
428
429    // should we check encryption settings at master side, default true
430    this.masterCheckEncryption = conf.getBoolean("hbase.master.check.encryption", true);
431
432    this.metricsMaster = new MetricsMaster(new MetricsMasterWrapperImpl(this));
433
434    // preload table descriptor at startup
435    this.preLoadTableDescriptors = conf.getBoolean("hbase.master.preload.tabledescriptors", true);
436
437    // Do we publish the status?
438
439    boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
440        HConstants.STATUS_PUBLISHED_DEFAULT);
441    Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
442        conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
443            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
444            ClusterStatusPublisher.Publisher.class);
445
446    if (shouldPublish) {
447      if (publisherClass == null) {
448        LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
449            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
450            " is not set - not publishing status");
451      } else {
452        clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
453        getChoreService().scheduleChore(clusterStatusPublisherChore);
454      }
455    }
456
457    // Some unit tests don't need a cluster, so no zookeeper at all
458    if (!conf.getBoolean("hbase.testing.nocluster", false)) {
459      setInitLatch(new CountDownLatch(1));
460      activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName, this);
461      int infoPort = putUpJettyServer();
462      startActiveMasterManager(infoPort);
463    } else {
464      activeMasterManager = null;
465    }
466  }

468  // return the actual infoPort, -1 means disable info server.
469  private int putUpJettyServer() throws IOException {
470    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
471      return -1;
472    }
473    int infoPort = conf.getInt("hbase.master.info.port.orig",
474      HConstants.DEFAULT_MASTER_INFOPORT);
475    // -1 is for disabling info server, so no redirecting
476    if (infoPort < 0 || infoServer == null) {
477      return -1;
478    }
479    String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
480    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
481      String msg =
482          "Failed to start redirecting jetty server. Address " + addr
483              + " does not belong to this host. Correct configuration parameter: "
484              + "hbase.master.info.bindAddress";
485      LOG.error(msg);
486      throw new IOException(msg);
487    }
488
489    RedirectServlet.regionServerInfoPort = infoServer.getPort();
490    if(RedirectServlet.regionServerInfoPort == infoPort) {
491      return infoPort;
492    }
493    masterJettyServer = new org.mortbay.jetty.Server();
494    Connector connector = new SelectChannelConnector();
495    connector.setHost(addr);
496    connector.setPort(infoPort);
497    masterJettyServer.addConnector(connector);
498    masterJettyServer.setStopAtShutdown(true);
499    Context context = new Context(masterJettyServer, "/", Context.NO_SESSIONS);
500    context.addServlet(RedirectServlet.class, "/*");
501    try {
502      masterJettyServer.start();
503    } catch (Exception e) {
504      throw new IOException("Failed to start redirecting jetty server", e);
505    }
506    return connector.getLocalPort();
507  }

653  /**
654   * Finish initialization of HMaster after becoming the primary master.
655   *
656   * <ol>
657   * <li>Initialize master components - file system manager, server manager,
658   *     assignment manager, region server tracker, etc</li>
659   * <li>Start necessary service threads - balancer, catalog janior,
660   *     executor services, etc</li>
661   * <li>Set cluster as UP in ZooKeeper</li>
662   * <li>Wait for RegionServers to check-in</li>
663   * <li>Split logs and perform data recovery, if necessary</li>
664   * <li>Ensure assignment of meta/namespace regions<li>
665   * <li>Handle either fresh cluster start or master failover</li>
666   * </ol>
667   */
668  private void finishActiveMasterInitialization(MonitoredTask status)
669      throws IOException, InterruptedException, KeeperException, CoordinatedStateException {
670
671    isActiveMaster = true;
672    Thread zombieDetector = new Thread(new InitializationMonitor(this));
673    zombieDetector.start();
674
675    /*
676     * We are active master now... go initialize components we need to run.
677     * Note, there may be dross in zk from previous runs; it'll get addressed
678     * below after we determine if cluster startup or failover.
679     */
680
681    status.setStatus("Initializing Master file system");
682
683    this.masterActiveTime = System.currentTimeMillis();
684    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
685    this.fileSystemManager = new MasterFileSystem(this);
686    this.walManager = new MasterWalManager(this);
687
688    // enable table descriptors cache
689    this.tableDescriptors.setCacheOn();
690    // set the META's descriptor to the correct replication
691    this.tableDescriptors.get(TableName.META_TABLE_NAME).setRegionReplication(
692        conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM));
693    // warm-up HTDs cache on master initialization
694    if (preLoadTableDescriptors) {
695      status.setStatus("Pre-loading table descriptors");
696      this.tableDescriptors.getAll();
697    }
698
699    // publish cluster ID
700    status.setStatus("Publishing Cluster ID in ZooKeeper");
701    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
702    this.initLatch.countDown();
703
704    this.serverManager = createServerManager(this);
705
706    // Invalidate all write locks held previously
707    this.tableLockManager.reapWriteLocks();
708    this.tableStateManager = new TableStateManager(this);
709
710    status.setStatus("Initializing ZK system trackers");
711    initializeZKBasedSystemTrackers();
712
713    // This is for backwards compatibility
714    // See HBASE-11393
715    status.setStatus("Update TableCFs node in ZNode");
716    TableCFsUpdater tableCFsUpdater = new TableCFsUpdater(zooKeeper,
717            conf, this.clusterConnection);
718    tableCFsUpdater.update();
719
720    // initialize master side coprocessors before we start handling requests
721    status.setStatus("Initializing master coprocessors");
722    this.cpHost = new MasterCoprocessorHost(this, this.conf);
723
724    // start up all service threads.
725    status.setStatus("Initializing master service threads");
726    startServiceThreads();
727
728    // Wake up this server to check in
729    sleeper.skipSleepCycle();
730
731    // Wait for region servers to report in
732    status.setStatus("Wait for region servers to report in");
733    waitForRegionServers(status);
734
735    // get a list for previously failed RS which need log splitting work
736    // we recover hbase:meta region servers inside master initialization and
737    // handle other failed servers in SSH in order to start up master node ASAP
738    MasterMetaBootstrap metaBootstrap = createMetaBootstrap(this, status);
739    metaBootstrap.splitMetaLogsBeforeAssignment();
740
741    this.initializationBeforeMetaAssignment = true;
742
743    // Wait for regionserver to finish initialization.
744    if (BaseLoadBalancer.tablesOnMaster(conf)) {
745      waitForServerOnline();
746    }
747
748    //initialize load balancer
749    this.balancer.setClusterStatus(getClusterStatus());
750    this.balancer.setMasterServices(this);
751    this.balancer.initialize();
752
753    // Check if master is shutting down because of some issue
754    // in initializing the regionserver or the balancer.
755    if (isStopped()) return;
756
757    // Make sure meta assigned before proceeding.
758    status.setStatus("Assigning Meta Region");
759    metaBootstrap.assignMeta();
760
761    // check if master is shutting down because above assignMeta could return even hbase:meta isn't
762    // assigned when master is shutting down
763    if (isStopped()) return;
764
765    // migrating existent table state from zk, so splitters
766    // and recovery process treat states properly.
767    for (Map.Entry<TableName, TableState.State> entry : ZKDataMigrator
768        .queryForTableStates(getZooKeeper()).entrySet()) {
769      LOG.info("Converting state from zk to new states:" + entry);
770      tableStateManager.setTableState(entry.getKey(), entry.getValue());
771    }
772    ZKUtil.deleteChildrenRecursively(getZooKeeper(), getZooKeeper().tableZNode);
773
774    status.setStatus("Submitting log splitting work for previously failed region servers");
775    metaBootstrap.processDeadServers();
776
777    // Fix up assignment manager status
778    status.setStatus("Starting assignment manager");
779    this.assignmentManager.joinCluster();
780
781    // set cluster status again after user regions are assigned
782    this.balancer.setClusterStatus(getClusterStatus());
783
784    // Start balancer and meta catalog janitor after meta and regions have been assigned.
785    status.setStatus("Starting balancer and catalog janitor");
786    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
787    getChoreService().scheduleChore(clusterStatusChore);
788    this.balancerChore = new BalancerChore(this);
789    getChoreService().scheduleChore(balancerChore);
790    this.normalizerChore = new RegionNormalizerChore(this);
791    getChoreService().scheduleChore(normalizerChore);
792    this.catalogJanitorChore = new CatalogJanitor(this, this);
793    getChoreService().scheduleChore(catalogJanitorChore);
794
795    // Do Metrics periodically
796    periodicDoMetricsChore = new PeriodicDoMetrics(msgInterval, this);
797    getChoreService().scheduleChore(periodicDoMetricsChore);
798
799    status.setStatus("Starting cluster schema service");
800    initClusterSchemaService();
801
802    if (this.cpHost != null) {
803      try {
804        this.cpHost.preMasterInitialization();
805      } catch (IOException e) {
806        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
807      }
808    }
809
810    status.markComplete("Initialization successful");
811    LOG.info("Master has completed initialization");
812    configurationManager.registerObserver(this.balancer);
813
814    // Set master as 'initialized'.
815    setInitialized(true);
816
817    status.setStatus("Assign meta replicas");
818    metaBootstrap.assignMetaReplicas();
819
820    status.setStatus("Starting quota manager");
821    initQuotaManager();
822
823    // clear the dead servers with same host name and port of online server because we are not
824    // removing dead server with same hostname and port of rs which is trying to check in before
825    // master initialization. See HBASE-5916.
826    this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
827
828    // Check and set the znode ACLs if needed in case we are overtaking a non-secure configuration
829    status.setStatus("Checking ZNode ACLs");
830    zooKeeper.checkAndSetZNodeAcls();
831
832    status.setStatus("Initializing MOB Cleaner");
833    initMobCleaner();
834
835    status.setStatus("Calling postStartMaster coprocessors");
836    if (this.cpHost != null) {
837      // don't let cp initialization errors kill the master
838      try {
839        this.cpHost.postStartMaster();
840      } catch (IOException ioe) {
841        LOG.error("Coprocessor postStartMaster() hook failed", ioe);
842      }
843    }
844
845    zombieDetector.interrupt();
846  }

946  private void startServiceThreads() throws IOException{
947   // Start the executor service pools
948   this.service.startExecutorService(ExecutorType.MASTER_OPEN_REGION,
949      conf.getInt("hbase.master.executor.openregion.threads", 5));
950   this.service.startExecutorService(ExecutorType.MASTER_CLOSE_REGION,
951      conf.getInt("hbase.master.executor.closeregion.threads", 5));
952   this.service.startExecutorService(ExecutorType.MASTER_SERVER_OPERATIONS,
953      conf.getInt("hbase.master.executor.serverops.threads", 5));
954   this.service.startExecutorService(ExecutorType.MASTER_META_SERVER_OPERATIONS,
955      conf.getInt("hbase.master.executor.meta.serverops.threads", 5));
956   this.service.startExecutorService(ExecutorType.M_LOG_REPLAY_OPS,
957      conf.getInt("hbase.master.executor.logreplayops.threads", 10));
958
959   // We depend on there being only one instance of this executor running
960   // at a time.  To do concurrency, would need fencing of enable/disable of
961   // tables.
962   // Any time changing this maxThreads to > 1, pls see the comment at
963   // AccessController#postCompletedCreateTableAction
964   this.service.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
965   startProcedureExecutor();
966
967   // Start log cleaner thread
968   int cleanerInterval = conf.getInt("hbase.master.cleaner.interval", 60 * 1000);
969   this.logCleaner =
970      new LogCleaner(cleanerInterval,
971         this, conf, getMasterWalManager().getFileSystem(),
972         getMasterWalManager().getOldLogDir());
973    getChoreService().scheduleChore(logCleaner);
974
975   //start the hfile archive cleaner thread
976    Path archiveDir = HFileArchiveUtil.getArchivePath(conf);
977    Map<String, Object> params = new HashMap<String, Object>();
978    params.put(MASTER, this);
979    this.hfileCleaner = new HFileCleaner(cleanerInterval, this, conf, getMasterFileSystem()
980        .getFileSystem(), archiveDir, params);
981    getChoreService().scheduleChore(hfileCleaner);
982    serviceStarted = true;
983    if (LOG.isTraceEnabled()) {
984      LOG.trace("Started service threads");
985    }
986    if (conf.getClass("hbase.region.replica.replication.replicationQueues.class",
987        ReplicationFactory.defaultReplicationQueueClass) == ReplicationQueuesZKImpl.class && !conf
988        .getBoolean(HConstants.ZOOKEEPER_USEMULTI, true)) {
989      try {
990        replicationZKLockCleanerChore = new ReplicationZKLockCleanerChore(this, this,
991            cleanerInterval, this.getZooKeeper(), this.conf);
992        getChoreService().scheduleChore(replicationZKLockCleanerChore);
993      } catch (Exception e) {
994        LOG.error("start replicationZKLockCleanerChore failed", e);
995      }
996    }
997    replicationMetaCleaner = new ReplicationMetaCleaner(this, this, cleanerInterval);
998    getChoreService().scheduleChore(replicationMetaCleaner);
999  }

1142  public boolean balance(boolean force) throws IOException {
1143    // if master not initialized, don't run balancer.
1144    if (!isInitialized()) {
1145      LOG.debug("Master has not been initialized, don't run balancer.");
1146      return false;
1147    }
1148
1149    if (isInMaintenanceMode()) {
1150      LOG.info("Master is in maintenanceMode mode, don't run balancer.");
1151      return false;
1152    }
1153
1154    // Do this call outside of synchronized block.
1155    int maximumBalanceTime = getBalancerCutoffTime();
1156    synchronized (this.balancer) {
1157      // If balance not true, don't run balancer.
1158      if (!this.loadBalancerTracker.isBalancerOn()) return false;
1159      // Only allow one balance run at at time.
1160      if (this.assignmentManager.getRegionStates().isRegionsInTransition()) {
1161        Set<RegionState> regionsInTransition =
1162          this.assignmentManager.getRegionStates().getRegionsInTransition();
1163        // if hbase:meta region is in transition, result of assignment cannot be recorded
1164        // ignore the force flag in that case
1165        boolean metaInTransition = assignmentManager.getRegionStates().isMetaRegionInTransition();
1166        String prefix = force && !metaInTransition ? "R" : "Not r";
1167        LOG.debug(prefix + "unning balancer because " + regionsInTransition.size() +
1168          " region(s) in transition: " + org.apache.commons.lang.StringUtils.
1169            abbreviate(regionsInTransition.toString(), 256));
1170        if (!force || metaInTransition) return false;
1171      }
1172      if (this.serverManager.areDeadServersInProgress()) {
1173        LOG.debug("Not running balancer because processing dead regionserver(s): " +
1174          this.serverManager.getDeadServers());
1175        return false;
1176      }
1177
1178      if (this.cpHost != null) {
1179        try {
1180          if (this.cpHost.preBalance()) {
1181            LOG.debug("Coprocessor bypassing balancer request");
1182            return false;
1183          }
1184        } catch (IOException ioe) {
1185          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1186          return false;
1187        }
1188      }
1189
1190      Map<TableName, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1191        this.assignmentManager.getRegionStates().getAssignmentsByTable();
1192
1193      List<RegionPlan> plans = new ArrayList<RegionPlan>();
1194
1195      //Give the balancer the current cluster state.
1196      this.balancer.setClusterStatus(getClusterStatus());
1197      for (Entry<TableName, Map<ServerName, List<HRegionInfo>>> e : assignmentsByTable.entrySet()) {
1198        List<RegionPlan> partialPlans = this.balancer.balanceCluster(e.getKey(), e.getValue());
1199        if (partialPlans != null) plans.addAll(partialPlans);
1200      }
1201
1202      long cutoffTime = System.currentTimeMillis() + maximumBalanceTime;
1203      int rpCount = 0;  // number of RegionPlans balanced so far
1204      long totalRegPlanExecTime = 0;
1205      if (plans != null && !plans.isEmpty()) {
1206        for (RegionPlan plan: plans) {
1207          LOG.info("balance " + plan);
1208          long balStartTime = System.currentTimeMillis();
1209          //TODO: bulk assign
1210          this.assignmentManager.balance(plan);
1211          totalRegPlanExecTime += System.currentTimeMillis()-balStartTime;
1212          rpCount++;
1213          if (rpCount < plans.size() &&
1214              // if performing next balance exceeds cutoff time, exit the loop
1215              (System.currentTimeMillis() + (totalRegPlanExecTime / rpCount)) > cutoffTime) {
1216            //TODO: After balance, there should not be a cutoff time (keeping it as
1217            // a security net for now)
1218            LOG.debug("No more balancing till next balance run; maximumBalanceTime=" +
1219              maximumBalanceTime);
1220            break;
1221          }
1222        }
1223      }
1224      if (this.cpHost != null) {
1225        try {
1226          this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1227        } catch (IOException ioe) {
1228          // balancing already succeeded so don't change the result
1229          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1230        }
1231      }
1232    }
1233    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1234    // Return true indicating a success.
1235    return true;
1236  }

1244  /**
1245   * Perform normalization of cluster (invoked by {@link RegionNormalizerChore}).
1246   *
1247   * @return true if normalization step was performed successfully, false otherwise
1248   *    (specifically, if HMaster hasn't been initialized properly or normalization
1249   *    is globally disabled)
1250   */
1251  public boolean normalizeRegions() throws IOException {
1252    if (!isInitialized()) {
1253      LOG.debug("Master has not been initialized, don't run region normalizer.");
1254      return false;
1255    }
1256
1257    if (isInMaintenanceMode()) {
1258      LOG.info("Master is in maintenance mode, don't run region normalizer.");
1259      return false;
1260    }
1261
1262    if (!this.regionNormalizerTracker.isNormalizerOn()) {
1263      LOG.debug("Region normalization is disabled, don't run region normalizer.");
1264      return false;
1265    }
1266
1267    synchronized (this.normalizer) {
1268      // Don't run the normalizer concurrently
1269      List<TableName> allEnabledTables = new ArrayList<>(
1270        this.tableStateManager.getTablesInStates(TableState.State.ENABLED));
1271
1272      Collections.shuffle(allEnabledTables);
1273
1274      for (TableName table : allEnabledTables) {
1275        if (isInMaintenanceMode()) {
1276          LOG.debug("Master is in maintenance mode, stop running region normalizer.");
1277          return false;
1278        }
1279
1280        HTableDescriptor tblDesc = getTableDescriptors().get(table);
1281        if (table.isSystemTable() || (tblDesc != null &&
1282            !tblDesc.isNormalizationEnabled())) {
1283          LOG.debug("Skipping normalization for table: " + table + ", as it's either system"
1284              + " table or doesn't have auto normalization turned on");
1285          continue;
1286        }
1287        List<NormalizationPlan> plans = this.normalizer.computePlanForTable(table);
1288        if (plans != null) {
1289          for (NormalizationPlan plan : plans) {
1290            plan.execute(clusterConnection.getAdmin());
1291            if (plan.getType() == PlanType.SPLIT) {
1292              splitPlanCount++;
1293            } else if (plan.getType() == PlanType.MERGE) {
1294              mergePlanCount++;
1295            }
1296          }
1297        }
1298      }
1299    }
1300    // If Region did not generate any plans, it means the cluster is already balanced.
1301    // Return true indicating a success.
1302    return true;
1303  }

1368  void move(final byte[] encodedRegionName,
1369      final byte[] destServerName) throws HBaseIOException {
1370    RegionState regionState = assignmentManager.getRegionStates().
1371      getRegionState(Bytes.toString(encodedRegionName));
1372
1373    HRegionInfo hri;
1374    if (regionState != null) {
1375      hri = regionState.getRegion();
1376    } else {
1377      throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1378    }
1379
1380    ServerName dest;
1381    if (destServerName == null || destServerName.length == 0) {
1382      LOG.info("Passed destination servername is null/empty so " +
1383        "choosing a server at random");
1384      final List<ServerName> destServers = this.serverManager.createDestinationServersList(
1385        regionState.getServerName());
1386      dest = balancer.randomAssignment(hri, destServers);
1387      if (dest == null) {
1388        LOG.debug("Unable to determine a plan to assign " + hri);
1389        return;
1390      }
1391    } else {
1392      ServerName candidate = ServerName.valueOf(Bytes.toString(destServerName));
1393      dest = balancer.randomAssignment(hri, Lists.newArrayList(candidate));
1394      if (dest == null) {
1395        LOG.debug("Unable to determine a plan to assign " + hri);
1396        return;
1397      }
1398      if (dest.equals(serverName) && balancer instanceof BaseLoadBalancer
1399          && !((BaseLoadBalancer)balancer).shouldBeOnMaster(hri)) {
1400        // To avoid unnecessary region moving later by balancer. Don't put user
1401        // regions on master. Regions on master could be put on other region
1402        // server intentionally by test however.
1403        LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1404          + " to avoid unnecessary region moving later by load balancer,"
1405          + " because it should not be on master");
1406        return;
1407      }
1408    }
1409
1410    if (dest.equals(regionState.getServerName())) {
1411      LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1412        + " because region already assigned to the same server " + dest + ".");
1413      return;
1414    }
1415
1416    // Now we can do the move
1417    RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1418
1419    try {
1420      checkInitialized();
1421      if (this.cpHost != null) {
1422        if (this.cpHost.preMove(hri, rp.getSource(), rp.getDestination())) {
1423          return;
1424        }
1425      }
1426      // warmup the region on the destination before initiating the move. this call
1427      // is synchronous and takes some time. doing it before the source region gets
1428      // closed
1429      serverManager.sendRegionWarmup(rp.getDestination(), hri);
1430
1431      LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
1432      this.assignmentManager.balance(rp);
1433      if (this.cpHost != null) {
1434        this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
1435      }
1436    } catch (IOException ioe) {
1437      if (ioe instanceof HBaseIOException) {
1438        throw (HBaseIOException)ioe;
1439      }
1440      throw new HBaseIOException(ioe);
1441    }
1442  }

1506  /**
1507   * Checks whether the table conforms to some sane limits, and configured
1508   * values (compression, etc) work. Throws an exception if something is wrong.
1509   * @throws IOException
1510   */
1511  private void sanityCheckTableDescriptor(final HTableDescriptor htd) throws IOException {
1512    final String CONF_KEY = "hbase.table.sanity.checks";
1513    boolean logWarn = false;
1514    if (!conf.getBoolean(CONF_KEY, true)) {
1515      logWarn = true;
1516    }
1517    String tableVal = htd.getConfigurationValue(CONF_KEY);
1518    if (tableVal != null && !Boolean.valueOf(tableVal)) {
1519      logWarn = true;
1520    }
1521
1522    // check max file size
1523    long maxFileSizeLowerLimit = 2 * 1024 * 1024L; // 2M is the default lower limit
1524    long maxFileSize = htd.getMaxFileSize();
1525    if (maxFileSize < 0) {
1526      maxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE, maxFileSizeLowerLimit);
1527    }
1528    if (maxFileSize < conf.getLong("hbase.hregion.max.filesize.limit", maxFileSizeLowerLimit)) {
1529      String message = "MAX_FILESIZE for table descriptor or "
1530          + "\"hbase.hregion.max.filesize\" (" + maxFileSize
1531          + ") is too small, which might cause over splitting into unmanageable "
1532          + "number of regions.";
1533      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1534    }
1535
1536    // check flush size
1537    long flushSizeLowerLimit = 1024 * 1024L; // 1M is the default lower limit
1538    long flushSize = htd.getMemStoreFlushSize();
1539    if (flushSize < 0) {
1540      flushSize = conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSizeLowerLimit);
1541    }
1542    if (flushSize < conf.getLong("hbase.hregion.memstore.flush.size.limit", flushSizeLowerLimit)) {
1543      String message = "MEMSTORE_FLUSHSIZE for table descriptor or "
1544          + "\"hbase.hregion.memstore.flush.size\" ("+flushSize+") is too small, which might cause"
1545          + " very frequent flushing.";
1546      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1547    }
1548
1549    // check that coprocessors and other specified plugin classes can be loaded
1550    try {
1551      checkClassLoading(conf, htd);
1552    } catch (Exception ex) {
1553      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, ex.getMessage(), null);
1554    }
1555
1556    // check compression can be loaded
1557    try {
1558      checkCompression(htd);
1559    } catch (IOException e) {
1560      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1561    }
1562
1563    // check encryption can be loaded
1564    try {
1565      checkEncryption(conf, htd);
1566    } catch (IOException e) {
1567      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1568    }
1569    // Verify compaction policy
1570    try{
1571      checkCompactionPolicy(conf, htd);
1572    } catch(IOException e){
1573      warnOrThrowExceptionForFailure(false, CONF_KEY, e.getMessage(), e);
1574    }
1575    // check that we have at least 1 CF
1576    if (htd.getColumnFamilies().length == 0) {
1577      String message = "Table should have at least one column family.";
1578      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1579    }
1580
1581    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1582      if (hcd.getTimeToLive() <= 0) {
1583        String message = "TTL for column family " + hcd.getNameAsString() + " must be positive.";
1584        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1585      }
1586
1587      // check blockSize
1588      if (hcd.getBlocksize() < 1024 || hcd.getBlocksize() > 16 * 1024 * 1024) {
1589        String message = "Block size for column family " + hcd.getNameAsString()
1590            + "  must be between 1K and 16MB.";
1591        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1592      }
1593
1594      // check versions
1595      if (hcd.getMinVersions() < 0) {
1596        String message = "Min versions for column family " + hcd.getNameAsString()
1597          + "  must be positive.";
1598        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1599      }
1600      // max versions already being checked
1601
1602      // HBASE-13776 Setting illegal versions for HColumnDescriptor
1603      //  does not throw IllegalArgumentException
1604      // check minVersions <= maxVerions
1605      if (hcd.getMinVersions() > hcd.getMaxVersions()) {
1606        String message = "Min versions for column family " + hcd.getNameAsString()
1607            + " must be less than the Max versions.";
1608        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1609      }
1610
1611      // check replication scope
1612      checkReplicationScope(hcd);
1613
1614      // check data replication factor, it can be 0(default value) when user has not explicitly
1615      // set the value, in this case we use default replication factor set in the file system.
1616      if (hcd.getDFSReplication() < 0) {
1617        String message = "HFile Replication for column family " + hcd.getNameAsString()
1618            + "  must be greater than zero.";
1619        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1620      }
1621
1622      // TODO: should we check coprocessors and encryption ?
1623    }
1624  }

1638  private void checkCompactionPolicy(Configuration conf, HTableDescriptor htd)
1639      throws IOException {
1640    // FIFO compaction has some requirements
1641    // Actually FCP ignores periodic major compactions
1642    String className =
1643        htd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1644    if (className == null) {
1645      className =
1646          conf.get(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
1647            ExploringCompactionPolicy.class.getName());
1648    }
1649
1650    int blockingFileCount = HStore.DEFAULT_BLOCKING_STOREFILE_COUNT;
1651    String sv = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1652    if (sv != null) {
1653      blockingFileCount = Integer.parseInt(sv);
1654    } else {
1655      blockingFileCount = conf.getInt(HStore.BLOCKING_STOREFILES_KEY, blockingFileCount);
1656    }
1657
1658    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1659      String compactionPolicy =
1660          hcd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1661      if (compactionPolicy == null) {
1662        compactionPolicy = className;
1663      }
1664      if (!compactionPolicy.equals(FIFOCompactionPolicy.class.getName())) {
1665        continue;
1666      }
1667      // FIFOCompaction
1668      String message = null;
1669
1670      // 1. Check TTL
1671      if (hcd.getTimeToLive() == HColumnDescriptor.DEFAULT_TTL) {
1672        message = "Default TTL is not supported for FIFO compaction";
1673        throw new IOException(message);
1674      }
1675
1676      // 2. Check min versions
1677      if (hcd.getMinVersions() > 0) {
1678        message = "MIN_VERSION > 0 is not supported for FIFO compaction";
1679        throw new IOException(message);
1680      }
1681
1682      // 3. blocking file count
1683      String sbfc = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1684      if (sbfc != null) {
1685        blockingFileCount = Integer.parseInt(sbfc);
1686      }
1687      if (blockingFileCount < 1000) {
1688        message =
1689            "blocking file count '" + HStore.BLOCKING_STOREFILES_KEY + "' " + blockingFileCount
1690                + " is below recommended minimum of 1000";
1691        throw new IOException(message);
1692      }
1693    }
1694  }

1706  private void startActiveMasterManager(int infoPort) throws KeeperException {
1707    String backupZNode = ZKUtil.joinZNode(
1708      zooKeeper.backupMasterAddressesZNode, serverName.toString());
1709    /*
1710    * Add a ZNode for ourselves in the backup master directory since we
1711    * may not become the active master. If so, we want the actual active
1712    * master to know we are backup masters, so that it won't assign
1713    * regions to us if so configured.
1714    *
1715    * If we become the active master later, ActiveMasterManager will delete
1716    * this node explicitly.  If we crash before then, ZooKeeper will delete
1717    * this node for us since it is ephemeral.
1718    */
1719    LOG.info("Adding backup master ZNode " + backupZNode);
1720    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode,
1721        serverName, infoPort)) {
1722      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
1723    }
1724
1725    activeMasterManager.setInfoPort(infoPort);
1726    // Start a thread to try to become the active master, so we won't block here
1727    Threads.setDaemonThreadRunning(new Thread(new Runnable() {
1728      @Override
1729      public void run() {
1730        int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
1731          HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
1732        // If we're a backup master, stall until a primary to writes his address
1733        if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP,
1734          HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
1735          LOG.debug("HMaster started in backup mode. "
1736            + "Stalling until master znode is written.");
1737          // This will only be a minute or so while the cluster starts up,
1738          // so don't worry about setting watches on the parent znode
1739          while (!activeMasterManager.hasActiveMaster()) {
1740            LOG.debug("Waiting for master address ZNode to be written "
1741              + "(Also watching cluster state node)");
1742            Threads.sleep(timeout);
1743          }
1744        }
1745        MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
1746        status.setDescription("Master startup");
1747        try {
1748          if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
1749            finishActiveMasterInitialization(status);
1750          }
1751        } catch (Throwable t) {
1752          status.setStatus("Failed to become active: " + t.getMessage());
1753          LOG.fatal("Failed to become active master", t);
1754          // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
1755          if (t instanceof NoClassDefFoundError &&
1756            t.getMessage()
1757              .contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
1758            // improved error message for this special case
1759            abort("HBase is having a problem with its Hadoop jars.  You may need to "
1760              + "recompile HBase against Hadoop version "
1761              + org.apache.hadoop.util.VersionInfo.getVersion()
1762              + " or change your hadoop jars to start properly", t);
1763          } else {
1764            abort("Unhandled exception. Starting shutdown.", t);
1765          }
1766        } finally {
1767          status.cleanup();
1768        }
1769      }
1770    }, getServerName().toShortString() + ".activeMasterManager"));
1771  }

2678  /**
2679   * @return list of table table descriptors after filtering by regex and whether to include system
2680   *    tables, etc.
2681   * @throws IOException
2682   */
2683  private List<HTableDescriptor> getTableDescriptors(final List<HTableDescriptor> htds,
2684      final String namespace, final String regex, final List<TableName> tableNameList,
2685      final boolean includeSysTables)
2686  throws IOException {
2687    if (tableNameList == null || tableNameList.size() == 0) {
2688      // request for all TableDescriptors
2689      Collection<HTableDescriptor> allHtds;
2690      if (namespace != null && namespace.length() > 0) {
2691        // Do a check on the namespace existence. Will fail if does not exist.
2692        this.clusterSchemaService.getNamespace(namespace);
2693        allHtds = tableDescriptors.getByNamespace(namespace).values();
2694      } else {
2695        allHtds = tableDescriptors.getAll().values();
2696      }
2697      for (HTableDescriptor desc: allHtds) {
2698        if (tableStateManager.isTablePresent(desc.getTableName())
2699            && (includeSysTables || !desc.getTableName().isSystemTable())) {
2700          htds.add(desc);
2701        }
2702      }
2703    } else {
2704      for (TableName s: tableNameList) {
2705        if (tableStateManager.isTablePresent(s)) {
2706          HTableDescriptor desc = tableDescriptors.get(s);
2707          if (desc != null) {
2708            htds.add(desc);
2709          }
2710        }
2711      }
2712    }
2713
2714    // Retains only those matched by regular expression.
2715    if (regex != null) filterTablesByRegex(htds, Pattern.compile(regex));
2716    return htds;
2717  }

2719  /**
2720   * Removes the table descriptors that don't match the pattern.
2721   * @param descriptors list of table descriptors to filter
2722   * @param pattern the regex to use
2723   */
2724  private static void filterTablesByRegex(final Collection<HTableDescriptor> descriptors,
2725      final Pattern pattern) {
2726    final String defaultNS = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
2727    Iterator<HTableDescriptor> itr = descriptors.iterator();
2728    while (itr.hasNext()) {
2729      HTableDescriptor htd = itr.next();
2730      String tableName = htd.getTableName().getNameAsString();
2731      boolean matched = pattern.matcher(tableName).matches();
2732      if (!matched && htd.getTableName().getNamespaceAsString().equals(defaultNS)) {
2733        matched = pattern.matcher(defaultNS + TableName.NAMESPACE_DELIM + tableName).matches();
2734      }
2735      if (!matched) {
2736        itr.remove();
2737      }
2738    }
2739  }

2768  public void reportMobCompactionStart(TableName tableName) throws IOException {
2769    IdLock.Entry lockEntry = null;
2770    try {
2771      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2772      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2773      if (compactionsCount == null) {
2774        compactionsCount = new AtomicInteger(0);
2775        mobCompactionStates.put(tableName, compactionsCount);
2776      }
2777      compactionsCount.incrementAndGet();
2778    } finally {
2779      if (lockEntry != null) {
2780        mobCompactionLock.releaseLockEntry(lockEntry);
2781      }
2782    }
2783  }
2784
2785  public void reportMobCompactionEnd(TableName tableName) throws IOException {
2786    IdLock.Entry lockEntry = null;
2787    try {
2788      lockEntry = mobCompactionLock.getLockEntry(tableName.hashCode());
2789      AtomicInteger compactionsCount = mobCompactionStates.get(tableName);
2790      if (compactionsCount != null) {
2791        int count = compactionsCount.decrementAndGet();
2792        // remove the entry if the count is 0.
2793        if (count == 0) {
2794          mobCompactionStates.remove(tableName);
2795        }
2796      }
2797    } finally {
2798      if (lockEntry != null) {
2799        mobCompactionLock.releaseLockEntry(lockEntry);
2800      }
2801    }
2802  }
