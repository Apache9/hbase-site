145  enum BalanceSwitchMode {
146    SYNC,
147    ASYNC
148  }

150  /**
151   * Assigns balancer switch according to BalanceSwitchMode
152   * @param b new balancer switch
153   * @param mode BalanceSwitchMode
154   * @return old balancer switch
155   */
156  boolean switchBalancer(final boolean b, BalanceSwitchMode mode) throws IOException {
157    boolean oldValue = master.loadBalancerTracker.isBalancerOn();
158    boolean newValue = b;
159    try {
160      if (master.cpHost != null) {
161        newValue = master.cpHost.preBalanceSwitch(newValue);
162      }
163      try {
164        if (mode == BalanceSwitchMode.SYNC) {
165          synchronized (master.balancer) {
166            master.loadBalancerTracker.setBalancerOn(newValue);
167          }
168        } else {
169          master.loadBalancerTracker.setBalancerOn(newValue);
170        }
171      } catch (KeeperException ke) {
172        throw new IOException(ke);
173      }
174      LOG.info(master.getClientIdAuditPrefix() + " set balanceSwitch=" + newValue);
175      if (master.cpHost != null) {
176        master.cpHost.postBalanceSwitch(oldValue, newValue);
177      }
178    } catch (IOException ioe) {
179      LOG.warn("Error flipping balance switch", ioe);
180    }
181    return oldValue;
182  }
183
184  boolean synchronousBalanceSwitch(final boolean b) throws IOException {
185    return switchBalancer(b, BalanceSwitchMode.SYNC);
186  }

189   * Sets normalizer on/off flag in ZK.
190   */
191  public boolean normalizerSwitch(boolean on) {
192    boolean oldValue = master.getRegionNormalizerTracker().isNormalizerOn();
193    boolean newValue = on;
194    try {
195      try {
196        master.getRegionNormalizerTracker().setNormalizerOn(newValue);
197      } catch (KeeperException ke) {
198        throw new IOException(ke);
199      }
200      LOG.info(master.getClientIdAuditPrefix() + " set normalizerSwitch=" + newValue);
201    } catch (IOException ioe) {
202      LOG.warn("Error flipping normalizer switch", ioe);
203    }
204    return oldValue;
205  }

238      RpcController controller, RegionServerReportRequest request) throws ServiceException {
239    try {
240      master.checkServiceStarted();
241      ClusterStatusProtos.ServerLoad sl = request.getLoad();
242      ServerName serverName = ProtobufUtil.toServerName(request.getServer());
243      ServerLoad oldLoad = master.serverManager.getLoad(serverName);
244      master.serverManager.regionServerReport(serverName, new ServerLoad(sl));
245      if (sl != null && master.metricsMaster != null) {
246        // Up our metrics.
247        master.metricsMaster.incrementRequests(sl.getTotalNumberOfRequests()
248            - (oldLoad != null ? oldLoad.getTotalNumberOfRequests() : 0));
249      }
250    } catch (IOException ioe) {
251      throw new ServiceException(ioe);
252    }
253    return RegionServerReportResponse.newBuilder().build();
254  }

313  @Override
314  public AssignRegionResponse assignRegion(RpcController controller,
315      AssignRegionRequest req) throws ServiceException {
316    try {
317      final byte [] regionName = req.getRegion().getValue().toByteArray();
318      RegionSpecifierType type = req.getRegion().getType();
319      AssignRegionResponse arr = AssignRegionResponse.newBuilder().build();
320
321      master.checkInitialized();
322      if (type != RegionSpecifierType.REGION_NAME) {
323        LOG.warn("assignRegion specifier type: expected: " + RegionSpecifierType.REGION_NAME
324          + " actual: " + type);
325      }
326      RegionStates regionStates = master.assignmentManager.getRegionStates();
327      HRegionInfo regionInfo = regionStates.getRegionInfo(regionName);
328      if (regionInfo == null) throw new UnknownRegionException(Bytes.toString(regionName));
329      if (master.cpHost != null) {
330        if (master.cpHost.preAssign(regionInfo)) {
331          return arr;
332        }
333      }
334      LOG.info(master.getClientIdAuditPrefix()
335        + " assign " + regionInfo.getRegionNameAsString());
336      master.assignmentManager.assign(regionInfo, true);
337      if (master.cpHost != null) {
338        master.cpHost.postAssign(regionInfo);
339      }
340      return arr;
341    } catch (IOException ioe) {
342      throw new ServiceException(ioe);
343    }
344  }

482  @Override
483  public DispatchMergingRegionsResponse dispatchMergingRegions(RpcController c,
484      DispatchMergingRegionsRequest request) throws ServiceException {
485    try {
486      master.checkInitialized();
487    } catch (IOException ioe) {
488      throw new ServiceException(ioe);
489    }
490
491    final byte[] encodedNameOfRegionA = request.getRegionA().getValue()
492      .toByteArray();
493    final byte[] encodedNameOfRegionB = request.getRegionB().getValue()
494      .toByteArray();
495    final boolean forcible = request.getForcible();
496    if (request.getRegionA().getType() != RegionSpecifierType.ENCODED_REGION_NAME
497        || request.getRegionB().getType() != RegionSpecifierType.ENCODED_REGION_NAME) {
498      LOG.warn("mergeRegions specifier type: expected: "
499        + RegionSpecifierType.ENCODED_REGION_NAME + " actual: region_a="
500        + request.getRegionA().getType() + ", region_b="
501        + request.getRegionB().getType());
502    }
503    RegionStates regionStates = master.assignmentManager.getRegionStates();
504    RegionState regionStateA = regionStates.getRegionState(Bytes.toString(encodedNameOfRegionA));
505    RegionState regionStateB = regionStates.getRegionState(Bytes.toString(encodedNameOfRegionB));
506    if (regionStateA == null || regionStateB == null) {
507      throw new ServiceException(new UnknownRegionException(
508          Bytes.toStringBinary(regionStateA == null ? encodedNameOfRegionA
509              : encodedNameOfRegionB)));
510    }
511
512    if (!regionStateA.isOpened() || !regionStateB.isOpened()) {
513      throw new ServiceException(new MergeRegionException(
514        "Unable to merge regions not online " + regionStateA + ", " + regionStateB));
515    }
516
517    HRegionInfo regionInfoA = regionStateA.getRegion();
518    HRegionInfo regionInfoB = regionStateB.getRegion();
519    if (regionInfoA.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID ||
520        regionInfoB.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID) {
521      throw new ServiceException(new MergeRegionException("Can't merge non-default replicas"));
522    }
523    if (regionInfoA.compareTo(regionInfoB) == 0) {
524      throw new ServiceException(new MergeRegionException(
525        "Unable to merge a region to itself " + regionInfoA + ", " + regionInfoB));
526    }
527
528    if (!forcible && !HRegionInfo.areAdjacent(regionInfoA, regionInfoB)) {
529      throw new ServiceException(new MergeRegionException(
530        "Unable to merge not adjacent regions "
531          + regionInfoA.getRegionNameAsString() + ", "
532          + regionInfoB.getRegionNameAsString()
533          + " where forcible = " + forcible));
534    }
535
536    try {
537      master.dispatchMergingRegions(regionInfoA, regionInfoB, forcible);
538    } catch (IOException ioe) {
539      throw new ServiceException(ioe);
540    }
541
542    return DispatchMergingRegionsResponse.newBuilder().build();
543  }

571  @Override
572  public ClientProtos.CoprocessorServiceResponse execMasterService(final RpcController controller,
573      final ClientProtos.CoprocessorServiceRequest request) throws ServiceException {
574    try {
575      master.checkInitialized();
576      ServerRpcController execController = new ServerRpcController();
577
578      ClientProtos.CoprocessorServiceCall call = request.getCall();
579      String serviceName = call.getServiceName();
580      String methodName = call.getMethodName();
581      if (!master.coprocessorServiceHandlers.containsKey(serviceName)) {
582        throw new UnknownProtocolException(null,
583          "No registered master coprocessor service found for name "+serviceName);
584      }
585
586      Service service = master.coprocessorServiceHandlers.get(serviceName);
587      Descriptors.ServiceDescriptor serviceDesc = service.getDescriptorForType();
588      Descriptors.MethodDescriptor methodDesc = serviceDesc.findMethodByName(methodName);
589      if (methodDesc == null) {
590        throw new UnknownProtocolException(service.getClass(),
591          "Unknown method "+methodName+" called on master service "+serviceName);
592      }
593
594      //invoke the method
595      Message.Builder builderForType = service.getRequestPrototype(methodDesc).newBuilderForType();
596      ProtobufUtil.mergeFrom(builderForType, call.getRequest());
597      Message execRequest = builderForType.build();
598      final Message.Builder responseBuilder =
599          service.getResponsePrototype(methodDesc).newBuilderForType();
600      service.callMethod(methodDesc, execController, execRequest, new RpcCallback<Message>() {
601        @Override
602        public void run(Message message) {
603          if (message != null) {
604            responseBuilder.mergeFrom(message);
605          }
606        }
607      });
608      Message execResult = responseBuilder.build();
609
610      if (execController.getFailedOn() != null) {
611        throw execController.getFailedOn();
612      }
613      ClientProtos.CoprocessorServiceResponse.Builder builder =
614        ClientProtos.CoprocessorServiceResponse.newBuilder();
615      builder.setRegion(RequestConverter.buildRegionSpecifier(
616        RegionSpecifierType.REGION_NAME, HConstants.EMPTY_BYTE_ARRAY));
617      builder.setValue(
618        builder.getValueBuilder().setName(execResult.getClass().getName())
619          .setValue(execResult.toByteString()));

984      } else {

986        if (proc == null) {
987          builder.setState(GetProcedureResultResponse.State.NOT_FOUND);
988        } else {
989          builder.setState(GetProcedureResultResponse.State.RUNNING);
990          builder.setStartTime(proc.getStartTime());
991          builder.setLastUpdate(proc.getLastUpdate());
992        }

996      throw new ServiceException(e);

1288  @Override
1289  public UnassignRegionResponse unassignRegion(RpcController controller,
1290      UnassignRegionRequest req) throws ServiceException {
1291    try {
1292      final byte [] regionName = req.getRegion().getValue().toByteArray();
1293      RegionSpecifierType type = req.getRegion().getType();
1294      final boolean force = req.getForce();
1295      UnassignRegionResponse urr = UnassignRegionResponse.newBuilder().build();
1296
1297      master.checkInitialized();
1298      if (type != RegionSpecifierType.REGION_NAME) {
1299        LOG.warn("unassignRegion specifier type: expected: " + RegionSpecifierType.REGION_NAME
1300          + " actual: " + type);
1301      }
1302      Pair<HRegionInfo, ServerName> pair =
1303        MetaTableAccessor.getRegion(master.getConnection(), regionName);
1304      if (pair == null) throw new UnknownRegionException(Bytes.toString(regionName));
1305      HRegionInfo hri = pair.getFirst();
1306      if (master.cpHost != null) {
1307        if (master.cpHost.preUnassign(hri, force)) {
1308          return urr;
1309        }
1310      }
1311      LOG.debug(master.getClientIdAuditPrefix() + " unassign " + hri.getRegionNameAsString()
1312          + " in current location if it is online and reassign.force=" + force);
1313      master.assignmentManager.unassign(hri);
1314      if (master.cpHost != null) {
1315        master.cpHost.postUnassign(hri, force);
1316      }
1317
1318      return urr;
1319    } catch (IOException ioe) {
1320      throw new ServiceException(ioe);
1321    }
1322  }

1325  public ReportRegionStateTransitionResponse reportRegionStateTransition(RpcController c,
1326      ReportRegionStateTransitionRequest req) throws ServiceException {
1327    try {
1328      master.checkServiceStarted();
1329      RegionStateTransition rt = req.getTransition(0);
1330      TableName tableName = ProtobufUtil.toTableName(
1331        rt.getRegionInfo(0).getTableName());
1332      RegionStates regionStates = master.assignmentManager.getRegionStates();
1333      if (!(TableName.META_TABLE_NAME.equals(tableName)
1334          && regionStates.getRegionState(HRegionInfo.FIRST_META_REGIONINFO) != null)
1335            && !master.assignmentManager.isFailoverCleanupDone()) {
1336        // Meta region is assigned before master finishes the
1337        // failover cleanup. So no need this check for it
1338        throw new PleaseHoldException("Master is rebuilding user regions");
1339      }
1340      ServerName sn = ProtobufUtil.toServerName(req.getServer());
1341      String error = master.assignmentManager.onRegionTransition(sn, rt);
1342      ReportRegionStateTransitionResponse.Builder rrtr =
1343        ReportRegionStateTransitionResponse.newBuilder();
1344      if (error != null) {
1345        rrtr.setErrorMessage(error);
1346      }
1347      return rrtr.build();
1348    } catch (IOException ioe) {
1349      throw new ServiceException(ioe);
1350    }

1441  /**
1442   * Compacts the mob files in the current table.
1443   * @param request the request.
1444   * @param tableName the current table name.
1445   * @return The response of the mob file compaction.
1446   * @throws IOException
1447   */
1448  private CompactRegionResponse compactMob(final CompactRegionRequest request,
1449    TableName tableName) throws IOException {
1450    if (!master.getTableStateManager().isTableState(tableName, TableState.State.ENABLED)) {
1451      throw new DoNotRetryIOException("Table " + tableName + " is not enabled");
1452    }
1453    boolean allFiles = false;
1454    List<HColumnDescriptor> compactedColumns = new ArrayList<HColumnDescriptor>();
1455    HColumnDescriptor[] hcds = master.getTableDescriptors().get(tableName).getColumnFamilies();
1456    byte[] family = null;
1457    if (request.hasFamily()) {
1458      family = request.getFamily().toByteArray();
1459      for (HColumnDescriptor hcd : hcds) {
1460        if (Bytes.equals(family, hcd.getName())) {
1461          if (!hcd.isMobEnabled()) {
1462            LOG.error("Column family " + hcd.getNameAsString() + " is not a mob column family");
1463            throw new DoNotRetryIOException("Column family " + hcd.getNameAsString()
1464                    + " is not a mob column family");
1465          }
1466          compactedColumns.add(hcd);
1467        }
1468      }
1469    } else {
1470      for (HColumnDescriptor hcd : hcds) {
1471        if (hcd.isMobEnabled()) {
1472          compactedColumns.add(hcd);
1473        }
1474      }
1475    }
1476    if (compactedColumns.isEmpty()) {
1477      LOG.error("No mob column families are assigned in the mob compaction");
1478      throw new DoNotRetryIOException(
1479              "No mob column families are assigned in the mob compaction");
1480    }
1481    if (request.hasMajor() && request.getMajor()) {
1482      allFiles = true;
1483    }
1484    String familyLogMsg = (family != null) ? Bytes.toString(family) : "";
1485    if (LOG.isTraceEnabled()) {
1486      LOG.trace("User-triggered mob compaction requested for table: "
1487              + tableName.getNameAsString() + " for column family: " + familyLogMsg);
1488    }
1489    master.requestMobCompaction(tableName, compactedColumns, allFiles);
1490    return CompactRegionResponse.newBuilder().build();
1491  }

1531  /**
1532   * Returns the security capabilities in effect on the cluster
1533   */
1534  @Override
1535  public SecurityCapabilitiesResponse getSecurityCapabilities(RpcController controller,
1536      SecurityCapabilitiesRequest request) throws ServiceException {
1537    SecurityCapabilitiesResponse.Builder response = SecurityCapabilitiesResponse.newBuilder();
1538    try {
1539      master.checkInitialized();
1540      Set<Capability> capabilities = new HashSet<>();
1541      // Authentication
1542      if (User.isHBaseSecurityEnabled(master.getConfiguration())) {
1543        capabilities.add(Capability.SECURE_AUTHENTICATION);
1544      } else {
1545        capabilities.add(Capability.SIMPLE_AUTHENTICATION);
1546      }
1547      // The AccessController can provide AUTHORIZATION and CELL_AUTHORIZATION
1548      if (master.cpHost != null &&
1549            master.cpHost.findCoprocessor(AccessController.class.getName()) != null) {
1550        if (AccessController.isAuthorizationSupported(master.getConfiguration())) {
1551          capabilities.add(Capability.AUTHORIZATION);
1552        }
1553        if (AccessController.isCellAuthorizationSupported(master.getConfiguration())) {
1554          capabilities.add(Capability.CELL_AUTHORIZATION);
1555        }
1556      }
1557      // The VisibilityController can provide CELL_VISIBILITY
1558      if (master.cpHost != null &&
1559            master.cpHost.findCoprocessor(VisibilityController.class.getName()) != null) {
1560        if (VisibilityController.isCellAuthorizationSupported(master.getConfiguration())) {
1561          capabilities.add(Capability.CELL_VISIBILITY);
1562        }
1563      }
1564      response.addAllCapabilities(capabilities);
1565    } catch (IOException e) {
1566      throw new ServiceException(e);
1567    }
1568    return response.build();
1569  }
