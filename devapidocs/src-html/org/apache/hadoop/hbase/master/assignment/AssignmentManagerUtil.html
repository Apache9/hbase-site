062  static GetRegionInfoResponse getRegionInfoResponse(final MasterProcedureEnv env,
063      final ServerName regionLocation, final RegionInfo hri) throws IOException {
064    return getRegionInfoResponse(env, regionLocation, hri, false);
065  }

067  static GetRegionInfoResponse getRegionInfoResponse(final MasterProcedureEnv env,
068      final ServerName regionLocation, final RegionInfo hri, boolean includeBestSplitRow)
069      throws IOException {
070    // TODO: There is no timeout on this controller. Set one!
071    HBaseRpcController controller =
072      env.getMasterServices().getClusterConnection().getRpcControllerFactory().newController();
073    final AdminService.BlockingInterface admin =
074      env.getMasterServices().getClusterConnection().getAdmin(regionLocation);
075    GetRegionInfoRequest request = null;
076    if (includeBestSplitRow) {
077      request = RequestConverter.buildGetRegionInfoRequest(hri.getRegionName(), false, true);
078    } else {
079      request = RequestConverter.buildGetRegionInfoRequest(hri.getRegionName());
080    }
081    try {
082      return admin.getRegionInfo(controller, request);
083    } catch (ServiceException e) {
084      throw ProtobufUtil.handleRemoteException(e);
085    }
086  }

088  private static void lock(List<RegionStateNode> regionNodes) {
089    regionNodes.iterator().forEachRemaining(RegionStateNode::lock);
090  }
091
092  private static void unlock(List<RegionStateNode> regionNodes) {
093    for (ListIterator<RegionStateNode> iter = regionNodes.listIterator(regionNodes.size()); iter
094      .hasPrevious();) {
095      iter.previous().unlock();
096    }
097  }

099  static TransitRegionStateProcedure[] createUnassignProceduresForSplitOrMerge(
100      MasterProcedureEnv env, Stream<RegionInfo> regions, int regionReplication)
101      throws IOException {
102    List<RegionStateNode> regionNodes = regions
103      .flatMap(hri -> IntStream.range(0, regionReplication)
104        .mapToObj(i -> RegionReplicaUtil.getRegionInfoForReplica(hri, i)))
105      .map(env.getAssignmentManager().getRegionStates()::getOrCreateRegionStateNode)
106      .collect(Collectors.toList());
107    TransitRegionStateProcedure[] procs = new TransitRegionStateProcedure[regionNodes.size()];
108    boolean rollback = true;
109    int i = 0;
110    // hold the lock at once, and then release it in finally. This is important as SCP may jump in
111    // if we release the lock in the middle when we want to do rollback, and cause problems.
112    lock(regionNodes);
113    try {
114      for (; i < procs.length; i++) {
115        RegionStateNode regionNode = regionNodes.get(i);
116        TransitRegionStateProcedure proc =
117          TransitRegionStateProcedure.unassign(env, regionNode.getRegionInfo());
118        if (regionNode.getProcedure() != null) {
119          throw new HBaseIOException(
120            "The parent region " + regionNode + " is currently in transition, give up");
121        }
122        regionNode.setProcedure(proc);
123        procs[i] = proc;
124      }
125      // all succeeded, set rollback to false
126      rollback = false;
127    } finally {
128      if (rollback) {
129        for (;;) {
130          i--;
131          if (i < 0) {
132            break;
133          }
134          RegionStateNode regionNode = regionNodes.get(i);
135          regionNode.unsetProcedure(procs[i]);
136        }
137      }
138      unlock(regionNodes);
139    }
140    return procs;
141  }

143  private static TransitRegionStateProcedure[] createAssignProcedures(MasterProcedureEnv env,
144      List<RegionInfo> regions, int regionReplication, ServerName targetServer) {
145    // create the assign procs only for the primary region using the targetServer
146    TransitRegionStateProcedure[] primaryRegionProcs = regions.stream()
147        .map(env.getAssignmentManager().getRegionStates()::getOrCreateRegionStateNode)
148        .map(regionNode -> {
149          TransitRegionStateProcedure proc =
150              TransitRegionStateProcedure.assign(env, regionNode.getRegionInfo(), targetServer);
151          regionNode.lock();
152          try {
153            // should never fail, as we have the exclusive region lock, and the region is newly
154            // created, or has been successfully closed so should not be on any servers, so SCP will
155            // not process it either.
156            assert !regionNode.isInTransition();
157            regionNode.setProcedure(proc);
158          } finally {
159            regionNode.unlock();
160          }
161          return proc;
162        }).toArray(TransitRegionStateProcedure[]::new);
163    if (regionReplication == DEFAULT_REGION_REPLICA) {
164      // this is the default case
165      return primaryRegionProcs;
166    }
167    // collect the replica region infos
168    List<RegionInfo> replicaRegionInfos =
169        new ArrayList<RegionInfo>(regions.size() * (regionReplication - 1));
170    for (RegionInfo hri : regions) {
171      // start the index from 1
172      for (int i = 1; i < regionReplication; i++) {
173        replicaRegionInfos.add(RegionReplicaUtil.getRegionInfoForReplica(hri, i));
174      }
175    }
176    // create round robin procs. Note that we exclude the primary region's target server
177    TransitRegionStateProcedure[] replicaRegionAssignProcs =
178        env.getAssignmentManager().createRoundRobinAssignProcedures(replicaRegionInfos,
179          Collections.singletonList(targetServer));
180    // combine both the procs and return the result
181    return ArrayUtils.addAll(primaryRegionProcs, replicaRegionAssignProcs);
182  }
183
184  static TransitRegionStateProcedure[] createAssignProceduresForOpeningNewRegions(
185      MasterProcedureEnv env, List<RegionInfo> regions, int regionReplication,
186      ServerName targetServer) {
187    return createAssignProcedures(env, regions, regionReplication, targetServer);
188  }
189
190  static void reopenRegionsForRollback(MasterProcedureEnv env, List<RegionInfo> regions,
191      int regionReplication, ServerName targetServer) {
192    TransitRegionStateProcedure[] procs =
193        createAssignProcedures(env, regions, regionReplication, targetServer);
194    env.getMasterServices().getMasterProcedureExecutor().submitProcedures(procs);
195  }
196
197  static void removeNonDefaultReplicas(MasterProcedureEnv env, Stream<RegionInfo> regions,
198      int regionReplication) {
199    // Remove from in-memory states
200    regions.flatMap(hri -> IntStream.range(1, regionReplication)
201      .mapToObj(i -> RegionReplicaUtil.getRegionInfoForReplica(hri, i))).forEach(hri -> {
202        env.getAssignmentManager().getRegionStates().deleteRegion(hri);
203        env.getMasterServices().getServerManager().removeRegion(hri);
204        FavoredNodesManager fnm = env.getMasterServices().getFavoredNodesManager();
205        if (fnm != null) {
206          fnm.deleteFavoredNodesForRegions(Collections.singletonList(hri));
207        }
208      });
209  }
210
211  static void checkClosedRegion(MasterProcedureEnv env, RegionInfo regionInfo) throws IOException {
212    if (WALSplitter.hasRecoveredEdits(env.getMasterServices().getFileSystem(),
213      env.getMasterConfiguration(), regionInfo)) {
214      throw new IOException("Recovered.edits are found in Region: " + regionInfo +
215        ", abort split/merge to prevent data loss");
216    }
217  }
