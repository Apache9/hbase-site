048/**
049 * Procedure that describes the unassignment of a single region.
050 * There can only be one RegionTransitionProcedure -- i.e. an assign or an unassign -- per region
051 * running at a time, since each procedure takes a lock on the region.
052 *
053 * <p>The Unassign starts by placing a "close region" request in the Remote Dispatcher
054 * queue, and the procedure will then go into a "waiting state" (suspend).
055 * The Remote Dispatcher will batch the various requests for that server and
056 * they will be sent to the RS for execution.
057 * The RS will complete the open operation by calling master.reportRegionStateTransition().
058 * The AM will intercept the transition report, and notify this procedure.
059 * The procedure will wakeup and finish the unassign by publishing its new state on meta.
060 * <p>If we are unable to contact the remote regionserver whether because of ConnectException
061 * or socket timeout, we will call expire on the server we were trying to contact. We will remain
062 * in suspended state waiting for a wake up from the ServerCrashProcedure that is processing the
063 * failed server. The basic idea is that if we notice a crashed server, then we have a
064 * responsibility; i.e. we should not let go of the region until we are sure the server that was
065 * hosting has had its crash processed. If we let go of the region before then, an assign might
066 * run before the logs have been split which would make for data loss.
067 *
068 * <p>TODO: Rather than this tricky coordination between SCP and this Procedure, instead, work on
069 * returning a SCP as our subprocedure; probably needs work on the framework to do this,
070 * especially if the SCP already created.
071 */
072@InterfaceAudience.Private
073public class UnassignProcedure extends RegionTransitionProcedure {
074  private static final Logger LOG = LoggerFactory.getLogger(UnassignProcedure.class);
075
076  /**
077   * Where to send the unassign RPC.
078   */
079  protected volatile ServerName hostingServer;
080  /**
081   * The Server we will subsequently assign the region too (can be null).
082   */
083  protected volatile ServerName destinationServer;
084
085  // TODO: should this be in a reassign procedure?
086  //       ...and keep unassign for 'disable' case?
087  private boolean force;
088
089  public UnassignProcedure() {
090    // Required by the Procedure framework to create the procedure on replay
091    super();
092  }
093
094  public UnassignProcedure(final RegionInfo regionInfo,  final ServerName hostingServer,
095                           final boolean force) {
096    this(regionInfo, hostingServer, null, force);
097  }
098
099  public UnassignProcedure(final RegionInfo regionInfo,
100      final ServerName hostingServer, final ServerName destinationServer, final boolean force) {
101    super(regionInfo);
102    this.hostingServer = hostingServer;
103    this.destinationServer = destinationServer;
104    this.force = force;
105
106    // we don't need REGION_TRANSITION_QUEUE, we jump directly to sending the request
107    setTransitionState(RegionTransitionState.REGION_TRANSITION_DISPATCH);
108  }
109
110  @Override
111  public TableOperationType getTableOperationType() {
112    return TableOperationType.REGION_UNASSIGN;
113  }
114
115  @Override
116  protected boolean isRollbackSupported(final RegionTransitionState state) {
117    switch (state) {
118      case REGION_TRANSITION_QUEUE:
119      case REGION_TRANSITION_DISPATCH:
120        return true;
121      default:
122        return false;
123    }
124  }
125
126  @Override
127  protected void serializeStateData(ProcedureStateSerializer serializer)
128      throws IOException {
129    UnassignRegionStateData.Builder state = UnassignRegionStateData.newBuilder()
130        .setTransitionState(getTransitionState())
131        .setHostingServer(ProtobufUtil.toServerName(this.hostingServer))
132        .setRegionInfo(ProtobufUtil.toRegionInfo(getRegionInfo()));
133    if (this.destinationServer != null) {
134      state.setDestinationServer(ProtobufUtil.toServerName(destinationServer));
135    }
136    if (force) {
137      state.setForce(true);
138    }
139    serializer.serialize(state.build());
140  }
141
142  @Override
143  protected void deserializeStateData(ProcedureStateSerializer serializer)
144      throws IOException {
145    final UnassignRegionStateData state =
146        serializer.deserialize(UnassignRegionStateData.class);
147    setTransitionState(state.getTransitionState());
148    setRegionInfo(ProtobufUtil.toRegionInfo(state.getRegionInfo()));
149    this.hostingServer = ProtobufUtil.toServerName(state.getHostingServer());
150    force = state.getForce();
151    if (state.hasDestinationServer()) {
152      this.destinationServer = ProtobufUtil.toServerName(state.getDestinationServer());
153    }
154  }
155
156  @Override
157  protected boolean startTransition(final MasterProcedureEnv env, final RegionStateNode regionNode) {
158    // nothing to do here. we skip the step in the constructor
159    // by jumping to REGION_TRANSITION_DISPATCH
160    throw new UnsupportedOperationException();
161  }
162
163  @Override
164  protected boolean updateTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
165        throws IOException {
166    // if the region is already closed or offline we can't do much...
167    if (regionNode.isInState(State.CLOSED, State.OFFLINE)) {
168      LOG.info("Not unassigned " + this + "; " + regionNode.toShortString());
169      return false;
170    }
171
172    // if we haven't started the operation yet, we can abort
173    if (aborted.get() && regionNode.isInState(State.OPEN)) {
174      setAbortFailure(getClass().getSimpleName(), "abort requested");
175      return false;
176    }
177
178    // Mark the region as CLOSING.
179    env.getAssignmentManager().markRegionAsClosing(regionNode);
180
181    // Add the close region operation the the server dispatch queue.
182    if (!addToRemoteDispatcher(env, regionNode.getRegionLocation())) {
183      // If addToRemoteDispatcher fails, it calls the callback #remoteCallFailed.
184    }
185
186    // Return true to keep the procedure running.
187    return true;
188  }
189
190  @Override
191  protected void finishTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
192      throws IOException {
193    env.getAssignmentManager().markRegionAsClosed(regionNode);
194  }
195
196  @Override
197  public RemoteOperation remoteCallBuild(final MasterProcedureEnv env, final ServerName serverName) {
198    assert serverName.equals(getRegionState(env).getRegionLocation());
199    return new RegionCloseOperation(this, getRegionInfo(), this.destinationServer);
200  }
201
202  @Override
203  protected void reportTransition(final MasterProcedureEnv env, final RegionStateNode regionNode,
204      final TransitionCode code, final long seqId) throws UnexpectedStateException {
205    switch (code) {
206      case CLOSED:
207        setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
208        break;
209      default:
210        throw new UnexpectedStateException(String.format(
211          "Received report unexpected transition state=%s for region=%s server=%s, expected CLOSED.",
212          code, regionNode.getRegionInfo(), regionNode.getRegionLocation()));
213    }
214  }
215
216  @Override
217  protected boolean remoteCallFailed(final MasterProcedureEnv env, final RegionStateNode regionNode,
218      final IOException exception) {
219    // TODO: Is there on-going rpc to cleanup?
220    if (exception instanceof ServerCrashException) {
221      // This exception comes from ServerCrashProcedure after log splitting.
222      // SCP found this region as a RIT. Its call into here says it is ok to let this procedure go
223      // on to a complete close now. This will release lock on this region so subsequent action on
224      // region can succeed; e.g. the assign that follows this unassign when a move (w/o wait on SCP
225      // the assign could run w/o logs being split so data loss).
226      try {
227        reportTransition(env, regionNode, TransitionCode.CLOSED, HConstants.NO_SEQNUM);
228      } catch (UnexpectedStateException e) {
229        // Should never happen.
230        throw new RuntimeException(e);
231      }
232    } else if (exception instanceof RegionServerAbortedException ||
233        exception instanceof RegionServerStoppedException ||
234        exception instanceof ServerNotRunningYetException) {
235      // TODO
236      // RS is aborting, we cannot offline the region since the region may need to do WAL
237      // recovery. Until we see the RS expiration, we should retry.
238      // TODO: This should be suspend like the below where we call expire on server?
239      LOG.info("Ignoring; waiting on ServerCrashProcedure", exception);
240    } else if (exception instanceof NotServingRegionException) {
241      LOG.info("IS THIS OK? ANY LOGS TO REPLAY; ACTING AS THOUGH ALL GOOD " + regionNode,
242        exception);
243      setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
244    } else {
245      LOG.warn("Expiring server " + this + "; " + regionNode.toShortString() +
246        ", exception=" + exception);
247      env.getMasterServices().getServerManager().expireServer(regionNode.getRegionLocation());
248      // Return false so this procedure stays in suspended state. It will be woken up by a
249      // ServerCrashProcedure when it notices this RIT.
250      // TODO: Add a SCP as a new subprocedure that we now come to depend on.
251      return false;
252    }
253    return true;
254  }
255
256  @Override
257  public void toStringClassDetails(StringBuilder sb) {
258    super.toStringClassDetails(sb);
259    sb.append(", server=").append(this.hostingServer);
260  }
261
262  @Override
263  public ServerName getServer(final MasterProcedureEnv env) {
264    return this.hostingServer;
265  }
266
267  @Override
268  protected ProcedureMetrics getProcedureMetrics(MasterProcedureEnv env) {
269    return env.getAssignmentManager().getAssignmentManagerMetrics().getUnassignProcMetrics();
270  }
