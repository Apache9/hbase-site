050/**
051 * Procedure that describe the unassignment of a single region.
052 * There can only be one RegionTransitionProcedure per region running at the time,
053 * since each procedure takes a lock on the region.
054 *
055 * <p>The Unassign starts by placing a "close region" request in the Remote Dispatcher
056 * queue, and the procedure will then go into a "waiting state".
057 * The Remote Dispatcher will batch the various requests for that server and
058 * they will be sent to the RS for execution.
059 * The RS will complete the open operation by calling master.reportRegionStateTransition().
060 * The AM will intercept the transition report, and notify the procedure.
061 * The procedure will finish the unassign by publishing its new state on meta
062 * or it will retry the unassign.
063 */
064@InterfaceAudience.Private
065public class UnassignProcedure extends RegionTransitionProcedure {
066  private static final Log LOG = LogFactory.getLog(UnassignProcedure.class);
067
068  /**
069   * Where to send the unassign RPC.
070   */
071  protected volatile ServerName hostingServer;
072  /**
073   * The Server we will subsequently assign the region too (can be null).
074   */
075  protected volatile ServerName destinationServer;
076
077  protected final AtomicBoolean serverCrashed = new AtomicBoolean(false);
078
079  // TODO: should this be in a reassign procedure?
080  //       ...and keep unassign for 'disable' case?
081  private boolean force;
082
083  public UnassignProcedure() {
084    // Required by the Procedure framework to create the procedure on replay
085    super();
086  }
087
088  public UnassignProcedure(final HRegionInfo regionInfo,  final ServerName hostingServer,
089                           final boolean force) {
090    this(regionInfo, hostingServer, null, force);
091  }
092
093  public UnassignProcedure(final HRegionInfo regionInfo,
094      final ServerName hostingServer, final ServerName destinationServer, final boolean force) {
095    super(regionInfo);
096    this.hostingServer = hostingServer;
097    this.destinationServer = destinationServer;
098    this.force = force;
099
100    // we don't need REGION_TRANSITION_QUEUE, we jump directly to sending the request
101    setTransitionState(RegionTransitionState.REGION_TRANSITION_DISPATCH);
102  }
103
104  @Override
105  public TableOperationType getTableOperationType() {
106    return TableOperationType.REGION_UNASSIGN;
107  }
108
109  @Override
110  protected boolean isRollbackSupported(final RegionTransitionState state) {
111    switch (state) {
112      case REGION_TRANSITION_QUEUE:
113      case REGION_TRANSITION_DISPATCH:
114        return true;
115      default:
116        return false;
117    }
118  }
119
120  @Override
121  public void serializeStateData(final OutputStream stream) throws IOException {
122    UnassignRegionStateData.Builder state = UnassignRegionStateData.newBuilder()
123        .setTransitionState(getTransitionState())
124        .setHostingServer(ProtobufUtil.toServerName(this.hostingServer))
125        .setRegionInfo(HRegionInfo.convert(getRegionInfo()));
126    if (this.destinationServer != null) {
127      state.setDestinationServer(ProtobufUtil.toServerName(destinationServer));
128    }
129    if (force) {
130      state.setForce(true);
131    }
132    state.build().writeDelimitedTo(stream);
133  }
134
135  @Override
136  public void deserializeStateData(final InputStream stream) throws IOException {
137    final UnassignRegionStateData state = UnassignRegionStateData.parseDelimitedFrom(stream);
138    setTransitionState(state.getTransitionState());
139    setRegionInfo(HRegionInfo.convert(state.getRegionInfo()));
140    this.hostingServer = ProtobufUtil.toServerName(state.getHostingServer());
141    force = state.getForce();
142    if (state.hasDestinationServer()) {
143      this.destinationServer = ProtobufUtil.toServerName(state.getDestinationServer());
144    }
145  }
146
147  @Override
148  protected boolean startTransition(final MasterProcedureEnv env, final RegionStateNode regionNode) {
149    // nothing to do here. we skip the step in the constructor
150    // by jumping to REGION_TRANSITION_DISPATCH
151    throw new UnsupportedOperationException();
152  }
153
154  @Override
155  protected boolean updateTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
156        throws IOException {
157    // if the region is already closed or offline we can't do much...
158    if (regionNode.isInState(State.CLOSED, State.OFFLINE)) {
159      LOG.info("Not unassigned " + this + "; " + regionNode.toShortString());
160      return false;
161    }
162
163    // if the server is down, mark the operation as complete
164    if (serverCrashed.get() || !isServerOnline(env, regionNode)) {
165      LOG.info("Server already down: " + this + "; " + regionNode.toShortString());
166      return false;
167    }
168
169    // if we haven't started the operation yet, we can abort
170    if (aborted.get() && regionNode.isInState(State.OPEN)) {
171      setAbortFailure(getClass().getSimpleName(), "abort requested");
172      return false;
173    }
174
175    // Mark the region as CLOSING.
176    env.getAssignmentManager().markRegionAsClosing(regionNode);
177
178    // Add the close region operation the the server dispatch queue.
179    if (!addToRemoteDispatcher(env, regionNode.getRegionLocation())) {
180      // If addToRemoteDispatcher fails, it calls #remoteCallFailed which
181      // does all cleanup.
182    }
183
184    // We always return true, even if we fail dispatch because addToRemoteDispatcher
185    // failure processing sets state back to REGION_TRANSITION_QUEUE so we try again;
186    // i.e. return true to keep the Procedure running; it has been reset to startover.
187    return true;
188  }
189
190  @Override
191  protected void finishTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
192      throws IOException {
193    env.getAssignmentManager().markRegionAsClosed(regionNode);
194  }
195
196  @Override
197  public RemoteOperation remoteCallBuild(final MasterProcedureEnv env, final ServerName serverName) {
198    assert serverName.equals(getRegionState(env).getRegionLocation());
199    return new RegionCloseOperation(this, getRegionInfo(), this.destinationServer);
200  }
201
202  @Override
203  protected void reportTransition(final MasterProcedureEnv env, final RegionStateNode regionNode,
204      final TransitionCode code, final long seqId) throws UnexpectedStateException {
205    switch (code) {
206      case CLOSED:
207        setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
208        break;
209      default:
210        throw new UnexpectedStateException(String.format(
211          "Received report unexpected transition state=%s for region=%s server=%s, expected CLOSED.",
212          code, regionNode.getRegionInfo(), regionNode.getRegionLocation()));
213    }
214  }
215
216  @Override
217  protected void remoteCallFailed(final MasterProcedureEnv env, final RegionStateNode regionNode,
218      final IOException exception) {
219    // TODO: Is there on-going rpc to cleanup?
220    if (exception instanceof ServerCrashException) {
221      // This exception comes from ServerCrashProcedure after log splitting.
222      // It is ok to let this procedure go on to complete close now.
223      // This will release lock on this region so the subsequent assign can succeed.
224      try {
225        reportTransition(env, regionNode, TransitionCode.CLOSED, HConstants.NO_SEQNUM);
226      } catch (UnexpectedStateException e) {
227        // Should never happen.
228        throw new RuntimeException(e);
229      }
230    } else if (exception instanceof RegionServerAbortedException ||
231        exception instanceof RegionServerStoppedException ||
232        exception instanceof ServerNotRunningYetException) {
233      // TODO
234      // RS is aborting, we cannot offline the region since the region may need to do WAL
235      // recovery. Until we see the RS expiration, we should retry.
236      LOG.info("Ignoring; waiting on ServerCrashProcedure", exception);
237      // serverCrashed.set(true);
238    } else if (exception instanceof NotServingRegionException) {
239      LOG.info("IS THIS OK? ANY LOGS TO REPLAY; ACTING AS THOUGH ALL GOOD " + regionNode, exception);
240      setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
241    } else {
242      // TODO: kill the server in case we get an exception we are not able to handle
243      LOG.warn("Killing server; unexpected exception; " +
244          this + "; " + regionNode.toShortString() +
245        " exception=" + exception);
246      env.getMasterServices().getServerManager().expireServer(regionNode.getRegionLocation());
247      serverCrashed.set(true);
248    }
249  }
250
251  @Override
252  public void toStringClassDetails(StringBuilder sb) {
253    super.toStringClassDetails(sb);
254    sb.append(", server=").append(this.hostingServer);
255  }
256
257  @Override
258  public ServerName getServer(final MasterProcedureEnv env) {
259    return this.hostingServer;
260  }
