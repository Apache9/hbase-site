048/**
049 * Procedure that describe the assignment of a single region.
050 * There can only be one RegionTransitionProcedure per region running at a time
051 * since each procedure takes a lock on the region.
052 *
053 * <p>The Assign starts by pushing the "assign" operation to the AssignmentManager
054 * and then will go in a "waiting" state.
055 * The AM will batch the "assign" requests and ask the Balancer where to put
056 * the region (the various policies will be respected: retain, round-robin, random).
057 * Once the AM and the balancer have found a place for the region the procedure
058 * will be resumed and an "open region" request will be placed in the Remote Dispatcher
059 * queue, and the procedure once again will go in a "waiting state".
060 * The Remote Dispatcher will batch the various requests for that server and
061 * they will be sent to the RS for execution.
062 * The RS will complete the open operation by calling master.reportRegionStateTransition().
063 * The AM will intercept the transition report, and notify the procedure.
064 * The procedure will finish the assignment by publishing to new state on meta
065 * or it will retry the assignment.
066 *
067 * <p>This procedure does not rollback when beyond the first
068 * REGION_TRANSITION_QUEUE step; it will press on trying to assign in the face of
069 * failure. Should we ignore rollback calls to Assign/Unassign then? Or just
070 * remove rollback here?
071 */
072@InterfaceAudience.Private
073public class AssignProcedure extends RegionTransitionProcedure {
074  private static final Logger LOG = LoggerFactory.getLogger(AssignProcedure.class);
075
076  /**
077   * Set to true when we need recalibrate -- choose a new target -- because original assign failed.
078   */
079  private boolean forceNewPlan = false;
080
081  /**
082   * Gets set as desired target on move, merge, etc., when we want to go to a particular server.
083   * We may not be able to respect this request but will try. When it is NOT set, then we ask
084   * the balancer to assign. This value is used below in startTransition to set regionLocation if
085   * non-null. Setting regionLocation in regionServerNode is how we override balancer setting
086   * destination.
087   */
088  protected volatile ServerName targetServer;
089
090  /**
091   * Comparator that will sort AssignProcedures so meta assigns come first, then system table
092   * assigns and finally user space assigns.
093   */
094  public static final CompareAssignProcedure COMPARATOR = new CompareAssignProcedure();
095

126  @Override
127  protected void serializeStateData(ProcedureStateSerializer serializer)
128      throws IOException {
129    final AssignRegionStateData.Builder state = AssignRegionStateData.newBuilder()
130        .setTransitionState(getTransitionState())
131        .setRegionInfo(ProtobufUtil.toRegionInfo(getRegionInfo()));
132    if (forceNewPlan) {
133      state.setForceNewPlan(true);
134    }
135    if (this.targetServer != null) {
136      state.setTargetServer(ProtobufUtil.toServerName(this.targetServer));
137    }
138    serializer.serialize(state.build());
139  }
140
141  @Override
142  protected void deserializeStateData(ProcedureStateSerializer serializer)
143      throws IOException {
144    final AssignRegionStateData state = serializer.deserialize(AssignRegionStateData.class);
145    setTransitionState(state.getTransitionState());
146    setRegionInfo(ProtobufUtil.toRegionInfo(state.getRegionInfo()));
147    forceNewPlan = state.getForceNewPlan();
148    if (state.hasTargetServer()) {
149      this.targetServer = ProtobufUtil.toServerName(state.getTargetServer());
150    }
151  }

153  @Override
154  protected boolean startTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
155      throws IOException {
156    // If the region is already open we can't do much...
157    if (regionNode.isInState(State.OPEN) && isServerOnline(env, regionNode)) {
158      LOG.info("Assigned, not reassigning; " + this + "; " + regionNode.toShortString());
159      return false;
160    }
161    // Don't assign if table is in disabling or disabled state.
162    TableStateManager tsm = env.getMasterServices().getTableStateManager();
163    TableName tn = regionNode.getRegionInfo().getTable();
164    if (tsm.isTableState(tn, TableState.State.DISABLING, TableState.State.DISABLED)) {
165      LOG.info("Table " + tn + " state=" + tsm.getTableState(tn) + ", skipping " + this);
166      return false;
167    }
168    // If the region is SPLIT, we can't assign it. But state might be CLOSED, rather than
169    // SPLIT which is what a region gets set to when unassigned as part of SPLIT. FIX.
170    if (regionNode.isInState(State.SPLIT) ||
171        (regionNode.getRegionInfo().isOffline() && regionNode.getRegionInfo().isSplit())) {
172      LOG.info("SPLIT, cannot be assigned; " + this + "; " + regionNode +
173        "; hri=" + regionNode.getRegionInfo());
174      return false;
175    }
176
177    // If we haven't started the operation yet, we can abort
178    if (aborted.get() && regionNode.isInState(State.CLOSED, State.OFFLINE)) {
179      if (incrementAndCheckMaxAttempts(env, regionNode)) {
180        regionNode.setState(State.FAILED_OPEN);
181        setFailure(getClass().getSimpleName(),
182          new RetriesExhaustedException("Max attempts exceeded"));
183      } else {
184        setAbortFailure(getClass().getSimpleName(), "Abort requested");
185      }
186      return false;
187    }
188
189    // Send assign (add into assign-pool). Region is now in OFFLINE state. Setting offline state
190    // scrubs what was the old region location. Setting a new regionLocation here is how we retain
191    // old assignment or specify target server if a move or merge. See
192    // AssignmentManager#processAssignQueue. Otherwise, balancer gives us location.
193    ServerName lastRegionLocation = regionNode.offline();
194    boolean retain = false;
195    if (!forceNewPlan) {
196      if (this.targetServer != null) {
197        retain = targetServer.equals(lastRegionLocation);
198        regionNode.setRegionLocation(targetServer);
199      } else {
200        if (lastRegionLocation != null) {
201          // Try and keep the location we had before we offlined.
202          retain = true;
203          regionNode.setRegionLocation(lastRegionLocation);
204        } else if (regionNode.getLastHost() != null) {
205          retain = true;
206          LOG.info("Setting lastHost as the region location " + regionNode.getLastHost());
207          regionNode.setRegionLocation(regionNode.getLastHost());
208        }
209      }
210    }
211    LOG.info("Start " + this + "; " + regionNode.toShortString() +
212        "; forceNewPlan=" + this.forceNewPlan +
213        ", retain=" + retain);
214    env.getAssignmentManager().queueAssign(regionNode);
215    return true;
216  }
217
218  @Override
219  protected boolean updateTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
220  throws IOException, ProcedureSuspendedException {
221    // TODO: crash if destinationServer is specified and not online
222    // which is also the case when the balancer provided us with a different location.
223    if (LOG.isTraceEnabled()) {
224      LOG.trace("Update " + this + "; " + regionNode.toShortString());
225    }
226    if (regionNode.getRegionLocation() == null) {
227      setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
228      return true;
229    }
230
231    if (!isServerOnline(env, regionNode)) {
232      // TODO: is this correct? should we wait the chore/ssh?
233      LOG.info("Server not online, re-queuing " + this + "; " + regionNode.toShortString());
234      setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
235      return true;
236    }
237
238    if (env.getAssignmentManager().waitServerReportEvent(regionNode.getRegionLocation(), this)) {
239      LOG.info("Early suspend! " + this + "; " + regionNode.toShortString());
240      throw new ProcedureSuspendedException();
241    }
242
243    if (regionNode.isInState(State.OPEN)) {
244      LOG.info("Already assigned: " + this + "; " + regionNode.toShortString());
245      return false;
246    }
247
248    // Transition regionNode State. Set it to OPENING. Update hbase:meta, and add
249    // region to list of regions on the target regionserver. Need to UNDO if failure!
250    env.getAssignmentManager().markRegionAsOpening(regionNode);
251
252    // TODO: Requires a migration to be open by the RS?
253    // regionNode.getFormatVersion()
254
255    if (!addToRemoteDispatcher(env, regionNode.getRegionLocation())) {
256      // Failed the dispatch BUT addToRemoteDispatcher internally does
257      // cleanup on failure -- even the undoing of markRegionAsOpening above --
258      // so nothing more to do here; in fact we need to get out of here
259      // fast since we've been put back on the scheduler.
260    }
261
262    // We always return true, even if we fail dispatch because addToRemoteDispatcher
263    // failure processing sets state back to REGION_TRANSITION_QUEUE so we try again;
264    // i.e. return true to keep the Procedure running; it has been reset to startover.
265    return true;
266  }
267
268  @Override
269  protected void finishTransition(final MasterProcedureEnv env, final RegionStateNode regionNode)
270      throws IOException {
271    env.getAssignmentManager().markRegionAsOpened(regionNode);
272    // This success may have been after we failed open a few times. Be sure to cleanup any
273    // failed open references. See #incrementAndCheckMaxAttempts and where it is called.
274    env.getAssignmentManager().getRegionStates().removeFromFailedOpen(regionNode.getRegionInfo());
275  }

277  @Override
278  protected void reportTransition(final MasterProcedureEnv env, final RegionStateNode regionNode,
279      final TransitionCode code, final long openSeqNum) throws UnexpectedStateException {
280    switch (code) {
281      case OPENED:
282        if (openSeqNum < 0) {
283          throw new UnexpectedStateException("Received report unexpected " + code +
284              " transition openSeqNum=" + openSeqNum + ", " + regionNode);
285        }
286        if (openSeqNum < regionNode.getOpenSeqNum()) {
287          LOG.warn("Skipping update of open seqnum with " + openSeqNum +
288              " because current seqnum=" + regionNode.getOpenSeqNum());
289        }
290        regionNode.setOpenSeqNum(openSeqNum);
291        // Leave the state here as OPENING for now. We set it to OPEN in
292        // REGION_TRANSITION_FINISH section where we do a bunch of checks.
293        // regionNode.setState(RegionState.State.OPEN, RegionState.State.OPENING);
294        setTransitionState(RegionTransitionState.REGION_TRANSITION_FINISH);
295        break;
296      case FAILED_OPEN:
297        handleFailure(env, regionNode);
298        break;
299      default:
300        throw new UnexpectedStateException("Received report unexpected " + code +
301            " transition openSeqNum=" + openSeqNum + ", " + regionNode.toShortString() +
302            ", " + this + ", expected OPENED or FAILED_OPEN.");
303    }
304  }

306  /**
307   * Called when dispatch or subsequent OPEN request fail. Can be run by the
308   * inline dispatch call or later by the ServerCrashProcedure. Our state is
309   * generally OPENING. Cleanup and reset to OFFLINE and put our Procedure
310   * State back to REGION_TRANSITION_QUEUE so the Assign starts over.
311   */
312  private void handleFailure(final MasterProcedureEnv env, final RegionStateNode regionNode) {
313    if (incrementAndCheckMaxAttempts(env, regionNode)) {
314      aborted.set(true);
315    }
316    this.forceNewPlan = true;
317    this.targetServer = null;
318    regionNode.offline();
319    // We were moved to OPENING state before dispatch. Undo. It is safe to call
320    // this method because it checks for OPENING first.
321    env.getAssignmentManager().undoRegionAsOpening(regionNode);
322    setTransitionState(RegionTransitionState.REGION_TRANSITION_QUEUE);
323  }
324
325  private boolean incrementAndCheckMaxAttempts(final MasterProcedureEnv env,
326      final RegionStateNode regionNode) {
327    final int retries = env.getAssignmentManager().getRegionStates().
328        addToFailedOpen(regionNode).incrementAndGetRetries();
329    int max = env.getAssignmentManager().getAssignMaxAttempts();
330    LOG.info("Retry=" + retries + " of max=" + max + "; " +
331        this + "; " + regionNode.toShortString());
332    return retries >= max;
333  }

335  @Override
336  public RemoteOperation remoteCallBuild(final MasterProcedureEnv env, final ServerName serverName) {
337    assert serverName.equals(getRegionState(env).getRegionLocation());
338    return new RegionOpenOperation(this, getRegionInfo(),
339        env.getAssignmentManager().getFavoredNodes(getRegionInfo()), false);
340  }
341
342  @Override
343  protected boolean remoteCallFailed(final MasterProcedureEnv env, final RegionStateNode regionNode,
344      final IOException exception) {
345    handleFailure(env, regionNode);
346    return true;
347  }

368  /**
369   * Sort AssignProcedures such that meta and system assigns come first before user-space assigns.
370   * Have to do it this way w/ distinct Comparator because Procedure is already Comparable on
371   * 'Env'(?).
372   */
373  public static class CompareAssignProcedure implements Comparator<AssignProcedure> {
374    @Override
375    public int compare(AssignProcedure left, AssignProcedure right) {
376      if (left.getRegionInfo().isMetaRegion()) {
377        if (right.getRegionInfo().isMetaRegion()) {
378          return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
379        }
380        return -1;
381      } else if (right.getRegionInfo().isMetaRegion()) {
382        return +1;
383      }
384      if (left.getRegionInfo().getTable().isSystemTable()) {
385        if (right.getRegionInfo().getTable().isSystemTable()) {
386          return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
387        }
388        return -1;
389      } else if (right.getRegionInfo().getTable().isSystemTable()) {
390        return +1;
391      }
392      return RegionInfo.COMPARATOR.compare(left.getRegionInfo(), right.getRegionInfo());
393    }
394  }
