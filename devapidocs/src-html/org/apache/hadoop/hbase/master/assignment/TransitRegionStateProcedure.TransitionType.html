056/**
057 * The procedure to deal with the state transition of a region. A region with a TRSP in place is
058 * called RIT, i.e, RegionInTransition.
059 * <p/>
060 * It can be used to assign/unassign/reopen/move a region, and for
061 * {@link #unassign(MasterProcedureEnv, RegionInfo)} and
062 * {@link #reopen(MasterProcedureEnv, RegionInfo)}, you do not need to specify a target server, and
063 * for {@link #assign(MasterProcedureEnv, RegionInfo, ServerName)} and
064 * {@link #move(MasterProcedureEnv, RegionInfo, ServerName)}, if you want to you can provide a
065 * target server. And for {@link #move(MasterProcedureEnv, RegionInfo, ServerName)}, if you do not
066 * specify a targetServer, we will select one randomly.
067 * <p/>
068 * <p/>
069 * The typical state transition for assigning a region is:
070 *
071 * <pre>
072 * GET_ASSIGN_CANDIDATE ------> OPEN -----> CONFIRM_OPENED
073 * </pre>
074 *
075 * Notice that, if there are failures we may go back to the {@code GET_ASSIGN_CANDIDATE} state to
076 * try again.
077 * <p/>
078 * The typical state transition for unassigning a region is:
079 *
080 * <pre>
081 * CLOSE -----> CONFIRM_CLOSED
082 * </pre>
083 *
084 * Here things go a bit different, if there are failures, especially that if there is a server
085 * crash, we will go to the {@code GET_ASSIGN_CANDIDATE} state to bring the region online first, and
086 * then go through the normal way to unassign it.
087 * <p/>
088 * The typical state transition for reopening/moving a region is:
089 *
090 * <pre>
091 * CLOSE -----> CONFIRM_CLOSED -----> GET_ASSIGN_CANDIDATE ------> OPEN -----> CONFIRM_OPENED
092 * </pre>
093 *
094 * The retry logic is the same with the above assign/unassign.
095 * <p/>
096 * Notice that, although we allow specify a target server, it just acts as a candidate, we do not
097 * guarantee that the region will finally be on the target server. If this is important for you, you
098 * should check whether the region is on the target server after the procedure is finished.
099 * <p/>
100 * When you want to schedule a TRSP, please check whether there is still one for this region, and
101 * the check should be under the RegionStateNode lock. We will remove the TRSP from a
102 * RegionStateNode when we are done, see the code in {@code reportTransition} method below. There
103 * could be at most one TRSP for a give region.
104 */

127  private void setInitalAndLastState() {
128    switch (type) {
129      case ASSIGN:
130        initialState = RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE;
131        lastState = RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_OPENED;
132        break;
133      case UNASSIGN:
134        initialState = RegionStateTransitionState.REGION_STATE_TRANSITION_CLOSE;
135        lastState = RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_CLOSED;
136        break;
137      case MOVE:
138      case REOPEN:
139        initialState = RegionStateTransitionState.REGION_STATE_TRANSITION_CLOSE;
140        lastState = RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_OPENED;
141        break;
142      default:
143        throw new IllegalArgumentException("Unknown TransitionType: " + type);
144    }
145  }

149    super(env, hri);
150    this.assignCandidate = assignCandidate;
151    this.forceNewPlan = forceNewPlan;
152    this.type = type;
153    setInitalAndLastState();

162  @Override
163  protected boolean waitInitialized(MasterProcedureEnv env) {
164    if (TableName.isMetaTableName(getTableName())) {
165      return false;
166    }
167    // First we need meta to be loaded, and second, if meta is not online then we will likely to

170    return am.waitMetaLoaded(this) || am.waitMetaAssigned(this, getRegion());

173  private void queueAssign(MasterProcedureEnv env, RegionStateNode regionNode)
174      throws ProcedureSuspendedException {
175    // Here the assumption is that, the region must be in CLOSED state, so the region location
176    // will be null. And if we fail to open the region and retry here, the forceNewPlan will be
177    // true, and also we will set the region location to null.
178    boolean retain = false;
179    if (!forceNewPlan) {
180      if (assignCandidate != null) {
181        retain = assignCandidate.equals(regionNode.getLastHost());
182        regionNode.setRegionLocation(assignCandidate);
183      } else if (regionNode.getLastHost() != null) {
184        retain = true;
185        LOG.info("Setting lastHost as the region location {}", regionNode.getLastHost());
186        regionNode.setRegionLocation(regionNode.getLastHost());
187      }
188    }
189    LOG.info("Starting {}; {}; forceNewPlan={}, retain={}", this, regionNode.toShortString(),
190      forceNewPlan, retain);
191    env.getAssignmentManager().queueAssign(regionNode);
192    setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_OPEN);
193    if (regionNode.getProcedureEvent().suspendIfNotReady(this)) {
194      throw new ProcedureSuspendedException();
195    }
196  }
197
198  private void openRegion(MasterProcedureEnv env, RegionStateNode regionNode) throws IOException {
199    ServerName loc = regionNode.getRegionLocation();
200    if (loc == null) {
201      LOG.warn("No location specified for {}, jump back to state {} to get one", getRegion(),
202        RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
203      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
204      return;
205    }
206    env.getAssignmentManager().regionOpening(regionNode);
207    addChildProcedure(new OpenRegionProcedure(getRegion(), loc));
208    setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_OPENED);
209  }

211  private Flow confirmOpened(MasterProcedureEnv env, RegionStateNode regionNode)
212      throws IOException {
213    // notice that, for normal case, if we successfully opened a region, we will not arrive here, as
214    // in reportTransition we will call unsetProcedure, and in executeFromState we will return
215    // directly. But if the master is crashed before we finish the procedure, then next time we will
216    // arrive here. So we still need to add code for normal cases.
217    if (regionNode.isInState(State.OPEN)) {
218      attempt = 0;
219      if (lastState == RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_OPENED) {
220        // we are the last state, finish
221        regionNode.unsetProcedure(this);
222        return Flow.NO_MORE_STATE;
223      }
224      // It is possible that we arrive here but confirm opened is not the last state, for example,
225      // when merging or splitting a region, we unassign the region from a RS and the RS is crashed,
226      // then there will be recovered edits for this region, we'd better make the region online
227      // again and then unassign it, otherwise we have to fail the merge/split procedure as we may
228      // loss data.
229      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_CLOSE);
230      return Flow.HAS_MORE_STATE;
231    }
232
233    if (incrementAndCheckMaxAttempts(env, regionNode)) {
234      env.getAssignmentManager().regionFailedOpen(regionNode, true);
235      setFailure(getClass().getSimpleName(), new RetriesExhaustedException(
236        "Max attempts " + env.getAssignmentManager().getAssignMaxAttempts() + " exceeded"));
237      regionNode.unsetProcedure(this);
238      return Flow.NO_MORE_STATE;
239    }
240    env.getAssignmentManager().regionFailedOpen(regionNode, false);
241    // we failed to assign the region, force a new plan
242    forceNewPlan = true;
243    regionNode.setRegionLocation(null);
244    setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
245    // Here we do not throw exception because we want to the region to be online ASAP
246    return Flow.HAS_MORE_STATE;
247  }
248
249  private void closeRegion(MasterProcedureEnv env, RegionStateNode regionNode) throws IOException {
250    if (regionNode.isInState(State.OPEN, State.CLOSING, State.MERGING, State.SPLITTING)) {
251      // this is the normal case
252      env.getAssignmentManager().regionClosing(regionNode);
253      addChildProcedure(
254        new CloseRegionProcedure(getRegion(), regionNode.getRegionLocation(), assignCandidate));
255      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_CLOSED);
256    } else {
257      forceNewPlan = true;
258      regionNode.setRegionLocation(null);
259      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
260    }
261  }
262
263  private Flow confirmClosed(MasterProcedureEnv env, RegionStateNode regionNode)
264      throws IOException {
265    // notice that, for normal case, if we successfully opened a region, we will not arrive here, as
266    // in reportTransition we will call unsetProcedure, and in executeFromState we will return
267    // directly. But if the master is crashed before we finish the procedure, then next time we will
268    // arrive here. So we still need to add code for normal cases.
269    if (regionNode.isInState(State.CLOSED)) {
270      attempt = 0;
271      if (lastState == RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_CLOSED) {
272        // we are the last state, finish
273        regionNode.unsetProcedure(this);
274        return Flow.NO_MORE_STATE;
275      }
276      // This means we need to open the region again, should be a move or reopen
277      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
278      return Flow.HAS_MORE_STATE;
279    }
280    if (regionNode.isInState(State.CLOSING)) {
281      // This is possible, think the target RS crashes and restarts immediately, the close region
282      // operation will return a NotServingRegionException soon, we can only recover after SCP takes
283      // care of this RS. So here we throw an IOException to let upper layer to retry with backoff.
284      setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_CLOSE);
285      throw new HBaseIOException("Failed to close region");
286    }
287    // abnormally closed, need to reopen it, no matter what is the last state, see the comment in
288    // confirmOpened for more details that why we need to reopen the region first even if we just
289    // want to close it.
290    // The only exception is for non-default replica, where we do not need to deal with recovered
291    // edits. Notice that the region will remain in ABNORMALLY_CLOSED state, the upper layer need to
292    // deal with this state. For non-default replica, this is usually the same with CLOSED.
293    assert regionNode.isInState(State.ABNORMALLY_CLOSED);
294    if (!RegionReplicaUtil.isDefaultReplica(getRegion()) &&
295      lastState == RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_CLOSED) {
296      regionNode.unsetProcedure(this);
297      return Flow.NO_MORE_STATE;
298    }
299    attempt = 0;
300    setNextState(RegionStateTransitionState.REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE);
301    return Flow.HAS_MORE_STATE;
302  }

306  @Override
307  protected Procedure[] execute(MasterProcedureEnv env)
308      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
309    RegionStateNode regionNode =
310      env.getAssignmentManager().getRegionStates().getOrCreateRegionStateNode(getRegion());
311    regionNode.lock();
312    try {
313      return super.execute(env);
314    } finally {
315      regionNode.unlock();
316    }
317  }

320    return env.getAssignmentManager().getRegionStates().getOrCreateRegionStateNode(getRegion());
321  }

323  @Override
324  protected Flow executeFromState(MasterProcedureEnv env, RegionStateTransitionState state)
325      throws ProcedureSuspendedException, ProcedureYieldException, InterruptedException {
326    RegionStateNode regionNode = getRegionStateNode(env);
327    if (regionNode.getProcedure() != this) {
328      // This is possible, and is the normal case, as we will call unsetProcedure in
329      // reportTransition, this means we have already done
330      // This is because that, when we mark the region as OPENED or CLOSED, then all the works
331      // should have already been done, and logically we could have another TRSP scheduled for this
332      // region immediately(think of a RS crash at the point...).
333      return Flow.NO_MORE_STATE;
334    }
335    try {
336      switch (state) {
337        case REGION_STATE_TRANSITION_GET_ASSIGN_CANDIDATE:
338          queueAssign(env, regionNode);
339          return Flow.HAS_MORE_STATE;
340        case REGION_STATE_TRANSITION_OPEN:
341          openRegion(env, regionNode);
342          return Flow.HAS_MORE_STATE;
343        case REGION_STATE_TRANSITION_CONFIRM_OPENED:
344          return confirmOpened(env, regionNode);
345        case REGION_STATE_TRANSITION_CLOSE:
346          closeRegion(env, regionNode);
347          return Flow.HAS_MORE_STATE;
348        case REGION_STATE_TRANSITION_CONFIRM_CLOSED:
349          return confirmClosed(env, regionNode);
350        default:
351          throw new UnsupportedOperationException("unhandled state=" + state);
352      }
353    } catch (IOException e) {
354      long backoff = ProcedureUtil.getBackoffTimeMs(this.attempt++);
355      LOG.warn(
356        "Failed transition, suspend {}secs {}; {}; waiting on rectified condition fixed " +
357          "by other Procedure or operator intervention",
358        backoff / 1000, this, regionNode.toShortString(), e);
359      setTimeout(Math.toIntExact(backoff));
360      setState(ProcedureProtos.ProcedureState.WAITING_TIMEOUT);
361      skipPersistence();
362      throw new ProcedureSuspendedException();
363    }
364  }

376  private boolean isOpening(RegionStateNode regionNode, ServerName serverName,
377      TransitionCode code) {
378    if (!regionNode.isInState(State.OPENING)) {
379      LOG.warn("Received report {} transition from {} for {}, pid={}, but the region is not in" +
380        " OPENING state, should be a retry, ignore", code, serverName, regionNode, getProcId());
381      return false;
382    }
383    if (getCurrentState() != REGION_STATE_TRANSITION_CONFIRM_OPENED) {
384      LOG.warn(
385        "Received report {} transition from {} for {}, pid={}," +
386          " but the TRSP is not in {} state, should be a retry, ignore",
387        code, serverName, regionNode, getProcId(), REGION_STATE_TRANSITION_CONFIRM_OPENED);
388      return false;
389    }
390    return true;
391  }

393  private void reportTransitionOpen(MasterProcedureEnv env, RegionStateNode regionNode,
394      ServerName serverName, long openSeqNum) throws IOException {
395    if (!isOpening(regionNode, serverName, TransitionCode.OPENED)) {
396      return;
397    }
398    if (openSeqNum < 0) {
399      throw new UnexpectedStateException("Received report unexpected " + TransitionCode.OPENED +
400        " transition openSeqNum=" + openSeqNum + ", " + regionNode + ", proc=" + this);
401    }
402    if (openSeqNum < regionNode.getOpenSeqNum()) {
403      // use the openSeqNum as a fence, if this is not a retry, then the openSeqNum should be
404      // greater than or equal to the existing one.
405      LOG.warn(
406        "Received report {} transition from {} for {}, pid={} but the new openSeqNum {}" +
407          " is less than the current one {}, should be a retry, ignore",
408        TransitionCode.OPENED, serverName, regionNode, getProcId(), openSeqNum,
409        regionNode.getOpenSeqNum());
410      return;
411    }
412    // notice that it is possible for a region to still have the same openSeqNum if it crashes and
413    // we haven't written anything into it. That's why we can not just change the above condition
414    // from '<' to '<='. So here we still need to check whether the serverName
415    // matches, to determine whether this is a retry when the openSeqNum is not changed.
416    if (!regionNode.getRegionLocation().equals(serverName)) {
417      LOG.warn("Received report {} transition from {} for {}, pid={} but the region is not on it," +
418        " should be a retry, ignore", TransitionCode.OPENED, serverName, regionNode, getProcId());
419      return;
420    }
421    regionNode.setOpenSeqNum(openSeqNum);
422    env.getAssignmentManager().regionOpened(regionNode);
423    if (lastState == RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_OPENED) {
424      // if parent procedure is ServerCrashProcedure, update progress
425      ServerCrashProcedure.updateProgress(env, getParentProcId());
426      // we are done
427      regionNode.unsetProcedure(this);
428    }
429    regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
430  }
431
432  private void reportTransitionFailedOpen(MasterProcedureEnv env, RegionStateNode regionNode,
433      ServerName serverName) {
434    if (!isOpening(regionNode, serverName, TransitionCode.FAILED_OPEN)) {
435      return;
436    }
437    // there is no openSeqNum for FAILED_OPEN, so we will check the target server instead
438    if (!regionNode.getRegionLocation().equals(serverName)) {
439      LOG.warn(
440        "Received report {} transition from {} for {}, pid={}," +
441          " but the region is not on it, should be a retry, ignore",
442        TransitionCode.FAILED_OPEN, regionNode, serverName, getProcId());
443      return;
444    }
445    // just wake up the procedure and see if we can retry
446    // Notice that, even if we arrive here, this call could still be a retry, as we may retry
447    // opening on the same server again. And the assumption here is that, once the region state is
448    // OPENING, and the TRSP state is REGION_STATE_TRANSITION_CONFIRM_OPENED, the TRSP must have
449    // been suspended on the procedure event, so after the waking operation here, the TRSP will be
450    // executed and try to schedule new OpenRegionProcedure again. Once there is a successful open
451    // then we are done, so the TRSP will not be stuck.
452    // TODO: maybe we could send the procedure id of the OpenRegionProcedure to the region server
453    // and let the region server send it back when done, so it will be easy to detect whether this
454    // is a retry.
455    regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
456  }

458  // we do not need seqId for closing a region
459  private void reportTransitionClosed(MasterProcedureEnv env, RegionStateNode regionNode,
460      ServerName serverName) throws IOException {
461    if (!regionNode.isInState(State.CLOSING)) {
462      LOG.warn(
463        "Received report {} transition from {} for {}, pid={}" +
464          ", but the region is not in CLOSING state, should be a retry, ignore",
465        TransitionCode.CLOSED, serverName, regionNode, getProcId());
466      return;
467    }
468    if (getCurrentState() != REGION_STATE_TRANSITION_CONFIRM_CLOSED) {
469      LOG.warn(
470        "Received report {} transition from {} for {}, pid={} but the proc is not in {}" +
471          " state, should be a retry, ignore",
472        TransitionCode.CLOSED, serverName, regionNode, getProcId(),
473        REGION_STATE_TRANSITION_CONFIRM_CLOSED);
474      return;
475    }
476    if (!regionNode.getRegionLocation().equals(serverName)) {
477      LOG.warn(
478        "Received report {} transition from {} for {}, pid={}," +
479          " but the region is not on it, should be a retry, ignore",
480        TransitionCode.CLOSED, serverName, regionNode, getProcId());
481      return;
482    }
483    env.getAssignmentManager().regionClosed(regionNode, true);
484    if (lastState == RegionStateTransitionState.REGION_STATE_TRANSITION_CONFIRM_CLOSED) {
485      // we are done
486      regionNode.unsetProcedure(this);
487    }
488    regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
489  }

491  // Should be called with RegionStateNode locked
492  public void reportTransition(MasterProcedureEnv env, RegionStateNode regionNode,
493      ServerName serverName, TransitionCode code, long seqId) throws IOException {
494    // It is possible that the previous reportRegionStateTransition call was succeeded at master
495    // side, but before returning the result to region server, the rpc connection was broken, or the
496    // master restarted. The region server will try calling reportRegionStateTransition again under
497    // this scenario, so here we need to check whether this is a retry.
498    switch (code) {
499      case OPENED:
500        reportTransitionOpen(env, regionNode, serverName, seqId);
501        break;
502      case FAILED_OPEN:
503        reportTransitionFailedOpen(env, regionNode, serverName);
504        break;
505      case CLOSED:
506        reportTransitionClosed(env, regionNode, serverName);
507        break;
508      default:
509        throw new UnexpectedStateException("Received report unexpected " + code + " transition, " +
510          regionNode.toShortString() + ", " + this + ", expected OPENED or FAILED_OPEN or CLOSED.");
511    }
512  }

514  // Should be called with RegionStateNode locked
515  public void serverCrashed(MasterProcedureEnv env, RegionStateNode regionNode,
516      ServerName serverName) throws IOException {
517    // Notice that, in this method, we do not change the procedure state, instead, we update the
518    // region state in hbase:meta. This is because that, the procedure state change will not be
519    // persisted until the region is woken up and finish one step, if we crash before that then the
520    // information will be lost. So here we will update the region state in hbase:meta, and when the
521    // procedure is woken up, it will process the error and jump to the correct procedure state.
522    RegionStateTransitionState currentState = getCurrentState();
523    switch (currentState) {
524      case REGION_STATE_TRANSITION_CLOSE:
525      case REGION_STATE_TRANSITION_CONFIRM_CLOSED:
526      case REGION_STATE_TRANSITION_CONFIRM_OPENED:
527        // for these 3 states, the region may still be online on the crashed server
528        env.getAssignmentManager().regionClosed(regionNode, false);
529        if (currentState != RegionStateTransitionState.REGION_STATE_TRANSITION_CLOSE) {
530          regionNode.getProcedureEvent().wake(env.getProcedureScheduler());
531        }
532        break;
533      default:
534        // If the procedure is in other 2 states, then actually we should not arrive here, as we
535        // know that the region is not online on any server, so we need to do nothing... But anyway
536        // let's add a log here
537        LOG.warn("{} received unexpected server crash call for region {} from {}", this, regionNode,
538          serverName);
539
540    }
541  }

543  private boolean incrementAndCheckMaxAttempts(MasterProcedureEnv env, RegionStateNode regionNode) {
544    int retries = env.getAssignmentManager().getRegionStates().addToFailedOpen(regionNode)
545      .incrementAndGetRetries();
546    int max = env.getAssignmentManager().getAssignMaxAttempts();
547    LOG.info("Retry={} of max={}; {}; {}", retries, max, this, regionNode.toShortString());
548    return retries >= max;
549  }

603  @Override
604  protected void serializeStateData(ProcedureStateSerializer serializer) throws IOException {
605    super.serializeStateData(serializer);
606    RegionStateTransitionStateData.Builder builder = RegionStateTransitionStateData.newBuilder()
607      .setType(convert(type)).setForceNewPlan(forceNewPlan);
608    if (assignCandidate != null) {
609      builder.setAssignCandidate(ProtobufUtil.toServerName(assignCandidate));
610    }
611    serializer.serialize(builder.build());
612  }
613
614  @Override
615  protected void deserializeStateData(ProcedureStateSerializer serializer) throws IOException {
616    super.deserializeStateData(serializer);
617    RegionStateTransitionStateData data =
618      serializer.deserialize(RegionStateTransitionStateData.class);
619    type = convert(data.getType());
620    setInitalAndLastState();
621    forceNewPlan = data.getForceNewPlan();
622    if (data.hasAssignCandidate()) {
623      assignCandidate = ProtobufUtil.toServerName(data.getAssignCandidate());
624    }
625  }
