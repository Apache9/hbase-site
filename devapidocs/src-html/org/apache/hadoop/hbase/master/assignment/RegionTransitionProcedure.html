042/**
043 * Base class for the Assign and Unassign Procedure.
044 * There can only be one RegionTransitionProcedure per region running at a time
045 * since each procedure takes a lock on the region (see MasterProcedureScheduler).
046 *
047 * <p>This procedure is asynchronous and responds to external events.
048 * The AssignmentManager will notify this procedure when the RS completes
049 * the operation and reports the transitioned state
050 * (see the Assign and Unassign class for more detail).
051 * <p>Procedures move from the REGION_TRANSITION_QUEUE state when they are
052 * first submitted, to the REGION_TRANSITION_DISPATCH state when the request
053 * to remote server is sent and the Procedure is suspended waiting on external
054 * event to be woken again. Once the external event is triggered, Procedure
055 * moves to the REGION_TRANSITION_FINISH state.
056 */
057@InterfaceAudience.Private
058public abstract class RegionTransitionProcedure
059    extends Procedure<MasterProcedureEnv>
060    implements TableProcedureInterface,
061      RemoteProcedure<MasterProcedureEnv, ServerName> {
062  private static final Log LOG = LogFactory.getLog(RegionTransitionProcedure.class);
063
064  protected final AtomicBoolean aborted = new AtomicBoolean(false);
065

067      RegionTransitionState.REGION_TRANSITION_QUEUE;

069  private volatile boolean lock = false;

149  @Override
150  public void remoteCallFailed(final MasterProcedureEnv env,
151      final ServerName serverName, final IOException exception) {
152    final RegionStateNode regionNode = getRegionState(env);
153    assert serverName.equals(regionNode.getRegionLocation());
154    String msg = exception.getMessage() == null? exception.getClass().getSimpleName():
155      exception.getMessage();
156    LOG.warn("Failed " + this + "; " + regionNode.toShortString() + "; exception=" + msg);
157    remoteCallFailed(env, regionNode, exception);
158    // NOTE: This call to wakeEvent puts this Procedure back on the scheduler.
159    // Thereafter, another Worker can be in here so DO NOT MESS WITH STATE beyond
160    // this method. Just get out of this current processing quickly.
161    env.getProcedureScheduler().wakeEvent(regionNode.getProcedureEvent());
162  }
163
164  /**
165   * Be careful! At the end of this method, the procedure has either succeeded
166   * and this procedure has been set into a suspended state OR, we failed and
167   * this procedure has been put back on the scheduler ready for another worker
168   * to pick it up. In both cases, we need to exit the current Worker processing
169   * toute de suite!
170   * @return True if we successfully dispatched the call and false if we failed;
171   * if failed, we need to roll back any setup done for the dispatch.
172   */
173  protected boolean addToRemoteDispatcher(final MasterProcedureEnv env,
174      final ServerName targetServer) {
175    assert targetServer.equals(getRegionState(env).getRegionLocation()) :
176      "targetServer=" + targetServer + " getRegionLocation=" +
177        getRegionState(env).getRegionLocation(); // TODO
178
179    LOG.info("Dispatch " + this + "; " + getRegionState(env).toShortString());
180
181    // Put this procedure into suspended mode to wait on report of state change
182    // from remote regionserver. Means Procedure associated ProcedureEvent is marked not 'ready'.
183    env.getProcedureScheduler().suspendEvent(getRegionState(env).getProcedureEvent());
184
185    // Tricky because this can fail. If it fails need to backtrack on stuff like
186    // the 'suspend' done above -- tricky as the 'wake' requeues us -- and ditto
187    // up in the caller; it needs to undo state changes.
188    if (!env.getRemoteDispatcher().addOperationToNode(targetServer, this)) {
189      remoteCallFailed(env, targetServer,
190          new FailedRemoteDispatchException(this + " to " + targetServer));
191      return false;
192    }
193    return true;
194  }
195
196  protected void reportTransition(final MasterProcedureEnv env, final ServerName serverName,
197      final TransitionCode code, final long seqId) throws UnexpectedStateException {
198    final RegionStateNode regionNode = getRegionState(env);
199    if (LOG.isDebugEnabled()) {
200      LOG.debug("Received report " + code + " seqId=" + seqId + ", " +
201            this + "; " + regionNode.toShortString());
202    }
203    if (!serverName.equals(regionNode.getRegionLocation())) {
204      if (isMeta() && regionNode.getRegionLocation() == null) {
205        regionNode.setRegionLocation(serverName);
206      } else {
207        throw new UnexpectedStateException(String.format(
208          "Unexpected state=%s from server=%s; expected server=%s; %s; %s",
209          code, serverName, regionNode.getRegionLocation(),
210          this, regionNode.toShortString()));
211      }
212    }
213
214    reportTransition(env, regionNode, code, seqId);
215
216    // NOTE: This call adds this procedure back on the scheduler.
217    // This makes it so this procedure can run again. Another worker will take
218    // processing to the next stage. At an extreme, the other worker may run in
219    // parallel so DO  NOT CHANGE any state hereafter! This should be last thing
220    // done in this processing step.
221    env.getProcedureScheduler().wakeEvent(regionNode.getProcedureEvent());
222  }

234    super.toStringState(builder);

241  @Override
242  protected Procedure[] execute(final MasterProcedureEnv env) throws ProcedureSuspendedException {
243    final AssignmentManager am = env.getAssignmentManager();
244    final RegionStateNode regionNode = getRegionState(env);
245    if (!am.addRegionInTransition(regionNode, this)) {
246      String msg = String.format(
247        "There is already another procedure running on this region this=%s owner=%s",
248        this, regionNode.getProcedure());
249      LOG.warn(msg + " " + this + "; " + regionNode.toShortString());
250      setAbortFailure(getClass().getSimpleName(), msg);
251      return null;
252    }
253    try {
254      boolean retry;
255      do {
256        retry = false;
257        switch (transitionState) {
258          case REGION_TRANSITION_QUEUE:
259            // 1. push into the AM queue for balancer policy
260            if (!startTransition(env, regionNode)) {
261              // The operation figured it is done or it aborted; check getException()
262              am.removeRegionInTransition(getRegionState(env), this);
263              return null;
264            }
265            transitionState = RegionTransitionState.REGION_TRANSITION_DISPATCH;
266            if (env.getProcedureScheduler().waitEvent(regionNode.getProcedureEvent(), this)) {
267              // Why this suspend? Because we want to ensure Store happens before proceed?
268              throw new ProcedureSuspendedException();
269            }
270            break;
271
272          case REGION_TRANSITION_DISPATCH:
273            // 2. send the request to the target server
274            if (!updateTransition(env, regionNode)) {
275              // The operation figured it is done or it aborted; check getException()
276              am.removeRegionInTransition(regionNode, this);
277              return null;
278            }
279            if (transitionState != RegionTransitionState.REGION_TRANSITION_DISPATCH) {
280              retry = true;
281              break;
282            }
283            if (env.getProcedureScheduler().waitEvent(regionNode.getProcedureEvent(), this)) {
284              throw new ProcedureSuspendedException();
285            }
286            break;
287
288          case REGION_TRANSITION_FINISH:
289            // 3. wait assignment response. completion/failure
290            finishTransition(env, regionNode);
291            am.removeRegionInTransition(regionNode, this);
292            return null;
293        }
294      } while (retry);
295    } catch (IOException e) {
296      LOG.warn("Retryable error trying to transition: " +
297          this + "; " + regionNode.toShortString(), e);
298    }
299
300    return new Procedure[] {this};
301  }
302
303  @Override
304  protected void rollback(final MasterProcedureEnv env) {
305    if (isRollbackSupported(transitionState)) {
306      // Nothing done up to this point. abort safely.
307      // This should happen when something like disableTable() is triggered.
308      env.getAssignmentManager().removeRegionInTransition(getRegionState(env), this);
309      return;
310    }
311
312    // There is no rollback for assignment unless we cancel the operation by
313    // dropping/disabling the table.
314    throw new UnsupportedOperationException("Unhandled state " + transitionState +
315        "; there is no rollback for assignment unless we cancel the operation by " +
316        "dropping/disabling the table");
317  }
318
319  protected abstract boolean isRollbackSupported(final RegionTransitionState state);
320
321  @Override
322  protected boolean abort(final MasterProcedureEnv env) {
323    if (isRollbackSupported(transitionState)) {
324      aborted.set(true);
325      return true;
326    }
327    return false;
328  }
329
330  @Override
331  protected LockState acquireLock(final MasterProcedureEnv env) {
332    // Unless we are assigning meta, wait for meta to be available and loaded.
333    if (!isMeta() && (env.waitFailoverCleanup(this) ||
334        env.getAssignmentManager().waitMetaInitialized(this, getRegionInfo()))) {
335      return LockState.LOCK_EVENT_WAIT;
336    }
337
338    // TODO: Revisit this and move it to the executor
339    if (env.getProcedureScheduler().waitRegion(this, getRegionInfo())) {
340      try {
341        LOG.debug(LockState.LOCK_EVENT_WAIT + " pid=" + getProcId() + " " +
342          env.getProcedureScheduler().dumpLocks());
343      } catch (IOException e) {
344        // TODO Auto-generated catch block
345        e.printStackTrace();
346      }
347      return LockState.LOCK_EVENT_WAIT;
348    }
349    this.lock = true;
350    return LockState.LOCK_ACQUIRED;
351  }
352
353  @Override
354  protected void releaseLock(final MasterProcedureEnv env) {
355    env.getProcedureScheduler().wakeRegion(this, getRegionInfo());
356    lock = false;
357  }
