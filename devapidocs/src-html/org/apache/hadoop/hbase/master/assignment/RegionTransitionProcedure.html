042/**
043 * Base class for the Assign and Unassign Procedure.
044 * There can only be one RegionTransitionProcedure per region running at a time
045 * since each procedure takes a lock on the region (see MasterProcedureScheduler).
046 *
047 * <p>This procedure is asynchronous and responds to external events.
048 * The AssignmentManager will notify this procedure when the RS completes
049 * the operation and reports the transitioned state
050 * (see the Assign and Unassign class for more detail).
051 *
052 * <p>Procedures move from the REGION_TRANSITION_QUEUE state when they are
053 * first submitted, to the REGION_TRANSITION_DISPATCH state when the request
054 * to remote server is sent and the Procedure is suspended waiting on external
055 * event to be woken again. Once the external event is triggered, Procedure
056 * moves to the REGION_TRANSITION_FINISH state.
057 *
058 * <p>NOTE: {@link AssignProcedure} and {@link UnassignProcedure} should not be thought of
059 * as being asymmetric, at least currently.
060 * <ul>
061 * <li>{@link AssignProcedure} moves through all the above described states and implements methods
062 * associated with each while {@link UnassignProcedure} starts at state
063 * REGION_TRANSITION_DISPATCH and state REGION_TRANSITION_QUEUE is not supported.</li>
064 *
065 * <li>When any step in {@link AssignProcedure} fails, failure handler
066 * AssignProcedure#handleFailure(MasterProcedureEnv, RegionStateNode) re-attempts the
067 * assignment by setting the procedure state to REGION_TRANSITION_QUEUE and forces
068 * assignment to a different target server by setting {@link AssignProcedure#forceNewPlan}. When
069 * the number of attempts reach hreshold configuration 'hbase.assignment.maximum.attempts',
070 * the procedure is aborted. For {@link UnassignProcedure}, similar re-attempts are
071 * intentionally not implemented. It is a 'one shot' procedure. See its class doc for how it
072 * handles failure.
073 * </li>
074 * </ul>
075 *
076 * <p>TODO: Considering it is a priority doing all we can to get make a region available as soon as possible,
077 * re-attempting with any target makes sense if specified target fails in case of
078 * {@link AssignProcedure}. For {@link UnassignProcedure}, our concern is preventing data loss
079 * on failed unassign. See class doc for explanation.
080 */

179  @Override
180  public void remoteCallFailed(final MasterProcedureEnv env,
181      final ServerName serverName, final IOException exception) {
182    final RegionStateNode regionNode = getRegionState(env);
183    assert serverName.equals(regionNode.getRegionLocation());
184    String msg = exception.getMessage() == null? exception.getClass().getSimpleName():
185      exception.getMessage();
186    LOG.warn("Remote call failed " + this + "; " + regionNode.toShortString() +
187      "; exception=" + msg);
188    if (remoteCallFailed(env, regionNode, exception)) {
189      // NOTE: This call to wakeEvent puts this Procedure back on the scheduler.
190      // Thereafter, another Worker can be in here so DO NOT MESS WITH STATE beyond
191      // this method. Just get out of this current processing quickly.
192      env.getProcedureScheduler().wakeEvent(regionNode.getProcedureEvent());
193    }
194    // else leave the procedure in suspended state; it is waiting on another call to this callback
195  }
196
197  /**
198   * Be careful! At the end of this method, the procedure has either succeeded
199   * and this procedure has been set into a suspended state OR, we failed and
200   * this procedure has been put back on the scheduler ready for another worker
201   * to pick it up. In both cases, we need to exit the current Worker processing
202   * toute de suite!
203   * @return True if we successfully dispatched the call and false if we failed;
204   * if failed, we need to roll back any setup done for the dispatch.
205   */
206  protected boolean addToRemoteDispatcher(final MasterProcedureEnv env,
207      final ServerName targetServer) {
208    assert targetServer.equals(getRegionState(env).getRegionLocation()) :
209      "targetServer=" + targetServer + " getRegionLocation=" +
210        getRegionState(env).getRegionLocation(); // TODO
211
212    LOG.info("Dispatch " + this + "; " + getRegionState(env).toShortString());
213
214    // Put this procedure into suspended mode to wait on report of state change
215    // from remote regionserver. Means Procedure associated ProcedureEvent is marked not 'ready'.
216    env.getProcedureScheduler().suspendEvent(getRegionState(env).getProcedureEvent());
217
218    // Tricky because the below call to addOperationToNode can fail. If it fails, we need to
219    // backtrack on stuff like the 'suspend' done above -- tricky as the 'wake' requeues us -- and
220    // ditto up in the caller; it needs to undo state changes. Inside in remoteCallFailed, it does
221    // wake to undo the above suspend.
222    if (!env.getRemoteDispatcher().addOperationToNode(targetServer, this)) {
223      remoteCallFailed(env, targetServer,
224          new FailedRemoteDispatchException(this + " to " + targetServer));
225      return false;
226    }
227    return true;
228  }
229
230  protected void reportTransition(final MasterProcedureEnv env, final ServerName serverName,
231      final TransitionCode code, final long seqId) throws UnexpectedStateException {
232    final RegionStateNode regionNode = getRegionState(env);
233    if (LOG.isDebugEnabled()) {
234      LOG.debug("Received report " + code + " seqId=" + seqId + ", " +
235            this + "; " + regionNode.toShortString());
236    }
237    if (!serverName.equals(regionNode.getRegionLocation())) {
238      if (isMeta() && regionNode.getRegionLocation() == null) {
239        regionNode.setRegionLocation(serverName);
240      } else {
241        throw new UnexpectedStateException(String.format(
242          "Unexpected state=%s from server=%s; expected server=%s; %s; %s",
243          code, serverName, regionNode.getRegionLocation(),
244          this, regionNode.toShortString()));
245      }
246    }
247
248    reportTransition(env, regionNode, code, seqId);
249
250    // NOTE: This call adds this procedure back on the scheduler.
251    // This makes it so this procedure can run again. Another worker will take
252    // processing to the next stage. At an extreme, the other worker may run in
253    // parallel so DO  NOT CHANGE any state hereafter! This should be last thing
254    // done in this processing step.
255    env.getProcedureScheduler().wakeEvent(regionNode.getProcedureEvent());
256  }

275  @Override
276  protected Procedure[] execute(final MasterProcedureEnv env) throws ProcedureSuspendedException {
277    final AssignmentManager am = env.getAssignmentManager();
278    final RegionStateNode regionNode = getRegionState(env);
279    if (!am.addRegionInTransition(regionNode, this)) {
280      String msg = String.format(
281        "There is already another procedure running on this region this=%s owner=%s",
282        this, regionNode.getProcedure());
283      LOG.warn(msg + " " + this + "; " + regionNode.toShortString());
284      setAbortFailure(getClass().getSimpleName(), msg);
285      return null;
286    }
287    try {
288      boolean retry;
289      do {
290        retry = false;
291        switch (transitionState) {
292          case REGION_TRANSITION_QUEUE:
293            // 1. push into the AM queue for balancer policy
294            if (!startTransition(env, regionNode)) {
295              // The operation figured it is done or it aborted; check getException()
296              am.removeRegionInTransition(getRegionState(env), this);
297              return null;
298            }
299            transitionState = RegionTransitionState.REGION_TRANSITION_DISPATCH;
300            if (env.getProcedureScheduler().waitEvent(regionNode.getProcedureEvent(), this)) {
301              // Why this suspend? Because we want to ensure Store happens before proceed?
302              throw new ProcedureSuspendedException();
303            }
304            break;
305
306          case REGION_TRANSITION_DISPATCH:
307            // 2. send the request to the target server
308            if (!updateTransition(env, regionNode)) {
309              // The operation figured it is done or it aborted; check getException()
310              am.removeRegionInTransition(regionNode, this);
311              return null;
312            }
313            if (transitionState != RegionTransitionState.REGION_TRANSITION_DISPATCH) {
314              retry = true;
315              break;
316            }
317            if (env.getProcedureScheduler().waitEvent(regionNode.getProcedureEvent(), this)) {
318              throw new ProcedureSuspendedException();
319            }
320            break;
321
322          case REGION_TRANSITION_FINISH:
323            // 3. wait assignment response. completion/failure
324            finishTransition(env, regionNode);
325            am.removeRegionInTransition(regionNode, this);
326            return null;
327        }
328      } while (retry);
329    } catch (IOException e) {
330      LOG.warn("Retryable error trying to transition: " +
331          this + "; " + regionNode.toShortString(), e);
332    }
333
334    return new Procedure[] {this};
335  }
336
337  @Override
338  protected void rollback(final MasterProcedureEnv env) {
339    if (isRollbackSupported(transitionState)) {
340      // Nothing done up to this point. abort safely.
341      // This should happen when something like disableTable() is triggered.
342      env.getAssignmentManager().removeRegionInTransition(getRegionState(env), this);
343      return;
344    }
345
346    // There is no rollback for assignment unless we cancel the operation by
347    // dropping/disabling the table.
348    throw new UnsupportedOperationException("Unhandled state " + transitionState +
349        "; there is no rollback for assignment unless we cancel the operation by " +
350        "dropping/disabling the table");
351  }
352
353  protected abstract boolean isRollbackSupported(final RegionTransitionState state);
354
355  @Override
356  protected boolean abort(final MasterProcedureEnv env) {
357    if (isRollbackSupported(transitionState)) {
358      aborted.set(true);
359      return true;
360    }
361    return false;
362  }

364  @Override
365  protected LockState acquireLock(final MasterProcedureEnv env) {
366    // Unless we are assigning meta, wait for meta to be available and loaded.
367    if (!isMeta() && (env.waitFailoverCleanup(this) ||
368        env.getAssignmentManager().waitMetaInitialized(this, getRegionInfo()))) {
369      return LockState.LOCK_EVENT_WAIT;
370    }
371
372    // TODO: Revisit this and move it to the executor
373    if (env.getProcedureScheduler().waitRegion(this, getRegionInfo())) {
374      try {
375        LOG.debug(LockState.LOCK_EVENT_WAIT + " pid=" + getProcId() + " " +
376          env.getProcedureScheduler().dumpLocks());
377      } catch (IOException e) {
378        // TODO Auto-generated catch block
379        e.printStackTrace();
380      }
381      return LockState.LOCK_EVENT_WAIT;
382    }
383    this.lock = true;
384    return LockState.LOCK_ACQUIRED;
385  }
386
387  @Override
388  protected void releaseLock(final MasterProcedureEnv env) {
389    env.getProcedureScheduler().wakeRegion(this, getRegionInfo());
390    lock = false;
391  }

396  }
