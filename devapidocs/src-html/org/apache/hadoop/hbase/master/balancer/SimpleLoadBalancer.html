042/**
043 * Makes decisions about the placement and movement of Regions across
044 * RegionServers.
045 *
046 * <p>Cluster-wide load balancing will occur only when there are no regions in
047 * transition and according to a fixed period of a time using {@link #balanceCluster(Map)}.
048 *
049 * <p>On cluster startup, bulk assignment can be used to determine
050 * locations for all Regions in a cluster.
051 *
052 * <p>This classes produces plans for the
053 * {@link org.apache.hadoop.hbase.master.AssignmentManager} to execute.
054 */
055@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
056public class SimpleLoadBalancer extends BaseLoadBalancer {
057  private static final Log LOG = LogFactory.getLog(SimpleLoadBalancer.class);
058  private static final Random RANDOM = new Random(System.currentTimeMillis());
059
060  private RegionInfoComparator riComparator = new RegionInfoComparator();
061  private RegionPlan.RegionPlanComparator rpComparator = new RegionPlan.RegionPlanComparator();
062
063
064  /**
065   * Stores additional per-server information about the regions added/removed
066   * during the run of the balancing algorithm.
067   *
068   * For servers that shed regions, we need to track which regions we have already
069   * shed. <b>nextRegionForUnload</b> contains the index in the list of regions on
070   * the server that is the next to be shed.
071   */
072  static class BalanceInfo {
073
074    private final int nextRegionForUnload;
075    private int numRegionsAdded;
076
077    public BalanceInfo(int nextRegionForUnload, int numRegionsAdded) {
078      this.nextRegionForUnload = nextRegionForUnload;
079      this.numRegionsAdded = numRegionsAdded;
080    }
081
082    int getNextRegionForUnload() {
083      return nextRegionForUnload;
084    }
085
086    int getNumRegionsAdded() {
087      return numRegionsAdded;
088    }
089
090    void setNumRegionsAdded(int numAdded) {
091      this.numRegionsAdded = numAdded;
092    }
093  }
094
095  /**
096   * Generate a global load balancing plan according to the specified map of
097   * server information to the most loaded regions of each server.
098   *
099   * The load balancing invariant is that all servers are within 1 region of the
100   * average number of regions per server.  If the average is an integer number,
101   * all servers will be balanced to the average.  Otherwise, all servers will
102   * have either floor(average) or ceiling(average) regions.
103   *
104   * HBASE-3609 Modeled regionsToMove using Guava's MinMaxPriorityQueue so that
105   *   we can fetch from both ends of the queue.
106   * At the beginning, we check whether there was empty region server
107   *   just discovered by Master. If so, we alternately choose new / old
108   *   regions from head / tail of regionsToMove, respectively. This alternation
109   *   avoids clustering young regions on the newly discovered region server.
110   *   Otherwise, we choose new regions from head of regionsToMove.
111   *
112   * Another improvement from HBASE-3609 is that we assign regions from
113   *   regionsToMove to underloaded servers in round-robin fashion.
114   *   Previously one underloaded server would be filled before we move onto
115   *   the next underloaded server, leading to clustering of young regions.
116   *
117   * Finally, we randomly shuffle underloaded servers so that they receive
118   *   offloaded regions relatively evenly across calls to balanceCluster().
119   *
120   * The algorithm is currently implemented as such:
121   *
122   * <ol>
123   * <li>Determine the two valid numbers of regions each server should have,
124   *     <b>MIN</b>=floor(average) and <b>MAX</b>=ceiling(average).
125   *
126   * <li>Iterate down the most loaded servers, shedding regions from each so
127   *     each server hosts exactly <b>MAX</b> regions.  Stop once you reach a
128   *     server that already has &lt;= <b>MAX</b> regions.
129   *     <p>
130   *     Order the regions to move from most recent to least.
131   *
132   * <li>Iterate down the least loaded servers, assigning regions so each server
133   *     has exactly <b>MIN</b> regions.  Stop once you reach a server that
134   *     already has &gt;= <b>MIN</b> regions.
135   *
136   *     Regions being assigned to underloaded servers are those that were shed
137   *     in the previous step.  It is possible that there were not enough
138   *     regions shed to fill each underloaded server to <b>MIN</b>.  If so we
139   *     end up with a number of regions required to do so, <b>neededRegions</b>.
140   *
141   *     It is also possible that we were able to fill each underloaded but ended
142   *     up with regions that were unassigned from overloaded servers but that
143   *     still do not have assignment.
144   *
145   *     If neither of these conditions hold (no regions needed to fill the
146   *     underloaded servers, no regions leftover from overloaded servers),
147   *     we are done and return.  Otherwise we handle these cases below.
148   *
149   * <li>If <b>neededRegions</b> is non-zero (still have underloaded servers),
150   *     we iterate the most loaded servers again, shedding a single server from
151   *     each (this brings them from having <b>MAX</b> regions to having
152   *     <b>MIN</b> regions).
153   *
154   * <li>We now definitely have more regions that need assignment, either from
155   *     the previous step or from the original shedding from overloaded servers.
156   *     Iterate the least loaded servers filling each to <b>MIN</b>.
157   *
158   * <li>If we still have more regions that need assignment, again iterate the
159   *     least loaded servers, this time giving each one (filling them to
160   *     <b>MAX</b>) until we run out.
161   *
162   * <li>All servers will now either host <b>MIN</b> or <b>MAX</b> regions.
163   *
164   *     In addition, any server hosting &gt;= <b>MAX</b> regions is guaranteed
165   *     to end up with <b>MAX</b> regions at the end of the balancing.  This
166   *     ensures the minimal number of regions possible are moved.
167   * </ol>
168   *
169   * TODO: We can at-most reassign the number of regions away from a particular
170   *       server to be how many they report as most loaded.
171   *       Should we just keep all assignment in memory?  Any objections?
172   *       Does this mean we need HeapSize on HMaster?  Or just careful monitor?
173   *       (current thinking is we will hold all assignments in memory)
174   *
175   * @param clusterMap Map of regionservers and their load/region information to
176   *                   a list of their most loaded regions
177   * @return a list of regions to be moved, including source and destination,
178   *         or null if cluster is already balanced
179   */
180  @Override
181  public List<RegionPlan> balanceCluster(
182      Map<ServerName, List<HRegionInfo>> clusterMap) {
183    List<RegionPlan> regionsToReturn = balanceMasterRegions(clusterMap);
184    if (regionsToReturn != null || clusterMap == null || clusterMap.size() <= 1) {
185      return regionsToReturn;
186    }
187    if (masterServerName != null && clusterMap.containsKey(masterServerName)) {
188      if (clusterMap.size() <= 2) {
189        return null;
190      }
191      clusterMap = new HashMap<ServerName, List<HRegionInfo>>(clusterMap);
192      clusterMap.remove(masterServerName);
193    }
194
195    long startTime = System.currentTimeMillis();
196
197    // construct a Cluster object with clusterMap and rest of the
198    // argument as defaults
199    Cluster c = new Cluster(clusterMap, null, this.regionFinder, this.rackManager);
200    if (!this.needsBalance(c)) return null;
201
202    ClusterLoadState cs = new ClusterLoadState(clusterMap);
203    int numServers = cs.getNumServers();
204    NavigableMap<ServerAndLoad, List<HRegionInfo>> serversByLoad = cs.getServersByLoad();
205    int numRegions = cs.getNumRegions();
206    float average = cs.getLoadAverage();
207    int max = (int)Math.ceil(average);
208    int min = (int)average;
209
210    // Using to check balance result.
211    StringBuilder strBalanceParam = new StringBuilder();
212    strBalanceParam.append("Balance parameter: numRegions=").append(numRegions)
213        .append(", numServers=").append(numServers).append(", max=").append(max)
214        .append(", min=").append(min);
215    LOG.debug(strBalanceParam.toString());
216
217    // Balance the cluster
218    // TODO: Look at data block locality or a more complex load to do this
219    MinMaxPriorityQueue<RegionPlan> regionsToMove =
220      MinMaxPriorityQueue.orderedBy(rpComparator).create();
221    regionsToReturn = new ArrayList<RegionPlan>();
222
223    // Walk down most loaded, pruning each to the max
224    int serversOverloaded = 0;
225    // flag used to fetch regions from head and tail of list, alternately
226    boolean fetchFromTail = false;
227    Map<ServerName, BalanceInfo> serverBalanceInfo =
228      new TreeMap<ServerName, BalanceInfo>();
229    for (Map.Entry<ServerAndLoad, List<HRegionInfo>> server:
230        serversByLoad.descendingMap().entrySet()) {
231      ServerAndLoad sal = server.getKey();
232      int load = sal.getLoad();
233      if (load <= max) {
234        serverBalanceInfo.put(sal.getServerName(), new BalanceInfo(0, 0));
235        break;
236      }
237      serversOverloaded++;
238      List<HRegionInfo> regions = server.getValue();
239      int numToOffload = Math.min(load - max, regions.size());
240      // account for the out-of-band regions which were assigned to this server
241      // after some other region server crashed
242      Collections.sort(regions, riComparator);
243      int numTaken = 0;
244      for (int i = 0; i <= numToOffload; ) {
245        HRegionInfo hri = regions.get(i); // fetch from head
246        if (fetchFromTail) {
247          hri = regions.get(regions.size() - 1 - i);
248        }
249        i++;
250        // Don't rebalance special regions.
251        if (shouldBeOnMaster(hri)
252            && masterServerName.equals(sal.getServerName())) continue;
253        regionsToMove.add(new RegionPlan(hri, sal.getServerName(), null));
254        numTaken++;
255        if (numTaken >= numToOffload) break;
256      }
257      serverBalanceInfo.put(sal.getServerName(),
258        new BalanceInfo(numToOffload, (-1)*numTaken));
259    }
260    int totalNumMoved = regionsToMove.size();
261
262    // Walk down least loaded, filling each to the min
263    int neededRegions = 0; // number of regions needed to bring all up to min
264    fetchFromTail = false;
265
266    Map<ServerName, Integer> underloadedServers = new HashMap<ServerName, Integer>();
267    int maxToTake = numRegions - min;
268    for (Map.Entry<ServerAndLoad, List<HRegionInfo>> server:
269        serversByLoad.entrySet()) {
270      if (maxToTake == 0) break; // no more to take
271      int load = server.getKey().getLoad();
272      if (load >= min && load > 0) {
273        continue; // look for other servers which haven't reached min
274      }
275      int regionsToPut = min - load;
276      if (regionsToPut == 0)
277      {
278        regionsToPut = 1;
279      }
280      maxToTake -= regionsToPut;
281      underloadedServers.put(server.getKey().getServerName(), regionsToPut);
282    }
283    // number of servers that get new regions
284    int serversUnderloaded = underloadedServers.size();
285    int incr = 1;
286    List<ServerName> sns =
287      Arrays.asList(underloadedServers.keySet().toArray(new ServerName[serversUnderloaded]));
288    Collections.shuffle(sns, RANDOM);
289    while (regionsToMove.size() > 0) {
290      int cnt = 0;
291      int i = incr > 0 ? 0 : underloadedServers.size()-1;
292      for (; i >= 0 && i < underloadedServers.size(); i += incr) {
293        if (regionsToMove.isEmpty()) break;
294        ServerName si = sns.get(i);
295        int numToTake = underloadedServers.get(si);
296        if (numToTake == 0) continue;
297
298        addRegionPlan(regionsToMove, fetchFromTail, si, regionsToReturn);
299
300        underloadedServers.put(si, numToTake-1);
301        cnt++;
302        BalanceInfo bi = serverBalanceInfo.get(si);
303        if (bi == null) {
304          bi = new BalanceInfo(0, 0);
305          serverBalanceInfo.put(si, bi);
306        }
307        bi.setNumRegionsAdded(bi.getNumRegionsAdded()+1);
308      }
309      if (cnt == 0) break;
310      // iterates underloadedServers in the other direction
311      incr = -incr;
312    }
313    for (Integer i : underloadedServers.values()) {
314      // If we still want to take some, increment needed
315      neededRegions += i;
316    }
317
318    // If none needed to fill all to min and none left to drain all to max,
319    // we are done
320    if (neededRegions == 0 && regionsToMove.isEmpty()) {
321      long endTime = System.currentTimeMillis();
322      LOG.info("Calculated a load balance in " + (endTime-startTime) + "ms. " +
323          "Moving " + totalNumMoved + " regions off of " +
324          serversOverloaded + " overloaded servers onto " +
325          serversUnderloaded + " less loaded servers");
326      return regionsToReturn;
327    }
328
329    // Need to do a second pass.
330    // Either more regions to assign out or servers that are still underloaded
331
332    // If we need more to fill min, grab one from each most loaded until enough
333    if (neededRegions != 0) {
334      // Walk down most loaded, grabbing one from each until we get enough
335      for (Map.Entry<ServerAndLoad, List<HRegionInfo>> server :
336        serversByLoad.descendingMap().entrySet()) {
337        BalanceInfo balanceInfo =
338          serverBalanceInfo.get(server.getKey().getServerName());
339        int idx =
340          balanceInfo == null ? 0 : balanceInfo.getNextRegionForUnload();
341        if (idx >= server.getValue().size()) break;
342        HRegionInfo region = server.getValue().get(idx);
343        if (region.isMetaRegion()) continue; // Don't move meta regions.
344        regionsToMove.add(new RegionPlan(region, server.getKey().getServerName(), null));
345        totalNumMoved++;
346        if (--neededRegions == 0) {
347          // No more regions needed, done shedding
348          break;
349        }
350      }
351    }
352
353    // Now we have a set of regions that must be all assigned out
354    // Assign each underloaded up to the min, then if leftovers, assign to max
355
356    // Walk down least loaded, assigning to each to fill up to min
357    for (Map.Entry<ServerAndLoad, List<HRegionInfo>> server :
358        serversByLoad.entrySet()) {
359      int regionCount = server.getKey().getLoad();
360      if (regionCount >= min) break;
361      BalanceInfo balanceInfo = serverBalanceInfo.get(server.getKey().getServerName());
362      if(balanceInfo != null) {
363        regionCount += balanceInfo.getNumRegionsAdded();
364      }
365      if(regionCount >= min) {
366        continue;
367      }
368      int numToTake = min - regionCount;
369      int numTaken = 0;
370      while(numTaken < numToTake && 0 < regionsToMove.size()) {
371        addRegionPlan(regionsToMove, fetchFromTail,
372          server.getKey().getServerName(), regionsToReturn);
373        numTaken++;
374      }
375    }
376
377    // If we still have regions to dish out, assign underloaded to max
378    if (0 < regionsToMove.size()) {
379      for (Map.Entry<ServerAndLoad, List<HRegionInfo>> server :
380        serversByLoad.entrySet()) {
381        int regionCount = server.getKey().getLoad();
382        BalanceInfo balanceInfo = serverBalanceInfo.get(server.getKey().getServerName());
383        if(balanceInfo != null) {
384          regionCount += balanceInfo.getNumRegionsAdded();
385        }
386        if(regionCount >= max) {
387          break;
388        }
389        addRegionPlan(regionsToMove, fetchFromTail,
390          server.getKey().getServerName(), regionsToReturn);
391        if (regionsToMove.isEmpty()) {
392          break;
393        }
394      }
395    }
396
397    long endTime = System.currentTimeMillis();
398
399    if (!regionsToMove.isEmpty() || neededRegions != 0) {
400      // Emit data so can diagnose how balancer went astray.
401      LOG.warn("regionsToMove=" + totalNumMoved +
402        ", numServers=" + numServers + ", serversOverloaded=" + serversOverloaded +
403        ", serversUnderloaded=" + serversUnderloaded);
404      StringBuilder sb = new StringBuilder();
405      for (Map.Entry<ServerName, List<HRegionInfo>> e: clusterMap.entrySet()) {
406        if (sb.length() > 0) sb.append(", ");
407        sb.append(e.getKey().toString());
408        sb.append(" ");
409        sb.append(e.getValue().size());
410      }
411      LOG.warn("Input " + sb.toString());
412    }
413
414    // All done!
415    LOG.info("Done. Calculated a load balance in " + (endTime-startTime) + "ms. " +
416        "Moving " + totalNumMoved + " regions off of " +
417        serversOverloaded + " overloaded servers onto " +
418        serversUnderloaded + " less loaded servers");
419
420    return regionsToReturn;
421  }
422
423  /**
424   * Add a region from the head or tail to the List of regions to return.
425   */
426  private void addRegionPlan(final MinMaxPriorityQueue<RegionPlan> regionsToMove,
427      final boolean fetchFromTail, final ServerName sn, List<RegionPlan> regionsToReturn) {
428    RegionPlan rp = null;
429    if (!fetchFromTail) rp = regionsToMove.remove();
430    else rp = regionsToMove.removeLast();
431    rp.setDestination(sn);
432    regionsToReturn.add(rp);
433  }
434
435  @Override
436  public List<RegionPlan> balanceCluster(TableName tableName,
437      Map<ServerName, List<HRegionInfo>> clusterState) throws HBaseIOException {
438    return balanceCluster(clusterState);
439  }
