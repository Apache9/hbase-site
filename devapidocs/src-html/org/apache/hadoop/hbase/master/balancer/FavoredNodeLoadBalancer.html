045/**
046 * An implementation of the {@link org.apache.hadoop.hbase.master.LoadBalancer} that 
047 * assigns favored nodes for each region. There is a Primary RegionServer that hosts 
048 * the region, and then there is Secondary and Tertiary RegionServers. Currently, the 
049 * favored nodes information is used in creating HDFS files - the Primary RegionServer 
050 * passes the primary, secondary, tertiary node addresses as hints to the 
051 * DistributedFileSystem API for creating files on the filesystem. These nodes are 
052 * treated as hints by the HDFS to place the blocks of the file. This alleviates the 
053 * problem to do with reading from remote nodes (since we can make the Secondary 
054 * RegionServer as the new Primary RegionServer) after a region is recovered. This 
055 * should help provide consistent read latencies for the regions even when their 
056 * primary region servers die.
057 *
058 */
059@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
060public class FavoredNodeLoadBalancer extends BaseLoadBalancer {
061  private static final Log LOG = LogFactory.getLog(FavoredNodeLoadBalancer.class);
062
063  private FavoredNodesPlan globalFavoredNodesAssignmentPlan;
064  private RackManager rackManager;
065
066  @Override
067  public void setConf(Configuration conf) {
068    super.setConf(conf);
069    globalFavoredNodesAssignmentPlan = new FavoredNodesPlan();
070    this.rackManager = new RackManager(conf);
071    super.setConf(conf);
072  }

074  @Override
075  public List<RegionPlan> balanceCluster(Map<ServerName, List<HRegionInfo>> clusterState)  {
076    //TODO. Look at is whether Stochastic loadbalancer can be integrated with this
077    List<RegionPlan> plans = new ArrayList<RegionPlan>();
078    //perform a scan of the meta to get the latest updates (if any)
079    SnapshotOfRegionAssignmentFromMeta snaphotOfRegionAssignment =
080        new SnapshotOfRegionAssignmentFromMeta(super.services.getConnection());
081    try {
082      snaphotOfRegionAssignment.initialize();
083    } catch (IOException ie) {
084      LOG.warn("Not running balancer since exception was thrown " + ie);
085      return plans;
086    }
087    globalFavoredNodesAssignmentPlan = snaphotOfRegionAssignment.getExistingAssignmentPlan();
088    Map<ServerName, ServerName> serverNameToServerNameWithoutCode =
089        new HashMap<ServerName, ServerName>();
090    Map<ServerName, ServerName> serverNameWithoutCodeToServerName =
091        new HashMap<ServerName, ServerName>();
092    ServerManager serverMgr = super.services.getServerManager();
093    for (ServerName sn: serverMgr.getOnlineServersList()) {
094      ServerName s = ServerName.valueOf(sn.getHostname(), sn.getPort(), ServerName.NON_STARTCODE);
095      serverNameToServerNameWithoutCode.put(sn, s);
096      serverNameWithoutCodeToServerName.put(s, sn);
097    }
098    for (Map.Entry<ServerName, List<HRegionInfo>> entry : clusterState.entrySet()) {
099      ServerName currentServer = entry.getKey();
100      //get a server without the startcode for the currentServer
101      ServerName currentServerWithoutStartCode = ServerName.valueOf(currentServer.getHostname(),
102          currentServer.getPort(), ServerName.NON_STARTCODE);
103      List<HRegionInfo> list = entry.getValue();
104      for (HRegionInfo region : list) {
105        if(region.getTable().getNamespaceAsString()
106            .equals(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR)) {
107          continue;
108        }
109        List<ServerName> favoredNodes = globalFavoredNodesAssignmentPlan.getFavoredNodes(region);
110        if (favoredNodes == null || favoredNodes.get(0).equals(currentServerWithoutStartCode)) {
111          continue; //either favorednodes does not exist or we are already on the primary node
112        }
113        ServerName destination = null;
114        //check whether the primary is available
115        destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(0));
116        if (destination == null) {
117          //check whether the region is on secondary/tertiary
118          if (currentServerWithoutStartCode.equals(favoredNodes.get(1)) ||
119              currentServerWithoutStartCode.equals(favoredNodes.get(2))) {
120            continue;
121          }
122          //the region is currently on none of the favored nodes
123          //get it on one of them if possible
124          ServerLoad l1 = super.services.getServerManager().getLoad(
125              serverNameWithoutCodeToServerName.get(favoredNodes.get(1)));
126          ServerLoad l2 = super.services.getServerManager().getLoad(
127              serverNameWithoutCodeToServerName.get(favoredNodes.get(2)));
128          if (l1 != null && l2 != null) {
129            if (l1.getLoad() > l2.getLoad()) {
130              destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(2));
131            } else {
132              destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(1));
133            }
134          } else if (l1 != null) {
135            destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(1));
136          } else if (l2 != null) {
137            destination = serverNameWithoutCodeToServerName.get(favoredNodes.get(2));
138          }
139        }
140
141        if (destination != null) {
142          RegionPlan plan = new RegionPlan(region, currentServer, destination);
143          plans.add(plan);
144        }
145      }
146    }
147    return plans;
148  }

150  @Override
151  public Map<ServerName, List<HRegionInfo>> roundRobinAssignment(List<HRegionInfo> regions,
152      List<ServerName> servers) {
153    Map<ServerName, List<HRegionInfo>> assignmentMap;
154    try {
155      FavoredNodeAssignmentHelper assignmentHelper =
156          new FavoredNodeAssignmentHelper(servers, rackManager);
157      assignmentHelper.initialize();
158      if (!assignmentHelper.canPlaceFavoredNodes()) {
159        return super.roundRobinAssignment(regions, servers);
160      }
161      // Segregate the regions into two types:
162      // 1. The regions that have favored node assignment, and where at least
163      //    one of the favored node is still alive. In this case, try to adhere
164      //    to the current favored nodes assignment as much as possible - i.e.,
165      //    if the current primary is gone, then make the secondary or tertiary
166      //    as the new host for the region (based on their current load).
167      //    Note that we don't change the favored
168      //    node assignments here (even though one or more favored node is currently
169      //    down). It is up to the balanceCluster to do this hard work. The HDFS
170      //    can handle the fact that some nodes in the favored nodes hint is down
171      //    It'd allocate some other DNs. In combination with stale settings for HDFS,
172      //    we should be just fine.
173      // 2. The regions that currently don't have favored node assignment. We will
174      //    need to come up with favored nodes assignments for them. The corner case
175      //    in (1) above is that all the nodes are unavailable and in that case, we
176      //    will note that this region doesn't have favored nodes.
177      Pair<Map<ServerName,List<HRegionInfo>>, List<HRegionInfo>> segregatedRegions =
178          segregateRegionsAndAssignRegionsWithFavoredNodes(regions, servers);
179      Map<ServerName,List<HRegionInfo>> regionsWithFavoredNodesMap = segregatedRegions.getFirst();
180      List<HRegionInfo> regionsWithNoFavoredNodes = segregatedRegions.getSecond();
181      assignmentMap = new HashMap<ServerName, List<HRegionInfo>>();
182      roundRobinAssignmentImpl(assignmentHelper, assignmentMap, regionsWithNoFavoredNodes,
183          servers);
184      // merge the assignment maps
185      assignmentMap.putAll(regionsWithFavoredNodesMap);
186    } catch (Exception ex) {
187      LOG.warn("Encountered exception while doing favored-nodes assignment " + ex +
188          " Falling back to regular assignment");
189      assignmentMap = super.roundRobinAssignment(regions, servers);
190    }
191    return assignmentMap;
192  }

194  @Override
195  public ServerName randomAssignment(HRegionInfo regionInfo, List<ServerName> servers) {
196    try {
197      FavoredNodeAssignmentHelper assignmentHelper =
198          new FavoredNodeAssignmentHelper(servers, rackManager);
199      assignmentHelper.initialize();
200      ServerName primary = super.randomAssignment(regionInfo, servers);
201      if (!assignmentHelper.canPlaceFavoredNodes()) {
202        return primary;
203      }
204      List<ServerName> favoredNodes = globalFavoredNodesAssignmentPlan.getFavoredNodes(regionInfo);
205      // check if we have a favored nodes mapping for this region and if so, return
206      // a server from the favored nodes list if the passed 'servers' contains this
207      // server as well (available servers, that is)
208      if (favoredNodes != null) {
209        for (ServerName s : favoredNodes) {
210          ServerName serverWithLegitStartCode = availableServersContains(servers, s);
211          if (serverWithLegitStartCode != null) {
212            return serverWithLegitStartCode;
213          }
214        }
215      }
216      List<HRegionInfo> regions = new ArrayList<HRegionInfo>(1);
217      regions.add(regionInfo);
218      Map<HRegionInfo, ServerName> primaryRSMap = new HashMap<HRegionInfo, ServerName>(1);
219      primaryRSMap.put(regionInfo, primary);
220      assignSecondaryAndTertiaryNodesForRegion(assignmentHelper, regions, primaryRSMap);
221      return primary;
222    } catch (Exception ex) {
223      LOG.warn("Encountered exception while doing favored-nodes (random)assignment " + ex +
224          " Falling back to regular assignment");
225      return super.randomAssignment(regionInfo, servers);
226    }
227  }

229  private Pair<Map<ServerName, List<HRegionInfo>>, List<HRegionInfo>>
230  segregateRegionsAndAssignRegionsWithFavoredNodes(List<HRegionInfo> regions,
231      List<ServerName> availableServers) {
232    Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes =
233        new HashMap<ServerName, List<HRegionInfo>>(regions.size() / 2);
234    List<HRegionInfo> regionsWithNoFavoredNodes = new ArrayList<HRegionInfo>(regions.size()/2);
235    for (HRegionInfo region : regions) {
236      List<ServerName> favoredNodes = globalFavoredNodesAssignmentPlan.getFavoredNodes(region);
237      ServerName primaryHost = null;
238      ServerName secondaryHost = null;
239      ServerName tertiaryHost = null;
240      if (favoredNodes != null) {
241        for (ServerName s : favoredNodes) {
242          ServerName serverWithLegitStartCode = availableServersContains(availableServers, s);
243          if (serverWithLegitStartCode != null) {
244            FavoredNodesPlan.Position position =
245                FavoredNodesPlan.getFavoredServerPosition(favoredNodes, s);
246            if (Position.PRIMARY.equals(position)) {
247              primaryHost = serverWithLegitStartCode;
248            } else if (Position.SECONDARY.equals(position)) {
249              secondaryHost = serverWithLegitStartCode;
250            } else if (Position.TERTIARY.equals(position)) {
251              tertiaryHost = serverWithLegitStartCode;
252            }
253          }
254        }
255        assignRegionToAvailableFavoredNode(assignmentMapForFavoredNodes, region,
256              primaryHost, secondaryHost, tertiaryHost);
257      }
258      if (primaryHost == null && secondaryHost == null && tertiaryHost == null) {
259        //all favored nodes unavailable
260        regionsWithNoFavoredNodes.add(region);
261      }
262    }
263    return new Pair<Map<ServerName, List<HRegionInfo>>, List<HRegionInfo>>(
264        assignmentMapForFavoredNodes, regionsWithNoFavoredNodes);
265  }

267  // Do a check of the hostname and port and return the servername from the servers list
268  // that matched (the favoredNode will have a startcode of -1 but we want the real
269  // server with the legit startcode
270  private ServerName availableServersContains(List<ServerName> servers, ServerName favoredNode) {
271    for (ServerName server : servers) {
272      if (ServerName.isSameHostnameAndPort(favoredNode, server)) {
273        return server;
274      }
275    }
276    return null;
277  }
278
279  private void assignRegionToAvailableFavoredNode(Map<ServerName,
280      List<HRegionInfo>> assignmentMapForFavoredNodes, HRegionInfo region, ServerName primaryHost,
281      ServerName secondaryHost, ServerName tertiaryHost) {
282    if (primaryHost != null) {
283      addRegionToMap(assignmentMapForFavoredNodes, region, primaryHost);
284    } else if (secondaryHost != null && tertiaryHost != null) {
285      // assign the region to the one with a lower load
286      // (both have the desired hdfs blocks)
287      ServerName s;
288      ServerLoad tertiaryLoad = super.services.getServerManager().getLoad(tertiaryHost);
289      ServerLoad secondaryLoad = super.services.getServerManager().getLoad(secondaryHost);
290      if (secondaryLoad.getLoad() < tertiaryLoad.getLoad()) {
291        s = secondaryHost;
292      } else {
293        s = tertiaryHost;
294      }
295      addRegionToMap(assignmentMapForFavoredNodes, region, s);
296    } else if (secondaryHost != null) {
297      addRegionToMap(assignmentMapForFavoredNodes, region, secondaryHost);
298    } else if (tertiaryHost != null) {
299      addRegionToMap(assignmentMapForFavoredNodes, region, tertiaryHost);
300    }
301  }
302
303  private void addRegionToMap(Map<ServerName, List<HRegionInfo>> assignmentMapForFavoredNodes,
304      HRegionInfo region, ServerName host) {
305    List<HRegionInfo> regionsOnServer = null;
306    if ((regionsOnServer = assignmentMapForFavoredNodes.get(host)) == null) {
307      regionsOnServer = new ArrayList<HRegionInfo>();
308      assignmentMapForFavoredNodes.put(host, regionsOnServer);
309    }
310    regionsOnServer.add(region);
311  }

313  public List<ServerName> getFavoredNodes(HRegionInfo regionInfo) {
314    return this.globalFavoredNodesAssignmentPlan.getFavoredNodes(regionInfo);
315  }
316
317  private void roundRobinAssignmentImpl(FavoredNodeAssignmentHelper assignmentHelper,
318      Map<ServerName, List<HRegionInfo>> assignmentMap,
319      List<HRegionInfo> regions, List<ServerName> servers) {
320    Map<HRegionInfo, ServerName> primaryRSMap = new HashMap<HRegionInfo, ServerName>();
321    // figure the primary RSs
322    assignmentHelper.placePrimaryRSAsRoundRobin(assignmentMap, primaryRSMap, regions);
323    assignSecondaryAndTertiaryNodesForRegion(assignmentHelper, regions, primaryRSMap);
324  }
325
326  private void assignSecondaryAndTertiaryNodesForRegion(
327      FavoredNodeAssignmentHelper assignmentHelper,
328      List<HRegionInfo> regions, Map<HRegionInfo, ServerName> primaryRSMap) {
329    // figure the secondary and tertiary RSs
330    Map<HRegionInfo, ServerName[]> secondaryAndTertiaryRSMap =
331        assignmentHelper.placeSecondaryAndTertiaryRS(primaryRSMap);
332    // now record all the assignments so that we can serve queries later
333    for (HRegionInfo region : regions) {
334      // Store the favored nodes without startCode for the ServerName objects
335      // We don't care about the startcode; but only the hostname really
336      List<ServerName> favoredNodesForRegion = new ArrayList<ServerName>(3);
337      ServerName sn = primaryRSMap.get(region);
338      favoredNodesForRegion.add(ServerName.valueOf(sn.getHostname(), sn.getPort(),
339          ServerName.NON_STARTCODE));
340      ServerName[] secondaryAndTertiaryNodes = secondaryAndTertiaryRSMap.get(region);
341      if (secondaryAndTertiaryNodes != null) {
342        favoredNodesForRegion.add(ServerName.valueOf(secondaryAndTertiaryNodes[0].getHostname(),
343            secondaryAndTertiaryNodes[0].getPort(), ServerName.NON_STARTCODE));
344        favoredNodesForRegion.add(ServerName.valueOf(secondaryAndTertiaryNodes[1].getHostname(),
345            secondaryAndTertiaryNodes[1].getPort(), ServerName.NON_STARTCODE));
346      }
347      globalFavoredNodesAssignmentPlan.updateFavoredNodesMap(region, favoredNodesForRegion);
348    }
349  }

351  @Override
352  public List<RegionPlan> balanceCluster(TableName tableName,
353      Map<ServerName, List<HRegionInfo>> clusterState) throws HBaseIOException {
354    return balanceCluster(clusterState);
