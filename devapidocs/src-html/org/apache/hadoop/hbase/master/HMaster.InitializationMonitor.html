205/**
206 * HMaster is the "master server" for HBase. An HBase cluster has one active
207 * master.  If many masters are started, all compete.  Whichever wins goes on to
208 * run the cluster.  All others park themselves in their constructor until
209 * master or cluster shutdown or until the active master loses its lease in
210 * zookeeper.  Thereafter, all running master jostle to take over master role.
211 *
212 * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
213 * this case it will tell all regionservers to go down and then wait on them
214 * all reporting in that they are down.  This master will then shut itself down.
215 *
216 * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
217 *
218 * @see org.apache.zookeeper.Watcher
219 */
220@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
221@SuppressWarnings("deprecation")
222public class HMaster extends HRegionServer implements MasterServices {
223  private static final Log LOG = LogFactory.getLog(HMaster.class.getName());

225  /**
226   * Protection against zombie master. Started once Master accepts active responsibility and
227   * starts taking over responsibilities. Allows a finite time window before giving up ownership.
228   */
229  private static class InitializationMonitor extends HasThread {
230    /** The amount of time in milliseconds to sleep before checking initialization status. */
231    public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
232    public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
233
234    /**
235     * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
236     * true, do nothing otherwise.
237     */
238    public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
239    public static final boolean HALT_DEFAULT = false;
240
241    private final HMaster master;
242    private final long timeout;
243    private final boolean haltOnTimeout;
244
245    /** Creates a Thread that monitors the {@link #isInitialized()} state. */
246    InitializationMonitor(HMaster master) {
247      super("MasterInitializationMonitor");
248      this.master = master;
249      this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
250      this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
251      this.setDaemon(true);
252    }
253
254    @Override
255    public void run() {
256      try {
257        while (!master.isStopped() && master.isActiveMaster()) {
258          Thread.sleep(timeout);
259          if (master.isInitialized()) {
260            LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
261          } else {
262            LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
263                + " consider submitting a bug report including a thread dump of this process.");
264            if (haltOnTimeout) {
265              LOG.error("Zombie Master exiting. Thread dump to stdout");
266              Threads.printThreadInfo(System.out, "Zombie HMaster");
267              System.exit(-1);
268            }
269          }
270        }
271      } catch (InterruptedException ie) {
272        LOG.trace("InitMonitor thread interrupted. Existing.");
273      }
274    }
275  }

484    // should we check encryption settings at master side, default true

496    // Do we publish the status?

501        conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
502            ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
503            ClusterStatusPublisher.Publisher.class);

505    if (shouldPublish) {

522    } else {
523      activeMasterManager = null;
524    }

527  // return the actual infoPort, -1 means disable info server.
528  private int putUpJettyServer() throws IOException {
529    if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
530      return -1;
531    }
532    final int infoPort = conf.getInt("hbase.master.info.port.orig",
533      HConstants.DEFAULT_MASTER_INFOPORT);
534    // -1 is for disabling info server, so no redirecting
535    if (infoPort < 0 || infoServer == null) {
536      return -1;
537    }
538    if(infoPort == infoServer.getPort()) {
539      return infoPort;
540    }
541    final String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
542    if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
543      String msg =
544          "Failed to start redirecting jetty server. Address " + addr
545              + " does not belong to this host. Correct configuration parameter: "
546              + "hbase.master.info.bindAddress";

707  /**
708   * Finish initialization of HMaster after becoming the primary master.
709   *
710   * <ol>
711   * <li>Initialize master components - file system manager, server manager,
712   *     assignment manager, region server tracker, etc</li>
713   * <li>Start necessary service threads - balancer, catalog janior,
714   *     executor services, etc</li>
715   * <li>Set cluster as UP in ZooKeeper</li>
716   * <li>Wait for RegionServers to check-in</li>
717   * <li>Split logs and perform data recovery, if necessary</li>
718   * <li>Ensure assignment of meta/namespace regions<li>
719   * <li>Handle either fresh cluster start or master failover</li>
720   * </ol>
721   */
722  private void finishActiveMasterInitialization(MonitoredTask status)
723      throws IOException, InterruptedException, KeeperException, CoordinatedStateException {
724
725    activeMaster = true;
726    Thread zombieDetector = new Thread(new InitializationMonitor(this),
727        "ActiveMasterInitializationMonitor-" + System.currentTimeMillis());
728    zombieDetector.start();
729
730    /*
731     * We are active master now... go initialize components we need to run.
732     * Note, there may be dross in zk from previous runs; it'll get addressed
733     * below after we determine if cluster startup or failover.
734     */
735
736    status.setStatus("Initializing Master file system");
737
738    this.masterActiveTime = System.currentTimeMillis();
739    // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
740    // Initialize the chunkCreator
741    initializeMemStoreChunkCreator();
742    this.fileSystemManager = new MasterFileSystem(this);
743    this.walManager = new MasterWalManager(this);
744
745    // enable table descriptors cache
746    this.tableDescriptors.setCacheOn();
747    // set the META's descriptor to the correct replication
748    this.tableDescriptors.get(TableName.META_TABLE_NAME).setRegionReplication(
749        conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM));
750    // warm-up HTDs cache on master initialization
751    if (preLoadTableDescriptors) {
752      status.setStatus("Pre-loading table descriptors");
753      this.tableDescriptors.getAll();
754    }
755
756    // publish cluster ID
757    status.setStatus("Publishing Cluster ID in ZooKeeper");
758    ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
759    this.initLatch.countDown();
760
761    this.serverManager = createServerManager(this);
762
763    this.tableStateManager = new TableStateManager(this);
764
765    status.setStatus("Initializing ZK system trackers");
766    initializeZKBasedSystemTrackers();
767
768    // This is for backwards compatibility
769    // See HBASE-11393
770    status.setStatus("Update TableCFs node in ZNode");
771    TableCFsUpdater tableCFsUpdater = new TableCFsUpdater(zooKeeper,
772            conf, this.clusterConnection);
773    tableCFsUpdater.update();
774
775    // Add the Observer to delete space quotas on table deletion before starting all CPs by
776    // default with quota support, avoiding if user specifically asks to not load this Observer.
777    if (QuotaUtil.isQuotaEnabled(conf)) {
778      updateConfigurationForSpaceQuotaObserver(conf);
779    }
780    // initialize master side coprocessors before we start handling requests
781    status.setStatus("Initializing master coprocessors");
782    this.cpHost = new MasterCoprocessorHost(this, this.conf);
783
784    // start up all service threads.
785    status.setStatus("Initializing master service threads");
786    startServiceThreads();
787
788    // Wake up this server to check in
789    sleeper.skipSleepCycle();
790
791    // Wait for region servers to report in
792    status.setStatus("Wait for region servers to report in");
793    waitForRegionServers(status);
794
795    // get a list for previously failed RS which need log splitting work
796    // we recover hbase:meta region servers inside master initialization and
797    // handle other failed servers in SSH in order to start up master node ASAP
798    MasterMetaBootstrap metaBootstrap = createMetaBootstrap(this, status);
799    metaBootstrap.splitMetaLogsBeforeAssignment();
800
801    this.initializationBeforeMetaAssignment = true;
802
803    if (this.balancer instanceof FavoredNodesPromoter) {
804      favoredNodesManager = new FavoredNodesManager(this);
805    }
806    // Wait for regionserver to finish initialization.
807    if (BaseLoadBalancer.tablesOnMaster(conf)) {
808      waitForServerOnline();
809    }
810
811    //initialize load balancer
812    this.balancer.setMasterServices(this);
813    this.balancer.setClusterStatus(getClusterStatus());
814    this.balancer.initialize();
815
816    // Check if master is shutting down because of some issue
817    // in initializing the regionserver or the balancer.
818    if (isStopped()) return;
819
820    // Make sure meta assigned before proceeding.
821    status.setStatus("Assigning Meta Region");
822    metaBootstrap.assignMeta();
823
824    // check if master is shutting down because above assignMeta could return even hbase:meta isn't
825    // assigned when master is shutting down
826    if (isStopped()) return;
827
828    //Initialize after meta as it scans meta
829    if (favoredNodesManager != null) {
830      SnapshotOfRegionAssignmentFromMeta snapshotOfRegionAssignment =
831          new SnapshotOfRegionAssignmentFromMeta(getConnection());
832      snapshotOfRegionAssignment.initialize();
833      favoredNodesManager.initialize(snapshotOfRegionAssignment);
834    }
835
836    // migrating existent table state from zk, so splitters
837    // and recovery process treat states properly.
838    for (Map.Entry<TableName, TableState.State> entry : ZKDataMigrator
839        .queryForTableStates(getZooKeeper()).entrySet()) {
840      LOG.info("Converting state from zk to new states:" + entry);
841      tableStateManager.setTableState(entry.getKey(), entry.getValue());
842    }
843    ZKUtil.deleteChildrenRecursively(getZooKeeper(), getZooKeeper().znodePaths.tableZNode);
844
845    status.setStatus("Submitting log splitting work for previously failed region servers");
846    metaBootstrap.processDeadServers();
847
848    // Fix up assignment manager status
849    status.setStatus("Starting assignment manager");
850    this.assignmentManager.joinCluster();
851
852    // set cluster status again after user regions are assigned
853    this.balancer.setClusterStatus(getClusterStatus());
854
855    // Start balancer and meta catalog janitor after meta and regions have been assigned.
856    status.setStatus("Starting balancer and catalog janitor");
857    this.clusterStatusChore = new ClusterStatusChore(this, balancer);
858    getChoreService().scheduleChore(clusterStatusChore);
859    this.balancerChore = new BalancerChore(this);
860    getChoreService().scheduleChore(balancerChore);
861    this.normalizerChore = new RegionNormalizerChore(this);
862    getChoreService().scheduleChore(normalizerChore);
863    this.catalogJanitorChore = new CatalogJanitor(this);
864    getChoreService().scheduleChore(catalogJanitorChore);
865
866    status.setStatus("Starting cluster schema service");
867    initClusterSchemaService();
868
869    if (this.cpHost != null) {
870      try {
871        this.cpHost.preMasterInitialization();
872      } catch (IOException e) {
873        LOG.error("Coprocessor preMasterInitialization() hook failed", e);
874      }
875    }
876
877    status.markComplete("Initialization successful");
878    LOG.info(String.format("Master has completed initialization %.3fsec",
879       (System.currentTimeMillis() - masterActiveTime) / 1000.0f));
880    configurationManager.registerObserver(this.balancer);
881    configurationManager.registerObserver(this.hfileCleaner);
882
883    // Set master as 'initialized'.
884    setInitialized(true);
885
886    status.setStatus("Assign meta replicas");
887    metaBootstrap.assignMetaReplicas();
888
889    status.setStatus("Starting quota manager");
890    initQuotaManager();
891    if (QuotaUtil.isQuotaEnabled(conf)) {
892      // Create the quota snapshot notifier
893      spaceQuotaSnapshotNotifier = createQuotaSnapshotNotifier();
894      spaceQuotaSnapshotNotifier.initialize(getClusterConnection());
895      this.quotaObserverChore = new QuotaObserverChore(this, getMasterMetrics());
896      // Start the chore to read the region FS space reports and act on them
897      getChoreService().scheduleChore(quotaObserverChore);
898    }
899
900    // clear the dead servers with same host name and port of online server because we are not
901    // removing dead server with same hostname and port of rs which is trying to check in before
902    // master initialization. See HBASE-5916.
903    this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
904
905    // Check and set the znode ACLs if needed in case we are overtaking a non-secure configuration
906    status.setStatus("Checking ZNode ACLs");
907    zooKeeper.checkAndSetZNodeAcls();
908
909    status.setStatus("Initializing MOB Cleaner");
910    initMobCleaner();
911
912    status.setStatus("Calling postStartMaster coprocessors");
913    if (this.cpHost != null) {
914      // don't let cp initialization errors kill the master
915      try {
916        this.cpHost.postStartMaster();
917      } catch (IOException ioe) {
918        LOG.error("Coprocessor postStartMaster() hook failed", ioe);
919      }
920    }
921
922    zombieDetector.interrupt();
923  }

1059   * Start up all services. If any of these threads gets an unhandled exception
1060   * then they just die with a logged message.  This should be fine because
1061   * in general, we do not expect the master to get such unhandled exceptions
1062   *  as OOMEs; it should be lightly loaded. See what HRegionServer does if
1063   *  need to install an unexpected exception handler.
1064   */
1065  private void startServiceThreads() throws IOException{
1066   // Start the executor service pools
1067   this.service.startExecutorService(ExecutorType.MASTER_OPEN_REGION,
1068      conf.getInt("hbase.master.executor.openregion.threads", 5));
1069   this.service.startExecutorService(ExecutorType.MASTER_CLOSE_REGION,
1070      conf.getInt("hbase.master.executor.closeregion.threads", 5));
1071   this.service.startExecutorService(ExecutorType.MASTER_SERVER_OPERATIONS,
1072      conf.getInt("hbase.master.executor.serverops.threads", 5));
1073   this.service.startExecutorService(ExecutorType.MASTER_META_SERVER_OPERATIONS,
1074      conf.getInt("hbase.master.executor.meta.serverops.threads", 5));
1075   this.service.startExecutorService(ExecutorType.M_LOG_REPLAY_OPS,
1076      conf.getInt("hbase.master.executor.logreplayops.threads", 10));
1077
1078   // We depend on there being only one instance of this executor running
1079   // at a time.  To do concurrency, would need fencing of enable/disable of
1080   // tables.
1081   // Any time changing this maxThreads to > 1, pls see the comment at
1082   // AccessController#postCompletedCreateTableAction
1083   this.service.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
1084   startProcedureExecutor();
1085
1086   // Start log cleaner thread
1087   int cleanerInterval = conf.getInt("hbase.master.cleaner.interval", 60 * 1000);
1088   this.logCleaner =
1089      new LogCleaner(cleanerInterval,
1090         this, conf, getMasterWalManager().getFileSystem(),
1091         getMasterWalManager().getOldLogDir());
1092    getChoreService().scheduleChore(logCleaner);
1093
1094   //start the hfile archive cleaner thread
1095    Path archiveDir = HFileArchiveUtil.getArchivePath(conf);
1096    Map<String, Object> params = new HashMap<>();
1097    params.put(MASTER, this);
1098    this.hfileCleaner = new HFileCleaner(cleanerInterval, this, conf, getMasterFileSystem()
1099        .getFileSystem(), archiveDir, params);
1100    getChoreService().scheduleChore(hfileCleaner);
1101    serviceStarted = true;
1102    if (LOG.isTraceEnabled()) {
1103      LOG.trace("Started service threads");
1104    }
1105
1106    // Start replication zk node cleaner
1107    if (conf.getClass("hbase.region.replica.replication.replicationQueues.class",
1108      ReplicationFactory.defaultReplicationQueueClass) == ReplicationQueuesZKImpl.class) {
1109      try {
1110        replicationZKNodeCleanerChore = new ReplicationZKNodeCleanerChore(this, cleanerInterval,
1111            new ReplicationZKNodeCleaner(this.conf, this.getZooKeeper(), this));
1112        getChoreService().scheduleChore(replicationZKNodeCleanerChore);
1113      } catch (Exception e) {
1114        LOG.error("start replicationZKNodeCleanerChore failed", e);
1115      }
1116    }
1117    replicationMetaCleaner = new ReplicationMetaCleaner(this, this, cleanerInterval);
1118    getChoreService().scheduleChore(replicationMetaCleaner);
1119  }

1122  protected void stopServiceThreads() {
1123    if (masterJettyServer != null) {
1124      LOG.info("Stopping master jetty server");
1125      try {
1126        masterJettyServer.stop();
1127      } catch (Exception e) {
1128        LOG.error("Failed to stop master jetty server", e);
1129      }
1130    }
1131    super.stopServiceThreads();
1132    stopChores();
1133
1134    // Wait for all the remaining region servers to report in IFF we were
1135    // running a cluster shutdown AND we were NOT aborting.
1136    if (!isAborted() && this.serverManager != null &&
1137        this.serverManager.isClusterShutdown()) {
1138      this.serverManager.letRegionServersShutdown();
1139    }
1140    if (LOG.isDebugEnabled()) {
1141      LOG.debug("Stopping service threads");
1142    }
1143
1144    // Clean up and close up shop
1145    if (this.logCleaner != null) this.logCleaner.cancel(true);
1146    if (this.hfileCleaner != null) this.hfileCleaner.cancel(true);
1147    if (this.replicationZKNodeCleanerChore != null) this.replicationZKNodeCleanerChore.cancel(true);
1148    if (this.replicationMetaCleaner != null) this.replicationMetaCleaner.cancel(true);
1149    if (this.quotaManager != null) this.quotaManager.stop();
1150
1151    if (this.activeMasterManager != null) this.activeMasterManager.stop();
1152    if (this.serverManager != null) this.serverManager.stop();
1153    if (this.assignmentManager != null) this.assignmentManager.stop();
1154
1155    stopProcedureExecutor();
1156

1285  /**
1286   * It first sleep to the next balance plan start time. Meanwhile, throttling by the max
1287   * number regions in transition to protect availability.
1288   * @param nextBalanceStartTime The next balance plan start time
1289   * @param maxRegionsInTransition max number of regions in transition
1290   * @param cutoffTime when to exit balancer
1291   */
1292  private void balanceThrottling(long nextBalanceStartTime, int maxRegionsInTransition,
1293      long cutoffTime) {
1294    boolean interrupted = false;
1295
1296    // Sleep to next balance plan start time
1297    // But if there are zero regions in transition, it can skip sleep to speed up.
1298    while (!interrupted && System.currentTimeMillis() < nextBalanceStartTime
1299        && this.assignmentManager.getRegionStates().hasRegionsInTransition()) {
1300      try {
1301        Thread.sleep(100);
1302      } catch (InterruptedException ie) {
1303        interrupted = true;
1304      }
1305    }
1306
1307    // Throttling by max number regions in transition
1308    while (!interrupted
1309        && maxRegionsInTransition > 0
1310        && this.assignmentManager.getRegionStates().getRegionsInTransition().size()
1311        >= maxRegionsInTransition && System.currentTimeMillis() <= cutoffTime) {
1312      try {
1313        // sleep if the number of regions in transition exceeds the limit
1314        Thread.sleep(100);
1315      } catch (InterruptedException ie) {
1316        interrupted = true;
1317      }
1318    }
1319
1320    if (interrupted) Thread.currentThread().interrupt();
1321  }

1327  public boolean balance(boolean force) throws IOException {
1328    // if master not initialized, don't run balancer.
1329    if (!isInitialized()) {
1330      LOG.debug("Master has not been initialized, don't run balancer.");
1331      return false;
1332    }
1333
1334    if (isInMaintenanceMode()) {
1335      LOG.info("Master is in maintenanceMode mode, don't run balancer.");
1336      return false;
1337    }
1338
1339    int maxRegionsInTransition = getMaxRegionsInTransition();
1340    synchronized (this.balancer) {
1341      // If balance not true, don't run balancer.
1342      if (!this.loadBalancerTracker.isBalancerOn()) return false;
1343        // Only allow one balance run at at time.
1344      if (this.assignmentManager.hasRegionsInTransition()) {
1345        List<RegionStateNode> regionsInTransition = assignmentManager.getRegionsInTransition();
1346        // if hbase:meta region is in transition, result of assignment cannot be recorded
1347        // ignore the force flag in that case
1348        boolean metaInTransition = assignmentManager.isMetaRegionInTransition();
1349        String prefix = force && !metaInTransition ? "R" : "Not r";
1350        List<RegionStateNode> toPrint = regionsInTransition;
1351        int max = 5;
1352        boolean truncated = false;
1353        if (regionsInTransition.size() > max) {
1354          toPrint = regionsInTransition.subList(0, max);
1355          truncated = true;
1356        }
1357        LOG.info(prefix + "unning balancer because " + regionsInTransition.size() +
1358          " region(s) in transition: " + toPrint + (truncated? "(truncated list)": ""));
1359        if (!force || metaInTransition) return false;
1360      }
1361      if (this.serverManager.areDeadServersInProgress()) {
1362        LOG.info("Not running balancer because processing dead regionserver(s): " +
1363          this.serverManager.getDeadServers());
1364        return false;
1365      }
1366
1367      if (this.cpHost != null) {
1368        try {
1369          if (this.cpHost.preBalance()) {
1370            LOG.debug("Coprocessor bypassing balancer request");
1371            return false;
1372          }
1373        } catch (IOException ioe) {
1374          LOG.error("Error invoking master coprocessor preBalance()", ioe);
1375          return false;
1376        }
1377      }
1378
1379      Map<TableName, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1380        this.assignmentManager.getRegionStates().getAssignmentsByTable();
1381
1382      List<RegionPlan> plans = new ArrayList<>();
1383
1384      //Give the balancer the current cluster state.
1385      this.balancer.setClusterStatus(getClusterStatus());
1386      this.balancer.setClusterLoad(
1387              this.assignmentManager.getRegionStates().getAssignmentsByTable());
1388
1389      for (Entry<TableName, Map<ServerName, List<HRegionInfo>>> e : assignmentsByTable.entrySet()) {
1390        List<RegionPlan> partialPlans = this.balancer.balanceCluster(e.getKey(), e.getValue());
1391        if (partialPlans != null) plans.addAll(partialPlans);
1392      }
1393
1394      long balanceStartTime = System.currentTimeMillis();
1395      long cutoffTime = balanceStartTime + this.maxBlancingTime;
1396      int rpCount = 0;  // number of RegionPlans balanced so far
1397      if (plans != null && !plans.isEmpty()) {
1398        int balanceInterval = this.maxBlancingTime / plans.size();
1399        LOG.info("Balancer plans size is " + plans.size() + ", the balance interval is "
1400            + balanceInterval + " ms, and the max number regions in transition is "
1401            + maxRegionsInTransition);
1402
1403        for (RegionPlan plan: plans) {
1404          LOG.info("balance " + plan);
1405          //TODO: bulk assign
1406          this.assignmentManager.moveAsync(plan);
1407          rpCount++;
1408
1409          balanceThrottling(balanceStartTime + rpCount * balanceInterval, maxRegionsInTransition,
1410            cutoffTime);
1411
1412          // if performing next balance exceeds cutoff time, exit the loop
1413          if (rpCount < plans.size() && System.currentTimeMillis() > cutoffTime) {
1414            // TODO: After balance, there should not be a cutoff time (keeping it as
1415            // a security net for now)
1416            LOG.debug("No more balancing till next balance run; maxBalanceTime="
1417                + this.maxBlancingTime);
1418            break;
1419          }
1420        }
1421      }
1422
1423      if (this.cpHost != null) {
1424        try {
1425          this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1426        } catch (IOException ioe) {
1427          // balancing already succeeded so don't change the result
1428          LOG.error("Error invoking master coprocessor postBalance()", ioe);
1429        }
1430      }
1431    }
1432    // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1433    // Return true indicating a success.
1434    return true;
1435  }

1443  /**
1444   * Perform normalization of cluster (invoked by {@link RegionNormalizerChore}).
1445   *
1446   * @return true if normalization step was performed successfully, false otherwise
1447   *    (specifically, if HMaster hasn't been initialized properly or normalization
1448   *    is globally disabled)
1449   */
1450  public boolean normalizeRegions() throws IOException {
1451    if (!isInitialized()) {
1452      LOG.debug("Master has not been initialized, don't run region normalizer.");
1453      return false;
1454    }
1455
1456    if (isInMaintenanceMode()) {
1457      LOG.info("Master is in maintenance mode, don't run region normalizer.");
1458      return false;
1459    }
1460
1461    if (!this.regionNormalizerTracker.isNormalizerOn()) {
1462      LOG.debug("Region normalization is disabled, don't run region normalizer.");
1463      return false;
1464    }
1465
1466    synchronized (this.normalizer) {
1467      // Don't run the normalizer concurrently
1468      List<TableName> allEnabledTables = new ArrayList<>(
1469        this.tableStateManager.getTablesInStates(TableState.State.ENABLED));
1470
1471      Collections.shuffle(allEnabledTables);
1472
1473      for (TableName table : allEnabledTables) {
1474        if (isInMaintenanceMode()) {
1475          LOG.debug("Master is in maintenance mode, stop running region normalizer.");
1476          return false;
1477        }
1478
1479        HTableDescriptor tblDesc = getTableDescriptors().get(table);
1480        if (table.isSystemTable() || (tblDesc != null &&
1481            !tblDesc.isNormalizationEnabled())) {
1482          LOG.debug("Skipping normalization for table: " + table + ", as it's either system"
1483              + " table or doesn't have auto normalization turned on");
1484          continue;
1485        }
1486        List<NormalizationPlan> plans = this.normalizer.computePlanForTable(table);
1487        if (plans != null) {
1488          for (NormalizationPlan plan : plans) {
1489            plan.execute(clusterConnection.getAdmin());
1490            if (plan.getType() == PlanType.SPLIT) {
1491              splitPlanCount++;
1492            } else if (plan.getType() == PlanType.MERGE) {
1493              mergePlanCount++;
1494            }
1495          }
1496        }
1497      }
1498    }
1499    // If Region did not generate any plans, it means the cluster is already balanced.
1500    // Return true indicating a success.
1501    return true;
1502  }

1650      final byte[] destServerName) throws HBaseIOException {
1651    RegionState regionState = assignmentManager.getRegionStates().
1652      getRegionState(Bytes.toString(encodedRegionName));
1653
1654    HRegionInfo hri;
1655    if (regionState != null) {
1656      hri = regionState.getRegion();

1658      throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1659    }
1660
1661    ServerName dest;
1662    if (destServerName == null || destServerName.length == 0) {
1663      LOG.info("Passed destination servername is null/empty so " +
1664        "choosing a server at random");
1665      final List<ServerName> destServers = this.serverManager.createDestinationServersList(
1666        regionState.getServerName());
1667      dest = balancer.randomAssignment(hri, destServers);
1668      if (dest == null) {
1669        LOG.debug("Unable to determine a plan to assign " + hri);
1670        return;
1671      }
1672    } else {
1673      ServerName candidate = ServerName.valueOf(Bytes.toString(destServerName));
1674      dest = balancer.randomAssignment(hri, Lists.newArrayList(candidate));
1675      if (dest == null) {
1676        LOG.debug("Unable to determine a plan to assign " + hri);
1677        return;
1678      }
1679      if (dest.equals(serverName) && balancer instanceof BaseLoadBalancer
1680          && !((BaseLoadBalancer)balancer).shouldBeOnMaster(hri)) {
1681        // To avoid unnecessary region moving later by balancer. Don't put user
1682        // regions on master. Regions on master could be put on other region
1683        // server intentionally by test however.
1684        LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1685          + " to avoid unnecessary region moving later by load balancer,"
1686          + " because it should not be on master");
1687        return;
1688      }
1689    }
1690
1691    if (dest.equals(regionState.getServerName())) {
1692      LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1693        + " because region already assigned to the same server " + dest + ".");
1694      return;
1695    }
1696
1697    // Now we can do the move
1698    RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1699    assert rp.getDestination() != null: rp.toString() + " " + dest;
1700    assert rp.getSource() != null: rp.toString();

1705        if (this.cpHost.preMove(hri, rp.getSource(), rp.getDestination())) {
1706          return;
1707        }

1710      // is synchronous and takes some time. doing it before the source region gets
1711      // closed
1712      serverManager.sendRegionWarmup(rp.getDestination(), hri);
1713
1714      LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
1715      Future<byte []> future = this.assignmentManager.moveAsync(rp);
1716      try {
1717        // Is this going to work? Will we throw exception on error?
1718        // TODO: CompletableFuture rather than this stunted Future.
1719        future.get();
1720      } catch (InterruptedException | ExecutionException e) {
1721        throw new HBaseIOException(e);
1722      }
1723      if (this.cpHost != null) {
1724        this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
1725      }
1726    } catch (IOException ioe) {
1727      if (ioe instanceof HBaseIOException) {
1728        throw (HBaseIOException)ioe;
1729      }
1730      throw new HBaseIOException(ioe);
1731    }

1797  /**
1798   * Checks whether the table conforms to some sane limits, and configured
1799   * values (compression, etc) work. Throws an exception if something is wrong.
1800   * @throws IOException
1801   */
1802  private void sanityCheckTableDescriptor(final HTableDescriptor htd) throws IOException {
1803    final String CONF_KEY = "hbase.table.sanity.checks";
1804    boolean logWarn = false;
1805    if (!conf.getBoolean(CONF_KEY, true)) {
1806      logWarn = true;
1807    }
1808    String tableVal = htd.getConfigurationValue(CONF_KEY);
1809    if (tableVal != null && !Boolean.valueOf(tableVal)) {
1810      logWarn = true;
1811    }
1812
1813    // check max file size
1814    long maxFileSizeLowerLimit = 2 * 1024 * 1024L; // 2M is the default lower limit
1815    long maxFileSize = htd.getMaxFileSize();
1816    if (maxFileSize < 0) {
1817      maxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE, maxFileSizeLowerLimit);
1818    }
1819    if (maxFileSize < conf.getLong("hbase.hregion.max.filesize.limit", maxFileSizeLowerLimit)) {
1820      String message = "MAX_FILESIZE for table descriptor or "
1821          + "\"hbase.hregion.max.filesize\" (" + maxFileSize
1822          + ") is too small, which might cause over splitting into unmanageable "
1823          + "number of regions.";
1824      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1825    }
1826
1827    // check flush size
1828    long flushSizeLowerLimit = 1024 * 1024L; // 1M is the default lower limit
1829    long flushSize = htd.getMemStoreFlushSize();
1830    if (flushSize < 0) {
1831      flushSize = conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSizeLowerLimit);
1832    }
1833    if (flushSize < conf.getLong("hbase.hregion.memstore.flush.size.limit", flushSizeLowerLimit)) {
1834      String message = "MEMSTORE_FLUSHSIZE for table descriptor or "
1835          + "\"hbase.hregion.memstore.flush.size\" ("+flushSize+") is too small, which might cause"
1836          + " very frequent flushing.";
1837      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1838    }
1839
1840    // check that coprocessors and other specified plugin classes can be loaded
1841    try {
1842      checkClassLoading(conf, htd);
1843    } catch (Exception ex) {
1844      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, ex.getMessage(), null);
1845    }
1846
1847    // check compression can be loaded
1848    try {
1849      checkCompression(htd);
1850    } catch (IOException e) {
1851      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1852    }
1853
1854    // check encryption can be loaded
1855    try {
1856      checkEncryption(conf, htd);
1857    } catch (IOException e) {
1858      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1859    }
1860    // Verify compaction policy
1861    try{
1862      checkCompactionPolicy(conf, htd);
1863    } catch(IOException e){
1864      warnOrThrowExceptionForFailure(false, CONF_KEY, e.getMessage(), e);
1865    }
1866    // check that we have at least 1 CF
1867    if (htd.getColumnFamilyCount() == 0) {
1868      String message = "Table should have at least one column family.";
1869      warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1870    }
1871
1872    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1873      if (hcd.getTimeToLive() <= 0) {
1874        String message = "TTL for column family " + hcd.getNameAsString() + " must be positive.";
1875        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1876      }
1877
1878      // check blockSize
1879      if (hcd.getBlocksize() < 1024 || hcd.getBlocksize() > 16 * 1024 * 1024) {
1880        String message = "Block size for column family " + hcd.getNameAsString()
1881            + "  must be between 1K and 16MB.";
1882        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1883      }
1884
1885      // check versions
1886      if (hcd.getMinVersions() < 0) {
1887        String message = "Min versions for column family " + hcd.getNameAsString()
1888          + "  must be positive.";
1889        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1890      }
1891      // max versions already being checked
1892
1893      // HBASE-13776 Setting illegal versions for HColumnDescriptor
1894      //  does not throw IllegalArgumentException
1895      // check minVersions <= maxVerions
1896      if (hcd.getMinVersions() > hcd.getMaxVersions()) {
1897        String message = "Min versions for column family " + hcd.getNameAsString()
1898            + " must be less than the Max versions.";
1899        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1900      }
1901
1902      // check replication scope
1903      checkReplicationScope(hcd);
1904
1905      // check data replication factor, it can be 0(default value) when user has not explicitly
1906      // set the value, in this case we use default replication factor set in the file system.
1907      if (hcd.getDFSReplication() < 0) {
1908        String message = "HFile Replication for column family " + hcd.getNameAsString()
1909            + "  must be greater than zero.";
1910        warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1911      }
1912
1913      // TODO: should we check coprocessors and encryption ?
1914    }
1915  }

1929  private void checkCompactionPolicy(Configuration conf, HTableDescriptor htd)
1930      throws IOException {
1931    // FIFO compaction has some requirements
1932    // Actually FCP ignores periodic major compactions
1933    String className =
1934        htd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1935    if (className == null) {
1936      className =
1937          conf.get(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY,
1938            ExploringCompactionPolicy.class.getName());
1939    }
1940
1941    int blockingFileCount = HStore.DEFAULT_BLOCKING_STOREFILE_COUNT;
1942    String sv = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1943    if (sv != null) {
1944      blockingFileCount = Integer.parseInt(sv);
1945    } else {
1946      blockingFileCount = conf.getInt(HStore.BLOCKING_STOREFILES_KEY, blockingFileCount);
1947    }
1948
1949    for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1950      String compactionPolicy =
1951          hcd.getConfigurationValue(DefaultStoreEngine.DEFAULT_COMPACTION_POLICY_CLASS_KEY);
1952      if (compactionPolicy == null) {
1953        compactionPolicy = className;
1954      }
1955      if (!compactionPolicy.equals(FIFOCompactionPolicy.class.getName())) {
1956        continue;
1957      }
1958      // FIFOCompaction
1959      String message = null;
1960
1961      // 1. Check TTL
1962      if (hcd.getTimeToLive() == HColumnDescriptor.DEFAULT_TTL) {
1963        message = "Default TTL is not supported for FIFO compaction";
1964        throw new IOException(message);
1965      }
1966
1967      // 2. Check min versions
1968      if (hcd.getMinVersions() > 0) {
1969        message = "MIN_VERSION > 0 is not supported for FIFO compaction";
1970        throw new IOException(message);
1971      }
1972
1973      // 3. blocking file count
1974      String sbfc = htd.getConfigurationValue(HStore.BLOCKING_STOREFILES_KEY);
1975      if (sbfc != null) {
1976        blockingFileCount = Integer.parseInt(sbfc);
1977      }
1978      if (blockingFileCount < 1000) {
1979        message =
1980            "blocking file count '" + HStore.BLOCKING_STOREFILES_KEY + "' " + blockingFileCount
1981                + " is below recommended minimum of 1000";
1982        throw new IOException(message);
1983      }
1984    }

1997  private void startActiveMasterManager(int infoPort) throws KeeperException {
1998    String backupZNode = ZKUtil.joinZNode(
1999      zooKeeper.znodePaths.backupMasterAddressesZNode, serverName.toString());
2000    /*
2001    * Add a ZNode for ourselves in the backup master directory since we
2002    * may not become the active master. If so, we want the actual active
2003    * master to know we are backup masters, so that it won't assign
2004    * regions to us if so configured.
2005    *
2006    * If we become the active master later, ActiveMasterManager will delete
2007    * this node explicitly.  If we crash before then, ZooKeeper will delete
2008    * this node for us since it is ephemeral.
2009    */
2010    LOG.info("Adding backup master ZNode " + backupZNode);
2011    if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode,
2012        serverName, infoPort)) {
2013      LOG.warn("Failed create of " + backupZNode + " by " + serverName);
2014    }
2015
2016    activeMasterManager.setInfoPort(infoPort);
2017    // Start a thread to try to become the active master, so we won't block here
2018    Threads.setDaemonThreadRunning(new Thread(new Runnable() {
2019      @Override
2020      public void run() {
2021        int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
2022          HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
2023        // If we're a backup master, stall until a primary to writes his address
2024        if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP,
2025          HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
2026          LOG.debug("HMaster started in backup mode. "
2027            + "Stalling until master znode is written.");
2028          // This will only be a minute or so while the cluster starts up,
2029          // so don't worry about setting watches on the parent znode
2030          while (!activeMasterManager.hasActiveMaster()) {
2031            LOG.debug("Waiting for master address ZNode to be written "
2032              + "(Also watching cluster state node)");
2033            Threads.sleep(timeout);
2034          }
2035        }
2036        MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
2037        status.setDescription("Master startup");
2038        try {
2039          if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
2040            finishActiveMasterInitialization(status);
2041          }
2042        } catch (Throwable t) {
2043          status.setStatus("Failed to become active: " + t.getMessage());
2044          LOG.fatal("Failed to become active master", t);
2045          // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
2046          if (t instanceof NoClassDefFoundError &&
2047            t.getMessage()
2048              .contains("org/apache/hadoop/hdfs/protocol/HdfsConstants$SafeModeAction")) {
2049            // improved error message for this special case
2050            abort("HBase is having a problem with its Hadoop jars.  You may need to "
2051              + "recompile HBase against Hadoop version "
2052              + org.apache.hadoop.util.VersionInfo.getVersion()
2053              + " or change your hadoop jars to start properly", t);
2054          } else {
2055            abort("Unhandled exception. Starting shutdown.", t);
2056          }
2057        } finally {
2058          status.cleanup();
2059        }
2060      }
2061    }, getServerName().toShortString() + ".masterManager"));
2062  }

2465   * @return cluster status
2466   */
2467  public ClusterStatus getClusterStatus() throws InterruptedIOException {
2468    // Build Set of backup masters from ZK nodes
2469    List<String> backupMasterStrings;
2470    try {
2471      backupMasterStrings = ZKUtil.listChildrenNoWatch(this.zooKeeper,
2472        this.zooKeeper.znodePaths.backupMasterAddressesZNode);
2473    } catch (KeeperException e) {
2474      LOG.warn(this.zooKeeper.prefix("Unable to list backup servers"), e);
2475      backupMasterStrings = null;
2476    }
2477
2478    List<ServerName> backupMasters = null;
2479    if (backupMasterStrings != null && !backupMasterStrings.isEmpty()) {
2480      backupMasters = new ArrayList<>(backupMasterStrings.size());
2481      for (String s: backupMasterStrings) {
2482        try {
2483          byte [] bytes;
2484          try {
2485            bytes = ZKUtil.getData(this.zooKeeper, ZKUtil.joinZNode(
2486                this.zooKeeper.znodePaths.backupMasterAddressesZNode, s));
2487          } catch (InterruptedException e) {
2488            throw new InterruptedIOException();
2489          }
2490          if (bytes != null) {
2491            ServerName sn;
2492            try {
2493              sn = ProtobufUtil.parseServerNameFrom(bytes);
2494            } catch (DeserializationException e) {
2495              LOG.warn("Failed parse, skipping registering backup server", e);
2496              continue;
2497            }
2498            backupMasters.add(sn);
2499          }
2500        } catch (KeeperException e) {
2501          LOG.warn(this.zooKeeper.prefix("Unable to get information about " +
2502                   "backup servers"), e);
2503        }
2504      }
2505      Collections.sort(backupMasters, new Comparator<ServerName>() {
2506        @Override
2507        public int compare(ServerName s1, ServerName s2) {
2508          return s1.getServerName().compareTo(s2.getServerName());
2509        }});
2510    }
2511
2512    String clusterId = fileSystemManager != null ?
2513      fileSystemManager.getClusterId().toString() : null;
2514    List<RegionState> regionsInTransition = assignmentManager != null ?
2515      assignmentManager.getRegionStates().getRegionsStateInTransition() : null;
2516
2517    String[] coprocessors = cpHost != null ? getMasterCoprocessors() : null;
2518    boolean balancerOn = loadBalancerTracker != null ?
2519      loadBalancerTracker.isBalancerOn() : false;
2520    Map<ServerName, ServerLoad> onlineServers = null;
2521    Set<ServerName> deadServers = null;
2522    if (serverManager != null) {
2523      deadServers = serverManager.getDeadServers().copyServerNames();
2524      onlineServers = serverManager.getOnlineServers();
2525    }
2526    return new ClusterStatus(VersionInfo.getVersion(), clusterId,
2527      onlineServers, deadServers, serverName, backupMasters,
2528      regionsInTransition, coprocessors, balancerOn);
2529  }
