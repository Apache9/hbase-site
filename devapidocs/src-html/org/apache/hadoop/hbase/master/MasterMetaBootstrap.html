021import java.io.IOException;
022import java.util.List;
023import java.util.Set;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.HConstants;
028import org.apache.hadoop.hbase.HRegionInfo;
029import org.apache.hadoop.hbase.ServerName;
030import org.apache.hadoop.hbase.TableName;
031import org.apache.yetus.audience.InterfaceAudience;
032import org.apache.hadoop.hbase.client.RegionReplicaUtil;
033import org.apache.hadoop.hbase.client.TableState;
034import org.apache.hadoop.hbase.master.assignment.AssignmentManager;
035import org.apache.hadoop.hbase.monitoring.MonitoredTask;
036import org.apache.hadoop.hbase.zookeeper.MetaTableLocator;
037import org.apache.hadoop.hbase.zookeeper.ZKUtil;
038import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
039import org.apache.zookeeper.KeeperException;
040
041/**
042 * Used by the HMaster on startup to split meta logs and assign the meta table.
043 */
044@InterfaceAudience.Private
045public class MasterMetaBootstrap {
046  private static final Log LOG = LogFactory.getLog(MasterMetaBootstrap.class);
047
048  private final MonitoredTask status;
049  private final HMaster master;
050
051  public MasterMetaBootstrap(final HMaster master, final MonitoredTask status) {
052    this.master = master;
053    this.status = status;
054  }
055
056  public void recoverMeta() throws InterruptedException, IOException {
057    master.recoverMeta();
058    master.getTableStateManager().start();
059    enableCrashedServerProcessing(false);
060  }
061
062  public void processDeadServers() {
063    // get a list for previously failed RS which need log splitting work
064    // we recover hbase:meta region servers inside master initialization and
065    // handle other failed servers in SSH in order to start up master node ASAP
066    Set<ServerName> previouslyFailedServers =
067        master.getMasterWalManager().getFailedServersFromLogFolders();
068
069    // Master has recovered hbase:meta region server and we put
070    // other failed region servers in a queue to be handled later by SSH
071    for (ServerName tmpServer : previouslyFailedServers) {
072      master.getServerManager().processDeadServer(tmpServer, true);
073    }
074  }
075
076  public void assignMetaReplicas()
077      throws IOException, InterruptedException, KeeperException {
078    int numReplicas = master.getConfiguration().getInt(HConstants.META_REPLICAS_NUM,
079           HConstants.DEFAULT_META_REPLICA_NUM);
080    for (int i = 1; i < numReplicas; i++) {
081      assignMeta(i);
082    }
083    unassignExcessMetaReplica(numReplicas);
084  }
085
086  private void unassignExcessMetaReplica(int numMetaReplicasConfigured) {
087    final ZooKeeperWatcher zooKeeper = master.getZooKeeper();
088    // unassign the unneeded replicas (for e.g., if the previous master was configured
089    // with a replication of 3 and now it is 2, we need to unassign the 1 unneeded replica)
090    try {
091      List<String> metaReplicaZnodes = zooKeeper.getMetaReplicaNodes();
092      for (String metaReplicaZnode : metaReplicaZnodes) {
093        int replicaId = zooKeeper.znodePaths.getMetaReplicaIdFromZnode(metaReplicaZnode);
094        if (replicaId >= numMetaReplicasConfigured) {
095          RegionState r = MetaTableLocator.getMetaRegionState(zooKeeper, replicaId);
096          LOG.info("Closing excess replica of meta region " + r.getRegion());
097          // send a close and wait for a max of 30 seconds
098          ServerManager.closeRegionSilentlyAndWait(master.getClusterConnection(),
099              r.getServerName(), r.getRegion(), 30000);
100          ZKUtil.deleteNode(zooKeeper, zooKeeper.znodePaths.getZNodeForReplica(replicaId));
101        }
102      }
103    } catch (Exception ex) {
104      // ignore the exception since we don't want the master to be wedged due to potential
105      // issues in the cleanup of the extra regions. We can do that cleanup via hbck or manually
106      LOG.warn("Ignoring exception " + ex);
107    }
108  }
109
110  /**
111   * Check <code>hbase:meta</code> is assigned. If not, assign it.
112   */
113  protected void assignMeta(int replicaId)
114      throws InterruptedException, IOException, KeeperException {
115    final AssignmentManager assignmentManager = master.getAssignmentManager();
116
117    // Work on meta region
118    // TODO: Unimplemented
119    // long timeout =
120    //   master.getConfiguration().getLong("hbase.catalog.verification.timeout", 1000);
121    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
122      status.setStatus("Assigning hbase:meta region");
123    } else {
124      status.setStatus("Assigning hbase:meta region, replicaId " + replicaId);
125    }
126
127    // Get current meta state from zk.
128    RegionState metaState = MetaTableLocator.getMetaRegionState(master.getZooKeeper(), replicaId);
129    LOG.debug("meta state from zookeeper: " + metaState);
130    HRegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(
131      HRegionInfo.FIRST_META_REGIONINFO, replicaId);
132    assignmentManager.assignMeta(hri, metaState.getServerName());
133
134    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
135      // TODO: should we prevent from using state manager before meta was initialized?
136      // tableStateManager.start();
137      master.getTableStateManager()
138        .setTableState(TableName.META_TABLE_NAME, TableState.State.ENABLED);
139    }
140
141    master.getTableStateManager().start();
142
143    // Make sure a hbase:meta location is set. We need to enable SSH here since
144    // if the meta region server is died at this time, we need it to be re-assigned
145    // by SSH so that system tables can be assigned.
146    // No need to wait for meta is assigned = 0 when meta is just verified.
147    if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableCrashedServerProcessing(false);
148    LOG.info("hbase:meta with replicaId " + replicaId + ", location="
149      + master.getMetaTableLocator().getMetaRegionLocation(master.getZooKeeper(), replicaId));
150    status.setStatus("META assigned.");
151  }
152
153  private void enableCrashedServerProcessing(final boolean waitForMeta)
154      throws InterruptedException {
155    // If crashed server processing is disabled, we enable it and expire those dead but not expired
156    // servers. This is required so that if meta is assigning to a server which dies after
157    // assignMeta starts assignment, ServerCrashProcedure can re-assign it. Otherwise, we will be
158    // stuck here waiting forever if waitForMeta is specified.
159    if (!master.isServerCrashProcessingEnabled()) {
160      master.setServerCrashProcessingEnabled(true);
161      master.getServerManager().processQueuedDeadServers();
162    }
163
164    if (waitForMeta) {
165      master.getMetaTableLocator().waitMetaRegionLocation(master.getZooKeeper());
166    }
167  }
168}
