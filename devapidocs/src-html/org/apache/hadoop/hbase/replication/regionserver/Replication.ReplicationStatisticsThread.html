107  public void initialize(final Server server, final FileSystem fs,
108      final Path logDir, final Path oldLogDir) throws IOException {
109    this.server = server;
110    this.conf = this.server.getConfiguration();
111    this.replicationForBulkLoadData = isReplicationForBulkLoadDataEnabled(this.conf);
112    this.scheduleThreadPool = Executors.newScheduledThreadPool(1,
113      new ThreadFactoryBuilder()
114        .setNameFormat(server.getServerName().toShortString() + "Replication Statistics #%d")
115        .setDaemon(true)
116        .build());
117    if (this.replicationForBulkLoadData) {
118      if (conf.get(HConstants.REPLICATION_CLUSTER_ID) == null
119          || conf.get(HConstants.REPLICATION_CLUSTER_ID).isEmpty()) {
120        throw new IllegalArgumentException(HConstants.REPLICATION_CLUSTER_ID
121            + " cannot be null/empty when " + HConstants.REPLICATION_BULKLOAD_ENABLE_KEY
122            + " is set to true.");
123      }
124    }
125
126    try {
127      this.replicationQueues =
128          ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, this.server,
129            server.getZooKeeper()));
130      this.replicationQueues.init(this.server.getServerName().toString());
131      this.replicationPeers =
132          ReplicationFactory.getReplicationPeers(server.getZooKeeper(), this.conf, this.server);
133      this.replicationPeers.init();
134      this.replicationTracker =
135          ReplicationFactory.getReplicationTracker(server.getZooKeeper(), this.replicationPeers,
136            this.conf, this.server, this.server);
137    } catch (Exception e) {
138      throw new IOException("Failed replication handler create", e);
139    }
140    UUID clusterId = null;
141    try {
142      clusterId = ZKClusterId.getUUIDForCluster(this.server.getZooKeeper());
143    } catch (KeeperException ke) {
144      throw new IOException("Could not read cluster id", ke);
145    }
146    this.replicationManager =
147        new ReplicationSourceManager(replicationQueues, replicationPeers, replicationTracker,
148            conf, this.server, fs, logDir, oldLogDir, clusterId);
149    this.statsThreadPeriod =
150        this.conf.getInt("replication.stats.thread.period.seconds", 5 * 60);
151    LOG.debug("ReplicationStatisticsThread " + this.statsThreadPeriod);
152    this.replicationLoad = new ReplicationLoad();
153  }

177  /**
178   * Join with the replication threads
179   */
180  public void join() {
181    this.replicationManager.join();
182    if (this.replicationSink != null) {
183      this.replicationSink.stopReplicationSinkServices();
184    }
185    scheduleThreadPool.shutdown();
186  }

208  /**
209   * If replication is enabled and this cluster is a master,
210   * it starts
211   * @throws IOException
212   */
213  public void startReplicationService() throws IOException {
214    try {
215      this.replicationManager.init();
216    } catch (ReplicationException e) {
217      throw new IOException(e);
218    }
219    this.replicationSink = new ReplicationSink(this.conf, this.server);
220    this.scheduleThreadPool.scheduleAtFixedRate(
221      new ReplicationStatisticsThread(this.replicationSink, this.replicationManager),
222      statsThreadPeriod, statsThreadPeriod, TimeUnit.SECONDS);
223  }

238  /**
239   * Utility method used to set the correct scopes on each log key. Doesn't set a scope on keys from
240   * compaction WAL edits and if the scope is local.
241   * @param logKey Key that may get scoped according to its edits
242   * @param logEdit Edits used to lookup the scopes
243   * @param replicationManager Manager used to add bulk load events hfile references
244   * @throws IOException If failed to parse the WALEdit
245   */
246  public static void scopeWALEdits(WALKey logKey,
247      WALEdit logEdit, Configuration conf, ReplicationSourceManager replicationManager)
248          throws IOException {
249    boolean replicationForBulkLoadEnabled = isReplicationForBulkLoadDataEnabled(conf);
250    boolean foundOtherEdits = false;
251    for (Cell cell : logEdit.getCells()) {
252      if (!CellUtil.matchingFamily(cell, WALEdit.METAFAMILY)) {
253        foundOtherEdits = true;
254        break;
255      }
256    }
257
258    if (!foundOtherEdits && logEdit.getCells().size() > 0) {
259      WALProtos.RegionEventDescriptor maybeEvent =
260          WALEdit.getRegionEventDescriptor(logEdit.getCells().get(0));
261      if (maybeEvent != null && (maybeEvent.getEventType() ==
262          WALProtos.RegionEventDescriptor.EventType.REGION_CLOSE)) {
263        // In serially replication, we use scopes when reading close marker.
264        foundOtherEdits = true;
265      }
266    }
267    if ((!replicationForBulkLoadEnabled && !foundOtherEdits) || logEdit.isReplay()) {
268      logKey.serializeReplicationScope(false);
269    }
270  }

292  /**
293   * This method modifies the master's configuration in order to inject replication-related features
294   * @param conf
295   */
296  public static void decorateMasterConfiguration(Configuration conf) {
297    String plugins = conf.get(HBASE_MASTER_LOGCLEANER_PLUGINS);
298    String cleanerClass = ReplicationLogCleaner.class.getCanonicalName();
299    if (!plugins.contains(cleanerClass)) {
300      conf.set(HBASE_MASTER_LOGCLEANER_PLUGINS, plugins + "," + cleanerClass);
301    }
302    if (isReplicationForBulkLoadDataEnabled(conf)) {
303      plugins = conf.get(HFileCleaner.MASTER_HFILE_CLEANER_PLUGINS);
304      cleanerClass = ReplicationHFileCleaner.class.getCanonicalName();
305      if (!plugins.contains(cleanerClass)) {
306        conf.set(HFileCleaner.MASTER_HFILE_CLEANER_PLUGINS, plugins + "," + cleanerClass);
307      }
308    }
309  }
310
311  /**
312   * This method modifies the region server's configuration in order to inject replication-related
313   * features
314   * @param conf region server configurations
315   */
316  public static void decorateRegionServerConfiguration(Configuration conf) {
317    if (isReplicationForBulkLoadDataEnabled(conf)) {
318      String plugins = conf.get(CoprocessorHost.REGIONSERVER_COPROCESSOR_CONF_KEY, "");
319      String rsCoprocessorClass = ReplicationObserver.class.getCanonicalName();
320      if (!plugins.contains(rsCoprocessorClass)) {
321        conf.set(CoprocessorHost.REGIONSERVER_COPROCESSOR_CONF_KEY,
322          plugins + "," + rsCoprocessorClass);
323      }
324    }
325  }

327  /*
328   * Statistics thread. Periodically prints the cache statistics to the log.
329   */
330  static class ReplicationStatisticsThread extends Thread {
331
332    private final ReplicationSink replicationSink;
333    private final ReplicationSourceManager replicationManager;
334
335    public ReplicationStatisticsThread(final ReplicationSink replicationSink,
336                            final ReplicationSourceManager replicationManager) {
337      super("ReplicationStatisticsThread");
338      this.replicationManager = replicationManager;
339      this.replicationSink = replicationSink;
340    }
341
342    @Override
343    public void run() {
344      printStats(this.replicationManager.getStats());
345      printStats(this.replicationSink.getStats());
346    }
347
348    private void printStats(String stats) {
349      if (!stats.isEmpty()) {
350        LOG.info(stats);
351      }
352    }
353  }
354
355  @Override
356  public ReplicationLoad refreshAndGetReplicationLoad() {
357    if (this.replicationLoad == null) {
358      return null;
359    }
360    // always build for latest data
361    buildReplicationLoad();
362    return this.replicationLoad;
363  }
364
365  private void buildReplicationLoad() {
366    List<MetricsSource> sourceMetricsList = new ArrayList<>();
367
368    // get source
369    List<ReplicationSourceInterface> sources = this.replicationManager.getSources();
370    for (ReplicationSourceInterface source : sources) {
371      sourceMetricsList.add(source.getSourceMetrics());
372    }
373
374    // get old source
375    List<ReplicationSourceInterface> oldSources = this.replicationManager.getOldSources();
376    for (ReplicationSourceInterface source : oldSources) {
377      if (source instanceof ReplicationSource) {
378        sourceMetricsList.add(((ReplicationSource) source).getSourceMetrics());
379      }
380    }
381
382    // get sink
383    MetricsSink sinkMetrics = this.replicationSink.getSinkMetrics();
384    this.replicationLoad.buildReplicationLoad(sourceMetricsList, sinkMetrics);
385  }
