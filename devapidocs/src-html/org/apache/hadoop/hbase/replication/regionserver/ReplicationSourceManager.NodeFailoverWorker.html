075/**
076 * This class is responsible to manage all the replication
077 * sources. There are two classes of sources:
078 * <ul>
079 * <li> Normal sources are persistent and one per peer cluster</li>
080 * <li> Old sources are recovered from a failed region server and our
081 * only goal is to finish replicating the WAL queue it had up in ZK</li>
082 * </ul>
083 *
084 * When a region server dies, this class uses a watcher to get notified and it
085 * tries to grab a lock in order to transfer all the queues in a local
086 * old source.
087 *
088 * This class implements the ReplicationListener interface so that it can track changes in
089 * replication state.
090 */
091@InterfaceAudience.Private
092public class ReplicationSourceManager implements ReplicationListener {
093  private static final Log LOG =
094      LogFactory.getLog(ReplicationSourceManager.class);
095  // List of all the sources that read this RS's logs
096  private final List<ReplicationSourceInterface> sources;
097  // List of all the sources we got from died RSs
098  private final List<ReplicationSourceInterface> oldsources;
099  private final ReplicationQueues replicationQueues;
100  private final ReplicationTracker replicationTracker;
101  private final ReplicationPeers replicationPeers;
102  // UUID for this cluster
103  private final UUID clusterId;
104  // All about stopping
105  private final Server server;
106  // All logs we are currently tracking
107  // Index structure of the map is: peer_id->logPrefix/logGroup->logs
108  private final Map<String, Map<String, SortedSet<String>>> walsById;
109  // Logs for recovered sources we are currently tracking
110  private final Map<String, Map<String, SortedSet<String>>> walsByIdRecoveredQueues;
111  private final Configuration conf;
112  private final FileSystem fs;
113  // The paths to the latest log of each wal group, for new coming peers
114  private Set<Path> latestPaths;
115  // Path to the wals directories
116  private final Path logDir;
117  // Path to the wal archive
118  private final Path oldLogDir;

121  // Homemade executer service for replication

132  /**
133   * Creates a replication manager and sets the watch on all the other registered region servers
134   * @param replicationQueues the interface for manipulating replication queues
135   * @param replicationPeers
136   * @param replicationTracker
137   * @param conf the configuration to use
138   * @param server the server for this region server
139   * @param fs the file system to use
140   * @param logDir the directory that contains all wal directories of live RSs
141   * @param oldLogDir the directory where old logs are archived
142   * @param clusterId
143   */
144  public ReplicationSourceManager(final ReplicationQueues replicationQueues,
145      final ReplicationPeers replicationPeers, final ReplicationTracker replicationTracker,
146      final Configuration conf, final Server server, final FileSystem fs, final Path logDir,
147      final Path oldLogDir, final UUID clusterId) throws IOException {
148    //CopyOnWriteArrayList is thread-safe.
149    //Generally, reading is more than modifying.
150    this.sources = new CopyOnWriteArrayList<>();
151    this.replicationQueues = replicationQueues;
152    this.replicationPeers = replicationPeers;
153    this.replicationTracker = replicationTracker;
154    this.server = server;
155    this.walsById = new HashMap<>();
156    this.walsByIdRecoveredQueues = new ConcurrentHashMap<>();
157    this.oldsources = new CopyOnWriteArrayList<>();
158    this.conf = conf;
159    this.fs = fs;
160    this.logDir = logDir;
161    this.oldLogDir = oldLogDir;
162    this.sleepBeforeFailover =
163        conf.getLong("replication.sleep.before.failover", 30000); // 30 seconds
164    this.clusterId = clusterId;
165    this.replicationTracker.registerListener(this);
166    this.replicationPeers.getAllPeerIds();
167    // It's preferable to failover 1 RS at a time, but with good zk servers
168    // more could be processed at the same time.
169    int nbWorkers = conf.getInt("replication.executor.workers", 1);
170    // use a short 100ms sleep since this could be done inline with a RS startup
171    // even if we fail, other region servers can take care of it
172    this.executor = new ThreadPoolExecutor(nbWorkers, nbWorkers,
173        100, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<>());
174    ThreadFactoryBuilder tfb = new ThreadFactoryBuilder();
175    tfb.setNameFormat("ReplicationExecutor-%d");
176    tfb.setDaemon(true);
177    this.executor.setThreadFactory(tfb.build());
178    this.rand = new Random();
179    this.latestPaths = Collections.synchronizedSet(new HashSet<Path>());
180    replicationForBulkLoadDataEnabled =
181        conf.getBoolean(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY,
182          HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT);
183    this.replicationWaitTime = conf.getLong(HConstants.REPLICATION_SERIALLY_WAITING_KEY,
184          HConstants.REPLICATION_SERIALLY_WAITING_DEFAULT);
185    connection = ConnectionFactory.createConnection(conf);
186  }

188  /**
189   * Provide the id of the peer and a log key and this method will figure which
190   * wal it belongs to and will log, for this region server, the current
191   * position. It will also clean old logs from the queue.
192   * @param log Path to the log currently being replicated from
193   * replication status in zookeeper. It will also delete older entries.
194   * @param id id of the peer cluster
195   * @param position current location in the log
196   * @param queueRecovered indicates if this queue comes from another region server
197   * @param holdLogInZK if true then the log is retained in ZK
198   */
199  public void logPositionAndCleanOldLogs(Path log, String id, long position,
200      boolean queueRecovered, boolean holdLogInZK) {
201    String fileName = log.getName();
202    this.replicationQueues.setLogPosition(id, fileName, position);
203    if (holdLogInZK) {
204     return;
205    }
206    cleanOldLogs(fileName, id, queueRecovered);
207  }

209  /**
210   * Cleans a log file and all older files from ZK. Called when we are sure that a
211   * log file is closed and has no more entries.
212   * @param key Path to the log
213   * @param id id of the peer cluster
214   * @param queueRecovered Whether this is a recovered queue
215   */
216  public void cleanOldLogs(String key, String id, boolean queueRecovered) {
217    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(key);
218    if (queueRecovered) {
219      SortedSet<String> wals = walsByIdRecoveredQueues.get(id).get(logPrefix);
220      if (wals != null && !wals.first().equals(key)) {
221        cleanOldLogs(wals, key, id);
222      }
223    } else {
224      synchronized (this.walsById) {
225        SortedSet<String> wals = walsById.get(id).get(logPrefix);
226        if (wals != null && !wals.first().equals(key)) {
227          cleanOldLogs(wals, key, id);
228        }
229      }
230    }
231 }
232
233  private void cleanOldLogs(SortedSet<String> wals, String key, String id) {
234    SortedSet<String> walSet = wals.headSet(key);
235    LOG.debug("Removing " + walSet.size() + " logs in the list: " + walSet);
236    for (String wal : walSet) {
237      this.replicationQueues.removeLog(id, wal);
238    }
239    walSet.clear();
240  }

242  /**
243   * Adds a normal source per registered peer cluster and tries to process all
244   * old region server wal queues
245   */
246  protected void init() throws IOException, ReplicationException {
247    for (String id : this.replicationPeers.getConnectedPeerIds()) {
248      addSource(id);
249      if (replicationForBulkLoadDataEnabled) {
250        // Check if peer exists in hfile-refs queue, if not add it. This can happen in the case
251        // when a peer was added before replication for bulk loaded data was enabled.
252        this.replicationQueues.addPeerToHFileRefs(id);
253      }
254    }
255    AdoptAbandonedQueuesWorker adoptionWorker = new AdoptAbandonedQueuesWorker();
256    try {
257      this.executor.execute(adoptionWorker);
258    } catch (RejectedExecutionException ex) {
259      LOG.info("Cancelling the adoption of abandoned queues because of " + ex.getMessage());
260    }
261  }

263  /**
264   * Add sources for the given peer cluster on this region server. For the newly added peer, we only
265   * need to enqueue the latest log of each wal group and do replication
266   * @param id the id of the peer cluster
267   * @return the source that was created
268   * @throws IOException
269   */
270  protected ReplicationSourceInterface addSource(String id) throws IOException,
271      ReplicationException {
272    ReplicationPeerConfig peerConfig = replicationPeers.getReplicationPeerConfig(id);
273    ReplicationPeer peer = replicationPeers.getConnectedPeer(id);
274    ReplicationSourceInterface src =
275        getReplicationSource(this.conf, this.fs, this, this.replicationQueues,
276          this.replicationPeers, server, id, this.clusterId, peerConfig, peer);
277    synchronized (this.walsById) {
278      this.sources.add(src);
279      Map<String, SortedSet<String>> walsByGroup = new HashMap<>();
280      this.walsById.put(id, walsByGroup);
281      // Add the latest wal to that source's queue
282      synchronized (latestPaths) {
283        if (this.latestPaths.size() > 0) {
284          for (Path logPath : latestPaths) {
285            String name = logPath.getName();
286            String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(name);
287            SortedSet<String> logs = new TreeSet<>();
288            logs.add(name);
289            walsByGroup.put(walPrefix, logs);
290            try {
291              this.replicationQueues.addLog(id, name);
292            } catch (ReplicationException e) {
293              String message =
294                  "Cannot add log to queue when creating a new source, queueId=" + id
295                      + ", filename=" + name;
296              server.stop(message);
297              throw e;
298            }
299            src.enqueueLog(logPath);
300          }
301        }
302      }
303    }
304    src.startup();
305    return src;
306  }

380  void preLogRoll(Path newLog) throws IOException {
381    recordLog(newLog);
382    String logName = newLog.getName();
383    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(logName);
384    synchronized (latestPaths) {
385      Iterator<Path> iterator = latestPaths.iterator();
386      while (iterator.hasNext()) {
387        Path path = iterator.next();
388        if (path.getName().contains(logPrefix)) {
389          iterator.remove();
390          break;
391        }
392      }
393      this.latestPaths.add(newLog);
394    }
395  }

397  /**
398   * Check and enqueue the given log to the correct source. If there's still no source for the
399   * group to which the given log belongs, create one
400   * @param logPath the log path to check and enqueue
401   * @throws IOException
402   */
403  private void recordLog(Path logPath) throws IOException {
404    String logName = logPath.getName();
405    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(logName);
406    // update replication queues on ZK
407    // synchronize on replicationPeers to avoid adding source for the to-be-removed peer
408    synchronized (replicationPeers) {
409      for (String id : replicationPeers.getConnectedPeerIds()) {
410        try {
411          this.replicationQueues.addLog(id, logName);
412        } catch (ReplicationException e) {
413          throw new IOException("Cannot add log to replication queue"
414              + " when creating a new source, queueId=" + id + ", filename=" + logName, e);
415        }
416      }
417    }
418    // update walsById map
419    synchronized (walsById) {
420      for (Map.Entry<String, Map<String, SortedSet<String>>> entry : this.walsById.entrySet()) {
421        String peerId = entry.getKey();
422        Map<String, SortedSet<String>> walsByPrefix = entry.getValue();
423        boolean existingPrefix = false;
424        for (Map.Entry<String, SortedSet<String>> walsEntry : walsByPrefix.entrySet()) {
425          SortedSet<String> wals = walsEntry.getValue();
426          if (this.sources.isEmpty()) {
427            // If there's no slaves, don't need to keep the old wals since
428            // we only consider the last one when a new slave comes in
429            wals.clear();
430          }
431          if (logPrefix.equals(walsEntry.getKey())) {
432            wals.add(logName);
433            existingPrefix = true;
434          }
435        }
436        if (!existingPrefix) {
437          // The new log belongs to a new group, add it into this peer
438          LOG.debug("Start tracking logs for wal group " + logPrefix + " for peer " + peerId);
439          SortedSet<String> wals = new TreeSet<>();
440          wals.add(logName);
441          walsByPrefix.put(logPrefix, wals);
442        }
443      }
444    }
445  }

447  void postLogRoll(Path newLog) throws IOException {
448    // This only updates the sources we own, not the recovered ones
449    for (ReplicationSourceInterface source : this.sources) {
450      source.enqueueLog(newLog);
451    }
452  }

459  /**
460   * Factory method to create a replication source
461   * @param conf the configuration to use
462   * @param fs the file system to use
463   * @param manager the manager to use
464   * @param server the server object for this region server
465   * @param peerId the id of the peer cluster
466   * @return the created source
467   * @throws IOException
468   */
469  protected ReplicationSourceInterface getReplicationSource(final Configuration conf,
470      final FileSystem fs, final ReplicationSourceManager manager,
471      final ReplicationQueues replicationQueues, final ReplicationPeers replicationPeers,
472      final Server server, final String peerId, final UUID clusterId,
473      final ReplicationPeerConfig peerConfig, final ReplicationPeer replicationPeer)
474      throws IOException {
475    RegionServerCoprocessorHost rsServerHost = null;
476    TableDescriptors tableDescriptors = null;
477    if (server instanceof HRegionServer) {
478      rsServerHost = ((HRegionServer) server).getRegionServerCoprocessorHost();
479      tableDescriptors = ((HRegionServer) server).getTableDescriptors();
480    }
481
482    ReplicationSourceInterface src = ReplicationSourceFactory.create(conf, peerId);
483
484    ReplicationEndpoint replicationEndpoint = null;
485    try {
486      String replicationEndpointImpl = peerConfig.getReplicationEndpointImpl();
487      if (replicationEndpointImpl == null) {
488        // Default to HBase inter-cluster replication endpoint
489        replicationEndpointImpl = HBaseInterClusterReplicationEndpoint.class.getName();
490      }
491      @SuppressWarnings("rawtypes")
492      Class c = Class.forName(replicationEndpointImpl);
493      replicationEndpoint = (ReplicationEndpoint) c.newInstance();
494      if(rsServerHost != null) {
495        ReplicationEndpoint newReplicationEndPoint = rsServerHost
496            .postCreateReplicationEndPoint(replicationEndpoint);
497        if(newReplicationEndPoint != null) {
498          // Override the newly created endpoint from the hook with configured end point
499          replicationEndpoint = newReplicationEndPoint;
500        }
501      }
502    } catch (Exception e) {
503      LOG.warn("Passed replication endpoint implementation throws errors"
504          + " while initializing ReplicationSource for peer: " + peerId, e);
505      throw new IOException(e);
506    }
507
508    MetricsSource metrics = new MetricsSource(peerId);
509    // init replication source
510    src.init(conf, fs, manager, replicationQueues, replicationPeers, server, peerId,
511      clusterId, replicationEndpoint, metrics);
512
513    // init replication endpoint
514    replicationEndpoint.init(new ReplicationEndpoint.Context(replicationPeer.getConfiguration(),
515      fs, peerId, clusterId, replicationPeer, metrics, tableDescriptors, server));
516
517    return src;
518  }

520  /**
521   * Transfer all the queues of the specified to this region server.
522   * First it tries to grab a lock and if it works it will move the
523   * znodes and finally will delete the old znodes.
524   *
525   * It creates one old source for any type of source of the old rs.
526   * @param rsZnode
527   */
528  private void transferQueues(String rsZnode) {
529    NodeFailoverWorker transfer =
530        new NodeFailoverWorker(rsZnode, this.replicationQueues, this.replicationPeers,
531            this.clusterId);
532    try {
533      this.executor.execute(transfer);
534    } catch (RejectedExecutionException ex) {
535      LOG.info("Cancelling the transfer of " + rsZnode + " because of " + ex.getMessage());
536    }
537  }

539  /**
540   * Clear the references to the specified old source
541   * @param src source to clear
542   */
543  public void closeRecoveredQueue(ReplicationSourceInterface src) {
544    LOG.info("Done with the recovered queue " + src.getPeerClusterZnode());
545    if (src instanceof ReplicationSource) {
546      ((ReplicationSource) src).getSourceMetrics().clear();
547    }
548    this.oldsources.remove(src);
549    deleteSource(src.getPeerClusterZnode(), false);
550    this.walsByIdRecoveredQueues.remove(src.getPeerClusterZnode());
551  }

553  /**
554   * Clear the references to the specified old source
555   * @param src source to clear
556   */
557  public void closeQueue(ReplicationSourceInterface src) {
558    LOG.info("Done with the queue " + src.getPeerClusterZnode());
559    src.getSourceMetrics().clear();
560    this.sources.remove(src);
561    deleteSource(src.getPeerClusterZnode(), true);
562    this.walsById.remove(src.getPeerClusterZnode());
563  }

565  /**
566   * Thie method first deletes all the recovered sources for the specified
567   * id, then deletes the normal source (deleting all related data in ZK).
568   * @param id The id of the peer cluster
569   */
570  public void removePeer(String id) {
571    LOG.info("Closing the following queue " + id + ", currently have "
572        + sources.size() + " and another "
573        + oldsources.size() + " that were recovered");
574    String terminateMessage = "Replication stream was removed by a user";
575    List<ReplicationSourceInterface> oldSourcesToDelete = new ArrayList<>();
576    // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
577    // see NodeFailoverWorker.run
578    synchronized (oldsources) {
579      // First close all the recovered sources for this peer
580      for (ReplicationSourceInterface src : oldsources) {
581        if (id.equals(src.getPeerId())) {
582          oldSourcesToDelete.add(src);
583        }
584      }
585      for (ReplicationSourceInterface src : oldSourcesToDelete) {
586        src.terminate(terminateMessage);
587        closeRecoveredQueue(src);
588      }
589    }
590    LOG.info("Number of deleted recovered sources for " + id + ": "
591        + oldSourcesToDelete.size());
592    // Now look for the one on this cluster
593    List<ReplicationSourceInterface> srcToRemove = new ArrayList<>();
594    // synchronize on replicationPeers to avoid adding source for the to-be-removed peer
595    synchronized (this.replicationPeers) {
596      for (ReplicationSourceInterface src : this.sources) {
597        if (id.equals(src.getPeerId())) {
598          srcToRemove.add(src);
599        }
600      }
601      if (srcToRemove.isEmpty()) {
602        LOG.error("The peer we wanted to remove is missing a ReplicationSourceInterface. " +
603            "This could mean that ReplicationSourceInterface initialization failed for this peer " +
604            "and that replication on this peer may not be caught up. peerId=" + id);
605      }
606      for (ReplicationSourceInterface toRemove : srcToRemove) {
607        toRemove.terminate(terminateMessage);
608        closeQueue(toRemove);
609      }
610      deleteSource(id, true);
611    }
612  }

625  @Override
626  public void peerListChanged(List<String> peerIds) {
627    for (String id : peerIds) {
628      try {
629        boolean added = this.replicationPeers.peerConnected(id);
630        if (added) {
631          addSource(id);
632          if (replicationForBulkLoadDataEnabled) {
633            this.replicationQueues.addPeerToHFileRefs(id);
634          }
635        }
636      } catch (Exception e) {

642  /**
643   * Class responsible to setup new ReplicationSources to take care of the
644   * queues from dead region servers.
645   */
646  class NodeFailoverWorker extends Thread {
647
648    private String rsZnode;
649    private final ReplicationQueues rq;
650    private final ReplicationPeers rp;
651    private final UUID clusterId;
652
653    /**
654     * @param rsZnode
655     */
656    public NodeFailoverWorker(String rsZnode) {
657      this(rsZnode, replicationQueues, replicationPeers, ReplicationSourceManager.this.clusterId);
658    }
659
660    public NodeFailoverWorker(String rsZnode, final ReplicationQueues replicationQueues,
661        final ReplicationPeers replicationPeers, final UUID clusterId) {
662      super("Failover-for-"+rsZnode);
663      this.rsZnode = rsZnode;
664      this.rq = replicationQueues;
665      this.rp = replicationPeers;
666      this.clusterId = clusterId;
667    }
668
669    @Override
670    public void run() {
671      if (this.rq.isThisOurRegionServer(rsZnode)) {
672        return;
673      }
674      // Wait a bit before transferring the queues, we may be shutting down.
675      // This sleep may not be enough in some cases.
676      try {
677        Thread.sleep(sleepBeforeFailover + (long) (rand.nextFloat() * sleepBeforeFailover));
678      } catch (InterruptedException e) {
679        LOG.warn("Interrupted while waiting before transferring a queue.");
680        Thread.currentThread().interrupt();
681      }
682      // We try to lock that rs' queue directory
683      if (server.isStopped()) {
684        LOG.info("Not transferring queue since we are shutting down");
685        return;
686      }
687      Map<String, Set<String>> newQueues = new HashMap<>();
688      List<String> peers = rq.getUnClaimedQueueIds(rsZnode);
689      while (peers != null && !peers.isEmpty()) {
690        Pair<String, SortedSet<String>> peer = this.rq.claimQueue(rsZnode,
691            peers.get(rand.nextInt(peers.size())));
692        long sleep = sleepBeforeFailover/2;
693        if (peer != null) {
694          newQueues.put(peer.getFirst(), peer.getSecond());
695          sleep = sleepBeforeFailover;
696        }
697        try {
698          Thread.sleep(sleep);
699        } catch (InterruptedException e) {
700          LOG.warn("Interrupted while waiting before transferring a queue.");
701          Thread.currentThread().interrupt();
702        }
703        peers = rq.getUnClaimedQueueIds(rsZnode);
704      }
705      if (peers != null) {
706        rq.removeReplicatorIfQueueIsEmpty(rsZnode);
707      }
708      // Copying over the failed queue is completed.
709      if (newQueues.isEmpty()) {
710        // We either didn't get the lock or the failed region server didn't have any outstanding
711        // WALs to replicate, so we are done.
712        return;
713      }
714
715      for (Map.Entry<String, Set<String>> entry : newQueues.entrySet()) {
716        String peerId = entry.getKey();
717        Set<String> walsSet = entry.getValue();
718        try {
719          // there is not an actual peer defined corresponding to peerId for the failover.
720          ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(peerId);
721          String actualPeerId = replicationQueueInfo.getPeerId();
722          ReplicationPeer peer = replicationPeers.getConnectedPeer(actualPeerId);
723          ReplicationPeerConfig peerConfig = null;
724          try {
725            peerConfig = replicationPeers.getReplicationPeerConfig(actualPeerId);
726          } catch (ReplicationException ex) {
727            LOG.warn("Received exception while getting replication peer config, skipping replay"
728                + ex);
729          }
730          if (peer == null || peerConfig == null) {
731            LOG.warn("Skipping failover for peer:" + actualPeerId + " of node" + rsZnode);
732            replicationQueues.removeQueue(peerId);
733            continue;
734          }
735          // track sources in walsByIdRecoveredQueues
736          Map<String, SortedSet<String>> walsByGroup = new HashMap<>();
737          walsByIdRecoveredQueues.put(peerId, walsByGroup);
738          for (String wal : walsSet) {
739            String walPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(wal);
740            SortedSet<String> wals = walsByGroup.get(walPrefix);
741            if (wals == null) {
742              wals = new TreeSet<>();
743              walsByGroup.put(walPrefix, wals);
744            }
745            wals.add(wal);
746          }
747
748          // enqueue sources
749          ReplicationSourceInterface src =
750              getReplicationSource(conf, fs, ReplicationSourceManager.this, this.rq, this.rp,
751                server, peerId, this.clusterId, peerConfig, peer);
752          // synchronized on oldsources to avoid adding recovered source for the to-be-removed peer
753          // see removePeer
754          synchronized (oldsources) {
755            if (!this.rp.getConnectedPeerIds().contains(src.getPeerId())) {
756              src.terminate("Recovered queue doesn't belong to any current peer");
757              closeRecoveredQueue(src);
758              continue;
759            }
760            oldsources.add(src);
761            for (String wal : walsSet) {
762              src.enqueueLog(new Path(oldLogDir, wal));
763            }
764            src.startup();
765          }
766        } catch (IOException e) {
767          // TODO manage it
768          LOG.error("Failed creating a source", e);
769        }
770      }
771    }
772  }

775
776    public AdoptAbandonedQueuesWorker() {}
777
778    @Override
779    public void run() {
780      List<String> currentReplicators = replicationQueues.getListOfReplicators();
781      if (currentReplicators == null || currentReplicators.isEmpty()) {
782        return;
783      }
784      List<String> otherRegionServers = replicationTracker.getListOfRegionServers();
785      LOG.info("Current list of replicators: " + currentReplicators + " other RSs: "
786        + otherRegionServers);
787
788      // Look if there's anything to process after a restart
789      for (String rs : currentReplicators) {
790        if (!otherRegionServers.contains(rs)) {
791          transferQueues(rs);
792        }
793      }

858  /**
859   * Whether an entry can be pushed to the peer or not right now.
860   * If we enable serial replication, we can not push the entry until all entries in its region
861   * whose sequence numbers are smaller than this entry have been pushed.
862   * For each ReplicationSource, we need only check the first entry in each region, as long as it
863   * can be pushed, we can push all in this ReplicationSource.
864   * This method will be blocked until we can push.
865   * @return the first barrier of entry's region, or -1 if there is no barrier. It is used to
866   *         prevent saving positions in the region of no barrier.
867   */
868  void waitUntilCanBePushed(byte[] encodedName, long seq, String peerId)
869      throws IOException, InterruptedException {
870
871    /**
872     * There are barriers for this region and position for this peer. N barriers form N intervals,
873     * (b1,b2) (b2,b3) ... (bn,max). Generally, there is no logs whose seq id is not greater than
874     * the first barrier and the last interval is start from the last barrier.
875     *
876     * There are several conditions that we can push now, otherwise we should block:
877     * 1) "Serial replication" is not enabled, we can push all logs just like before. This case
878     *    should not call this method.
879     * 2) There is no barriers for this region, or the seq id is smaller than the first barrier.
880     *    It is mainly because we alter REPLICATION_SCOPE = 2. We can not guarantee the
881     *    order of logs that is written before altering.
882     * 3) This entry is in the first interval of barriers. We can push them because it is the
883     *    start of a region. But if the region is created by region split, we should check
884     *    if the parent regions are fully pushed.
885     * 4) If the entry's seq id and the position are in same section, or the pos is the last
886     *    number of previous section. Because when open a region we put a barrier the number
887     *    is the last log's id + 1.
888     * 5) Log's seq is smaller than pos in meta, we are retrying. It may happen when a RS crashes
889     *    after save replication meta and before save zk offset.
890     */
891    List<Long> barriers = MetaTableAccessor.getReplicationBarriers(connection, encodedName);
892    if (barriers.isEmpty() || seq <= barriers.get(0)) {
893      // Case 2
894      return;
895    }
896    int interval = Collections.binarySearch(barriers, seq);
897    if (interval < 0) {
898      interval = -interval - 1;// get the insert position if negative
899    }
900    if (interval == 1) {
901      // Case 3
902      // Check if there are parent regions
903      String parentValue = MetaTableAccessor.getSerialReplicationParentRegion(connection,
904          encodedName);
905      if (parentValue == null) {
906        // This region has no parent or the parent's log entries are fully pushed.
907        return;
908      }
909      while (true) {
910        boolean allParentDone = true;
911        String[] parentRegions = parentValue.split(",");
912        for (String parent : parentRegions) {
913          byte[] region = Bytes.toBytes(parent);
914          long pos = MetaTableAccessor.getReplicationPositionForOnePeer(connection, region, peerId);
915          List<Long> parentBarriers = MetaTableAccessor.getReplicationBarriers(connection, region);
916          if (parentBarriers.size() > 0
917              && parentBarriers.get(parentBarriers.size() - 1) - 1 > pos) {
918            allParentDone = false;
919            // For a closed region, we will write a close event marker to WAL whose sequence id is
920            // larger than final barrier but still smaller than next region's openSeqNum.
921            // So if the pos is larger than last barrier, we can say we have read the event marker
922            // which means the parent region has been fully pushed.
923            LOG.info(Bytes.toString(encodedName) + " can not start pushing because parent region's"
924                + " log has not been fully pushed: parent=" + Bytes.toString(region) + " pos=" + pos
925                + " barriers=" + Arrays.toString(barriers.toArray()));
926            break;
927          }
928        }
929        if (allParentDone) {
930          return;
931        } else {
932          Thread.sleep(replicationWaitTime);
933        }
934      }
935
936    }
937
938    while (true) {
939      long pos = MetaTableAccessor.getReplicationPositionForOnePeer(connection, encodedName, peerId);
940      if (seq <= pos) {
941        // Case 5
942      }
943      if (pos >= 0) {
944        // Case 4
945        int posInterval = Collections.binarySearch(barriers, pos);
946        if (posInterval < 0) {
947          posInterval = -posInterval - 1;// get the insert position if negative
948        }
949        if (posInterval == interval || pos == barriers.get(interval - 1) - 1) {
950          return;
951        }
952      }
953
954      LOG.info(Bytes.toString(encodedName) + " can not start pushing to peer " + peerId
955          + " because previous log has not been pushed: sequence=" + seq + " pos=" + pos
956          + " barriers=" + Arrays.toString(barriers.toArray()));
957      Thread.sleep(replicationWaitTime);
958    }
959  }
