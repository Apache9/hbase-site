082/**
083 * Class that handles the source of a replication stream.
084 * Currently does not handle more than 1 slave
085 * For each slave cluster it selects a random number of peers
086 * using a replication ratio. For example, if replication ration = 0.1
087 * and slave cluster has 100 region servers, 10 will be selected.
088 * <p>
089 * A stream is considered down when we cannot contact a region server on the
090 * peer cluster for more than 55 seconds by default.
091 * </p>
092 *
093 */

096    implements ReplicationSourceInterface {

224  @Override
225  public void enqueueLog(Path log) {
226    String logPrefix = AbstractFSWALProvider.getWALPrefixFromWALName(log.getName());
227    PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
228    if (queue == null) {
229      queue = new PriorityBlockingQueue<Path>(queueSizePerGroup, new LogsComparator());
230      queues.put(logPrefix, queue);
231      if (this.sourceRunning) {
232        // new wal group observed after source startup, start a new worker thread to track it
233        // notice: it's possible that log enqueued when this.running is set but worker thread
234        // still not launched, so it's necessary to check workerThreads before start the worker
235        final ReplicationSourceWorkerThread worker =
236            new ReplicationSourceWorkerThread(logPrefix, queue, replicationQueueInfo, this);
237        ReplicationSourceWorkerThread extant = workerThreads.putIfAbsent(logPrefix, worker);
238        if (extant != null) {
239          LOG.debug("Someone has beat us to start a worker thread for wal group " + logPrefix);
240        } else {
241          LOG.debug("Starting up worker for wal group " + logPrefix);
242          worker.startup();
243        }
244      }
245    }
246    queue.put(log);
247    this.metrics.incrSizeOfLogQueue();
248    // This will log a warning for each new log that gets created above the warn threshold
249    int queueSize = queue.size();
250    if (queueSize > this.logQueueWarnThreshold) {
251      LOG.warn("WAL group " + logPrefix + " queue size: " + queueSize
252          + " exceeds value of replication.source.log.queue.warn: " + logQueueWarnThreshold);
253    }
254  }

256  @Override
257  public void addHFileRefs(TableName tableName, byte[] family, List<Pair<Path, Path>> pairs)
258      throws ReplicationException {
259    String peerId = peerClusterZnode;
260    if (peerId.contains("-")) {
261      // peerClusterZnode will be in the form peerId + "-" + rsZNode.
262      // A peerId will not have "-" in its name, see HBASE-11394
263      peerId = peerClusterZnode.split("-")[0];
264    }
265    Map<TableName, List<String>> tableCFMap = replicationPeers.getConnectedPeer(peerId).getTableCFs();
266    if (tableCFMap != null) {
267      List<String> tableCfs = tableCFMap.get(tableName);
268      if (tableCFMap.containsKey(tableName)
269          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
270        this.replicationQueues.addHFileRefs(peerId, pairs);
271        metrics.incrSizeOfHFileRefsQueue(pairs.size());
272      } else {
273        LOG.debug("HFiles will not be replicated belonging to the table " + tableName + " family "
274            + Bytes.toString(family) + " to peer id " + peerId);
275      }
276    } else {
277      // user has explicitly not defined any table cfs for replication, means replicate all the
278      // data
279      this.replicationQueues.addHFileRefs(peerId, pairs);
280      metrics.incrSizeOfHFileRefsQueue(pairs.size());
281    }
282  }

284  private void uninitialize() {
285    LOG.debug("Source exiting " + this.peerId);
286    metrics.clear();
287    if (replicationEndpoint.state() == Service.State.STARTING
288        || replicationEndpoint.state() == Service.State.RUNNING) {
289      replicationEndpoint.stopAndWait();
290    }
291  }

293  @Override
294  public void run() {
295    // mark we are running now
296    this.sourceRunning = true;
297    try {
298      // start the endpoint, connect to the cluster
299      Service.State state = replicationEndpoint.start().get();
300      if (state != Service.State.RUNNING) {
301        LOG.warn("ReplicationEndpoint was not started. Exiting");
302        uninitialize();
303        return;
304      }
305    } catch (Exception ex) {
306      LOG.warn("Error starting ReplicationEndpoint, exiting", ex);
307      throw new RuntimeException(ex);
308    }
309
310    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
311    ArrayList<WALEntryFilter> filters = Lists.newArrayList(
312      (WALEntryFilter)new SystemTableWALEntryFilter());
313    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
314    if (filterFromEndpoint != null) {
315      filters.add(filterFromEndpoint);
316    }
317    this.walEntryFilter = new ChainWALEntryFilter(filters);
318
319    int sleepMultiplier = 1;
320    // delay this until we are in an asynchronous thread
321    while (this.isSourceActive() && this.peerClusterId == null) {
322      this.peerClusterId = replicationEndpoint.getPeerUUID();
323      if (this.isSourceActive() && this.peerClusterId == null) {
324        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
325          sleepMultiplier++;
326        }
327      }
328    }
329
330    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
331    // peerClusterId value, which is the same as the source clusterId
332    if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
333      this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
334          + peerClusterId + " which is not allowed by ReplicationEndpoint:"
335          + replicationEndpoint.getClass().getName(), null, false);
336      this.manager.closeQueue(this);
337      return;
338    }
339    LOG.info("Replicating " + clusterId + " -> " + peerClusterId);
340    // start workers
341    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
342      String walGroupId = entry.getKey();
343      PriorityBlockingQueue<Path> queue = entry.getValue();
344      final ReplicationSourceWorkerThread worker =
345          new ReplicationSourceWorkerThread(walGroupId, queue, replicationQueueInfo, this);
346      ReplicationSourceWorkerThread extant = workerThreads.putIfAbsent(walGroupId, worker);
347      if (extant != null) {
348        LOG.debug("Someone has beat us to start a worker thread for wal group " + walGroupId);
349      } else {
350        LOG.debug("Starting up worker for wal group " + walGroupId);
351        worker.startup();
352      }
353    }
354  }

356  /**
357   * Do the sleeping logic
358   * @param msg Why we sleep
359   * @param sleepMultiplier by how many times the default sleeping time is augmented
360   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
361   */
362  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
363    try {
364      if (LOG.isTraceEnabled()) {
365        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
366      }
367      Thread.sleep(this.sleepForRetries * sleepMultiplier);
368    } catch (InterruptedException e) {
369      LOG.debug("Interrupted while sleeping between retries");
370      Thread.currentThread().interrupt();
371    }
372    return sleepMultiplier < maxRetriesMultiplier;
373  }

407  public void terminate(String reason, Exception cause, boolean join) {
408    if (cause == null) {
409      LOG.info("Closing source "
410          + this.peerClusterZnode + " because: " + reason);
411
412    } else {
413      LOG.error("Closing source " + this.peerClusterZnode
414          + " because an error occurred: " + reason, cause);
415    }
416    this.sourceRunning = false;
417    Collection<ReplicationSourceWorkerThread> workers = workerThreads.values();
418    for (ReplicationSourceWorkerThread worker : workers) {
419      worker.setWorkerRunning(false);
420      worker.interrupt();
421    }
422    ListenableFuture<Service.State> future = null;
423    if (this.replicationEndpoint != null) {
424      future = this.replicationEndpoint.stop();
425    }
426    if (join) {
427      for (ReplicationSourceWorkerThread worker : workers) {
428        Threads.shutdown(worker, this.sleepForRetries);
429        LOG.info("ReplicationSourceWorker " + worker.getName() + " terminated");
430      }
431      if (future != null) {
432        try {
433          future.get(sleepForRetries * maxRetriesMultiplier, TimeUnit.MILLISECONDS);
434        } catch (Exception e) {
435          LOG.warn("Got exception while waiting for endpoint to shutdown for replication source :"
436              + this.peerClusterZnode,
437            e);
438        }
439      }
440    }
441  }

466  /**
467   * Comparator used to compare logs together based on their start time
468   */
469  public static class LogsComparator implements Comparator<Path> {
470
471    @Override
472    public int compare(Path o1, Path o2) {
473      return Long.valueOf(getTS(o1)).compareTo(getTS(o2));
474    }
475
476    /**
477     * Split a path to get the start time
478     * For example: 10.20.20.171%3A60020.1277499063250
479     * @param p path to split
480     * @return start time
481     */
482    private static long getTS(Path p) {
483      int tsIndex = p.getName().lastIndexOf('.') + 1;
484      return Long.parseLong(p.getName().substring(tsIndex));
485    }
486  }

488  @Override
489  public String getStats() {
490    StringBuilder sb = new StringBuilder();
491    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
492        .append(", current progress: \n");
493    for (Map.Entry<String, ReplicationSourceWorkerThread> entry : workerThreads.entrySet()) {
494      String walGroupId = entry.getKey();
495      ReplicationSourceWorkerThread worker = entry.getValue();
496      long position = worker.getCurrentPosition();
497      Path currentPath = worker.getCurrentPath();
498      sb.append("walGroup [").append(walGroupId).append("]: ");
499      if (currentPath != null) {
500        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
501            .append(position).append("\n");
502      } else {
503        sb.append("no replication ongoing, waiting for new log");
504      }
505    }
506    return sb.toString();
507  }

524  public class ReplicationSourceWorkerThread extends Thread {
525    ReplicationSource source;
526    String walGroupId;
527    PriorityBlockingQueue<Path> queue;
528    ReplicationQueueInfo replicationQueueInfo;
529    // Our reader for the current log. open/close handled by repLogReader
530    private WAL.Reader reader;
531    // Last position in the log that we sent to ZooKeeper
532    private long lastLoggedPosition = -1;
533    // Path of the current log
534    private volatile Path currentPath;
535    // Handle on the log reader helper
536    private ReplicationWALReaderManager repLogReader;
537    // Current number of operations (Put/Delete) that we need to replicate
538    private int currentNbOperations = 0;
539    // Current size of data we need to replicate
540    private int currentSize = 0;
541    // Indicates whether this particular worker is running
542    private boolean workerRunning = true;
543    // Current number of hfiles that we need to replicate
544    private long currentNbHFiles = 0;
545    List<WAL.Entry> entries;
546    // Use guava cache to set ttl for each key
547    private LoadingCache<String, Boolean> canSkipWaitingSet = CacheBuilder.newBuilder()
548        .expireAfterAccess(1, TimeUnit.DAYS).build(
549        new CacheLoader<String, Boolean>() {
550          @Override
551          public Boolean load(String key) throws Exception {
552            return false;
553          }
554        }
555    );
556
557    public ReplicationSourceWorkerThread(String walGroupId,
558        PriorityBlockingQueue<Path> queue, ReplicationQueueInfo replicationQueueInfo,
559        ReplicationSource source) {
560      this.walGroupId = walGroupId;
561      this.queue = queue;
562      this.replicationQueueInfo = replicationQueueInfo;
563      this.repLogReader = new ReplicationWALReaderManager(fs, conf);
564      this.source = source;
565      this.entries = new ArrayList<>();
566    }
567
568    @Override
569    public void run() {
570      // If this is recovered, the queue is already full and the first log
571      // normally has a position (unless the RS failed between 2 logs)
572      if (this.replicationQueueInfo.isQueueRecovered()) {
573        try {
574          this.repLogReader.setPosition(replicationQueues.getLogPosition(peerClusterZnode,
575            this.queue.peek().getName()));
576          if (LOG.isTraceEnabled()) {
577            LOG.trace("Recovered queue started with log " + this.queue.peek() + " at position "
578                + this.repLogReader.getPosition());
579          }
580        } catch (ReplicationException e) {
581          terminate("Couldn't get the position of this recovered queue " + peerClusterZnode, e);
582        }
583      }
584      // Loop until we close down
585      while (isWorkerActive()) {
586        int sleepMultiplier = 1;
587        // Sleep until replication is enabled again
588        if (!isPeerEnabled()) {
589          if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
590            sleepMultiplier++;
591          }
592          continue;
593        }
594        Path oldPath = getCurrentPath(); //note that in the current scenario,
595                                         //oldPath will be null when a log roll
596                                         //happens.
597        // Get a new path
598        boolean hasCurrentPath = getNextPath();
599        if (getCurrentPath() != null && oldPath == null) {
600          sleepMultiplier = 1; //reset the sleepMultiplier on a path change
601        }
602        if (!hasCurrentPath) {
603          if (sleepForRetries("No log to process", sleepMultiplier)) {
604            sleepMultiplier++;
605          }
606          continue;
607        }
608        boolean currentWALisBeingWrittenTo = false;
609        //For WAL files we own (rather than recovered), take a snapshot of whether the
610        //current WAL file (this.currentPath) is in use (for writing) NOW!
611        //Since the new WAL paths are enqueued only after the prev WAL file
612        //is 'closed', presence of an element in the queue means that
613        //the previous WAL file was closed, else the file is in use (currentPath)
614        //We take the snapshot now so that we are protected against races
615        //where a new file gets enqueued while the current file is being processed
616        //(and where we just finished reading the current file).
617        if (!this.replicationQueueInfo.isQueueRecovered() && queue.isEmpty()) {
618          currentWALisBeingWrittenTo = true;
619        }
620        // Open a reader on it
621        if (!openReader(sleepMultiplier)) {
622          // Reset the sleep multiplier, else it'd be reused for the next file
623          sleepMultiplier = 1;
624          continue;
625        }
626
627        // If we got a null reader but didn't continue, then sleep and continue
628        if (this.reader == null) {
629          if (sleepForRetries("Unable to open a reader", sleepMultiplier)) {
630            sleepMultiplier++;
631          }
632          continue;
633        }
634
635        boolean gotIOE = false;
636        currentNbOperations = 0;
637        currentNbHFiles = 0;
638        entries.clear();
639        Map<String, Long> lastPositionsForSerialScope = new HashMap<>();
640        currentSize = 0;
641        try {
642          if (readAllEntriesToReplicateOrNextFile(currentWALisBeingWrittenTo, entries,
643              lastPositionsForSerialScope)) {
644            for (Map.Entry<String, Long> entry : lastPositionsForSerialScope.entrySet()) {
645              waitingUntilCanPush(entry);
646            }
647            try {
648              MetaTableAccessor
649                  .updateReplicationPositions(manager.getConnection(), actualPeerId,
650                      lastPositionsForSerialScope);
651            } catch (IOException e) {
652              LOG.error("updateReplicationPositions fail", e);
653              stopper.stop("updateReplicationPositions fail");
654            }
655
656            continue;
657          }
658        } catch (IOException ioe) {
659          LOG.warn(peerClusterZnode + " Got: ", ioe);
660          gotIOE = true;
661          if (ioe.getCause() instanceof EOFException) {
662
663            boolean considerDumping = false;
664            if (this.replicationQueueInfo.isQueueRecovered()) {
665              try {
666                FileStatus stat = fs.getFileStatus(this.currentPath);
667                if (stat.getLen() == 0) {
668                  LOG.warn(peerClusterZnode + " Got EOF and the file was empty");
669                }
670                considerDumping = true;
671              } catch (IOException e) {
672                LOG.warn(peerClusterZnode + " Got while getting file size: ", e);
673              }
674            }
675
676            if (considerDumping &&
677                sleepMultiplier == maxRetriesMultiplier &&
678                processEndOfFile()) {
679              continue;
680            }
681          }
682        } finally {
683          try {
684            this.reader = null;
685            this.repLogReader.closeReader();
686          } catch (IOException e) {
687            gotIOE = true;
688            LOG.warn("Unable to finalize the tailing of a file", e);
689          }
690        }
691        for(Map.Entry<String, Long> entry: lastPositionsForSerialScope.entrySet()) {
692          waitingUntilCanPush(entry);
693        }
694        // If we didn't get anything to replicate, or if we hit a IOE,
695        // wait a bit and retry.
696        // But if we need to stop, don't bother sleeping
697        if (isWorkerActive() && (gotIOE || entries.isEmpty())) {
698          if (this.lastLoggedPosition != this.repLogReader.getPosition()) {
699
700            // Save positions to meta table before zk.
701            if (!gotIOE) {
702              try {
703                MetaTableAccessor.updateReplicationPositions(manager.getConnection(), actualPeerId,
704                    lastPositionsForSerialScope);
705              } catch (IOException e) {
706                LOG.error("updateReplicationPositions fail", e);
707                stopper.stop("updateReplicationPositions fail");
708              }
709            }
710
711            manager.logPositionAndCleanOldLogs(this.currentPath, peerClusterZnode,
712                this.repLogReader.getPosition(),
713                this.replicationQueueInfo.isQueueRecovered(), currentWALisBeingWrittenTo);
714
715            this.lastLoggedPosition = this.repLogReader.getPosition();
716          }
717          // Reset the sleep multiplier if nothing has actually gone wrong
718          if (!gotIOE) {
719            sleepMultiplier = 1;
720            // if there was nothing to ship and it's not an error
721            // set "ageOfLastShippedOp" to <now> to indicate that we're current
722            metrics.setAgeOfLastShippedOp(EnvironmentEdgeManager.currentTime(), walGroupId);
723          }
724          if (sleepForRetries("Nothing to replicate", sleepMultiplier)) {
725            sleepMultiplier++;
726          }
727          continue;
728        }
729        shipEdits(currentWALisBeingWrittenTo, entries, lastPositionsForSerialScope);
730        releaseBufferQuota();
731      }
732      if (replicationQueueInfo.isQueueRecovered()) {
733        // use synchronize to make sure one last thread will clean the queue
734        synchronized (workerThreads) {
735          Threads.sleep(100);// wait a short while for other worker thread to fully exit
736          boolean allOtherTaskDone = true;
737          for (ReplicationSourceWorkerThread worker : workerThreads.values()) {
738            if (!worker.equals(this) && worker.isAlive()) {
739              allOtherTaskDone = false;
740              break;
741            }
742          }
743          if (allOtherTaskDone) {
744            manager.closeRecoveredQueue(this.source);
745            LOG.info("Finished recovering queue " + peerClusterZnode
746                + " with the following stats: " + getStats());
747          }
748        }
749      }
750    }

752    private void waitingUntilCanPush(Map.Entry<String, Long> entry) {
753      String key = entry.getKey();
754      long seq = entry.getValue();
755      boolean deleteKey = false;
756      if (seq <= 0) {
757        // There is a REGION_CLOSE marker, we can not continue skipping after this entry.
758        deleteKey = true;
759        seq = -seq;
760      }
761
762      if (!canSkipWaitingSet.getUnchecked(key)) {
763        try {
764          manager.waitUntilCanBePushed(Bytes.toBytes(key), seq, actualPeerId);
765        } catch (Exception e) {
766          LOG.error("waitUntilCanBePushed fail", e);
767          stopper.stop("waitUntilCanBePushed fail");
768        }
769        canSkipWaitingSet.put(key, true);
770      }
771      if (deleteKey) {
772        canSkipWaitingSet.invalidate(key);
773      }
774    }

776    /**
777     * Read all the entries from the current log files and retain those that need to be replicated.
778     * Else, process the end of the current file.
779     * @param currentWALisBeingWrittenTo is the current WAL being written to
780     * @param entries resulting entries to be replicated
781     * @param lastPosition save the last sequenceid for each region if the table has
782     *                     serial-replication scope
783     * @return true if we got nothing and went to the next file, false if we got entries
784     * @throws IOException
785     */
786    protected boolean readAllEntriesToReplicateOrNextFile(boolean currentWALisBeingWrittenTo,
787        List<WAL.Entry> entries, Map<String, Long> lastPosition) throws IOException {
788      long seenEntries = 0;
789      if (LOG.isTraceEnabled()) {
790        LOG.trace("Seeking in " + this.currentPath + " at position "
791            + this.repLogReader.getPosition());
792      }
793      this.repLogReader.seek();
794      long positionBeforeRead = this.repLogReader.getPosition();
795      WAL.Entry entry = this.repLogReader.readNextAndSetPosition();
796      while (entry != null) {
797        metrics.incrLogEditsRead();
798        seenEntries++;
799
800        if (entry.hasSerialReplicationScope()) {
801          String key = Bytes.toString(entry.getKey().getEncodedRegionName());
802          lastPosition.put(key, entry.getKey().getSequenceId());
803          if (entry.getEdit().getCells().size() > 0) {
804            WALProtos.RegionEventDescriptor maybeEvent =
805                WALEdit.getRegionEventDescriptor(entry.getEdit().getCells().get(0));
806            if (maybeEvent != null && maybeEvent.getEventType()
807                == WALProtos.RegionEventDescriptor.EventType.REGION_CLOSE) {
808              // In serially replication, if we move a region to another RS and move it back, we may
809              // read logs crossing two sections. We should break at REGION_CLOSE and push the first
810              // section first in case of missing the middle section belonging to the other RS.
811              // In a worker thread, if we can push the first log of a region, we can push all logs
812              // in the same region without waiting until we read a close marker because next time
813              // we read logs in this region, it must be a new section and not adjacent with this
814              // region. Mark it negative.
815              lastPosition.put(key, -entry.getKey().getSequenceId());
816              break;
817            }
818          }
819        }
820        boolean totalBufferTooLarge = false;
821        // don't replicate if the log entries have already been consumed by the cluster
822        if (replicationEndpoint.canReplicateToSameCluster()
823            || !entry.getKey().getClusterIds().contains(peerClusterId)) {
824          // Remove all KVs that should not be replicated
825          entry = walEntryFilter.filter(entry);
826          WALEdit edit = null;
827          WALKey logKey = null;
828          if (entry != null) {
829            edit = entry.getEdit();
830            logKey = entry.getKey();
831          }
832
833          if (edit != null && edit.size() != 0) {
834            // Mark that the current cluster has the change
835            logKey.addClusterId(clusterId);
836            currentNbOperations += countDistinctRowKeys(edit);
837            entries.add(entry);
838            int delta = (int)entry.getEdit().heapSize() + calculateTotalSizeOfStoreFiles(edit);
839            currentSize += delta;
840            totalBufferTooLarge = acquireBufferQuota(delta);
841          } else {
842            metrics.incrLogEditsFiltered();
843          }
844        }
845        // Stop if too many entries or too big
846        // FIXME check the relationship between single wal group and overall
847        if (totalBufferTooLarge || currentSize >= replicationQueueSizeCapacity
848            || entries.size() >= replicationQueueNbCapacity) {
849          break;
850        }
851
852        try {
853          entry = this.repLogReader.readNextAndSetPosition();
854        } catch (IOException ie) {
855          LOG.debug("Break on IOE: " + ie.getMessage());
856          break;
857        }
858      }
859      metrics.incrLogReadInBytes(this.repLogReader.getPosition() - positionBeforeRead);
860      if (currentWALisBeingWrittenTo) {
861        return false;
862      }
863      // If we didn't get anything and the queue has an object, it means we
864      // hit the end of the file for sure
865      return seenEntries == 0 && processEndOfFile();
866    }

868    /**
869     * Calculate the total size of all the store files
870     * @param edit edit to count row keys from
871     * @return the total size of the store files
872     */
873    private int calculateTotalSizeOfStoreFiles(WALEdit edit) {
874      List<Cell> cells = edit.getCells();
875      int totalStoreFilesSize = 0;
876
877      int totalCells = edit.size();
878      for (int i = 0; i < totalCells; i++) {
879        if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
880          try {
881            BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
882            List<StoreDescriptor> stores = bld.getStoresList();
883            int totalStores = stores.size();
884            for (int j = 0; j < totalStores; j++) {
885              totalStoreFilesSize += stores.get(j).getStoreFileSizeBytes();
886            }
887          } catch (IOException e) {
888            LOG.error("Failed to deserialize bulk load entry from wal edit. "
889                + "Size of HFiles part of cell will not be considered in replication "
890                + "request size calculation.", e);
891          }
892        }
893      }
894      return totalStoreFilesSize;
895    }

897    private void cleanUpHFileRefs(WALEdit edit) throws IOException {
898      String peerId = peerClusterZnode;
899      if (peerId.contains("-")) {
900        // peerClusterZnode will be in the form peerId + "-" + rsZNode.
901        // A peerId will not have "-" in its name, see HBASE-11394
902        peerId = peerClusterZnode.split("-")[0];
903      }
904      List<Cell> cells = edit.getCells();
905      int totalCells = cells.size();
906      for (int i = 0; i < totalCells; i++) {
907        Cell cell = cells.get(i);
908        if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
909          BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
910          List<StoreDescriptor> stores = bld.getStoresList();
911          int totalStores = stores.size();
912          for (int j = 0; j < totalStores; j++) {
913            List<String> storeFileList = stores.get(j).getStoreFileList();
914            manager.cleanUpHFileRefs(peerId, storeFileList);
915            metrics.decrSizeOfHFileRefsQueue(storeFileList.size());
916          }
917        }
918      }
919    }

923     * @return true if a path was obtained, false if not
924     */
925    protected boolean getNextPath() {
926      try {
927        if (this.currentPath == null) {
928          this.currentPath = queue.poll(sleepForRetries, TimeUnit.MILLISECONDS);
929          metrics.decrSizeOfLogQueue();
930          if (this.currentPath != null) {
931            // For recovered queue: must use peerClusterZnode since peerId is a parsed value
932            manager.cleanOldLogs(this.currentPath.getName(), peerClusterZnode,
933              this.replicationQueueInfo.isQueueRecovered());
934            if (LOG.isTraceEnabled()) {
935              LOG.trace("New log: " + this.currentPath);
936            }
937          }
938        }
939      } catch (InterruptedException e) {
940        LOG.warn("Interrupted while reading edits", e);
941      }
942      return this.currentPath != null;
943    }

945    /**
946     * Open a reader on the current path
947     *
948     * @param sleepMultiplier by how many times the default sleeping time is augmented
949     * @return true if we should continue with that file, false if we are over with it
950     */
951    protected boolean openReader(int sleepMultiplier) {
952      try {
953        try {
954          if (LOG.isTraceEnabled()) {
955            LOG.trace("Opening log " + this.currentPath);
956          }
957          this.reader = repLogReader.openReader(this.currentPath);
958        } catch (FileNotFoundException fnfe) {
959          if (this.replicationQueueInfo.isQueueRecovered()) {
960            // We didn't find the log in the archive directory, look if it still
961            // exists in the dead RS folder (there could be a chain of failures
962            // to look at)
963            List<String> deadRegionServers = this.replicationQueueInfo.getDeadRegionServers();
964            LOG.info("NB dead servers : " + deadRegionServers.size());
965            final Path walDir = FSUtils.getWALRootDir(conf);
966            for (String curDeadServerName : deadRegionServers) {
967              final Path deadRsDirectory = new Path(walDir,
968                AbstractFSWALProvider.getWALDirectoryName(curDeadServerName));
969              Path[] locs = new Path[] { new Path(deadRsDirectory, currentPath.getName()),
970                new Path(deadRsDirectory.suffix(AbstractFSWALProvider.SPLITTING_EXT),
971                  currentPath.getName()) };
972              for (Path possibleLogLocation : locs) {
973                LOG.info("Possible location " + possibleLogLocation.toUri().toString());
974                if (manager.getFs().exists(possibleLogLocation)) {
975                  // We found the right new location
976                  LOG.info("Log " + this.currentPath + " still exists at " +
977                      possibleLogLocation);
978                  // Breaking here will make us sleep since reader is null
979                  // TODO why don't we need to set currentPath and call openReader here?
980                  return true;
981                }
982              }
983            }
984            // In the case of disaster/recovery, HMaster may be shutdown/crashed before flush data
985            // from .logs to .oldlogs. Loop into .logs folders and check whether a match exists
986            if (stopper instanceof ReplicationSyncUp.DummyServer) {
987              // N.B. the ReplicationSyncUp tool sets the manager.getWALDir to the root of the wal
988              //      area rather than to the wal area for a particular region server.
989              FileStatus[] rss = fs.listStatus(manager.getLogDir());
990              for (FileStatus rs : rss) {
991                Path p = rs.getPath();
992                FileStatus[] logs = fs.listStatus(p);
993                for (FileStatus log : logs) {
994                  p = new Path(p, log.getPath().getName());
995                  if (p.getName().equals(currentPath.getName())) {
996                    currentPath = p;
997                    LOG.info("Log " + currentPath.getName() + " found at " + currentPath);
998                    // Open the log at the new location
999                    this.openReader(sleepMultiplier);
1000                    return true;
1001                  }
1002                }
1003              }
1004            }
1005
1006            // TODO What happens if the log was missing from every single location?
1007            // Although we need to check a couple of times as the log could have
1008            // been moved by the master between the checks
1009            // It can also happen if a recovered queue wasn't properly cleaned,
1010            // such that the znode pointing to a log exists but the log was
1011            // deleted a long time ago.
1012            // For the moment, we'll throw the IO and processEndOfFile
1013            throw new IOException("File from recovered queue is " +
1014                "nowhere to be found", fnfe);
1015          } else {
1016            // If the log was archived, continue reading from there
1017            Path archivedLogLocation =
1018                new Path(manager.getOldLogDir(), currentPath.getName());
1019            if (manager.getFs().exists(archivedLogLocation)) {
1020              currentPath = archivedLogLocation;
1021              LOG.info("Log " + this.currentPath + " was moved to " +
1022                  archivedLogLocation);
1023              // Open the log at the new location
1024              this.openReader(sleepMultiplier);
1025
1026            }
1027            // TODO What happens the log is missing in both places?
1028          }
1029        }
1030      } catch (LeaseNotRecoveredException lnre) {
1031        // HBASE-15019 the WAL was not closed due to some hiccup.
1032        LOG.warn(peerClusterZnode + " Try to recover the WAL lease " + currentPath, lnre);
1033        recoverLease(conf, currentPath);
1034        this.reader = null;
1035      } catch (IOException ioe) {
1036        if (ioe instanceof EOFException && isCurrentLogEmpty()) return true;
1037        LOG.warn(peerClusterZnode + " Got: ", ioe);
1038        this.reader = null;
1039        if (ioe.getCause() instanceof NullPointerException) {
1040          // Workaround for race condition in HDFS-4380
1041          // which throws a NPE if we open a file before any data node has the most recent block
1042          // Just sleep and retry. Will require re-reading compressed WALs for compressionContext.
1043          LOG.warn("Got NPE opening reader, will retry.");
1044        } else if (sleepMultiplier >= maxRetriesMultiplier) {
1045          // TODO Need a better way to determine if a file is really gone but
1046          // TODO without scanning all logs dir
1047          LOG.warn("Waited too long for this file, considering dumping");
1048          return !processEndOfFile();
1049        }
1050      }
1051      return true;
1052    }

1056        final FileSystem dfs = FSUtils.getCurrentFileSystem(conf);
1057        FSUtils fsUtils = FSUtils.getInstance(dfs, conf);

1081    /**
1082     * Count the number of different row keys in the given edit because of mini-batching. We assume
1083     * that there's at least one Cell in the WALEdit.
1084     * @param edit edit to count row keys from
1085     * @return number of different row keys
1086     */
1087    private int countDistinctRowKeys(WALEdit edit) {
1088      List<Cell> cells = edit.getCells();
1089      int distinctRowKeys = 1;
1090      int totalHFileEntries = 0;
1091      Cell lastCell = cells.get(0);
1092
1093      int totalCells = edit.size();
1094      for (int i = 0; i < totalCells; i++) {
1095        // Count HFiles to be replicated
1096        if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
1097          try {
1098            BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
1099            List<StoreDescriptor> stores = bld.getStoresList();
1100            int totalStores = stores.size();
1101            for (int j = 0; j < totalStores; j++) {
1102              totalHFileEntries += stores.get(j).getStoreFileList().size();
1103            }
1104          } catch (IOException e) {
1105            LOG.error("Failed to deserialize bulk load entry from wal edit. "
1106                + "Then its hfiles count will not be added into metric.");
1107          }
1108        }
1109
1110        if (!CellUtil.matchingRows(cells.get(i), lastCell)) {
1111          distinctRowKeys++;
1112        }
1113        lastCell = cells.get(i);
1114      }
1115      currentNbHFiles += totalHFileEntries;
1116      return distinctRowKeys + totalHFileEntries;
1117    }

1129    /**
1130     * Do the shipping logic
1131     * @param currentWALisBeingWrittenTo was the current WAL being (seemingly)
1132     * written to when this method was called
1133     */
1134    protected void shipEdits(boolean currentWALisBeingWrittenTo, List<WAL.Entry> entries,
1135        Map<String, Long> lastPositionsForSerialScope) {
1136      int sleepMultiplier = 0;
1137      if (entries.isEmpty()) {
1138        LOG.warn("Was given 0 edits to ship");
1139        return;
1140      }
1141      while (isWorkerActive()) {
1142        try {
1143          checkBandwidthChangeAndResetThrottler();
1144          if (throttler.isEnabled()) {
1145            long sleepTicks = throttler.getNextSleepInterval(currentSize);
1146            if (sleepTicks > 0) {
1147              try {
1148                if (LOG.isTraceEnabled()) {
1149                  LOG.trace("To sleep " + sleepTicks + "ms for throttling control");
1150                }
1151                Thread.sleep(sleepTicks);
1152              } catch (InterruptedException e) {
1153                LOG.debug("Interrupted while sleeping for throttling control");
1154                Thread.currentThread().interrupt();
1155                // current thread might be interrupted to terminate
1156                // directly go back to while() for confirm this
1157                continue;
1158              }
1159              // reset throttler's cycle start tick when sleep for throttling occurs
1160              throttler.resetStartTick();
1161            }
1162          }
1163          // create replicateContext here, so the entries can be GC'd upon return from this call
1164          // stack
1165          ReplicationEndpoint.ReplicateContext replicateContext =
1166              new ReplicationEndpoint.ReplicateContext();
1167          replicateContext.setEntries(entries).setSize(currentSize);
1168          replicateContext.setWalGroupId(walGroupId);
1169
1170          long startTimeNs = System.nanoTime();
1171          // send the edits to the endpoint. Will block until the edits are shipped and acknowledged
1172          boolean replicated = replicationEndpoint.replicate(replicateContext);
1173          long endTimeNs = System.nanoTime();
1174
1175          if (!replicated) {
1176            continue;
1177          } else {
1178            sleepMultiplier = Math.max(sleepMultiplier - 1, 0);
1179          }
1180
1181          if (this.lastLoggedPosition != this.repLogReader.getPosition()) {
1182            //Clean up hfile references
1183            int size = entries.size();
1184            for (int i = 0; i < size; i++) {
1185              cleanUpHFileRefs(entries.get(i).getEdit());
1186            }
1187
1188            // Save positions to meta table before zk.
1189            try {
1190              MetaTableAccessor.updateReplicationPositions(manager.getConnection(), actualPeerId,
1191                  lastPositionsForSerialScope);
1192            } catch (IOException e) {
1193              LOG.error("updateReplicationPositions fail", e);
1194              stopper.stop("updateReplicationPositions fail");
1195            }
1196
1197            //Log and clean up WAL logs
1198            manager.logPositionAndCleanOldLogs(this.currentPath, peerClusterZnode,
1199              this.repLogReader.getPosition(), this.replicationQueueInfo.isQueueRecovered(),
1200              currentWALisBeingWrittenTo);
1201            this.lastLoggedPosition = this.repLogReader.getPosition();
1202          }
1203          if (throttler.isEnabled()) {
1204            throttler.addPushSize(currentSize);
1205          }
1206          totalReplicatedEdits.addAndGet(entries.size());
1207          totalReplicatedOperations.addAndGet(currentNbOperations);
1208          // FIXME check relationship between wal group and overall
1209          metrics.shipBatch(currentNbOperations, currentSize, currentNbHFiles);
1210          metrics.setAgeOfLastShippedOp(entries.get(entries.size() - 1).getKey().getWriteTime(),
1211            walGroupId);
1212          if (LOG.isTraceEnabled()) {
1213            LOG.trace("Replicated " + totalReplicatedEdits + " entries in total, or "
1214                + totalReplicatedOperations + " operations in "
1215                + ((endTimeNs - startTimeNs) / 1000000) + " ms");
1216          }
1217          break;
1218        } catch (Exception ex) {
1219          LOG.warn(replicationEndpoint.getClass().getName() + " threw unknown exception:"
1220              + org.apache.hadoop.util.StringUtils.stringifyException(ex));
1221          if (sleepForRetries("ReplicationEndpoint threw exception", sleepMultiplier)) {
1222            sleepMultiplier++;
1223          }
1224        }
1225      }
1226    }

1228    /**
1229     * If the queue isn't empty, switch to the next one Else if this is a recovered queue, it means
1230     * we're done! Else we'll just continue to try reading the log file
1231     * @return true if we're done with the current file, false if we should continue trying to read
1232     *         from it
1233     */
1234    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "DE_MIGHT_IGNORE",
1235        justification = "Yeah, this is how it works")
1236    protected boolean processEndOfFile() {
1237      // We presume this means the file we're reading is closed.
1238      if (this.queue.size() != 0) {
1239        // -1 means the wal wasn't closed cleanly.
1240        final long trailerSize = this.repLogReader.currentTrailerSize();
1241        final long currentPosition = this.repLogReader.getPosition();
1242        FileStatus stat = null;
1243        try {
1244          stat = fs.getFileStatus(this.currentPath);
1245        } catch (IOException exception) {
1246          LOG.warn("Couldn't get file length information about log " + this.currentPath + ", it " + (trailerSize < 0 ? "was not" : "was") + " closed cleanly"
1247              + ", stats: " + getStats());
1248          metrics.incrUnknownFileLengthForClosedWAL();
1249        }
1250        if (stat != null) {
1251          if (trailerSize < 0) {
1252            if (currentPosition < stat.getLen()) {
1253              final long skippedBytes = stat.getLen() - currentPosition;
1254              LOG.info("Reached the end of WAL file '" + currentPath + "'. It was not closed cleanly, so we did not parse " + skippedBytes + " bytes of data.");
1255              metrics.incrUncleanlyClosedWALs();
1256              metrics.incrBytesSkippedInUncleanlyClosedWALs(skippedBytes);
1257            }
1258          } else if (currentPosition + trailerSize < stat.getLen()){
1259            LOG.warn("Processing end of WAL file '" + currentPath + "'. At position " + currentPosition + ", which is too far away from reported file length " + stat.getLen() +
1260                ". Restarting WAL reading (see HBASE-15983 for details). stats: " + getStats());
1261            repLogReader.setPosition(0);
1262            metrics.incrRestartedWALReading();
1263            metrics.incrRepeatedFileBytes(currentPosition);
1264            return false;
1265          }
1266        }
1267        if (LOG.isTraceEnabled()) {
1268          LOG.trace("Reached the end of log " + this.currentPath + ", stats: " + getStats()
1269              + ", and the length of the file is " + (stat == null ? "N/A" : stat.getLen()));
1270        }
1271        this.currentPath = null;
1272        this.repLogReader.finishCurrentFile();
1273        this.reader = null;
1274        metrics.incrCompletedWAL();
1275        return true;
1276      } else if (this.replicationQueueInfo.isQueueRecovered()) {
1277        LOG.debug("Finished recovering queue for group " + walGroupId + " of peer "
1278            + peerClusterZnode);
1279        metrics.incrCompletedRecoveryQueue();
1280        workerRunning = false;
1281        return true;
1282      }
1283      return false;
1284    }

1287      String n = Thread.currentThread().getName();
1288      Thread.UncaughtExceptionHandler handler = new Thread.UncaughtExceptionHandler() {
1289        @Override

1291          LOG.error("Unexpected exception in ReplicationSourceWorkerThread," + " currentPath="

1293        }
1294      };

1296          + peerClusterZnode, handler);
1297      workerThreads.put(walGroupId, this);
1298    }

1312    private void terminate(String reason, Exception cause) {
1313      if (cause == null) {
1314        LOG.info("Closing worker for wal group " + this.walGroupId + " because: " + reason);
1315
1316      } else {
1317        LOG.error("Closing worker for wal group " + this.walGroupId
1318            + " because an error occurred: " + reason, cause);
1319      }
1320      this.interrupt();
1321      Threads.shutdown(this, sleepForRetries);
1322      LOG.info("ReplicationSourceWorker " + this.getName() + " terminated");
1323    }
