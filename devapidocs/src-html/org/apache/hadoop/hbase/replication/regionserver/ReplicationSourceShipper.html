040/**
041 * This thread reads entries from a queue and ships them. Entries are placed onto the queue by
042 * ReplicationSourceWALReaderThread
043 */
044@InterfaceAudience.Private
045public class ReplicationSourceShipper extends Thread {
046  private static final Logger LOG = LoggerFactory.getLogger(ReplicationSourceShipper.class);
047
048  // Hold the state of a replication worker thread
049  public enum WorkerState {
050    RUNNING,
051    STOPPED,
052    FINISHED,  // The worker is done processing a recovered queue
053  }
054
055  protected final Configuration conf;
056  protected final String walGroupId;
057  protected final PriorityBlockingQueue<Path> queue;
058  protected final ReplicationSourceInterface source;
059
060  // Last position in the log that we sent to ZooKeeper
061  protected long lastLoggedPosition = -1;
062  // Path of the current log
063  protected volatile Path currentPath;
064  // Current state of the worker thread
065  private WorkerState state;
066  protected ReplicationSourceWALReader entryReader;
067
068  // How long should we sleep for each retry
069  protected final long sleepForRetries;
070  // Maximum number of retries before taking bold actions
071  protected final int maxRetriesMultiplier;
072
073  public ReplicationSourceShipper(Configuration conf, String walGroupId,
074      PriorityBlockingQueue<Path> queue, ReplicationSourceInterface source) {
075    this.conf = conf;
076    this.walGroupId = walGroupId;
077    this.queue = queue;
078    this.source = source;
079    this.sleepForRetries =
080        this.conf.getLong("replication.source.sleepforretries", 1000);    // 1 second
081    this.maxRetriesMultiplier =
082        this.conf.getInt("replication.source.maxretriesmultiplier", 300); // 5 minutes @ 1 sec per
083  }
084
085  @Override
086  public void run() {
087    setWorkerState(WorkerState.RUNNING);
088    // Loop until we close down
089    while (isActive()) {
090      int sleepMultiplier = 1;
091      // Sleep until replication is enabled again
092      if (!source.isPeerEnabled()) {
093        if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
094          sleepMultiplier++;
095        }
096        continue;
097      }
098
099      while (entryReader == null) {
100        if (sleepForRetries("Replication WAL entry reader thread not initialized",
101          sleepMultiplier)) {
102          sleepMultiplier++;
103        }
104      }
105
106      try {
107        WALEntryBatch entryBatch = entryReader.take();
108        shipEdits(entryBatch);
109      } catch (InterruptedException e) {
110        LOG.trace("Interrupted while waiting for next replication entry batch", e);
111        Thread.currentThread().interrupt();
112      }
113    }
114    // If the worker exits run loop without finishing its task, mark it as stopped.
115    if (state != WorkerState.FINISHED) {
116      setWorkerState(WorkerState.STOPPED);
117    }
118  }
119
120  /**
121   * Do the shipping logic
122   */
123  protected void shipEdits(WALEntryBatch entryBatch) {
124    List<Entry> entries = entryBatch.getWalEntries();
125    long lastReadPosition = entryBatch.getLastWalPosition();
126    currentPath = entryBatch.getLastWalPath();
127    int sleepMultiplier = 0;
128    if (entries.isEmpty()) {
129      if (lastLoggedPosition != lastReadPosition) {
130        updateLogPosition(lastReadPosition, entryBatch.getLastSeqIds());
131        // if there was nothing to ship and it's not an error
132        // set "ageOfLastShippedOp" to <now> to indicate that we're current
133        source.getSourceMetrics().setAgeOfLastShippedOp(EnvironmentEdgeManager.currentTime(),
134          walGroupId);
135      }
136      return;
137    }
138    int currentSize = (int) entryBatch.getHeapSize();
139    while (isActive()) {
140      try {
141        try {
142          source.tryThrottle(currentSize);
143        } catch (InterruptedException e) {
144          LOG.debug("Interrupted while sleeping for throttling control");
145          Thread.currentThread().interrupt();
146          // current thread might be interrupted to terminate
147          // directly go back to while() for confirm this
148          continue;
149        }
150
151        // create replicateContext here, so the entries can be GC'd upon return from this call
152        // stack
153        ReplicationEndpoint.ReplicateContext replicateContext =
154            new ReplicationEndpoint.ReplicateContext();
155        replicateContext.setEntries(entries).setSize(currentSize);
156        replicateContext.setWalGroupId(walGroupId);
157
158        long startTimeNs = System.nanoTime();
159        // send the edits to the endpoint. Will block until the edits are shipped and acknowledged
160        boolean replicated = source.getReplicationEndpoint().replicate(replicateContext);
161        long endTimeNs = System.nanoTime();
162
163        if (!replicated) {
164          continue;
165        } else {
166          sleepMultiplier = Math.max(sleepMultiplier - 1, 0);
167        }
168
169        if (this.lastLoggedPosition != lastReadPosition) {
170          // Clean up hfile references
171          int size = entries.size();
172          for (int i = 0; i < size; i++) {
173            cleanUpHFileRefs(entries.get(i).getEdit());
174          }
175          // Log and clean up WAL logs
176          updateLogPosition(lastReadPosition, entryBatch.getLastSeqIds());
177        }
178
179        source.postShipEdits(entries, currentSize);
180        // FIXME check relationship between wal group and overall
181        source.getSourceMetrics().shipBatch(entryBatch.getNbOperations(), currentSize,
182          entryBatch.getNbHFiles());
183        source.getSourceMetrics().setAgeOfLastShippedOp(
184          entries.get(entries.size() - 1).getKey().getWriteTime(), walGroupId);
185        if (LOG.isTraceEnabled()) {
186          LOG.trace("Replicated " + entries.size() + " entries or " + entryBatch.getNbOperations()
187              + " operations in " + ((endTimeNs - startTimeNs) / 1000000) + " ms");
188        }
189        break;
190      } catch (Exception ex) {
191        LOG.warn(source.getReplicationEndpoint().getClass().getName() + " threw unknown exception:"
192            + org.apache.hadoop.util.StringUtils.stringifyException(ex));
193        if (sleepForRetries("ReplicationEndpoint threw exception", sleepMultiplier)) {
194          sleepMultiplier++;
195        }
196      }
197    }
198  }
199
200  private void cleanUpHFileRefs(WALEdit edit) throws IOException {
201    String peerId = source.getPeerId();
202    if (peerId.contains("-")) {
203      // peerClusterZnode will be in the form peerId + "-" + rsZNode.
204      // A peerId will not have "-" in its name, see HBASE-11394
205      peerId = peerId.split("-")[0];
206    }
207    List<Cell> cells = edit.getCells();
208    int totalCells = cells.size();
209    for (int i = 0; i < totalCells; i++) {
210      Cell cell = cells.get(i);
211      if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
212        BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
213        List<StoreDescriptor> stores = bld.getStoresList();
214        int totalStores = stores.size();
215        for (int j = 0; j < totalStores; j++) {
216          List<String> storeFileList = stores.get(j).getStoreFileList();
217          source.getSourceManager().cleanUpHFileRefs(peerId, storeFileList);
218          source.getSourceMetrics().decrSizeOfHFileRefsQueue(storeFileList.size());
219        }
220      }
221    }
222  }
223
224  private void updateLogPosition(long lastReadPosition, Map<String, Long> lastSeqIds) {
225    source.getSourceManager().logPositionAndCleanOldLogs(currentPath, source.getQueueId(),
226      lastReadPosition, lastSeqIds, source.isRecovered());
227    lastLoggedPosition = lastReadPosition;
228  }
229
230  public void startup(UncaughtExceptionHandler handler) {
231    String name = Thread.currentThread().getName();
232    Threads.setDaemonThreadRunning(this, name + ".replicationSource." + walGroupId + ","
233        + source.getQueueId(), handler);
234  }
235
236  public PriorityBlockingQueue<Path> getLogQueue() {
237    return this.queue;
238  }
239
240  public Path getCurrentPath() {
241    return this.entryReader.getCurrentPath();
242  }
243
244  public long getCurrentPosition() {
245    return this.lastLoggedPosition;
246  }
247
248  public void setWALReader(ReplicationSourceWALReader entryReader) {
249    this.entryReader = entryReader;
250  }
251
252  public long getStartPosition() {
253    return 0;
254  }
255
256  protected boolean isActive() {
257    return source.isSourceActive() && state == WorkerState.RUNNING && !isInterrupted();
258  }
259
260  public void setWorkerState(WorkerState state) {
261    this.state = state;
262  }
263
264  public WorkerState getWorkerState() {
265    return state;
266  }
267
268  public void stopWorker() {
269    setWorkerState(WorkerState.STOPPED);
270  }
271
272  public boolean isFinished() {
273    return state == WorkerState.FINISHED;
274  }
275
276  /**
277   * Do the sleeping logic
278   * @param msg Why we sleep
279   * @param sleepMultiplier by how many times the default sleeping time is augmented
280   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
281   */
282  public boolean sleepForRetries(String msg, int sleepMultiplier) {
283    try {
284      if (LOG.isTraceEnabled()) {
285        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
286      }
287      Thread.sleep(this.sleepForRetries * sleepMultiplier);
288    } catch (InterruptedException e) {
289      LOG.debug("Interrupted while sleeping between retries");
290      Thread.currentThread().interrupt();
291    }
292    return sleepMultiplier < maxRetriesMultiplier;
293  }
