073/**
074 * Class that handles the source of a replication stream.
075 * Currently does not handle more than 1 slave
076 * For each slave cluster it selects a random number of peers
077 * using a replication ratio. For example, if replication ration = 0.1
078 * and slave cluster has 100 region servers, 10 will be selected.
079 * <p>
080 * A stream is considered down when we cannot contact a region server on the
081 * peer cluster for more than 55 seconds by default.
082 * </p>
083 *
084 */
085@InterfaceAudience.Private
086public class ReplicationSource extends Thread
087    implements ReplicationSourceInterface {

198  @Override
199  public void enqueueLog(Path log) {
200    String logPrefix = DefaultWALProvider.getWALPrefixFromWALName(log.getName());
201    PriorityBlockingQueue<Path> queue = queues.get(logPrefix);
202    if (queue == null) {
203      queue = new PriorityBlockingQueue<Path>(queueSizePerGroup, new LogsComparator());
204      queues.put(logPrefix, queue);
205      if (this.sourceRunning) {
206        // new wal group observed after source startup, start a new worker thread to track it
207        // notice: it's possible that log enqueued when this.running is set but worker thread
208        // still not launched, so it's necessary to check workerThreads before start the worker
209        final ReplicationSourceWorkerThread worker =
210            new ReplicationSourceWorkerThread(logPrefix, queue, replicationQueueInfo, this);
211        ReplicationSourceWorkerThread extant = workerThreads.putIfAbsent(logPrefix, worker);
212        if (extant != null) {
213          LOG.debug("Someone has beat us to start a worker thread for wal group " + logPrefix);
214        } else {
215          LOG.debug("Starting up worker for wal group " + logPrefix);
216          worker.startup();
217        }
218      }
219    }
220    queue.put(log);
221    int queueSize = logQueueSize.incrementAndGet();
222    this.metrics.setSizeOfLogQueue(queueSize);
223    // This will log a warning for each new log that gets created above the warn threshold
224    if (queue.size() > this.logQueueWarnThreshold) {
225      LOG.warn("WAL group " + logPrefix + " queue size: " + queueSize
226          + " exceeds value of replication.source.log.queue.warn: " + logQueueWarnThreshold);
227    }
228  }

230  @Override
231  public void addHFileRefs(TableName tableName, byte[] family, List<String> files)
232      throws ReplicationException {
233    String peerId = peerClusterZnode;
234    if (peerId.contains("-")) {
235      // peerClusterZnode will be in the form peerId + "-" + rsZNode.
236      // A peerId will not have "-" in its name, see HBASE-11394
237      peerId = peerClusterZnode.split("-")[0];
238    }
239    Map<TableName, List<String>> tableCFMap = replicationPeers.getPeer(peerId).getTableCFs();
240    if (tableCFMap != null) {
241      List<String> tableCfs = tableCFMap.get(tableName);
242      if (tableCFMap.containsKey(tableName)
243          && (tableCfs == null || tableCfs.contains(Bytes.toString(family)))) {
244        this.replicationQueues.addHFileRefs(peerId, files);
245        metrics.incrSizeOfHFileRefsQueue(files.size());
246      } else {
247        LOG.debug("HFiles will not be replicated belonging to the table " + tableName + " family "
248            + Bytes.toString(family) + " to peer id " + peerId);
249      }
250    } else {
251      // user has explicitly not defined any table cfs for replication, means replicate all the
252      // data
253      this.replicationQueues.addHFileRefs(peerId, files);
254      metrics.incrSizeOfHFileRefsQueue(files.size());
255    }
256  }

258  private void uninitialize() {
259    LOG.debug("Source exiting " + this.peerId);
260    metrics.clear();
261    if (replicationEndpoint.state() == Service.State.STARTING
262        || replicationEndpoint.state() == Service.State.RUNNING) {
263      replicationEndpoint.stopAndWait();
264    }
265  }

267  @Override
268  public void run() {
269    // mark we are running now
270    this.sourceRunning = true;
271    try {
272      // start the endpoint, connect to the cluster
273      Service.State state = replicationEndpoint.start().get();
274      if (state != Service.State.RUNNING) {
275        LOG.warn("ReplicationEndpoint was not started. Exiting");
276        uninitialize();
277        return;
278      }
279    } catch (Exception ex) {
280      LOG.warn("Error starting ReplicationEndpoint, exiting", ex);
281      throw new RuntimeException(ex);
282    }
283
284    // get the WALEntryFilter from ReplicationEndpoint and add it to default filters
285    ArrayList<WALEntryFilter> filters = Lists.newArrayList(
286      (WALEntryFilter)new SystemTableWALEntryFilter());
287    WALEntryFilter filterFromEndpoint = this.replicationEndpoint.getWALEntryfilter();
288    if (filterFromEndpoint != null) {
289      filters.add(filterFromEndpoint);
290    }
291    this.walEntryFilter = new ChainWALEntryFilter(filters);
292
293    int sleepMultiplier = 1;
294    // delay this until we are in an asynchronous thread
295    while (this.isSourceActive() && this.peerClusterId == null) {
296      this.peerClusterId = replicationEndpoint.getPeerUUID();
297      if (this.isSourceActive() && this.peerClusterId == null) {
298        if (sleepForRetries("Cannot contact the peer's zk ensemble", sleepMultiplier)) {
299          sleepMultiplier++;
300        }
301      }
302    }
303
304    // In rare case, zookeeper setting may be messed up. That leads to the incorrect
305    // peerClusterId value, which is the same as the source clusterId
306    if (clusterId.equals(peerClusterId) && !replicationEndpoint.canReplicateToSameCluster()) {
307      this.terminate("ClusterId " + clusterId + " is replicating to itself: peerClusterId "
308          + peerClusterId + " which is not allowed by ReplicationEndpoint:"
309          + replicationEndpoint.getClass().getName(), null, false);
310    }
311    LOG.info("Replicating " + clusterId + " -> " + peerClusterId);
312    // start workers
313    for (Map.Entry<String, PriorityBlockingQueue<Path>> entry : queues.entrySet()) {
314      String walGroupId = entry.getKey();
315      PriorityBlockingQueue<Path> queue = entry.getValue();
316      final ReplicationSourceWorkerThread worker =
317          new ReplicationSourceWorkerThread(walGroupId, queue, replicationQueueInfo, this);
318      ReplicationSourceWorkerThread extant = workerThreads.putIfAbsent(walGroupId, worker);
319      if (extant != null) {
320        LOG.debug("Someone has beat us to start a worker thread for wal group " + walGroupId);
321      } else {
322        LOG.debug("Starting up worker for wal group " + walGroupId);
323        worker.startup();
324      }
325    }
326  }

328  /**
329   * Do the sleeping logic
330   * @param msg Why we sleep
331   * @param sleepMultiplier by how many times the default sleeping time is augmented
332   * @return True if <code>sleepMultiplier</code> is &lt; <code>maxRetriesMultiplier</code>
333   */
334  protected boolean sleepForRetries(String msg, int sleepMultiplier) {
335    try {
336      if (LOG.isTraceEnabled()) {
337        LOG.trace(msg + ", sleeping " + sleepForRetries + " times " + sleepMultiplier);
338      }
339      Thread.sleep(this.sleepForRetries * sleepMultiplier);
340    } catch (InterruptedException e) {
341      LOG.debug("Interrupted while sleeping between retries");
342      Thread.currentThread().interrupt();
343    }
344    return sleepMultiplier < maxRetriesMultiplier;
345  }

379  public void terminate(String reason, Exception cause, boolean join) {
380    if (cause == null) {
381      LOG.info("Closing source "
382          + this.peerClusterZnode + " because: " + reason);
383
384    } else {
385      LOG.error("Closing source " + this.peerClusterZnode
386          + " because an error occurred: " + reason, cause);
387    }
388    this.sourceRunning = false;
389    Collection<ReplicationSourceWorkerThread> workers = workerThreads.values();
390    for (ReplicationSourceWorkerThread worker : workers) {
391      worker.setWorkerRunning(false);
392      worker.interrupt();
393    }
394    ListenableFuture<Service.State> future = null;
395    if (this.replicationEndpoint != null) {
396      future = this.replicationEndpoint.stop();
397    }
398    if (join) {
399      for (ReplicationSourceWorkerThread worker : workers) {
400        Threads.shutdown(worker, this.sleepForRetries);
401        LOG.info("ReplicationSourceWorker " + worker.getName() + " terminated");
402      }
403      if (future != null) {
404        try {
405          future.get();
406        } catch (Exception e) {
407          LOG.warn("Got exception:" + e);
408        }
409      }
410    }
411  }

436  /**
437   * Comparator used to compare logs together based on their start time
438   */
439  public static class LogsComparator implements Comparator<Path> {
440
441    @Override
442    public int compare(Path o1, Path o2) {
443      return Long.valueOf(getTS(o1)).compareTo(getTS(o2));
444    }
445
446    /**
447     * Split a path to get the start time
448     * For example: 10.20.20.171%3A60020.1277499063250
449     * @param p path to split
450     * @return start time
451     */
452    private long getTS(Path p) {
453      String[] parts = p.getName().split("\\.");
454      return Long.parseLong(parts[parts.length-1]);
455    }
456  }

458  @Override
459  public String getStats() {
460    StringBuilder sb = new StringBuilder();
461    sb.append("Total replicated edits: ").append(totalReplicatedEdits)
462        .append(", current progress: \n");
463    for (Map.Entry<String, ReplicationSourceWorkerThread> entry : workerThreads.entrySet()) {
464      String walGroupId = entry.getKey();
465      ReplicationSourceWorkerThread worker = entry.getValue();
466      long position = worker.getCurrentPosition();
467      Path currentPath = worker.getCurrentPath();
468      sb.append("walGroup [").append(walGroupId).append("]: ");
469      if (currentPath != null) {
470        sb.append("currently replicating from: ").append(currentPath).append(" at position: ")
471            .append(position).append("\n");
472      } else {
473        sb.append("no replication ongoing, waiting for new log");
474      }
475    }
476    return sb.toString();
477  }

487  public class ReplicationSourceWorkerThread extends Thread {
488    ReplicationSource source;
489    String walGroupId;
490    PriorityBlockingQueue<Path> queue;
491    ReplicationQueueInfo replicationQueueInfo;
492    // Our reader for the current log. open/close handled by repLogReader
493    private WAL.Reader reader;
494    // Last position in the log that we sent to ZooKeeper
495    private long lastLoggedPosition = -1;
496    // Path of the current log
497    private volatile Path currentPath;
498    // Handle on the log reader helper
499    private ReplicationWALReaderManager repLogReader;
500    // Current number of operations (Put/Delete) that we need to replicate
501    private int currentNbOperations = 0;
502    // Current size of data we need to replicate
503    private int currentSize = 0;
504    // Indicates whether this particular worker is running
505    private boolean workerRunning = true;
506    // Current number of hfiles that we need to replicate
507    private long currentNbHFiles = 0;
508
509    public ReplicationSourceWorkerThread(String walGroupId,
510        PriorityBlockingQueue<Path> queue, ReplicationQueueInfo replicationQueueInfo, ReplicationSource source) {
511      this.walGroupId = walGroupId;
512      this.queue = queue;
513      this.replicationQueueInfo = replicationQueueInfo;
514      this.repLogReader = new ReplicationWALReaderManager(fs, conf);
515      this.source = source;
516    }
517
518    @Override
519    public void run() {
520      // If this is recovered, the queue is already full and the first log
521      // normally has a position (unless the RS failed between 2 logs)
522      if (this.replicationQueueInfo.isQueueRecovered()) {
523        try {
524          this.repLogReader.setPosition(replicationQueues.getLogPosition(peerClusterZnode,
525            this.queue.peek().getName()));
526          if (LOG.isTraceEnabled()) {
527            LOG.trace("Recovered queue started with log " + this.queue.peek() + " at position "
528                + this.repLogReader.getPosition());
529          }
530        } catch (ReplicationException e) {
531          terminate("Couldn't get the position of this recovered queue " + peerClusterZnode, e);
532        }
533      }
534      // Loop until we close down
535      while (isWorkerActive()) {
536        int sleepMultiplier = 1;
537        // Sleep until replication is enabled again
538        if (!isPeerEnabled()) {
539          if (sleepForRetries("Replication is disabled", sleepMultiplier)) {
540            sleepMultiplier++;
541          }
542          continue;
543        }
544        Path oldPath = getCurrentPath(); //note that in the current scenario,
545                                         //oldPath will be null when a log roll
546                                         //happens.
547        // Get a new path
548        boolean hasCurrentPath = getNextPath();
549        if (getCurrentPath() != null && oldPath == null) {
550          sleepMultiplier = 1; //reset the sleepMultiplier on a path change
551        }
552        if (!hasCurrentPath) {
553          if (sleepForRetries("No log to process", sleepMultiplier)) {
554            sleepMultiplier++;
555          }
556          continue;
557        }
558        boolean currentWALisBeingWrittenTo = false;
559        //For WAL files we own (rather than recovered), take a snapshot of whether the
560        //current WAL file (this.currentPath) is in use (for writing) NOW!
561        //Since the new WAL paths are enqueued only after the prev WAL file
562        //is 'closed', presence of an element in the queue means that
563        //the previous WAL file was closed, else the file is in use (currentPath)
564        //We take the snapshot now so that we are protected against races
565        //where a new file gets enqueued while the current file is being processed
566        //(and where we just finished reading the current file).
567        if (!this.replicationQueueInfo.isQueueRecovered() && queue.size() == 0) {
568          currentWALisBeingWrittenTo = true;
569        }
570        // Open a reader on it
571        if (!openReader(sleepMultiplier)) {
572          // Reset the sleep multiplier, else it'd be reused for the next file
573          sleepMultiplier = 1;
574          continue;
575        }
576
577        // If we got a null reader but didn't continue, then sleep and continue
578        if (this.reader == null) {
579          if (sleepForRetries("Unable to open a reader", sleepMultiplier)) {
580            sleepMultiplier++;
581          }
582          continue;
583        }
584
585        boolean gotIOE = false;
586        currentNbOperations = 0;
587        currentNbHFiles = 0;
588        List<WAL.Entry> entries = new ArrayList<WAL.Entry>(1);
589        currentSize = 0;
590        try {
591          if (readAllEntriesToReplicateOrNextFile(currentWALisBeingWrittenTo, entries)) {
592            continue;
593          }
594        } catch (IOException ioe) {
595          LOG.warn(peerClusterZnode + " Got: ", ioe);
596          gotIOE = true;
597          if (ioe.getCause() instanceof EOFException) {
598
599            boolean considerDumping = false;
600            if (this.replicationQueueInfo.isQueueRecovered()) {
601              try {
602                FileStatus stat = fs.getFileStatus(this.currentPath);
603                if (stat.getLen() == 0) {
604                  LOG.warn(peerClusterZnode + " Got EOF and the file was empty");
605                }
606                considerDumping = true;
607              } catch (IOException e) {
608                LOG.warn(peerClusterZnode + " Got while getting file size: ", e);
609              }
610            }
611
612            if (considerDumping &&
613                sleepMultiplier == maxRetriesMultiplier &&
614                processEndOfFile()) {
615              continue;
616            }
617          }
618        } finally {
619          try {
620            this.reader = null;
621            this.repLogReader.closeReader();
622          } catch (IOException e) {
623            gotIOE = true;
624            LOG.warn("Unable to finalize the tailing of a file", e);
625          }
626        }
627
628        // If we didn't get anything to replicate, or if we hit a IOE,
629        // wait a bit and retry.
630        // But if we need to stop, don't bother sleeping
631        if (isWorkerActive() && (gotIOE || entries.isEmpty())) {
632          if (this.lastLoggedPosition != this.repLogReader.getPosition()) {
633            manager.logPositionAndCleanOldLogs(this.currentPath,
634                peerClusterZnode, this.repLogReader.getPosition(),
635                this.replicationQueueInfo.isQueueRecovered(), currentWALisBeingWrittenTo);
636            this.lastLoggedPosition = this.repLogReader.getPosition();
637          }
638          // Reset the sleep multiplier if nothing has actually gone wrong
639          if (!gotIOE) {
640            sleepMultiplier = 1;
641            // if there was nothing to ship and it's not an error
642            // set "ageOfLastShippedOp" to <now> to indicate that we're current
643            metrics.setAgeOfLastShippedOp(System.currentTimeMillis(), walGroupId);
644          }
645          if (sleepForRetries("Nothing to replicate", sleepMultiplier)) {
646            sleepMultiplier++;
647          }
648          continue;
649        }
650        sleepMultiplier = 1;
651        shipEdits(currentWALisBeingWrittenTo, entries);
652      }
653      if (replicationQueueInfo.isQueueRecovered()) {
654        // use synchronize to make sure one last thread will clean the queue
655        synchronized (workerThreads) {
656          Threads.sleep(100);// wait a short while for other worker thread to fully exit
657          boolean allOtherTaskDone = true;
658          for (ReplicationSourceWorkerThread worker : workerThreads.values()) {
659            if (!worker.equals(this) && worker.isAlive()) {
660              allOtherTaskDone = false;
661              break;
662            }
663          }
664          if (allOtherTaskDone) {
665            manager.closeRecoveredQueue(this.source);
666            LOG.info("Finished recovering queue " + peerClusterZnode
667                + " with the following stats: " + getStats());
668          }
669        }
670      }
671    }

673    /**
674     * Read all the entries from the current log files and retain those that need to be replicated.
675     * Else, process the end of the current file.
676     * @param currentWALisBeingWrittenTo is the current WAL being written to
677     * @param entries resulting entries to be replicated
678     * @return true if we got nothing and went to the next file, false if we got entries
679     * @throws IOException
680     */
681    protected boolean readAllEntriesToReplicateOrNextFile(boolean currentWALisBeingWrittenTo,
682        List<WAL.Entry> entries) throws IOException {
683      long seenEntries = 0;
684      if (LOG.isTraceEnabled()) {
685        LOG.trace("Seeking in " + this.currentPath + " at position "
686            + this.repLogReader.getPosition());
687      }
688      this.repLogReader.seek();
689      long positionBeforeRead = this.repLogReader.getPosition();
690      WAL.Entry entry = this.repLogReader.readNextAndSetPosition();
691      while (entry != null) {
692        metrics.incrLogEditsRead();
693        seenEntries++;
694
695        // don't replicate if the log entries have already been consumed by the cluster
696        if (replicationEndpoint.canReplicateToSameCluster()
697            || !entry.getKey().getClusterIds().contains(peerClusterId)) {
698          // Remove all KVs that should not be replicated
699          entry = walEntryFilter.filter(entry);
700          WALEdit edit = null;
701          WALKey logKey = null;
702          if (entry != null) {
703            edit = entry.getEdit();
704            logKey = entry.getKey();
705          }
706
707          if (edit != null && edit.size() != 0) {
708            // Mark that the current cluster has the change
709            logKey.addClusterId(clusterId);
710            currentNbOperations += countDistinctRowKeys(edit);
711            entries.add(entry);
712            currentSize += entry.getEdit().heapSize();
713          } else {
714            metrics.incrLogEditsFiltered();
715          }
716        }
717        // Stop if too many entries or too big
718        // FIXME check the relationship between single wal group and overall
719        if (currentSize >= replicationQueueSizeCapacity
720            || entries.size() >= replicationQueueNbCapacity) {
721          break;
722        }
723        try {
724          entry = this.repLogReader.readNextAndSetPosition();
725        } catch (IOException ie) {
726          LOG.debug("Break on IOE: " + ie.getMessage());
727          break;
728        }
729      }
730      metrics.incrLogReadInBytes(this.repLogReader.getPosition() - positionBeforeRead);
731      if (currentWALisBeingWrittenTo) {
732        return false;
733      }
734      // If we didn't get anything and the queue has an object, it means we
735      // hit the end of the file for sure
736      return seenEntries == 0 && processEndOfFile();
737    }

739    private void cleanUpHFileRefs(WALEdit edit) throws IOException {
740      String peerId = peerClusterZnode;
741      if (peerId.contains("-")) {
742        // peerClusterZnode will be in the form peerId + "-" + rsZNode.
743        // A peerId will not have "-" in its name, see HBASE-11394
744        peerId = peerClusterZnode.split("-")[0];
745      }
746      List<Cell> cells = edit.getCells();
747      for (int i = 0; i < cells.size(); i++) {
748        Cell cell = cells.get(i);
749        if (CellUtil.matchingQualifier(cell, WALEdit.BULK_LOAD)) {
750          BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cell);
751          List<StoreDescriptor> stores = bld.getStoresList();
752          for (int j = 0; j < stores.size(); j++) {
753            List<String> storeFileList = stores.get(j).getStoreFileList();
754            manager.cleanUpHFileRefs(peerId, storeFileList);
755            metrics.decrSizeOfHFileRefsQueue(storeFileList.size());
756          }
757        }
758      }
759    }

761    /**
762     * Poll for the next path
763     * @return true if a path was obtained, false if not
764     */
765    protected boolean getNextPath() {
766      try {
767        if (this.currentPath == null) {
768          this.currentPath = queue.poll(sleepForRetries, TimeUnit.MILLISECONDS);
769          int queueSize = logQueueSize.decrementAndGet();
770          metrics.setSizeOfLogQueue(queueSize);
771          if (this.currentPath != null) {
772            // For recovered queue: must use peerClusterZnode since peerId is a parsed value
773            manager.cleanOldLogs(this.currentPath.getName(), peerClusterZnode,
774              this.replicationQueueInfo.isQueueRecovered());
775            if (LOG.isTraceEnabled()) {
776              LOG.trace("New log: " + this.currentPath);
777            }
778          }
779        }
780      } catch (InterruptedException e) {
781        LOG.warn("Interrupted while reading edits", e);
782      }
783      return this.currentPath != null;
784    }

786    /**
787     * Open a reader on the current path
788     *
789     * @param sleepMultiplier by how many times the default sleeping time is augmented
790     * @return true if we should continue with that file, false if we are over with it
791     */
792    protected boolean openReader(int sleepMultiplier) {
793
794      try {
795        try {
796          if (LOG.isTraceEnabled()) {
797            LOG.trace("Opening log " + this.currentPath);
798          }
799          this.reader = repLogReader.openReader(this.currentPath);
800        } catch (FileNotFoundException fnfe) {
801          if (this.replicationQueueInfo.isQueueRecovered()) {
802            // We didn't find the log in the archive directory, look if it still
803            // exists in the dead RS folder (there could be a chain of failures
804            // to look at)
805            List<String> deadRegionServers = this.replicationQueueInfo.getDeadRegionServers();
806            LOG.info("NB dead servers : " + deadRegionServers.size());
807            final Path rootDir = FSUtils.getRootDir(conf);
808            for (String curDeadServerName : deadRegionServers) {
809              final Path deadRsDirectory = new Path(rootDir,
810                  DefaultWALProvider.getWALDirectoryName(curDeadServerName));
811              Path[] locs = new Path[] {
812                  new Path(deadRsDirectory, currentPath.getName()),
813                  new Path(deadRsDirectory.suffix(DefaultWALProvider.SPLITTING_EXT),
814                                            currentPath.getName()),
815              };
816              for (Path possibleLogLocation : locs) {
817                LOG.info("Possible location " + possibleLogLocation.toUri().toString());
818                if (manager.getFs().exists(possibleLogLocation)) {
819                  // We found the right new location
820                  LOG.info("Log " + this.currentPath + " still exists at " +
821                      possibleLogLocation);
822                  // Breaking here will make us sleep since reader is null
823                  // TODO why don't we need to set currentPath and call openReader here?
824                  return true;
825                }
826              }
827            }
828            // In the case of disaster/recovery, HMaster may be shutdown/crashed before flush data
829            // from .logs to .oldlogs. Loop into .logs folders and check whether a match exists
830            if (stopper instanceof ReplicationSyncUp.DummyServer) {
831              // N.B. the ReplicationSyncUp tool sets the manager.getLogDir to the root of the wal
832              //      area rather than to the wal area for a particular region server.
833              FileStatus[] rss = fs.listStatus(manager.getLogDir());
834              for (FileStatus rs : rss) {
835                Path p = rs.getPath();
836                FileStatus[] logs = fs.listStatus(p);
837                for (FileStatus log : logs) {
838                  p = new Path(p, log.getPath().getName());
839                  if (p.getName().equals(currentPath.getName())) {
840                    currentPath = p;
841                    LOG.info("Log " + currentPath.getName() + " found at " + currentPath);
842                    // Open the log at the new location
843                    this.openReader(sleepMultiplier);
844                    return true;
845                  }
846                }
847              }
848            }
849
850            // TODO What happens if the log was missing from every single location?
851            // Although we need to check a couple of times as the log could have
852            // been moved by the master between the checks
853            // It can also happen if a recovered queue wasn't properly cleaned,
854            // such that the znode pointing to a log exists but the log was
855            // deleted a long time ago.
856            // For the moment, we'll throw the IO and processEndOfFile
857            throw new IOException("File from recovered queue is " +
858                "nowhere to be found", fnfe);
859          } else {
860            // If the log was archived, continue reading from there
861            Path archivedLogLocation =
862                new Path(manager.getOldLogDir(), currentPath.getName());
863            if (manager.getFs().exists(archivedLogLocation)) {
864              currentPath = archivedLogLocation;
865              LOG.info("Log " + this.currentPath + " was moved to " +
866                  archivedLogLocation);
867              // Open the log at the new location
868              this.openReader(sleepMultiplier);
869
870            }
871            // TODO What happens the log is missing in both places?
872          }
873        }
874      } catch (IOException ioe) {
875        if (ioe instanceof EOFException && isCurrentLogEmpty()) return true;
876        LOG.warn(peerClusterZnode + " Got: ", ioe);
877        this.reader = null;
878        if (ioe.getCause() instanceof NullPointerException) {
879          // Workaround for race condition in HDFS-4380
880          // which throws a NPE if we open a file before any data node has the most recent block
881          // Just sleep and retry. Will require re-reading compressed WALs for compressionContext.
882          LOG.warn("Got NPE opening reader, will retry.");
883        } else if (sleepMultiplier == maxRetriesMultiplier) {
884          // TODO Need a better way to determine if a file is really gone but
885          // TODO without scanning all logs dir
886          LOG.warn("Waited too long for this file, considering dumping");
887          return !processEndOfFile();
888        }
889      }
890      return true;
891    }

893    /*
894     * Checks whether the current log file is empty, and it is not a recovered queue. This is to
895     * handle scenario when in an idle cluster, there is no entry in the current log and we keep on
896     * trying to read the log file and get EOFException. In case of a recovered queue the last log
897     * file may be empty, and we don't want to retry that.
898     */
899    private boolean isCurrentLogEmpty() {
900      return (this.repLogReader.getPosition() == 0 &&
901          !this.replicationQueueInfo.isQueueRecovered() && queue.size() == 0);
902    }

904    /**
905     * Count the number of different row keys in the given edit because of mini-batching. We assume
906     * that there's at least one Cell in the WALEdit.
907     * @param edit edit to count row keys from
908     * @return number of different row keys
909     */
910    private int countDistinctRowKeys(WALEdit edit) {
911      List<Cell> cells = edit.getCells();
912      int distinctRowKeys = 1;
913      int totalHFileEntries = 0;
914      Cell lastCell = cells.get(0);
915
916      for (int i = 0; i < edit.size(); i++) {
917        // Count HFiles to be replicated
918        if (CellUtil.matchingQualifier(cells.get(i), WALEdit.BULK_LOAD)) {
919          try {
920            BulkLoadDescriptor bld = WALEdit.getBulkLoadDescriptor(cells.get(i));
921            List<StoreDescriptor> stores = bld.getStoresList();
922            for (int j = 0; j < stores.size(); j++) {
923              totalHFileEntries += stores.get(j).getStoreFileList().size();
924            }
925          } catch (IOException e) {
926            LOG.error("Failed to deserialize bulk load entry from wal edit. "
927                + "This its hfiles count will not be added into metric.");
928          }
929        }
930
931        if (!CellUtil.matchingRows(cells.get(i), lastCell)) {
932          distinctRowKeys++;
933        }
934        lastCell = cells.get(i);
935      }
936      currentNbHFiles += totalHFileEntries;
937      return distinctRowKeys + totalHFileEntries;
938    }

940    /**
941     * Do the shipping logic
942     * @param currentWALisBeingWrittenTo was the current WAL being (seemingly)
943     * written to when this method was called
944     */
945    protected void shipEdits(boolean currentWALisBeingWrittenTo, List<WAL.Entry> entries) {
946      int sleepMultiplier = 0;
947      if (entries.isEmpty()) {
948        LOG.warn("Was given 0 edits to ship");
949        return;
950      }
951      while (isWorkerActive()) {
952        try {
953          if (throttler.isEnabled()) {
954            long sleepTicks = throttler.getNextSleepInterval(currentSize);
955            if (sleepTicks > 0) {
956              try {
957                if (LOG.isTraceEnabled()) {
958                  LOG.trace("To sleep " + sleepTicks + "ms for throttling control");
959                }
960                Thread.sleep(sleepTicks);
961              } catch (InterruptedException e) {
962                LOG.debug("Interrupted while sleeping for throttling control");
963                Thread.currentThread().interrupt();
964                // current thread might be interrupted to terminate
965                // directly go back to while() for confirm this
966                continue;
967              }
968              // reset throttler's cycle start tick when sleep for throttling occurs
969              throttler.resetStartTick();
970            }
971          }
972          // create replicateContext here, so the entries can be GC'd upon return from this call
973          // stack
974          ReplicationEndpoint.ReplicateContext replicateContext =
975              new ReplicationEndpoint.ReplicateContext();
976          replicateContext.setEntries(entries).setSize(currentSize);
977          replicateContext.setWalGroupId(walGroupId);
978
979          long startTimeNs = System.nanoTime();
980          // send the edits to the endpoint. Will block until the edits are shipped and acknowledged
981          boolean replicated = replicationEndpoint.replicate(replicateContext);
982          long endTimeNs = System.nanoTime();
983
984          if (!replicated) {
985            continue;
986          } else {
987            sleepMultiplier = Math.max(sleepMultiplier - 1, 0);
988          }
989
990          if (this.lastLoggedPosition != this.repLogReader.getPosition()) {
991            //Clean up hfile references
992            int size = entries.size();
993            for (int i = 0; i < size; i++) {
994              cleanUpHFileRefs(entries.get(i).getEdit());
995            }
996            //Log and clean up WAL logs
997            manager.logPositionAndCleanOldLogs(this.currentPath, peerClusterZnode,
998              this.repLogReader.getPosition(), this.replicationQueueInfo.isQueueRecovered(),
999              currentWALisBeingWrittenTo);
1000            this.lastLoggedPosition = this.repLogReader.getPosition();
1001          }
1002          if (throttler.isEnabled()) {
1003            throttler.addPushSize(currentSize);
1004          }
1005          totalReplicatedEdits.addAndGet(entries.size());
1006          totalReplicatedOperations.addAndGet(currentNbOperations);
1007          // FIXME check relationship between wal group and overall
1008          metrics.shipBatch(currentNbOperations, currentSize / 1024, currentNbHFiles);
1009          metrics.setAgeOfLastShippedOp(entries.get(entries.size() - 1).getKey().getWriteTime(),
1010            walGroupId);
1011          if (LOG.isTraceEnabled()) {
1012            LOG.trace("Replicated " + totalReplicatedEdits + " entries in total, or "
1013                + totalReplicatedOperations + " operations in "
1014                + ((endTimeNs - startTimeNs) / 1000000) + " ms");
1015          }
1016          break;
1017        } catch (Exception ex) {
1018          LOG.warn(replicationEndpoint.getClass().getName() + " threw unknown exception:"
1019              + org.apache.hadoop.util.StringUtils.stringifyException(ex));
1020          if (sleepForRetries("ReplicationEndpoint threw exception", sleepMultiplier)) {
1021            sleepMultiplier++;
1022          }
1023        }
1024      }
1025    }

1027    /**
1028     * If the queue isn't empty, switch to the next one Else if this is a recovered queue, it means
1029     * we're done! Else we'll just continue to try reading the log file
1030     * @return true if we're done with the current file, false if we should continue trying to read
1031     *         from it
1032     */
1033    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "DE_MIGHT_IGNORE",
1034        justification = "Yeah, this is how it works")
1035    protected boolean processEndOfFile() {
1036      if (this.queue.size() != 0) {
1037        if (LOG.isTraceEnabled()) {
1038          String filesize = "N/A";
1039          try {
1040            FileStatus stat = fs.getFileStatus(this.currentPath);
1041            filesize = stat.getLen() + "";
1042          } catch (IOException ex) {
1043          }
1044          LOG.trace("Reached the end of log " + this.currentPath + ", stats: " + getStats()
1045              + ", and the length of the file is " + filesize);
1046        }
1047        this.currentPath = null;
1048        this.repLogReader.finishCurrentFile();
1049        this.reader = null;
1050        return true;
1051      } else if (this.replicationQueueInfo.isQueueRecovered()) {
1052        LOG.debug("Finished recovering queue for group " + walGroupId + " of peer "
1053            + peerClusterZnode);
1054        workerRunning = false;
1055        return true;
1056      }
1057      return false;
1058    }

1060    public void startup() {
1061      String n = Thread.currentThread().getName();
1062      Thread.UncaughtExceptionHandler handler = new Thread.UncaughtExceptionHandler() {
1063        @Override
1064        public void uncaughtException(final Thread t, final Throwable e) {
1065          LOG.error("Unexpected exception in ReplicationSourceWorkerThread," + " currentPath="
1066              + getCurrentPath(), e);
1067        }

1069      Threads.setDaemonThreadRunning(this, n + ".replicationSource." + walGroupId + ","
1070          + peerClusterZnode, handler);
1071      workerThreads.put(walGroupId, this);
1072    }
