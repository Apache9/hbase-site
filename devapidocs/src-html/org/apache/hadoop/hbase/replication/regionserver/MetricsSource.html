019package org.apache.hadoop.hbase.replication.regionserver;
020
021import java.util.HashMap;
022import java.util.Map;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.CompatibilitySingletonFactory;
028import org.apache.hadoop.hbase.HBaseInterfaceAudience;
029import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
030
031/**
032 * This class is for maintaining the various replication statistics for a source and publishing them
033 * through the metrics interfaces.
034 */
035@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.REPLICATION)
036public class MetricsSource {
037
038  private static final Log LOG = LogFactory.getLog(MetricsSource.class);
039
040  // tracks last shipped timestamp for each wal group
041  private Map<String, Long> lastTimeStamps = new HashMap<String, Long>();
042  private int lastQueueSize = 0;
043  private String id;
044
045  private final MetricsReplicationSourceSource singleSourceSource;
046  private final MetricsReplicationSourceSource globalSourceSource;
047
048  /**
049   * Constructor used to register the metrics
050   *
051   * @param id Name of the source this class is monitoring
052   */
053  public MetricsSource(String id) {
054    this.id = id;
055    singleSourceSource =
056        CompatibilitySingletonFactory.getInstance(MetricsReplicationSourceFactory.class)
057            .getSource(id);
058    globalSourceSource = CompatibilitySingletonFactory.getInstance(MetricsReplicationSourceFactory.class).getGlobalSource();
059  }
060
061  /**
062   * Set the age of the last edit that was shipped
063   * @param timestamp write time of the edit
064   * @param walGroup which group we are setting
065   */
066  public void setAgeOfLastShippedOp(long timestamp, String walGroup) {
067    long age = EnvironmentEdgeManager.currentTime() - timestamp;
068    singleSourceSource.setLastShippedAge(age);
069    globalSourceSource.setLastShippedAge(age);
070    this.lastTimeStamps.put(walGroup, timestamp);
071  }
072
073  /**
074   * Convenience method to use the last given timestamp to refresh the age of the last edit. Used
075   * when replication fails and need to keep that metric accurate.
076   * @param walGroupId id of the group to update
077   */
078  public void refreshAgeOfLastShippedOp(String walGroupId) {
079    Long lastTimestamp = this.lastTimeStamps.get(walGroupId);
080    if (lastTimestamp == null) {
081      this.lastTimeStamps.put(walGroupId, 0L);
082      lastTimestamp = 0L;
083    }
084    if (lastTimestamp > 0) {
085      setAgeOfLastShippedOp(lastTimestamp, walGroupId);
086    }
087  }
088
089  /**
090   * Set the size of the log queue
091   *
092   * @param size the size.
093   */
094  public void setSizeOfLogQueue(int size) {
095    singleSourceSource.setSizeOfLogQueue(size);
096    globalSourceSource.incrSizeOfLogQueue(size - lastQueueSize);
097    lastQueueSize = size;
098  }
099
100  /**
101   * Add on the the number of log edits read
102   *
103   * @param delta the number of log edits read.
104   */
105  private void incrLogEditsRead(long delta) {
106    singleSourceSource.incrLogReadInEdits(delta);
107    globalSourceSource.incrLogReadInEdits(delta);
108  }
109
110  /** Increment the number of log edits read by one. */
111  public void incrLogEditsRead() {
112    incrLogEditsRead(1);
113  }
114
115  /**
116   * Add on the number of log edits filtered
117   *
118   * @param delta the number filtered.
119   */
120  public void incrLogEditsFiltered(long delta) {
121    singleSourceSource.incrLogEditsFiltered(delta);
122    globalSourceSource.incrLogEditsFiltered(delta);
123  }
124
125  /** The number of log edits filtered out. */
126  public void incrLogEditsFiltered() {
127    incrLogEditsFiltered(1);
128  }
129
130  /**
131   * Convience method to apply changes to metrics do to shipping a batch of logs.
132   *
133   * @param batchSize the size of the batch that was shipped to sinks.
134   */
135  public void shipBatch(long batchSize, int sizeInKB) {
136    singleSourceSource.incrBatchesShipped(1);
137    globalSourceSource.incrBatchesShipped(1);
138
139    singleSourceSource.incrOpsShipped(batchSize);
140    globalSourceSource.incrOpsShipped(batchSize);
141
142    singleSourceSource.incrShippedKBs(sizeInKB);
143    globalSourceSource.incrShippedKBs(sizeInKB);
144  }
145
146  /** increase the byte number read by source from log file */
147  public void incrLogReadInBytes(long readInBytes) {
148    singleSourceSource.incrLogReadInBytes(readInBytes);
149    globalSourceSource.incrLogReadInBytes(readInBytes);
150  }
151
152  /** Removes all metrics about this Source. */
153  public void clear() {
154    singleSourceSource.clear();
155    globalSourceSource.decrSizeOfLogQueue(lastQueueSize);
156    lastTimeStamps.clear();
157    lastQueueSize = 0;
158  }
159
160  /**
161   * Get AgeOfLastShippedOp
162   * @return AgeOfLastShippedOp
163   */
164  public Long getAgeOfLastShippedOp() {
165    return singleSourceSource.getLastShippedAge();
166  }
167
168  /**
169   * Get the sizeOfLogQueue
170   * @return sizeOfLogQueue
171   */
172  public int getSizeOfLogQueue() {
173    return this.lastQueueSize;
174  }
175
176  /**
177   * Get the timeStampsOfLastShippedOp, if there are multiple groups, return the latest one
178   * @return lastTimestampForAge
179   */
180  public long getTimeStampOfLastShippedOp() {
181    long lastTimestamp = 0L;
182    for (long ts : lastTimeStamps.values()) {
183      if (ts > lastTimestamp) {
184        lastTimestamp = ts;
185      }
186    }
187    return lastTimestamp;
188  }
189
190  /**
191   * Get the slave peer ID
192   * @return peerID
193   */
194  public String getPeerID() {
195    return id;
196  }
197}
