051/**
052 * This class provides an implementation of the ReplicationPeers interface using ZooKeeper. The
053 * peers znode contains a list of all peer replication clusters and the current replication state of
054 * those clusters. It has one child peer znode for each peer cluster. The peer znode is named with
055 * the cluster id provided by the user in the HBase shell. The value of the peer znode contains the
056 * peers cluster key provided by the user in the HBase Shell. The cluster key contains a list of
057 * zookeeper quorum peers, the client port for the zookeeper quorum, and the base znode for HBase.
058 * For example:
059 *
060 *  /hbase/replication/peers/1 [Value: zk1.host.com,zk2.host.com,zk3.host.com:2181:/hbase]
061 *  /hbase/replication/peers/2 [Value: zk5.host.com,zk6.host.com,zk7.host.com:2181:/hbase]
062 *
063 * Each of these peer znodes has a child znode that indicates whether or not replication is enabled
064 * on that peer cluster. These peer-state znodes do not have child znodes and simply contain a
065 * boolean value (i.e. ENABLED or DISABLED). This value is read/maintained by the
066 * ReplicationPeer.PeerStateTracker class. For example:
067 *
068 * /hbase/replication/peers/1/peer-state [Value: ENABLED]
069 *
070 * Each of these peer znodes has a child znode that indicates which data will be replicated
071 * to the peer cluster. These peer-tableCFs znodes do not have child znodes and only have a
072 * table/cf list config. This value is read/maintained by the ReplicationPeer.TableCFsTracker
073 * class. For example:
074 *
075 * /hbase/replication/peers/1/tableCFs [Value: "table1; table2:cf1,cf3; table3:cfx,cfy"]
076 */
077@InterfaceAudience.Private
078public class ReplicationPeersZKImpl extends ReplicationStateZKBase implements ReplicationPeers {

087  public ReplicationPeersZKImpl(final ZKWatcher zk, final Configuration conf,
088                                final ReplicationQueuesClient queuesClient, Abortable abortable) {
089    super(zk, conf, abortable);
090    this.abortable = abortable;
091    this.peerClusters = new ConcurrentHashMap<>();
092    this.queuesClient = queuesClient;
093  }

095  @Override
096  public void init() throws ReplicationException {
097    try {
098      if (ZKUtil.checkExists(this.zookeeper, this.peersZNode) < 0) {
099        ZKUtil.createWithParents(this.zookeeper, this.peersZNode);
100      }
101    } catch (KeeperException e) {
102      throw new ReplicationException("Could not initialize replication peers", e);
103    }
104    addExistingPeers();
105  }

107  @Override
108  public void registerPeer(String id, ReplicationPeerConfig peerConfig, boolean enabled)
109      throws ReplicationException {
110    try {
111      if (peerExists(id)) {
112        throw new IllegalArgumentException("Cannot add a peer with id=" + id
113            + " because that id already exists.");
114      }
115
116      if(id.contains("-")){
117        throw new IllegalArgumentException("Found invalid peer name:" + id);
118      }
119
120      if (peerConfig.getClusterKey() != null) {
121        try {
122          ZKConfig.validateClusterKey(peerConfig.getClusterKey());
123        } catch (IOException ioe) {
124          throw new IllegalArgumentException(ioe.getMessage());
125        }
126      }
127
128      checkQueuesDeleted(id);
129
130      ZKUtil.createWithParents(this.zookeeper, this.peersZNode);
131
132      List<ZKUtilOp> listOfOps = new ArrayList<>(2);
133      ZKUtilOp op1 =
134          ZKUtilOp.createAndFailSilent(getPeerNode(id),
135            ReplicationPeerConfigUtil.toByteArray(peerConfig));
136      ZKUtilOp op2 =
137          ZKUtilOp.createAndFailSilent(getPeerStateNode(id), enabled ? ENABLED_ZNODE_BYTES
138              : DISABLED_ZNODE_BYTES);
139      listOfOps.add(op1);
140      listOfOps.add(op2);
141      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, false);
142    } catch (KeeperException e) {
143      throw new ReplicationException("Could not add peer with id=" + id + ", peerConfif=>"
144          + peerConfig + ", state=" + (enabled ? "ENABLED" : "DISABLED"), e);
145    }
146  }

148  @Override
149  public void unregisterPeer(String id) throws ReplicationException {
150    try {
151      if (!peerExists(id)) {
152        throw new IllegalArgumentException("Cannot remove peer with id=" + id
153            + " because that id does not exist.");
154      }
155      ZKUtil.deleteNodeRecursively(this.zookeeper, ZNodePaths.joinZNode(this.peersZNode, id));
156    } catch (KeeperException e) {
157      throw new ReplicationException("Could not remove peer with id=" + id, e);
158    }
159  }

173  @Override
174  public Map<TableName, List<String>> getPeerTableCFsConfig(String id) throws ReplicationException {
175    try {
176      if (!peerExists(id)) {
177        throw new IllegalArgumentException("peer " + id + " doesn't exist");
178      }
179      try {
180        ReplicationPeerConfig rpc = getReplicationPeerConfig(id);
181        if (rpc == null) {
182          throw new ReplicationException("Unable to get tableCFs of the peer with id=" + id);
183        }
184        return rpc.getTableCFsMap();
185      } catch (Exception e) {
186        throw new ReplicationException(e);
187      }
188    } catch (KeeperException e) {
189      throw new ReplicationException("Unable to get tableCFs of the peer with id=" + id, e);
190    }
191  }

193  @Override
194  public void setPeerTableCFsConfig(String id,
195                                    Map<TableName, ? extends Collection<String>>  tableCFs)
196      throws ReplicationException {
197    try {
198      if (!peerExists(id)) {
199        throw new IllegalArgumentException("Cannot set peer tableCFs because id=" + id
200            + " does not exist.");
201      }
202      ReplicationPeerConfig rpc = getReplicationPeerConfig(id);
203      if (rpc == null) {
204        throw new ReplicationException("Unable to get tableCFs of the peer with id=" + id);
205      }
206      rpc.setTableCFsMap(tableCFs);
207      ZKUtil.setData(this.zookeeper, getPeerNode(id),
208          ReplicationPeerConfigUtil.toByteArray(rpc));
209      LOG.info("Peer tableCFs with id= " + id + " is now " +
210        ReplicationPeerConfigUtil.convertToString(tableCFs));
211    } catch (KeeperException e) {
212      throw new ReplicationException("Unable to change tableCFs of the peer with id=" + id, e);
213    }
214  }

216  @Override
217  public boolean getStatusOfPeer(String id) {
218    ReplicationPeer replicationPeer = this.peerClusters.get(id);
219    if (replicationPeer == null) {
220      throw new IllegalArgumentException("Peer with id= " + id + " is not cached");
221    }
222    return replicationPeer.getPeerState() == PeerState.ENABLED;

225  @Override
226  public boolean getStatusOfPeerFromBackingStore(String id) throws ReplicationException {
227    try {
228      if (!peerExists(id)) {
229        throw new IllegalArgumentException("peer " + id + " doesn't exist");
230      }
231      String peerStateZNode = getPeerStateNode(id);
232      try {
233        return ReplicationPeerZKImpl.isStateEnabled(ZKUtil.getData(this.zookeeper, peerStateZNode));
234      } catch (KeeperException e) {
235        throw new ReplicationException(e);
236      } catch (DeserializationException e) {
237        throw new ReplicationException(e);
238      }
239    } catch (KeeperException e) {
240      throw new ReplicationException("Unable to get status of the peer with id=" + id +
241          " from backing store", e);
242    } catch (InterruptedException e) {
243      throw new ReplicationException(e);
244    }
245  }

247  @Override
248  public Map<String, ReplicationPeerConfig> getAllPeerConfigs() {
249    Map<String, ReplicationPeerConfig> peers = new TreeMap<>();
250    List<String> ids = null;
251    try {
252      ids = ZKUtil.listChildrenNoWatch(this.zookeeper, this.peersZNode);
253      for (String id : ids) {
254        ReplicationPeerConfig peerConfig = getReplicationPeerConfig(id);
255        if (peerConfig == null) {
256          LOG.warn("Failed to get replication peer configuration of clusterid=" + id
257            + " znode content, continuing.");
258          continue;
259        }
260        peers.put(id, peerConfig);
261      }
262    } catch (KeeperException e) {
263      this.abortable.abort("Cannot get the list of peers ", e);
264    } catch (ReplicationException e) {
265      this.abortable.abort("Cannot get the list of peers ", e);
266    }
267    return peers;
268  }

280  /**
281   * Returns a ReplicationPeerConfig from the znode or null for the given peerId.
282   */
283  @Override
284  public ReplicationPeerConfig getReplicationPeerConfig(String peerId)
285      throws ReplicationException {
286    String znode = getPeerNode(peerId);
287    byte[] data = null;
288    try {
289      data = ZKUtil.getData(this.zookeeper, znode);
290    } catch (InterruptedException e) {
291      LOG.warn("Could not get configuration for peer because the thread " +
292          "was interrupted. peerId=" + peerId);
293      Thread.currentThread().interrupt();
294      return null;
295    } catch (KeeperException e) {
296      throw new ReplicationException("Error getting configuration for peer with id="
297          + peerId, e);
298    }
299    if (data == null) {
300      LOG.error("Could not get configuration for peer because it doesn't exist. peerId=" + peerId);
301      return null;
302    }
303
304    try {
305      return ReplicationPeerConfigUtil.parsePeerFrom(data);
306    } catch (DeserializationException e) {
307      LOG.warn("Failed to parse cluster key from peerId=" + peerId
308          + ", specifically the content from the following znode: " + znode);
309      return null;
310    }
311  }

313  @Override
314  public Pair<ReplicationPeerConfig, Configuration> getPeerConf(String peerId)
315      throws ReplicationException {
316    ReplicationPeerConfig peerConfig = getReplicationPeerConfig(peerId);
317
318    if (peerConfig == null) {
319      return null;
320    }
321
322    Configuration otherConf;
323    try {
324      otherConf = HBaseConfiguration.createClusterConf(this.conf, peerConfig.getClusterKey());
325    } catch (IOException e) {
326      LOG.error("Can't get peer configuration for peerId=" + peerId + " because:", e);
327      return null;
328    }
329
330    if (!peerConfig.getConfiguration().isEmpty()) {
331      CompoundConfiguration compound = new CompoundConfiguration();
332      compound.add(otherConf);
333      compound.addStringMap(peerConfig.getConfiguration());
334      return new Pair<>(peerConfig, compound);
335    }
336
337    return new Pair<>(peerConfig, otherConf);
338  }

340  @Override
341  public void updatePeerConfig(String id, ReplicationPeerConfig newConfig)
342      throws ReplicationException {
343    ReplicationPeer peer = getConnectedPeer(id);
344    if (peer == null){
345      throw new ReplicationException("Could not find peer Id " + id + " in connected peers");
346    }
347    ReplicationPeerConfig existingConfig = peer.getPeerConfig();
348    if (newConfig.getClusterKey() != null && !newConfig.getClusterKey().isEmpty() &&
349        !newConfig.getClusterKey().equals(existingConfig.getClusterKey())){
350      throw new ReplicationException("Changing the cluster key on an existing peer is not allowed."
351          + " Existing key '" + existingConfig.getClusterKey() + "' does not match new key '"
352          + newConfig.getClusterKey() +
353      "'");
354    }
355    String existingEndpointImpl = existingConfig.getReplicationEndpointImpl();
356    if (newConfig.getReplicationEndpointImpl() != null &&
357        !newConfig.getReplicationEndpointImpl().isEmpty() &&
358        !newConfig.getReplicationEndpointImpl().equals(existingEndpointImpl)){
359      throw new ReplicationException("Changing the replication endpoint implementation class " +
360          "on an existing peer is not allowed. Existing class '"
361          + existingConfig.getReplicationEndpointImpl()
362          + "' does not match new class '" + newConfig.getReplicationEndpointImpl() + "'");
363    }
364    //Update existingConfig's peer config and peer data with the new values, but don't touch config
365    // or data that weren't explicitly changed
366    existingConfig.getConfiguration().putAll(newConfig.getConfiguration());
367    existingConfig.getPeerData().putAll(newConfig.getPeerData());
368    existingConfig.setTableCFsMap(newConfig.getTableCFsMap());
369    existingConfig.setNamespaces(newConfig.getNamespaces());
370    existingConfig.setBandwidth(newConfig.getBandwidth());
371
372    try {
373      ZKUtil.setData(this.zookeeper, getPeerNode(id),
374          ReplicationPeerConfigUtil.toByteArray(existingConfig));
375    }
376    catch(KeeperException ke){
377      throw new ReplicationException("There was a problem trying to save changes to the " +
378          "replication peer " + id, ke);
379    }
380  }

396  /**
397   * A private method used during initialization. This method attempts to add all registered
398   * peer clusters. This method does not set a watch on the peer cluster znodes.
399   */
400  private void addExistingPeers() throws ReplicationException {
401    List<String> znodes = null;
402    try {
403      znodes = ZKUtil.listChildrenNoWatch(this.zookeeper, this.peersZNode);
404    } catch (KeeperException e) {
405      throw new ReplicationException("Error getting the list of peer clusters.", e);
406    }
407    if (znodes != null) {
408      for (String z : znodes) {
409        createAndAddPeer(z);
410      }
411    }
412  }

414  @Override
415  public boolean peerConnected(String peerId) throws ReplicationException {
416    return createAndAddPeer(peerId);
417  }

419  @Override
420  public void peerDisconnected(String peerId) {
421    ReplicationPeer rp = this.peerClusters.get(peerId);
422    if (rp != null) {
423      ((ConcurrentMap<String, ReplicationPeerZKImpl>) peerClusters).remove(peerId, rp);
424    }
425  }

427  /**
428   * Attempt to connect to a new remote slave cluster.
429   * @param peerId a short that identifies the cluster
430   * @return true if a new connection was made, false if no new connection was made.
431   */
432  public boolean createAndAddPeer(String peerId) throws ReplicationException {
433    if (peerClusters == null) {
434      return false;
435    }
436    if (this.peerClusters.containsKey(peerId)) {
437      return false;
438    }
439
440    ReplicationPeerZKImpl peer = null;
441    try {
442      peer = createPeer(peerId);
443    } catch (Exception e) {
444      throw new ReplicationException("Error adding peer with id=" + peerId, e);
445    }
446    if (peer == null) {
447      return false;
448    }
449    ReplicationPeerZKImpl previous =
450      ((ConcurrentMap<String, ReplicationPeerZKImpl>) peerClusters).putIfAbsent(peerId, peer);
451    if (previous == null) {
452      LOG.info("Added new peer cluster=" + peer.getPeerConfig().getClusterKey());
453    } else {
454      LOG.info("Peer already present, " + previous.getPeerConfig().getClusterKey() +
455        ", new cluster=" + peer.getPeerConfig().getClusterKey());
456    }
457    return true;
458  }

460  /**
461   * Update the state znode of a peer cluster.
462   * @param id
463   * @param state
464   */
465  private void changePeerState(String id, ReplicationProtos.ReplicationState.State state)
466      throws ReplicationException {
467    try {
468      if (!peerExists(id)) {
469        throw new IllegalArgumentException("Cannot enable/disable peer because id=" + id
470            + " does not exist.");
471      }
472      String peerStateZNode = getPeerStateNode(id);
473      byte[] stateBytes =
474          (state == ReplicationProtos.ReplicationState.State.ENABLED) ? ENABLED_ZNODE_BYTES
475              : DISABLED_ZNODE_BYTES;
476      if (ZKUtil.checkExists(this.zookeeper, peerStateZNode) != -1) {
477        ZKUtil.setData(this.zookeeper, peerStateZNode, stateBytes);
478      } else {
479        ZKUtil.createAndWatch(this.zookeeper, peerStateZNode, stateBytes);
480      }
481      LOG.info("Peer with id= " + id + " is now " + state.name());
482    } catch (KeeperException e) {
483      throw new ReplicationException("Unable to change state of the peer with id=" + id, e);
484    }
485  }

487  /**
488   * Helper method to connect to a peer
489   * @param peerId peer's identifier
490   * @return object representing the peer
491   * @throws ReplicationException
492   */
493  private ReplicationPeerZKImpl createPeer(String peerId) throws ReplicationException {
494    Pair<ReplicationPeerConfig, Configuration> pair = getPeerConf(peerId);
495    if (pair == null) {
496      return null;
497    }
498    Configuration peerConf = pair.getSecond();
499
500    ReplicationPeerZKImpl peer = new ReplicationPeerZKImpl(zookeeper,
501        peerConf, peerId, pair.getFirst(), abortable);
502    try {
503      peer.startStateTracker(this.getPeerStateNode(peerId));
504    } catch (KeeperException e) {
505      throw new ReplicationException("Error starting the peer state tracker for peerId=" +
506          peerId, e);
507    }
508
509    try {
510      peer.startPeerConfigTracker(this.getPeerNode(peerId));
511    } catch (KeeperException e) {
512      throw new ReplicationException("Error starting the peer tableCFs tracker for peerId=" +
513          peerId, e);
514    }
515
516    return peer;
517  }

519  private void checkQueuesDeleted(String peerId) throws ReplicationException {
520    if (queuesClient == null) return;
521    try {
522      List<String> replicators = queuesClient.getListOfReplicators();
523      if (replicators == null || replicators.isEmpty()) {
524        return;
525      }
526      for (String replicator : replicators) {
527        List<String> queueIds = queuesClient.getAllQueues(replicator);
528        for (String queueId : queueIds) {
529          ReplicationQueueInfo queueInfo = new ReplicationQueueInfo(queueId);
530          if (queueInfo.getPeerId().equals(peerId)) {
531            throw new ReplicationException("undeleted queue for peerId: " + peerId
532                + ", replicator: " + replicator + ", queueId: " + queueId);
533          }
534        }
535      }
536      // Check for hfile-refs queue
537      if (-1 != ZKUtil.checkExists(zookeeper, hfileRefsZNode)
538          && queuesClient.getAllPeersFromHFileRefsQueue().contains(peerId)) {
539        throw new ReplicationException("Undeleted queue for peerId: " + peerId
540            + ", found in hfile-refs node path " + hfileRefsZNode);
541      }
542    } catch (KeeperException e) {
543      throw new ReplicationException("Could not check queues deleted with id=" + peerId, e);
544    }
545  }
