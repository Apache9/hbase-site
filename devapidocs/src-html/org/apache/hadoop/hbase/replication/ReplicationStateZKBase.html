019package org.apache.hadoop.hbase.replication;
020
021import java.util.List;
022
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.Abortable;
026import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
027import org.apache.hadoop.hbase.protobuf.generated.ZooKeeperProtos;
028import org.apache.hadoop.hbase.zookeeper.ZKUtil;
029import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
030import org.apache.zookeeper.KeeperException;
031
032
033/**
034 * This is a base class for maintaining replication state in zookeeper.
035 */
036@InterfaceAudience.Private
037public abstract class ReplicationStateZKBase {
038
039  /**
040   * The name of the znode that contains the replication status of a remote slave (i.e. peer)
041   * cluster.
042   */
043  protected final String peerStateNodeName;
044  /** The name of the base znode that contains all replication state. */
045  protected final String replicationZNode;
046  /** The name of the znode that contains a list of all remote slave (i.e. peer) clusters. */
047  protected final String peersZNode;
048  /** The name of the znode that contains all replication queues */
049  protected final String queuesZNode;
050  /** The cluster key of the local cluster */
051  protected final String ourClusterKey;
052  protected final ZooKeeperWatcher zookeeper;
053  protected final Configuration conf;
054  protected final Abortable abortable;
055
056  // Public for testing
057  public static final byte[] ENABLED_ZNODE_BYTES =
058      toByteArray(ZooKeeperProtos.ReplicationState.State.ENABLED);
059  public static final byte[] DISABLED_ZNODE_BYTES =
060      toByteArray(ZooKeeperProtos.ReplicationState.State.DISABLED);
061
062  public ReplicationStateZKBase(ZooKeeperWatcher zookeeper, Configuration conf,
063      Abortable abortable) {
064    this.zookeeper = zookeeper;
065    this.conf = conf;
066    this.abortable = abortable;
067
068    String replicationZNodeName = conf.get("zookeeper.znode.replication", "replication");
069    String peersZNodeName = conf.get("zookeeper.znode.replication.peers", "peers");
070    String queuesZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
071    this.peerStateNodeName = conf.get("zookeeper.znode.replication.peers.state", "peer-state");
072    this.ourClusterKey = ZKUtil.getZooKeeperClusterKey(this.conf);
073    this.replicationZNode = ZKUtil.joinZNode(this.zookeeper.baseZNode, replicationZNodeName);
074    this.peersZNode = ZKUtil.joinZNode(replicationZNode, peersZNodeName);
075    this.queuesZNode = ZKUtil.joinZNode(replicationZNode, queuesZNodeName);
076  }
077
078  public List<String> getListOfReplicators() {
079    List<String> result = null;
080    try {
081      result = ZKUtil.listChildrenNoWatch(this.zookeeper, this.queuesZNode);
082    } catch (KeeperException e) {
083      this.abortable.abort("Failed to get list of replicators", e);
084    }
085    return result;
086  }
087
088  /**
089   * @param state
090   * @return Serialized protobuf of <code>state</code> with pb magic prefix prepended suitable for
091   *         use as content of a peer-state znode under a peer cluster id as in
092   *         /hbase/replication/peers/PEER_ID/peer-state.
093   */
094  protected static byte[] toByteArray(final ZooKeeperProtos.ReplicationState.State state) {
095    byte[] bytes =
096        ZooKeeperProtos.ReplicationState.newBuilder().setState(state).build().toByteArray();
097    return ProtobufUtil.prependPBMagic(bytes);
098  }
099
100  protected boolean peerExists(String id) throws KeeperException {
101    return ZKUtil.checkExists(this.zookeeper, ZKUtil.joinZNode(this.peersZNode, id)) >= 0;
102  }
103
104  /**
105   * Determine if a ZK path points to a peer node.
106   * @param path path to be checked
107   * @return true if the path points to a peer node, otherwise false
108   */
109  protected boolean isPeerPath(String path) {
110    return path.split("/").length == peersZNode.split("/").length + 1;
111  }
112}
