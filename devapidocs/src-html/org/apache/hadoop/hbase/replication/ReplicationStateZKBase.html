019package org.apache.hadoop.hbase.replication;
020
021import java.util.List;
022
023import com.google.common.annotations.VisibleForTesting;
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.Abortable;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
028import org.apache.hadoop.hbase.protobuf.generated.ZooKeeperProtos;
029import org.apache.hadoop.hbase.zookeeper.ZKConfig;
030import org.apache.hadoop.hbase.zookeeper.ZKUtil;
031import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
032import org.apache.zookeeper.KeeperException;
033
034
035/**
036 * This is a base class for maintaining replication state in zookeeper.
037 */
038@InterfaceAudience.Private
039public abstract class ReplicationStateZKBase {
040
041  /**
042   * The name of the znode that contains the replication status of a remote slave (i.e. peer)
043   * cluster.
044   */
045  protected final String peerStateNodeName;
046  /** The name of the base znode that contains all replication state. */
047  protected final String replicationZNode;
048  /** The name of the znode that contains a list of all remote slave (i.e. peer) clusters. */
049  protected final String peersZNode;
050  /** The name of the znode that contains all replication queues */
051  protected final String queuesZNode;
052  /** The name of the znode that contains queues of hfile references to be replicated */
053  protected final String hfileRefsZNode;
054  /** The cluster key of the local cluster */
055  protected final String ourClusterKey;
056  /** The name of the znode that contains tableCFs */
057  protected final String tableCFsNodeName;
058
059  protected final ZooKeeperWatcher zookeeper;
060  protected final Configuration conf;
061  protected final Abortable abortable;
062
063  // Public for testing
064  public static final byte[] ENABLED_ZNODE_BYTES =
065      toByteArray(ZooKeeperProtos.ReplicationState.State.ENABLED);
066  public static final byte[] DISABLED_ZNODE_BYTES =
067      toByteArray(ZooKeeperProtos.ReplicationState.State.DISABLED);
068  public static final String ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY =
069      "zookeeper.znode.replication.hfile.refs";
070  public static final String ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT = "hfile-refs";
071
072  public ReplicationStateZKBase(ZooKeeperWatcher zookeeper, Configuration conf,
073      Abortable abortable) {
074    this.zookeeper = zookeeper;
075    this.conf = conf;
076    this.abortable = abortable;
077
078    String replicationZNodeName = conf.get("zookeeper.znode.replication", "replication");
079    String peersZNodeName = conf.get("zookeeper.znode.replication.peers", "peers");
080    String queuesZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
081    String hfileRefsZNodeName = conf.get(ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY,
082      ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT);
083    this.peerStateNodeName = conf.get("zookeeper.znode.replication.peers.state", "peer-state");
084    this.tableCFsNodeName = conf.get("zookeeper.znode.replication.peers.tableCFs", "tableCFs");
085    this.ourClusterKey = ZKConfig.getZooKeeperClusterKey(this.conf);
086    this.replicationZNode = ZKUtil.joinZNode(this.zookeeper.baseZNode, replicationZNodeName);
087    this.peersZNode = ZKUtil.joinZNode(replicationZNode, peersZNodeName);
088    this.queuesZNode = ZKUtil.joinZNode(replicationZNode, queuesZNodeName);
089    this.hfileRefsZNode = ZKUtil.joinZNode(replicationZNode, hfileRefsZNodeName);
090  }
091
092  public List<String> getListOfReplicators() {
093    List<String> result = null;
094    try {
095      result = ZKUtil.listChildrenNoWatch(this.zookeeper, this.queuesZNode);
096    } catch (KeeperException e) {
097      this.abortable.abort("Failed to get list of replicators", e);
098    }
099    return result;
100  }
101
102  /**
103   * @param state
104   * @return Serialized protobuf of <code>state</code> with pb magic prefix prepended suitable for
105   *         use as content of a peer-state znode under a peer cluster id as in
106   *         /hbase/replication/peers/PEER_ID/peer-state.
107   */
108  protected static byte[] toByteArray(final ZooKeeperProtos.ReplicationState.State state) {
109    byte[] bytes =
110        ZooKeeperProtos.ReplicationState.newBuilder().setState(state).build().toByteArray();
111    return ProtobufUtil.prependPBMagic(bytes);
112  }
113
114  protected boolean peerExists(String id) throws KeeperException {
115    return ZKUtil.checkExists(this.zookeeper, ZKUtil.joinZNode(this.peersZNode, id)) >= 0;
116  }
117
118  /**
119   * Determine if a ZK path points to a peer node.
120   * @param path path to be checked
121   * @return true if the path points to a peer node, otherwise false
122   */
123  protected boolean isPeerPath(String path) {
124    return path.split("/").length == peersZNode.split("/").length + 1;
125  }
126
127  @VisibleForTesting
128  protected String getTableCFsNode(String id) {
129    return ZKUtil.joinZNode(this.peersZNode, ZKUtil.joinZNode(id, this.tableCFsNodeName));
130  }
131
132  @VisibleForTesting
133  protected String getPeerStateNode(String id) {
134    return ZKUtil.joinZNode(this.peersZNode, ZKUtil.joinZNode(id, this.peerStateNodeName));
135  }
136  @VisibleForTesting
137  protected String getPeerNode(String id) {
138    return ZKUtil.joinZNode(this.peersZNode, id);
139  }
140}
