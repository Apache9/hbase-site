019package org.apache.hadoop.hbase.replication;
020
021import java.util.Collection;
022import java.util.List;
023import java.util.Map;
024import java.util.Set;
025
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.util.Pair;
030
031/**
032 * This provides an interface for maintaining a set of peer clusters. These peers are remote slave
033 * clusters that data is replicated to. A peer cluster can be in three different states:
034 *
035 * 1. Not-Registered - There is no notion of the peer cluster.
036 * 2. Registered - The peer has an id and is being tracked but there is no connection.
037 * 3. Connected - There is an active connection to the remote peer.
038 *
039 * In the registered or connected state, a peer cluster can either be enabled or disabled.
040 */
041@InterfaceAudience.Private
042public interface ReplicationPeers {
043
044  /**
045   * Initialize the ReplicationPeers interface.
046   */
047  void init() throws ReplicationException;
048
049  /**
050   * Add a new remote slave cluster for replication.
051   * @param peerId a short that identifies the cluster
052   * @param peerConfig configuration for the replication slave cluster
053   */
054  void addPeer(String peerId, ReplicationPeerConfig peerConfig)
055      throws ReplicationException;
056
057  /**
058   * Removes a remote slave cluster and stops the replication to it.
059   * @param peerId a short that identifies the cluster
060   */
061  void removePeer(String peerId) throws ReplicationException;
062
063  boolean peerAdded(String peerId) throws ReplicationException;
064
065  void peerRemoved(String peerId);
066
067  /**
068   * Restart the replication to the specified remote slave cluster.
069   * @param peerId a short that identifies the cluster
070   */
071  void enablePeer(String peerId) throws ReplicationException;
072
073  /**
074   * Stop the replication to the specified remote slave cluster.
075   * @param peerId a short that identifies the cluster
076   */
077  void disablePeer(String peerId) throws ReplicationException;
078
079  /**
080   * Get the table and column-family list string of the peer from ZK.
081   * @param peerId a short that identifies the cluster
082   */
083  public Map<TableName, List<String>> getPeerTableCFsConfig(String peerId)
084      throws ReplicationException;
085
086  /**
087   * Set the table and column-family list string of the peer to ZK.
088   * @param peerId a short that identifies the cluster
089   * @param tableCFs the table and column-family list which will be replicated for this peer
090   */
091  public void setPeerTableCFsConfig(String peerId,
092                                    Map<TableName, ? extends Collection<String>>  tableCFs)
093      throws ReplicationException;
094
095  /**
096   * Returns the ReplicationPeer
097   * @param peerId id for the peer
098   * @return ReplicationPeer object
099   */
100  ReplicationPeer getPeer(String peerId);
101
102  /**
103   * Returns the set of peerIds defined
104   * @return a Set of Strings for peerIds
105   */
106  public Set<String> getPeerIds();
107
108  /**
109   * Get the replication status for the specified connected remote slave cluster.
110   * The value might be read from cache, so it is recommended to
111   * use {@link #getStatusOfPeerFromBackingStore(String)}
112   * if reading the state after enabling or disabling it.
113   * @param peerId a short that identifies the cluster
114   * @return true if replication is enabled, false otherwise.
115   */
116  boolean getStatusOfPeer(String peerId);
117
118  /**
119   * Get the replication status for the specified remote slave cluster, which doesn't
120   * have to be connected. The state is read directly from the backing store.
121   * @param peerId a short that identifies the cluster
122   * @return true if replication is enabled, false otherwise.
123   * @throws ReplicationException thrown if there's an error contacting the store
124   */
125  boolean getStatusOfPeerFromBackingStore(String peerId) throws ReplicationException;
126
127  /**
128   * List the cluster replication configs of all remote slave clusters (whether they are
129   * enabled/disabled or connected/disconnected).
130   * @return A map of peer ids to peer cluster keys
131   */
132  Map<String, ReplicationPeerConfig> getAllPeerConfigs();
133
134  /**
135   * List the peer ids of all remote slave clusters (whether they are enabled/disabled or
136   * connected/disconnected).
137   * @return A list of peer ids
138   */
139  List<String> getAllPeerIds();
140
141  /**
142   * Returns the configured ReplicationPeerConfig for this peerId
143   * @param peerId a short name that identifies the cluster
144   * @return ReplicationPeerConfig for the peer
145   */
146  ReplicationPeerConfig getReplicationPeerConfig(String peerId) throws ReplicationException;
147
148  /**
149   * Returns the configuration needed to talk to the remote slave cluster.
150   * @param peerId a short that identifies the cluster
151   * @return the configuration for the peer cluster, null if it was unable to get the configuration
152   */
153  Pair<ReplicationPeerConfig, Configuration> getPeerConf(String peerId) throws ReplicationException;
154}
