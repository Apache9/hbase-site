019package org.apache.hadoop.hbase.replication;
020
021import java.util.List;
022import java.util.SortedMap;
023import java.util.SortedSet;
024
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026
027/**
028 * This provides an interface for maintaining a region server's replication queues. These queues
029 * keep track of the WALs that still need to be replicated to remote clusters.
030 */
031@InterfaceAudience.Private
032public interface ReplicationQueues {
033
034  /**
035   * Initialize the region server replication queue interface.
036   * @param serverName The server name of the region server that owns the replication queues this
037   *          interface manages.
038   */
039  void init(String serverName) throws ReplicationException;
040
041  /**
042   * Remove a replication queue.
043   * @param queueId a String that identifies the queue.
044   */
045  void removeQueue(String queueId);
046
047  /**
048   * Add a new WAL file to the given queue. If the queue does not exist it is created.
049   * @param queueId a String that identifies the queue.
050   * @param filename name of the WAL
051   */
052  void addLog(String queueId, String filename) throws ReplicationException;
053
054  /**
055   * Remove an WAL file from the given queue.
056   * @param queueId a String that identifies the queue.
057   * @param filename name of the WAL
058   */
059  void removeLog(String queueId, String filename);
060
061  /**
062   * Set the current position for a specific WAL in a given queue.
063   * @param queueId a String that identifies the queue
064   * @param filename name of the WAL
065   * @param position the current position in the file
066   */
067  void setLogPosition(String queueId, String filename, long position);
068
069  /**
070   * Get the current position for a specific WAL in a given queue.
071   * @param queueId a String that identifies the queue
072   * @param filename name of the WAL
073   * @return the current position in the file
074   */
075  long getLogPosition(String queueId, String filename) throws ReplicationException;
076
077  /**
078   * Remove all replication queues for this region server.
079   */
080  void removeAllQueues();
081
082  /**
083   * Get a list of all WALs in the given queue.
084   * @param queueId a String that identifies the queue
085   * @return a list of WALs, null if this region server is dead and has no outstanding queues
086   */
087  List<String> getLogsInQueue(String queueId);
088
089  /**
090   * Get a list of all queues for this region server.
091   * @return a list of queueIds, null if this region server is dead and has no outstanding queues
092   */
093  List<String> getAllQueues();
094
095  /**
096   * Take ownership for the set of queues belonging to a dead region server.
097   * @param regionserver the id of the dead region server
098   * @return A SortedMap of the queues that have been claimed, including a SortedSet of WALs in
099   *         each queue. Returns an empty map if no queues were failed-over.
100   */
101  SortedMap<String, SortedSet<String>> claimQueues(String regionserver);
102
103  /**
104   * Get a list of all region servers that have outstanding replication queues. These servers could
105   * be alive, dead or from a previous run of the cluster.
106   * @return a list of server names
107   */
108  List<String> getListOfReplicators();
109
110  /**
111   * Checks if the provided znode is the same as this region server's
112   * @param znode to check
113   * @return if this is this rs's znode
114   */
115  boolean isThisOurZnode(String znode);
116}
