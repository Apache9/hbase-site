041/**
042 * Implementation of a file cleaner that checks if a hfile is still scheduled for replication before
043 * deleting it from hfile archive directory.
044 */
045@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
046public class ReplicationHFileCleaner extends BaseHFileCleanerDelegate {
047  private static final Log LOG = LogFactory.getLog(ReplicationHFileCleaner.class);
048  private ZooKeeperWatcher zkw;
049  private ReplicationQueuesClient rqc;
050  private boolean stopped = false;
051
052  @Override
053  public Iterable<FileStatus> getDeletableFiles(Iterable<FileStatus> files) {
054    // all members of this class are null if replication is disabled,
055    // so we cannot filter the files
056    if (this.getConf() == null) {
057      return files;
058    }
059
060    final Set<String> hfileRefs;
061    try {
062      // The concurrently created new hfile entries in ZK may not be included in the return list,
063      // but they won't be deleted because they're not in the checking set.
064      hfileRefs = loadHFileRefsFromPeers();
065    } catch (KeeperException e) {
066      LOG.warn("Failed to read hfile references from zookeeper, skipping checking deletable files");
067      return Collections.emptyList();
068    }
069    return Iterables.filter(files, new Predicate<FileStatus>() {
070      @Override
071      public boolean apply(FileStatus file) {
072        String hfile = file.getPath().getName();
073        boolean foundHFileRefInQueue = hfileRefs.contains(hfile);
074        if (LOG.isDebugEnabled()) {
075          if (foundHFileRefInQueue) {
076            LOG.debug("Found hfile reference in ZK, keeping: " + hfile);
077          } else {
078            LOG.debug("Did not find hfile reference in ZK, deleting: " + hfile);
079          }
080        }
081        return !foundHFileRefInQueue;
082      }
083    });
084  }
085
086  /**
087   * Load all hfile references in all replication queues from ZK. This method guarantees to return a
088   * snapshot which contains all hfile references in the zookeeper at the start of this call.
089   * However, some newly created hfile references during the call may not be included.
090   */
091  private Set<String> loadHFileRefsFromPeers() throws KeeperException {
092    Set<String> hfileRefs = Sets.newHashSet();
093    List<String> listOfPeers;
094    for (int retry = 0;; retry++) {
095      int v0 = rqc.getHFileRefsNodeChangeVersion();
096      hfileRefs.clear();
097      listOfPeers = rqc.getAllPeersFromHFileRefsQueue();
098      if (listOfPeers == null) {
099        LOG.debug("Didn't find any peers with hfile references, won't prevent any deletions.");
100        return ImmutableSet.of();
101      }
102      for (String id : listOfPeers) {
103        List<String> peerHFileRefs = rqc.getReplicableHFiles(id);
104        if (peerHFileRefs != null) {
105          hfileRefs.addAll(peerHFileRefs);
106        }
107      }
108      int v1 = rqc.getHFileRefsNodeChangeVersion();
109      if (v0 == v1) {
110        return hfileRefs;
111      }
112      LOG.debug(String.format("Replication hfile references node cversion changed from "
113          + "%d to %d, retry = %d", v0, v1, retry));
114    }
115  }
116
117  @Override
118  public void setConf(Configuration config) {
119    // If either replication or replication of bulk load hfiles is disabled, keep all members null
120    if (!(config.getBoolean(
121      HConstants.REPLICATION_BULKLOAD_ENABLE_KEY,
122      HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT))) {
123      LOG.warn(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY
124          + " is not enabled. Better to remove "
125          + ReplicationHFileCleaner.class + " from " + HFileCleaner.MASTER_HFILE_CLEANER_PLUGINS
126          + " configuration.");
127      return;
128    }
129    // Make my own Configuration. Then I'll have my own connection to zk that
130    // I can close myself when time comes.
131    Configuration conf = new Configuration(config);
132    try {
133      setConf(conf, new ZooKeeperWatcher(conf, "replicationHFileCleaner", null));
134    } catch (IOException e) {
135      LOG.error("Error while configuring " + this.getClass().getName(), e);
136    }
137  }
138
139  @VisibleForTesting
140  public void setConf(Configuration conf, ZooKeeperWatcher zk) {
141    super.setConf(conf);
142    try {
143      initReplicationQueuesClient(conf, zk);
144    } catch (Exception e) {
145      LOG.error("Error while configuring " + this.getClass().getName(), e);
146    }
147  }
148
149  private void initReplicationQueuesClient(Configuration conf, ZooKeeperWatcher zk)
150      throws Exception {
151    this.zkw = zk;
152    this.rqc = ReplicationFactory.getReplicationQueuesClient(new ReplicationQueuesClientArguments(
153        conf, new WarnOnlyAbortable(), zkw));
154  }
155
156  @Override
157  public void stop(String why) {
158    if (this.stopped) {
159      return;
160    }
161    this.stopped = true;
162    if (this.zkw != null) {
163      LOG.info("Stopping " + this.zkw);
164      this.zkw.close();
165    }
166  }
167
168  @Override
169  public boolean isStopped() {
170    return this.stopped;
171  }
172
173  @Override
174  public boolean isFileDeletable(FileStatus fStat) {
175    Set<String> hfileRefsFromQueue;
176    // all members of this class are null if replication is disabled,
177    // so do not stop from deleting the file
178    if (getConf() == null) {
179      return true;
180    }
181
182    try {
183      hfileRefsFromQueue = loadHFileRefsFromPeers();
184    } catch (KeeperException e) {
185      LOG.warn("Failed to read hfile references from zookeeper, skipping checking deletable "
186          + "file for " + fStat.getPath());
187      return false;
188    }
189    return !hfileRefsFromQueue.contains(fStat.getPath().getName());
190  }
191
192  private static class WarnOnlyAbortable implements Abortable {
193    @Override
194    public void abort(String why, Throwable e) {
195      LOG.warn("ReplicationHFileCleaner received abort, ignoring.  Reason: " + why);
196      if (LOG.isDebugEnabled()) {
197        LOG.debug(e);
198      }
199    }
200
201    @Override
202    public boolean isAborted() {
203      return false;
204    }
205  }
