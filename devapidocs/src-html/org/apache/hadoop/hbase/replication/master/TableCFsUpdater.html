019package org.apache.hadoop.hbase.replication.master;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.hbase.Abortable;
025import org.apache.hadoop.hbase.HBaseConfiguration;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027import org.apache.hadoop.hbase.classification.InterfaceStability;
028import org.apache.hadoop.hbase.client.replication.ReplicationSerDeHelper;
029import org.apache.hadoop.hbase.exceptions.DeserializationException;
030import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos;
031import org.apache.hadoop.hbase.replication.ReplicationPeerConfig;
032import org.apache.hadoop.hbase.replication.ReplicationStateZKBase;
033import org.apache.hadoop.hbase.zookeeper.ZKUtil;
034import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
035import org.apache.zookeeper.KeeperException;
036
037import java.io.IOException;
038import java.util.List;
039
040/**
041 * This class is used to upgrade TableCFs from HBase 1.0, 1.1, 1.2, 1.3 to HBase 1.4 or 2.x.
042 * It will be removed in HBase 3.x. See HBASE-11393
043 */
044@InterfaceAudience.Private
045@InterfaceStability.Unstable
046public class TableCFsUpdater extends ReplicationStateZKBase {
047
048  private static final Log LOG = LogFactory.getLog(TableCFsUpdater.class);
049
050  public TableCFsUpdater(ZooKeeperWatcher zookeeper,
051                         Configuration conf, Abortable abortable) {
052    super(zookeeper, conf, abortable);
053  }
054
055  public void update() {
056    List<String> znodes = null;
057    try {
058      znodes = ZKUtil.listChildrenNoWatch(this.zookeeper, this.peersZNode);
059    } catch (KeeperException e) {
060      LOG.error("Failed to get peers znode", e);
061    }
062    if (znodes != null) {
063      for (String peerId : znodes) {
064        if (!update(peerId)) {
065          LOG.error("upgrade tableCFs failed for peerId=" + peerId);
066        }
067      }
068    }
069  }
070
071  public boolean update(String peerId) {
072    String tableCFsNode = getTableCFsNode(peerId);
073    try {
074      if (ZKUtil.checkExists(zookeeper, tableCFsNode) != -1) {
075        String peerNode = getPeerNode(peerId);
076        ReplicationPeerConfig rpc = getReplicationPeerConig(peerNode);
077        // We only need to copy data from tableCFs node to rpc Node the first time hmaster start.
078        if (rpc.getTableCFsMap() == null || rpc.getTableCFsMap().size() == 0) {
079          // we copy TableCFs node into PeerNode
080          LOG.info("copy tableCFs into peerNode:" + peerId);
081          ZooKeeperProtos.TableCF[] tableCFs =
082                  ReplicationSerDeHelper.parseTableCFs(
083                          ZKUtil.getData(this.zookeeper, tableCFsNode));
084          if (tableCFs != null && tableCFs.length > 0) {
085            rpc.setTableCFsMap(ReplicationSerDeHelper.convert2Map(tableCFs));
086            ZKUtil.setData(this.zookeeper, peerNode,
087              ReplicationSerDeHelper.toByteArray(rpc));
088          }
089        } else {
090          LOG.info("No tableCFs in peerNode:" + peerId);
091        }
092      }
093    } catch (KeeperException e) {
094      LOG.warn("NOTICE!! Update peerId failed, peerId=" + peerId, e);
095      return false;
096    } catch (InterruptedException e) {
097      LOG.warn("NOTICE!! Update peerId failed, peerId=" + peerId, e);
098      return false;
099    } catch (IOException e) {
100      LOG.warn("NOTICE!! Update peerId failed, peerId=" + peerId, e);
101      return false;
102    }
103    return true;
104  }
105
106  private ReplicationPeerConfig getReplicationPeerConig(String peerNode)
107          throws KeeperException, InterruptedException {
108    byte[] data = null;
109    data = ZKUtil.getData(this.zookeeper, peerNode);
110    if (data == null) {
111      LOG.error("Could not get configuration for " +
112              "peer because it doesn't exist. peer=" + peerNode);
113      return null;
114    }
115    try {
116      return ReplicationSerDeHelper.parsePeerFrom(data);
117    } catch (DeserializationException e) {
118      LOG.warn("Failed to parse cluster key from peer=" + peerNode);
119      return null;
120    }
121  }
122
123  private static void printUsageAndExit() {
124    System.err.printf("Usage: bin/hbase org.apache.hadoop.hbase.replication.master.TableCFsUpdater [options]");
125    System.err.println(" where [options] are:");
126    System.err.println("  -h|-help    Show this help and exit.");
127    System.err.println("  update      Copy table-cfs to replication peer config");
128    System.err.println();
129    System.exit(1);
130  }
131
132  public static void main(String[] args) throws Exception {
133    if (args.length != 1) {
134      printUsageAndExit();
135    }
136    if (args[0].equals("-help") || args[0].equals("-h")) {
137      printUsageAndExit();
138    } else if (args[0].equals("update")) {
139      Configuration conf = HBaseConfiguration.create();
140      ZooKeeperWatcher zkw = new ZooKeeperWatcher(conf, "TableCFsUpdater", null);
141      try {
142        TableCFsUpdater tableCFsUpdater = new TableCFsUpdater(zkw, conf, null);
143        tableCFsUpdater.update();
144      } finally {
145        zkw.close();
146      }
147    } else {
148      printUsageAndExit();
149    }
150  }
151}
