097  @Override
098  public void removeQueue(String queueId) {
099    try {
100      byte[] rowKey = queueIdToRowKey(queueId);
101      if (checkQueueExists(queueId)) {
102        Delete deleteQueue = new Delete(rowKey);
103        safeQueueUpdate(deleteQueue);
104      } else {
105        LOG.info("No logs were registered for queue id=" + queueId + " so no rows were removed " +
106            "from the replication table while removing the queue");
107      }
108    } catch (IOException | ReplicationException e) {
109      String errMsg = "Failed removing queue queueId=" + queueId;
110      abortable.abort(errMsg, e);
111    }
112  }

114  @Override
115  public void addLog(String queueId, String filename) throws ReplicationException {
116    try (Table replicationTable = getOrBlockOnReplicationTable()) {
117      if (!checkQueueExists(queueId)) {
118        // Each queue will have an Owner, OwnerHistory, and a collection of [WAL:offset] key values
119        Put putNewQueue = new Put(Bytes.toBytes(buildQueueRowKey(queueId)));
120        putNewQueue.addColumn(CF_QUEUE, COL_QUEUE_OWNER, serverNameBytes);
121        putNewQueue.addColumn(CF_QUEUE, COL_QUEUE_OWNER_HISTORY, EMPTY_STRING_BYTES);
122        putNewQueue.addColumn(CF_QUEUE, Bytes.toBytes(filename), INITIAL_OFFSET_BYTES);
123        replicationTable.put(putNewQueue);
124      } else {
125        // Otherwise simply add the new log and offset as a new column
126        Put putNewLog = new Put(queueIdToRowKey(queueId));
127        putNewLog.addColumn(CF_QUEUE, Bytes.toBytes(filename), INITIAL_OFFSET_BYTES);
128        safeQueueUpdate(putNewLog);
129      }
130    } catch (IOException | ReplicationException e) {
131      String errMsg = "Failed adding log queueId=" + queueId + " filename=" + filename;
132      abortable.abort(errMsg, e);
133    }
134  }

136  @Override
137  public void removeLog(String queueId, String filename) {
138    try {
139      byte[] rowKey = queueIdToRowKey(queueId);
140      Delete delete = new Delete(rowKey);
141      delete.addColumns(CF_QUEUE, Bytes.toBytes(filename));
142      safeQueueUpdate(delete);
143    } catch (IOException | ReplicationException e) {
144      String errMsg = "Failed removing log queueId=" + queueId + " filename=" + filename;
145      abortable.abort(errMsg, e);
146    }
147  }

149  @Override
150  public void setLogPosition(String queueId, String filename, long position) {
151    try (Table replicationTable = getOrBlockOnReplicationTable()) {
152      byte[] rowKey = queueIdToRowKey(queueId);
153      // Check that the log exists. addLog() must have been called before setLogPosition().
154      Get checkLogExists = new Get(rowKey);
155      checkLogExists.addColumn(CF_QUEUE, Bytes.toBytes(filename));
156      if (!replicationTable.exists(checkLogExists)) {
157        String errMsg = "Could not set position of non-existent log from queueId=" + queueId +
158          ", filename=" + filename;
159        abortable.abort(errMsg, new ReplicationException(errMsg));
160        return;
161      }
162      // Update the log offset if it exists
163      Put walAndOffset = new Put(rowKey);
164      walAndOffset.addColumn(CF_QUEUE, Bytes.toBytes(filename), Bytes.toBytes(position));
165      safeQueueUpdate(walAndOffset);
166    } catch (IOException | ReplicationException e) {
167      String errMsg = "Failed writing log position queueId=" + queueId + "filename=" +
168        filename + " position=" + position;
169      abortable.abort(errMsg, e);
170    }
171  }

173  @Override
174  public long getLogPosition(String queueId, String filename) throws ReplicationException {
175    try {
176      byte[] rowKey = queueIdToRowKey(queueId);
177      Get getOffset = new Get(rowKey);
178      getOffset.addColumn(CF_QUEUE, Bytes.toBytes(filename));
179      Result result = getResultIfOwner(getOffset);
180      if (result == null || !result.containsColumn(CF_QUEUE, Bytes.toBytes(filename))) {
181        throw new ReplicationException("Could not read empty result while getting log position " +
182          "queueId=" + queueId + ", filename=" + filename);
183      }
184      return Bytes.toLong(result.getValue(CF_QUEUE, Bytes.toBytes(filename)));
185    } catch (IOException e) {
186      throw new ReplicationException("Could not get position in log for queueId=" + queueId +
187        ", filename=" + filename);
188    }
189  }

199  @Override
200  public List<String> getLogsInQueue(String queueId) {
201    String errMsg = "Failed getting logs in queue queueId=" + queueId;
202    byte[] rowKey = queueIdToRowKey(queueId);
203    List<String> logs = new ArrayList<>();
204    try {
205      Get getQueue = new Get(rowKey);
206      Result queue = getResultIfOwner(getQueue);
207      if (queue == null || queue.isEmpty()) {
208        String errMsgLostOwnership = "Failed getting logs for queue queueId=" +
209            Bytes.toString(rowKey) + " because the queue was missing or we lost ownership";
210        abortable.abort(errMsg, new ReplicationException(errMsgLostOwnership));
211        return null;
212      }
213      Map<byte[], byte[]> familyMap = queue.getFamilyMap(CF_QUEUE);
214      for(byte[] cQualifier : familyMap.keySet()) {
215        if (Arrays.equals(cQualifier, COL_QUEUE_OWNER) || Arrays.equals(cQualifier,
216            COL_QUEUE_OWNER_HISTORY)) {
217          continue;
218        }
219        logs.add(Bytes.toString(cQualifier));
220      }
221    } catch (IOException e) {
222      abortable.abort(errMsg, e);
223      return null;
224    }
225    return logs;
226  }

233  @Override public List<String> getUnClaimedQueueIds(String regionserver) {
234    if (isThisOurRegionServer(regionserver)) {
235      return null;
236    }
237    try (ResultScanner queuesToClaim = getQueuesBelongingToServer(regionserver)) {
238      List<String> res = new ArrayList<>();
239      for (Result queue : queuesToClaim) {
240        String rowKey = Bytes.toString(queue.getRow());
241        res.add(rowKey);
242      }
243      return res.isEmpty() ? null : res;
244    } catch (IOException e) {
245      String errMsg = "Failed getUnClaimedQueueIds";
246      abortable.abort(errMsg, e);
247    }
248    return null;
249  }

255  @Override
256  public Pair<String, SortedSet<String>> claimQueue(String regionserver, String queueId) {
257    if (isThisOurRegionServer(regionserver)) {
258      return null;
259    }
260
261    try (ResultScanner queuesToClaim = getQueuesBelongingToServer(regionserver)){
262      for (Result queue : queuesToClaim) {
263        String rowKey = Bytes.toString(queue.getRow());
264        if (!rowKey.equals(queueId)){
265          continue;
266        }
267        if (attemptToClaimQueue(queue, regionserver)) {
268          ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(rowKey);
269          if (replicationState.peerExists(replicationQueueInfo.getPeerId())) {
270            SortedSet<String> sortedLogs = new TreeSet<>();
271            List<String> logs = getLogsInQueue(queue.getRow());
272            for (String log : logs) {
273              sortedLogs.add(log);
274            }
275            LOG.info(serverName + " has claimed queue " + rowKey + " from " + regionserver);
276            return new Pair<>(rowKey, sortedLogs);
277          } else {
278            // Delete orphaned queues
279            removeQueue(Bytes.toString(queue.getRow()));
280            LOG.info(serverName + " has deleted abandoned queue " + queueId + " from " +
281              regionserver);
282          }
283        }
284      }
285    } catch (IOException | KeeperException e) {
286      String errMsg = "Failed claiming queues for regionserver=" + regionserver;
287      abortable.abort(errMsg, e);
288    }
289    return null;
290  }

335  /**
336   * See safeQueueUpdate(RowMutations mutate)
337   *
338   * @param put Row mutation to perform on the queue
339   */
340  private void safeQueueUpdate(Put put) throws ReplicationException, IOException {
341    RowMutations mutations = new RowMutations(put.getRow());
342    mutations.add(put);
343    safeQueueUpdate(mutations);
344  }

346  /**
347   * See safeQueueUpdate(RowMutations mutate)
348   *
349   * @param delete Row mutation to perform on the queue
350   */
351  private void safeQueueUpdate(Delete delete) throws ReplicationException,
352    IOException{
353    RowMutations mutations = new RowMutations(delete.getRow());
354    mutations.add(delete);
355    safeQueueUpdate(mutations);
356  }

358  /**
359   * Attempt to mutate a given queue in the Replication Table with a checkAndPut on the OWNER column
360   * of the queue. Abort the server if this checkAndPut fails: which means we have somehow lost
361   * ownership of the column or an IO Exception has occurred during the transaction.
362   *
363   * @param mutate Mutation to perform on a given queue
364   */
365  private void safeQueueUpdate(RowMutations mutate) throws ReplicationException, IOException{
366    try (Table replicationTable = getOrBlockOnReplicationTable()) {
367      boolean updateSuccess = replicationTable.checkAndMutate(mutate.getRow(),
368          CF_QUEUE, COL_QUEUE_OWNER, CompareOperator.EQUAL, serverNameBytes, mutate);
369      if (!updateSuccess) {
370        throw new ReplicationException("Failed to update Replication Table because we lost queue " +
371            " ownership");
372      }
373    }
374  }

376  /**
377   * Check if the queue specified by queueId is stored in HBase
378   *
379   * @param queueId Either raw or reclaimed format of the queueId
380   * @return Whether the queue is stored in HBase
381   * @throws IOException
382   */
383  private boolean checkQueueExists(String queueId) throws IOException {
384    try (Table replicationTable = getOrBlockOnReplicationTable()) {
385      byte[] rowKey = queueIdToRowKey(queueId);
386      return replicationTable.exists(new Get(rowKey));
387    }
388  }

390  /**
391   * Attempt to claim the given queue with a checkAndPut on the OWNER column. We check that the
392   * recently killed server is still the OWNER before we claim it.
393   *
394   * @param queue The queue that we are trying to claim
395   * @param originalServer The server that originally owned the queue
396   * @return Whether we successfully claimed the queue
397   * @throws IOException
398   */
399  private boolean attemptToClaimQueue (Result queue, String originalServer) throws IOException{
400    Put putQueueNameAndHistory = new Put(queue.getRow());
401    putQueueNameAndHistory.addColumn(CF_QUEUE, COL_QUEUE_OWNER, Bytes.toBytes(serverName));
402    String newOwnerHistory = buildClaimedQueueHistory(Bytes.toString(queue.getValue(CF_QUEUE,
403      COL_QUEUE_OWNER_HISTORY)), originalServer);
404    putQueueNameAndHistory.addColumn(CF_QUEUE, COL_QUEUE_OWNER_HISTORY,
405        Bytes.toBytes(newOwnerHistory));
406    RowMutations claimAndRenameQueue = new RowMutations(queue.getRow());
407    claimAndRenameQueue.add(putQueueNameAndHistory);
408    // Attempt to claim ownership for this queue by checking if the current OWNER is the original
409    // server. If it is not then another RS has already claimed it. If it is we set ourselves as the
410    // new owner and update the queue's history
411    try (Table replicationTable = getOrBlockOnReplicationTable()) {
412      boolean success = replicationTable.checkAndMutate(queue.getRow(),
413          CF_QUEUE, COL_QUEUE_OWNER, CompareOperator.EQUAL, Bytes.toBytes(originalServer),
414          claimAndRenameQueue);
415      return success;
416    }
417  }

419  /**
420   * Attempts to run a Get on some queue. Will only return a non-null result if we currently own
421   * the queue.
422   *
423   * @param get The Get that we want to query
424   * @return The result of the Get if this server is the owner of the queue. Else it returns null.
425   * @throws IOException
426   */
427  private Result getResultIfOwner(Get get) throws IOException {
428    Scan scan = new Scan(get);
429    // Check if the Get currently contains all columns or only specific columns
430    if (scan.getFamilyMap().size() > 0) {
431      // Add the OWNER column if the scan is already only over specific columns
432      scan.addColumn(CF_QUEUE, COL_QUEUE_OWNER);
433    }
434    scan.setMaxResultSize(1);
435    SingleColumnValueFilter checkOwner = new SingleColumnValueFilter(CF_QUEUE, COL_QUEUE_OWNER,
436    CompareOperator.EQUAL, serverNameBytes);
437    scan.setFilter(checkOwner);
438    ResultScanner scanner = null;
439    try (Table replicationTable = getOrBlockOnReplicationTable()) {
440      scanner = replicationTable.getScanner(scan);
441      Result result = scanner.next();
442      return (result == null || result.isEmpty()) ? null : result;
443    } finally {
444      if (scanner != null) {
445        scanner.close();
446      }
447    }
448  }
