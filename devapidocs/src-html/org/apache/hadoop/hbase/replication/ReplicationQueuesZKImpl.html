043/**
044 * This class provides an implementation of the
045 * interface using ZooKeeper. The
046 * base znode that this class works at is the myQueuesZnode. The myQueuesZnode contains a list of
047 * all outstanding WAL files on this region server that need to be replicated. The myQueuesZnode is
048 * the regionserver name (a concatenation of the region server?s hostname, client port and start
049 * code). For example:
050 *
051 * /hbase/replication/rs/hostname.example.org,6020,1234
052 *
053 * Within this znode, the region server maintains a set of WAL replication queues. These queues are
054 * represented by child znodes named using there give queue id. For example:
055 *
056 * /hbase/replication/rs/hostname.example.org,6020,1234/1
057 * /hbase/replication/rs/hostname.example.org,6020,1234/2
058 *
059 * Each queue has one child znode for every WAL that still needs to be replicated. The value of
060 * these WAL child znodes is the latest position that has been replicated. This position is updated
061 * every time a WAL entry is replicated. For example:
062 *
063 * /hbase/replication/rs/hostname.example.org,6020,1234/1/23522342.23422 [VALUE: 254]
064 */
065@InterfaceAudience.Private
066public class ReplicationQueuesZKImpl extends ReplicationStateZKBase implements ReplicationQueues {

069  private String myQueuesZnode;
070  /** Name of znode we use to lock during failover */
071  private final static String RS_LOCK_ZNODE = "lock";
072

084  @Override
085  public void init(String serverName) throws ReplicationException {
086    this.myQueuesZnode = ZKUtil.joinZNode(this.queuesZNode, serverName);
087    try {
088      ZKUtil.createWithParents(this.zookeeper, this.myQueuesZnode);
089    } catch (KeeperException e) {
090      throw new ReplicationException("Could not initialize replication queues.", e);
091    }
092    if (conf.getBoolean(HConstants.REPLICATION_BULKLOAD_ENABLE_KEY,
093      HConstants.REPLICATION_BULKLOAD_ENABLE_DEFAULT)) {
094      try {
095        ZKUtil.createWithParents(this.zookeeper, this.hfileRefsZNode);
096      } catch (KeeperException e) {
097        throw new ReplicationException("Could not initialize hfile references replication queue.",
098            e);
099      }
100    }
101  }

150  @Override
151  public long getLogPosition(String queueId, String filename) throws ReplicationException {
152    String clusterZnode = ZKUtil.joinZNode(this.myQueuesZnode, queueId);
153    String znode = ZKUtil.joinZNode(clusterZnode, filename);
154    byte[] bytes = null;
155    try {
156      bytes = ZKUtil.getData(this.zookeeper, znode);
157    } catch (KeeperException e) {
158      throw new ReplicationException("Internal Error: could not get position in log for queueId="
159          + queueId + ", filename=" + filename, e);
160    } catch (InterruptedException e) {
161      Thread.currentThread().interrupt();
162      return 0;
163    }
164    try {
165      return ZKUtil.parseWALPositionFrom(bytes);
166    } catch (DeserializationException de) {
167      LOG.warn("Failed to parse WALPosition for queueId=" + queueId + " and wal=" + filename
168          + " znode content, continuing.");
169    }
170    // if we can not parse the position, start at the beginning of the wal file
171    // again
172    return 0;
173  }

175  @Override
176  public boolean isThisOurRegionServer(String regionserver) {
177    return ZKUtil.joinZNode(this.queuesZNode, regionserver).equals(this.myQueuesZnode);
178  }

180  @Override
181  public List<String> getUnClaimedQueueIds(String regionserver) {
182    if (isThisOurRegionServer(regionserver)) {
183      return null;
184    }
185    String rsZnodePath = ZKUtil.joinZNode(this.queuesZNode, regionserver);
186    List<String> queues = null;
187    try {
188      queues = ZKUtil.listChildrenNoWatch(this.zookeeper, rsZnodePath);
189    } catch (KeeperException e) {
190      this.abortable.abort("Failed to getUnClaimedQueueIds for RS" + regionserver, e);
191    }
192    if (queues != null) {
193      queues.remove(RS_LOCK_ZNODE);
194    }
195    return queues;
196  }

198  @Override
199  public Pair<String, SortedSet<String>> claimQueue(String regionserver, String queueId) {
200    if (conf.getBoolean(HConstants.ZOOKEEPER_USEMULTI, true)) {
201      LOG.info("Atomically moving " + regionserver + "/" + queueId + "'s WALs to my queue");
202      return moveQueueUsingMulti(regionserver, queueId);
203    } else {
204      LOG.info("Moving " + regionserver + "/" + queueId + "'s wals to my queue");
205      if (!lockOtherRS(regionserver)) {
206        LOG.info("Can not take the lock now");
207        return null;
208      }
209      Pair<String, SortedSet<String>> newQueues;
210      try {
211        newQueues = copyQueueFromLockedRS(regionserver, queueId);
212        removeQueueFromLockedRS(regionserver, queueId);
213      } finally {
214        unlockOtherRS(regionserver);
215      }
216      return newQueues;
217    }
218  }

220  private void removeQueueFromLockedRS(String znode, String peerId) {
221    String nodePath = ZKUtil.joinZNode(this.queuesZNode, znode);
222    String peerPath = ZKUtil.joinZNode(nodePath, peerId);
223    try {
224      ZKUtil.deleteNodeRecursively(this.zookeeper, peerPath);
225    } catch (KeeperException e) {
226      LOG.warn("Remove copied queue failed", e);
227    }
228  }

245    try {
246      ZKUtil.deleteNodeRecursively(this.zookeeper, this.myQueuesZnode);
247    } catch (KeeperException e) {
248      // if the znode is already expired, don't bother going further
249      if (e instanceof KeeperException.SessionExpiredException) {

253          + this.myQueuesZnode, e);

281  /**
282   * Try to set a lock in another region server's znode.
283   * @param znode the server names of the other server
284   * @return true if the lock was acquired, false in every other cases
285   */
286  @VisibleForTesting
287  public boolean lockOtherRS(String znode) {
288    try {
289      String parent = ZKUtil.joinZNode(this.queuesZNode, znode);
290      if (parent.equals(this.myQueuesZnode)) {
291        LOG.warn("Won't lock because this is us, we're dead!");
292        return false;
293      }
294      String p = ZKUtil.joinZNode(parent, RS_LOCK_ZNODE);
295      ZKUtil.createAndWatch(this.zookeeper, p, lockToByteArray(this.myQueuesZnode));
296    } catch (KeeperException e) {
297      // This exception will pop up if the znode under which we're trying to
298      // create the lock is already deleted by another region server, meaning
299      // that the transfer already occurred.
300      // NoNode => transfer is done and znodes are already deleted
301      // NodeExists => lock znode already created by another RS
302      if (e instanceof KeeperException.NoNodeException
303          || e instanceof KeeperException.NodeExistsException) {
304        LOG.info("Won't transfer the queue," + " another RS took care of it because of: "
305            + e.getMessage());
306      } else {
307        LOG.info("Failed lock other rs", e);
308      }
309      return false;
310    }
311    return true;
312  }

314  public String getLockZNode(String znode) {
315    return this.queuesZNode + "/" + znode + "/" + RS_LOCK_ZNODE;
316  }

318  @VisibleForTesting

323  private void unlockOtherRS(String znode){
324    String parent = ZKUtil.joinZNode(this.queuesZNode, znode);
325    String p = ZKUtil.joinZNode(parent, RS_LOCK_ZNODE);
326    try {
327      ZKUtil.deleteNode(this.zookeeper, p);
328    } catch (KeeperException e) {
329      this.abortable.abort("Remove lock failed", e);
330    }
331  }

333  /**
334   * It "atomically" copies all the wals queues from another region server and returns them all
335   * sorted per peer cluster (appended with the dead server's znode).
336   * @param znode pertaining to the region server to copy the queues from
337   */
338  private Pair<String, SortedSet<String>> moveQueueUsingMulti(String znode, String peerId) {
339    try {
340      // hbase/replication/rs/deadrs
341      String deadRSZnodePath = ZKUtil.joinZNode(this.queuesZNode, znode);
342      List<ZKUtilOp> listOfOps = new ArrayList<>();
343      ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(peerId);
344      if (!peerExists(replicationQueueInfo.getPeerId())) {
345        // the orphaned queues must be moved, otherwise the delete op of dead rs will fail,
346        // this will cause the whole multi op fail.
347        // NodeFailoverWorker will skip the orphaned queues.
348        LOG.warn("Peer " + peerId +
349            " didn't exist, will move its queue to avoid the failure of multi op");
350      }
351      String newPeerId = peerId + "-" + znode;
352      String newPeerZnode = ZKUtil.joinZNode(this.myQueuesZnode, newPeerId);
353      // check the logs queue for the old peer cluster
354      String oldClusterZnode = ZKUtil.joinZNode(deadRSZnodePath, peerId);
355      List<String> wals = ZKUtil.listChildrenNoWatch(this.zookeeper, oldClusterZnode);
356      SortedSet<String> logQueue = new TreeSet<>();
357      if (wals == null || wals.size() == 0) {
358        listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldClusterZnode));
359      } else {
360        // create the new cluster znode
361        ZKUtilOp op = ZKUtilOp.createAndFailSilent(newPeerZnode, HConstants.EMPTY_BYTE_ARRAY);
362        listOfOps.add(op);
363        // get the offset of the logs and set it to new znodes
364        for (String wal : wals) {
365          String oldWalZnode = ZKUtil.joinZNode(oldClusterZnode, wal);
366          byte[] logOffset = ZKUtil.getData(this.zookeeper, oldWalZnode);
367          LOG.debug("Creating " + wal + " with data " + Bytes.toString(logOffset));
368          String newLogZnode = ZKUtil.joinZNode(newPeerZnode, wal);
369          listOfOps.add(ZKUtilOp.createAndFailSilent(newLogZnode, logOffset));
370          listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldWalZnode));
371          logQueue.add(wal);
372        }
373        // add delete op for peer
374        listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldClusterZnode));
375
376        if (LOG.isTraceEnabled())
377          LOG.trace(" The multi list size is: " + listOfOps.size());
378      }
379      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, false);
380      if (LOG.isTraceEnabled())
381        LOG.trace("Atomically moved the dead regionserver logs. ");
382      return new Pair<>(newPeerId, logQueue);
383    } catch (KeeperException e) {
384      // Multi call failed; it looks like some other regionserver took away the logs.
385      LOG.warn("Got exception in copyQueuesFromRSUsingMulti: ", e);
386    } catch (InterruptedException e) {
387      LOG.warn("Got exception in copyQueuesFromRSUsingMulti: ", e);
388      Thread.currentThread().interrupt();
389    }
390    return null;
391  }

393  /**
394   * This methods moves all the wals queues from another region server and returns them all sorted
395   * per peer cluster (appended with the dead server's znode)
396   * @param znode server names to copy
397   * @return all wals for the peer of that cluster, null if an error occurred
398   */
399  private Pair<String, SortedSet<String>> copyQueueFromLockedRS(String znode, String peerId) {
400    // TODO this method isn't atomic enough, we could start copying and then
401    // TODO fail for some reason and we would end up with znodes we don't want.
402    try {
403      String nodePath = ZKUtil.joinZNode(this.queuesZNode, znode);
404      ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(peerId);
405      String clusterPath = ZKUtil.joinZNode(nodePath, peerId);
406      if (!peerExists(replicationQueueInfo.getPeerId())) {
407        LOG.warn("Peer " + peerId + " didn't exist, skipping the replay");
408        // Protection against moving orphaned queues
409        return null;
410      }
411      // We add the name of the recovered RS to the new znode, we can even
412      // do that for queues that were recovered 10 times giving a znode like
413      // number-startcode-number-otherstartcode-number-anotherstartcode-etc
414      String newCluster = peerId + "-" + znode;
415      String newClusterZnode = ZKUtil.joinZNode(this.myQueuesZnode, newCluster);
416
417      List<String> wals = ZKUtil.listChildrenNoWatch(this.zookeeper, clusterPath);
418      // That region server didn't have anything to replicate for this cluster
419      if (wals == null || wals.size() == 0) {
420        return null;
421      }
422      ZKUtil.createNodeIfNotExistsAndWatch(this.zookeeper, newClusterZnode,
423          HConstants.EMPTY_BYTE_ARRAY);
424      SortedSet<String> logQueue = new TreeSet<>();
425      for (String wal : wals) {
426        String z = ZKUtil.joinZNode(clusterPath, wal);
427        byte[] positionBytes = ZKUtil.getData(this.zookeeper, z);
428        long position = 0;
429        try {
430          position = ZKUtil.parseWALPositionFrom(positionBytes);
431        } catch (DeserializationException e) {
432          LOG.warn("Failed parse of wal position from the following znode: " + z
433              + ", Exception: " + e);
434        }
435        LOG.debug("Creating " + wal + " with data " + position);
436        String child = ZKUtil.joinZNode(newClusterZnode, wal);
437        // Position doesn't actually change, we are just deserializing it for
438        // logging, so just use the already serialized version
439        ZKUtil.createNodeIfNotExistsAndWatch(this.zookeeper, child, positionBytes);
440        logQueue.add(wal);
441      }
442      return new Pair<>(newCluster, logQueue);
443    } catch (KeeperException e) {
444      LOG.warn("Got exception in copyQueueFromLockedRS: ", e);
445    } catch (InterruptedException e) {
446      LOG.warn(e);
447      Thread.currentThread().interrupt();
448    }
449    return null;
450  }

452  /**
453   * @param lockOwner
454   * @return Serialized protobuf of <code>lockOwner</code> with pb magic prefix prepended suitable
455   *         for use as content of an replication lock during region server fail over.
456   */
457  static byte[] lockToByteArray(final String lockOwner) {
458    byte[] bytes =
459        ZooKeeperProtos.ReplicationLock.newBuilder().setLockOwner(lockOwner).build().toByteArray();
460    return ProtobufUtil.prependPBMagic(bytes);
461  }

463  @Override
464  public void addHFileRefs(String peerId, List<String> files) throws ReplicationException {
465    String peerZnode = ZKUtil.joinZNode(this.hfileRefsZNode, peerId);
466    boolean debugEnabled = LOG.isDebugEnabled();
467    if (debugEnabled) {
468      LOG.debug("Adding hfile references " + files + " in queue " + peerZnode);
469    }
470    List<ZKUtilOp> listOfOps = new ArrayList<ZKUtil.ZKUtilOp>();
471    int size = files.size();
472    for (int i = 0; i < size; i++) {
473      listOfOps.add(ZKUtilOp.createAndFailSilent(ZKUtil.joinZNode(peerZnode, files.get(i)),
474        HConstants.EMPTY_BYTE_ARRAY));
475    }
476    if (debugEnabled) {
477      LOG.debug(" The multi list size for adding hfile references in zk for node " + peerZnode
478          + " is " + listOfOps.size());
479    }
480    try {
481      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
482    } catch (KeeperException e) {
483      throw new ReplicationException("Failed to create hfile reference znode=" + e.getPath(), e);
484    }
485  }

487  @Override
488  public void removeHFileRefs(String peerId, List<String> files) {
489    String peerZnode = ZKUtil.joinZNode(this.hfileRefsZNode, peerId);
490    boolean debugEnabled = LOG.isDebugEnabled();
491    if (debugEnabled) {
492      LOG.debug("Removing hfile references " + files + " from queue " + peerZnode);
493    }
494    List<ZKUtilOp> listOfOps = new ArrayList<ZKUtil.ZKUtilOp>();
495    int size = files.size();
496    for (int i = 0; i < size; i++) {
497      listOfOps.add(ZKUtilOp.deleteNodeFailSilent(ZKUtil.joinZNode(peerZnode, files.get(i))));
498    }
499    if (debugEnabled) {
500      LOG.debug(" The multi list size for removing hfile references in zk for node " + peerZnode
501          + " is " + listOfOps.size());
502    }
503    try {
504      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
505    } catch (KeeperException e) {
506      LOG.error("Failed to remove hfile reference znode=" + e.getPath(), e);
507    }
508  }

510  @Override
511  public void addPeerToHFileRefs(String peerId) throws ReplicationException {
512    String peerZnode = ZKUtil.joinZNode(this.hfileRefsZNode, peerId);
513    try {
514      if (ZKUtil.checkExists(this.zookeeper, peerZnode) == -1) {
515        LOG.info("Adding peer " + peerId + " to hfile reference queue.");
516        ZKUtil.createWithParents(this.zookeeper, peerZnode);
517      }
518    } catch (KeeperException e) {
519      throw new ReplicationException("Failed to add peer " + peerId + " to hfile reference queue.",
520          e);
521    }
522  }
523
524  @Override
525  public void removePeerFromHFileRefs(String peerId) {
526    final String peerZnode = ZKUtil.joinZNode(this.hfileRefsZNode, peerId);
527    try {
528      if (ZKUtil.checkExists(this.zookeeper, peerZnode) == -1) {
529        if (LOG.isDebugEnabled()) {
530          LOG.debug("Peer " + peerZnode + " not found in hfile reference queue.");
531        }
532        return;
533      } else {
534        LOG.info("Removing peer " + peerZnode + " from hfile reference queue.");
535        ZKUtil.deleteNodeRecursively(this.zookeeper, peerZnode);
536      }
537    } catch (KeeperException e) {
538      LOG.error("Ignoring the exception to remove peer " + peerId + " from hfile reference queue.",
539        e);
540    }
541  }
