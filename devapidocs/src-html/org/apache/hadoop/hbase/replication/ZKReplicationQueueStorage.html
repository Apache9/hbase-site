054/**
055 * ZK based replication queue storage.
056 * <p>
057 * The base znode for each regionserver is the regionserver name. For example:
058 *
059 * <pre>
060 * /hbase/replication/rs/hostname.example.org,6020,1234
061 * </pre>
062 *
063 * Within this znode, the region server maintains a set of WAL replication queues. These queues are
064 * represented by child znodes named using there give queue id. For example:
065 *
066 * <pre>
067 * /hbase/replication/rs/hostname.example.org,6020,1234/1
068 * /hbase/replication/rs/hostname.example.org,6020,1234/2
069 * </pre>
070 *
071 * Each queue has one child znode for every WAL that still needs to be replicated. The value of
072 * these WAL child znodes is the latest position that has been replicated. This position is updated
073 * every time a WAL entry is replicated. For example:
074 *
075 * <pre>
076 * /hbase/replication/rs/hostname.example.org,6020,1234/1/23522342.23422 [VALUE: 254]
077 * </pre>
078 */
079@InterfaceAudience.Private
080class ZKReplicationQueueStorage extends ZKReplicationStorageBase
081    implements ReplicationQueueStorage {
082

099  public ZKReplicationQueueStorage(ZKWatcher zookeeper, Configuration conf) {
100    super(zookeeper, conf);
101
102    String queuesZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
103    String hfileRefsZNodeName = conf.get(ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_KEY,
104      ZOOKEEPER_ZNODE_REPLICATION_HFILE_REFS_DEFAULT);
105    this.queuesZNode = ZNodePaths.joinZNode(replicationZNode, queuesZNodeName);
106    this.hfileRefsZNode = ZNodePaths.joinZNode(replicationZNode, hfileRefsZNodeName);
107  }

109  private String getRsNode(ServerName serverName) {
110    return ZNodePaths.joinZNode(queuesZNode, serverName.getServerName());
111  }

122    return getFileNode(getQueueNode(serverName, queueId), fileName);

126  public void removeQueue(ServerName serverName, String queueId) throws ReplicationException {
127    try {
128      ZKUtil.deleteNodeRecursively(zookeeper, getQueueNode(serverName, queueId));
129    } catch (KeeperException e) {

131          "Failed to delete queue (serverName=" + serverName + ", queueId=" + queueId + ")", e);
132    }
133  }

136  public void addWAL(ServerName serverName, String queueId, String fileName)
137      throws ReplicationException {
138    try {
139      ZKUtil.createWithParents(zookeeper, getFileNode(serverName, queueId, fileName));
140    } catch (KeeperException e) {
141      throw new ReplicationException("Failed to add wal to queue (serverName=" + serverName +
142        ", queueId=" + queueId + ", fileName=" + fileName + ")", e);

146  @Override
147  public void removeWAL(ServerName serverName, String queueId, String fileName)
148      throws ReplicationException {
149    String fileNode = getFileNode(serverName, queueId, fileName);
150    try {
151      ZKUtil.deleteNode(zookeeper, fileNode);
152    } catch (NoNodeException e) {
153      LOG.warn(fileNode + " has already been deleted when removing log");
154    } catch (KeeperException e) {
155      throw new ReplicationException("Failed to remove wal from queue (serverName=" + serverName +
156        ", queueId=" + queueId + ", fileName=" + fileName + ")", e);
157    }
158  }

160  @Override
161  public void setWALPosition(ServerName serverName, String queueId, String fileName, long position)
162      throws ReplicationException {
163    try {
164      ZKUtil.setData(zookeeper, getFileNode(serverName, queueId, fileName),
165        ZKUtil.positionToByteArray(position));
166    } catch (KeeperException e) {
167      throw new ReplicationException("Failed to set log position (serverName=" + serverName +
168        ", queueId=" + queueId + ", fileName=" + fileName + ", position=" + position + ")", e);
169    }
170  }

172  @Override
173  public long getWALPosition(ServerName serverName, String queueId, String fileName)
174      throws ReplicationException {
175    byte[] bytes;
176    try {
177      bytes = ZKUtil.getData(zookeeper, getFileNode(serverName, queueId, fileName));
178    } catch (KeeperException | InterruptedException e) {
179      throw new ReplicationException("Failed to get log position (serverName=" + serverName +
180        ", queueId=" + queueId + ", fileName=" + fileName + ")", e);
181    }
182    try {
183      return ZKUtil.parseWALPositionFrom(bytes);
184    } catch (DeserializationException de) {
185      LOG.warn("Failed to parse log position (serverName=" + serverName + ", queueId=" + queueId +
186        ", fileName=" + fileName + ")");
187    }
188    // if we can not parse the position, start at the beginning of the wal file again
189    return 0;
190  }

192  @Override
193  public Pair<String, SortedSet<String>> claimQueue(ServerName sourceServerName, String queueId,
194      ServerName destServerName) throws ReplicationException {
195    LOG.info(
196      "Atomically moving " + sourceServerName + "/" + queueId + "'s WALs to " + destServerName);
197    try {
198      ZKUtil.createWithParents(zookeeper, getRsNode(destServerName));
199    } catch (KeeperException e) {
200      throw new ReplicationException(
201          "Claim queue queueId=" + queueId + " from " + sourceServerName + " to " + destServerName +
202            " failed when creating the node for " + destServerName,
203          e);
204    }
205    try {
206      String oldQueueNode = getQueueNode(sourceServerName, queueId);
207      List<String> wals = ZKUtil.listChildrenNoWatch(zookeeper, oldQueueNode);
208      String newQueueId = queueId + "-" + sourceServerName;
209      if (CollectionUtils.isEmpty(wals)) {
210        ZKUtil.deleteNodeFailSilent(zookeeper, oldQueueNode);
211        LOG.info("Removed " + sourceServerName + "/" + queueId + " since it's empty");
212        return new Pair<>(newQueueId, Collections.emptySortedSet());
213      }
214      String newQueueNode = getQueueNode(destServerName, newQueueId);
215      List<ZKUtilOp> listOfOps = new ArrayList<>();
216      SortedSet<String> logQueue = new TreeSet<>();
217      // create the new cluster znode
218      listOfOps.add(ZKUtilOp.createAndFailSilent(newQueueNode, HConstants.EMPTY_BYTE_ARRAY));
219      // get the offset of the logs and set it to new znodes
220      for (String wal : wals) {
221        String oldWalNode = getFileNode(oldQueueNode, wal);
222        byte[] logOffset = ZKUtil.getData(this.zookeeper, oldWalNode);
223        if (LOG.isDebugEnabled()) {
224          LOG.debug("Creating " + wal + " with data " + Bytes.toStringBinary(logOffset));
225        }
226        String newWalNode = getFileNode(newQueueNode, wal);
227        listOfOps.add(ZKUtilOp.createAndFailSilent(newWalNode, logOffset));
228        listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldWalNode));
229        logQueue.add(wal);
230      }
231      // add delete op for peer
232      listOfOps.add(ZKUtilOp.deleteNodeFailSilent(oldQueueNode));
233
234      if (LOG.isTraceEnabled()) {
235        LOG.trace("The multi list size is: " + listOfOps.size());
236      }
237      ZKUtil.multiOrSequential(zookeeper, listOfOps, false);
238
239      LOG.info(
240        "Atomically moved " + sourceServerName + "/" + queueId + "'s WALs to " + destServerName);
241      return new Pair<>(newQueueId, logQueue);
242    } catch (NoNodeException | NodeExistsException | NotEmptyException | BadVersionException e) {
243      // Multi call failed; it looks like some other regionserver took away the logs.
244      // These exceptions mean that zk tells us the request can not be execute so it is safe to just
245      // return a null. For other types of exception should be thrown out to notify the upper layer.
246      LOG.info(
247        "Claim queue queueId=" + queueId + " from " + sourceServerName + " to " + destServerName +
248          " failed with " + e.toString() + ", maybe someone else has already took away the logs");
249      return null;
250    } catch (KeeperException | InterruptedException e) {
251      throw new ReplicationException("Claim queue queueId=" + queueId + " from " +
252        sourceServerName + " to " + destServerName + " failed", e);
253    }
254  }

256  @Override
257  public void removeReplicatorIfQueueIsEmpty(ServerName serverName) throws ReplicationException {
258    try {
259      ZKUtil.deleteNodeFailSilent(zookeeper, getRsNode(serverName));

319  @Override
320  public Set<String> getAllWALs() throws ReplicationException {
321    try {
322      for (int retry = 0;; retry++) {
323        int v0 = getQueuesZNodeCversion();
324        List<ServerName> rss = getListOfReplicators0();
325        if (rss.isEmpty()) {
326          LOG.debug("Didn't find any region server that replicates, won't prevent any deletions.");
327          return Collections.emptySet();
328        }
329        Set<String> wals = new HashSet<>();
330        for (ServerName rs : rss) {
331          for (String queueId : getAllQueues0(rs)) {
332            wals.addAll(getWALsInQueue0(rs, queueId));
333          }
334        }
335        int v1 = getQueuesZNodeCversion();
336        if (v0 == v1) {
337          return wals;
338        }
339        LOG.info(String.format("Replication queue node cversion changed from %d to %d, retry = %d",
340          v0, v1, retry));
341      }
342    } catch (KeeperException e) {
343      throw new ReplicationException("Failed to get all wals", e);
344    }
345  }

355  @Override
356  public void addPeerToHFileRefs(String peerId) throws ReplicationException {
357    String peerNode = getHFileRefsPeerNode(peerId);
358    try {
359      if (ZKUtil.checkExists(zookeeper, peerNode) == -1) {
360        LOG.info("Adding peer " + peerId + " to hfile reference queue.");
361        ZKUtil.createWithParents(zookeeper, peerNode);
362      }
363    } catch (KeeperException e) {
364      throw new ReplicationException("Failed to add peer " + peerId + " to hfile reference queue.",
365          e);
366    }
367  }
368
369  @Override
370  public void removePeerFromHFileRefs(String peerId) throws ReplicationException {
371    String peerNode = getHFileRefsPeerNode(peerId);
372    try {
373      if (ZKUtil.checkExists(zookeeper, peerNode) == -1) {
374        if (LOG.isDebugEnabled()) {
375          LOG.debug("Peer " + peerNode + " not found in hfile reference queue.");
376        }
377      } else {
378        LOG.info("Removing peer " + peerNode + " from hfile reference queue.");
379        ZKUtil.deleteNodeRecursively(zookeeper, peerNode);
380      }
381    } catch (KeeperException e) {
382      throw new ReplicationException(
383          "Failed to remove peer " + peerId + " from hfile reference queue.", e);
384    }
385  }

387  @Override
388  public void addHFileRefs(String peerId, List<Pair<Path, Path>> pairs)
389      throws ReplicationException {
390    String peerNode = getHFileRefsPeerNode(peerId);
391    boolean debugEnabled = LOG.isDebugEnabled();
392    if (debugEnabled) {
393      LOG.debug("Adding hfile references " + pairs + " in queue " + peerNode);
394    }
395    List<ZKUtilOp> listOfOps = pairs.stream().map(p -> p.getSecond().getName())
396        .map(n -> getHFileNode(peerNode, n))
397        .map(f -> ZKUtilOp.createAndFailSilent(f, HConstants.EMPTY_BYTE_ARRAY)).collect(toList());
398    if (debugEnabled) {
399      LOG.debug("The multi list size for adding hfile references in zk for node " + peerNode +
400        " is " + listOfOps.size());
401    }
402    try {
403      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
404    } catch (KeeperException e) {
405      throw new ReplicationException("Failed to add hfile reference to peer " + peerId, e);
406    }
407  }

409  @Override
410  public void removeHFileRefs(String peerId, List<String> files) throws ReplicationException {
411    String peerNode = getHFileRefsPeerNode(peerId);
412    boolean debugEnabled = LOG.isDebugEnabled();
413    if (debugEnabled) {
414      LOG.debug("Removing hfile references " + files + " from queue " + peerNode);
415    }
416
417    List<ZKUtilOp> listOfOps = files.stream().map(n -> getHFileNode(peerNode, n))
418        .map(ZKUtilOp::deleteNodeFailSilent).collect(toList());
419    if (debugEnabled) {
420      LOG.debug("The multi list size for removing hfile references in zk for node " + peerNode +
421        " is " + listOfOps.size());
422    }
423    try {
424      ZKUtil.multiOrSequential(this.zookeeper, listOfOps, true);
425    } catch (KeeperException e) {
426      throw new ReplicationException("Failed to remove hfile reference from peer " + peerId, e);
427    }
428  }

470  @Override
471  public Set<String> getAllHFileRefs() throws ReplicationException {
472    try {
473      for (int retry = 0;; retry++) {
474        int v0 = getHFileRefsZNodeCversion();
475        List<String> peers = getAllPeersFromHFileRefsQueue();
476        if (peers.isEmpty()) {
477          LOG.debug("Didn't find any peers with hfile references, won't prevent any deletions.");
478          return Collections.emptySet();
479        }
480        Set<String> hfileRefs = new HashSet<>();
481        for (String peer : peers) {
482          hfileRefs.addAll(getReplicableHFiles0(peer));
483        }
484        int v1 = getHFileRefsZNodeCversion();
485        if (v0 == v1) {
486          return hfileRefs;
487        }
488        LOG.debug(String.format(
489          "Replication hfile references node cversion changed from " + "%d to %d, retry = %d", v0,
490          v1, retry));
491      }
492    } catch (KeeperException e) {
493      throw new ReplicationException("Failed to get all hfile refs", e);
494    }
495  }
