018package org.apache.hadoop.hbase.replication;
019
020import static org.apache.hadoop.hbase.replication.ReplicationUtils.PEER_STATE_DISABLED_BYTES;
021import static org.apache.hadoop.hbase.replication.ReplicationUtils.PEER_STATE_ENABLED_BYTES;
022
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.Arrays;
026import java.util.List;
027
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.client.Delete;
030import org.apache.hadoop.hbase.client.Get;
031import org.apache.hadoop.hbase.client.Put;
032import org.apache.hadoop.hbase.client.Result;
033import org.apache.hadoop.hbase.client.ResultScanner;
034import org.apache.hadoop.hbase.client.Scan;
035import org.apache.hadoop.hbase.client.Table;
036import org.apache.hadoop.hbase.client.replication.ReplicationPeerConfigUtil;
037import org.apache.hadoop.hbase.exceptions.DeserializationException;
038import org.apache.hadoop.hbase.util.Bytes;
039import org.apache.hadoop.hbase.zookeeper.ZKWatcher;
040import org.apache.yetus.audience.InterfaceAudience;
041
042/**
043 * Table based replication peer storage.
044 */
045@InterfaceAudience.Private
046public class TableReplicationPeerStorage extends TableReplicationStorageBase
047    implements ReplicationPeerStorage {
048
049  public TableReplicationPeerStorage(ZKWatcher zookeeper, Configuration conf) throws IOException {
050    super(zookeeper, conf);
051  }
052
053  @Override
054  public void addPeer(String peerId, ReplicationPeerConfig peerConfig, boolean enabled)
055      throws ReplicationException {
056    try (Table table = getReplicationMetaTable()) {
057      Put put = new Put(Bytes.toBytes(peerId));
058      put.addColumn(FAMILY_PEER, QUALIFIER_PEER_CONFIG,
059        ReplicationPeerConfigUtil.toByteArray(peerConfig));
060      put.addColumn(FAMILY_PEER, QUALIFIER_PEER_STATE,
061        enabled ? PEER_STATE_ENABLED_BYTES : PEER_STATE_DISABLED_BYTES);
062      table.put(put);
063    } catch (IOException e) {
064      throw new ReplicationException("Failed to add peer " + peerId, e);
065    }
066  }
067
068  @Override
069  public void removePeer(String peerId) throws ReplicationException {
070    try (Table table = getReplicationMetaTable()) {
071      Delete delete = new Delete(Bytes.toBytes(peerId));
072      table.delete(delete);
073    } catch (IOException e) {
074      throw new ReplicationException("Failed to remove peer " + peerId, e);
075    }
076  }
077
078  // TODO make it to be a checkExistAndMutate operation.
079  private boolean peerExist(String peerId, Table table) throws IOException {
080    Get get = new Get(Bytes.toBytes(peerId));
081    get.addColumn(FAMILY_PEER, QUALIFIER_PEER_STATE);
082    return table.exists(get);
083  }
084
085  @Override
086  public void setPeerState(String peerId, boolean enabled) throws ReplicationException {
087    try (Table table = getReplicationMetaTable()) {
088      if (!peerExist(peerId, table)) {
089        throw new ReplicationException("Peer " + peerId + " does not exist.");
090      }
091      Put put = new Put(Bytes.toBytes(peerId));
092      put.addColumn(FAMILY_PEER, QUALIFIER_PEER_STATE,
093        enabled ? PEER_STATE_ENABLED_BYTES : PEER_STATE_DISABLED_BYTES);
094      table.put(put);
095    } catch (IOException e) {
096      throw new ReplicationException(
097          "Failed to set peer state, peerId=" + peerId + ", state=" + enabled, e);
098    }
099  }
100
101  @Override
102  public void updatePeerConfig(String peerId, ReplicationPeerConfig peerConfig)
103      throws ReplicationException {
104    try (Table table = getReplicationMetaTable()) {
105      if (!peerExist(peerId, table)) {
106        throw new ReplicationException("Peer " + peerId + " does not exist.");
107      }
108      Put put = new Put(Bytes.toBytes(peerId));
109      put.addColumn(FAMILY_PEER, QUALIFIER_PEER_CONFIG,
110        ReplicationPeerConfigUtil.toByteArray(peerConfig));
111      table.put(put);
112    } catch (IOException e) {
113      throw new ReplicationException("Failed to update peer configuration, peerId=" + peerId, e);
114    }
115  }
116
117  @Override
118  public List<String> listPeerIds() throws ReplicationException {
119    try (Table table = getReplicationMetaTable()) {
120      Scan scan = new Scan().addColumn(FAMILY_PEER, QUALIFIER_PEER_STATE);
121      try (ResultScanner scanner = table.getScanner(scan)) {
122        List<String> peerIds = new ArrayList<>();
123        for (Result r : scanner) {
124          peerIds.add(Bytes.toString(r.getRow()));
125        }
126        return peerIds;
127      }
128    } catch (IOException e) {
129      throw new ReplicationException("Failed to list peers", e);
130    }
131  }
132
133  @Override
134  public boolean isPeerEnabled(String peerId) throws ReplicationException {
135    try (Table table = getReplicationMetaTable()) {
136      Get get = new Get(Bytes.toBytes(peerId)).addColumn(FAMILY_PEER, QUALIFIER_PEER_STATE);
137      Result r = table.get(get);
138      if (r == null) {
139        throw new ReplicationException("Peer " + peerId + " does not found");
140      }
141      return Arrays.equals(PEER_STATE_ENABLED_BYTES, r.getValue(FAMILY_PEER, QUALIFIER_PEER_STATE));
142    } catch (IOException e) {
143      throw new ReplicationException("Failed to read the peer state, peerId=" + peerId, e);
144    }
145  }
146
147  @Override
148  public ReplicationPeerConfig getPeerConfig(String peerId) throws ReplicationException {
149    try (Table table = getReplicationMetaTable()) {
150      Get get = new Get(Bytes.toBytes(peerId)).addColumn(FAMILY_PEER, QUALIFIER_PEER_CONFIG);
151      Result r = table.get(get);
152      if (r == null) {
153        throw new ReplicationException("Peer " + peerId + " does not found");
154      }
155      byte[] data = r.getValue(FAMILY_PEER, QUALIFIER_PEER_CONFIG);
156      if (data == null || data.length == 0) {
157        throw new ReplicationException(
158            "Replication peer config data shouldn't be empty, peerId=" + peerId);
159      }
160      try {
161        return ReplicationPeerConfigUtil.parsePeerFrom(data);
162      } catch (DeserializationException e) {
163        throw new ReplicationException(
164            "Failed to parse replication peer config for peer with id=" + peerId, e);
165      }
166    } catch (IOException e) {
167      throw new ReplicationException(
168          "Failed to read the peer configuration in hbase:replication, peerId=" + peerId, e);
169    }
170  }
171}
