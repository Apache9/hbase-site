018package org.apache.hadoop.hbase.replication;
019
020import java.io.IOException;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.HConstants;
024import org.apache.hadoop.hbase.NamespaceDescriptor;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.TableName;
027import org.apache.hadoop.hbase.client.ColumnFamilyDescriptorBuilder;
028import org.apache.hadoop.hbase.client.Connection;
029import org.apache.hadoop.hbase.client.ConnectionFactory;
030import org.apache.hadoop.hbase.client.Table;
031import org.apache.hadoop.hbase.client.TableDescriptorBuilder;
032import org.apache.hadoop.hbase.regionserver.BloomType;
033import org.apache.hadoop.hbase.util.Bytes;
034import org.apache.hadoop.hbase.zookeeper.ZKWatcher;
035import org.apache.yetus.audience.InterfaceAudience;
036
037import org.apache.hbase.thirdparty.com.google.common.annotations.VisibleForTesting;
038
039@InterfaceAudience.Private
040public class TableReplicationStorageBase {
041  protected final ZKWatcher zookeeper;
042  protected final Configuration conf;
043
044  public static final TableName REPLICATION_TABLE =
045      TableName.valueOf(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR, "replication");
046
047  // Peer family, the row key would be peer id.
048  public static final byte[] FAMILY_PEER = Bytes.toBytes("peer");
049  public static final byte[] QUALIFIER_PEER_CONFIG = Bytes.toBytes("config");
050  public static final byte[] QUALIFIER_PEER_STATE = Bytes.toBytes("state");
051
052  // Region server state family, the row key would be name of region server.
053  public static final byte[] FAMILY_RS_STATE = Bytes.toBytes("rs_state");
054  public static final byte[] QUALIFIER_STATE_ENABLED = Bytes.toBytes("enabled");
055
056  // Queue and wal family, the row key would be name of region server.
057  public static final byte[] FAMILY_QUEUE = Bytes.toBytes("queue");
058  public static final byte[] FAMILY_WAL = Bytes.toBytes("wal");
059
060  // HFile-Refs family, the row key would be peer id.
061  public static final byte[] FAMILY_HFILE_REFS = Bytes.toBytes("hfile-refs");
062
063  // Region family, the row key would be peer id.
064  public static final byte[] FAMILY_REGIONS = Bytes.toBytes("regions");
065
066  private Connection connection;
067
068  protected static byte[] getServerNameRowKey(ServerName serverName) {
069    return Bytes.toBytes(serverName.toString());
070  }
071
072  protected static byte[] getRegionQualifier(String encodedRegionName) {
073    return Bytes.toBytes(encodedRegionName);
074  }
075
076  @VisibleForTesting
077  public static TableDescriptorBuilder createReplicationTableDescBuilder(final Configuration conf)
078      throws IOException {
079    int metaMaxVersion =
080        conf.getInt(HConstants.HBASE_META_VERSIONS, HConstants.DEFAULT_HBASE_META_VERSIONS);
081    int metaBlockSize =
082        conf.getInt(HConstants.HBASE_META_BLOCK_SIZE, HConstants.DEFAULT_HBASE_META_BLOCK_SIZE);
083    return TableDescriptorBuilder
084        .newBuilder(REPLICATION_TABLE)
085        .setColumnFamily(
086          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_PEER).setMaxVersions(metaMaxVersion)
087              .setInMemory(true).setBlocksize(metaBlockSize)
088              .setScope(HConstants.REPLICATION_SCOPE_LOCAL).setBloomFilterType(BloomType.NONE)
089              .build())
090        .setColumnFamily(
091          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_RS_STATE).setMaxVersions(metaMaxVersion)
092              .setInMemory(true).setBlocksize(metaBlockSize)
093              .setScope(HConstants.REPLICATION_SCOPE_LOCAL).setBloomFilterType(BloomType.NONE)
094              .build())
095        .setColumnFamily(
096          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_QUEUE).setMaxVersions(metaMaxVersion)
097              .setInMemory(true).setBlocksize(metaBlockSize)
098              .setScope(HConstants.REPLICATION_SCOPE_LOCAL).setBloomFilterType(BloomType.NONE)
099              .build())
100        .setColumnFamily(
101          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_WAL)
102              .setMaxVersions(HConstants.ALL_VERSIONS).setInMemory(true)
103              .setBlocksize(metaBlockSize).setScope(HConstants.REPLICATION_SCOPE_LOCAL)
104              .setBloomFilterType(BloomType.NONE).build())
105        .setColumnFamily(
106          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_REGIONS).setMaxVersions(metaMaxVersion)
107              .setInMemory(true).setBlocksize(metaBlockSize)
108              .setScope(HConstants.REPLICATION_SCOPE_LOCAL).setBloomFilterType(BloomType.NONE)
109              .build())
110        .setColumnFamily(
111          ColumnFamilyDescriptorBuilder.newBuilder(FAMILY_HFILE_REFS)
112              .setMaxVersions(metaMaxVersion).setInMemory(true).setBlocksize(metaBlockSize)
113              .setScope(HConstants.REPLICATION_SCOPE_LOCAL).setBloomFilterType(BloomType.NONE)
114              .build());
115  }
116
117  protected TableReplicationStorageBase(ZKWatcher zookeeper, Configuration conf)
118      throws IOException {
119    this.zookeeper = zookeeper;
120    this.conf = conf;
121    this.connection = ConnectionFactory.createConnection(conf);
122  }
123
124  protected Table getReplicationMetaTable() throws IOException {
125    return this.connection.getTable(REPLICATION_TABLE);
126  }
127}
