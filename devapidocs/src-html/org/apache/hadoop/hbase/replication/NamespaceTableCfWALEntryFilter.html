019package org.apache.hadoop.hbase.replication;
020
021import java.util.List;
022import java.util.Map;
023import java.util.Set;
024import org.apache.hadoop.hbase.Cell;
025import org.apache.hadoop.hbase.CellUtil;
026import org.apache.hadoop.hbase.TableName;
027import org.apache.hadoop.hbase.util.Bytes;
028import org.apache.hadoop.hbase.wal.WAL.Entry;
029import org.apache.hadoop.hbase.wal.WALEdit;
030import org.apache.yetus.audience.InterfaceAudience;
031
032/**
033 * Filter a WAL Entry by the peer config: replicate_all flag, namespaces config, table-cfs config,
034 * exclude namespaces config, and exclude table-cfs config.
035 *
036 * If replicate_all flag is true, it means all user tables will be replicated to peer cluster. But
037 * you can set exclude namespaces or exclude table-cfs which can't be replicated to peer cluster.
038 * Note: set a exclude namespace means that all tables in this namespace can't be replicated.
039 *
040 * If replicate_all flag is false, it means all user tables can't be replicated to peer cluster.
041 * But you can set namespaces or table-cfs which will be replicated to peer cluster.
042 * Note: set a namespace means that all tables in this namespace will be replicated.
043 */
044@InterfaceAudience.Private
045public class NamespaceTableCfWALEntryFilter implements WALEntryFilter, WALCellFilter {
046
047  private final ReplicationPeer peer;
048  private BulkLoadCellFilter bulkLoadFilter = new BulkLoadCellFilter();
049
050  public NamespaceTableCfWALEntryFilter(ReplicationPeer peer) {
051    this.peer = peer;
052  }
053
054  @Override
055  public Entry filter(Entry entry) {
056    TableName tabName = entry.getKey().getTableName();
057    String namespace = tabName.getNamespaceAsString();
058    ReplicationPeerConfig peerConfig = this.peer.getPeerConfig();
059
060    if (peerConfig.replicateAllUserTables()) {
061      // replicate all user tables, but filter by exclude namespaces config
062      Set<String> excludeNamespaces = peerConfig.getExcludeNamespaces();
063
064      // return null(prevent replicating) if logKey's table is in this peer's
065      // exclude namespaces list
066      if (excludeNamespaces != null && excludeNamespaces.contains(namespace)) {
067        return null;
068      }
069
070      return entry;
071    } else {
072      // Not replicate all user tables, so filter by namespaces and table-cfs config
073      Set<String> namespaces = peerConfig.getNamespaces();
074      Map<TableName, List<String>> tableCFs = peerConfig.getTableCFsMap();
075
076      if (namespaces == null && tableCFs == null) {
077        return null;
078      }
079
080      // First filter by namespaces config
081      // If table's namespace in peer config, all the tables data are applicable for replication
082      if (namespaces != null && namespaces.contains(namespace)) {
083        return entry;
084      }
085
086      // Then filter by table-cfs config
087      // return null(prevent replicating) if logKey's table isn't in this peer's
088      // replicable tables list
089      if (tableCFs == null || !tableCFs.containsKey(tabName)) {
090        return null;
091      }
092
093      return entry;
094    }
095  }
096
097  @Override
098  public Cell filterCell(final Entry entry, Cell cell) {
099    ReplicationPeerConfig peerConfig = this.peer.getPeerConfig();
100    if (peerConfig.replicateAllUserTables()) {
101      // replicate all user tables, but filter by exclude table-cfs config
102      final Map<TableName, List<String>> excludeTableCfs = peerConfig.getExcludeTableCFsMap();
103      if (excludeTableCfs == null) {
104        return cell;
105      }
106
107      if (CellUtil.matchingColumn(cell, WALEdit.METAFAMILY, WALEdit.BULK_LOAD)) {
108        cell = bulkLoadFilter.filterCell(cell,
109          fam -> filterByExcludeTableCfs(entry.getKey().getTableName(), Bytes.toString(fam),
110            excludeTableCfs));
111      } else {
112        if (filterByExcludeTableCfs(entry.getKey().getTableName(),
113          Bytes.toString(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength()),
114          excludeTableCfs)) {
115          return null;
116        }
117      }
118
119      return cell;
120    } else {
121      // not replicate all user tables, so filter by table-cfs config
122      final Map<TableName, List<String>> tableCfs = peerConfig.getTableCFsMap();
123      if (tableCfs == null) {
124        return cell;
125      }
126
127      if (CellUtil.matchingColumn(cell, WALEdit.METAFAMILY, WALEdit.BULK_LOAD)) {
128        cell = bulkLoadFilter.filterCell(cell,
129          fam -> filterByTableCfs(entry.getKey().getTableName(), Bytes.toString(fam), tableCfs));
130      } else {
131        if (filterByTableCfs(entry.getKey().getTableName(),
132          Bytes.toString(cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength()),
133          tableCfs)) {
134          return null;
135        }
136      }
137
138      return cell;
139    }
140  }
141
142  private boolean filterByExcludeTableCfs(TableName tableName, String family,
143      Map<TableName, List<String>> excludeTableCfs) {
144    List<String> excludeCfs = excludeTableCfs.get(tableName);
145    if (excludeCfs != null) {
146      // empty cfs means all cfs of this table are excluded
147      if (excludeCfs.isEmpty()) {
148        return true;
149      }
150      // ignore(remove) kv if its cf is in the exclude cfs list
151      if (excludeCfs.contains(family)) {
152        return true;
153      }
154    }
155    return false;
156  }
157
158  private boolean filterByTableCfs(TableName tableName, String family,
159      Map<TableName, List<String>> tableCfs) {
160    List<String> cfs = tableCfs.get(tableName);
161    // ignore(remove) kv if its cf isn't in the replicable cf list
162    // (empty cfs means all cfs of this table are replicable)
163    if (cfs != null && !cfs.contains(family)) {
164      return true;
165    }
166    return false;
167  }
168}
