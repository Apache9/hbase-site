048/**
049 * Coprocessor service for bulk loads in secure mode.
050 * @deprecated As of release 2.0.0, this will be removed in HBase 3.0.0
051 */
052@InterfaceAudience.Private
053@Deprecated
054public class SecureBulkLoadEndpoint extends SecureBulkLoadService
055    implements CoprocessorService, Coprocessor {
056
057  public static final long VERSION = 0L;
058
059  private static final Log LOG = LogFactory.getLog(SecureBulkLoadEndpoint.class);
060
061  private RegionCoprocessorEnvironment env;
062
063  @Override
064  public void start(CoprocessorEnvironment env) {
065    this.env = (RegionCoprocessorEnvironment)env;
066    LOG.warn("SecureBulkLoadEndpoint is deprecated. It will be removed in future releases.");
067    LOG.warn("Secure bulk load has been integrated into HBase core.");
068  }
069
070  @Override
071  public void stop(CoprocessorEnvironment env) throws IOException {
072  }
073
074  @Override
075  public void prepareBulkLoad(RpcController controller, PrepareBulkLoadRequest request,
076      RpcCallback<PrepareBulkLoadResponse> done) {
077    try {
078      SecureBulkLoadManager secureBulkLoadManager =
079          this.env.getRegionServerServices().getSecureBulkLoadManager();
080      String bulkToken = secureBulkLoadManager.prepareBulkLoad(this.env.getRegion(),
081          convert(request));
082      done.run(PrepareBulkLoadResponse.newBuilder().setBulkToken(bulkToken).build());
083    } catch (IOException e) {
084      CoprocessorRpcUtils.setControllerException(controller, e);
085    }
086    done.run(null);
087  }

127  @Override
128  public void secureBulkLoadHFiles(RpcController controller, SecureBulkLoadHFilesRequest request,
129      RpcCallback<SecureBulkLoadHFilesResponse> done) {
130    boolean loaded = false;
131    try {
132      SecureBulkLoadManager secureBulkLoadManager =
133          this.env.getRegionServerServices().getSecureBulkLoadManager();
134      BulkLoadHFileRequest bulkLoadHFileRequest = ConvertSecureBulkLoadHFilesRequest(request);
135      loaded = secureBulkLoadManager.secureBulkLoadHFiles(this.env.getRegion(),
136          convert(bulkLoadHFileRequest));
137    } catch (IOException e) {
138      CoprocessorRpcUtils.setControllerException(controller, e);
139    }
140    done.run(SecureBulkLoadHFilesResponse.newBuilder().setLoaded(loaded).build());
141  }

148        builder =

152    return builder.build();

154
155  private BulkLoadHFileRequest ConvertSecureBulkLoadHFilesRequest(
156      SecureBulkLoadHFilesRequest request) {
157    BulkLoadHFileRequest.Builder bulkLoadHFileRequest = BulkLoadHFileRequest.newBuilder();
158    RegionSpecifier region =
159        ProtobufUtil.buildRegionSpecifier(RegionSpecifierType.REGION_NAME, this.env
160            .getRegionInfo().getRegionName());
161    bulkLoadHFileRequest.setRegion(region).setFsToken(request.getFsToken())
162        .setBulkToken(request.getBulkToken()).setAssignSeqNum(request.getAssignSeqNum())
163        .addAllFamilyPath(request.getFamilyPathList());
164    return bulkLoadHFileRequest.build();
165  }
