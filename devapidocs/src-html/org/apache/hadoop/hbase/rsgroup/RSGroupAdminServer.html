082  private void checkOnlineServersOnly(Set<Address> servers) throws ConstraintException {
083    Set<Address> onlineServers = new HashSet<Address>();
084    for(ServerName server: master.getServerManager().getOnlineServers().keySet()) {
085      onlineServers.add(server.getAddress());
086    }
087    for (Address el: servers) {
088      if (!onlineServers.contains(el)) {
089        throw new ConstraintException(
090            "Server " + el + " is not an online server in 'default' RSGroup.");
091      }
092    }
093  }

095  /**
096   * Check passed name. Fail if nulls or if corresponding RSGroupInfo not found.
097   * @return The RSGroupInfo named <code>name</code>
098   * @throws IOException
099   */
100  private RSGroupInfo getAndCheckRSGroupInfo(String name)
101  throws IOException {
102    if (StringUtils.isEmpty(name)) {
103      throw new ConstraintException("RSGroup cannot be null.");
104    }
105    RSGroupInfo rsgi = getRSGroupInfo(name);
106    if (rsgi == null) {
107      throw new ConstraintException("RSGroup does not exist: " + name);
108    }
109    return rsgi;
110  }

112  /**
113   * @return List of Regions associated with this <code>server</code>.
114   */
115  private List<HRegionInfo> getRegions(final Address server) {
116    LinkedList<HRegionInfo> regions = new LinkedList<HRegionInfo>();
117    for (Map.Entry<HRegionInfo, ServerName> el :
118        master.getAssignmentManager().getRegionStates().getRegionAssignments().entrySet()) {
119      if (el.getValue().getAddress().equals(server)) {
120        addRegion(regions, el.getKey());
121      }
122    }
123    for (RegionState state:
124        this.master.getAssignmentManager().getRegionStates().getRegionsInTransition()) {
125      if (state.getServerName().getAddress().equals(server)) {
126        addRegion(regions, state.getRegion());
127      }
128    }
129    return regions;
130  }
131
132  private void addRegion(final LinkedList<HRegionInfo> regions, HRegionInfo hri) {
133    // If meta, move it last otherwise other unassigns fail because meta is not
134    // online for them to update state in. This is dodgy. Needs to be made more
135    // robust. See TODO below.
136    if (hri.isMetaRegion()) regions.addLast(hri);
137    else regions.addFirst(hri);
138  }

140  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="RCN_REDUNDANT_NULLCHECK_WOULD_HAVE_BEEN_A_NPE",
141      justification="Ignoring complaint because don't know what it is complaining about")
142  @Override
143  public void moveServers(Set<Address> servers, String targetGroupName)
144  throws IOException {
145    if (servers == null) {
146      throw new ConstraintException("The list of servers to move cannot be null.");
147    }
148    if (servers.isEmpty()) {
149      // For some reason this difference between null servers and isEmpty is important distinction.
150      // TODO. Why? Stuff breaks if I equate them.
151      return;
152    }
153    RSGroupInfo targetGrp = getAndCheckRSGroupInfo(targetGroupName);
154    RSGroupInfoManager manager = getRSGroupInfoManager();
155    // Lock the manager during the below manipulations.
156    synchronized (manager) {
157      if (master.getMasterCoprocessorHost() != null) {
158        master.getMasterCoprocessorHost().preMoveServers(servers, targetGroupName);
159      }
160      // Presume first server is the source group. Later we check all servers are from
161      // this same group.
162      Address firstServer = servers.iterator().next();
163      RSGroupInfo srcGrp = manager.getRSGroupOfServer(firstServer);
164      if (srcGrp == null) {
165        // Be careful. This message is tested for in TestRSGroupsBase...
166        throw new ConstraintException("Source RSGroup for server " + firstServer + " does not exist.");
167      }
168      if (srcGrp.getName().equals(targetGroupName)) {
169        throw new ConstraintException( "Target RSGroup " + targetGroupName +
170            " is same as source " + srcGrp + " RSGroup.");
171      }
172      // Only move online servers (when from 'default') or servers from other groups.
173      // This prevents bogus servers from entering groups
174      if (RSGroupInfo.DEFAULT_GROUP.equals(srcGrp.getName())) {
175        checkOnlineServersOnly(servers);
176      }
177      // Check all servers are of same rsgroup.
178      for (Address server: servers) {
179        String tmpGroup = manager.getRSGroupOfServer(server).getName();
180        if (!tmpGroup.equals(srcGrp.getName())) {
181          throw new ConstraintException("Move server request should only come from one source " +
182              "RSGroup. Expecting only " + srcGrp.getName() + " but contains " + tmpGroup);
183        }
184      }
185      if (srcGrp.getServers().size() <= servers.size() && srcGrp.getTables().size() > 0) {
186        throw new ConstraintException("Cannot leave a RSGroup " + srcGrp.getName() +
187            " that contains tables without servers to host them.");
188      }
189
190      // MovedServers may be < passed in 'servers'.
191      Set<Address> movedServers = manager.moveServers(servers, srcGrp.getName(), targetGroupName);
192      // Appy makes note that if we were passed in a List of servers,
193      // we'd save having to do stuff like the below.
194      List<Address> editableMovedServers = Lists.newArrayList(movedServers);
195      boolean foundRegionsToUnassign;
196      do {
197        foundRegionsToUnassign = false;
198        for (Iterator<Address> iter = editableMovedServers.iterator(); iter.hasNext();) {
199          Address rs = iter.next();
200          // Get regions that are associated with this server.
201          List<HRegionInfo> regions = getRegions(rs);
202
203          // Unassign regions for a server
204          // TODO: This is problematic especially if hbase:meta is in the mix.
205          // We need to update state in hbase:meta and if unassigned we hang
206          // around in here. There is a silly sort on linked list done above
207          // in getRegions putting hbase:meta last which helps but probably holes.
208          LOG.info("Unassigning " + regions.size() +
209              " region(s) from " + rs + " for server move to " + targetGroupName);
210          if (!regions.isEmpty()) {
211            // TODO bulk unassign or throttled unassign?
212            for (HRegionInfo region: regions) {
213              // Regions might get assigned from tables of target group so we need to filter
214              if (!targetGrp.containsTable(region.getTable())) {
215                this.master.getAssignmentManager().unassign(region);
216                if (master.getAssignmentManager().getRegionStates().
217                    getRegionState(region).isFailedOpen()) {
218                  // If region is in FAILED_OPEN state, it won't recover, not without
219                  // operator intervention... in hbase-2.0.0 at least. Continue rather
220                  // than mark region as 'foundRegionsToUnassign'.
221                  continue;
222                }
223                foundRegionsToUnassign = true;
224              }
225            }
226          }
227          if (!foundRegionsToUnassign) {
228            iter.remove();
229          }
230        }
231        try {
232          manager.wait(1000);
233        } catch (InterruptedException e) {
234          LOG.warn("Sleep interrupted", e);
235          Thread.currentThread().interrupt();
236        }
237      } while (foundRegionsToUnassign);
238      if (master.getMasterCoprocessorHost() != null) {
239        master.getMasterCoprocessorHost().postMoveServers(servers, targetGroupName);
240      }
241      LOG.info("Move server done: " + srcGrp.getName() + "=>" + targetGroupName);
242    }
243  }

245  @Override
246  public void moveTables(Set<TableName> tables, String targetGroup) throws IOException {
247    if (tables == null) {
248      throw new ConstraintException(
249          "The list of servers cannot be null.");
250    }
251    if (tables.size() < 1) {
252      LOG.debug("moveTables() passed an empty set. Ignoring.");
253      return;
254    }
255    RSGroupInfoManager manager = getRSGroupInfoManager();
256    // Lock the manager during below machinations.
257    synchronized (manager) {
258      if (master.getMasterCoprocessorHost() != null) {
259        master.getMasterCoprocessorHost().preMoveTables(tables, targetGroup);
260      }
261
262      if(targetGroup != null) {
263        RSGroupInfo destGroup = manager.getRSGroup(targetGroup);
264        if(destGroup == null) {
265          throw new ConstraintException("Target " + targetGroup + " RSGroup does not exist.");
266        }
267        if(destGroup.getServers().size() < 1) {
268          throw new ConstraintException("Target RSGroup must have at least one server.");
269        }
270      }
271
272      for (TableName table : tables) {
273        String srcGroup = manager.getRSGroupOfTable(table);
274        if(srcGroup != null && srcGroup.equals(targetGroup)) {
275          throw new ConstraintException(
276              "Source RSGroup " + srcGroup + " is same as target " + targetGroup +
277              " RSGroup for table " + table);
278        }
279      }
280      manager.moveTables(tables, targetGroup);
281      if (master.getMasterCoprocessorHost() != null) {
282        master.getMasterCoprocessorHost().postMoveTables(tables, targetGroup);
283      }
284    }
285    for (TableName table: tables) {
286      LockManager.MasterLock lock = master.getLockManager().createMasterLock(table,
287          LockProcedure.LockType.EXCLUSIVE, this.getClass().getName() + ": RSGroup: table move");
288      try {
289        try {
290          lock.acquire();
291        } catch (InterruptedException e) {
292          throw new IOException("Interrupted when waiting for table lock", e);
293        }
294        for (HRegionInfo region :
295            master.getAssignmentManager().getRegionStates().getRegionsOfTable(table)) {
296          master.getAssignmentManager().unassign(region);
297        }
298      } finally {
299        lock.release();
300      }
301    }
302  }

306    if (master.getMasterCoprocessorHost() != null) {
307      master.getMasterCoprocessorHost().preAddRSGroup(name);

315  @Override
316  public void removeRSGroup(String name) throws IOException {
317    RSGroupInfoManager manager = getRSGroupInfoManager();
318    // Hold lock across coprocessor calls.
319    synchronized (manager) {
320      if (master.getMasterCoprocessorHost() != null) {
321        master.getMasterCoprocessorHost().preRemoveRSGroup(name);
322      }
323      RSGroupInfo RSGroupInfo = manager.getRSGroup(name);
324      if (RSGroupInfo == null) {
325        throw new ConstraintException("RSGroup " + name + " does not exist");
326      }
327      int tableCount = RSGroupInfo.getTables().size();
328      if (tableCount > 0) {
329        throw new ConstraintException("RSGroup " + name + " has " + tableCount +
330            " tables; you must remove these tables from the rsgroup before " +
331            "the rsgroup can be removed.");
332      }
333      int serverCount = RSGroupInfo.getServers().size();
334      if (serverCount > 0) {
335        throw new ConstraintException("RSGroup " + name + " has " + serverCount +
336            " servers; you must remove these servers from the RSGroup before" +
337            "the RSGroup can be removed.");
338      }
339      for (NamespaceDescriptor ns: master.getClusterSchema().getNamespaces()) {
340        String nsGroup = ns.getConfigurationValue(RSGroupInfo.NAMESPACEDESC_PROP_GROUP);
341        if (nsGroup != null &&  nsGroup.equals(name)) {
342          throw new ConstraintException("RSGroup " + name + " is referenced by namespace: " +
343              ns.getName());
344        }
345      }
346      manager.removeRSGroup(name);
347      if (master.getMasterCoprocessorHost() != null) {
348        master.getMasterCoprocessorHost().postRemoveRSGroup(name);
349      }
350    }
351  }

353  @Override
354  public boolean balanceRSGroup(String groupName) throws IOException {
355    ServerManager serverManager = master.getServerManager();
356    AssignmentManager assignmentManager = master.getAssignmentManager();
357    LoadBalancer balancer = master.getLoadBalancer();
358
359    boolean balancerRan;
360    synchronized (balancer) {
361      if (master.getMasterCoprocessorHost() != null) {
362        master.getMasterCoprocessorHost().preBalanceRSGroup(groupName);
363      }
364      if (getRSGroupInfo(groupName) == null) {
365        throw new ConstraintException("RSGroup does not exist: "+groupName);
366      }
367      // Only allow one balance run at at time.
368      Map<String, RegionState> groupRIT = rsGroupGetRegionsInTransition(groupName);
369      if (groupRIT.size() > 0) {
370        LOG.debug("Not running balancer because " +
371          groupRIT.size() +
372          " region(s) in transition: " +
373          StringUtils.abbreviate(
374              master.getAssignmentManager().getRegionStates().getRegionsInTransition().toString(),
375              256));
376        return false;
377      }
378      if (serverManager.areDeadServersInProgress()) {
379        LOG.debug("Not running balancer because processing dead regionserver(s): " +
380            serverManager.getDeadServers());
381        return false;
382      }
383
384      //We balance per group instead of per table
385      List<RegionPlan> plans = new ArrayList<RegionPlan>();
386      for(Map.Entry<TableName, Map<ServerName, List<HRegionInfo>>> tableMap:
387          getRSGroupAssignmentsByTable(groupName).entrySet()) {
388        LOG.info("Creating partial plan for table "+tableMap.getKey()+": "+tableMap.getValue());
389        List<RegionPlan> partialPlans = balancer.balanceCluster(tableMap.getValue());
390        LOG.info("Partial plan for table "+tableMap.getKey()+": "+partialPlans);
391        if (partialPlans != null) {
392          plans.addAll(partialPlans);
393        }
394      }
395      long startTime = System.currentTimeMillis();
396      balancerRan = plans != null;
397      if (plans != null && !plans.isEmpty()) {
398        LOG.info("RSGroup balance "+groupName+" starting with plan count: "+plans.size());
399        for (RegionPlan plan: plans) {
400          LOG.info("balance " + plan);
401          assignmentManager.balance(plan);
402        }
403        LOG.info("RSGroup balance "+groupName+" completed after "+
404            (System.currentTimeMillis()-startTime)+" seconds");
405      }
406      if (master.getMasterCoprocessorHost() != null) {
407        master.getMasterCoprocessorHost().postBalanceRSGroup(groupName, balancerRan);
408      }
409    }
410    return balancerRan;
411  }

427  private Map<String, RegionState> rsGroupGetRegionsInTransition(String groupName)
428      throws IOException {
429    Map<String, RegionState> rit = Maps.newTreeMap();
430    AssignmentManager am = master.getAssignmentManager();
431    RSGroupInfo RSGroupInfo = getRSGroupInfo(groupName);
432    for(TableName tableName : RSGroupInfo.getTables()) {
433      for(HRegionInfo regionInfo: am.getRegionStates().getRegionsOfTable(tableName)) {
434        RegionState state =
435            master.getAssignmentManager().getRegionStates().getRegionTransitionState(regionInfo);
436        if(state != null) {
437          rit.put(regionInfo.getEncodedName(), state);
438        }
439      }
440    }
441    return rit;
442  }

444  private Map<TableName, Map<ServerName, List<HRegionInfo>>>
445      getRSGroupAssignmentsByTable(String groupName) throws IOException {
446    Map<TableName, Map<ServerName, List<HRegionInfo>>> result = Maps.newHashMap();
447    RSGroupInfo RSGroupInfo = getRSGroupInfo(groupName);
448    Map<TableName, Map<ServerName, List<HRegionInfo>>> assignments = Maps.newHashMap();
449    for(Map.Entry<HRegionInfo, ServerName> entry:
450        master.getAssignmentManager().getRegionStates().getRegionAssignments().entrySet()) {
451      TableName currTable = entry.getKey().getTable();
452      ServerName currServer = entry.getValue();
453      HRegionInfo currRegion = entry.getKey();
454      if(RSGroupInfo.getTables().contains(currTable)) {
455        if(!assignments.containsKey(entry.getKey().getTable())) {
456          assignments.put(currTable, new HashMap<ServerName, List<HRegionInfo>>());
457        }
458        if(!assignments.get(currTable).containsKey(currServer)) {
459          assignments.get(currTable).put(currServer, new ArrayList<HRegionInfo>());
460        }
461        assignments.get(currTable).get(currServer).add(currRegion);
462      }
463    }
464
465    Map<ServerName, List<HRegionInfo>> serverMap = Maps.newHashMap();
466    for(ServerName serverName: master.getServerManager().getOnlineServers().keySet()) {
467      if(RSGroupInfo.getServers().contains(serverName.getAddress())) {
468        serverMap.put(serverName, Collections.EMPTY_LIST);
469      }
470    }
471
472    //add all tables that are members of the group
473    for(TableName tableName : RSGroupInfo.getTables()) {
474      if(assignments.containsKey(tableName)) {
475        result.put(tableName, new HashMap<ServerName, List<HRegionInfo>>());
476        result.get(tableName).putAll(serverMap);
477        result.get(tableName).putAll(assignments.get(tableName));
478        LOG.debug("Adding assignments for "+tableName+": "+assignments.get(tableName));
479      }
480    }
481
482    return result;
483  }

485  public void prepareRSGroupForTable(HTableDescriptor desc) throws IOException {
486    String groupName =
487        master.getClusterSchema().getNamespace(desc.getTableName().getNamespaceAsString())
488                .getConfigurationValue(RSGroupInfo.NAMESPACEDESC_PROP_GROUP);
489    if (groupName == null) {
490      groupName = RSGroupInfo.DEFAULT_GROUP;
491    }
492    RSGroupInfo RSGroupInfo = getRSGroupInfo(groupName);
493    if (RSGroupInfo == null) {
494      throw new ConstraintException("RSGroup " + groupName + " does not exist.");
495    }
496    if (!RSGroupInfo.containsTable(desc.getTableName())) {
497      LOG.debug("Pre-moving table " + desc.getTableName() + " to RSGroup " + groupName);
498      moveTables(Sets.newHashSet(desc.getTableName()), groupName);
499    }
500  }
501
502  public void cleanupRSGroupForTable(TableName tableName) throws IOException {
503    try {
504      RSGroupInfo group = getRSGroupInfoOfTable(tableName);
505      if (group != null) {
506        LOG.debug("Removing deleted table from table rsgroup " + group.getName());
507        moveTables(Sets.newHashSet(tableName), null);
508      }
509    } catch (ConstraintException ex) {
510      LOG.debug("Failed to perform RSGroup information cleanup for table: " + tableName, ex);
511    } catch (IOException ex) {
512      LOG.debug("Failed to perform RSGroup information cleanup for table: " + tableName, ex);
513    }
514  }
