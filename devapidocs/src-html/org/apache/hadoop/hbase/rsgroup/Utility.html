068  /**
069   * @param master the master to get online servers for
070   * @return Set of online Servers named for their hostname and port (not ServerName).
071   */
072  static Set<Address> getOnlineServers(final MasterServices master) {
073    Set<Address> onlineServers = new HashSet<Address>();
074    if (master == null) {
075      return onlineServers;
076    }
077
078    for (ServerName server : master.getServerManager().getOnlineServers().keySet()) {
079      onlineServers.add(server.getAddress());
080    }
081    return onlineServers;
082  }

084  /**
085   * Verify <code>hbase:meta</code> is deployed and accessible.
086   * @param hConnection the connection to use
087   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
088   * @param timeout How long to wait on zk for meta address (passed through to the internal call to
089   *          {@link #getMetaServerConnection}.
090   * @return True if the <code>hbase:meta</code> location is healthy.
091   * @throws IOException if the number of retries for getting the connection is exceeded
092   * @throws InterruptedException if waiting for the socket operation fails
093   */
094  public static boolean verifyMetaRegionLocation(ClusterConnection hConnection, ZKWatcher zkw,
095      final long timeout) throws InterruptedException, IOException {
096    return verifyMetaRegionLocation(hConnection, zkw, timeout, RegionInfo.DEFAULT_REPLICA_ID);
097  }
098
099  /**
100   * Verify <code>hbase:meta</code> is deployed and accessible.
101   * @param connection the connection to use
102   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
103   * @param timeout How long to wait on zk for meta address (passed through to
104   * @param replicaId the ID of the replica
105   * @return True if the <code>hbase:meta</code> location is healthy.
106   * @throws InterruptedException if waiting for the socket operation fails
107   * @throws IOException if the number of retries for getting the connection is exceeded
108   */
109  public static boolean verifyMetaRegionLocation(ClusterConnection connection, ZKWatcher zkw,
110      final long timeout, int replicaId) throws InterruptedException, IOException {
111    AdminProtos.AdminService.BlockingInterface service = null;
112    try {
113      service = getMetaServerConnection(connection, zkw, timeout, replicaId);
114    } catch (NotAllMetaRegionsOnlineException e) {
115      // Pass
116    } catch (ServerNotRunningYetException e) {
117      // Pass -- remote server is not up so can't be carrying root
118    } catch (UnknownHostException e) {
119      // Pass -- server name doesn't resolve so it can't be assigned anything.
120    } catch (RegionServerStoppedException e) {
121      // Pass -- server name sends us to a server that is dying or already dead.
122    }
123    return (service != null) && verifyRegionLocation(connection, service,
124      MetaTableLocator.getMetaRegionLocation(zkw, replicaId),
125      RegionReplicaUtil.getRegionInfoForReplica(RegionInfoBuilder.FIRST_META_REGIONINFO, replicaId)
126        .getRegionName());
127  }
128
129  /**
130   * Verify we can connect to <code>hostingServer</code> and that its carrying
131   * <code>regionName</code>.
132   * @param hostingServer Interface to the server hosting <code>regionName</code>
133   * @param address The servername that goes with the <code>metaServer</code> interface. Used
134   *          logging.
135   * @param regionName The regionname we are interested in.
136   * @return True if we were able to verify the region located at other side of the interface.
137   */
138  // TODO: We should be able to get the ServerName from the AdminProtocol
139  // rather than have to pass it in. Its made awkward by the fact that the
140  // HRI is likely a proxy against remote server so the getServerName needs
141  // to be fixed to go to a local method or to a cache before we can do this.
142  private static boolean verifyRegionLocation(final ClusterConnection connection,
143      AdminService.BlockingInterface hostingServer, final ServerName address,
144      final byte[] regionName) {
145    if (hostingServer == null) {
146      LOG.info("Passed hostingServer is null");
147      return false;
148    }
149    Throwable t;
150    HBaseRpcController controller = connection.getRpcControllerFactory().newController();
151    try {
152      // Try and get regioninfo from the hosting server.
153      return ProtobufUtil.getRegionInfo(controller, hostingServer, regionName) != null;
154    } catch (ConnectException e) {
155      t = e;
156    } catch (RetriesExhaustedException e) {
157      t = e;
158    } catch (RemoteException e) {
159      IOException ioe = e.unwrapRemoteException();
160      t = ioe;
161    } catch (IOException e) {
162      Throwable cause = e.getCause();
163      if (cause != null && cause instanceof EOFException) {
164        t = cause;
165      } else if (cause != null && cause.getMessage() != null &&
166        cause.getMessage().contains("Connection reset")) {
167        t = cause;
168      } else {
169        t = e;
170      }
171    }
172    LOG.info("Failed verification of " + Bytes.toStringBinary(regionName) + " at address=" +
173      address + ", exception=" + t.getMessage());
174    return false;
175  }
176
177  /**
178   * Gets a connection to the server hosting meta, as reported by ZooKeeper, waiting up to the
179   * specified timeout for availability.
180   * <p>
181   * WARNING: Does not retry. Use an {@link org.apache.hadoop.hbase.client.HTable} instead.
182   * @param connection the connection to use
183   * @param zkw reference to the {@link ZKWatcher} which also contains configuration and operation
184   * @param timeout How long to wait on meta location
185   * @param replicaId the ID of the replica
186   * @return connection to server hosting meta
187   * @throws InterruptedException if waiting for the socket operation fails
188   * @throws IOException if the number of retries for getting the connection is exceeded
189   */
190  private static AdminService.BlockingInterface getMetaServerConnection(
191      ClusterConnection connection, ZKWatcher zkw, long timeout, int replicaId)
192      throws InterruptedException, IOException {
193    return getCachedConnection(connection,
194      MetaTableLocator.waitMetaRegionLocation(zkw, replicaId, timeout));
195  }
196
197  /**
198   * @param sn ServerName to get a connection against.
199   * @return The AdminProtocol we got when we connected to <code>sn</code> May have come from cache,
200   *         may not be good, may have been setup by this invocation, or may be null.
201   * @throws IOException if the number of retries for getting the connection is exceeded
202   */
203  private static AdminService.BlockingInterface getCachedConnection(ClusterConnection connection,
204      ServerName sn) throws IOException {
205    if (sn == null) {
206      return null;
207    }
208    AdminService.BlockingInterface service = null;
209    try {
210      service = connection.getAdmin(sn);
211    } catch (RetriesExhaustedException e) {
212      if (e.getCause() != null && e.getCause() instanceof ConnectException) {
213        LOG.debug("Catch this; presume it means the cached connection has gone bad.");
214      } else {
215        throw e;
216      }
217    } catch (SocketTimeoutException e) {
218      LOG.debug("Timed out connecting to " + sn);
219    } catch (NoRouteToHostException e) {
220      LOG.debug("Connecting to " + sn, e);
221    } catch (SocketException e) {
222      LOG.debug("Exception connecting to " + sn);
223    } catch (UnknownHostException e) {
224      LOG.debug("Unknown host exception connecting to  " + sn);
225    } catch (FailedServerException e) {
226      if (LOG.isDebugEnabled()) {
227        LOG.debug("Server " + sn + " is in failed server list.");
228      }
229    } catch (IOException ioe) {
230      Throwable cause = ioe.getCause();
231      if (ioe instanceof ConnectException) {
232        LOG.debug("Catch. Connect refused.");
233      } else if (cause != null && cause instanceof EOFException) {
234        LOG.debug("Catch. Other end disconnected us.");
235      } else if (cause != null && cause.getMessage() != null &&
236        cause.getMessage().toLowerCase(Locale.ROOT).contains("connection reset")) {
237        LOG.debug("Catch. Connection reset.");
238      } else {
239        throw ioe;
240      }
241
242    }
243    return service;
244  }
