073@InterfaceAudience.Private
074public class RSGroupAdminEndpoint implements MasterObserver, CoprocessorService {
075  private static final Log LOG = LogFactory.getLog(RSGroupAdminEndpoint.class);
076
077  private MasterServices master = null;
078  // Only instance of RSGroupInfoManager. RSGroup aware load balancers ask for this instance on
079  // their setup.
080  private RSGroupInfoManager groupInfoManager;
081  private RSGroupAdminServer groupAdminServer;
082  private final RSGroupAdminService groupAdminService = new RSGroupAdminServiceImpl();
083
084  @Override
085  public void start(CoprocessorEnvironment env) throws IOException {
086    master = ((MasterCoprocessorEnvironment)env).getMasterServices();
087    groupInfoManager = RSGroupInfoManagerImpl.getInstance(master);
088    groupAdminServer = new RSGroupAdminServer(master, groupInfoManager);
089    Class<?> clazz =
090        master.getConfiguration().getClass(HConstants.HBASE_MASTER_LOADBALANCER_CLASS, null);
091    if (!RSGroupableBalancer.class.isAssignableFrom(clazz)) {
092      throw new IOException("Configured balancer does not support RegionServer groups.");
093    }
094  }
095
096  @Override
097  public Service getService() {
098    return groupAdminService;
099  }

101  RSGroupInfoManager getGroupInfoManager() {
102    return groupInfoManager;
103  }

105  /**
106   * Implementation of RSGroupAdminService defined in RSGroupAdmin.proto.
107   * This class calls {@link RSGroupAdminServer} for actual work, converts result to protocol
108   * buffer response, handles exceptions if any occurred and then calls the {@code RpcCallback} with
109   * the response.
110   * Since our CoprocessorHost asks the Coprocessor for a Service
111   * ({@link CoprocessorService#getService()}) instead of doing "coproc instanceOf Service"
112   * and requiring Coprocessor itself to be Service (something we do with our observers),
113   * we can use composition instead of inheritance here. That makes it easy to manage
114   * functionalities in concise classes (sometimes inner classes) instead of single class doing
115   * many different things.
116   */
117  private class RSGroupAdminServiceImpl extends RSGroupAdminProtos.RSGroupAdminService {
118    @Override
119    public void getRSGroupInfo(RpcController controller,
120        GetRSGroupInfoRequest request, RpcCallback<GetRSGroupInfoResponse> done) {
121      GetRSGroupInfoResponse.Builder builder = GetRSGroupInfoResponse.newBuilder();
122      String groupName = request.getRSGroupName();
123      LOG.info(master.getClientIdAuditPrefix() + " initiates rsgroup info retrieval, group="
124              + groupName);
125      try {
126        RSGroupInfo rsGroupInfo = groupAdminServer.getRSGroupInfo(groupName);
127        if (rsGroupInfo != null) {
128          builder.setRSGroupInfo(RSGroupProtobufUtil.toProtoGroupInfo(rsGroupInfo));
129        }
130      } catch (IOException e) {
131        CoprocessorRpcUtils.setControllerException(controller, e);
132      }
133      done.run(builder.build());
134    }
135
136    @Override
137    public void getRSGroupInfoOfTable(RpcController controller,
138        GetRSGroupInfoOfTableRequest request, RpcCallback<GetRSGroupInfoOfTableResponse> done) {
139      GetRSGroupInfoOfTableResponse.Builder builder = GetRSGroupInfoOfTableResponse.newBuilder();
140      try {
141        TableName tableName = ProtobufUtil.toTableName(request.getTableName());
142        LOG.info(master.getClientIdAuditPrefix() + " initiates rsgroup info retrieval, table="
143                + tableName);
144        RSGroupInfo RSGroupInfo = groupAdminServer.getRSGroupInfoOfTable(tableName);
145        if (RSGroupInfo != null) {
146          builder.setRSGroupInfo(RSGroupProtobufUtil.toProtoGroupInfo(RSGroupInfo));
147        }
148      } catch (IOException e) {
149        CoprocessorRpcUtils.setControllerException(controller, e);
150      }
151      done.run(builder.build());
152    }
153
154    @Override
155    public void moveServers(RpcController controller, MoveServersRequest request,
156        RpcCallback<MoveServersResponse> done) {
157      MoveServersResponse.Builder builder = MoveServersResponse.newBuilder();
158      try {
159        Set<Address> hostPorts = Sets.newHashSet();
160        for (HBaseProtos.ServerName el : request.getServersList()) {
161          hostPorts.add(Address.fromParts(el.getHostName(), el.getPort()));
162        }
163        LOG.info(master.getClientIdAuditPrefix() + " move servers " + hostPorts +" to rsgroup "
164                + request.getTargetGroup());
165        groupAdminServer.moveServers(hostPorts, request.getTargetGroup());
166      } catch (IOException e) {
167        CoprocessorRpcUtils.setControllerException(controller, e);
168      }
169      done.run(builder.build());
170    }
171
172    @Override
173    public void moveTables(RpcController controller, MoveTablesRequest request,
174        RpcCallback<MoveTablesResponse> done) {
175      MoveTablesResponse.Builder builder = MoveTablesResponse.newBuilder();
176      try {
177        Set<TableName> tables = new HashSet<>(request.getTableNameList().size());
178        for (HBaseProtos.TableName tableName : request.getTableNameList()) {
179          tables.add(ProtobufUtil.toTableName(tableName));
180        }
181        LOG.info(master.getClientIdAuditPrefix() + " move tables " + tables +" to rsgroup "
182                + request.getTargetGroup());
183        groupAdminServer.moveTables(tables, request.getTargetGroup());
184      } catch (IOException e) {
185        CoprocessorRpcUtils.setControllerException(controller, e);
186      }
187      done.run(builder.build());
188    }
189

237        for (RSGroupInfo RSGroupInfo : groupAdminServer.listRSGroups()) {

246    @Override
247    public void getRSGroupInfoOfServer(RpcController controller,
248        GetRSGroupInfoOfServerRequest request, RpcCallback<GetRSGroupInfoOfServerResponse> done) {
249      GetRSGroupInfoOfServerResponse.Builder builder = GetRSGroupInfoOfServerResponse.newBuilder();
250      try {
251        Address hp = Address.fromParts(request.getServer().getHostName(),
252            request.getServer().getPort());
253        LOG.info(master.getClientIdAuditPrefix() + " initiates rsgroup info retrieval, server=" + hp);
254        RSGroupInfo RSGroupInfo = groupAdminServer.getRSGroupOfServer(hp);
255        if (RSGroupInfo != null) {
256          builder.setRSGroupInfo(RSGroupProtobufUtil.toProtoGroupInfo(RSGroupInfo));
257        }
258      } catch (IOException e) {
259        CoprocessorRpcUtils.setControllerException(controller, e);
260      }
261      done.run(builder.build());
262    }
263
264    @Override
265    public void moveServersAndTables(RpcController controller,
266        MoveServersAndTablesRequest request, RpcCallback<MoveServersAndTablesResponse> done) {
267      MoveServersAndTablesResponse.Builder builder = MoveServersAndTablesResponse.newBuilder();
268      try {
269        Set<Address> hostPorts = Sets.newHashSet();
270        for (HBaseProtos.ServerName el : request.getServersList()) {
271          hostPorts.add(Address.fromParts(el.getHostName(), el.getPort()));
272        }
273        Set<TableName> tables = new HashSet<>(request.getTableNameList().size());
274        for (HBaseProtos.TableName tableName : request.getTableNameList()) {
275          tables.add(ProtobufUtil.toTableName(tableName));
276        }
277        LOG.info(master.getClientIdAuditPrefix() + " move servers " + hostPorts
278                + " and tables " + tables + " to rsgroup" + request.getTargetGroup());
279        groupAdminServer.moveServersAndTables(hostPorts, tables, request.getTargetGroup());
280      } catch (IOException e) {
281        CoprocessorRpcUtils.setControllerException(controller, e);
282      }
283      done.run(builder.build());
284    }

287  void assignTableToGroup(TableDescriptor desc) throws IOException {
288    String groupName =
289        master.getClusterSchema().getNamespace(desc.getTableName().getNamespaceAsString())
290                .getConfigurationValue(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP);
291    if (groupName == null) {
292      groupName = RSGroupInfo.DEFAULT_GROUP;
293    }
294    RSGroupInfo rsGroupInfo = groupAdminServer.getRSGroupInfo(groupName);
295    if (rsGroupInfo == null) {
296      throw new ConstraintException("Default RSGroup (" + groupName + ") for this table's "
297          + "namespace does not exist.");
298    }
299    if (!rsGroupInfo.containsTable(desc.getTableName())) {
300      LOG.debug("Pre-moving table " + desc.getTableName() + " to RSGroup " + groupName);
301      groupAdminServer.moveTables(Sets.newHashSet(desc.getTableName()), groupName);
302    }
303  }

309  // Assign table to default RSGroup.
310  @Override
311  public void preCreateTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
312      TableDescriptor desc, HRegionInfo[] regions) throws IOException {
313    assignTableToGroup(desc);
314  }
315
316  // Remove table from its RSGroup.
317  @Override
318  public void postDeleteTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
319                              TableName tableName) throws IOException {
320    try {
321      RSGroupInfo group = groupAdminServer.getRSGroupInfoOfTable(tableName);
322      if (group != null) {
323        LOG.debug(String.format("Removing deleted table '%s' from rsgroup '%s'", tableName,
324            group.getName()));
325        groupAdminServer.moveTables(Sets.newHashSet(tableName), null);
326      }
327    } catch (IOException ex) {
328      LOG.debug("Failed to perform RSGroup information cleanup for table: " + tableName, ex);
329    }
330  }
331
332  @Override
333  public void preCreateNamespace(ObserverContext<MasterCoprocessorEnvironment> ctx,
334                                 NamespaceDescriptor ns) throws IOException {
335    String group = ns.getConfigurationValue(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP);
336    if(group != null && groupAdminServer.getRSGroupInfo(group) == null) {
337      throw new ConstraintException("Region server group "+group+" does not exit");
338    }
339  }
340
341  @Override
342  public void preModifyNamespace(ObserverContext<MasterCoprocessorEnvironment> ctx,
343      NamespaceDescriptor ns) throws IOException {
344    preCreateNamespace(ctx, ns);
345  }
346
347  @Override
348  public void preCloneSnapshot(ObserverContext<MasterCoprocessorEnvironment> ctx,
349      SnapshotDescription snapshot, TableDescriptor desc) throws IOException {
350    assignTableToGroup(desc);
351  }
