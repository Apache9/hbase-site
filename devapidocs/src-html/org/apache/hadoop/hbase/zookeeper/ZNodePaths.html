018package org.apache.hadoop.hbase.zookeeper;
019
020import static org.apache.hadoop.hbase.HConstants.DEFAULT_META_REPLICA_NUM;
021import static org.apache.hadoop.hbase.HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT;
022import static org.apache.hadoop.hbase.HConstants.META_REPLICAS_NUM;
023import static org.apache.hadoop.hbase.HConstants.SPLIT_LOGDIR_NAME;
024import static org.apache.hadoop.hbase.HConstants.ZOOKEEPER_ZNODE_PARENT;
025import static org.apache.hadoop.hbase.HRegionInfo.DEFAULT_REPLICA_ID;
026
027import org.apache.hadoop.hbase.shaded.com.google.common.collect.ImmutableMap;
028
029import java.util.Optional;
030import java.util.stream.IntStream;
031
032import org.apache.hadoop.conf.Configuration;
033import org.apache.hadoop.hbase.HRegionInfo;
034import org.apache.hadoop.hbase.classification.InterfaceAudience;
035
036/**
037 * Class that hold all the paths of znode for HBase.
038 */
039@InterfaceAudience.Private
040public class ZNodePaths {
041
042  public final static String META_ZNODE_PREFIX = "meta-region-server";
043
044  // base znode for this cluster
045  public final String baseZNode;
046  // the prefix of meta znode, does not include baseZNode.
047  public final String metaZNodePrefix;
048  // znodes containing the locations of the servers hosting the meta replicas
049  public final ImmutableMap<Integer, String> metaReplicaZNodes;
050  // znode containing ephemeral nodes of the regionservers
051  public final String rsZNode;
052  // znode containing ephemeral nodes of the draining regionservers
053  public final String drainingZNode;
054  // znode of currently active master
055  public final String masterAddressZNode;
056  // znode of this master in backup master directory, if not the active master
057  public final String backupMasterAddressesZNode;
058  // znode containing the current cluster state
059  public final String clusterStateZNode;
060  // znode used for table disabling/enabling
061  @Deprecated
062  public final String tableZNode;
063  // znode containing the unique cluster ID
064  public final String clusterIdZNode;
065  // znode used for log splitting work assignment
066  public final String splitLogZNode;
067  // znode containing the state of the load balancer
068  public final String balancerZNode;
069  // znode containing the state of region normalizer
070  public final String regionNormalizerZNode;
071  // znode containing the state of all switches, currently there are split and merge child node.
072  public final String switchZNode;
073  // znode containing the lock for the tables
074  public final String tableLockZNode;
075  // znode containing the state of recovering regions
076  public final String recoveringRegionsZNode;
077  // znode containing namespace descriptors
078  public final String namespaceZNode;
079  // znode of indicating master maintenance mode
080  public final String masterMaintZNode;
081
082  public ZNodePaths(Configuration conf) {
083    baseZNode = conf.get(ZOOKEEPER_ZNODE_PARENT, DEFAULT_ZOOKEEPER_ZNODE_PARENT);
084    ImmutableMap.Builder<Integer, String> builder = ImmutableMap.builder();
085    metaZNodePrefix = conf.get("zookeeper.znode.metaserver", META_ZNODE_PREFIX);
086    String defaultMetaReplicaZNode = ZKUtil.joinZNode(baseZNode, metaZNodePrefix);
087    builder.put(DEFAULT_REPLICA_ID, defaultMetaReplicaZNode);
088    int numMetaReplicas = conf.getInt(META_REPLICAS_NUM, DEFAULT_META_REPLICA_NUM);
089    IntStream.range(1, numMetaReplicas)
090        .forEachOrdered(i -> builder.put(i, defaultMetaReplicaZNode + "-" + i));
091    metaReplicaZNodes = builder.build();
092    rsZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.rs", "rs"));
093    drainingZNode = ZKUtil.joinZNode(baseZNode,
094      conf.get("zookeeper.znode.draining.rs", "draining"));
095    masterAddressZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.master", "master"));
096    backupMasterAddressesZNode = ZKUtil.joinZNode(baseZNode,
097      conf.get("zookeeper.znode.backup.masters", "backup-masters"));
098    clusterStateZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.state", "running"));
099    tableZNode = ZKUtil.joinZNode(baseZNode,
100      conf.get("zookeeper.znode.tableEnableDisable", "table"));
101    clusterIdZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.clusterId", "hbaseid"));
102    splitLogZNode = ZKUtil.joinZNode(baseZNode,
103      conf.get("zookeeper.znode.splitlog", SPLIT_LOGDIR_NAME));
104    balancerZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.balancer", "balancer"));
105    regionNormalizerZNode = ZKUtil.joinZNode(baseZNode,
106      conf.get("zookeeper.znode.regionNormalizer", "normalizer"));
107    switchZNode = ZKUtil.joinZNode(baseZNode, conf.get("zookeeper.znode.switch", "switch"));
108    tableLockZNode = ZKUtil.joinZNode(baseZNode,
109      conf.get("zookeeper.znode.tableLock", "table-lock"));
110    recoveringRegionsZNode = ZKUtil.joinZNode(baseZNode,
111      conf.get("zookeeper.znode.recovering.regions", "recovering-regions"));
112    namespaceZNode = ZKUtil.joinZNode(baseZNode,
113      conf.get("zookeeper.znode.namespace", "namespace"));
114    masterMaintZNode = ZKUtil.joinZNode(baseZNode,
115      conf.get("zookeeper.znode.masterMaintenance", "master-maintenance"));
116  }
117
118  @Override
119  public String toString() {
120    return "ZNodePaths [baseZNode=" + baseZNode + ", metaReplicaZNodes=" + metaReplicaZNodes
121        + ", rsZNode=" + rsZNode + ", drainingZNode=" + drainingZNode + ", masterAddressZNode="
122        + masterAddressZNode + ", backupMasterAddressesZNode=" + backupMasterAddressesZNode
123        + ", clusterStateZNode=" + clusterStateZNode + ", tableZNode=" + tableZNode
124        + ", clusterIdZNode=" + clusterIdZNode + ", splitLogZNode=" + splitLogZNode
125        + ", balancerZNode=" + balancerZNode + ", regionNormalizerZNode=" + regionNormalizerZNode
126        + ", switchZNode=" + switchZNode + ", tableLockZNode=" + tableLockZNode
127        + ", recoveringRegionsZNode=" + recoveringRegionsZNode + ", namespaceZNode="
128        + namespaceZNode + ", masterMaintZNode=" + masterMaintZNode + "]";
129  }
130
131  /**
132   * Is the znode of any meta replica
133   * @param node
134   * @return true or false
135   */
136  public boolean isAnyMetaReplicaZNode(String node) {
137    if (metaReplicaZNodes.containsValue(node)) {
138      return true;
139    }
140    return false;
141  }
142
143  /**
144   * Get the znode string corresponding to a replicaId
145   * @param replicaId
146   * @return znode
147   */
148  public String getZNodeForReplica(int replicaId) {
149    // return a newly created path but don't update the cache of paths
150    // This is mostly needed for tests that attempt to create meta replicas
151    // from outside the master
152    return Optional.ofNullable(metaReplicaZNodes.get(replicaId))
153        .orElseGet(() -> metaReplicaZNodes.get(DEFAULT_REPLICA_ID) + "-" + replicaId);
154  }
155
156  /**
157   * Parse the meta replicaId from the passed znode
158   * @param znode
159   * @return replicaId
160   */
161  public int getMetaReplicaIdFromZnode(String znode) {
162    if (znode.equals(metaZNodePrefix)) {
163      return HRegionInfo.DEFAULT_REPLICA_ID;
164    }
165    return Integer.parseInt(znode.substring(metaZNodePrefix.length() + 1));
166  }
167
168  /**
169   * Is it the default meta replica's znode
170   * @param znode
171   * @return true or false
172   */
173  public boolean isDefaultMetaReplicaZnode(String znode) {
174    return metaReplicaZNodes.get(DEFAULT_REPLICA_ID).equals(znode);
175  }
176}
