019package org.apache.hadoop.hbase.zookeeper;
020
021import java.util.Map.Entry;
022import java.util.Properties;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.hbase.HConstants;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027
028/**
029 * Utility methods for reading, and building the ZooKeeper configuration.
030 *
031 * The order and priority for reading the config are as follows:
032 * (1). Property with "hbase.zookeeper.property." prefix from HBase XML
033 * (2). other zookeeper related properties in HBASE XML
034 */
035@InterfaceAudience.Private
036public class ZKConfig {
037
038  private static final String VARIABLE_START = "${";
039
040  /**
041   * Make a Properties object holding ZooKeeper config.
042   * Parses the corresponding config options from the HBase XML configs
043   * and generates the appropriate ZooKeeper properties.
044   * @param conf Configuration to read from.
045   * @return Properties holding mappings representing ZooKeeper config file.
046   */
047  public static Properties makeZKProps(Configuration conf) {
048    return makeZKPropsFromHbaseConfig(conf);
049  }
050
051  /**
052   * Make a Properties object holding ZooKeeper config.
053   * Parses the corresponding config options from the HBase XML configs
054   * and generates the appropriate ZooKeeper properties.
055   *
056   * @param conf Configuration to read from.
057   * @return Properties holding mappings representing ZooKeeper config file.
058   */
059  private static Properties makeZKPropsFromHbaseConfig(Configuration conf) {
060    Properties zkProperties = new Properties();
061
062    // Directly map all of the hbase.zookeeper.property.KEY properties.
063    // Synchronize on conf so no loading of configs while we iterate
064    synchronized (conf) {
065      for (Entry<String, String> entry : conf) {
066        String key = entry.getKey();
067        if (key.startsWith(HConstants.ZK_CFG_PROPERTY_PREFIX)) {
068          String zkKey = key.substring(HConstants.ZK_CFG_PROPERTY_PREFIX_LEN);
069          String value = entry.getValue();
070          // If the value has variables substitutions, need to do a get.
071          if (value.contains(VARIABLE_START)) {
072            value = conf.get(key);
073          }
074          zkProperties.setProperty(zkKey, value);
075        }
076      }
077    }
078
079    // If clientPort is not set, assign the default.
080    if (zkProperties.getProperty(HConstants.CLIENT_PORT_STR) == null) {
081      zkProperties.put(HConstants.CLIENT_PORT_STR,
082          HConstants.DEFAULT_ZOOKEPER_CLIENT_PORT);
083    }
084
085    // Create the server.X properties.
086    int peerPort = conf.getInt("hbase.zookeeper.peerport", 2888);
087    int leaderPort = conf.getInt("hbase.zookeeper.leaderport", 3888);
088
089    final String[] serverHosts = conf.getStrings(HConstants.ZOOKEEPER_QUORUM,
090                                                 HConstants.LOCALHOST);
091    String serverHost;
092    String address;
093    String key;
094    for (int i = 0; i < serverHosts.length; ++i) {
095      if (serverHosts[i].contains(":")) {
096        serverHost = serverHosts[i].substring(0, serverHosts[i].indexOf(':'));
097      } else {
098        serverHost = serverHosts[i];
099      }
100      address = serverHost + ":" + peerPort + ":" + leaderPort;
101      key = "server." + i;
102      zkProperties.put(key, address);
103    }
104
105    return zkProperties;
106  }
107
108  /**
109   * Return the ZK Quorum servers string given the specified configuration
110   *
111   * @param conf
112   * @return Quorum servers String
113   */
114  private static String getZKQuorumServersStringFromHbaseConfig(Configuration conf) {
115    String defaultClientPort = Integer.toString(
116        conf.getInt(HConstants.ZOOKEEPER_CLIENT_PORT, HConstants.DEFAULT_ZOOKEPER_CLIENT_PORT));
117
118    // Build the ZK quorum server string with "server:clientport" list, separated by ','
119    final String[] serverHosts =
120        conf.getStrings(HConstants.ZOOKEEPER_QUORUM, HConstants.LOCALHOST);
121    return buildQuorumServerString(serverHosts, defaultClientPort);
122  }
123
124  /**
125   * Build the ZK quorum server string with "server:clientport" list, separated by ','
126   *
127   * @param serverHosts a list of servers for ZK quorum
128   * @param clientPort the default client port
129   * @return the string for a list of "server:port" separated by ","
130   */
131  public static String buildQuorumServerString(String[] serverHosts, String clientPort) {
132    StringBuilder quorumStringBuilder = new StringBuilder();
133    String serverHost;
134    for (int i = 0; i < serverHosts.length; ++i) {
135      if (serverHosts[i].contains(":")) {
136        serverHost = serverHosts[i]; // just use the port specified from the input
137      } else {
138        serverHost = serverHosts[i] + ":" + clientPort;
139      }
140      if (i > 0) {
141        quorumStringBuilder.append(',');
142      }
143      quorumStringBuilder.append(serverHost);
144    }
145    return quorumStringBuilder.toString();
146  }
147
148  /**
149   * Return the ZK Quorum servers string given the specified configuration.
150   * @return Quorum servers
151   */
152  public static String getZKQuorumServersString(Configuration conf) {
153    return getZKQuorumServersStringFromHbaseConfig(conf);
154  }
155}
