020package org.apache.hadoop.hbase.zookeeper.lock;
021
022import java.io.IOException;
023import java.util.List;
024import java.util.SortedSet;
025import java.util.TreeSet;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030import org.apache.hadoop.hbase.zookeeper.ZKUtil;
031import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
032
033/**
034 * ZooKeeper based read lock: does not exclude other read locks, but excludes
035 * and is excluded by write locks.
036 */
037@InterfaceAudience.Private
038public class ZKInterProcessReadLock extends ZKInterProcessLockBase {
039
040  private static final Log LOG = LogFactory.getLog(ZKInterProcessReadLock.class);
041
042  public ZKInterProcessReadLock(ZooKeeperWatcher zooKeeperWatcher,
043      String znode, byte[] metadata, MetadataHandler handler) {
044    super(zooKeeperWatcher, znode, metadata, handler, READ_LOCK_CHILD_NODE_PREFIX);
045  }
046
047  /**
048   * {@inheritDoc}
049   */
050  @Override
051  protected String getLockPath(String createdZNode, List<String> children) throws IOException {
052    TreeSet<String> writeChildren =
053        new TreeSet<String>(ZNodeComparator.COMPARATOR);
054    for (String child : children) {
055      if (isChildWriteLock(child)) {
056        writeChildren.add(child);
057      }
058    }
059    if (writeChildren.isEmpty()) {
060      return null;
061    }
062    SortedSet<String> lowerChildren = writeChildren.headSet(createdZNode);
063    if (lowerChildren.isEmpty()) {
064      return null;
065    }
066    String pathToWatch = lowerChildren.last();
067    String nodeHoldingLock = lowerChildren.first();
068    String znode = ZKUtil.joinZNode(parentLockNode, nodeHoldingLock);
069    handleLockMetadata(znode);
070
071    return pathToWatch;
072  }
073}
