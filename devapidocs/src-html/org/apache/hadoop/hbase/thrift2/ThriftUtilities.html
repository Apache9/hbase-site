084  /**
085   * Creates a {@link Get} (HBase) from a {@link TGet} (Thrift).
086   *
087   * This ignores any timestamps set on {@link TColumn} objects.
088   *
089   * @param in the <code>TGet</code> to convert
090   *
091   * @return <code>Get</code> object
092   *
093   * @throws IOException if an invalid time range or max version parameter is given
094   */
095  public static Get getFromThrift(TGet in) throws IOException {
096    Get out = new Get(in.getRow());
097
098    // Timestamp overwrites time range if both are set
099    if (in.isSetTimestamp()) {
100      out.setTimestamp(in.getTimestamp());
101    } else if (in.isSetTimeRange()) {
102      out.setTimeRange(in.getTimeRange().getMinStamp(), in.getTimeRange().getMaxStamp());
103    }
104
105    if (in.isSetMaxVersions()) {
106      out.setMaxVersions(in.getMaxVersions());
107    }
108
109    if (in.isSetFilterString()) {
110      ParseFilter parseFilter = new ParseFilter();
111      out.setFilter(parseFilter.parseFilterString(in.getFilterString()));
112    }
113
114    if (in.isSetAttributes()) {
115      addAttributes(out,in.getAttributes());
116    }
117
118    if (in.isSetAuthorizations()) {
119      out.setAuthorizations(new Authorizations(in.getAuthorizations().getLabels()));
120    }
121
122    if (!in.isSetColumns()) {
123      return out;
124    }
125
126    for (TColumn column : in.getColumns()) {
127      if (column.isSetQualifier()) {
128        out.addColumn(column.getFamily(), column.getQualifier());
129      } else {
130        out.addFamily(column.getFamily());
131      }
132    }
133
134    return out;
135  }

138   * Converts multiple {@link TGet}s (Thrift) into a list of {@link Get}s (HBase).
139   *
140   * @param in list of <code>TGet</code>s to convert
141   *
142   * @return list of <code>Get</code> objects
143   *
144   * @throws IOException if an invalid time range or max version parameter is given

147  public static List<Get> getsFromThrift(List<TGet> in) throws IOException {
148    List<Get> out = new ArrayList<>(in.size());
149    for (TGet get : in) {
150      out.add(getFromThrift(get));
151    }
152    return out;

155  /**
156   * Creates a {@link TResult} (Thrift) from a {@link Result} (HBase).
157   *
158   * @param in the <code>Result</code> to convert
159   *
160   * @return converted result, returns an empty result if the input is <code>null</code>
161   */
162  public static TResult resultFromHBase(Result in) {
163    Cell[] raw = in.rawCells();
164    TResult out = new TResult();
165    byte[] row = in.getRow();
166    if (row != null) {
167      out.setRow(in.getRow());
168    }
169    List<TColumnValue> columnValues = new ArrayList<>(raw.length);
170    for (Cell kv : raw) {
171      TColumnValue col = new TColumnValue();
172      col.setFamily(CellUtil.cloneFamily(kv));
173      col.setQualifier(CellUtil.cloneQualifier(kv));
174      col.setTimestamp(kv.getTimestamp());
175      col.setValue(CellUtil.cloneValue(kv));
176      if (kv.getTagsLength() > 0) {
177        col.setTags(PrivateCellUtil.cloneTags(kv));
178      }
179      columnValues.add(col);
180    }
181    out.setColumnValues(columnValues);
182    return out;
183  }

202  /**
203   * Creates a {@link Put} (HBase) from a {@link TPut} (Thrift)
204   *
205   * @param in the <code>TPut</code> to convert
206   *
207   * @return converted <code>Put</code>
208   */
209  public static Put putFromThrift(TPut in) {
210    Put out;
211
212    if (in.isSetTimestamp()) {
213      out = new Put(in.getRow(), in.getTimestamp());
214    } else {
215      out = new Put(in.getRow());
216    }
217
218    if (in.isSetDurability()) {
219      out.setDurability(durabilityFromThrift(in.getDurability()));
220    }
221
222    for (TColumnValue columnValue : in.getColumnValues()) {
223      try {
224        if (columnValue.isSetTimestamp()) {
225          out.add(CellBuilderFactory.create(CellBuilderType.DEEP_COPY)
226              .setRow(out.getRow())
227              .setFamily(columnValue.getFamily())
228              .setQualifier(columnValue.getQualifier())
229              .setTimestamp(columnValue.getTimestamp())
230              .setType(Cell.Type.Put)
231              .setValue(columnValue.getValue())
232              .build());
233        } else {
234          out.add(CellBuilderFactory.create(CellBuilderType.DEEP_COPY)
235              .setRow(out.getRow())
236              .setFamily(columnValue.getFamily())
237              .setQualifier(columnValue.getQualifier())
238              .setTimestamp(out.getTimestamp())
239              .setType(Cell.Type.Put)
240              .setValue(columnValue.getValue())
241              .build());
242        }
243      } catch (IOException e) {
244        throw new IllegalArgumentException((e));
245      }
246    }
247
248    if (in.isSetAttributes()) {
249      addAttributes(out,in.getAttributes());
250    }
251
252    if (in.getCellVisibility() != null) {
253      out.setCellVisibility(new CellVisibility(in.getCellVisibility().getExpression()));
254    }
255
256    return out;
257  }

276  /**
277   * Creates a {@link Delete} (HBase) from a {@link TDelete} (Thrift).
278   *
279   * @param in the <code>TDelete</code> to convert
280   *
281   * @return converted <code>Delete</code>
282   */
283  public static Delete deleteFromThrift(TDelete in) {
284    Delete out;
285
286    if (in.isSetColumns()) {
287      out = new Delete(in.getRow());
288      for (TColumn column : in.getColumns()) {
289        if (in.isSetDeleteType()) {
290          switch (in.getDeleteType()) {
291            case DELETE_COLUMN:
292              if (column.isSetTimestamp()) {
293                out.addColumn(column.getFamily(), column.getQualifier(), column.getTimestamp());
294              } else {
295                out.addColumn(column.getFamily(), column.getQualifier());
296              }
297              break;
298            case DELETE_COLUMNS:
299              if (column.isSetTimestamp()) {
300                out.addColumns(column.getFamily(), column.getQualifier(), column.getTimestamp());
301              } else {
302                out.addColumns(column.getFamily(), column.getQualifier());
303              }
304              break;
305            case DELETE_FAMILY:
306              if (column.isSetTimestamp()) {
307                out.addFamily(column.getFamily(), column.getTimestamp());
308              } else {
309                out.addFamily(column.getFamily());
310              }
311              break;
312            case DELETE_FAMILY_VERSION:
313              if (column.isSetTimestamp()) {
314                out.addFamilyVersion(column.getFamily(), column.getTimestamp());
315              } else {
316                throw new IllegalArgumentException(
317                    "Timestamp is required for TDelete with DeleteFamilyVersion type");
318              }
319              break;
320            default:
321              throw new IllegalArgumentException("DeleteType is required for TDelete");
322          }
323        } else {
324          throw new IllegalArgumentException("DeleteType is required for TDelete");
325        }
326      }
327    } else {
328      if (in.isSetTimestamp()) {
329        out = new Delete(in.getRow(), in.getTimestamp());
330      } else {
331        out = new Delete(in.getRow());
332      }
333    }
334
335    if (in.isSetAttributes()) {
336      addAttributes(out,in.getAttributes());
337    }
338
339    if (in.isSetDurability()) {
340      out.setDurability(durabilityFromThrift(in.getDurability()));
341    }
342
343    return out;
344  }

364  public static TDelete deleteFromHBase(Delete in) {
365    TDelete out = new TDelete(ByteBuffer.wrap(in.getRow()));
366
367    List<TColumn> columns = new ArrayList<>(in.getFamilyCellMap().entrySet().size());
368    long rowTimestamp = in.getTimestamp();
369    if (rowTimestamp != HConstants.LATEST_TIMESTAMP) {
370      out.setTimestamp(rowTimestamp);
371    }
372
373    // Map<family, List<KeyValue>>
374    for (Map.Entry<byte[], List<org.apache.hadoop.hbase.Cell>> familyEntry:
375        in.getFamilyCellMap().entrySet()) {
376      TColumn column = new TColumn(ByteBuffer.wrap(familyEntry.getKey()));
377      for (org.apache.hadoop.hbase.Cell cell: familyEntry.getValue()) {
378        byte[] family = CellUtil.cloneFamily(cell);
379        byte[] qualifier = CellUtil.cloneQualifier(cell);
380        long timestamp = cell.getTimestamp();
381        if (family != null) {
382          column.setFamily(family);
383        }
384        if (qualifier != null) {
385          column.setQualifier(qualifier);
386        }
387        if (timestamp != HConstants.LATEST_TIMESTAMP) {
388          column.setTimestamp(timestamp);
389        }
390      }
391      columns.add(column);
392    }
393    out.setColumns(columns);
394
395    return out;
396  }

398  /**
399   * Creates a {@link RowMutations} (HBase) from a {@link TRowMutations} (Thrift)
400   *
401   * @param in the <code>TRowMutations</code> to convert
402   *
403   * @return converted <code>RowMutations</code>
404   */
405  public static RowMutations rowMutationsFromThrift(TRowMutations in) throws IOException {
406    List<TMutation> mutations = in.getMutations();
407    RowMutations out = new RowMutations(in.getRow(), mutations.size());
408    for (TMutation mutation : mutations) {
409      if (mutation.isSetPut()) {
410        out.add(putFromThrift(mutation.getPut()));
411      }
412      if (mutation.isSetDeleteSingle()) {
413        out.add(deleteFromThrift(mutation.getDeleteSingle()));
414      }
415    }
416    return out;
417  }

419  public static Scan scanFromThrift(TScan in) throws IOException {
420    Scan out = new Scan();
421
422    if (in.isSetStartRow()) {
423      out.setStartRow(in.getStartRow());
424    }
425    if (in.isSetStopRow()) {
426      out.setStopRow(in.getStopRow());
427    }
428    if (in.isSetCaching()) {
429      out.setCaching(in.getCaching());
430    }
431    if (in.isSetMaxVersions()) {
432      out.setMaxVersions(in.getMaxVersions());
433    }
434
435    if (in.isSetColumns()) {
436      for (TColumn column : in.getColumns()) {
437        if (column.isSetQualifier()) {
438          out.addColumn(column.getFamily(), column.getQualifier());
439        } else {
440          out.addFamily(column.getFamily());
441        }
442      }
443    }
444
445    TTimeRange timeRange = in.getTimeRange();
446    if (timeRange != null &&
447        timeRange.isSetMinStamp() && timeRange.isSetMaxStamp()) {
448      out.setTimeRange(timeRange.getMinStamp(), timeRange.getMaxStamp());
449    }
450
451    if (in.isSetBatchSize()) {
452      out.setBatch(in.getBatchSize());
453    }
454
455    if (in.isSetFilterString()) {
456      ParseFilter parseFilter = new ParseFilter();
457      out.setFilter(parseFilter.parseFilterString(in.getFilterString()));
458    }
459
460    if (in.isSetAttributes()) {
461      addAttributes(out,in.getAttributes());
462    }
463
464    if (in.isSetAuthorizations()) {
465      out.setAuthorizations(new Authorizations(in.getAuthorizations().getLabels()));
466    }
467
468    if (in.isSetReversed()) {
469      out.setReversed(in.isReversed());
470    }
471
472    if (in.isSetCacheBlocks()) {
473      out.setCacheBlocks(in.isCacheBlocks());
474    }
475
476    if (in.isSetColFamTimeRangeMap()) {
477      Map<ByteBuffer, TTimeRange> colFamTimeRangeMap = in.getColFamTimeRangeMap();
478      if (MapUtils.isNotEmpty(colFamTimeRangeMap)) {
479        for (Map.Entry<ByteBuffer, TTimeRange> entry : colFamTimeRangeMap.entrySet()) {
480          out.setColumnFamilyTimeRange(Bytes.toBytes(entry.getKey()),
481              entry.getValue().getMinStamp(), entry.getValue().getMaxStamp());
482        }
483      }
484    }
485
486    if (in.isSetReadType()) {
487      out.setReadType(readTypeFromThrift(in.getReadType()));
488    }
489
490    if (in.isSetLimit()) {
491      out.setLimit(in.getLimit());
492    }
493
494    return out;
495  }

539  public static THRegionLocation regionLocationFromHBase(HRegionLocation hrl) {
540    HRegionInfo hri = hrl.getRegionInfo();
541    ServerName serverName = hrl.getServerName();
542
543    THRegionInfo thRegionInfo = new THRegionInfo();
544    THRegionLocation thRegionLocation = new THRegionLocation();
545    TServerName tServerName = new TServerName();
546
547    tServerName.setHostName(serverName.getHostname());
548    tServerName.setPort(serverName.getPort());
549    tServerName.setStartCode(serverName.getStartcode());
550
551    thRegionInfo.setTableName(hri.getTable().getName());
552    thRegionInfo.setEndKey(hri.getEndKey());
553    thRegionInfo.setStartKey(hri.getStartKey());
554    thRegionInfo.setOffline(hri.isOffline());
555    thRegionInfo.setSplit(hri.isSplit());
556    thRegionInfo.setReplicaId(hri.getReplicaId());
557
558    thRegionLocation.setRegionInfo(thRegionInfo);
559    thRegionLocation.setServerName(tServerName);
560
561    return thRegionLocation;
562  }

572  /**
573   * Adds all the attributes into the Operation object
574   */
575  private static void addAttributes(OperationWithAttributes op,
576                                    Map<ByteBuffer, ByteBuffer> attributes) {
577    if (attributes == null || attributes.isEmpty()) {
578      return;
579    }
580    for (Map.Entry<ByteBuffer, ByteBuffer> entry : attributes.entrySet()) {
581      String name = Bytes.toStringBinary(getBytes(entry.getKey()));
582      byte[] value =  getBytes(entry.getValue());
583      op.setAttribute(name, value);
584    }
585  }

587  private static Durability durabilityFromThrift(TDurability tDurability) {
588    switch (tDurability.getValue()) {
589      case 1: return Durability.SKIP_WAL;
590      case 2: return Durability.ASYNC_WAL;
591      case 3: return Durability.SYNC_WAL;
592      case 4: return Durability.FSYNC_WAL;
593      default: return null;
594    }
595  }

597  public static CompareOperator compareOpFromThrift(TCompareOp tCompareOp) {
598    switch (tCompareOp.getValue()) {
599      case 0: return CompareOperator.LESS;
600      case 1: return CompareOperator.LESS_OR_EQUAL;
601      case 2: return CompareOperator.EQUAL;
602      case 3: return CompareOperator.NOT_EQUAL;
603      case 4: return CompareOperator.GREATER_OR_EQUAL;
604      case 5: return CompareOperator.GREATER;
605      case 6: return CompareOperator.NO_OP;
606      default: return null;
607    }
608  }

611    switch (tReadType.getValue()) {
612      case 1: return ReadType.DEFAULT;
613      case 2: return ReadType.STREAM;
614      case 3: return ReadType.PREAD;
615      default: return null;
616    }
