038/**
039 * This class is used for exporting current state of load on a RegionServer.
040 */
041@InterfaceAudience.Public
042public class ServerLoad {
043  private int stores = 0;
044  private int storefiles = 0;
045  private int storeUncompressedSizeMB = 0;
046  private int storefileSizeMB = 0;
047  private int memstoreSizeMB = 0;
048  private long storefileIndexSizeKB = 0;
049  private long readRequestsCount = 0;
050  private long filteredReadRequestsCount = 0;
051  private long writeRequestsCount = 0;
052  private int rootIndexSizeKB = 0;
053  private int totalStaticIndexSizeKB = 0;
054  private int totalStaticBloomSizeKB = 0;
055  private long totalCompactingKVs = 0;
056  private long currentCompactedKVs = 0;
057  private long reportTime = 0;
058
059  @InterfaceAudience.Private
060  public ServerLoad(ClusterStatusProtos.ServerLoad serverLoad) {
061    this.serverLoad = serverLoad;
062    this.reportTime = System.currentTimeMillis();
063    for (ClusterStatusProtos.RegionLoad rl: serverLoad.getRegionLoadsList()) {
064      stores += rl.getStores();
065      storefiles += rl.getStorefiles();
066      storeUncompressedSizeMB += rl.getStoreUncompressedSizeMB();
067      storefileSizeMB += rl.getStorefileSizeMB();
068      memstoreSizeMB += rl.getMemstoreSizeMB();
069      storefileIndexSizeKB += rl.getStorefileIndexSizeKB();
070      readRequestsCount += rl.getReadRequestsCount();
071      filteredReadRequestsCount += rl.getFilteredReadRequestsCount();
072      writeRequestsCount += rl.getWriteRequestsCount();
073      rootIndexSizeKB += rl.getRootIndexSizeKB();
074      totalStaticIndexSizeKB += rl.getTotalStaticIndexSizeKB();
075      totalStaticBloomSizeKB += rl.getTotalStaticBloomSizeKB();
076      totalCompactingKVs += rl.getTotalCompactingKVs();
077      currentCompactedKVs += rl.getCurrentCompactedKVs();
078    }
079  }

238  /**
239   * Originally, this method factored in the effect of requests going to the
240   * server as well. However, this does not interact very well with the current
241   * region rebalancing code, which only factors number of regions. For the
242   * interim, until we can figure out how to make rebalancing use all the info
243   * available, we're just going to make load purely the number of regions.
244   *
245   * @return load factor for this server
246   */
247  public int getLoad() {
248    // See above comment
249    // int load = numberOfRequests == 0 ? 1 : numberOfRequests;
250    // load *= numberOfRegions == 0 ? 1 : numberOfRegions;
251    // return load;
252    return getNumberOfRegions();
253  }

255  /**
256   * @return region load metrics
257   */
258  public Map<byte[], RegionLoad> getRegionsLoad() {
259    Map<byte[], RegionLoad> regionLoads =
260      new TreeMap<>(Bytes.BYTES_COMPARATOR);
261    for (ClusterStatusProtos.RegionLoad rl : serverLoad.getRegionLoadsList()) {
262      RegionLoad regionLoad = new RegionLoad(rl);
263      regionLoads.put(regionLoad.getName(), regionLoad);
264    }
265    return regionLoads;
266  }

268  /**
269   * Return the RegionServer-level coprocessors
270   * @return string array of loaded RegionServer-level coprocessors
271   */
272  public String[] getRegionServerCoprocessors() {
273    List<Coprocessor> list = obtainServerLoadPB().getCoprocessorsList();
274    String [] ret = new String[list.size()];
275    int i = 0;
276    for (Coprocessor elem : list) {
277      ret[i++] = elem.getName();
278    }
279
280    return ret;
281  }

283  /**
284   * Return the RegionServer-level and Region-level coprocessors
285   * @return string array of loaded RegionServer-level and
286   *         Region-level coprocessors
287   */
288  public String[] getRsCoprocessors() {
289    // Need a set to remove duplicates, but since generated Coprocessor class
290    // is not Comparable, make it a Set<String> instead of Set<Coprocessor>
291    TreeSet<String> coprocessSet = new TreeSet<>();
292    for (Coprocessor coprocessor : obtainServerLoadPB().getCoprocessorsList()) {
293      coprocessSet.add(coprocessor.getName());
294    }
295    return coprocessSet.toArray(new String[coprocessSet.size()]);
296  }

305  /**
306   * @see java.lang.Object#toString()
307   */
308  @Override
309  public String toString() {
310     StringBuilder sb =
311        Strings.appendKeyValue(new StringBuilder(), "requestsPerSecond",
312          Double.valueOf(getRequestsPerSecond()));
313    Strings.appendKeyValue(sb, "numberOfOnlineRegions", Integer.valueOf(getNumberOfRegions()));
314    sb = Strings.appendKeyValue(sb, "usedHeapMB", Integer.valueOf(this.getUsedHeapMB()));
315    sb = Strings.appendKeyValue(sb, "maxHeapMB", Integer.valueOf(getMaxHeapMB()));
316    sb = Strings.appendKeyValue(sb, "numberOfStores", Integer.valueOf(this.stores));
317    sb = Strings.appendKeyValue(sb, "numberOfStorefiles", Integer.valueOf(this.storefiles));
318    sb =
319        Strings.appendKeyValue(sb, "storefileUncompressedSizeMB",
320          Integer.valueOf(this.storeUncompressedSizeMB));
321    sb = Strings.appendKeyValue(sb, "storefileSizeMB", Integer.valueOf(this.storefileSizeMB));
322    if (this.storeUncompressedSizeMB != 0) {
323      sb =
324          Strings.appendKeyValue(
325            sb,
326            "compressionRatio",
327            String.format("%.4f", (float) this.storefileSizeMB
328                / (float) this.storeUncompressedSizeMB));
329    }
330    sb = Strings.appendKeyValue(sb, "memstoreSizeMB", Integer.valueOf(this.memstoreSizeMB));
331    sb =
332        Strings.appendKeyValue(sb, "storefileIndexSizeKB",
333          Long.valueOf(this.storefileIndexSizeKB));
334    sb = Strings.appendKeyValue(sb, "readRequestsCount", Long.valueOf(this.readRequestsCount));
335    sb = Strings.appendKeyValue(sb, "filteredReadRequestsCount",
336      Long.valueOf(this.filteredReadRequestsCount));
337    sb = Strings.appendKeyValue(sb, "writeRequestsCount", Long.valueOf(this.writeRequestsCount));
338    sb = Strings.appendKeyValue(sb, "rootIndexSizeKB", Integer.valueOf(this.rootIndexSizeKB));
339    sb =
340        Strings.appendKeyValue(sb, "totalStaticIndexSizeKB",
341          Integer.valueOf(this.totalStaticIndexSizeKB));
342    sb =
343        Strings.appendKeyValue(sb, "totalStaticBloomSizeKB",
344          Integer.valueOf(this.totalStaticBloomSizeKB));
345    sb = Strings.appendKeyValue(sb, "totalCompactingKVs", Long.valueOf(this.totalCompactingKVs));
346    sb = Strings.appendKeyValue(sb, "currentCompactedKVs", Long.valueOf(this.currentCompactedKVs));
347    float compactionProgressPct = Float.NaN;
348    if (this.totalCompactingKVs > 0) {
349      compactionProgressPct =
350          Float.valueOf((float) this.currentCompactedKVs / this.totalCompactingKVs);
351    }
352    sb = Strings.appendKeyValue(sb, "compactionProgressPct", compactionProgressPct);
353
354    String[] coprocessorStrings = getRsCoprocessors();
355    if (coprocessorStrings != null) {
356      sb = Strings.appendKeyValue(sb, "coprocessors", Arrays.toString(coprocessorStrings));
357    }
358    return sb.toString();
359  }
