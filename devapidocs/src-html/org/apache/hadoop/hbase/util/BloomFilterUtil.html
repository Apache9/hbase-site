038  /** Record separator for the Bloom filter statistics human-readable string */
039  public static final String STATS_RECORD_SEP = "; ";
040  /**
041   * Used in computing the optimal Bloom filter size. This approximately equals
042   * 0.480453.
043   */
044  public static final double LOG2_SQUARED = Math.log(2) * Math.log(2);
045  
046  /**
047   * A random number generator to use for "fake lookups" when testing to
048   * estimate the ideal false positive rate.
049   */
050  private static Random randomGeneratorForTest;
051
052  public static final String PREFIX_LENGTH_KEY = "RowPrefixBloomFilter.prefix_length";
053  public static final String DELIMITER_KEY = "RowPrefixDelimitedBloomFilter.delimiter";
054  
055  /** Bit-value lookup array to prevent doing the same work over and over */
056  public static final byte [] bitvals = {
057    (byte) 0x01,
058    (byte) 0x02,
059    (byte) 0x04,
060    (byte) 0x08,
061    (byte) 0x10,
062    (byte) 0x20,
063    (byte) 0x40,
064    (byte) 0x80
065  };

073  /**
074   * @param maxKeys
075   * @param errorRate
076   * @return the number of bits for a Bloom filter than can hold the given
077   *         number of keys and provide the given error rate, assuming that the
078   *         optimal number of hash functions is used and it does not have to
079   *         be an integer.
080   */
081  public static long computeBitSize(long maxKeys, double errorRate) {
082    return (long) Math.ceil(maxKeys * (-Math.log(errorRate) / LOG2_SQUARED));
083  }
084
085  /**
086   * Sets a random generator to be used for look-ups instead of computing hashes. Can be used to
087   * simulate uniformity of accesses better in a test environment. Should not be set in a real
088   * environment where correctness matters!
089   * <p>
090 *   This gets used in {@link #contains(ByteBuff, int, int, Hash, int, HashKey)}
091   * @param random The random number source to use, or null to compute actual hashes
092   */
093  @VisibleForTesting
094  public static void setRandomGeneratorForTest(Random random) {
095    randomGeneratorForTest = random;
096  }
097
098  /**
099   * The maximum number of keys we can put into a Bloom filter of a certain
100   * size to maintain the given error rate, assuming the number of hash
101   * functions is chosen optimally and does not even have to be an integer
102   * (hence the "ideal" in the function name).
103   *
104   * @param bitSize
105   * @param errorRate
106   * @return maximum number of keys that can be inserted into the Bloom filter
107   * @see #computeMaxKeys(long, double, int) for a more precise estimate
108   */
109  public static long idealMaxKeys(long bitSize, double errorRate) {
110    // The reason we need to use floor here is that otherwise we might put
111    // more keys in a Bloom filter than is allowed by the target error rate.
112    return (long) (bitSize * (LOG2_SQUARED / -Math.log(errorRate)));
113  }
114
115  /**
116   * The maximum number of keys we can put into a Bloom filter of a certain
117   * size to get the given error rate, with the given number of hash functions.
118   *
119   * @param bitSize
120   * @param errorRate
121   * @param hashCount
122   * @return the maximum number of keys that can be inserted in a Bloom filter
123   *         to maintain the target error rate, if the number of hash functions
124   *         is provided.
125   */
126  public static long computeMaxKeys(long bitSize, double errorRate,
127      int hashCount) {
128    return (long) (-bitSize * 1.0 / hashCount *
129        Math.log(1 - Math.exp(Math.log(errorRate) / hashCount)));
130  }
131
132  /**
133   * Computes the actual error rate for the given number of elements, number
134   * of bits, and number of hash functions. Taken directly from the
135   * <a href=
136   * "http://en.wikipedia.org/wiki/Bloom_filter#Probability_of_false_positives"
137   * > Wikipedia Bloom filter article</a>.
138   *
139   * @param maxKeys
140   * @param bitSize
141   * @param functionCount
142   * @return the actual error rate
143   */
144  public static double actualErrorRate(long maxKeys, long bitSize,
145      int functionCount) {
146    return Math.exp(Math.log(1 - Math.exp(-functionCount * maxKeys * 1.0
147        / bitSize)) * functionCount);
148  }
149
150  /**
151   * Increases the given byte size of a Bloom filter until it can be folded by
152   * the given factor.
153   *
154   * @param bitSize
155   * @param foldFactor
156   * @return Foldable byte size
157   */
158  public static int computeFoldableByteSize(long bitSize, int foldFactor) {
159    long byteSizeLong = (bitSize + 7) / 8;
160    int mask = (1 << foldFactor) - 1;
161    if ((mask & byteSizeLong) != 0) {
162      byteSizeLong >>= foldFactor;
163      ++byteSizeLong;
164      byteSizeLong <<= foldFactor;
165    }
166    if (byteSizeLong > Integer.MAX_VALUE) {
167      throw new IllegalArgumentException("byteSize=" + byteSizeLong + " too "
168          + "large for bitSize=" + bitSize + ", foldFactor=" + foldFactor);
169    }
170    return (int) byteSizeLong;
171  }
172
173  public static int optimalFunctionCount(int maxKeys, long bitSize) {
174    long i = bitSize / maxKeys;
175    double result = Math.ceil(Math.log(2) * i);
176    if (result > Integer.MAX_VALUE){
177      throw new IllegalArgumentException("result too large for integer value.");
178    }
179    return (int)result;
180  }
181  
182  /**
183   * Creates a Bloom filter chunk of the given size.
184   *
185   * @param byteSizeHint the desired number of bytes for the Bloom filter bit
186   *          array. Will be increased so that folding is possible.
187   * @param errorRate target false positive rate of the Bloom filter
188   * @param hashType Bloom filter hash function type
189   * @param foldFactor
190   * @param bloomType
191   * @return the new Bloom filter of the desired size
192   */
193  public static BloomFilterChunk createBySize(int byteSizeHint,
194      double errorRate, int hashType, int foldFactor, BloomType bloomType) {
195    BloomFilterChunk bbf = new BloomFilterChunk(hashType, bloomType);
196
197    bbf.byteSize = computeFoldableByteSize(byteSizeHint * 8L, foldFactor);
198    long bitSize = bbf.byteSize * 8;
199    bbf.maxKeys = (int) idealMaxKeys(bitSize, errorRate);
200    bbf.hashCount = optimalFunctionCount(bbf.maxKeys, bitSize);
201
202    // Adjust max keys to bring error rate closer to what was requested,
203    // because byteSize was adjusted to allow for folding, and hashCount was
204    // rounded.
205    bbf.maxKeys = (int) computeMaxKeys(bitSize, errorRate, bbf.hashCount);
206
207    return bbf;
208  }

210  public static boolean contains(byte[] buf, int offset, int length,
211      ByteBuff bloomBuf, int bloomOffset, int bloomSize, Hash hash,
212      int hashCount) {
213    HashKey<byte[]> hashKey = new ByteArrayHashKey(buf, offset, length);
214    return contains(bloomBuf, bloomOffset, bloomSize, hash, hashCount, hashKey);
215  }
216
217  private static <T> boolean contains(ByteBuff bloomBuf, int bloomOffset, int bloomSize, Hash hash,
218      int hashCount, HashKey<T> hashKey) {
219    int hash1 = hash.hash(hashKey, 0);
220    int bloomBitSize = bloomSize << 3;
221
222    int hash2 = 0;
223    int compositeHash = 0;
224
225    if (randomGeneratorForTest == null) {
226      // Production mode
227      compositeHash = hash1;
228      hash2 = hash.hash(hashKey, hash1);
229    }
230
231    for (int i = 0; i < hashCount; i++) {
232      int hashLoc = (randomGeneratorForTest == null
233          // Production mode
234          ? Math.abs(compositeHash % bloomBitSize)
235          // Test mode with "fake look-ups" to estimate "ideal false positive rate"
236          : randomGeneratorForTest.nextInt(bloomBitSize));
237      compositeHash += hash2;
238      if (!checkBit(hashLoc, bloomBuf, bloomOffset)) {
239        return false;
240      }
241    }
242    return true;
243  }
244
245  public static boolean contains(Cell cell, ByteBuff bloomBuf, int bloomOffset, int bloomSize,
246      Hash hash, int hashCount, BloomType type) {
247    HashKey<Cell> hashKey = type == BloomType.ROWCOL ? new RowColBloomHashKey(cell)
248        : new RowBloomHashKey(cell);
249    return contains(bloomBuf, bloomOffset, bloomSize, hash, hashCount, hashKey);
250  }

252  /**
253   * Check if bit at specified index is 1.
254   *
255   * @param pos index of bit
256   * @return true if bit at specified index is 1, false if 0.
257   */
258   static boolean checkBit(int pos, ByteBuff bloomBuf, int bloomOffset) {
259    int bytePos = pos >> 3; //pos / 8
260    int bitPos = pos & 0x7; //pos % 8
261    byte curByte = bloomBuf.get(bloomOffset + bytePos);
262    curByte &= bitvals[bitPos];
263    return (curByte != 0);
264  }

266  /**
267   * A human-readable string with statistics for the given Bloom filter.
268   *
269   * @param bloomFilter the Bloom filter to output statistics for;
270   * @return a string consisting of "&lt;key&gt;: &lt;value&gt;" parts
271   *         separated by {@link #STATS_RECORD_SEP}.
272   */
273  public static String formatStats(BloomFilterBase bloomFilter) {
274    StringBuilder sb = new StringBuilder();
275    long k = bloomFilter.getKeyCount();
276    long m = bloomFilter.getMaxKeys();
277
278    sb.append("BloomSize: " + bloomFilter.getByteSize() + STATS_RECORD_SEP);
279    sb.append("No of Keys in bloom: " + k + STATS_RECORD_SEP);
280    sb.append("Max Keys for bloom: " + m);
281    if (m > 0) {
282      sb.append(STATS_RECORD_SEP + "Percentage filled: "
283          + NumberFormat.getPercentInstance().format(k * 1.0 / m));
284    }
285    return sb.toString();
286  }
287
288  public static String toString(BloomFilterChunk bloomFilter) {
289    return formatStats(bloomFilter) + STATS_RECORD_SEP + "Actual error rate: "
290        + String.format("%.8f", bloomFilter.actualErrorRate());
291  }

293  public static byte[] getBloomFilterParam(BloomType bloomFilterType, Configuration conf)
294      throws IllegalArgumentException{
295    byte[] bloomParam = null;
296    String message = "Bloom filter type is " + bloomFilterType + ", ";
297    switch (bloomFilterType) {
298      case ROWPREFIX_FIXED_LENGTH:
299        String prefixLengthString = conf.get(PREFIX_LENGTH_KEY);
300        if (prefixLengthString == null) {
301          message += PREFIX_LENGTH_KEY + " not specified.";
302          throw new IllegalArgumentException(message);
303        }
304        int prefixLength;
305        try {
306          prefixLength = Integer.parseInt(prefixLengthString);
307          if (prefixLength <= 0 || prefixLength > HConstants.MAX_ROW_LENGTH) {
308            message += "the value of " + PREFIX_LENGTH_KEY
309                + " must >=0 and < " + HConstants.MAX_ROW_LENGTH;
310            throw new IllegalArgumentException(message);
311          }
312        } catch (NumberFormatException nfe) {
313          message = "Number format exception when parsing " + PREFIX_LENGTH_KEY + " for BloomType "
314              + bloomFilterType.toString() + ":"
315              + prefixLengthString;
316          throw new IllegalArgumentException(message, nfe);
317        }
318        bloomParam = Bytes.toBytes(prefixLength);
319        break;
320      case ROWPREFIX_DELIMITED:
321        String delimiterString = conf.get(DELIMITER_KEY);
322        if (delimiterString == null || delimiterString.length() == 0) {
323          message += DELIMITER_KEY + " not specified.";
324          throw new IllegalArgumentException(message);
325        }
326        bloomParam = Bytes.toBytes(delimiterString);
327        break;
328      default:
329        break;
330    }
331    return bloomParam;
332  }
