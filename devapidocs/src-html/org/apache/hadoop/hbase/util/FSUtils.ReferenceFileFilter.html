126  /**
127   * Sets storage policy for given path according to config setting.
128   * If the passed path is a directory, we'll set the storage policy for all files
129   * created in the future in said directory. Note that this change in storage
130   * policy takes place at the HDFS level; it will persist beyond this RS's lifecycle.
131   * If we're running on a version of HDFS that doesn't support the given storage policy
132   * (or storage policies at all), then we'll issue a log message and continue.
133   *
134   * See http://hadoop.apache.org/docs/r2.6.0/hadoop-project-dist/hadoop-hdfs/ArchivalStorage.html
135   *
136   * @param fs We only do anything if an instance of DistributedFileSystem
137   * @param conf used to look up storage policy with given key; not modified.
138   * @param path the Path whose storage policy is to be set
139   * @param policyKey Key to use pulling a policy from Configuration:
140   * e.g. HConstants.WAL_STORAGE_POLICY (hbase.wal.storage.policy).
141   * @param defaultPolicy usually should be the policy NONE to delegate to HDFS
142   */
143  public static void setStoragePolicy(final FileSystem fs, final Configuration conf,
144      final Path path, final String policyKey, final String defaultPolicy) {
145    String storagePolicy = conf.get(policyKey, defaultPolicy).toUpperCase(Locale.ROOT);
146    if (storagePolicy.equals(defaultPolicy)) {
147      if (LOG.isTraceEnabled()) {
148        LOG.trace("default policy of " + defaultPolicy + " requested, exiting early.");
149      }
150      return;
151    }
152    setStoragePolicy(fs, path, storagePolicy);
153  }

158  /**
159   * Sets storage policy for given path.
160   * If the passed path is a directory, we'll set the storage policy for all files
161   * created in the future in said directory. Note that this change in storage
162   * policy takes place at the HDFS level; it will persist beyond this RS's lifecycle.
163   * If we're running on a version of HDFS that doesn't support the given storage policy
164   * (or storage policies at all), then we'll issue a log message and continue.
165   *
166   * See http://hadoop.apache.org/docs/r2.6.0/hadoop-project-dist/hadoop-hdfs/ArchivalStorage.html
167   *
168   * @param fs We only do anything if an instance of DistributedFileSystem
169   * @param path the Path whose storage policy is to be set
170   * @param storagePolicy Policy to set on <code>path</code>; see hadoop 2.6+
171   * org.apache.hadoop.hdfs.protocol.HdfsConstants for possible list e.g
172   * 'COLD', 'WARM', 'HOT', 'ONE_SSD', 'ALL_SSD', 'LAZY_PERSIST'.
173   */
174  public static void setStoragePolicy(final FileSystem fs, final Path path,
175      final String storagePolicy) {
176    if (storagePolicy == null) {
177      if (LOG.isTraceEnabled()) {
178        LOG.trace("We were passed a null storagePolicy, exiting early.");
179      }
180      return;
181    }
182    final String trimmedStoragePolicy = storagePolicy.trim();
183    if (trimmedStoragePolicy.isEmpty()) {
184      if (LOG.isTraceEnabled()) {
185        LOG.trace("We were passed an empty storagePolicy, exiting early.");
186      }
187      return;
188    }
189    boolean distributed = false;
190    try {
191      distributed = isDistributedFileSystem(fs);
192    } catch (IOException ioe) {
193      if (!warningMap.containsKey(fs)) {
194        warningMap.put(fs, true);
195        LOG.warn("FileSystem isn't an instance of DistributedFileSystem; presuming it doesn't "
196            + "support setStoragePolicy. Unable to set storagePolicy=" + trimmedStoragePolicy
197            + " on path=" + path);
198      } else if (LOG.isDebugEnabled()) {
199        LOG.debug("FileSystem isn't an instance of DistributedFileSystem; presuming it doesn't "
200            + "support setStoragePolicy. Unable to set storagePolicy=" + trimmedStoragePolicy
201            + " on path=" + path);
202      }
203      return;
204    }
205    if (distributed) {
206      invokeSetStoragePolicy(fs, path, trimmedStoragePolicy);
207    }
208  }

210  /*
211   * All args have been checked and are good. Run the setStoragePolicy invocation.
212   */
213  private static void invokeSetStoragePolicy(final FileSystem fs, final Path path,
214      final String storagePolicy) {
215    Method m = null;
216    try {
217      m = fs.getClass().getDeclaredMethod("setStoragePolicy",
218        new Class<?>[] { Path.class, String.class });
219      m.setAccessible(true);
220    } catch (NoSuchMethodException e) {
221      final String msg = "FileSystem doesn't support setStoragePolicy; HDFS-6584 not available";
222      if (!warningMap.containsKey(fs)) {
223        warningMap.put(fs, true);
224        LOG.warn(msg, e);
225      } else if (LOG.isDebugEnabled()) {
226        LOG.debug(msg, e);
227      }
228      m = null;
229    } catch (SecurityException e) {
230      final String msg = "No access to setStoragePolicy on FileSystem; HDFS-6584 not available";
231      if (!warningMap.containsKey(fs)) {
232        warningMap.put(fs, true);
233        LOG.warn(msg, e);
234      } else if (LOG.isDebugEnabled()) {
235        LOG.debug(msg, e);
236      }
237      m = null; // could happen on setAccessible()
238    }
239    if (m != null) {
240      try {
241        m.invoke(fs, path, storagePolicy);
242        if (LOG.isDebugEnabled()) {
243          LOG.debug("Set storagePolicy=" + storagePolicy + " for path=" + path);
244        }
245      } catch (Exception e) {
246        // This swallows FNFE, should we be throwing it? seems more likely to indicate dev
247        // misuse than a runtime problem with HDFS.
248        if (!warningMap.containsKey(fs)) {
249          warningMap.put(fs, true);
250          LOG.warn("Unable to set storagePolicy=" + storagePolicy + " for path=" + path, e);
251        } else if (LOG.isDebugEnabled()) {
252          LOG.debug("Unable to set storagePolicy=" + storagePolicy + " for path=" + path, e);
253        }
254        // check for lack of HDFS-7228
255        if (e instanceof InvocationTargetException) {
256          final Throwable exception = e.getCause();
257          if (exception instanceof RemoteException &&
258              HadoopIllegalArgumentException.class.getName().equals(
259                ((RemoteException)exception).getClassName())) {
260            if (LOG.isDebugEnabled()) {
261              LOG.debug("Given storage policy, '" +storagePolicy +"', was rejected and probably " +
262                "isn't a valid policy for the version of Hadoop you're running. I.e. if you're " +
263                "trying to use SSD related policies then you're likely missing HDFS-7228. For " +
264                "more information see the 'ArchivalStorage' docs for your Hadoop release.");
265            }
266          }
267        }
268      }
269    }
270  }

312  /**
313   * Compare path component of the Path URI; e.g. if hdfs://a/b/c and /a/b/c, it will compare the
314   * '/a/b/c' part. If you passed in 'hdfs://a/b/c and b/c, it would return true.  Does not consider
315   * schema; i.e. if schemas different but path or subpath matches, the two will equate.
316   * @param pathToSearch Path we will be trying to match.
317   * @param pathTail
318   * @return True if <code>pathTail</code> is tail on the path of <code>pathToSearch</code>
319   */
320  public static boolean isMatchingTail(final Path pathToSearch, final Path pathTail) {
321    if (pathToSearch.depth() != pathTail.depth()) return false;
322    Path tailPath = pathTail;
323    String tailName;
324    Path toSearch = pathToSearch;
325    String toSearchName;
326    boolean result = false;
327    do {
328      tailName = tailPath.getName();
329      if (tailName == null || tailName.length() <= 0) {
330        result = true;
331        break;
332      }
333      toSearchName = toSearch.getName();
334      if (toSearchName == null || toSearchName.length() <= 0) break;
335      // Move up a parent on each path for next go around.  Path doesn't let us go off the end.
336      tailPath = tailPath.getParent();
337      toSearch = toSearch.getParent();
338    } while(tailName.equals(toSearchName));
339    return result;
340  }

464  /**
465   * Create the specified file on the filesystem. By default, this will:
466   * <ol>
467   * <li>overwrite the file if it exists</li>
468   * <li>apply the umask in the configuration (if it is enabled)</li>
469   * <li>use the fs configured buffer size (or 4096 if not set)</li>
470   * <li>use the configured column family replication or default replication if
471   * {@link HColumnDescriptor#DEFAULT_DFS_REPLICATION}</li>
472   * <li>use the default block size</li>
473   * <li>not track progress</li>
474   * </ol>
475   * @param conf configurations
476   * @param fs {@link FileSystem} on which to write the file
477   * @param path {@link Path} to the file to write
478   * @param perm permissions
479   * @param favoredNodes
480   * @return output stream to the created file
481   * @throws IOException if the file cannot be created
482   */
483  public static FSDataOutputStream create(Configuration conf, FileSystem fs, Path path,
484      FsPermission perm, InetSocketAddress[] favoredNodes) throws IOException {
485    if (fs instanceof HFileSystem) {
486      FileSystem backingFs = ((HFileSystem)fs).getBackingFs();
487      if (backingFs instanceof DistributedFileSystem) {
488        // Try to use the favoredNodes version via reflection to allow backwards-
489        // compatibility.
490        short replication = Short.parseShort(conf.get(HColumnDescriptor.DFS_REPLICATION,
491          String.valueOf(HColumnDescriptor.DEFAULT_DFS_REPLICATION)));
492        try {
493          return (FSDataOutputStream) (DistributedFileSystem.class.getDeclaredMethod("create",
494            Path.class, FsPermission.class, boolean.class, int.class, short.class, long.class,
495            Progressable.class, InetSocketAddress[].class).invoke(backingFs, path, perm, true,
496            getDefaultBufferSize(backingFs),
497            replication > 0 ? replication : getDefaultReplication(backingFs, path),
498            getDefaultBlockSize(backingFs, path), null, favoredNodes));
499        } catch (InvocationTargetException ite) {
500          // Function was properly called, but threw it's own exception.
501          throw new IOException(ite.getCause());
502        } catch (NoSuchMethodException e) {
503          LOG.debug("DFS Client does not support most favored nodes create; using default create");
504          if (LOG.isTraceEnabled()) LOG.trace("Ignoring; use default create", e);
505        } catch (IllegalArgumentException e) {
506          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
507        } catch (SecurityException e) {
508          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
509        } catch (IllegalAccessException e) {
510          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
511        }
512      }
513    }
514    return create(fs, path, perm, true);
515  }

543  /**
544   * Get the file permissions specified in the configuration, if they are
545   * enabled.
546   *
547   * @param fs filesystem that the file will be created on.
548   * @param conf configuration to read for determining if permissions are
549   *          enabled and which to use
550   * @param permssionConfKey property key in the configuration to use when
551   *          finding the permission
552   * @return the permission to use when creating a new file on the fs. If
553   *         special permissions are not specified in the configuration, then
554   *         the default permissions on the the fs will be returned.
555   */
556  public static FsPermission getFilePermissions(final FileSystem fs,
557      final Configuration conf, final String permssionConfKey) {
558    boolean enablePermissions = conf.getBoolean(
559        HConstants.ENABLE_DATA_FILE_UMASK, false);
560
561    if (enablePermissions) {
562      try {
563        FsPermission perm = new FsPermission(FULL_RWX_PERMISSIONS);
564        // make sure that we have a mask, if not, go default.
565        String mask = conf.get(permssionConfKey);
566        if (mask == null)
567          return FsPermission.getFileDefault();
568        // appy the umask
569        FsPermission umask = new FsPermission(mask);
570        return perm.applyUMask(umask);
571      } catch (IllegalArgumentException e) {
572        LOG.warn(
573            "Incorrect umask attempted to be created: "
574                + conf.get(permssionConfKey)
575                + ", using default file permissions.", e);
576        return FsPermission.getFileDefault();
577      }
578    }
579    return FsPermission.getFileDefault();
580  }

582  /**
583   * Checks to see if the specified file system is available
584   *
585   * @param fs filesystem
586   * @throws IOException e
587   */
588  public static void checkFileSystemAvailable(final FileSystem fs)
589  throws IOException {
590    if (!(fs instanceof DistributedFileSystem)) {
591      return;
592    }
593    IOException exception = null;
594    DistributedFileSystem dfs = (DistributedFileSystem) fs;
595    try {
596      if (dfs.exists(new Path("/"))) {
597        return;
598      }
599    } catch (IOException e) {
600      exception = e instanceof RemoteException ?
601              ((RemoteException)e).unwrapRemoteException() : e;
602    }
603    try {
604      fs.close();
605    } catch (Exception e) {

657   * Verifies current version of file system
658   *
659   * @param fs filesystem object
660   * @param rootdir root hbase directory
661   * @return null if no version file exists, version string otherwise.
662   * @throws IOException e
663   * @throws org.apache.hadoop.hbase.exceptions.DeserializationException
664   */
665  public static String getVersion(FileSystem fs, Path rootdir)
666  throws IOException, DeserializationException {
667    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
668    FileStatus[] status = null;
669    try {
670      // hadoop 2.0 throws FNFE if directory does not exist.
671      // hadoop 1.0 returns null if directory does not exist.
672      status = fs.listStatus(versionFile);
673    } catch (FileNotFoundException fnfe) {
674      return null;
675    }
676    if (status == null || status.length == 0) return null;
677    String version = null;
678    byte [] content = new byte [(int)status[0].getLen()];
679    FSDataInputStream s = fs.open(versionFile);
680    try {
681      IOUtils.readFully(s, content, 0, content.length);
682      if (ProtobufUtil.isPBMagicPrefix(content)) {
683        version = parseVersionFrom(content);
684      } else {
685        // Presume it pre-pb format.
686        InputStream is = new ByteArrayInputStream(content);
687        DataInputStream dis = new DataInputStream(is);
688        try {
689          version = dis.readUTF();
690        } finally {
691          dis.close();
692        }
693      }
694    } catch (EOFException eof) {
695      LOG.warn("Version file was empty, odd, will try to set it.");
696    } finally {
697      s.close();
698    }
699    return version;
700  }

817   * Sets version of file system
818   *
819   * @param fs filesystem object
820   * @param rootdir hbase root directory
821   * @param version version to set
822   * @param wait time to wait for retry
823   * @param retries number of times to retry before throwing an IOException
824   * @throws IOException e
825   */
826  public static void setVersion(FileSystem fs, Path rootdir, String version,
827      int wait, int retries) throws IOException {
828    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
829    Path tempVersionFile = new Path(rootdir, HConstants.HBASE_TEMP_DIRECTORY + Path.SEPARATOR +
830      HConstants.VERSION_FILE_NAME);
831    while (true) {
832      try {
833        // Write the version to a temporary file
834        FSDataOutputStream s = fs.create(tempVersionFile);
835        try {
836          s.write(toVersionByteArray(version));
837          s.close();
838          s = null;
839          // Move the temp version file to its normal location. Returns false
840          // if the rename failed. Throw an IOE in that case.
841          if (!fs.rename(tempVersionFile, versionFile)) {
842            throw new IOException("Unable to move temp version file to " + versionFile);
843          }
844        } finally {
845          // Cleaning up the temporary if the rename failed would be trying
846          // too hard. We'll unconditionally create it again the next time
847          // through anyway, files are overwritten by default by create().
848
849          // Attempt to close the stream on the way out if it is still open.
850          try {
851            if (s != null) s.close();
852          } catch (IOException ignore) { }
853        }
854        LOG.info("Created version file at " + rootdir.toString() + " with version=" + version);
855        return;
856      } catch (IOException e) {
857        if (retries > 0) {
858          LOG.debug("Unable to create version file at " + rootdir.toString() + ", retrying", e);
859          fs.delete(versionFile, false);
860          try {
861            if (wait > 0) {
862              Thread.sleep(wait);
863            }
864          } catch (InterruptedException ie) {
865            throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
866          }
867          retries--;
868        } else {
869          throw e;
870        }
871      }
872    }
873  }

913  throws IOException {
914    Path idPath = new Path(rootdir, HConstants.CLUSTER_ID_FILE_NAME);
915    ClusterId clusterId = null;
916    FileStatus status = fs.exists(idPath)? fs.getFileStatus(idPath):  null;
917    if (status != null) {
918      int len = Ints.checkedCast(status.getLen());
919      byte [] content = new byte[len];
920      FSDataInputStream in = fs.open(idPath);
921      try {
922        in.readFully(content);
923      } catch (EOFException eof) {
924        LOG.warn("Cluster ID file " + idPath.toString() + " was empty");
925      } finally{
926        in.close();
927      }
928      try {
929        clusterId = ClusterId.parseFrom(content);
930      } catch (DeserializationException e) {
931        throw new IOException("content=" + Bytes.toString(content), e);
932      }
933      // If not pb'd, make it so.
934      if (!ProtobufUtil.isPBMagicPrefix(content)) {
935        String cid = null;
936        in = fs.open(idPath);
937        try {
938          cid = in.readUTF();
939          clusterId = new ClusterId(cid);
940        } catch (EOFException eof) {
941          LOG.warn("Cluster ID file " + idPath.toString() + " was empty");
942        } finally {
943          in.close();
944        }
945        rewriteAsPb(fs, rootdir, idPath, clusterId);
946      }
947      return clusterId;
948    } else {
949      LOG.warn("Cluster ID file does not exist at " + idPath.toString());
950    }
951    return clusterId;

1251  /**
1252   * Runs through the HBase rootdir and checks how many stores for each table
1253   * have more than one file in them. Checks -ROOT- and hbase:meta too. The total
1254   * percentage across all tables is stored under the special key "-TOTAL-".
1255   *
1256   * @param fs  The file system to use.
1257   * @param hbaseRootDir  The root directory to scan.
1258   * @return A map for each table and its percentage.
1259   * @throws IOException When scanning the directory fails.
1260   */
1261  public static Map<String, Integer> getTableFragmentation(
1262    final FileSystem fs, final Path hbaseRootDir)
1263  throws IOException {
1264    Map<String, Integer> frags = new HashMap<>();
1265    int cfCountTotal = 0;
1266    int cfFragTotal = 0;
1267    PathFilter regionFilter = new RegionDirFilter(fs);
1268    PathFilter familyFilter = new FamilyDirFilter(fs);
1269    List<Path> tableDirs = getTableDirs(fs, hbaseRootDir);
1270    for (Path d : tableDirs) {
1271      int cfCount = 0;
1272      int cfFrag = 0;
1273      FileStatus[] regionDirs = fs.listStatus(d, regionFilter);
1274      for (FileStatus regionDir : regionDirs) {
1275        Path dd = regionDir.getPath();
1276        // else its a region name, now look in region for families
1277        FileStatus[] familyDirs = fs.listStatus(dd, familyFilter);
1278        for (FileStatus familyDir : familyDirs) {
1279          cfCount++;
1280          cfCountTotal++;
1281          Path family = familyDir.getPath();
1282          // now in family make sure only one file
1283          FileStatus[] familyStatus = fs.listStatus(family);
1284          if (familyStatus.length > 1) {
1285            cfFrag++;
1286            cfFragTotal++;
1287          }
1288        }
1289      }
1290      // compute percentage per table and store in result list
1291      frags.put(FSUtils.getTableName(d).getNameAsString(),
1292        cfCount == 0? 0: Math.round((float) cfFrag / cfCount * 100));
1293    }
1294    // set overall percentage for all tables
1295    frags.put("-TOTAL-",
1296      cfCountTotal == 0? 0: Math.round((float) cfFragTotal / cfCountTotal * 100));
1297    return frags;
1298  }

1705  /**
1706   * Runs through the HBase rootdir/tablename and creates a reverse lookup map for
1707   * table StoreFile names to the full Path.  Note that because this method can be called
1708   * on a 'live' HBase system that we will skip files that no longer exist by the time
1709   * we traverse them and similarly the user of the result needs to consider that some
1710   * entries in this map may not exist by the time this call completes.
1711   * <br>
1712   * Example...<br>
1713   * Key = 3944417774205889744  <br>
1714   * Value = hdfs://localhost:51169/user/userid/-ROOT-/70236052/info/3944417774205889744
1715   *
1716   * @param resultMap map to add values.  If null, this method will create and populate one to return
1717   * @param fs  The file system to use.
1718   * @param hbaseRootDir  The root directory to scan.
1719   * @param tableName name of the table to scan.
1720   * @param sfFilter optional path filter to apply to store files
1721   * @param executor optional executor service to parallelize this operation
1722   * @param errors ErrorReporter instance or null
1723   * @return Map keyed by StoreFile name with a value of the full Path.
1724   * @throws IOException When scanning the directory fails.
1725   * @throws InterruptedException
1726   */
1727  public static Map<String, Path> getTableStoreFilePathMap(
1728      Map<String, Path> resultMap,
1729      final FileSystem fs, final Path hbaseRootDir, TableName tableName, final PathFilter sfFilter,
1730      ExecutorService executor, final ErrorReporter errors) throws IOException, InterruptedException {
1731
1732    final Map<String, Path> finalResultMap =
1733        resultMap == null ? new ConcurrentHashMap<>(128, 0.75f, 32) : resultMap;
1734
1735    // only include the directory paths to tables
1736    Path tableDir = FSUtils.getTableDir(hbaseRootDir, tableName);
1737    // Inside a table, there are compaction.dir directories to skip.  Otherwise, all else
1738    // should be regions.
1739    final FamilyDirFilter familyFilter = new FamilyDirFilter(fs);
1740    final Vector<Exception> exceptions = new Vector<>();
1741
1742    try {
1743      List<FileStatus> regionDirs = FSUtils.listStatusWithStatusFilter(fs, tableDir, new RegionDirFilter(fs));
1744      if (regionDirs == null) {
1745        return finalResultMap;
1746      }
1747
1748      final List<Future<?>> futures = new ArrayList<>(regionDirs.size());
1749
1750      for (FileStatus regionDir : regionDirs) {
1751        if (null != errors) {
1752          errors.progress();
1753        }
1754        final Path dd = regionDir.getPath();
1755
1756        if (!exceptions.isEmpty()) {
1757          break;
1758        }
1759
1760        Runnable getRegionStoreFileMapCall = new Runnable() {
1761          @Override
1762          public void run() {
1763            try {
1764              HashMap<String,Path> regionStoreFileMap = new HashMap<>();
1765              List<FileStatus> familyDirs = FSUtils.listStatusWithStatusFilter(fs, dd, familyFilter);
1766              if (familyDirs == null) {
1767                if (!fs.exists(dd)) {
1768                  LOG.warn("Skipping region because it no longer exists: " + dd);
1769                } else {
1770                  LOG.warn("Skipping region because it has no family dirs: " + dd);
1771                }
1772                return;
1773              }
1774              for (FileStatus familyDir : familyDirs) {
1775                if (null != errors) {
1776                  errors.progress();
1777                }
1778                Path family = familyDir.getPath();
1779                if (family.getName().equals(HConstants.RECOVERED_EDITS_DIR)) {
1780                  continue;
1781                }
1782                // now in family, iterate over the StoreFiles and
1783                // put in map
1784                FileStatus[] familyStatus = fs.listStatus(family);
1785                for (FileStatus sfStatus : familyStatus) {
1786                  if (null != errors) {
1787                    errors.progress();
1788                  }
1789                  Path sf = sfStatus.getPath();
1790                  if (sfFilter == null || sfFilter.accept(sf)) {
1791                    regionStoreFileMap.put( sf.getName(), sf);
1792                  }
1793                }
1794              }
1795              finalResultMap.putAll(regionStoreFileMap);
1796            } catch (Exception e) {
1797              LOG.error("Could not get region store file map for region: " + dd, e);
1798              exceptions.add(e);
1799            }
1800          }
1801        };
1802
1803        // If executor is available, submit async tasks to exec concurrently, otherwise
1804        // just do serial sync execution
1805        if (executor != null) {
1806          Future<?> future = executor.submit(getRegionStoreFileMapCall);
1807          futures.add(future);
1808        } else {
1809          FutureTask<?> future = new FutureTask<>(getRegionStoreFileMapCall, null);
1810          future.run();
1811          futures.add(future);
1812        }
1813      }
1814
1815      // Ensure all pending tasks are complete (or that we run into an exception)
1816      for (Future<?> f : futures) {
1817        if (!exceptions.isEmpty()) {
1818          break;
1819        }
1820        try {
1821          f.get();
1822        } catch (ExecutionException e) {
1823          LOG.error("Unexpected exec exception!  Should've been caught already.  (Bug?)", e);
1824          // Shouldn't happen, we already logged/caught any exceptions in the Runnable
1825        }
1826      }
1827    } catch (IOException e) {
1828      LOG.error("Cannot execute getTableStoreFilePathMap for " + tableName, e);
1829      exceptions.add(e);
1830    } finally {
1831      if (!exceptions.isEmpty()) {
1832        // Just throw the first exception as an indication something bad happened
1833        // Don't need to propagate all the exceptions, we already logged them all anyway
1834        Throwables.propagateIfInstanceOf(exceptions.firstElement(), IOException.class);
1835        throw Throwables.propagate(exceptions.firstElement());
1836      }
1837    }
1838
1839    return finalResultMap;
1840  }

2199   * mapping between the region name and its best locality region server or the
2200   * degree of locality of each region on each of the servers having at least
2201   * one block of that region. The output map parameters are both optional.
2202   *
2203   * @param conf
2204   *          the configuration to use
2205   * @param desiredTable
2206   *          the table you wish to scan locality for
2207   * @param threadPoolSize
2208   *          the thread pool size to use
2209   * @param regionToBestLocalityRSMapping
2210   *          the map into which to put the best locality mapping or null
2211   * @param regionDegreeLocalityMapping
2212   *          the map into which to put the locality degree mapping or null,
2213   *          must be a thread-safe implementation
2214   * @throws IOException
2215   *           in case of file system errors or interrupts
2216   */
2217  private static void getRegionLocalityMappingFromFS(
2218      final Configuration conf, final String desiredTable,
2219      int threadPoolSize,
2220      Map<String, String> regionToBestLocalityRSMapping,
2221      Map<String, Map<String, Float>> regionDegreeLocalityMapping)
2222      throws IOException {
2223    FileSystem fs =  FileSystem.get(conf);
2224    Path rootPath = FSUtils.getRootDir(conf);
2225    long startTime = EnvironmentEdgeManager.currentTime();
2226    Path queryPath;
2227    // The table files are in ${hbase.rootdir}/data/<namespace>/<table>/*
2228    if (null == desiredTable) {
2229      queryPath = new Path(new Path(rootPath, HConstants.BASE_NAMESPACE_DIR).toString() + "/*/*/*/");
2230    } else {
2231      queryPath = new Path(FSUtils.getTableDir(rootPath, TableName.valueOf(desiredTable)).toString() + "/*/");
2232    }
2233
2234    // reject all paths that are not appropriate
2235    PathFilter pathFilter = new PathFilter() {
2236      @Override
2237      public boolean accept(Path path) {
2238        // this is the region name; it may get some noise data
2239        if (null == path) {
2240          return false;
2241        }
2242
2243        // no parent?
2244        Path parent = path.getParent();
2245        if (null == parent) {
2246          return false;
2247        }
2248
2249        String regionName = path.getName();
2250        if (null == regionName) {
2251          return false;
2252        }
2253
2254        if (!regionName.toLowerCase(Locale.ROOT).matches("[0-9a-f]+")) {
2255          return false;
2256        }
2257        return true;
2258      }
2259    };
2260
2261    FileStatus[] statusList = fs.globStatus(queryPath, pathFilter);
2262
2263    if (null == statusList) {
2264      return;
2265    } else {
2266      LOG.debug("Query Path: " + queryPath + " ; # list of files: " +
2267          statusList.length);
2268    }
2269
2270    // lower the number of threads in case we have very few expected regions
2271    threadPoolSize = Math.min(threadPoolSize, statusList.length);
2272
2273    // run in multiple threads
2274    ThreadPoolExecutor tpe = new ThreadPoolExecutor(threadPoolSize,
2275        threadPoolSize, 60, TimeUnit.SECONDS,
2276        new ArrayBlockingQueue<>(statusList.length));
2277    try {
2278      // ignore all file status items that are not of interest
2279      for (FileStatus regionStatus : statusList) {
2280        if (null == regionStatus) {
2281          continue;
2282        }
2283
2284        if (!regionStatus.isDirectory()) {
2285          continue;
2286        }
2287
2288        Path regionPath = regionStatus.getPath();
2289        if (null == regionPath) {
2290          continue;
2291        }
2292
2293        tpe.execute(new FSRegionScanner(fs, regionPath,
2294            regionToBestLocalityRSMapping, regionDegreeLocalityMapping));
2295      }
2296    } finally {
2297      tpe.shutdown();
2298      int threadWakeFrequency = conf.getInt(HConstants.THREAD_WAKE_FREQUENCY,
2299          60 * 1000);
2300      try {
2301        // here we wait until TPE terminates, which is either naturally or by
2302        // exceptions in the execution of the threads
2303        while (!tpe.awaitTermination(threadWakeFrequency,
2304            TimeUnit.MILLISECONDS)) {
2305          // printing out rough estimate, so as to not introduce
2306          // AtomicInteger
2307          LOG.info("Locality checking is underway: { Scanned Regions : "
2308              + tpe.getCompletedTaskCount() + "/"
2309              + tpe.getTaskCount() + " }");
2310        }
2311      } catch (InterruptedException e) {
2312        throw (InterruptedIOException)new InterruptedIOException().initCause(e);
2313      }
2314    }
2315
2316    long overhead = EnvironmentEdgeManager.currentTime() - startTime;
2317    String overheadMsg = "Scan DFS for locality info takes " + overhead + " ms";
2318
2319    LOG.info(overheadMsg);
2320  }
