019package org.apache.hadoop.hbase.util.vint;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.OutputStream;
024
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026import org.apache.hadoop.hbase.nio.ByteBuff;
027
028/**
029 * Simple Variable Length Integer encoding.  Left bit of 0 means we are on the last byte.  If left
030 * bit of the current byte is 1, then there is at least one more byte.
031 */
032@InterfaceAudience.Private
033public class UVIntTool {
034
035  public static final byte
036    BYTE_7_RIGHT_BITS_SET = 127,
037    BYTE_LEFT_BIT_SET = -128;
038
039  public static final long
040    INT_7_RIGHT_BITS_SET = 127,
041    INT_8TH_BIT_SET = 128;
042
043  public static final byte[]
044    MAX_VALUE_BYTES = new byte[] { -1, -1, -1, -1, 7 };
045
046  /********************* int -&gt; bytes **************************/
047
048  public static int numBytes(int in) {
049    if (in == 0) {
050      // doesn't work with the formula below
051      return 1;
052    }
053    return (38 - Integer.numberOfLeadingZeros(in)) / 7;// 38 comes from 32+(7-1)
054  }
055
056  public static byte[] getBytes(int value) {
057    int numBytes = numBytes(value);
058    byte[] bytes = new byte[numBytes];
059    int remainder = value;
060    for (int i = 0; i < numBytes - 1; ++i) {
061      // set the left bit
062      bytes[i] = (byte) ((remainder & INT_7_RIGHT_BITS_SET) | INT_8TH_BIT_SET);
063      remainder >>= 7;
064    }
065    // do not set the left bit
066    bytes[numBytes - 1] = (byte) (remainder & INT_7_RIGHT_BITS_SET);
067    return bytes;
068  }
069
070  public static int writeBytes(int value, OutputStream os) throws IOException {
071    int numBytes = numBytes(value);
072    int remainder = value;
073    for (int i = 0; i < numBytes - 1; ++i) {
074      // set the left bit
075      os.write((byte) ((remainder & INT_7_RIGHT_BITS_SET) | INT_8TH_BIT_SET));
076      remainder >>= 7;
077    }
078    // do not set the left bit
079    os.write((byte) (remainder & INT_7_RIGHT_BITS_SET));
080    return numBytes;
081  }
082
083  /******************** bytes -&gt; int **************************/
084
085  public static int getInt(ByteBuff buffer, int offset) {
086    int value = 0;
087    for (int i = 0;; ++i) {
088      byte b = buffer.get(offset + i);
089      int shifted = BYTE_7_RIGHT_BITS_SET & b;// kill leftmost bit
090      shifted <<= 7 * i;
091      value |= shifted;
092      if (b >= 0) {
093        break;
094      }
095    }
096    return value;
097  }
098
099  public static int getInt(InputStream is) throws IOException {
100    int value = 0;
101    int i = 0;
102    int b;
103    do{
104      b = is.read();
105      int shifted = BYTE_7_RIGHT_BITS_SET & b;// kill leftmost bit
106      shifted <<= 7 * i;
107      value |= shifted;
108      ++i;
109    }while(b > Byte.MAX_VALUE);
110    return value;
111  }
112}
