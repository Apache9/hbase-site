018package org.apache.hadoop.hbase.util;
019
020import java.io.IOException;
021import java.util.HashMap;
022import java.util.List;
023import java.util.Map;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.TableName;
028import org.apache.hadoop.hbase.client.TableState;
029import org.apache.hadoop.hbase.exceptions.DeserializationException;
030import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
031import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos;
032import org.apache.hadoop.hbase.zookeeper.ZKUtil;
033import org.apache.hadoop.hbase.zookeeper.ZKWatcher;
034import org.apache.hadoop.hbase.zookeeper.ZNodePaths;
035import org.apache.yetus.audience.InterfaceAudience;
036import org.apache.zookeeper.KeeperException;
037
038/**
039 * utlity method to migrate zookeeper data across HBase versions.
040 */
041@InterfaceAudience.Private
042public class ZKDataMigrator {
043
044  private static final Log LOG = LogFactory.getLog(ZKDataMigrator.class);
045
046  /**
047   * Method for table states migration.
048   * Used when upgrading from pre-2.0 to 2.0
049   * Reading state from zk, applying them to internal state
050   * and delete.
051   * Used by master to clean migration from zk based states to
052   * table descriptor based states.
053   */
054  @Deprecated
055  public static Map<TableName, TableState.State> queryForTableStates(ZKWatcher zkw)
056      throws KeeperException, InterruptedException {
057    Map<TableName, TableState.State> rv = new HashMap<>();
058    List<String> children = ZKUtil.listChildrenNoWatch(zkw, zkw.znodePaths.tableZNode);
059    if (children == null)
060      return rv;
061    for (String child: children) {
062      TableName tableName = TableName.valueOf(child);
063      ZooKeeperProtos.DeprecatedTableState.State state = getTableState(zkw, tableName);
064      TableState.State newState = TableState.State.ENABLED;
065      if (state != null) {
066        switch (state) {
067        case ENABLED:
068          newState = TableState.State.ENABLED;
069          break;
070        case DISABLED:
071          newState = TableState.State.DISABLED;
072          break;
073        case DISABLING:
074          newState = TableState.State.DISABLING;
075          break;
076        case ENABLING:
077          newState = TableState.State.ENABLING;
078          break;
079        default:
080        }
081      }
082      rv.put(tableName, newState);
083    }
084    return rv;
085  }
086
087  /**
088   * Gets table state from ZK.
089   * @param zkw ZKWatcher instance to use
090   * @param tableName table we're checking
091   * @return Null or {@link ZooKeeperProtos.DeprecatedTableState.State} found in znode.
092   * @throws KeeperException
093   */
094  @Deprecated
095  private static  ZooKeeperProtos.DeprecatedTableState.State getTableState(
096          final ZKWatcher zkw, final TableName tableName)
097      throws KeeperException, InterruptedException {
098    String znode = ZNodePaths.joinZNode(zkw.znodePaths.tableZNode, tableName.getNameAsString());
099    byte [] data = ZKUtil.getData(zkw, znode);
100    if (data == null || data.length <= 0) return null;
101    try {
102      ProtobufUtil.expectPBMagicPrefix(data);
103      ZooKeeperProtos.DeprecatedTableState.Builder builder =
104          ZooKeeperProtos.DeprecatedTableState.newBuilder();
105      int magicLen = ProtobufUtil.lengthOfPBMagic();
106      ProtobufUtil.mergeFrom(builder, data, magicLen, data.length - magicLen);
107      return builder.getState();
108    } catch (IOException e) {
109      KeeperException ke = new KeeperException.DataInconsistencyException();
110      ke.initCause(e);
111      throw ke;
112    } catch (DeserializationException e) {
113      throw ZKUtil.convert(e);
114    }
115  }
116
117}
