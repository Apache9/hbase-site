126  /**
127   * Sets storage policy for given path according to config setting.
128   * If the passed path is a directory, we'll set the storage policy for all files
129   * created in the future in said directory. Note that this change in storage
130   * policy takes place at the HDFS level; it will persist beyond this RS's lifecycle.
131   * If we're running on a version of HDFS that doesn't support the given storage policy
132   * (or storage policies at all), then we'll issue a log message and continue.
133   *
134   * See http://hadoop.apache.org/docs/r2.6.0/hadoop-project-dist/hadoop-hdfs/ArchivalStorage.html
135   *
136   * @param fs We only do anything if an instance of DistributedFileSystem
137   * @param conf used to look up storage policy with given key; not modified.
138   * @param path the Path whose storage policy is to be set
139   * @param policyKey Key to use pulling a policy from Configuration:
140   * e.g. HConstants.WAL_STORAGE_POLICY (hbase.wal.storage.policy).
141   * @param defaultPolicy usually should be the policy NONE to delegate to HDFS
142   */
143  public static void setStoragePolicy(final FileSystem fs, final Configuration conf,
144      final Path path, final String policyKey, final String defaultPolicy) {
145    String storagePolicy = conf.get(policyKey, defaultPolicy).toUpperCase(Locale.ROOT);
146    if (storagePolicy.equals(defaultPolicy)) {
147      if (LOG.isTraceEnabled()) {
148        LOG.trace("default policy of " + defaultPolicy + " requested, exiting early.");
149      }
150      return;
151    }
152    setStoragePolicy(fs, path, storagePolicy);
153  }

155  /**
156   * Sets storage policy for given path.
157   * If the passed path is a directory, we'll set the storage policy for all files
158   * created in the future in said directory. Note that this change in storage
159   * policy takes place at the HDFS level; it will persist beyond this RS's lifecycle.
160   * If we're running on a version of HDFS that doesn't support the given storage policy
161   * (or storage policies at all), then we'll issue a log message and continue.
162   *
163   * See http://hadoop.apache.org/docs/r2.6.0/hadoop-project-dist/hadoop-hdfs/ArchivalStorage.html
164   *
165   * @param fs We only do anything if an instance of DistributedFileSystem
166   * @param path the Path whose storage policy is to be set
167   * @param storagePolicy Policy to set on <code>path</code>; see hadoop 2.6+
168   * org.apache.hadoop.hdfs.protocol.HdfsConstants for possible list e.g
169   * 'COLD', 'WARM', 'HOT', 'ONE_SSD', 'ALL_SSD', 'LAZY_PERSIST'.
170   */
171  public static void setStoragePolicy(final FileSystem fs, final Path path,
172      final String storagePolicy) {
173    if (storagePolicy == null) {
174      if (LOG.isTraceEnabled()) {
175        LOG.trace("We were passed a null storagePolicy, exiting early.");
176      }
177      return;
178    }
179    final String trimmedStoragePolicy = storagePolicy.trim();
180    if (trimmedStoragePolicy.isEmpty()) {
181      if (LOG.isTraceEnabled()) {
182        LOG.trace("We were passed an empty storagePolicy, exiting early.");
183      }
184      return;
185    }
186    boolean distributed = false;
187    try {
188      distributed = isDistributedFileSystem(fs);
189    } catch (IOException ioe) {
190      // This should NEVER happen.
191      LOG.warn("Failed setStoragePolicy=" + trimmedStoragePolicy + " on path=" +
192          path + "; failed isDFS test", ioe);
193      return;
194    }
195    if (distributed) {
196      invokeSetStoragePolicy(fs, path, trimmedStoragePolicy);
197    } else {
198      LOG.info("FileSystem isn't an instance of DistributedFileSystem; presuming it doesn't " +
199          "support setStoragePolicy. Unable to set storagePolicy=" + trimmedStoragePolicy +
200          " on path=" + path);
201    }
202  }

204  /*
205   * All args have been checked and are good. Run the setStoragePolicy invocation.
206   */
207  private static void invokeSetStoragePolicy(final FileSystem fs, final Path path,
208      final String storagePolicy) {
209    Method m = null;
210    try {
211      m = fs.getClass().getDeclaredMethod("setStoragePolicy",
212          new Class<?>[] { Path.class, String.class });
213      m.setAccessible(true);
214    } catch (NoSuchMethodException e) {
215      LOG.info("FileSystem doesn't support setStoragePolicy; HDFS-6584 not available "
216          + "(hadoop-2.6.0+): " + e.getMessage());
217    } catch (SecurityException e) {
218      LOG.info("Don't have access to setStoragePolicy on FileSystems; HDFS-6584 not available "
219          + "(hadoop-2.6.0+): ", e);
220      m = null; // could happen on setAccessible()
221    }
222    if (m != null) {
223      try {
224        m.invoke(fs, path, storagePolicy);
225        LOG.info("Set storagePolicy=" + storagePolicy + " for path=" + path);
226      } catch (Exception e) {
227        // check for lack of HDFS-7228
228        boolean probablyBadPolicy = false;
229        if (e instanceof InvocationTargetException) {
230          final Throwable exception = e.getCause();
231          if (exception instanceof RemoteException &&
232              HadoopIllegalArgumentException.class.getName().equals(
233                  ((RemoteException)exception).getClassName())) {
234            LOG.warn("Given storage policy, '" + storagePolicy + "', was rejected and probably " +
235                "isn't a valid policy for the version of Hadoop you're running. I.e. if you're " +
236                "trying to use SSD related policies then you're likely missing HDFS-7228. For " +
237                "more information see the 'ArchivalStorage' docs for your Hadoop release.");
238            LOG.debug("More information about the invalid storage policy.", exception);
239            probablyBadPolicy = true;
240          }
241        }
242        if (!probablyBadPolicy) {
243          // This swallows FNFE, should we be throwing it? seems more likely to indicate dev
244          // misuse than a runtime problem with HDFS.
245          LOG.warn("Unable to set storagePolicy=" + storagePolicy + " for path=" + path, e);
246        }
247      }
248    }
249  }

291  /**
292   * Compare path component of the Path URI; e.g. if hdfs://a/b/c and /a/b/c, it will compare the
293   * '/a/b/c' part. If you passed in 'hdfs://a/b/c and b/c, it would return true.  Does not consider
294   * schema; i.e. if schemas different but path or subpath matches, the two will equate.
295   * @param pathToSearch Path we will be trying to match.
296   * @param pathTail
297   * @return True if <code>pathTail</code> is tail on the path of <code>pathToSearch</code>
298   */
299  public static boolean isMatchingTail(final Path pathToSearch, final Path pathTail) {
300    if (pathToSearch.depth() != pathTail.depth()) return false;
301    Path tailPath = pathTail;
302    String tailName;
303    Path toSearch = pathToSearch;
304    String toSearchName;
305    boolean result = false;
306    do {
307      tailName = tailPath.getName();
308      if (tailName == null || tailName.length() <= 0) {
309        result = true;
310        break;
311      }
312      toSearchName = toSearch.getName();
313      if (toSearchName == null || toSearchName.length() <= 0) break;
314      // Move up a parent on each path for next go around.  Path doesn't let us go off the end.
315      tailPath = tailPath.getParent();
316      toSearch = toSearch.getParent();
317    } while(tailName.equals(toSearchName));
318    return result;
319  }

443  /**
444   * Create the specified file on the filesystem. By default, this will:
445   * <ol>
446   * <li>overwrite the file if it exists</li>
447   * <li>apply the umask in the configuration (if it is enabled)</li>
448   * <li>use the fs configured buffer size (or 4096 if not set)</li>
449   * <li>use the configured column family replication or default replication if
450   * {@link HColumnDescriptor#DEFAULT_DFS_REPLICATION}</li>
451   * <li>use the default block size</li>
452   * <li>not track progress</li>
453   * </ol>
454   * @param conf configurations
455   * @param fs {@link FileSystem} on which to write the file
456   * @param path {@link Path} to the file to write
457   * @param perm permissions
458   * @param favoredNodes
459   * @return output stream to the created file
460   * @throws IOException if the file cannot be created
461   */
462  public static FSDataOutputStream create(Configuration conf, FileSystem fs, Path path,
463      FsPermission perm, InetSocketAddress[] favoredNodes) throws IOException {
464    if (fs instanceof HFileSystem) {
465      FileSystem backingFs = ((HFileSystem)fs).getBackingFs();
466      if (backingFs instanceof DistributedFileSystem) {
467        // Try to use the favoredNodes version via reflection to allow backwards-
468        // compatibility.
469        short replication = Short.parseShort(conf.get(HColumnDescriptor.DFS_REPLICATION,
470          String.valueOf(HColumnDescriptor.DEFAULT_DFS_REPLICATION)));
471        try {
472          return (FSDataOutputStream) (DistributedFileSystem.class.getDeclaredMethod("create",
473            Path.class, FsPermission.class, boolean.class, int.class, short.class, long.class,
474            Progressable.class, InetSocketAddress[].class).invoke(backingFs, path, perm, true,
475            getDefaultBufferSize(backingFs),
476            replication > 0 ? replication : getDefaultReplication(backingFs, path),
477            getDefaultBlockSize(backingFs, path), null, favoredNodes));
478        } catch (InvocationTargetException ite) {
479          // Function was properly called, but threw it's own exception.
480          throw new IOException(ite.getCause());
481        } catch (NoSuchMethodException e) {
482          LOG.debug("DFS Client does not support most favored nodes create; using default create");
483          if (LOG.isTraceEnabled()) LOG.trace("Ignoring; use default create", e);
484        } catch (IllegalArgumentException e) {
485          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
486        } catch (SecurityException e) {
487          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
488        } catch (IllegalAccessException e) {
489          LOG.debug("Ignoring (most likely Reflection related exception) " + e);
490        }
491      }
492    }
493    return create(fs, path, perm, true);
494  }

522  /**
523   * Get the file permissions specified in the configuration, if they are
524   * enabled.
525   *
526   * @param fs filesystem that the file will be created on.
527   * @param conf configuration to read for determining if permissions are
528   *          enabled and which to use
529   * @param permssionConfKey property key in the configuration to use when
530   *          finding the permission
531   * @return the permission to use when creating a new file on the fs. If
532   *         special permissions are not specified in the configuration, then
533   *         the default permissions on the the fs will be returned.
534   */
535  public static FsPermission getFilePermissions(final FileSystem fs,
536      final Configuration conf, final String permssionConfKey) {
537    boolean enablePermissions = conf.getBoolean(
538        HConstants.ENABLE_DATA_FILE_UMASK, false);
539
540    if (enablePermissions) {
541      try {
542        FsPermission perm = new FsPermission(FULL_RWX_PERMISSIONS);
543        // make sure that we have a mask, if not, go default.
544        String mask = conf.get(permssionConfKey);
545        if (mask == null)
546          return FsPermission.getFileDefault();
547        // appy the umask
548        FsPermission umask = new FsPermission(mask);
549        return perm.applyUMask(umask);
550      } catch (IllegalArgumentException e) {
551        LOG.warn(
552            "Incorrect umask attempted to be created: "
553                + conf.get(permssionConfKey)
554                + ", using default file permissions.", e);
555        return FsPermission.getFileDefault();
556      }
557    }
558    return FsPermission.getFileDefault();
559  }

561  /**
562   * Checks to see if the specified file system is available
563   *
564   * @param fs filesystem
565   * @throws IOException e
566   */
567  public static void checkFileSystemAvailable(final FileSystem fs)
568  throws IOException {
569    if (!(fs instanceof DistributedFileSystem)) {
570      return;
571    }
572    IOException exception = null;
573    DistributedFileSystem dfs = (DistributedFileSystem) fs;
574    try {
575      if (dfs.exists(new Path("/"))) {
576        return;
577      }
578    } catch (IOException e) {
579      exception = e instanceof RemoteException ?
580              ((RemoteException)e).unwrapRemoteException() : e;
581    }
582    try {
583      fs.close();
584    } catch (Exception e) {
585      LOG.error("file system close failed: ", e);
586    }
587    IOException io = new IOException("File system is not available");
588    io.initCause(exception);
589    throw io;
590  }

592  /**
593   * We use reflection because {@link DistributedFileSystem#setSafeMode(
594   * HdfsConstants.SafeModeAction action, boolean isChecked)} is not in hadoop 1.1
595   *
596   * @param dfs
597   * @return whether we're in safe mode
598   * @throws IOException
599   */
600  private static boolean isInSafeMode(DistributedFileSystem dfs) throws IOException {
601    boolean inSafeMode = false;
602    try {
603      Method m = DistributedFileSystem.class.getMethod("setSafeMode", new Class<?> []{
604          org.apache.hadoop.hdfs.protocol.HdfsConstants.SafeModeAction.class, boolean.class});
605      inSafeMode = (Boolean) m.invoke(dfs,

636   * Verifies current version of file system
637   *
638   * @param fs filesystem object
639   * @param rootdir root hbase directory
640   * @return null if no version file exists, version string otherwise.
641   * @throws IOException e
642   * @throws org.apache.hadoop.hbase.exceptions.DeserializationException
643   */
644  public static String getVersion(FileSystem fs, Path rootdir)
645  throws IOException, DeserializationException {
646    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
647    FileStatus[] status = null;
648    try {
649      // hadoop 2.0 throws FNFE if directory does not exist.
650      // hadoop 1.0 returns null if directory does not exist.
651      status = fs.listStatus(versionFile);
652    } catch (FileNotFoundException fnfe) {
653      return null;
654    }
655    if (status == null || status.length == 0) return null;
656    String version = null;
657    byte [] content = new byte [(int)status[0].getLen()];
658    FSDataInputStream s = fs.open(versionFile);
659    try {
660      IOUtils.readFully(s, content, 0, content.length);
661      if (ProtobufUtil.isPBMagicPrefix(content)) {
662        version = parseVersionFrom(content);
663      } else {
664        // Presume it pre-pb format.
665        InputStream is = new ByteArrayInputStream(content);
666        DataInputStream dis = new DataInputStream(is);
667        try {
668          version = dis.readUTF();
669        } finally {
670          dis.close();
671        }
672      }
673    } catch (EOFException eof) {
674      LOG.warn("Version file was empty, odd, will try to set it.");
675    } finally {
676      s.close();
677    }
678    return version;
679  }

796   * Sets version of file system
797   *
798   * @param fs filesystem object
799   * @param rootdir hbase root directory
800   * @param version version to set
801   * @param wait time to wait for retry
802   * @param retries number of times to retry before throwing an IOException
803   * @throws IOException e
804   */
805  public static void setVersion(FileSystem fs, Path rootdir, String version,
806      int wait, int retries) throws IOException {
807    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
808    Path tempVersionFile = new Path(rootdir, HConstants.HBASE_TEMP_DIRECTORY + Path.SEPARATOR +
809      HConstants.VERSION_FILE_NAME);
810    while (true) {
811      try {
812        // Write the version to a temporary file
813        FSDataOutputStream s = fs.create(tempVersionFile);
814        try {
815          s.write(toVersionByteArray(version));
816          s.close();
817          s = null;
818          // Move the temp version file to its normal location. Returns false
819          // if the rename failed. Throw an IOE in that case.
820          if (!fs.rename(tempVersionFile, versionFile)) {
821            throw new IOException("Unable to move temp version file to " + versionFile);
822          }
823        } finally {
824          // Cleaning up the temporary if the rename failed would be trying
825          // too hard. We'll unconditionally create it again the next time
826          // through anyway, files are overwritten by default by create().
827
828          // Attempt to close the stream on the way out if it is still open.
829          try {
830            if (s != null) s.close();
831          } catch (IOException ignore) { }
832        }
833        LOG.info("Created version file at " + rootdir.toString() + " with version=" + version);
834        return;
835      } catch (IOException e) {
836        if (retries > 0) {
837          LOG.debug("Unable to create version file at " + rootdir.toString() + ", retrying", e);
838          fs.delete(versionFile, false);
839          try {
840            if (wait > 0) {
841              Thread.sleep(wait);
842            }
843          } catch (InterruptedException ie) {
844            throw (InterruptedIOException)new InterruptedIOException().initCause(ie);
845          }
846          retries--;
847        } else {
848          throw e;
849        }
850      }
851    }
852  }

887   * @param rootdir the path to the HBase root directory
888   * @return the unique cluster identifier
889   * @throws IOException if reading the cluster ID file fails
890   */
891  public static ClusterId getClusterId(FileSystem fs, Path rootdir)
892  throws IOException {
893    Path idPath = new Path(rootdir, HConstants.CLUSTER_ID_FILE_NAME);
894    ClusterId clusterId = null;
895    FileStatus status = fs.exists(idPath)? fs.getFileStatus(idPath):  null;
896    if (status != null) {
897      int len = Ints.checkedCast(status.getLen());
898      byte [] content = new byte[len];
899      FSDataInputStream in = fs.open(idPath);
900      try {
901        in.readFully(content);
902      } catch (EOFException eof) {
903        LOG.warn("Cluster ID file " + idPath.toString() + " was empty");
904      } finally{
905        in.close();
906      }
907      try {
908        clusterId = ClusterId.parseFrom(content);
909      } catch (DeserializationException e) {
910        throw new IOException("content=" + Bytes.toString(content), e);
911      }
912      // If not pb'd, make it so.
913      if (!ProtobufUtil.isPBMagicPrefix(content)) {
914        String cid = null;
915        in = fs.open(idPath);
916        try {
917          cid = in.readUTF();
918          clusterId = new ClusterId(cid);
919        } catch (EOFException eof) {
920          LOG.warn("Cluster ID file " + idPath.toString() + " was empty");
921        } finally {
922          in.close();
923        }
924        rewriteAsPb(fs, rootdir, idPath, clusterId);
925      }
926      return clusterId;
927    } else {
928      LOG.warn("Cluster ID file does not exist at " + idPath.toString());
929    }
930    return clusterId;
931  }

1245    int cfFragTotal = 0;
1246    PathFilter regionFilter = new RegionDirFilter(fs);
1247    PathFilter familyFilter = new FamilyDirFilter(fs);
1248    List<Path> tableDirs = getTableDirs(fs, hbaseRootDir);
1249    for (Path d : tableDirs) {
1250      int cfCount = 0;
1251      int cfFrag = 0;
1252      FileStatus[] regionDirs = fs.listStatus(d, regionFilter);
1253      for (FileStatus regionDir : regionDirs) {
1254        Path dd = regionDir.getPath();
1255        // else its a region name, now look in region for families
1256        FileStatus[] familyDirs = fs.listStatus(dd, familyFilter);
1257        for (FileStatus familyDir : familyDirs) {
1258          cfCount++;
1259          cfCountTotal++;
1260          Path family = familyDir.getPath();
1261          // now in family make sure only one file
1262          FileStatus[] familyStatus = fs.listStatus(family);
1263          if (familyStatus.length > 1) {
1264            cfFrag++;
1265            cfFragTotal++;
1266          }
1267        }
1268      }
1269      // compute percentage per table and store in result list
1270      frags.put(FSUtils.getTableName(d).getNameAsString(),
1271        cfCount == 0? 0: Math.round((float) cfFrag / cfCount * 100));
1272    }
1273    // set overall percentage for all tables
1274    frags.put("-TOTAL-",
1275      cfCountTotal == 0? 0: Math.round((float) cfFragTotal / cfCountTotal * 100));
1276    return frags;
1277  }

1684  /**
1685   * Runs through the HBase rootdir/tablename and creates a reverse lookup map for
1686   * table StoreFile names to the full Path.  Note that because this method can be called
1687   * on a 'live' HBase system that we will skip files that no longer exist by the time
1688   * we traverse them and similarly the user of the result needs to consider that some
1689   * entries in this map may not exist by the time this call completes.
1690   * <br>
1691   * Example...<br>
1692   * Key = 3944417774205889744  <br>
1693   * Value = hdfs://localhost:51169/user/userid/-ROOT-/70236052/info/3944417774205889744
1694   *
1695   * @param resultMap map to add values.  If null, this method will create and populate one to return
1696   * @param fs  The file system to use.
1697   * @param hbaseRootDir  The root directory to scan.
1698   * @param tableName name of the table to scan.
1699   * @param sfFilter optional path filter to apply to store files
1700   * @param executor optional executor service to parallelize this operation
1701   * @param errors ErrorReporter instance or null
1702   * @return Map keyed by StoreFile name with a value of the full Path.
1703   * @throws IOException When scanning the directory fails.
1704   * @throws InterruptedException
1705   */
1706  public static Map<String, Path> getTableStoreFilePathMap(
1707      Map<String, Path> resultMap,
1708      final FileSystem fs, final Path hbaseRootDir, TableName tableName, final PathFilter sfFilter,
1709      ExecutorService executor, final ErrorReporter errors) throws IOException, InterruptedException {
1710
1711    final Map<String, Path> finalResultMap =
1712        resultMap == null ? new ConcurrentHashMap<>(128, 0.75f, 32) : resultMap;
1713
1714    // only include the directory paths to tables
1715    Path tableDir = FSUtils.getTableDir(hbaseRootDir, tableName);
1716    // Inside a table, there are compaction.dir directories to skip.  Otherwise, all else
1717    // should be regions.
1718    final FamilyDirFilter familyFilter = new FamilyDirFilter(fs);
1719    final Vector<Exception> exceptions = new Vector<>();
1720
1721    try {
1722      List<FileStatus> regionDirs = FSUtils.listStatusWithStatusFilter(fs, tableDir, new RegionDirFilter(fs));
1723      if (regionDirs == null) {
1724        return finalResultMap;
1725      }
1726
1727      final List<Future<?>> futures = new ArrayList<>(regionDirs.size());
1728
1729      for (FileStatus regionDir : regionDirs) {
1730        if (null != errors) {
1731          errors.progress();
1732        }
1733        final Path dd = regionDir.getPath();
1734
1735        if (!exceptions.isEmpty()) {
1736          break;
1737        }
1738
1739        Runnable getRegionStoreFileMapCall = new Runnable() {
1740          @Override
1741          public void run() {
1742            try {
1743              HashMap<String,Path> regionStoreFileMap = new HashMap<>();
1744              List<FileStatus> familyDirs = FSUtils.listStatusWithStatusFilter(fs, dd, familyFilter);
1745              if (familyDirs == null) {
1746                if (!fs.exists(dd)) {
1747                  LOG.warn("Skipping region because it no longer exists: " + dd);
1748                } else {
1749                  LOG.warn("Skipping region because it has no family dirs: " + dd);
1750                }
1751                return;
1752              }
1753              for (FileStatus familyDir : familyDirs) {
1754                if (null != errors) {
1755                  errors.progress();
1756                }
1757                Path family = familyDir.getPath();
1758                if (family.getName().equals(HConstants.RECOVERED_EDITS_DIR)) {
1759                  continue;
1760                }
1761                // now in family, iterate over the StoreFiles and
1762                // put in map
1763                FileStatus[] familyStatus = fs.listStatus(family);
1764                for (FileStatus sfStatus : familyStatus) {
1765                  if (null != errors) {
1766                    errors.progress();
1767                  }
1768                  Path sf = sfStatus.getPath();
1769                  if (sfFilter == null || sfFilter.accept(sf)) {
1770                    regionStoreFileMap.put( sf.getName(), sf);
1771                  }
1772                }
1773              }
1774              finalResultMap.putAll(regionStoreFileMap);
1775            } catch (Exception e) {
1776              LOG.error("Could not get region store file map for region: " + dd, e);
1777              exceptions.add(e);
1778            }
1779          }
1780        };
1781
1782        // If executor is available, submit async tasks to exec concurrently, otherwise
1783        // just do serial sync execution
1784        if (executor != null) {
1785          Future<?> future = executor.submit(getRegionStoreFileMapCall);
1786          futures.add(future);
1787        } else {
1788          FutureTask<?> future = new FutureTask<>(getRegionStoreFileMapCall, null);
1789          future.run();
1790          futures.add(future);
1791        }
1792      }
1793
1794      // Ensure all pending tasks are complete (or that we run into an exception)
1795      for (Future<?> f : futures) {
1796        if (!exceptions.isEmpty()) {
1797          break;
1798        }
1799        try {
1800          f.get();
1801        } catch (ExecutionException e) {
1802          LOG.error("Unexpected exec exception!  Should've been caught already.  (Bug?)", e);
1803          // Shouldn't happen, we already logged/caught any exceptions in the Runnable
1804        }
1805      }
1806    } catch (IOException e) {
1807      LOG.error("Cannot execute getTableStoreFilePathMap for " + tableName, e);
1808      exceptions.add(e);
1809    } finally {
1810      if (!exceptions.isEmpty()) {
1811        // Just throw the first exception as an indication something bad happened
1812        // Don't need to propagate all the exceptions, we already logged them all anyway
1813        Throwables.propagateIfInstanceOf(exceptions.firstElement(), IOException.class);
1814        throw Throwables.propagate(exceptions.firstElement());
1815      }
1816    }
1817
1818    return finalResultMap;
1819  }

2181   *

2183   *          the configuration to use
2184   * @param desiredTable
2185   *          the table you wish to scan locality for
2186   * @param threadPoolSize
2187   *          the thread pool size to use
2188   * @param regionToBestLocalityRSMapping
2189   *          the map into which to put the best locality mapping or null
2190   * @param regionDegreeLocalityMapping
2191   *          the map into which to put the locality degree mapping or null,
2192   *          must be a thread-safe implementation
2193   * @throws IOException
2194   *           in case of file system errors or interrupts
2195   */
2196  private static void getRegionLocalityMappingFromFS(
2197      final Configuration conf, final String desiredTable,
2198      int threadPoolSize,
2199      Map<String, String> regionToBestLocalityRSMapping,
2200      Map<String, Map<String, Float>> regionDegreeLocalityMapping)
2201      throws IOException {
2202    FileSystem fs =  FileSystem.get(conf);
2203    Path rootPath = FSUtils.getRootDir(conf);
2204    long startTime = EnvironmentEdgeManager.currentTime();
2205    Path queryPath;
2206    // The table files are in ${hbase.rootdir}/data/<namespace>/<table>/*
2207    if (null == desiredTable) {
2208      queryPath = new Path(new Path(rootPath, HConstants.BASE_NAMESPACE_DIR).toString() + "/*/*/*/");
2209    } else {
2210      queryPath = new Path(FSUtils.getTableDir(rootPath, TableName.valueOf(desiredTable)).toString() + "/*/");
2211    }
2212
2213    // reject all paths that are not appropriate
2214    PathFilter pathFilter = new PathFilter() {
2215      @Override
2216      public boolean accept(Path path) {
2217        // this is the region name; it may get some noise data
2218        if (null == path) {
2219          return false;
2220        }
2221
2222        // no parent?
2223        Path parent = path.getParent();
2224        if (null == parent) {
2225          return false;
2226        }
2227
2228        String regionName = path.getName();
2229        if (null == regionName) {
2230          return false;
2231        }
2232
2233        if (!regionName.toLowerCase(Locale.ROOT).matches("[0-9a-f]+")) {
2234          return false;
2235        }
2236        return true;
2237      }
2238    };
2239
2240    FileStatus[] statusList = fs.globStatus(queryPath, pathFilter);
2241
2242    if (null == statusList) {
2243      return;
2244    } else {
2245      LOG.debug("Query Path: " + queryPath + " ; # list of files: " +
2246          statusList.length);
2247    }
2248
2249    // lower the number of threads in case we have very few expected regions
2250    threadPoolSize = Math.min(threadPoolSize, statusList.length);
2251
2252    // run in multiple threads
2253    ThreadPoolExecutor tpe = new ThreadPoolExecutor(threadPoolSize,
2254        threadPoolSize, 60, TimeUnit.SECONDS,
2255        new ArrayBlockingQueue<>(statusList.length));
2256    try {
2257      // ignore all file status items that are not of interest
2258      for (FileStatus regionStatus : statusList) {
2259        if (null == regionStatus) {

2261        }
2262
2263        if (!regionStatus.isDirectory()) {
2264          continue;
2265        }

2268        if (null == regionPath) {
2269          continue;
2270        }
2271
2272        tpe.execute(new FSRegionScanner(fs, regionPath,
2273            regionToBestLocalityRSMapping, regionDegreeLocalityMapping));
2274      }
2275    } finally {
2276      tpe.shutdown();

2278          60 * 1000);
2279      try {
2280        // here we wait until TPE terminates, which is either naturally or by
2281        // exceptions in the execution of the threads
2282        while (!tpe.awaitTermination(threadWakeFrequency,
2283            TimeUnit.MILLISECONDS)) {
2284          // printing out rough estimate, so as to not introduce
2285          // AtomicInteger
2286          LOG.info("Locality checking is underway: { Scanned Regions : "
2287              + tpe.getCompletedTaskCount() + "/"
2288              + tpe.getTaskCount() + " }");
2289        }
2290      } catch (InterruptedException e) {
2291        throw (InterruptedIOException)new InterruptedIOException().initCause(e);
2292      }
2293    }

2297
2298    LOG.info(overheadMsg);
