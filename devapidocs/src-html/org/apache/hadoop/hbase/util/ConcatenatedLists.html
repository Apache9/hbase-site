019package org.apache.hadoop.hbase.util;
020
021import java.lang.reflect.Array;
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.List;
025import java.util.NoSuchElementException;
026
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028
029/**
030 * A collection class that contains multiple sub-lists, which allows us to not copy lists.
031 * This class does not support modification. The derived classes that add modifications are
032 * not thread-safe.
033 * NOTE: Doesn't implement list as it is not necessary for current usage, feel free to add.
034 */
035@InterfaceAudience.Private
036public class ConcatenatedLists<T> implements Collection<T> {
037  protected final ArrayList<List<T>> components = new ArrayList<List<T>>();
038  protected int size = 0;
039
040  public void addAllSublists(List<? extends List<T>> items) {
041    for (List<T> list : items) {
042      addSublist(list);
043    }
044  }
045
046  public void addSublist(List<T> items) {
047    if (!items.isEmpty()) {
048      this.components.add(items);
049      this.size += items.size();
050    }
051  }
052
053  @Override
054  public int size() {
055    return this.size;
056  }
057
058  @Override
059  public boolean isEmpty() {
060    return this.size == 0;
061  }
062
063  @Override
064  public boolean contains(Object o) {
065    for (List<T> component : this.components) {
066      if (component.contains(o)) return true;
067    }
068    return false;
069  }
070
071  @Override
072  public boolean containsAll(Collection<?> c) {
073    for (Object o : c) {
074      if (!contains(o)) return false;
075    }
076    return true;
077  }
078
079  @Override
080  public Object[] toArray() {
081    return toArray((Object[])Array.newInstance(Object.class, this.size));
082  }
083
084  @Override
085  @SuppressWarnings("unchecked")
086  public <U> U[] toArray(U[] a) {
087    U[] result = (a.length == this.size()) ? a
088        : (U[])Array.newInstance(a.getClass().getComponentType(), this.size);
089    int i = 0;
090    for (List<T> component : this.components) {
091      for (T t : component) {
092        result[i] = (U)t;
093        ++i;
094      }
095    }
096    return result;
097  }
098
099  @Override
100  public boolean add(T e) {
101    throw new UnsupportedOperationException();
102  }
103
104  @Override
105  public boolean remove(Object o) {
106    throw new UnsupportedOperationException();
107  }
108
109  @Override
110  public boolean addAll(Collection<? extends T> c) {
111    throw new UnsupportedOperationException();
112  }
113
114  @Override
115  public boolean removeAll(Collection<?> c) {
116    throw new UnsupportedOperationException();
117  }
118
119  @Override
120  public boolean retainAll(Collection<?> c) {
121    throw new UnsupportedOperationException();
122  }
123
124  @Override
125  public void clear() {
126    throw new UnsupportedOperationException();
127  }
128
129  @Override
130  public java.util.Iterator<T> iterator() {
131    return new Iterator();
132  }
133
134  @edu.umd.cs.findbugs.annotations.SuppressWarnings(
135    value="URF_UNREAD_PUBLIC_OR_PROTECTED_FIELD",
136    justification="nextWasCalled is using by StripeStoreFileManager")
137  public class Iterator implements java.util.Iterator<T> {
138    protected int currentComponent = 0;
139    protected int indexWithinComponent = -1;
140    protected boolean nextWasCalled = false;
141
142    @Override
143    public boolean hasNext() {
144      return (currentComponent + 1) < components.size()
145          || ((currentComponent + 1) == components.size()
146              && ((indexWithinComponent + 1) < components.get(currentComponent).size()));
147    }
148
149    @Override
150    public T next() {
151      if (!components.isEmpty()) {
152        this.nextWasCalled = true;
153        List<T> src = components.get(currentComponent);
154        if (++indexWithinComponent < src.size()) return src.get(indexWithinComponent);
155        if (++currentComponent < components.size()) {
156          indexWithinComponent = 0;
157          src = components.get(currentComponent);
158          assert src.size() > 0;
159          return src.get(indexWithinComponent);
160        }
161      }
162      this.nextWasCalled = false;
163      throw new NoSuchElementException();
164    }
165
166    @Override
167    public void remove() {
168      throw new UnsupportedOperationException();
169    }
170  }
171}
