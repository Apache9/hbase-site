019package org.apache.hadoop.hbase.util;
020
021import java.nio.ByteBuffer;
022import java.util.concurrent.locks.Lock;
023import java.util.concurrent.locks.ReentrantLock;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.nio.ByteBuff;
029import org.apache.hadoop.hbase.nio.MultiByteBuff;
030import org.apache.hadoop.hbase.nio.SingleByteBuff;
031import org.apache.hadoop.util.StringUtils;
032
033/**
034 * This class manages an array of ByteBuffers with a default size 4MB. These
035 * buffers are sequential and could be considered as a large buffer.It supports
036 * reading/writing data from this large buffer with a position and offset
037 */
038@InterfaceAudience.Private
039public final class ByteBufferArray {
040  private static final Log LOG = LogFactory.getLog(ByteBufferArray.class);
041
042  static final int DEFAULT_BUFFER_SIZE = 4 * 1024 * 1024;
043  private ByteBuffer buffers[];
044  private Lock locks[];
045  private int bufferSize;
046  private int bufferCount;
047
048  /**
049   * We allocate a number of byte buffers as the capacity. In order not to out
050   * of the array bounds for the last byte(see {@link ByteBufferArray#multiple}),
051   * we will allocate one additional buffer with capacity 0;
052   * @param capacity total size of the byte buffer array
053   * @param directByteBuffer true if we allocate direct buffer
054   */
055  public ByteBufferArray(long capacity, boolean directByteBuffer) {
056    this.bufferSize = DEFAULT_BUFFER_SIZE;
057    if (this.bufferSize > (capacity / 16))
058      this.bufferSize = (int) roundUp(capacity / 16, 32768);
059    this.bufferCount = (int) (roundUp(capacity, bufferSize) / bufferSize);
060    LOG.info("Allocating buffers total=" + StringUtils.byteDesc(capacity)
061        + ", sizePerBuffer=" + StringUtils.byteDesc(bufferSize) + ", count="
062        + bufferCount + ", direct=" + directByteBuffer);
063    buffers = new ByteBuffer[bufferCount + 1];
064    locks = new Lock[bufferCount + 1];
065    for (int i = 0; i <= bufferCount; i++) {
066      locks[i] = new ReentrantLock();
067      if (i < bufferCount) {
068        buffers[i] = directByteBuffer ? ByteBuffer.allocateDirect(bufferSize)
069            : ByteBuffer.allocate(bufferSize);
070      } else {
071        buffers[i] = ByteBuffer.allocate(0);
072      }
073
074    }
075  }
076
077  private long roundUp(long n, long to) {
078    return ((n + to - 1) / to) * to;
079  }
080
081  /**
082   * Transfers bytes from this buffer array into the given destination array
083   * @param start start position in the ByteBufferArray
084   * @param len The maximum number of bytes to be written to the given array
085   * @param dstArray The array into which bytes are to be written
086   * @return number of bytes read
087   */
088  public int getMultiple(long start, int len, byte[] dstArray) {
089    return getMultiple(start, len, dstArray, 0);
090  }
091
092  /**
093   * Transfers bytes from this buffer array into the given destination array
094   * @param start start offset of this buffer array
095   * @param len The maximum number of bytes to be written to the given array
096   * @param dstArray The array into which bytes are to be written
097   * @param dstOffset The offset within the given array of the first byte to be
098   *          written
099   * @return number of bytes read
100   */
101  public int getMultiple(long start, int len, byte[] dstArray, int dstOffset) {
102    multiple(start, len, dstArray, dstOffset, GET_MULTIPLE_VISTOR);
103    return len;
104  }
105
106  private final static Visitor GET_MULTIPLE_VISTOR = new Visitor() {
107    @Override
108    public void visit(ByteBuffer bb, byte[] array, int arrayIdx, int len) {
109      bb.get(array, arrayIdx, len);
110    }
111  };
112
113  /**
114   * Transfers bytes from the given source array into this buffer array
115   * @param start start offset of this buffer array
116   * @param len The maximum number of bytes to be read from the given array
117   * @param srcArray The array from which bytes are to be read
118   */
119  public void putMultiple(long start, int len, byte[] srcArray) {
120    putMultiple(start, len, srcArray, 0);
121  }
122
123  /**
124   * Transfers bytes from the given source array into this buffer array
125   * @param start start offset of this buffer array
126   * @param len The maximum number of bytes to be read from the given array
127   * @param srcArray The array from which bytes are to be read
128   * @param srcOffset The offset within the given array of the first byte to be
129   *          read
130   */
131  public void putMultiple(long start, int len, byte[] srcArray, int srcOffset) {
132    multiple(start, len, srcArray, srcOffset, PUT_MULTIPLE_VISITOR);
133  }
134
135  private final static Visitor PUT_MULTIPLE_VISITOR = new Visitor() {
136    @Override
137    public void visit(ByteBuffer bb, byte[] array, int arrayIdx, int len) {
138      bb.put(array, arrayIdx, len);
139    }
140  };
141
142  private interface Visitor {
143    /**
144     * Visit the given byte buffer, if it is a read action, we will transfer the
145     * bytes from the buffer to the destination array, else if it is a write
146     * action, we will transfer the bytes from the source array to the buffer
147     * @param bb byte buffer
148     * @param array a source or destination byte array
149     * @param arrayOffset offset of the byte array
150     * @param len read/write length
151     */
152    void visit(ByteBuffer bb, byte[] array, int arrayOffset, int len);
153  }
154
155  /**
156   * Access(read or write) this buffer array with a position and length as the
157   * given array. Here we will only lock one buffer even if it may be need visit
158   * several buffers. The consistency is guaranteed by the caller.
159   * @param start start offset of this buffer array
160   * @param len The maximum number of bytes to be accessed
161   * @param array The array from/to which bytes are to be read/written
162   * @param arrayOffset The offset within the given array of the first byte to
163   *          be read or written
164   * @param visitor implement of how to visit the byte buffer
165   */
166  void multiple(long start, int len, byte[] array, int arrayOffset, Visitor visitor) {
167    assert len >= 0;
168    long end = start + len;
169    int startBuffer = (int) (start / bufferSize), startOffset = (int) (start % bufferSize);
170    int endBuffer = (int) (end / bufferSize), endOffset = (int) (end % bufferSize);
171    assert array.length >= len + arrayOffset;
172    assert startBuffer >= 0 && startBuffer < bufferCount;
173    assert endBuffer >= 0 && endBuffer < bufferCount
174        || (endBuffer == bufferCount && endOffset == 0);
175    if (startBuffer >= locks.length || startBuffer < 0) {
176      String msg = "Failed multiple, start=" + start + ",startBuffer="
177          + startBuffer + ",bufferSize=" + bufferSize;
178      LOG.error(msg);
179      throw new RuntimeException(msg);
180    }
181    int srcIndex = 0, cnt = -1;
182    for (int i = startBuffer; i <= endBuffer; ++i) {
183      Lock lock = locks[i];
184      lock.lock();
185      try {
186        ByteBuffer bb = buffers[i];
187        if (i == startBuffer) {
188          cnt = bufferSize - startOffset;
189          if (cnt > len) cnt = len;
190          bb.limit(startOffset + cnt).position(startOffset);
191        } else if (i == endBuffer) {
192          cnt = endOffset;
193          bb.limit(cnt).position(0);
194        } else {
195          cnt = bufferSize;
196          bb.limit(cnt).position(0);
197        }
198        visitor.visit(bb, array, srcIndex + arrayOffset, cnt);
199        srcIndex += cnt;
200      } finally {
201        lock.unlock();
202      }
203    }
204    assert srcIndex == len;
205  }
206
207  /**
208   * Creates a ByteBuff from a given array of ByteBuffers from the given offset to the
209   * length specified. For eg, if there are 4 buffers forming an array each with length 10 and
210   * if we call asSubBuffer(5, 10) then we will create an MBB consisting of two BBs
211   * and the first one be a BB from 'position' 5 to a 'length' 5 and the 2nd BB will be from
212   * 'position' 0 to 'length' 5.
213   * @param offset
214   * @param len
215   * @return a ByteBuff formed from the underlying ByteBuffers
216   */
217  public ByteBuff asSubByteBuff(long offset, int len) {
218    assert len >= 0;
219    long end = offset + len;
220    int startBuffer = (int) (offset / bufferSize), startBufferOffset = (int) (offset % bufferSize);
221    int endBuffer = (int) (end / bufferSize), endBufferOffset = (int) (end % bufferSize);
222    // Last buffer in the array is a dummy one with 0 capacity. Avoid sending back that
223    if (endBuffer == this.bufferCount) {
224      endBuffer--;
225      endBufferOffset = bufferSize;
226    }
227    assert startBuffer >= 0 && startBuffer < bufferCount;
228    assert endBuffer >= 0 && endBuffer < bufferCount
229        || (endBuffer == bufferCount && endBufferOffset == 0);
230    if (startBuffer >= locks.length || startBuffer < 0) {
231      String msg = "Failed subArray, start=" + offset + ",startBuffer=" + startBuffer
232          + ",bufferSize=" + bufferSize;
233      LOG.error(msg);
234      throw new RuntimeException(msg);
235    }
236    int srcIndex = 0, cnt = -1;
237    ByteBuffer[] mbb = new ByteBuffer[endBuffer - startBuffer + 1];
238    for (int i = startBuffer,j=0; i <= endBuffer; ++i,j++) {
239      Lock lock = locks[i];
240      lock.lock();
241      try {
242        ByteBuffer bb = buffers[i];
243        if (i == startBuffer) {
244          cnt = bufferSize - startBufferOffset;
245          if (cnt > len) cnt = len;
246          ByteBuffer dup = bb.duplicate();
247          dup.limit(startBufferOffset + cnt).position(startBufferOffset);
248          mbb[j] = dup.slice();
249        } else if (i == endBuffer) {
250          cnt = endBufferOffset;
251          ByteBuffer dup = bb.duplicate();
252          dup.position(0).limit(cnt);
253          mbb[j] = dup.slice();
254        } else {
255          cnt = bufferSize ;
256          ByteBuffer dup = bb.duplicate();
257          dup.position(0).limit(cnt);
258          mbb[j] = dup.slice();
259        }
260        srcIndex += cnt;
261      } finally {
262        lock.unlock();
263      }
264    }
265    assert srcIndex == len;
266    if (mbb.length > 1) {
267      return new MultiByteBuff(mbb);
268    } else {
269      return new SingleByteBuff(mbb[0]);
270    }
271  }
