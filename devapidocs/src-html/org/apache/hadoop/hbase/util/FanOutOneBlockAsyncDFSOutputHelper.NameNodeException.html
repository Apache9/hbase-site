258  private static PipelineAckStatusGetter createPipelineAckStatusGetter() {
259    try {
260      final Method getFlagListMethod = PipelineAckProto.class.getMethod("getFlagList");
261      @SuppressWarnings("rawtypes")
262      Class<? extends Enum> ecnClass;
263      try {
264        ecnClass = Class.forName("org.apache.hadoop.hdfs.protocol.datatransfer.PipelineAck$ECN")
265            .asSubclass(Enum.class);
266      } catch (ClassNotFoundException e) {
267        throw new Error(e);
268      }
269      @SuppressWarnings("unchecked")
270      final Enum<?> disabledECN = Enum.valueOf(ecnClass, "DISABLED");
271      final Method getReplyMethod = PipelineAckProto.class.getMethod("getReply", int.class);
272      final Method combineHeaderMethod = PipelineAck.class.getMethod("combineHeader", ecnClass,
273        Status.class);
274      final Method getStatusFromHeaderMethod = PipelineAck.class.getMethod("getStatusFromHeader",
275        int.class);
276      return new PipelineAckStatusGetter() {
277
278        @Override
279        public Status get(PipelineAckProto ack) {
280          try {
281            @SuppressWarnings("unchecked")
282            List<Integer> flagList = (List<Integer>) getFlagListMethod.invoke(ack);
283            Integer headerFlag;
284            if (flagList.isEmpty()) {
285              Status reply = (Status) getReplyMethod.invoke(ack, 0);
286              headerFlag = (Integer) combineHeaderMethod.invoke(null, disabledECN, reply);
287            } else {
288              headerFlag = flagList.get(0);
289            }
290            return (Status) getStatusFromHeaderMethod.invoke(null, headerFlag);
291          } catch (IllegalAccessException | InvocationTargetException e) {
292            throw new RuntimeException(e);
293          }
294        }
295      };
296    } catch (NoSuchMethodException e) {
297      LOG.warn("Can not get expected methods, should be hadoop 2.6-", e);
298    }
299    try {
300      final Method getStatusMethod = PipelineAckProto.class.getMethod("getStatus", int.class);
301      return new PipelineAckStatusGetter() {
302
303        @Override
304        public Status get(PipelineAckProto ack) {
305          try {
306            return (Status) getStatusMethod.invoke(ack, 0);
307          } catch (IllegalAccessException | InvocationTargetException e) {
308            throw new RuntimeException(e);
309          }
310        }
311      };
312    } catch (NoSuchMethodException e) {
313      throw new Error(e);
314    }
315  }

317  private static StorageTypeSetter createStorageTypeSetter() {
318    final Method setStorageTypeMethod;
319    try {
320      setStorageTypeMethod = OpWriteBlockProto.Builder.class.getMethod("setStorageType",
321        StorageTypeProto.class);
322    } catch (NoSuchMethodException e) {
323      LOG.warn("noSetStorageType method found, should be hadoop 2.5-", e);
324      return new StorageTypeSetter() {
325
326        @Override
327        public Builder set(Builder builder, Enum<?> storageType) {
328          return builder;
329        }
330      };
331    }
332    ImmutableMap.Builder<String, StorageTypeProto> builder = ImmutableMap.builder();
333    for (StorageTypeProto storageTypeProto : StorageTypeProto.values()) {
334      builder.put(storageTypeProto.name(), storageTypeProto);
335    }
336    final ImmutableMap<String, StorageTypeProto> name2ProtoEnum = builder.build();
337    return new StorageTypeSetter() {
338
339      @Override
340      public Builder set(Builder builder, Enum<?> storageType) {
341        Object protoEnum = name2ProtoEnum.get(storageType.name());
342        try {
343          setStorageTypeMethod.invoke(builder, protoEnum);
344        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
345          throw new RuntimeException(e);
346        }
347        return builder;
348      }
349    };
350  }

352  private static FileCreater createFileCreater() {
353    for (Method method : ClientProtocol.class.getMethods()) {
354      if (method.getName().equals("create")) {
355        final Method createMethod = method;
356        Class<?>[] paramTypes = createMethod.getParameterTypes();
357        if (paramTypes[paramTypes.length - 1] == long.class) {
358          return new FileCreater() {
359
360            @Override
361            public HdfsFileStatus create(ClientProtocol namenode, String src, FsPermission masked,
362                String clientName, EnumSetWritable<CreateFlag> flag, boolean createParent,
363                short replication, long blockSize) throws IOException {
364              try {
365                return (HdfsFileStatus) createMethod.invoke(namenode, src, masked, clientName, flag,
366                  createParent, replication, blockSize);
367              } catch (IllegalAccessException e) {
368                throw new RuntimeException(e);
369              } catch (InvocationTargetException e) {
370                Throwables.propagateIfPossible(e.getTargetException(), IOException.class);
371                throw new RuntimeException(e);
372              }
373            }
374          };
375        } else {
376          try {
377            Class<?> cryptoProtocolVersionClass = Class
378                .forName("org.apache.hadoop.crypto.CryptoProtocolVersion");
379            Method supportedMethod = cryptoProtocolVersionClass.getMethod("supported");
380            final Object supported = supportedMethod.invoke(null);
381            return new FileCreater() {
382
383              @Override
384              public HdfsFileStatus create(ClientProtocol namenode, String src, FsPermission masked,
385                  String clientName, EnumSetWritable<CreateFlag> flag, boolean createParent,
386                  short replication, long blockSize) throws IOException {
387                try {
388                  return (HdfsFileStatus) createMethod.invoke(namenode, src, masked, clientName,
389                    flag, createParent, replication, blockSize, supported);
390                } catch (IllegalAccessException e) {
391                  throw new RuntimeException(e);
392                } catch (InvocationTargetException e) {
393                  Throwables.propagateIfPossible(e.getTargetException(), IOException.class);
394                  throw new RuntimeException(e);
395                }
396              }
397            };
398          } catch (ClassNotFoundException | NoSuchMethodException | IllegalAccessException
399              | InvocationTargetException e) {
400            throw new Error(e);
401          }
402        }
403      }
404    }
405    throw new Error("No create method found for " + ClientProtocol.class.getName());
406  }

458  private static void processWriteBlockResponse(Channel channel, final DatanodeInfo dnInfo,
459      final Promise<Channel> promise, final int timeoutMs) {
460    channel.pipeline().addLast(new IdleStateHandler(timeoutMs, 0, 0, TimeUnit.MILLISECONDS),
461      new ProtobufVarint32FrameDecoder(),
462      new ProtobufDecoder(BlockOpResponseProto.getDefaultInstance()),
463      new SimpleChannelInboundHandler<BlockOpResponseProto>() {
464
465        @Override
466        protected void channelRead0(ChannelHandlerContext ctx, BlockOpResponseProto resp)
467            throws Exception {
468          Status pipelineStatus = resp.getStatus();
469          if (PipelineAck.isRestartOOBStatus(pipelineStatus)) {
470            throw new IOException("datanode " + dnInfo + " is restarting");
471          }
472          String logInfo = "ack with firstBadLink as " + resp.getFirstBadLink();
473          if (resp.getStatus() != Status.SUCCESS) {
474            if (resp.getStatus() == Status.ERROR_ACCESS_TOKEN) {
475              throw new InvalidBlockTokenException("Got access token error" + ", status message "
476                  + resp.getMessage() + ", " + logInfo);
477            } else {
478              throw new IOException("Got error" + ", status=" + resp.getStatus().name()
479                  + ", status message " + resp.getMessage() + ", " + logInfo);
480            }
481          }
482          // success
483          ChannelPipeline p = ctx.pipeline();
484          while (p.first() != null) {
485            p.removeFirst();
486          }
487          // Disable auto read here. Enable it after we setup the streaming pipeline in
488          // FanOutOneBLockAsyncDFSOutput.
489          ctx.channel().config().setAutoRead(false);
490          promise.trySuccess(ctx.channel());
491        }
492
493        @Override
494        public void channelInactive(ChannelHandlerContext ctx) throws Exception {
495          promise.tryFailure(new IOException("connection to " + dnInfo + " is closed"));
496        }
497
498        @Override
499        public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
500          if (evt instanceof IdleStateEvent
501              && ((IdleStateEvent) evt).state() == IdleState.READER_IDLE) {
502            promise
503                .tryFailure(new IOException("Timeout(" + timeoutMs + "ms) waiting for response"));
504          } else {
505            super.userEventTriggered(ctx, evt);
506          }
507        }
508
509        @Override
510        public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
511          promise.tryFailure(cause);
512        }
513      });
514  }

516  private static void requestWriteBlock(Channel channel, Enum<?> storageType,
517      OpWriteBlockProto.Builder writeBlockProtoBuilder) throws IOException {
518    // TODO: SASL negotiation. should be done using a netty Handler.
519    OpWriteBlockProto proto = STORAGE_TYPE_SETTER.set(writeBlockProtoBuilder, storageType).build();
520    int protoLen = proto.getSerializedSize();
521    ByteBuf buffer = channel.alloc()
522        .buffer(3 + CodedOutputStream.computeRawVarint32Size(protoLen) + protoLen);
523    buffer.writeShort(DataTransferProtocol.DATA_TRANSFER_VERSION);
524    buffer.writeByte(Op.WRITE_BLOCK.code);
525    proto.writeDelimitedTo(new ByteBufOutputStream(buffer));
526    channel.writeAndFlush(buffer);
527  }

529  private static List<Future<Channel>> connectToDataNodes(Configuration conf, String clientName,
530      LocatedBlock locatedBlock, long maxBytesRcvd, long latestGS, BlockConstructionStage stage,
531      DataChecksum summer, EventLoop eventLoop) {
532    Enum<?>[] storageTypes = locatedBlock.getStorageTypes();
533    DatanodeInfo[] datanodeInfos = locatedBlock.getLocations();
534    boolean connectToDnViaHostname = conf.getBoolean(DFS_CLIENT_USE_DN_HOSTNAME,
535      DFS_CLIENT_USE_DN_HOSTNAME_DEFAULT);
536    final int timeoutMs = conf.getInt(DFS_CLIENT_SOCKET_TIMEOUT_KEY,
537      HdfsServerConstants.READ_TIMEOUT);
538    ExtendedBlock blockCopy = new ExtendedBlock(locatedBlock.getBlock());
539    blockCopy.setNumBytes(locatedBlock.getBlockSize());
540    ClientOperationHeaderProto header = ClientOperationHeaderProto.newBuilder()
541        .setBaseHeader(BaseHeaderProto.newBuilder().setBlock(PBHelper.convert(blockCopy))
542            .setToken(PBHelper.convert(locatedBlock.getBlockToken())))
543        .setClientName(clientName).build();
544    ChecksumProto checksumProto = DataTransferProtoUtil.toProto(summer);
545    final OpWriteBlockProto.Builder writeBlockProtoBuilder = OpWriteBlockProto.newBuilder()
546        .setHeader(header).setStage(OpWriteBlockProto.BlockConstructionStage.valueOf(stage.name()))
547        .setPipelineSize(1).setMinBytesRcvd(locatedBlock.getBlock().getNumBytes())
548        .setMaxBytesRcvd(maxBytesRcvd).setLatestGenerationStamp(latestGS)
549        .setRequestedChecksum(checksumProto)
550        .setCachingStrategy(CachingStrategyProto.newBuilder().setDropBehind(true).build());
551    List<Future<Channel>> futureList = new ArrayList<>(datanodeInfos.length);
552    for (int i = 0; i < datanodeInfos.length; i++) {
553      final DatanodeInfo dnInfo = datanodeInfos[i];
554      // Use Enum here because StoregType is moved to another package in hadoop 2.6. Use StorageType
555      // will cause compilation error for hadoop 2.5 or before.
556      final Enum<?> storageType = storageTypes[i];
557      final Promise<Channel> promise = eventLoop.newPromise();
558      futureList.add(promise);
559      String dnAddr = dnInfo.getXferAddr(connectToDnViaHostname);
560      new Bootstrap().group(eventLoop).channel(NioSocketChannel.class)
561          .option(CONNECT_TIMEOUT_MILLIS, timeoutMs).handler(new ChannelInitializer<Channel>() {
562
563            @Override
564            protected void initChannel(Channel ch) throws Exception {
565              processWriteBlockResponse(ch, dnInfo, promise, timeoutMs);
566            }
567          }).connect(NetUtils.createSocketAddr(dnAddr)).addListener(new ChannelFutureListener() {
568
569            @Override
570            public void operationComplete(ChannelFuture future) throws Exception {
571              if (future.isSuccess()) {
572                requestWriteBlock(future.channel(), storageType, writeBlockProtoBuilder);
573              } else {
574                promise.tryFailure(future.cause());
575              }
576            }
577          });
578    }
579    return futureList;
580  }

596      EventLoop eventLoop) throws IOException {
597    Configuration conf = dfs.getConf();
598    FSUtils fsUtils = FSUtils.getInstance(dfs, conf);
599    DFSClient client = dfs.getClient();
600    String clientName = client.getClientName();
601    ClientProtocol namenode = client.getNamenode();
602    HdfsFileStatus stat;
603    try {
604      stat = FILE_CREATER.create(namenode, src,
605        FsPermission.getFileDefault().applyUMask(FsPermission.getUMask(conf)), clientName,
606        new EnumSetWritable<CreateFlag>(
607            overwrite ? EnumSet.of(CREATE, OVERWRITE) : EnumSet.of(CREATE)),
608        createParent, replication, blockSize);
609    } catch (Exception e) {
610      if (e instanceof RemoteException) {
611        throw (RemoteException) e;
612      } else {
613        throw new NameNodeException(e);
614      }
615    }
616    beginFileLease(client, src, stat.getFileId());
617    boolean succ = false;
618    LocatedBlock locatedBlock = null;
619    List<Future<Channel>> futureList = null;
620    try {
621      DataChecksum summer = createChecksum(client);
622      locatedBlock = namenode.addBlock(src, client.getClientName(), null, null, stat.getFileId(),
623        null);
624      List<Channel> datanodeList = new ArrayList<>();
625      futureList = connectToDataNodes(conf, clientName, locatedBlock, 0L, 0L, PIPELINE_SETUP_CREATE,
626        summer, eventLoop);
627      for (Future<Channel> future : futureList) {
628        // fail the creation if there are connection failures since we are fail-fast. The upper
629        // layer should retry itself if needed.
630        datanodeList.add(future.syncUninterruptibly().getNow());
631      }
632      succ = true;
633      return new FanOutOneBlockAsyncDFSOutput(conf, fsUtils, dfs, client, namenode, clientName, src,
634          stat.getFileId(), locatedBlock, eventLoop, datanodeList, summer, ALLOC);
635    } finally {
636      if (!succ) {
637        if (futureList != null) {
638          for (Future<Channel> f : futureList) {
639            f.addListener(new FutureListener<Channel>() {
640
641              @Override
642              public void operationComplete(Future<Channel> future) throws Exception {
643                if (future.isSuccess()) {
644                  future.getNow().close();
645                }
646              }
647            });
648          }
649        }
650        endFileLease(client, src, stat.getFileId());
651        fsUtils.recoverFileLease(dfs, new Path(src), conf, new CancelOnClose(client));
652      }
653    }
654  }

656  /**
657   * Create a {@link FanOutOneBlockAsyncDFSOutput}. The method maybe blocked so do not call it
658   * inside {@link EventLoop}.
659   * @param eventLoop all connections to datanode will use the same event loop.
660   */
661  public static FanOutOneBlockAsyncDFSOutput createOutput(final DistributedFileSystem dfs, Path f,
662      final boolean overwrite, final boolean createParent, final short replication,
663      final long blockSize, final EventLoop eventLoop) throws IOException {
664    return new FileSystemLinkResolver<FanOutOneBlockAsyncDFSOutput>() {
665
666      @Override
667      public FanOutOneBlockAsyncDFSOutput doCall(Path p)
668          throws IOException, UnresolvedLinkException {
669        return createOutput(dfs, p.toUri().getPath(), overwrite, createParent, replication,
670          blockSize, eventLoop);
671      }
672
673      @Override
674      public FanOutOneBlockAsyncDFSOutput next(FileSystem fs, Path p) throws IOException {
675        throw new UnsupportedOperationException();
676      }
677    }.resolve(dfs, f);
678  }

680  public static boolean shouldRetryCreate(RemoteException e) {
681    // RetryStartFileException is introduced in HDFS 2.6+, so here we can only use the class name.
682    // For exceptions other than this, we just throw it out. This is same with
683    // DFSOutputStream.newStreamForCreate.
684    return e.getClassName().endsWith("RetryStartFileException");
685  }

687  static void completeFile(DFSClient client, ClientProtocol namenode, String src, String clientName,
688      ExtendedBlock block, long fileId) {
689    for (int retry = 0;; retry++) {
690      try {
691        if (namenode.complete(src, clientName, block, fileId)) {
692          endFileLease(client, src, fileId);
693          return;
694        } else {
695          LOG.warn("complete file " + src + " not finished, retry = " + retry);
696        }
697      } catch (LeaseExpiredException e) {
698        LOG.warn("lease for file " + src + " is expired, give up", e);
699        return;
700      } catch (Exception e) {
701        LOG.warn("complete file " + src + " failed, retry = " + retry, e);
702      }
703      sleepIgnoreInterrupt(retry);
704    }
705  }
706
707  static void sleepIgnoreInterrupt(int retry) {
708    try {
709      Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
710    } catch (InterruptedException e) {
711    }
712  }
