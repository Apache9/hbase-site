020package org.apache.hadoop.hbase.coordination;
021
022import java.io.IOException;
023import java.io.InterruptedIOException;
024import java.util.Set;
025import java.util.concurrent.ConcurrentMap;
026
027import org.apache.hadoop.hbase.ServerName;
028import org.apache.hadoop.hbase.client.RegionInfo;
029import org.apache.hadoop.hbase.master.MasterServices;
030import org.apache.hadoop.hbase.master.SplitLogManager.ResubmitDirective;
031import org.apache.hadoop.hbase.master.SplitLogManager.Task;
032import org.apache.yetus.audience.InterfaceAudience;
033
034import org.apache.hadoop.hbase.shaded.com.google.common.annotations.VisibleForTesting;
035import org.apache.hadoop.hbase.shaded.protobuf.generated.ZooKeeperProtos.SplitLogTask.RecoveryMode;
036
037/**
038 * Coordination for SplitLogManager. It creates and works with tasks for split log operations<BR>
039 * Manager prepares task by calling {@link #prepareTask} and submit it by
040 * {@link #submitTask(String)}. After that it periodically check the number of remaining tasks by
041 * {@link #remainingTasksInCoordination()} and waits until it become zero.
042 * <P>
043 * Methods required for task life circle: <BR>
044 * {@link #markRegionsRecovering(ServerName, Set)} mark regions for log replaying. Used by
045 * {@link org.apache.hadoop.hbase.master.MasterFileSystem} <BR>
046 * {@link #removeRecoveringRegions(Set, Boolean)} make regions cleanup that previous were marked as
047 * recovering. Called after all tasks processed <BR>
048 * {@link #removeStaleRecoveringRegions(Set)} remove stale recovering. called by
049 * {@link org.apache.hadoop.hbase.master.MasterFileSystem} after Active Master is initialized <BR>
050 * {@link #getLastRecoveryTime()} required for garbage collector and should indicate when the last
051 * recovery has been made<BR>
052 * {@link #checkTaskStillAvailable(String)} Check that task is still there <BR>
053 * {@link #checkTasks()} check for unassigned tasks and resubmit them
054 */
055@InterfaceAudience.Private
056public interface SplitLogManagerCoordination {
057
058  /**
059   * Detail class that shares data between coordination and split log manager
060   */
061  class SplitLogManagerDetails {
062    final private ConcurrentMap<String, Task> tasks;
063    final private MasterServices master;
064    final private Set<String> failedDeletions;
065
066    public SplitLogManagerDetails(ConcurrentMap<String, Task> tasks, MasterServices master,
067        Set<String> failedDeletions) {
068      this.tasks = tasks;
069      this.master = master;
070      this.failedDeletions = failedDeletions;
071    }
072
073    /**
074     * @return the master value
075     */
076    public MasterServices getMaster() {
077      return master;
078    }
079
080    /**
081     * @return map of tasks
082     */
083    public ConcurrentMap<String, Task> getTasks() {
084      return tasks;
085    }
086
087    /**
088     * @return a set of failed deletions
089     */
090    public Set<String> getFailedDeletions() {
091      return failedDeletions;
092    }
093
094    /**
095     * @return server name
096     */
097    public ServerName getServerName() {
098      return master.getServerName();
099    }
100  }
101
102  /**
103   * Provide the configuration from the SplitLogManager
104   */
105  void setDetails(SplitLogManagerDetails details);
106
107  /**
108   * Returns the configuration that was provided previously
109   */
110  SplitLogManagerDetails getDetails();
111
112  /**
113   * Prepare the new task
114   * @param taskName name of the task
115   * @return the task id
116   */
117  String prepareTask(String taskName);
118
119  /**
120   * Mark regions in recovering state for distributed log replay
121   * @param serverName server name
122   * @param userRegions set of regions to be marked
123   * @throws IOException in case of failure
124   * @throws InterruptedIOException
125   */
126  void markRegionsRecovering(final ServerName serverName, Set<RegionInfo> userRegions)
127      throws IOException, InterruptedIOException;
128
129  /**
130   * tells Coordination that it should check for new tasks
131   */
132  void checkTasks();
133
134  /**
135   * It removes recovering regions from Coordination
136   * @param serverNames servers which are just recovered
137   * @param isMetaRecovery whether current recovery is for the meta region on
138   *          <code>serverNames</code>
139   */
140  void removeRecoveringRegions(Set<String> serverNames, Boolean isMetaRecovery) throws IOException;
141
142  /**
143   * Return the number of remaining tasks
144   */
145  int remainingTasksInCoordination();
146
147  /**
148   * Check that the task is still there
149   * @param task node to check
150   */
151  void checkTaskStillAvailable(String task);
152
153  /**
154   * Change the recovery mode.
155   * @param b the recovery mode state
156   * @throws InterruptedIOException
157   * @throws IOException in case of failure
158   */
159  void setRecoveryMode(boolean b) throws IOException;
160
161  /**
162   * Removes known stale servers
163   * @param knownServers set of previously failed servers
164   * @throws IOException in case of failure
165   * @throws InterruptedIOException
166   */
167  void removeStaleRecoveringRegions(Set<String> knownServers) throws IOException,
168      InterruptedIOException;
169
170  /**
171   * Resubmit the task in case if found unassigned or failed
172   * @param taskName path related to task
173   * @param task to resubmit
174   * @param force whether it should be forced
175   * @return whether it was successful
176   */
177
178  boolean resubmitTask(String taskName, Task task, ResubmitDirective force);
179
180  /**
181   * @param taskName to be submitted
182   */
183  void submitTask(String taskName);
184
185  /**
186   * @param taskName to be removed
187   */
188  void deleteTask(String taskName);
189
190  /**
191   * @return shows whether the log recovery mode is in replaying state
192   */
193  boolean isReplaying();
194
195  /**
196   * @return shows whether the log recovery mode is in splitting state
197   */
198  boolean isSplitting();
199
200  /**
201   * @return the time of last attempt to recover
202   */
203  long getLastRecoveryTime();
204
205  /**
206   * Temporary function, mostly for UTs. In the regular code isReplaying or isSplitting should be
207   * used.
208   * @return the current log recovery mode.
209   */
210  RecoveryMode getRecoveryMode();
211
212  /**
213   * Support method to init constants such as timeout. Mostly required for UTs.
214   * @throws IOException
215   */
216  @VisibleForTesting
217  void init() throws IOException;
218}
