018package org.apache.hadoop.hbase.classification.tools;
019
020import org.apache.hadoop.hbase.classification.InterfaceAudience;
021import com.sun.javadoc.DocErrorReporter;
022import com.sun.javadoc.LanguageVersion;
023import com.sun.javadoc.RootDoc;
024import com.sun.tools.doclets.standard.Standard;
025
026/**
027 * A <a href="http://java.sun.com/javase/6/docs/jdk/api/javadoc/doclet/">Doclet</a>
028 * for excluding elements that are annotated with
029 * {@link org.apache.hadoop.hbase.classification.InterfaceAudience.Private} or
030 * {@link org.apache.hadoop.hbase.classification.InterfaceAudience.LimitedPrivate}.
031 * It delegates to the Standard Doclet, and takes the same options.
032 */
033@InterfaceAudience.Private
034public final class ExcludePrivateAnnotationsStandardDoclet {
035
036  private ExcludePrivateAnnotationsStandardDoclet() {}
037
038  public static LanguageVersion languageVersion() {
039    return LanguageVersion.JAVA_1_5;
040  }
041
042  public static boolean start(RootDoc root) {
043    System.out.println(
044      ExcludePrivateAnnotationsStandardDoclet.class.getSimpleName());
045    return Standard.start(RootDocProcessor.process(root));
046  }
047
048  public static int optionLength(String option) {
049    Integer length = StabilityOptions.optionLength(option);
050    if (length != null) {
051      return length;
052    }
053    return Standard.optionLength(option);
054  }
055
056  public static boolean validOptions(String[][] options,
057      DocErrorReporter reporter) {
058    StabilityOptions.validOptions(options, reporter);
059    String[][] filteredOptions = StabilityOptions.filterOptions(options);
060    return Standard.validOptions(filteredOptions, reporter);
061  }
062}
