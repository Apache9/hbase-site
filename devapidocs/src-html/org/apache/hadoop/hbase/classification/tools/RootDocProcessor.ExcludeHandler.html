018package org.apache.hadoop.hbase.classification.tools;
019
020import com.sun.javadoc.AnnotationDesc;
021import com.sun.javadoc.AnnotationTypeDoc;
022import com.sun.javadoc.ClassDoc;
023import com.sun.javadoc.ConstructorDoc;
024import com.sun.javadoc.Doc;
025import com.sun.javadoc.FieldDoc;
026import com.sun.javadoc.MethodDoc;
027import com.sun.javadoc.PackageDoc;
028import com.sun.javadoc.ProgramElementDoc;
029import com.sun.javadoc.RootDoc;
030
031import java.lang.reflect.Array;
032import java.lang.reflect.InvocationHandler;
033import java.lang.reflect.InvocationTargetException;
034import java.lang.reflect.Method;
035import java.lang.reflect.Proxy;
036import java.util.ArrayList;
037import java.util.List;
038import java.util.Map;
039import java.util.WeakHashMap;
040
041import org.apache.hadoop.hbase.classification.InterfaceAudience;
042import org.apache.hadoop.hbase.classification.InterfaceStability;
043
044/**
045 * Process the {@link RootDoc} by substituting with (nested) proxy objects that
046 * exclude elements with Private or LimitedPrivate annotations.
047 * <p>
048 * Based on code from http://www.sixlegs.com/blog/java/exclude-javadoc-tag.html.
049 */
050final class RootDocProcessor {
051
052  static String stability = StabilityOptions.UNSTABLE_OPTION;
053  static boolean treatUnannotatedClassesAsPrivate = false;
054
055  private RootDocProcessor() {}
056
057  public static RootDoc process(RootDoc root) {
058    return (RootDoc) process(root, RootDoc.class);
059  }
060
061  private static Object process(Object obj, Class<?> type) {
062    if (obj == null) {
063      return null;
064    }
065    Class<?> cls = obj.getClass();
066    if (cls.getName().startsWith("com.sun.")) {
067      return getProxy(obj);
068    } else if (obj instanceof Object[]) {
069      Class<?> componentType = type.isArray() ? type.getComponentType()
070        : cls.getComponentType();
071      Object[] array = (Object[]) obj;
072      Object[] newArray = (Object[]) Array.newInstance(componentType,
073        array.length);
074      for (int i = 0; i < array.length; ++i) {
075        newArray[i] = process(array[i], componentType);
076      }
077      return newArray;
078    }
079    return obj;
080  }
081
082  private static Map<Object, Object> proxies =
083    new WeakHashMap<Object, Object>();
084
085  private static Object getProxy(Object obj) {
086    Object proxy = proxies.get(obj);
087    if (proxy == null) {
088      proxy = Proxy.newProxyInstance(obj.getClass().getClassLoader(),
089        obj.getClass().getInterfaces(), new ExcludeHandler(obj));
090      proxies.put(obj, proxy);
091    }
092    return proxy;
093  }
094
095  private static class ExcludeHandler implements InvocationHandler {
096    private Object target;
097
098    public ExcludeHandler(Object target) {
099      this.target = target;
100    }
101
102    @Override
103    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
104      String methodName = method.getName();
105      if (target instanceof Doc) {
106        if (methodName.equals("isIncluded")) {
107          Doc doc = (Doc) target;
108          return !exclude(doc) && doc.isIncluded();
109        }
110        if (target instanceof RootDoc) {
111          if (methodName.equals("classes")) {
112            return filter(((RootDoc) target).classes(), ClassDoc.class);
113          } else if (methodName.equals("specifiedClasses")) {
114            return filter(((RootDoc) target).specifiedClasses(), ClassDoc.class);
115          } else if (methodName.equals("specifiedPackages")) {
116            return filter(((RootDoc) target).specifiedPackages(), PackageDoc.class);
117          }
118        } else if (target instanceof ClassDoc) {
119          if (isFiltered(args)) {
120            if (methodName.equals("methods")) {
121              return filter(((ClassDoc) target).methods(true), MethodDoc.class);
122            } else if (methodName.equals("fields")) {
123              return filter(((ClassDoc) target).fields(true), FieldDoc.class);
124            } else if (methodName.equals("innerClasses")) {
125              return filter(((ClassDoc) target).innerClasses(true), ClassDoc.class);
126            } else if (methodName.equals("constructors")) {
127              return filter(((ClassDoc) target).constructors(true), ConstructorDoc.class);
128            }
129          }
130        } else if (target instanceof PackageDoc) {
131          if (methodName.equals("allClasses")) {
132            if (isFiltered(args)) {
133              return filter(((PackageDoc) target).allClasses(true), ClassDoc.class);
134            } else {
135              return filter(((PackageDoc) target).allClasses(), ClassDoc.class);
136            }
137          } else if (methodName.equals("annotationTypes")) {
138            return filter(((PackageDoc) target).annotationTypes(), AnnotationTypeDoc.class);
139          } else if (methodName.equals("enums")) {
140            return filter(((PackageDoc) target).enums(), ClassDoc.class);
141          } else if (methodName.equals("errors")) {
142            return filter(((PackageDoc) target).errors(), ClassDoc.class);
143          } else if (methodName.equals("exceptions")) {
144            return filter(((PackageDoc) target).exceptions(), ClassDoc.class);
145          } else if (methodName.equals("interfaces")) {
146            return filter(((PackageDoc) target).interfaces(), ClassDoc.class);
147          } else if (methodName.equals("ordinaryClasses")) {
148            return filter(((PackageDoc) target).ordinaryClasses(), ClassDoc.class);
149          }
150        }
151      }
152
153      if (args != null) {
154        if (methodName.equals("compareTo") || methodName.equals("equals")
155            || methodName.equals("overrides") || methodName.equals("subclassOf")) {
156          args[0] = unwrap(args[0]);
157        }
158      }
159      try {
160        return process(method.invoke(target, args), method.getReturnType());
161      } catch (InvocationTargetException e) {
162        throw e.getTargetException();
163      }
164    }
165
166    private static boolean exclude(Doc doc) {
167      AnnotationDesc[] annotations = null;
168      if (doc instanceof ProgramElementDoc) {
169        annotations = ((ProgramElementDoc) doc).annotations();
170      } else if (doc instanceof PackageDoc) {
171        annotations = ((PackageDoc) doc).annotations();
172      }
173      if (annotations != null) {
174        for (AnnotationDesc annotation : annotations) {
175          String qualifiedTypeName = annotation.annotationType().qualifiedTypeName();
176          if (qualifiedTypeName.equals(InterfaceAudience.Private.class.getCanonicalName())
177              || qualifiedTypeName
178                  .equals(InterfaceAudience.LimitedPrivate.class.getCanonicalName())) {
179            return true;
180          }
181          if (stability.equals(StabilityOptions.EVOLVING_OPTION)) {
182            if (qualifiedTypeName.equals(InterfaceStability.Unstable.class.getCanonicalName())) {
183              return true;
184            }
185          }
186          if (stability.equals(StabilityOptions.STABLE_OPTION)) {
187            if (qualifiedTypeName.equals(InterfaceStability.Unstable.class.getCanonicalName())
188                || qualifiedTypeName.equals(InterfaceStability.Evolving.class.getCanonicalName())) {
189              return true;
190            }
191          }
192        }
193        for (AnnotationDesc annotation : annotations) {
194          String qualifiedTypeName = annotation.annotationType().qualifiedTypeName();
195          if (qualifiedTypeName.equals(InterfaceAudience.Public.class.getCanonicalName())) {
196            return false;
197          }
198        }
199      }
200      if (treatUnannotatedClassesAsPrivate) {
201        return doc.isClass() || doc.isInterface() || doc.isAnnotationType();
202      }
203      return false;
204    }
205
206    private static Object[] filter(Doc[] array, Class<?> componentType) {
207      if (array == null || array.length == 0) {
208        return array;
209      }
210      List<Object> list = new ArrayList<Object>(array.length);
211      for (Doc entry : array) {
212        if (!exclude(entry)) {
213          list.add(process(entry, componentType));
214        }
215      }
216      return list.toArray((Object[]) Array.newInstance(componentType, list.size()));
217    }
218
219    private Object unwrap(Object proxy) {
220      if (proxy instanceof Proxy) {
221        return ((ExcludeHandler) Proxy.getInvocationHandler(proxy)).target;
222      }
223      return proxy;
224    }
225
226    private boolean isFiltered(Object[] args) {
227      return args != null && Boolean.TRUE.equals(args[0]);
228    }
229  }
