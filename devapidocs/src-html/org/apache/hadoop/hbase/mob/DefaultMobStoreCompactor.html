051/**
052 * Compact passed set of files in the mob-enabled column family.
053 */
054@InterfaceAudience.Private
055public class DefaultMobStoreCompactor extends DefaultCompactor {
056
057  private static final Log LOG = LogFactory.getLog(DefaultMobStoreCompactor.class);
058  private long mobSizeThreshold;
059  private HMobStore mobStore;
060
061  private final InternalScannerFactory scannerFactory = new InternalScannerFactory() {
062
063    @Override
064    public ScanType getScanType(CompactionRequest request) {
065      return request.isRetainDeleteMarkers() ? ScanType.COMPACT_RETAIN_DELETES
066          : ScanType.COMPACT_DROP_DELETES;
067    }
068
069    @Override
070    public InternalScanner createScanner(List<StoreFileScanner> scanners,
071        ScanType scanType, FileDetails fd, long smallestReadPoint) throws IOException {
072      Scan scan = new Scan();
073      scan.setMaxVersions(store.getFamily().getMaxVersions());
074      if (scanType == ScanType.COMPACT_DROP_DELETES) {
075        // In major compaction, we need to write the delete markers to del files, so we have to
076        // retain the them in scanning.
077        scanType = ScanType.COMPACT_RETAIN_DELETES;
078        return new MobCompactionStoreScanner(store, store.getScanInfo(), scan, scanners,
079            scanType, smallestReadPoint, fd.earliestPutTs, true);
080      } else {
081        return new MobCompactionStoreScanner(store, store.getScanInfo(), scan, scanners,
082            scanType, smallestReadPoint, fd.earliestPutTs, false);
083      }
084    }
085  };
086
087  private final CellSinkFactory<StoreFileWriter> writerFactory =
088      new CellSinkFactory<StoreFileWriter>() {
089        @Override
090        public StoreFileWriter createWriter(InternalScanner scanner,
091            org.apache.hadoop.hbase.regionserver.compactions.Compactor.FileDetails fd,
092            boolean shouldDropBehind) throws IOException {
093          // make this writer with tags always because of possible new cells with tags.
094          return store.createWriterInTmp(fd.maxKeyCount, compactionCompression, true, true, true,
095            shouldDropBehind);
096        }
097      };
098
099  public DefaultMobStoreCompactor(Configuration conf, Store store) {
100    super(conf, store);
101    // The mob cells reside in the mob-enabled column family which is held by HMobStore.
102    // During the compaction, the compactor reads the cells from the mob files and
103    // probably creates new mob files. All of these operations are included in HMobStore,
104    // so we need to cast the Store to HMobStore.
105    if (!(store instanceof HMobStore)) {
106      throw new IllegalArgumentException("The store " + store + " is not a HMobStore");
107    }
108    mobStore = (HMobStore) store;
109    mobSizeThreshold = store.getFamily().getMobThreshold();
110  }
111
112  @Override
113  public List<Path> compact(CompactionRequest request, ThroughputController throughputController,
114      User user) throws IOException {
115    return compact(request, scannerFactory, writerFactory, throughputController, user);
116  }

120  /**
121   * Performs compaction on a column family with the mob flag enabled.
122   * This is for when the mob threshold size has changed or if the mob
123   * column family mode has been toggled via an alter table statement.
124   * Compacts the files by the following rules.
125   * 1. If the cell has a mob reference tag, the cell's value is the path of the mob file.
126   * <ol>
127   * <li>
128   * If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
129   * directly copy the (with mob tag) cell into the new store file.
130   * </li>
131   * <li>
132   * Otherwise, retrieve the mob cell from the mob file, and writes a copy of the cell into
133   * the new store file.
134   * </li>
135   * </ol>
136   * 2. If the cell doesn't have a reference tag.
137   * <ol>
138   * <li>
139   * If the value size of a cell is larger than the threshold, this cell is regarded as a mob,
140   * write this cell to a mob file, and write the path of this mob file to the store file.
141   * </li>
142   * <li>
143   * Otherwise, directly write this cell into the store file.
144   * </li>
145   * </ol>
146   * In the mob compaction, the {@link MobCompactionStoreScanner} is used as a scanner
147   * which could output the normal cells and delete markers together when required.
148   * After the major compaction on the normal hfiles, we have a guarantee that we have purged all
149   * deleted or old version mob refs, and the delete markers are written to a del file with the
150   * suffix _del. Because of this, it is safe to use the del file in the mob compaction.
151   * The mob compaction doesn't take place in the normal hfiles, it occurs directly in the
152   * mob files. When the small mob files are merged into bigger ones, the del file is added into
153   * the scanner to filter the deleted cells.
154   * @param fd File details
155   * @param scanner Where to read from.
156   * @param writer Where to write to.
157   * @param smallestReadPoint Smallest read point.
158   * @param cleanSeqId When true, remove seqId(used to be mvcc) value which is <= smallestReadPoint
159   * @param throughputController The compaction throughput controller.
160   * @param major Is a major compaction.
161   * @param numofFilesToCompact the number of files to compact
162   * @return Whether compaction ended; false if it was interrupted for any reason.
163   */
164  @Override
165  protected boolean performCompaction(FileDetails fd, InternalScanner scanner, CellSink writer,
166      long smallestReadPoint, boolean cleanSeqId, ThroughputController throughputController,
167      boolean major, int numofFilesToCompact) throws IOException {
168    if (!(scanner instanceof MobCompactionStoreScanner)) {
169      throw new IllegalArgumentException(
170        "The scanner should be an instance of MobCompactionStoreScanner");
171    }
172    MobCompactionStoreScanner compactionScanner = (MobCompactionStoreScanner) scanner;
173    int bytesWritten = 0;
174    // Since scanner.next() can return 'false' but still be delivering data,
175    // we have to use a do/while loop.
176    List<Cell> cells = new ArrayList<Cell>();
177    // Limit to "hbase.hstore.compaction.kv.max" (default 10) to avoid OOME
178    int closeCheckInterval = HStore.getCloseCheckInterval();
179    boolean hasMore;
180    Path path = MobUtils.getMobFamilyPath(conf, store.getTableName(), store.getColumnFamilyName());
181    byte[] fileName = null;
182    StoreFileWriter mobFileWriter = null, delFileWriter = null;
183    long mobCells = 0, deleteMarkersCount = 0;
184    long cellsCountCompactedToMob = 0, cellsCountCompactedFromMob = 0;
185    long cellsSizeCompactedToMob = 0, cellsSizeCompactedFromMob = 0;
186    try {
187      try {
188        // If the mob file writer could not be created, directly write the cell to the store file.
189        mobFileWriter = mobStore.createWriterInTmp(new Date(fd.latestPutTs), fd.maxKeyCount,
190            store.getFamily().getCompression(), store.getRegionInfo().getStartKey());
191        fileName = Bytes.toBytes(mobFileWriter.getPath().getName());
192      } catch (IOException e) {
193        LOG.error("Failed to create mob writer, "
194               + "we will continue the compaction by writing MOB cells directly in store files", e);
195      }
196      delFileWriter = mobStore.createDelFileWriterInTmp(new Date(fd.latestPutTs), fd.maxKeyCount,
197          store.getFamily().getCompression(), store.getRegionInfo().getStartKey());
198      ScannerContext scannerContext =
199              ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
200      do {
201        hasMore = compactionScanner.next(cells, scannerContext);
202        for (Cell c : cells) {
203          if (compactionScanner.isOutputDeleteMarkers() && CellUtil.isDelete(c)) {
204            delFileWriter.append(c);
205            deleteMarkersCount++;
206          } else if (mobFileWriter == null || c.getTypeByte() != KeyValue.Type.Put.getCode()) {
207            // If the mob file writer is null or the kv type is not put, directly write the cell
208            // to the store file.
209            writer.append(c);
210          } else if (MobUtils.isMobReferenceCell(c)) {
211            if (MobUtils.hasValidMobRefCellValue(c)) {
212              int size = MobUtils.getMobValueLength(c);
213              if (size > mobSizeThreshold) {
214                // If the value size is larger than the threshold, it's regarded as a mob. Since
215                // its value is already in the mob file, directly write this cell to the store file
216                writer.append(c);
217              } else {
218                // If the value is not larger than the threshold, it's not regarded a mob. Retrieve
219                // the mob cell from the mob file, and write it back to the store file.
220                Cell mobCell = mobStore.resolve(c, false);
221                if (mobCell.getValueLength() != 0) {
222                  // put the mob data back to the store file
223                  CellUtil.setSequenceId(mobCell, c.getSequenceId());
224                  writer.append(mobCell);
225                  cellsCountCompactedFromMob++;
226                  cellsSizeCompactedFromMob += mobCell.getValueLength();
227                } else {
228                  // If the value of a file is empty, there might be issues when retrieving,
229                  // directly write the cell to the store file, and leave it to be handled by the
230                  // next compaction.
231                  writer.append(c);
232                }
233              }
234            } else {
235              LOG.warn("The value format of the KeyValue " + c
236                  + " is wrong, its length is less than " + Bytes.SIZEOF_INT);
237              writer.append(c);
238            }
239          } else if (c.getValueLength() <= mobSizeThreshold) {
240            //If value size of a cell is not larger than the threshold, directly write to store file
241            writer.append(c);
242          } else {
243            // If the value size of a cell is larger than the threshold, it's regarded as a mob,
244            // write this cell to a mob file, and write the path to the store file.
245            mobCells++;
246            // append the original keyValue in the mob file.
247            mobFileWriter.append(c);
248            Cell reference = MobUtils.createMobRefCell(c, fileName,
249                this.mobStore.getRefCellTags());
250            // write the cell whose value is the path of a mob file to the store file.
251            writer.append(reference);
252            cellsCountCompactedToMob++;
253            cellsSizeCompactedToMob += c.getValueLength();
254          }
255          ++progress.currentCompactedKVs;
256          // check periodically to see if a system stop is requested
257          if (closeCheckInterval > 0) {
258            bytesWritten += KeyValueUtil.length(c);
259            if (bytesWritten > closeCheckInterval) {
260              bytesWritten = 0;
261              if (!store.areWritesEnabled()) {
262                progress.cancel();
263                return false;
264              }
265            }
266          }
267        }
268        cells.clear();
269      } while (hasMore);
270    } finally {
271      if (mobFileWriter != null) {
272        mobFileWriter.appendMetadata(fd.maxSeqId, major, mobCells);
273        mobFileWriter.close();
274      }
275      if (delFileWriter != null) {
276        delFileWriter.appendMetadata(fd.maxSeqId, major, deleteMarkersCount);
277        delFileWriter.close();
278      }
279    }
280    if (mobFileWriter != null) {
281      if (mobCells > 0) {
282        // If the mob file is not empty, commit it.
283        mobStore.commitFile(mobFileWriter.getPath(), path);
284      } else {
285        try {
286          // If the mob file is empty, delete it instead of committing.
287          store.getFileSystem().delete(mobFileWriter.getPath(), true);
288        } catch (IOException e) {
289          LOG.error("Failed to delete the temp mob file", e);
290        }
291      }
292    }
293    if (delFileWriter != null) {
294      if (deleteMarkersCount > 0) {
295        // If the del file is not empty, commit it.
296        // If the commit fails, the compaction is re-performed again.
297        mobStore.commitFile(delFileWriter.getPath(), path);
298      } else {
299        try {
300          // If the del file is empty, delete it instead of committing.
301          store.getFileSystem().delete(delFileWriter.getPath(), true);
302        } catch (IOException e) {
303          LOG.error("Failed to delete the temp del file", e);
304        }
305      }
306    }
307    mobStore.updateCellsCountCompactedFromMob(cellsCountCompactedFromMob);
308    mobStore.updateCellsCountCompactedToMob(cellsCountCompactedToMob);
309    mobStore.updateCellsSizeCompactedFromMob(cellsSizeCompactedFromMob);
310    mobStore.updateCellsSizeCompactedToMob(cellsSizeCompactedToMob);
311    progress.complete();
312    return true;
313  }
