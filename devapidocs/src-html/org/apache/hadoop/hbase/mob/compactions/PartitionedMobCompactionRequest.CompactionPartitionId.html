019package org.apache.hadoop.hbase.mob.compactions;
020
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Collections;
024import java.util.List;
025
026import org.apache.hadoop.fs.FileStatus;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.mob.MobConstants;
029import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
030
031/**
032 * An implementation of {@link MobCompactionRequest} that is used in
033 * {@link PartitionedMobCompactor}.
034 * The mob files that have the same start key and date in their names belong to
035 * the same partition.
036 */
037@InterfaceAudience.Private
038public class PartitionedMobCompactionRequest extends MobCompactionRequest {
039
040  protected Collection<FileStatus> delFiles;
041  protected Collection<CompactionPartition> compactionPartitions;
042
043  public PartitionedMobCompactionRequest(Collection<CompactionPartition> compactionPartitions,
044    Collection<FileStatus> delFiles) {
045    this.selectionTime = EnvironmentEdgeManager.currentTime();
046    this.compactionPartitions = compactionPartitions;
047    this.delFiles = delFiles;
048  }
049
050  /**
051   * Gets the compaction partitions.
052   * @return The compaction partitions.
053   */
054  public Collection<CompactionPartition> getCompactionPartitions() {
055    return this.compactionPartitions;
056  }
057
058  /**
059   * Gets the del files.
060   * @return The del files.
061   */
062  public Collection<FileStatus> getDelFiles() {
063    return this.delFiles;
064  }
065
066  /**
067   * The partition in the mob compaction.
068   * The mob files that have the same start key and date in their names belong to
069   * the same partition.
070   */
071  protected static class CompactionPartition {
072    private List<FileStatus> files = new ArrayList<FileStatus>();
073    private CompactionPartitionId partitionId;
074
075    public CompactionPartition(CompactionPartitionId partitionId) {
076      this.partitionId = partitionId;
077    }
078
079    public CompactionPartitionId getPartitionId() {
080      return this.partitionId;
081    }
082
083    public void addFile(FileStatus file) {
084      files.add(file);
085    }
086
087    public List<FileStatus> listFiles() {
088      return Collections.unmodifiableList(files);
089    }
090
091    public int getFileCount () {
092      return files.size();
093    }
094  }
095
096  /**
097   * The partition id that consists of start key and date of the mob file name.
098   */
099  public static class CompactionPartitionId {
100    private String startKey;
101    private String date;
102    private String latestDate;
103    private long threshold;
104
105    public CompactionPartitionId() {
106      // initialize these fields to empty string
107      this.startKey = "";
108      this.date = "";
109      this.latestDate = "";
110      this.threshold = 0;
111    }
112
113    public CompactionPartitionId(String startKey, String date) {
114      if (startKey == null || date == null) {
115        throw new IllegalArgumentException("Neither of start key and date could be null");
116      }
117      this.startKey = startKey;
118      this.date = date;
119      this.latestDate = "";
120      this.threshold = 0;
121    }
122
123    public void setThreshold (final long threshold) {
124      this.threshold = threshold;
125    }
126
127    public long getThreshold () {
128      return this.threshold;
129    }
130
131    public String getStartKey() {
132      return this.startKey;
133    }
134
135    public void setStartKey(final String startKey) {
136      this.startKey = startKey;
137    }
138
139    public String getDate() {
140      return this.date;
141    }
142
143    public void setDate(final String date) {
144      this.date = date;
145    }
146
147    public String getLatestDate () { return this.latestDate; }
148
149    public void updateLatestDate(final String latestDate) {
150      if (this.latestDate.compareTo(latestDate) < 0) {
151        this.latestDate = latestDate;
152      }
153    }
154
155    @Override
156    public int hashCode() {
157      int result = 17;
158      result = 31 * result + startKey.hashCode();
159      result = 31 * result + date.hashCode();
160      return result;
161    }
162
163    @Override
164    public boolean equals(Object obj) {
165      if (this == obj) {
166        return true;
167      }
168      if (!(obj instanceof CompactionPartitionId)) {
169        return false;
170      }
171      CompactionPartitionId another = (CompactionPartitionId) obj;
172      if (!this.startKey.equals(another.startKey)) {
173        return false;
174      }
175      if (!this.date.equals(another.date)) {
176        return false;
177      }
178      return true;
179    }
180
181    @Override
182    public String toString() {
183      return new StringBuilder(startKey).append(date).toString();
184    }
185  }
186}
