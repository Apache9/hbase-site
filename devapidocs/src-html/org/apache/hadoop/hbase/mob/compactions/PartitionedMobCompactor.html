098  public PartitionedMobCompactor(Configuration conf, FileSystem fs, TableName tableName,
099    HColumnDescriptor column, ExecutorService pool) throws IOException {
100    super(conf, fs, tableName, column, pool);
101    mergeableSize = conf.getLong(MobConstants.MOB_COMPACTION_MERGEABLE_THRESHOLD,
102      MobConstants.DEFAULT_MOB_COMPACTION_MERGEABLE_THRESHOLD);
103    delFileMaxCount = conf.getInt(MobConstants.MOB_DELFILE_MAX_COUNT,
104      MobConstants.DEFAULT_MOB_DELFILE_MAX_COUNT);
105    // default is 100
106    compactionBatchSize = conf.getInt(MobConstants.MOB_COMPACTION_BATCH_SIZE,
107      MobConstants.DEFAULT_MOB_COMPACTION_BATCH_SIZE);
108    tempPath = new Path(MobUtils.getMobHome(conf), MobConstants.TEMP_DIR_NAME);
109    bulkloadPath = new Path(tempPath, new Path(MobConstants.BULKLOAD_DIR_NAME, new Path(
110      tableName.getNamespaceAsString(), tableName.getQualifierAsString())));
111    compactionKVMax = this.conf.getInt(HConstants.COMPACTION_KV_MAX,
112      HConstants.COMPACTION_KV_MAX_DEFAULT);
113    Configuration copyOfConf = new Configuration(conf);
114    copyOfConf.setFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY, 0f);
115    compactionCacheConfig = new CacheConfig(copyOfConf);
116    tableNameTag = new ArrayBackedTag(TagType.MOB_TABLE_NAME_TAG_TYPE, tableName.getName());
117    cryptoContext = EncryptionUtil.createEncryptionContext(copyOfConf, column);
118  }

120  @Override
121  public List<Path> compact(List<FileStatus> files, boolean allFiles) throws IOException {
122    if (files == null || files.isEmpty()) {
123      LOG.info("No candidate mob files");
124      return null;
125    }
126    LOG.info("is allFiles: " + allFiles);
127    // find the files to compact.
128    PartitionedMobCompactionRequest request = select(files, allFiles);
129    // compact the files.
130    return performCompaction(request);
131  }

133  /**
134   * Selects the compacted mob/del files.
135   * Iterates the candidates to find out all the del files and small mob files.
136   * @param candidates All the candidates.
137   * @param allFiles Whether add all mob files into the compaction.
138   * @return A compaction request.
139   * @throws IOException if IO failure is encountered
140   */
141  protected PartitionedMobCompactionRequest select(List<FileStatus> candidates,
142    boolean allFiles) throws IOException {
143    Collection<FileStatus> allDelFiles = new ArrayList<>();
144    Map<CompactionPartitionId, CompactionPartition> filesToCompact = new HashMap<>();
145    int selectedFileCount = 0;
146    int irrelevantFileCount = 0;
147    for (FileStatus file : candidates) {
148      if (!file.isFile()) {
149        irrelevantFileCount++;
150        continue;
151      }
152      // group the del files and small files.
153      FileStatus linkedFile = file;
154      if (HFileLink.isHFileLink(file.getPath())) {
155        HFileLink link = HFileLink.buildFromHFileLinkPattern(conf, file.getPath());
156        linkedFile = getLinkedFileStatus(link);
157        if (linkedFile == null) {
158          // If the linked file cannot be found, regard it as an irrelevantFileCount file
159          irrelevantFileCount++;
160          continue;
161        }
162      }
163      if (StoreFileInfo.isDelFile(linkedFile.getPath())) {
164        allDelFiles.add(file);
165      } else if (allFiles || linkedFile.getLen() < mergeableSize) {
166        // add all files if allFiles is true,
167        // otherwise add the small files to the merge pool
168        MobFileName fileName = MobFileName.create(linkedFile.getPath().getName());
169        CompactionPartitionId id = new CompactionPartitionId(fileName.getStartKey(),
170          fileName.getDate());
171        CompactionPartition compactionPartition = filesToCompact.get(id);
172        if (compactionPartition == null) {
173          compactionPartition = new CompactionPartition(id);
174          compactionPartition.addFile(file);
175          filesToCompact.put(id, compactionPartition);
176        } else {
177          compactionPartition.addFile(file);
178        }
179        selectedFileCount++;
180      }
181    }
182    PartitionedMobCompactionRequest request = new PartitionedMobCompactionRequest(
183      filesToCompact.values(), allDelFiles);
184    if (candidates.size() == (allDelFiles.size() + selectedFileCount + irrelevantFileCount)) {
185      // all the files are selected
186      request.setCompactionType(CompactionType.ALL_FILES);
187    }
188    LOG.info("The compaction type is " + request.getCompactionType() + ", the request has "
189      + allDelFiles.size() + " del files, " + selectedFileCount + " selected files, and "
190      + irrelevantFileCount + " irrelevant files");
191    return request;
192  }

194  /**
195   * Performs the compaction on the selected files.
196   * <ol>
197   * <li>Compacts the del files.</li>
198   * <li>Compacts the selected small mob files and all the del files.</li>
199   * <li>If all the candidates are selected, delete the del files.</li>
200   * </ol>
201   * @param request The compaction request.
202   * @return The paths of new mob files generated in the compaction.
203   * @throws IOException if IO failure is encountered
204   */
205  protected List<Path> performCompaction(PartitionedMobCompactionRequest request)
206    throws IOException {
207    // merge the del files
208    List<Path> delFilePaths = new ArrayList<>();
209    for (FileStatus delFile : request.delFiles) {
210      delFilePaths.add(delFile.getPath());
211    }
212    List<Path> newDelPaths = compactDelFiles(request, delFilePaths);
213    List<StoreFile> newDelFiles = new ArrayList<>();
214    List<Path> paths = null;
215    try {
216      for (Path newDelPath : newDelPaths) {
217        StoreFile sf = new StoreFile(fs, newDelPath, conf, compactionCacheConfig, BloomType.NONE);
218        // pre-create reader of a del file to avoid race condition when opening the reader in each
219        // partition.
220        sf.createReader();
221        newDelFiles.add(sf);
222      }
223      LOG.info("After merging, there are " + newDelFiles.size() + " del files");
224      // compact the mob files by partitions.
225      paths = compactMobFiles(request, newDelFiles);
226      LOG.info("After compaction, there are " + paths.size() + " mob files");
227    } finally {
228      closeStoreFileReaders(newDelFiles);
229    }
230    // archive the del files if all the mob files are selected.
231    if (request.type == CompactionType.ALL_FILES && !newDelPaths.isEmpty()) {
232      LOG.info("After a mob compaction with all files selected, archiving the del files "
233        + newDelPaths);
234      try {
235        MobUtils.removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), newDelFiles);
236      } catch (IOException e) {
237        LOG.error("Failed to archive the del files " + newDelPaths, e);
238      }
239    }
240    return paths;
241  }

243  /**
244   * Compacts the selected small mob files and all the del files.
245   * @param request The compaction request.
246   * @param delFiles The del files.
247   * @return The paths of new mob files after compactions.
248   * @throws IOException if IO failure is encountered
249   */
250  protected List<Path> compactMobFiles(final PartitionedMobCompactionRequest request,
251    final List<StoreFile> delFiles) throws IOException {
252    Collection<CompactionPartition> partitions = request.compactionPartitions;
253    if (partitions == null || partitions.isEmpty()) {
254      LOG.info("No partitions of mob files");
255      return Collections.emptyList();
256    }
257    List<Path> paths = new ArrayList<>();
258    final Connection c = ConnectionFactory.createConnection(conf);
259    final Table table = c.getTable(tableName);
260    try {
261      Map<CompactionPartitionId, Future<List<Path>>> results = new HashMap<>();
262      // compact the mob files by partitions in parallel.
263      for (final CompactionPartition partition : partitions) {
264        results.put(partition.getPartitionId(), pool.submit(new Callable<List<Path>>() {
265          @Override
266          public List<Path> call() throws Exception {
267            LOG.info("Compacting mob files for partition " + partition.getPartitionId());
268            return compactMobFilePartition(request, partition, delFiles, c, table);
269          }
270        }));
271      }
272      // compact the partitions in parallel.
273      List<CompactionPartitionId> failedPartitions = new ArrayList<>();
274      for (Entry<CompactionPartitionId, Future<List<Path>>> result : results.entrySet()) {
275        try {
276          paths.addAll(result.getValue().get());
277        } catch (Exception e) {
278          // just log the error
279          LOG.error("Failed to compact the partition " + result.getKey(), e);
280          failedPartitions.add(result.getKey());
281        }
282      }
283      if (!failedPartitions.isEmpty()) {
284        // if any partition fails in the compaction, directly throw an exception.
285        throw new IOException("Failed to compact the partitions " + failedPartitions);
286      }
287    } finally {
288      try {
289        table.close();
290      } catch (IOException e) {
291        LOG.error("Failed to close the Table", e);
292      }
293    }
294    return paths;
295  }

297  /**
298   * Compacts a partition of selected small mob files and all the del files.
299   * @param request The compaction request.
300   * @param partition A compaction partition.
301   * @param delFiles The del files.
302   * @param connection to use
303   * @param table The current table.  @return The paths of new mob files after compactions.
304   * @throws IOException if IO failure is encountered
305   */
306  private List<Path> compactMobFilePartition(PartitionedMobCompactionRequest request,
307                                             CompactionPartition partition,
308                                             List<StoreFile> delFiles,
309                                             Connection connection,
310                                             Table table) throws IOException {
311    List<Path> newFiles = new ArrayList<>();
312    List<FileStatus> files = partition.listFiles();
313    int offset = 0;
314    Path bulkloadPathOfPartition = new Path(bulkloadPath, partition.getPartitionId().toString());
315    Path bulkloadColumnPath = new Path(bulkloadPathOfPartition, column.getNameAsString());
316    while (offset < files.size()) {
317      int batch = compactionBatchSize;
318      if (files.size() - offset < compactionBatchSize) {
319        batch = files.size() - offset;
320      }
321      if (batch == 1 && delFiles.isEmpty()) {
322        // only one file left and no del files, do not compact it,
323        // and directly add it to the new files.
324        newFiles.add(files.get(offset).getPath());
325        offset++;
326        continue;
327      }
328      // clean the bulkload directory to avoid loading old files.
329      fs.delete(bulkloadPathOfPartition, true);
330      // add the selected mob files and del files into filesToCompact
331      List<StoreFile> filesToCompact = new ArrayList<>();
332      for (int i = offset; i < batch + offset; i++) {
333        StoreFile sf = new StoreFile(fs, files.get(i).getPath(), conf, compactionCacheConfig,
334          BloomType.NONE);
335        filesToCompact.add(sf);
336      }
337      filesToCompact.addAll(delFiles);
338      // compact the mob files in a batch.
339      compactMobFilesInBatch(request, partition, connection, table, filesToCompact, batch,
340        bulkloadPathOfPartition, bulkloadColumnPath, newFiles);
341      // move to the next batch.
342      offset += batch;
343    }
344    LOG.info("Compaction is finished. The number of mob files is changed from " + files.size()
345      + " to " + newFiles.size());
346    return newFiles;
347  }

363  /**
364   * Compacts a partition of selected small mob files and all the del files in a batch.
365   * @param request The compaction request.
366   * @param partition A compaction partition.
367   * @param connection To use for transport
368   * @param table The current table.
369   * @param filesToCompact The files to be compacted.
370   * @param batch The number of mob files to be compacted in a batch.
371   * @param bulkloadPathOfPartition The directory where the bulkload column of the current
372   *   partition is saved.
373   * @param bulkloadColumnPath The directory where the bulkload files of current partition
374   *   are saved.
375   * @param newFiles The paths of new mob files after compactions.
376   * @throws IOException if IO failure is encountered
377   */
378  private void compactMobFilesInBatch(PartitionedMobCompactionRequest request,
379                                      CompactionPartition partition,
380                                      Connection connection, Table table,
381                                      List<StoreFile> filesToCompact, int batch,
382                                      Path bulkloadPathOfPartition, Path bulkloadColumnPath,
383                                      List<Path> newFiles)
384    throws IOException {
385    // open scanner to the selected mob files and del files.
386    StoreScanner scanner = createScanner(filesToCompact, ScanType.COMPACT_DROP_DELETES);
387    // the mob files to be compacted, not include the del files.
388    List<StoreFile> mobFilesToCompact = filesToCompact.subList(0, batch);
389    // Pair(maxSeqId, cellsCount)
390    Pair<Long, Long> fileInfo = getFileInfo(mobFilesToCompact);
391    // open writers for the mob files and new ref store files.
392    StoreFileWriter writer = null;
393    StoreFileWriter refFileWriter = null;
394    Path filePath = null;
395    Path refFilePath = null;
396    long mobCells = 0;
397    try {
398      writer = MobUtils.createWriter(conf, fs, column, partition.getPartitionId().getDate(),
399        tempPath, Long.MAX_VALUE, column.getCompactionCompression(), partition.getPartitionId()
400          .getStartKey(), compactionCacheConfig, cryptoContext);
401      filePath = writer.getPath();
402      byte[] fileName = Bytes.toBytes(filePath.getName());
403      // create a temp file and open a writer for it in the bulkloadPath
404      refFileWriter = MobUtils.createRefFileWriter(conf, fs, column, bulkloadColumnPath, fileInfo
405        .getSecond().longValue(), compactionCacheConfig, cryptoContext);
406      refFilePath = refFileWriter.getPath();
407      List<Cell> cells = new ArrayList<>();
408      boolean hasMore;
409      ScannerContext scannerContext =
410              ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
411      do {
412        hasMore = scanner.next(cells, scannerContext);
413        for (Cell cell : cells) {
414          // write the mob cell to the mob file.
415          writer.append(cell);
416          // write the new reference cell to the store file.
417          KeyValue reference = MobUtils.createMobRefKeyValue(cell, fileName, tableNameTag);
418          refFileWriter.append(reference);
419          mobCells++;
420        }
421        cells.clear();
422      } while (hasMore);
423    } finally {
424      // close the scanner.
425      scanner.close();
426      // append metadata to the mob file, and close the mob file writer.
427      closeMobFileWriter(writer, fileInfo.getFirst(), mobCells);
428      // append metadata and bulkload info to the ref mob file, and close the writer.
429      closeRefFileWriter(refFileWriter, fileInfo.getFirst(), request.selectionTime);
430    }
431    if (mobCells > 0) {
432      // commit mob file
433      MobUtils.commitFile(conf, fs, filePath, mobFamilyDir, compactionCacheConfig);
434      // bulkload the ref file
435      bulkloadRefFile(connection, table, bulkloadPathOfPartition, filePath.getName());
436      newFiles.add(new Path(mobFamilyDir, filePath.getName()));
437    } else {
438      // remove the new files
439      // the mob file is empty, delete it instead of committing.
440      deletePath(filePath);
441      // the ref file is empty, delete it instead of committing.
442      deletePath(refFilePath);
443    }
444    // archive the old mob files, do not archive the del files.
445    try {
446      closeStoreFileReaders(mobFilesToCompact);
447      MobUtils
448        .removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), mobFilesToCompact);
449    } catch (IOException e) {
450      LOG.error("Failed to archive the files " + mobFilesToCompact, e);
451    }
452  }

454  /**
455   * Compacts the del files in batches which avoids opening too many files.
456   * @param request The compaction request.
457   * @param delFilePaths Del file paths to compact
458   * @return The paths of new del files after merging or the original files if no merging
459   *         is necessary.
460   * @throws IOException if IO failure is encountered
461   */
462  protected List<Path> compactDelFiles(PartitionedMobCompactionRequest request,
463    List<Path> delFilePaths) throws IOException {
464    if (delFilePaths.size() <= delFileMaxCount) {
465      return delFilePaths;
466    }
467    // when there are more del files than the number that is allowed, merge it firstly.
468    int offset = 0;
469    List<Path> paths = new ArrayList<>();
470    while (offset < delFilePaths.size()) {
471      // get the batch
472      int batch = compactionBatchSize;
473      if (delFilePaths.size() - offset < compactionBatchSize) {
474        batch = delFilePaths.size() - offset;
475      }
476      List<StoreFile> batchedDelFiles = new ArrayList<>();
477      if (batch == 1) {
478        // only one file left, do not compact it, directly add it to the new files.
479        paths.add(delFilePaths.get(offset));
480        offset++;
481        continue;
482      }
483      for (int i = offset; i < batch + offset; i++) {
484        batchedDelFiles.add(new StoreFile(fs, delFilePaths.get(i), conf, compactionCacheConfig,
485          BloomType.NONE));
486      }
487      // compact the del files in a batch.
488      paths.add(compactDelFilesInBatch(request, batchedDelFiles));
489      // move to the next batch.
490      offset += batch;
491    }
492    return compactDelFiles(request, paths);
493  }

495  /**
496   * Compacts the del file in a batch.
497   * @param request The compaction request.
498   * @param delFiles The del files.
499   * @return The path of new del file after merging.
500   * @throws IOException if IO failure is encountered
501   */
502  private Path compactDelFilesInBatch(PartitionedMobCompactionRequest request,
503    List<StoreFile> delFiles) throws IOException {
504    // create a scanner for the del files.
505    StoreScanner scanner = createScanner(delFiles, ScanType.COMPACT_RETAIN_DELETES);
506    StoreFileWriter writer = null;
507    Path filePath = null;
508    try {
509      writer = MobUtils.createDelFileWriter(conf, fs, column,
510        MobUtils.formatDate(new Date(request.selectionTime)), tempPath, Long.MAX_VALUE,
511        column.getCompactionCompression(), HConstants.EMPTY_START_ROW, compactionCacheConfig,
512        cryptoContext);
513      filePath = writer.getPath();
514      List<Cell> cells = new ArrayList<>();
515      boolean hasMore;
516      ScannerContext scannerContext =
517              ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
518      do {
519        hasMore = scanner.next(cells, scannerContext);
520        for (Cell cell : cells) {
521          writer.append(cell);
522        }
523        cells.clear();
524      } while (hasMore);
525    } finally {
526      scanner.close();
527      if (writer != null) {
528        try {
529          writer.close();
530        } catch (IOException e) {
531          LOG.error("Failed to close the writer of the file " + filePath, e);
532        }
533      }
534    }
535    // commit the new del file
536    Path path = MobUtils.commitFile(conf, fs, filePath, mobFamilyDir, compactionCacheConfig);
537    // archive the old del files
538    try {
539      MobUtils.removeMobFiles(conf, fs, tableName, mobTableDir, column.getName(), delFiles);
540    } catch (IOException e) {
541      LOG.error("Failed to archive the old del files " + delFiles, e);
542    }
543    return path;
544  }

546  /**
547   * Creates a store scanner.
548   * @param filesToCompact The files to be compacted.
549   * @param scanType The scan type.
550   * @return The store scanner.
551   * @throws IOException if IO failure is encountered
552   */
553  private StoreScanner createScanner(List<StoreFile> filesToCompact, ScanType scanType)
554    throws IOException {
555    List scanners = StoreFileScanner.getScannersForStoreFiles(filesToCompact, false, true, false,
556      false, HConstants.LATEST_TIMESTAMP);
557    Scan scan = new Scan();
558    scan.setMaxVersions(column.getMaxVersions());
559    long ttl = HStore.determineTTLFromFamily(column);
560    ScanInfo scanInfo = new ScanInfo(conf, column, ttl, 0, CellComparator.COMPARATOR);
561    StoreScanner scanner = new StoreScanner(scan, scanInfo, scanType, null, scanners, 0L,
562      HConstants.LATEST_TIMESTAMP);
563    return scanner;
564  }

566  /**
567   * Bulkloads the current file.
568   *
569   * @param connection to use to get admin/RegionLocator
570   * @param table The current table.
571   * @param bulkloadDirectory The path of bulkload directory.
572   * @param fileName The current file name.
573   * @throws IOException if IO failure is encountered
574   */
575  private void bulkloadRefFile(Connection connection, Table table, Path bulkloadDirectory,
576                               String fileName)
577    throws IOException {
578    // bulkload the ref file
579    try {
580      LoadIncrementalHFiles bulkload = new LoadIncrementalHFiles(conf);
581      bulkload.doBulkLoad(bulkloadDirectory,
582        connection.getAdmin(),
583        table,
584        connection.getRegionLocator(table.getName()));
585    } catch (Exception e) {
586      // delete the committed mob file
587      deletePath(new Path(mobFamilyDir, fileName));
588      throw new IOException(e);
589    } finally {
590      // delete the bulkload files in bulkloadPath
591      deletePath(bulkloadDirectory);
592    }
593  }

597   * @param writer The mob file writer.
598   * @param maxSeqId Maximum sequence id.
599   * @param mobCellsCount The number of mob cells.
600   * @throws IOException if IO failure is encountered
601   */
602  private void closeMobFileWriter(StoreFileWriter writer, long maxSeqId, long mobCellsCount)
603    throws IOException {
604    if (writer != null) {
605      writer.appendMetadata(maxSeqId, false, mobCellsCount);
606      try {
607        writer.close();
608      } catch (IOException e) {
609        LOG.error("Failed to close the writer of the file " + writer.getPath(), e);
610      }
611    }
612  }

614  /**
615   * Closes the ref file writer.
616   * @param writer The ref file writer.
617   * @param maxSeqId Maximum sequence id.
618   * @param bulkloadTime The timestamp at which the bulk load file is created.
619   * @throws IOException if IO failure is encountered
620   */
621  private void closeRefFileWriter(StoreFileWriter writer, long maxSeqId, long bulkloadTime)
622    throws IOException {
623    if (writer != null) {
624      writer.appendMetadata(maxSeqId, false);
625      writer.appendFileInfo(StoreFile.BULKLOAD_TIME_KEY, Bytes.toBytes(bulkloadTime));
626      writer.appendFileInfo(StoreFile.SKIP_RESET_SEQ_ID, Bytes.toBytes(true));
627      try {
628        writer.close();
629      } catch (IOException e) {
630        LOG.error("Failed to close the writer of the ref file " + writer.getPath(), e);
631      }
632    }
633  }

635  /**
636   * Gets the max seqId and number of cells of the store files.
637   * @param storeFiles The store files.
638   * @return The pair of the max seqId and number of cells of the store files.
639   * @throws IOException if IO failure is encountered
640   */
641  private Pair<Long, Long> getFileInfo(List<StoreFile> storeFiles) throws IOException {
642    long maxSeqId = 0;
643    long maxKeyCount = 0;
644    for (StoreFile sf : storeFiles) {
645      // the readers will be closed later after the merge.
646      maxSeqId = Math.max(maxSeqId, sf.getMaxSequenceId());
647      byte[] count = sf.createReader().loadFileInfo().get(StoreFile.MOB_CELLS_COUNT);
648      if (count != null) {
649        maxKeyCount += Bytes.toLong(count);
650      }
651    }
652    return new Pair<>(Long.valueOf(maxSeqId), Long.valueOf(maxKeyCount));
653  }

669  private FileStatus getLinkedFileStatus(HFileLink link) throws IOException {
670    Path[] locations = link.getLocations();
671    for (Path location : locations) {
672      FileStatus file = getFileStatus(location);
673      if (file != null) {
674        return file;
675      }
676    }
677    return null;
678  }
679
680  private FileStatus getFileStatus(Path path) throws IOException {
681    try {
682      if (path != null) {
683        FileStatus file = fs.getFileStatus(path);
684        return file;
685      }
686    } catch (FileNotFoundException e) {
687      LOG.warn("The file " + path + " can not be found", e);
688    }
689    return null;
690  }
