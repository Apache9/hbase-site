042/**
043 * Base class of a WAL Provider that returns a single thread safe WAL that writes to Hadoop FS. By
044 * default, this implementation picks a directory in Hadoop FS based on a combination of
045 * <ul>
046 * <li>the HBase root directory
047 * <li>HConstants.HREGION_LOGDIR_NAME
048 * <li>the given factory's factoryId (usually identifying the regionserver by host:port)
049 * </ul>
050 * It also uses the providerId to differentiate among files.
051 */
052@InterfaceAudience.Private
053@InterfaceStability.Evolving
054public abstract class AbstractFSWALProvider<T extends AbstractFSWAL<?>> implements WALProvider {
055
056  private static final Log LOG = LogFactory.getLog(AbstractFSWALProvider.class);
057
058  // Only public so classes back in regionserver.wal can access
059  public interface Reader extends WAL.Reader {
060    /**
061     * @param fs File system.
062     * @param path Path.
063     * @param c Configuration.
064     * @param s Input stream that may have been pre-opened by the caller; may be null.
065     */
066    void init(FileSystem fs, Path path, Configuration c, FSDataInputStream s) throws IOException;
067  }
068
069  protected volatile T wal;
070  protected WALFactory factory = null;
071  protected Configuration conf = null;
072  protected List<WALActionsListener> listeners = null;
073  protected String providerId = null;
074  protected AtomicBoolean initialized = new AtomicBoolean(false);
075  // for default wal provider, logPrefix won't change
076  protected String logPrefix = null;
077
078  /**
079   * we synchronized on walCreateLock to prevent wal recreation in different threads
080   */
081  private final Object walCreateLock = new Object();
082
083  /**
084   * @param factory factory that made us, identity used for FS layout. may not be null
085   * @param conf may not be null
086   * @param listeners may be null
087   * @param providerId differentiate between providers from one factory, used for FS layout. may be
088   *          null
089   */
090  @Override
091  public void init(WALFactory factory, Configuration conf, List<WALActionsListener> listeners,
092      String providerId) throws IOException {
093    if (!initialized.compareAndSet(false, true)) {
094      throw new IllegalStateException("WALProvider.init should only be called once.");
095    }
096    this.factory = factory;
097    this.conf = conf;
098    this.listeners = listeners;
099    this.providerId = providerId;
100    // get log prefix
101    StringBuilder sb = new StringBuilder().append(factory.factoryId);
102    if (providerId != null) {
103      if (providerId.startsWith(WAL_FILE_NAME_DELIMITER)) {
104        sb.append(providerId);
105      } else {
106        sb.append(WAL_FILE_NAME_DELIMITER).append(providerId);
107      }
108    }
109    logPrefix = sb.toString();
110    doInit(conf);
111  }
112
113  @Override
114  public List<WAL> getWALs() throws IOException {
115    if (wal == null) {
116      return Collections.emptyList();
117    }
118    List<WAL> wals = new ArrayList<>(1);
119    wals.add(wal);
120    return wals;
121  }
122
123  @Override
124  public T getWAL(byte[] identifier, byte[] namespace) throws IOException {
125    T walCopy = wal;
126    if (walCopy == null) {
127      // only lock when need to create wal, and need to lock since
128      // creating hlog on fs is time consuming
129      synchronized (walCreateLock) {
130        walCopy = wal;
131        if (walCopy == null) {
132          walCopy = createWAL();
133          wal = walCopy;
134        }
135      }
136    }
137    return walCopy;
138  }
139
140  protected abstract T createWAL() throws IOException;
141
142  protected abstract void doInit(Configuration conf) throws IOException;
143
144  @Override
145  public void shutdown() throws IOException {
146    T log = this.wal;
147    if (log != null) {
148      log.shutdown();
149    }
150  }
151
152  @Override
153  public void close() throws IOException {
154    T log = this.wal;
155    if (log != null) {
156      log.close();
157    }
158  }

212  // should be package private; more visible for use in AbstractFSWAL
213  public static final String WAL_FILE_NAME_DELIMITER = ".";
214  /** The hbase:meta region's WAL filename extension */
215  @VisibleForTesting

217  static final String DEFAULT_PROVIDER_ID = "default";

220  /** File Extension used while splitting an WAL into regions (HBASE-2312) */
221  public static final String SPLITTING_EXT = "-splitting";

223  /**
224   * It returns the file create timestamp from the file name. For name format see
225   * {@link #validateWALFilename(String)} public until remaining tests move to o.a.h.h.wal
226   * @param wal must not be null
227   * @return the file number that is part of the WAL file name
228   */
229  @VisibleForTesting
230  public static long extractFileNumFromWAL(final WAL wal) {
231    final Path walName = ((AbstractFSWAL<?>) wal).getCurrentFileName();
232    if (walName == null) {
233      throw new IllegalArgumentException("The WAL path couldn't be null");
234    }
235    final String[] walPathStrs = walName.toString().split("\\" + WAL_FILE_NAME_DELIMITER);
236    return Long.parseLong(walPathStrs[walPathStrs.length - (isMetaFile(walName) ? 2 : 1)]);
237  }

239  /**
240   * Pattern used to validate a WAL file name see {@link #validateWALFilename(String)} for
241   * description.
242   */
243  private static final Pattern pattern = Pattern
244      .compile(".*\\.\\d*(" + META_WAL_PROVIDER_ID + ")*");
245
246  /**
247   * A WAL file name is of the format: &lt;wal-name&gt;{@link #WAL_FILE_NAME_DELIMITER}
248   * &lt;file-creation-timestamp&gt;[.meta]. provider-name is usually made up of a server-name and a
249   * provider-id
250   * @param filename name of the file to validate
251   * @return <tt>true</tt> if the filename matches an WAL, <tt>false</tt> otherwise
252   */
253  public static boolean validateWALFilename(String filename) {
254    return pattern.matcher(filename).matches();
255  }

257  /**
258   * Construct the directory name for all WALs on a given server. Dir names currently look like
259   * this for WALs: <code>hbase//WALs/kalashnikov.att.net,61634,1486865297088</code>.
260   * @param serverName Server name formatted as described in {@link ServerName}
261   * @return the relative WAL directory name, e.g. <code>.logs/1.example.org,60030,12345</code> if
262   *         <code>serverName</code> passed is <code>1.example.org,60030,12345</code>
263   */
264  public static String getWALDirectoryName(final String serverName) {
265    StringBuilder dirName = new StringBuilder(HConstants.HREGION_LOGDIR_NAME);
266    dirName.append("/");
267    dirName.append(serverName);
268    return dirName.toString();
269  }

271  /**
272   * Pulls a ServerName out of a Path generated according to our layout rules. In the below layouts,
273   * this method ignores the format of the logfile component. Current format: [base directory for
274   * hbase]/hbase/.logs/ServerName/logfile or [base directory for
275   * hbase]/hbase/.logs/ServerName-splitting/logfile Expected to work for individual log files and
276   * server-specific directories.
277   * @return null if it's not a log file. Returns the ServerName of the region server that created
278   *         this log file otherwise.
279   */
280  public static ServerName getServerNameFromWALDirectoryName(Configuration conf, String path)
281      throws IOException {
282    if (path == null || path.length() <= HConstants.HREGION_LOGDIR_NAME.length()) {
283      return null;
284    }
285
286    if (conf == null) {
287      throw new IllegalArgumentException("parameter conf must be set");
288    }
289
290    final String rootDir = conf.get(HConstants.HBASE_DIR);
291    if (rootDir == null || rootDir.isEmpty()) {
292      throw new IllegalArgumentException(HConstants.HBASE_DIR + " key not found in conf.");
293    }
294
295    final StringBuilder startPathSB = new StringBuilder(rootDir);
296    if (!rootDir.endsWith("/")) {
297      startPathSB.append('/');
298    }
299    startPathSB.append(HConstants.HREGION_LOGDIR_NAME);
300    if (!HConstants.HREGION_LOGDIR_NAME.endsWith("/")) {
301      startPathSB.append('/');
302    }
303    final String startPath = startPathSB.toString();
304
305    String fullPath;
306    try {
307      fullPath = FileSystem.get(conf).makeQualified(new Path(path)).toString();
308    } catch (IllegalArgumentException e) {
309      LOG.info("Call to makeQualified failed on " + path + " " + e.getMessage());
310      return null;
311    }
312
313    if (!fullPath.startsWith(startPath)) {
314      return null;
315    }
316
317    final String serverNameAndFile = fullPath.substring(startPath.length());
318
319    if (serverNameAndFile.indexOf('/') < "a,0,0".length()) {
320      // Either it's a file (not a directory) or it's not a ServerName format
321      return null;
322    }
323
324    Path p = new Path(path);
325    return getServerNameFromWALDirectoryName(p);
326  }

328  /**
329   * This function returns region server name from a log file name which is in one of the following
330   * formats:
331   * <ul>
332   * <li>hdfs://&lt;name node&gt;/hbase/.logs/&lt;server name&gt;-splitting/...</li>
333   * <li>hdfs://&lt;name node&gt;/hbase/.logs/&lt;server name&gt;/...</li>
334   * </ul>
335   * @return null if the passed in logFile isn't a valid WAL file path
336   */
337  public static ServerName getServerNameFromWALDirectoryName(Path logFile) {
338    String logDirName = logFile.getParent().getName();
339    // We were passed the directory and not a file in it.
340    if (logDirName.equals(HConstants.HREGION_LOGDIR_NAME)) {
341      logDirName = logFile.getName();
342    }
343    ServerName serverName = null;
344    if (logDirName.endsWith(SPLITTING_EXT)) {
345      logDirName = logDirName.substring(0, logDirName.length() - SPLITTING_EXT.length());
346    }
347    try {
348      serverName = ServerName.parseServerName(logDirName);
349    } catch (IllegalArgumentException|IllegalStateException ex) {
350      serverName = null;
351      LOG.warn("Cannot parse a server name from path=" + logFile + "; " + ex.getMessage());
352    }
353    if (serverName != null && serverName.getStartcode() < 0) {
354      LOG.warn("Invalid log file path=" + logFile);
355      serverName = null;
356    }
357    return serverName;
358  }

360  public static boolean isMetaFile(Path p) {
361    return isMetaFile(p.getName());
362  }

364  public static boolean isMetaFile(String p) {
365    if (p != null && p.endsWith(META_WAL_PROVIDER_ID)) {
366      return true;
367    }
368    return false;
369  }

371  public static boolean isArchivedLogFile(Path p) {
372    String oldLog = Path.SEPARATOR + HConstants.HREGION_OLDLOGDIR_NAME + Path.SEPARATOR;
373    return p.toString().contains(oldLog);
374  }

376  /**
377   * Get prefix of the log from its name, assuming WAL name in format of
378   * log_prefix.filenumber.log_suffix
379   * @param name Name of the WAL to parse
380   * @return prefix of the log
