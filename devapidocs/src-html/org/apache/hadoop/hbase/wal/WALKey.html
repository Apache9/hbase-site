055/**
056 * A Key for an entry in the WAL.
057 *
058 * The log intermingles edits to many tables and rows, so each log entry
059 * identifies the appropriate table and row.  Within a table and row, they're
060 * also sorted.
061 *
062 * <p>Some Transactional edits (START, COMMIT, ABORT) will not have an associated row.
063 *
064 * Note that protected members marked @InterfaceAudience.Private are only protected
065 * to support the legacy HLogKey class, which is in a different package.
066 */
067// TODO: Key and WALEdit are never used separately, or in one-to-many relation, for practical
068//       purposes. They need to be merged into WALEntry.
069@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.REPLICATION)
070public class WALKey implements SequenceId, Comparable<WALKey> {
071  private static final Log LOG = LogFactory.getLog(WALKey.class);
072  private final CountDownLatch sequenceIdAssignedLatch = new CountDownLatch(1);
073  /**
074   * Used to represent when a particular wal key doesn't know/care about the sequence ordering.
075   */
076  public static final long NO_SEQUENCE_ID = -1;

083  /**
084   * Use it to complete mvcc transaction. This WALKey was part of
085   * (the transaction is started when you call append; see the comment on FSHLog#append). To
086   * complete call
087   * {@link MultiVersionConcurrencyControl#complete(MultiVersionConcurrencyControl.WriteEntry)}
088   * or {@link MultiVersionConcurrencyControl#complete(MultiVersionConcurrencyControl.WriteEntry)}
089   * @return A WriteEntry gotten from local WAL subsystem.
090   * @see #setWriteEntry(MultiVersionConcurrencyControl.WriteEntry)
091   */
092  @InterfaceAudience.Private // For internal use only.
093  public MultiVersionConcurrencyControl.WriteEntry getWriteEntry() throws InterruptedIOException {
094    try {
095      this.sequenceIdAssignedLatch.await();
096    } catch (InterruptedException ie) {
097      MultiVersionConcurrencyControl mvcc = getMvcc();
098      if (LOG.isDebugEnabled()) {
099        LOG.debug("mvcc=" + mvcc + ", writeEntry=" + this.writeEntry);
100      }
101      InterruptedIOException iie = new InterruptedIOException();
102      iie.initCause(ie);
103      throw iie;
104    }
105    return this.writeEntry;
106  }
107
108  @InterfaceAudience.Private // For internal use only.
109  public void setWriteEntry(MultiVersionConcurrencyControl.WriteEntry writeEntry) {
110    if (this.writeEntry != null) {
111      throw new RuntimeException("Non-null!!!");
112    }
113    this.writeEntry = writeEntry;
114    // Set our sequenceid now using WriteEntry.
115    if (this.writeEntry != null) {
116      this.sequenceId = this.writeEntry.getWriteNumber();
117    }
118    this.sequenceIdAssignedLatch.countDown();
119  }

125  @InterfaceAudience.Private
126  protected enum Version {
127    UNVERSIONED(0),
128    // Initial number we put on WALKey when we introduced versioning.
129    INITIAL(-1),
130    // Version -2 introduced a dictionary compression facility.  Only this
131    // dictionary-based compression is available in version -2.
132    COMPRESSED(-2);
133
134    public final int code;
135    static final Version[] byCode;
136    static {
137      byCode = Version.values();
138      for (int i = 0; i < byCode.length; i++) {
139        if (byCode[i].code != -1 * i) {
140          throw new AssertionError("Values in this enum should be descending by one");
141        }
142      }
143    }
144
145    Version(int code) {
146      this.code = code;
147    }
148
149    public boolean atLeast(Version other) {
150      return code <= other.code;
151    }
152
153    public static Version fromCode(int code) {
154      return byCode[code * -1];
155    }
156  }

439  public void readOlderScopes(NavigableMap<byte[], Integer> scopes) {
440    if (scopes != null) {
441      Iterator<Map.Entry<byte[], Integer>> iterator = scopes.entrySet()
442          .iterator();
443      while (iterator.hasNext()) {
444        Map.Entry<byte[], Integer> scope = iterator.next();
445        String key = Bytes.toString(scope.getKey());
446        if (key.startsWith(PREFIX_CLUSTER_KEY)) {
447          addClusterId(UUID.fromString(key.substring(PREFIX_CLUSTER_KEY
448              .length())));
449          iterator.remove();
450        }
451      }
452      if (scopes.size() > 0) {
453        this.scopes = scopes;
454      }
455    }
456  }

522  @Override
523  public int compareTo(WALKey o) {
524    int result = Bytes.compareTo(this.encodedRegionName, o.encodedRegionName);
525    if (result == 0) {
526      long sid = getSequenceId();
527      long otherSid = o.getSequenceId();
528      if (sid < otherSid) {
529        result = -1;
530      } else if (sid  > otherSid) {
531        result = 1;
532      }
533      if (result == 0) {
534        if (this.writeTime < o.writeTime) {
535          result = -1;
536        } else if (this.writeTime > o.writeTime) {
537          return 1;
538        }
539      }
540    }
541    // why isn't cluster id accounted for?
542    return result;
543  }

571  public org.apache.hadoop.hbase.protobuf.generated.WALProtos.WALKey.Builder getBuilder(
572      WALCellCodec.ByteStringCompressor compressor) throws IOException {
573    org.apache.hadoop.hbase.protobuf.generated.WALProtos.WALKey.Builder builder =
574        org.apache.hadoop.hbase.protobuf.generated.WALProtos.WALKey.newBuilder();
575    if (compressionContext == null) {
576      builder.setEncodedRegionName(ByteStringer.wrap(this.encodedRegionName));
577      builder.setTableName(ByteStringer.wrap(this.tablename.getName()));
578    } else {
579      builder.setEncodedRegionName(compressor.compress(this.encodedRegionName,
580          compressionContext.regionDict));
581      builder.setTableName(compressor.compress(this.tablename.getName(),
582          compressionContext.tableDict));
583    }
584    builder.setLogSequenceNumber(getSequenceId());
585    builder.setWriteTime(writeTime);
586    if (this.origLogSeqNum > 0) {
587      builder.setOrigSequenceNumber(this.origLogSeqNum);
588    }
589    if (this.nonce != HConstants.NO_NONCE) {
590      builder.setNonce(nonce);
591    }
592    if (this.nonceGroup != HConstants.NO_NONCE) {
593      builder.setNonceGroup(nonceGroup);
594    }
595    HBaseProtos.UUID.Builder uuidBuilder = HBaseProtos.UUID.newBuilder();
596    for (UUID clusterId : clusterIds) {
597      uuidBuilder.setLeastSigBits(clusterId.getLeastSignificantBits());
598      uuidBuilder.setMostSigBits(clusterId.getMostSignificantBits());
599      builder.addClusterIds(uuidBuilder.build());
600    }
601    if (scopes != null) {
602      for (Map.Entry<byte[], Integer> e : scopes.entrySet()) {
603        ByteString family = (compressionContext == null) ? ByteStringer.wrap(e.getKey())
604            : compressor.compress(e.getKey(), compressionContext.familyDict);
605        builder.addScopes(FamilyScope.newBuilder()
606            .setFamily(family).setScopeType(ScopeType.valueOf(e.getValue())));
607      }
608    }
609    return builder;
610  }

612  public void readFieldsFromPb(org.apache.hadoop.hbase.protobuf.generated.WALProtos.WALKey walKey,
613                               WALCellCodec.ByteStringUncompressor uncompressor)
614      throws IOException {
615    if (this.compressionContext != null) {
616      this.encodedRegionName = uncompressor.uncompress(
617          walKey.getEncodedRegionName(), compressionContext.regionDict);
618      byte[] tablenameBytes = uncompressor.uncompress(
619          walKey.getTableName(), compressionContext.tableDict);
620      this.tablename = TableName.valueOf(tablenameBytes);
621    } else {
622      this.encodedRegionName = walKey.getEncodedRegionName().toByteArray();
623      this.tablename = TableName.valueOf(walKey.getTableName().toByteArray());
624    }
625    clusterIds.clear();
626    if (walKey.hasClusterId()) {
627      //When we are reading the older log (0.95.1 release)
628      //This is definitely the originating cluster
629      clusterIds.add(new UUID(walKey.getClusterId().getMostSigBits(), walKey.getClusterId()
630          .getLeastSigBits()));
631    }
632    for (HBaseProtos.UUID clusterId : walKey.getClusterIdsList()) {
633      clusterIds.add(new UUID(clusterId.getMostSigBits(), clusterId.getLeastSigBits()));
634    }
635    if (walKey.hasNonceGroup()) {
636      this.nonceGroup = walKey.getNonceGroup();
637    }
638    if (walKey.hasNonce()) {
639      this.nonce = walKey.getNonce();
640    }
641    this.scopes = null;
642    if (walKey.getScopesCount() > 0) {
643      this.scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
644      for (FamilyScope scope : walKey.getScopesList()) {
645        byte[] family = (compressionContext == null) ? scope.getFamily().toByteArray() :
646          uncompressor.uncompress(scope.getFamily(), compressionContext.familyDict);
647        this.scopes.put(family, scope.getScopeType().getNumber());
648      }
649    }
650    setSequenceId(walKey.getLogSequenceNumber());
651    this.writeTime = walKey.getWriteTime();
652    if(walKey.hasOrigSequenceNumber()) {
653      this.origLogSeqNum = walKey.getOrigSequenceNumber();
654    }
655  }
