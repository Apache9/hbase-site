057/**
058 * A Key for an entry in the WAL.
059 *
060 * The log intermingles edits to many tables and rows, so each log entry
061 * identifies the appropriate table and row.  Within a table and row, they're
062 * also sorted.
063 *
064 * <p>Some Transactional edits (START, COMMIT, ABORT) will not have an associated row.
065 *
066 * Note that protected members marked @InterfaceAudience.Private are only protected
067 * to support the legacy HLogKey class, which is in a different package.
068 */
069// TODO: Key and WALEdit are never used separately, or in one-to-many relation, for practical
070//       purposes. They need to be merged into WALEntry.
071@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.REPLICATION)
072public class WALKey implements SequenceId, Comparable<WALKey> {
073  private static final Log LOG = LogFactory.getLog(WALKey.class);
074  private final CountDownLatch sequenceIdAssignedLatch = new CountDownLatch(1);

085  /**
086   * Use it to complete mvcc transaction. This WALKey was part of
087   * (the transaction is started when you call append; see the comment on FSHLog#append). To
088   * complete call
089   * {@link MultiVersionConcurrencyControl#complete(MultiVersionConcurrencyControl.WriteEntry)}
090   * or {@link MultiVersionConcurrencyControl#complete(MultiVersionConcurrencyControl.WriteEntry)}
091   * @return A WriteEntry gotten from local WAL subsystem.
092   * @see #setWriteEntry(MultiVersionConcurrencyControl.WriteEntry)
093   */
094  @InterfaceAudience.Private // For internal use only.
095  public MultiVersionConcurrencyControl.WriteEntry getWriteEntry() throws InterruptedIOException {
096    if (this.preAssignedWriteEntry != null) {
097      // don't wait for seqNumAssignedLatch if writeEntry is preassigned
098      return this.preAssignedWriteEntry;
099    }
100    try {
101      this.sequenceIdAssignedLatch.await();
102    } catch (InterruptedException ie) {
103      MultiVersionConcurrencyControl mvcc = getMvcc();
104      if (LOG.isDebugEnabled()) {
105        LOG.debug("mvcc=" + mvcc + ", writeEntry=" + this.writeEntry);
106      }
107      InterruptedIOException iie = new InterruptedIOException();
108      iie.initCause(ie);
109      throw iie;
110    }
111    return this.writeEntry;
112  }
113
114  @InterfaceAudience.Private // For internal use only.
115  public void setWriteEntry(MultiVersionConcurrencyControl.WriteEntry writeEntry) {
116    if (this.writeEntry != null) {
117      throw new RuntimeException("Non-null!!!");
118    }
119    this.writeEntry = writeEntry;
120    // Set our sequenceid now using WriteEntry.
121    if (this.writeEntry != null) {
122      this.sequenceId = this.writeEntry.getWriteNumber();
123    }
124    this.sequenceIdAssignedLatch.countDown();
125  }

132  protected enum Version {
133    UNVERSIONED(0),
134    // Initial number we put on WALKey when we introduced versioning.
135    INITIAL(-1),
136    // Version -2 introduced a dictionary compression facility.  Only this
137    // dictionary-based compression is available in version -2.
138    COMPRESSED(-2);
139
140    public final int code;
141    static final Version[] byCode;
142    static {
143      byCode = Version.values();
144      for (int i = 0; i < byCode.length; i++) {
145        if (byCode[i].code != -1 * i) {
146          throw new AssertionError("Values in this enum should be descending by one");
147        }
148      }
149    }
150
151    Version(int code) {
152      this.code = code;
153    }
154
155    public boolean atLeast(Version other) {
156      return code <= other.code;
157    }
158
159    public static Version fromCode(int code) {
160      return byCode[code * -1];
161    }
162  }

514  public void readOlderScopes(NavigableMap<byte[], Integer> scopes) {
515    if (scopes != null) {
516      Iterator<Map.Entry<byte[], Integer>> iterator = scopes.entrySet()
517          .iterator();
518      while (iterator.hasNext()) {
519        Map.Entry<byte[], Integer> scope = iterator.next();
520        String key = Bytes.toString(scope.getKey());
521        if (key.startsWith(PREFIX_CLUSTER_KEY)) {
522          addClusterId(UUID.fromString(key.substring(PREFIX_CLUSTER_KEY
523              .length())));
524          iterator.remove();
525        }
526      }
527      if (scopes.size() > 0) {
528        this.replicationScope = scopes;
529      }
530    }
531  }

589  @Override

594    return result;

597  @Override
598  public int compareTo(WALKey o) {
599    int result = Bytes.compareTo(this.encodedRegionName, o.encodedRegionName);
600    if (result == 0) {
601      long sid = getSequenceId();
602      long otherSid = o.getSequenceId();
603      if (sid < otherSid) {
604        result = -1;
605      } else if (sid  > otherSid) {
606        result = 1;
607      }
608      if (result == 0) {
609        if (this.writeTime < o.writeTime) {
610          result = -1;
611        } else if (this.writeTime > o.writeTime) {
612          return 1;
613        }
614      }
615    }
616    // why isn't cluster id accounted for?
617    return result;
618  }

646  public WALProtos.WALKey.Builder getBuilder(
647      WALCellCodec.ByteStringCompressor compressor) throws IOException {
648    WALProtos.WALKey.Builder builder = WALProtos.WALKey.newBuilder();
649    if (compressionContext == null) {
650      builder.setEncodedRegionName(UnsafeByteOperations.unsafeWrap(this.encodedRegionName));
651      builder.setTableName(UnsafeByteOperations.unsafeWrap(this.tablename.getName()));
652    } else {
653      builder.setEncodedRegionName(compressor.compress(this.encodedRegionName,
654          compressionContext.regionDict));
655      builder.setTableName(compressor.compress(this.tablename.getName(),
656          compressionContext.tableDict));
657    }
658    builder.setLogSequenceNumber(getSequenceId());
659    builder.setWriteTime(writeTime);
660    if (this.origLogSeqNum > 0) {
661      builder.setOrigSequenceNumber(this.origLogSeqNum);
662    }
663    if (this.nonce != HConstants.NO_NONCE) {
664      builder.setNonce(nonce);
665    }
666    if (this.nonceGroup != HConstants.NO_NONCE) {
667      builder.setNonceGroup(nonceGroup);
668    }
669    HBaseProtos.UUID.Builder uuidBuilder = HBaseProtos.UUID.newBuilder();
670    for (UUID clusterId : clusterIds) {
671      uuidBuilder.setLeastSigBits(clusterId.getLeastSignificantBits());
672      uuidBuilder.setMostSigBits(clusterId.getMostSignificantBits());
673      builder.addClusterIds(uuidBuilder.build());
674    }
675    if (replicationScope != null) {
676      for (Map.Entry<byte[], Integer> e : replicationScope.entrySet()) {
677        ByteString family = (compressionContext == null)
678            ? UnsafeByteOperations.unsafeWrap(e.getKey())
679            : compressor.compress(e.getKey(), compressionContext.familyDict);
680        builder.addScopes(FamilyScope.newBuilder()
681            .setFamily(family).setScopeType(ScopeType.forNumber(e.getValue())));
682      }
683    }
684    return builder;
685  }
686
687  public void readFieldsFromPb(WALProtos.WALKey walKey,
688                               WALCellCodec.ByteStringUncompressor uncompressor)
689      throws IOException {
690    if (this.compressionContext != null) {
691      this.encodedRegionName = uncompressor.uncompress(
692          walKey.getEncodedRegionName(), compressionContext.regionDict);
693      byte[] tablenameBytes = uncompressor.uncompress(
694          walKey.getTableName(), compressionContext.tableDict);
695      this.tablename = TableName.valueOf(tablenameBytes);
696    } else {
697      this.encodedRegionName = walKey.getEncodedRegionName().toByteArray();
698      this.tablename = TableName.valueOf(walKey.getTableName().toByteArray());
699    }
700    clusterIds.clear();
701    for (HBaseProtos.UUID clusterId : walKey.getClusterIdsList()) {
702      clusterIds.add(new UUID(clusterId.getMostSigBits(), clusterId.getLeastSigBits()));
703    }
704    if (walKey.hasNonceGroup()) {
705      this.nonceGroup = walKey.getNonceGroup();
706    }
707    if (walKey.hasNonce()) {
708      this.nonce = walKey.getNonce();
709    }
710    this.replicationScope = null;
711    if (walKey.getScopesCount() > 0) {
712      this.replicationScope = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
713      for (FamilyScope scope : walKey.getScopesList()) {
714        byte[] family = (compressionContext == null) ? scope.getFamily().toByteArray() :
715          uncompressor.uncompress(scope.getFamily(), compressionContext.familyDict);
716        this.replicationScope.put(family, scope.getScopeType().getNumber());
717      }
718    }
719    setSequenceId(walKey.getLogSequenceNumber());
720    this.writeTime = walKey.getWriteTime();
721    if(walKey.hasOrigSequenceNumber()) {
722      this.origLogSeqNum = walKey.getOrigSequenceNumber();
723    }
724  }
