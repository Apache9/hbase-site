050/**
051 * Entry point for users of the Write Ahead Log.
052 * Acts as the shim between internal use and the particular WALProvider we use to handle wal
053 * requests.
054 *
055 * Configure which provider gets used with the configuration setting "hbase.wal.provider". Available
056 * implementations:
057 * <ul>
058 *   <li><em>defaultProvider</em> : whatever provider is standard for the hbase version. Currently
059 *                                  "filesystem"</li>
060 *   <li><em>filesystem</em> : a provider that will run on top of an implementation of the Hadoop
061 *                             FileSystem interface, normally HDFS.</li>
062 *   <li><em>multiwal</em> : a provider that will use multiple "filesystem" wal instances per region
063 *                           server.</li>
064 * </ul>
065 *
066 * Alternatively, you may provide a custom implementation of {@link WALProvider} by class name.
067 */
068@InterfaceAudience.Private
069public class WALFactory {
070
071  private static final Log LOG = LogFactory.getLog(WALFactory.class);
072
073  /**
074   * Maps between configuration names for providers and implementation classes.
075   */
076  static enum Providers {
077    defaultProvider(DefaultWALProvider.class),
078    filesystem(DefaultWALProvider.class),
079    multiwal(RegionGroupingProvider.class),
080    asyncfs(AsyncFSWALProvider.class);
081
082    Class<? extends WALProvider> clazz;
083    Providers(Class<? extends WALProvider> clazz) {
084      this.clazz = clazz;
085    }
086  }

092  static final String DEFAULT_META_WAL_PROVIDER = Providers.defaultProvider.name();

113  // Used for the singleton WALFactory, see below.
114  private WALFactory(Configuration conf) {
115    // this code is duplicated here so we can keep our members final.
116    // until we've moved reader/writer construction down into providers, this initialization must
117    // happen prior to provider initialization, in case they need to instantiate a reader/writer.
118    timeoutMillis = conf.getInt("hbase.hlog.open.timeout", 300000);
119    /* TODO Both of these are probably specific to the fs wal provider */
120    logReaderClass = conf.getClass("hbase.regionserver.hlog.reader.impl", ProtobufLogReader.class,
121        DefaultWALProvider.Reader.class);
122    this.conf = conf;
123    // end required early initialization
124
125    // this instance can't create wals, just reader/writers.
126    provider = null;
127    factoryId = SINGLETON_ID;
128  }

130  /**
131   * instantiate a provider from a config property.
132   * requires conf to have already been set (as well as anything the provider might need to read).
133   */
134  WALProvider getProvider(final String key, final String defaultValue,
135      final List<WALActionsListener> listeners, final String providerId) throws IOException {
136    Class<? extends WALProvider> clazz;
137    try {
138      clazz = Providers.valueOf(conf.get(key, defaultValue)).clazz;
139    } catch (IllegalArgumentException exception) {
140      // Fall back to them specifying a class name
141      // Note that the passed default class shouldn't actually be used, since the above only fails
142      // when there is a config value present.
143      clazz = conf.getClass(key, DefaultWALProvider.class, WALProvider.class);
144    }
145    LOG.info("Instantiating WALProvider of type " + clazz);
146    try {
147      final WALProvider result = clazz.newInstance();
148      result.init(this, conf, listeners, providerId);
149      return result;
150    } catch (InstantiationException exception) {
151      LOG.error("couldn't set up WALProvider, check config key " + key);
152      LOG.debug("Exception details for failure to load WALProvider.", exception);
153      throw new IOException("couldn't set up WALProvider", exception);
154    } catch (IllegalAccessException exception) {
155      LOG.error("couldn't set up WALProvider, check config key " + key);
156      LOG.debug("Exception details for failure to load WALProvider.", exception);
157      throw new IOException("couldn't set up WALProvider", exception);
158    }
159  }

161  /**
162   * @param conf must not be null, will keep a reference to read params in later reader/writer
163   *     instances.
164   * @param listeners may be null. will be given to all created wals (and not meta-wals)
165   * @param factoryId a unique identifier for this factory. used i.e. by filesystem implementations
166   *     to make a directory
167   */
168  public WALFactory(final Configuration conf, final List<WALActionsListener> listeners,
169      final String factoryId) throws IOException {
170    // until we've moved reader/writer construction down into providers, this initialization must
171    // happen prior to provider initialization, in case they need to instantiate a reader/writer.
172    timeoutMillis = conf.getInt("hbase.hlog.open.timeout", 300000);
173    /* TODO Both of these are probably specific to the fs wal provider */
174    logReaderClass = conf.getClass("hbase.regionserver.hlog.reader.impl", ProtobufLogReader.class,
175        DefaultWALProvider.Reader.class);
176    this.conf = conf;
177    this.factoryId = factoryId;
178    // end required early initialization
179    if (conf.getBoolean("hbase.regionserver.hlog.enabled", true)) {
180      provider = getProvider(WAL_PROVIDER, DEFAULT_WAL_PROVIDER, listeners, null);
181    } else {
182      // special handling of existing configuration behavior.
183      LOG.warn("Running with WAL disabled.");
184      provider = new DisabledWALProvider();
185      provider.init(this, conf, null, factoryId);
186    }
187  }

189  /**
190   * Shutdown all WALs and clean up any underlying storage.
191   * Use only when you will not need to replay and edits that have gone to any wals from this
192   * factory.
193   */
194  public void close() throws IOException {
195    final WALProvider metaProvider = this.metaProvider.get();
196    if (null != metaProvider) {
197      metaProvider.close();
198    }
199    // close is called on a WALFactory with null provider in the case of contention handling
200    // within the getInstance method.
201    if (null != provider) {
202      provider.close();
203    }
204  }
205
206  /**
207   * Tell the underlying WAL providers to shut down, but do not clean up underlying storage.
208   * If you are not ending cleanly and will need to replay edits from this factory's wals,
209   * use this method if you can as it will try to leave things as tidy as possible.
210   */
211  public void shutdown() throws IOException {
212    IOException exception = null;
213    final WALProvider metaProvider = this.metaProvider.get();
214    if (null != metaProvider) {
215      try {
216        metaProvider.shutdown();
217      } catch(IOException ioe) {
218        exception = ioe;
219      }
220    }
221    provider.shutdown();
222    if (null != exception) {
223      throw exception;
224    }
225  }

235  /**
236   * @param identifier may not be null, contents will not be altered
237   */
238  public WAL getMetaWAL(final byte[] identifier) throws IOException {
239    WALProvider metaProvider = this.metaProvider.get();
240    if (null == metaProvider) {
241      final WALProvider temp = getProvider(META_WAL_PROVIDER, DEFAULT_META_WAL_PROVIDER,
242          Collections.<WALActionsListener>singletonList(new MetricsWAL()),
243          DefaultWALProvider.META_WAL_PROVIDER_ID);
244      if (this.metaProvider.compareAndSet(null, temp)) {
245        metaProvider = temp;
246      } else {
247        // reference must now be to a provider created in another thread.
248        temp.close();
249        metaProvider = this.metaProvider.get();
250      }
251    }
252    return metaProvider.getWAL(identifier, null);
253  }

271  public Reader createReader(final FileSystem fs, final Path path,
272      CancelableProgressable reporter, boolean allowCustom)
273      throws IOException {
274    Class<? extends DefaultWALProvider.Reader> lrClass =
275        allowCustom ? logReaderClass : ProtobufLogReader.class;
276
277    try {
278      // A wal file could be under recovery, so it may take several
279      // tries to get it open. Instead of claiming it is corrupted, retry
280      // to open it up to 5 minutes by default.
281      long startWaiting = EnvironmentEdgeManager.currentTime();
282      long openTimeout = timeoutMillis + startWaiting;
283      int nbAttempt = 0;
284      FSDataInputStream stream = null;
285      while (true) {
286        try {
287          if (lrClass != ProtobufLogReader.class) {
288            // User is overriding the WAL reader, let them.
289            DefaultWALProvider.Reader reader = lrClass.newInstance();
290            reader.init(fs, path, conf, null);
291            return reader;
292          } else {
293            stream = fs.open(path);
294            // Note that zero-length file will fail to read PB magic, and attempt to create
295            // a non-PB reader and fail the same way existing code expects it to. If we get
296            // rid of the old reader entirely, we need to handle 0-size files differently from
297            // merely non-PB files.
298            byte[] magic = new byte[ProtobufLogReader.PB_WAL_MAGIC.length];
299            boolean isPbWal = (stream.read(magic) == magic.length)
300                && Arrays.equals(magic, ProtobufLogReader.PB_WAL_MAGIC);
301            DefaultWALProvider.Reader reader =
302                isPbWal ? new ProtobufLogReader() : new SequenceFileLogReader();
303            reader.init(fs, path, conf, stream);
304            return reader;
305          }
306        } catch (IOException e) {
307          try {
308            if (stream != null) {
309              stream.close();
310            }
311          } catch (IOException exception) {
312            LOG.warn("Could not close FSDataInputStream" + exception.getMessage());
313            LOG.debug("exception details", exception);
314          }
315          String msg = e.getMessage();
316          if (msg != null && (msg.contains("Cannot obtain block length")
317              || msg.contains("Could not obtain the last block")
318              || msg.matches("Blocklist for [^ ]* has changed.*"))) {
319            if (++nbAttempt == 1) {
320              LOG.warn("Lease should have recovered. This is not expected. Will retry", e);
321            }
322            if (reporter != null && !reporter.progress()) {
323              throw new InterruptedIOException("Operation is cancelled");
324            }
325            if (nbAttempt > 2 && openTimeout < EnvironmentEdgeManager.currentTime()) {
326              LOG.error("Can't open after " + nbAttempt + " attempts and "
327                + (EnvironmentEdgeManager.currentTime() - startWaiting)
328                + "ms " + " for " + path);
329            } else {
330              try {
331                Thread.sleep(nbAttempt < 3 ? 500 : 1000);
332                continue; // retry
333              } catch (InterruptedException ie) {
334                InterruptedIOException iioe = new InterruptedIOException();
335                iioe.initCause(ie);
336                throw iioe;
337              }
338            }
339            throw new LeaseNotRecoveredException(e);
340          } else {
341            throw e;
342          }
343        }
344      }
345    } catch (IOException ie) {
346      throw ie;
347    } catch (Exception e) {
348      throw new IOException("Cannot get log reader", e);
349    }
350  }

374  // These static methods are currently used where it's impractical to
375  // untangle the reliance on state in the filesystem. They rely on singleton
376  // WALFactory that just provides Reader / Writers.
377  // For now, first Configuration object wins. Practically this just impacts the reader/writer class
378  private static final AtomicReference<WALFactory> singleton = new AtomicReference<WALFactory>();
379  private static final String SINGLETON_ID = WALFactory.class.getName();
380  
381  // public only for FSHLog
382  public static WALFactory getInstance(Configuration configuration) {
383    WALFactory factory = singleton.get();
384    if (null == factory) {
385      WALFactory temp = new WALFactory(configuration);
386      if (singleton.compareAndSet(null, temp)) {
387        factory = temp;
388      } else {
389        // someone else beat us to initializing
390        try {
391          temp.close();
392        } catch (IOException exception) {
393          LOG.debug("failed to close temporary singleton. ignoring.", exception);
394        }
395        factory = singleton.get();
396      }
397    }
398    return factory;
399  }
