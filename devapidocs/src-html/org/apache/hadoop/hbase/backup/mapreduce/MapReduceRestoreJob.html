018package org.apache.hadoop.hbase.backup.mapreduce;
019
020import java.io.IOException;
021
022import org.apache.commons.lang.StringUtils;
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.fs.FileSystem;
027import org.apache.hadoop.fs.Path;
028import org.apache.hadoop.hbase.HConstants;
029import org.apache.hadoop.hbase.TableName;
030import org.apache.hadoop.hbase.backup.BackupRestoreConstants;
031import org.apache.hadoop.hbase.backup.RestoreJob;
032import org.apache.hadoop.hbase.classification.InterfaceAudience;
033import org.apache.hadoop.hbase.mapreduce.LoadIncrementalHFiles;
034import org.apache.hadoop.hbase.mapreduce.WALPlayer;
035import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
036import org.apache.hadoop.util.Tool;
037
038/**
039 * MapReduce implementation of {@link RestoreJob}
040 *
041 * For full backup restore, it runs {@link HFileSplitterJob} job and creates
042 * HFiles which are aligned with a region boundaries of a table being
043 * restored, for incremental backup restore it runs {@link WALPlayer} in
044 * bulk load mode (creates HFiles from WAL edits).
045 *
046 * The resulting HFiles then are loaded using HBase bulk load tool
047 * {@link LoadIncrementalHFiles}
048 */
049@InterfaceAudience.Private
050public class MapReduceRestoreJob implements RestoreJob {
051  public static final Log LOG = LogFactory.getLog(MapReduceRestoreJob.class);
052
053  private Tool player;
054  private Configuration conf;
055
056  public MapReduceRestoreJob() {
057  }
058
059  @Override
060  public void run(Path[] dirPaths, TableName[] tableNames, TableName[] newTableNames,
061      boolean fullBackupRestore) throws IOException {
062
063    String bulkOutputConfKey;
064
065    player = new HFileSplitterJob();
066    bulkOutputConfKey = HFileSplitterJob.BULK_OUTPUT_CONF_KEY;
067    // Player reads all files in arbitrary directory structure and creates
068    // a Map task for each file
069    String dirs = StringUtils.join(dirPaths, ",");
070
071    if (LOG.isDebugEnabled()) {
072      LOG.debug("Restore " + (fullBackupRestore ? "full" : "incremental")
073          + " backup from directory " + dirs + " from hbase tables "
074          + StringUtils.join(tableNames, BackupRestoreConstants.TABLENAME_DELIMITER_IN_COMMAND) +
075          " to tables "
076          + StringUtils.join(newTableNames, BackupRestoreConstants.TABLENAME_DELIMITER_IN_COMMAND));
077    }
078
079    for (int i = 0; i < tableNames.length; i++) {
080
081      LOG.info("Restore " + tableNames[i] + " into " + newTableNames[i]);
082
083      Path bulkOutputPath = getBulkOutputDir(getFileNameCompatibleString(newTableNames[i]));
084      Configuration conf = getConf();
085      conf.set(bulkOutputConfKey, bulkOutputPath.toString());
086      String[] playerArgs =
087        { dirs,
088          fullBackupRestore? newTableNames[i].getNameAsString():tableNames[i].getNameAsString()
089        };
090
091      int result = 0;
092      int loaderResult = 0;
093      try {
094
095        player.setConf(getConf());
096        result = player.run(playerArgs);
097        if (succeeded(result)) {
098          // do bulk load
099          LoadIncrementalHFiles loader = createLoader(getConf());
100          if (LOG.isDebugEnabled()) {
101            LOG.debug("Restoring HFiles from directory " + bulkOutputPath);
102          }
103          String[] args = { bulkOutputPath.toString(), newTableNames[i].getNameAsString() };
104          loaderResult = loader.run(args);
105
106          if (failed(loaderResult)) {
107            throw new IOException("Can not restore from backup directory " + dirs
108                + " (check Hadoop and HBase logs). Bulk loader return code =" + loaderResult);
109          }
110        } else {
111          throw new IOException("Can not restore from backup directory " + dirs
112              + " (check Hadoop/MR and HBase logs). Player return code =" + result);
113        }
114        LOG.debug("Restore Job finished:" + result);
115      } catch (Exception e) {
116        throw new IOException("Can not restore from backup directory " + dirs
117            + " (check Hadoop and HBase logs) ", e);
118      }
119
120    }
121  }
122
123  private String getFileNameCompatibleString(TableName table) {
124    return table.getNamespaceAsString() + "-" + table.getQualifierAsString();
125  }
126
127  private boolean failed(int result) {
128    return result != 0;
129  }
130
131  private boolean succeeded(int result) {
132    return result == 0;
133  }
134
135  public static LoadIncrementalHFiles createLoader(Configuration config) throws IOException {
136    // set configuration for restore:
137    // LoadIncrementalHFile needs more time
138    // <name>hbase.rpc.timeout</name> <value>600000</value>
139    // calculates
140    Integer milliSecInHour = 3600000;
141    Configuration conf = new Configuration(config);
142    conf.setInt(HConstants.HBASE_RPC_TIMEOUT_KEY, milliSecInHour);
143
144    // By default, it is 32 and loader will fail if # of files in any region exceed this
145    // limit. Bad for snapshot restore.
146    conf.setInt(LoadIncrementalHFiles.MAX_FILES_PER_REGION_PER_FAMILY, Integer.MAX_VALUE);
147    conf.set(LoadIncrementalHFiles.IGNORE_UNMATCHED_CF_CONF_KEY, "yes");
148    LoadIncrementalHFiles loader = null;
149    try {
150      loader = new LoadIncrementalHFiles(conf);
151    } catch (Exception e) {
152      throw new IOException(e);
153    }
154    return loader;
155  }
156
157  private Path getBulkOutputDir(String tableName) throws IOException {
158    Configuration conf = getConf();
159    FileSystem fs = FileSystem.get(conf);
160    String tmp =
161        conf.get(HConstants.TEMPORARY_FS_DIRECTORY_KEY,
162          HConstants.DEFAULT_TEMPORARY_HDFS_DIRECTORY);
163    Path path =
164        new Path(tmp + Path.SEPARATOR + "bulk_output-" + tableName + "-"
165            + EnvironmentEdgeManager.currentTime());
166    fs.deleteOnExit(path);
167    return path;
168  }
169
170  @Override
171  public Configuration getConf() {
172    return conf;
173  }
174
175  @Override
176  public void setConf(Configuration conf) {
177    this.conf = conf;
178  }
