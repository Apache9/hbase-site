043/**
044 * Full table backup implementation
045 *
046 */
047@InterfaceAudience.Private
048public class FullTableBackupClient extends TableBackupClient {
049  private static final Log LOG = LogFactory.getLog(FullTableBackupClient.class);
050
051  public FullTableBackupClient(final Connection conn, final String backupId, BackupRequest request)
052      throws IOException {
053    super(conn, backupId, request);
054  }
055
056  /**
057   * Do snapshot copy.
058   * @param backupInfo backup info
059   * @throws Exception exception
060   */
061  private void snapshotCopy(BackupInfo backupInfo) throws Exception {
062    LOG.info("Snapshot copy is starting.");
063
064    // set overall backup phase: snapshot_copy
065    backupInfo.setPhase(BackupPhase.SNAPSHOTCOPY);
066
067    // call ExportSnapshot to copy files based on hbase snapshot for backup
068    // ExportSnapshot only support single snapshot export, need loop for multiple tables case
069    BackupCopyJob copyService = BackupRestoreFactory.getBackupCopyJob(conf);
070
071    // number of snapshots matches number of tables
072    float numOfSnapshots = backupInfo.getSnapshotNames().size();
073
074    LOG.debug("There are " + (int) numOfSnapshots + " snapshots to be copied.");
075
076    for (TableName table : backupInfo.getTables()) {
077      // Currently we simply set the sub copy tasks by counting the table snapshot number, we can
078      // calculate the real files' size for the percentage in the future.
079      // backupCopier.setSubTaskPercntgInWholeTask(1f / numOfSnapshots);
080      int res = 0;
081      String[] args = new String[4];
082      args[0] = "-snapshot";
083      args[1] = backupInfo.getSnapshotName(table);
084      args[2] = "-copy-to";
085      args[3] = backupInfo.getTableBackupDir(table);
086
087      LOG.debug("Copy snapshot " + args[1] + " to " + args[3]);
088      res = copyService.copy(backupInfo, backupManager, conf, BackupType.FULL, args);
089      // if one snapshot export failed, do not continue for remained snapshots
090      if (res != 0) {
091        LOG.error("Exporting Snapshot " + args[1] + " failed with return code: " + res + ".");
092
093        throw new IOException("Failed of exporting snapshot " + args[1] + " to " + args[3]
094            + " with reason code " + res);
095      }
096      LOG.info("Snapshot copy " + args[1] + " finished.");
097    }
098  }
099
100  /**
101   * Backup request execution
102   * @throws IOException
103   */
104  @Override
105  public void execute() throws IOException {
106
107    try (Admin admin = conn.getAdmin();) {
108
109      // Begin BACKUP
110      beginBackup(backupManager, backupInfo);
111      String savedStartCode = null;
112      boolean firstBackup = false;
113      // do snapshot for full table backup
114
115      savedStartCode = backupManager.readBackupStartCode();
116      firstBackup = savedStartCode == null || Long.parseLong(savedStartCode) == 0L;
117      if (firstBackup) {
118        // This is our first backup. Let's put some marker to system table so that we can hold the logs
119        // while we do the backup.
120        backupManager.writeBackupStartCode(0L);
121      }
122      // We roll log here before we do the snapshot. It is possible there is duplicate data
123      // in the log that is already in the snapshot. But if we do it after the snapshot, we
124      // could have data loss.
125      // A better approach is to do the roll log on each RS in the same global procedure as
126      // the snapshot.
127      LOG.info("Execute roll log procedure for full backup ...");
128
129      Map<String, String> props = new HashMap<String, String>();
130      props.put("backupRoot", backupInfo.getBackupRootDir());
131      admin.execProcedure(LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_SIGNATURE,
132        LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_NAME, props);
133
134      newTimestamps = backupManager.readRegionServerLastLogRollResult();
135      if (firstBackup) {
136        // Updates registered log files
137        // We record ALL old WAL files as registered, because
138        // this is a first full backup in the system and these
139        // files are not needed for next incremental backup
140        List<String> logFiles = BackupUtils.getWALFilesOlderThan(conf, newTimestamps);
141        backupManager.recordWALFiles(logFiles);
142      }
143
144      // SNAPSHOT_TABLES:
145      backupInfo.setPhase(BackupPhase.SNAPSHOT);
146      for (TableName tableName : tableList) {
147        String snapshotName =
148            "snapshot_" + Long.toString(EnvironmentEdgeManager.currentTime()) + "_"
149                + tableName.getNamespaceAsString() + "_" + tableName.getQualifierAsString();
150
151        admin.snapshot(snapshotName, tableName);
152
153        backupInfo.setSnapshotName(tableName, snapshotName);
154      }
155
156      // SNAPSHOT_COPY:
157      // do snapshot copy
158      LOG.debug("snapshot copy for " + backupId);
159      snapshotCopy(backupInfo);
160      // Updates incremental backup table set
161      backupManager.addIncrementalBackupTableSet(backupInfo.getTables());
162
163      // BACKUP_COMPLETE:
164      // set overall backup status: complete. Here we make sure to complete the backup.
165      // After this checkpoint, even if entering cancel process, will let the backup finished
166      backupInfo.setState(BackupState.COMPLETE);
167      // The table list in backupInfo is good for both full backup and incremental backup.
168      // For incremental backup, it contains the incremental backup table set.
169      backupManager.writeRegionServerLogTimestamp(backupInfo.getTables(), newTimestamps);
170
171      HashMap<TableName, HashMap<String, Long>> newTableSetTimestampMap =
172          backupManager.readLogTimestampMap();
173
174      Long newStartCode =
175          BackupUtils.getMinValue(BackupUtils
176              .getRSLogTimestampMins(newTableSetTimestampMap));
177      backupManager.writeBackupStartCode(newStartCode);
178
179      // backup complete
180      completeBackup(conn, backupInfo, backupManager, BackupType.FULL, conf);
181    } catch (Exception e) {
182      failBackup(conn, backupInfo, backupManager, e, "Unexpected BackupException : ",
183        BackupType.FULL, conf);
184      throw new IOException(e);
185    }
186
187  }
