074  @Override
075  public BackupInfo getBackupInfo(String backupId) throws IOException {
076    BackupInfo backupInfo = null;
077    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
078      if (backupId == null) {
079        ArrayList<BackupInfo> recentSessions = table.getBackupInfos(BackupState.RUNNING);
080        if (recentSessions.isEmpty()) {
081          LOG.warn("No ongoing sessions found.");
082          return null;
083        }
084        // else show status for ongoing session
085        // must be one maximum
086        return recentSessions.get(0);
087      } else {
088        backupInfo = table.readBackupInfo(backupId);
089        return backupInfo;
090      }
091    }
092  }

094  @Override
095  public int deleteBackups(String[] backupIds) throws IOException {
096    // TODO: requires Fault tolerance support, failure will leave system
097    // in a non-consistent state
098    // see HBASE-15227
099    int totalDeleted = 0;
100    Map<String, HashSet<TableName>> allTablesMap = new HashMap<String, HashSet<TableName>>();
101
102    try (final BackupSystemTable sysTable = new BackupSystemTable(conn)) {
103      for (int i = 0; i < backupIds.length; i++) {
104        BackupInfo info = sysTable.readBackupInfo(backupIds[i]);
105        if (info != null) {
106          String rootDir = info.getBackupRootDir();
107          HashSet<TableName> allTables = allTablesMap.get(rootDir);
108          if (allTables == null) {
109            allTables = new HashSet<TableName>();
110            allTablesMap.put(rootDir, allTables);
111          }
112          allTables.addAll(info.getTableNames());
113          totalDeleted += deleteBackup(backupIds[i], sysTable);
114        }
115      }
116      finalizeDelete(allTablesMap, sysTable);
117    }
118    return totalDeleted;
119  }

121  /**
122   * Updates incremental backup set for every backupRoot
123   * @param tablesMap map [backupRoot: Set<TableName>]
124   * @param table backup system table
125   * @throws IOException
126   */
127
128  private void finalizeDelete(Map<String, HashSet<TableName>> tablesMap, BackupSystemTable table)
129      throws IOException {
130    for (String backupRoot : tablesMap.keySet()) {
131      Set<TableName> incrTableSet = table.getIncrementalBackupTableSet(backupRoot);
132      Map<TableName, ArrayList<BackupInfo>> tableMap =
133          table.getBackupHistoryForTableSet(incrTableSet, backupRoot);
134      for (Map.Entry<TableName, ArrayList<BackupInfo>> entry : tableMap.entrySet()) {
135        if (entry.getValue() == null) {
136          // No more backups for a table
137          incrTableSet.remove(entry.getKey());
138        }
139      }
140      if (!incrTableSet.isEmpty()) {
141        table.addIncrementalBackupTableSet(incrTableSet, backupRoot);
142      } else { // empty
143        table.deleteIncrementalBackupTableSet(backupRoot);
144      }
145    }
146  }

148  /**
149   * Delete single backup and all related backups <br>
150   * Algorithm:<br>
151   *  Backup type: FULL or INCREMENTAL <br>
152   *  Is this last backup session for table T: YES or NO <br>
153   *  For every table T from table list 'tables':<br>
154   *  if(FULL, YES) deletes only physical data (PD) <br>
155   *  if(FULL, NO), deletes PD, scans all newer backups and removes T from backupInfo,<br>
156   *  until we either reach the most recent backup for T in the system or FULL backup<br>
157   *  which includes T<br>
158   *  if(INCREMENTAL, YES) deletes only physical data (PD)
159   *  if(INCREMENTAL, NO) deletes physical data and for table T scans all backup images between last<br>
160   *  FULL backup, which is older than the backup being deleted and the next FULL backup (if exists) <br>
161   *  or last one for a particular table T and removes T from list of backup tables.
162   * @param backupId backup id
163   * @param sysTable backup system table
164   * @return total number of deleted backup images
165   * @throws IOException
166   */
167  private int deleteBackup(String backupId, BackupSystemTable sysTable) throws IOException {
168
169    BackupInfo backupInfo = sysTable.readBackupInfo(backupId);
170
171    int totalDeleted = 0;
172    if (backupInfo != null) {
173      LOG.info("Deleting backup " + backupInfo.getBackupId() + " ...");
174      BackupUtils.cleanupBackupData(backupInfo, conn.getConfiguration());
175      // List of tables in this backup;
176      List<TableName> tables = backupInfo.getTableNames();
177      long startTime = backupInfo.getStartTs();
178      for (TableName tn : tables) {
179        boolean isLastBackupSession = isLastBackupSession(sysTable, tn, startTime);
180        if (isLastBackupSession) {
181          continue;
182        }
183        // else
184        List<BackupInfo> affectedBackups = getAffectedBackupInfos(backupInfo, tn, sysTable);
185        for (BackupInfo info : affectedBackups) {
186          if (info.equals(backupInfo)) {
187            continue;
188          }
189          removeTableFromBackupImage(info, tn, sysTable);
190        }
191      }
192      Map<byte[], String> map = sysTable.readBulkLoadedFiles(backupId);
193      FileSystem fs = FileSystem.get(conn.getConfiguration());
194      boolean succ = true;
195      int numDeleted = 0;
196      for (String f : map.values()) {
197        Path p = new Path(f);
198        try {
199          LOG.debug("Delete backup info " + p + " for " + backupInfo.getBackupId());
200          if (!fs.delete(p)) {
201            if (fs.exists(p)) {
202              LOG.warn(f + " was not deleted");
203              succ = false;
204            }
205          } else {
206            numDeleted++;
207          }
208        } catch (IOException ioe) {
209          LOG.warn(f + " was not deleted", ioe);
210          succ = false;
211        }
212      }
213      if (LOG.isDebugEnabled()) {
214        LOG.debug(numDeleted + " bulk loaded files out of " + map.size() + " were deleted");
215      }
216      if (succ) {
217        sysTable.deleteBulkLoadedFiles(map);
218      }
219
220      sysTable.deleteBackupInfo(backupInfo.getBackupId());
221      LOG.info("Delete backup " + backupInfo.getBackupId() + " completed.");
222      totalDeleted++;
223    } else {
224      LOG.warn("Delete backup failed: no information found for backupID=" + backupId);
225    }
226    return totalDeleted;
227  }

229  private void removeTableFromBackupImage(BackupInfo info, TableName tn,
230      BackupSystemTable sysTable) throws IOException {
231    List<TableName> tables = info.getTableNames();
232    LOG.debug("Remove " + tn + " from " + info.getBackupId() + " tables="
233        + info.getTableListAsString());
234    if (tables.contains(tn)) {
235      tables.remove(tn);
236
237      if (tables.isEmpty()) {
238        LOG.debug("Delete backup info " + info.getBackupId());
239
240        sysTable.deleteBackupInfo(info.getBackupId());
241        BackupUtils.cleanupBackupData(info, conn.getConfiguration());
242      } else {
243        info.setTables(tables);
244        sysTable.updateBackupInfo(info);
245        // Now, clean up directory for table
246        cleanupBackupDir(info, tn, conn.getConfiguration());
247      }
248    }
249  }

251  private List<BackupInfo> getAffectedBackupInfos(BackupInfo backupInfo, TableName tn,
252      BackupSystemTable table) throws IOException {
253    LOG.debug("GetAffectedBackupInfos for: " + backupInfo.getBackupId() + " table=" + tn);
254    long ts = backupInfo.getStartTs();
255    List<BackupInfo> list = new ArrayList<BackupInfo>();
256    List<BackupInfo> history = table.getBackupHistory(backupInfo.getBackupRootDir());
257    // Scan from most recent to backupInfo
258    // break when backupInfo reached
259    for (BackupInfo info : history) {
260      if (info.getStartTs() == ts) {
261        break;
262      }
263      List<TableName> tables = info.getTableNames();
264      if (tables.contains(tn)) {
265        BackupType bt = info.getType();
266        if (bt == BackupType.FULL) {
267          // Clear list if we encounter FULL backup
268          list.clear();
269        } else {
270          LOG.debug("GetAffectedBackupInfos for: " + backupInfo.getBackupId() + " table=" + tn
271              + " added " + info.getBackupId() + " tables=" + info.getTableListAsString());
272          list.add(info);
273        }
274      }
275    }
276    return list;
277  }

279  /**
280   * Clean up the data at target directory
281   * @throws IOException
282   */
283  private void cleanupBackupDir(BackupInfo backupInfo, TableName table, Configuration conf)
284      throws IOException {
285    try {
286      // clean up the data at target directory
287      String targetDir = backupInfo.getBackupRootDir();
288      if (targetDir == null) {
289        LOG.warn("No target directory specified for " + backupInfo.getBackupId());
290        return;
291      }
292
293      FileSystem outputFs = FileSystem.get(new Path(backupInfo.getBackupRootDir()).toUri(), conf);
294
295      Path targetDirPath =
296          new Path(BackupUtils.getTableBackupDir(backupInfo.getBackupRootDir(),
297            backupInfo.getBackupId(), table));
298      if (outputFs.delete(targetDirPath, true)) {
299        LOG.info("Cleaning up backup data at " + targetDirPath.toString() + " done.");
300      } else {
301        LOG.info("No data has been found in " + targetDirPath.toString() + ".");
302      }
303
304    } catch (IOException e1) {
305      LOG.error("Cleaning up backup data of " + backupInfo.getBackupId() + " for table " + table
306          + "at " + backupInfo.getBackupRootDir() + " failed due to " + e1.getMessage() + ".");
307      throw e1;
308    }
309  }

311  private boolean isLastBackupSession(BackupSystemTable table, TableName tn, long startTime)
312      throws IOException {
313    List<BackupInfo> history = table.getBackupHistory();
314    for (BackupInfo info : history) {
315      List<TableName> tables = info.getTableNames();
316      if (!tables.contains(tn)) {
317        continue;
318      }
319      if (info.getStartTs() <= startTime) {
320        return true;
321      } else {
322        return false;
323      }
324    }
325    return false;
326  }

328  @Override
329  public List<BackupInfo> getHistory(int n) throws IOException {
330    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
331      List<BackupInfo> history = table.getBackupHistory();
332      if (history.size() <= n) return history;
333      List<BackupInfo> list = new ArrayList<BackupInfo>();
334      for (int i = 0; i < n; i++) {
335        list.add(history.get(i));
336      }
337      return list;
338    }
339  }

341  @Override
342  public List<BackupInfo> getHistory(int n, BackupInfo.Filter... filters) throws IOException {
343    if (filters.length == 0) return getHistory(n);
344    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
345      List<BackupInfo> history = table.getBackupHistory();
346      List<BackupInfo> result = new ArrayList<BackupInfo>();
347      for (BackupInfo bi : history) {
348        if (result.size() == n) break;
349        boolean passed = true;
350        for (int i = 0; i < filters.length; i++) {
351          if (!filters[i].apply(bi)) {
352            passed = false;
353            break;
354          }
355        }
356        if (passed) {
357          result.add(bi);
358        }
359      }
360      return result;
361    }
362  }

399  @Override
400  public void addToBackupSet(String name, TableName[] tables) throws IOException {
401    String[] tableNames = new String[tables.length];
402    try (final BackupSystemTable table = new BackupSystemTable(conn);
403        final Admin admin = conn.getAdmin();) {
404      for (int i = 0; i < tables.length; i++) {
405        tableNames[i] = tables[i].getNameAsString();
406        if (!admin.tableExists(TableName.valueOf(tableNames[i]))) {
407          throw new IOException("Cannot add " + tableNames[i] + " because it doesn't exist");
408        }
409      }
410      table.addToBackupSet(name, tableNames);
411      LOG.info("Added tables [" + StringUtils.join(tableNames, " ") + "] to '" + name
412          + "' backup set");
413    }
414  }

416  @Override
417  public void removeFromBackupSet(String name, TableName[] tables) throws IOException {
418    LOG.info("Removing tables [" + StringUtils.join(tables, " ") + "] from '" + name + "'");
419    try (final BackupSystemTable table = new BackupSystemTable(conn)) {
420      table.removeFromBackupSet(name, toStringArray(tables));
421      LOG.info("Removing tables [" + StringUtils.join(tables, " ") + "] from '" + name
422          + "' completed.");
423    }
424  }

426  private String[] toStringArray(TableName[] list) {
427    String[] arr = new String[list.length];
428    for(int i=0; i < list.length; i++) {
429      arr[i] = list[i].toString();
430    }
431    return arr;

434  @Override
435  public void restore(RestoreRequest request) throws IOException {
436    if (request.isCheck()) {
437      HashMap<TableName, BackupManifest> backupManifestMap = new HashMap<>();
438      // check and load backup image manifest for the tables
439      Path rootPath = new Path(request.getBackupRootDir());
440      String backupId = request.getBackupId();
441      TableName[] sTableArray = request.getFromTables();
442      HBackupFileSystem.checkImageManifestExist(backupManifestMap, sTableArray,
443        conn.getConfiguration(), rootPath, backupId);
444
445      // Check and validate the backup image and its dependencies
446
447      if (BackupUtils.validate(backupManifestMap, conn.getConfiguration())) {
448        LOG.info(CHECK_OK);
449      } else {
450        LOG.error(CHECK_FAILED);
451      }
452      return;
453    }
454    // Execute restore request
455    new RestoreTablesClient(conn, request).execute();
456  }

458  @Override
459  public String backupTables(BackupRequest request) throws IOException {
460    BackupType type = request.getBackupType();
461    String targetRootDir = request.getTargetRootDir();
462    List<TableName> tableList = request.getTableList();
463
464    String backupId =BackupRestoreConstants.BACKUPID_PREFIX + EnvironmentEdgeManager.currentTime();
465    if (type == BackupType.INCREMENTAL) {
466      Set<TableName> incrTableSet = null;
467      try (BackupSystemTable table = new BackupSystemTable(conn)) {
468        incrTableSet = table.getIncrementalBackupTableSet(targetRootDir);
469      }
470
471      if (incrTableSet.isEmpty()) {
472        String msg = "Incremental backup table set contains no tables. "
473            + "You need to run full backup first " +
474            (tableList != null ? "on "+StringUtils.join(tableList, ","): "");
475
476        throw new IOException(msg);
477      }
478      if(tableList != null) {
479        tableList.removeAll(incrTableSet);
480        if (!tableList.isEmpty()) {
481          String extraTables = StringUtils.join(tableList, ",");
482          String msg = "Some tables (" + extraTables + ") haven't gone through full backup. "+
483            "Perform full backup on " + extraTables + " first, "
484            + "then retry the command";
485          throw new IOException(msg);
486        }
487      }
488      tableList = Lists.newArrayList(incrTableSet);
489    }
490    if (tableList != null && !tableList.isEmpty()) {
491      for (TableName table : tableList) {
492        String targetTableBackupDir =
493            HBackupFileSystem.getTableBackupDir(targetRootDir, backupId, table);
494        Path targetTableBackupDirPath = new Path(targetTableBackupDir);
495        FileSystem outputFs =
496            FileSystem.get(targetTableBackupDirPath.toUri(), conn.getConfiguration());
497        if (outputFs.exists(targetTableBackupDirPath)) {
498          throw new IOException("Target backup directory " + targetTableBackupDir
499              + " exists already.");
500        }
501      }
502      ArrayList<TableName> nonExistingTableList = null;
503      try (Admin admin = conn.getAdmin();) {
504        for (TableName tableName : tableList) {
505          if (!admin.tableExists(tableName)) {
506            if (nonExistingTableList == null) {
507              nonExistingTableList = new ArrayList<>();
508            }
509            nonExistingTableList.add(tableName);
510          }
511        }
512      }
513      if (nonExistingTableList != null) {
514        if (type == BackupType.INCREMENTAL) {
515          // Update incremental backup set
516          tableList = excludeNonExistingTables(tableList, nonExistingTableList);
517        } else {
518          // Throw exception only in full mode - we try to backup non-existing table
519          throw new IOException("Non-existing tables found in the table list: "
520              + nonExistingTableList);
521        }
522      }
523    }
524
525    // update table list
526    BackupRequest.Builder builder = new BackupRequest.Builder();
527    request = builder.withBackupType(request.getBackupType()).
528                      withTableList(tableList).
529                      withTargetRootDir(request.getTargetRootDir()).
530                      withBackupSetName(request.getBackupSetName()).
531                      withTotalTasks(request.getTotalTasks()).
532                      withBandwidthPerTasks((int)request.getBandwidth()).build();
533
534    TableBackupClient client =null;
535    try {
536      client = BackupClientFactory.create(conn, backupId, request);
537    } catch (IOException e) {
538      LOG.error("There is an active session already running");
539      throw e;
540    }
541
542    client.execute();
543
544    return backupId;
545  }

547  private List<TableName> excludeNonExistingTables(List<TableName> tableList,
548      List<TableName> nonExistingTableList) {
549
550    for (TableName table : nonExistingTableList) {
551      tableList.remove(table);
552    }
553    return tableList;
554  }
