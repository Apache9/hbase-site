047/**
048 * After a full backup was created, the incremental backup will only store the changes made after
049 * the last full or incremental backup. Creating the backup copies the logfiles in .logs and
050 * .oldlogs since the last backup timestamp.
051 */
052@InterfaceAudience.Private
053public class IncrementalBackupManager extends BackupManager {
054  public static final Logger LOG = LoggerFactory.getLogger(IncrementalBackupManager.class);
055
056  public IncrementalBackupManager(Connection conn, Configuration conf) throws IOException {
057    super(conn, conf);
058  }
059
060  /**
061   * Obtain the list of logs that need to be copied out for this incremental backup. The list is set
062   * in BackupInfo.
063   * @return The new HashMap of RS log time stamps after the log roll for this incremental backup.
064   * @throws IOException exception
065   */
066  public HashMap<String, Long> getIncrBackupLogFileMap()
067      throws IOException {
068    List<String> logList;
069    HashMap<String, Long> newTimestamps;
070    HashMap<String, Long> previousTimestampMins;
071
072    String savedStartCode = readBackupStartCode();
073
074    // key: tableName
075    // value: <RegionServer,PreviousTimeStamp>
076    HashMap<TableName, HashMap<String, Long>> previousTimestampMap = readLogTimestampMap();
077
078    previousTimestampMins = BackupUtils.getRSLogTimestampMins(previousTimestampMap);
079
080    if (LOG.isDebugEnabled()) {
081      LOG.debug("StartCode " + savedStartCode + "for backupID " + backupInfo.getBackupId());
082    }
083    // get all new log files from .logs and .oldlogs after last TS and before new timestamp
084    if (savedStartCode == null || previousTimestampMins == null
085        || previousTimestampMins.isEmpty()) {
086      throw new IOException(
087          "Cannot read any previous back up timestamps from backup system table. "
088              + "In order to create an incremental backup, at least one full backup is needed.");
089    }
090
091    LOG.info("Execute roll log procedure for incremental backup ...");
092    HashMap<String, String> props = new HashMap<String, String>();
093    props.put("backupRoot", backupInfo.getBackupRootDir());
094
095    try (Admin admin = conn.getAdmin()) {
096      admin.execProcedure(LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_SIGNATURE,
097        LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_NAME, props);
098    }
099    newTimestamps = readRegionServerLastLogRollResult();
100
101    logList = getLogFilesForNewBackup(previousTimestampMins, newTimestamps, conf, savedStartCode);
102    List<WALItem> logFromSystemTable =
103        getLogFilesFromBackupSystem(previousTimestampMins, newTimestamps, getBackupInfo()
104            .getBackupRootDir());
105    logList = excludeAlreadyBackedUpWALs(logList, logFromSystemTable);
106    backupInfo.setIncrBackupFileList(logList);
107
108    return newTimestamps;
109  }

111  /**

113   * @return list of WAL files

155  private List<String> excludeAlreadyBackedUpWALs(List<String> logList,
156      List<WALItem> logFromSystemTable) {
157
158    Set<String> walFileNameSet = convertToSet(logFromSystemTable);
159
160    List<String> list = new ArrayList<String>();
161    for (int i=0; i < logList.size(); i++) {
162      Path p = new Path(logList.get(i));
163      String name  = p.getName();
164      if (walFileNameSet.contains(name)) continue;
165      list.add(logList.get(i));
166    }
167    return list;
168  }
169
170  /**
171   * Create Set of WAL file names (not full path names)
172   * @param logFromSystemTable
173   * @return set of WAL file names
174   */
175  private Set<String> convertToSet(List<WALItem> logFromSystemTable) {
176
177    Set<String> set = new HashSet<String>();
178    for (int i=0; i < logFromSystemTable.size(); i++) {
179      WALItem item = logFromSystemTable.get(i);
180      set.add(item.walFile);
181    }
182    return set;
183  }

185  /**
186   * For each region server: get all log files newer than the last timestamps, but not newer than
187   * the newest timestamps.
188   * @param olderTimestamps timestamp map for each region server of the last backup.
189   * @param newestTimestamps timestamp map for each region server that the backup should lead to.
190   * @return list of log files which needs to be added to this backup
191   * @throws IOException
192   */
193  private List<WALItem> getLogFilesFromBackupSystem(HashMap<String, Long> olderTimestamps,
194      HashMap<String, Long> newestTimestamps, String backupRoot) throws IOException {
195    List<WALItem> logFiles = new ArrayList<WALItem>();
196    Iterator<WALItem> it = getWALFilesFromBackupSystem();
197    while (it.hasNext()) {
198      WALItem item = it.next();
199      String rootDir = item.getBackupRoot();
200      if (!rootDir.equals(backupRoot)) {
201        continue;
202      }
203      String walFileName = item.getWalFile();
204      String server = BackupUtils.parseHostNameFromLogFile(new Path(walFileName));
205      if (server == null) {
206        continue;
207      }
208      Long tss = getTimestamp(walFileName);
209      Long oldTss = olderTimestamps.get(server);
210      Long newTss = newestTimestamps.get(server);
211      if (oldTss == null) {
212        logFiles.add(item);
213        continue;
214      }
215      if (newTss == null) {
216        newTss = Long.MAX_VALUE;
217      }
218      if (tss > oldTss && tss < newTss) {
219        logFiles.add(item);
220      }
221    }
222    return logFiles;
223  }
224
225  private Long getTimestamp(String walFileName) {
226    int index = walFileName.lastIndexOf(BackupUtils.LOGNAME_SEPARATOR);
227    return Long.parseLong(walFileName.substring(index + 1));
228  }

230  /**
231   * For each region server: get all log files newer than the last timestamps but not newer than the
232   * newest timestamps.
233   * @param olderTimestamps the timestamp for each region server of the last backup.
234   * @param newestTimestamps the timestamp for each region server that the backup should lead to.
235   * @param conf the Hadoop and Hbase configuration
236   * @param savedStartCode the startcode (timestamp) of last successful backup.
237   * @return a list of log files to be backed up
238   * @throws IOException exception
239   */
240  private List<String> getLogFilesForNewBackup(HashMap<String, Long> olderTimestamps,
241      HashMap<String, Long> newestTimestamps, Configuration conf, String savedStartCode)
242      throws IOException {
243    LOG.debug("In getLogFilesForNewBackup()\n" + "olderTimestamps: " + olderTimestamps
244        + "\n newestTimestamps: " + newestTimestamps);
245    Path rootdir = FSUtils.getRootDir(conf);
246    Path logDir = new Path(rootdir, HConstants.HREGION_LOGDIR_NAME);
247    Path oldLogDir = new Path(rootdir, HConstants.HREGION_OLDLOGDIR_NAME);
248    FileSystem fs = rootdir.getFileSystem(conf);
249    NewestLogFilter pathFilter = new NewestLogFilter();
250
251    List<String> resultLogFiles = new ArrayList<String>();
252    List<String> newestLogs = new ArrayList<String>();
253
254    /*
255     * The old region servers and timestamps info we kept in backup system table may be out of sync
256     * if new region server is added or existing one lost. We'll deal with it here when processing
257     * the logs. If data in backup system table has more hosts, just ignore it. If the .logs
258     * directory includes more hosts, the additional hosts will not have old timestamps to compare
259     * with. We'll just use all the logs in that directory. We always write up-to-date region server
260     * and timestamp info to backup system table at the end of successful backup.
261     */
262
263    FileStatus[] rss;
264    Path p;
265    String host;
266    Long oldTimeStamp;
267    String currentLogFile;
268    long currentLogTS;
269
270    // Get the files in .logs.
271    rss = fs.listStatus(logDir);
272    for (FileStatus rs : rss) {
273      p = rs.getPath();
274      host = BackupUtils.parseHostNameFromLogFile(p);
275      if (host == null) {
276        continue;
277      }
278      FileStatus[] logs;
279      oldTimeStamp = olderTimestamps.get(host);
280      // It is possible that there is no old timestamp in backup system table for this host if
281      // this region server is newly added after our last backup.
282      if (oldTimeStamp == null) {
283        logs = fs.listStatus(p);
284      } else {
285        pathFilter.setLastBackupTS(oldTimeStamp);
286        logs = fs.listStatus(p, pathFilter);
287      }
288      for (FileStatus log : logs) {
289        LOG.debug("currentLogFile: " + log.getPath().toString());
290        if (AbstractFSWALProvider.isMetaFile(log.getPath())) {
291          if (LOG.isDebugEnabled()) {
292            LOG.debug("Skip hbase:meta log file: " + log.getPath().getName());
293          }
294          continue;
295        }
296        currentLogFile = log.getPath().toString();
297        resultLogFiles.add(currentLogFile);
298        currentLogTS = BackupUtils.getCreationTime(log.getPath());
299        // newestTimestamps is up-to-date with the current list of hosts
300        // so newestTimestamps.get(host) will not be null.
301        if (currentLogTS > newestTimestamps.get(host)) {
302          newestLogs.add(currentLogFile);
303        }
304      }
305    }
306
307    // Include the .oldlogs files too.
308    FileStatus[] oldlogs = fs.listStatus(oldLogDir);
309    for (FileStatus oldlog : oldlogs) {
310      p = oldlog.getPath();
311      currentLogFile = p.toString();
312      if (AbstractFSWALProvider.isMetaFile(p)) {
313        if (LOG.isDebugEnabled()) {
314          LOG.debug("Skip .meta log file: " + currentLogFile);
315        }
316        continue;
317      }
318      host = BackupUtils.parseHostFromOldLog(p);
319      if (host == null) {
320        continue;
321      }
322      currentLogTS = BackupUtils.getCreationTime(p);
323      oldTimeStamp = olderTimestamps.get(host);
324      /*
325       * It is possible that there is no old timestamp in backup system table for this host. At the
326       * time of our last backup operation, this rs did not exist. The reason can be one of the two:
327       * 1. The rs already left/crashed. Its logs were moved to .oldlogs. 2. The rs was added after
328       * our last backup.
329       */
330      if (oldTimeStamp == null) {
331        if (currentLogTS < Long.parseLong(savedStartCode)) {
332          // This log file is really old, its region server was before our last backup.
333          continue;
334        } else {
335          resultLogFiles.add(currentLogFile);
336        }
337      } else if (currentLogTS > oldTimeStamp) {
338        resultLogFiles.add(currentLogFile);
339      }
340
341      // It is possible that a host in .oldlogs is an obsolete region server
342      // so newestTimestamps.get(host) here can be null.
343      // Even if these logs belong to a obsolete region server, we still need
344      // to include they to avoid loss of edits for backup.
345      Long newTimestamp = newestTimestamps.get(host);
346      if (newTimestamp != null && currentLogTS > newTimestamp) {
347        newestLogs.add(currentLogFile);
348      }
349    }
350    // remove newest log per host because they are still in use
351    resultLogFiles.removeAll(newestLogs);
352    return resultLogFiles;
353  }
354
355  static class NewestLogFilter implements PathFilter {
356    private Long lastBackupTS = 0L;
357
358    public NewestLogFilter() {
359    }
360
361    protected void setLastBackupTS(Long ts) {
362      this.lastBackupTS = ts;
363    }
364
365    @Override
366    public boolean accept(Path path) {
367      // skip meta table log -- ts.meta file
368      if (AbstractFSWALProvider.isMetaFile(path)) {
369        if (LOG.isDebugEnabled()) {
370          LOG.debug("Skip .meta log file: " + path.getName());
371        }
372        return false;
373      }
374      long timestamp;
375      try {
376        timestamp = BackupUtils.getCreationTime(path);
377        return timestamp > lastBackupTS;
378      } catch (Exception e) {
379        LOG.warn("Cannot read timestamp of log file " + path);
380        return false;
381      }
382    }
383  }
