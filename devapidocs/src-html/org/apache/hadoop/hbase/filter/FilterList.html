039/**
040 * Implementation of {@link Filter} that represents an ordered List of Filters
041 * which will be evaluated with a specified boolean operator {@link Operator#MUST_PASS_ALL}
042 * (<code>AND</code>) or {@link Operator#MUST_PASS_ONE} (<code>OR</code>).
043 * Since you can use Filter Lists as children of Filter Lists, you can create a
044 * hierarchy of filters to be evaluated.
045 *
046 * <br>
047 * {@link Operator#MUST_PASS_ALL} evaluates lazily: evaluation stops as soon as one filter does
048 * not include the KeyValue.
049 *
050 * <br>
051 * {@link Operator#MUST_PASS_ONE} evaluates non-lazily: all filters are always evaluated.
052 *
053 * <br>
054 * Defaults to {@link Operator#MUST_PASS_ALL}.
055 */
056@InterfaceAudience.Public
057final public class FilterList extends FilterBase {
058  /** set operator */
059  @InterfaceAudience.Public
060  public static enum Operator {
061    /** !AND */
062    MUST_PASS_ALL,
063    /** !OR */
064    MUST_PASS_ONE
065  }

070  private Set<Filter> seekHintFilter = new HashSet<>();

076   */
077  private List<ReturnCode> prevFilterRCList = null;

139  private void initPrevListForMustPassOne(int size) {
140    if (operator == Operator.MUST_PASS_ONE) {

143      }
144      if (this.prevCellList == null) {

147    }

174  private static boolean checkAndGetReversed(List<Filter> rowFilters, boolean defaultValue) {
175    if (rowFilters.isEmpty()) {
176      return defaultValue;
177    }
178    Boolean retValue = rowFilters.get(0).isReversed();
179    boolean allEqual = rowFilters.stream().map(Filter::isReversed).allMatch(retValue::equals);
180    if (!allEqual) {
181      throw new IllegalArgumentException("Filters in the list must have the same reversed flag");
182    }
183    return retValue;
184  }
185
186  public void addFilter(List<Filter> filters) {
187    if (checkAndGetReversed(filters, isReversed()) != isReversed()) {
188      throw new IllegalArgumentException("Filters in the list must have the same reversed flag");
189    }
190    this.filters.addAll(filters);
191    if (operator == Operator.MUST_PASS_ONE) {
192      this.prevFilterRCList.addAll(Collections.nCopies(filters.size(), null));
193      this.prevCellList.addAll(Collections.nCopies(filters.size(), null));
194    }
195  }

205  @Override
206  public void reset() throws IOException {
207    for (int i = 0, n = filters.size(); i < n; i++) {
208      filters.get(i).reset();
209      if (operator == Operator.MUST_PASS_ONE) {
210        prevFilterRCList.set(i, null);
211        prevCellList.set(i, null);
212      }
213    }
214    seekHintFilter.clear();
215  }

217  @Override
218  public boolean filterRowKey(byte[] rowKey, int offset, int length) throws IOException {
219    if (isEmpty()) {
220      return super.filterRowKey(rowKey, offset, length);
221    }
222    boolean flag = this.operator == Operator.MUST_PASS_ONE;
223    for (int i = 0, n = filters.size(); i < n; i++) {
224      Filter filter = filters.get(i);
225      if (this.operator == Operator.MUST_PASS_ALL) {
226        if (filter.filterAllRemaining() || filter.filterRowKey(rowKey, offset, length)) {
227          flag = true;
228        }
229      } else if (this.operator == Operator.MUST_PASS_ONE) {
230        if (!filter.filterAllRemaining() && !filter.filterRowKey(rowKey, offset, length)) {
231          flag = false;
232        }
233      }
234    }
235    return flag;
236  }

238  @Override
239  public boolean filterRowKey(Cell firstRowCell) throws IOException {
240    if (isEmpty()) {
241      return super.filterRowKey(firstRowCell);
242    }
243    boolean flag = this.operator == Operator.MUST_PASS_ONE;
244    for (int i = 0, n = filters.size(); i < n; i++) {
245      Filter filter = filters.get(i);
246      if (this.operator == Operator.MUST_PASS_ALL) {
247        if (filter.filterAllRemaining() || filter.filterRowKey(firstRowCell)) {
248          flag = true;
249        }
250      } else if (this.operator == Operator.MUST_PASS_ONE) {
251        if (!filter.filterAllRemaining() && !filter.filterRowKey(firstRowCell)) {
252          flag = false;
253        }
254      }
255    }
256    return flag;
257  }

259  @Override
260  public boolean filterAllRemaining() throws IOException {
261    if (isEmpty()) {
262      return super.filterAllRemaining();
263    }
264    for (int i = 0, n = filters.size(); i < n; i++) {
265      if (filters.get(i).filterAllRemaining()) {
266        if (operator == Operator.MUST_PASS_ALL) {
267          return true;
268        }
269      } else {
270        if (operator == Operator.MUST_PASS_ONE) {
271          return false;
272        }
273      }
274    }
275    return operator == Operator.MUST_PASS_ONE;
276  }

278  @Override
279  public Cell transformCell(Cell c) throws IOException {
280    if (isEmpty()) {
281      return super.transformCell(c);
282    }
283    if (!CellUtil.equals(c, referenceCell)) {
284      throw new IllegalStateException(
285          "Reference Cell: " + this.referenceCell + " does not match: " + c);
286    }
287    // Copy transformedCell into a new cell and reset transformedCell & referenceCell to null for
288    // Java GC optimization
289    Cell cell = KeyValueUtil.copyToNewKeyValue(this.transformedCell);
290    this.transformedCell = null;
291    this.referenceCell = null;
292    return cell;
293  }

295  /**
296   * For MUST_PASS_ONE, we cannot make sure that when filter-A in filter list return NEXT_COL then
297   * the next cell passing to filterList will be the first cell in next column, because if filter-B
298   * in filter list return SKIP, then the filter list will return SKIP. In this case, we should pass
299   * the cell following the previous cell, and it's possible that the next cell has the same column
300   * as the previous cell even if filter-A has NEXT_COL returned for the previous cell. So we should
301   * save the previous cell and the return code list when checking previous cell for every filter in
302   * filter list, and verify if currentCell fit the previous return code, if fit then pass the currentCell
303   * to the corresponding filter. (HBASE-17678)
304   */
305  private boolean shouldPassCurrentCellToFilter(Cell prevCell, Cell currentCell, int filterIdx)
306      throws IOException {
307    ReturnCode prevCode = this.prevFilterRCList.get(filterIdx);
308    if (prevCell == null || prevCode == null) {
309      return true;
310    }
311    switch (prevCode) {
312    case INCLUDE:
313    case SKIP:
314      return true;
315    case SEEK_NEXT_USING_HINT:
316      Cell nextHintCell = getNextCellHint(prevCell);
317      return nextHintCell == null
318          || CellComparator.COMPARATOR.compare(currentCell, nextHintCell) >= 0;
319    case NEXT_COL:
320    case INCLUDE_AND_NEXT_COL:
321      return !CellUtil.matchingRowColumn(prevCell, currentCell);
322    case NEXT_ROW:
323    case INCLUDE_AND_SEEK_NEXT_ROW:
324      return !CellUtil.matchingRows(prevCell, currentCell);
325    default:
326      throw new IllegalStateException("Received code is not valid.");
327    }
328  }

330  /**
331   * FilterList with MUST_PASS_ALL choose the maximal forward step among sub-filters in filter list.
332   * Let's call it: The Maximal Step Rule. So if filter-A in filter list return INCLUDE and filter-B
333   * in filter list return INCLUDE_AND_NEXT_COL, then the filter list should return
334   * INCLUDE_AND_NEXT_COL. For SEEK_NEXT_USING_HINT, it's more special, and in method
335   * filterKeyValueWithMustPassAll(), if any sub-filter return SEEK_NEXT_USING_HINT, then our filter
336   * list will return SEEK_NEXT_USING_HINT. so we don't care about the SEEK_NEXT_USING_HINT here. <br/>
337   * <br/>
338   * The jump step will be:
339   *
340   * <pre>
341   * INCLUDE &lt; SKIP &lt; INCLUDE_AND_NEXT_COL &lt; NEXT_COL &lt; INCLUDE_AND_SEEK_NEXT_ROW &lt; NEXT_ROW &lt; SEEK_NEXT_USING_HINT
342   * </pre>
343   *
344   * Here, we have the following map to describe The Maximal Step Rule. if current return code (for
345   * previous sub-filters in filter list) is <strong>ReturnCode</strong>, and current filter returns
346   * <strong>localRC</strong>, then we should return map[ReturnCode][localRC] for the merged result,
347   * according to The Maximal Step Rule. <br/>
348   *
349   * <pre>
350   * LocalCode\ReturnCode       INCLUDE                    INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
351   * INCLUDE                    INCLUDE                    INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
352   * INCLUDE_AND_NEXT_COL       INCLUDE_AND_NEXT_COL       INCLUDE_AND_NEXT_COL      INCLUDE_AND_SEEK_NEXT_ROW  NEXT_COL              NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
353   * INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE_AND_SEEK_NEXT_ROW INCLUDE_AND_SEEK_NEXT_ROW  NEXT_ROW              NEXT_ROW              NEXT_ROW              SEEK_NEXT_USING_HINT
354   * SKIP                       SKIP                       NEXT_COL                  NEXT_ROW                   SKIP                  NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
355   * NEXT_COL                   NEXT_COL                   NEXT_COL                  NEXT_ROW                   NEXT_COL              NEXT_COL              NEXT_ROW              SEEK_NEXT_USING_HINT
356   * NEXT_ROW                   NEXT_ROW                   NEXT_ROW                  NEXT_ROW                   NEXT_ROW              NEXT_ROW              NEXT_ROW              SEEK_NEXT_USING_HINT
357   * SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT      SEEK_NEXT_USING_HINT       SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT  SEEK_NEXT_USING_HINT
358   * </pre>
359   * @param rc Return code which is calculated by previous sub-filter(s) in filter list.
360   * @param localRC Return code of the current sub-filter in filter list.
361   * @return Return code which is merged by the return code of previous sub-filter(s) and the return
362   *         code of current sub-filter.
363   */
364  private ReturnCode mergeReturnCodeForAndOperator(ReturnCode rc, ReturnCode localRC) {
365    if (rc == ReturnCode.SEEK_NEXT_USING_HINT || localRC == ReturnCode.SEEK_NEXT_USING_HINT) {
366      return ReturnCode.SEEK_NEXT_USING_HINT;
367    }
368    switch (localRC) {
369    case INCLUDE:
370      return rc;
371    case INCLUDE_AND_NEXT_COL:
372      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL)) {
373        return ReturnCode.INCLUDE_AND_NEXT_COL;
374      }
375      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
376        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
377      }
378      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL)) {
379        return ReturnCode.NEXT_COL;
380      }
381      if (isInReturnCodes(rc, ReturnCode.NEXT_ROW)) {
382        return ReturnCode.NEXT_ROW;
383      }
384      break;
385    case INCLUDE_AND_SEEK_NEXT_ROW:
386      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
387        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
388        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
389      }
390      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL, ReturnCode.NEXT_ROW)) {
391        return ReturnCode.NEXT_ROW;
392      }
393      break;
394    case SKIP:
395      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.SKIP)) {
396        return ReturnCode.SKIP;
397      }
398      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.NEXT_COL)) {
399        return ReturnCode.NEXT_COL;
400      }
401      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW, ReturnCode.NEXT_ROW)) {
402        return ReturnCode.NEXT_ROW;
403      }
404      break;
405    case NEXT_COL:
406      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.SKIP,
407        ReturnCode.NEXT_COL)) {
408        return ReturnCode.NEXT_COL;
409      }
410      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW, ReturnCode.NEXT_ROW)) {
411        return ReturnCode.NEXT_ROW;
412      }
413      break;
414    case NEXT_ROW:
415      return ReturnCode.NEXT_ROW;
416    }
417    throw new IllegalStateException("Received code is not valid. rc: " + rc + ", localRC: "
418        + localRC);
419  }

421  private ReturnCode filterKeyValueWithMustPassAll(Cell c) throws IOException {
422    ReturnCode rc = ReturnCode.INCLUDE;
423    Cell transformed = c;
424    this.seekHintFilter.clear();
425    for (int i = 0, n = filters.size(); i < n; i++) {
426      Filter filter = filters.get(i);
427      if (filter.filterAllRemaining()) {
428        return ReturnCode.NEXT_ROW;
429      }
430      ReturnCode localRC = filter.filterKeyValue(c);
431      rc = mergeReturnCodeForAndOperator(rc, localRC);
432
433      // For INCLUDE* case, we need to update the transformed cell.
434      if (isInReturnCodes(localRC, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
435        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
436        transformed = filter.transformCell(transformed);
437      }
438      if (localRC == ReturnCode.SEEK_NEXT_USING_HINT) {
439        seekHintFilter.add(filter);
440      }
441    }
442    this.transformedCell = transformed;
443    if (!seekHintFilter.isEmpty()) {
444      return ReturnCode.SEEK_NEXT_USING_HINT;
445    }
446    return rc;
447  }

453  private void updatePrevCellList(int index, Cell currentCell, ReturnCode currentRC) {
454    if (currentCell == null || currentRC == ReturnCode.INCLUDE || currentRC == ReturnCode.SKIP) {
455      // If previous return code is INCLUDE or SKIP, we should always pass the next cell to the
456      // corresponding sub-filter(need not test shouldPassCurrentCellToFilter() method), So we
457      // need not save current cell to prevCellList for saving heap memory.
458      prevCellList.set(index, null);
459    } else {
460      prevCellList.set(index, KeyValueUtil.toNewKeyCell(currentCell));
461    }
462  }

465    return Arrays.stream(returnCodes).anyMatch(testRC::equals);

468  /**
469   * FilterList with MUST_PASS_ONE choose the minimal forward step among sub-filter in filter list.
470   * Let's call it: The Minimal Step Rule. So if filter-A in filter list return INCLUDE and filter-B
471   * in filter list return INCLUDE_AND_NEXT_COL, then the filter list should return INCLUDE. For
472   * SEEK_NEXT_USING_HINT, it's more special, because we do not know how far it will forward, so we
473   * use SKIP by default.<br/>
474   * <br/>
475   * The jump step will be:
476   *
477   * <pre>
478   * INCLUDE &lt; SKIP &lt; INCLUDE_AND_NEXT_COL &lt; NEXT_COL &lt; INCLUDE_AND_SEEK_NEXT_ROW &lt; NEXT_ROW &lt; SEEK_NEXT_USING_HINT
479   * </pre>
480   *
481   * Here, we have the following map to describe The Minimal Step Rule. if current return code (for
482   * previous sub-filters in filter list) is <strong>ReturnCode</strong>, and current filter returns
483   * <strong>localRC</strong>, then we should return map[ReturnCode][localRC] for the merged result,
484   * according to The Minimal Step Rule.<br/>
485   *
486   * <pre>
487   * LocalCode\ReturnCode       INCLUDE INCLUDE_AND_NEXT_COL     INCLUDE_AND_SEEK_NEXT_ROW  SKIP      NEXT_COL              NEXT_ROW                  SEEK_NEXT_USING_HINT
488   * INCLUDE                    INCLUDE INCLUDE                  INCLUDE                    INCLUDE   INCLUDE               INCLUDE                   INCLUDE
489   * INCLUDE_AND_NEXT_COL       INCLUDE INCLUDE_AND_NEXT_COL     INCLUDE_AND_NEXT_COL       INCLUDE   INCLUDE_AND_NEXT_COL  INCLUDE_AND_NEXT_COL      INCLUDE
490   * INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE INCLUDE_AND_NEXT_COL     INCLUDE_AND_SEEK_NEXT_ROW  INCLUDE   INCLUDE_AND_NEXT_COL  INCLUDE_AND_SEEK_NEXT_ROW INCLUDE
491   * SKIP                       INCLUDE INCLUDE                  INCLUDE                    SKIP      SKIP                  SKIP                      SKIP
492   * NEXT_COL                   INCLUDE INCLUDE_AND_NEXT_COL     INCLUDE_AND_NEXT_COL       SKIP      NEXT_COL              NEXT_COL                  SKIP
493   * NEXT_ROW                   INCLUDE INCLUDE_AND_NEXT_COL     INCLUDE_AND_SEEK_NEXT_ROW  SKIP      NEXT_COL              NEXT_ROW                  SKIP
494   * SEEK_NEXT_USING_HINT       INCLUDE INCLUDE                  INCLUDE                    SKIP      SKIP                  SKIP                      SEEK_NEXT_USING_HINT
495   * </pre>
496   * @param rc Return code which is calculated by previous sub-filter(s) in filter list.
497   * @param localRC Return code of the current sub-filter in filter list.
498   * @return Return code which is merged by the return code of previous sub-filter(s) and the return
499   *         code of current sub-filter.
500   */
501  private ReturnCode mergeReturnCodeForOrOperator(ReturnCode rc, ReturnCode localRC) {
502    if (rc == null) return localRC;
503    switch (localRC) {
504    case INCLUDE:
505      return ReturnCode.INCLUDE;
506    case INCLUDE_AND_NEXT_COL:
507      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.SKIP,
508        ReturnCode.SEEK_NEXT_USING_HINT)) {
509        return ReturnCode.INCLUDE;
510      }
511      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW,
512        ReturnCode.NEXT_COL, ReturnCode.NEXT_ROW)) {
513        return ReturnCode.INCLUDE_AND_NEXT_COL;
514      }
515      break;
516    case INCLUDE_AND_SEEK_NEXT_ROW:
517      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.SKIP,
518        ReturnCode.SEEK_NEXT_USING_HINT)) {
519        return ReturnCode.INCLUDE;
520      }
521      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL, ReturnCode.NEXT_COL)) {
522        return ReturnCode.INCLUDE_AND_NEXT_COL;
523      }
524      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW, ReturnCode.NEXT_ROW)) {
525        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
526      }
527      break;
528    case SKIP:
529      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
530        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
531        return ReturnCode.INCLUDE;
532      }
533      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL, ReturnCode.NEXT_ROW,
534        ReturnCode.SEEK_NEXT_USING_HINT)) {
535        return ReturnCode.SKIP;
536      }
537      break;
538    case NEXT_COL:
539      if (isInReturnCodes(rc, ReturnCode.INCLUDE)) {
540        return ReturnCode.INCLUDE;
541      }
542      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL,
543        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
544        return ReturnCode.INCLUDE_AND_NEXT_COL;
545      }
546      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.SEEK_NEXT_USING_HINT)) {
547        return ReturnCode.SKIP;
548      }
549      break;
550    case NEXT_ROW:
551      if (isInReturnCodes(rc, ReturnCode.INCLUDE)) {
552        return ReturnCode.INCLUDE;
553      }
554      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_NEXT_COL)) {
555        return ReturnCode.INCLUDE_AND_NEXT_COL;
556      }
557      if (isInReturnCodes(rc, ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
558        return ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW;
559      }
560      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.SEEK_NEXT_USING_HINT)) {
561        return ReturnCode.SKIP;
562      }
563      if (isInReturnCodes(rc, ReturnCode.NEXT_COL)) {
564        return ReturnCode.NEXT_COL;
565      }
566      if (isInReturnCodes(rc, ReturnCode.NEXT_ROW)) {
567        return ReturnCode.NEXT_ROW;
568      }
569    case SEEK_NEXT_USING_HINT:
570      if (isInReturnCodes(rc, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
571        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
572        return ReturnCode.INCLUDE;
573      }
574      if (isInReturnCodes(rc, ReturnCode.SKIP, ReturnCode.NEXT_COL, ReturnCode.NEXT_ROW)) {
575        return ReturnCode.SKIP;
576      }
577      if (isInReturnCodes(rc, ReturnCode.SEEK_NEXT_USING_HINT)) {
578        return ReturnCode.SEEK_NEXT_USING_HINT;
579      }
580    }
581    throw new IllegalStateException(
582        "Received code is not valid. rc: " + rc + ", localRC: " + localRC);
583  }
584
585  private ReturnCode filterKeyValueWithMustPassOne(Cell c) throws IOException {
586    ReturnCode rc = null;
587    boolean everyFilterReturnHint = true;
588    Cell transformed = c;
589    for (int i = 0, n = filters.size(); i < n; i++) {
590      Filter filter = filters.get(i);
591
592      Cell prevCell = this.prevCellList.get(i);
593      if (filter.filterAllRemaining() || !shouldPassCurrentCellToFilter(prevCell, c, i)) {
594        everyFilterReturnHint = false;
595        continue;
596      }
597
598      ReturnCode localRC = filter.filterKeyValue(c);
599
600      // Update previous return code and previous cell for filter[i].
601      updatePrevFilterRCList(i, localRC);
602      updatePrevCellList(i, c, localRC);
603
604      if (localRC != ReturnCode.SEEK_NEXT_USING_HINT) {
605        everyFilterReturnHint = false;
606      }
607
608      rc = mergeReturnCodeForOrOperator(rc, localRC);
609
610      // For INCLUDE* case, we need to update the transformed cell.
611      if (isInReturnCodes(localRC, ReturnCode.INCLUDE, ReturnCode.INCLUDE_AND_NEXT_COL,
612        ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW)) {
613        transformed = filter.transformCell(transformed);
614      }
615    }
616
617    this.transformedCell = transformed;
618    if (everyFilterReturnHint) {
619      return ReturnCode.SEEK_NEXT_USING_HINT;
620    } else if (rc == null) {
621      // Each sub-filter in filter list got true for filterAllRemaining().
622      return ReturnCode.SKIP;
623    } else {
624      return rc;
625    }
626  }

628  @Override
629  public ReturnCode filterKeyValue(Cell c) throws IOException {
630    if (isEmpty()) {
631      return ReturnCode.INCLUDE;
632    }
633    this.referenceCell = c;
634
635    if (operator == Operator.MUST_PASS_ALL) {
636      return filterKeyValueWithMustPassAll(c);
637    } else {
638      return filterKeyValueWithMustPassOne(c);
639    }
640  }

667    if (isEmpty()) {
668      return super.filterRow();
669    }

680      }
681    }
682    return operator == Operator.MUST_PASS_ONE;

697  /**
698   * @param pbBytes A pb serialized {@link FilterList} instance
699   * @return An instance of {@link FilterList} made from <code>bytes</code>
700   * @throws DeserializationException
701   * @see #toByteArray
702   */
703  public static FilterList parseFrom(final byte [] pbBytes)
704      throws DeserializationException {
705    FilterProtos.FilterList proto;
706    try {
707      proto = FilterProtos.FilterList.parseFrom(pbBytes);
708    } catch (InvalidProtocolBufferException e) {
709      throw new DeserializationException(e);
710    }
711
712    List<Filter> rowFilters = new ArrayList<>(proto.getFiltersCount());
713    try {
714      List<FilterProtos.Filter> filtersList = proto.getFiltersList();
715      for (int i = 0, n = filtersList.size(); i < n; i++) {
716        rowFilters.add(ProtobufUtil.toFilter(filtersList.get(i)));
717      }
718    } catch (IOException ioe) {
719      throw new DeserializationException(ioe);
720    }
721    return new FilterList(Operator.valueOf(proto.getOperator().name()), rowFilters);
722  }

725   * @param other
726   * @return true if and only if the fields of the filter that are serialized
727   * are equal to the corresponding fields in other.  Used for testing.
728   */
729  boolean areSerializedFieldsEqual(Filter other) {
730    if (other == this) return true;
731    if (!(other instanceof FilterList)) return false;
732
733    FilterList o = (FilterList)other;
734    return this.getOperator().equals(o.getOperator()) &&
735      ((this.getFilters() == o.getFilters())
736      || this.getFilters().equals(o.getFilters()));
737  }

739  @Override
740  public Cell getNextCellHint(Cell currentCell) throws IOException {
741    if (isEmpty()) {
742      return super.getNextCellHint(currentCell);
743    }
744    Cell keyHint = null;
745    if (operator == Operator.MUST_PASS_ALL) {
746      for (Filter filter : seekHintFilter) {
747        if (filter.filterAllRemaining()) continue;
748        Cell curKeyHint = filter.getNextCellHint(currentCell);
749        if (keyHint == null) {
750          keyHint = curKeyHint;
751          continue;
752        }
753        if (CellComparator.COMPARATOR.compare(keyHint, curKeyHint) < 0) {
754          keyHint = curKeyHint;
755        }
756      }
757      return keyHint;
758    }
759
760    // If any condition can pass, we need to keep the min hint
761    for (int i = 0, n = filters.size(); i < n; i++) {
762      if (filters.get(i).filterAllRemaining()) {
763        continue;
764      }
765      Cell curKeyHint = filters.get(i).getNextCellHint(currentCell);
766      if (curKeyHint == null) {
767        // If we ever don't have a hint and this is must-pass-one, then no hint
768        return null;
769      }
770      // If this is the first hint we find, set it
771      if (keyHint == null) {
772        keyHint = curKeyHint;
773        continue;
774      }
775      if (CellComparator.COMPARATOR.compare(keyHint, curKeyHint) > 0) {
776        keyHint = curKeyHint;
777      }
778    }
779    return keyHint;
780  }
