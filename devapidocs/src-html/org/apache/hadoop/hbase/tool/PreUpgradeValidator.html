019package org.apache.hadoop.hbase.tool;
020
021import java.io.IOException;
022import java.util.List;
023
024import org.apache.hadoop.hbase.HBaseInterfaceAudience;
025import org.apache.hadoop.hbase.client.Admin;
026import org.apache.hadoop.hbase.client.ColumnFamilyDescriptor;
027import org.apache.hadoop.hbase.client.Connection;
028import org.apache.hadoop.hbase.client.ConnectionFactory;
029import org.apache.hadoop.hbase.client.TableDescriptor;
030import org.apache.hadoop.hbase.io.encoding.DataBlockEncoding;
031import org.apache.hadoop.hbase.util.AbstractHBaseTool;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.yetus.audience.InterfaceAudience;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036
037import org.apache.hbase.thirdparty.org.apache.commons.cli.CommandLine;
038
039/**
040 * Tool for validating that cluster can be upgraded from HBase 1.x to 2.0
041 * <p>
042 * Available validations:
043 * <ul>
044 * <li>all: Run all pre-upgrade validations</li>
045 * <li>validateDBE: Check Data Block Encoding for column families</li>
046 * </ul>
047 * </p>
048 */
049@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
050public class PreUpgradeValidator extends AbstractHBaseTool {
051
052  public static final String NAME = "pre-upgrade";
053  private static final Logger LOG = LoggerFactory.getLogger(PreUpgradeValidator.class);
054  private static final byte[] DATA_BLOCK_ENCODING = Bytes.toBytes("DATA_BLOCK_ENCODING");
055  private boolean validateAll;
056  private boolean validateDBE;
057
058  /**
059   * Check DataBlockEncodings for column families.
060   *
061   * @return DataBlockEncoding compatible with HBase 2
062   * @throws IOException if a remote or network exception occurs
063   */
064  private boolean validateDBE() throws IOException {
065    int incompatibilities = 0;
066
067    LOG.info("Validating Data Block Encodings");
068
069    try (Connection connection = ConnectionFactory.createConnection(getConf());
070        Admin admin = connection.getAdmin()) {
071      List<TableDescriptor> tableDescriptors = admin.listTableDescriptors();
072      String encoding = "";
073
074      for (TableDescriptor td : tableDescriptors) {
075        ColumnFamilyDescriptor[] columnFamilies = td.getColumnFamilies();
076        for (ColumnFamilyDescriptor cfd : columnFamilies) {
077          try {
078            encoding = Bytes.toString(cfd.getValue(DATA_BLOCK_ENCODING));
079            // IllegalArgumentException will be thrown if encoding is incompatible with 2.0
080            DataBlockEncoding.valueOf(encoding);
081          } catch (IllegalArgumentException e) {
082            incompatibilities++;
083            LOG.warn("Incompatible DataBlockEncoding for table: {}, cf: {}, encoding: {}",
084                td.getTableName().getNameAsString(), cfd.getNameAsString(), encoding);
085          }
086        }
087      }
088    }
089
090    if (incompatibilities > 0) {
091      LOG.warn("There are {} column families with incompatible Data Block Encodings. Do not "
092          + "upgrade until these encodings are converted to a supported one.", incompatibilities);
093      LOG.warn("Check http://hbase.apache.org/book.html#upgrade2.0.prefix-tree.removed "
094          + "for instructions.");
095      return false;
096    } else {
097      LOG.info("The used Data Block Encodings are compatible with HBase 2.0.");
098      return true;
099    }
100  }
101
102  @Override
103  protected void addOptions() {
104    addOptNoArg("all", "Run all pre-upgrade validations");
105    addOptNoArg("validateDBE", "Validate DataBlockEncoding are compatible on the cluster");
106  }
107
108  @Override
109  protected void processOptions(CommandLine cmd) {
110    validateAll = cmd.hasOption("all");
111    validateDBE = cmd.hasOption("validateDBE");
112  }
113
114  @Override
115  protected int doWork() throws Exception {
116    boolean validationFailed = false;
117    if (validateDBE || validateAll) {
118      if (validateDBE()) {
119        validationFailed = true;
120      }
121    }
122
123    return validationFailed ? 1 : 0;
124  }
125
126  public static void main(String[] args) {
127    new PreUpgradeValidator().doStaticMain(args);
128  }
129}
