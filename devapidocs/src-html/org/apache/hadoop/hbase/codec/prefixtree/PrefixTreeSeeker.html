040/**
041 * These methods have the same definition as any implementation of the EncodedSeeker.
042 *
043 * In the future, the EncodedSeeker could be modified to work with the Cell interface directly.  It
044 * currently returns a new KeyValue object each time getKeyValue is called.  This is not horrible,
045 * but in order to create a new KeyValue object, we must first allocate a new byte[] and copy in
046 * the data from the PrefixTreeCell.  It is somewhat heavyweight right now.
047 */
048@InterfaceAudience.Private
049public class PrefixTreeSeeker implements EncodedSeeker {
050
051  protected boolean includeMvccVersion;
052  protected PrefixTreeArraySearcher ptSearcher;

054  public PrefixTreeSeeker(boolean includeMvccVersion) {
055    this.includeMvccVersion = includeMvccVersion;
056  }

060    ptSearcher = DecoderFactory.checkOut(fullBlockBuffer, includeMvccVersion);

087  /**
088   * currently must do deep copy into new array
089   */
090  @Override
091  public Cell getCell() {
092    // The PrefixTreecell is of type BytebufferedCell and the value part of the cell
093    // determines whether we are offheap cell or onheap cell.  All other parts of the cell-
094    // row, fam and col are all represented as onheap byte[]
095    ByteBufferCell cell = (ByteBufferCell)ptSearcher.current();
096    if (cell == null) {
097      return null;
098    }
099    // Use the ByteBuffered cell to see if the Cell is onheap or offheap
100    if (cell.getValueByteBuffer().hasArray()) {
101      return new OnheapPrefixTreeCell(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
102          cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
103          cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength(),
104          cell.getValueArray(), cell.getValueOffset(), cell.getValueLength(), cell.getTagsArray(),
105          cell.getTagsOffset(), cell.getTagsLength(), cell.getTimestamp(), cell.getTypeByte(),
106          cell.getSequenceId());
107    } else {
108      return new OffheapPrefixTreeCell(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
109          cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
110          cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength(),
111          cell.getValueByteBuffer(), cell.getValuePosition(), cell.getValueLength(),
112          cell.getTagsArray(), cell.getTagsOffset(), cell.getTagsLength(), cell.getTimestamp(),
113          cell.getTypeByte(), cell.getSequenceId());
114    }
115  }

147  private static final boolean USE_POSITION_BEFORE = false;
148
149  /*
150   * Support both of these options since the underlying PrefixTree supports
151   * both. Possibly expand the EncodedSeeker to utilize them both.
152   */
153
154  protected int seekToOrBeforeUsingPositionAtOrBefore(Cell kv, boolean seekBefore) {
155    // this does a deep copy of the key byte[] because the CellSearcher
156    // interface wants a Cell
157    CellScannerPosition position = ptSearcher.seekForwardToOrBefore(kv);
158
159    if (CellScannerPosition.AT == position) {
160      if (seekBefore) {
161        ptSearcher.previous();
162        return 1;
163      }
164      return 0;
165    }
166
167    return 1;
168  }
169
170  protected int seekToOrBeforeUsingPositionAtOrAfter(Cell kv, boolean seekBefore) {
171    // should probably switch this to use the seekForwardToOrBefore method
172    CellScannerPosition position = ptSearcher.seekForwardToOrAfter(kv);
173
174    if (CellScannerPosition.AT == position) {
175      if (seekBefore) {
176        ptSearcher.previous();
177        return 1;
178      }
179      return 0;
180
181    }
182
183    if (CellScannerPosition.AFTER == position) {
184      if (!ptSearcher.isBeforeFirst()) {
185        ptSearcher.previous();
186      }
187      return 1;
188    }
189
190    if (position == CellScannerPosition.AFTER_LAST) {
191      if (seekBefore) {
192        ptSearcher.previous();
193      }
194      return 1;
195    }
196
197    throw new RuntimeException("unexpected CellScannerPosition:" + position);
198  }
199
200  @Override
201  public int seekToKeyInBlock(Cell key, boolean forceBeforeOnExactMatch) {
202    if (USE_POSITION_BEFORE) {
203      return seekToOrBeforeUsingPositionAtOrBefore(key, forceBeforeOnExactMatch);
204    } else {
205      return seekToOrBeforeUsingPositionAtOrAfter(key, forceBeforeOnExactMatch);
206    }
207  }
208
209  @Override
210  public int compareKey(CellComparator comparator, Cell key) {
211    return comparator.compare(key,
212        ptSearcher.current());
213  }

215  /**
216   * Cloned version of the PrefixTreeCell where except the value part, the rest
217   * of the key part is deep copied
218   *
219   */
220  private static class OnheapPrefixTreeCell implements Cell, SettableSequenceId, HeapSize {
221    private static final long FIXED_OVERHEAD = ClassSize.align(ClassSize.OBJECT
222        + (5 * ClassSize.REFERENCE) + (2 * Bytes.SIZEOF_LONG) + (4 * Bytes.SIZEOF_INT)
223        + (Bytes.SIZEOF_SHORT) + (2 * Bytes.SIZEOF_BYTE) + (5 * ClassSize.ARRAY));
224    private byte[] row;
225    private short rowLength;
226    private byte[] fam;
227    private byte famLength;
228    private byte[] qual;
229    private int qualLength;
230    private byte[] val;
231    private int valOffset;
232    private int valLength;
233    private byte[] tag;
234    private int tagsLength;
235    private long ts;
236    private long seqId;
237    private byte type;
238
239    public OnheapPrefixTreeCell(byte[] row, int rowOffset, short rowLength, byte[] fam,
240        int famOffset, byte famLength, byte[] qual, int qualOffset, int qualLength, byte[] val,
241        int valOffset, int valLength, byte[] tag, int tagOffset, int tagLength, long ts, byte type,
242        long seqId) {
243      this.row = new byte[rowLength];
244      System.arraycopy(row, rowOffset, this.row, 0, rowLength);
245      this.rowLength = rowLength;
246      this.fam = new byte[famLength];
247      System.arraycopy(fam, famOffset, this.fam, 0, famLength);
248      this.famLength = famLength;
249      this.qual = new byte[qualLength];
250      System.arraycopy(qual, qualOffset, this.qual, 0, qualLength);
251      this.qualLength = qualLength;
252      this.tag = new byte[tagLength];
253      System.arraycopy(tag, tagOffset, this.tag, 0, tagLength);
254      this.tagsLength = tagLength;
255      this.val = val;
256      this.valLength = valLength;
257      this.valOffset = valOffset;
258      this.ts = ts;
259      this.seqId = seqId;
260      this.type = type;
261    }

265      this.seqId = seqId;

358    @Override
359    public String toString() {
360      String row = Bytes.toStringBinary(getRowArray(), getRowOffset(), getRowLength());
361      String family = Bytes.toStringBinary(getFamilyArray(), getFamilyOffset(), getFamilyLength());
362      String qualifier = Bytes.toStringBinary(getQualifierArray(), getQualifierOffset(),
363          getQualifierLength());
364      String timestamp = String.valueOf((getTimestamp()));
365      return row + "/" + family + (family != null && family.length() > 0 ? ":" : "") + qualifier
366          + "/" + timestamp + "/" + Type.codeToType(type);
367    }

369    @Override
370    public long heapSize() {
371      return FIXED_OVERHEAD + rowLength + famLength + qualLength + valLength + tagsLength;
372    }

375  private static class OffheapPrefixTreeCell extends ByteBufferCell implements Cell,
376      SettableSequenceId, HeapSize {
377    private static final long FIXED_OVERHEAD = ClassSize.align(ClassSize.OBJECT
378        + (5 * ClassSize.REFERENCE) + (2 * Bytes.SIZEOF_LONG) + (4 * Bytes.SIZEOF_INT)
379        + (Bytes.SIZEOF_SHORT) + (2 * Bytes.SIZEOF_BYTE) + (5 * ClassSize.BYTE_BUFFER));
380    private ByteBuffer rowBuff;
381    private short rowLength;
382    private ByteBuffer famBuff;
383    private byte famLength;
384    private ByteBuffer qualBuff;
385    private int qualLength;
386    private ByteBuffer val;
387    private int valOffset;
388    private int valLength;
389    private ByteBuffer tagBuff;
390    private int tagsLength;
391    private long ts;
392    private long seqId;
393    private byte type;
394    public OffheapPrefixTreeCell(byte[] row, int rowOffset, short rowLength, byte[] fam,
395        int famOffset, byte famLength, byte[] qual, int qualOffset, int qualLength, ByteBuffer val,
396        int valOffset, int valLength, byte[] tag, int tagOffset, int tagLength, long ts, byte type,
397        long seqId) {
398      byte[] tmpRow = new byte[rowLength];
399      System.arraycopy(row, rowOffset, tmpRow, 0, rowLength);
400      this.rowBuff = ByteBuffer.wrap(tmpRow);
401      this.rowLength = rowLength;
402      byte[] tmpFam = new byte[famLength];
403      System.arraycopy(fam, famOffset, tmpFam, 0, famLength);
404      this.famBuff = ByteBuffer.wrap(tmpFam);
405      this.famLength = famLength;
406      byte[] tmpQual = new byte[qualLength];
407      System.arraycopy(qual, qualOffset, tmpQual, 0, qualLength);
408      this.qualBuff = ByteBuffer.wrap(tmpQual);
409      this.qualLength = qualLength;
410      byte[] tmpTag = new byte[tagLength];
411      System.arraycopy(tag, tagOffset, tmpTag, 0, tagLength);
412      this.tagBuff = ByteBuffer.wrap(tmpTag);
413      this.tagsLength = tagLength;
414      this.val = val;
415      this.valLength = valLength;
416      this.valOffset = valOffset;
417      this.ts = ts;
418      this.seqId = seqId;
419      this.type = type;
420    }
