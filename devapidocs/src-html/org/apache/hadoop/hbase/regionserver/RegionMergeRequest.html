036/**
037 * Handles processing region merges. Put in a queue, owned by HRegionServer.
038 */
039// TODO:UNUSED: REMOVE!!!
040@InterfaceAudience.Private
041class RegionMergeRequest implements Runnable {
042  private static final Log LOG = LogFactory.getLog(RegionMergeRequest.class);
043  private final RegionInfo region_a;
044  private final RegionInfo region_b;
045  private final HRegionServer server;
046  private final boolean forcible;
047  private final User user;
048
049  RegionMergeRequest(Region a, Region b, HRegionServer hrs, boolean forcible,
050      long masterSystemTime, User user) {
051    Preconditions.checkNotNull(hrs);
052    this.region_a = a.getRegionInfo();
053    this.region_b = b.getRegionInfo();
054    this.server = hrs;
055    this.forcible = forcible;
056    this.user = user;
057  }
058
059  @Override
060  public String toString() {
061    return "MergeRequest,regions:" + region_a + ", " + region_b + ", forcible="
062        + forcible;
063  }
064
065  private void doMerge() {
066    boolean success = false;
067    //server.metricsRegionServer.incrMergeRequest();
068
069    if (user != null && user.getUGI() != null) {
070      user.getUGI().doAs (new PrivilegedAction<Void>() {
071        @Override
072        public Void run() {
073          requestRegionMerge();
074          return null;
075        }
076      });
077    } else {
078      requestRegionMerge();
079    }
080  }
081
082  private void requestRegionMerge() {
083    final TableName table = region_a.getTable();
084    if (!table.equals(region_b.getTable())) {
085      LOG.error("Can't merge regions from two different tables: " + region_a + ", " + region_b);
086      return;
087    }
088
089    // TODO: fake merged region for compat with the report protocol
090    final RegionInfo merged = RegionInfoBuilder.newBuilder(table).build();
091
092    // Send the split request to the master. the master will do the validation on the split-key.
093    // The parent region will be unassigned and the two new regions will be assigned.
094    // hri_a and hri_b objects may not reflect the regions that will be created, those objectes
095    // are created just to pass the information to the reportRegionStateTransition().
096    if (!server.reportRegionStateTransition(new RegionStateTransitionContext(
097        TransitionCode.READY_TO_MERGE, HConstants.NO_SEQNUM, -1, merged, region_a, region_b))) {
098      LOG.error("Unable to ask master to merge: " + region_a + ", " + region_b);
099    }
100  }
101
102  @Override
103  public void run() {
104    if (this.server.isStopping() || this.server.isStopped()) {
105      LOG.debug("Skipping merge because server is stopping="
106          + this.server.isStopping() + " or stopped=" + this.server.isStopped());
107      return;
108    }
109
110    doMerge();
111  }
112}
