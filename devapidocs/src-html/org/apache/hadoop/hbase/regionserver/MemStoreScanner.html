020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.IOException;
023import java.util.List;
024
025import org.apache.hadoop.hbase.Cell;
026import org.apache.hadoop.hbase.CellComparator;
027import org.apache.hadoop.hbase.CellUtil;
028import org.apache.hadoop.hbase.HConstants;
029import org.apache.hadoop.hbase.KeyValue;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.client.Scan;
032import org.apache.htrace.Trace;
033
034/**
035 * This is the scanner for any MemStore implementation, derived from MemStore.
036 * The MemStoreScanner combines KeyValueScanner from different Segments and
037 * uses the key-value heap and the reversed key-value heap for the aggregated key-values set.
038 * It is assumed that only traversing forward or backward is used (without zigzagging in between)
039 */
040@InterfaceAudience.Private
041public class MemStoreScanner extends NonLazyKeyValueScanner {
042
043  // heap of scanners, lazily initialized
044  private KeyValueHeap heap;
045
046  // indicates if the scanner is created for inmemoryCompaction
047  private boolean inmemoryCompaction;
048
049  // remember the initial version of the scanners list
050  List<KeyValueScanner> scanners;
051
052  private final CellComparator comparator;
053
054  private boolean closed;
055
056  /**
057   * Creates either a forward KeyValue heap or Reverse KeyValue heap based on the type of scan
058   * and the heap is lazily initialized
059   * @param comparator Cell Comparator
060   * @param scanners List of scanners, from which the heap will be built
061   * @param inmemoryCompaction true if used for inmemoryCompaction.
062   *        In this case, creates a forward heap always.
063   */
064  public MemStoreScanner(CellComparator comparator, List<KeyValueScanner> scanners,
065      boolean inmemoryCompaction) throws IOException {
066    super();
067    this.comparator = comparator;
068    this.scanners = scanners;
069    if (Trace.isTracing() && Trace.currentSpan() != null) {
070      Trace.currentSpan().addTimelineAnnotation("Creating MemStoreScanner");
071    }
072    this.inmemoryCompaction = inmemoryCompaction;
073    if (inmemoryCompaction) {
074      // init the forward scanner in case of inmemoryCompaction
075      initForwardKVHeapIfNeeded(comparator, scanners);
076    }
077  }
078
079  /**
080   * Creates either a forward KeyValue heap or Reverse KeyValue heap based on the type of scan
081   * and the heap is lazily initialized
082   * @param comparator Cell Comparator
083   * @param scanners List of scanners, from which the heap will be built
084   */
085  public MemStoreScanner(CellComparator comparator, List<KeyValueScanner> scanners)
086      throws IOException {
087    this(comparator, scanners, false);
088  }
089
090  private void initForwardKVHeapIfNeeded(CellComparator comparator, List<KeyValueScanner> scanners)
091      throws IOException {
092    if (heap == null) {
093      // lazy init
094      // In a normal scan case, at the StoreScanner level before the KVHeap is
095      // created we do a seek or reseek. So that will happen
096      // on all the scanners that the StoreScanner is
097      // made of. So when we get any of those call to this scanner we init the
098      // heap here with normal forward KVHeap.
099      this.heap = new KeyValueHeap(scanners, comparator);
100    }
101  }
102
103  private boolean initReverseKVHeapIfNeeded(Cell seekKey, CellComparator comparator,
104      List<KeyValueScanner> scanners) throws IOException {
105    boolean res = false;
106    if (heap == null) {
107      // lazy init
108      // In a normal reverse scan case, at the ReversedStoreScanner level before the
109      // ReverseKeyValueheap is
110      // created we do a seekToLastRow or backwardSeek. So that will happen
111      // on all the scanners that the ReversedStoreSCanner is
112      // made of. So when we get any of those call to this scanner we init the
113      // heap here with ReversedKVHeap.
114      if (CellUtil.matchingRow(seekKey, HConstants.EMPTY_START_ROW)) {
115        for (KeyValueScanner scanner : scanners) {
116          res |= scanner.seekToLastRow();
117        }
118      } else {
119        for (KeyValueScanner scanner : scanners) {
120          res |= scanner.backwardSeek(seekKey);
121        }
122      }
123      this.heap = new ReversedKeyValueHeap(scanners, comparator);
124    }
125    return res;
126  }
127
128  /**
129   * Returns the cell from the top-most scanner without advancing the iterator.
130   * The backward traversal is assumed, only if specified explicitly
131   */
132  @Override
133  public Cell peek() {
134    if (this.heap != null) {
135      return this.heap.peek();
136    }
137    // Doing this way in case some test cases tries to peek directly to avoid NPE
138    return null;
139  }
140
141  /**
142   * Gets the next cell from the top-most scanner. Assumed forward scanning.
143   */
144  @Override
145  public Cell next() throws IOException {
146    if(this.heap != null) {
147      // loop over till the next suitable value
148      // take next value from the heap
149      for (Cell currentCell = heap.next();
150          currentCell != null;
151          currentCell = heap.next()) {
152        // all the logic of presenting cells is inside the internal KeyValueScanners
153        // located inside the heap
154        return currentCell;
155      }
156    }
157    return null;
158  }
159
160  /**
161   * Set the scanner at the seek key. Assumed forward scanning.
162   * Must be called only once: there is no thread safety between the scanner
163   * and the memStore.
164   *
165   * @param cell seek value
166   * @return false if the key is null or if there is no data
167   */
168  @Override
169  public boolean seek(Cell cell) throws IOException {
170    initForwardKVHeapIfNeeded(comparator, scanners);
171
172    if (cell == null) {
173      close();
174      return false;
175    }
176
177    return heap.seek(cell);
178  }
179
180  /**
181   * Move forward on the sub-lists set previously by seek. Assumed forward scanning.
182   *
183   * @param cell seek value (should be non-null)
184   * @return true if there is at least one KV to read, false otherwise
185   */
186  @Override
187  public boolean reseek(Cell cell) throws IOException {
188    /*
189    * See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
190    * This code is executed concurrently with flush and puts, without locks.
191    * Two points must be known when working on this code:
192    * 1) It's not possible to use the 'kvTail' and 'snapshot'
193    *  variables, as they are modified during a flush.
194    * 2) The ideal implementation for performance would use the sub skip list
195    *  implicitly pointed by the iterators 'kvsetIt' and
196    *  'snapshotIt'. Unfortunately the Java API does not offer a method to
197    *  get it. So we remember the last keys we iterated to and restore
198    *  the reseeked set to at least that point.
199    *
200    *  TODO: The above comment copied from the original MemStoreScanner
201    */
202    initForwardKVHeapIfNeeded(comparator, scanners);
203    return heap.reseek(cell);
204  }
205
206  /**
207   * MemStoreScanner returns Long.MAX_VALUE because it will always have the latest data among all
208   * scanners.
209   * @see KeyValueScanner#getScannerOrder()
210   */
211  @Override
212  public long getScannerOrder() {
213    return Long.MAX_VALUE;
214  }
215
216  @Override
217  public void close() {
218    if (closed) {
219      return;
220    }
221    // Ensuring that all the segment scanners are closed
222    if (heap != null) {
223      heap.close();
224      // It is safe to do close as no new calls will be made to this scanner.
225      heap = null;
226    } else {
227      for (KeyValueScanner scanner : scanners) {
228        scanner.close();
229      }
230    }
231    closed = true;
232  }
233
234  /**
235   * Set the scanner at the seek key. Assumed backward scanning.
236   *
237   * @param cell seek value
238   * @return false if the key is null or if there is no data
239   */
240  @Override
241  public boolean backwardSeek(Cell cell) throws IOException {
242    // The first time when this happens it sets the scanners to the seek key
243    // passed by the incoming scan's start row
244    initReverseKVHeapIfNeeded(cell, comparator, scanners);
245    return heap.backwardSeek(cell);
246  }
247
248  /**
249   * Assumed backward scanning.
250   *
251   * @param cell seek value
252   * @return false if the key is null or if there is no data
253   */
254  @Override
255  public boolean seekToPreviousRow(Cell cell) throws IOException {
256    initReverseKVHeapIfNeeded(cell, comparator, scanners);
257    if (heap.peek() == null) {
258      restartBackwardHeap(cell);
259    }
260    return heap.seekToPreviousRow(cell);
261  }
262
263  @Override
264  public boolean seekToLastRow() throws IOException {
265    return initReverseKVHeapIfNeeded(KeyValue.LOWESTKEY, comparator, scanners);
266  }
267
268  /**
269   * Check if this memstore may contain the required keys
270   * @return False if the key definitely does not exist in this Memstore
271   */
272  @Override
273  public boolean shouldUseScanner(Scan scan, Store store, long oldestUnexpiredTS) {
274    // TODO : Check if this can be removed.
275    if (inmemoryCompaction) {
276      return true;
277    }
278
279    for (KeyValueScanner sc : scanners) {
280      if (sc.shouldUseScanner(scan, store, oldestUnexpiredTS)) {
281        return true;
282      }
283    }
284    return false;
285  }
286
287  // debug method
288  @Override
289  public String toString() {
290    StringBuffer buf = new StringBuffer();
291    int i = 1;
292    for (KeyValueScanner scanner : scanners) {
293      buf.append("scanner (" + i + ") " + scanner.toString() + " ||| ");
294      i++;
295    }
296    return buf.toString();
297  }
298  /****************** Private methods ******************/
299  /**
300   * Restructure the ended backward heap after rerunning a seekToPreviousRow()
301   * on each scanner
302   * @return false if given Cell does not exist in any scanner
303   */
304  private boolean restartBackwardHeap(Cell cell) throws IOException {
305    boolean res = false;
306    for (KeyValueScanner scan : scanners) {
307      res |= scan.seekToPreviousRow(cell);
308    }
309    this.heap =
310        new ReversedKeyValueHeap(scanners, comparator);
311    return res;
312  }
