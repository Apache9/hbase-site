018package org.apache.hadoop.hbase.regionserver;
019
020import java.util.List;
021
022import org.apache.commons.logging.Log;
023import org.apache.commons.logging.LogFactory;
024import org.apache.hadoop.hbase.ScheduledChore;
025import org.apache.hadoop.hbase.Server;
026import org.apache.hadoop.hbase.Stoppable;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.executor.EventType;
029import org.apache.hadoop.hbase.regionserver.Region;
030import org.apache.hadoop.hbase.regionserver.RegionServerServices;
031import org.apache.hadoop.hbase.regionserver.Store;
032
033import com.google.common.annotations.VisibleForTesting;
034
035/**
036 * A chore service that periodically cleans up the compacted files when there are no active readers
037 * using those compacted files and also helps in clearing the block cache with these compacted
038 * file entries
039 */
040@InterfaceAudience.Private
041public class CompactedHFilesDischarger extends ScheduledChore {
042  private static final Log LOG = LogFactory.getLog(CompactedHFilesDischarger.class);
043  private RegionServerServices regionServerServices;
044  // Default is to use executor
045  @VisibleForTesting
046  private boolean useExecutor = true;
047
048  /**
049   * @param period the period of time to sleep between each run
050   * @param stopper the stopper
051   * @param regionServerServices the region server that starts this chore
052   */
053  public CompactedHFilesDischarger(final int period, final Stoppable stopper,
054      final RegionServerServices regionServerServices) {
055    // Need to add the config classes
056    super("CompactedHFilesCleaner", stopper, period);
057    this.regionServerServices = regionServerServices;
058  }
059
060  /**
061   * @param period the period of time to sleep between each run
062   * @param stopper the stopper
063   * @param regionServerServices the region server that starts this chore
064   * @param useExecutor true if to use the region server's executor service, false otherwise
065   */
066  @VisibleForTesting
067  public CompactedHFilesDischarger(final int period, final Stoppable stopper,
068      final RegionServerServices regionServerServices, boolean useExecutor) {
069    // Need to add the config classes
070    this(period, stopper, regionServerServices);
071    this.useExecutor = useExecutor;
072  }
073
074  @Override
075  public void chore() {
076    // Noop if rss is null. This will never happen in a normal condition except for cases
077    // when the test case is not spinning up a cluster
078    if (regionServerServices == null) return;
079    List<Region> onlineRegions = regionServerServices.getOnlineRegions();
080    if (onlineRegions != null) {
081      for (Region region : onlineRegions) {
082        if (LOG.isTraceEnabled()) {
083          LOG.trace(
084              "Started the compacted hfiles cleaner for the region " + region.getRegionInfo());
085        }
086        for (Store store : region.getStores()) {
087          try {
088            if (useExecutor && regionServerServices != null) {
089              CompactedHFilesDischargeHandler handler = new CompactedHFilesDischargeHandler(
090                  (Server) regionServerServices, EventType.RS_COMPACTED_FILES_DISCHARGER,
091                  (HStore) store);
092              regionServerServices.getExecutorService().submit(handler);
093            } else {
094              // call synchronously if the RegionServerServices are not
095              // available
096              store.closeAndArchiveCompactedFiles();
097            }
098            if (LOG.isTraceEnabled()) {
099              LOG.trace("Completed archiving the compacted files for the region "
100                  + region.getRegionInfo() + " under the store " + store.getColumnFamilyName());
101            }
102          } catch (Exception e) {
103            LOG.error("Exception while trying to close and archive the comapcted store "
104                + "files of the store  " + store.getColumnFamilyName() + " in the" + " region "
105                + region.getRegionInfo(), e);
106          }
107        }
108        if (LOG.isTraceEnabled()) {
109          LOG.trace(
110              "Completed the compacted hfiles cleaner for the region " + region.getRegionInfo());
111        }
112      }
113    }
114  }
115}
