019package org.apache.hadoop.hbase.regionserver;

021import java.util.concurrent.BlockingQueue;
022import java.util.concurrent.Executors;
023import java.util.concurrent.LinkedBlockingQueue;
024import java.util.concurrent.ScheduledExecutorService;
025import java.util.concurrent.TimeUnit;
026import java.util.concurrent.atomic.AtomicLong;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.regionserver.HeapMemoryManager.HeapMemoryTuneObserver;
032import org.apache.hadoop.util.StringUtils;
033
034import com.google.common.annotations.VisibleForTesting;
035import com.google.common.util.concurrent.ThreadFactoryBuilder;
036
037/**
038 * A pool of {@link Chunk} instances.
039 * 
040 * MemStoreChunkPool caches a number of retired chunks for reusing, it could
041 * decrease allocating bytes when writing, thereby optimizing the garbage
042 * collection on JVM.
043 * 
044 * The pool instance is globally unique and could be obtained through
045 * {@link MemStoreChunkPool#initialize(long, float, float, int, boolean)}
046 * 
047 * {@link MemStoreChunkPool#getChunk()} is called when MemStoreLAB allocating
048 * bytes, and {@link MemStoreChunkPool#putbackChunks(BlockingQueue)} is called
049 * when MemStore clearing snapshot for flush
050 */
051@SuppressWarnings("javadoc")
052@InterfaceAudience.Private
053public class MemStoreChunkPool implements HeapMemoryTuneObserver {
054  private static final Log LOG = LogFactory.getLog(MemStoreChunkPool.class);
055
056  // Static reference to the MemStoreChunkPool
057  static MemStoreChunkPool GLOBAL_INSTANCE;
058  /** Boolean whether we have disabled the memstore chunk pool entirely. */
059  static boolean chunkPoolDisabled = false;
060
061  private int maxCount;
062
063  // A queue of reclaimed chunks
064  private final BlockingQueue<Chunk> reclaimedChunks;
065  private final int chunkSize;
066  private final float poolSizePercentage;
067
068  /** Statistics thread schedule pool */
069  private final ScheduledExecutorService scheduleThreadPool;
070  /** Statistics thread */
071  private static final int statThreadPeriod = 60 * 5;
072  private final AtomicLong chunkCount = new AtomicLong();
073  private final AtomicLong reusedChunkCount = new AtomicLong();
074  private final boolean offheap;
075
076  MemStoreChunkPool(int chunkSize, int maxCount, int initialCount, float poolSizePercentage,
077      boolean offheap) {
078    this.maxCount = maxCount;
079    this.chunkSize = chunkSize;
080    this.poolSizePercentage = poolSizePercentage;
081    this.offheap = offheap;
082    this.reclaimedChunks = new LinkedBlockingQueue<>();
083    for (int i = 0; i < initialCount; i++) {
084      Chunk chunk = this.offheap ? new OffheapChunk(chunkSize) : new OnheapChunk(chunkSize);
085      chunk.init();
086      reclaimedChunks.add(chunk);
087    }
088    chunkCount.set(initialCount);
089    final String n = Thread.currentThread().getName();
090    scheduleThreadPool = Executors.newScheduledThreadPool(1, new ThreadFactoryBuilder()
091        .setNameFormat(n + "-MemStoreChunkPool Statistics").setDaemon(true).build());
092    this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(), statThreadPeriod,
093        statThreadPeriod, TimeUnit.SECONDS);
094  }
095
096  /**
097   * Poll a chunk from the pool, reset it if not null, else create a new chunk to return if we have
098   * not yet created max allowed chunks count. When we have already created max allowed chunks and
099   * no free chunks as of now, return null. It is the responsibility of the caller to make a chunk
100   * then.
101   * Note: Chunks returned by this pool must be put back to the pool after its use.
102   * @return a chunk
103   * @see #putbackChunk(Chunk)
104   * @see #putbackChunks(BlockingQueue)
105   */
106  Chunk getChunk() {
107    Chunk chunk = reclaimedChunks.poll();
108    if (chunk != null) {
109      chunk.reset();
110      reusedChunkCount.incrementAndGet();
111    } else {
112      // Make a chunk iff we have not yet created the maxCount chunks
113      while (true) {
114        long created = this.chunkCount.get();
115        if (created < this.maxCount) {
116          chunk = this.offheap ? new OffheapChunk(this.chunkSize) : new OnheapChunk(this.chunkSize);
117          if (this.chunkCount.compareAndSet(created, created + 1)) {
118            break;
119          }
120        } else {
121          break;
122        }
123      }
124    }
125    return chunk;
126  }
127
128  /**
129   * Add the chunks to the pool, when the pool achieves the max size, it will
130   * skip the remaining chunks
131   * @param chunks
132   */
133  synchronized void putbackChunks(BlockingQueue<Chunk> chunks) {
134    int toAdd = Math.min(chunks.size(), this.maxCount - reclaimedChunks.size());
135    Chunk chunk = null;
136    while ((chunk = chunks.poll()) != null && toAdd > 0) {
137      reclaimedChunks.add(chunk);
138      toAdd--;
139    }
140  }
141
142  /**
143   * Add the chunk to the pool, if the pool has achieved the max size, it will
144   * skip it
145   * @param chunk
146   */
147  synchronized void putbackChunk(Chunk chunk) {
148    if (reclaimedChunks.size() < this.maxCount) {
149      reclaimedChunks.add(chunk);
150    }
151  }
152
153  int getPoolSize() {
154    return this.reclaimedChunks.size();
155  }
156
157  /*
158   * Only used in testing
159   */
160  void clearChunks() {
161    this.reclaimedChunks.clear();
162  }
163
164  private class StatisticsThread extends Thread {
165    StatisticsThread() {
166      super("MemStoreChunkPool.StatisticsThread");
167      setDaemon(true);
168    }
169
170    @Override
171    public void run() {
172      logStats();
173    }
174
175    private void logStats() {
176      if (!LOG.isDebugEnabled()) return;
177      long created = chunkCount.get();
178      long reused = reusedChunkCount.get();
179      long total = created + reused;
180      LOG.debug("Stats: current pool size=" + reclaimedChunks.size()
181          + ",created chunk count=" + created
182          + ",reused chunk count=" + reused
183          + ",reuseRatio=" + (total == 0 ? "0" : StringUtils.formatPercent(
184              (float) reused / (float) total, 2)));
185    }
186  }
187
188  /**
189   * @return the global MemStoreChunkPool instance
190   */
191  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "LI_LAZY_INIT_STATIC",
192      justification = "Method is called by single thread at the starting of RS")
193  static MemStoreChunkPool initialize(long globalMemStoreSize, float poolSizePercentage,
194      float initialCountPercentage, int chunkSize, boolean offheap) {
195    if (GLOBAL_INSTANCE != null) return GLOBAL_INSTANCE;
196    if (chunkPoolDisabled) return null;
197
198    if (poolSizePercentage <= 0) {
199      chunkPoolDisabled = true;
200      return null;
201    }
202    if (poolSizePercentage > 1.0) {
203      throw new IllegalArgumentException(
204          MemStoreLAB.CHUNK_POOL_MAXSIZE_KEY + " must be between 0.0 and 1.0");
205    }
206    int maxCount = (int) (globalMemStoreSize * poolSizePercentage / chunkSize);
207    if (initialCountPercentage > 1.0 || initialCountPercentage < 0) {
208      throw new IllegalArgumentException(
209          MemStoreLAB.CHUNK_POOL_INITIALSIZE_KEY + " must be between 0.0 and 1.0");
210    }
211    int initialCount = (int) (initialCountPercentage * maxCount);
212    LOG.info("Allocating MemStoreChunkPool with chunk size " + StringUtils.byteDesc(chunkSize)
213        + ", max count " + maxCount + ", initial count " + initialCount);
214    GLOBAL_INSTANCE = new MemStoreChunkPool(chunkSize, maxCount, initialCount, poolSizePercentage,
215        offheap);
216    return GLOBAL_INSTANCE;
217  }
218
219  /**
220   * @return The singleton instance of this pool.
221   */
222  static MemStoreChunkPool getPool() {
223    return GLOBAL_INSTANCE;
224  }
225
226  int getMaxCount() {
227    return this.maxCount;
228  }
229
230  @VisibleForTesting
231  static void clearDisableFlag() {
232    chunkPoolDisabled = false;
233  }
234
235  @Override
236  public void onHeapMemoryTune(long newMemstoreSize, long newBlockCacheSize) {
237    // don't do any tuning in case of offheap memstore
238    if (this.offheap) {
239      LOG.warn("Not tuning the chunk pool as it is offheap");
240      return;
241    }
242    int newMaxCount = (int) (newMemstoreSize * poolSizePercentage / chunkSize);
243    if (newMaxCount != this.maxCount) {
244      // We need an adjustment in the chunks numbers
245      if (newMaxCount > this.maxCount) {
246        // Max chunks getting increased. Just change the variable. Later calls to getChunk() would
247        // create and add them to Q
248        LOG.info("Max count for chunks increased from " + this.maxCount + " to " + newMaxCount);
249        this.maxCount = newMaxCount;
250      } else {
251        // Max chunks getting decreased. We may need to clear off some of the pooled chunks now
252        // itself. If the extra chunks are serving already, do not pool those when we get them back
253        LOG.info("Max count for chunks decreased from " + this.maxCount + " to " + newMaxCount);
254        this.maxCount = newMaxCount;
255        if (this.reclaimedChunks.size() > newMaxCount) {
256          synchronized (this) {
257            while (this.reclaimedChunks.size() > newMaxCount) {
258              this.reclaimedChunks.poll();
259            }
260          }
261        }
262      }
263    }
264  }
