040/**
041 * A pool of {@link Chunk} instances.
042 * 
043 * MemStoreChunkPool caches a number of retired chunks for reusing, it could
044 * decrease allocating bytes when writing, thereby optimizing the garbage
045 * collection on JVM.
046 * 
047 * The pool instance is globally unique and could be obtained through
048 * {@link MemStoreChunkPool#getPool(Configuration)}
049 * 
050 * {@link MemStoreChunkPool#getChunk()} is called when MemStoreLAB allocating
051 * bytes, and {@link MemStoreChunkPool#putbackChunks(BlockingQueue)} is called
052 * when MemStore clearing snapshot for flush
053 */
054@SuppressWarnings("javadoc")
055@InterfaceAudience.Private
056public class MemStoreChunkPool implements HeapMemoryTuneObserver {
057  private static final Log LOG = LogFactory.getLog(MemStoreChunkPool.class);
058  final static String CHUNK_POOL_MAXSIZE_KEY = "hbase.hregion.memstore.chunkpool.maxsize";
059  final static String CHUNK_POOL_INITIALSIZE_KEY = "hbase.hregion.memstore.chunkpool.initialsize";
060  final static float POOL_MAX_SIZE_DEFAULT = 0.0f;
061  final static float POOL_INITIAL_SIZE_DEFAULT = 0.0f;
062
063  // Static reference to the MemStoreChunkPool
064  static MemStoreChunkPool GLOBAL_INSTANCE;
065  /** Boolean whether we have disabled the memstore chunk pool entirely. */
066  static boolean chunkPoolDisabled = false;
067
068  private int maxCount;
069
070  // A queue of reclaimed chunks
071  private final BlockingQueue<PooledChunk> reclaimedChunks;
072  private final int chunkSize;
073  private final float poolSizePercentage;
074
075  /** Statistics thread schedule pool */
076  private final ScheduledExecutorService scheduleThreadPool;
077  /** Statistics thread */
078  private static final int statThreadPeriod = 60 * 5;
079  private final AtomicLong chunkCount = new AtomicLong();
080  private final AtomicLong reusedChunkCount = new AtomicLong();
081
082  MemStoreChunkPool(Configuration conf, int chunkSize, int maxCount,
083      int initialCount, float poolSizePercentage) {
084    this.maxCount = maxCount;
085    this.chunkSize = chunkSize;
086    this.poolSizePercentage = poolSizePercentage;
087    this.reclaimedChunks = new LinkedBlockingQueue<PooledChunk>();
088    for (int i = 0; i < initialCount; i++) {
089      PooledChunk chunk = new PooledChunk(chunkSize);
090      chunk.init();
091      reclaimedChunks.add(chunk);
092    }
093    chunkCount.set(initialCount);
094    final String n = Thread.currentThread().getName();
095    scheduleThreadPool = Executors.newScheduledThreadPool(1, new ThreadFactoryBuilder()
096        .setNameFormat(n + "-MemStoreChunkPool Statistics").setDaemon(true).build());
097    this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(), statThreadPeriod,
098        statThreadPeriod, TimeUnit.SECONDS);
099  }
100
101  /**
102   * Poll a chunk from the pool, reset it if not null, else create a new chunk to return if we have
103   * not yet created max allowed chunks count. When we have already created max allowed chunks and
104   * no free chunks as of now, return null. It is the responsibility of the caller to make a chunk
105   * then.
106   * Note: Chunks returned by this pool must be put back to the pool after its use.
107   * @return a chunk
108   * @see #putbackChunk(Chunk)
109   * @see #putbackChunks(BlockingQueue)
110   */
111  PooledChunk getChunk() {
112    PooledChunk chunk = reclaimedChunks.poll();
113    if (chunk != null) {
114      chunk.reset();
115      reusedChunkCount.incrementAndGet();
116    } else {
117      // Make a chunk iff we have not yet created the maxCount chunks
118      while (true) {
119        long created = this.chunkCount.get();
120        if (created < this.maxCount) {
121          chunk = new PooledChunk(chunkSize);
122          if (this.chunkCount.compareAndSet(created, created + 1)) {
123            break;
124          }
125        } else {
126          break;
127        }
128      }
129    }
130    return chunk;
131  }
132
133  /**
134   * Add the chunks to the pool, when the pool achieves the max size, it will
135   * skip the remaining chunks
136   * @param chunks
137   */
138  synchronized void putbackChunks(BlockingQueue<PooledChunk> chunks) {
139    int toAdd = Math.min(chunks.size(), this.maxCount - reclaimedChunks.size());
140    PooledChunk chunk = null;
141    while ((chunk = chunks.poll()) != null && toAdd > 0) {
142      reclaimedChunks.add(chunk);
143      toAdd--;
144    }
145  }
146
147  /**
148   * Add the chunk to the pool, if the pool has achieved the max size, it will
149   * skip it
150   * @param chunk
151   */
152  synchronized void putbackChunk(PooledChunk chunk) {
153    if (reclaimedChunks.size() < this.maxCount) {
154      reclaimedChunks.add(chunk);
155    }
156  }
157
158  int getPoolSize() {
159    return this.reclaimedChunks.size();
160  }
161
162  /*
163   * Only used in testing
164   */
165  void clearChunks() {
166    this.reclaimedChunks.clear();
167  }
168
169  private class StatisticsThread extends Thread {
170    StatisticsThread() {
171      super("MemStoreChunkPool.StatisticsThread");
172      setDaemon(true);
173    }
174
175    @Override
176    public void run() {
177      logStats();
178    }
179
180    private void logStats() {
181      if (!LOG.isDebugEnabled()) return;
182      long created = chunkCount.get();
183      long reused = reusedChunkCount.get();
184      long total = created + reused;
185      LOG.debug("Stats: current pool size=" + reclaimedChunks.size()
186          + ",created chunk count=" + created
187          + ",reused chunk count=" + reused
188          + ",reuseRatio=" + (total == 0 ? "0" : StringUtils.formatPercent(
189              (float) reused / (float) total, 2)));
190    }
191  }
192
193  /**
194   * @param conf
195   * @return the global MemStoreChunkPool instance
196   */
197  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="DC_DOUBLECHECK",
198      justification="Intentional")
199  static MemStoreChunkPool getPool(Configuration conf) {
200    if (GLOBAL_INSTANCE != null) return GLOBAL_INSTANCE;
201
202    synchronized (MemStoreChunkPool.class) {
203      if (chunkPoolDisabled) return null;
204      if (GLOBAL_INSTANCE != null) return GLOBAL_INSTANCE;
205      float poolSizePercentage = conf.getFloat(CHUNK_POOL_MAXSIZE_KEY, POOL_MAX_SIZE_DEFAULT);
206      if (poolSizePercentage <= 0) {
207        chunkPoolDisabled = true;
208        return null;
209      }
210      if (poolSizePercentage > 1.0) {
211        throw new IllegalArgumentException(CHUNK_POOL_MAXSIZE_KEY + " must be between 0.0 and 1.0");
212      }
213      long heapMax = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage().getMax();
214      long globalMemStoreLimit = (long) (heapMax * HeapMemorySizeUtil.getGlobalMemStorePercent(conf,
215          false));
216      int chunkSize = conf.getInt(HeapMemStoreLAB.CHUNK_SIZE_KEY,
217          HeapMemStoreLAB.CHUNK_SIZE_DEFAULT);
218      int maxCount = (int) (globalMemStoreLimit * poolSizePercentage / chunkSize);
219
220      float initialCountPercentage = conf.getFloat(CHUNK_POOL_INITIALSIZE_KEY,
221          POOL_INITIAL_SIZE_DEFAULT);
222      if (initialCountPercentage > 1.0 || initialCountPercentage < 0) {
223        throw new IllegalArgumentException(CHUNK_POOL_INITIALSIZE_KEY
224            + " must be between 0.0 and 1.0");
225      }
226
227      int initialCount = (int) (initialCountPercentage * maxCount);
228      LOG.info("Allocating MemStoreChunkPool with chunk size " + StringUtils.byteDesc(chunkSize)
229          + ", max count " + maxCount + ", initial count " + initialCount);
230      GLOBAL_INSTANCE = new MemStoreChunkPool(conf, chunkSize, maxCount, initialCount,
231          poolSizePercentage);
232      return GLOBAL_INSTANCE;
233    }
234  }
235
236  int getMaxCount() {
237    return this.maxCount;
238  }
239
240  @VisibleForTesting
241  static void clearDisableFlag() {
242    chunkPoolDisabled = false;
243  }
244
245  public static class PooledChunk extends Chunk {
246    PooledChunk(int size) {
247      super(size);
248    }
249  }
250
251  @Override
252  public void onHeapMemoryTune(long newMemstoreSize, long newBlockCacheSize) {
253    int newMaxCount = (int) (newMemstoreSize * poolSizePercentage / chunkSize);
254    if (newMaxCount != this.maxCount) {
255      // We need an adjustment in the chunks numbers
256      if (newMaxCount > this.maxCount) {
257        // Max chunks getting increased. Just change the variable. Later calls to getChunk() would
258        // create and add them to Q
259        LOG.info("Max count for chunks increased from " + this.maxCount + " to " + newMaxCount);
260        this.maxCount = newMaxCount;
261      } else {
262        // Max chunks getting decreased. We may need to clear off some of the pooled chunks now
263        // itself. If the extra chunks are serving already, do not pool those when we get them back
264        LOG.info("Max count for chunks decreased from " + this.maxCount + " to " + newMaxCount);
265        this.maxCount = newMaxCount;
266        if (this.reclaimedChunks.size() > newMaxCount) {
267          synchronized (this) {
268            while (this.reclaimedChunks.size() > newMaxCount) {
269              this.reclaimedChunks.poll();
270            }
271          }
272        }
273      }
274    }
275  }
