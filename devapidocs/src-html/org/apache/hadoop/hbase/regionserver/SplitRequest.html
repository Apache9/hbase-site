019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026import org.apache.hadoop.hbase.DroppedSnapshotException;
027import org.apache.hadoop.hbase.master.TableLockManager.TableLock;
028import org.apache.hadoop.hbase.security.User;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
031import org.apache.hadoop.ipc.RemoteException;
032import org.apache.hadoop.util.StringUtils;
033
034import com.google.common.base.Preconditions;
035
036/**
037 * Handles processing region splits. Put in a queue, owned by HRegionServer.
038 */
039@InterfaceAudience.Private
040class SplitRequest implements Runnable {
041  private static final Log LOG = LogFactory.getLog(SplitRequest.class);
042  private final HRegion parent;
043  private final byte[] midKey;
044  private final HRegionServer server;
045  private final User user;
046  private TableLock tableLock;
047
048  SplitRequest(Region region, byte[] midKey, HRegionServer hrs, User user) {
049    Preconditions.checkNotNull(hrs);
050    this.parent = (HRegion)region;
051    this.midKey = midKey;
052    this.server = hrs;
053    this.user = user;
054  }
055
056  @Override
057  public String toString() {
058    return "regionName=" + parent + ", midKey=" + Bytes.toStringBinary(midKey);
059  }
060
061  private void doSplitting(User user) {
062    boolean success = false;
063    server.metricsRegionServer.incrSplitRequest();
064    long startTime = EnvironmentEdgeManager.currentTime();
065    SplitTransactionImpl st = new SplitTransactionImpl(parent, midKey);
066    try {
067      //acquire a shared read lock on the table, so that table schema modifications
068      //do not happen concurrently
069      tableLock = server.getTableLockManager().readLock(parent.getTableDesc().getTableName()
070          , "SPLIT_REGION:" + parent.getRegionInfo().getRegionNameAsString());
071      try {
072        tableLock.acquire();
073      } catch (IOException ex) {
074        tableLock = null;
075        throw ex;
076      }
077
078      // If prepare does not return true, for some reason -- logged inside in
079      // the prepare call -- we are not ready to split just now. Just return.
080      if (!st.prepare()) return;
081      try {
082        st.execute(this.server, this.server, user);
083        success = true;
084      } catch (Exception e) {
085        if (this.server.isStopping() || this.server.isStopped()) {
086          LOG.info(
087              "Skip rollback/cleanup of failed split of "
088                  + parent.getRegionInfo().getRegionNameAsString() + " because server is"
089                  + (this.server.isStopping() ? " stopping" : " stopped"), e);
090          return;
091        }
092        if (e instanceof DroppedSnapshotException) {
093          server.abort("Replay of WAL required. Forcing server shutdown", e);
094          return;
095        }
096        try {
097          LOG.info("Running rollback/cleanup of failed split of " +
098            parent.getRegionInfo().getRegionNameAsString() + "; " + e.getMessage(), e);
099          if (st.rollback(this.server, this.server)) {
100            LOG.info("Successful rollback of failed split of " +
101              parent.getRegionInfo().getRegionNameAsString());
102          } else {
103            this.server.abort("Abort; we got an error after point-of-no-return");
104          }
105        } catch (RuntimeException ee) {
106          String msg = "Failed rollback of failed split of " +
107            parent.getRegionInfo().getRegionNameAsString() + " -- aborting server";
108          // If failed rollback, kill this server to avoid having a hole in table.
109          LOG.info(msg, ee);
110          this.server.abort(msg + " -- Cause: " + ee.getMessage());
111        }
112        return;
113      }
114    } catch (IOException ex) {
115      ex = ex instanceof RemoteException ? ((RemoteException) ex).unwrapRemoteException() : ex;
116      LOG.error("Split failed " + this, ex);
117      server.checkFileSystem();
118    } finally {
119      if (this.parent.getCoprocessorHost() != null) {
120        try {
121          this.parent.getCoprocessorHost().postCompleteSplit();
122        } catch (IOException io) {
123          LOG.error("Split failed " + this,
124            io instanceof RemoteException ? ((RemoteException) io).unwrapRemoteException() : io);
125        }
126      }
127      if (parent.shouldForceSplit()) {
128        parent.clearSplit();
129      }
130      releaseTableLock();
131      long endTime = EnvironmentEdgeManager.currentTime();
132      // Update regionserver metrics with the split transaction total running time
133      server.metricsRegionServer.updateSplitTime(endTime - startTime);
134      if (success) {
135        server.metricsRegionServer.incrSplitSuccess();
136        // Log success
137        LOG.info("Region split, hbase:meta updated, and report to master. Parent="
138            + parent.getRegionInfo().getRegionNameAsString() + ", new regions: "
139            + st.getFirstDaughter().getRegionNameAsString() + ", "
140            + st.getSecondDaughter().getRegionNameAsString() + ". Split took "
141            + StringUtils.formatTimeDiff(EnvironmentEdgeManager.currentTime(), startTime));
142      }
143      // Always log the split transaction journal
144      LOG.info("Split transaction journal:\n\t" + StringUtils.join("\n\t", st.getJournal()));
145    }
146  }
147
148  @Override
149  public void run() {
150    if (this.server.isStopping() || this.server.isStopped()) {
151      LOG.debug("Skipping split because server is stopping=" +
152        this.server.isStopping() + " or stopped=" + this.server.isStopped());
153      return;
154    }
155    doSplitting(user);
156  }
157
158  protected void releaseTableLock() {
159    if (this.tableLock != null) {
160      try {
161        this.tableLock.release();
162      } catch (IOException ex) {
163        LOG.error("Could not release the table lock (something is really wrong). "
164           + "Aborting this server to avoid holding the lock forever.");
165        this.server.abort("Abort; we got an error when releasing the table lock "
166                         + "on " + parent.getRegionInfo().getRegionNameAsString());
167      }
168    }
169  }
170}
