042/**
043 * Does the management of memstoreLAB chunk creations. A monotonically incrementing id is associated
044 * with every chunk
045 */
046@InterfaceAudience.Private
047public class ChunkCreator {
048  private static final Logger LOG = LoggerFactory.getLogger(ChunkCreator.class);
049  // monotonically increasing chunkid
050  private AtomicInteger chunkID = new AtomicInteger(1);
051  // maps the chunk against the monotonically increasing chunk id. We need to preserve the
052  // natural ordering of the key
053  // CellChunkMap creation should convert the weak ref to hard reference
054
055  // chunk id of each chunk is the first integer written on each chunk,
056  // the header size need to be changed in case chunk id size is changed
057  public static final int SIZEOF_CHUNK_HEADER = Bytes.SIZEOF_INT;
058
059  // mapping from chunk IDs to chunks
060  private Map<Integer, Chunk> chunkIdMap = new ConcurrentHashMap<Integer, Chunk>();
061
062  private final int chunkSize;
063  private final boolean offheap;
064  @VisibleForTesting
065  static ChunkCreator INSTANCE;
066  @VisibleForTesting
067  static boolean chunkPoolDisabled = false;
068  private MemStoreChunkPool pool;
069
070  @VisibleForTesting
071  ChunkCreator(int chunkSize, boolean offheap, long globalMemStoreSize, float poolSizePercentage,
072      float initialCountPercentage, HeapMemoryManager heapMemoryManager) {
073    this.chunkSize = chunkSize;
074    this.offheap = offheap;
075    this.pool = initializePool(globalMemStoreSize, poolSizePercentage, initialCountPercentage);
076    if (heapMemoryManager != null && this.pool != null) {
077      // Register with Heap Memory manager
078      heapMemoryManager.registerTuneObserver(this.pool);
079    }
080  }

123  /**
124   * Creates and inits a chunk.
125   * @return the chunk that was initialized
126   * @param chunkIndexType whether the requested chunk is going to be used with CellChunkMap index
127   * @param size the size of the chunk to be allocated, in bytes
128   */
129  Chunk getChunk(CompactingMemStore.IndexType chunkIndexType, int size) {
130    Chunk chunk = null;
131    // if we have pool and this is not jumbo chunk (when size != chunkSize this is jumbo chunk)
132    if ((pool != null) && (size == chunkSize)) {
133      //  the pool creates the chunk internally. The chunk#init() call happens here
134      chunk = this.pool.getChunk();
135      // the pool has run out of maxCount
136      if (chunk == null) {
137        if (LOG.isTraceEnabled()) {
138          LOG.trace("Chunk pool full (maxCount={}); creating chunk offheap.",
139              this.pool.getMaxCount());
140        }
141      }
142    }
143    if (chunk == null) {
144      // the second parameter explains whether CellChunkMap index is requested,
145      // in that case, put allocated on demand chunk mapping into chunkIdMap
146      chunk = createChunk(false, chunkIndexType, size);
147    }
148
149    // now we need to actually do the expensive memory allocation step in case of a new chunk,
150    // else only the offset is set to the beginning of the chunk to accept allocations
151    chunk.init();
152    return chunk;
153  }
154
155  /**
156   * Creates and inits a chunk of a special size, bigger than a regular chunk size.
157   * Such a chunk will never come from pool and will always be on demand allocated.
158   * @return the chunk that was initialized
159   * @param chunkIndexType whether the requested chunk is going to be used with CellChunkMap index
160   * @param jumboSize the special size to be used
161   */
162  Chunk getJumboChunk(CompactingMemStore.IndexType chunkIndexType, int jumboSize) {
163    if (jumboSize <= chunkSize) {
164      LOG.warn("Jumbo chunk size=" + jumboSize + " must be more than regular chunk size="
165          + chunkSize + "; converting to regular chunk.");
166      return getChunk(chunkIndexType,chunkSize);
167    }
168    // the size of the allocation includes
169    // both the size requested and a place for the Chunk's header
170    return getChunk(chunkIndexType, jumboSize + SIZEOF_CHUNK_HEADER);
171  }
172
173  /**
174   * Creates the chunk either onheap or offheap
175   * @param pool indicates if the chunks have to be created which will be used by the Pool
176   * @param chunkIndexType whether the requested chunk is going to be used with CellChunkMap index
177   * @param size the size of the chunk to be allocated, in bytes
178   * @return the chunk
179   */
180  private Chunk createChunk(boolean pool, CompactingMemStore.IndexType chunkIndexType, int size) {
181    Chunk chunk = null;
182    int id = chunkID.getAndIncrement();
183    assert id > 0;
184    // do not create offheap chunk on demand
185    if (pool && this.offheap) {
186      chunk = new OffheapChunk(size, id, pool);
187    } else {
188      chunk = new OnheapChunk(size, id, pool);
189    }
190    if (pool || (chunkIndexType == CompactingMemStore.IndexType.CHUNK_MAP)) {
191      // put the pool chunk into the chunkIdMap so it is not GC-ed
192      this.chunkIdMap.put(chunk.getId(), chunk);
193    }
194    return chunk;
195  }
196
197  // Chunks from pool are created covered with strong references anyway
198  // TODO: change to CHUNK_MAP if it is generally defined
199  private Chunk createChunkForPool() {
200    return createChunk(true, CompactingMemStore.IndexType.ARRAY_MAP, chunkSize);
201  }

238  /**
239   * A pool of {@link Chunk} instances.
240   *
241   * MemStoreChunkPool caches a number of retired chunks for reusing, it could
242   * decrease allocating bytes when writing, thereby optimizing the garbage
243   * collection on JVM.
244   */
245  private  class MemStoreChunkPool implements HeapMemoryTuneObserver {
246    private int maxCount;
247
248    // A queue of reclaimed chunks
249    private final BlockingQueue<Chunk> reclaimedChunks;
250    private final float poolSizePercentage;
251
252    /** Statistics thread schedule pool */
253    private final ScheduledExecutorService scheduleThreadPool;
254    /** Statistics thread */
255    private static final int statThreadPeriod = 60 * 5;
256    private final AtomicLong chunkCount = new AtomicLong();
257    private final LongAdder reusedChunkCount = new LongAdder();
258
259    MemStoreChunkPool(int maxCount, int initialCount, float poolSizePercentage) {
260      this.maxCount = maxCount;
261      this.poolSizePercentage = poolSizePercentage;
262      this.reclaimedChunks = new LinkedBlockingQueue<>();
263      for (int i = 0; i < initialCount; i++) {
264        Chunk chunk = createChunkForPool();
265        chunk.init();
266        reclaimedChunks.add(chunk);
267      }
268      chunkCount.set(initialCount);
269      final String n = Thread.currentThread().getName();
270      scheduleThreadPool = Executors.newScheduledThreadPool(1, new ThreadFactoryBuilder()
271          .setNameFormat(n + "-MemStoreChunkPool Statistics").setDaemon(true).build());
272      this.scheduleThreadPool.scheduleAtFixedRate(new StatisticsThread(), statThreadPeriod,
273          statThreadPeriod, TimeUnit.SECONDS);
274    }
275
276    /**
277     * Poll a chunk from the pool, reset it if not null, else create a new chunk to return if we have
278     * not yet created max allowed chunks count. When we have already created max allowed chunks and
279     * no free chunks as of now, return null. It is the responsibility of the caller to make a chunk
280     * then.
281     * Note: Chunks returned by this pool must be put back to the pool after its use.
282     * @return a chunk
283     * @see #putbackChunks(Chunk)
284     */
285    Chunk getChunk() {
286      Chunk chunk = reclaimedChunks.poll();
287      if (chunk != null) {
288        chunk.reset();
289        reusedChunkCount.increment();
290      } else {
291        // Make a chunk iff we have not yet created the maxCount chunks
292        while (true) {
293          long created = this.chunkCount.get();
294          if (created < this.maxCount) {
295            if (this.chunkCount.compareAndSet(created, created + 1)) {
296              chunk = createChunkForPool();
297              break;
298            }
299          } else {
300            break;
301          }
302        }
303      }
304      return chunk;
305    }
306
307    /**
308     * Add the chunks to the pool, when the pool achieves the max size, it will skip the remaining
309     * chunks
310     * @param c
311     */
312    private void putbackChunks(Chunk c) {
313      int toAdd = this.maxCount - reclaimedChunks.size();
314      if (c.isFromPool() && toAdd > 0) {
315        reclaimedChunks.add(c);
316      } else {
317        // remove the chunk (that is not going to pool)
318        // though it is initially from the pool or not
319        ChunkCreator.this.removeChunk(c.getId());
320      }
321    }
322
323    private class StatisticsThread extends Thread {
324      StatisticsThread() {
325        super("MemStoreChunkPool.StatisticsThread");
326        setDaemon(true);
327      }
328
329      @Override
330      public void run() {
331        logStats();
332      }
333
334      private void logStats() {
335        if (!LOG.isDebugEnabled()) return;
336        long created = chunkCount.get();
337        long reused = reusedChunkCount.sum();
338        long total = created + reused;
339        LOG.debug("Stats: current pool size=" + reclaimedChunks.size()
340            + ",created chunk count=" + created
341            + ",reused chunk count=" + reused
342            + ",reuseRatio=" + (total == 0 ? "0" : StringUtils.formatPercent(
343                (float) reused / (float) total, 2)));
344      }
345    }
346
347    private int getMaxCount() {
348      return this.maxCount;
349    }
350
351    @Override
352    public void onHeapMemoryTune(long newMemstoreSize, long newBlockCacheSize) {
353      // don't do any tuning in case of offheap memstore
354      if (isOffheap()) {
355        LOG.warn("Not tuning the chunk pool as it is offheap");
356        return;
357      }
358      int newMaxCount =
359          (int) (newMemstoreSize * poolSizePercentage / getChunkSize());
360      if (newMaxCount != this.maxCount) {
361        // We need an adjustment in the chunks numbers
362        if (newMaxCount > this.maxCount) {
363          // Max chunks getting increased. Just change the variable. Later calls to getChunk() would
364          // create and add them to Q
365          LOG.info("Max count for chunks increased from " + this.maxCount + " to " + newMaxCount);
366          this.maxCount = newMaxCount;
367        } else {
368          // Max chunks getting decreased. We may need to clear off some of the pooled chunks now
369          // itself. If the extra chunks are serving already, do not pool those when we get them back
370          LOG.info("Max count for chunks decreased from " + this.maxCount + " to " + newMaxCount);
371          this.maxCount = newMaxCount;
372          if (this.reclaimedChunks.size() > newMaxCount) {
373            synchronized (this) {
374              while (this.reclaimedChunks.size() > newMaxCount) {
375                this.reclaimedChunks.poll();
376              }
377            }
378          }
379        }
380      }
381    }
382  }

389  private MemStoreChunkPool initializePool(long globalMemStoreSize, float poolSizePercentage,
390      float initialCountPercentage) {
391    if (poolSizePercentage <= 0) {
392      LOG.info("PoolSizePercentage is less than 0. So not using pool");
393      return null;
394    }
395    if (chunkPoolDisabled) {
396      return null;
397    }
398    if (poolSizePercentage > 1.0) {
399      throw new IllegalArgumentException(
400          MemStoreLAB.CHUNK_POOL_MAXSIZE_KEY + " must be between 0.0 and 1.0");
401    }
402    int maxCount = (int) (globalMemStoreSize * poolSizePercentage / getChunkSize());
403    if (initialCountPercentage > 1.0 || initialCountPercentage < 0) {
404      throw new IllegalArgumentException(
405          MemStoreLAB.CHUNK_POOL_INITIALSIZE_KEY + " must be between 0.0 and 1.0");
406    }
407    int initialCount = (int) (initialCountPercentage * maxCount);
408    LOG.info("Allocating MemStoreChunkPool with chunk size="
409        + StringUtils.byteDesc(getChunkSize()) + ", max count=" + maxCount
410        + ", initial count=" + initialCount);
411    return new MemStoreChunkPool(maxCount, initialCount, poolSizePercentage);
412  }

430  @VisibleForTesting
431  boolean isChunkInPool(int chunkId) {
432    if (pool != null) {
433      // chunks that are from pool will return true chunk reference not null
434      Chunk c = getChunk(chunkId);
435      if (c==null) {
436        return false;
437      }
438      return pool.reclaimedChunks.contains(c);
439    }
440
441    return false;
442  }

454  synchronized void putbackChunks(Set<Integer> chunks) {
455    // if there is no pool just try to clear the chunkIdMap in case there is something
456    if ( pool == null ) {
457      this.removeChunks(chunks);
458      return;
459    }
460
461    // if there is pool, go over all chunk IDs that came back, the chunks may be from pool or not
462    for (int chunkID : chunks) {
463      // translate chunk ID to chunk, if chunk initially wasn't in pool
464      // this translation will (most likely) return null
465      Chunk chunk = ChunkCreator.this.getChunk(chunkID);
466      if (chunk != null) {
467        // Jumbo chunks are covered with chunkIdMap, but are not from pool, so such a chunk should
468        // be released here without going to pool.
469        // Removing them from chunkIdMap will cause their removal by the GC.
470        if (chunk.isJumbo()) {
471          this.removeChunk(chunkID);
472        } else {
473          pool.putbackChunks(chunk);
474        }
475      }
476      // if chunk is null, it was never covered by the chunkIdMap (and so wasn't in pool also),
477      // so we have nothing to do on its release
478    }
479    return;
480  }
