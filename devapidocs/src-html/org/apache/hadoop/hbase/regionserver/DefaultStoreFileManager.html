019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.Comparator;
026import java.util.Iterator;
027import java.util.List;
028import java.util.Optional;
029
030import org.apache.commons.logging.Log;
031import org.apache.commons.logging.LogFactory;
032import org.apache.hadoop.conf.Configuration;
033import org.apache.hadoop.hbase.Cell;
034import org.apache.hadoop.hbase.CellComparator;
035import org.apache.hadoop.hbase.KeyValue;
036import org.apache.hadoop.hbase.classification.InterfaceAudience;
037import org.apache.hadoop.hbase.regionserver.compactions.CompactionConfiguration;
038
039import com.google.common.collect.ImmutableCollection;
040import com.google.common.collect.ImmutableList;
041import com.google.common.collect.Lists;
042
043/**
044 * Default implementation of StoreFileManager. Not thread-safe.
045 */
046@InterfaceAudience.Private
047class DefaultStoreFileManager implements StoreFileManager {
048  private static final Log LOG = LogFactory.getLog(DefaultStoreFileManager.class);
049
050  private final CellComparator kvComparator;
051  private final CompactionConfiguration comConf;
052  private final int blockingFileCount;
053  private final Comparator<StoreFile> storeFileComparator;
054  /**
055   * List of store files inside this store. This is an immutable list that
056   * is atomically replaced when its contents change.
057   */
058  private volatile ImmutableList<StoreFile> storefiles = null;
059  /**
060   * List of compacted files inside this store that needs to be excluded in reads
061   * because further new reads will be using only the newly created files out of compaction.
062   * These compacted files will be deleted/cleared once all the existing readers on these
063   * compacted files are done.
064   */
065  private volatile List<StoreFile> compactedfiles = null;
066
067  public DefaultStoreFileManager(CellComparator kvComparator,
068      Comparator<StoreFile> storeFileComparator, Configuration conf,
069      CompactionConfiguration comConf) {
070    this.kvComparator = kvComparator;
071    this.storeFileComparator = storeFileComparator;
072    this.comConf = comConf;
073    this.blockingFileCount =
074        conf.getInt(HStore.BLOCKING_STOREFILES_KEY, HStore.DEFAULT_BLOCKING_STOREFILE_COUNT);
075  }
076
077  @Override
078  public void loadFiles(List<StoreFile> storeFiles) {
079    sortAndSetStoreFiles(storeFiles);
080  }
081
082  @Override
083  public final Collection<StoreFile> getStorefiles() {
084    // TODO: I can return a null list of StoreFiles? That'll mess up clients. St.Ack 20151111
085    return storefiles;
086  }
087
088  @Override
089  public Collection<StoreFile> getCompactedfiles() {
090    return compactedfiles;
091  }
092
093  @Override
094  public void insertNewFiles(Collection<StoreFile> sfs) throws IOException {
095    ArrayList<StoreFile> newFiles = new ArrayList<>(storefiles);
096    newFiles.addAll(sfs);
097    sortAndSetStoreFiles(newFiles);
098  }
099
100  @Override
101  public ImmutableCollection<StoreFile> clearFiles() {
102    ImmutableList<StoreFile> result = storefiles;
103    storefiles = ImmutableList.of();
104    return result;
105  }
106
107  @Override
108  public Collection<StoreFile> clearCompactedFiles() {
109    List<StoreFile> result = compactedfiles;
110    compactedfiles = new ArrayList<>();
111    return result;
112  }
113
114  @Override
115  public final int getStorefileCount() {
116    return storefiles.size();
117  }
118
119  @Override
120  public final int getCompactedFilesCount() {
121    if (compactedfiles == null) {
122      return 0;
123    }
124    return compactedfiles.size();
125  }
126
127  @Override
128  public void addCompactionResults(
129    Collection<StoreFile> newCompactedfiles, Collection<StoreFile> results) {
130    ArrayList<StoreFile> newStoreFiles = Lists.newArrayList(storefiles);
131    newStoreFiles.removeAll(newCompactedfiles);
132    if (!results.isEmpty()) {
133      newStoreFiles.addAll(results);
134    }
135    sortAndSetStoreFiles(newStoreFiles);
136    ArrayList<StoreFile> updatedCompactedfiles = null;
137    if (this.compactedfiles != null) {
138      updatedCompactedfiles = new ArrayList<>(this.compactedfiles);
139      updatedCompactedfiles.addAll(newCompactedfiles);
140    } else {
141      updatedCompactedfiles = new ArrayList<>(newCompactedfiles);
142    }
143    markCompactedAway(newCompactedfiles);
144    this.compactedfiles = sortCompactedfiles(updatedCompactedfiles);
145  }
146
147  // Mark the files as compactedAway once the storefiles and compactedfiles list is finalized
148  // Let a background thread close the actual reader on these compacted files and also
149  // ensure to evict the blocks from block cache so that they are no longer in
150  // cache
151  private void markCompactedAway(Collection<StoreFile> compactedFiles) {
152    for (StoreFile file : compactedFiles) {
153      file.markCompactedAway();
154    }
155  }
156
157  @Override
158  public void removeCompactedFiles(Collection<StoreFile> removedCompactedfiles) throws IOException {
159    ArrayList<StoreFile> updatedCompactedfiles = null;
160    if (this.compactedfiles != null) {
161      updatedCompactedfiles = new ArrayList<>(this.compactedfiles);
162      updatedCompactedfiles.removeAll(removedCompactedfiles);
163      this.compactedfiles = sortCompactedfiles(updatedCompactedfiles);
164    }
165  }
166
167  @Override
168  public final Iterator<StoreFile> getCandidateFilesForRowKeyBefore(final KeyValue targetKey) {
169    return new ArrayList<>(Lists.reverse(this.storefiles)).iterator();
170  }
171
172  @Override
173  public Iterator<StoreFile> updateCandidateFilesForRowKeyBefore(
174      Iterator<StoreFile> candidateFiles, final KeyValue targetKey, final Cell candidate) {
175    // Default store has nothing useful to do here.
176    // TODO: move this comment when implementing Level:
177    // Level store can trim the list by range, removing all the files which cannot have
178    // any useful candidates less than "candidate".
179    return candidateFiles;
180  }
181
182  @Override
183  public final byte[] getSplitPoint() throws IOException {
184    List<StoreFile> storefiles = this.storefiles;
185    if (storefiles.isEmpty()) {
186      return null;
187    }
188    Optional<StoreFile> largestFile = StoreUtils.getLargestFile(storefiles);
189    return largestFile.isPresent()
190        ? StoreUtils.getFileSplitPoint(largestFile.get(), kvComparator).orElse(null) : null;
191  }
192
193  @Override
194  public final Collection<StoreFile> getFilesForScan(byte[] startRow, boolean includeStartRow,
195      byte[] stopRow, boolean includeStopRow) {
196    // We cannot provide any useful input and already have the files sorted by seqNum.
197    return getStorefiles();
198  }
199
200  @Override
201  public int getStoreCompactionPriority() {
202    int priority = blockingFileCount - storefiles.size();
203    return (priority == HStore.PRIORITY_USER) ? priority + 1 : priority;
204  }
205
206  @Override
207  public Collection<StoreFile> getUnneededFiles(long maxTs, List<StoreFile> filesCompacting) {
208    Collection<StoreFile> expiredStoreFiles = null;
209    ImmutableList<StoreFile> files = storefiles;
210    // 1) We can never get rid of the last file which has the maximum seqid.
211    // 2) Files that are not the latest can't become one due to (1), so the rest are fair game.
212    for (int i = 0; i < files.size() - 1; ++i) {
213      StoreFile sf = files.get(i);
214      long fileTs = sf.getReader().getMaxTimestamp();
215      if (fileTs < maxTs && !filesCompacting.contains(sf)) {
216        LOG.info("Found an expired store file: " + sf.getPath()
217            + " whose maxTimeStamp is " + fileTs + ", which is below " + maxTs);
218        if (expiredStoreFiles == null) {
219          expiredStoreFiles = new ArrayList<>();
220        }
221        expiredStoreFiles.add(sf);
222      }
223    }
224    return expiredStoreFiles;
225  }
226
227  private void sortAndSetStoreFiles(List<StoreFile> storeFiles) {
228    Collections.sort(storeFiles, storeFileComparator);
229    storefiles = ImmutableList.copyOf(storeFiles);
230  }
231
232  private List<StoreFile> sortCompactedfiles(List<StoreFile> storefiles) {
233    // Sorting may not be really needed here for the compacted files?
234    Collections.sort(storefiles, storeFileComparator);
235    return new ArrayList<>(storefiles);
236  }
237
238  @Override
239  public double getCompactionPressure() {
240    int storefileCount = getStorefileCount();
241    int minFilesToCompact = comConf.getMinFilesToCompact();
242    if (storefileCount <= minFilesToCompact) {
243      return 0.0;
244    }
245    return (double) (storefileCount - minFilesToCompact) / (blockingFileCount - minFilesToCompact);
246  }
247
248  @Override
249  public Comparator<StoreFile> getStoreFileComparator() {
250    return storeFileComparator;
251  }
252}
