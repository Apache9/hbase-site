043/**
044 * An abstract class, which implements the behaviour shared by all concrete memstore instances.
045 */
046@InterfaceAudience.Private
047public abstract class AbstractMemStore implements MemStore {
048
049  private static final long NO_SNAPSHOT_ID = -1;
050
051  private final Configuration conf;
052  private final CellComparator comparator;
053
054  // active segment absorbs write operations
055  protected volatile MutableSegment active;
056  // Snapshot of memstore.  Made for flusher.
057  protected volatile ImmutableSegment snapshot;
058  protected volatile long snapshotId;
059  // Used to track when to flush
060  private volatile long timeOfOldestEdit;
061
062  public final static long FIXED_OVERHEAD = ClassSize
063      .align(ClassSize.OBJECT + (4 * ClassSize.REFERENCE) + (2 * Bytes.SIZEOF_LONG));
064
065  public final static long DEEP_OVERHEAD = FIXED_OVERHEAD;
066
067  protected AbstractMemStore(final Configuration conf, final CellComparator c) {
068    this.conf = conf;
069    this.comparator = c;
070    resetActive();
071    this.snapshot = SegmentFactory.instance().createImmutableSegment(c);
072    this.snapshotId = NO_SNAPSHOT_ID;
073  }
074
075  protected void resetActive() {
076    // Reset heap to not include any keys
077    this.active = SegmentFactory.instance().createMutableSegment(conf, comparator);
078    this.timeOfOldestEdit = Long.MAX_VALUE;
079  }

081  /*
082  * Calculate how the MemStore size has changed.  Includes overhead of the
083  * backing Map.
084  * @param cell
085  * @param notPresent True if the cell was NOT present in the set.
086  * @return change in size
087  */
088  static long heapSizeChange(final Cell cell, final boolean notPresent) {
089    return notPresent ? ClassSize.align(ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY
090        + CellUtil.estimatedHeapSizeOf(cell)) : 0;
091  }

100  /**
101   * Write an update
102   * @param cell the cell to be added
103   * @return approximate size of the passed cell & newly added cell which maybe different than the
104   *         passed-in cell
105   */
106  @Override
107  public long add(Cell cell) {
108    Cell toAdd = maybeCloneWithAllocator(cell);
109    boolean mslabUsed = (toAdd != cell);
110    // This cell data is backed by the same byte[] where we read request in RPC(See HBASE-15180). By
111    // default MSLAB is ON and we might have copied cell to MSLAB area. If not we must do below deep
112    // copy. Or else we will keep referring to the bigger chunk of memory and prevent it from
113    // getting GCed.
114    // Copy to MSLAB would not have happened if
115    // 1. MSLAB is turned OFF. See "hbase.hregion.memstore.mslab.enabled"
116    // 2. When the size of the cell is bigger than the max size supported by MSLAB. See
117    // "hbase.hregion.memstore.mslab.max.allocation". This defaults to 256 KB
118    // 3. When cells are from Append/Increment operation.
119    if (!mslabUsed) {
120      toAdd = deepCopyIfNeeded(toAdd);
121    }
122    return internalAdd(toAdd, mslabUsed);
123  }
124
125  private static Cell deepCopyIfNeeded(Cell cell) {
126    // When Cell is backed by a shared memory chunk (this can be a chunk of memory where we read the
127    // req into) the Cell instance will be of type ShareableMemory. Later we will add feature to
128    // read the RPC request into pooled direct ByteBuffers.
129    if (cell instanceof ShareableMemory) {
130      return ((ShareableMemory) cell).cloneToCell();
131    }
132    return cell;
133  }

135  /**
136   * Update or insert the specified Cells.
137   * <p>
138   * For each Cell, insert into MemStore.  This will atomically upsert the
139   * value for that row/family/qualifier.  If a Cell did already exist,
140   * it will then be removed.
141   * <p>
142   * Currently the memstoreTS is kept at 0 so as each insert happens, it will
143   * be immediately visible.  May want to change this so it is atomic across
144   * all Cells.
145   * <p>
146   * This is called under row lock, so Get operations will still see updates
147   * atomically.  Scans will only see each Cell update as atomic.
148   *
149   * @param cells the cells to be updated
150   * @param readpoint readpoint below which we can safely remove duplicate KVs
151   * @return change in memstore size
152   */
153  @Override
154  public long upsert(Iterable<Cell> cells, long readpoint) {
155    long size = 0;
156    for (Cell cell : cells) {
157      size += upsert(cell, readpoint);
158    }
159    return size;
160  }

172   * Write a delete

174   * @return approximate size of the passed key and value.

176  @Override
177  public long delete(Cell deleteCell) {
178    // Delete operation just adds the delete marker cell coming here.
179    return add(deleteCell);

182  /**
183   * The passed snapshot was successfully persisted; it can be let go.
184   * @param id Id of the snapshot to clean out.
185   * @see MemStore#snapshot()
186   */
187  @Override
188  public void clearSnapshot(long id) throws UnexpectedStateException {
189    if (this.snapshotId == -1) return;  // already cleared
190    if (this.snapshotId != id) {
191      throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
192          + id);
193    }
194    // OK. Passed in snapshot is same as current snapshot. If not-empty,
195    // create a new snapshot and let the old one go.
196    Segment oldSnapshot = this.snapshot;
197    if (!this.snapshot.isEmpty()) {
198      this.snapshot = SegmentFactory.instance().createImmutableSegment(this.comparator);
199    }
200    this.snapshotId = NO_SNAPSHOT_ID;
201    oldSnapshot.close();
202  }

218  @Override
219  public String toString() {
220    StringBuffer buf = new StringBuffer();
221    int i = 1;
222    try {
223      for (Segment segment : getSegments()) {
224        buf.append("Segment (" + i + ") " + segment.toString() + "; ");
225        i++;
226      }
227    } catch (IOException e){
228      return e.toString();
229    }
230    return buf.toString();
231  }

243  /**
244   * Inserts the specified Cell into MemStore and deletes any existing
245   * versions of the same row/family/qualifier as the specified Cell.
246   * <p>
247   * First, the specified Cell is inserted into the Memstore.
248   * <p>
249   * If there are any existing Cell in this MemStore with the same row,
250   * family, and qualifier, they are removed.
251   * <p>
252   * Callers must hold the read lock.
253   *
254   * @param cell the cell to be updated
255   * @param readpoint readpoint below which we can safely remove duplicate KVs
256   * @return change in size of MemStore
257   */
258  private long upsert(Cell cell, long readpoint) {
259    // Add the Cell to the MemStore
260    // Use the internalAdd method here since we (a) already have a lock
261    // and (b) cannot safely use the MSLAB here without potentially
262    // hitting OOME - see TestMemStore.testUpsertMSLAB for a
263    // test that triggers the pathological case if we don't avoid MSLAB
264    // here.
265    // This cell data is backed by the same byte[] where we read request in RPC(See HBASE-15180). We
266    // must do below deep copy. Or else we will keep referring to the bigger chunk of memory and
267    // prevent it from getting GCed.
268    cell = deepCopyIfNeeded(cell);
269    long addedSize = internalAdd(cell, false);
270
271    // Get the Cells for the row/family/qualifier regardless of timestamp.
272    // For this case we want to clean up any other puts
273    Cell firstCell = CellUtil.createFirstOnRow(
274        cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
275        cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
276        cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength());
277    SortedSet<Cell> ss = active.tailSet(firstCell);
278    Iterator<Cell> it = ss.iterator();
279    // versions visible to oldest scanner
280    int versionsVisible = 0;
281    while (it.hasNext()) {
282      Cell cur = it.next();
283
284      if (cell == cur) {
285        // ignore the one just put in
286        continue;
287      }
288      // check that this is the row and column we are interested in, otherwise bail
289      if (CellUtil.matchingRow(cell, cur) && CellUtil.matchingQualifier(cell, cur)) {
290        // only remove Puts that concurrent scanners cannot possibly see
291        if (cur.getTypeByte() == KeyValue.Type.Put.getCode() &&
292            cur.getSequenceId() <= readpoint) {
293          if (versionsVisible >= 1) {
294            // if we get here we have seen at least one version visible to the oldest scanner,
295            // which means we can prove that no scanner will see this version
296
297            // false means there was a change, so give us the size.
298            long delta = heapSizeChange(cur, true);
299            addedSize -= delta;
300            active.incSize(-delta);
301            it.remove();
302            setOldestEditTimeToNow();
303          } else {
304            versionsVisible++;
305          }
306        }
307      } else {
308        // past the row or column, done
309        break;
310      }
311    }
312    return addedSize;
313  }

315  /*
316   * @param a
317   * @param b
318   * @return Return lowest of a or b or null if both a and b are null
319   */
320  protected Cell getLowest(final Cell a, final Cell b) {
321    if (a == null) {
322      return b;
323    }
324    if (b == null) {
325      return a;
326    }
327    return comparator.compareRows(a, b) <= 0? a: b;
328  }

330  /*
331   * @param key Find row that follows this one.  If null, return first.
332   * @param set Set to look in for a row beyond <code>row</code>.
333   * @return Next row or null if none found.  If one found, will be a new
334   * KeyValue -- can be destroyed by subsequent calls to this method.
335   */
336  protected Cell getNextRow(final Cell key,
337      final NavigableSet<Cell> set) {
338    Cell result = null;
339    SortedSet<Cell> tail = key == null? set: set.tailSet(key);
340    // Iterate until we fall into the next row; i.e. move off current row
341    for (Cell cell: tail) {
342      if (comparator.compareRows(cell, key) <= 0) {
343        continue;
344      }
345      // Note: Not suppressing deletes or expired cells.  Needs to be handled
346      // by higher up functions.
347      result = cell;
348      break;
349    }
350    return result;
351  }

353  /**
354   * Given the specs of a column, update it, first by inserting a new record,
355   * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
356   * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
357   * store will ensure that the insert/delete each are atomic. A scanner/reader will either
358   * get the new value, or the old value and all readers will eventually only see the new
359   * value after the old was removed.
360   */
361  @VisibleForTesting
362  @Override
363  public long updateColumnValue(byte[] row, byte[] family, byte[] qualifier,
364      long newValue, long now) {
365    Cell firstCell = KeyValueUtil.createFirstOnRow(row, family, qualifier);
366    // Is there a Cell in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
367    Cell snc = snapshot.getFirstAfter(firstCell);
368    if(snc != null) {
369      // is there a matching Cell in the snapshot?
370      if (CellUtil.matchingRow(snc, firstCell) && CellUtil.matchingQualifier(snc, firstCell)) {
371        if (snc.getTimestamp() == now) {
372          now += 1;
373        }
374      }
375    }
376    // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
377    // But the timestamp should also be max(now, mostRecentTsInMemstore)
378
379    // so we cant add the new Cell w/o knowing what's there already, but we also
380    // want to take this chance to delete some cells. So two loops (sad)
381
382    SortedSet<Cell> ss = this.active.tailSet(firstCell);
383    for (Cell cell : ss) {
384      // if this isnt the row we are interested in, then bail:
385      if (!CellUtil.matchingColumn(cell, family, qualifier)
386          || !CellUtil.matchingRow(cell, firstCell)) {
387        break; // rows dont match, bail.
388      }
389
390      // if the qualifier matches and it's a put, just RM it out of the active.
391      if (cell.getTypeByte() == KeyValue.Type.Put.getCode() &&
392          cell.getTimestamp() > now && CellUtil.matchingQualifier(firstCell, cell)) {
393        now = cell.getTimestamp();
394      }
395    }
396
397    // create or update (upsert) a new Cell with
398    // 'now' and a 0 memstoreTS == immediately visible
399    List<Cell> cells = new ArrayList<Cell>(1);
400    cells.add(new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue)));
401    return upsert(cells, 1L);
402  }

404  private Cell maybeCloneWithAllocator(Cell cell) {
405    return active.maybeCloneWithAllocator(cell);
406  }

408  /**
409   * Internal version of add() that doesn't clone Cells with the
410   * allocator, and doesn't take the lock.
411   *
412   * Callers should ensure they already have the read lock taken
413   * @param toAdd the cell to add
414   * @param mslabUsed whether using MSLAB
415   * @return the heap size change in bytes
416   */
417  private long internalAdd(final Cell toAdd, final boolean mslabUsed) {
418    long s = active.add(toAdd, mslabUsed);
419    setOldestEditTimeToNow();
420    checkActiveSize();
421    return s;
422  }
423
424  private void setOldestEditTimeToNow() {
425    if (timeOfOldestEdit == Long.MAX_VALUE) {
426      timeOfOldestEdit = EnvironmentEdgeManager.currentTime();
427    }
428  }
