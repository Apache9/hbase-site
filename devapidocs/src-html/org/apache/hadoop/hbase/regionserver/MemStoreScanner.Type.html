020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.IOException;
023import java.util.List;
024
025import org.apache.hadoop.hbase.Cell;
026import org.apache.hadoop.hbase.KeyValue;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.client.Scan;
029import org.apache.htrace.Trace;
030
031/**
032 * This is the scanner for any MemStore implementation, derived from MemStore.
033 * The MemStoreScanner combines SegmentScanner from different Segments and
034 * uses the key-value heap and the reversed key-value heap for the aggregated key-values set.
035 * It is assumed that only traversing forward or backward is used (without zigzagging in between)
036 */
037@InterfaceAudience.Private
038public class MemStoreScanner extends NonLazyKeyValueScanner {
039  /**
040   * Types of cell MemStoreScanner
041   */
042  static public enum Type {
043    UNDEFINED,
044    COMPACT_FORWARD,
045    USER_SCAN_FORWARD,
046    USER_SCAN_BACKWARD
047  }
048
049  // heap of scanners used for traversing forward
050  private KeyValueHeap forwardHeap;
051  // reversed scanners heap for traversing backward
052  private ReversedKeyValueHeap backwardHeap;
053
054  // The type of the scan is defined by constructor
055  // or according to the first usage
056  private Type type = Type.UNDEFINED;
057
058  private long readPoint;
059  // remember the initial version of the scanners list
060  List<KeyValueScanner> scanners;
061  // pointer back to the relevant MemStore
062  // is needed for shouldSeek() method
063  private AbstractMemStore backwardReferenceToMemStore;
064
065  /**
066   * If UNDEFINED type for MemStoreScanner is provided, the forward heap is used as default!
067   * After constructor only one heap is going to be initialized for entire lifespan
068   * of the MemStoreScanner. A specific scanner can only be one directional!
069   *
070   * @param ms        Pointer back to the MemStore
071   * @param scanners  List of scanners over the segments
072   * @param readPt    Read point below which we can safely remove duplicate KVs
073   */
074  public MemStoreScanner(AbstractMemStore ms, List<KeyValueScanner> scanners, long readPt)
075      throws IOException {
076    this(ms, scanners, readPt, Type.UNDEFINED);
077  }
078
079  /**
080   * If UNDEFINED type for MemStoreScanner is provided, the forward heap is used as default!
081   * After constructor only one heap is going to be initialized for entire lifespan
082   * of the MemStoreScanner. A specific scanner can only be one directional!
083   *
084   * @param ms        Pointer back to the MemStore
085   * @param scanners  List of scanners over the segments
086   * @param readPt Read point below which we can safely remove duplicate KVs
087   * @param type      The scan type COMPACT_FORWARD should be used for compaction
088   */
089  public MemStoreScanner(AbstractMemStore ms, List<KeyValueScanner> scanners, long readPt,
090      Type type) throws IOException {
091    super();
092    this.readPoint = readPt;
093    this.type = type;
094    switch (type) {
095      case UNDEFINED:
096      case USER_SCAN_FORWARD:
097      case COMPACT_FORWARD:
098        this.forwardHeap = new KeyValueHeap(scanners, ms.getComparator());
099        break;
100      case USER_SCAN_BACKWARD:
101        this.backwardHeap = new ReversedKeyValueHeap(scanners, ms.getComparator());
102        break;
103      default:
104        throw new IllegalArgumentException("Unknown scanner type in MemStoreScanner");
105    }
106    this.backwardReferenceToMemStore = ms;
107    this.scanners = scanners;
108    if (Trace.isTracing() && Trace.currentSpan() != null) {
109      Trace.currentSpan().addTimelineAnnotation("Creating MemStoreScanner");
110    }
111  }
112
113  /**
114   * Returns the cell from the top-most scanner without advancing the iterator.
115   * The backward traversal is assumed, only if specified explicitly
116   */
117  @Override
118  public synchronized Cell peek() {
119    if (type == Type.USER_SCAN_BACKWARD) {
120      return backwardHeap.peek();
121    }
122    return forwardHeap.peek();
123  }
124
125  /**
126   * Gets the next cell from the top-most scanner. Assumed forward scanning.
127   */
128  @Override
129  public synchronized Cell next() throws IOException {
130    KeyValueHeap heap = (Type.USER_SCAN_BACKWARD == type) ? backwardHeap : forwardHeap;
131
132    // loop over till the next suitable value
133    // take next value from the heap
134    for (Cell currentCell = heap.next();
135         currentCell != null;
136         currentCell = heap.next()) {
137
138      // all the logic of presenting cells is inside the internal SegmentScanners
139      // located inside the heap
140
141      return currentCell;
142    }
143    return null;
144  }
145
146  /**
147   * Set the scanner at the seek key. Assumed forward scanning.
148   * Must be called only once: there is no thread safety between the scanner
149   * and the memStore.
150   *
151   * @param cell seek value
152   * @return false if the key is null or if there is no data
153   */
154  @Override
155  public synchronized boolean seek(Cell cell) throws IOException {
156    assertForward();
157
158    if (cell == null) {
159      close();
160      return false;
161    }
162
163    return forwardHeap.seek(cell);
164  }
165
166  /**
167   * Move forward on the sub-lists set previously by seek. Assumed forward scanning.
168   *
169   * @param cell seek value (should be non-null)
170   * @return true if there is at least one KV to read, false otherwise
171   */
172  @Override
173  public synchronized boolean reseek(Cell cell) throws IOException {
174    /*
175    * See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
176    * This code is executed concurrently with flush and puts, without locks.
177    * Two points must be known when working on this code:
178    * 1) It's not possible to use the 'kvTail' and 'snapshot'
179    *  variables, as they are modified during a flush.
180    * 2) The ideal implementation for performance would use the sub skip list
181    *  implicitly pointed by the iterators 'kvsetIt' and
182    *  'snapshotIt'. Unfortunately the Java API does not offer a method to
183    *  get it. So we remember the last keys we iterated to and restore
184    *  the reseeked set to at least that point.
185    *
186    *  TODO: The above comment copied from the original MemStoreScanner
187    */
188    assertForward();
189    return forwardHeap.reseek(cell);
190  }
191
192  /**
193   * MemStoreScanner returns Long.MAX_VALUE because it will always have the latest data among all
194   * scanners.
195   * @see KeyValueScanner#getScannerOrder()
196   */
197  @Override
198  public long getScannerOrder() {
199    return Long.MAX_VALUE;
200  }
201
202  @Override
203  public synchronized void close() {
204
205    if (forwardHeap != null) {
206      assert ((type == Type.USER_SCAN_FORWARD) ||
207          (type == Type.COMPACT_FORWARD) || (type == Type.UNDEFINED));
208      forwardHeap.close();
209      forwardHeap = null;
210      if (backwardHeap != null) {
211        backwardHeap.close();
212        backwardHeap = null;
213      }
214    } else if (backwardHeap != null) {
215      assert (type == Type.USER_SCAN_BACKWARD);
216      backwardHeap.close();
217      backwardHeap = null;
218    }
219  }
220
221  /**
222   * Set the scanner at the seek key. Assumed backward scanning.
223   *
224   * @param cell seek value
225   * @return false if the key is null or if there is no data
226   */
227  @Override
228  public synchronized boolean backwardSeek(Cell cell) throws IOException {
229    initBackwardHeapIfNeeded(cell, false);
230    return backwardHeap.backwardSeek(cell);
231  }
232
233  /**
234   * Assumed backward scanning.
235   *
236   * @param cell seek value
237   * @return false if the key is null or if there is no data
238   */
239  @Override
240  public synchronized boolean seekToPreviousRow(Cell cell) throws IOException {
241    initBackwardHeapIfNeeded(cell, false);
242    if (backwardHeap.peek() == null) {
243      restartBackwardHeap(cell);
244    }
245    return backwardHeap.seekToPreviousRow(cell);
246  }
247
248  @Override
249  public synchronized boolean seekToLastRow() throws IOException {
250    // TODO: it looks like this is how it should be, however ReversedKeyValueHeap class doesn't
251    // implement seekToLastRow() method :(
252    // however seekToLastRow() was implemented in internal MemStoreScanner
253    // so I wonder whether we need to come with our own workaround, or to update
254    // ReversedKeyValueHeap
255    return initBackwardHeapIfNeeded(KeyValue.LOWESTKEY, true);
256  }
257
258  /**
259   * Check if this memstore may contain the required keys
260   * @return False if the key definitely does not exist in this Memstore
261   */
262  @Override
263  public synchronized boolean shouldUseScanner(Scan scan, Store store, long oldestUnexpiredTS) {
264
265    if (type == Type.COMPACT_FORWARD) {
266      return true;
267    }
268
269    for (KeyValueScanner sc : scanners) {
270      if (sc.shouldUseScanner(scan, store, oldestUnexpiredTS)) {
271        return true;
272      }
273    }
274    return false;
275  }
276
277  // debug method
278  @Override
279  public String toString() {
280    StringBuffer buf = new StringBuffer();
281    int i = 1;
282    for (KeyValueScanner scanner : scanners) {
283      buf.append("scanner (" + i + ") " + scanner.toString() + " ||| ");
284      i++;
285    }
286    return buf.toString();
287  }
288  /****************** Private methods ******************/
289  /**
290   * Restructure the ended backward heap after rerunning a seekToPreviousRow()
291   * on each scanner
292   * @return false if given Cell does not exist in any scanner
293   */
294  private boolean restartBackwardHeap(Cell cell) throws IOException {
295    boolean res = false;
296    for (KeyValueScanner scan : scanners) {
297      res |= scan.seekToPreviousRow(cell);
298    }
299    this.backwardHeap =
300        new ReversedKeyValueHeap(scanners, backwardReferenceToMemStore.getComparator());
301    return res;
302  }
303
304  /**
305   * Checks whether the type of the scan suits the assumption of moving backward
306   */
307  private boolean initBackwardHeapIfNeeded(Cell cell, boolean toLast) throws IOException {
308    boolean res = false;
309    if (toLast && (type != Type.UNDEFINED)) {
310      throw new IllegalStateException(
311          "Wrong usage of initBackwardHeapIfNeeded in parameters. The type is:" + type.toString());
312    }
313    if (type == Type.UNDEFINED) {
314      // In case we started from peek, release the forward heap
315      // and build backward. Set the correct type. Thus this turn
316      // can happen only once
317      if ((backwardHeap == null) && (forwardHeap != null)) {
318        forwardHeap.close();
319        forwardHeap = null;
320        // before building the heap seek for the relevant key on the scanners,
321        // for the heap to be built from the scanners correctly
322        for (KeyValueScanner scan : scanners) {
323          if (toLast) {
324            res |= scan.seekToLastRow();
325          } else {
326            res |= scan.backwardSeek(cell);
327          }
328        }
329        this.backwardHeap =
330            new ReversedKeyValueHeap(scanners, backwardReferenceToMemStore.getComparator());
331        type = Type.USER_SCAN_BACKWARD;
332      }
333    }
334
335    if (type == Type.USER_SCAN_FORWARD) {
336      throw new IllegalStateException("Traversing backward with forward scan");
337    }
338    return res;
339  }
340
341  /**
342   * Checks whether the type of the scan suits the assumption of moving forward
343   */
344  private void assertForward() throws IllegalStateException {
345    if (type == Type.UNDEFINED) {
346      type = Type.USER_SCAN_FORWARD;
347    }
348
349    if (type == Type.USER_SCAN_BACKWARD) {
350      throw new IllegalStateException("Traversing forward with backward scan");
351    }
352  }
