019package org.apache.hadoop.hbase.regionserver;
020
021import com.google.common.base.Preconditions;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.CellComparator;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.util.ReflectionUtils;
026
027import java.io.IOException;
028
029/**
030 * A singleton store segment factory.
031 * Generate concrete store segments.
032 */
033@InterfaceAudience.Private
034public final class SegmentFactory {
035
036  static final String USEMSLAB_KEY = "hbase.hregion.memstore.mslab.enabled";
037  static final boolean USEMSLAB_DEFAULT = true;
038  static final String MSLAB_CLASS_NAME = "hbase.regionserver.mslab.class";
039
040  private SegmentFactory() {}
041  private static SegmentFactory instance = new SegmentFactory();
042  public static SegmentFactory instance() {
043    return instance;
044  }
045
046  // create skip-list-based (non-flat) immutable segment from compacting old immutable segments
047  public ImmutableSegment createImmutableSegment(final Configuration conf,
048      final CellComparator comparator, MemStoreCompactorIterator iterator) {
049    MemStoreLAB memStoreLAB = getMemStoreLAB(conf);
050    return
051        new ImmutableSegment(comparator, iterator, memStoreLAB);
052  }
053
054  // create empty immutable segment
055  public ImmutableSegment createImmutableSegment(CellComparator comparator, long size) {
056    MutableSegment segment = generateMutableSegment(null, comparator, null, size);
057    return createImmutableSegment(segment);
058  }
059
060  // create immutable segment from mutable
061  public ImmutableSegment createImmutableSegment(MutableSegment segment) {
062    return new ImmutableSegment(segment);
063  }
064
065  // create mutable segment
066  public MutableSegment createMutableSegment(final Configuration conf,
067      CellComparator comparator, long size) {
068    MemStoreLAB memStoreLAB = getMemStoreLAB(conf);
069    return generateMutableSegment(conf, comparator, memStoreLAB, size);
070  }
071
072  // create new flat immutable segment from compacting old immutable segment
073  public ImmutableSegment createImmutableSegment(final Configuration conf, final CellComparator comparator,
074      MemStoreCompactorIterator iterator, int numOfCells, ImmutableSegment.Type segmentType)
075      throws IOException {
076    Preconditions.checkArgument(
077        segmentType != ImmutableSegment.Type.SKIPLIST_MAP_BASED, "wrong immutable segment type");
078    MemStoreLAB memStoreLAB = getMemStoreLAB(conf);
079    return
080        new ImmutableSegment(comparator, iterator, memStoreLAB, numOfCells, segmentType);
081  }
082
083  //****** private methods to instantiate concrete store segments **********//
084
085  private MutableSegment generateMutableSegment(
086      final Configuration conf, CellComparator comparator, MemStoreLAB memStoreLAB, long size) {
087    // TBD use configuration to set type of segment
088    CellSet set = new CellSet(comparator);
089    return new MutableSegment(set, comparator, memStoreLAB, size);
090  }
091
092  private MemStoreLAB getMemStoreLAB(Configuration conf) {
093    MemStoreLAB memStoreLAB = null;
094    if (conf.getBoolean(USEMSLAB_KEY, USEMSLAB_DEFAULT)) {
095      String className = conf.get(MSLAB_CLASS_NAME, HeapMemStoreLAB.class.getName());
096      memStoreLAB = ReflectionUtils.instantiateWithCustomCtor(className,
097          new Class[] { Configuration.class }, new Object[] { conf });
098    }
099    return memStoreLAB;
100  }
101
102}
