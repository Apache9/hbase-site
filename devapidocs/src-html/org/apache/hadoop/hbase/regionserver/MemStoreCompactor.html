019package org.apache.hadoop.hbase.regionserver;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.HConstants;
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026import org.apache.hadoop.hbase.client.Scan;
027
028import java.io.IOException;
029import java.util.ArrayList;
030import java.util.Collections;
031import java.util.List;
032import java.util.concurrent.atomic.AtomicBoolean;
033
034/**
035 * The ongoing MemStore Compaction manager, dispatches a solo running compaction
036 * and interrupts the compaction if requested.
037 * The MemStoreScanner is used to traverse the compaction pipeline. The MemStoreScanner
038 * is included in internal store scanner, where all compaction logic is implemented.
039 * Threads safety: It is assumed that the compaction pipeline is immutable,
040 * therefore no special synchronization is required.
041 */
042@InterfaceAudience.Private
043class MemStoreCompactor {
044
045  private static final Log LOG = LogFactory.getLog(MemStoreCompactor.class);
046  private CompactingMemStore compactingMemStore;
047  private MemStoreScanner scanner;            // scanner for pipeline only
048  // scanner on top of MemStoreScanner that uses ScanQueryMatcher
049  private StoreScanner compactingScanner;
050
051  // smallest read point for any ongoing MemStore scan
052  private long smallestReadPoint;
053
054  // a static version of the segment list from the pipeline
055  private VersionedSegmentsList versionedList;
056  private final AtomicBoolean isInterrupted = new AtomicBoolean(false);
057
058  public MemStoreCompactor(CompactingMemStore compactingMemStore) {
059    this.compactingMemStore = compactingMemStore;
060  }
061
062  /**
063   * The request to dispatch the compaction asynchronous task.
064   * The method returns true if compaction was successfully dispatched, or false if there
065   * is already an ongoing compaction or nothing to compact.
066   */
067  public boolean startCompaction() throws IOException {
068    if (!compactingMemStore.hasCompactibleSegments()) return false;  // no compaction on empty
069
070    List<SegmentScanner> scanners = new ArrayList<SegmentScanner>();
071    // get the list of segments from the pipeline
072    versionedList = compactingMemStore.getCompactibleSegments();
073    // the list is marked with specific version
074
075    // create the list of scanners with maximally possible read point, meaning that
076    // all KVs are going to be returned by the pipeline traversing
077    for (Segment segment : versionedList.getStoreSegments()) {
078      scanners.add(segment.getSegmentScanner(Long.MAX_VALUE));
079    }
080    scanner =
081        new MemStoreScanner(compactingMemStore, scanners, Long.MAX_VALUE,
082            MemStoreScanner.Type.COMPACT_FORWARD);
083
084    smallestReadPoint = compactingMemStore.getSmallestReadPoint();
085    compactingScanner = createScanner(compactingMemStore.getStore());
086
087    LOG.info("Starting the MemStore in-memory compaction for store " +
088        compactingMemStore.getStore().getColumnFamilyName());
089
090    doCompaction();
091    return true;
092  }
093
094  /**
095  * The request to cancel the compaction asynchronous task
096  * The compaction may still happen if the request was sent too late
097  * Non-blocking request
098  */
099  public void stopCompact() {
100    isInterrupted.compareAndSet(false, true);
101  }
102
103
104  /**
105  * Close the scanners and clear the pointers in order to allow good
106  * garbage collection
107  */
108  private void releaseResources() {
109    isInterrupted.set(false);
110    scanner.close();
111    scanner = null;
112    compactingScanner.close();
113    compactingScanner = null;
114    versionedList = null;
115  }
116
117  /**
118  * The worker thread performs the compaction asynchronously.
119  * The solo (per compactor) thread only reads the compaction pipeline.
120  * There is at most one thread per memstore instance.
121  */
122  private void doCompaction() {
123
124    ImmutableSegment result = SegmentFactory.instance()  // create the scanner
125        .createImmutableSegment(
126            compactingMemStore.getConfiguration(), compactingMemStore.getComparator(),
127            CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM);
128
129    // the compaction processing
130    try {
131      // Phase I: create the compacted MutableCellSetSegment
132      compactSegments(result);
133
134      // Phase II: swap the old compaction pipeline
135      if (!isInterrupted.get()) {
136        compactingMemStore.swapCompactedSegments(versionedList, result);
137        // update the wal so it can be truncated and not get too long
138        compactingMemStore.updateLowestUnflushedSequenceIdInWAL(true); // only if greater
139      }
140    } catch (Exception e) {
141      LOG.debug("Interrupting the MemStore in-memory compaction for store " + compactingMemStore
142          .getFamilyName());
143      Thread.currentThread().interrupt();
144      return;
145    } finally {
146      releaseResources();
147      compactingMemStore.setInMemoryFlushInProgress(false);
148    }
149
150  }
151
152  /**
153   * Creates the scanner for compacting the pipeline.
154   *
155   * @return the scanner
156   */
157  private StoreScanner createScanner(Store store) throws IOException {
158
159    Scan scan = new Scan();
160    scan.setMaxVersions();  //Get all available versions
161
162    StoreScanner internalScanner =
163        new StoreScanner(store, store.getScanInfo(), scan, Collections.singletonList(scanner),
164            ScanType.COMPACT_RETAIN_DELETES, smallestReadPoint, HConstants.OLDEST_TIMESTAMP);
165
166    return internalScanner;
167  }
168
169  /**
170   * Updates the given single Segment using the internal store scanner,
171   * who in turn uses ScanQueryMatcher
172   */
173  private void compactSegments(Segment result) throws IOException {
174
175    List<Cell> kvs = new ArrayList<Cell>();
176    // get the limit to the size of the groups to be returned by compactingScanner
177    int compactionKVMax = compactingMemStore.getConfiguration().getInt(
178        HConstants.COMPACTION_KV_MAX,
179        HConstants.COMPACTION_KV_MAX_DEFAULT);
180
181    ScannerContext scannerContext =
182        ScannerContext.newBuilder().setBatchLimit(compactionKVMax).build();
183
184    boolean hasMore;
185    do {
186      hasMore = compactingScanner.next(kvs, scannerContext);
187      if (!kvs.isEmpty()) {
188        for (Cell c : kvs) {
189          // The scanner is doing all the elimination logic
190          // now we just copy it to the new segment
191          Cell newKV = result.maybeCloneWithAllocator(c);
192          result.internalAdd(newKV);
193
194        }
195        kvs.clear();
196      }
197    } while (hasMore && (!isInterrupted.get()));
198  }
199}
