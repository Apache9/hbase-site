020package org.apache.hadoop.hbase.regionserver.wal;
021
022import java.io.IOException;
023import java.util.NavigableMap;
024
025import org.apache.hadoop.hbase.client.RegionInfo;
026import org.apache.hadoop.hbase.regionserver.MultiVersionConcurrencyControl;
027import org.apache.hadoop.hbase.wal.WAL;
028import org.apache.hadoop.hbase.wal.WALEdit;
029import org.apache.hadoop.hbase.wal.WALKeyImpl;
030import org.apache.yetus.audience.InterfaceAudience;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033import org.apache.hbase.thirdparty.com.google.protobuf.TextFormat;
034import org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos;
035import org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.CompactionDescriptor;
036import org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.FlushDescriptor;
037import org.apache.hadoop.hbase.shaded.protobuf.generated.WALProtos.RegionEventDescriptor;
038
039/**
040 * Helper methods to ease Region Server integration with the Write Ahead Log (WAL).
041 * Note that methods in this class specifically should not require access to anything
042 * other than the API found in {@link WAL}. For internal use only.
043 */
044@InterfaceAudience.Private
045public class WALUtil {
046  private static final Logger LOG = LoggerFactory.getLogger(WALUtil.class);
047
048  private WALUtil() {
049    // Shut down construction of this class.
050  }
051
052  /**
053   * Write the marker that a compaction has succeeded and is about to be committed.
054   * This provides info to the HMaster to allow it to recover the compaction if this regionserver
055   * dies in the middle. It also prevents the compaction from finishing if this regionserver has
056   * already lost its lease on the log.
057   *
058   * <p>This write is for internal use only. Not for external client consumption.
059   * @param mvcc Used by WAL to get sequence Id for the waledit.
060   */
061  public static WALKeyImpl writeCompactionMarker(WAL wal,
062      NavigableMap<byte[], Integer> replicationScope, RegionInfo hri, final CompactionDescriptor c,
063      MultiVersionConcurrencyControl mvcc)
064  throws IOException {
065    WALKeyImpl walKey =
066        writeMarker(wal, replicationScope, hri, WALEdit.createCompaction(hri, c), mvcc);
067    if (LOG.isTraceEnabled()) {
068      LOG.trace("Appended compaction marker " + TextFormat.shortDebugString(c));
069    }
070    return walKey;
071  }
072
073  /**
074   * Write a flush marker indicating a start / abort or a complete of a region flush
075   *
076   * <p>This write is for internal use only. Not for external client consumption.
077   */
078  public static WALKeyImpl writeFlushMarker(WAL wal, NavigableMap<byte[], Integer> replicationScope,
079      RegionInfo hri, final FlushDescriptor f, boolean sync, MultiVersionConcurrencyControl mvcc)
080          throws IOException {
081    WALKeyImpl walKey = doFullAppendTransaction(wal, replicationScope, hri,
082        WALEdit.createFlushWALEdit(hri, f), mvcc, sync);
083    if (LOG.isTraceEnabled()) {
084      LOG.trace("Appended flush marker " + TextFormat.shortDebugString(f));
085    }
086    return walKey;
087  }
088
089  /**
090   * Write a region open marker indicating that the region is opened.
091   * This write is for internal use only. Not for external client consumption.
092   */
093  public static WALKeyImpl writeRegionEventMarker(WAL wal,
094      NavigableMap<byte[], Integer> replicationScope, RegionInfo hri,
095      final RegionEventDescriptor r, final MultiVersionConcurrencyControl mvcc)
096  throws IOException {
097    WALKeyImpl walKey = writeMarker(wal, replicationScope, hri,
098        WALEdit.createRegionEventWALEdit(hri, r), mvcc);
099    if (LOG.isTraceEnabled()) {
100      LOG.trace("Appended region event marker " + TextFormat.shortDebugString(r));
101    }
102    return walKey;
103  }
104
105  /**
106   * Write a log marker that a bulk load has succeeded and is about to be committed.
107   * This write is for internal use only. Not for external client consumption.
108   * @param wal The log to write into.
109   * @param replicationScope The replication scope of the families in the HRegion
110   * @param hri A description of the region in the table that we are bulk loading into.
111   * @param desc A protocol buffers based description of the client's bulk loading request
112   * @return walKey with sequenceid filled out for this bulk load marker
113   * @throws IOException We will throw an IOException if we can not append to the HLog.
114   */
115  public static WALKeyImpl writeBulkLoadMarkerAndSync(final WAL wal,
116      final NavigableMap<byte[], Integer> replicationScope, final RegionInfo hri,
117      final WALProtos.BulkLoadDescriptor desc, final MultiVersionConcurrencyControl mvcc)
118          throws IOException {
119    WALKeyImpl walKey =
120        writeMarker(wal, replicationScope, hri, WALEdit.createBulkLoadEvent(hri, desc), mvcc);
121    if (LOG.isTraceEnabled()) {
122      LOG.trace("Appended Bulk Load marker " + TextFormat.shortDebugString(desc));
123    }
124    return walKey;
125  }
126
127  private static WALKeyImpl writeMarker(final WAL wal,
128      final NavigableMap<byte[], Integer> replicationScope, final RegionInfo hri,
129      final WALEdit edit, final MultiVersionConcurrencyControl mvcc)
130  throws IOException {
131    // If sync == true in below, then timeout is not used; safe to pass UNSPECIFIED_TIMEOUT
132    return doFullAppendTransaction(wal, replicationScope, hri, edit, mvcc, true);
133  }
134
135  /**
136   * A 'full' WAL transaction involves starting an mvcc transaction followed by an append,
137   * an optional sync, and then a call to complete the mvcc transaction. This method does it all.
138   * Good for case of adding a single edit or marker to the WAL.
139   *
140   * <p>This write is for internal use only. Not for external client consumption.
141   * @return WALKeyImpl that was added to the WAL.
142   */
143  public static WALKeyImpl doFullAppendTransaction(final WAL wal,
144      final NavigableMap<byte[], Integer> replicationScope, final RegionInfo hri,
145      final WALEdit edit, final MultiVersionConcurrencyControl mvcc, final boolean sync)
146  throws IOException {
147    // TODO: Pass in current time to use?
148    WALKeyImpl walKey = new WALKeyImpl(hri.getEncodedNameAsBytes(), hri.getTable(),
149        System.currentTimeMillis(), mvcc, replicationScope);
150    long trx = MultiVersionConcurrencyControl.NONE;
151    try {
152      trx = wal.append(hri, walKey, edit, false);
153      if (sync) {
154        wal.sync(trx);
155      }
156      // Call complete only here because these are markers only. They are not for clients to read.
157      mvcc.complete(walKey.getWriteEntry());
158    } catch (IOException ioe) {
159      if (walKey.getWriteEntry() != null) {
160        mvcc.complete(walKey.getWriteEntry());
161      }
162      throw ioe;
163    }
164    return walKey;
165  }
166}
