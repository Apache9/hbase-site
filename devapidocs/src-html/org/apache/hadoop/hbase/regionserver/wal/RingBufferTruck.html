019package org.apache.hadoop.hbase.regionserver.wal;
020
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022import org.apache.htrace.Span;
023
024import com.lmax.disruptor.EventFactory;
025
026/**
027 * A 'truck' to carry a payload across the {@link FSHLog} ring buffer from Handler to WAL.
028 * Has EITHER a {@link FSWALEntry} for making an append OR it has a {@link SyncFuture} to
029 * represent a 'sync' invocation. Truck instances are reused by the disruptor when it gets
030 * around to it so their payload references must be discarded on consumption to release them
031 * to GC.
032 */
033@InterfaceAudience.Private
034class RingBufferTruck {
035  /**
036   * Either this syncFuture is set or entry is set, but not both.
037   */
038  private SyncFuture syncFuture;
039  private FSWALEntry entry;
040
041  /**
042   * The tracing span for this entry.  Can be null.
043   * TODO: Fix up tracing.
044   */
045  private Span span;
046
047  /**
048   * Load the truck with a {@link FSWALEntry} and associated {@link Span}.
049   */
050  void loadPayload(final FSWALEntry entry, final Span span) {
051    this.entry = entry;
052    this.span = span;
053    this.syncFuture = null;
054  }
055
056  /**
057   * Load the truck with a {@link SyncFuture}.
058   */
059  void loadPayload(final SyncFuture syncFuture) {
060    this.syncFuture = syncFuture;
061    this.entry = null;
062    this.span = null;
063  }
064
065  /**
066   * return {@code true} when this truck is carrying a {@link FSWALEntry},
067   * {@code false} otherwise.
068   */
069  boolean hasFSWALEntryPayload() {
070    return this.entry != null;
071  }
072
073  /**
074   * return {@code true} when this truck is carrying a {@link SyncFuture},
075   * {@code false} otherwise.
076   */
077  boolean hasSyncFuturePayload() {
078    return this.syncFuture != null;
079  }
080
081  /**
082   * Unload the truck of its {@link FSWALEntry} payload. The internal refernce is released.
083   */
084  FSWALEntry unloadFSWALEntryPayload() {
085    FSWALEntry ret = this.entry;
086    this.entry = null;
087    return ret;
088  }
089
090  /**
091   * Unload the truck of its {@link SyncFuture} payload. The internal refernce is released.
092   */
093  SyncFuture unloadSyncFuturePayload() {
094    SyncFuture ret = this.syncFuture;
095    this.syncFuture = null;
096    return ret;
097  }
098
099  /**
100   * Unload the truck of its {@link Span} payload. The internal reference is released.
101   */
102  Span unloadSpanPayload() {
103    Span ret = this.span;
104    this.span = null;
105    return ret;
106  }
107
108  /**
109   * Factory for making a bunch of these.  Needed by the ringbuffer/disruptor.
110   */
111  final static EventFactory<RingBufferTruck> EVENT_FACTORY = new EventFactory<RingBufferTruck>() {
112    public RingBufferTruck newInstance() {
113      return new RingBufferTruck();
114    }
115  };
116}
