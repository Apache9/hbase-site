018package org.apache.hadoop.hbase.regionserver.wal;
019
020import java.util.concurrent.ExecutionException;
021import java.util.concurrent.TimeUnit;
022
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.htrace.Span;
025
026/**
027 * A Future on a filesystem sync call.  It given to a client or 'Handler' for it to wait on till
028 * the sync completes.
029 *
030 * <p>Handlers coming in call append, append, append, and then do a flush/sync of
031 * the edits they have appended the WAL before returning. Since sync takes a while to
032 * complete, we give the Handlers back this sync future to wait on until the
033 * actual HDFS sync completes. Meantime this sync future goes across the ringbuffer and into a
034 * sync runner thread; when it completes, it finishes up the future, the handler get or failed
035 * check completes and the Handler can then progress.
036 * <p>
037 * This is just a partial implementation of Future; we just implement get and
038 * failure.  Unimplemented methods throw {@link UnsupportedOperationException}.
039 * <p>
040 * There is not a one-to-one correlation between dfs sync invocations and
041 * instances of this class. A single dfs sync call may complete and mark many
042 * SyncFutures as done; i.e. we batch up sync calls rather than do a dfs sync
043 * call every time a Handler asks for it.
044 * <p>
045 * SyncFutures are immutable but recycled. Call #reset(long, Span) before use even
046 * if it the first time, start the sync, then park the 'hitched' thread on a call to
047 * #get().
048 */
049@InterfaceAudience.Private
050class SyncFuture {
051  // Implementation notes: I tried using a cyclicbarrier in here for handler and sync threads
052  // to coordinate on but it did not give any obvious advantage and some issues with order in which
053  // events happen.
054  private static final long NOT_DONE = 0;
055
056  /**
057   * The sequence at which we were added to the ring buffer.
058   */
059  private long ringBufferSequence;
060
061  /**
062   * The sequence that was set in here when we were marked done. Should be equal
063   * or > ringBufferSequence.  Put this data member into the NOT_DONE state while this
064   * class is in use.  But for the first position on construction, let it be -1 so we can
065   * immediately call {@link #reset(long, Span)} below and it will work.
066   */
067  private long doneSequence = -1;
068
069  /**
070   * If error, the associated throwable. Set when the future is 'done'.
071   */
072  private Throwable throwable = null;
073
074  private Thread t;
075
076  /**
077   * Optionally carry a disconnected scope to the SyncRunner.
078   */
079  private Span span;
080
081  /**
082   * Call this method to clear old usage and get it ready for new deploy. Call
083   * this method even if it is being used for the first time.
084   *
085   * @param sequence sequenceId from this Future's position in the RingBuffer
086   * @return this
087   */
088  synchronized SyncFuture reset(final long sequence) {
089    return reset(sequence, null);
090  }
091
092  /**
093   * Call this method to clear old usage and get it ready for new deploy. Call
094   * this method even if it is being used for the first time.
095   *
096   * @param sequence sequenceId from this Future's position in the RingBuffer
097   * @param span curren span, detached from caller. Don't forget to attach it when
098   *             resuming after a call to {@link #get()}.
099   * @return this
100   */
101  synchronized SyncFuture reset(final long sequence, Span span) {
102    if (t != null && t != Thread.currentThread()) throw new IllegalStateException();
103    t = Thread.currentThread();
104    if (!isDone()) throw new IllegalStateException("" + sequence + " " + Thread.currentThread());
105    this.doneSequence = NOT_DONE;
106    this.ringBufferSequence = sequence;
107    this.span = span;
108    return this;
109  }
110
111  @Override
112  public synchronized String toString() {
113    return "done=" + isDone() + ", ringBufferSequence=" + this.ringBufferSequence;
114  }
115
116  synchronized long getRingBufferSequence() {
117    return this.ringBufferSequence;
118  }
119
120  /**
121   * Retrieve the {@code span} instance from this Future. EventHandler calls
122   * this method to continue the span. Thread waiting on this Future musn't call
123   * this method until AFTER calling {@link #get()} and the future has been
124   * released back to the originating thread.
125   */
126  synchronized Span getSpan() {
127    return this.span;
128  }
129
130  /**
131   * Used to re-attach a {@code span} to the Future. Called by the EventHandler
132   * after a it has completed processing and detached the span from its scope.
133   */
134  synchronized void setSpan(Span span) {
135    this.span = span;
136  }
137
138  /**
139   * @param sequence Sync sequence at which this future 'completed'.
140   * @param t Can be null.  Set if we are 'completing' on error (and this 't' is the error).
141   * @return True if we successfully marked this outstanding future as completed/done.
142   * Returns false if this future is already 'done' when this method called.
143   */
144  synchronized boolean done(final long sequence, final Throwable t) {
145    if (isDone()) return false;
146    this.throwable = t;
147    if (sequence < this.ringBufferSequence) {
148      // Something badly wrong.
149      if (throwable == null) {
150        this.throwable = new IllegalStateException("sequence=" + sequence +
151          ", ringBufferSequence=" + this.ringBufferSequence);
152      }
153    }
154    // Mark done.
155    this.doneSequence = sequence;
156    // Wake up waiting threads.
157    notify();
158    return true;
159  }
160
161  public boolean cancel(boolean mayInterruptIfRunning) {
162    throw new UnsupportedOperationException();
163  }
164
165  public synchronized long get() throws InterruptedException, ExecutionException {
166    while (!isDone()) {
167      wait(1000);
168    }
169    if (this.throwable != null) throw new ExecutionException(this.throwable);
170    return this.doneSequence;
171  }
172
173  public Long get(long timeout, TimeUnit unit)
174  throws InterruptedException, ExecutionException {
175    throw new UnsupportedOperationException();
176  }
177
178  public boolean isCancelled() {
179    throw new UnsupportedOperationException();
180  }
181
182  synchronized boolean isDone() {
183    return this.doneSequence != NOT_DONE;
184  }
185
186  synchronized boolean isThrowable() {
187    return isDone() && getThrowable() != null;
188  }
189
190  synchronized Throwable getThrowable() {
191    return this.throwable;
192  }
193}
