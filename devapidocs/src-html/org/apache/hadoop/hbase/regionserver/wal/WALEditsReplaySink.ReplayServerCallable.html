052/**
053 * This class is responsible for replaying the edits coming from a failed region server.
054 * <p>
055 * This class uses the native HBase client in order to replay WAL entries.
056 * </p>
057 */
058@InterfaceAudience.Private
059public class WALEditsReplaySink {
060  private static final Log LOG = LogFactory.getLog(WALEditsReplaySink.class);
061  private static final int MAX_BATCH_SIZE = 1024;
062  private final Configuration conf;
063  private final ClusterConnection conn;
064  private final TableName tableName;
065  private final MetricsWALEditsReplay metrics;
066  private final AtomicLong totalReplayedEdits = new AtomicLong();
067  private final boolean skipErrors;
068  private final int replayTimeout;
069  private final RpcControllerFactory rpcControllerFactory;
070
071  /**
072   * Create a sink for WAL log entries replay
073   * @param conf configuration
074   * @param tableName of table to replay edits of
075   * @param conn connection to use
076   * @throws IOException on IO failure
077   */
078  public WALEditsReplaySink(Configuration conf, TableName tableName, ClusterConnection conn)
079      throws IOException {
080    this.conf = conf;
081    this.metrics = new MetricsWALEditsReplay();
082    this.conn = conn;
083    this.tableName = tableName;
084    this.skipErrors = conf.getBoolean(HConstants.HREGION_EDITS_REPLAY_SKIP_ERRORS,
085      HConstants.DEFAULT_HREGION_EDITS_REPLAY_SKIP_ERRORS);
086    // a single replay operation time out and default is 60 seconds
087    this.replayTimeout = conf.getInt("hbase.regionserver.logreplay.timeout", 60000);
088    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
089  }
090
091  /**
092   * Replay an array of actions of the same region directly into the newly assigned Region Server
093   * @param entries to replay
094   * @throws IOException on IO failure
095   */
096  public void replayEntries(List<Pair<HRegionLocation, Entry>> entries) throws IOException {
097    if (entries.isEmpty()) {
098      return;
099    }
100
101    int batchSize = entries.size();
102    Map<HRegionInfo, List<Entry>> entriesByRegion =
103        new HashMap<>();
104    HRegionLocation loc = null;
105    Entry entry = null;
106    List<Entry> regionEntries = null;
107    // Build the action list.
108    for (int i = 0; i < batchSize; i++) {
109      loc = entries.get(i).getFirst();
110      entry = entries.get(i).getSecond();
111      if (entriesByRegion.containsKey(loc.getRegionInfo())) {
112        regionEntries = entriesByRegion.get(loc.getRegionInfo());
113      } else {
114        regionEntries = new ArrayList<Entry>();
115        entriesByRegion.put(loc.getRegionInfo(), regionEntries);
116      }
117      regionEntries.add(entry);
118    }
119
120    long startTime = EnvironmentEdgeManager.currentTime();
121
122    // replaying edits by region
123    for (Map.Entry<HRegionInfo, List<Entry>> _entry : entriesByRegion.entrySet()) {
124      HRegionInfo curRegion = _entry.getKey();
125      List<Entry> allActions = _entry.getValue();
126      // send edits in chunks
127      int totalActions = allActions.size();
128      int replayedActions = 0;
129      int curBatchSize = 0;
130      for (; replayedActions < totalActions;) {
131        curBatchSize = (totalActions > (MAX_BATCH_SIZE + replayedActions)) ? MAX_BATCH_SIZE
132                : (totalActions - replayedActions);
133        replayEdits(loc, curRegion, allActions.subList(replayedActions,
134          replayedActions + curBatchSize));
135        replayedActions += curBatchSize;
136      }
137    }
138
139    long endTime = EnvironmentEdgeManager.currentTime() - startTime;
140    LOG.debug("number of rows:" + entries.size() + " are sent by batch! spent " + endTime
141        + "(ms)!");
142
143    metrics.updateReplayTime(endTime);
144    metrics.updateReplayBatchSize(batchSize);
145
146    this.totalReplayedEdits.addAndGet(batchSize);
147  }
148
149  /**
150   * Get a string representation of this sink's metrics
151   * @return string with the total replayed edits count
152   */
153  public String getStats() {
154    return this.totalReplayedEdits.get() == 0 ? "" : "Sink: total replayed edits: "
155        + this.totalReplayedEdits;
156  }
157
158  private void replayEdits(final HRegionLocation regionLoc, final HRegionInfo regionInfo,
159      final List<Entry> entries) throws IOException {
160    try {
161      RpcRetryingCallerFactory factory = RpcRetryingCallerFactory.instantiate(conf, null);
162      ReplayServerCallable<ReplicateWALEntryResponse> callable =
163          new ReplayServerCallable<ReplicateWALEntryResponse>(this.conn, this.rpcControllerFactory,
164              this.tableName, regionLoc, entries);
165      factory.<ReplicateWALEntryResponse> newCaller().callWithRetries(callable, this.replayTimeout);
166    } catch (IOException ie) {
167      if (skipErrors) {
168        LOG.warn(HConstants.HREGION_EDITS_REPLAY_SKIP_ERRORS
169            + "=true so continuing replayEdits with error:" + ie.getMessage());
170      } else {
171        throw ie;
172      }
173    }
174  }
175
176  /**
177   * Callable that handles the <code>replay</code> method call going against a single regionserver
178   * @param <R>
179   */
180  class ReplayServerCallable<R> extends ClientServiceCallable<ReplicateWALEntryResponse> {
181    private List<Entry> entries;
182
183    ReplayServerCallable(final Connection connection, RpcControllerFactory rpcControllerFactory,
184        final TableName tableName, final HRegionLocation regionLoc, final List<Entry> entries) {
185      super(connection, tableName, HConstants.EMPTY_BYTE_ARRAY,
186          rpcControllerFactory.newController());
187      this.entries = entries;
188      setLocation(regionLoc);
189    }
190
191    @Override
192    protected ReplicateWALEntryResponse rpcCall() throws Exception {
193      if (entries.isEmpty()) return null;
194
195      Entry[] entriesArray = new Entry[entries.size()];
196      entriesArray = entries.toArray(entriesArray);
197      AdminService.BlockingInterface remoteSvr = conn.getAdmin(getLocation().getServerName());
198
199      Pair<AdminProtos.ReplicateWALEntryRequest, CellScanner> p =
200          ReplicationProtbufUtil.buildReplicateWALEntryRequest(entriesArray);
201      setRpcControllerCellScanner(p.getSecond());
202      return remoteSvr.replay(getRpcController(), p.getFirst());
203    }
204
205    @Override
206    public void prepare(boolean reload) throws IOException {
207      if (!reload) return;
208      // relocate regions in case we have a new dead server or network hiccup
209      // if not due to connection issue, the following code should run fast because it uses
210      // cached location
211      boolean skip = false;
212      for (Entry entry : this.entries) {
213        WALEdit edit = entry.getEdit();
214        List<Cell> cells = edit.getCells();
215        for (Cell cell : cells) {
216          // filtering WAL meta entries
217          setLocation(conn.locateRegion(tableName, CellUtil.cloneRow(cell)));
218          skip = true;
219          break;
220        }
221        // use first log entry to relocate region because all entries are for one region
222        if (skip) break;
223      }
224    }
225  }
