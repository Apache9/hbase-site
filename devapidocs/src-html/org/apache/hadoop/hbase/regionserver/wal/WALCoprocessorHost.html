040/**
041 * Implements the coprocessor environment and runtime support for coprocessors
042 * loaded within a {@link WAL}.
043 */
044@InterfaceAudience.Private
045public class WALCoprocessorHost
046    extends CoprocessorHost<WALCoprocessorHost.WALEnvironment> {
047
048  /**
049   * Encapsulation of the environment of each coprocessor
050   */
051  static class WALEnvironment extends CoprocessorHost.Environment
052    implements WALCoprocessorEnvironment {
053
054    private final WAL wal;
055
056    private final MetricRegistry metricRegistry;
057
058    @Override
059    public WAL getWAL() {
060      return wal;
061    }
062
063    /**
064     * Constructor
065     * @param implClass - not used
066     * @param impl the coprocessor instance
067     * @param priority chaining priority
068     * @param seq load sequence
069     * @param conf configuration
070     * @param wal WAL
071     */
072    public WALEnvironment(Class<?> implClass, final Coprocessor impl,
073        final int priority, final int seq, final Configuration conf,
074        final WAL wal) {
075      super(impl, priority, seq, conf);
076      this.wal = wal;
077      this.metricRegistry = MetricsCoprocessor.createRegistryForWALCoprocessor(implClass.getName());
078    }
079
080    @Override
081    public MetricRegistry getMetricRegistryForRegionServer() {
082      return metricRegistry;
083    }
084
085    @Override
086    protected void shutdown() {
087      super.shutdown();
088      MetricsCoprocessor.removeRegistry(this.metricRegistry);
089    }
090  }
091
092  private final WAL wal;
093
094  /**
095   * Constructor
096   * @param log the write ahead log
097   * @param conf the configuration
098   */
099  public WALCoprocessorHost(final WAL log, final Configuration conf) {
100    // We don't want to require an Abortable passed down through (FS)HLog, so
101    // this means that a failure to load of a WAL coprocessor won't abort the
102    // server. This isn't ideal, and means that security components that
103    // utilize a WALObserver will have to check the observer initialization
104    // state manually. However, WALObservers will eventually go away so it
105    // should be an acceptable state of affairs.
106    super(null);
107    this.wal = log;
108    // load system default cp's from configuration.
109    loadSystemCoprocessors(conf, WAL_COPROCESSOR_CONF_KEY);
110  }
111
112  @Override
113  public WALEnvironment createEnvironment(final Class<?> implClass,
114      final Coprocessor instance, final int priority, final int seq,
115      final Configuration conf) {
116    return new WALEnvironment(implClass, instance, priority, seq, conf,
117        this.wal);
118  }
119
120  /**
121   * @param info
122   * @param logKey
123   * @param logEdit
124   * @return true if default behavior should be bypassed, false otherwise
125   * @throws IOException
126   */
127  public boolean preWALWrite(final HRegionInfo info, final WALKey logKey, final WALEdit logEdit)
128      throws IOException {
129    boolean bypass = false;
130    if (this.coprocessors == null || this.coprocessors.isEmpty()) return bypass;
131    ObserverContext<WALCoprocessorEnvironment> ctx = null;
132    List<WALEnvironment> envs = coprocessors.get();
133    for (int i = 0; i < envs.size(); i++) {
134      WALEnvironment env = envs.get(i);
135      if (env.getInstance() instanceof WALObserver) {
136        final WALObserver observer = (WALObserver)env.getInstance();
137        ctx = ObserverContext.createAndPrepare(env, ctx);
138        Thread currentThread = Thread.currentThread();
139        ClassLoader cl = currentThread.getContextClassLoader();
140        try {
141          currentThread.setContextClassLoader(env.getClassLoader());
142          observer.preWALWrite(ctx, info, logKey, logEdit);
143        } catch (Throwable e) {
144          handleCoprocessorThrowable(env, e);
145        } finally {
146          currentThread.setContextClassLoader(cl);
147        }
148        bypass |= ctx.shouldBypass();
149        if (ctx.shouldComplete()) {
150          break;
151        }
152      }
153    }
154    return bypass;
155  }
156

159   * @param logKey

161   * @throws IOException
162   */

223   * @param oldPath the path of the wal that we replaced
