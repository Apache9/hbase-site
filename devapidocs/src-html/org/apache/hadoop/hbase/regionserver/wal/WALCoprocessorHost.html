021package org.apache.hadoop.hbase.regionserver.wal;
022
023import java.io.IOException;
024import java.util.List;
025
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.fs.Path;
028import org.apache.hadoop.hbase.Coprocessor;
029import org.apache.hadoop.hbase.HRegionInfo;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.coprocessor.CoprocessorHost;
032import org.apache.hadoop.hbase.coprocessor.ObserverContext;
033import org.apache.hadoop.hbase.coprocessor.WALCoprocessorEnvironment;
034import org.apache.hadoop.hbase.coprocessor.WALObserver;
035import org.apache.hadoop.hbase.wal.WAL;
036import org.apache.hadoop.hbase.wal.WALKey;

038/**
039 * Implements the coprocessor environment and runtime support for coprocessors
040 * loaded within a {@link WAL}.
041 */
042@InterfaceAudience.Private
043public class WALCoprocessorHost
044    extends CoprocessorHost<WALCoprocessorHost.WALEnvironment> {
045
046  /**
047   * Encapsulation of the environment of each coprocessor
048   */
049  static class WALEnvironment extends CoprocessorHost.Environment
050    implements WALCoprocessorEnvironment {
051
052    private final WAL wal;
053
054    @Override
055    public WAL getWAL() {
056      return wal;
057    }
058
059    /**
060     * Constructor
061     * @param implClass - not used
062     * @param impl the coprocessor instance
063     * @param priority chaining priority
064     * @param seq load sequence
065     * @param conf configuration
066     * @param wal WAL
067     */
068    public WALEnvironment(Class<?> implClass, final Coprocessor impl,
069        final int priority, final int seq, final Configuration conf,
070        final WAL wal) {
071      super(impl, priority, seq, conf);
072      this.wal = wal;
073    }
074  }
075
076  private final WAL wal;
077
078  /**
079   * Constructor
080   * @param log the write ahead log
081   * @param conf the configuration
082   */
083  public WALCoprocessorHost(final WAL log, final Configuration conf) {
084    // We don't want to require an Abortable passed down through (FS)HLog, so
085    // this means that a failure to load of a WAL coprocessor won't abort the
086    // server. This isn't ideal, and means that security components that
087    // utilize a WALObserver will have to check the observer initialization
088    // state manually. However, WALObservers will eventually go away so it
089    // should be an acceptable state of affairs.
090    super(null);
091    this.wal = log;
092    // load system default cp's from configuration.
093    loadSystemCoprocessors(conf, WAL_COPROCESSOR_CONF_KEY);
094  }
095
096  @Override
097  public WALEnvironment createEnvironment(final Class<?> implClass,
098      final Coprocessor instance, final int priority, final int seq,
099      final Configuration conf) {
100    return new WALEnvironment(implClass, instance, priority, seq, conf,
101        this.wal);
102  }
103
104  /**
105   * @param info
106   * @param logKey
107   * @param logEdit
108   * @return true if default behavior should be bypassed, false otherwise
109   * @throws IOException
110   */
111  public boolean preWALWrite(final HRegionInfo info, final WALKey logKey, final WALEdit logEdit)
112      throws IOException {
113    boolean bypass = false;
114    if (this.coprocessors == null || this.coprocessors.isEmpty()) return bypass;
115    ObserverContext<WALCoprocessorEnvironment> ctx = null;
116    List<WALEnvironment> envs = coprocessors.get();
117    for (int i = 0; i < envs.size(); i++) {
118      WALEnvironment env = envs.get(i);
119      if (env.getInstance() instanceof WALObserver) {
120        final WALObserver observer = (WALObserver)env.getInstance();
121        ctx = ObserverContext.createAndPrepare(env, ctx);
122        Thread currentThread = Thread.currentThread();
123        ClassLoader cl = currentThread.getContextClassLoader();
124        try {
125          currentThread.setContextClassLoader(env.getClassLoader());
126          observer.preWALWrite(ctx, info, logKey, logEdit);
127        } catch (Throwable e) {
128          handleCoprocessorThrowable(env, e);
129        } finally {
130          currentThread.setContextClassLoader(cl);
131        }
132        bypass |= ctx.shouldBypass();
133        if (ctx.shouldComplete()) {
134          break;
135        }
136      }
137    }
138    return bypass;
139  }
140
141  /**
142   * @param info
143   * @param logKey
144   * @param logEdit
145   * @throws IOException
146   */
147  public void postWALWrite(final HRegionInfo info, final WALKey logKey, final WALEdit logEdit)
148      throws IOException {
149    if (this.coprocessors == null || this.coprocessors.isEmpty()) return;
150    ObserverContext<WALCoprocessorEnvironment> ctx = null;
151    List<WALEnvironment> envs = coprocessors.get();
152    for (int i = 0; i < envs.size(); i++) {
153      WALEnvironment env = envs.get(i);
154      if (env.getInstance() instanceof WALObserver) {
155        final WALObserver observer = (WALObserver)env.getInstance();
156        ctx = ObserverContext.createAndPrepare(env, ctx);
157        Thread currentThread = Thread.currentThread();
158        ClassLoader cl = currentThread.getContextClassLoader();
159        try {
160          currentThread.setContextClassLoader(env.getClassLoader());
161          observer.postWALWrite(ctx, info, logKey, logEdit);
162        } catch (Throwable e) {
163          handleCoprocessorThrowable(env, e);
164        } finally {
165          currentThread.setContextClassLoader(cl);
166        }
167        if (ctx.shouldComplete()) {
168          break;
169        }
170      }
171    }
172  }

174  /**
175   * Called before rolling the current WAL
176   * @param oldPath the path of the current wal that we are replacing
177   * @param newPath the path of the wal we are going to create
178   */

203  }

205  /**
206   * Called after rolling the current WAL
207   * @param oldPath the path of the wal that we replaced
208   * @param newPath the path of the wal we have created and now is the current
209   */
