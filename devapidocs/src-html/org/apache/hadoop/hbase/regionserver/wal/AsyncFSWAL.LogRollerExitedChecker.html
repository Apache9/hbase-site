062/**
063 * An asynchronous implementation of FSWAL.
064 * <p>
065 * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog. We do not use RingBuffer here
066 * because RingBuffer need an exclusive thread to consume the entries in it, and here we want to run
067 * the append and sync operation inside EventLoop. We can not use EventLoop as the RingBuffer's
068 * executor otherwise the EventLoop can not process any other events such as socket read and write.
069 * <p>
070 * For append, we process it as follow:
071 * <ol>
072 * <li>In the caller thread(typically, in the rpc handler thread):
073 * <ol>
074 * <li>Lock 'waitingConsumePayloads', bump nextTxid, and insert the entry to
075 * 'waitingConsumePayloads'.</li>
076 * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
077 * </li>
078 * </ol>
079 * </li>
080 * <li>In the consumer task(in the EventLoop thread)
081 * <ol>
082 * <li>Poll the entry from 'waitingConsumePayloads' and insert it into 'waitingAppendEntries'</li>
083 * <li>Poll the entry from 'waitingAppendEntries', append it to the AsyncWriter, and insert it into
084 * 'unackedEntries'</li>
085 * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
086 * sync on the AsyncWriter.</li>
087 * <li>In the callback methods(CompletionHandler):
088 * <ul>
089 * <li>If succeeded, poll the entry from 'unackedEntries' and drop it.</li>
090 * <li>If failed, add all the entries in 'unackedEntries' back to 'waitingAppendEntries' and wait
091 * for writing them again.</li>
092 * </ul>
093 * </li>
094 * </ol>
095 * </li>
096 * </ol>
097 * For sync, the processing stages are almost same except that if it is not assigned with a new
098 * 'txid', we just assign the previous 'txid' to it without bumping the 'nextTxid'. And different
099 * from FSHLog, we will open a new writer and rewrite unacked entries to the new writer and sync
100 * again if we hit a sync error.
101 * <p>
102 * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
103 * FSHLog.<br>
104 * For a normal roll request(for example, we have reached the log roll size):
105 * <ol>
106 * <li>In the log roller thread, we add a roll payload to 'waitingConsumePayloads', and then wait on
107 * the rollPromise(see {@link #waitForSafePoint()}).</li>
108 * <li>In the consumer thread, we will stop polling entries from 'waitingConsumePayloads' if we hit
109 * a Payload which contains a roll request.</li>
110 * <li>Append all entries to current writer, issue a sync request if possible.</li>
111 * <li>If sync succeeded, check if we could finish a roll request. There 3 conditions:
112 * <ul>
113 * <li>'rollPromise' is not null which means we have a pending roll request.</li>
114 * <li>'waitingAppendEntries' is empty.</li>
115 * <li>'unackedEntries' is empty.</li>
116 * </ul>
117 * </li>
118 * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
119 * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
120 * <li>Acquire 'waitingConsumePayloads' lock, set 'writerBroken' and 'waitingRoll' to false, cancel
121 * log roller exit checker if any(see the comments in the 'failed' method of the sync
122 * CompletionHandler to see why we need a checker here).</li>
123 * <li>Schedule the consumer task if needed.</li>
124 * <li>Schedule a background task to close the old writer.</li>
125 * </ol>
126 * For a broken writer roll request, the only difference is that we can bypass the wait for safe
127 * point stage. See the comments in the 'failed' method of the sync CompletionHandler for more
128 * details.
129 */

230  private final class LogRollerExitedChecker implements Runnable {
231
232    private boolean cancelled;
233
234    private ScheduledFuture<?> future;
235
236    public synchronized void setFuture(ScheduledFuture<?> future) {
237      this.future = future;
238    }
239
240    @Override
241    public void run() {
242      if (!logRollerExited) {
243        return;
244      }
245      // rollWriter is called in the log roller thread, and logRollerExited will be set just before
246      // the log rolled exit. So here we can confirm that no one could cancel us if the 'canceled'
247      // check passed. So it is safe to release the lock after checking 'canceled' flag.
248      synchronized (this) {
249        if (cancelled) {
250          return;
251        }
252      }
253      unackedEntries.clear();
254      waitingAppendEntries.clear();
255      IOException error = new IOException("sync failed but log roller exited");
256      for (SyncFuture future; (future = syncFutures.peek()) != null;) {
257        future.done(highestProcessedTxid, error);
258        syncFutures.remove();
259      }
260      synchronized (waitingConsumePayloads) {
261        for (Payload p : waitingConsumePayloads) {
262          if (p.entry != null) {
263            try {
264              p.entry.stampRegionSequenceId();
265            } catch (IOException e) {
266              throw new AssertionError("should not happen", e);
267            }
268          } else if (p.sync != null) {
269            p.sync.done(nextTxid, error);
270          }
271        }
272        waitingConsumePayloads.clear();
273      }
274    }
275
276    public synchronized void cancel() {
277      future.cancel(false);
278      cancelled = true;
279    }
280  }

300  private void tryFinishRoll() {
301    // 1. a roll is requested
302    // 2. we have written out all entries before the roll point.
303    // 3. all entries have been acked.
304    if (rollPromise != null && waitingAppendEntries.isEmpty() && unackedEntries.isEmpty()) {
305      rollPromise.trySuccess(null);
306      rollPromise = null;
307    }
308  }

310  private void syncFailed(Throwable error) {
311    LOG.warn("sync failed", error);
312    // Here we depends on the implementation of FanOutOneBlockAsyncDFSOutput and netty.
313    // When error occur, FanOutOneBlockAsyncDFSOutput will fail all pending flush requests. It
314    // is execute inside EventLoop. And in DefaultPromise in netty, it will notifyListener
315    // directly if it is already in the EventLoop thread. And in the listener method, it will
316    // call us. So here we know that all failed flush request will call us continuously, and
317    // before the last one finish, no other task can be executed in EventLoop. So here we are
318    // safe to use writerBroken as a guard.
319    // Do not forget to revisit this if we change the implementation of
320    // FanOutOneBlockAsyncDFSOutput!
321    synchronized (waitingConsumePayloads) {
322      if (writerBroken) {
323        return;
324      }
325      // schedule a periodical task to check if log roller is exited. Otherwise the the sync
326      // request maybe blocked forever since we are still waiting for a new writer to write the
327      // pending data and sync it...
328      logRollerExitedChecker = new LogRollerExitedChecker();
329      // we are currently in the EventLoop thread, so it is safe to set the future after
330      // schedule it since the task can not be executed before we release the thread.
331      logRollerExitedChecker.setFuture(eventLoop.scheduleAtFixedRate(logRollerExitedChecker,
332        logRollerExitedCheckIntervalMs, logRollerExitedCheckIntervalMs, TimeUnit.MILLISECONDS));
333      writerBroken = true;
334    }
335    for (Iterator<FSWALEntry> iter = unackedEntries.descendingIterator(); iter.hasNext();) {
336      waitingAppendEntries.addFirst(iter.next());
337    }
338    highestUnsyncedTxid = highestSyncedTxid.get();
339    if (rollPromise != null) {
340      rollPromise.trySuccess(null);
341      rollPromise = null;
342      return;
343    }
344    // request a roll.
345    if (!rollWriterLock.tryLock()) {
346      return;
347    }
348    try {
349      requestLogRoll();
350    } finally {
351      rollWriterLock.unlock();
352    }
353  }
354
355  private void syncCompleted(AsyncWriter writer, long processedTxid, long startTimeNs) {
356    highestSyncedTxid.set(processedTxid);
357    int syncCount = finishSync(true);
358    for (Iterator<FSWALEntry> iter = unackedEntries.iterator(); iter.hasNext();) {
359      if (iter.next().getTxid() <= processedTxid) {
360        iter.remove();
361      } else {
362        break;
363      }
364    }
365    postSync(System.nanoTime() - startTimeNs, syncCount);
366    tryFinishRoll();
367    if (!rollWriterLock.tryLock()) {
368      return;
369    }
370    try {
371      if (writer.getLength() >= logrollsize) {
372        requestLogRoll();
373      }
374    } finally {
375      rollWriterLock.unlock();
376    }
377  }

379  private void sync(final AsyncWriter writer, final long processedTxid) {
380    fileLengthAtLastSync = writer.getLength();
381    final long startTimeNs = System.nanoTime();
382    writer.sync().whenComplete((result, error) -> {
383      if (error != null) {
384        syncFailed(error);
385      } else {
386        syncCompleted(writer, processedTxid, startTimeNs);
387      }
388    });
389  }

397  private int finishSync(boolean addSyncTrace) {
398    long doneTxid = highestSyncedTxid.get();
399    int finished = 0;
400    for (SyncFuture future; (future = syncFutures.peek()) != null;) {
401      if (future.getTxid() <= doneTxid) {
402        future.done(doneTxid, null);
403        syncFutures.remove();
404        finished++;
405        addTimeAnnotation(future, "writer synced");
406      } else {
407        break;
408      }
409    }
410    return finished;
411  }

413  private void consume() {
414    final AsyncWriter writer = this.writer;
415    // maybe a sync request is not queued when we issue a sync, so check here to see if we could
416    // finish some.
417    finishSync(false);
418    long newHighestProcessedTxid = -1L;
419    for (Iterator<FSWALEntry> iter = waitingAppendEntries.iterator(); iter.hasNext();) {
420      FSWALEntry entry = iter.next();
421      boolean appended;
422      try {
423        appended = append(writer, entry);
424      } catch (IOException e) {
425        throw new AssertionError("should not happen", e);
426      }
427      newHighestProcessedTxid = entry.getTxid();
428      iter.remove();
429      if (appended) {
430        unackedEntries.addLast(entry);
431        if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
432          break;
433        }
434      }
435    }
436    // if we have a newer transaction id, update it.
437    // otherwise, use the previous transaction id.
438    if (newHighestProcessedTxid > 0) {
439      highestProcessedTxid = newHighestProcessedTxid;
440    } else {
441      newHighestProcessedTxid = highestProcessedTxid;
442    }
443    if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
444      // sync because buffer size limit.
445      sync(writer, newHighestProcessedTxid);
446    } else if ((!syncFutures.isEmpty() || rollPromise != null)
447        && writer.getLength() > fileLengthAtLastSync) {
448      // first we should have at least one sync request or a roll request
449      // second we should have some unsynced data.
450      sync(writer, newHighestProcessedTxid);
451    } else if (writer.getLength() == fileLengthAtLastSync) {
452      // we haven't written anything out, just advance the highestSyncedSequence since we may only
453      // stamped some region sequence id.
454      highestSyncedTxid.set(newHighestProcessedTxid);
455      finishSync(false);
456      tryFinishRoll();
457    }
458  }

460  private static final Comparator<SyncFuture> SEQ_COMPARATOR = (o1, o2) -> {
461    int c = Long.compare(o1.getTxid(), o2.getTxid());
462    return c != 0 ? c : Integer.compare(System.identityHashCode(o1), System.identityHashCode(o2));
463  };

465  private final Runnable consumer = new Runnable() {
466
467    @Override
468    public void run() {
469      synchronized (waitingConsumePayloads) {
470        assert consumerScheduled;
471        if (writerBroken) {
472          // waiting for reschedule after rollWriter.
473          consumerScheduled = false;
474          return;
475        }
476        if (waitingRoll) {
477          // we may have toWriteEntries if the consume method does not write all pending entries
478          // out, this is usually happen if we have too many toWriteEntries that exceeded the
479          // batchSize limit.
480          if (waitingAppendEntries.isEmpty()) {
481            consumerScheduled = false;
482            return;
483          }
484        } else {
485          for (Payload p; (p = waitingConsumePayloads.pollFirst()) != null;) {
486            if (p.entry != null) {
487              waitingAppendEntries.addLast(p.entry);
488            } else if (p.sync != null) {
489              syncFutures.add(p.sync);
490            } else {
491              rollPromise = p.roll;
492              waitingRoll = true;
493              break;
494            }
495          }
496        }
497      }
498      consume();
499      synchronized (waitingConsumePayloads) {
500        if (waitingRoll) {
501          if (waitingAppendEntries.isEmpty()) {
502            consumerScheduled = false;
503            return;
504          }
505        } else {
506          if (waitingConsumePayloads.isEmpty() && waitingAppendEntries.isEmpty()) {
507            consumerScheduled = false;
508            return;
509          }
510        }
511      }
512      // reschedule if we still have something to write.
513      eventLoop.execute(this);
514    }
515  };
516
517  private boolean shouldScheduleConsumer() {
518    if (writerBroken || waitingRoll) {
519      return false;
520    }
521    if (consumerScheduled) {
522      return false;
523    }
524    consumerScheduled = true;
525    return true;
526  }

528  @Override
529  public long append(HRegionInfo hri, WALKey key, WALEdit edits, boolean inMemstore)
530      throws IOException {
531    boolean scheduleTask;
532    long txid;
533    synchronized (waitingConsumePayloads) {
534      if (this.closed) {
535        throw new IOException("Cannot append; log is closed");
536      }
537      txid = nextTxid++;
538      FSWALEntry entry = new FSWALEntry(txid, key, edits, hri, inMemstore);
539      scheduleTask = shouldScheduleConsumer();
540      waitingConsumePayloads.add(new Payload(entry));
541    }
542    if (scheduleTask) {
543      eventLoop.execute(consumer);
544    }
545    return txid;
546  }

548  @Override
549  public void sync() throws IOException {
550    TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
551    try {
552      SyncFuture future;
553      boolean scheduleTask;
554      synchronized (waitingConsumePayloads) {
555        scheduleTask = shouldScheduleConsumer();
556        future = getSyncFuture(nextTxid - 1, scope.detach());
557        waitingConsumePayloads.addLast(new Payload(future));
558      }
559      if (scheduleTask) {
560        eventLoop.execute(consumer);
561      }
562      scope = Trace.continueSpan(blockOnSync(future));
563    } finally {
564      assert scope == NullScope.INSTANCE || !scope.isDetached();
565      scope.close();
566    }
567  }
568
569  @Override
570  public void sync(long txid) throws IOException {
571    if (highestSyncedTxid.get() >= txid) {
572      return;
573    }
574    TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
575    try {
576      SyncFuture future = getSyncFuture(txid, scope.detach());
577      boolean scheduleTask;
578      synchronized (waitingConsumePayloads) {
579        scheduleTask = shouldScheduleConsumer();
580        waitingConsumePayloads.addLast(new Payload(future));
581      }
582      if (scheduleTask) {
583        eventLoop.execute(consumer);
584      }
585      scope = Trace.continueSpan(blockOnSync(future));
586    } finally {
587      assert scope == NullScope.INSTANCE || !scope.isDetached();
588      scope.close();
589    }
590  }

597  @Override
598  protected AsyncWriter createWriterInstance(Path path) throws IOException {
599    boolean overwrite = false;
600    for (int retry = 0;; retry++) {
601      try {
602        return AsyncFSWALProvider.createAsyncWriter(conf, fs, path, overwrite, eventLoop);
603      } catch (RemoteException e) {
604        LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
605        if (shouldRetryCreate(e)) {
606          if (retry >= createMaxRetries) {
607            break;
608          }
609        } else {
610          throw e.unwrapRemoteException();
611        }
612      } catch (NameNodeException e) {
613        throw e;
614      } catch (IOException e) {
615        LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
616        if (retry >= createMaxRetries) {
617          break;
618        }
619        // overwrite the old broken file.
620        overwrite = true;
621        try {
622          Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
623        } catch (InterruptedException ie) {
624          throw new InterruptedIOException();
625        }
626      }
627    }
628    throw new IOException("Failed to create wal log writer " + path + " after retrying "
629        + createMaxRetries + " time(s)");
630  }

632  private void waitForSafePoint() {
633    Future<Void> roll;
634    boolean scheduleTask;
635    synchronized (waitingConsumePayloads) {
636      if (!writerBroken && this.writer != null) {
637        Promise<Void> promise = eventLoop.newPromise();
638        if (consumerScheduled) {
639          scheduleTask = false;
640        } else {
641          scheduleTask = consumerScheduled = true;
642        }
643        waitingConsumePayloads.addLast(new Payload(promise));
644        roll = promise;
645      } else {
646        roll = eventLoop.newSucceededFuture(null);
647        scheduleTask = false;
648      }
649    }
650    if (scheduleTask) {
651      eventLoop.execute(consumer);
652    }
653    roll.awaitUninterruptibly();
654  }

656  @Override
657  protected long doReplaceWriter(Path oldPath, Path newPath, AsyncWriter nextWriter)
658      throws IOException {
659    waitForSafePoint();
660    final AsyncWriter oldWriter = this.writer;
661    this.writer = nextWriter;
662    if (nextWriter != null && nextWriter instanceof AsyncProtobufLogWriter) {
663      this.fsOut = ((AsyncProtobufLogWriter) nextWriter).getOutput();
664    }
665    this.fileLengthAtLastSync = 0L;
666    boolean scheduleTask;
667    synchronized (waitingConsumePayloads) {
668      writerBroken = waitingRoll = false;
669      if (logRollerExitedChecker != null) {
670        logRollerExitedChecker.cancel();
671        logRollerExitedChecker = null;
672      }
673      if (consumerScheduled) {
674        scheduleTask = false;
675      } else {
676        if (waitingConsumePayloads.isEmpty() && waitingAppendEntries.isEmpty()) {
677          scheduleTask = false;
678        } else {
679          scheduleTask = consumerScheduled = true;
680        }
681      }
682    }
683    if (scheduleTask) {
684      eventLoop.execute(consumer);
685    }
686    long oldFileLen;
687    if (oldWriter != null) {
688      oldFileLen = oldWriter.getLength();
689      closeExecutor.execute(() -> {
690        try {
691          oldWriter.close();
692        } catch (IOException e) {
693          LOG.warn("close old writer failed", e);
694        }
695      });
696    } else {
697      oldFileLen = 0L;
698    }
699    return oldFileLen;
700  }
701
702  @Override
703  protected void doShutdown() throws IOException {
704    waitForSafePoint();
705    this.writer.close();
706    this.writer = null;
707    closeExecutor.shutdown();
708  }
