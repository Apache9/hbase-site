072/**
073 * An asynchronous implementation of FSWAL.
074 * <p>
075 * Here 'waitingConsumePayloads' acts as the RingBuffer in FSHLog.
076 * <p>
077 * For append, we process it as follow:
078 * <ol>
079 * <li>In the caller thread(typically, in the rpc handler thread):
080 * <ol>
081 * <li>Insert the entry into 'waitingConsumePayloads'. Use ringbuffer sequence as txid.</li>
082 * <li>Schedule the consumer task if needed. See {@link #shouldScheduleConsumer()} for more details.
083 * </li>
084 * </ol>
085 * </li>
086 * <li>In the consumer task(in the EventLoop thread)
087 * <ol>
088 * <li>Poll the entry from {@link #waitingConsumePayloads} and insert it into
089 * {@link #toWriteAppends}</li>
090 * <li>Poll the entry from {@link #toWriteAppends}, append it to the AsyncWriter, and insert it into
091 * {@link #unackedAppends}</li>
092 * <li>If the buffered size reaches {@link #batchSize}, or there is a sync request, then we call
093 * sync on the AsyncWriter.</li>
094 * <li>In the callback methods:
095 * <ul>
096 * <li>If succeeded, poll the entry from {@link #unackedAppends} and drop it.</li>
097 * <li>If failed, add all the entries in {@link #unackedAppends} back to {@link #toWriteAppends} and
098 * wait for writing them again.</li>
099 * </ul>
100 * </li>
101 * </ol>
102 * </li>
103 * </ol>
104 * For sync, the processing stages are almost same. And different from FSHLog, we will open a new
105 * writer and rewrite unacked entries to the new writer and sync again if we hit a sync error.
106 * <p>
107 * Here we only describe the logic of doReplaceWriter. The main logic of rollWriter is same with
108 * FSHLog.<br>
109 * For a normal roll request(for example, we have reached the log roll size):
110 * <ol>
111 * <li>In the log roller thread, we will set {@link #waitingRoll} to true and
112 * {@link #readyForRolling} to false, and then wait on {@link #readyForRolling}(see
113 * {@link #waitForSafePoint()}).</li>
114 * <li>In the consumer thread, we will stop polling entries from {@link #waitingConsumePayloads} if
115 * {@link #waitingRoll} is true, and also stop writing the entries in {@link #toWriteAppends} out.
116 * </li>
117 * <li>If there are unflush data in the writer, sync them.</li>
118 * <li>When all out-going sync request is finished, i.e, the {@link #unackedAppends} is empty,
119 * signal the {@link #readyForRollingCond}.</li>
120 * <li>Back to the log roller thread, now we can confirm that there are no out-going entries, i.e.,
121 * we reach a safe point. So it is safe to replace old writer with new writer now.</li>
122 * <li>Set {@link #writerBroken} and {@link #waitingRoll} to false, cancel log roller exit checker
123 * if any(see the comments in the {@link #syncFailed(Throwable)} method to see why we need a checker
124 * here).</li>
125 * <li>Schedule the consumer task.</li>
126 * <li>Schedule a background task to close the old writer.</li>
127 * </ol>
128 * For a broken writer roll request, the only difference is that we can bypass the wait for safe
129 * point stage. See the comments in the {@link #syncFailed(Throwable)} method for more details.
130 */

197  public AsyncFSWAL(FileSystem fs, Path rootDir, String logDir, String archiveDir,
198      Configuration conf, List<WALActionsListener> listeners, boolean failIfWALExists,
199      String prefix, String suffix, EventLoop eventLoop, Class<? extends Channel> channelClass)
200      throws FailedLogCloseException, IOException {
201    super(fs, rootDir, logDir, archiveDir, conf, listeners, failIfWALExists, prefix, suffix);
202    this.eventLoop = eventLoop;
203    this.channelClass = channelClass;
204    Supplier<Boolean> hasConsumerTask;
205    if (eventLoop instanceof SingleThreadEventExecutor) {
206
207      try {
208        Field field = SingleThreadEventExecutor.class.getDeclaredField("taskQueue");
209        field.setAccessible(true);
210        Queue<?> queue = (Queue<?>) field.get(eventLoop);
211        hasConsumerTask = () -> queue.peek() == consumer;
212      } catch (Exception e) {
213        LOG.warn("Can not get task queue of " + eventLoop + ", this is not necessary, just give up",
214          e);
215        hasConsumerTask = () -> false;
216      }
217    } else {
218      hasConsumerTask = () -> false;
219    }
220    this.hasConsumerTask = hasConsumerTask;
221    int preallocatedEventCount =
222        this.conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
223    waitingConsumePayloads =
224        RingBuffer.createMultiProducer(RingBufferTruck::new, preallocatedEventCount);
225    waitingConsumePayloadsGatingSequence = new Sequence(Sequencer.INITIAL_CURSOR_VALUE);
226    waitingConsumePayloads.addGatingSequences(waitingConsumePayloadsGatingSequence);
227
228    // inrease the ringbuffer sequence so our txid is start from 1
229    waitingConsumePayloads.publish(waitingConsumePayloads.next());
230    waitingConsumePayloadsGatingSequence.set(waitingConsumePayloads.getCursor());
231
232    batchSize = conf.getLong(WAL_BATCH_SIZE, DEFAULT_WAL_BATCH_SIZE);
233    createMaxRetries =
234        conf.getInt(ASYNC_WAL_CREATE_MAX_RETRIES, DEFAULT_ASYNC_WAL_CREATE_MAX_RETRIES);
235    rollWriter();
236  }

238  // return whether we have successfully set readyForRolling to true.
239  private boolean trySetReadyForRolling() {
240    // Check without holding lock first. Usually we will just return here.
241    // waitingRoll is volatile and unacedEntries is only accessed inside event loop so it is safe to
242    // check them outside the consumeLock.
243    if (!waitingRoll || !unackedAppends.isEmpty()) {
244      return false;
245    }
246    consumeLock.lock();
247    try {
248      // 1. a roll is requested
249      // 2. all out-going entries have been acked(we have confirmed above).
250      if (waitingRoll) {
251        readyForRolling = true;
252        readyForRollingCond.signalAll();
253        return true;
254      } else {
255        return false;
256      }
257    } finally {
258      consumeLock.unlock();
259    }
260  }

262  private void syncFailed(Throwable error) {
263    LOG.warn("sync failed", error);
264    // Here we depends on the implementation of FanOutOneBlockAsyncDFSOutput and netty.
265    // When error occur, FanOutOneBlockAsyncDFSOutput will fail all pending flush requests. It
266    // is execute inside EventLoop. And in DefaultPromise in netty, it will notifyListener
267    // directly if it is already in the EventLoop thread. And in the listener method, it will
268    // call us. So here we know that all failed flush request will call us continuously, and
269    // before the last one finish, no other task can be executed in EventLoop. So here we are
270    // safe to use writerBroken as a guard.
271    // Do not forget to revisit this if we change the implementation of
272    // FanOutOneBlockAsyncDFSOutput!
273    consumeLock.lock();
274    try {
275      if (writerBroken) {
276        return;
277      }
278      writerBroken = true;
279      if (waitingRoll) {
280        readyForRolling = true;
281        readyForRollingCond.signalAll();
282      }
283    } finally {
284      consumeLock.unlock();
285    }
286    for (Iterator<FSWALEntry> iter = unackedAppends.descendingIterator(); iter.hasNext();) {
287      toWriteAppends.addFirst(iter.next());
288    }
289    highestUnsyncedTxid = highestSyncedTxid.get();
290    // request a roll.
291    requestLogRoll();
292  }

294  private void syncCompleted(AsyncWriter writer, long processedTxid, long startTimeNs) {
295    highestSyncedTxid.set(processedTxid);
296    for (Iterator<FSWALEntry> iter = unackedAppends.iterator(); iter.hasNext();) {
297      if (iter.next().getTxid() <= processedTxid) {
298        iter.remove();
299      } else {
300        break;
301      }
302    }
303    postSync(System.nanoTime() - startTimeNs, finishSync(true));
304    // Ideally, we should set a flag to indicate that the log roll has already been requested for
305    // the current writer and give up here, and reset the flag when roll is finished. But we
306    // finish roll in the log roller thread so the flag need to be set by different thread which
307    // typically means we need to use a lock to protect it and do fencing. As the log roller will
308    // aggregate the roll requests of the same WAL, so it is safe to call requestLogRoll multiple
309    // times before the roll actual happens. But we need to stop if we set readyForRolling to true
310    // and wake up the log roller thread waiting in waitForSafePoint as the rollWriter call may
311    // return firstly and then we run the code below and request a roll on the new writer.
312    if (trySetReadyForRolling()) {
313      // we have just finished a roll, then do not need to check for log rolling, the writer will be
314      // closed soon.
315      return;
316    }
317    if (writer.getLength() < logrollsize) {
318      return;
319    }
320    if (!rollWriterLock.tryLock()) {
321      return;
322    }
323    try {
324      requestLogRoll();
325    } finally {
326      rollWriterLock.unlock();
327    }
328  }

330  private void sync(AsyncWriter writer) {
331    fileLengthAtLastSync = writer.getLength();
332    long currentHighestProcessedAppendTxid = highestProcessedAppendTxid;
333    highestProcessedAppendTxidAtLastSync = currentHighestProcessedAppendTxid;
334    final long startTimeNs = System.nanoTime();
335    writer.sync().whenComplete((result, error) -> {
336      if (error != null) {
337        syncFailed(error);
338      } else {
339        syncCompleted(writer, currentHighestProcessedAppendTxid, startTimeNs);
340      }
341    });
342  }

346    Trace.addTimelineAnnotation(annotation);

350  private int finishSyncLowerThanTxid(long txid, boolean addSyncTrace) {
351    int finished = 0;
352    for (Iterator<SyncFuture> iter = syncFutures.iterator(); iter.hasNext();) {
353      SyncFuture sync = iter.next();
354      if (sync.getTxid() <= txid) {
355        sync.done(txid, null);
356        iter.remove();
357        finished++;
358        if (addSyncTrace) {
359          addTimeAnnotation(sync, "writer synced");
360        }
361      } else {
362        break;
363      }
364    }
365    return finished;
366  }

368  // try advancing the highestSyncedTxid as much as possible
369  private int finishSync(boolean addSyncTrace) {
370    if (unackedAppends.isEmpty()) {
371      // All outstanding appends have been acked.
372      if (toWriteAppends.isEmpty()) {
373        // Also no appends that wait to be written out, then just finished all pending syncs.
374        long maxSyncTxid = highestSyncedTxid.get();
375        for (SyncFuture sync : syncFutures) {
376          maxSyncTxid = Math.max(maxSyncTxid, sync.getTxid());
377          sync.done(maxSyncTxid, null);
378          if (addSyncTrace) {
379            addTimeAnnotation(sync, "writer synced");
380          }
381        }
382        highestSyncedTxid.set(maxSyncTxid);
383        int finished = syncFutures.size();
384        syncFutures.clear();
385        return finished;
386      } else {
387        // There is no append between highestProcessedAppendTxid and lowestUnprocessedAppendTxid, so
388        // if highestSyncedTxid >= highestProcessedAppendTxid, then all syncs whose txid are between
389        // highestProcessedAppendTxid and lowestUnprocessedAppendTxid can be finished.
390        long lowestUnprocessedAppendTxid = toWriteAppends.peek().getTxid();
391        assert lowestUnprocessedAppendTxid > highestProcessedAppendTxid;
392        long doneTxid = lowestUnprocessedAppendTxid - 1;
393        highestSyncedTxid.set(doneTxid);
394        return finishSyncLowerThanTxid(doneTxid, addSyncTrace);
395      }
396    } else {
397      // There are still unacked appends. So let's move the highestSyncedTxid to the txid of the
398      // first unacked append minus 1.
399      long lowestUnackedAppendTxid = unackedAppends.peek().getTxid();
400      long doneTxid = Math.max(lowestUnackedAppendTxid - 1, highestSyncedTxid.get());
401      highestSyncedTxid.set(doneTxid);
402      return finishSyncLowerThanTxid(doneTxid, addSyncTrace);
403    }
404  }

406  private void appendAndSync() {
407    final AsyncWriter writer = this.writer;
408    // maybe a sync request is not queued when we issue a sync, so check here to see if we could
409    // finish some.
410    finishSync(false);
411    long newHighestProcessedAppendTxid = -1L;
412    for (Iterator<FSWALEntry> iter = toWriteAppends.iterator(); iter.hasNext();) {
413      FSWALEntry entry = iter.next();
414      boolean appended;
415      Span span = entry.detachSpan();
416      // the span maybe null if this is a retry after rolling.
417      if (span != null) {
418        TraceScope scope = Trace.continueSpan(span);
419        try {
420          appended = append(writer, entry);
421        } catch (IOException e) {
422          throw new AssertionError("should not happen", e);
423        } finally {
424          assert scope == NullScope.INSTANCE || !scope.isDetached();
425          scope.close(); // append scope is complete
426        }
427      } else {
428        try {
429          appended = append(writer, entry);
430        } catch (IOException e) {
431          throw new AssertionError("should not happen", e);
432        }
433      }
434      newHighestProcessedAppendTxid = entry.getTxid();
435      iter.remove();
436      if (appended) {
437        unackedAppends.addLast(entry);
438        if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
439          break;
440        }
441      }
442    }
443    // if we have a newer transaction id, update it.
444    // otherwise, use the previous transaction id.
445    if (newHighestProcessedAppendTxid > 0) {
446      highestProcessedAppendTxid = newHighestProcessedAppendTxid;
447    } else {
448      newHighestProcessedAppendTxid = highestProcessedAppendTxid;
449    }
450
451    if (writer.getLength() - fileLengthAtLastSync >= batchSize) {
452      // sync because buffer size limit.
453      sync(writer);
454      return;
455    }
456    if (writer.getLength() == fileLengthAtLastSync) {
457      // we haven't written anything out, just advance the highestSyncedSequence since we may only
458      // stamped some region sequence id.
459      if (unackedAppends.isEmpty()) {
460        highestSyncedTxid.set(highestProcessedAppendTxid);
461        finishSync(false);
462        trySetReadyForRolling();
463      }
464      return;
465    }
466    // we have some unsynced data but haven't reached the batch size yet
467    if (!syncFutures.isEmpty()
468        && syncFutures.last().getTxid() > highestProcessedAppendTxidAtLastSync) {
469      // we have at least one sync request
470      sync(writer);
471    }
472  }

474  private void consume() {
475    consumeLock.lock();
476    try {
477      if (writerBroken) {
478        return;
479      }
480      if (waitingRoll) {
481        if (writer.getLength() > fileLengthAtLastSync) {
482          // issue a sync
483          sync(writer);
484        } else {
485          if (unackedAppends.isEmpty()) {
486            readyForRolling = true;
487            readyForRollingCond.signalAll();
488          }
489        }
490        return;
491      }
492    } finally {
493      consumeLock.unlock();
494    }
495    long nextCursor = waitingConsumePayloadsGatingSequence.get() + 1;
496    for (long cursorBound =
497        waitingConsumePayloads.getCursor(); nextCursor <= cursorBound; nextCursor++) {
498      if (!waitingConsumePayloads.isPublished(nextCursor)) {
499        break;
500      }
501      RingBufferTruck truck = waitingConsumePayloads.get(nextCursor);
502      switch (truck.type()) {
503        case APPEND:
504          toWriteAppends.addLast(truck.unloadAppend());
505          break;
506        case SYNC:
507          syncFutures.add(truck.unloadSync());
508          break;
509        default:
510          LOG.warn("RingBufferTruck with unexpected type: " + truck.type());
511          break;
512      }
513      waitingConsumePayloadsGatingSequence.set(nextCursor);
514    }
515    appendAndSync();
516    if (hasConsumerTask.get()) {
517      return;
518    }
519    if (toWriteAppends.isEmpty()) {
520      if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
521        consumerScheduled.set(false);
522        // recheck here since in append and sync we do not hold the consumeLock. Thing may
523        // happen like
524        // 1. we check cursor, no new entry
525        // 2. someone publishes a new entry to ringbuffer and the consumerScheduled is true and
526        // give up scheduling the consumer task.
527        // 3. we set consumerScheduled to false and also give up scheduling consumer task.
528        if (waitingConsumePayloadsGatingSequence.get() == waitingConsumePayloads.getCursor()) {
529          return;
530        } else {
531          // maybe someone has grabbed this before us
532          if (!consumerScheduled.compareAndSet(false, true)) {
533            return;
534          }
535        }
536      }
537    }
538    // reschedule if we still have something to write.
539    eventLoop.execute(consumer);
540  }

542  private boolean shouldScheduleConsumer() {
543    if (writerBroken || waitingRoll) {
544      return false;
545    }
546    return consumerScheduled.compareAndSet(false, true);
547  }

549  @Override
550  public long append(HRegionInfo hri, WALKey key, WALEdit edits, boolean inMemstore)
551      throws IOException {
552    long txid =
553        stampSequenceIdAndPublishToRingBuffer(hri, key, edits, inMemstore, waitingConsumePayloads);
554    if (shouldScheduleConsumer()) {
555      eventLoop.execute(consumer);
556    }
557    return txid;
558  }

560  @Override
561  public void sync() throws IOException {
562    TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
563    try {
564      long txid = waitingConsumePayloads.next();
565      SyncFuture future;
566      try {
567        future = getSyncFuture(txid, scope.detach());
568        RingBufferTruck truck = waitingConsumePayloads.get(txid);
569        truck.load(future);
570      } finally {
571        waitingConsumePayloads.publish(txid);
572      }
573      if (shouldScheduleConsumer()) {
574        eventLoop.execute(consumer);
575      }
576      scope = Trace.continueSpan(blockOnSync(future));
577    } finally {
578      assert scope == NullScope.INSTANCE || !scope.isDetached();
579      scope.close();
580    }
581  }

583  @Override
584  public void sync(long txid) throws IOException {
585    if (highestSyncedTxid.get() >= txid) {
586      return;
587    }
588    TraceScope scope = Trace.startSpan("AsyncFSWAL.sync");
589    try {
590      // here we do not use ring buffer sequence as txid
591      long sequence = waitingConsumePayloads.next();
592      SyncFuture future;
593      try {
594        future = getSyncFuture(txid, scope.detach());
595        RingBufferTruck truck = waitingConsumePayloads.get(sequence);
596        truck.load(future);
597      } finally {
598        waitingConsumePayloads.publish(sequence);
599      }
600      if (shouldScheduleConsumer()) {
601        eventLoop.execute(consumer);
602      }
603      scope = Trace.continueSpan(blockOnSync(future));
604    } finally {
605      assert scope == NullScope.INSTANCE || !scope.isDetached();
606      scope.close();
607    }
608  }

610  @Override
611  protected AsyncWriter createWriterInstance(Path path) throws IOException {
612    boolean overwrite = false;
613    for (int retry = 0;; retry++) {
614      try {
615        return AsyncFSWALProvider.createAsyncWriter(conf, fs, path, overwrite, eventLoop,
616          channelClass);
617      } catch (RemoteException e) {
618        LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
619        if (shouldRetryCreate(e)) {
620          if (retry >= createMaxRetries) {
621            break;
622          }
623        } else {
624          IOException ioe = e.unwrapRemoteException();
625          // this usually means master already think we are dead so let's fail all the pending
626          // syncs. The shutdown process of RS will wait for all regions to be closed before calling
627          // WAL.close so if we do not wake up the thread blocked by sync here it will cause dead
628          // lock.
629          if (e.getMessage().contains("Parent directory doesn't exist:")) {
630            syncFutures.forEach(f -> f.done(f.getTxid(), ioe));
631          }
632          throw ioe;
633        }
634      } catch (NameNodeException e) {
635        throw e;
636      } catch (IOException e) {
637        LOG.warn("create wal log writer " + path + " failed, retry = " + retry, e);
638        if (retry >= createMaxRetries) {
639          break;
640        }
641        // overwrite the old broken file.
642        overwrite = true;
643        try {
644          Thread.sleep(ConnectionUtils.getPauseTime(100, retry));
645        } catch (InterruptedException ie) {
646          throw new InterruptedIOException();
647        }
648      }
649    }
650    throw new IOException("Failed to create wal log writer " + path + " after retrying "
651        + createMaxRetries + " time(s)");
652  }
653
654  private void waitForSafePoint() {
655    consumeLock.lock();
656    try {
657      if (writerBroken || this.writer == null) {
658        return;
659      }
660      consumerScheduled.set(true);
661      waitingRoll = true;
662      readyForRolling = false;
663      eventLoop.execute(consumer);
664      while (!readyForRolling) {
665        readyForRollingCond.awaitUninterruptibly();
666      }
667    } finally {
668      consumeLock.unlock();
669    }
670  }
671
672  @Override
673  protected long doReplaceWriter(Path oldPath, Path newPath, AsyncWriter nextWriter)
674      throws IOException {
675    waitForSafePoint();
676    final AsyncWriter oldWriter = this.writer;
677    this.writer = nextWriter;
678    if (nextWriter != null && nextWriter instanceof AsyncProtobufLogWriter) {
679      this.fsOut = ((AsyncProtobufLogWriter) nextWriter).getOutput();
680    }
681    this.fileLengthAtLastSync = 0L;
682    this.highestProcessedAppendTxidAtLastSync = 0L;
683    consumeLock.lock();
684    try {
685      consumerScheduled.set(true);
686      writerBroken = waitingRoll = false;
687      eventLoop.execute(consumer);
688    } finally {
689      consumeLock.unlock();
690    }
691    long oldFileLen;
692    if (oldWriter != null) {
693      oldFileLen = oldWriter.getLength();
694      closeExecutor.execute(() -> {
695        try {
696          oldWriter.close();
697        } catch (IOException e) {
698          LOG.warn("close old writer failed", e);
699        }
700      });
701    } else {
702      oldFileLen = 0L;
703    }
704    return oldFileLen;
705  }
706
707  @Override
708  protected void doShutdown() throws IOException {
709    waitForSafePoint();
710    this.writer.close();
711    this.writer = null;
712    closeExecutor.shutdown();
713    IOException error = new IOException("WAL has been closed");
714    syncFutures.forEach(f -> f.done(f.getTxid(), error));
715  }
