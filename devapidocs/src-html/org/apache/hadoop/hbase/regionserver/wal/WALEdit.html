045/**
046 * WALEdit: Used in HBase's transaction log (WAL) to represent
047 * the collection of edits (KeyValue objects) corresponding to a
048 * single transaction.
049 *
050 * All the edits for a given transaction are written out as a single record, in PB format followed
051 * by Cells written via the WALCellEncoder.
052 */
053@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.REPLICATION,
054    HBaseInterfaceAudience.COPROC })
055public class WALEdit implements HeapSize {
056  private static final Log LOG = LogFactory.getLog(WALEdit.class);
057
058  // TODO: Get rid of this; see HBASE-8457
059  public static final byte [] METAFAMILY = Bytes.toBytes("METAFAMILY");
060  @VisibleForTesting
061  public static final byte [] METAROW = Bytes.toBytes("METAROW");
062  @VisibleForTesting
063  public static final byte[] COMPACTION = Bytes.toBytes("HBASE::COMPACTION");
064  @VisibleForTesting
065  public static final byte [] FLUSH = Bytes.toBytes("HBASE::FLUSH");
066  @VisibleForTesting
067  public static final byte [] REGION_EVENT = Bytes.toBytes("HBASE::REGION_EVENT");
068  @VisibleForTesting
069  public static final byte [] BULK_LOAD = Bytes.toBytes("HBASE::BULK_LOAD");
070
071  private final boolean isReplay;
072
073  private ArrayList<Cell> cells = null;
074
075  public WALEdit() {
076    this(false);
077  }
078
079  public WALEdit(boolean isReplay) {
080    this(1, isReplay);
081  }
082
083  public WALEdit(int cellCount) {
084    this(cellCount, false);
085  }
086
087  public WALEdit(int cellCount, boolean isReplay) {
088    this.isReplay = isReplay;
089    cells = new ArrayList<>(cellCount);
090  }
091
092  /**
093   * @param f
094   * @return True is <code>f</code> is {@link #METAFAMILY}
095   */
096  public static boolean isMetaEditFamily(final byte [] f) {
097    return Bytes.equals(METAFAMILY, f);
098  }
099
100  public static boolean isMetaEditFamily(Cell cell) {
101    return CellUtil.matchingFamily(cell, METAFAMILY);
102  }
103
104  public boolean isMetaEdit() {
105    for (Cell cell: cells) {
106      if (!isMetaEditFamily(cell)) {
107        return false;
108      }
109    }
110    return true;
111  }
112
113  /**
114   * @return True when current WALEdit is created by log replay. Replication skips WALEdits from
115   *         replay.
116   */
117  public boolean isReplay() {
118    return this.isReplay;
119  }
120
121  public WALEdit add(Cell cell) {
122    this.cells.add(cell);
123    return this;
124  }
125
126  public boolean isEmpty() {
127    return cells.isEmpty();
128  }
129
130  public int size() {
131    return cells.size();
132  }
133
134  public ArrayList<Cell> getCells() {
135    return cells;
136  }
137
138  /**
139   * This is not thread safe.
140   * This will change the WALEdit and shouldn't be used unless you are sure that nothing
141   * else depends on the contents being immutable.
142   *
143   * @param cells the list of cells that this WALEdit now contains.
144   */
145  @InterfaceAudience.Private
146  public void setCells(ArrayList<Cell> cells) {
147    this.cells = cells;
148  }
149
150  /**
151   * Reads WALEdit from cells.
152   * @param cellDecoder Cell decoder.
153   * @param expectedCount Expected cell count.
154   * @return Number of KVs read.
155   */
156  public int readFromCells(Codec.Decoder cellDecoder, int expectedCount) throws IOException {
157    cells.clear();
158    cells.ensureCapacity(expectedCount);
159    while (cells.size() < expectedCount && cellDecoder.advance()) {
160      cells.add(cellDecoder.current());
161    }
162    return cells.size();
163  }
164
165  @Override
166  public long heapSize() {
167    long ret = ClassSize.ARRAYLIST;
168    for (Cell cell : cells) {
169      ret += CellUtil.estimatedHeapSizeOf(cell);
170    }
171    return ret;
172  }
173
174  public long estimatedSerializedSizeOf() {
175    long ret = 0;
176    for (Cell cell: cells) {
177      ret += CellUtil.estimatedSerializedSizeOf(cell);
178    }
179    return ret;
180  }
181
182  @Override
183  public String toString() {
184    StringBuilder sb = new StringBuilder();
185
186    sb.append("[#edits: " + cells.size() + " = <");
187    for (Cell cell : cells) {
188      sb.append(cell);
189      sb.append("; ");
190    }
191    sb.append(">]");
192    return sb.toString();
193  }
194
195  public static WALEdit createFlushWALEdit(HRegionInfo hri, FlushDescriptor f) {
196    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, FLUSH,
197      EnvironmentEdgeManager.currentTime(), f.toByteArray());
198    return new WALEdit().add(kv);
199  }
200
201  public static FlushDescriptor getFlushDescriptor(Cell cell) throws IOException {
202    if (CellUtil.matchingColumn(cell, METAFAMILY, FLUSH)) {
203      return FlushDescriptor.parseFrom(CellUtil.cloneValue(cell));
204    }
205    return null;
206  }
207
208  public static WALEdit createRegionEventWALEdit(HRegionInfo hri,
209      RegionEventDescriptor regionEventDesc) {
210    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, REGION_EVENT,
211      EnvironmentEdgeManager.currentTime(), regionEventDesc.toByteArray());
212    return new WALEdit().add(kv);
213  }
214
215  public static RegionEventDescriptor getRegionEventDescriptor(Cell cell) throws IOException {
216    if (CellUtil.matchingColumn(cell, METAFAMILY, REGION_EVENT)) {
217      return RegionEventDescriptor.parseFrom(CellUtil.cloneValue(cell));
218    }
219    return null;
220  }
221
222  /**
223   * Create a compaction WALEdit
224   * @param c
225   * @return A WALEdit that has <code>c</code> serialized as its value
226   */
227  public static WALEdit createCompaction(final HRegionInfo hri, final CompactionDescriptor c) {
228    byte [] pbbytes = c.toByteArray();
229    KeyValue kv = new KeyValue(getRowForRegion(hri), METAFAMILY, COMPACTION,
230      EnvironmentEdgeManager.currentTime(), pbbytes);
231    return new WALEdit().add(kv); //replication scope null so that this won't be replicated
232  }
233
234  public static byte[] getRowForRegion(HRegionInfo hri) {
235    byte[] startKey = hri.getStartKey();
236    if (startKey.length == 0) {
237      // empty row key is not allowed in mutations because it is both the start key and the end key
238      // we return the smallest byte[] that is bigger (in lex comparison) than byte[0].
239      return new byte[] {0};
240    }
241    return startKey;
242  }
243
244  /**
245   * Deserialized and returns a CompactionDescriptor is the KeyValue contains one.
246   * @param kv the key value
247   * @return deserialized CompactionDescriptor or null.
248   */
249  public static CompactionDescriptor getCompaction(Cell kv) throws IOException {
250    if (isCompactionMarker(kv)) {
251      return CompactionDescriptor.parseFrom(CellUtil.cloneValue(kv));
252    }
253    return null;
254  }
255
256  /**
257   * Returns true if the given cell is a serialized {@link CompactionDescriptor}
258   *
259   * @see #getCompaction(Cell)
260   */
261  public static boolean isCompactionMarker(Cell cell) {
262    return CellUtil.matchingColumn(cell, METAFAMILY, COMPACTION);
263  }
264
265  /**
266   * Create a bulk loader WALEdit
267   *
268   * @param hri                The HRegionInfo for the region in which we are bulk loading
269   * @param bulkLoadDescriptor The descriptor for the Bulk Loader
270   * @return The WALEdit for the BulkLoad
271   */
272  public static WALEdit createBulkLoadEvent(HRegionInfo hri,
273                                            WALProtos.BulkLoadDescriptor bulkLoadDescriptor) {
274    KeyValue kv = new KeyValue(getRowForRegion(hri),
275        METAFAMILY,
276        BULK_LOAD,
277        EnvironmentEdgeManager.currentTime(),
278        bulkLoadDescriptor.toByteArray());
279    return new WALEdit().add(kv);
280  }
281
282  /**
283   * Deserialized and returns a BulkLoadDescriptor from the passed in Cell
284   * @param cell the key value
285   * @return deserialized BulkLoadDescriptor or null.
286   */
287  public static WALProtos.BulkLoadDescriptor getBulkLoadDescriptor(Cell cell) throws IOException {
288    if (CellUtil.matchingColumn(cell, METAFAMILY, BULK_LOAD)) {
289      return WALProtos.BulkLoadDescriptor.parseFrom(CellUtil.cloneValue(cell));
290    }
291    return null;
292  }
