019package org.apache.hadoop.hbase.regionserver;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.hbase.HConstants;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025
026import java.io.IOException;
027import java.util.concurrent.atomic.AtomicBoolean;
028
029/**
030 * The ongoing MemStore Compaction manager, dispatches a solo running compaction and interrupts
031 * the compaction if requested. The compaction is interrupted and stopped by CompactingMemStore,
032 * for example when another compaction needs to be started.
033 * Prior to compaction the MemStoreCompactor evaluates
034 * the compacting ratio and aborts the compaction if it is not worthy.
035 * The MemStoreScanner is used to traverse the compaction pipeline. The MemStoreScanner
036 * is included in internal store scanner, where all compaction logic is implemented.
037 * Threads safety: It is assumed that the compaction pipeline is immutable,
038 * therefore no special synchronization is required.
039 */
040@InterfaceAudience.Private
041class MemStoreCompactor {
042
043  // Option for external guidance whether flattening is allowed
044  static final String MEMSTORE_COMPACTOR_FLATTENING = "hbase.hregion.compacting.memstore.flatten";
045  static final boolean MEMSTORE_COMPACTOR_FLATTENING_DEFAULT = true;
046
047  // Option for external setting of the compacted structure (SkipList, CellArray, etc.)
048  static final String COMPACTING_MEMSTORE_TYPE_KEY = "hbase.hregion.compacting.memstore.type";
049  static final int COMPACTING_MEMSTORE_TYPE_DEFAULT = 2;  // COMPACT_TO_ARRAY_MAP as default
050
051  // What percentage of the duplications is causing compaction?
052  static final String COMPACTION_THRESHOLD_REMAIN_FRACTION
053      = "hbase.hregion.compacting.memstore.comactPercent";
054  static final double COMPACTION_THRESHOLD_REMAIN_FRACTION_DEFAULT = 0.2;
055
056  // Option for external guidance whether the flattening is allowed
057  static final String MEMSTORE_COMPACTOR_AVOID_SPECULATIVE_SCAN
058      = "hbase.hregion.compacting.memstore.avoidSpeculativeScan";
059  static final boolean MEMSTORE_COMPACTOR_AVOID_SPECULATIVE_SCAN_DEFAULT = false;
060
061  private static final Log LOG = LogFactory.getLog(MemStoreCompactor.class);
062  private CompactingMemStore compactingMemStore;
063
064  // a static version of the segment list from the pipeline
065  private VersionedSegmentsList versionedList;
066
067  // a flag raised when compaction is requested to stop
068  private final AtomicBoolean isInterrupted = new AtomicBoolean(false);
069
070  // the limit to the size of the groups to be later provided to MemStoreCompactorIterator
071  private final int compactionKVMax;
072
073  double fraction = 0.8;
074
075  int immutCellsNum = 0;  // number of immutable for compaction cells
076  /**
077   * Types of Compaction
078   */
079  private enum Type {
080    COMPACT_TO_SKIPLIST_MAP,
081    COMPACT_TO_ARRAY_MAP
082  }
083
084  private Type type = Type.COMPACT_TO_ARRAY_MAP;
085
086  public MemStoreCompactor(CompactingMemStore compactingMemStore) {
087    this.compactingMemStore = compactingMemStore;
088    this.compactionKVMax = compactingMemStore.getConfiguration().getInt(
089        HConstants.COMPACTION_KV_MAX, HConstants.COMPACTION_KV_MAX_DEFAULT);
090    this.fraction = 1 - compactingMemStore.getConfiguration().getDouble(
091        COMPACTION_THRESHOLD_REMAIN_FRACTION,
092        COMPACTION_THRESHOLD_REMAIN_FRACTION_DEFAULT);
093  }
094
095  /**----------------------------------------------------------------------
096   * The request to dispatch the compaction asynchronous task.
097   * The method returns true if compaction was successfully dispatched, or false if there
098   * is already an ongoing compaction or no segments to compact.
099   */
100  public boolean start() throws IOException {
101    if (!compactingMemStore.hasImmutableSegments()) return false;  // no compaction on empty
102
103    int t = compactingMemStore.getConfiguration().getInt(COMPACTING_MEMSTORE_TYPE_KEY,
104        COMPACTING_MEMSTORE_TYPE_DEFAULT);
105
106    switch (t) {
107      case 1: type = Type.COMPACT_TO_SKIPLIST_MAP;
108        break;
109      case 2: type = Type.COMPACT_TO_ARRAY_MAP;
110        break;
111      default: throw new RuntimeException("Unknown type " + type); // sanity check
112    }
113
114    // get a snapshot of the list of the segments from the pipeline,
115    // this local copy of the list is marked with specific version
116    versionedList = compactingMemStore.getImmutableSegments();
117    immutCellsNum = versionedList.getNumOfCells();
118
119    if (LOG.isDebugEnabled()) {
120      LOG.debug("Starting the MemStore In-Memory Shrink of type " + type + " for store "
121          + compactingMemStore.getStore().getColumnFamilyName());
122    }
123
124    doCompaction();
125    return true;
126  }
127
128  /**----------------------------------------------------------------------
129  * The request to cancel the compaction asynchronous task
130  * The compaction may still happen if the request was sent too late
131  * Non-blocking request
132  */
133  public void stop() {
134      isInterrupted.compareAndSet(false, true);
135  }
136
137  /**----------------------------------------------------------------------
138  * Close the scanners and clear the pointers in order to allow good
139  * garbage collection
140  */
141  private void releaseResources() {
142    isInterrupted.set(false);
143    versionedList = null;
144  }
145
146  /**----------------------------------------------------------------------
147   * Check whether there are some signs to definitely not to flatten,
148   * returns false if we must compact. If this method returns true we
149   * still need to evaluate the compaction.
150   */
151  private boolean shouldFlatten() {
152    boolean userToFlatten =         // the user configurable option to flatten or not to flatten
153        compactingMemStore.getConfiguration().getBoolean(MEMSTORE_COMPACTOR_FLATTENING,
154            MEMSTORE_COMPACTOR_FLATTENING_DEFAULT);
155    if (userToFlatten==false) {
156      LOG.debug("In-Memory shrink is doing compaction, as user asked to avoid flattening");
157      return false;                 // the user doesn't want to flatten
158    }
159
160    // limit the number of the segments in the pipeline
161    int numOfSegments = versionedList.getNumOfSegments();
162    if (numOfSegments > 3) {        // hard-coded for now as it is going to move to policy
163      LOG.debug("In-Memory shrink is doing compaction, as there already are " + numOfSegments
164          + " segments in the compaction pipeline");
165      return false;                 // to avoid "too many open files later", compact now
166    }
167    // till here we hvae all the signs that it is possible to flatten, run the speculative scan
168    // (if allowed by the user) to check the efficiency of compaction
169    boolean avoidSpeculativeScan =   // the user configurable option to avoid the speculative scan
170        compactingMemStore.getConfiguration().getBoolean(MEMSTORE_COMPACTOR_AVOID_SPECULATIVE_SCAN,
171            MEMSTORE_COMPACTOR_AVOID_SPECULATIVE_SCAN_DEFAULT);
172    if (avoidSpeculativeScan==true) {
173      LOG.debug("In-Memory shrink is doing flattening, as user asked to avoid compaction "
174          + "evaluation");
175      return true;                  // flatten without checking the compaction expedience
176    }
177    try {
178      immutCellsNum = countCellsForCompaction();
179      if (immutCellsNum > fraction * versionedList.getNumOfCells()) {
180        return true;
181      }
182    } catch(Exception e) {
183      return true;
184    }
185    return false;
186  }
187
188  /**----------------------------------------------------------------------
189  * The worker thread performs the compaction asynchronously.
190  * The solo (per compactor) thread only reads the compaction pipeline.
191  * There is at most one thread per memstore instance.
192  */
193  private void doCompaction() {
194    ImmutableSegment result = null;
195    boolean resultSwapped = false;
196
197    try {
198      // PHASE I: estimate the compaction expedience - EVALUATE COMPACTION
199      if (shouldFlatten()) {
200        // too much cells "survive" the possible compaction, we do not want to compact!
201        LOG.debug("In-Memory compaction does not pay off - storing the flattened segment"
202            + " for store: " + compactingMemStore.getFamilyName());
203        // Looking for Segment in the pipeline with SkipList index, to make it flat
204        compactingMemStore.flattenOneSegment(versionedList.getVersion());
205        return;
206      }
207
208      // PHASE II: create the new compacted ImmutableSegment - START COPY-COMPACTION
209      if (!isInterrupted.get()) {
210        result = compact(immutCellsNum);
211      }
212
213      // Phase III: swap the old compaction pipeline - END COPY-COMPACTION
214      if (!isInterrupted.get()) {
215        if (resultSwapped = compactingMemStore.swapCompactedSegments(versionedList, result)) {
216          // update the wal so it can be truncated and not get too long
217          compactingMemStore.updateLowestUnflushedSequenceIdInWAL(true); // only if greater
218        }
219      }
220    } catch (Exception e) {
221      LOG.debug("Interrupting the MemStore in-memory compaction for store "
222          + compactingMemStore.getFamilyName());
223      Thread.currentThread().interrupt();
224    } finally {
225      if ((result != null) && (!resultSwapped)) result.close();
226      releaseResources();
227    }
228
229  }
230
231  /**----------------------------------------------------------------------
232   * The copy-compaction is the creation of the ImmutableSegment (from the relevant type)
233   * based on the Compactor Iterator. The new ImmutableSegment is returned.
234   */
235  private ImmutableSegment compact(int numOfCells) throws IOException {
236
237    LOG.debug("In-Memory compaction does pay off - The estimated number of cells "
238        + "after compaction is " + numOfCells + ", while number of cells before is " + versionedList
239        .getNumOfCells() + ". The fraction of remaining cells should be: " + fraction);
240
241    ImmutableSegment result = null;
242    MemStoreCompactorIterator iterator =
243        new MemStoreCompactorIterator(versionedList.getStoreSegments(),
244            compactingMemStore.getComparator(),
245            compactionKVMax, compactingMemStore.getStore());
246    try {
247      switch (type) {
248      case COMPACT_TO_SKIPLIST_MAP:
249        result = SegmentFactory.instance().createImmutableSegment(
250            compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator);
251        break;
252      case COMPACT_TO_ARRAY_MAP:
253        result = SegmentFactory.instance().createImmutableSegment(
254            compactingMemStore.getConfiguration(), compactingMemStore.getComparator(), iterator,
255            numOfCells, ImmutableSegment.Type.ARRAY_MAP_BASED);
256        break;
257      default: throw new RuntimeException("Unknown type " + type); // sanity check
258      }
259    } finally {
260      iterator.close();
261    }
262
263    return result;
264  }
265
266  /**----------------------------------------------------------------------
267   * Count cells to estimate the efficiency of the future compaction
268   */
269  private int countCellsForCompaction() throws IOException {
270
271    int cnt = 0;
272    MemStoreCompactorIterator iterator =
273        new MemStoreCompactorIterator(
274            versionedList.getStoreSegments(), compactingMemStore.getComparator(),
275            compactionKVMax, compactingMemStore.getStore());
276
277    try {
278      while (iterator.next() != null) {
279        cnt++;
280      }
281    } finally {
282      iterator.close();
283    }
284
285    return cnt;
286  }
