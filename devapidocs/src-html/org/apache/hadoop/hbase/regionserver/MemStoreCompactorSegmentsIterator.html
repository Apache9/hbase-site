020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.Iterator;
025import java.util.List;
026import java.util.OptionalInt;
027
028import org.apache.hadoop.hbase.Cell;
029import org.apache.hadoop.hbase.CellComparator;
030import org.apache.hadoop.hbase.HConstants;
031import org.apache.yetus.audience.InterfaceAudience;
032
033/**
034 * The MemStoreCompactorSegmentsIterator extends MemStoreSegmentsIterator
035 * and performs the scan for compaction operation meaning it is based on SQM
036 */
037@InterfaceAudience.Private
038public class MemStoreCompactorSegmentsIterator extends MemStoreSegmentsIterator {
039
040  private List<Cell> kvs = new ArrayList<>();
041  private boolean hasMore;
042  private Iterator<Cell> kvsIterator;
043
044  // scanner on top of pipeline scanner that uses ScanQueryMatcher
045  private StoreScanner compactingScanner;
046
047  // C-tor
048  public MemStoreCompactorSegmentsIterator(
049      List<ImmutableSegment> segments,
050      CellComparator comparator, int compactionKVMax, Store store
051  ) throws IOException {
052    super(compactionKVMax);
053
054    List<KeyValueScanner> scanners = new ArrayList<KeyValueScanner>();
055    // create the list of scanners to traverse over all the data
056    // no dirty reads here as these are immutable segments
057    int order = segments.size();
058    AbstractMemStore.addToScanners(segments, Integer.MAX_VALUE, order, scanners);
059    // build the scanner based on Query Matcher
060    // reinitialize the compacting scanner for each instance of iterator
061    compactingScanner = createScanner(store, scanners);
062
063    hasMore = compactingScanner.next(kvs, scannerContext);
064
065    if (!kvs.isEmpty()) {
066      kvsIterator = kvs.iterator();
067    }
068
069  }
070
071  @Override
072  public boolean hasNext() {
073    if (kvsIterator == null)  { // for the case when the result is empty
074      return false;
075    }
076    if (!kvsIterator.hasNext()) {
077      // refillKVS() method should be invoked only if !kvsIterator.hasNext()
078      if (!refillKVS()) {
079        return false;
080      }
081    }
082    return kvsIterator.hasNext();
083  }
084
085  @Override
086  public Cell next()  {
087    if (kvsIterator == null)  { // for the case when the result is empty
088      return null;
089    }
090    if (!kvsIterator.hasNext()) {
091      // refillKVS() method should be invoked only if !kvsIterator.hasNext()
092      if (!refillKVS())  return null;
093    }
094    return (!hasMore) ? null : kvsIterator.next();
095  }
096
097  public void close() {
098    compactingScanner.close();
099    compactingScanner = null;
100  }
101
102  @Override
103  public void remove() {
104    throw new UnsupportedOperationException();
105  }
106
107  /**
108   * Creates the scanner for compacting the pipeline.
109   * @return the scanner
110   */
111  private StoreScanner createScanner(Store store, List<KeyValueScanner> scanners)
112      throws IOException {
113    // Get all available versions
114    return new StoreScanner(store, store.getScanInfo(), OptionalInt.of(Integer.MAX_VALUE), scanners,
115        ScanType.COMPACT_RETAIN_DELETES, store.getSmallestReadPoint(), HConstants.OLDEST_TIMESTAMP);
116  }
117
118  /* Refill kev-value set (should be invoked only when KVS is empty)
119   * Returns true if KVS is non-empty */
120  private boolean refillKVS() {
121    kvs.clear();          // clear previous KVS, first initiated in the constructor
122    if (!hasMore) {       // if there is nothing expected next in compactingScanner
123      return false;
124    }
125
126    try {                 // try to get next KVS
127      hasMore = compactingScanner.next(kvs, scannerContext);
128    } catch (IOException ie) {
129      throw new IllegalStateException(ie);
130    }
131
132    if (!kvs.isEmpty() ) {// is the new KVS empty ?
133      kvsIterator = kvs.iterator();
134      return true;
135    } else {
136      // KVS is empty, but hasMore still true?
137      if (hasMore) {      // try to move to next row
138        return refillKVS();
139      }
140
141    }
142    return hasMore;
143  }
144}
