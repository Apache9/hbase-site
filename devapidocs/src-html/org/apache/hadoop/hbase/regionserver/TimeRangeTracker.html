019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.DataInput;
022import java.io.DataOutput;
023import java.io.IOException;
024
025import org.apache.hadoop.hbase.Cell;
026import org.apache.hadoop.hbase.CellUtil;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.io.TimeRange;
029import org.apache.hadoop.io.Writable;
030
031/**
032 * Stores the minimum and maximum timestamp values (both are inclusive).
033 * Can be used to find if any given time range overlaps with its time range
034 * MemStores use this class to track its minimum and maximum timestamps.
035 * When writing StoreFiles, this information is stored in meta blocks and used
036 * at read time to match against the required TimeRange.
037 */
038@InterfaceAudience.Private
039public class TimeRangeTracker implements Writable {
040  static final long INITIAL_MINIMUM_TIMESTAMP = Long.MAX_VALUE;
041  long minimumTimestamp = INITIAL_MINIMUM_TIMESTAMP;
042  long maximumTimestamp = -1;
043
044  /**
045   * Default constructor.
046   * Initializes TimeRange to be null
047   */
048  public TimeRangeTracker() {}
049
050  /**
051   * Copy Constructor
052   * @param trt source TimeRangeTracker
053   */
054  public TimeRangeTracker(final TimeRangeTracker trt) {
055    set(trt.getMinimumTimestamp(), trt.getMaximumTimestamp());
056  }
057
058  public TimeRangeTracker(long minimumTimestamp, long maximumTimestamp) {
059    set(minimumTimestamp, maximumTimestamp);
060  }
061
062  private void set(final long min, final long max) {
063    this.minimumTimestamp = min;
064    this.maximumTimestamp = max;
065  }
066
067  /**
068   * @param l
069   * @return True if we initialized values
070   */
071  private boolean init(final long l) {
072    if (this.minimumTimestamp != INITIAL_MINIMUM_TIMESTAMP) return false;
073    set(l, l);
074    return true;
075  }
076
077  /**
078   * Update the current TimestampRange to include the timestamp from Cell
079   * If the Key is of type DeleteColumn or DeleteFamily, it includes the
080   * entire time range from 0 to timestamp of the key.
081   * @param cell the Cell to include
082   */
083  public void includeTimestamp(final Cell cell) {
084    includeTimestamp(cell.getTimestamp());
085    if (CellUtil.isDeleteColumnOrFamily(cell)) {
086      includeTimestamp(0);
087    }
088  }
089
090  /**
091   * If required, update the current TimestampRange to include timestamp
092   * @param timestamp the timestamp value to include
093   */
094  void includeTimestamp(final long timestamp) {
095    // Do test outside of synchronization block.  Synchronization in here can be problematic
096    // when many threads writing one Store -- they can all pile up trying to add in here.
097    // Happens when doing big write upload where we are hammering on one region.
098    if (timestamp < this.minimumTimestamp) {
099      synchronized (this) {
100        if (!init(timestamp)) {
101          if (timestamp < this.minimumTimestamp) {
102            this.minimumTimestamp = timestamp;
103          }
104        }
105      }
106    } else if (timestamp > this.maximumTimestamp) {
107      synchronized (this) {
108        if (!init(timestamp)) {
109          if (this.maximumTimestamp < timestamp) {
110            this.maximumTimestamp =  timestamp;
111          }
112        }
113      }
114    }
115  }
116
117  /**
118   * Check if the range has any overlap with TimeRange
119   * @param tr TimeRange
120   * @return True if there is overlap, false otherwise
121   */
122  public synchronized boolean includesTimeRange(final TimeRange tr) {
123    return (this.minimumTimestamp < tr.getMax() && this.maximumTimestamp >= tr.getMin());
124  }
125
126  /**
127   * @return the minimumTimestamp
128   */
129  public synchronized long getMinimumTimestamp() {
130    return minimumTimestamp;
131  }
132
133  /**
134   * @return the maximumTimestamp
135   */
136  public synchronized long getMaximumTimestamp() {
137    return maximumTimestamp;
138  }
139
140  public synchronized void write(final DataOutput out) throws IOException {
141    out.writeLong(minimumTimestamp);
142    out.writeLong(maximumTimestamp);
143  }
144
145  public synchronized void readFields(final DataInput in) throws IOException {
146    this.minimumTimestamp = in.readLong();
147    this.maximumTimestamp = in.readLong();
148  }
149
150  @Override
151  public synchronized String toString() {
152    return "[" + minimumTimestamp + "," + maximumTimestamp + "]";
153  }
154}
