019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022import java.util.List;
023
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.client.Scan;
026
027/**
028 * Scanner scans the MOB Store. Coalesce KeyValue stream into {@code List<KeyValue>}
029 * for a single row. It's only used in the compaction of mob-enabled columns.
030 * It outputs the normal cells and delete markers when outputDeleteMarkers is set as true.
031 */
032@InterfaceAudience.Private
033public class MobCompactionStoreScanner extends StoreScanner {
034
035  /*
036   * The delete markers are probably contained in the output of the scanner, for instance the
037   * minor compaction. If outputDeleteMarkers is set as true, these delete markers could be
038   * written to the del file, otherwise it's not allowed.
039   */
040  protected boolean outputDeleteMarkers;
041
042  /**
043   * Used for compactions.<p>
044   *
045   * Opens a scanner across specified StoreFiles.
046   * @param store who we scan
047   * @param scan the spec
048   * @param scanners ancillary scanners
049   * @param smallestReadPoint the readPoint that we should use for tracking
050   *          versions
051   */
052  public MobCompactionStoreScanner(Store store, ScanInfo scanInfo, Scan scan,
053      List<? extends KeyValueScanner> scanners, ScanType scanType, long smallestReadPoint,
054      long earliestPutTs, boolean outputDeleteMarkers) throws IOException {
055    super(store, scanInfo, scan, scanners, scanType, smallestReadPoint, earliestPutTs);
056    this.outputDeleteMarkers = outputDeleteMarkers;
057  }
058
059  /**
060   * Gets whether the delete markers could be written to the del files.
061   * @return True if the delete markers could be written del files, false if it's not allowed.
062   */
063  public boolean isOutputDeleteMarkers() {
064    return this.outputDeleteMarkers;
065  }
066}
