052/**
053 * The MemStore holds in-memory modifications to the Store.  Modifications
054 * are {@link Cell}s.  When asked to flush, current memstore is moved
055 * to snapshot and is cleared.  We continue to serve edits out of new memstore
056 * and backing snapshot until flusher reports in that the flush succeeded. At
057 * this point we let the snapshot go.
058 *  <p>
059 * The MemStore functions should not be called in parallel. Callers should hold
060 *  write and read locks. This is done in {@link HStore}.
061 *  </p>
062 *
063 * TODO: Adjust size of the memstore when we remove items because they have
064 * been deleted.
065 * TODO: With new KVSLS, need to make sure we update HeapSize with difference
066 * in KV size.
067 */
068@InterfaceAudience.Private
069public class DefaultMemStore implements MemStore {
070  private static final Log LOG = LogFactory.getLog(DefaultMemStore.class);
071  static final String USEMSLAB_KEY = "hbase.hregion.memstore.mslab.enabled";
072  private static final boolean USEMSLAB_DEFAULT = true;
073  static final String MSLAB_CLASS_NAME = "hbase.regionserver.mslab.class";

143  /**
144   * Creates a snapshot of the current memstore.
145   * Snapshot must be cleared by call to {@link #clearSnapshot(long)}
146   */
147  @Override
148  public MemStoreSnapshot snapshot() {
149    // If snapshot currently has entries, then flusher failed or didn't call
150    // cleanup.  Log a warning.
151    if (!this.snapshot.isEmpty()) {
152      LOG.warn("Snapshot called again without clearing previous. " +
153          "Doing nothing. Another ongoing flush or did we fail last attempt?");
154    } else {
155      this.snapshotId = EnvironmentEdgeManager.currentTime();
156      this.snapshotSize = keySize();
157      if (!this.cellSet.isEmpty()) {
158        this.snapshot = this.cellSet;
159        this.cellSet = new CellSkipListSet(this.comparator);
160        this.snapshotTimeRangeTracker = this.timeRangeTracker;
161        this.timeRangeTracker = new TimeRangeTracker();
162        // Reset heap to not include any keys
163        this.size.set(DEEP_OVERHEAD);
164        this.snapshotAllocator = this.allocator;
165        // Reset allocator so we get a fresh buffer for the new memstore
166        if (allocator != null) {
167          String className = conf.get(MSLAB_CLASS_NAME, HeapMemStoreLAB.class.getName());
168          this.allocator = ReflectionUtils.instantiateWithCustomCtor(className,
169              new Class[] { Configuration.class }, new Object[] { conf });
170        }
171        timeOfOldestEdit = Long.MAX_VALUE;
172      }
173    }
174    MemStoreSnapshot memStoreSnapshot = new MemStoreSnapshot(this.snapshotId, snapshot.size(), this.snapshotSize,
175        this.snapshotTimeRangeTracker, new CollectionBackedScanner(snapshot, this.comparator),
176        this.tagsPresent);
177    this.tagsPresent = false;
178    return memStoreSnapshot;
179  }

181  /**
182   * The passed snapshot was successfully persisted; it can be let go.
183   * @param id Id of the snapshot to clean out.
184   * @throws UnexpectedStateException
185   * @see #snapshot()
186   */
187  @Override
188  public void clearSnapshot(long id) throws UnexpectedStateException {
189    MemStoreLAB tmpAllocator = null;
190    if (this.snapshotId != id) {
191      throw new UnexpectedStateException("Current snapshot id is " + this.snapshotId + ",passed "
192          + id);
193    }
194    // OK. Passed in snapshot is same as current snapshot. If not-empty,
195    // create a new snapshot and let the old one go.
196    if (!this.snapshot.isEmpty()) {
197      this.snapshot = new CellSkipListSet(this.comparator);
198      this.snapshotTimeRangeTracker = new TimeRangeTracker();
199    }
200    this.snapshotSize = 0;
201    this.snapshotId = -1;
202    if (this.snapshotAllocator != null) {
203      tmpAllocator = this.snapshotAllocator;
204      this.snapshotAllocator = null;
205    }
206    if (tmpAllocator != null) {
207      tmpAllocator.close();
208    }
209  }

237  private boolean addToCellSet(Cell e) {
238    boolean b = this.cellSet.add(e);
239    // In no tags case this NoTagsKeyValue.getTagsLength() is a cheap call.
240    // When we use ACL CP or Visibility CP which deals with Tags during
241    // mutation, the TagRewriteCell.getTagsLength() is a cheaper call. We do not
242    // parse the byte[] to identify the tags length.
243    if(e.getTagsLength() > 0) {
244      tagsPresent = true;
245    }
246    setOldestEditTimeToNow();
247    return b;
248  }
249
250  private boolean removeFromCellSet(Cell e) {
251    boolean b = this.cellSet.remove(e);
252    setOldestEditTimeToNow();
253    return b;
254  }
255
256  void setOldestEditTimeToNow() {
257    if (timeOfOldestEdit == Long.MAX_VALUE) {
258      timeOfOldestEdit = EnvironmentEdgeManager.currentTime();
259    }
260  }

262  /**
263   * Internal version of add() that doesn't clone Cells with the
264   * allocator, and doesn't take the lock.
265   *
266   * Callers should ensure they already have the read lock taken
267   */
268  private long internalAdd(final Cell toAdd) {
269    long s = heapSizeChange(toAdd, addToCellSet(toAdd));
270    timeRangeTracker.includeTimestamp(toAdd);
271    this.size.addAndGet(s);
272    return s;
273  }
274
275  private Cell maybeCloneWithAllocator(Cell cell) {
276    if (allocator == null) {
277      return cell;
278    }
279
280    int len = KeyValueUtil.length(cell);
281    ByteRange alloc = allocator.allocateBytes(len);
282    if (alloc == null) {
283      // The allocation was too large, allocator decided
284      // not to do anything with it.
285      return cell;
286    }
287    assert alloc.getBytes() != null;
288    KeyValueUtil.appendToByteArray(cell, alloc.getBytes(), alloc.getOffset());
289    KeyValue newKv = new KeyValue(alloc.getBytes(), alloc.getOffset(), len);
290    newKv.setSequenceId(cell.getSequenceId());
291    return newKv;
292  }

294  /**
295   * Remove n key from the memstore. Only cells that have the same key and the
296   * same memstoreTS are removed.  It is ok to not update timeRangeTracker
297   * in this call. It is possible that we can optimize this method by using
298   * tailMap/iterator, but since this method is called rarely (only for
299   * error recovery), we can leave those optimization for the future.
300   * @param cell
301   */
302  @Override
303  public void rollback(Cell cell) {
304    // If the key is in the snapshot, delete it. We should not update
305    // this.size, because that tracks the size of only the memstore and
306    // not the snapshot. The flush of this snapshot to disk has not
307    // yet started because Store.flush() waits for all rwcc transactions to
308    // commit before starting the flush to disk.
309    Cell found = this.snapshot.get(cell);
310    if (found != null && found.getSequenceId() == cell.getSequenceId()) {
311      this.snapshot.remove(cell);
312      long sz = heapSizeChange(cell, true);
313      this.snapshotSize -= sz;
314    }
315    // If the key is in the memstore, delete it. Update this.size.
316    found = this.cellSet.get(cell);
317    if (found != null && found.getSequenceId() == cell.getSequenceId()) {
318      removeFromCellSet(cell);
319      long s = heapSizeChange(cell, true);
320      this.size.addAndGet(-s);
321    }
322  }

324  /**
325   * Write a delete
326   * @param deleteCell
327   * @return approximate size of the passed key and value.
328   */
329  @Override
330  public long delete(Cell deleteCell) {
331    long s = 0;
332    Cell toAdd = maybeCloneWithAllocator(deleteCell);
333    s += heapSizeChange(toAdd, addToCellSet(toAdd));
334    timeRangeTracker.includeTimestamp(toAdd);
335    this.size.addAndGet(s);
336    return s;
337  }

340   * @param cell Find the row that comes after this one.  If null, we return the
341   * first.
342   * @return Next row or null if none found.

344  Cell getNextRow(final Cell cell) {

348  /*
349   * @param a
350   * @param b
351   * @return Return lowest of a or b or null if both a and b are null
352   */
353  private Cell getLowest(final Cell a, final Cell b) {
354    if (a == null) {
355      return b;
356    }
357    if (b == null) {
358      return a;
359    }
360    return comparator.compareRows(a, b) <= 0? a: b;
361  }

363  /*
364   * @param key Find row that follows this one.  If null, return first.
365   * @param map Set to look in for a row beyond <code>row</code>.
366   * @return Next row or null if none found.  If one found, will be a new
367   * KeyValue -- can be destroyed by subsequent calls to this method.
368   */
369  private Cell getNextRow(final Cell key,
370      final NavigableSet<Cell> set) {
371    Cell result = null;
372    SortedSet<Cell> tail = key == null? set: set.tailSet(key);
373    // Iterate until we fall into the next row; i.e. move off current row
374    for (Cell cell: tail) {
375      if (comparator.compareRows(cell, key) <= 0)
376        continue;
377      // Note: Not suppressing deletes or expired cells.  Needs to be handled
378      // by higher up functions.
379      result = cell;
380      break;
381    }
382    return result;
383  }

385  /**
386   * Only used by tests. TODO: Remove
387   *
388   * Given the specs of a column, update it, first by inserting a new record,
389   * then removing the old one.  Since there is only 1 KeyValue involved, the memstoreTS
390   * will be set to 0, thus ensuring that they instantly appear to anyone. The underlying
391   * store will ensure that the insert/delete each are atomic. A scanner/reader will either
392   * get the new value, or the old value and all readers will eventually only see the new
393   * value after the old was removed.
394   *
395   * @param row
396   * @param family
397   * @param qualifier
398   * @param newValue
399   * @param now
400   * @return  Timestamp
401   */
402  @Override
403  public long updateColumnValue(byte[] row,
404                                byte[] family,
405                                byte[] qualifier,
406                                long newValue,
407                                long now) {
408    Cell firstCell = KeyValueUtil.createFirstOnRow(row, family, qualifier);
409    // Is there a Cell in 'snapshot' with the same TS? If so, upgrade the timestamp a bit.
410    SortedSet<Cell> snSs = snapshot.tailSet(firstCell);
411    if (!snSs.isEmpty()) {
412      Cell snc = snSs.first();
413      // is there a matching Cell in the snapshot?
414      if (CellUtil.matchingRow(snc, firstCell) && CellUtil.matchingQualifier(snc, firstCell)) {
415        if (snc.getTimestamp() == now) {
416          // poop,
417          now += 1;
418        }
419      }
420    }
421
422    // logic here: the new ts MUST be at least 'now'. But it could be larger if necessary.
423    // But the timestamp should also be max(now, mostRecentTsInMemstore)
424
425    // so we cant add the new Cell w/o knowing what's there already, but we also
426    // want to take this chance to delete some cells. So two loops (sad)
427
428    SortedSet<Cell> ss = cellSet.tailSet(firstCell);
429    for (Cell cell : ss) {
430      // if this isnt the row we are interested in, then bail:
431      if (!CellUtil.matchingColumn(cell, family, qualifier)
432          || !CellUtil.matchingRow(cell, firstCell)) {
433        break; // rows dont match, bail.
434      }
435
436      // if the qualifier matches and it's a put, just RM it out of the cellSet.
437      if (cell.getTypeByte() == KeyValue.Type.Put.getCode() &&
438          cell.getTimestamp() > now && CellUtil.matchingQualifier(firstCell, cell)) {
439        now = cell.getTimestamp();
440      }
441    }
442
443    // create or update (upsert) a new Cell with
444    // 'now' and a 0 memstoreTS == immediately visible
445    List<Cell> cells = new ArrayList<Cell>(1);
446    cells.add(new KeyValue(row, family, qualifier, now, Bytes.toBytes(newValue)));
447    return upsert(cells, 1L);
448  }

450  /**
451   * Update or insert the specified KeyValues.
452   * <p>
453   * For each KeyValue, insert into MemStore.  This will atomically upsert the
454   * value for that row/family/qualifier.  If a KeyValue did already exist,
455   * it will then be removed.
456   * <p>
457   * This is called under row lock, so Get operations will still see updates
458   * atomically.  Scans will only see each KeyValue update as atomic.
459   *
460   * @param readpoint readpoint below which we can safely remove duplicate KVs
461   * @return change in memstore size
462   */
463  @Override
464  public long upsert(Iterable<Cell> cells, long readpoint) {
465    long size = 0;
466    for (Cell cell : cells) {
467      size += upsert(cell, readpoint);
468    }
469    return size;
470  }

472  /**
473   * Inserts the specified KeyValue into MemStore and deletes any existing
474   * versions of the same row/family/qualifier as the specified KeyValue.
475   * <p>
476   * First, the specified KeyValue is inserted into the Memstore.
477   * <p>
478   * If there are any existing KeyValues in this MemStore with the same row,
479   * family, and qualifier, they are removed.
480   * <p>
481   * Callers must hold the read lock.
482   * @param readpoint Smallest outstanding readpoint; below which we can remove duplicate Cells.
483   * @return change in size of MemStore
484   */
485  private long upsert(Cell cell, long readpoint) {
486    // Add the Cell to the MemStore
487    // Use the internalAdd method here since we (a) already have a lock
488    // and (b) cannot safely use the MSLAB here without potentially
489    // hitting OOME - see TestMemStore.testUpsertMSLAB for a
490    // test that triggers the pathological case if we don't avoid MSLAB
491    // here.
492    long addedSize = internalAdd(cell);
493
494    // Get the Cells for the row/family/qualifier regardless of timestamp.
495    // For this case we want to clean up any other puts
496    Cell firstCell = KeyValueUtil.createFirstOnRow(
497        cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
498        cell.getFamilyArray(), cell.getFamilyOffset(), cell.getFamilyLength(),
499        cell.getQualifierArray(), cell.getQualifierOffset(), cell.getQualifierLength());
500    SortedSet<Cell> ss = cellSet.tailSet(firstCell);
501    Iterator<Cell> it = ss.iterator();
502    // Versions visible to oldest scanner.
503    int versionsVisible = 0;
504    while ( it.hasNext() ) {
505      Cell cur = it.next();
506
507      if (cell == cur) {
508        // ignore the one just put in
509        continue;
510      }
511      // check that this is the row and column we are interested in, otherwise bail
512      if (CellUtil.matchingRow(cell, cur) && CellUtil.matchingQualifier(cell, cur)) {
513        // only remove Puts that concurrent scanners cannot possibly see
514        if (cur.getTypeByte() == KeyValue.Type.Put.getCode() &&
515            cur.getSequenceId() <= readpoint) {
516          if (versionsVisible >= 1) {
517            // if we get here we have seen at least one version visible to the oldest scanner,
518            // which means we can prove that no scanner will see this version
519
520            // false means there was a change, so give us the size.
521            long delta = heapSizeChange(cur, true);
522            addedSize -= delta;
523            this.size.addAndGet(-delta);
524            it.remove();
525            setOldestEditTimeToNow();
526          } else {
527            versionsVisible++;
528          }
529        }
530      } else {
531        // past the row or column, done
532        break;
533      }
534    }
535    return addedSize;
536  }

546  /**
547   * Check if this memstore may contain the required keys
548   * @param scan scan
549   * @param store holds reference to cf
550   * @param oldestUnexpiredTS
551   * @return False if the key definitely does not exist in this Memstore
552   */
553  public boolean shouldSeek(Scan scan, Store store, long oldestUnexpiredTS) {
554    byte[] cf = store.getFamily().getName();
555    TimeRange timeRange = scan.getColumnFamilyTimeRange().get(cf);
556    if (timeRange == null) {
557      timeRange = scan.getTimeRange();
558    }
559    return (timeRangeTracker.includesTimeRange(timeRange) ||
560        snapshotTimeRangeTracker.includesTimeRange(timeRange))
561        && (Math.max(timeRangeTracker.getMaximumTimestamp(),
562                     snapshotTimeRangeTracker.getMaximumTimestamp()) >=
563            oldestUnexpiredTS);
564  }

566  /*
567   * MemStoreScanner implements the KeyValueScanner.
568   * It lets the caller scan the contents of a memstore -- both current
569   * map and snapshot.
570   * This behaves as if it were a real scanner but does not maintain position.
571   */
572  protected class MemStoreScanner extends NonLazyKeyValueScanner {
573    // Next row information for either cellSet or snapshot
574    private Cell cellSetNextRow = null;
575    private Cell snapshotNextRow = null;
576
577    // last iterated Cells for cellSet and snapshot (to restore iterator state after reseek)
578    private Cell cellSetItRow = null;
579    private Cell snapshotItRow = null;
580
581    // iterator based scanning.
582    private Iterator<Cell> cellSetIt;
583    private Iterator<Cell> snapshotIt;
584
585    // The cellSet and snapshot at the time of creating this scanner
586    private CellSkipListSet cellSetAtCreation;
587    private CellSkipListSet snapshotAtCreation;
588
589    // the pre-calculated Cell to be returned by peek() or next()
590    private Cell theNext;
591
592    // The allocator and snapshot allocator at the time of creating this scanner
593    volatile MemStoreLAB allocatorAtCreation;
594    volatile MemStoreLAB snapshotAllocatorAtCreation;
595
596    // A flag represents whether could stop skipping Cells for MVCC
597    // if have encountered the next row. Only used for reversed scan
598    private boolean stopSkippingCellsIfNextRow = false;
599
600    private long readPoint;
601
602    /*
603    Some notes...
604
605     So memstorescanner is fixed at creation time. this includes pointers/iterators into
606    existing kvset/snapshot.  during a snapshot creation, the kvset is null, and the
607    snapshot is moved.  since kvset is null there is no point on reseeking on both,
608      we can save us the trouble. During the snapshot->hfile transition, the memstore
609      scanner is re-created by StoreScanner#updateReaders().  StoreScanner should
610      potentially do something smarter by adjusting the existing memstore scanner.
611
612      But there is a greater problem here, that being once a scanner has progressed
613      during a snapshot scenario, we currently iterate past the kvset then 'finish' up.
614      if a scan lasts a little while, there is a chance for new entries in kvset to
615      become available but we will never see them.  This needs to be handled at the
616      StoreScanner level with coordination with MemStoreScanner.
617
618      Currently, this problem is only partly managed: during the small amount of time
619      when the StoreScanner has not yet created a new MemStoreScanner, we will miss
620      the adds to kvset in the MemStoreScanner.
621    */
622
623    MemStoreScanner(long readPoint) {
624      super();
625
626      this.readPoint = readPoint;
627      cellSetAtCreation = cellSet;
628      snapshotAtCreation = snapshot;
629      if (allocator != null) {
630        this.allocatorAtCreation = allocator;
631        this.allocatorAtCreation.incScannerCount();
632      }
633      if (snapshotAllocator != null) {
634        this.snapshotAllocatorAtCreation = snapshotAllocator;
635        this.snapshotAllocatorAtCreation.incScannerCount();
636      }
637      if (Trace.isTracing() && Trace.currentSpan() != null) {
638        Trace.currentSpan().addTimelineAnnotation("Creating MemStoreScanner");
639      }
640    }
641
642    /**
643     * Lock on 'this' must be held by caller.
644     * @param it
645     * @return Next Cell
646     */
647    private Cell getNext(Iterator<Cell> it) {
648      Cell startCell = theNext;
649      Cell v = null;
650      try {
651        while (it.hasNext()) {
652          v = it.next();
653          if (v.getSequenceId() <= this.readPoint) {
654            return v;
655          }
656          if (stopSkippingCellsIfNextRow && startCell != null
657              && comparator.compareRows(v, startCell) > 0) {
658            return null;
659          }
660        }
661
662        return null;
663      } finally {
664        if (v != null) {
665          // in all cases, remember the last Cell iterated to
666          if (it == snapshotIt) {
667            snapshotItRow = v;
668          } else {
669            cellSetItRow = v;
670          }
671        }
672      }
673    }

675    /**
676     *  Set the scanner at the seek key.
677     *  Must be called only once: there is no thread safety between the scanner
678     *   and the memStore.
679     * @param key seek value
680     * @return false if the key is null or if there is no data
681     */
682    @Override
683    public synchronized boolean seek(Cell key) {
684      if (key == null) {
685        close();
686        return false;
687      }
688      // kvset and snapshot will never be null.
689      // if tailSet can't find anything, SortedSet is empty (not null).
690      cellSetIt = cellSetAtCreation.tailSet(key).iterator();
691      snapshotIt = snapshotAtCreation.tailSet(key).iterator();
692      cellSetItRow = null;
693      snapshotItRow = null;
694
695      return seekInSubLists(key);
696    }

699    /**
700     * (Re)initialize the iterators after a seek or a reseek.
701     */
702    private synchronized boolean seekInSubLists(Cell key){
703      cellSetNextRow = getNext(cellSetIt);
704      snapshotNextRow = getNext(snapshotIt);
705
706      // Calculate the next value
707      theNext = getLowest(cellSetNextRow, snapshotNextRow);
708
709      // has data
710      return (theNext != null);
711    }

714    /**
715     * Move forward on the sub-lists set previously by seek.
716     * @param key seek value (should be non-null)
717     * @return true if there is at least one KV to read, false otherwise
718     */
719    @Override
720    public synchronized boolean reseek(Cell key) {
721      /*
722      See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
723      This code is executed concurrently with flush and puts, without locks.
724      Two points must be known when working on this code:
725      1) It's not possible to use the 'kvTail' and 'snapshot'
726       variables, as they are modified during a flush.
727      2) The ideal implementation for performance would use the sub skip list
728       implicitly pointed by the iterators 'kvsetIt' and
729       'snapshotIt'. Unfortunately the Java API does not offer a method to
730       get it. So we remember the last keys we iterated to and restore
731       the reseeked set to at least that point.
732       */
733      cellSetIt = cellSetAtCreation.tailSet(getHighest(key, cellSetItRow)).iterator();
734      snapshotIt = snapshotAtCreation.tailSet(getHighest(key, snapshotItRow)).iterator();
735
736      return seekInSubLists(key);
737    }

741    public synchronized Cell peek() {
742      //DebugPrint.println(" MS@" + hashCode() + " peek = " + getLowest());
743      return theNext;
744    }

746    @Override
747    public synchronized Cell next() {
748      if (theNext == null) {
749          return null;
750      }
751
752      final Cell ret = theNext;
753
754      // Advance one of the iterators
755      if (theNext == cellSetNextRow) {
756        cellSetNextRow = getNext(cellSetIt);
757      } else {
758        snapshotNextRow = getNext(snapshotIt);
759      }
760
761      // Calculate the next value
762      theNext = getLowest(cellSetNextRow, snapshotNextRow);
763
764      //long readpoint = ReadWriteConsistencyControl.getThreadReadPoint();
765      //DebugPrint.println(" MS@" + hashCode() + " next: " + theNext + " next_next: " +
766      //    getLowest() + " threadpoint=" + readpoint);
767      return ret;
768    }

770    /*
771     * Returns the lower of the two key values, or null if they are both null.
772     * This uses comparator.compare() to compare the KeyValue using the memstore
773     * comparator.
774     */
775    private Cell getLowest(Cell first, Cell second) {
776      if (first == null && second == null) {
777        return null;
778      }
779      if (first != null && second != null) {
780        int compare = comparator.compare(first, second);
781        return (compare <= 0 ? first : second);
782      }
783      return (first != null ? first : second);
784    }
785
786    /*
787     * Returns the higher of the two cells, or null if they are both null.
788     * This uses comparator.compare() to compare the Cell using the memstore
789     * comparator.
790     */
791    private Cell getHighest(Cell first, Cell second) {
792      if (first == null && second == null) {
793        return null;
794      }
795      if (first != null && second != null) {
796        int compare = comparator.compare(first, second);
797        return (compare > 0 ? first : second);
798      }
799      return (first != null ? first : second);
800    }

802    public synchronized void close() {
803      this.cellSetNextRow = null;
804      this.snapshotNextRow = null;
805
806      this.cellSetIt = null;
807      this.snapshotIt = null;
808
809      if (allocatorAtCreation != null) {
810        this.allocatorAtCreation.decScannerCount();
811        this.allocatorAtCreation = null;
812      }
813      if (snapshotAllocatorAtCreation != null) {
814        this.snapshotAllocatorAtCreation.decScannerCount();
815        this.snapshotAllocatorAtCreation = null;
816      }
817
818      this.cellSetItRow = null;
819      this.snapshotItRow = null;
820    }

836    /**
837     * Seek scanner to the given key first. If it returns false(means
838     * peek()==null) or scanner's peek row is bigger than row of given key, seek
839     * the scanner to the previous row of given key
840     */
841    @Override
842    public synchronized boolean backwardSeek(Cell key) {
843      seek(key);
844      if (peek() == null || comparator.compareRows(peek(), key) > 0) {
845        return seekToPreviousRow(key);
846      }
847      return true;
848    }

850    /**
851     * Separately get the KeyValue before the specified key from kvset and
852     * snapshotset, and use the row of higher one as the previous row of
853     * specified key, then seek to the first KeyValue of previous row
854     */
855    @Override
856    public synchronized boolean seekToPreviousRow(Cell originalKey) {
857      boolean keepSeeking = false;
858      Cell key = originalKey;
859      do {
860        Cell firstKeyOnRow = CellUtil.createFirstOnRow(key);
861        SortedSet<Cell> cellHead = cellSetAtCreation.headSet(firstKeyOnRow);
862        Cell cellSetBeforeRow = cellHead.isEmpty() ? null : cellHead.last();
863        SortedSet<Cell> snapshotHead = snapshotAtCreation
864            .headSet(firstKeyOnRow);
865        Cell snapshotBeforeRow = snapshotHead.isEmpty() ? null : snapshotHead
866            .last();
867        Cell lastCellBeforeRow = getHighest(cellSetBeforeRow, snapshotBeforeRow);
868        if (lastCellBeforeRow == null) {
869          theNext = null;
870          return false;
871        }
872        Cell firstKeyOnPreviousRow = CellUtil.createFirstOnRow(lastCellBeforeRow);
873        this.stopSkippingCellsIfNextRow = true;
874        seek(firstKeyOnPreviousRow);
875        this.stopSkippingCellsIfNextRow = false;
876        if (peek() == null
877            || comparator.compareRows(peek(), firstKeyOnPreviousRow) > 0) {
878          keepSeeking = true;
879          key = firstKeyOnPreviousRow;
880          continue;
881        } else {
882          keepSeeking = false;
883        }
884      } while (keepSeeking);
885      return true;
886    }
887
888    @Override
889    public synchronized boolean seekToLastRow() {
890      Cell first = cellSetAtCreation.isEmpty() ? null : cellSetAtCreation
891          .last();
892      Cell second = snapshotAtCreation.isEmpty() ? null
893          : snapshotAtCreation.last();
894      Cell higherCell = getHighest(first, second);
895      if (higherCell == null) {
896        return false;
897      }
898      Cell firstCellOnLastRow = CellUtil.createFirstOnRow(higherCell);
899      if (seek(firstCellOnLastRow)) {
900        return true;
901      } else {
902        return seekToPreviousRow(higherCell);
903      }
904
905    }
