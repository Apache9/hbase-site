056/**
057 * Regions store data for a certain region of a table.  It stores all columns
058 * for each row. A given table consists of one or more Regions.
059 *
060 * <p>An Region is defined by its table and its key extent.
061 *
062 * <p>Locking at the Region level serves only one purpose: preventing the
063 * region from being closed (and consequently split) while other operations
064 * are ongoing. Each row level operation obtains both a row lock and a region
065 * read lock for the duration of the operation. While a scanner is being
066 * constructed, getScanner holds a read lock. If the scanner is successfully
067 * constructed, it holds a read lock until it is closed. A close takes out a
068 * write lock and consequently will block for ongoing operations and will block
069 * new operations from starting while the close is in progress.
070 */
071@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
072@InterfaceStability.Evolving
073public interface Region extends ConfigurationObserver {

681  /**
682   * Flush the cache.
683   *
684   * <p>When this method is called the cache will be flushed unless:
685   * <ol>
686   *   <li>the cache is empty</li>
687   *   <li>the region is closed.</li>
688   *   <li>a flush is already in progress</li>
689   *   <li>writes are disabled</li>
690   * </ol>
691   *
692   * <p>This method may block for some time, so it should not be called from a
693   * time-sensitive thread.
694   * @param force whether we want to force a flush of all stores
695   * @return FlushResult indicating whether the flush was successful or not and if
696   * the region needs compacting
697   *
698   * @throws IOException general io exceptions
699   * because a snapshot was not properly persisted.
700   */
701  FlushResult flush(boolean force) throws IOException;
702
703  /**
704   * Synchronously compact all stores in the region.
705   * <p>This operation could block for a long time, so don't call it from a
706   * time-sensitive thread.
707   * <p>Note that no locks are taken to prevent possible conflicts between
708   * compaction and splitting activities. The regionserver does not normally compact
709   * and split in parallel. However by calling this method you may introduce
710   * unexpected and unhandled concurrency. Don't do this unless you know what
711   * you are doing.
712   *
713   * @param majorCompaction True to force a major compaction regardless of thresholds
714   * @throws IOException
715   */
716  void compact(final boolean majorCompaction) throws IOException;

719   * Trigger major compaction on all stores in the region.
720   * <p>
721   * Compaction will be performed asynchronously to this call by the RegionServer's
722   * CompactSplitThread. See also {@link Store#triggerMajorCompaction()}
723   * @throws IOException
724   */
725  void triggerMajorCompaction() throws IOException;

728   * @return if a given region is in compaction now.

730  CompactionState getCompactionState();

733  void waitForFlushesAndCompactions();
