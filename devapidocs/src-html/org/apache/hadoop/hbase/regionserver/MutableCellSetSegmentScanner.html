018package org.apache.hadoop.hbase.regionserver;
019
020import java.io.IOException;
021import java.util.Iterator;
022import java.util.SortedSet;
023
024import org.apache.hadoop.hbase.Cell;
025import org.apache.hadoop.hbase.CellUtil;
026import org.apache.hadoop.hbase.classification.InterfaceAudience;
027
028/**
029 * A scanner of a single cells segment {@link MutableCellSetSegment}.
030 */
031@InterfaceAudience.Private
032class MutableCellSetSegmentScanner extends SegmentScanner {
033
034  // the observed structure
035  private final MutableCellSetSegment segment;
036  // the highest relevant MVCC
037  private long readPoint;
038  // the current iterator that can be reinitialized by
039  // seek(), backwardSeek(), or reseek()
040  private Iterator<Cell> iter;
041  // the pre-calculated cell to be returned by peek()
042  private Cell current = null;
043  // or next()
044  // A flag represents whether could stop skipping KeyValues for MVCC
045  // if have encountered the next row. Only used for reversed scan
046  private boolean stopSkippingKVsIfNextRow = false;
047  // last iterated KVs by seek (to restore the iterator state after reseek)
048  private Cell last = null;
049
050  public MutableCellSetSegmentScanner(MutableCellSetSegment segment, long readPoint) {
051    super();
052    this.segment = segment;
053    this.readPoint = readPoint;
054    iter = segment.iterator();
055    // the initialization of the current is required for working with heap of SegmentScanners
056    current = getNext();
057    //increase the reference count so the underlying structure will not be de-allocated
058    this.segment.incScannerCount();
059  }
060
061  /**
062   * Look at the next Cell in this scanner, but do not iterate the scanner
063   * @return the currently observed Cell
064   */
065  @Override
066  public Cell peek() {          // sanity check, the current should be always valid
067    if (current!=null && current.getSequenceId() > readPoint) {
068      throw new RuntimeException("current is invalid: read point is "+readPoint+", " +
069          "while current sequence id is " +current.getSequenceId());
070    }
071
072    return current;
073  }
074
075  /**
076   * Return the next Cell in this scanner, iterating the scanner
077   * @return the next Cell or null if end of scanner
078   */
079  @Override
080  public Cell next() throws IOException {
081    Cell oldCurrent = current;
082    current = getNext();                  // update the currently observed Cell
083    return oldCurrent;
084  }
085
086  /**
087   * Seek the scanner at or after the specified Cell.
088   * @param cell seek value
089   * @return true if scanner has values left, false if end of scanner
090   */
091  @Override
092  public boolean seek(Cell cell) throws IOException {
093    if(cell == null) {
094      close();
095      return false;
096    }
097    // restart the iterator from new key
098    iter = segment.tailSet(cell).iterator();
099    // last is going to be reinitialized in the next getNext() call
100    last = null;
101    current = getNext();
102    return (current != null);
103  }
104
105  /**
106   * Reseek the scanner at or after the specified KeyValue.
107   * This method is guaranteed to seek at or after the required key only if the
108   * key comes after the current position of the scanner. Should not be used
109   * to seek to a key which may come before the current position.
110   *
111   * @param cell seek value (should be non-null)
112   * @return true if scanner has values left, false if end of scanner
113   */
114  @Override
115  public boolean reseek(Cell cell) throws IOException {
116
117    /*
118    See HBASE-4195 & HBASE-3855 & HBASE-6591 for the background on this implementation.
119    This code is executed concurrently with flush and puts, without locks.
120    The ideal implementation for performance would use the sub skip list implicitly
121    pointed by the iterator. Unfortunately the Java API does not offer a method to
122    get it. So we remember the last keys we iterated to and restore
123    the reseeked set to at least that point.
124    */
125    iter = segment.tailSet(getHighest(cell, last)).iterator();
126    current = getNext();
127    return (current != null);
128  }
129
130  /**
131   * Seek the scanner at or before the row of specified Cell, it firstly
132   * tries to seek the scanner at or after the specified Cell, return if
133   * peek KeyValue of scanner has the same row with specified Cell,
134   * otherwise seek the scanner at the first Cell of the row which is the
135   * previous row of specified KeyValue
136   *
137   * @param key seek Cell
138   * @return true if the scanner is at the valid KeyValue, false if such Cell does not exist
139   */
140  @Override
141  public boolean backwardSeek(Cell key) throws IOException {
142    seek(key);    // seek forward then go backward
143    if (peek() == null || segment.compareRows(peek(), key) > 0) {
144      return seekToPreviousRow(key);
145    }
146    return true;
147  }
148
149  /**
150   * Seek the scanner at the first Cell of the row which is the previous row
151   * of specified key
152   *
153   * @param cell seek value
154   * @return true if the scanner at the first valid Cell of previous row,
155   *     false if not existing such Cell
156   */
157  @Override
158  public boolean seekToPreviousRow(Cell cell) throws IOException {
159    boolean keepSeeking = false;
160    Cell key = cell;
161
162    do {
163      Cell firstKeyOnRow = CellUtil.createFirstOnRow(key);
164      SortedSet<Cell> cellHead = segment.headSet(firstKeyOnRow);
165      Cell lastCellBeforeRow = cellHead.isEmpty() ? null : cellHead.last();
166      if (lastCellBeforeRow == null) {
167        current = null;
168        return false;
169      }
170      Cell firstKeyOnPreviousRow = CellUtil.createFirstOnRow(lastCellBeforeRow);
171      this.stopSkippingKVsIfNextRow = true;
172      seek(firstKeyOnPreviousRow);
173      this.stopSkippingKVsIfNextRow = false;
174      if (peek() == null
175          || segment.getComparator().compareRows(peek(), firstKeyOnPreviousRow) > 0) {
176        keepSeeking = true;
177        key = firstKeyOnPreviousRow;
178        continue;
179      } else {
180        keepSeeking = false;
181      }
182    } while (keepSeeking);
183    return true;
184  }
185
186  /**
187   * Seek the scanner at the first KeyValue of last row
188   *
189   * @return true if scanner has values left, false if the underlying data is empty
190   */
191  @Override
192  public boolean seekToLastRow() throws IOException {
193    Cell higherCell = segment.isEmpty() ? null : segment.last();
194    if (higherCell == null) {
195      return false;
196    }
197
198    Cell firstCellOnLastRow = CellUtil.createFirstOnRow(higherCell);
199
200    if (seek(firstCellOnLastRow)) {
201      return true;
202    } else {
203      return seekToPreviousRow(higherCell);
204    }
205  }
206
207  @Override protected Segment getSegment() {
208    return segment;
209  }
210
211  /********************* Private Methods **********************/
212
213  /**
214   * Private internal method for iterating over the segment,
215   * skipping the cells with irrelevant MVCC
216   */
217  private Cell getNext() {
218    Cell startKV = current;
219    Cell next = null;
220
221    try {
222      while (iter.hasNext()) {
223        next = iter.next();
224        if (next.getSequenceId() <= this.readPoint) {
225          return next;                    // skip irrelevant versions
226        }
227        if (stopSkippingKVsIfNextRow &&   // for backwardSeek() stay in the
228            startKV != null &&        // boundaries of a single row
229            segment.compareRows(next, startKV) > 0) {
230          return null;
231        }
232      } // end of while
233
234      return null; // nothing found
235    } finally {
236      if (next != null) {
237        // in all cases, remember the last KV we iterated to, needed for reseek()
238        last = next;
239      }
240    }
241  }
242
243  /**
244   * Private internal method that returns the higher of the two key values, or null
245   * if they are both null
246   */
247  private Cell getHighest(Cell first, Cell second) {
248    if (first == null && second == null) {
249      return null;
250    }
251    if (first != null && second != null) {
252      int compare = segment.compare(first, second);
253      return (compare > 0 ? first : second);
254    }
255    return (first != null ? first : second);
256  }
