019package org.apache.hadoop.hbase.regionserver;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025import org.apache.hadoop.hbase.client.Scan;
026
027/**
028 * An abstraction for store segment scanner.
029 */
030@InterfaceAudience.Private
031public abstract class SegmentScanner implements KeyValueScanner {
032
033  private long sequenceID = Long.MAX_VALUE;
034
035  protected abstract Segment getSegment();
036
037  /**
038   * Get the sequence id associated with this KeyValueScanner. This is required
039   * for comparing multiple files (or memstore segments) scanners to find out
040   * which one has the latest data.
041   *
042   */
043  @Override
044  public long getSequenceID() {
045    return sequenceID;
046  }
047
048  /**
049   * Close the KeyValue scanner.
050   */
051  @Override
052  public void close() {
053    getSegment().decScannerCount();
054  }
055
056  /**
057   * This functionality should be resolved in the higher level which is
058   * MemStoreScanner, currently returns true as default. Doesn't throw
059   * IllegalStateException in order not to change the signature of the
060   * overridden method
061   */
062  @Override
063  public boolean shouldUseScanner(Scan scan, Store store, long oldestUnexpiredTS) {
064    return true;
065  }
066  /**
067   * This scanner is working solely on the in-memory MemStore therefore this
068   * interface is not relevant.
069   */
070  @Override
071  public boolean requestSeek(Cell c, boolean forward, boolean useBloom)
072      throws IOException {
073
074    throw new IllegalStateException(
075        "requestSeek cannot be called on MutableCellSetSegmentScanner");
076  }
077
078  /**
079   * This scanner is working solely on the in-memory MemStore and doesn't work on
080   * store files, MutableCellSetSegmentScanner always does the seek,
081   * therefore always returning true.
082   */
083  @Override
084  public boolean realSeekDone() {
085    return true;
086  }
087
088  /**
089   * This function should be never called on scanners that always do real seek operations (i.e. most
090   * of the scanners and also this one). The easiest way to achieve this is to call
091   * {@link #realSeekDone()} first.
092   */
093  @Override
094  public void enforceSeek() throws IOException {
095    throw new IllegalStateException(
096        "enforceSeek cannot be called on MutableCellSetSegmentScanner");
097  }
098
099  /**
100   * @return true if this is a file scanner. Otherwise a memory scanner is assumed.
101   */
102  @Override
103  public boolean isFileScanner() {
104    return false;
105  }
106
107  /**
108   * @return the next key in the index (the key to seek to the next block)
109   *     if known, or null otherwise
110   *     Not relevant for in-memory scanner
111   */
112  @Override
113  public Cell getNextIndexedKey() {
114    return null;
115  }
116
117  /**
118   * Called after a batch of rows scanned (RPC) and set to be returned to client. Any in between
119   * cleanup can be done here. Nothing to be done for MutableCellSetSegmentScanner.
120   */
121  @Override
122  public void shipped() throws IOException {
123    // do nothing
124  }
125
126  /**
127   * Set the sequence id of the scanner.
128   * This is used to determine an order between memory segment scanners.
129   * @param x a unique sequence id
130   */
131  public void setSequenceID(long x) {
132    sequenceID = x;
133  }
134
135  /**
136   * Returns whether the given scan should seek in this segment
137   * @return whether the given scan should seek in this segment
138   */
139  public boolean shouldSeek(Scan scan, long oldestUnexpiredTS) {
140    return getSegment().shouldSeek(scan,oldestUnexpiredTS);
141  }
142
143  //debug method
144  @Override
145  public String toString() {
146    String res = "Store segment scanner of type "+this.getClass().getName()+"; ";
147    res += "sequence id "+getSequenceID()+"; ";
148    res += getSegment().toString();
149    return res;
150  }
151
152}
