104/**
105 * A Store holds a column family in a Region.  Its a memstore and a set of zero
106 * or more StoreFiles, which stretch backwards over time.
107 *
108 * <p>There's no reason to consider append-logging at this level; all logging
109 * and locking is handled at the HRegion level.  Store just provides
110 * services to manage sets of StoreFiles.  One of the most important of those
111 * services is compaction services where files are aggregated once they pass
112 * a configurable threshold.
113 *
114 * <p>Locking and transactions are handled at a higher level.  This API should
115 * not be called directly but by an HRegion manager.

210  /**
211   * Constructor
212   * @param region
213   * @param family HColumnDescriptor for this column
214   * @param confParam configuration object
215   * failed.  Can be null.
216   * @throws IOException
217   */
218  protected HStore(final HRegion region, final ColumnFamilyDescriptor family,
219      final Configuration confParam) throws IOException {
220
221    this.fs = region.getRegionFileSystem();
222
223    // Assemble the store's home directory and Ensure it exists.
224    fs.createStoreDir(family.getNameAsString());
225    this.region = region;
226    this.family = family;
227    // 'conf' renamed to 'confParam' b/c we use this.conf in the constructor
228    // CompoundConfiguration will look for keys in reverse order of addition, so we'd
229    // add global config first, then table and cf overrides, then cf metadata.
230    this.conf = new CompoundConfiguration()
231      .add(confParam)
232      .addBytesMap(region.getTableDescriptor().getValues())
233      .addStringMap(family.getConfiguration())
234      .addBytesMap(family.getValues());
235    this.blocksize = family.getBlocksize();
236
237    // set block storage policy for store directory
238    String policyName = family.getStoragePolicy();
239    if (null == policyName) {
240      policyName = this.conf.get(BLOCK_STORAGE_POLICY_KEY, DEFAULT_BLOCK_STORAGE_POLICY);
241    }
242    this.fs.setStoragePolicy(family.getNameAsString(), policyName.trim());
243
244    this.dataBlockEncoder =
245        new HFileDataBlockEncoderImpl(family.getDataBlockEncoding());
246
247    this.comparator = region.getCellComparator();
248    // used by ScanQueryMatcher
249    long timeToPurgeDeletes =
250        Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
251    LOG.trace("Time to purge deletes set to " + timeToPurgeDeletes +
252        "ms in store " + this);
253    // Get TTL
254    long ttl = determineTTLFromFamily(family);
255    // Why not just pass a HColumnDescriptor in here altogether?  Even if have
256    // to clone it?
257    scanInfo = new ScanInfo(conf, family, ttl, timeToPurgeDeletes, this.comparator);
258    MemoryCompactionPolicy inMemoryCompaction = family.getInMemoryCompaction();
259    if(inMemoryCompaction == null) {
260      inMemoryCompaction = MemoryCompactionPolicy.valueOf(
261          conf.get(CompactingMemStore.COMPACTING_MEMSTORE_TYPE_KEY,
262              CompactingMemStore.COMPACTING_MEMSTORE_TYPE_DEFAULT));
263    }
264    String className;
265    switch (inMemoryCompaction) {
266      case BASIC :
267      case EAGER :
268        Class<? extends CompactingMemStore> clz = conf.getClass(MEMSTORE_CLASS_NAME,
269          CompactingMemStore.class, CompactingMemStore.class);
270        className = clz.getName();
271        this.memstore = ReflectionUtils.newInstance(clz, new Object[] { conf, this.comparator, this,
272            this.getHRegion().getRegionServicesForStores(), inMemoryCompaction});
273        break;
274      case NONE :
275      default:
276        className = DefaultMemStore.class.getName();
277        this.memstore = ReflectionUtils.instantiateWithCustomCtor(className, new Class[] {
278        Configuration.class, CellComparator.class }, new Object[] { conf, this.comparator });
279    }
280    LOG.info("Memstore class name is " + className);
281    this.offPeakHours = OffPeakHours.getInstance(conf);
282
283    // Setting up cache configuration for this family
284    createCacheConf(family);
285
286    this.verifyBulkLoads = conf.getBoolean("hbase.hstore.bulkload.verify", false);
287
288    this.blockingFileCount =
289        conf.getInt(BLOCKING_STOREFILES_KEY, DEFAULT_BLOCKING_STOREFILE_COUNT);
290    this.compactionCheckMultiplier = conf.getInt(
291        COMPACTCHECKER_INTERVAL_MULTIPLIER_KEY, DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
292    if (this.compactionCheckMultiplier <= 0) {
293      LOG.error("Compaction check period multiplier must be positive, setting default: "
294          + DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER);
295      this.compactionCheckMultiplier = DEFAULT_COMPACTCHECKER_INTERVAL_MULTIPLIER;
296    }
297
298    if (HStore.closeCheckInterval == 0) {
299      HStore.closeCheckInterval = conf.getInt(
300          "hbase.hstore.close.check.interval", 10*1000*1000 /* 10 MB */);
301    }
302
303    this.storeEngine = createStoreEngine(this, this.conf, this.comparator);
304    this.storeEngine.getStoreFileManager().loadFiles(loadStoreFiles());
305
306    // Initialize checksum type from name. The names are CRC32, CRC32C, etc.
307    this.checksumType = getChecksumType(conf);
308    // initilize bytes per checksum
309    this.bytesPerChecksum = getBytesPerChecksum(conf);
310    flushRetriesNumber = conf.getInt(
311        "hbase.hstore.flush.retries.number", DEFAULT_FLUSH_RETRIES_NUMBER);
312    pauseTime = conf.getInt(HConstants.HBASE_SERVER_PAUSE, HConstants.DEFAULT_HBASE_SERVER_PAUSE);
313    if (flushRetriesNumber <= 0) {
314      throw new IllegalArgumentException(
315          "hbase.hstore.flush.retries.number must be > 0, not "
316              + flushRetriesNumber);
317    }
318    cryptoContext = EncryptionUtil.createEncryptionContext(conf, family);
319  }

342  /**
343   * @param family
344   * @return TTL in seconds of the specified family
345   */
346  public static long determineTTLFromFamily(final ColumnFamilyDescriptor family) {
347    // HCD.getTimeToLive returns ttl in seconds.  Convert to milliseconds.
348    long ttl = family.getTimeToLive();
349    if (ttl == HConstants.FOREVER) {
350      // Default is unlimited ttl.
351      ttl = Long.MAX_VALUE;
352    } else if (ttl == -1) {
353      ttl = Long.MAX_VALUE;
354    } else {
355      // Second -> ms adjust for user data
356      ttl *= 1000;
357    }
358    return ttl;
359  }

524  private List<StoreFile> openStoreFiles(Collection<StoreFileInfo> files) throws IOException {
525    if (files == null || files.isEmpty()) {
526      return new ArrayList<>();
527    }
528    // initialize the thread pool for opening store files in parallel..
529    ThreadPoolExecutor storeFileOpenerThreadPool =
530      this.region.getStoreFileOpenAndCloseThreadPool("StoreFileOpenerThread-" +
531          this.getColumnFamilyName());
532    CompletionService<StoreFile> completionService = new ExecutorCompletionService<>(storeFileOpenerThreadPool);
533
534    int totalValidStoreFile = 0;
535    for (final StoreFileInfo storeFileInfo: files) {
536      // open each store file in parallel
537      completionService.submit(new Callable<StoreFile>() {
538        @Override
539        public StoreFile call() throws IOException {
540          StoreFile storeFile = createStoreFileAndReader(storeFileInfo);
541          return storeFile;
542        }
543      });
544      totalValidStoreFile++;
545    }
546
547    ArrayList<StoreFile> results = new ArrayList<>(files.size());
548    IOException ioe = null;
549    try {
550      for (int i = 0; i < totalValidStoreFile; i++) {
551        try {
552          Future<StoreFile> future = completionService.take();
553          StoreFile storeFile = future.get();
554          if (storeFile != null) {
555            long length = storeFile.getReader().length();
556            this.storeSize += length;
557            this.totalUncompressedBytes += storeFile.getReader().getTotalUncompressedBytes();
558            if (LOG.isDebugEnabled()) {
559              LOG.debug("loaded " + storeFile.toStringDetailed());
560            }
561            results.add(storeFile);
562          }
563        } catch (InterruptedException e) {
564          if (ioe == null) ioe = new InterruptedIOException(e.getMessage());
565        } catch (ExecutionException e) {
566          if (ioe == null) ioe = new IOException(e.getCause());
567        }
568      }
569    } finally {
570      storeFileOpenerThreadPool.shutdownNow();
571    }
572    if (ioe != null) {
573      // close StoreFile readers
574      boolean evictOnClose =
575          cacheConf != null? cacheConf.shouldEvictOnClose(): true;
576      for (StoreFile file : results) {
577        try {
578          if (file != null) file.closeReader(evictOnClose);
579        } catch (IOException e) {
580          LOG.warn(e.getMessage());
581        }
582      }
583      throw ioe;
584    }
585
586    return results;
587  }

611  /**
612   * Checks the underlying store files, and opens the files that  have not
613   * been opened, and removes the store file readers for store files no longer
614   * available. Mainly used by secondary region replicas to keep up to date with
615   * the primary region files.
616   * @throws IOException
617   */
618  private void refreshStoreFilesInternal(Collection<StoreFileInfo> newFiles) throws IOException {
619    StoreFileManager sfm = storeEngine.getStoreFileManager();
620    Collection<StoreFile> currentFiles = sfm.getStorefiles();
621    Collection<StoreFile> compactedFiles = sfm.getCompactedfiles();
622    if (currentFiles == null) currentFiles = Collections.emptySet();
623    if (newFiles == null) newFiles = Collections.emptySet();
624    if (compactedFiles == null) compactedFiles = Collections.emptySet();
625
626    HashMap<StoreFileInfo, StoreFile> currentFilesSet = new HashMap<>(currentFiles.size());
627    for (StoreFile sf : currentFiles) {
628      currentFilesSet.put(sf.getFileInfo(), sf);
629    }
630    HashMap<StoreFileInfo, StoreFile> compactedFilesSet =
631        new HashMap<StoreFileInfo, StoreFile>(compactedFiles.size());
632    for (StoreFile sf : compactedFiles) {
633      compactedFilesSet.put(sf.getFileInfo(), sf);
634    }
635
636    Set<StoreFileInfo> newFilesSet = new HashSet<StoreFileInfo>(newFiles);
637    // Exclude the files that have already been compacted
638    newFilesSet = Sets.difference(newFilesSet, compactedFilesSet.keySet());
639    Set<StoreFileInfo> toBeAddedFiles = Sets.difference(newFilesSet, currentFilesSet.keySet());
640    Set<StoreFileInfo> toBeRemovedFiles = Sets.difference(currentFilesSet.keySet(), newFilesSet);
641
642    if (toBeAddedFiles.isEmpty() && toBeRemovedFiles.isEmpty()) {
643      return;
644    }
645
646    LOG.info("Refreshing store files for region " + this.getRegionInfo().getRegionNameAsString()
647      + " files to add: " + toBeAddedFiles + " files to remove: " + toBeRemovedFiles);
648
649    Set<StoreFile> toBeRemovedStoreFiles = new HashSet<>(toBeRemovedFiles.size());
650    for (StoreFileInfo sfi : toBeRemovedFiles) {
651      toBeRemovedStoreFiles.add(currentFilesSet.get(sfi));
652    }
653
654    // try to open the files
655    List<StoreFile> openedFiles = openStoreFiles(toBeAddedFiles);
656
657    // propogate the file changes to the underlying store file manager
658    replaceStoreFiles(toBeRemovedStoreFiles, openedFiles); //won't throw an exception
659
660    // Advance the memstore read point to be at least the new store files seqIds so that
661    // readers might pick it up. This assumes that the store is not getting any writes (otherwise
662    // in-flight transactions might be made visible)
663    if (!toBeAddedFiles.isEmpty()) {
664      region.getMVCC().advanceTo(this.getMaxSequenceId());
665    }
666
667    completeCompaction(toBeRemovedStoreFiles);
668  }

745  /**
746   * This throws a WrongRegionException if the HFile does not fit in this region, or an
747   * InvalidHFileException if the HFile is not valid.
748   */
749  public void assertBulkLoadHFileOk(Path srcPath) throws IOException {
750    HFile.Reader reader  = null;
751    try {
752      LOG.info("Validating hfile at " + srcPath + " for inclusion in "
753          + "store " + this + " region " + this.getRegionInfo().getRegionNameAsString());
754      reader = HFile.createReader(srcPath.getFileSystem(conf), srcPath, cacheConf,
755        isPrimaryReplicaStore(), conf);
756      reader.loadFileInfo();
757
758      byte[] firstKey = reader.getFirstRowKey();
759      Preconditions.checkState(firstKey != null, "First key can not be null");
760      Cell lk = reader.getLastKey();
761      Preconditions.checkState(lk != null, "Last key can not be null");
762      byte[] lastKey =  CellUtil.cloneRow(lk);
763
764      LOG.debug("HFile bounds: first=" + Bytes.toStringBinary(firstKey) +
765          " last=" + Bytes.toStringBinary(lastKey));
766      LOG.debug("Region bounds: first=" +
767          Bytes.toStringBinary(getRegionInfo().getStartKey()) +
768          " last=" + Bytes.toStringBinary(getRegionInfo().getEndKey()));
769
770      if (!this.getRegionInfo().containsRange(firstKey, lastKey)) {
771        throw new WrongRegionException(
772            "Bulk load file " + srcPath.toString() + " does not fit inside region "
773            + this.getRegionInfo().getRegionNameAsString());
774      }
775
776      if(reader.length() > conf.getLong(HConstants.HREGION_MAX_FILESIZE,
777          HConstants.DEFAULT_MAX_FILE_SIZE)) {
778        LOG.warn("Trying to bulk load hfile " + srcPath.toString() + " with size: " +
779            reader.length() + " bytes can be problematic as it may lead to oversplitting.");
780      }
781
782      if (verifyBulkLoads) {
783        long verificationStartTime = EnvironmentEdgeManager.currentTime();
784        LOG.info("Full verification started for bulk load hfile: " + srcPath.toString());
785        Cell prevCell = null;
786        HFileScanner scanner = reader.getScanner(false, false, false);
787        scanner.seekTo();
788        do {
789          Cell cell = scanner.getCell();
790          if (prevCell != null) {
791            if (comparator.compareRows(prevCell, cell) > 0) {
792              throw new InvalidHFileException("Previous row is greater than"
793                  + " current row: path=" + srcPath + " previous="
794                  + CellUtil.getCellKeyAsString(prevCell) + " current="
795                  + CellUtil.getCellKeyAsString(cell));
796            }
797            if (CellComparator.compareFamilies(prevCell, cell) != 0) {
798              throw new InvalidHFileException("Previous key had different"
799                  + " family compared to current key: path=" + srcPath
800                  + " previous="
801                  + Bytes.toStringBinary(prevCell.getFamilyArray(), prevCell.getFamilyOffset(),
802                      prevCell.getFamilyLength())
803                  + " current="
804                  + Bytes.toStringBinary(cell.getFamilyArray(), cell.getFamilyOffset(),
805                      cell.getFamilyLength()));
806            }
807          }
808          prevCell = cell;
809        } while (scanner.next());
810      LOG.info("Full verification complete for bulk load hfile: " + srcPath.toString()
811         + " took " + (EnvironmentEdgeManager.currentTime() - verificationStartTime)
812         + " ms");
813      }
814    } finally {
815      if (reader != null) reader.close();
816    }
817  }

884  @Override
885  public ImmutableCollection<StoreFile> close() throws IOException {
886    this.archiveLock.lock();
887    this.lock.writeLock().lock();
888    try {
889      // Clear so metrics doesn't find them.
890      ImmutableCollection<StoreFile> result = storeEngine.getStoreFileManager().clearFiles();
891      Collection<StoreFile> compactedfiles =
892          storeEngine.getStoreFileManager().clearCompactedFiles();
893      // clear the compacted files
894      if (compactedfiles != null && !compactedfiles.isEmpty()) {
895        removeCompactedfiles(compactedfiles);
896      }
897      if (!result.isEmpty()) {
898        // initialize the thread pool for closing store files in parallel.
899        ThreadPoolExecutor storeFileCloserThreadPool = this.region
900            .getStoreFileOpenAndCloseThreadPool("StoreFileCloserThread-"
901                + this.getColumnFamilyName());
902
903        // close each store file in parallel
904        CompletionService<Void> completionService =
905          new ExecutorCompletionService<>(storeFileCloserThreadPool);
906        for (final StoreFile f : result) {
907          completionService.submit(new Callable<Void>() {
908            @Override
909            public Void call() throws IOException {
910              boolean evictOnClose =
911                  cacheConf != null? cacheConf.shouldEvictOnClose(): true;
912              f.closeReader(evictOnClose);
913              return null;
914            }
915          });
916        }
917
918        IOException ioe = null;
919        try {
920          for (int i = 0; i < result.size(); i++) {
921            try {
922              Future<Void> future = completionService.take();
923              future.get();
924            } catch (InterruptedException e) {
925              if (ioe == null) {
926                ioe = new InterruptedIOException();
927                ioe.initCause(e);
928              }
929            } catch (ExecutionException e) {
930              if (ioe == null) ioe = new IOException(e.getCause());
931            }
932          }
933        } finally {
934          storeFileCloserThreadPool.shutdownNow();
935        }
936        if (ioe != null) throw ioe;
937      }
938      LOG.info("Closed " + this);
939      return result;
940    } finally {
941      this.lock.writeLock().unlock();
942      this.archiveLock.unlock();
943    }
944  }

960  /**
961   * Write out current snapshot. Presumes {@link #snapshot()} has been called previously.
962   * @param logCacheFlushId flush sequence number
963   * @param snapshot
964   * @param status
965   * @param throughputController
966   * @return The path name of the tmp file to which the store was flushed
967   * @throws IOException if exception occurs during process
968   */
969  protected List<Path> flushCache(final long logCacheFlushId, MemStoreSnapshot snapshot,
970      MonitoredTask status, ThroughputController throughputController) throws IOException {
971    // If an exception happens flushing, we let it out without clearing
972    // the memstore snapshot.  The old snapshot will be returned when we say
973    // 'snapshot', the next time flush comes around.
974    // Retry after catching exception when flushing, otherwise server will abort
975    // itself
976    StoreFlusher flusher = storeEngine.getStoreFlusher();
977    IOException lastException = null;
978    for (int i = 0; i < flushRetriesNumber; i++) {
979      try {
980        List<Path> pathNames =
981            flusher.flushSnapshot(snapshot, logCacheFlushId, status, throughputController);
982        Path lastPathName = null;
983        try {
984          for (Path pathName : pathNames) {
985            lastPathName = pathName;
986            validateStoreFile(pathName);
987          }
988          return pathNames;
989        } catch (Exception e) {
990          LOG.warn("Failed validating store file " + lastPathName + ", retrying num=" + i, e);
991          if (e instanceof IOException) {
992            lastException = (IOException) e;
993          } else {
994            lastException = new IOException(e);
995          }
996        }
997      } catch (IOException e) {
998        LOG.warn("Failed flushing store file, retrying num=" + i, e);
999        lastException = e;
1000      }
1001      if (lastException != null && i < (flushRetriesNumber - 1)) {
1002        try {
1003          Thread.sleep(pauseTime);
1004        } catch (InterruptedException e) {
1005          IOException iie = new InterruptedIOException();
1006          iie.initCause(e);
1007          throw iie;
1008        }
1009      }
1010    }
1011    throw lastException;
1012  }

1263  /**
1264   * Compact the StoreFiles.  This method may take some time, so the calling
1265   * thread must be able to block for long periods.
1266   *
1267   * <p>During this time, the Store can work as usual, getting values from
1268   * StoreFiles and writing new StoreFiles from the memstore.
1269   *
1270   * Existing StoreFiles are not destroyed until the new compacted StoreFile is
1271   * completely written-out to disk.
1272   *
1273   * <p>The compactLock prevents multiple simultaneous compactions.
1274   * The structureLock prevents us from interfering with other write operations.
1275   *
1276   * <p>We don't want to hold the structureLock for the whole time, as a compact()
1277   * can be lengthy and we want to allow cache-flushes during this period.
1278   *
1279   * <p> Compaction event should be idempotent, since there is no IO Fencing for
1280   * the region directory in hdfs. A region server might still try to complete the
1281   * compaction after it lost the region. That is why the following events are carefully
1282   * ordered for a compaction:
1283   *  1. Compaction writes new files under region/.tmp directory (compaction output)
1284   *  2. Compaction atomically moves the temporary file under region directory
1285   *  3. Compaction appends a WAL edit containing the compaction input and output files.
1286   *  Forces sync on WAL.
1287   *  4. Compaction deletes the input files from the region directory.
1288   *
1289   * Failure conditions are handled like this:
1290   *  - If RS fails before 2, compaction wont complete. Even if RS lives on and finishes
1291   *  the compaction later, it will only write the new data file to the region directory.
1292   *  Since we already have this data, this will be idempotent but we will have a redundant
1293   *  copy of the data.
1294   *  - If RS fails between 2 and 3, the region will have a redundant copy of the data. The
1295   *  RS that failed won't be able to finish snyc() for WAL because of lease recovery in WAL.
1296   *  - If RS fails after 3, the region region server who opens the region will pick up the
1297   *  the compaction marker from the WAL and replay it by removing the compaction input files.
1298   *  Failed RS can also attempt to delete those files, but the operation will be idempotent
1299   *
1300   * See HBASE-2231 for details.
1301   *
1302   * @param compaction compaction details obtained from requestCompaction()
1303   * @throws IOException
1304   * @return Storefile we compacted into or null if we failed or opted out early.
1305   */
1306  @Override
1307  public List<StoreFile> compact(CompactionContext compaction,
1308      ThroughputController throughputController) throws IOException {
1309    return compact(compaction, throughputController, null);
1310  }

1312  @Override
1313  public List<StoreFile> compact(CompactionContext compaction,
1314    ThroughputController throughputController, User user) throws IOException {
1315    assert compaction != null;
1316    List<StoreFile> sfs = null;
1317    CompactionRequest cr = compaction.getRequest();
1318    try {
1319      // Do all sanity checking in here if we have a valid CompactionRequest
1320      // because we need to clean up after it on the way out in a finally
1321      // block below
1322      long compactionStartTime = EnvironmentEdgeManager.currentTime();
1323      assert compaction.hasSelection();
1324      Collection<StoreFile> filesToCompact = cr.getFiles();
1325      assert !filesToCompact.isEmpty();
1326      synchronized (filesCompacting) {
1327        // sanity check: we're compacting files that this store knows about
1328        // TODO: change this to LOG.error() after more debugging
1329        Preconditions.checkArgument(filesCompacting.containsAll(filesToCompact));
1330      }
1331
1332      // Ready to go. Have list of files to compact.
1333      LOG.info("Starting compaction of " + filesToCompact +
1334        " into tmpdir=" + fs.getTempDir() + ", totalSize=" +
1335          TraditionalBinaryPrefix.long2String(cr.getSize(), "", 1));
1336
1337      // Commence the compaction.
1338      List<Path> newFiles = compaction.compact(throughputController, user);
1339
1340      long outputBytes = 0L;
1341      // TODO: get rid of this!
1342      if (!this.conf.getBoolean("hbase.hstore.compaction.complete", true)) {
1343        LOG.warn("hbase.hstore.compaction.complete is set to false");
1344        sfs = new ArrayList<>(newFiles.size());
1345        final boolean evictOnClose =
1346            cacheConf != null? cacheConf.shouldEvictOnClose(): true;
1347        for (Path newFile : newFiles) {
1348          // Create storefile around what we wrote with a reader on it.
1349          StoreFile sf = createStoreFileAndReader(newFile);
1350          sf.closeReader(evictOnClose);
1351          sfs.add(sf);
1352        }
1353        return sfs;
1354      }
1355      // Do the steps necessary to complete the compaction.
1356      sfs = moveCompatedFilesIntoPlace(cr, newFiles, user);
1357      writeCompactionWalRecord(filesToCompact, sfs);
1358      replaceStoreFiles(filesToCompact, sfs);
1359      if (cr.isMajor()) {
1360        majorCompactedCellsCount += getCompactionProgress().totalCompactingKVs;
1361        majorCompactedCellsSize += getCompactionProgress().totalCompactedSize;
1362      } else {
1363        compactedCellsCount += getCompactionProgress().totalCompactingKVs;
1364        compactedCellsSize += getCompactionProgress().totalCompactedSize;
1365      }
1366
1367      for (StoreFile sf : sfs) {
1368        outputBytes += sf.getReader().length();
1369      }
1370
1371      // At this point the store will use new files for all new scanners.
1372      completeCompaction(filesToCompact); // update store size.
1373
1374      long now = EnvironmentEdgeManager.currentTime();
1375      if (region.getRegionServerServices() != null
1376          && region.getRegionServerServices().getMetrics() != null) {
1377        region.getRegionServerServices().getMetrics().updateCompaction(cr.isMajor(),
1378          now - compactionStartTime, cr.getFiles().size(), newFiles.size(), cr.getSize(),
1379          outputBytes);
1380      }
1381
1382      logCompactionEndMessage(cr, sfs, now, compactionStartTime);
1383      return sfs;
1384    } finally {
1385      finishCompactionRequest(cr);
1386    }
1387  }

1493  /**
1494   * Call to complete a compaction. Its for the case where we find in the WAL a compaction
1495   * that was not finished.  We could find one recovering a WAL after a regionserver crash.
1496   * See HBASE-2231.
1497   * @param compaction
1498   */
1499  public void replayCompactionMarker(CompactionDescriptor compaction,
1500      boolean pickCompactionFiles, boolean removeFiles)
1501      throws IOException {
1502    LOG.debug("Completing compaction from the WAL marker");
1503    List<String> compactionInputs = compaction.getCompactionInputList();
1504    List<String> compactionOutputs = Lists.newArrayList(compaction.getCompactionOutputList());
1505
1506    // The Compaction Marker is written after the compaction is completed,
1507    // and the files moved into the region/family folder.
1508    //
1509    // If we crash after the entry is written, we may not have removed the
1510    // input files, but the output file is present.
1511    // (The unremoved input files will be removed by this function)
1512    //
1513    // If we scan the directory and the file is not present, it can mean that:
1514    //   - The file was manually removed by the user
1515    //   - The file was removed as consequence of subsequent compaction
1516    // so, we can't do anything with the "compaction output list" because those
1517    // files have already been loaded when opening the region (by virtue of
1518    // being in the store's folder) or they may be missing due to a compaction.
1519
1520    String familyName = this.getColumnFamilyName();
1521    List<String> inputFiles = new ArrayList<>(compactionInputs.size());
1522    for (String compactionInput : compactionInputs) {
1523      Path inputPath = fs.getStoreFilePath(familyName, compactionInput);
1524      inputFiles.add(inputPath.getName());
1525    }
1526
1527    //some of the input files might already be deleted
1528    List<StoreFile> inputStoreFiles = new ArrayList<>(compactionInputs.size());
1529    for (StoreFile sf : this.getStorefiles()) {
1530      if (inputFiles.contains(sf.getPath().getName())) {
1531        inputStoreFiles.add(sf);
1532      }
1533    }
1534
1535    // check whether we need to pick up the new files
1536    List<StoreFile> outputStoreFiles = new ArrayList<>(compactionOutputs.size());
1537
1538    if (pickCompactionFiles) {
1539      for (StoreFile sf : this.getStorefiles()) {
1540        compactionOutputs.remove(sf.getPath().getName());
1541      }
1542      for (String compactionOutput : compactionOutputs) {
1543        StoreFileInfo storeFileInfo = fs.getStoreFileInfo(getColumnFamilyName(), compactionOutput);
1544        StoreFile storeFile = createStoreFileAndReader(storeFileInfo);
1545        outputStoreFiles.add(storeFile);
1546      }
1547    }
1548
1549    if (!inputStoreFiles.isEmpty() || !outputStoreFiles.isEmpty()) {
1550      LOG.info("Replaying compaction marker, replacing input files: " +
1551          inputStoreFiles + " with output files : " + outputStoreFiles);
1552      this.replaceStoreFiles(inputStoreFiles, outputStoreFiles);
1553      this.completeCompaction(inputStoreFiles);
1554    }
1555  }

1558   * This method tries to compact N recent files for testing.
1559   * Note that because compacting "recent" files only makes sense for some policies,
1560   * e.g. the default one, it assumes default policy is used. It doesn't use policy,
1561   * but instead makes a compaction candidate list by itself.
1562   * @param N Number of files.
1563   */
1564  public void compactRecentForTestingAssumingDefaultPolicy(int N) throws IOException {
1565    List<StoreFile> filesToCompact;
1566    boolean isMajor;
1567
1568    this.lock.readLock().lock();
1569    try {
1570      synchronized (filesCompacting) {
1571        filesToCompact = Lists.newArrayList(storeEngine.getStoreFileManager().getStorefiles());
1572        if (!filesCompacting.isEmpty()) {
1573          // exclude all files older than the newest file we're currently
1574          // compacting. this allows us to preserve contiguity (HBASE-2856)
1575          StoreFile last = filesCompacting.get(filesCompacting.size() - 1);
1576          int idx = filesToCompact.indexOf(last);

1582          throw new RuntimeException("Not enough files");
1583        }

1585        filesToCompact = filesToCompact.subList(count - N, count);
1586        isMajor = (filesToCompact.size() == storeEngine.getStoreFileManager().getStorefileCount());
1587        filesCompacting.addAll(filesToCompact);
1588        Collections.sort(filesCompacting, storeEngine.getStoreFileManager()
1589            .getStoreFileComparator());
1590      }
1591    } finally {
1592      this.lock.readLock().unlock();
1593    }
1594
1595    try {
1596      // Ready to go. Have list of files to compact.
1597      List<Path> newFiles = ((DefaultCompactor)this.storeEngine.getCompactor())
1598          .compactForTesting(filesToCompact, isMajor);
1599      for (Path newFile: newFiles) {
1600        // Move the compaction into place.

1602        if (this.getCoprocessorHost() != null) {
1603          this.getCoprocessorHost().postCompact(this, sf, null, null);
1604        }
1605        replaceStoreFiles(filesToCompact, Lists.newArrayList(sf));
1606        completeCompaction(filesToCompact);
1607      }
1608    } finally {
1609      synchronized (filesCompacting) {
1610        filesCompacting.removeAll(filesToCompact);
1611      }
1612    }
1613  }

1648  @Override
1649  public CompactionContext requestCompaction(int priority, final CompactionRequest baseRequest,
1650      User user) throws IOException {
1651    // don't even select for compaction if writes are disabled
1652    if (!this.areWritesEnabled()) {
1653      return null;
1654    }
1655
1656    // Before we do compaction, try to get rid of unneeded files to simplify things.
1657    removeUnneededFiles();
1658
1659    final CompactionContext compaction = storeEngine.createCompaction();
1660    CompactionRequest request = null;
1661    this.lock.readLock().lock();
1662    try {
1663      synchronized (filesCompacting) {
1664        // First, see if coprocessor would want to override selection.
1665        if (this.getCoprocessorHost() != null) {
1666          final List<StoreFile> candidatesForCoproc = compaction.preSelect(this.filesCompacting);
1667          boolean override = false;
1668          override = getCoprocessorHost().preCompactSelection(this, candidatesForCoproc,
1669              baseRequest, user);
1670          if (override) {
1671            // Coprocessor is overriding normal file selection.
1672            compaction.forceSelect(new CompactionRequest(candidatesForCoproc));
1673          }
1674        }
1675
1676        // Normal case - coprocessor is not overriding file selection.
1677        if (!compaction.hasSelection()) {
1678          boolean isUserCompaction = priority == Store.PRIORITY_USER;
1679          boolean mayUseOffPeak = offPeakHours.isOffPeakHour() &&
1680              offPeakCompactionTracker.compareAndSet(false, true);
1681          try {
1682            compaction.select(this.filesCompacting, isUserCompaction,
1683              mayUseOffPeak, forceMajor && filesCompacting.isEmpty());
1684          } catch (IOException e) {
1685            if (mayUseOffPeak) {
1686              offPeakCompactionTracker.set(false);
1687            }
1688            throw e;
1689          }
1690          assert compaction.hasSelection();
1691          if (mayUseOffPeak && !compaction.getRequest().isOffPeak()) {
1692            // Compaction policy doesn't want to take advantage of off-peak.
1693            offPeakCompactionTracker.set(false);
1694          }
1695        }
1696        if (this.getCoprocessorHost() != null) {
1697          this.getCoprocessorHost().postCompactSelection(
1698              this, ImmutableList.copyOf(compaction.getRequest().getFiles()), baseRequest, user);
1699        }
1700
1701        // Selected files; see if we have a compaction with some custom base request.
1702        if (baseRequest != null) {
1703          // Update the request with what the system thinks the request should be;
1704          // its up to the request if it wants to listen.
1705          compaction.forceSelect(
1706              baseRequest.combineWith(compaction.getRequest()));
1707        }
1708        // Finally, we have the resulting files list. Check if we have any files at all.
1709        request = compaction.getRequest();
1710        final Collection<StoreFile> selectedFiles = request.getFiles();
1711        if (selectedFiles.isEmpty()) {
1712          return null;
1713        }
1714
1715        addToCompactingFiles(selectedFiles);
1716
1717        // If we're enqueuing a major, clear the force flag.
1718        this.forceMajor = this.forceMajor && !request.isMajor();
1719
1720        // Set common request properties.
1721        // Set priority, either override value supplied by caller or from store.
1722        request.setPriority((priority != Store.NO_PRIORITY) ? priority : getCompactPriority());
1723        request.setDescription(getRegionInfo().getRegionNameAsString(), getColumnFamilyName());
1724      }
1725    } finally {
1726      this.lock.readLock().unlock();
1727    }
1728
1729    LOG.debug(getRegionInfo().getEncodedName() + " - " + getColumnFamilyName()
1730        + ": Initiating " + (request.isMajor() ? "major" : "minor") + " compaction"
1731        + (request.isAllFiles() ? " (all files)" : ""));
1732    this.region.reportCompactionRequestStart(request.isMajor());
1733    return compaction;
1734  }

1747  private void removeUnneededFiles() throws IOException {
1748    if (!conf.getBoolean("hbase.store.delete.expired.storefile", true)) return;
1749    if (getColumnFamilyDescriptor().getMinVersions() > 0) {
1750      LOG.debug("Skipping expired store file removal due to min version being " +
1751          getColumnFamilyDescriptor().getMinVersions());
1752      return;
1753    }
1754    this.lock.readLock().lock();
1755    Collection<StoreFile> delSfs = null;
1756    try {
1757      synchronized (filesCompacting) {
1758        long cfTtl = getStoreFileTtl();
1759        if (cfTtl != Long.MAX_VALUE) {
1760          delSfs = storeEngine.getStoreFileManager().getUnneededFiles(
1761              EnvironmentEdgeManager.currentTime() - cfTtl, filesCompacting);
1762          addToCompactingFiles(delSfs);
1763        }
1764      }
1765    } finally {
1766      this.lock.readLock().unlock();
1767    }
1768    if (delSfs == null || delSfs.isEmpty()) return;
1769
1770    Collection<StoreFile> newFiles = new ArrayList<>(); // No new files.
1771    writeCompactionWalRecord(delSfs, newFiles);
1772    replaceStoreFiles(delSfs, newFiles);
1773    completeCompaction(delSfs);
1774    LOG.info("Completed removal of " + delSfs.size() + " unnecessary (expired) file(s) in "
1775        + this + " of " + this.getRegionInfo().getRegionNameAsString()
1776        + "; total size for store is " + TraditionalBinaryPrefix.long2String(storeSize, "", 1));
1777  }

2531        }
2532      } finally {
2533        lock.readLock().unlock();
2534      }
2535      if (copyCompactedfiles != null && !copyCompactedfiles.isEmpty()) {
2536        removeCompactedfiles(copyCompactedfiles);
2537      }
2538    } finally {
2539      archiveLock.unlock();
2540    }
2541  }

2543  /**
2544   * Archives and removes the compacted files
2545   * @param compactedfiles The compacted files in this store that are not active in reads
2546   * @throws IOException
2547   */
2548  private void removeCompactedfiles(Collection<StoreFile> compactedfiles)
2549      throws IOException {
2550    final List<StoreFile> filesToRemove = new ArrayList<>(compactedfiles.size());
2551    for (final StoreFile file : compactedfiles) {
2552      synchronized (file) {
2553        try {
2554          StoreFileReader r = file.getReader();
2555          if (r == null) {
2556            if (LOG.isDebugEnabled()) {
2557              LOG.debug("The file " + file + " was closed but still not archived.");
2558            }
2559            filesToRemove.add(file);
2560            continue;
2561          }
2562          if (file.isCompactedAway() && !file.isReferencedInReads()) {
2563            // Even if deleting fails we need not bother as any new scanners won't be
2564            // able to use the compacted file as the status is already compactedAway
2565            if (LOG.isTraceEnabled()) {
2566              LOG.trace("Closing and archiving the file " + file.getPath());
2567            }
2568            r.close(true);
2569            // Just close and return
2570            filesToRemove.add(file);
2571          }
2572        } catch (Exception e) {
2573          LOG.error(
2574            "Exception while trying to close the compacted store file " + file.getPath().getName());
2575        }
2576      }
2577    }
2578    if (this.isPrimaryReplicaStore()) {
2579      // Only the primary region is allowed to move the file to archive.
2580      // The secondary region does not move the files to archive. Any active reads from
2581      // the secondary region will still work because the file as such has active readers on it.
2582      if (!filesToRemove.isEmpty()) {
2583        if (LOG.isDebugEnabled()) {
2584          LOG.debug("Moving the files " + filesToRemove + " to archive");
2585        }
2586        // Only if this is successful it has to be removed
2587        try {
2588          this.fs.removeStoreFiles(this.getColumnFamilyDescriptor().getNameAsString(), filesToRemove);
2589        } catch (FailedArchiveException fae) {
2590          // Even if archiving some files failed, we still need to clear out any of the
2591          // files which were successfully archived.  Otherwise we will receive a
2592          // FileNotFoundException when we attempt to re-archive them in the next go around.
2593          Collection<Path> failedFiles = fae.getFailedFiles();
2594          Iterator<StoreFile> iter = filesToRemove.iterator();
2595          while (iter.hasNext()) {
2596            if (failedFiles.contains(iter.next().getPath())) {
2597              iter.remove();
2598            }
2599          }
2600          if (!filesToRemove.isEmpty()) {
2601            clearCompactedfiles(filesToRemove);
2602          }
2603          throw fae;
2604        }
2605      }
2606    }
2607    if (!filesToRemove.isEmpty()) {
2608      // Clear the compactedfiles from the store file manager
2609      clearCompactedfiles(filesToRemove);
2610    }
2611  }
