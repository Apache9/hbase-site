419  /**
420   * Instantiated as a scanner lease. If the lease times out, the scanner is
421   * closed
422   */
423  private class ScannerListener implements LeaseListener {
424    private final String scannerName;
425
426    ScannerListener(final String n) {
427      this.scannerName = n;
428    }
429
430    @Override
431    public void leaseExpired() {
432      RegionScannerHolder rsh = scanners.remove(this.scannerName);
433      if (rsh != null) {
434        RegionScanner s = rsh.s;
435        LOG.info("Scanner " + this.scannerName + " lease expired on region "
436          + s.getRegionInfo().getRegionNameAsString());
437        Region region = null;
438        try {
439          region = regionServer.getRegion(s.getRegionInfo().getRegionName());
440          if (region != null && region.getCoprocessorHost() != null) {
441            region.getCoprocessorHost().preScannerClose(s);
442          }
443        } catch (IOException e) {
444          LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
445        } finally {
446          try {
447            s.close();
448            if (region != null && region.getCoprocessorHost() != null) {
449              region.getCoprocessorHost().postScannerClose(s);
450            }
451          } catch (IOException e) {
452            LOG.error("Closing scanner for " + s.getRegionInfo().getRegionNameAsString(), e);
453          }
454        }
455      } else {
456        LOG.warn("Scanner " + this.scannerName + " lease expired, but no related" +
457          " scanner found, hence no chance to close that related scanner!");
458      }
459    }
460  }

476  /**
477   * Starts the nonce operation for a mutation, if needed.
478   * @param mutation Mutation.
479   * @param nonceGroup Nonce group from the request.
480   * @returns whether to proceed this mutation.
481   */
482  private boolean startNonceOperation(final MutationProto mutation, long nonceGroup)
483      throws IOException {
484    if (regionServer.nonceManager == null || !mutation.hasNonce()) return true;
485    boolean canProceed = false;
486    try {
487      canProceed = regionServer.nonceManager.startOperation(
488        nonceGroup, mutation.getNonce(), regionServer);
489    } catch (InterruptedException ex) {
490      throw new InterruptedIOException("Nonce start operation interrupted");
491    }
492    return canProceed;
493  }

501  private void endNonceOperation(final MutationProto mutation,

546   *
547   * @param region
548   * @param actions
549   * @param cellScanner if non-null, the mutation data -- the Cell content.
550   * @throws IOException
551   */
552  private void mutateRows(final Region region,
553      final List<ClientProtos.Action> actions,
554      final CellScanner cellScanner, RegionActionResult.Builder builder) throws IOException {
555    if (!region.getRegionInfo().isMetaTable()) {
556      regionServer.cacheFlusher.reclaimMemStoreMemory();
557    }
558    RowMutations rm = null;
559    int i = 0;
560    ClientProtos.ResultOrException.Builder resultOrExceptionOrBuilder =
561        ClientProtos.ResultOrException.newBuilder();
562    for (ClientProtos.Action action: actions) {
563      if (action.hasGet()) {
564        throw new DoNotRetryIOException("Atomic put and/or delete only, not a Get=" +
565          action.getGet());
566      }
567      MutationType type = action.getMutation().getMutateType();
568      if (rm == null) {
569        rm = new RowMutations(action.getMutation().getRow().toByteArray(), actions.size());
570      }
571      switch (type) {
572        case PUT:
573          Put put = ProtobufUtil.toPut(action.getMutation(), cellScanner);
574          checkCellSizeLimit(region, put);
575          rm.add(put);
576          break;
577        case DELETE:
578          rm.add(ProtobufUtil.toDelete(action.getMutation(), cellScanner));
579          break;
580        default:
581          throw new DoNotRetryIOException("Atomic put and/or delete only, not " + type.name());
582      }
583      // To unify the response format with doNonAtomicRegionMutation and read through client's
584      // AsyncProcess we have to add an empty result instance per operation
585      resultOrExceptionOrBuilder.clear();
586      resultOrExceptionOrBuilder.setIndex(i++);
587      builder.addResultOrException(
588          resultOrExceptionOrBuilder.build());
589    }
590    region.mutateRow(rm);
591  }

652   * Execute an append mutation.
653   *
654   * @param region
655   * @param m
656   * @param cellScanner
657   * @return result to return to client if default operation should be
658   * bypassed as indicated by RegionObserver, null otherwise
659   * @throws IOException
660   */
661  private Result append(final Region region, final OperationQuota quota,
662      final MutationProto mutation, final CellScanner cellScanner, long nonceGroup,
663      ActivePolicyEnforcement spaceQuota)
664      throws IOException {
665    long before = EnvironmentEdgeManager.currentTime();
666    Append append = ProtobufUtil.toAppend(mutation, cellScanner);
667    checkCellSizeLimit(region, append);
668    spaceQuota.getPolicyEnforcement(region).check(append);
669    quota.addMutation(append);
670    Result r = null;
671    if (region.getCoprocessorHost() != null) {
672      r = region.getCoprocessorHost().preAppend(append);
673    }
674    if (r == null) {
675      boolean canProceed = startNonceOperation(mutation, nonceGroup);
676      boolean success = false;
677      try {
678        long nonce = mutation.hasNonce() ? mutation.getNonce() : HConstants.NO_NONCE;
679        if (canProceed) {
680          r = region.append(append, nonceGroup, nonce);
681        } else {
682          // convert duplicate append to get
683          List<Cell> results = region.get(ProtobufUtil.toGet(mutation, cellScanner), false,
684            nonceGroup, nonce);
685          r = Result.create(results);
686        }
687        success = true;
688      } finally {
689        if (canProceed) {
690          endNonceOperation(mutation, nonceGroup, success);
691        }
692      }
693      if (region.getCoprocessorHost() != null) {
694        region.getCoprocessorHost().postAppend(append, r);
695      }
696    }
697    if (regionServer.metricsRegionServer != null) {
698      regionServer.metricsRegionServer.updateAppend(
699        EnvironmentEdgeManager.currentTime() - before);
700    }
701    return r;
702  }

755  /**
756   * Run through the regionMutation <code>rm</code> and per Mutation, do the work, and then when
757   * done, add an instance of a {@link ResultOrException} that corresponds to each Mutation.
758   * @param region
759   * @param actions
760   * @param cellScanner
761   * @param builder
762   * @param cellsToReturn  Could be null. May be allocated in this method.  This is what this
763   * method returns as a 'result'.
764   * @param closeCallBack the callback to be used with multigets
765   * @param context the current RpcCallContext
766   * @return Return the <code>cellScanner</code> passed
767   */
768  private List<CellScannable> doNonAtomicRegionMutation(final Region region,
769      final OperationQuota quota, final RegionAction actions, final CellScanner cellScanner,
770      final RegionActionResult.Builder builder, List<CellScannable> cellsToReturn, long nonceGroup,
771      final RegionScannersCloseCallBack closeCallBack, RpcCallContext context,
772      ActivePolicyEnforcement spaceQuotaEnforcement) {
773    // Gather up CONTIGUOUS Puts and Deletes in this mutations List.  Idea is that rather than do
774    // one at a time, we instead pass them in batch.  Be aware that the corresponding
775    // ResultOrException instance that matches each Put or Delete is then added down in the
776    // doBatchOp call.  We should be staying aligned though the Put and Delete are deferred/batched
777    List<ClientProtos.Action> mutations = null;
778    long maxQuotaResultSize = Math.min(maxScannerResultSize, quota.getReadAvailable());
779    IOException sizeIOE = null;
780    Object lastBlock = null;
781    ClientProtos.ResultOrException.Builder resultOrExceptionBuilder = ResultOrException.newBuilder();
782    boolean hasResultOrException = false;
783    for (ClientProtos.Action action : actions.getActionList()) {
784      hasResultOrException = false;
785      resultOrExceptionBuilder.clear();
786      try {
787        Result r = null;
788
789        if (context != null
790            && context.isRetryImmediatelySupported()
791            && (context.getResponseCellSize() > maxQuotaResultSize
792              || context.getResponseBlockSize() + context.getResponseExceptionSize()
793              > maxQuotaResultSize)) {
794
795          // We're storing the exception since the exception and reason string won't
796          // change after the response size limit is reached.
797          if (sizeIOE == null ) {
798            // We don't need the stack un-winding do don't throw the exception.
799            // Throwing will kill the JVM's JIT.
800            //
801            // Instead just create the exception and then store it.
802            sizeIOE = new MultiActionResultTooLarge("Max size exceeded"
803                + " CellSize: " + context.getResponseCellSize()
804                + " BlockSize: " + context.getResponseBlockSize());
805
806            // Only report the exception once since there's only one request that
807            // caused the exception. Otherwise this number will dominate the exceptions count.
808            rpcServer.getMetrics().exception(sizeIOE);
809          }
810
811          // Now that there's an exception is known to be created
812          // use it for the response.
813          //
814          // This will create a copy in the builder.
815          hasResultOrException = true;
816          NameBytesPair pair = ResponseConverter.buildException(sizeIOE);
817          resultOrExceptionBuilder.setException(pair);
818          context.incrementResponseExceptionSize(pair.getSerializedSize());
819          resultOrExceptionBuilder.setIndex(action.getIndex());
820          builder.addResultOrException(resultOrExceptionBuilder.build());
821          if (cellScanner != null) {
822            skipCellsForMutation(action, cellScanner);
823          }
824          continue;
825        }
826        if (action.hasGet()) {
827          long before = EnvironmentEdgeManager.currentTime();
828          try {
829            Get get = ProtobufUtil.toGet(action.getGet());
830            if (context != null) {
831              r = get(get, ((HRegion) region), closeCallBack, context);
832            } else {
833              r = region.get(get);
834            }
835          } finally {
836            if (regionServer.metricsRegionServer != null) {
837              regionServer.metricsRegionServer.updateGet(
838                EnvironmentEdgeManager.currentTime() - before);
839            }
840          }
841        } else if (action.hasServiceCall()) {
842          hasResultOrException = true;
843          try {
844            com.google.protobuf.Message result =
845                execServiceOnRegion(region, action.getServiceCall());
846            ClientProtos.CoprocessorServiceResult.Builder serviceResultBuilder =
847                ClientProtos.CoprocessorServiceResult.newBuilder();
848            resultOrExceptionBuilder.setServiceResult(
849                serviceResultBuilder.setValue(
850                  serviceResultBuilder.getValueBuilder()
851                    .setName(result.getClass().getName())
852                    // TODO: Copy!!!
853                    .setValue(UnsafeByteOperations.unsafeWrap(result.toByteArray()))));
854          } catch (IOException ioe) {
855            rpcServer.getMetrics().exception(ioe);
856            NameBytesPair pair = ResponseConverter.buildException(ioe);
857            resultOrExceptionBuilder.setException(pair);
858            context.incrementResponseExceptionSize(pair.getSerializedSize());
859          }
860        } else if (action.hasMutation()) {
861          MutationType type = action.getMutation().getMutateType();
862          if (type != MutationType.PUT && type != MutationType.DELETE && mutations != null &&
863              !mutations.isEmpty()) {
864            // Flush out any Puts or Deletes already collected.
865            doBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
866            mutations.clear();
867          }
868          switch (type) {
869            case APPEND:
870              r = append(region, quota, action.getMutation(), cellScanner, nonceGroup,
871                  spaceQuotaEnforcement);
872              break;
873            case INCREMENT:
874              r = increment(region, quota, action.getMutation(), cellScanner, nonceGroup,
875                  spaceQuotaEnforcement);
876              break;
877            case PUT:
878            case DELETE:
879              // Collect the individual mutations and apply in a batch
880              if (mutations == null) {
881                mutations = new ArrayList<>(actions.getActionCount());
882              }
883              mutations.add(action);
884              break;
885            default:
886              throw new DoNotRetryIOException("Unsupported mutate type: " + type.name());
887          }
888        } else {
889          throw new HBaseIOException("Unexpected Action type");
890        }
891        if (r != null) {
892          ClientProtos.Result pbResult = null;
893          if (isClientCellBlockSupport(context)) {
894            pbResult = ProtobufUtil.toResultNoData(r);
895            //  Hard to guess the size here.  Just make a rough guess.
896            if (cellsToReturn == null) {
897              cellsToReturn = new ArrayList<>();
898            }
899            cellsToReturn.add(r);
900          } else {
901            pbResult = ProtobufUtil.toResult(r);
902          }
903          lastBlock = addSize(context, r, lastBlock);
904          hasResultOrException = true;
905          resultOrExceptionBuilder.setResult(pbResult);
906        }
907        // Could get to here and there was no result and no exception.  Presumes we added
908        // a Put or Delete to the collecting Mutations List for adding later.  In this
909        // case the corresponding ResultOrException instance for the Put or Delete will be added
910        // down in the doBatchOp method call rather than up here.
911      } catch (IOException ie) {
912        rpcServer.getMetrics().exception(ie);
913        hasResultOrException = true;
914        NameBytesPair pair = ResponseConverter.buildException(ie);
915        resultOrExceptionBuilder.setException(pair);
916        context.incrementResponseExceptionSize(pair.getSerializedSize());
917      }
918      if (hasResultOrException) {
919        // Propagate index.
920        resultOrExceptionBuilder.setIndex(action.getIndex());
921        builder.addResultOrException(resultOrExceptionBuilder.build());
922      }
923    }
924    // Finish up any outstanding mutations
925    if (mutations != null && !mutations.isEmpty()) {
926      doBatchOp(builder, region, quota, mutations, cellScanner, spaceQuotaEnforcement);
927    }
928    return cellsToReturn;
929  }

931  private void checkCellSizeLimit(final Region region, final Mutation m) throws IOException {
932    if (!(region instanceof HRegion)) {
933      return;
934    }
935    HRegion r = (HRegion)region;
936    if (r.maxCellSize > 0) {
937      CellScanner cells = m.cellScanner();
938      while (cells.advance()) {
939        int size = CellUtil.estimatedSerializedSizeOf(cells.current());
940        if (size > r.maxCellSize) {
941          String msg = "Cell with size " + size + " exceeds limit of " + r.maxCellSize + " bytes";
942          if (LOG.isDebugEnabled()) {
943            LOG.debug(msg);
944          }
945          throw new DoNotRetryIOException(msg);
946        }
947      }
948    }
949  }

951  /**
952   * Execute a list of Put/Delete mutations.
953   *
954   * @param builder
955   * @param region
956   * @param mutations
957   */
958  private void doBatchOp(final RegionActionResult.Builder builder, final Region region,
959      final OperationQuota quota, final List<ClientProtos.Action> mutations,
960      final CellScanner cells, ActivePolicyEnforcement spaceQuotaEnforcement) {
961    Mutation[] mArray = new Mutation[mutations.size()];
962    long before = EnvironmentEdgeManager.currentTime();
963    boolean batchContainsPuts = false, batchContainsDelete = false;
964    try {
965      /** HBASE-17924
966       * mutationActionMap is a map to map the relation between mutations and actions
967       * since mutation array may have been reoredered.In order to return the right
968       * result or exception to the corresponding actions, We need to know which action
969       * is the mutation belong to. We can't sort ClientProtos.Action array, since they
970       * are bonded to cellscanners.
971       */
972      Map<Mutation, ClientProtos.Action> mutationActionMap = new HashMap<Mutation, ClientProtos.Action>();
973      int i = 0;
974      for (ClientProtos.Action action: mutations) {
975        MutationProto m = action.getMutation();
976        Mutation mutation;
977        if (m.getMutateType() == MutationType.PUT) {
978          mutation = ProtobufUtil.toPut(m, cells);
979          batchContainsPuts = true;
980        } else {
981          mutation = ProtobufUtil.toDelete(m, cells);
982          batchContainsDelete = true;
983        }
984        mutationActionMap.put(mutation, action);
985        mArray[i++] = mutation;
986        checkCellSizeLimit(region, mutation);
987        // Check if a space quota disallows this mutation
988        spaceQuotaEnforcement.getPolicyEnforcement(region).check(mutation);
989        quota.addMutation(mutation);
990      }
991
992      if (!region.getRegionInfo().isMetaTable()) {
993        regionServer.cacheFlusher.reclaimMemStoreMemory();
994      }
995
996      // HBASE-17924
997      // sort to improve lock efficiency
998      Arrays.sort(mArray);
999
1000      OperationStatus[] codes = region.batchMutate(mArray, HConstants.NO_NONCE,
1001        HConstants.NO_NONCE);
1002      for (i = 0; i < codes.length; i++) {
1003        Mutation currentMutation = mArray[i];
1004        ClientProtos.Action currentAction = mutationActionMap.get(currentMutation);
1005        int index = currentAction.getIndex();
1006        Exception e = null;
1007        switch (codes[i].getOperationStatusCode()) {
1008          case BAD_FAMILY:
1009            e = new NoSuchColumnFamilyException(codes[i].getExceptionMsg());
1010            builder.addResultOrException(getResultOrException(e, index));
1011            break;
1012
1013          case SANITY_CHECK_FAILURE:
1014            e = new FailedSanityCheckException(codes[i].getExceptionMsg());
1015            builder.addResultOrException(getResultOrException(e, index));
1016            break;
1017
1018          default:
1019            e = new DoNotRetryIOException(codes[i].getExceptionMsg());
1020            builder.addResultOrException(getResultOrException(e, index));
1021            break;
1022
1023          case SUCCESS:
1024            builder.addResultOrException(getResultOrException(
1025              ClientProtos.Result.getDefaultInstance(), index));
1026            break;
1027        }
1028      }
1029    } catch (IOException ie) {
1030      for (int i = 0; i < mutations.size(); i++) {
1031        builder.addResultOrException(getResultOrException(ie, mutations.get(i).getIndex()));
1032      }
1033    }
1034    if (regionServer.metricsRegionServer != null) {
1035      long after = EnvironmentEdgeManager.currentTime();
1036      if (batchContainsPuts) {
1037        regionServer.metricsRegionServer.updatePut(after - before);
1038      }
1039      if (batchContainsDelete) {
1040        regionServer.metricsRegionServer.updateDelete(after - before);
1041      }
1042    }
1043  }

1045  /**
1046   * Execute a list of Put/Delete mutations. The function returns OperationStatus instead of
1047   * constructing MultiResponse to save a possible loop if caller doesn't need MultiResponse.
1048   * @param region
1049   * @param mutations
1050   * @param replaySeqId
1051   * @return an array of OperationStatus which internally contains the OperationStatusCode and the
1052   *         exceptionMessage if any
1053   * @throws IOException
1054   */
1055  private OperationStatus [] doReplayBatchOp(final Region region,
1056      final List<WALSplitter.MutationReplay> mutations, long replaySeqId) throws IOException {
1057    long before = EnvironmentEdgeManager.currentTime();
1058    boolean batchContainsPuts = false, batchContainsDelete = false;
1059    try {
1060      for (Iterator<WALSplitter.MutationReplay> it = mutations.iterator(); it.hasNext();) {
1061        WALSplitter.MutationReplay m = it.next();
1062
1063        if (m.type == MutationType.PUT) {
1064          batchContainsPuts = true;
1065        } else {
1066          batchContainsDelete = true;
1067        }
1068
1069        NavigableMap<byte[], List<Cell>> map = m.mutation.getFamilyCellMap();
1070        List<Cell> metaCells = map.get(WALEdit.METAFAMILY);
1071        if (metaCells != null && !metaCells.isEmpty()) {
1072          for (Cell metaCell : metaCells) {
1073            CompactionDescriptor compactionDesc = WALEdit.getCompaction(metaCell);
1074            boolean isDefaultReplica = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
1075            HRegion hRegion = (HRegion)region;
1076            if (compactionDesc != null) {
1077              // replay the compaction. Remove the files from stores only if we are the primary
1078              // region replica (thus own the files)
1079              hRegion.replayWALCompactionMarker(compactionDesc, !isDefaultReplica, isDefaultReplica,
1080                replaySeqId);
1081              continue;
1082            }
1083            FlushDescriptor flushDesc = WALEdit.getFlushDescriptor(metaCell);
1084            if (flushDesc != null && !isDefaultReplica) {
1085              hRegion.replayWALFlushMarker(flushDesc, replaySeqId);
1086              continue;
1087            }
1088            RegionEventDescriptor regionEvent = WALEdit.getRegionEventDescriptor(metaCell);
1089            if (regionEvent != null && !isDefaultReplica) {
1090              hRegion.replayWALRegionEventMarker(regionEvent);
1091              continue;
1092            }
1093            BulkLoadDescriptor bulkLoadEvent = WALEdit.getBulkLoadDescriptor(metaCell);
1094            if (bulkLoadEvent != null) {
1095              hRegion.replayWALBulkLoadEventMarker(bulkLoadEvent);
1096              continue;
1097            }
1098          }
1099          it.remove();
1100        }
1101      }
1102      requestCount.add(mutations.size());
1103      if (!region.getRegionInfo().isMetaTable()) {
1104        regionServer.cacheFlusher.reclaimMemStoreMemory();
1105      }
1106      return region.batchReplay(mutations.toArray(
1107        new WALSplitter.MutationReplay[mutations.size()]), replaySeqId);
1108    } finally {
1109      if (regionServer.metricsRegionServer != null) {
1110        long after = EnvironmentEdgeManager.currentTime();
1111          if (batchContainsPuts) {
1112          regionServer.metricsRegionServer.updatePut(after - before);
1113        }
1114        if (batchContainsDelete) {
1115          regionServer.metricsRegionServer.updateDelete(after - before);
1116        }
1117      }
1118    }
1119  }

1121  private void closeAllScanners() {
1122    // Close any outstanding scanners. Means they'll get an UnknownScanner
1123    // exception next time they come in.
1124    for (Map.Entry<String, RegionScannerHolder> e : scanners.entrySet()) {
1125      try {
1126        e.getValue().s.close();

1300  /**
1301   * Method to account for the size of retained cells and retained data blocks.
1302   * @return an object that represents the last referenced block from this response.
1303   */
1304  Object addSize(RpcCallContext context, Result r, Object lastBlock) {
1305    if (context != null && r != null && !r.isEmpty()) {
1306      for (Cell c : r.rawCells()) {
1307        context.incrementResponseCellSize(CellUtil.estimatedSerializedSizeOf(c));
1308
1309        // Since byte buffers can point all kinds of crazy places it's harder to keep track
1310        // of which blocks are kept alive by what byte buffer.
1311        // So we make a guess.
1312        if (c instanceof ByteBufferCell) {
1313          ByteBufferCell bbCell = (ByteBufferCell) c;
1314          ByteBuffer bb = bbCell.getValueByteBuffer();
1315          if (bb != lastBlock) {
1316            context.incrementResponseBlockSize(bb.capacity());
1317            lastBlock = bb;
1318          }
1319        } else {
1320          // We're using the last block being the same as the current block as
1321          // a proxy for pointing to a new block. This won't be exact.
1322          // If there are multiple gets that bounce back and forth
1323          // Then it's possible that this will over count the size of
1324          // referenced blocks. However it's better to over count and
1325          // use two rpcs than to OOME the regionserver.
1326          byte[] valueArray = c.getValueArray();
1327          if (valueArray != lastBlock) {
1328            context.incrementResponseBlockSize(valueArray.length);
1329            lastBlock = valueArray;
1330          }
1331        }
1332
1333      }
1334    }
1335    return lastBlock;
1336  }

1805  /**
1806   * Open asynchronously a region or a set of regions on the region server.
1807   *
1808   * The opening is coordinated by ZooKeeper, and this method requires the znode to be created
1809   *  before being called. As a consequence, this method should be called only from the master.
1810   * <p>
1811   * Different manages states for the region are:
1812   * </p><ul>
1813   *  <li>region not opened: the region opening will start asynchronously.</li>
1814   *  <li>a close is already in progress: this is considered as an error.</li>
1815   *  <li>an open is already in progress: this new open request will be ignored. This is important
1816   *  because the Master can do multiple requests if it crashes.</li>
1817   *  <li>the region is already opened:  this new open request will be ignored.</li>
1818   *  </ul>
1819   * <p>
1820   * Bulk assign: If there are more than 1 region to open, it will be considered as a bulk assign.
1821   * For a single region opening, errors are sent through a ServiceException. For bulk assign,
1822   * errors are put in the response as FAILED_OPENING.
1823   * </p>
1824   * @param controller the RPC controller
1825   * @param request the request
1826   * @throws ServiceException
1827   */
1828  @Override
1829  @QosPriority(priority=HConstants.ADMIN_QOS)
1830  public OpenRegionResponse openRegion(final RpcController controller,
1831      final OpenRegionRequest request) throws ServiceException {
1832    requestCount.increment();
1833    if (request.hasServerStartCode()) {
1834      // check that we are the same server that this RPC is intended for.
1835      long serverStartCode = request.getServerStartCode();
1836      if (regionServer.serverName.getStartcode() !=  serverStartCode) {
1837        throw new ServiceException(new DoNotRetryIOException("This RPC was intended for a " +
1838            "different server with startCode: " + serverStartCode + ", this server is: "
1839            + regionServer.serverName));
1840      }
1841    }
1842
1843    OpenRegionResponse.Builder builder = OpenRegionResponse.newBuilder();
1844    final int regionCount = request.getOpenInfoCount();
1845    final Map<TableName, HTableDescriptor> htds = new HashMap<>(regionCount);
1846    final boolean isBulkAssign = regionCount > 1;
1847    try {
1848      checkOpen();
1849    } catch (IOException ie) {
1850      TableName tableName = null;
1851      if (regionCount == 1) {
1852        RegionInfo ri = request.getOpenInfo(0).getRegion();
1853        if (ri != null) {
1854          tableName = ProtobufUtil.toTableName(ri.getTableName());
1855        }
1856      }
1857      if (!TableName.META_TABLE_NAME.equals(tableName)) {
1858        throw new ServiceException(ie);
1859      }
1860      // We are assigning meta, wait a little for regionserver to finish initialization.
1861      int timeout = regionServer.conf.getInt(HConstants.HBASE_RPC_TIMEOUT_KEY,
1862        HConstants.DEFAULT_HBASE_RPC_TIMEOUT) >> 2; // Quarter of RPC timeout
1863      long endTime = System.currentTimeMillis() + timeout;
1864      synchronized (regionServer.online) {
1865        try {
1866          while (System.currentTimeMillis() <= endTime
1867              && !regionServer.isStopped() && !regionServer.isOnline()) {
1868            regionServer.online.wait(regionServer.msgInterval);
1869          }
1870          checkOpen();
1871        } catch (InterruptedException t) {
1872          Thread.currentThread().interrupt();
1873          throw new ServiceException(t);
1874        } catch (IOException e) {
1875          throw new ServiceException(e);
1876        }
1877      }
1878    }
1879
1880    long masterSystemTime = request.hasMasterSystemTime() ? request.getMasterSystemTime() : -1;
1881
1882    for (RegionOpenInfo regionOpenInfo : request.getOpenInfoList()) {
1883      final HRegionInfo region = HRegionInfo.convert(regionOpenInfo.getRegion());
1884      HTableDescriptor htd;
1885      try {
1886        String encodedName = region.getEncodedName();
1887        byte[] encodedNameBytes = region.getEncodedNameAsBytes();
1888        final Region onlineRegion = regionServer.getFromOnlineRegions(encodedName);
1889        if (onlineRegion != null) {
1890          // The region is already online. This should not happen any more.
1891          String error = "Received OPEN for the region:"
1892            + region.getRegionNameAsString() + ", which is already online";
1893          LOG.warn(error);
1894          //regionServer.abort(error);
1895          //throw new IOException(error);
1896          builder.addOpeningState(RegionOpeningState.OPENED);
1897          continue;
1898        }
1899        LOG.info("Open " + region.getRegionNameAsString());
1900
1901        final Boolean previous = regionServer.regionsInTransitionInRS.putIfAbsent(
1902          encodedNameBytes, Boolean.TRUE);
1903
1904        if (Boolean.FALSE.equals(previous)) {
1905          if (regionServer.getFromOnlineRegions(encodedName) != null) {
1906            // There is a close in progress. This should not happen any more.
1907            String error = "Received OPEN for the region:"
1908              + region.getRegionNameAsString() + ", which we are already trying to CLOSE";
1909            regionServer.abort(error);
1910            throw new IOException(error);
1911          }
1912          regionServer.regionsInTransitionInRS.put(encodedNameBytes, Boolean.TRUE);
1913        }
1914
1915        if (Boolean.TRUE.equals(previous)) {
1916          // An open is in progress. This is supported, but let's log this.
1917          LOG.info("Receiving OPEN for the region:" +
1918            region.getRegionNameAsString() + ", which we are already trying to OPEN"
1919              + " - ignoring this new request for this region.");
1920        }
1921
1922        // We are opening this region. If it moves back and forth for whatever reason, we don't
1923        // want to keep returning the stale moved record while we are opening/if we close again.
1924        regionServer.removeFromMovedRegions(region.getEncodedName());
1925
1926        if (previous == null || !previous.booleanValue()) {
1927          // check if the region to be opened is marked in recovering state in ZK
1928          if (ZKSplitLog.isRegionMarkedRecoveringInZK(regionServer.getZooKeeper(),
1929              region.getEncodedName())) {
1930            // Check if current region open is for distributedLogReplay. This check is to support
1931            // rolling restart/upgrade where we want to Master/RS see same configuration
1932            if (!regionOpenInfo.hasOpenForDistributedLogReplay()
1933                  || regionOpenInfo.getOpenForDistributedLogReplay()) {
1934              regionServer.recoveringRegions.put(region.getEncodedName(), null);
1935            } else {
1936              // Remove stale recovery region from ZK when we open region not for recovering which
1937              // could happen when turn distributedLogReplay off from on.
1938              List<String> tmpRegions = new ArrayList<>();
1939              tmpRegions.add(region.getEncodedName());
1940              ZKSplitLog.deleteRecoveringRegionZNodes(regionServer.getZooKeeper(),
1941                tmpRegions);
1942            }
1943          }
1944          htd = htds.get(region.getTable());
1945          if (htd == null) {
1946            htd = regionServer.tableDescriptors.get(region.getTable());
1947            htds.put(region.getTable(), htd);
1948          }
1949          if (htd == null) {
1950            throw new IOException("Missing table descriptor for " + region.getEncodedName());
1951          }
1952          // If there is no action in progress, we can submit a specific handler.
1953          // Need to pass the expected version in the constructor.
1954          if (region.isMetaRegion()) {
1955            regionServer.service.submit(new OpenMetaHandler(
1956              regionServer, regionServer, region, htd, masterSystemTime));
1957          } else {
1958            if (regionOpenInfo.getFavoredNodesCount() > 0) {
1959              regionServer.updateRegionFavoredNodesMapping(region.getEncodedName(),
1960                  regionOpenInfo.getFavoredNodesList());
1961            }
1962            if (htd.getPriority() >= HConstants.ADMIN_QOS || region.getTable().isSystemTable()) {
1963              regionServer.service.submit(new OpenPriorityRegionHandler(
1964                regionServer, regionServer, region, htd, masterSystemTime));
1965            } else {
1966              regionServer.service.submit(new OpenRegionHandler(
1967                regionServer, regionServer, region, htd, masterSystemTime));
1968            }
1969          }
1970        }
1971
1972        builder.addOpeningState(RegionOpeningState.OPENED);
1973
1974      } catch (KeeperException zooKeeperEx) {
1975        LOG.error("Can't retrieve recovering state from zookeeper", zooKeeperEx);
1976        throw new ServiceException(zooKeeperEx);
1977      } catch (IOException ie) {
1978        LOG.warn("Failed opening region " + region.getRegionNameAsString(), ie);
1979        if (isBulkAssign) {
1980          builder.addOpeningState(RegionOpeningState.FAILED_OPENING);
1981        } else {
1982          throw new ServiceException(ie);
1983        }
1984      }
1985    }
1986    return builder.build();
1987  }

2051  @QosPriority(priority = HConstants.REPLAY_QOS)
2052  public ReplicateWALEntryResponse replay(final RpcController controller,
2053      final ReplicateWALEntryRequest request) throws ServiceException {
2054    long before = EnvironmentEdgeManager.currentTime();
2055    CellScanner cells = ((HBaseRpcController) controller).cellScanner();
2056    try {
2057      checkOpen();
2058      List<WALEntry> entries = request.getEntryList();
2059      if (entries == null || entries.isEmpty()) {
2060        // empty input
2061        return ReplicateWALEntryResponse.newBuilder().build();
2062      }
2063      ByteString regionName = entries.get(0).getKey().getEncodedRegionName();
2064      Region region = regionServer.getRegionByEncodedName(regionName.toStringUtf8());
2065      RegionCoprocessorHost coprocessorHost =
2066          ServerRegionReplicaUtil.isDefaultReplica(region.getRegionInfo())
2067            ? region.getCoprocessorHost()
2068            : null; // do not invoke coprocessors if this is a secondary region replica
2069      List<Pair<WALKey, WALEdit>> walEntries = new ArrayList<>();
2070
2071      // Skip adding the edits to WAL if this is a secondary region replica
2072      boolean isPrimary = RegionReplicaUtil.isDefaultReplica(region.getRegionInfo());
2073      Durability durability = isPrimary ? Durability.USE_DEFAULT : Durability.SKIP_WAL;
2074
2075      for (WALEntry entry : entries) {
2076        if (!regionName.equals(entry.getKey().getEncodedRegionName())) {
2077          throw new NotServingRegionException("Replay request contains entries from multiple " +
2078              "regions. First region:" + regionName.toStringUtf8() + " , other region:"
2079              + entry.getKey().getEncodedRegionName());
2080        }
2081        if (regionServer.nonceManager != null && isPrimary) {
2082          long nonceGroup = entry.getKey().hasNonceGroup()
2083            ? entry.getKey().getNonceGroup() : HConstants.NO_NONCE;
2084          long nonce = entry.getKey().hasNonce() ? entry.getKey().getNonce() : HConstants.NO_NONCE;
2085          regionServer.nonceManager.reportOperationFromWal(
2086              nonceGroup,
2087              nonce,
2088              entry.getKey().getWriteTime());
2089        }
2090        Pair<WALKey, WALEdit> walEntry = (coprocessorHost == null) ? null : new Pair<>();
2091        List<WALSplitter.MutationReplay> edits = WALSplitter.getMutationsFromWALEntry(entry,
2092          cells, walEntry, durability);

2094          // Start coprocessor replay here. The coprocessor is for each WALEdit instead of a
2095          // KeyValue.

2097            walEntry.getSecond())) {
2098            // if bypass this log entry, ignore it ...
2099            continue;
2100          }
2101          walEntries.add(walEntry);
2102        }
2103        if(edits!=null && !edits.isEmpty()) {
2104          // HBASE-17924
2105          // sort to improve lock efficiency
2106          Collections.sort(edits);
2107          long replaySeqId = (entry.getKey().hasOrigSequenceNumber()) ?
2108            entry.getKey().getOrigSequenceNumber() : entry.getKey().getLogSequenceNumber();
2109          OperationStatus[] result = doReplayBatchOp(region, edits, replaySeqId);
2110          // check if it's a partial success
2111          for (int i = 0; result != null && i < result.length; i++) {
2112            if (result[i] != OperationStatus.SUCCESS) {
2113              throw new IOException(result[i].getExceptionMsg());
2114            }
2115          }
2116        }
2117      }
2118
2119      //sync wal at the end because ASYNC_WAL is used above
2120      WAL wal = getWAL(region);
2121      if (wal != null) {
2122        wal.sync();
2123      }
2124
2125      if (coprocessorHost != null) {
2126        for (Pair<WALKey, WALEdit> entry : walEntries) {
2127          coprocessorHost.postWALRestore(region.getRegionInfo(), entry.getFirst(),
2128            entry.getSecond());
2129        }
2130      }
2131      return ReplicateWALEntryResponse.newBuilder().build();
2132    } catch (IOException ie) {
2133      throw new ServiceException(ie);
2134    } finally {
2135      if (regionServer.metricsRegionServer != null) {
2136        regionServer.metricsRegionServer.updateReplay(
2137          EnvironmentEdgeManager.currentTime() - before);
2138      }
2139    }
2140  }

2793                qualifier, compareOp, comparator, delete, result);

2821      }

2939  private long getTimeLimit(HBaseRpcController controller, boolean allowHeartbeatMessages) {
2940    // Set the time limit to be half of the more restrictive timeout value (one of the
2941    // timeout values must be positive). In the event that both values are positive, the
2942    // more restrictive of the two is used to calculate the limit.
2943    if (allowHeartbeatMessages && (scannerLeaseTimeoutPeriod > 0 || rpcTimeout > 0)) {
2944      long timeLimitDelta;
2945      if (scannerLeaseTimeoutPeriod > 0 && rpcTimeout > 0) {
2946        timeLimitDelta = Math.min(scannerLeaseTimeoutPeriod, rpcTimeout);
2947      } else {
2948        timeLimitDelta = scannerLeaseTimeoutPeriod > 0 ? scannerLeaseTimeoutPeriod : rpcTimeout;
2949      }
2950      if (controller != null && controller.getCallTimeout() > 0) {
2951        timeLimitDelta = Math.min(timeLimitDelta, controller.getCallTimeout());
2952      }
2953      // Use half of whichever timeout value was more restrictive... But don't allow
2954      // the time limit to be less than the allowable minimum (could cause an
2955      // immediatate timeout before scanning any data).
2956      timeLimitDelta = Math.max(timeLimitDelta / 2, minimumScanTimeLimitDelta);
2957      // XXX: Can not use EnvironmentEdge here because TestIncrementTimeRange use a
2958      // ManualEnvironmentEdge. Consider using System.nanoTime instead.
2959      return System.currentTimeMillis() + timeLimitDelta;
2960    }
2961    // Default value of timeLimit is negative to indicate no timeLimit should be
2962    // enforced.
2963    return -1L;
2964  }
