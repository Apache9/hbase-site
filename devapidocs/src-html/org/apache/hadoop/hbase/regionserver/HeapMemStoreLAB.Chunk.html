019package org.apache.hadoop.hbase.regionserver;
020
021import java.util.concurrent.BlockingQueue;
022import java.util.concurrent.LinkedBlockingQueue;
023import java.util.concurrent.atomic.AtomicBoolean;
024import java.util.concurrent.atomic.AtomicInteger;
025import java.util.concurrent.atomic.AtomicReference;
026
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.util.ByteRange;
030import org.apache.hadoop.hbase.util.SimpleMutableByteRange;
031
032import com.google.common.annotations.VisibleForTesting;
033import com.google.common.base.Preconditions;
034
035/**
036 * A memstore-local allocation buffer.
037 * <p>
038 * The MemStoreLAB is basically a bump-the-pointer allocator that allocates
039 * big (2MB) byte[] chunks from and then doles it out to threads that request
040 * slices into the array.
041 * <p>
042 * The purpose of this class is to combat heap fragmentation in the
043 * regionserver. By ensuring that all KeyValues in a given memstore refer
044 * only to large chunks of contiguous memory, we ensure that large blocks
045 * get freed up when the memstore is flushed.
046 * <p>
047 * Without the MSLAB, the byte array allocated during insertion end up
048 * interleaved throughout the heap, and the old generation gets progressively
049 * more fragmented until a stop-the-world compacting collection occurs.
050 * <p>
051 * TODO: we should probably benchmark whether word-aligning the allocations
052 * would provide a performance improvement - probably would speed up the
053 * Bytes.toLong/Bytes.toInt calls in KeyValue, but some of those are cached
054 * anyway
055 */
056@InterfaceAudience.Private
057public class HeapMemStoreLAB implements MemStoreLAB {
058
059  static final String CHUNK_SIZE_KEY = "hbase.hregion.memstore.mslab.chunksize";
060  static final int CHUNK_SIZE_DEFAULT = 2048 * 1024;
061  static final String MAX_ALLOC_KEY = "hbase.hregion.memstore.mslab.max.allocation";
062  static final int MAX_ALLOC_DEFAULT = 256 * 1024; // allocs bigger than this don't go through
063                                                   // allocator
064
065  private AtomicReference<Chunk> curChunk = new AtomicReference<Chunk>();
066  // A queue of chunks contained by this memstore
067  private BlockingQueue<Chunk> chunkQueue = new LinkedBlockingQueue<Chunk>();
068  final int chunkSize;
069  final int maxAlloc;
070  private final MemStoreChunkPool chunkPool;
071
072  // This flag is for closing this instance, its set when clearing snapshot of
073  // memstore
074  private volatile boolean closed = false;
075  // This flag is for reclaiming chunks. Its set when putting chunks back to
076  // pool
077  private AtomicBoolean reclaimed = new AtomicBoolean(false);
078  // Current count of open scanners which reading data from this MemStoreLAB
079  private final AtomicInteger openScannerCount = new AtomicInteger();
080
081  // Used in testing
082  public HeapMemStoreLAB() {
083    this(new Configuration());
084  }
085
086  public HeapMemStoreLAB(Configuration conf) {
087    chunkSize = conf.getInt(CHUNK_SIZE_KEY, CHUNK_SIZE_DEFAULT);
088    maxAlloc = conf.getInt(MAX_ALLOC_KEY, MAX_ALLOC_DEFAULT);
089    this.chunkPool = MemStoreChunkPool.getPool(conf);
090
091    // if we don't exclude allocations >CHUNK_SIZE, we'd infiniteloop on one!
092    Preconditions.checkArgument(
093      maxAlloc <= chunkSize,
094      MAX_ALLOC_KEY + " must be less than " + CHUNK_SIZE_KEY);
095  }
096
097  /**
098   * Allocate a slice of the given length.
099   *
100   * If the size is larger than the maximum size specified for this
101   * allocator, returns null.
102   */
103  @Override
104  public ByteRange allocateBytes(int size) {
105    Preconditions.checkArgument(size >= 0, "negative size");
106
107    // Callers should satisfy large allocations directly from JVM since they
108    // don't cause fragmentation as badly.
109    if (size > maxAlloc) {
110      return null;
111    }
112
113    while (true) {
114      Chunk c = getOrMakeChunk();
115
116      // Try to allocate from this chunk
117      int allocOffset = c.alloc(size);
118      if (allocOffset != -1) {
119        // We succeeded - this is the common case - small alloc
120        // from a big buffer
121        return new SimpleMutableByteRange(c.data, allocOffset, size);
122      }
123
124      // not enough space!
125      // try to retire this chunk
126      tryRetireChunk(c);
127    }
128  }
129
130  /**
131   * Close this instance since it won't be used any more, try to put the chunks
132   * back to pool
133   */
134  @Override
135  public void close() {
136    this.closed = true;
137    // We could put back the chunks to pool for reusing only when there is no
138    // opening scanner which will read their data
139    if (chunkPool != null && openScannerCount.get() == 0
140        && reclaimed.compareAndSet(false, true)) {
141      chunkPool.putbackChunks(this.chunkQueue);
142    }
143  }
144
145  /**
146   * Called when opening a scanner on the data of this MemStoreLAB
147   */
148  @Override
149  public void incScannerCount() {
150    this.openScannerCount.incrementAndGet();
151  }
152
153  /**
154   * Called when closing a scanner on the data of this MemStoreLAB
155   */
156  @Override
157  public void decScannerCount() {
158    int count = this.openScannerCount.decrementAndGet();
159    if (chunkPool != null && count == 0 && this.closed
160        && reclaimed.compareAndSet(false, true)) {
161      chunkPool.putbackChunks(this.chunkQueue);
162    }
163  }
164
165  /**
166   * Try to retire the current chunk if it is still
167   * <code>c</code>. Postcondition is that curChunk.get()
168   * != c
169   */
170  private void tryRetireChunk(Chunk c) {
171    curChunk.compareAndSet(c, null);
172    // If the CAS succeeds, that means that we won the race
173    // to retire the chunk. We could use this opportunity to
174    // update metrics on external fragmentation.
175    //
176    // If the CAS fails, that means that someone else already
177    // retired the chunk for us.
178  }
179
180  /**
181   * Get the current chunk, or, if there is no current chunk,
182   * allocate a new one from the JVM.
183   */
184  private Chunk getOrMakeChunk() {
185    while (true) {
186      // Try to get the chunk
187      Chunk c = curChunk.get();
188      if (c != null) {
189        return c;
190      }
191
192      // No current chunk, so we want to allocate one. We race
193      // against other allocators to CAS in an uninitialized chunk
194      // (which is cheap to allocate)
195      c = (chunkPool != null) ? chunkPool.getChunk() : new Chunk(chunkSize);
196      if (curChunk.compareAndSet(null, c)) {
197        // we won race - now we need to actually do the expensive
198        // allocation step
199        c.init();
200        this.chunkQueue.add(c);
201        return c;
202      } else if (chunkPool != null) {
203        chunkPool.putbackChunk(c);
204      }
205      // someone else won race - that's fine, we'll try to grab theirs
206      // in the next iteration of the loop.
207    }
208  }
209
210  @VisibleForTesting
211  Chunk getCurrentChunk() {
212    return this.curChunk.get();
213  }
214
215  /**
216   * A chunk of memory out of which allocations are sliced.
217   */
218  static class Chunk {
219    /** Actual underlying data */
220    private byte[] data;
221
222    private static final int UNINITIALIZED = -1;
223    private static final int OOM = -2;
224    /**
225     * Offset for the next allocation, or the sentinel value -1
226     * which implies that the chunk is still uninitialized.
227     * */
228    private AtomicInteger nextFreeOffset = new AtomicInteger(UNINITIALIZED);
229
230    /** Total number of allocations satisfied from this buffer */
231    private AtomicInteger allocCount = new AtomicInteger();
232
233    /** Size of chunk in bytes */
234    private final int size;
235
236    /**
237     * Create an uninitialized chunk. Note that memory is not allocated yet, so
238     * this is cheap.
239     * @param size in bytes
240     */
241    Chunk(int size) {
242      this.size = size;
243    }
244
245    /**
246     * Actually claim the memory for this chunk. This should only be called from
247     * the thread that constructed the chunk. It is thread-safe against other
248     * threads calling alloc(), who will block until the allocation is complete.
249     */
250    public void init() {
251      assert nextFreeOffset.get() == UNINITIALIZED;
252      try {
253        if (data == null) {
254          data = new byte[size];
255        }
256      } catch (OutOfMemoryError e) {
257        boolean failInit = nextFreeOffset.compareAndSet(UNINITIALIZED, OOM);
258        assert failInit; // should be true.
259        throw e;
260      }
261      // Mark that it's ready for use
262      boolean initted = nextFreeOffset.compareAndSet(
263          UNINITIALIZED, 0);
264      // We should always succeed the above CAS since only one thread
265      // calls init()!
266      Preconditions.checkState(initted,
267          "Multiple threads tried to init same chunk");
268    }
269
270    /**
271     * Reset the offset to UNINITIALIZED before before reusing an old chunk
272     */
273    void reset() {
274      if (nextFreeOffset.get() != UNINITIALIZED) {
275        nextFreeOffset.set(UNINITIALIZED);
276        allocCount.set(0);
277      }
278    }
279
280    /**
281     * Try to allocate <code>size</code> bytes from the chunk.
282     * @return the offset of the successful allocation, or -1 to indicate not-enough-space
283     */
284    public int alloc(int size) {
285      while (true) {
286        int oldOffset = nextFreeOffset.get();
287        if (oldOffset == UNINITIALIZED) {
288          // The chunk doesn't have its data allocated yet.
289          // Since we found this in curChunk, we know that whoever
290          // CAS-ed it there is allocating it right now. So spin-loop
291          // shouldn't spin long!
292          Thread.yield();
293          continue;
294        }
295        if (oldOffset == OOM) {
296          // doh we ran out of ram. return -1 to chuck this away.
297          return -1;
298        }
299
300        if (oldOffset + size > data.length) {
301          return -1; // alloc doesn't fit
302        }
303
304        // Try to atomically claim this chunk
305        if (nextFreeOffset.compareAndSet(oldOffset, oldOffset + size)) {
306          // we got the alloc
307          allocCount.incrementAndGet();
308          return oldOffset;
309        }
310        // we raced and lost alloc, try again
311      }
312    }
313
314    @Override
315    public String toString() {
316      return "Chunk@" + System.identityHashCode(this) +
317        " allocs=" + allocCount.get() + "waste=" +
318        (data.length - nextFreeOffset.get());
319    }
320
321    @VisibleForTesting
322    int getNextFreeOffset() {
323      return this.nextFreeOffset.get();
324    }
325  }
