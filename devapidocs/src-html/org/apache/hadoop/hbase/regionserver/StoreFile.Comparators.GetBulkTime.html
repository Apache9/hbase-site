051/**
052 * A Store data file.  Stores usually have one or more of these files.  They
053 * are produced by flushing the memstore to disk.  To
054 * create, instantiate a writer using {@link StoreFileWriter.Builder}
055 * and append data. Be sure to add any metadata before calling close on the
056 * Writer (Use the appendMetadata convenience methods). On close, a StoreFile
057 * is sitting in the Filesystem.  To refer to it, create a StoreFile instance
058 * passing filesystem and path.  To read, call {@link #initReader()}
059 * <p>StoreFiles may also reference store files in another Store.
060 *
061 * The reason for this weird pattern where you use a different instance for the
062 * writer and a reader is that we write once but read a lot more.
063 */
064@InterfaceAudience.LimitedPrivate("Coprocessor")
065public class StoreFile {

261  /**
262   * Constructor, loads a reader and it's indices, etc. May allocate a substantial amount of ram
263   * depending on the underlying files (10-20MB?).
264   * @param fs fs The current file system to use.
265   * @param fileInfo The store file information.
266   * @param conf The current configuration.
267   * @param cacheConf The cache configuration and block cache reference.
268   * @param cfBloomType cfBloomType The bloom type to use for this store file as specified by column
269   *          family configuration. This may or may not be the same as the Bloom filter type
270   *          actually present in the HFile, because column family configuration might change. If
271   *          this is {@link BloomType#NONE}, the existing Bloom filter is ignored.
272   * @param primaryReplica true if this is a store file for primary replica, otherwise false.
273   */
274  public StoreFile(FileSystem fs, StoreFileInfo fileInfo, Configuration conf, CacheConfig cacheConf,
275      BloomType cfBloomType, boolean primaryReplica) {
276    this.fs = fs;
277    this.fileInfo = fileInfo;
278    this.cacheConf = cacheConf;
279    this.noReadahead =
280        conf.getBoolean(STORE_FILE_READER_NO_READAHEAD, DEFAULT_STORE_FILE_READER_NO_READAHEAD);
281    if (BloomFilterFactory.isGeneralBloomEnabled(conf)) {
282      this.cfBloomType = cfBloomType;
283    } else {
284      LOG.info("Ignoring bloom filter check for file " + this.getPath() + ": " + "cfBloomType=" +
285          cfBloomType + " (disabled in config)");
286      this.cfBloomType = BloomType.NONE;
287    }
288    this.primaryReplica = primaryReplica;
289  }

398  /**
399   * Check if this storefile was created by bulk load.
400   * When a hfile is bulk loaded into HBase, we append
401   * {@code '_SeqId_<id-when-loaded>'} to the hfile name, unless
402   * "hbase.mapreduce.bulkload.assign.sequenceNumbers" is
403   * explicitly turned off.
404   * If "hbase.mapreduce.bulkload.assign.sequenceNumbers"
405   * is turned off, fall back to BULKLOAD_TIME_KEY.
406   * @return true if this storefile was created by bulk load.
407   */
408  public boolean isBulkLoadResult() {
409    boolean bulkLoadedHFile = false;
410    String fileName = this.getPath().getName();
411    int startPos = fileName.indexOf("SeqId_");
412    if (startPos != -1) {
413      bulkLoadedHFile = true;
414    }
415    return bulkLoadedHFile || (metadataMap != null && metadataMap.containsKey(BULKLOAD_TIME_KEY));
416  }

453  /**
454   * Opens reader on this store file. Called by Constructor.
455   * @throws IOException
456   * @see #closeReader(boolean)
457   */
458  private void open() throws IOException {
459    if (this.reader != null) {
460      throw new IllegalAccessError("Already open");
461    }
462
463    // Open the StoreFile.Reader
464    this.reader = fileInfo.open(this.fs, this.cacheConf, false, noReadahead ? 0L : -1L,
465      primaryReplica, refCount, true);
466
467    // Load up indices and fileinfo. This also loads Bloom filter type.
468    metadataMap = Collections.unmodifiableMap(this.reader.loadFileInfo());
469
470    // Read in our metadata.
471    byte [] b = metadataMap.get(MAX_SEQ_ID_KEY);
472    if (b != null) {
473      // By convention, if halfhfile, top half has a sequence number > bottom
474      // half. Thats why we add one in below. Its done for case the two halves
475      // are ever merged back together --rare.  Without it, on open of store,
476      // since store files are distinguished by sequence id, the one half would
477      // subsume the other.
478      this.sequenceid = Bytes.toLong(b);
479      if (fileInfo.isTopReference()) {
480        this.sequenceid += 1;
481      }
482    }
483
484    if (isBulkLoadResult()){
485      // generate the sequenceId from the fileName
486      // fileName is of the form <randomName>_SeqId_<id-when-loaded>_
487      String fileName = this.getPath().getName();
488      // Use lastIndexOf() to get the last, most recent bulk load seqId.
489      int startPos = fileName.lastIndexOf("SeqId_");
490      if (startPos != -1) {
491        this.sequenceid = Long.parseLong(fileName.substring(startPos + 6,
492            fileName.indexOf('_', startPos + 6)));
493        // Handle reference files as done above.
494        if (fileInfo.isTopReference()) {
495          this.sequenceid += 1;
496        }
497      }
498      // SKIP_RESET_SEQ_ID only works in bulk loaded file.
499      // In mob compaction, the hfile where the cells contain the path of a new mob file is bulk
500      // loaded to hbase, these cells have the same seqIds with the old ones. We do not want
501      // to reset new seqIds for them since this might make a mess of the visibility of cells that
502      // have the same row key but different seqIds.
503      boolean skipResetSeqId = isSkipResetSeqId(metadataMap.get(SKIP_RESET_SEQ_ID));
504      if (skipResetSeqId) {
505        // increase the seqId when it is a bulk loaded file from mob compaction.
506        this.sequenceid += 1;
507      }
508      this.reader.setSkipResetSeqId(skipResetSeqId);
509      this.reader.setBulkLoaded(true);
510    }
511    this.reader.setSequenceID(this.sequenceid);
512
513    b = metadataMap.get(HFile.Writer.MAX_MEMSTORE_TS_KEY);
514    if (b != null) {
515      this.maxMemstoreTS = Bytes.toLong(b);
516    }
517
518    b = metadataMap.get(MAJOR_COMPACTION_KEY);
519    if (b != null) {
520      boolean mc = Bytes.toBoolean(b);
521      if (this.majorCompaction == null) {
522        this.majorCompaction = new AtomicBoolean(mc);
523      } else {
524        this.majorCompaction.set(mc);
525      }
526    } else {
527      // Presume it is not major compacted if it doesn't explicity say so
528      // HFileOutputFormat explicitly sets the major compacted key.
529      this.majorCompaction = new AtomicBoolean(false);
530    }
531
532    b = metadataMap.get(EXCLUDE_FROM_MINOR_COMPACTION_KEY);
533    this.excludeFromMinorCompaction = (b != null && Bytes.toBoolean(b));
534
535    BloomType hfileBloomType = reader.getBloomFilterType();
536    if (cfBloomType != BloomType.NONE) {
537      reader.loadBloomfilter(BlockType.GENERAL_BLOOM_META);
538      if (hfileBloomType != cfBloomType) {
539        LOG.info("HFile Bloom filter type for "
540            + reader.getHFileReader().getName() + ": " + hfileBloomType
541            + ", but " + cfBloomType + " specified in column family "
542            + "configuration");
543      }
544    } else if (hfileBloomType != BloomType.NONE) {
545      LOG.info("Bloom filter turned off by CF config for "
546          + reader.getHFileReader().getName());
547    }
548
549    // load delete family bloom filter
550    reader.loadBloomfilter(BlockType.DELETE_FAMILY_BLOOM_META);
551
552    try {
553      this.reader.timeRange = TimeRangeTracker.getTimeRange(metadataMap.get(TIMERANGE_KEY));
554    } catch (IllegalArgumentException e) {
555      LOG.error("Error reading timestamp range data from meta -- " +
556          "proceeding without", e);
557      this.reader.timeRange = null;
558    }
559    // initialize so we can reuse them after reader closed.
560    firstKey = reader.getFirstKey();
561    lastKey = reader.getLastKey();
562    comparator = reader.getComparator();
563  }

565  /**
566   * Initialize the reader used for pread.
567   */
568  public void initReader() throws IOException {
569    if (reader == null) {
570      try {
571        open();
572      } catch (Exception e) {
573        try {
574          boolean evictOnClose = cacheConf != null ? cacheConf.shouldEvictOnClose() : true;
575          this.closeReader(evictOnClose);
576        } catch (IOException ee) {
577          LOG.warn("failed to close reader", ee);
578        }
579        throw e;
580      }
581    }
582  }

584  private StoreFileReader createStreamReader(boolean canUseDropBehind) throws IOException {
585    initReader();
586    StoreFileReader reader = fileInfo.open(this.fs, this.cacheConf, canUseDropBehind, -1L,
587      primaryReplica, refCount, false);
588    reader.copyFields(this.reader);
589    return reader;
590  }

648  /**
649   * @return a length description of this StoreFile, suitable for debug output
650   */
651  public String toStringDetailed() {
652    StringBuilder sb = new StringBuilder();
653    sb.append(this.getPath().toString());
654    sb.append(", isReference=").append(isReference());
655    sb.append(", isBulkLoadResult=").append(isBulkLoadResult());
656    if (isBulkLoadResult()) {
657      sb.append(", bulkLoadTS=").append(getBulkLoadTimestamp());
658    } else {
659      sb.append(", seqid=").append(getMaxSequenceId());
660    }
661    sb.append(", majorCompaction=").append(isMajorCompaction());
662
663    return sb.toString();
664  }

701  /**
702   * Gets the approximate mid-point of this file that is optimal for use in splitting it.
703   * @param comparator Comparator used to compare KVs.
704   * @return The split point row, or null if splitting is not possible, or reader is null.
705   */
706  byte[] getFileSplitPoint(CellComparator comparator) throws IOException {
707    if (this.reader == null) {
708      LOG.warn("Storefile " + this + " Reader is null; cannot get split point");
709      return null;
710    }
711    // Get first, last, and mid keys.  Midkey is the key that starts block
712    // in middle of hfile.  Has column and timestamp.  Need to return just
713    // the row we want to split on as midkey.
714    Cell midkey = this.reader.midkey();
715    if (midkey != null) {
716      Cell firstKey = this.reader.getFirstKey();
717      Cell lastKey = this.reader.getLastKey();
718      // if the midkey is the same as the first or last keys, we cannot (ever) split this region.
719      if (comparator.compareRows(midkey, firstKey) == 0
720          || comparator.compareRows(midkey, lastKey) == 0) {
721        if (LOG.isDebugEnabled()) {
722          LOG.debug("cannot split because midkey is the same as first or last row");
723        }
724        return null;
725      }
726      return CellUtil.cloneRow(midkey);
727    }
728    return null;
729  }

731  /**
732   * Useful comparators for comparing StoreFiles.
733   */
734  public abstract static class Comparators {
735    /**
736     * Comparator that compares based on the Sequence Ids of the
737     * the StoreFiles. Bulk loads that did not request a seq ID
738     * are given a seq id of -1; thus, they are placed before all non-
739     * bulk loads, and bulk loads with sequence Id. Among these files,
740     * the size is used to determine the ordering, then bulkLoadTime.
741     * If there are ties, the path name is used as a tie-breaker.
742     */
743    public static final Comparator<StoreFile> SEQ_ID =
744      Ordering.compound(ImmutableList.of(
745          Ordering.natural().onResultOf(new GetSeqId()),
746          Ordering.natural().onResultOf(new GetFileSize()).reverse(),
747          Ordering.natural().onResultOf(new GetBulkTime()),
748          Ordering.natural().onResultOf(new GetPathName())
749      ));
750
751    /**
752     * Comparator for time-aware compaction. SeqId is still the first
753     *   ordering criterion to maintain MVCC.
754     */
755    public static final Comparator<StoreFile> SEQ_ID_MAX_TIMESTAMP =
756      Ordering.compound(ImmutableList.of(
757        Ordering.natural().onResultOf(new GetSeqId()),
758        Ordering.natural().onResultOf(new GetMaxTimestamp()),
759        Ordering.natural().onResultOf(new GetFileSize()).reverse(),
760        Ordering.natural().onResultOf(new GetBulkTime()),
761        Ordering.natural().onResultOf(new GetPathName())
762      ));
763
764    private static class GetSeqId implements Function<StoreFile, Long> {
765      @Override
766      public Long apply(StoreFile sf) {
767        return sf.getMaxSequenceId();
768      }
769    }
770
771    private static class GetFileSize implements Function<StoreFile, Long> {
772      @Override
773      public Long apply(StoreFile sf) {
774        if (sf.getReader() != null) {
775          return sf.getReader().length();
776        } else {
777          // the reader may be null for the compacted files and if the archiving
778          // had failed.
779          return -1L;
780        }
781      }
782    }
783
784    private static class GetBulkTime implements Function<StoreFile, Long> {
785      @Override
786      public Long apply(StoreFile sf) {
787        if (!sf.isBulkLoadResult()) return Long.MAX_VALUE;
788        return sf.getBulkLoadTimestamp();
789      }
790    }
791
792    private static class GetPathName implements Function<StoreFile, String> {
793      @Override
794      public String apply(StoreFile sf) {
795        return sf.getPath().getName();
796      }
797    }
798
799    private static class GetMaxTimestamp implements Function<StoreFile, Long> {
800      @Override
801      public Long apply(StoreFile sf) {
802        return sf.getMaximumTimestamp() == null? (Long)Long.MAX_VALUE : sf.getMaximumTimestamp();
803      }
804    }
805  }
