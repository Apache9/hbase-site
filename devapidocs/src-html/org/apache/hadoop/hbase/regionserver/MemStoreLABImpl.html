019package org.apache.hadoop.hbase.regionserver;

021import java.nio.ByteBuffer;
022import java.util.Set;
023import java.util.concurrent.BlockingQueue;
024import java.util.concurrent.ConcurrentSkipListSet;
025import java.util.concurrent.LinkedBlockingQueue;
026import java.util.concurrent.atomic.AtomicBoolean;
027import java.util.concurrent.atomic.AtomicInteger;
028import java.util.concurrent.atomic.AtomicReference;
029import java.util.concurrent.locks.ReentrantLock;
030
031import org.apache.commons.logging.Log;
032import org.apache.commons.logging.LogFactory;
033import org.apache.hadoop.conf.Configuration;
034import org.apache.hadoop.hbase.Cell;
035import org.apache.hadoop.hbase.ExtendedCell;
036import org.apache.hadoop.hbase.KeyValueUtil;
037import org.apache.hadoop.hbase.classification.InterfaceAudience;
038
039import com.google.common.annotations.VisibleForTesting;
040import com.google.common.base.Preconditions;
041/**
042 * A memstore-local allocation buffer.
043 * <p>
044 * The MemStoreLAB is basically a bump-the-pointer allocator that allocates
045 * big (2MB) byte[] chunks from and then doles it out to threads that request
046 * slices into the array.
047 * <p>
048 * The purpose of this class is to combat heap fragmentation in the
049 * regionserver. By ensuring that all Cells in a given memstore refer
050 * only to large chunks of contiguous memory, we ensure that large blocks
051 * get freed up when the memstore is flushed.
052 * <p>
053 * Without the MSLAB, the byte array allocated during insertion end up
054 * interleaved throughout the heap, and the old generation gets progressively
055 * more fragmented until a stop-the-world compacting collection occurs.
056 * <p>
057 * TODO: we should probably benchmark whether word-aligning the allocations
058 * would provide a performance improvement - probably would speed up the
059 * Bytes.toLong/Bytes.toInt calls in KeyValue, but some of those are cached
060 * anyway.
061 * The chunks created by this MemStoreLAB can get pooled at {@link ChunkCreator}.
062 * When the Chunk comes from pool, it can be either an on heap or an off heap backed chunk. The chunks,
063 * which this MemStoreLAB creates on its own (when no chunk available from pool), those will be
064 * always on heap backed.
065 */
066@InterfaceAudience.Private
067public class MemStoreLABImpl implements MemStoreLAB {
068
069  static final Log LOG = LogFactory.getLog(MemStoreLABImpl.class);
070
071  private AtomicReference<Chunk> curChunk = new AtomicReference<>();
072  // Lock to manage multiple handlers requesting for a chunk
073  private ReentrantLock lock = new ReentrantLock();
074
075  // A set of chunks contained by this memstore LAB
076  @VisibleForTesting
077  Set<Integer> chunks = new ConcurrentSkipListSet<Integer>();
078  private final int chunkSize;
079  private final int maxAlloc;
080  private final ChunkCreator chunkCreator;
081
082  // This flag is for closing this instance, its set when clearing snapshot of
083  // memstore
084  private volatile boolean closed = false;
085  // This flag is for reclaiming chunks. Its set when putting chunks back to
086  // pool
087  private AtomicBoolean reclaimed = new AtomicBoolean(false);
088  // Current count of open scanners which reading data from this MemStoreLAB
089  private final AtomicInteger openScannerCount = new AtomicInteger();
090
091  // Used in testing
092  public MemStoreLABImpl() {
093    this(new Configuration());
094  }
095
096  public MemStoreLABImpl(Configuration conf) {
097    chunkSize = conf.getInt(CHUNK_SIZE_KEY, CHUNK_SIZE_DEFAULT);
098    maxAlloc = conf.getInt(MAX_ALLOC_KEY, MAX_ALLOC_DEFAULT);
099    this.chunkCreator = ChunkCreator.getInstance();
100    // if we don't exclude allocations >CHUNK_SIZE, we'd infiniteloop on one!
101    Preconditions.checkArgument(maxAlloc <= chunkSize,
102        MAX_ALLOC_KEY + " must be less than " + CHUNK_SIZE_KEY);
103  }
104
105  @Override
106  public Cell copyCellInto(Cell cell) {
107    int size = KeyValueUtil.length(cell);
108    Preconditions.checkArgument(size >= 0, "negative size");
109    // Callers should satisfy large allocations directly from JVM since they
110    // don't cause fragmentation as badly.
111    if (size > maxAlloc) {
112      return null;
113    }
114    Chunk c = null;
115    int allocOffset = 0;
116    while (true) {
117      // Try to get the chunk
118      c = getOrMakeChunk();
119      // we may get null because the some other thread succeeded in getting the lock
120      // and so the current thread has to try again to make its chunk or grab the chunk
121      // that the other thread created
122      // Try to allocate from this chunk
123      if (c != null) {
124        allocOffset = c.alloc(size);
125        if (allocOffset != -1) {
126          // We succeeded - this is the common case - small alloc
127          // from a big buffer
128          break;
129        }
130        // not enough space!
131        // try to retire this chunk
132        tryRetireChunk(c);
133      }
134    }
135    return copyToChunkCell(cell, c.getData(), allocOffset, size);
136  }
137
138  /**
139   * Clone the passed cell by copying its data into the passed buf and create a cell with a chunkid
140   * out of it
141   */
142  private Cell copyToChunkCell(Cell cell, ByteBuffer buf, int offset, int len) {
143    int tagsLen = cell.getTagsLength();
144    if (cell instanceof ExtendedCell) {
145      ((ExtendedCell) cell).write(buf, offset);
146    } else {
147      // Normally all Cell impls within Server will be of type ExtendedCell. Just considering the
148      // other case also. The data fragments within Cell is copied into buf as in KeyValue
149      // serialization format only.
150      KeyValueUtil.appendTo(cell, buf, offset, true);
151    }
152    // TODO : write the seqid here. For writing seqId we should create a new cell type so
153    // that seqId is not used as the state
154    if (tagsLen == 0) {
155      // When tagsLen is 0, make a NoTagsByteBufferKeyValue version. This is an optimized class
156      // which directly return tagsLen as 0. So we avoid parsing many length components in
157      // reading the tagLength stored in the backing buffer. The Memstore addition of every Cell
158      // call getTagsLength().
159      return new NoTagByteBufferChunkCell(buf, offset, len, cell.getSequenceId());
160    } else {
161      return new ByteBufferChunkCell(buf, offset, len, cell.getSequenceId());
162    }
163  }
164
165  /**
166   * Close this instance since it won't be used any more, try to put the chunks
167   * back to pool
168   */
169  @Override
170  public void close() {
171    this.closed = true;
172    // We could put back the chunks to pool for reusing only when there is no
173    // opening scanner which will read their data
174    int count  = openScannerCount.get();
175    if(count == 0) {
176      recycleChunks();
177    }
178  }
179
180  /**
181   * Called when opening a scanner on the data of this MemStoreLAB
182   */
183  @Override
184  public void incScannerCount() {
185    this.openScannerCount.incrementAndGet();
186  }
187
188  /**
189   * Called when closing a scanner on the data of this MemStoreLAB
190   */
191  @Override
192  public void decScannerCount() {
193    int count = this.openScannerCount.decrementAndGet();
194    if (this.closed && count == 0) {
195      recycleChunks();
196    }
197  }
198
199  private void recycleChunks() {
200    if (reclaimed.compareAndSet(false, true)) {
201      chunkCreator.putbackChunks(chunks);
202    }
203  }
204
205  /**
206   * Try to retire the current chunk if it is still
207   * <code>c</code>. Postcondition is that curChunk.get()
208   * != c
209   * @param c the chunk to retire
210   * @return true if we won the race to retire the chunk
211   */
212  private void tryRetireChunk(Chunk c) {
213    curChunk.compareAndSet(c, null);
214    // If the CAS succeeds, that means that we won the race
215    // to retire the chunk. We could use this opportunity to
216    // update metrics on external fragmentation.
217    //
218    // If the CAS fails, that means that someone else already
219    // retired the chunk for us.
220  }
221
222  /**
223   * Get the current chunk, or, if there is no current chunk,
224   * allocate a new one from the JVM.
225   */
226  private Chunk getOrMakeChunk() {
227    // Try to get the chunk
228    Chunk c = curChunk.get();
229    if (c != null) {
230      return c;
231    }
232    // No current chunk, so we want to allocate one. We race
233    // against other allocators to CAS in an uninitialized chunk
234    // (which is cheap to allocate)
235    if (lock.tryLock()) {
236      try {
237        // once again check inside the lock
238        c = curChunk.get();
239        if (c != null) {
240          return c;
241        }
242        c = this.chunkCreator.getChunk();
243        if (c != null) {
244          // set the curChunk. No need of CAS as only one thread will be here
245          curChunk.set(c);
246          chunks.add(c.getId());
247          return c;
248        }
249      } finally {
250        lock.unlock();
251      }
252    }
253    return null;
254  }
255
256  @VisibleForTesting
257  Chunk getCurrentChunk() {
258    return this.curChunk.get();
259  }
260
261  @VisibleForTesting
262  BlockingQueue<Chunk> getPooledChunks() {
263    BlockingQueue<Chunk> pooledChunks = new LinkedBlockingQueue<>();
264    for (Integer id : this.chunks) {
265      Chunk chunk = chunkCreator.getChunk(id);
266      if (chunk != null && chunk.isFromPool()) {
267        pooledChunks.add(chunk);
268      }
269    }
270    return pooledChunks;
271  }
