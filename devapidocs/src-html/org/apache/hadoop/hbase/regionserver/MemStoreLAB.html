018package org.apache.hadoop.hbase.regionserver;
019
020import org.apache.hadoop.hbase.Cell;
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022
023/**
024 * A memstore-local allocation buffer.
025 * <p>
026 * The MemStoreLAB is basically a bump-the-pointer allocator that allocates big (2MB) chunks from
027 * and then doles it out to threads that request slices into the array. These chunks can get pooled
028 * as well. See {@link MemStoreChunkPool}.
029 * <p>
030 * The purpose of this is to combat heap fragmentation in the regionserver. By ensuring that all
031 * Cells in a given memstore refer only to large chunks of contiguous memory, we ensure that
032 * large blocks get freed up when the memstore is flushed.
033 * <p>
034 * Without the MSLAB, the byte array allocated during insertion end up interleaved throughout the
035 * heap, and the old generation gets progressively more fragmented until a stop-the-world compacting
036 * collection occurs.
037 * <p>
038 * This manages the large sized chunks. When Cells are to be added to Memstore, MemStoreLAB's
039 * {@link #copyCellInto(Cell)} gets called. This allocates enough size in the chunk to hold this
040 * cell's data and copies into this area and then recreate a Cell over this copied data.
041 * <p>
042 * @see MemStoreChunkPool
043 */
044@InterfaceAudience.Private
045public interface MemStoreLAB {
046
047  /**
048   * Allocates slice in this LAB and copy the passed Cell into this area. Returns new Cell instance
049   * over the copied the data. When this MemStoreLAB can not copy this Cell, it returns null.
050   */
051  Cell copyCellInto(Cell cell);
052
053  /**
054   * Close instance since it won't be used any more, try to put the chunks back to pool
055   */
056  void close();
057
058  /**
059   * Called when opening a scanner on the data of this MemStoreLAB
060   */
061  void incScannerCount();
062
063  /**
064   * Called when closing a scanner on the data of this MemStoreLAB
065   */
066  void decScannerCount();
067}
