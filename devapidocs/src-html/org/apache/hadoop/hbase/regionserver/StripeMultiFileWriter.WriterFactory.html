036/**
037 * Base class for cell sink that separates the provided cells into multiple files.
038 */
039@InterfaceAudience.Private
040public abstract class StripeMultiFileWriter implements Compactor.CellSink {
041  private static final Log LOG = LogFactory.getLog(StripeMultiFileWriter.class);
042
043  /** Factory that is used to produce single StoreFile.Writer-s */
044  protected WriterFactory writerFactory;
045  protected CellComparator comparator;
046
047  protected List<StoreFile.Writer> existingWriters;
048  protected List<byte[]> boundaries;
049  /** Source scanner that is tracking KV count; may be null if source is not StoreScanner */
050  protected StoreScanner sourceScanner;
051
052  /** Whether to write stripe metadata */
053  private boolean doWriteStripeMetadata = true;
054
055  public interface WriterFactory {
056    public StoreFile.Writer createWriter() throws IOException;
057  }
058
059  /**
060   * Initializes multi-writer before usage.
061   * @param sourceScanner Optional store scanner to obtain the information about read progress.
062   * @param factory Factory used to produce individual file writers.
063   * @param comparator Comparator used to compare rows.
064   */
065  public void init(StoreScanner sourceScanner, WriterFactory factory, CellComparator comparator)
066      throws IOException {
067    this.writerFactory = factory;
068    this.sourceScanner = sourceScanner;
069    this.comparator = comparator;
070  }
071
072  public void setNoStripeMetadata() {
073    this.doWriteStripeMetadata = false;
074  }
075
076  public List<Path> commitWriters(long maxSeqId, boolean isMajor) throws IOException {
077    assert this.existingWriters != null;
078    commitWritersInternal();
079    assert this.boundaries.size() == (this.existingWriters.size() + 1);
080    LOG.debug((this.doWriteStripeMetadata ? "W" : "Not w")
081      + "riting out metadata for " + this.existingWriters.size() + " writers");
082    List<Path> paths = new ArrayList<Path>();
083    for (int i = 0; i < this.existingWriters.size(); ++i) {
084      StoreFile.Writer writer = this.existingWriters.get(i);
085      if (writer == null) continue; // writer was skipped due to 0 KVs
086      if (doWriteStripeMetadata) {
087        writer.appendFileInfo(StripeStoreFileManager.STRIPE_START_KEY, this.boundaries.get(i));
088        writer.appendFileInfo(StripeStoreFileManager.STRIPE_END_KEY, this.boundaries.get(i + 1));
089      }
090      writer.appendMetadata(maxSeqId, isMajor);
091      paths.add(writer.getPath());
092      writer.close();
093    }
094    this.existingWriters = null;
095    return paths;
096  }
097
098  public List<Path> abortWriters() {
099    assert this.existingWriters != null;
100    List<Path> paths = new ArrayList<Path>();
101    for (StoreFile.Writer writer : this.existingWriters) {
102      try {
103        paths.add(writer.getPath());
104        writer.close();
105      } catch (Exception ex) {
106        LOG.error("Failed to close the writer after an unfinished compaction.", ex);
107      }
108    }
109    this.existingWriters = null;
110    return paths;
111  }
112
113  /**
114   * Subclasses can call this method to make sure the first KV is within multi-writer range.
115   * @param left The left boundary of the writer.
116   * @param cell The cell whose row has to be checked.
117   */
118  protected void sanityCheckLeft(
119      byte[] left, Cell cell) throws IOException {
120    if (!Arrays.equals(StripeStoreFileManager.OPEN_KEY, left) &&
121        comparator.compareRows(cell, left, 0, left.length) < 0) {
122      String error = "The first row is lower than the left boundary of [" + Bytes.toString(left)
123          + "]: [" + Bytes.toString(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength())
124          + "]";
125      LOG.error(error);
126      throw new IOException(error);
127    }
128  }
129
130  /**
131   * Subclasses can call this method to make sure the last KV is within multi-writer range.
132   * @param right The right boundary of the writer.
133   */
134  protected void sanityCheckRight(
135      byte[] right, Cell cell) throws IOException {
136    if (!Arrays.equals(StripeStoreFileManager.OPEN_KEY, right) &&
137        comparator.compareRows(cell, right, 0, right.length) >= 0) {
138      String error = "The last row is higher or equal than the right boundary of ["
139          + Bytes.toString(right) + "]: ["
140          + Bytes.toString(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength()) + "]";
141      LOG.error(error);
142      throw new IOException(error);
143    }
144  }
145
146  /**
147   * Subclasses override this method to be called at the end of a successful sequence of
148   * append; all appends are processed before this method is called.
149   */
150  protected abstract void commitWritersInternal() throws IOException;
151
152  /**
153   * MultiWriter that separates the cells based on fixed row-key boundaries.
154   * All the KVs between each pair of neighboring boundaries from the list supplied to ctor
155   * will end up in one file, and separate from all other such pairs.
156   */
157  public static class BoundaryMultiWriter extends StripeMultiFileWriter {
158    private StoreFile.Writer currentWriter;
159    private byte[] currentWriterEndKey;
160
161    private Cell lastCell;
162    private long cellsInCurrentWriter = 0;
163    private int majorRangeFromIndex = -1, majorRangeToIndex = -1;
164    private boolean hasAnyWriter = false;
165
166    /**
167     * @param targetBoundaries The boundaries on which writers/files are separated.
168     * @param majorRangeFrom Major range is the range for which at least one file should be
169     *                       written (because all files are included in compaction).
170     *                       majorRangeFrom is the left boundary.
171     * @param majorRangeTo The right boundary of majorRange (see majorRangeFrom).
172     */
173    public BoundaryMultiWriter(List<byte[]> targetBoundaries,
174        byte[] majorRangeFrom, byte[] majorRangeTo) throws IOException {
175      super();
176      this.boundaries = targetBoundaries;
177      this.existingWriters = new ArrayList<StoreFile.Writer>(this.boundaries.size() - 1);
178      // "major" range (range for which all files are included) boundaries, if any,
179      // must match some target boundaries, let's find them.
180      assert  (majorRangeFrom == null) == (majorRangeTo == null);
181      if (majorRangeFrom != null) {
182        majorRangeFromIndex = Arrays.equals(majorRangeFrom, StripeStoreFileManager.OPEN_KEY)
183                                ? 0
184                                : Collections.binarySearch(boundaries, majorRangeFrom,
185                                                           Bytes.BYTES_COMPARATOR);
186        majorRangeToIndex = Arrays.equals(majorRangeTo, StripeStoreFileManager.OPEN_KEY)
187                              ? boundaries.size()
188                              : Collections.binarySearch(boundaries, majorRangeTo,
189                                                         Bytes.BYTES_COMPARATOR);
190        if (this.majorRangeFromIndex < 0 || this.majorRangeToIndex < 0) {
191          throw new IOException("Major range does not match writer boundaries: [" +
192              Bytes.toString(majorRangeFrom) + "] [" + Bytes.toString(majorRangeTo) + "]; from "
193              + majorRangeFromIndex + " to " + majorRangeToIndex);
194        }
195      }
196    }
197
198    @Override
199    public void append(Cell cell) throws IOException {
200      if (currentWriter == null && existingWriters.isEmpty()) {
201        // First append ever, do a sanity check.
202        sanityCheckLeft(this.boundaries.get(0),
203            cell);
204      }
205      prepareWriterFor(cell);
206      currentWriter.append(cell);
207      lastCell = cell; // for the sanity check
208      ++cellsInCurrentWriter;
209    }
210
211    private boolean isCellAfterCurrentWriter(Cell cell) {
212      return !Arrays.equals(currentWriterEndKey, StripeStoreFileManager.OPEN_KEY) &&
213            (comparator.compareRows(cell, currentWriterEndKey, 0, currentWriterEndKey.length) >= 0);
214    }
215
216    @Override
217    protected void commitWritersInternal() throws IOException {
218      stopUsingCurrentWriter();
219      while (existingWriters.size() < boundaries.size() - 1) {
220        createEmptyWriter();
221      }
222      if (lastCell != null) {
223        sanityCheckRight(boundaries.get(boundaries.size() - 1),
224            lastCell);
225      }
226    }
227
228    private void prepareWriterFor(Cell cell) throws IOException {
229      if (currentWriter != null && !isCellAfterCurrentWriter(cell)) return; // Use same writer.
230
231      stopUsingCurrentWriter();
232      // See if KV will be past the writer we are about to create; need to add another one.
233      while (isCellAfterCurrentWriter(cell)) {
234        checkCanCreateWriter();
235        createEmptyWriter();
236      }
237      checkCanCreateWriter();
238      hasAnyWriter = true;
239      currentWriter = writerFactory.createWriter();
240      existingWriters.add(currentWriter);
241    }
242
243    /**
244     * Called if there are no cells for some stripe.
245     * We need to have something in the writer list for this stripe, so that writer-boundary
246     * list indices correspond to each other. We can insert null in the writer list for that
247     * purpose, except in the following cases where we actually need a file:
248     * 1) If we are in range for which we are compacting all the files, we need to create an
249     * empty file to preserve stripe metadata.
250     * 2) If we have not produced any file at all for this compactions, and this is the
251     * last chance (the last stripe), we need to preserve last seqNum (see also HBASE-6059).
252     */
253    private void createEmptyWriter() throws IOException {
254      int index = existingWriters.size();
255      boolean isInMajorRange = (index >= majorRangeFromIndex) && (index < majorRangeToIndex);
256      // Stripe boundary count = stripe count + 1, so last stripe index is (#boundaries minus 2)
257      boolean isLastWriter = !hasAnyWriter && (index == (boundaries.size() - 2));
258      boolean needEmptyFile = isInMajorRange || isLastWriter;
259      existingWriters.add(needEmptyFile ? writerFactory.createWriter() : null);
260      hasAnyWriter |= needEmptyFile;
261      currentWriterEndKey = (existingWriters.size() + 1 == boundaries.size())
262          ? null : boundaries.get(existingWriters.size() + 1);
263    }
264
265    private void checkCanCreateWriter() throws IOException {
266      int maxWriterCount =  boundaries.size() - 1;
267      assert existingWriters.size() <= maxWriterCount;
268      if (existingWriters.size() >= maxWriterCount) {
269        throw new IOException("Cannot create any more writers (created " + existingWriters.size()
270            + " out of " + maxWriterCount + " - row might be out of range of all valid writers");
271      }
272    }
273
274    private void stopUsingCurrentWriter() {
275      if (currentWriter != null) {
276        if (LOG.isDebugEnabled()) {
277          LOG.debug("Stopping to use a writer after [" + Bytes.toString(currentWriterEndKey)
278              + "] row; wrote out " + cellsInCurrentWriter + " kvs");
279        }
280        cellsInCurrentWriter = 0;
281      }
282      currentWriter = null;
283      currentWriterEndKey = (existingWriters.size() + 1 == boundaries.size())
284          ? null : boundaries.get(existingWriters.size() + 1);
285    }
286  }

288  /**
289   * MultiWriter that separates the cells based on target cell number per file and file count.
290   * New file is started every time the target number of KVs is reached, unless the fixed
291   * count of writers has already been created (in that case all the remaining KVs go into
292   * the last writer).
293   */
294  public static class SizeMultiWriter extends StripeMultiFileWriter {
295    private int targetCount;
296    private long targetCells;
297    private byte[] left;
298    private byte[] right;
299
300    private Cell lastCell;
301    private StoreFile.Writer currentWriter;
302    protected byte[] lastRowInCurrentWriter = null;
303    private long cellsInCurrentWriter = 0;
304    private long cellsSeen = 0;
305    private long cellsSeenInPrevious = 0;
306
307    /**
308     * @param targetCount The maximum count of writers that can be created.
309     * @param targetKvs The number of KVs to read from source before starting each new writer.
310     * @param left The left boundary of the first writer.
311     * @param right The right boundary of the last writer.
312     */
313    public SizeMultiWriter(int targetCount, long targetKvs, byte[] left, byte[] right) {
314      super();
315      this.targetCount = targetCount;
316      this.targetCells = targetKvs;
317      this.left = left;
318      this.right = right;
319      int preallocate = Math.min(this.targetCount, 64);
320      this.existingWriters = new ArrayList<StoreFile.Writer>(preallocate);
321      this.boundaries = new ArrayList<byte[]>(preallocate + 1);
322    }
323
324    @Override
325    public void append(Cell cell) throws IOException {
326      // If we are waiting for opportunity to close and we started writing different row,
327      // discard the writer and stop waiting.
328      boolean doCreateWriter = false;
329      if (currentWriter == null) {
330        // First append ever, do a sanity check.
331        sanityCheckLeft(left, cell);
332        doCreateWriter = true;
333      } else if (lastRowInCurrentWriter != null
334          && !CellUtil.matchingRow(cell,
335              lastRowInCurrentWriter, 0, lastRowInCurrentWriter.length)) {
336        if (LOG.isDebugEnabled()) {
337          LOG.debug("Stopping to use a writer after [" + Bytes.toString(lastRowInCurrentWriter)
338              + "] row; wrote out "  + cellsInCurrentWriter + " kvs");
339        }
340        lastRowInCurrentWriter = null;
341        cellsInCurrentWriter = 0;
342        cellsSeenInPrevious += cellsSeen;
343        doCreateWriter = true;
344      }
345      if (doCreateWriter) {
346        byte[] boundary = existingWriters.isEmpty() ? left : CellUtil.cloneRow(cell); // make a copy
347        if (LOG.isDebugEnabled()) {
348          LOG.debug("Creating new writer starting at [" + Bytes.toString(boundary) + "]");
349        }
350        currentWriter = writerFactory.createWriter();
351        boundaries.add(boundary);
352        existingWriters.add(currentWriter);
353      }
354
355      currentWriter.append(cell);
356      lastCell = cell; // for the sanity check
357      ++cellsInCurrentWriter;
358      cellsSeen = cellsInCurrentWriter;
359      if (this.sourceScanner != null) {
360        cellsSeen = Math.max(cellsSeen,
361            this.sourceScanner.getEstimatedNumberOfKvsScanned() - cellsSeenInPrevious);
362      }
363
364      // If we are not already waiting for opportunity to close, start waiting if we can
365      // create any more writers and if the current one is too big.
366      if (lastRowInCurrentWriter == null
367          && existingWriters.size() < targetCount
368          && cellsSeen >= targetCells) {
369        lastRowInCurrentWriter = CellUtil.cloneRow(cell); // make a copy
370        if (LOG.isDebugEnabled()) {
371          LOG.debug("Preparing to start a new writer after [" + Bytes.toString(
372              lastRowInCurrentWriter) + "] row; observed " + cellsSeen + " kvs and wrote out "
373              + cellsInCurrentWriter + " kvs");
374        }
375      }
376    }
377
378    @Override
379    protected void commitWritersInternal() throws IOException {
380      if (LOG.isDebugEnabled()) {
381        LOG.debug("Stopping with "  + cellsInCurrentWriter + " kvs in last writer" +
382            ((this.sourceScanner == null) ? "" : ("; observed estimated "
383                + this.sourceScanner.getEstimatedNumberOfKvsScanned() + " KVs total")));
384      }
385      if (lastCell != null) {
386        sanityCheckRight(
387            right, lastCell);
388      }
389
390      // When expired stripes were going to be merged into one, and if no writer was created during
391      // the compaction, we need to create an empty file to preserve metadata.
392      if (existingWriters.isEmpty() && 1 == targetCount) {
393        if (LOG.isDebugEnabled()) {
394          LOG.debug("Merge expired stripes into one, create an empty file to preserve metadata.");
395        }
396        boundaries.add(left);
397        existingWriters.add(writerFactory.createWriter());
398      }
399
400      this.boundaries.add(right);
401    }
402  }
