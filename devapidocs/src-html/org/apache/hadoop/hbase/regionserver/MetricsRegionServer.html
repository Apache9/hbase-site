018package org.apache.hadoop.hbase.regionserver;
019
020import org.apache.hadoop.hbase.CompatibilitySingletonFactory;
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022import org.apache.hadoop.hbase.classification.InterfaceStability;
023import org.apache.hadoop.hbase.metrics.MetricRegistries;
024import org.apache.hadoop.hbase.metrics.MetricRegistry;
025import org.apache.hadoop.hbase.metrics.Timer;
026
027import com.google.common.annotations.VisibleForTesting;
028
029/**
030 * <p>
031 * This class is for maintaining the various regionserver statistics
032 * and publishing them through the metrics interfaces.
033 * </p>
034 * This class has a number of metrics variables that are publicly accessible;
035 * these variables (objects) have methods to update their values.
036 */
037@InterfaceStability.Evolving
038@InterfaceAudience.Private
039public class MetricsRegionServer {
040  private MetricsRegionServerSource serverSource;
041  private MetricsRegionServerWrapper regionServerWrapper;
042
043  private MetricRegistry metricRegistry;
044  private Timer bulkLoadTimer;
045
046  public MetricsRegionServer(MetricsRegionServerWrapper regionServerWrapper) {
047    this(regionServerWrapper,
048        CompatibilitySingletonFactory.getInstance(MetricsRegionServerSourceFactory.class)
049            .createServer(regionServerWrapper));
050
051    // Create hbase-metrics module based metrics. The registry should already be registered by the
052    // MetricsRegionServerSource
053    metricRegistry = MetricRegistries.global().get(serverSource.getMetricRegistryInfo()).get();
054
055    // create and use metrics from the new hbase-metrics based registry.
056    bulkLoadTimer = metricRegistry.timer("Bulkload");
057  }
058
059  MetricsRegionServer(MetricsRegionServerWrapper regionServerWrapper,
060                      MetricsRegionServerSource serverSource) {
061    this.regionServerWrapper = regionServerWrapper;
062    this.serverSource = serverSource;
063  }
064
065  @VisibleForTesting
066  public MetricsRegionServerSource getMetricsSource() {
067    return serverSource;
068  }
069
070  public MetricsRegionServerWrapper getRegionServerWrapper() {
071    return regionServerWrapper;
072  }
073
074  public void updatePut(long t) {
075    if (t > 1000) {
076      serverSource.incrSlowPut();
077    }
078    serverSource.updatePut(t);
079  }
080
081  public void updateDelete(long t) {
082    if (t > 1000) {
083      serverSource.incrSlowDelete();
084    }
085    serverSource.updateDelete(t);
086  }
087
088  public void updateGet(long t) {
089    if (t > 1000) {
090      serverSource.incrSlowGet();
091    }
092    serverSource.updateGet(t);
093  }
094
095  public void updateIncrement(long t) {
096    if (t > 1000) {
097      serverSource.incrSlowIncrement();
098    }
099    serverSource.updateIncrement(t);
100  }
101
102  public void updateAppend(long t) {
103    if (t > 1000) {
104      serverSource.incrSlowAppend();
105    }
106    serverSource.updateAppend(t);
107  }
108
109  public void updateReplay(long t){
110    serverSource.updateReplay(t);
111  }
112
113  public void updateScanSize(long scanSize){
114    serverSource.updateScanSize(scanSize);
115  }
116
117  public void updateScanTime(long t) {
118    serverSource.updateScanTime(t);
119  }
120
121  public void updateSplitTime(long t) {
122    serverSource.updateSplitTime(t);
123  }
124
125  public void incrSplitRequest() {
126    serverSource.incrSplitRequest();
127  }
128
129  public void incrSplitSuccess() {
130    serverSource.incrSplitSuccess();
131  }
132
133  public void updateFlush(long t, long memstoreSize, long fileSize) {
134    serverSource.updateFlushTime(t);
135    serverSource.updateFlushMemstoreSize(memstoreSize);
136    serverSource.updateFlushOutputSize(fileSize);
137  }
138
139  public void updateCompaction(boolean isMajor, long t, int inputFileCount, int outputFileCount,
140      long inputBytes, long outputBytes) {
141    serverSource.updateCompactionTime(isMajor, t);
142    serverSource.updateCompactionInputFileCount(isMajor, inputFileCount);
143    serverSource.updateCompactionOutputFileCount(isMajor, outputFileCount);
144    serverSource.updateCompactionInputSize(isMajor, inputBytes);
145    serverSource.updateCompactionOutputSize(isMajor, outputBytes);
146  }
147
148  public void updateBulkLoad(long millis) {
149    this.bulkLoadTimer.updateMillis(millis);
150  }
151}
