018package org.apache.hadoop.hbase.regionserver;
019
020import com.google.common.annotations.VisibleForTesting;
021import org.apache.hadoop.hbase.classification.InterfaceAudience;
022import org.apache.hadoop.hbase.classification.InterfaceStability;
023import org.apache.hadoop.hbase.CompatibilitySingletonFactory;
024
025/**
026 * <p>
027 * This class is for maintaining the various regionserver statistics
028 * and publishing them through the metrics interfaces.
029 * </p>
030 * This class has a number of metrics variables that are publicly accessible;
031 * these variables (objects) have methods to update their values.
032 */
033@InterfaceStability.Evolving
034@InterfaceAudience.Private
035public class MetricsRegionServer {
036  private MetricsRegionServerSource serverSource;
037  private MetricsRegionServerWrapper regionServerWrapper;
038
039  public MetricsRegionServer(MetricsRegionServerWrapper regionServerWrapper) {
040    this(regionServerWrapper,
041        CompatibilitySingletonFactory.getInstance(MetricsRegionServerSourceFactory.class)
042            .createServer(regionServerWrapper));
043
044  }
045
046  MetricsRegionServer(MetricsRegionServerWrapper regionServerWrapper,
047                      MetricsRegionServerSource serverSource) {
048    this.regionServerWrapper = regionServerWrapper;
049    this.serverSource = serverSource;
050  }
051
052  @VisibleForTesting
053  public MetricsRegionServerSource getMetricsSource() {
054    return serverSource;
055  }
056
057  public MetricsRegionServerWrapper getRegionServerWrapper() {
058    return regionServerWrapper;
059  }
060
061  public void updatePut(long t) {
062    if (t > 1000) {
063      serverSource.incrSlowPut();
064    }
065    serverSource.updatePut(t);
066  }
067
068  public void updateDelete(long t) {
069    if (t > 1000) {
070      serverSource.incrSlowDelete();
071    }
072    serverSource.updateDelete(t);
073  }
074
075  public void updateGet(long t) {
076    if (t > 1000) {
077      serverSource.incrSlowGet();
078    }
079    serverSource.updateGet(t);
080  }
081
082  public void updateIncrement(long t) {
083    if (t > 1000) {
084      serverSource.incrSlowIncrement();
085    }
086    serverSource.updateIncrement(t);
087  }
088
089  public void updateAppend(long t) {
090    if (t > 1000) {
091      serverSource.incrSlowAppend();
092    }
093    serverSource.updateAppend(t);
094  }
095
096  public void updateReplay(long t){
097    serverSource.updateReplay(t);
098  }
099
100  public void updateScanSize(long scanSize){
101    serverSource.updateScanSize(scanSize);
102  }
103
104  public void updateScanTime(long t) {
105    serverSource.updateScanTime(t);
106  }
107
108  public void updateSplitTime(long t) {
109    serverSource.updateSplitTime(t);
110  }
111
112  public void incrSplitRequest() {
113    serverSource.incrSplitRequest();
114  }
115
116  public void incrSplitSuccess() {
117    serverSource.incrSplitSuccess();
118  }
119
120  public void updateFlush(long t, long memstoreSize, long fileSize) {
121    serverSource.updateFlushTime(t);
122    serverSource.updateFlushMemstoreSize(memstoreSize);
123    serverSource.updateFlushOutputSize(fileSize);
124  }
125
126  public void updateCompaction(boolean isMajor, long t, int inputFileCount, int outputFileCount,
127      long inputBytes, long outputBytes) {
128    serverSource.updateCompactionTime(isMajor, t);
129    serverSource.updateCompactionInputFileCount(isMajor, inputFileCount);
130    serverSource.updateCompactionOutputFileCount(isMajor, outputFileCount);
131    serverSource.updateCompactionInputSize(isMajor, inputBytes);
132    serverSource.updateCompactionOutputSize(isMajor, outputBytes);
133  }
134}
