071/**
072 * A Store data file.  Stores usually have one or more of these files.  They
073 * are produced by flushing the memstore to disk.  To
074 * create, instantiate a writer using {@link StoreFile.WriterBuilder}

076 * Writer (Use the appendMetadata convenience methods). On close, a StoreFile

078 * passing filesystem and path.  To read, call {@link #createReader()}.
079 * <p>StoreFiles may also reference store files in another Store.
080 *
081 * The reason for this weird pattern where you use a different instance for the
082 * writer and a reader is that we write once but read a lot more.

419  /**
420   * Opens reader on this store file.  Called by Constructor.
421   * @return Reader for the store file.
422   * @throws IOException
423   * @see #closeReader(boolean)
424   */
425  private Reader open(boolean canUseDropBehind) throws IOException {
426    if (this.reader != null) {
427      throw new IllegalAccessError("Already open");
428    }
429
430    // Open the StoreFile.Reader
431    this.reader = fileInfo.open(this.fs, this.cacheConf, canUseDropBehind);
432
433    // Load up indices and fileinfo. This also loads Bloom filter type.
434    metadataMap = Collections.unmodifiableMap(this.reader.loadFileInfo());
435
436    // Read in our metadata.
437    byte [] b = metadataMap.get(MAX_SEQ_ID_KEY);
438    if (b != null) {
439      // By convention, if halfhfile, top half has a sequence number > bottom
440      // half. Thats why we add one in below. Its done for case the two halves
441      // are ever merged back together --rare.  Without it, on open of store,
442      // since store files are distinguished by sequence id, the one half would
443      // subsume the other.
444      this.sequenceid = Bytes.toLong(b);
445      if (fileInfo.isTopReference()) {
446        this.sequenceid += 1;
447      }
448    }
449
450    if (isBulkLoadResult()){
451      // generate the sequenceId from the fileName
452      // fileName is of the form <randomName>_SeqId_<id-when-loaded>_
453      String fileName = this.getPath().getName();
454      // Use lastIndexOf() to get the last, most recent bulk load seqId.
455      int startPos = fileName.lastIndexOf("SeqId_");
456      if (startPos != -1) {
457        this.sequenceid = Long.parseLong(fileName.substring(startPos + 6,
458            fileName.indexOf('_', startPos + 6)));
459        // Handle reference files as done above.
460        if (fileInfo.isTopReference()) {
461          this.sequenceid += 1;
462        }
463      }
464      // SKIP_RESET_SEQ_ID only works in bulk loaded file.
465      // In mob compaction, the hfile where the cells contain the path of a new mob file is bulk
466      // loaded to hbase, these cells have the same seqIds with the old ones. We do not want
467      // to reset new seqIds for them since this might make a mess of the visibility of cells that
468      // have the same row key but different seqIds.
469      this.reader.setSkipResetSeqId(isSkipResetSeqId(metadataMap.get(SKIP_RESET_SEQ_ID)));
470      this.reader.setBulkLoaded(true);
471    }
472    this.reader.setSequenceID(this.sequenceid);
473
474    b = metadataMap.get(HFile.Writer.MAX_MEMSTORE_TS_KEY);
475    if (b != null) {
476      this.maxMemstoreTS = Bytes.toLong(b);
477    }
478
479    b = metadataMap.get(MAJOR_COMPACTION_KEY);
480    if (b != null) {
481      boolean mc = Bytes.toBoolean(b);
482      if (this.majorCompaction == null) {
483        this.majorCompaction = new AtomicBoolean(mc);
484      } else {
485        this.majorCompaction.set(mc);
486      }
487    } else {
488      // Presume it is not major compacted if it doesn't explicity say so
489      // HFileOutputFormat explicitly sets the major compacted key.
490      this.majorCompaction = new AtomicBoolean(false);
491    }
492
493    b = metadataMap.get(EXCLUDE_FROM_MINOR_COMPACTION_KEY);
494    this.excludeFromMinorCompaction = (b != null && Bytes.toBoolean(b));
495
496    BloomType hfileBloomType = reader.getBloomFilterType();
497    if (cfBloomType != BloomType.NONE) {
498      reader.loadBloomfilter(BlockType.GENERAL_BLOOM_META);
499      if (hfileBloomType != cfBloomType) {
500        LOG.info("HFile Bloom filter type for "
501            + reader.getHFileReader().getName() + ": " + hfileBloomType
502            + ", but " + cfBloomType + " specified in column family "
503            + "configuration");
504      }
505    } else if (hfileBloomType != BloomType.NONE) {
506      LOG.info("Bloom filter turned off by CF config for "
507          + reader.getHFileReader().getName());
508    }
509
510    // load delete family bloom filter
511    reader.loadBloomfilter(BlockType.DELETE_FAMILY_BLOOM_META);
512
513    try {
514      byte [] timerangeBytes = metadataMap.get(TIMERANGE_KEY);
515      if (timerangeBytes != null) {
516        this.reader.timeRangeTracker = new TimeRangeTracker();
517        Writables.copyWritable(timerangeBytes, this.reader.timeRangeTracker);
518      }
519    } catch (IllegalArgumentException e) {
520      LOG.error("Error reading timestamp range data from meta -- " +
521          "proceeding without", e);
522      this.reader.timeRangeTracker = null;
523    }
524    // initialize so we can reuse them after reader closed.
525    firstKey = reader.getFirstKey();
526    lastKey = reader.getLastKey();
527    comparator = reader.getComparator();
528    return this.reader;
529  }

536   * @return Reader for StoreFile. creates if necessary
537   * @throws IOException
538   */
539  public Reader createReader(boolean canUseDropBehind) throws IOException {
540    if (this.reader == null) {
541      try {
542        this.reader = open(canUseDropBehind);
543      } catch (IOException e) {
544        try {
545          boolean evictOnClose =
546              cacheConf != null? cacheConf.shouldEvictOnClose(): true; 
547          this.closeReader(evictOnClose);
548        } catch (IOException ee) {
549        }
550        throw e;
551      }
552
553    }
554    return this.reader;
555  }

774   * @return The split point row, or null if splitting is not possible, or reader is null.

776  @SuppressWarnings("deprecation")

778    if (this.reader == null) {
779      LOG.warn("Storefile " + this + " Reader is null; cannot get split point");
780      return null;
781    }
782    // Get first, last, and mid keys.  Midkey is the key that starts block
783    // in middle of hfile.  Has column and timestamp.  Need to return just
784    // the row we want to split on as midkey.
785    Cell midkey = this.reader.midkey();
786    if (midkey != null) {
787      Cell firstKey = this.reader.getFirstKey();
788      Cell lastKey = this.reader.getLastKey();
789      // if the midkey is the same as the first or last keys, we cannot (ever) split this region.
790      if (comparator.compareRows(midkey, firstKey) == 0
791          || comparator.compareRows(midkey, lastKey) == 0) {
792        if (LOG.isDebugEnabled()) {
793          LOG.debug("cannot split because midkey is the same as first or last row");
794        }
795        return null;
796      }
797      return CellUtil.cloneRow(midkey);
798    }
799    return null;
800  }

950    private void appendGeneralBloomfilter(final Cell cell) throws IOException {
951      if (this.generalBloomFilterWriter != null) {
952        // only add to the bloom filter on a new, unique key
953        boolean newKey = true;
954        if (this.lastCell != null) {
955          switch(bloomType) {
956          case ROW:
957            newKey = ! CellUtil.matchingRows(cell, lastCell);
958            break;
959          case ROWCOL:
960            newKey = ! CellUtil.matchingRowColumn(cell, lastCell);
961            break;
962          case NONE:
963            newKey = false;
964            break;
965          default:
966            throw new IOException("Invalid Bloom filter type: " + bloomType +
967                " (ROW or ROWCOL expected)");
968          }
969        }
970        if (newKey) {
971          /*
972           * http://2.bp.blogspot.com/_Cib_A77V54U/StZMrzaKufI/AAAAAAAAADo/ZhK7bGoJdMQ/s400/KeyValue.png
973           * Key = RowLen + Row + FamilyLen + Column [Family + Qualifier] + TimeStamp
974           *
975           * 2 Types of Filtering:
976           *  1. Row = Row
977           *  2. RowCol = Row + Qualifier
978           */
979          byte[] bloomKey = null;
980          // Used with ROW_COL bloom
981          KeyValue bloomKeyKV = null;
982          int bloomKeyOffset, bloomKeyLen;
983
984          switch (bloomType) {
985          case ROW:
986            bloomKey = cell.getRowArray();
987            bloomKeyOffset = cell.getRowOffset();
988            bloomKeyLen = cell.getRowLength();
989            break;
990          case ROWCOL:
991            // merge(row, qualifier)
992            // TODO: could save one buffer copy in case of compound Bloom
993            // filters when this involves creating a KeyValue
994            // TODO : Handle while writes also
995            bloomKeyKV = KeyValueUtil.createFirstOnRow(cell.getRowArray(), cell.getRowOffset(),
996                cell.getRowLength(), 
997                HConstants.EMPTY_BYTE_ARRAY, 0, 0, cell.getQualifierArray(),
998                cell.getQualifierOffset(),
999                cell.getQualifierLength());
1000            bloomKey = bloomKeyKV.getBuffer();
1001            bloomKeyOffset = bloomKeyKV.getKeyOffset();
1002            bloomKeyLen = bloomKeyKV.getKeyLength();
1003            break;
1004          default:
1005            throw new IOException("Invalid Bloom filter type: " + bloomType +
1006                " (ROW or ROWCOL expected)");
1007          }
1008          generalBloomFilterWriter.add(bloomKey, bloomKeyOffset, bloomKeyLen);
1009          if (lastBloomKey != null) {
1010            int res = 0;
1011            // hbase:meta does not have blooms. So we need not have special interpretation
1012            // of the hbase:meta cells.  We can safely use Bytes.BYTES_RAWCOMPARATOR for ROW Bloom
1013            if (bloomType == BloomType.ROW) {
1014              res = Bytes.BYTES_RAWCOMPARATOR.compare(bloomKey, bloomKeyOffset, bloomKeyLen,
1015                  lastBloomKey, lastBloomKeyOffset, lastBloomKeyLen);
1016            } else {
1017              // TODO : Caching of kv components becomes important in these cases
1018              res = CellComparator.COMPARATOR.compare(bloomKeyKV, lastBloomKeyOnlyKV);
1019            }
1020            if (res <= 0) {
1021              throw new IOException("Non-increasing Bloom keys: "
1022                  + Bytes.toStringBinary(bloomKey, bloomKeyOffset, bloomKeyLen) + " after "
1023                  + Bytes.toStringBinary(lastBloomKey, lastBloomKeyOffset, lastBloomKeyLen));
1024            }
1025          }
1026          lastBloomKey = bloomKey;
1027          lastBloomKeyOffset = bloomKeyOffset;
1028          lastBloomKeyLen = bloomKeyLen;
1029          if (bloomType == BloomType.ROWCOL) {
1030            lastBloomKeyOnlyKV.setKey(bloomKey, bloomKeyOffset, bloomKeyLen);
1031          }
1032          this.lastCell = cell;
1033        }
1034      }
1035    }

1037    private void appendDeleteFamilyBloomFilter(final Cell cell)
1038        throws IOException {
1039      if (!CellUtil.isDeleteFamily(cell) && !CellUtil.isDeleteFamilyVersion(cell)) {
1040        return;
1041      }
1042
1043      // increase the number of delete family in the store file
1044      deleteFamilyCnt++;
1045      if (null != this.deleteFamilyBloomFilterWriter) {
1046        boolean newKey = true;
1047        if (lastDeleteFamilyCell != null) {

1049          // of the hbase:meta cells
1050          newKey = !CellUtil.matchingRows(cell, lastDeleteFamilyCell);
1051        }
1052        if (newKey) {
1053          this.deleteFamilyBloomFilterWriter.add(cell.getRowArray(),
1054              cell.getRowOffset(), cell.getRowLength());
1055          this.lastDeleteFamilyCell = cell;
1056        }
1057      }

1328     * @param scan the scan specification. Used to determine the row, and to
1329     *          check whether this is a single-row ("get") scan.
1330     * @param columns the set of columns. Only used for row-column Bloom
1331     *          filters.
1332     * @return true if the scan with the given column set passes the Bloom
1333     *         filter, or if the Bloom filter is not applicable for the scan.
1334     *         False if the Bloom filter is applicable and the scan fails it.
1335     */
1336     boolean passesBloomFilter(Scan scan,
1337        final SortedSet<byte[]> columns) {
1338      // Multi-column non-get scans will use Bloom filters through the
1339      // lower-level API function that this function calls.
1340      if (!scan.isGetScan()) {
1341        return true;
1342      }
1343
1344      byte[] row = scan.getStartRow();
1345      switch (this.bloomFilterType) {
1346        case ROW:
1347          return passesGeneralRowBloomFilter(row, 0, row.length);
1348
1349        case ROWCOL:
1350          if (columns != null && columns.size() == 1) {
1351            byte[] column = columns.first();
1352            // create the required fake key
1353            Cell kvKey = KeyValueUtil.createFirstOnRow(row, 0, row.length,
1354              HConstants.EMPTY_BYTE_ARRAY, 0, 0, column, 0,
1355              column.length);
1356            return passesGeneralRowColBloomFilter(kvKey);
1357          }
1358
1359          // For multi-column queries the Bloom filter is checked from the
1360          // seekExact operation.
1361          return true;
1362
1363        default:
1364          return true;
1365      }
1366    }

1368    public boolean passesDeleteFamilyBloomFilter(byte[] row, int rowOffset,
1369        int rowLen) {
1370      // Cache Bloom filter as a local variable in case it is set to null by
1371      // another thread on an IO error.
1372      BloomFilter bloomFilter = this.deleteFamilyBloomFilter;
1373
1374      // Empty file or there is no delete family at all
1375      if (reader.getTrailer().getEntryCount() == 0 || deleteFamilyCnt == 0) {
1376        return false;
1377      }
1378
1379      if (bloomFilter == null) {
1380        return true;
1381      }
1382
1383      try {
1384        if (!bloomFilter.supportsAutoLoading()) {
1385          return true;
1386        }
1387        return bloomFilter.contains(row, rowOffset, rowLen, null);
1388      } catch (IllegalArgumentException e) {
1389        LOG.error("Bad Delete Family bloom filter data -- proceeding without",
1390            e);
1391        setDeleteFamilyBloomFilterFaulty();
1392      }
1393
1394      return true;
1395    }

1398     * A method for checking Bloom filters. Called directly from
1399     * StoreFileScanner in case of a multi-column query.
1400     *
1401     * @param row
1402     * @param rowOffset
1403     * @param rowLen
1404     * @return True if passes
1405     */
1406    public boolean passesGeneralRowBloomFilter(byte[] row, int rowOffset, int rowLen) {
1407      BloomFilter bloomFilter = this.generalBloomFilter;
1408      if (bloomFilter == null) {
1409        return true;
1410      }
1411
1412      // Used in ROW bloom
1413      byte[] key = null;
1414      if (rowOffset != 0 || rowLen != row.length) {
1415        throw new AssertionError(
1416            "For row-only Bloom filters the row " + "must occupy the whole array");
1417      }
1418      key = row;
1419      return checkGeneralBloomFilter(key, null, bloomFilter);
1420    }

1422    /**
1423     * A method for checking Bloom filters. Called directly from
1424     * StoreFileScanner in case of a multi-column query.
1425     *
1426     * @param cell
1427     *          the cell to check if present in BloomFilter
1428     * @return True if passes
1429     */
1430    public boolean passesGeneralRowColBloomFilter(Cell cell) {
1431      BloomFilter bloomFilter = this.generalBloomFilter;
1432      if (bloomFilter == null) {
1433        return true;
1434      }
1435      // Used in ROW_COL bloom
1436      Cell kvKey = null;
1437      // Already if the incoming key is a fake rowcol key then use it as it is
1438      if (cell.getTypeByte() == KeyValue.Type.Maximum.getCode() && cell.getFamilyLength() == 0) {
1439        kvKey = cell;
1440      } else {
1441        kvKey = CellUtil.createFirstOnRowCol(cell);
1442      }
1443      return checkGeneralBloomFilter(null, kvKey, bloomFilter);
1444    }

1446    private boolean checkGeneralBloomFilter(byte[] key, Cell kvKey, BloomFilter bloomFilter) {
1447      // Empty file
1448      if (reader.getTrailer().getEntryCount() == 0)
1449        return false;
1450      HFileBlock bloomBlock = null;
1451      try {
1452        boolean shouldCheckBloom;
1453        ByteBuff bloom;
1454        if (bloomFilter.supportsAutoLoading()) {
1455          bloom = null;
1456          shouldCheckBloom = true;
1457        } else {
1458          bloomBlock = reader.getMetaBlock(HFile.BLOOM_FILTER_DATA_KEY, true);
1459          bloom = bloomBlock.getBufferWithoutHeader();
1460          shouldCheckBloom = bloom != null;
1461        }
1462
1463        if (shouldCheckBloom) {
1464          boolean exists;
1465
1466          // Whether the primary Bloom key is greater than the last Bloom key
1467          // from the file info. For row-column Bloom filters this is not yet
1468          // a sufficient condition to return false.
1469          boolean keyIsAfterLast = (lastBloomKey != null);
1470          // hbase:meta does not have blooms. So we need not have special interpretation
1471          // of the hbase:meta cells.  We can safely use Bytes.BYTES_RAWCOMPARATOR for ROW Bloom
1472          if (keyIsAfterLast) {
1473            if (bloomFilterType == BloomType.ROW) {
1474              keyIsAfterLast = (Bytes.BYTES_RAWCOMPARATOR.compare(key, lastBloomKey) > 0);
1475            } else {
1476              keyIsAfterLast = (CellComparator.COMPARATOR.compare(kvKey, lastBloomKeyOnlyKV)) > 0;
1477            }
1478          }
1479
1480          if (bloomFilterType == BloomType.ROWCOL) {
1481            // Since a Row Delete is essentially a DeleteFamily applied to all
1482            // columns, a file might be skipped if using row+col Bloom filter.
1483            // In order to ensure this file is included an additional check is
1484            // required looking only for a row bloom.
1485            Cell rowBloomKey = CellUtil.createFirstOnRow(kvKey);
1486            // hbase:meta does not have blooms. So we need not have special interpretation
1487            // of the hbase:meta cells.  We can safely use Bytes.BYTES_RAWCOMPARATOR for ROW Bloom
1488            if (keyIsAfterLast
1489                && (CellComparator.COMPARATOR.compare(rowBloomKey, lastBloomKeyOnlyKV)) > 0) {
1490              exists = false;
1491            } else {
1492              exists =
1493                  bloomFilter.contains(kvKey, bloom) ||
1494                  bloomFilter.contains(rowBloomKey, bloom);
1495            }
1496          } else {
1497            exists = !keyIsAfterLast
1498                && bloomFilter.contains(key, 0, key.length, bloom);
1499          }
1500
1501          return exists;
1502        }
1503      } catch (IOException e) {
1504        LOG.error("Error reading bloom filter data -- proceeding without",
1505            e);
1506        setGeneralBloomFilterFaulty();
1507      } catch (IllegalArgumentException e) {
1508        LOG.error("Bad bloom filter data -- proceeding without", e);
1509        setGeneralBloomFilterFaulty();
1510      } finally {
1511        // Return the bloom block so that its ref count can be decremented.
1512        reader.returnBlock(bloomBlock);
1513      }
1514      return true;
1515    }

1517    /**
1518     * Checks whether the given scan rowkey range overlaps with the current storefile's
1519     * @param scan the scan specification. Used to determine the rowkey range.
1520     * @return true if there is overlap, false otherwise
1521     */
1522    public boolean passesKeyRangeFilter(Scan scan) {
1523      if (this.getFirstKey() == null || this.getLastKey() == null) {
1524        // the file is empty
1525        return false;
1526      }
1527      if (Bytes.equals(scan.getStartRow(), HConstants.EMPTY_START_ROW)
1528          && Bytes.equals(scan.getStopRow(), HConstants.EMPTY_END_ROW)) {
1529        return true;
1530      }
1531      byte[] smallestScanRow = scan.isReversed() ? scan.getStopRow() : scan.getStartRow();
1532      byte[] largestScanRow = scan.isReversed() ? scan.getStartRow() : scan.getStopRow();
1533      Cell firstKeyKV = this.getFirstKey();
1534      Cell lastKeyKV = this.getLastKey();
1535      boolean nonOverLapping = (getComparator().compareRows(firstKeyKV,
1536          largestScanRow, 0, largestScanRow.length) > 0 
1537          && !Bytes
1538          .equals(scan.isReversed() ? scan.getStartRow() : scan.getStopRow(),
1539              HConstants.EMPTY_END_ROW))
1540          || getComparator().compareRows(lastKeyKV, smallestScanRow, 0, smallestScanRow.length) < 0;
1541      return !nonOverLapping;
1542    }

1569    private void loadBloomfilter(BlockType blockType) {
1570      try {
1571        if (blockType == BlockType.GENERAL_BLOOM_META) {
1572          if (this.generalBloomFilter != null)
1573            return; // Bloom has been loaded
1574
1575          DataInput bloomMeta = reader.getGeneralBloomFilterMetadata();
1576          if (bloomMeta != null) {
1577            // sanity check for NONE Bloom filter
1578            if (bloomFilterType == BloomType.NONE) {
1579              throw new IOException(
1580                  "valid bloom filter type not found in FileInfo");
1581            } else {
1582              generalBloomFilter = BloomFilterFactory.createFromMeta(bloomMeta,
1583                  reader);
1584              if (LOG.isTraceEnabled()) {
1585                LOG.trace("Loaded " + bloomFilterType.toString() + " "
1586                  + generalBloomFilter.getClass().getSimpleName()
1587                  + " metadata for " + reader.getName());
1588              }
1589            }
1590          }
1591        } else if (blockType == BlockType.DELETE_FAMILY_BLOOM_META) {
1592          if (this.deleteFamilyBloomFilter != null)
1593            return; // Bloom has been loaded
1594
1595          DataInput bloomMeta = reader.getDeleteBloomFilterMetadata();
1596          if (bloomMeta != null) {
1597            deleteFamilyBloomFilter = BloomFilterFactory.createFromMeta(
1598                bloomMeta, reader);
1599            LOG.info("Loaded Delete Family Bloom ("
1600                + deleteFamilyBloomFilter.getClass().getSimpleName()
1601                + ") metadata for " + reader.getName());
1602          }
1603        } else {
1604          throw new RuntimeException("Block Type: " + blockType.toString()
1605              + "is not supported for Bloom filter");
1606        }
1607      } catch (IOException e) {
1608        LOG.error("Error reading bloom filter meta for " + blockType
1609            + " -- proceeding without", e);
1610        setBloomFilterFaulty(blockType);
1611      } catch (IllegalArgumentException e) {
1612        LOG.error("Bad bloom filter meta " + blockType
1613            + " -- proceeding without", e);
1614        setBloomFilterFaulty(blockType);
1615      }
1616    }

1746  /**
1747   * Useful comparators for comparing StoreFiles.
1748   */
1749  public abstract static class Comparators {
1750    /**
1751     * Comparator that compares based on the Sequence Ids of the
1752     * the StoreFiles. Bulk loads that did not request a seq ID
1753     * are given a seq id of -1; thus, they are placed before all non-
1754     * bulk loads, and bulk loads with sequence Id. Among these files,
1755     * the size is used to determine the ordering, then bulkLoadTime.
1756     * If there are ties, the path name is used as a tie-breaker.
1757     */
1758    public static final Comparator<StoreFile> SEQ_ID =
1759      Ordering.compound(ImmutableList.of(
1760          Ordering.natural().onResultOf(new GetSeqId()),
1761          Ordering.natural().onResultOf(new GetFileSize()).reverse(),
1762          Ordering.natural().onResultOf(new GetBulkTime()),
1763          Ordering.natural().onResultOf(new GetPathName())
1764      ));

1766    private static class GetSeqId implements Function<StoreFile, Long> {
1767      @Override
1768      public Long apply(StoreFile sf) {
1769        return sf.getMaxSequenceId();
1770      }
1771    }

1773    private static class GetFileSize implements Function<StoreFile, Long> {
1774      @Override
1775      public Long apply(StoreFile sf) {
1776        if (sf.getReader() != null) {
1777          return sf.getReader().length();
1778        } else {
1779          // the reader may be null for the compacted files and if the archiving
1780          // had failed.
1781          return -1L;
1782        }
1783      }
1784    }

1787      @Override
1788      public Long apply(StoreFile sf) {
1789        if (!sf.isBulkLoadResult()) return Long.MAX_VALUE;
1790        return sf.getBulkLoadTimestamp();
