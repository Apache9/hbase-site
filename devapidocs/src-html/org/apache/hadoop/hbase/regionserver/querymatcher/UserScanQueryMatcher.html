033/**
034 * Query matcher for user scan.
035 * <p>
036 * We do not consider mvcc here because
037 * {@link org.apache.hadoop.hbase.regionserver.StoreFileScanner} and
038 * {@link org.apache.hadoop.hbase.regionserver.SegmentScanner} will only return a cell whose mvcc is
039 * less than or equal to given read point. For
040 * {@link org.apache.hadoop.hbase.client.IsolationLevel#READ_UNCOMMITTED}, we just set the read
041 * point to {@link Long#MAX_VALUE}, i.e. still do not need to consider it.
042 */
043@InterfaceAudience.Private
044public abstract class UserScanQueryMatcher extends ScanQueryMatcher {
045
046  protected final boolean hasNullColumn;
047
048  protected final Filter filter;
049
050  protected final byte[] stopRow;
051
052  protected final TimeRange tr;
053
054  private static Cell createStartKey(Scan scan, ScanInfo scanInfo) {
055    if (scan.includeStartRow()) {
056      return createStartKeyFromRow(scan.getStartRow(), scanInfo);
057    } else {
058      return CellUtil.createLastOnRow(scan.getStartRow());
059    }
060  }
061
062  protected UserScanQueryMatcher(Scan scan, ScanInfo scanInfo, ColumnTracker columns,
063      boolean hasNullColumn, long oldestUnexpiredTS, long now) {
064    super(createStartKey(scan, scanInfo), scanInfo, columns, oldestUnexpiredTS, now);
065    this.hasNullColumn = hasNullColumn;
066    this.filter = scan.getFilter();
067    this.stopRow = scan.getStopRow();
068    TimeRange timeRange = scan.getColumnFamilyTimeRange().get(scanInfo.getFamily());
069    if (timeRange == null) {
070      this.tr = scan.getTimeRange();
071    } else {
072      this.tr = timeRange;
073    }
074  }
075
076  @Override
077  public boolean hasNullColumnInQuery() {
078    return hasNullColumn;
079  }
080
081  @Override
082  public boolean isUserScan() {
083    return true;
084  }
085
086  @Override
087  public Filter getFilter() {
088    return filter;
089  }
090
091  @Override
092  public Cell getNextKeyHint(Cell cell) throws IOException {
093    if (filter == null) {
094      return null;
095    } else {
096      return filter.getNextCellHint(cell);
097    }
098  }
099
100  protected final MatchCode matchColumn(Cell cell, long timestamp, byte typeByte)
101      throws IOException {
102    int tsCmp = tr.compare(timestamp);
103    if (tsCmp > 0) {
104      return MatchCode.SKIP;
105    }
106    if (tsCmp < 0) {
107      return columns.getNextRowOrNextColumn(cell);
108    }
109    // STEP 1: Check if the column is part of the requested columns
110    MatchCode colChecker = columns.checkColumn(cell, typeByte);
111    if (colChecker != MatchCode.INCLUDE) {
112      return colChecker;
113    }
114    ReturnCode filterResponse = ReturnCode.SKIP;
115    // STEP 2: Yes, the column is part of the requested columns. Check if filter is present
116    if (filter != null) {
117      // STEP 3: Filter the key value and return if it filters out
118      filterResponse = filter.filterKeyValue(cell);
119      switch (filterResponse) {
120        case SKIP:
121          return MatchCode.SKIP;
122        case NEXT_COL:
123          return columns.getNextRowOrNextColumn(cell);
124        case NEXT_ROW:
125          return MatchCode.SEEK_NEXT_ROW;
126        case SEEK_NEXT_USING_HINT:
127          return MatchCode.SEEK_NEXT_USING_HINT;
128        default:
129          // It means it is either include or include and seek next
130          break;
131      }
132    }
133    /*
134     * STEP 4: Reaching this step means the column is part of the requested columns and either
135     * the filter is null or the filter has returned INCLUDE or INCLUDE_AND_NEXT_COL response.
136     * Now check the number of versions needed. This method call returns SKIP, INCLUDE,
137     * INCLUDE_AND_SEEK_NEXT_ROW, INCLUDE_AND_SEEK_NEXT_COL.
138     *
139     * FilterResponse            ColumnChecker               Desired behavior
140     * INCLUDE                   SKIP                        row has already been included, SKIP.
141     * INCLUDE                   INCLUDE                     INCLUDE
142     * INCLUDE                   INCLUDE_AND_SEEK_NEXT_COL   INCLUDE_AND_SEEK_NEXT_COL
143     * INCLUDE                   INCLUDE_AND_SEEK_NEXT_ROW   INCLUDE_AND_SEEK_NEXT_ROW
144     * INCLUDE_AND_SEEK_NEXT_COL SKIP                        row has already been included, SKIP.
145     * INCLUDE_AND_SEEK_NEXT_COL INCLUDE                     INCLUDE_AND_SEEK_NEXT_COL
146     * INCLUDE_AND_SEEK_NEXT_COL INCLUDE_AND_SEEK_NEXT_COL   INCLUDE_AND_SEEK_NEXT_COL
147     * INCLUDE_AND_SEEK_NEXT_COL INCLUDE_AND_SEEK_NEXT_ROW   INCLUDE_AND_SEEK_NEXT_ROW
148     *
149     * In all the above scenarios, we return the column checker return value except for
150     * FilterResponse (INCLUDE_AND_SEEK_NEXT_COL) and ColumnChecker(INCLUDE)
151     */
152    colChecker = columns.checkVersions(cell, timestamp, typeByte, false);
153    if (filterResponse == ReturnCode.INCLUDE_AND_SEEK_NEXT_ROW) {
154      if (colChecker != MatchCode.SKIP) {
155        return MatchCode.INCLUDE_AND_SEEK_NEXT_ROW;
156      }
157      return MatchCode.SEEK_NEXT_ROW;
158    }
159    return (filterResponse == ReturnCode.INCLUDE_AND_NEXT_COL && colChecker == MatchCode.INCLUDE)
160        ? MatchCode.INCLUDE_AND_SEEK_NEXT_COL : colChecker;
161  }
162
163  protected abstract boolean isGet();
164
165  protected abstract boolean moreRowsMayExistsAfter(int cmpToStopRow);
166
167  @Override
168  public boolean moreRowsMayExistAfter(Cell cell) {
169    // If a 'get' Scan -- we are doing a Get (every Get is a single-row Scan in implementation) --
170    // then we are looking at one row only, the one specified in the Get coordinate..so we know
171    // for sure that there are no more rows on this Scan
172    if (isGet()) {
173      return false;
174    }
175    // If no stopRow, return that there may be more rows. The tests that follow depend on a
176    // non-empty, non-default stopRow so this little test below short-circuits out doing the
177    // following compares.
178    if (this.stopRow == null || this.stopRow.length == 0) {
179      return true;
180    }
181    return moreRowsMayExistsAfter(rowComparator.compareRows(cell, stopRow, 0, stopRow.length));
182  }
183
184  public static UserScanQueryMatcher create(Scan scan, ScanInfo scanInfo,
185      NavigableSet<byte[]> columns, long oldestUnexpiredTS, long now,
186      RegionCoprocessorHost regionCoprocessorHost) throws IOException {
187    int maxVersions = scan.isRaw() ? scan.getMaxVersions()
188        : Math.min(scan.getMaxVersions(), scanInfo.getMaxVersions());
189    boolean hasNullColumn;
190    ColumnTracker columnTracker;
191    if (columns == null || columns.isEmpty()) {
192      // there is always a null column in the wildcard column query.
193      hasNullColumn = true;
194      // use a specialized scan for wildcard column tracker.
195      columnTracker = new ScanWildcardColumnTracker(scanInfo.getMinVersions(), maxVersions,
196          oldestUnexpiredTS);
197    } else {
198      // We can share the ExplicitColumnTracker, diff is we reset
199      // between rows, not between storefiles.
200      // whether there is null column in the explicit column query
201      hasNullColumn = columns.first().length == 0;
202      columnTracker = new ExplicitColumnTracker(columns, scanInfo.getMinVersions(), maxVersions,
203          oldestUnexpiredTS);
204    }
205    if (scan.isRaw()) {
206      return RawScanQueryMatcher.create(scan, scanInfo, columnTracker, hasNullColumn,
207        oldestUnexpiredTS, now);
208    } else {
209      return NormalUserScanQueryMatcher.create(scan, scanInfo, columnTracker, hasNullColumn,
210        oldestUnexpiredTS, now, regionCoprocessorHost);
211    }
212  }
