047/**
048 * Base class for implementing a Compactor which will generate multiple output files after
049 * compaction.
050 */
051@InterfaceAudience.Private
052public abstract class AbstractMultiOutputCompactor<T extends AbstractMultiFileWriter>
053    extends Compactor {
054
055  private static final Log LOG = LogFactory.getLog(AbstractMultiOutputCompactor.class);
056
057  public AbstractMultiOutputCompactor(Configuration conf, Store store) {
058    super(conf, store);
059  }
060
061  protected interface InternalScannerFactory {
062
063    ScanType getScanType(CompactionRequest request);
064
065    InternalScanner createScanner(List<StoreFileScanner> scanners, ScanType scanType,
066        FileDetails fd, long smallestReadPoint) throws IOException;
067  }
068
069  protected List<Path> compact(T writer, final CompactionRequest request,
070      InternalScannerFactory scannerFactory, ThroughputController throughputController, User user)
071          throws IOException {
072    final FileDetails fd = getFileDetails(request.getFiles(), request.isAllFiles());
073    this.progress = new CompactionProgress(fd.maxKeyCount);
074
075    // Find the smallest read point across all the Scanners.
076    long smallestReadPoint = getSmallestReadPoint();
077
078    List<StoreFileScanner> scanners;
079    Collection<StoreFile> readersToClose;
080    if (this.conf.getBoolean("hbase.regionserver.compaction.private.readers", true)) {
081      // clone all StoreFiles, so we'll do the compaction on a independent copy of StoreFiles,
082      // HFiles, and their readers
083      readersToClose = new ArrayList<StoreFile>(request.getFiles().size());
084      for (StoreFile f : request.getFiles()) {
085        readersToClose.add(f.cloneForReader());
086      }
087      scanners = createFileScanners(readersToClose, smallestReadPoint,
088        store.throttleCompaction(request.getSize()));
089    } else {
090      readersToClose = Collections.emptyList();
091      scanners = createFileScanners(request.getFiles(), smallestReadPoint,
092        store.throttleCompaction(request.getSize()));
093    }
094    InternalScanner scanner = null;
095    boolean finished = false;
096    try {
097      /* Include deletes, unless we are doing a major compaction */
098      ScanType scanType = scannerFactory.getScanType(request);
099      scanner = preCreateCoprocScanner(request, scanType, fd.earliestPutTs, scanners);
100      if (scanner == null) {
101        scanner = scannerFactory.createScanner(scanners, scanType, fd, smallestReadPoint);
102      }
103      scanner = postCreateCoprocScanner(request, scanType, scanner, user);
104      if (scanner == null) {
105        // NULL scanner returned from coprocessor hooks means skip normal processing.
106        return new ArrayList<Path>();
107      }
108      boolean cleanSeqId = false;
109      if (fd.minSeqIdToKeep > 0) {
110        smallestReadPoint = Math.min(fd.minSeqIdToKeep, smallestReadPoint);
111        cleanSeqId = true;
112      }
113      // Create the writer factory for compactions.
114      final boolean needMvcc = fd.maxMVCCReadpoint >= 0;
115      WriterFactory writerFactory = new WriterFactory() {
116        @Override
117        public Writer createWriter() throws IOException {
118          return store.createWriterInTmp(fd.maxKeyCount, compactionCompression, true, needMvcc,
119            fd.maxTagsLength > 0, store.throttleCompaction(request.getSize()));
120        }
121      };
122      // Prepare multi-writer, and perform the compaction using scanner and writer.
123      // It is ok here if storeScanner is null.
124      StoreScanner storeScanner
125        = (scanner instanceof StoreScanner) ? (StoreScanner) scanner : null;
126      writer.init(storeScanner, writerFactory);
127      finished = performCompaction(fd, scanner, writer, smallestReadPoint, cleanSeqId,
128        throughputController, request.isAllFiles());
129      if (!finished) {
130        throw new InterruptedIOException("Aborting compaction of store " + store + " in region "
131            + store.getRegionInfo().getRegionNameAsString() + " because it was interrupted.");
132      }
133    } finally {
134      Closeables.close(scanner, true);
135      for (StoreFile f : readersToClose) {
136        try {
137          f.closeReader(true);
138        } catch (IOException e) {
139          LOG.warn("Exception closing " + f, e);
140        }
141      }
142      if (!finished) {
143        FileSystem fs = store.getFileSystem();
144        for (Path leftoverFile : writer.abortWriters()) {
145          try {
146            fs.delete(leftoverFile, false);
147          } catch (IOException e) {
148            LOG.error("Failed to delete the leftover file " + leftoverFile
149                + " after an unfinished compaction.",
150              e);
151          }
152        }
153      }
154    }
155    assert finished : "We should have exited the method on all error paths";
156    return commitMultiWriter(writer, fd, request);
157  }
158
159  protected abstract List<Path> commitMultiWriter(T writer, FileDetails fd,
160      CompactionRequest request) throws IOException;
161}
