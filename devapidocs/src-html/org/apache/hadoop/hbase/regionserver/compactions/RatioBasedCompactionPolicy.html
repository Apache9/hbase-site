042/**
043 * The default algorithm for selecting files for compaction.
044 * Combines the compaction configuration and the provisional file selection that
045 * it's given to produce the list of suitable candidates for compaction.
046 */
047@InterfaceAudience.Private
048public class RatioBasedCompactionPolicy extends CompactionPolicy {
049  private static final Log LOG = LogFactory.getLog(RatioBasedCompactionPolicy.class);
050
051  public RatioBasedCompactionPolicy(Configuration conf,
052                                    StoreConfigInformation storeConfigInfo) {
053    super(conf, storeConfigInfo);
054  }
055
056  private ArrayList<StoreFile> getCurrentEligibleFiles(
057      ArrayList<StoreFile> candidateFiles, final List<StoreFile> filesCompacting) {
058    // candidates = all storefiles not already in compaction queue
059    if (!filesCompacting.isEmpty()) {
060      // exclude all files older than the newest file we're currently
061      // compacting. this allows us to preserve contiguity (HBASE-2856)
062      StoreFile last = filesCompacting.get(filesCompacting.size() - 1);
063      int idx = candidateFiles.indexOf(last);
064      Preconditions.checkArgument(idx != -1);
065      candidateFiles.subList(0, idx + 1).clear();
066    }
067    return candidateFiles;
068  }
069
070  public List<StoreFile> preSelectCompactionForCoprocessor(
071      final Collection<StoreFile> candidates, final List<StoreFile> filesCompacting) {
072    return getCurrentEligibleFiles(new ArrayList<StoreFile>(candidates), filesCompacting);
073  }

075  /**
076   * @param candidateFiles candidate files, ordered from oldest to newest by seqId. We rely on
077   *   DefaultStoreFileManager to sort the files by seqId to guarantee contiguous compaction based
078   *   on seqId for data consistency.
079   * @return subset copy of candidate list that meets compaction criteria
080   * @throws java.io.IOException
081   */
082  public CompactionRequest selectCompaction(Collection<StoreFile> candidateFiles,
083      final List<StoreFile> filesCompacting, final boolean isUserCompaction,
084      final boolean mayUseOffPeak, final boolean forceMajor) throws IOException {
085    // Preliminary compaction subject to filters
086    ArrayList<StoreFile> candidateSelection = new ArrayList<StoreFile>(candidateFiles);
087    // Stuck and not compacting enough (estimate). It is not guaranteed that we will be
088    // able to compact more if stuck and compacting, because ratio policy excludes some
089    // non-compacting files from consideration during compaction (see getCurrentEligibleFiles).
090    int futureFiles = filesCompacting.isEmpty() ? 0 : 1;
091    boolean mayBeStuck = (candidateFiles.size() - filesCompacting.size() + futureFiles)
092        >= storeConfigInfo.getBlockingFileCount();
093    candidateSelection = getCurrentEligibleFiles(candidateSelection, filesCompacting);
094    LOG.debug("Selecting compaction from " + candidateFiles.size() + " store files, " +
095        filesCompacting.size() + " compacting, " + candidateSelection.size() +
096        " eligible, " + storeConfigInfo.getBlockingFileCount() + " blocking");
097
098    // If we can't have all files, we cannot do major anyway
099    boolean isAllFiles = candidateFiles.size() == candidateSelection.size();
100    if (!(forceMajor && isAllFiles)) {
101      candidateSelection = skipLargeFiles(candidateSelection, mayUseOffPeak);
102      isAllFiles = candidateFiles.size() == candidateSelection.size();
103    }
104
105    // Try a major compaction if this is a user-requested major compaction,
106    // or if we do not have too many files to compact and this was requested as a major compaction
107    boolean isTryingMajor = (forceMajor && isAllFiles && isUserCompaction)
108        || (((forceMajor && isAllFiles) || isMajorCompaction(candidateSelection))
109          && (candidateSelection.size() < comConf.getMaxFilesToCompact()));
110    // Or, if there are any references among the candidates.
111    boolean isAfterSplit = StoreUtils.hasReferences(candidateSelection);
112    if (!isTryingMajor && !isAfterSplit) {
113      // We're are not compacting all files, let's see what files are applicable
114      candidateSelection = filterBulk(candidateSelection);
115      candidateSelection = applyCompactionPolicy(candidateSelection, mayUseOffPeak, mayBeStuck);
116      candidateSelection = checkMinFilesCriteria(candidateSelection);
117    }
118    candidateSelection = removeExcessFiles(candidateSelection, isUserCompaction, isTryingMajor);
119    // Now we have the final file list, so we can determine if we can do major/all files.
120    isAllFiles = (candidateFiles.size() == candidateSelection.size());
121    CompactionRequest result = new CompactionRequest(candidateSelection);
122    result.setOffPeak(!candidateSelection.isEmpty() && !isAllFiles && mayUseOffPeak);
123    result.setIsMajor(isTryingMajor && isAllFiles, isAllFiles);
124    return result;
125  }

127  /**
128   * @param candidates pre-filtrate
129   * @return filtered subset
130   * exclude all files above maxCompactSize
131   * Also save all references. We MUST compact them
132   */
133  protected ArrayList<StoreFile> skipLargeFiles(ArrayList<StoreFile> candidates,
134    boolean mayUseOffpeak) {
135    int pos = 0;
136    while (pos < candidates.size() && !candidates.get(pos).isReference()
137      && (candidates.get(pos).getReader().length() > comConf.getMaxCompactSize(mayUseOffpeak))) {
138      ++pos;
139    }
140    if (pos > 0) {
141      LOG.debug("Some files are too large. Excluding " + pos
142          + " files from compaction candidates");
143      candidates.subList(0, pos).clear();
144    }
145    return candidates;
146  }
147
148  /**
149   * @param candidates pre-filtrate
150   * @return filtered subset
151   * exclude all bulk load files if configured
152   */
153  protected ArrayList<StoreFile> filterBulk(ArrayList<StoreFile> candidates) {
154    candidates.removeAll(Collections2.filter(candidates,
155        new Predicate<StoreFile>() {
156          @Override
157          public boolean apply(StoreFile input) {
158            return input.excludeFromMinorCompaction();
159          }
160        }));
161    return candidates;
162  }
163
164  /**
165   * @param candidates pre-filtrate
166   * @return filtered subset
167   * take upto maxFilesToCompact from the start
168   */
169  private ArrayList<StoreFile> removeExcessFiles(ArrayList<StoreFile> candidates,
170      boolean isUserCompaction, boolean isMajorCompaction) {
171    int excess = candidates.size() - comConf.getMaxFilesToCompact();
172    if (excess > 0) {
173      if (isMajorCompaction && isUserCompaction) {
174        LOG.debug("Warning, compacting more than " + comConf.getMaxFilesToCompact() +
175            " files because of a user-requested major compaction");
176      } else {
177        LOG.debug("Too many admissible files. Excluding " + excess
178          + " files from compaction candidates");
179        candidates.subList(comConf.getMaxFilesToCompact(), candidates.size()).clear();
180      }
181    }
182    return candidates;
183  }
184  /**
185   * @param candidates pre-filtrate
186   * @return filtered subset
187   * forget the compactionSelection if we don't have enough files
188   */
189  protected ArrayList<StoreFile> checkMinFilesCriteria(ArrayList<StoreFile> candidates) {
190    int minFiles = comConf.getMinFilesToCompact();
191    if (candidates.size() < minFiles) {
192      if(LOG.isDebugEnabled()) {
193        LOG.debug("Not compacting files because we only have " + candidates.size() +
194          " files ready for compaction. Need " + minFiles + " to initiate.");
195      }
196      candidates.clear();
197    }
198    return candidates;
199  }

201  /**
202    * @param candidates pre-filtrate
203    * @return filtered subset
204    * -- Default minor compaction selection algorithm:
205    * choose CompactSelection from candidates --
206    * First exclude bulk-load files if indicated in configuration.
207    * Start at the oldest file and stop when you find the first file that
208    * meets compaction criteria:
209    * (1) a recently-flushed, small file (i.e. <= minCompactSize)
210    * OR
211    * (2) within the compactRatio of sum(newer_files)
212    * Given normal skew, any newer files will also meet this criteria
213    * <p/>
214    * Additional Note:
215    * If fileSizes.size() >> maxFilesToCompact, we will recurse on
216    * compact().  Consider the oldest files first to avoid a
217    * situation where we always compact [end-threshold,end).  Then, the
218    * last file becomes an aggregate of the previous compactions.
219    *
220    * normal skew:
221    *
222    *         older ----> newer (increasing seqID)
223    *     _
224    *    | |   _
225    *    | |  | |   _
226    *  --|-|- |-|- |-|---_-------_-------  minCompactSize
227    *    | |  | |  | |  | |  _  | |
228    *    | |  | |  | |  | | | | | |
229    *    | |  | |  | |  | | | | | |
230    */
231  ArrayList<StoreFile> applyCompactionPolicy(ArrayList<StoreFile> candidates,
232      boolean mayUseOffPeak, boolean mayBeStuck) throws IOException {
233    if (candidates.isEmpty()) {
234      return candidates;
235    }
236
237    // we're doing a minor compaction, let's see what files are applicable
238    int start = 0;
239    double ratio = comConf.getCompactionRatio();
240    if (mayUseOffPeak) {
241      ratio = comConf.getCompactionRatioOffPeak();
242      LOG.info("Running an off-peak compaction, selection ratio = " + ratio);
243    }
244
245    // get store file sizes for incremental compacting selection.
246    final int countOfFiles = candidates.size();
247    long[] fileSizes = new long[countOfFiles];
248    long[] sumSize = new long[countOfFiles];
249    for (int i = countOfFiles - 1; i >= 0; --i) {
250      StoreFile file = candidates.get(i);
251      fileSizes[i] = file.getReader().length();
252      // calculate the sum of fileSizes[i,i+maxFilesToCompact-1) for algo
253      int tooFar = i + comConf.getMaxFilesToCompact() - 1;
254      sumSize[i] = fileSizes[i]
255        + ((i + 1 < countOfFiles) ? sumSize[i + 1] : 0)
256        - ((tooFar < countOfFiles) ? fileSizes[tooFar] : 0);
257    }
258
259
260    while (countOfFiles - start >= comConf.getMinFilesToCompact() &&
261      fileSizes[start] > Math.max(comConf.getMinCompactSize(),
262          (long) (sumSize[start + 1] * ratio))) {
263      ++start;
264    }
265    if (start < countOfFiles) {
266      LOG.info("Default compaction algorithm has selected " + (countOfFiles - start)
267        + " files from " + countOfFiles + " candidates");
268    } else if (mayBeStuck) {
269      // We may be stuck. Compact the latest files if we can.
270      int filesToLeave = candidates.size() - comConf.getMinFilesToCompact();
271      if (filesToLeave >= 0) {
272        start = filesToLeave;
273      }
274    }
275    candidates.subList(0, start).clear();
276    return candidates;
277  }

279  /*
280   * @param filesToCompact Files to compact. Can be null.
281   * @return True if we should run a major compaction.
282   */
283  @Override
284  public boolean isMajorCompaction(final Collection<StoreFile> filesToCompact)
285      throws IOException {
286    boolean result = false;
287    long mcTime = getNextMajorCompactTime(filesToCompact);
288    if (filesToCompact == null || filesToCompact.isEmpty() || mcTime == 0) {
289      return result;
290    }
291    // TODO: Use better method for determining stamp of last major (HBASE-2990)
292    long lowTimestamp = StoreUtils.getLowestTimestamp(filesToCompact);
293    long now = System.currentTimeMillis();
294    if (lowTimestamp > 0l && lowTimestamp < (now - mcTime)) {
295      // Major compaction time has elapsed.
296      long cfTtl = this.storeConfigInfo.getStoreFileTtl();
297      if (filesToCompact.size() == 1) {
298        // Single file
299        StoreFile sf = filesToCompact.iterator().next();
300        Long minTimestamp = sf.getMinimumTimestamp();
301        long oldest = (minTimestamp == null)
302            ? Long.MIN_VALUE
303            : now - minTimestamp.longValue();
304        if (sf.isMajorCompaction() &&
305            (cfTtl == HConstants.FOREVER || oldest < cfTtl)) {
306          float blockLocalityIndex = sf.getHDFSBlockDistribution().getBlockLocalityIndex(
307              RSRpcServices.getHostname(comConf.conf, false)
308          );
309          if (blockLocalityIndex < comConf.getMinLocalityToForceCompact()) {
310            if (LOG.isDebugEnabled()) {
311              LOG.debug("Major compaction triggered on only store " + this +
312                  "; to make hdfs blocks local, current blockLocalityIndex is " +
313                  blockLocalityIndex + " (min " + comConf.getMinLocalityToForceCompact() +
314                  ")");
315            }
316            result = true;
317          } else {
318            if (LOG.isDebugEnabled()) {
319              LOG.debug("Skipping major compaction of " + this +
320                  " because one (major) compacted file only, oldestTime " +
321                  oldest + "ms is < ttl=" + cfTtl + " and blockLocalityIndex is " +
322                  blockLocalityIndex + " (min " + comConf.getMinLocalityToForceCompact() +
323                  ")");
324            }
325          }
326        } else if (cfTtl != HConstants.FOREVER && oldest > cfTtl) {
327          LOG.debug("Major compaction triggered on store " + this +
328            ", because keyvalues outdated; time since last major compaction " +
329            (now - lowTimestamp) + "ms");
330          result = true;
331        }
332      } else {
333        if (LOG.isDebugEnabled()) {
334          LOG.debug("Major compaction triggered on store " + this +
335              "; time since last major compaction " + (now - lowTimestamp) + "ms");
336        }
337        result = true;
338      }
339    }
340    return result;
341  }

343  /**
344   * Used calculation jitter
345   */
346  private final Random random = new Random();
347
348  /**
349   * @param filesToCompact
350   * @return When to run next major compaction
351   */
352  public long getNextMajorCompactTime(final Collection<StoreFile> filesToCompact) {
353    // default = 24hrs
354    long ret = comConf.getMajorCompactionPeriod();
355    if (ret > 0) {
356      // default = 20% = +/- 4.8 hrs
357      double jitterPct = comConf.getMajorCompactionJitter();
358      if (jitterPct > 0) {
359        long jitter = Math.round(ret * jitterPct);
360        // deterministic jitter avoids a major compaction storm on restart
361        Integer seed = StoreUtils.getDeterministicRandomSeed(filesToCompact);
362        if (seed != null) {
363          // Synchronized to ensure one user of random instance at a time.
364          double rnd = -1;
365          synchronized (this) {
366            this.random.setSeed(seed);
367            rnd = this.random.nextDouble();
368          }
369          ret += jitter - Math.round(2L * jitter * rnd);
370        } else {
371          ret = 0; // If seed is null, then no storefiles == no major compaction
372        }
373      }
374    }
375    return ret;
376  }

378  /**
379   * @param compactionSize Total size of some compaction
380   * @return whether this should be a large or small compaction
381   */
382  @Override
383  public boolean throttleCompaction(long compactionSize) {
384    return compactionSize > comConf.getThrottlePoint();
385  }
386
387  public boolean needsCompaction(final Collection<StoreFile> storeFiles,
388      final List<StoreFile> filesCompacting) {
389    int numCandidates = storeFiles.size() - filesCompacting.size();
390    return numCandidates >= comConf.getMinFilesToCompact();
391  }
