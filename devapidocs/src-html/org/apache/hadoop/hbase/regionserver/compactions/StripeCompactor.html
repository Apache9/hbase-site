044/**
045 * This is the placeholder for stripe compactor. The implementation,
046 * as well as the proper javadoc, will be added in HBASE-7967.
047 */
048@InterfaceAudience.Private
049public class StripeCompactor extends Compactor {
050  private static final Log LOG = LogFactory.getLog(StripeCompactor.class);
051  public StripeCompactor(Configuration conf, Store store) {
052    super(conf, store);
053  }
054
055  public List<Path> compact(CompactionRequest request, List<byte[]> targetBoundaries,
056    byte[] majorRangeFromRow, byte[] majorRangeToRow,
057    ThroughputController throughputController) throws IOException {
058    return compact(request, targetBoundaries, majorRangeFromRow, majorRangeToRow,
059      throughputController, null);
060  }
061
062  public List<Path> compact(CompactionRequest request, List<byte[]> targetBoundaries,
063      byte[] majorRangeFromRow, byte[] majorRangeToRow,
064      ThroughputController throughputController, User user) throws IOException {
065    if (LOG.isDebugEnabled()) {
066      StringBuilder sb = new StringBuilder();
067      sb.append("Executing compaction with " + targetBoundaries.size() + " boundaries:");
068      for (byte[] tb : targetBoundaries) {
069        sb.append(" [").append(Bytes.toString(tb)).append("]");
070      }
071      LOG.debug(sb.toString());
072    }
073    StripeMultiFileWriter writer = new StripeMultiFileWriter.BoundaryMultiWriter(
074        targetBoundaries, majorRangeFromRow, majorRangeToRow);
075    return compactInternal(writer, request, majorRangeFromRow, majorRangeToRow,
076      throughputController, user);
077  }
078
079  public List<Path> compact(CompactionRequest request, int targetCount, long targetSize,
080    byte[] left, byte[] right, byte[] majorRangeFromRow, byte[] majorRangeToRow,
081    ThroughputController throughputController) throws IOException {
082    return compact(request, targetCount, targetSize, left, right, majorRangeFromRow,
083      majorRangeToRow, throughputController, null);
084  }
085
086  public List<Path> compact(CompactionRequest request, int targetCount, long targetSize,
087      byte[] left, byte[] right, byte[] majorRangeFromRow, byte[] majorRangeToRow,
088      ThroughputController throughputController, User user) throws IOException {
089    if (LOG.isDebugEnabled()) {
090      LOG.debug("Executing compaction with " + targetSize
091          + " target file size, no more than " + targetCount + " files, in ["
092          + Bytes.toString(left) + "] [" + Bytes.toString(right) + "] range");
093    }
094    StripeMultiFileWriter writer = new StripeMultiFileWriter.SizeMultiWriter(
095        targetCount, targetSize, left, right);
096    return compactInternal(writer, request, majorRangeFromRow, majorRangeToRow,
097      throughputController, user);
098  }
099
100  private List<Path> compactInternal(StripeMultiFileWriter mw, final CompactionRequest request,
101      byte[] majorRangeFromRow, byte[] majorRangeToRow,
102      ThroughputController throughputController, User user) throws IOException {
103    final Collection<StoreFile> filesToCompact = request.getFiles();
104    final FileDetails fd = getFileDetails(filesToCompact, request.isMajor());
105    this.progress = new CompactionProgress(fd.maxKeyCount);
106
107    long smallestReadPoint = getSmallestReadPoint();
108    List<StoreFileScanner> scanners = createFileScanners(filesToCompact,
109        smallestReadPoint, store.throttleCompaction(request.getSize()));
110
111    boolean finished = false;
112    InternalScanner scanner = null;
113    boolean cleanSeqId = false;
114    try {
115      // Get scanner to use.
116      ScanType coprocScanType = ScanType.COMPACT_RETAIN_DELETES;
117      scanner = preCreateCoprocScanner(request, coprocScanType, fd.earliestPutTs, scanners, user);
118      if (scanner == null) {
119        scanner = (majorRangeFromRow == null)
120            ? createScanner(store, scanners,
121                ScanType.COMPACT_RETAIN_DELETES, smallestReadPoint, fd.earliestPutTs)
122            : createScanner(store, scanners,
123                smallestReadPoint, fd.earliestPutTs, majorRangeFromRow, majorRangeToRow);
124      }
125      scanner = postCreateCoprocScanner(request, coprocScanType, scanner, user);
126      if (scanner == null) {
127        // NULL scanner returned from coprocessor hooks means skip normal processing.
128        return new ArrayList<Path>();
129      }
130
131      // Create the writer factory for compactions.
132      if(fd.minSeqIdToKeep > 0) {
133        smallestReadPoint = Math.min(fd.minSeqIdToKeep, smallestReadPoint);
134        cleanSeqId = true;
135      }
136
137      final boolean needMvcc = fd.maxMVCCReadpoint > 0;
138
139      final Compression.Algorithm compression = store.getFamily().getCompactionCompressionType();
140      StripeMultiFileWriter.WriterFactory factory = new StripeMultiFileWriter.WriterFactory() {
141        @Override
142        public Writer createWriter() throws IOException {
143          return store.createWriterInTmp(
144              fd.maxKeyCount, compression, true, needMvcc, fd.maxTagsLength > 0,
145              store.throttleCompaction(request.getSize()));
146        }
147      };
148
149      // Prepare multi-writer, and perform the compaction using scanner and writer.
150      // It is ok here if storeScanner is null.
151      StoreScanner storeScanner = (scanner instanceof StoreScanner) ? (StoreScanner)scanner : null;
152      mw.init(storeScanner, factory, store.getComparator());
153      finished =
154          performCompaction(fd, scanner, mw, smallestReadPoint, cleanSeqId, throughputController,
155                  request.isMajor());
156      if (!finished) {
157        throw new InterruptedIOException( "Aborting compaction of store " + store +
158            " in region " + store.getRegionInfo().getRegionNameAsString() +
159            " because it was interrupted.");
160      }
161    } finally {
162      if (scanner != null) {
163        try {
164          scanner.close();
165        } catch (Throwable t) {
166          // Don't fail the compaction if this fails.
167          LOG.error("Failed to close scanner after compaction.", t);
168        }
169      }
170      if (!finished) {
171        for (Path leftoverFile : mw.abortWriters()) {
172          try {
173            store.getFileSystem().delete(leftoverFile, false);
174          } catch (Exception ex) {
175            LOG.error("Failed to delete the leftover file after an unfinished compaction.", ex);
176          }
177        }
178      }
179    }
180
181    assert finished : "We should have exited the method on all error paths";
182    List<Path> newFiles = mw.commitWriters(fd.maxSeqId, request.isMajor());
183    assert !newFiles.isEmpty() : "Should have produced an empty file to preserve metadata.";
184    return newFiles;
185  }
