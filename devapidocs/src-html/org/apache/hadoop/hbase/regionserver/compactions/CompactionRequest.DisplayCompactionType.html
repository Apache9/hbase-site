019package org.apache.hadoop.hbase.regionserver.compactions;
020
021import java.util.Collection;
022import java.util.stream.Collectors;
023
024import org.apache.hadoop.hbase.regionserver.HStoreFile;
025import org.apache.hadoop.hbase.regionserver.Store;
026import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
027import org.apache.hadoop.util.StringUtils.TraditionalBinaryPrefix;
028import org.apache.yetus.audience.InterfaceAudience;
029
030import org.apache.hadoop.hbase.shaded.com.google.common.base.Preconditions;
031
032/**
033 * This class holds all logical details necessary to run a compaction.
034 */
035@InterfaceAudience.Private
036public class CompactionRequest {
037
038  // was this compaction promoted to an off-peak
039  private boolean isOffPeak = false;
040  private enum DisplayCompactionType { MINOR, ALL_FILES, MAJOR }
041  private DisplayCompactionType isMajor = DisplayCompactionType.MINOR;
042  private int priority = Store.NO_PRIORITY;
043  private Collection<HStoreFile> filesToCompact;
044
045  // CompactRequest object creation time.
046  private long selectionTime;
047  // System time used to compare objects in FIFO order. TODO: maybe use selectionTime?
048  private long timeInNanos;
049  private String regionName = "";
050  private String storeName = "";
051  private long totalSize = -1L;
052  private CompactionLifeCycleTracker tracker = CompactionLifeCycleTracker.DUMMY;
053
054  public CompactionRequest(Collection<HStoreFile> files) {
055    this.selectionTime = EnvironmentEdgeManager.currentTime();
056    this.timeInNanos = System.nanoTime();
057    this.filesToCompact = Preconditions.checkNotNull(files, "files for compaction can not null");
058    recalculateSize();
059  }
060
061  public void updateFiles(Collection<HStoreFile> files) {
062    this.filesToCompact = Preconditions.checkNotNull(files, "files for compaction can not null");
063    recalculateSize();
064  }
065
066  public Collection<HStoreFile> getFiles() {
067    return this.filesToCompact;
068  }
069
070  /**
071   * Sets the region/store name, for logging.
072   */
073  public void setDescription(String regionName, String storeName) {
074    this.regionName = regionName;
075    this.storeName = storeName;
076  }
077
078  /** Gets the total size of all StoreFiles in compaction */
079  public long getSize() {
080    return totalSize;
081  }
082
083  public boolean isAllFiles() {
084    return this.isMajor == DisplayCompactionType.MAJOR
085        || this.isMajor == DisplayCompactionType.ALL_FILES;
086  }
087
088  public boolean isMajor() {
089    return this.isMajor == DisplayCompactionType.MAJOR;
090  }
091
092  /** Gets the priority for the request */
093  public int getPriority() {
094    return priority;
095  }
096
097  /** Sets the priority for the request */
098  public void setPriority(int p) {
099    this.priority = p;
100  }
101
102  public boolean isOffPeak() {
103    return this.isOffPeak;
104  }
105
106  public void setOffPeak(boolean value) {
107    this.isOffPeak = value;
108  }
109
110  public long getSelectionTime() {
111    return this.selectionTime;
112  }
113
114  public long getSelectionNanoTime() {
115    return this.timeInNanos;
116  }
117
118  /**
119   * Specify if this compaction should be a major compaction based on the state of the store
120   * @param isMajor <tt>true</tt> if the system determines that this compaction should be a major
121   *          compaction
122   */
123  public void setIsMajor(boolean isMajor, boolean isAllFiles) {
124    assert isAllFiles || !isMajor;
125    this.isMajor = !isAllFiles ? DisplayCompactionType.MINOR
126        : (isMajor ? DisplayCompactionType.MAJOR : DisplayCompactionType.ALL_FILES);
127  }
128
129  public void setTracker(CompactionLifeCycleTracker tracker) {
130    this.tracker = tracker;
131  }
132
133  public CompactionLifeCycleTracker getTracker() {
134    return tracker;
135  }
136
137  @Override
138  public String toString() {
139    String fsList = filesToCompact.stream().filter(f -> f.getReader() != null)
140        .map(f -> TraditionalBinaryPrefix.long2String(f.getReader().length(), "", 1))
141        .collect(Collectors.joining(", "));
142
143    return "regionName=" + regionName + ", storeName=" + storeName + ", fileCount=" +
144        this.getFiles().size() + ", fileSize=" +
145        TraditionalBinaryPrefix.long2String(totalSize, "", 1) +
146        ((fsList.isEmpty()) ? "" : " (" + fsList + ")") + ", priority=" + priority + ", time=" +
147        timeInNanos;
148  }
149
150  /**
151   * Recalculate the size of the compaction based on current files.
152   * @param files files that should be included in the compaction
153   */
154  private void recalculateSize() {
155    this.totalSize = filesToCompact.stream().map(HStoreFile::getReader)
156        .mapToLong(r -> r != null ? r.length() : 0L).sum();
157  }
158}
