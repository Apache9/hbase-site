045/**
046 * HBASE-15181 This is a simple implementation of date-based tiered compaction similar to
047 * Cassandra's for the following benefits:
048 * 1. Improve date-range-based scan by structuring store files in date-based tiered layout.
049 * 2. Reduce compaction overhead.
050 * 3. Improve TTL efficiency.
051 * Perfect fit for the use cases that:
052 * 1. has mostly date-based data write and scan and a focus on the most recent data.
053 * 2. never or rarely deletes data. Out-of-order writes are handled gracefully. Time range
054 * overlapping among store files is tolerated and the performance impact is minimized. Configuration
055 * can be set at hbase-site or overriden at per-table or per-column-famly level by hbase shell.
056 * Design spec is at
057 * https://docs.google.com/document/d/1_AmlNb2N8Us1xICsTeGDLKIqL6T-oHoRLZ323MG_uy8/
058 */
059@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
060public class DateTieredCompactionPolicy extends RatioBasedCompactionPolicy {
061  private static final Log LOG = LogFactory.getLog(DateTieredCompactionPolicy.class);
062
063  private RatioBasedCompactionPolicy compactionPolicyPerWindow;
064
065  public DateTieredCompactionPolicy(Configuration conf, StoreConfigInformation storeConfigInfo)
066      throws IOException {
067    super(conf, storeConfigInfo);
068    try {
069      compactionPolicyPerWindow =
070          ReflectionUtils.instantiateWithCustomCtor(comConf.getCompactionPolicyForTieredWindow(),
071            new Class[] { Configuration.class, StoreConfigInformation.class }, new Object[] { conf,
072              storeConfigInfo });
073    } catch (Exception e) {
074      throw new IOException("Unable to load configured compaction policy '"
075          + comConf.getCompactionPolicyForTieredWindow() + "'", e);
076    }
077  }
078
079  @Override
080  public boolean isMajorCompaction(Collection<StoreFile> filesToCompact) throws IOException {
081    // Never do major compaction unless forced
082    return false;
083  }
084
085  @Override
086  /**
087   * Heuristics for guessing whether we need compaction.
088   */
089  public boolean needsCompaction(final Collection<StoreFile> storeFiles,
090      final List<StoreFile> filesCompacting) {
091    return needsCompaction(storeFiles, filesCompacting, EnvironmentEdgeManager.currentTime());
092  }
093
094  @VisibleForTesting
095  public boolean needsCompaction(final Collection<StoreFile> storeFiles,
096      final List<StoreFile> filesCompacting, long now) {
097    if (!super.needsCompaction(storeFiles, filesCompacting)) {
098      return false;
099    }
100
101    ArrayList<StoreFile> candidates = new ArrayList<StoreFile>(storeFiles);
102    candidates = filterBulk(candidates);
103    candidates = skipLargeFiles(candidates, true);
104    try {
105      candidates = applyCompactionPolicy(candidates, true, false, now);
106    } catch (Exception e) {
107      LOG.error("Can not check for compaction: ", e);
108      return false;
109    }
110
111    return candidates != null && candidates.size() >= comConf.getMinFilesToCompact();
112  }
113
114  /**
115   * Could return null if no candidates are found
116   */
117  @Override
118  public ArrayList<StoreFile> applyCompactionPolicy(ArrayList<StoreFile> candidates,
119      boolean mayUseOffPeak, boolean mayBeStuck) throws IOException {
120    return applyCompactionPolicy(candidates, mayUseOffPeak, mayBeStuck,
121      EnvironmentEdgeManager.currentTime());
122  }
123
124  /**
125   * Input candidates are sorted from oldest to newest by seqId. Could return null if no candidates
126   * are found.
127   */
128  @VisibleForTesting
129  public ArrayList<StoreFile> applyCompactionPolicy(ArrayList<StoreFile> candidates,
130      boolean mayUseOffPeak, boolean mayBeStuck, long now) throws IOException {
131    Iterable<StoreFile> candidatesInWindow =
132      filterOldStoreFiles(Lists.newArrayList(candidates), comConf.getMaxStoreFileAgeMillis(), now);
133
134    List<ArrayList<StoreFile>> buckets =
135        partitionFilesToBuckets(candidatesInWindow, comConf.getBaseWindowMillis(),
136          comConf.getWindowsPerTier(), now);
137    LOG.debug("Compaction buckets are: " + buckets);
138
139    return newestBucket(buckets, comConf.getIncomingWindowMin(), now,
140      comConf.getBaseWindowMillis(), mayUseOffPeak);
141  }
142
143  /**
144   * @param buckets the list of buckets, sorted from newest to oldest, from which to return the
145   *          newest bucket within thresholds.
146   * @param incomingWindowThreshold minimum number of storeFiles in a bucket to qualify.
147   * @param maxThreshold maximum number of storeFiles to compact at once (the returned bucket will
148   *          be trimmed down to this).
149   * @return a bucket (a list of store files within a window to be compacted).
150   * @throws IOException error
151   */
152  private ArrayList<StoreFile> newestBucket(List<ArrayList<StoreFile>> buckets,
153      int incomingWindowThreshold, long now, long baseWindowMillis, boolean mayUseOffPeak)
154      throws IOException {
155    Window incomingWindow = getInitialWindow(now, baseWindowMillis);
156    for (ArrayList<StoreFile> bucket : buckets) {
157      int minThreshold =
158          incomingWindow.compareToTimestamp(bucket.get(0).getMaximumTimestamp()) <= 0 ? comConf
159              .getIncomingWindowMin() : comConf.getMinFilesToCompact();
160      compactionPolicyPerWindow.setMinThreshold(minThreshold);
161      ArrayList<StoreFile> candidates =
162          compactionPolicyPerWindow.applyCompactionPolicy(bucket, mayUseOffPeak, false);
163      if (candidates != null && !candidates.isEmpty()) {
164        return candidates;
165      }
166    }
167    return null;
168  }
169
170  /**
171   * We receive store files sorted in ascending order by seqId then scan the list of files. If the
172   * current file has a maxTimestamp older than last known maximum, treat this file as it carries
173   * the last known maximum. This way both seqId and timestamp are in the same order. If files carry
174   * the same maxTimestamps, they are ordered by seqId. We then reverse the list so they are ordered
175   * by seqId and maxTimestamp in decending order and build the time windows. All the out-of-order
176   * data into the same compaction windows, guaranteeing contiguous compaction based on sequence id.
177   */
178  private static List<ArrayList<StoreFile>> partitionFilesToBuckets(Iterable<StoreFile> storeFiles,
179      long baseWindowSizeMillis, int windowsPerTier, long now) {
180    List<ArrayList<StoreFile>> buckets = Lists.newArrayList();
181    Window window = getInitialWindow(now, baseWindowSizeMillis);
182
183    List<Pair<StoreFile, Long>> storefileMaxTimestampPairs =
184        Lists.newArrayListWithCapacity(Iterables.size(storeFiles));
185    long maxTimestampSeen = Long.MIN_VALUE;
186    for (StoreFile storeFile : storeFiles) {
187      // if there is out-of-order data,
188      // we put them in the same window as the last file in increasing order
189      maxTimestampSeen = Math.max(maxTimestampSeen, storeFile.getMaximumTimestamp());
190      storefileMaxTimestampPairs.add(new Pair<StoreFile, Long>(storeFile, maxTimestampSeen));
191    }
192
193    Collections.reverse(storefileMaxTimestampPairs);
194    PeekingIterator<Pair<StoreFile, Long>> it =
195        Iterators.peekingIterator(storefileMaxTimestampPairs.iterator());
196
197    while (it.hasNext()) {
198      int compResult = window.compareToTimestamp(it.peek().getSecond());
199      if (compResult > 0) {
200        // If the file is too old for the window, switch to the next window
201        window = window.nextWindow(windowsPerTier);
202      } else {
203        // The file is within the target window
204        ArrayList<StoreFile> bucket = Lists.newArrayList();
205        // Add all files in the same window to current bucket. For incoming window
206        // we tolerate files with future data although it is sub-optimal
207        while (it.hasNext() && window.compareToTimestamp(it.peek().getSecond()) <= 0) {
208          bucket.add(it.next().getFirst());
209        }
210        if (!bucket.isEmpty()) {
211          buckets.add(bucket);
212        }
213      }
214    }
215
216    return buckets;
217  }
218
219  /**
220   * Removes all store files with max timestamp older than (current - maxAge).
221   * @param storeFiles all store files to consider
222   * @param maxAge the age in milliseconds when a store file stops participating in compaction.
223   * @param now current time. store files with max timestamp less than (now - maxAge) are filtered.
224   * @return a list of storeFiles with the store file older than maxAge excluded
225   */
226  private static Iterable<StoreFile> filterOldStoreFiles(List<StoreFile> storeFiles, long maxAge,
227      long now) {
228    if (maxAge == 0) {
229      return ImmutableList.of();
230    }
231    final long cutoff = now - maxAge;
232    return Iterables.filter(storeFiles, new Predicate<StoreFile>() {
233      @Override
234      public boolean apply(StoreFile storeFile) {
235        // Known findbugs issue to guava. SuppressWarning or Nonnull annotation don't work.
236        if (storeFile == null) {
237          throw new NullPointerException();
238        }
239        return storeFile.getMaximumTimestamp() >= cutoff;
240      }
241    });
242  }
243
244  private static Window getInitialWindow(long now, long timeUnit) {
245    return new Window(timeUnit, now / timeUnit);
246  }
247
248  /**
249   * This is the class we use to partition from epoch time to now into tiers of exponential sizes of
250   * windows.
251   */
252  private static final class Window {
253    /**
254     * How big a range of timestamps fit inside the window in milliseconds.
255     */
256    private final long windowMillis;
257
258    /**
259     * A timestamp t is within the window iff t / size == divPosition.
260     */
261    private final long divPosition;
262
263    private Window(long baseWindowMillis, long divPosition) {
264      this.windowMillis = baseWindowMillis;
265      this.divPosition = divPosition;
266    }
267
268    /**
269     * Compares the window to a timestamp.
270     * @param timestamp the timestamp to compare.
271     * @return a negative integer, zero, or a positive integer as the window lies before, covering,
272     *         or after than the timestamp.
273     */
274    public int compareToTimestamp(long timestamp) {
275      long pos = timestamp / windowMillis;
276      return divPosition == pos ? 0 : divPosition < pos ? -1 : 1;
277    }
278
279    /**
280     * Move to the new window of the same tier or of the next tier, which represents an earlier time
281     * span.
282     * @param windowsPerTier The number of contiguous windows that will have the same size. Windows
283     *          following those will be <code>tierBase</code> times as big.
284     * @return The next window
285     */
286    public Window nextWindow(int windowsPerTier) {
287      if (divPosition % windowsPerTier > 0) {
288        return new Window(windowMillis, divPosition - 1);
289      } else {
290        return new Window(windowMillis * windowsPerTier, divPosition / windowsPerTier - 1);
291      }
292    }
293  }
