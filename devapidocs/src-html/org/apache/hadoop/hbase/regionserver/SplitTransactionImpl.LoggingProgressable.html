133  public SplitTransactionImpl(final Region r, final byte [] splitrow) {

139  private void transition(SplitTransactionPhase nextPhase) throws IOException {
140    transition(nextPhase, false);
141  }
142
143  private void transition(SplitTransactionPhase nextPhase, boolean isRollback)
144      throws IOException {
145    if (!isRollback) {
146      // Add to the journal first, because if the listener throws an exception
147      // we need to roll back starting at 'nextPhase'
148      this.journal.add(new JournalEntryImpl(nextPhase));
149    }
150    for (int i = 0; i < listeners.size(); i++) {
151      TransactionListener listener = listeners.get(i);
152      if (!isRollback) {
153        listener.transition(this, currentPhase, nextPhase);
154      } else {
155        listener.rollback(this, currentPhase, nextPhase);
156      }
157    }
158    currentPhase = nextPhase;
159  }

161  @Override
162  public boolean prepare() throws IOException {
163    if (!this.parent.isSplittable()) return false;
164    // Split key can be null if this region is unsplittable; i.e. has refs.
165    if (this.splitrow == null) return false;
166    HRegionInfo hri = this.parent.getRegionInfo();
167    parent.prepareToSplit();
168    // Check splitrow.
169    byte [] startKey = hri.getStartKey();
170    byte [] endKey = hri.getEndKey();
171    if (Bytes.equals(startKey, splitrow) ||
172        !this.parent.getRegionInfo().containsRow(splitrow)) {
173      LOG.info("Split row is not inside region key range or is equal to " +
174          "startkey: " + Bytes.toStringBinary(this.splitrow));
175      return false;
176    }
177    long rid = getDaughterRegionIdTimestamp(hri);
178    this.hri_a = new HRegionInfo(hri.getTable(), startKey, this.splitrow, false, rid);
179    this.hri_b = new HRegionInfo(hri.getTable(), this.splitrow, endKey, false, rid);
180
181    transition(SplitTransactionPhase.PREPARED);
182
183    return true;
184  }

186  /**
187   * Calculate daughter regionid to use.
188   * @param hri Parent {@link HRegionInfo}
189   * @return Daughter region id (timestamp) to use.
190   */
191  private static long getDaughterRegionIdTimestamp(final HRegionInfo hri) {
192    long rid = EnvironmentEdgeManager.currentTime();
193    // Regionid is timestamp.  Can't be less than that of parent else will insert
194    // at wrong location in hbase:meta (See HBASE-710).
195    if (rid < hri.getRegionId()) {
196      LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
197        " but current time here is " + rid);
198      rid = hri.getRegionId() + 1;
199    }
200    return rid;
201  }

206  /**
207   * Prepare the regions and region files.
208   * @param server Hosting server instance.  Can be null when testing (won't try
209   * and update in zk if a null server)
210   * @param services Used to online/offline regions.
211   * @param user
212   * @throws IOException If thrown, transaction failed.
213   *    Call {@link #rollback(Server, RegionServerServices)}
214   * @return Regions created
215   */
216  @VisibleForTesting
217  PairOfSameType<Region> createDaughters(final Server server,
218      final RegionServerServices services, User user) throws IOException {
219    LOG.info("Starting split of region " + this.parent);
220    if ((server != null && server.isStopped()) ||
221        (services != null && services.isStopping())) {
222      throw new IOException("Server is stopped or stopping");
223    }
224    assert !this.parent.lock.writeLock().isHeldByCurrentThread():
225      "Unsafe to hold write lock while performing RPCs";
226
227    transition(SplitTransactionPhase.BEFORE_PRE_SPLIT_HOOK);
228
229    // Coprocessor callback
230    if (this.parent.getCoprocessorHost() != null) {
231      // TODO: Remove one of these
232      parent.getCoprocessorHost().preSplit(user);
233      parent.getCoprocessorHost().preSplit(splitrow, user);
234    }
235
236    transition(SplitTransactionPhase.AFTER_PRE_SPLIT_HOOK);
237
238    // If true, no cluster to write meta edits to or to update znodes in.
239    boolean testing = server == null? true:
240        server.getConfiguration().getBoolean("hbase.testing.nocluster", false);
241    this.fileSplitTimeout = testing ? this.fileSplitTimeout :
242        server.getConfiguration().getLong("hbase.regionserver.fileSplitTimeout",
243          this.fileSplitTimeout);
244
245    PairOfSameType<Region> daughterRegions = stepsBeforePONR(server, services, testing);
246
247    final List<Mutation> metaEntries = new ArrayList<Mutation>();
248    boolean ret = false;
249    if (this.parent.getCoprocessorHost() != null) {
250      ret = parent.getCoprocessorHost().preSplitBeforePONR(splitrow, metaEntries, user);
251      if (ret) {
252          throw new IOException("Coprocessor bypassing region "
253            + parent.getRegionInfo().getRegionNameAsString() + " split.");
254      }
255      try {
256        for (Mutation p : metaEntries) {
257          HRegionInfo.parseRegionName(p.getRow());
258        }
259      } catch (IOException e) {
260        LOG.error("Row key of mutation from coprossor is not parsable as region name."
261            + "Mutations from coprocessor should only for hbase:meta table.");
262        throw e;
263      }
264    }
265
266    // This is the point of no return.  Adding subsequent edits to .META. as we
267    // do below when we do the daughter opens adding each to .META. can fail in
268    // various interesting ways the most interesting of which is a timeout
269    // BUT the edits all go through (See HBASE-3872).  IF we reach the PONR
270    // then subsequent failures need to crash out this regionserver; the
271    // server shutdown processing should be able to fix-up the incomplete split.
272    // The offlined parent will have the daughters as extra columns.  If
273    // we leave the daughter regions in place and do not remove them when we
274    // crash out, then they will have their references to the parent in place
275    // still and the server shutdown fixup of .META. will point to these
276    // regions.
277    // We should add PONR JournalEntry before offlineParentInMeta,so even if
278    // OfflineParentInMeta timeout,this will cause regionserver exit,and then
279    // master ServerShutdownHandler will fix daughter & avoid data loss. (See
280    // HBase-4562).
281
282    transition(SplitTransactionPhase.PONR);
283
284    // Edit parent in meta.  Offlines parent region and adds splita and splitb
285    // as an atomic update. See HBASE-7721. This update to META makes the region
286    // will determine whether the region is split or not in case of failures.
287    // If it is successful, master will roll-forward, if not, master will rollback
288    // and assign the parent region.
289    if (services != null && !services.reportRegionStateTransition(TransitionCode.SPLIT_PONR,
290        parent.getRegionInfo(), hri_a, hri_b)) {
291      // Passed PONR, let SSH clean it up
292      throw new IOException("Failed to notify master that split passed PONR: "
293        + parent.getRegionInfo().getRegionNameAsString());
294    }
295    return daughterRegions;
296  }

308  @VisibleForTesting
309  public PairOfSameType<Region> stepsBeforePONR(final Server server,
310      final RegionServerServices services, boolean testing) throws IOException {
311    if (services != null && !services.reportRegionStateTransition(TransitionCode.READY_TO_SPLIT,
312        parent.getRegionInfo(), hri_a, hri_b)) {
313      throw new IOException("Failed to get ok from master to split "
314        + parent.getRegionInfo().getRegionNameAsString());
315    }
316
317    transition(SplitTransactionPhase.SET_SPLITTING);
318
319    this.parent.getRegionFileSystem().createSplitsDir();
320
321    transition(SplitTransactionPhase.CREATE_SPLIT_DIR);
322
323    Map<byte[], List<StoreFile>> hstoreFilesToSplit = null;
324    Exception exceptionToThrow = null;
325    try{
326      hstoreFilesToSplit = this.parent.close(false);
327    } catch (Exception e) {
328      exceptionToThrow = e;
329    }
330    if (exceptionToThrow == null && hstoreFilesToSplit == null) {
331      // The region was closed by a concurrent thread.  We can't continue
332      // with the split, instead we must just abandon the split.  If we
333      // reopen or split this could cause problems because the region has
334      // probably already been moved to a different server, or is in the
335      // process of moving to a different server.
336      exceptionToThrow = closedByOtherException;
337    }
338    if (exceptionToThrow != closedByOtherException) {
339      transition(SplitTransactionPhase.CLOSED_PARENT_REGION);
340    }
341    if (exceptionToThrow != null) {
342      if (exceptionToThrow instanceof IOException) throw (IOException)exceptionToThrow;
343      throw new IOException(exceptionToThrow);
344    }
345    if (!testing) {
346      services.removeFromOnlineRegions(this.parent, null);
347    }
348
349    transition(SplitTransactionPhase.OFFLINED_PARENT);
350
351    // TODO: If splitStoreFiles were multithreaded would we complete steps in
352    // less elapsed time?  St.Ack 20100920
353    //
354    // splitStoreFiles creates daughter region dirs under the parent splits dir
355    // Nothing to unroll here if failure -- clean up of CREATE_SPLIT_DIR will
356    // clean this up.
357    Pair<Integer, Integer> expectedReferences = splitStoreFiles(hstoreFilesToSplit);
358
359    // Log to the journal that we are creating region A, the first daughter
360    // region.  We could fail halfway through.  If we do, we could have left
361    // stuff in fs that needs cleanup -- a storefile or two.  Thats why we
362    // add entry to journal BEFORE rather than AFTER the change.
363
364    transition(SplitTransactionPhase.STARTED_REGION_A_CREATION);
365
366    assertReferenceFileCount(expectedReferences.getFirst(),
367        this.parent.getRegionFileSystem().getSplitsDir(this.hri_a));
368    HRegion a = this.parent.createDaughterRegionFromSplits(this.hri_a);
369    assertReferenceFileCount(expectedReferences.getFirst(),
370        new Path(this.parent.getRegionFileSystem().getTableDir(), this.hri_a.getEncodedName()));
371
372    // Ditto
373
374    transition(SplitTransactionPhase.STARTED_REGION_B_CREATION);
375
376    assertReferenceFileCount(expectedReferences.getSecond(),
377        this.parent.getRegionFileSystem().getSplitsDir(this.hri_b));
378    HRegion b = this.parent.createDaughterRegionFromSplits(this.hri_b);
379    assertReferenceFileCount(expectedReferences.getSecond(),
380        new Path(this.parent.getRegionFileSystem().getTableDir(), this.hri_b.getEncodedName()));
381
382    return new PairOfSameType<Region>(a, b);
383  }

395  /**
396   * Perform time consuming opening of the daughter regions.
397   * @param server Hosting server instance.  Can be null when testing
398   * @param services Used to online/offline regions.
399   * @param a first daughter region
400   * @param a second daughter region
401   * @throws IOException If thrown, transaction failed.
402   *          Call {@link #rollback(Server, RegionServerServices)}
403   */
404  @VisibleForTesting
405  void openDaughters(final Server server, final RegionServerServices services, Region a,
406      Region b) throws IOException {
407    boolean stopped = server != null && server.isStopped();
408    boolean stopping = services != null && services.isStopping();
409    // TODO: Is this check needed here?
410    if (stopped || stopping) {
411      LOG.info("Not opening daughters " +
412          b.getRegionInfo().getRegionNameAsString() +
413          " and " +
414          a.getRegionInfo().getRegionNameAsString() +
415          " because stopping=" + stopping + ", stopped=" + stopped);
416    } else {
417      // Open daughters in parallel.
418      DaughterOpener aOpener = new DaughterOpener(server, a);
419      DaughterOpener bOpener = new DaughterOpener(server, b);
420      aOpener.start();
421      bOpener.start();
422      try {
423        aOpener.join();
424        if (aOpener.getException() == null) {
425          transition(SplitTransactionPhase.OPENED_REGION_A);
426        }
427        bOpener.join();
428        if (bOpener.getException() == null) {
429          transition(SplitTransactionPhase.OPENED_REGION_B);
430        }
431      } catch (InterruptedException e) {
432        throw (InterruptedIOException)new InterruptedIOException().initCause(e);
433      }
434      if (aOpener.getException() != null) {
435        throw new IOException("Failed " +
436          aOpener.getName(), aOpener.getException());
437      }
438      if (bOpener.getException() != null) {
439        throw new IOException("Failed " +
440          bOpener.getName(), bOpener.getException());
441      }
442      if (services != null) {
443        if (!services.reportRegionStateTransition(TransitionCode.SPLIT,
444            parent.getRegionInfo(), hri_a, hri_b)) {
445          throw new IOException("Failed to report split region to master: "
446            + parent.getRegionInfo().getShortNameToLog());
447        }
448        // Should add it to OnlineRegions
449        services.addToOnlineRegions(b);
450        services.addToOnlineRegions(a);
451      }
452    }
453  }

465  @Override
466  public PairOfSameType<Region> execute(final Server server, final RegionServerServices services,
467    User user) throws IOException {
468    this.server = server;
469    this.rsServices = services;
470    PairOfSameType<Region> regions = createDaughters(server, services, user);
471    stepsAfterPONR(server, services, regions, user);
472    transition(SplitTransactionPhase.COMPLETED);

476  @VisibleForTesting
477  void stepsAfterPONR(final Server server,
478      final RegionServerServices services, final PairOfSameType<Region> regions, User user)
479      throws IOException {
480    if (this.parent.getCoprocessorHost() != null) {
481      parent.getCoprocessorHost().preSplitAfterPONR(user);
482    }
483
484    openDaughters(server, services, regions.getFirst(), regions.getSecond());
485
486    transition(SplitTransactionPhase.BEFORE_POST_SPLIT_HOOK);
487
488    // Coprocessor callback
489    if (parent.getCoprocessorHost() != null) {
490      this.parent.getCoprocessorHost().postSplit(regions.getFirst(), regions.getSecond(), user);
491    }
492
493    transition(SplitTransactionPhase.AFTER_POST_SPLIT_HOOK);
494  }

496  /*
497   * Open daughter region in its own thread.
498   * If we fail, abort this hosting server.
499   */
500  private class DaughterOpener extends HasThread {
501    private final Server server;
502    private final Region r;
503    private Throwable t = null;
504
505    DaughterOpener(final Server s, final Region r) {
506      super((s == null? "null-services": s.getServerName()) +
507        "-daughterOpener=" + r.getRegionInfo().getEncodedName());
508      setDaemon(true);
509      this.server = s;
510      this.r = r;
511    }
512
513    /**
514     * @return Null if open succeeded else exception that causes us fail open.
515     * Call it after this thread exits else you may get wrong view on result.
516     */
517    Throwable getException() {
518      return this.t;
519    }
520
521    @Override
522    public void run() {
523      try {
524        openDaughterRegion(this.server, r);
525      } catch (Throwable t) {
526        this.t = t;
527      }
528    }

548  static class LoggingProgressable implements CancelableProgressable {
549    private final HRegionInfo hri;
550    private long lastLog = -1;
551    private final long interval;
552
553    LoggingProgressable(final HRegionInfo hri, final long interval) {
554      this.hri = hri;
555      this.interval = interval;
556    }
557
558    @Override
559    public boolean progress() {
560      long now = EnvironmentEdgeManager.currentTime();
561      if (now - lastLog > this.interval) {

563        this.lastLog = now;
564      }
565      return true;
566    }

569  /**
570   * Creates reference files for top and bottom half of the
571   * @param hstoreFilesToSplit map of store files to create half file references for.
572   * @return the number of reference files that were created.
573   * @throws IOException
574   */
575  private Pair<Integer, Integer> splitStoreFiles(
576      final Map<byte[], List<StoreFile>> hstoreFilesToSplit)
577      throws IOException {
578    if (hstoreFilesToSplit == null) {
579      // Could be null because close didn't succeed -- for now consider it fatal
580      throw new IOException("Close returned empty list of StoreFiles");
581    }
582    // The following code sets up a thread pool executor with as many slots as
583    // there's files to split. It then fires up everything, waits for
584    // completion and finally checks for any exception
585    int nbFiles = 0;
586    for (Map.Entry<byte[], List<StoreFile>> entry: hstoreFilesToSplit.entrySet()) {
587        nbFiles += entry.getValue().size();
588    }
589    if (nbFiles == 0) {
590      // no file needs to be splitted.
591      return new Pair<Integer, Integer>(0,0);
592    }
593    // Default max #threads to use is the smaller of table's configured number of blocking store
594    // files or the available number of logical cores.
595    int defMaxThreads = Math.min(parent.conf.getInt(HStore.BLOCKING_STOREFILES_KEY,
596                HStore.DEFAULT_BLOCKING_STOREFILE_COUNT),
597            Runtime.getRuntime().availableProcessors());
598    // Max #threads is the smaller of the number of storefiles or the default max determined above.
599    int maxThreads = Math.min(parent.conf.getInt(HConstants.REGION_SPLIT_THREADS_MAX,
600                defMaxThreads), nbFiles);
601    LOG.info("Preparing to split " + nbFiles + " storefiles for region " + this.parent +
602            " using " + maxThreads + " threads");
603    ThreadFactoryBuilder builder = new ThreadFactoryBuilder();
604    builder.setNameFormat("StoreFileSplitter-%1$d");
605    ThreadFactory factory = builder.build();
606    ThreadPoolExecutor threadPool =
607      (ThreadPoolExecutor) Executors.newFixedThreadPool(maxThreads, factory);
608    List<Future<Pair<Path,Path>>> futures = new ArrayList<Future<Pair<Path,Path>>> (nbFiles);
609
610    // Split each store file.
611    for (Map.Entry<byte[], List<StoreFile>> entry: hstoreFilesToSplit.entrySet()) {
612      for (StoreFile sf: entry.getValue()) {
613        StoreFileSplitter sfs = new StoreFileSplitter(entry.getKey(), sf);
614        futures.add(threadPool.submit(sfs));
615      }
616    }
617    // Shutdown the pool
618    threadPool.shutdown();
619
620    // Wait for all the tasks to finish
621    try {
622      boolean stillRunning = !threadPool.awaitTermination(
623          this.fileSplitTimeout, TimeUnit.MILLISECONDS);
624      if (stillRunning) {
625        threadPool.shutdownNow();
626        // wait for the thread to shutdown completely.
627        while (!threadPool.isTerminated()) {
628          Thread.sleep(50);
629        }
630        throw new IOException("Took too long to split the" +
631            " files and create the references, aborting split");
632      }
633    } catch (InterruptedException e) {
634      throw (InterruptedIOException)new InterruptedIOException().initCause(e);
635    }
636
637    int created_a = 0;
638    int created_b = 0;
639    // Look for any exception
640    for (Future<Pair<Path, Path>> future : futures) {
641      try {
642        Pair<Path, Path> p = future.get();
643        created_a += p.getFirst() != null ? 1 : 0;
644        created_b += p.getSecond() != null ? 1 : 0;
645      } catch (InterruptedException e) {
646        throw (InterruptedIOException) new InterruptedIOException().initCause(e);
647      } catch (ExecutionException e) {
648        throw new IOException(e);
649      }
650    }
651
652    if (LOG.isDebugEnabled()) {
653      LOG.debug("Split storefiles for region " + this.parent + " Daughter A: " + created_a
654          + " storefiles, Daughter B: " + created_b + " storefiles.");
655    }
656    return new Pair<Integer, Integer>(created_a, created_b);
657  }

659  private Pair<Path, Path> splitStoreFile(final byte[] family, final StoreFile sf)
660      throws IOException {
661    if (LOG.isDebugEnabled()) {
662        LOG.debug("Splitting started for store file: " + sf.getPath() + " for region: " +
663                  this.parent);
664    }
665    HRegionFileSystem fs = this.parent.getRegionFileSystem();
666    String familyName = Bytes.toString(family);
667    Path path_a =
668        fs.splitStoreFile(this.hri_a, familyName, sf, this.splitrow, false,
669          this.parent.getSplitPolicy());
670    Path path_b =
671        fs.splitStoreFile(this.hri_b, familyName, sf, this.splitrow, true,
672          this.parent.getSplitPolicy());
673    if (LOG.isDebugEnabled()) {
674        LOG.debug("Splitting complete for store file: " + sf.getPath() + " for region: " +
675                  this.parent);
676    }
677    return new Pair<Path,Path>(path_a, path_b);
678  }

705      throws IOException {

712  @Override
713  public boolean rollback(final Server server, final RegionServerServices services, User user)
714      throws IOException {
715    this.server = server;
716    this.rsServices = services;
717    // Coprocessor callback
718    if (this.parent.getCoprocessorHost() != null) {
719      this.parent.getCoprocessorHost().preRollBackSplit(user);
720    }
721
722    boolean result = true;
723    ListIterator<JournalEntry> iterator =
724      this.journal.listIterator(this.journal.size());
725    // Iterate in reverse.
726    while (iterator.hasPrevious()) {
727      JournalEntry je = iterator.previous();
728
729      transition(je.getPhase(), true);
730
731      switch (je.getPhase()) {
732
733      case SET_SPLITTING:
734        if (services != null
735            && !services.reportRegionStateTransition(TransitionCode.SPLIT_REVERTED,
736                parent.getRegionInfo(), hri_a, hri_b)) {
737          return false;
738        }
739        break;
740
741      case CREATE_SPLIT_DIR:
742        this.parent.writestate.writesEnabled = true;
743        this.parent.getRegionFileSystem().cleanupSplitsDir();
744        break;
745
746      case CLOSED_PARENT_REGION:
747        try {
748          // So, this returns a seqid but if we just closed and then reopened, we
749          // should be ok. On close, we flushed using sequenceid obtained from
750          // hosting regionserver so no need to propagate the sequenceid returned
751          // out of initialize below up into regionserver as we normally do.
752          // TODO: Verify.
753          this.parent.initialize();
754        } catch (IOException e) {
755          LOG.error("Failed rollbacking CLOSED_PARENT_REGION of region " +
756            parent.getRegionInfo().getRegionNameAsString(), e);
757          throw new RuntimeException(e);
758        }
759        break;
760
761      case STARTED_REGION_A_CREATION:
762        this.parent.getRegionFileSystem().cleanupDaughterRegion(this.hri_a);
763        break;
764
765      case STARTED_REGION_B_CREATION:
766        this.parent.getRegionFileSystem().cleanupDaughterRegion(this.hri_b);
767        break;
768
769      case OFFLINED_PARENT:
770        if (services != null) services.addToOnlineRegions(this.parent);
771        break;
772
773      case PONR:
774        // We got to the point-of-no-return so we need to just abort. Return
775        // immediately.  Do not clean up created daughter regions.  They need
776        // to be in place so we don't delete the parent region mistakenly.
777        // See HBASE-3872.
778        return false;
779
780      // Informational only cases
781      case STARTED:
782      case PREPARED:
783      case BEFORE_PRE_SPLIT_HOOK:
784      case AFTER_PRE_SPLIT_HOOK:
785      case BEFORE_POST_SPLIT_HOOK:
786      case AFTER_POST_SPLIT_HOOK:
787      case OPENED_REGION_A:
788      case OPENED_REGION_B:
789      case COMPLETED:
790        break;
791
792      default:
793        throw new RuntimeException("Unhandled journal entry: " + je);
794      }
795    }
796    // Coprocessor callback
797    if (this.parent.getCoprocessorHost() != null) {
798      this.parent.getCoprocessorHost().postRollBackSplit(user);
799    }
800    return result;
801  }
