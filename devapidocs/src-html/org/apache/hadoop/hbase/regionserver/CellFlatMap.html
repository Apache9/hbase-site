033/**
034 * CellFlatMap stores a constant number of elements and is immutable after creation stage.
035 * Being immutable, the CellFlatMap can be implemented as array.
036 * The actual array can be on- or off-heap and is implemented in concrete class derived from CellFlatMap.
037 * The CellFlatMap uses no synchronization primitives, it is assumed to be created by a
038 * single thread and then it can be read-only by multiple threads.
039 *
040 * The "flat" in the name, means that the memory layout of the Map is sequential array and thus
041 * requires less memory than ConcurrentSkipListMap.
042 */
043@InterfaceAudience.Private
044public abstract class CellFlatMap implements NavigableMap<Cell,Cell> {
045
046  private final Comparator<? super Cell> comparator;
047  protected int minCellIdx   = 0;   // the index of the minimal cell (for sub-sets)
048  protected int maxCellIdx   = 0;   // the index of the cell after the maximal cell (for sub-sets)
049  private boolean descending = false;
050
051  /* C-tor */
052  public CellFlatMap(Comparator<? super Cell> comparator, int min, int max, boolean d){
053    this.comparator = comparator;
054    this.minCellIdx = min;
055    this.maxCellIdx = max;
056    this.descending = d;
057  }
058
059  /* Used for abstract CellFlatMap creation, implemented by derived class */
060  protected abstract CellFlatMap createSubCellFlatMap(int min, int max, boolean descending);
061
062  /* Returns the i-th cell in the cell block */
063  protected abstract Cell getCell(int i);
064
065  /**
066   * Binary search for a given key in between given boundaries of the array.
067   * Positive returned numbers mean the index.
068   * Negative returned numbers means the key not found.
069   *
070   * The absolute value of the output is the
071   * possible insert index for the searched key
072   *
073   * In twos-complement, (-1 * insertion point)-1 is the bitwise not of the insert point.
074   *
075   *
076   * @param needle The key to look for in all of the entries
077   * @return Same return value as Arrays.binarySearch.
078   */
079  private int find(Cell needle) {
080    int begin = minCellIdx;
081    int end = maxCellIdx - 1;
082
083    while (begin <= end) {
084      int mid = (begin + end) >>> 1;
085      Cell midCell = getCell(mid);
086      int compareRes = comparator.compare(midCell, needle);
087
088      if (compareRes == 0) {
089        return mid;  // 0 means equals. We found the key
090      }
091
092      if (compareRes < 0) {
093        // midCell is less than needle so we need to look at farther up
094        begin = mid + 1;
095      } else {
096        // midCell is greater than needle so we need to look down
097        end = mid - 1;
098      }
099    }
100
101    return (-1 * begin)-1;
102  }
103
104  /* Get the index of the given anchor key for creating subsequent set.
105  ** It doesn't matter whether the given key exists in the set or not.
106  **
107  ** taking into consideration whether
108  ** the key should be inclusive or exclusive */
109  private int getValidIndex(Cell key, boolean inclusive, boolean tail) {
110    int index = find(key);
111    int result = -1;
112
113    // if the key is found and to be included, for all possibilities, the answer is the found index
114    if (index >= 0 && inclusive) result = index;
115
116    // The compliment Operator (~) converts the returned insertion point to the real one
117    if (index<0) result = ~index;
118
119    if (tail && result==-1) {
120      if (index >= 0 && !inclusive)
121        result = (descending) ? index - 1 : index + 1;
122    } else if (result==-1) {
123      if (index >= 0 && !inclusive)
124        result = (descending) ? index + 1 : index - 1;
125    }
126
127    if (result < minCellIdx || result > maxCellIdx) {
128      throw new IllegalArgumentException("Index " + result + " (initial index " + index + ") "
129          + " out of boundary, when looking for key " + key + ". The minCellIdx is " + minCellIdx
130          + " and the maxCellIdx is " + maxCellIdx + ". Finally, descending? " + descending
131          + " and was the key requested inclusively? " + inclusive);
132    }
133    return result;
134  }
135
136  @Override
137  public Comparator<? super Cell> comparator() {
138    return comparator;
139  }
140
141  @Override
142  public int size() {
143    return maxCellIdx-minCellIdx;
144  }
145
146  @Override
147  public boolean isEmpty() {
148    return ( size() == 0 );
149  }

152  // ---------------- Sub-Maps ----------------
153  @Override
154  public NavigableMap<Cell, Cell> subMap( Cell fromKey,
155                                                    boolean fromInclusive,
156                                                    Cell toKey,
157                                                    boolean toInclusive) {
158    int toIndex = getValidIndex(toKey, toInclusive, false);
159    int fromIndex = (getValidIndex(fromKey, fromInclusive, true));
160
161    if (fromIndex > toIndex) {
162      throw new IllegalArgumentException("Inconsistent range, when looking from "
163          + fromKey + " to " + toKey);
164    }
165    return createSubCellFlatMap(fromIndex, toIndex+1, descending);
166  }
167
168  @Override
169  public NavigableMap<Cell, Cell> headMap(Cell toKey, boolean inclusive) {
170    int index = getValidIndex(toKey, inclusive, false);
171    // "+1" because the max index is one after the true index
172    return createSubCellFlatMap(minCellIdx, index+1, descending);
173  }
174
175  @Override
176  public NavigableMap<Cell, Cell> tailMap(Cell fromKey, boolean inclusive) {
177    int index = (getValidIndex(fromKey, inclusive, true));
178    return createSubCellFlatMap(index, maxCellIdx, descending);
179  }
180
181  @Override
182  public NavigableMap<Cell, Cell> descendingMap() {
183    return createSubCellFlatMap(minCellIdx, maxCellIdx, true);
184  }
185
186  @Override
187  public NavigableMap<Cell, Cell> subMap(Cell k1, Cell k2) {
188    return this.subMap(k1, true, k2, true);
189  }

193    return this.headMap(k, true);
194  }

198    return this.tailMap(k, true);
199  }

202  // -------------------------------- Key's getters --------------------------------
203  @Override
204  public Cell firstKey() {
205    if (isEmpty()) {
206      return null;
207    }
208    return descending ? getCell(maxCellIdx - 1) : getCell(minCellIdx);
209  }
210
211  @Override
212  public Cell lastKey() {
213    if (isEmpty()) {
214      return null;
215    }
216    return descending ? getCell(minCellIdx) : getCell(maxCellIdx - 1);
217  }
218
219  @Override
220  public Cell lowerKey(Cell k) {
221    if (isEmpty()) {
222      return null;
223    }
224    int index = find(k);
225    // If index>=0 there's a key exactly equal
226    index = (index>=0) ? index-1 : -(index);
227    return (index < minCellIdx || index >= maxCellIdx) ? null : getCell(index);
228  }
229
230  @Override
231  public Cell floorKey(Cell k) {
232    if (isEmpty()) {
233      return null;
234    }
235    int index = find(k);
236    index = (index>=0) ? index : -(index);
237    return (index < minCellIdx || index >= maxCellIdx) ? null : getCell(index);
238  }
239
240  @Override
241  public Cell ceilingKey(Cell k) {
242    if (isEmpty()) {
243      return null;
244    }
245    int index = find(k);
246    index = (index>=0) ? index : -(index)+1;
247    return (index < minCellIdx || index >= maxCellIdx) ? null : getCell(index);
248  }
249
250  @Override
251  public Cell higherKey(Cell k) {
252    if (isEmpty()) {
253      return null;
254    }
255    int index = find(k);
256    index = (index>=0) ? index+1 : -(index)+1;
257    return (index < minCellIdx || index >= maxCellIdx) ? null : getCell(index);
258  }
259
260  @Override
261  public boolean containsKey(Object o) {
262    int index = find((Cell) o);
263    return (index >= 0);
264  }
265
266  @Override
267  public boolean containsValue(Object o) { // use containsKey(Object o) instead
268    throw new UnsupportedOperationException("Use containsKey(Object o) instead");
269  }
270
271  @Override
272  public Cell get(Object o) {
273    int index = find((Cell) o);
274    return (index >= 0) ? getCell(index) : null;
275  }

372  // -------------------------------- Iterator K --------------------------------
373  private final class CellFlatMapIterator implements Iterator<Cell> {
374    int index;
375
376    private CellFlatMapIterator() {
377      index = descending ? maxCellIdx-1 : minCellIdx;
378    }
379
380    @Override
381    public boolean hasNext() {
382      return descending ? (index >= minCellIdx) : (index < maxCellIdx);
383    }
384
385    @Override
386    public Cell next() {
387      Cell result = getCell(index);
388      if (descending) {
389        index--;
390      } else {
391        index++;
392      }
393      return result;
394    }
395
396    @Override
397    public void remove() {
398      throw new UnsupportedOperationException();
399    }
400  }
