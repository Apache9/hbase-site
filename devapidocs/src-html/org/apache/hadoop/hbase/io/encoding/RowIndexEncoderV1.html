011package org.apache.hadoop.hbase.io.encoding;
012
013import java.io.DataOutputStream;
014import java.io.IOException;
015
016import org.apache.commons.logging.Log;
017import org.apache.commons.logging.LogFactory;
018import org.apache.hadoop.hbase.Cell;
019import org.apache.hadoop.hbase.CellComparator;
020import org.apache.hadoop.hbase.CellUtil;
021import org.apache.hadoop.hbase.KeyValue;
022import org.apache.hadoop.hbase.KeyValueUtil;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.io.ByteArrayOutputStream;
025import org.apache.hadoop.io.WritableUtils;
026
027@InterfaceAudience.Private
028public class RowIndexEncoderV1 {
029  private static final Log LOG = LogFactory.getLog(RowIndexEncoderV1.class);
030
031  /** The Cell previously appended. */
032  private Cell lastCell = null;
033
034  private DataOutputStream out;
035  private HFileBlockDefaultEncodingContext encodingCtx;
036  private int startOffset = -1;
037  private ByteArrayOutputStream rowsOffsetBAOS = new ByteArrayOutputStream(
038      64 * 4);
039
040  public RowIndexEncoderV1(DataOutputStream out, HFileBlockDefaultEncodingContext encodingCtx) {
041    this.out = out;
042    this.encodingCtx = encodingCtx;
043  }
044
045  public int write(Cell cell) throws IOException {
046    // checkRow uses comparator to check we are writing in order.
047    if (!checkRow(cell)) {
048      if (startOffset < 0) {
049        startOffset = out.size();
050      }
051      rowsOffsetBAOS.writeInt(out.size() - startOffset);
052    }
053    int klength = KeyValueUtil.keyLength(cell);
054    int vlength = cell.getValueLength();
055    out.writeInt(klength);
056    out.writeInt(vlength);
057    CellUtil.writeFlatKey(cell, out);
058    // Write the value part
059    CellUtil.writeValue(out, cell, vlength);
060    int encodedKvSize = klength + vlength
061        + KeyValue.KEYVALUE_INFRASTRUCTURE_SIZE;
062    // Write the additional tag into the stream
063    if (encodingCtx.getHFileContext().isIncludesTags()) {
064      int tagsLength = cell.getTagsLength();
065      out.writeShort(tagsLength);
066      if (tagsLength > 0) {
067        CellUtil.writeTags(out, cell, tagsLength);
068      }
069      encodedKvSize += tagsLength + KeyValue.TAGS_LENGTH_SIZE;
070    }
071    if (encodingCtx.getHFileContext().isIncludesMvcc()) {
072      WritableUtils.writeVLong(out, cell.getSequenceId());
073      encodedKvSize += WritableUtils.getVIntSize(cell.getSequenceId());
074    }
075    lastCell = cell;
076    return encodedKvSize;
077  }
078
079  protected boolean checkRow(final Cell cell) throws IOException {
080    boolean isDuplicateRow = false;
081    if (cell == null) {
082      throw new IOException("Key cannot be null or empty");
083    }
084    if (lastCell != null) {
085      int keyComp = CellComparator.COMPARATOR.compareRows(lastCell, cell);
086      if (keyComp > 0) {
087        throw new IOException("Added a key not lexically larger than"
088            + " previous. Current cell = " + cell + ", lastCell = " + lastCell);
089      } else if (keyComp == 0) {
090        isDuplicateRow = true;
091      }
092    }
093    return isDuplicateRow;
094  }
095
096  public void flush() throws IOException {
097    int onDiskDataSize = 0;
098    if (startOffset >= 0) {
099      onDiskDataSize = out.size() - startOffset;
100    }
101    out.writeInt(rowsOffsetBAOS.size() / 4);
102    if (rowsOffsetBAOS.size() > 0) {
103      out.write(rowsOffsetBAOS.getBuffer(), 0, rowsOffsetBAOS.size());
104    }
105    out.writeInt(onDiskDataSize);
106    if (LOG.isTraceEnabled()) {
107      LOG.trace("RowNumber: " + rowsOffsetBAOS.size() / 4
108          + ", onDiskDataSize: " + onDiskDataSize + ", totalOnDiskSize: "
109          + (out.size() - startOffset));
110    }
111  }
112
113}
