018package org.apache.hadoop.hbase.io.util;
019
020import java.lang.management.ManagementFactory;
021import java.lang.management.MemoryUsage;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.HConstants;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028
029@InterfaceAudience.Private
030public class HeapMemorySizeUtil {
031
032  public static final String MEMSTORE_SIZE_KEY = "hbase.regionserver.global.memstore.size";
033  public static final String MEMSTORE_SIZE_OLD_KEY =
034      "hbase.regionserver.global.memstore.upperLimit";
035  public static final String MEMSTORE_SIZE_LOWER_LIMIT_KEY =
036      "hbase.regionserver.global.memstore.size.lower.limit";
037  public static final String MEMSTORE_SIZE_LOWER_LIMIT_OLD_KEY =
038      "hbase.regionserver.global.memstore.lowerLimit";
039
040  public static final float DEFAULT_MEMSTORE_SIZE = 0.4f;
041  // Default lower water mark limit is 95% size of memstore size.
042  public static final float DEFAULT_MEMSTORE_SIZE_LOWER_LIMIT = 0.95f;
043
044  private static final Log LOG = LogFactory.getLog(HeapMemorySizeUtil.class);
045  // a constant to convert a fraction to a percentage
046  private static final int CONVERT_TO_PERCENTAGE = 100;
047
048  /**
049   * Checks whether we have enough heap memory left out after portion for Memstore and Block cache.
050   * We need atleast 20% of heap left out for other RS functions.
051   * @param conf
052   */
053  public static void checkForClusterFreeMemoryLimit(Configuration conf) {
054    if (conf.get(MEMSTORE_SIZE_OLD_KEY) != null) {
055      LOG.warn(MEMSTORE_SIZE_OLD_KEY + " is deprecated by " + MEMSTORE_SIZE_KEY);
056    }
057    float globalMemstoreSize = getGlobalMemStorePercent(conf, false);
058    int gml = (int)(globalMemstoreSize * CONVERT_TO_PERCENTAGE);
059    float blockCacheUpperLimit = getBlockCacheHeapPercent(conf);
060    int bcul = (int)(blockCacheUpperLimit * CONVERT_TO_PERCENTAGE);
061    if (CONVERT_TO_PERCENTAGE - (gml + bcul)
062            < (int)(CONVERT_TO_PERCENTAGE *
063                    HConstants.HBASE_CLUSTER_MINIMUM_MEMORY_THRESHOLD)) {
064      throw new RuntimeException("Current heap configuration for MemStore and BlockCache exceeds "
065          + "the threshold required for successful cluster operation. "
066          + "The combined value cannot exceed 0.8. Please check "
067          + "the settings for hbase.regionserver.global.memstore.size and "
068          + "hfile.block.cache.size in your configuration. "
069          + "hbase.regionserver.global.memstore.size is " + globalMemstoreSize
070          + " hfile.block.cache.size is " + blockCacheUpperLimit);
071    }
072  }
073
074  /**
075   * Retrieve global memstore configured size as percentage of total heap.
076   * @param c
077   * @param logInvalid
078   */
079  public static float getGlobalMemStorePercent(final Configuration c, final boolean logInvalid) {
080    float limit = c.getFloat(MEMSTORE_SIZE_KEY,
081        c.getFloat(MEMSTORE_SIZE_OLD_KEY, DEFAULT_MEMSTORE_SIZE));
082    if (limit > 0.8f || limit <= 0.0f) {
083      if (logInvalid) {
084        LOG.warn("Setting global memstore limit to default of " + DEFAULT_MEMSTORE_SIZE
085            + " because supplied value outside allowed range of (0 -> 0.8]");
086      }
087      limit = DEFAULT_MEMSTORE_SIZE;
088    }
089    return limit;
090  }
091
092  /**
093   * Retrieve configured size for global memstore lower water mark as fraction of global memstore
094   * size.
095   */
096  public static float getGlobalMemStoreLowerMark(final Configuration conf, float globalMemStorePercent) {
097    String lowMarkPercentStr = conf.get(MEMSTORE_SIZE_LOWER_LIMIT_KEY);
098    if (lowMarkPercentStr != null) {
099      float lowMarkPercent = Float.parseFloat(lowMarkPercentStr);
100      if (lowMarkPercent > 1.0f) {
101        LOG.error("Bad configuration value for " + MEMSTORE_SIZE_LOWER_LIMIT_KEY + ": " +
102            lowMarkPercent + ". Using 1.0f instead.");
103        lowMarkPercent = 1.0f;
104      }
105      return lowMarkPercent;
106    }
107    String lowerWaterMarkOldValStr = conf.get(MEMSTORE_SIZE_LOWER_LIMIT_OLD_KEY);
108    if (lowerWaterMarkOldValStr != null) {
109      LOG.warn(MEMSTORE_SIZE_LOWER_LIMIT_OLD_KEY + " is deprecated. Instead use "
110          + MEMSTORE_SIZE_LOWER_LIMIT_KEY);
111      float lowerWaterMarkOldVal = Float.parseFloat(lowerWaterMarkOldValStr);
112      if (lowerWaterMarkOldVal > globalMemStorePercent) {
113        lowerWaterMarkOldVal = globalMemStorePercent;
114        LOG.error("Value of " + MEMSTORE_SIZE_LOWER_LIMIT_OLD_KEY + " (" + lowerWaterMarkOldVal
115            + ") is greater than global memstore limit (" + globalMemStorePercent + ") set by "
116            + MEMSTORE_SIZE_KEY + "/" + MEMSTORE_SIZE_OLD_KEY + ". Setting memstore lower limit "
117            + "to " + globalMemStorePercent);
118      }
119      return lowerWaterMarkOldVal / globalMemStorePercent;
120    }
121    return DEFAULT_MEMSTORE_SIZE_LOWER_LIMIT;
122  }
123
124  /**
125   * Retrieve configured size for on heap block cache as percentage of total heap.
126   * @param conf
127   */
128  public static float getBlockCacheHeapPercent(final Configuration conf) {
129    // L1 block cache is always on heap
130    float l1CachePercent = conf.getFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY,
131        HConstants.HFILE_BLOCK_CACHE_SIZE_DEFAULT);
132    float l2CachePercent = getL2BlockCacheHeapPercent(conf);
133    return l1CachePercent + l2CachePercent;
134  }
135
136  /**
137   * @param conf
138   * @return The on heap size for L2 block cache.
139   */
140  public static float getL2BlockCacheHeapPercent(Configuration conf) {
141    float l2CachePercent = 0.0F;
142    String bucketCacheIOEngineName = conf.get(HConstants.BUCKET_CACHE_IOENGINE_KEY, null);
143    // L2 block cache can be on heap when IOEngine is "heap"
144    if (bucketCacheIOEngineName != null && bucketCacheIOEngineName.startsWith("heap")) {
145      float bucketCachePercentage = conf.getFloat(HConstants.BUCKET_CACHE_SIZE_KEY, 0F);
146      MemoryUsage mu = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
147      l2CachePercent = bucketCachePercentage < 1 ? bucketCachePercentage
148          : (bucketCachePercentage * 1024 * 1024) / mu.getMax();
149    }
150    return l2CachePercent;
151  }
152}
