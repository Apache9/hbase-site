019package org.apache.hadoop.hbase.io.hfile.bucket;
020
021import java.io.File;
022import java.io.IOException;
023import java.io.RandomAccessFile;
024import java.nio.ByteBuffer;
025import java.nio.channels.FileChannel;
026import java.util.Arrays;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.io.hfile.Cacheable;
032import org.apache.hadoop.hbase.io.hfile.CacheableDeserializer;
033import org.apache.hadoop.hbase.io.hfile.Cacheable.MemoryType;
034import org.apache.hadoop.hbase.nio.ByteBuff;
035import org.apache.hadoop.hbase.nio.SingleByteBuff;
036import org.apache.hadoop.hbase.shaded.com.google.common.base.Preconditions;
037import org.apache.hadoop.util.StringUtils;
038
039/**
040 * IO engine that stores data to a file on the local file system.
041 */
042@InterfaceAudience.Private
043public class FileIOEngine implements IOEngine {
044  private static final Log LOG = LogFactory.getLog(FileIOEngine.class);
045  public static final String FILE_DELIMITER = ",";
046  private final String[] filePaths;
047  private final FileChannel[] fileChannels;
048  private final RandomAccessFile[] rafs;
049
050  private final long sizePerFile;
051  private final long capacity;
052
053  private FileReadAccessor readAccessor = new FileReadAccessor();
054  private FileWriteAccessor writeAccessor = new FileWriteAccessor();
055
056  public FileIOEngine(long capacity, boolean maintainPersistence, String... filePaths)
057      throws IOException {
058    this.sizePerFile = capacity / filePaths.length;
059    this.capacity = this.sizePerFile * filePaths.length;
060    this.filePaths = filePaths;
061    this.fileChannels = new FileChannel[filePaths.length];
062    if (!maintainPersistence) {
063      for (String filePath : filePaths) {
064        File file = new File(filePath);
065        if (file.exists()) {
066          if (LOG.isDebugEnabled()) {
067            LOG.debug("File " + filePath + " already exists. Deleting!!");
068          }
069          file.delete();
070          // If deletion fails still we can manage with the writes
071        }
072      }
073    }
074    this.rafs = new RandomAccessFile[filePaths.length];
075    for (int i = 0; i < filePaths.length; i++) {
076      String filePath = filePaths[i];
077      try {
078        rafs[i] = new RandomAccessFile(filePath, "rw");
079        long totalSpace = new File(filePath).getTotalSpace();
080        if (totalSpace < sizePerFile) {
081          // The next setting length will throw exception,logging this message
082          // is just used for the detail reason of exceptionï¼Œ
083          String msg = "Only " + StringUtils.byteDesc(totalSpace)
084              + " total space under " + filePath + ", not enough for requested "
085              + StringUtils.byteDesc(sizePerFile);
086          LOG.warn(msg);
087        }
088        rafs[i].setLength(sizePerFile);
089        fileChannels[i] = rafs[i].getChannel();
090        LOG.info("Allocating cache " + StringUtils.byteDesc(sizePerFile)
091            + ", on the path:" + filePath);
092      } catch (IOException fex) {
093        LOG.error("Failed allocating cache on " + filePath, fex);
094        shutdown();
095        throw fex;
096      }
097    }
098  }
099
100  @Override
101  public String toString() {
102    return "ioengine=" + this.getClass().getSimpleName() + ", paths="
103        + Arrays.asList(filePaths) + ", capacity=" + String.format("%,d", this.capacity);
104  }
105
106  /**
107   * File IO engine is always able to support persistent storage for the cache
108   * @return true
109   */
110  @Override
111  public boolean isPersistent() {
112    return true;
113  }
114
115  /**
116   * Transfers data from file to the given byte buffer
117   * @param offset The offset in the file where the first byte to be read
118   * @param length The length of buffer that should be allocated for reading
119   *               from the file channel
120   * @return number of bytes read
121   * @throws IOException
122   */
123  @Override
124  public Cacheable read(long offset, int length, CacheableDeserializer<Cacheable> deserializer)
125      throws IOException {
126    Preconditions.checkArgument(length >= 0, "Length of read can not be less than 0.");
127    ByteBuffer dstBuffer = ByteBuffer.allocate(length);
128    if (length != 0) {
129      accessFile(readAccessor, dstBuffer, offset);
130      // The buffer created out of the fileChannel is formed by copying the data from the file
131      // Hence in this case there is no shared memory that we point to. Even if the BucketCache evicts
132      // this buffer from the file the data is already copied and there is no need to ensure that
133      // the results are not corrupted before consuming them.
134      if (dstBuffer.limit() != length) {
135        throw new RuntimeException("Only " + dstBuffer.limit() + " bytes read, " + length
136            + " expected");
137      }
138    }
139    return deserializer.deserialize(new SingleByteBuff(dstBuffer), true, MemoryType.EXCLUSIVE);
140  }
141
142  /**
143   * Transfers data from the given byte buffer to file
144   * @param srcBuffer the given byte buffer from which bytes are to be read
145   * @param offset The offset in the file where the first byte to be written
146   * @throws IOException
147   */
148  @Override
149  public void write(ByteBuffer srcBuffer, long offset) throws IOException {
150    if (!srcBuffer.hasRemaining()) {
151      return;
152    }
153    accessFile(writeAccessor, srcBuffer, offset);
154  }
155
156  /**
157   * Sync the data to file after writing
158   * @throws IOException
159   */
160  @Override
161  public void sync() throws IOException {
162    for (int i = 0; i < fileChannels.length; i++) {
163      try {
164        if (fileChannels[i] != null) {
165          fileChannels[i].force(true);
166        }
167      } catch (IOException ie) {
168        LOG.warn("Failed syncing data to " + this.filePaths[i]);
169        throw ie;
170      }
171    }
172  }
173
174  /**
175   * Close the file
176   */
177  @Override
178  public void shutdown() {
179    for (int i = 0; i < filePaths.length; i++) {
180      try {
181        if (fileChannels[i] != null) {
182          fileChannels[i].close();
183        }
184        if (rafs[i] != null) {
185          rafs[i].close();
186        }
187      } catch (IOException ex) {
188        LOG.error("Failed closing " + filePaths[i] + " when shudown the IOEngine", ex);
189      }
190    }
191  }
192
193  @Override
194  public void write(ByteBuff srcBuffer, long offset) throws IOException {
195    // When caching block into BucketCache there will be single buffer backing for this HFileBlock.
196    assert srcBuffer.hasArray();
197    write(ByteBuffer.wrap(srcBuffer.array(), srcBuffer.arrayOffset(),
198            srcBuffer.remaining()), offset);
199  }
200
201  private void accessFile(FileAccessor accessor, ByteBuffer buffer,
202      long globalOffset) throws IOException {
203    int startFileNum = getFileNum(globalOffset);
204    int remainingAccessDataLen = buffer.remaining();
205    int endFileNum = getFileNum(globalOffset + remainingAccessDataLen - 1);
206    int accessFileNum = startFileNum;
207    long accessOffset = getAbsoluteOffsetInFile(accessFileNum, globalOffset);
208    int bufLimit = buffer.limit();
209    while (true) {
210      FileChannel fileChannel = fileChannels[accessFileNum];
211      if (endFileNum > accessFileNum) {
212        // short the limit;
213        buffer.limit((int) (buffer.limit() - remainingAccessDataLen
214            + sizePerFile - accessOffset));
215      }
216      int accessLen = accessor.access(fileChannel, buffer, accessOffset);
217      // recover the limit
218      buffer.limit(bufLimit);
219      if (accessLen < remainingAccessDataLen) {
220        remainingAccessDataLen -= accessLen;
221        accessFileNum++;
222        accessOffset = 0;
223      } else {
224        break;
225      }
226      if (accessFileNum >= fileChannels.length) {
227        throw new IOException("Required data len "
228            + StringUtils.byteDesc(buffer.remaining())
229            + " exceed the engine's capacity " + StringUtils.byteDesc(capacity)
230            + " where offset=" + globalOffset);
231      }
232    }
233  }
234
235  /**
236   * Get the absolute offset in given file with the relative global offset.
237   * @param fileNum
238   * @param globalOffset
239   * @return the absolute offset
240   */
241  private long getAbsoluteOffsetInFile(int fileNum, long globalOffset) {
242    return globalOffset - fileNum * sizePerFile;
243  }
244
245  private int getFileNum(long offset) {
246    if (offset < 0) {
247      throw new IllegalArgumentException("Unexpected offset " + offset);
248    }
249    int fileNum = (int) (offset / sizePerFile);
250    if (fileNum >= fileChannels.length) {
251      throw new RuntimeException("Not expected offset " + offset
252          + " where capacity=" + capacity);
253    }
254    return fileNum;
255  }
256
257  private static interface FileAccessor {
258    int access(FileChannel fileChannel, ByteBuffer byteBuffer, long accessOffset)
259        throws IOException;
260  }
261
262  private static class FileReadAccessor implements FileAccessor {
263    @Override
264    public int access(FileChannel fileChannel, ByteBuffer byteBuffer,
265        long accessOffset) throws IOException {
266      return fileChannel.read(byteBuffer, accessOffset);
267    }
268  }
269
270  private static class FileWriteAccessor implements FileAccessor {
271    @Override
272    public int access(FileChannel fileChannel, ByteBuffer byteBuffer,
273        long accessOffset) throws IOException {
274      return fileChannel.write(byteBuffer, accessOffset);
275    }
276  }
