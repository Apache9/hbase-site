036/**
037 * Utilty for aggregating counts in CachedBlocks and toString/toJSON CachedBlocks and BlockCaches.
038 * No attempt has been made at making this thread safe.
039 */
040@InterfaceAudience.Private
041public class BlockCacheUtil {
042
043
044  public static final long NANOS_PER_SECOND = 1000000000;
045
046  /**
047   * Needed generating JSON.
048   */
049  private static final ObjectMapper MAPPER = new ObjectMapper();
050  static {
051    MAPPER.configure(SerializationConfig.Feature.FAIL_ON_EMPTY_BEANS, false);
052    MAPPER.configure(SerializationConfig.Feature.FLUSH_AFTER_WRITE_VALUE, true);
053    MAPPER.configure(SerializationConfig.Feature.INDENT_OUTPUT, true);
054  }
055
056  /**
057   * @param cb
058   * @return The block content as String.
059   */
060  public static String toString(final CachedBlock cb, final long now) {
061    return "filename=" + cb.getFilename() + ", " + toStringMinusFileName(cb, now);
062  }
063
064  /**
065   * Little data structure to hold counts for a file.
066   * Used doing a toJSON.
067   */
068  static class CachedBlockCountsPerFile {
069    private int count = 0;
070    private long size = 0;
071    private int countData = 0;
072    private long sizeData = 0;
073    private final String filename;
074
075    CachedBlockCountsPerFile(final String filename) {
076      this.filename = filename;
077    }
078
079    public int getCount() {
080      return count;
081    }
082
083    public long getSize() {
084      return size;
085    }
086
087    public int getCountData() {
088      return countData;
089    }
090
091    public long getSizeData() {
092      return sizeData;
093    }
094
095    public String getFilename() {
096      return filename;
097    }
098  }
099
100  /**
101   * @param filename
102   * @param blocks
103   * @return A JSON String of <code>filename</code> and counts of <code>blocks</code>
104   * @throws JsonGenerationException
105   * @throws JsonMappingException
106   * @throws IOException
107   */
108  public static String toJSON(final String filename, final NavigableSet<CachedBlock> blocks)
109  throws JsonGenerationException, JsonMappingException, IOException {
110    CachedBlockCountsPerFile counts = new CachedBlockCountsPerFile(filename);
111    for (CachedBlock cb: blocks) {
112      counts.count++;
113      counts.size += cb.getSize();
114      BlockType bt = cb.getBlockType();
115      if (bt != null && bt.isData()) {
116        counts.countData++;
117        counts.sizeData += cb.getSize();
118      }
119    }
120    return MAPPER.writeValueAsString(counts);
121  }
122
123  /**
124   * @param cbsbf
125   * @return JSON string of <code>cbsf</code> aggregated
126   * @throws JsonGenerationException
127   * @throws JsonMappingException
128   * @throws IOException
129   */
130  public static String toJSON(final CachedBlocksByFile cbsbf)
131  throws JsonGenerationException, JsonMappingException, IOException {
132    return MAPPER.writeValueAsString(cbsbf);
133  }
134
135  /**
136   * @param bc
137   * @return JSON string of <code>bc</code> content.
138   * @throws JsonGenerationException
139   * @throws JsonMappingException
140   * @throws IOException
141   */
142  public static String toJSON(final BlockCache bc)
143  throws JsonGenerationException, JsonMappingException, IOException {
144    return MAPPER.writeValueAsString(bc);
145  }
146
147  /**
148   * @param cb
149   * @return The block content of <code>bc</code> as a String minus the filename.
150   */
151  public static String toStringMinusFileName(final CachedBlock cb, final long now) {
152    return "offset=" + cb.getOffset() +
153      ", size=" + cb.getSize() +
154      ", age=" + (now - cb.getCachedTime()) +
155      ", type=" + cb.getBlockType() +
156      ", priority=" + cb.getBlockPriority();
157  }
158
159  /**
160   * Get a {@link CachedBlocksByFile} instance and load it up by iterating content in
161   * {@link BlockCache}.
162   * @param conf Used to read configurations
163   * @param bc Block Cache to iterate.
164   * @return Laoded up instance of CachedBlocksByFile
165   */
166  public static CachedBlocksByFile getLoadedCachedBlocksByFile(final Configuration conf,
167      final BlockCache bc) {
168    CachedBlocksByFile cbsbf = new CachedBlocksByFile(conf);
169    for (CachedBlock cb: bc) {
170      if (cbsbf.update(cb)) break;
171    }
172    return cbsbf;
173  }
174
175  /**
176   * Use one of these to keep a running account of cached blocks by file.  Throw it away when done.
177   * This is different than metrics in that it is stats on current state of a cache.
178   * See getLoadedCachedBlocksByFile
179   */
180  @JsonIgnoreProperties({"cachedBlockStatsByFile"})
181  public static class CachedBlocksByFile {
182    private int count;
183    private int dataBlockCount;
184    private long size;
185    private long dataSize;
186    private final long now = System.nanoTime();
187    /**
188     * How many blocks to look at before we give up.
189     * There could be many millions of blocks. We don't want the
190     * ui to freeze while we run through 1B blocks... users will
191     * think hbase dead. UI displays warning in red when stats
192     * are incomplete.
193     */
194    private final int max;
195    public static final int DEFAULT_MAX = 1000000;
196
197    CachedBlocksByFile() {
198      this(null);
199    }
200
201    CachedBlocksByFile(final Configuration c) {
202      this.max = c == null? DEFAULT_MAX: c.getInt("hbase.ui.blockcache.by.file.max", DEFAULT_MAX);
203    }
204
205    /**
206     * Map by filename. use concurent utils because we want our Map and contained blocks sorted.
207     */
208    private NavigableMap<String, NavigableSet<CachedBlock>> cachedBlockByFile = new ConcurrentSkipListMap<>();
209    FastLongHistogram hist = new FastLongHistogram();
210
211    /**
212     * @param cb
213     * @return True if full.... if we won't be adding any more.
214     */
215    public boolean update(final CachedBlock cb) {
216      if (isFull()) return true;
217      NavigableSet<CachedBlock> set = this.cachedBlockByFile.get(cb.getFilename());
218      if (set == null) {
219        set = new ConcurrentSkipListSet<>();
220        this.cachedBlockByFile.put(cb.getFilename(), set);
221      }
222      set.add(cb);
223      this.size += cb.getSize();
224      this.count++;
225      BlockType bt = cb.getBlockType();
226      if (bt != null && bt.isData()) {
227        this.dataBlockCount++;
228        this.dataSize += cb.getSize();
229      }
230      long age = (this.now - cb.getCachedTime())/NANOS_PER_SECOND;
231      this.hist.add(age, 1);
232      return false;
233    }
234
235    /**
236     * @return True if full; i.e. there are more items in the cache but we only loaded up
237     * the maximum set in configuration <code>hbase.ui.blockcache.by.file.max</code>
238     * (Default: DEFAULT_MAX).
239     */
240    public boolean isFull() {
241      return this.count >= this.max;
242    }
243
244    public NavigableMap<String, NavigableSet<CachedBlock>> getCachedBlockStatsByFile() {
245      return this.cachedBlockByFile;
246    }
247
248    /**
249     * @return count of blocks in the cache
250     */
251    public int getCount() {
252      return count;
253    }
254
255    public int getDataCount() {
256      return dataBlockCount;
257    }
258
259    /**
260     * @return size of blocks in the cache
261     */
262    public long getSize() {
263      return size;
264    }
265
266    /**
267     * @return Size of data.
268     */
269    public long getDataSize() {
270      return dataSize;
271    }
272
273    public AgeSnapshot getAgeInCacheSnapshot() {
274      return new AgeSnapshot(this.hist);
275    }
276
277    @Override
278    public String toString() {
279      AgeSnapshot snapshot = getAgeInCacheSnapshot();
280      return "count=" + count + ", dataBlockCount=" + dataBlockCount + ", size=" + size +
281          ", dataSize=" + getDataSize() +
282          ", mean age=" + snapshot.getMean() +
283          ", min age=" + snapshot.getMin() +
284          ", max age=" + snapshot.getMax() +
285          ", 75th percentile age="   + snapshot.get75thPercentile() +
286          ", 95th percentile age="   + snapshot.get95thPercentile() +
287          ", 98th percentile age="   + snapshot.get98thPercentile() +
288          ", 99th percentile age="   + snapshot.get99thPercentile() +
289          ", 99.9th percentile age=" + snapshot.get99thPercentile();
290    }
291  }
