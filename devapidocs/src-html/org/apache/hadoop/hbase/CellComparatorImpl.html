031/**
032 * Compare two HBase cells.  Do not use this method comparing <code>-ROOT-</code> or
033 * <code>hbase:meta</code> cells.  Cells from these tables need a specialized comparator, one that
034 * takes account of the special formatting of the row where we have commas to delimit table from
035 * regionname, from row.  See KeyValue for how it has a special comparator to do hbase:meta cells
036 * and yet another for -ROOT-.
037 * While using this comparator for {{@link #compareRows(Cell, Cell)} et al, the hbase:meta cells
038 * format should be taken into consideration, for which the instance of this comparator
039 * should be used.  In all other cases the static APIs in this comparator would be enough
040 */
041@edu.umd.cs.findbugs.annotations.SuppressWarnings(
042    value="UNKNOWN",
043    justification="Findbugs doesn't like the way we are negating the result of a compare in below")
044@InterfaceAudience.Private
045@InterfaceStability.Evolving
046public class CellComparatorImpl implements CellComparator {
047  static final Logger LOG = LoggerFactory.getLogger(CellComparatorImpl.class);
048  /**
049   * Comparator for plain key/values; i.e. non-catalog table key/values. Works on Key portion
050   * of KeyValue only.
051   */
052  public static final CellComparatorImpl COMPARATOR = new CellComparatorImpl();
053  /**
054   * A {@link CellComparatorImpl} for <code>hbase:meta</code> catalog table
055   * {@link KeyValue}s.
056   */
057  public static final CellComparatorImpl META_COMPARATOR = new MetaCellComparator();
058
059  @Override
060  public int compare(Cell a, Cell b) {
061    return compare(a, b, false);
062  }
063
064  /**
065   * Compare cells.
066   * @param a
067   * @param b
068   * @param ignoreSequenceid True if we are to compare the key portion only and ignore
069   * the sequenceid. Set to false to compare key and consider sequenceid.
070   * @return 0 if equal, -1 if a &lt; b, and +1 if a &gt; b.
071   */
072  public final int compare(final Cell a, final Cell b, boolean ignoreSequenceid) {
073    // row
074    int c = compareRows(a, b);
075    if (c != 0) return c;
076
077    c = compareWithoutRow(a, b);
078    if(c != 0) return c;
079
080    if (!ignoreSequenceid) {
081      // Negate following comparisons so later edits show up first
082      // mvccVersion: later sorts first
083      return Longs.compare(b.getSequenceId(), a.getSequenceId());
084    } else {
085      return c;
086    }
087  }
088
089  /**
090   * Compares the family and qualifier part of the cell
091   * @param left the left cell
092   * @param right the right cell
093   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
094   */
095  public final int compareColumns(final Cell left, final Cell right) {
096    int diff = compareFamilies(left, right);
097    if (diff != 0) {
098      return diff;
099    }
100    return compareQualifiers(left, right);
101  }
102
103  /**
104   * Compare the families of left and right cell
105   * @param left
106   * @param right
107   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
108   */
109  @Override
110  public final int compareFamilies(Cell left, Cell right) {
111    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
112      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getFamilyByteBuffer(),
113          ((ByteBufferExtendedCell) left).getFamilyPosition(), left.getFamilyLength(),
114          ((ByteBufferExtendedCell) right).getFamilyByteBuffer(),
115          ((ByteBufferExtendedCell) right).getFamilyPosition(), right.getFamilyLength());
116    }
117    if (left instanceof ByteBufferExtendedCell) {
118      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getFamilyByteBuffer(),
119          ((ByteBufferExtendedCell) left).getFamilyPosition(), left.getFamilyLength(),
120          right.getFamilyArray(), right.getFamilyOffset(), right.getFamilyLength());
121    }
122    if (right instanceof ByteBufferExtendedCell) {
123      // Notice how we flip the order of the compare here. We used to negate the return value but
124      // see what FindBugs says
125      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
126      // It suggest flipping the order to get same effect and 'safer'.
127      return ByteBufferUtils.compareTo(
128          left.getFamilyArray(), left.getFamilyOffset(), left.getFamilyLength(),
129          ((ByteBufferExtendedCell)right).getFamilyByteBuffer(),
130          ((ByteBufferExtendedCell)right).getFamilyPosition(), right.getFamilyLength());
131    }
132    return Bytes.compareTo(left.getFamilyArray(), left.getFamilyOffset(), left.getFamilyLength(),
133        right.getFamilyArray(), right.getFamilyOffset(), right.getFamilyLength());
134  }
135
136  /**
137   * Compare the qualifiers part of the left and right cells.
138   * @param left
139   * @param right
140   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
141   */
142  @Override
143  public final int compareQualifiers(Cell left, Cell right) {
144    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
145      return ByteBufferUtils
146          .compareTo(((ByteBufferExtendedCell) left).getQualifierByteBuffer(),
147              ((ByteBufferExtendedCell) left).getQualifierPosition(),
148              left.getQualifierLength(), ((ByteBufferExtendedCell) right).getQualifierByteBuffer(),
149              ((ByteBufferExtendedCell) right).getQualifierPosition(),
150              right.getQualifierLength());
151    }
152    if (left instanceof ByteBufferExtendedCell) {
153      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getQualifierByteBuffer(),
154          ((ByteBufferExtendedCell) left).getQualifierPosition(), left.getQualifierLength(),
155          right.getQualifierArray(), right.getQualifierOffset(), right.getQualifierLength());
156    }
157    if (right instanceof ByteBufferExtendedCell) {
158      // Notice how we flip the order of the compare here. We used to negate the return value but
159      // see what FindBugs says
160      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
161      // It suggest flipping the order to get same effect and 'safer'.
162      return ByteBufferUtils.compareTo(left.getQualifierArray(),
163          left.getQualifierOffset(), left.getQualifierLength(),
164          ((ByteBufferExtendedCell)right).getQualifierByteBuffer(),
165          ((ByteBufferExtendedCell)right).getQualifierPosition(), right.getQualifierLength());
166    }
167    return Bytes.compareTo(left.getQualifierArray(), left.getQualifierOffset(),
168        left.getQualifierLength(), right.getQualifierArray(), right.getQualifierOffset(),
169        right.getQualifierLength());
170  }
171
172  /**
173   * Compares the rows of the left and right cell.
174   * For the hbase:meta case this method is overridden such that it can handle hbase:meta cells.
175   * The caller should ensure using the appropriate comparator for hbase:meta.
176   * @param left
177   * @param right
178   * @return 0 if both cells are equal, 1 if left cell is bigger than right, -1 otherwise
179   */
180  @Override
181  public int compareRows(final Cell left, final Cell right) {
182    // left and right can be exactly the same at the beginning of a row
183    if (left == right) {
184      return 0;
185    }
186    if (left instanceof ByteBufferExtendedCell && right instanceof ByteBufferExtendedCell) {
187      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
188          ((ByteBufferExtendedCell) left).getRowPosition(), left.getRowLength(),
189          ((ByteBufferExtendedCell) right).getRowByteBuffer(),
190          ((ByteBufferExtendedCell) right).getRowPosition(), right.getRowLength());
191    }
192    if (left instanceof ByteBufferExtendedCell) {
193      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
194          ((ByteBufferExtendedCell) left).getRowPosition(), left.getRowLength(),
195          right.getRowArray(), right.getRowOffset(), right.getRowLength());
196    }
197    if (right instanceof ByteBufferExtendedCell) {
198      // Notice how we flip the order of the compare here. We used to negate the return value but
199      // see what FindBugs says
200      // http://findbugs.sourceforge.net/bugDescriptions.html#RV_NEGATING_RESULT_OF_COMPARETO
201      // It suggest flipping the order to get same effect and 'safer'.
202      return ByteBufferUtils.compareTo(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
203          ((ByteBufferExtendedCell)right).getRowByteBuffer(),
204          ((ByteBufferExtendedCell)right).getRowPosition(), right.getRowLength());
205    }
206    return Bytes.compareTo(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
207        right.getRowArray(), right.getRowOffset(), right.getRowLength());
208  }

210  /**
211   * Compares the row part of the cell with a simple plain byte[] like the
212   * stopRow in Scan. This should be used with context where for hbase:meta
213   * cells the {{@link #META_COMPARATOR} should be used
214   *
215   * @param left
216   *          the cell to be compared
217   * @param right
218   *          the kv serialized byte[] to be compared with
219   * @param roffset
220   *          the offset in the byte[]
221   * @param rlength
222   *          the length in the byte[]
223   * @return 0 if both cell and the byte[] are equal, 1 if the cell is bigger
224   *         than byte[], -1 otherwise
225   */
226  @Override
227  public int compareRows(Cell left, byte[] right, int roffset, int rlength) {
228    if (left instanceof ByteBufferExtendedCell) {
229      return ByteBufferUtils.compareTo(((ByteBufferExtendedCell) left).getRowByteBuffer(),
230          ((ByteBufferExtendedCell) left).getRowPosition(), left.getRowLength(), right,
231          roffset, rlength);
232    }
233    return Bytes.compareTo(left.getRowArray(), left.getRowOffset(), left.getRowLength(), right,
234        roffset, rlength);
235  }
236
237  @Override
238  public final int compareWithoutRow(final Cell left, final Cell right) {
239    // If the column is not specified, the "minimum" key type appears the
240    // latest in the sorted order, regardless of the timestamp. This is used
241    // for specifying the last key/value in a given row, because there is no
242    // "lexicographically last column" (it would be infinitely long). The
243    // "maximum" key type does not need this behavior.
244    // Copied from KeyValue. This is bad in that we can't do memcmp w/ special rules like this.
245    int lFamLength = left.getFamilyLength();
246    int rFamLength = right.getFamilyLength();
247    int lQualLength = left.getQualifierLength();
248    int rQualLength = right.getQualifierLength();
249    if (lFamLength + lQualLength == 0
250          && left.getTypeByte() == Type.Minimum.getCode()) {
251      // left is "bigger", i.e. it appears later in the sorted order
252      return 1;
253    }
254    if (rFamLength + rQualLength == 0
255        && right.getTypeByte() == Type.Minimum.getCode()) {
256      return -1;
257    }
258    if (lFamLength != rFamLength) {
259      // comparing column family is enough.
260      return compareFamilies(left, right);
261    }
262    // Compare cf:qualifier
263    int diff = compareColumns(left, right);
264    if (diff != 0) return diff;
265
266    diff = compareTimestamps(left, right);
267    if (diff != 0) return diff;
268
269    // Compare types. Let the delete types sort ahead of puts; i.e. types
270    // of higher numbers sort before those of lesser numbers. Maximum (255)
271    // appears ahead of everything, and minimum (0) appears after
272    // everything.
273    return (0xff & right.getTypeByte()) - (0xff & left.getTypeByte());
274  }

276  /**
277   * Compares cell's timestamps in DESCENDING order.
278   * The below older timestamps sorting ahead of newer timestamps looks
279   * wrong but it is intentional. This way, newer timestamps are first
280   * found when we iterate over a memstore and newer versions are the
281   * first we trip over when reading from a store file.
282   * @return 1 if left's timestamp &lt; right's timestamp
283   *         -1 if left's timestamp &gt; right's timestamp
284   *         0 if both timestamps are equal
285   */
286  @Override
287  public int compareTimestamps(final Cell left, final Cell right) {
288    return compareTimestamps(left.getTimestamp(), right.getTimestamp());
289  }
290
291
292  /**
293   * Compares timestamps in DESCENDING order.
294   * The below older timestamps sorting ahead of newer timestamps looks
295   * wrong but it is intentional. This way, newer timestamps are first
296   * found when we iterate over a memstore and newer versions are the
297   * first we trip over when reading from a store file.
298   * @return 1 if left timestamp &lt; right timestamp
299   *         -1 if left timestamp &gt; right timestamp
300   *         0 if both timestamps are equal
301   */
302  @Override
303  public int compareTimestamps(final long ltimestamp, final long rtimestamp) {
304    if (ltimestamp < rtimestamp) {
305      return 1;
306    } else if (ltimestamp > rtimestamp) {
307      return -1;
308    }
309    return 0;
310  }

312  /**
313   * A {@link CellComparatorImpl} for <code>hbase:meta</code> catalog table
314   * {@link KeyValue}s.
315   */
316  public static class MetaCellComparator extends CellComparatorImpl {
317
318    @Override
319    public int compareRows(final Cell left, final Cell right) {
320      return compareRows(left.getRowArray(), left.getRowOffset(), left.getRowLength(),
321          right.getRowArray(), right.getRowOffset(), right.getRowLength());
322    }
323
324    @Override
325    public int compareRows(Cell left, byte[] right, int roffset, int rlength) {
326      return compareRows(left.getRowArray(), left.getRowOffset(), left.getRowLength(), right,
327          roffset, rlength);
328    }
329
330    private int compareRows(byte[] left, int loffset, int llength, byte[] right, int roffset,
331        int rlength) {
332      int leftDelimiter = Bytes.searchDelimiterIndex(left, loffset, llength, HConstants.DELIMITER);
333      int rightDelimiter = Bytes
334          .searchDelimiterIndex(right, roffset, rlength, HConstants.DELIMITER);
335      // Compare up to the delimiter
336      int lpart = (leftDelimiter < 0 ? llength : leftDelimiter - loffset);
337      int rpart = (rightDelimiter < 0 ? rlength : rightDelimiter - roffset);
338      int result = Bytes.compareTo(left, loffset, lpart, right, roffset, rpart);
339      if (result != 0) {
340        return result;
341      } else {
342        if (leftDelimiter < 0 && rightDelimiter >= 0) {
343          return -1;
344        } else if (rightDelimiter < 0 && leftDelimiter >= 0) {
345          return 1;
346        } else if (leftDelimiter < 0 && rightDelimiter < 0) {
347          return 0;
348        }
349      }
350      // Compare middle bit of the row.
351      // Move past delimiter
352      leftDelimiter++;
353      rightDelimiter++;
354      int leftFarDelimiter = Bytes.searchDelimiterIndexInReverse(left, leftDelimiter, llength
355          - (leftDelimiter - loffset), HConstants.DELIMITER);
356      int rightFarDelimiter = Bytes.searchDelimiterIndexInReverse(right, rightDelimiter, rlength
357          - (rightDelimiter - roffset), HConstants.DELIMITER);
358      // Now compare middlesection of row.
359      lpart = (leftFarDelimiter < 0 ? llength + loffset : leftFarDelimiter) - leftDelimiter;
360      rpart = (rightFarDelimiter < 0 ? rlength + roffset : rightFarDelimiter) - rightDelimiter;
361      result = Bytes.compareTo(left, leftDelimiter, lpart, right, rightDelimiter, rpart);
362      if (result != 0) {
363        return result;
364      } else {
365        if (leftDelimiter < 0 && rightDelimiter >= 0) {
366          return -1;
367        } else if (rightDelimiter < 0 && leftDelimiter >= 0) {
368          return 1;
369        } else if (leftDelimiter < 0 && rightDelimiter < 0) {
370          return 0;
371        }
372      }
373      // Compare last part of row, the rowid.
374      leftFarDelimiter++;
375      rightFarDelimiter++;
376      result = Bytes.compareTo(left, leftFarDelimiter, llength - (leftFarDelimiter - loffset),
377          right, rightFarDelimiter, rlength - (rightFarDelimiter - roffset));
378      return result;
379    }
380  }
