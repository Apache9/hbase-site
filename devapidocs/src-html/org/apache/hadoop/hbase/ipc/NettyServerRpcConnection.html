048/**
049 * RpcConnection implementation for netty rpc server.
050 */
051@InterfaceAudience.Private
052class NettyServerRpcConnection extends ServerRpcConnection {
053
054  final Channel channel;
055
056  NettyServerRpcConnection(NettyRpcServer rpcServer, Channel channel) {
057    super(rpcServer);
058    this.channel = channel;
059    InetSocketAddress inetSocketAddress = ((InetSocketAddress) channel.remoteAddress());
060    this.addr = inetSocketAddress.getAddress();
061    if (addr == null) {
062      this.hostAddress = "*Unknown*";
063    } else {
064      this.hostAddress = inetSocketAddress.getAddress().getHostAddress();
065    }
066    this.remotePort = inetSocketAddress.getPort();
067    this.saslCall = new NettyServerCall(SASL_CALLID, null, null, null, null, null, this, 0, null,
068        null, System.currentTimeMillis(), 0, rpcServer.reservoir, rpcServer.cellBlockBuilder, null);
069    this.setConnectionHeaderResponseCall = new NettyServerCall(CONNECTION_HEADER_RESPONSE_CALLID,
070        null, null, null, null, null, this, 0, null, null, System.currentTimeMillis(), 0,
071        rpcServer.reservoir, rpcServer.cellBlockBuilder, null);
072    this.authFailedCall = new NettyServerCall(AUTHORIZATION_FAILED_CALLID, null, null, null, null,
073        null, this, 0, null, null, System.currentTimeMillis(), 0, rpcServer.reservoir,
074        rpcServer.cellBlockBuilder, null);
075  }
076
077  void readPreamble(ByteBuf buffer) throws IOException {
078    byte[] rpcHead = { buffer.readByte(), buffer.readByte(), buffer.readByte(), buffer.readByte() };
079    if (!Arrays.equals(HConstants.RPC_HEADER, rpcHead)) {
080      doBadPreambleHandling("Expected HEADER=" + Bytes.toStringBinary(HConstants.RPC_HEADER) +
081          " but received HEADER=" + Bytes.toStringBinary(rpcHead) + " from " + toString());
082      return;
083    }
084    // Now read the next two bytes, the version and the auth to use.
085    int version = buffer.readByte();
086    byte authbyte = buffer.readByte();
087    this.authMethod = AuthMethod.valueOf(authbyte);
088    if (version != NettyRpcServer.CURRENT_VERSION) {
089      String msg = getFatalConnectionString(version, authbyte);
090      doBadPreambleHandling(msg, new WrongVersionException(msg));
091      return;
092    }
093    if (authMethod == null) {
094      String msg = getFatalConnectionString(version, authbyte);
095      doBadPreambleHandling(msg, new BadAuthException(msg));
096      return;
097    }
098    if (this.rpcServer.isSecurityEnabled && authMethod == AuthMethod.SIMPLE) {
099      if (this.rpcServer.allowFallbackToSimpleAuth) {
100        this.rpcServer.metrics.authenticationFallback();
101        authenticatedWithFallback = true;
102      } else {
103        AccessDeniedException ae = new AccessDeniedException("Authentication is required");
104        this.rpcServer.setupResponse(authFailedResponse, authFailedCall, ae, ae.getMessage());
105        ((NettyServerCall) authFailedCall).sendResponseIfReady(ChannelFutureListener.CLOSE);
106        return;
107      }
108    }
109    if (!this.rpcServer.isSecurityEnabled && authMethod != AuthMethod.SIMPLE) {
110      doRawSaslReply(SaslStatus.SUCCESS, new IntWritable(SaslUtil.SWITCH_TO_SIMPLE_AUTH), null,
111        null);
112      authMethod = AuthMethod.SIMPLE;
113      // client has already sent the initial Sasl message and we
114      // should ignore it. Both client and server should fall back
115      // to simple auth from now on.
116      skipInitialSaslHandshake = true;
117    }
118    if (authMethod != AuthMethod.SIMPLE) {
119      useSasl = true;
120    }
121    connectionPreambleRead = true;
122  }
123
124  private void doBadPreambleHandling(final String msg) throws IOException {
125    doBadPreambleHandling(msg, new FatalConnectionException(msg));
126  }
127
128  private void doBadPreambleHandling(final String msg, final Exception e) throws IOException {
129    NettyRpcServer.LOG.warn(msg);
130    NettyServerCall fakeCall = new NettyServerCall(-1, null, null, null, null, null, this, -1, null,
131        null, System.currentTimeMillis(), 0, this.rpcServer.reservoir,
132        this.rpcServer.cellBlockBuilder, null);
133    this.rpcServer.setupResponse(null, fakeCall, e, msg);
134    // closes out the connection.
135    fakeCall.sendResponseIfReady(ChannelFutureListener.CLOSE);
136  }
137
138  void process(final ByteBuf buf) throws IOException, InterruptedException {
139    if (connectionHeaderRead) {
140      this.callCleanup = new RpcServer.CallCleanup() {
141        @Override
142        public void run() {
143          buf.release();
144        }
145      };
146      process(new SingleByteBuff(buf.nioBuffer()));
147    } else {
148      byte[] data = new byte[buf.readableBytes()];
149      buf.readBytes(data, 0, data.length);
150      ByteBuffer connectionHeader = ByteBuffer.wrap(data);
151      buf.release();
152      process(connectionHeader);
153    }
154  }
155
156  void process(ByteBuffer buf) throws IOException, InterruptedException {
157    process(new SingleByteBuff(buf));
158  }
159
160  void process(ByteBuff buf) throws IOException, InterruptedException {
161    try {
162      if (skipInitialSaslHandshake) {
163        skipInitialSaslHandshake = false;
164        if (callCleanup != null) {
165          callCleanup.run();
166        }
167        return;
168      }
169
170      if (useSasl) {
171        saslReadAndProcess(buf);
172      } else {
173        processOneRpc(buf);
174      }
175    } catch (Exception e) {
176      if (callCleanup != null) {
177        callCleanup.run();
178      }
179      throw e;
180    } finally {
181      this.callCleanup = null;
182    }
183  }
184
185  @Override
186  public synchronized void close() {
187    disposeSasl();
188    channel.close();
189    callCleanup = null;
190  }
191
192  @Override
193  public boolean isConnectionOpen() {
194    return channel.isOpen();
195  }
196
197  @Override
198  public NettyServerCall createCall(int id, final BlockingService service,
199      final MethodDescriptor md, RequestHeader header, Message param, CellScanner cellScanner,
200      long size, TraceInfo tinfo, final InetAddress remoteAddress, int timeout,
201      CallCleanup reqCleanup) {
202    return new NettyServerCall(id, service, md, header, param, cellScanner, this, size, tinfo,
203        remoteAddress, System.currentTimeMillis(), timeout, this.rpcServer.reservoir,
204        this.rpcServer.cellBlockBuilder, reqCleanup);
205  }
