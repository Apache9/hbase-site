018package org.apache.hadoop.hbase.ipc;
019
020import java.io.IOException;
021
022import org.apache.hadoop.hbase.CellScanner;
023import org.apache.hadoop.hbase.classification.InterfaceAudience;
024import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
025import org.apache.hadoop.hbase.protobuf.generated.RPCProtos;
026import org.apache.hadoop.ipc.RemoteException;
027
028import com.google.protobuf.Message;
029
030import io.netty.buffer.ByteBuf;
031import io.netty.buffer.ByteBufInputStream;
032import io.netty.channel.ChannelHandlerContext;
033import io.netty.channel.SimpleChannelInboundHandler;
034
035/**
036 * Handles Hbase responses
037 */
038@InterfaceAudience.Private
039public class AsyncServerResponseHandler extends SimpleChannelInboundHandler<ByteBuf> {
040  private final AsyncRpcChannel channel;
041
042  /**
043   * Constructor
044   * @param channel on which this response handler operates
045   */
046  public AsyncServerResponseHandler(AsyncRpcChannel channel) {
047    this.channel = channel;
048  }
049
050  @Override
051  protected void channelRead0(ChannelHandlerContext ctx, ByteBuf inBuffer) throws Exception {
052    ByteBufInputStream in = new ByteBufInputStream(inBuffer);
053    int totalSize = inBuffer.readableBytes();
054    // Read the header
055    RPCProtos.ResponseHeader responseHeader = RPCProtos.ResponseHeader.parseDelimitedFrom(in);
056    int id = responseHeader.getCallId();
057    AsyncCall call = channel.removePendingCall(id);
058    if (call == null) {
059      // So we got a response for which we have no corresponding 'call' here on the client-side.
060      // We probably timed out waiting, cleaned up all references, and now the server decides
061      // to return a response. There is nothing we can do w/ the response at this stage. Clean
062      // out the wire of the response so its out of the way and we can get other responses on
063      // this connection.
064      int readSoFar = IPCUtil.getTotalSizeWhenWrittenDelimited(responseHeader);
065      int whatIsLeftToRead = totalSize - readSoFar;
066
067      // This is done through a Netty ByteBuf which has different behavior than InputStream.
068      // It does not return number of bytes read but will update pointer internally and throws an
069      // exception when too many bytes are to be skipped.
070      inBuffer.skipBytes(whatIsLeftToRead);
071      return;
072    }
073
074    if (responseHeader.hasException()) {
075      RPCProtos.ExceptionResponse exceptionResponse = responseHeader.getException();
076      RemoteException re = createRemoteException(exceptionResponse);
077      if (exceptionResponse.getExceptionClassName()
078          .equals(FatalConnectionException.class.getName())) {
079        channel.close(re);
080      } else {
081        call.setFailed(re);
082      }
083    } else {
084      Message value = null;
085      // Call may be null because it may have timedout and been cleaned up on this side already
086      if (call.responseDefaultType != null) {
087        Message.Builder builder = call.responseDefaultType.newBuilderForType();
088        ProtobufUtil.mergeDelimitedFrom(builder, in);
089        value = builder.build();
090      }
091      CellScanner cellBlockScanner = null;
092      if (responseHeader.hasCellBlockMeta()) {
093        int size = responseHeader.getCellBlockMeta().getLength();
094        byte[] cellBlock = new byte[size];
095        inBuffer.readBytes(cellBlock, 0, cellBlock.length);
096        cellBlockScanner = channel.client.createCellScanner(cellBlock);
097      }
098      call.setSuccess(value, cellBlockScanner);
099      call.callStats.setResponseSizeBytes(totalSize);
100    }
101  }
102
103  @Override
104  public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
105    channel.close(cause);
106  }
107
108  @Override
109  public void channelInactive(ChannelHandlerContext ctx) throws Exception {
110    channel.close(new IOException("connection closed"));
111  }
112
113  /**
114   * @param e Proto exception
115   * @return RemoteException made from passed <code>e</code>
116   */
117  private RemoteException createRemoteException(final RPCProtos.ExceptionResponse e) {
118    String innerExceptionClassName = e.getExceptionClassName();
119    boolean doNotRetry = e.getDoNotRetry();
120    return e.hasHostname() ?
121        // If a hostname then add it to the RemoteWithExtrasException
122        new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), e.getHostname(),
123            e.getPort(), doNotRetry)
124        : new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), doNotRetry);
125  }
126}
