018package org.apache.hadoop.hbase.ipc;
019
020import com.google.protobuf.Descriptors;
021import com.google.protobuf.Message;
022import com.google.protobuf.RpcController;
023
024import java.io.IOException;
025
026import org.apache.commons.logging.Log;
027import org.apache.commons.logging.LogFactory;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030import org.apache.hadoop.hbase.client.ClusterConnection;
031import org.apache.hadoop.hbase.client.RegionServerCallable;
032import org.apache.hadoop.hbase.client.RpcRetryingCallerFactory;
033import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
034import org.apache.hadoop.hbase.protobuf.generated.ClientProtos;
035import org.apache.hadoop.hbase.protobuf.generated.ClientProtos.CoprocessorServiceResponse;
036import org.apache.hadoop.hbase.util.Bytes;
037
038/**
039 * Provides clients with an RPC connection to call Coprocessor Endpoint
040 * {@link com.google.protobuf.Service}s
041 * against a given table region.  An instance of this class may be obtained
042 * by calling {@link org.apache.hadoop.hbase.client.Table#coprocessorService(byte[])},
043 * but should normally only be used in creating a new {@link com.google.protobuf.Service} stub to
044 * call the endpoint methods.
045 * @see org.apache.hadoop.hbase.client.Table#coprocessorService(byte[])
046 */
047@InterfaceAudience.Private
048public class RegionCoprocessorRpcChannel extends SyncCoprocessorRpcChannel {
049  private static final Log LOG = LogFactory.getLog(RegionCoprocessorRpcChannel.class);
050
051  private final ClusterConnection connection;
052  private final TableName table;
053  private final byte[] row;
054  private byte[] lastRegion;
055  private final int operationTimeout;
056
057  private final RpcRetryingCallerFactory rpcCallerFactory;
058  private final RpcControllerFactory rpcControllerFactory;
059
060  /**
061   * Constructor
062   * @param conn connection to use
063   * @param table to connect to
064   * @param row to locate region with
065   */
066  public RegionCoprocessorRpcChannel(ClusterConnection conn, TableName table, byte[] row) {
067    this.connection = conn;
068    this.table = table;
069    this.row = row;
070    this.rpcCallerFactory = conn.getRpcRetryingCallerFactory();
071    this.rpcControllerFactory = conn.getRpcControllerFactory();
072    this.operationTimeout = conn.getConnectionConfiguration().getOperationTimeout();
073  }
074
075  @Override
076  protected Message callExecService(RpcController controller,
077      Descriptors.MethodDescriptor method, Message request, Message responsePrototype)
078          throws IOException {
079    if (LOG.isTraceEnabled()) {
080      LOG.trace("Call: " + method.getName() + ", " + request.toString());
081    }
082    if (row == null) {
083      throw new IllegalArgumentException("Missing row property for remote region location");
084    }
085    final ClientProtos.CoprocessorServiceCall call =
086        CoprocessorRpcUtils.buildServiceCall(row, method, request);
087    RegionServerCallable<CoprocessorServiceResponse> callable =
088        new RegionServerCallable<CoprocessorServiceResponse>(connection,
089          controller == null? this.rpcControllerFactory.newController(): controller,
090          table, row) {
091      @Override
092      protected CoprocessorServiceResponse rpcCall() throws Exception {
093        byte[] regionName = getLocation().getRegionInfo().getRegionName();
094        return ProtobufUtil.execService(getRpcController(), getStub(), call, regionName);
095      }
096    };
097    CoprocessorServiceResponse result = rpcCallerFactory.<CoprocessorServiceResponse> newCaller()
098        .callWithRetries(callable, operationTimeout);
099    Message response;
100    if (result.getValue().hasValue()) {
101      Message.Builder builder = responsePrototype.newBuilderForType();
102      ProtobufUtil.mergeFrom(builder, result.getValue().getValue());
103      response = builder.build();
104    } else {
105      response = responsePrototype.getDefaultInstanceForType();
106    }
107    lastRegion = result.getRegion().getValue().toByteArray();
108    if (LOG.isTraceEnabled()) {
109      LOG.trace("Result is region=" + Bytes.toStringBinary(lastRegion) + ", value=" + response);
110    }
111    return response;
112  }
113
114  /**
115   * Get last region this RpcChannel communicated with
116   * @return region name as byte array
117   */
118  public byte[] getLastRegion() {
119    return lastRegion;
120  }
121}
