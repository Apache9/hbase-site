406    @Override
407    public synchronized void setResponse(Message m, final CellScanner cells,
408        Throwable t, String errorMsg) {
409      if (this.isError) return;
410      if (t != null) this.isError = true;
411      BufferChain bc = null;
412      try {
413        ResponseHeader.Builder headerBuilder = ResponseHeader.newBuilder();
414        // Call id.
415        headerBuilder.setCallId(this.id);
416        if (t != null) {
417          setExceptionResponse(t, errorMsg, headerBuilder);
418        }
419        // Pass reservoir to buildCellBlock. Keep reference to returne so can add it back to the
420        // reservoir when finished. This is hacky and the hack is not contained but benefits are
421        // high when we can avoid a big buffer allocation on each rpc.
422        List<ByteBuffer> cellBlock = null;
423        int cellBlockSize = 0;
424        if (reservoir != null) {
425          this.cellBlockStream = cellBlockBuilder.buildCellBlockStream(this.connection.codec,
426              this.connection.compressionCodec, cells, reservoir);
427          if (this.cellBlockStream != null) {
428            cellBlock = this.cellBlockStream.getByteBuffers();
429            cellBlockSize = this.cellBlockStream.size();
430          }
431        } else {
432          ByteBuffer b = cellBlockBuilder.buildCellBlock(this.connection.codec,
433              this.connection.compressionCodec, cells);
434          if (b != null) {
435            cellBlockSize = b.remaining();
436            cellBlock = new ArrayList<>(1);
437            cellBlock.add(b);
438          }
439        }
440
441        if (cellBlockSize > 0) {
442          CellBlockMeta.Builder cellBlockBuilder = CellBlockMeta.newBuilder();
443          // Presumes the cellBlock bytebuffer has been flipped so limit has total size in it.
444          cellBlockBuilder.setLength(cellBlockSize);
445          headerBuilder.setCellBlockMeta(cellBlockBuilder.build());
446        }
447        Message header = headerBuilder.build();
448        ByteBuffer headerBuf =
449            createHeaderAndMessageBytes(m, header, cellBlockSize, cellBlock);
450        ByteBuffer[] responseBufs = null;
451        int cellBlockBufferSize = 0;
452        if (cellBlock != null) {
453          cellBlockBufferSize = cellBlock.size();
454          responseBufs = new ByteBuffer[1 + cellBlockBufferSize];
455        } else {
456          responseBufs = new ByteBuffer[1];
457        }
458        responseBufs[0] = headerBuf;
459        if (cellBlock != null) {
460          for (int i = 0; i < cellBlockBufferSize; i++) {
461            responseBufs[i + 1] = cellBlock.get(i);
462          }
463        }
464        bc = new BufferChain(responseBufs);
465        if (connection.useWrap) {
466          bc = wrapWithSasl(bc);
467        }
468      } catch (IOException e) {
469        LOG.warn("Exception while creating response " + e);
470      }
471      this.response = bc;
472      // Once a response message is created and set to this.response, this Call can be treated as
473      // done. The Responder thread will do the n/w write of this message back to client.
474      if (this.rpcCallback != null) {
475        try {
476          this.rpcCallback.run();
477        } catch (Exception e) {
478          // Don't allow any exception here to kill this handler thread.
479          LOG.warn("Exception while running the Rpc Callback.", e);
480        }
481      }
482    }

502    protected ByteBuffer createHeaderAndMessageBytes(Message result, Message header,
503        int cellBlockSize, List<ByteBuffer> cellBlock) throws IOException {
504      // Organize the response as a set of bytebuffers rather than collect it all together inside
505      // one big byte array; save on allocations.
506      // for writing the header, we check if there is available space in the buffers
507      // created for the cellblock itself. If there is space for the header, we reuse
508      // the last buffer in the cellblock. This applies to the cellblock created from the
509      // pool or even the onheap cellblock buffer in case there is no pool enabled.
510      // Possible reuse would avoid creating a temporary array for storing the header every time.
511      ByteBuffer possiblePBBuf =
512          (cellBlockSize > 0) ? cellBlock.get(cellBlock.size() - 1) : null;
513      int headerSerializedSize = 0, resultSerializedSize = 0, headerVintSize = 0,
514          resultVintSize = 0;
515      if (header != null) {
516        headerSerializedSize = header.getSerializedSize();
517        headerVintSize = CodedOutputStream.computeRawVarint32Size(headerSerializedSize);
518      }
519      if (result != null) {
520        resultSerializedSize = result.getSerializedSize();
521        resultVintSize = CodedOutputStream.computeRawVarint32Size(resultSerializedSize);
522      }
523      // calculate the total size
524      int totalSize = headerSerializedSize + headerVintSize
525          + (resultSerializedSize + resultVintSize)
526          + cellBlockSize;
527      int totalPBSize = headerSerializedSize + headerVintSize + resultSerializedSize
528          + resultVintSize + Bytes.SIZEOF_INT;
529      // Only if the last buffer has enough space for header use it. Else allocate
530      // a new buffer. Assume they are all flipped
531      if (possiblePBBuf != null
532          && possiblePBBuf.limit() + totalPBSize <= possiblePBBuf.capacity()) {
533        // duplicate the buffer. This is where the header is going to be written
534        ByteBuffer pbBuf = possiblePBBuf.duplicate();
535        // get the current limit
536        int limit = pbBuf.limit();
537        // Position such that we write the header to the end of the buffer
538        pbBuf.position(limit);
539        // limit to the header size
540        pbBuf.limit(totalPBSize + limit);
541        // mark the current position
542        pbBuf.mark();
543        writeToCOS(result, header, totalSize, pbBuf);
544        // reset the buffer back to old position
545        pbBuf.reset();
546        return pbBuf;
547      } else {
548        return createHeaderAndMessageBytes(result, header, totalSize, totalPBSize);
549      }
550    }
551
552    private void writeToCOS(Message result, Message header, int totalSize, ByteBuffer pbBuf)
553        throws IOException {
554      ByteBufferUtils.putInt(pbBuf, totalSize);
555      // create COS that works on BB
556      CodedOutputStream cos = CodedOutputStream.newInstance(pbBuf);
557      if (header != null) {
558        cos.writeMessageNoTag(header);
559      }
560      if (result != null) {
561        cos.writeMessageNoTag(result);
562      }
563      cos.flush();
564      cos.checkNoSpaceLeft();
565    }

567    private ByteBuffer createHeaderAndMessageBytes(Message result, Message header,
568        int totalSize, int totalPBSize) throws IOException {
569      ByteBuffer pbBuf = ByteBuffer.allocate(totalPBSize);
570      writeToCOS(result, header, totalSize, pbBuf);
571      pbBuf.flip();
572      return pbBuf;

575    protected BufferChain wrapWithSasl(BufferChain bc)
576        throws IOException {
577      if (!this.connection.useSasl) return bc;
578      // Looks like no way around this; saslserver wants a byte array.  I have to make it one.
579      // THIS IS A BIG UGLY COPY.
580      byte [] responseBytes = bc.getBytes();
581      byte [] token;
582      // synchronization may be needed since there can be multiple Handler
583      // threads using saslServer or Crypto AES to wrap responses.
584      if (connection.useCryptoAesWrap) {
585        // wrap with Crypto AES
586        synchronized (connection.cryptoAES) {
587          token = connection.cryptoAES.wrap(responseBytes, 0, responseBytes.length);
588        }
589      } else {
590        synchronized (connection.saslServer) {
591          token = connection.saslServer.wrap(responseBytes, 0, responseBytes.length);
592        }

885    protected void setupCryptoCipher(final ConnectionHeader header,
886        RPCProtos.ConnectionHeaderResponse.Builder chrBuilder)
887        throws FatalConnectionException {
888      // If simple auth, return
889      if (saslServer == null) return;
890      // check if rpc encryption with Crypto AES
891      String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
892      boolean isEncryption = SaslUtil.QualityOfProtection.PRIVACY
893          .getSaslQop().equalsIgnoreCase(qop);
894      boolean isCryptoAesEncryption = isEncryption && conf.getBoolean(
895          "hbase.rpc.crypto.encryption.aes.enabled", false);
896      if (!isCryptoAesEncryption) return;
897      if (!header.hasRpcCryptoCipherTransformation()) return;
898      String transformation = header.getRpcCryptoCipherTransformation();
899      if (transformation == null || transformation.length() == 0) return;
900       // Negotiates AES based on complete saslServer.
901       // The Crypto metadata need to be encrypted and send to client.
902      Properties properties = new Properties();
903      // the property for SecureRandomFactory
904      properties.setProperty(CryptoRandomFactory.CLASSES_KEY,
905          conf.get("hbase.crypto.sasl.encryption.aes.crypto.random",
906              "org.apache.commons.crypto.random.JavaCryptoRandom"));
907      // the property for cipher class
908      properties.setProperty(CryptoCipherFactory.CLASSES_KEY,
909          conf.get("hbase.rpc.crypto.encryption.aes.cipher.class",
910              "org.apache.commons.crypto.cipher.JceCipher"));
911
912      int cipherKeyBits = conf.getInt(
913          "hbase.rpc.crypto.encryption.aes.cipher.keySizeBits", 128);
914      // generate key and iv
915      if (cipherKeyBits % 8 != 0) {
916        throw new IllegalArgumentException("The AES cipher key size in bits" +
917            " should be a multiple of byte");
918      }
919      int len = cipherKeyBits / 8;
920      byte[] inKey = new byte[len];
921      byte[] outKey = new byte[len];
922      byte[] inIv = new byte[len];
923      byte[] outIv = new byte[len];
924
925      try {
926        // generate the cipher meta data with SecureRandom
927        CryptoRandom secureRandom = CryptoRandomFactory.getCryptoRandom(properties);
928        secureRandom.nextBytes(inKey);
929        secureRandom.nextBytes(outKey);
930        secureRandom.nextBytes(inIv);
931        secureRandom.nextBytes(outIv);
932
933        // create CryptoAES for server
934        cryptoAES = new CryptoAES(transformation, properties,
935            inKey, outKey, inIv, outIv);
936        // create SaslCipherMeta and send to client,
937        //  for client, the [inKey, outKey], [inIv, outIv] should be reversed
938        RPCProtos.CryptoCipherMeta.Builder ccmBuilder = RPCProtos.CryptoCipherMeta.newBuilder();
939        ccmBuilder.setTransformation(transformation);
940        ccmBuilder.setInIv(getByteString(outIv));
941        ccmBuilder.setInKey(getByteString(outKey));
942        ccmBuilder.setOutIv(getByteString(inIv));
943        ccmBuilder.setOutKey(getByteString(inKey));
944        chrBuilder.setCryptoCipherMeta(ccmBuilder);
945        useCryptoAesWrap = true;
946      } catch (GeneralSecurityException | IOException ex) {
947        throw new UnsupportedCryptoException(ex.getMessage(), ex);
948      }
949    }

1014        saslCall.sendResponseIfReady();
1015      } finally {
1016        if (saslResponse != null) {
1017          saslResponse.close();
1018        }
1019        if (out != null) {
1020          out.close();
1021        }
1022      }

1025    public void saslReadAndProcess(ByteBuff saslToken) throws IOException,
1026        InterruptedException {
1027      if (saslContextEstablished) {
1028        if (LOG.isTraceEnabled())
1029          LOG.trace("Have read input token of size " + saslToken.limit()
1030              + " for processing by saslServer.unwrap()");
1031
1032        if (!useWrap) {
1033          processOneRpc(saslToken);
1034        } else {
1035          byte[] b = saslToken.hasArray() ? saslToken.array() : saslToken.toBytes();
1036          byte [] plaintextData;
1037          if (useCryptoAesWrap) {
1038            // unwrap with CryptoAES
1039            plaintextData = cryptoAES.unwrap(b, 0, b.length);
1040          } else {
1041            plaintextData = saslServer.unwrap(b, 0, b.length);
1042          }
1043          processUnwrappedData(plaintextData);
1044        }
1045      } else {
1046        byte[] replyToken;
1047        try {
1048          if (saslServer == null) {
1049            switch (authMethod) {
1050            case DIGEST:
1051              if (secretManager == null) {
1052                throw new AccessDeniedException(
1053                    "Server is not configured to do DIGEST authentication.");
1054              }
1055              saslServer = Sasl.createSaslServer(AuthMethod.DIGEST
1056                  .getMechanismName(), null, SaslUtil.SASL_DEFAULT_REALM,
1057                  HBaseSaslRpcServer.getSaslProps(), new SaslDigestCallbackHandler(
1058                      secretManager, this));
1059              break;
1060            default:
1061              UserGroupInformation current = UserGroupInformation.getCurrentUser();
1062              String fullName = current.getUserName();
1063              if (LOG.isDebugEnabled()) {
1064                LOG.debug("Kerberos principal name is " + fullName);
1065              }
1066              final String names[] = SaslUtil.splitKerberosName(fullName);
1067              if (names.length != 3) {
1068                throw new AccessDeniedException(
1069                    "Kerberos principal name does NOT have the expected "
1070                        + "hostname part: " + fullName);
1071              }
1072              current.doAs(new PrivilegedExceptionAction<Object>() {
1073                @Override
1074                public Object run() throws SaslException {
1075                  saslServer = Sasl.createSaslServer(AuthMethod.KERBEROS
1076                      .getMechanismName(), names[0], names[1],
1077                      HBaseSaslRpcServer.getSaslProps(), new SaslGssCallbackHandler());
1078                  return null;
1079                }
1080              });
1081            }
1082            if (saslServer == null)
1083              throw new AccessDeniedException(
1084                  "Unable to find SASL server implementation for "
1085                      + authMethod.getMechanismName());
1086            if (LOG.isDebugEnabled()) {
1087              LOG.debug("Created SASL server with mechanism = " + authMethod.getMechanismName());
1088            }
1089          }
1090          if (LOG.isDebugEnabled()) {
1091            LOG.debug("Have read input token of size " + saslToken.limit()
1092                + " for processing by saslServer.evaluateResponse()");
1093          }
1094          replyToken = saslServer
1095              .evaluateResponse(saslToken.hasArray() ? saslToken.array() : saslToken.toBytes());
1096        } catch (IOException e) {
1097          IOException sendToClient = e;
1098          Throwable cause = e;
1099          while (cause != null) {
1100            if (cause instanceof InvalidToken) {
1101              sendToClient = (InvalidToken) cause;
1102              break;
1103            }
1104            cause = cause.getCause();
1105          }
1106          doRawSaslReply(SaslStatus.ERROR, null, sendToClient.getClass().getName(),
1107            sendToClient.getLocalizedMessage());
1108          metrics.authenticationFailure();
1109          String clientIP = this.toString();
1110          // attempting user could be null
1111          AUDITLOG.warn(AUTH_FAILED_FOR + clientIP + ":" + attemptingUser);
1112          throw e;
1113        }
1114        if (replyToken != null) {
1115          if (LOG.isDebugEnabled()) {
1116            LOG.debug("Will send token of size " + replyToken.length
1117                + " from saslServer.");
1118          }
1119          doRawSaslReply(SaslStatus.SUCCESS, new BytesWritable(replyToken), null,
1120              null);
1121        }
1122        if (saslServer.isComplete()) {
1123          String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
1124          useWrap = qop != null && !"auth".equalsIgnoreCase(qop);
1125          ugi = getAuthorizedUgi(saslServer.getAuthorizationID());
1126          if (LOG.isDebugEnabled()) {
1127            LOG.debug("SASL server context established. Authenticated client: "
1128              + ugi + ". Negotiated QoP is "
1129              + saslServer.getNegotiatedProperty(Sasl.QOP));
1130          }
1131          metrics.authenticationSuccess();
1132          AUDITLOG.info(AUTH_SUCCESSFUL_FOR + ugi);
1133          saslContextEstablished = true;
1134        }
1135      }
1136    }

1138    private void processUnwrappedData(byte[] inBuf) throws IOException,
1139    InterruptedException {
1140      ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(inBuf));
1141      // Read all RPCs contained in the inBuf, even partial ones
1142      while (true) {
1143        int count;
1144        if (unwrappedDataLengthBuffer.remaining() > 0) {
1145          count = channelRead(ch, unwrappedDataLengthBuffer);
1146          if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0)
1147            return;
1148        }
1149
1150        if (unwrappedData == null) {
1151          unwrappedDataLengthBuffer.flip();
1152          int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
1153
1154          if (unwrappedDataLength == RpcClient.PING_CALL_ID) {
1155            if (LOG.isDebugEnabled())
1156              LOG.debug("Received ping message");
1157            unwrappedDataLengthBuffer.clear();
1158            continue; // ping message
1159          }
1160          unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
1161        }
1162
1163        count = channelRead(ch, unwrappedData);
1164        if (count <= 0 || unwrappedData.remaining() > 0)
1165          return;
1166
1167        if (unwrappedData.remaining() == 0) {
1168          unwrappedDataLengthBuffer.clear();
1169          unwrappedData.flip();
1170          processOneRpc(new SingleByteBuff(unwrappedData));
1171          unwrappedData = null;
1172        }
1173      }
1174    }

1193    protected boolean authorizeConnection() throws IOException {
1194      try {
1195        // If auth method is DIGEST, the token was obtained by the
1196        // real user for the effective user, therefore not required to
1197        // authorize real user. doAs is allowed only for simple or kerberos
1198        // authentication
1199        if (ugi != null && ugi.getRealUser() != null
1200            && (authMethod != AuthMethod.DIGEST)) {
1201          ProxyUsers.authorize(ugi, this.getHostAddress(), conf);
1202        }
1203        authorize(ugi, connectionHeader, getHostInetAddress());
1204        metrics.authorizationSuccess();
1205      } catch (AuthorizationException ae) {
1206        if (LOG.isDebugEnabled()) {
1207          LOG.debug("Connection authorization failed: " + ae.getMessage(), ae);
1208        }
1209        metrics.authorizationFailure();
1210        setupResponse(authFailedResponse, authFailedCall,
1211          new AccessDeniedException(ae), ae.getMessage());
1212        authFailedCall.sendResponseIfReady();
1213        return false;
1214      }
1215      return true;
1216    }

1218    // Reads the connection header following version
1219    protected void processConnectionHeader(ByteBuff buf) throws IOException {
1220      if (buf.hasArray()) {
1221        this.connectionHeader = ConnectionHeader.parseFrom(buf.array());
1222      } else {
1223        CodedInputStream cis = UnsafeByteOperations
1224            .unsafeWrap(new ByteBuffByteInput(buf, 0, buf.limit()), 0, buf.limit()).newCodedInput();
1225        cis.enableAliasing(true);
1226        this.connectionHeader = ConnectionHeader.parseFrom(cis);
1227      }
1228      String serviceName = connectionHeader.getServiceName();
1229      if (serviceName == null) throw new EmptyServiceNameException();
1230      this.service = getService(services, serviceName);
1231      if (this.service == null) throw new UnknownServiceException(serviceName);
1232      setupCellBlockCodecs(this.connectionHeader);
1233      RPCProtos.ConnectionHeaderResponse.Builder chrBuilder =
1234          RPCProtos.ConnectionHeaderResponse.newBuilder();
1235      setupCryptoCipher(this.connectionHeader, chrBuilder);
1236      responseConnectionHeader(chrBuilder);
1237      UserGroupInformation protocolUser = createUser(connectionHeader);
1238      if (!useSasl) {
1239        ugi = protocolUser;
1240        if (ugi != null) {
1241          ugi.setAuthenticationMethod(AuthMethod.SIMPLE.authenticationMethod);
1242        }
1243        // audit logging for SASL authenticated users happens in saslReadAndProcess()
1244        if (authenticatedWithFallback) {
1245          LOG.warn("Allowed fallback to SIMPLE auth for " + ugi
1246              + " connecting from " + getHostAddress());
1247        }
1248        AUDITLOG.info(AUTH_SUCCESSFUL_FOR + ugi);
1249      } else {
1250        // user is authenticated
1251        ugi.setAuthenticationMethod(authMethod.authenticationMethod);
1252        //Now we check if this is a proxy user case. If the protocol user is
1253        //different from the 'user', it is a proxy user scenario. However,
1254        //this is not allowed if user authenticated with DIGEST.
1255        if ((protocolUser != null)
1256            && (!protocolUser.getUserName().equals(ugi.getUserName()))) {
1257          if (authMethod == AuthMethod.DIGEST) {
1258            // Not allowed to doAs if token authentication is used
1259            throw new AccessDeniedException("Authenticated user (" + ugi
1260                + ") doesn't match what the client claims to be ("
1261                + protocolUser + ")");
1262          } else {
1263            // Effective user can be different from authenticated user
1264            // for simple auth or kerberos auth
1265            // The user is the real user. Now we create a proxy user
1266            UserGroupInformation realUser = ugi;
1267            ugi = UserGroupInformation.createProxyUser(protocolUser
1268                .getUserName(), realUser);
1269            // Now the user is a proxy user, set Authentication method Proxy.
1270            ugi.setAuthenticationMethod(AuthenticationMethod.PROXY);
1271          }
1272        }
1273      }
1274      if (connectionHeader.hasVersionInfo()) {
1275        // see if this connection will support RetryImmediatelyException
1276        retryImmediatelySupported = VersionInfoUtil.hasMinimumVersion(getVersionInfo(), 1, 2);

1278        AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
1279            + " with version info: "
1280            + TextFormat.shortDebugString(connectionHeader.getVersionInfo()));
1281      } else {
1282        AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
1283            + " with unknown version info");
1284      }
1285    }

1287    private void responseConnectionHeader(RPCProtos.ConnectionHeaderResponse.Builder chrBuilder)
1288        throws FatalConnectionException {
1289      // Response the connection header if Crypto AES is enabled
1290      if (!chrBuilder.hasCryptoCipherMeta()) return;
1291      try {
1292        byte[] connectionHeaderResBytes = chrBuilder.build().toByteArray();
1293        // encrypt the Crypto AES cipher meta data with sasl server, and send to client
1294        byte[] unwrapped = new byte[connectionHeaderResBytes.length + 4];
1295        Bytes.putBytes(unwrapped, 0, Bytes.toBytes(connectionHeaderResBytes.length), 0, 4);
1296        Bytes.putBytes(unwrapped, 4, connectionHeaderResBytes, 0, connectionHeaderResBytes.length);
1297
1298        doConnectionHeaderResponse(saslServer.wrap(unwrapped, 0, unwrapped.length));
1299      } catch (IOException ex) {
1300        throw new UnsupportedCryptoException(ex.getMessage(), ex);
1301      }
1302    }

1330    /**
1331     * @param buf
1332     *          Has the request header and the request param and optionally
1333     *          encoded data buffer all in this one array.
1334     * @throws IOException
1335     * @throws InterruptedException
1336     */
1337    protected void processRequest(ByteBuff buf) throws IOException,
1338        InterruptedException {
1339      long totalRequestSize = buf.limit();
1340      int offset = 0;
1341      // Here we read in the header. We avoid having pb
1342      // do its default 4k allocation for CodedInputStream. We force it to use
1343      // backing array.
1344      CodedInputStream cis;
1345      if (buf.hasArray()) {
1346        cis = UnsafeByteOperations.unsafeWrap(buf.array(), 0, buf.limit())
1347            .newCodedInput();
1348      } else {
1349        cis = UnsafeByteOperations.unsafeWrap(
1350            new ByteBuffByteInput(buf, 0, buf.limit()), 0, buf.limit())
1351            .newCodedInput();
1352      }
1353      cis.enableAliasing(true);
1354      int headerSize = cis.readRawVarint32();
1355      offset = cis.getTotalBytesRead();
1356      Message.Builder builder = RequestHeader.newBuilder();
1357      ProtobufUtil.mergeFrom(builder, cis, headerSize);
1358      RequestHeader header = (RequestHeader) builder.build();
1359      offset += headerSize;
1360      int id = header.getCallId();
1361      if (LOG.isTraceEnabled()) {
1362        LOG.trace("RequestHeader " + TextFormat.shortDebugString(header)
1363            + " totalRequestSize: " + totalRequestSize + " bytes");
1364      }
1365      // Enforcing the call queue size, this triggers a retry in the client
1366      // This is a bit late to be doing this check - we have already read in the
1367      // total request.
1368      if ((totalRequestSize + callQueueSizeInBytes.sum()) > maxQueueSizeInBytes) {
1369        final RpcServer.Call callTooBig = createCall(id, this.service, null,
1370            null, null, null, this, totalRequestSize, null, null, 0,
1371            this.callCleanup);
1372        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1373        metrics.exception(CALL_QUEUE_TOO_BIG_EXCEPTION);
1374        setupResponse(responseBuffer, callTooBig, CALL_QUEUE_TOO_BIG_EXCEPTION,
1375            "Call queue is full on " + server.getServerName()
1376                + ", is hbase.ipc.server.max.callqueue.size too small?");
1377        callTooBig.sendResponseIfReady();
1378        return;
1379      }
1380      MethodDescriptor md = null;
1381      Message param = null;
1382      CellScanner cellScanner = null;
1383      try {
1384        if (header.hasRequestParam() && header.getRequestParam()) {
1385          md = this.service.getDescriptorForType().findMethodByName(
1386              header.getMethodName());
1387          if (md == null)
1388            throw new UnsupportedOperationException(header.getMethodName());
1389          builder = this.service.getRequestPrototype(md).newBuilderForType();
1390          cis.resetSizeCounter();
1391          int paramSize = cis.readRawVarint32();
1392          offset += cis.getTotalBytesRead();
1393          if (builder != null) {
1394            ProtobufUtil.mergeFrom(builder, cis, paramSize);
1395            param = builder.build();
1396          }
1397          offset += paramSize;
1398        } else {
1399          // currently header must have request param, so we directly throw
1400          // exception here
1401          String msg = "Invalid request header: "
1402              + TextFormat.shortDebugString(header)
1403              + ", should have param set in it";
1404          LOG.warn(msg);
1405          throw new DoNotRetryIOException(msg);
1406        }
1407        if (header.hasCellBlockMeta()) {
1408          buf.position(offset);
1409          ByteBuff dup = buf.duplicate();
1410          dup.limit(offset + header.getCellBlockMeta().getLength());
1411          cellScanner = cellBlockBuilder.createCellScannerReusingBuffers(
1412              this.codec, this.compressionCodec, dup);
1413        }
1414      } catch (Throwable t) {
1415        InetSocketAddress address = getListenerAddress();
1416        String msg = (address != null ? address : "(channel closed)")
1417            + " is unable to read call parameter from client "
1418            + getHostAddress();
1419        LOG.warn(msg, t);
1420
1421        metrics.exception(t);
1422
1423        // probably the hbase hadoop version does not match the running hadoop
1424        // version
1425        if (t instanceof LinkageError) {
1426          t = new DoNotRetryIOException(t);
1427        }
1428        // If the method is not present on the server, do not retry.
1429        if (t instanceof UnsupportedOperationException) {
1430          t = new DoNotRetryIOException(t);
1431        }
1432
1433        final RpcServer.Call readParamsFailedCall = createCall(id,
1434            this.service, null, null, null, null, this, totalRequestSize, null,
1435            null, 0, this.callCleanup);
1436        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1437        setupResponse(responseBuffer, readParamsFailedCall, t,
1438            msg + "; " + t.getMessage());
1439        readParamsFailedCall.sendResponseIfReady();
1440        return;
1441      }
1442
1443      TraceInfo traceInfo = header.hasTraceInfo() ? new TraceInfo(header
1444          .getTraceInfo().getTraceId(), header.getTraceInfo().getParentId())
1445          : null;
1446      int timeout = 0;
1447      if (header.hasTimeout() && header.getTimeout() > 0) {
1448        timeout = Math.max(minClientRequestTimeout, header.getTimeout());
1449      }
1450      RpcServer.Call call = createCall(id, this.service, md, header, param,
1451          cellScanner, this, totalRequestSize, traceInfo, this.addr, timeout,
1452          this.callCleanup);
1453
1454      if (!scheduler.dispatch(new CallRunner(RpcServer.this, call))) {
1455        callQueueSizeInBytes.add(-1 * call.getSize());
1456
1457        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1458        metrics.exception(CALL_QUEUE_TOO_BIG_EXCEPTION);
1459        setupResponse(responseBuffer, call, CALL_QUEUE_TOO_BIG_EXCEPTION,
1460            "Call queue is full on " + server.getServerName()
1461                + ", too many items queued ?");
1462        call.sendResponseIfReady();
1463      }
1464    }

1497  /**
1498   * Constructs a server listening on the named port and address.
1499   * @param server hosting instance of {@link Server}. We will do authentications if an
1500   * instance else pass null for no authentication check.
1501   * @param name Used keying this rpc servers' metrics and for naming the Listener thread.
1502   * @param services A list of services.
1503   * @param bindAddress Where to listen
1504   * @param conf
1505   * @param scheduler
1506   */
1507  public RpcServer(final Server server, final String name,
1508      final List<BlockingServiceAndInterface> services,
1509      final InetSocketAddress bindAddress, Configuration conf,
1510      RpcScheduler scheduler)
1511      throws IOException {
1512    if (conf.getBoolean("hbase.ipc.server.reservoir.enabled", true)) {
1513      int poolBufSize = conf.getInt(ByteBufferPool.BUFFER_SIZE_KEY,
1514          ByteBufferPool.DEFAULT_BUFFER_SIZE);
1515      // The max number of buffers to be pooled in the ByteBufferPool. The default value been
1516      // selected based on the #handlers configured. When it is read request, 2 MB is the max size
1517      // at which we will send back one RPC request. Means max we need 2 MB for creating the
1518      // response cell block. (Well it might be much lesser than this because in 2 MB size calc, we
1519      // include the heap size overhead of each cells also.) Considering 2 MB, we will need
1520      // (2 * 1024 * 1024) / poolBufSize buffers to make the response cell block. Pool buffer size
1521      // is by default 64 KB.
1522      // In case of read request, at the end of the handler process, we will make the response
1523      // cellblock and add the Call to connection's response Q and a single Responder thread takes
1524      // connections and responses from that one by one and do the socket write. So there is chances
1525      // that by the time a handler originated response is actually done writing to socket and so
1526      // released the BBs it used, the handler might have processed one more read req. On an avg 2x
1527      // we consider and consider that also for the max buffers to pool
1528      int bufsForTwoMB = (2 * 1024 * 1024) / poolBufSize;
1529      int maxPoolSize = conf.getInt(ByteBufferPool.MAX_POOL_SIZE_KEY,
1530          conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT,
1531              HConstants.DEFAULT_REGION_SERVER_HANDLER_COUNT) * bufsForTwoMB * 2);
1532      this.reservoir = new ByteBufferPool(poolBufSize, maxPoolSize);
1533      this.minSizeForReservoirUse = getMinSizeForReservoirUse(this.reservoir);
1534    } else {
1535      reservoir = null;
1536      this.minSizeForReservoirUse = Integer.MAX_VALUE;// reservoir itself not in place.
1537    }
1538    this.server = server;
1539    this.services = services;
1540    this.bindAddress = bindAddress;
1541    this.conf = conf;
1542    // See declaration above for documentation on what this size is.
1543    this.maxQueueSizeInBytes =
1544      this.conf.getLong("hbase.ipc.server.max.callqueue.size", DEFAULT_MAX_CALLQUEUE_SIZE);
1545
1546    this.warnResponseTime = conf.getInt(WARN_RESPONSE_TIME, DEFAULT_WARN_RESPONSE_TIME);
1547    this.warnResponseSize = conf.getInt(WARN_RESPONSE_SIZE, DEFAULT_WARN_RESPONSE_SIZE);
1548    this.minClientRequestTimeout = conf.getInt(MIN_CLIENT_REQUEST_TIMEOUT,
1549        DEFAULT_MIN_CLIENT_REQUEST_TIMEOUT);
1550    this.maxRequestSize = conf.getInt(MAX_REQUEST_SIZE, DEFAULT_MAX_REQUEST_SIZE);
1551
1552    this.metrics = new MetricsHBaseServer(name, new MetricsHBaseServerWrapperImpl(this));
1553    this.tcpNoDelay = conf.getBoolean("hbase.ipc.server.tcpnodelay", true);
1554    this.tcpKeepAlive = conf.getBoolean("hbase.ipc.server.tcpkeepalive", true);
1555
1556    this.cellBlockBuilder = new CellBlockBuilder(conf);
1557
1558    this.authorize = conf.getBoolean(HADOOP_SECURITY_AUTHORIZATION, false);
1559    this.userProvider = UserProvider.instantiate(conf);
1560    this.isSecurityEnabled = userProvider.isHBaseSecurityEnabled();
1561    if (isSecurityEnabled) {
1562      HBaseSaslRpcServer.init(conf);
1563    }
1564
1565    this.scheduler = scheduler;
1566  }

1646  /**
1647   * This is a server side method, which is invoked over RPC. On success
1648   * the return response has protobuf response payload. On failure, the
1649   * exception name and the stack trace are returned in the protobuf response.
1650   */
1651  @Override
1652  public Pair<Message, CellScanner> call(RpcCall call,
1653      MonitoredRPCHandler status) throws IOException {
1654    try {
1655      MethodDescriptor md = call.getMethod();
1656      Message param = call.getParam();
1657      status.setRPC(md.getName(), new Object[]{param},
1658        call.getReceiveTime());
1659      // TODO: Review after we add in encoded data blocks.
1660      status.setRPCPacket(param);
1661      status.resume("Servicing call");
1662      //get an instance of the method arg type
1663      HBaseRpcController controller = new HBaseRpcControllerImpl(call.getCellScanner());
1664      controller.setCallTimeout(call.getTimeout());
1665      Message result = call.getService().callBlockingMethod(md, controller, param);
1666      long receiveTime = call.getReceiveTime();
1667      long startTime = call.getStartTime();
1668      long endTime = System.currentTimeMillis();
1669      int processingTime = (int) (endTime - startTime);
1670      int qTime = (int) (startTime - receiveTime);
1671      int totalTime = (int) (endTime - receiveTime);
1672      if (LOG.isTraceEnabled()) {
1673        LOG.trace(CurCall.get().toString() +
1674            ", response " + TextFormat.shortDebugString(result) +
1675            " queueTime: " + qTime +
1676            " processingTime: " + processingTime +
1677            " totalTime: " + totalTime);
1678      }
1679      // Use the raw request call size for now.
1680      long requestSize = call.getSize();
1681      long responseSize = result.getSerializedSize();
1682      if (call.isClientCellBlockSupported()) {
1683        // Include the payload size in HBaseRpcController
1684        responseSize += call.getResponseCellSize();
1685      }
1686
1687      metrics.dequeuedCall(qTime);
1688      metrics.processedCall(processingTime);
1689      metrics.totalCall(totalTime);
1690      metrics.receivedRequest(requestSize);
1691      metrics.sentResponse(responseSize);
1692      // log any RPC responses that are slower than the configured warn
1693      // response time or larger than configured warning size
1694      boolean tooSlow = (processingTime > warnResponseTime && warnResponseTime > -1);
1695      boolean tooLarge = (responseSize > warnResponseSize && warnResponseSize > -1);
1696      if (tooSlow || tooLarge) {
1697        // when tagging, we let TooLarge trump TooSmall to keep output simple
1698        // note that large responses will often also be slow.
1699        logResponse(param,
1700            md.getName(), md.getName() + "(" + param.getClass().getName() + ")",
1701            (tooLarge ? "TooLarge" : "TooSlow"),
1702            status.getClient(), startTime, processingTime, qTime,
1703            responseSize);
1704      }
1705      return new Pair<>(result, controller.cellScanner());
1706    } catch (Throwable e) {
1707      // The above callBlockingMethod will always return a SE.  Strip the SE wrapper before
1708      // putting it on the wire.  Its needed to adhere to the pb Service Interface but we don't
1709      // need to pass it over the wire.
1710      if (e instanceof ServiceException) {
1711        if (e.getCause() == null) {
1712          LOG.debug("Caught a ServiceException with null cause", e);
1713        } else {
1714          e = e.getCause();
1715        }
1716      }
1717
1718      // increment the number of requests that were exceptions.
1719      metrics.exception(e);
1720
1721      if (e instanceof LinkageError) throw new DoNotRetryIOException(e);
1722      if (e instanceof IOException) throw (IOException)e;
1723      LOG.error("Unexpected throwable object ", e);
1724      throw new IOException(e.getMessage(), e);
1725    }
1726  }

1830   * @throws java.io.IOException e
1831   * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
1832   */
1833  protected int channelRead(ReadableByteChannel channel,
1834                                   ByteBuffer buffer) throws IOException {
1835
1836    int count = (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
1837           channel.read(buffer) : channelIO(channel, null, buffer);
1838    if (count > 0) {
1839      metrics.receivedBytes(count);
1840    }
1841    return count;
1842  }

1844  /**
1845   * Helper for {@link #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)}
1846   * and {@link #channelWrite(GatheringByteChannel, BufferChain)}. Only
1847   * one of readCh or writeCh should be non-null.
1848   *
1849   * @param readCh read channel
1850   * @param writeCh write channel
1851   * @param buf buffer to read or write into/out of
1852   * @return bytes written
1853   * @throws java.io.IOException e
1854   * @see #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)
1855   * @see #channelWrite(GatheringByteChannel, BufferChain)
1856   */
1857  private static int channelIO(ReadableByteChannel readCh,
1858                               WritableByteChannel writeCh,
1859                               ByteBuffer buf) throws IOException {
1860
1861    int originalLimit = buf.limit();
1862    int initialRemaining = buf.remaining();
1863    int ret = 0;
1864
1865    while (buf.remaining() > 0) {
1866      try {
1867        int ioSize = Math.min(buf.remaining(), NIO_BUFFER_LIMIT);
1868        buf.limit(buf.position() + ioSize);
1869
1870        ret = (readCh == null) ? writeCh.write(buf) : readCh.read(buf);
1871
1872        if (ret < ioSize) {
1873          break;
1874        }
1875
1876      } finally {
1877        buf.limit(originalLimit);
1878      }
1879    }
1880
1881    int nBytes = initialRemaining - buf.remaining();
1882    return (nBytes > 0) ? nBytes : ret;
1883  }

1885  /**
1886   * This is extracted to a static method for better unit testing. We try to get buffer(s) from pool
1887   * as much as possible.
1888   *
1889   * @param pool The ByteBufferPool to use
1890   * @param minSizeForPoolUse Only for buffer size above this, we will try to use pool. Any buffer
1891   *           need of size below this, create on heap ByteBuffer.
1892   * @param reqLen Bytes count in request
1893   */
1894  @VisibleForTesting
1895  static Pair<ByteBuff, CallCleanup> allocateByteBuffToReadInto(ByteBufferPool pool,
1896      int minSizeForPoolUse, int reqLen) {
1897    ByteBuff resultBuf;
1898    List<ByteBuffer> bbs = new ArrayList<>((reqLen / pool.getBufferSize()) + 1);
1899    int remain = reqLen;
1900    ByteBuffer buf = null;
1901    while (remain >= minSizeForPoolUse && (buf = pool.getBuffer()) != null) {
1902      bbs.add(buf);
1903      remain -= pool.getBufferSize();
1904    }
1905    ByteBuffer[] bufsFromPool = null;
1906    if (bbs.size() > 0) {
1907      bufsFromPool = new ByteBuffer[bbs.size()];
1908      bbs.toArray(bufsFromPool);
1909    }
1910    if (remain > 0) {
1911      bbs.add(ByteBuffer.allocate(remain));
1912    }
1913    if (bbs.size() > 1) {
1914      ByteBuffer[] items = new ByteBuffer[bbs.size()];
1915      bbs.toArray(items);
1916      resultBuf = new MultiByteBuff(items);
1917    } else {
1918      // We are backed by single BB
1919      resultBuf = new SingleByteBuff(bbs.get(0));
1920    }
1921    resultBuf.limit(reqLen);
1922    if (bufsFromPool != null) {
1923      final ByteBuffer[] bufsFromPoolFinal = bufsFromPool;
1924      return new Pair<>(resultBuf, () -> {
1925        // Return back all the BBs to pool
1926        for (int i = 0; i < bufsFromPoolFinal.length; i++) {
1927          pool.putbackBuffer(bufsFromPoolFinal[i]);
1928        }
1929      });
1930    }
1931    return new Pair<>(resultBuf, null);
1932  }
