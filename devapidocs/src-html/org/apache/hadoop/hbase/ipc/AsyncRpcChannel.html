151  /**
152   * Connect to channel
153   * @param bootstrap to connect to
154   * @return future of connection
155   */
156  private ChannelFuture connect(final Bootstrap bootstrap) {
157    return bootstrap.remoteAddress(address).connect()
158        .addListener(new GenericFutureListener<ChannelFuture>() {
159          @Override
160          public void operationComplete(final ChannelFuture f) throws Exception {
161            if (!f.isSuccess()) {
162              retryOrClose(bootstrap, failureCounter++, client.failureSleep, f.cause());
163              return;
164            }
165            channel = f.channel();
166
167            setupAuthorization();
168
169            ByteBuf b = channel.alloc().directBuffer(6);
170            createPreamble(b, authMethod);
171            channel.writeAndFlush(b).addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
172            if (useSasl) {
173              UserGroupInformation ticket = AsyncRpcChannel.this.ticket.getUGI();
174              if (authMethod == AuthMethod.KERBEROS) {
175                if (ticket != null && ticket.getRealUser() != null) {
176                  ticket = ticket.getRealUser();
177                }
178              }
179              SaslClientHandler saslHandler;
180              if (ticket == null) {
181                throw new FatalConnectionException("ticket/user is null");
182              }
183              final UserGroupInformation realTicket = ticket;
184              saslHandler = ticket.doAs(new PrivilegedExceptionAction<SaslClientHandler>() {
185                @Override
186                public SaslClientHandler run() throws IOException {
187                  return getSaslHandler(realTicket, bootstrap);
188                }
189              });
190              if (saslHandler != null) {
191                // Sasl connect is successful. Let's set up Sasl channel handler
192                channel.pipeline().addFirst(saslHandler);
193              } else {
194                // fall back to simple auth because server told us so.
195                authMethod = AuthMethod.SIMPLE;
196                useSasl = false;
197              }
198            } else {
199              startHBaseConnection(f.channel());
200            }
201          }
202        });
203  }

205  /**
206   * Start HBase connection
207   * @param ch channel to start connection on
208   */
209  private void startHBaseConnection(Channel ch) {
210    ch.pipeline().addLast("frameDecoder",
211      new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 4));
212    ch.pipeline().addLast(new AsyncServerResponseHandler(this));
213    try {
214      writeChannelHeader(ch).addListener(new GenericFutureListener<ChannelFuture>() {
215        @Override
216        public void operationComplete(ChannelFuture future) throws Exception {
217          if (!future.isSuccess()) {
218            close(future.cause());
219            return;
220          }
221          List<AsyncCall> callsToWrite;
222          synchronized (pendingCalls) {
223            connected = true;
224            callsToWrite = new ArrayList<AsyncCall>(pendingCalls.values());
225          }
226          for (AsyncCall call : callsToWrite) {
227            writeRequest(call);
228          }
229        }
230      });
231    } catch (IOException e) {
232      close(e);
233    }
234  }
235
236  private void startConnectionWithEncryption(Channel ch) {
237    // for rpc encryption, the order of ChannelInboundHandler should be:
238    // LengthFieldBasedFrameDecoder->SaslClientHandler->LengthFieldBasedFrameDecoder
239    // Don't skip the first 4 bytes for length in beforeUnwrapDecoder,
240    // SaslClientHandler will handler this
241    ch.pipeline().addFirst("beforeUnwrapDecoder",
242        new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 0));
243    ch.pipeline().addLast("afterUnwrapDecoder",
244        new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 4));
245    ch.pipeline().addLast(new AsyncServerResponseHandler(this));
246    List<AsyncCall> callsToWrite;
247    synchronized (pendingCalls) {
248      connected = true;
249      callsToWrite = new ArrayList<AsyncCall>(pendingCalls.values());
250    }
251    for (AsyncCall call : callsToWrite) {
252      writeRequest(call);
253    }
254  }

256  /**
257   * Get SASL handler
258   * @param bootstrap to reconnect to
259   * @return new SASL handler
260   * @throws java.io.IOException if handler failed to create
261   */
262  private SaslClientHandler getSaslHandler(final UserGroupInformation realTicket,
263      final Bootstrap bootstrap) throws IOException {
264    return new SaslClientHandler(realTicket, authMethod, token, serverPrincipal,
265        client.fallbackAllowed,
266        client.conf.get("hbase.rpc.protection",
267          SaslUtil.QualityOfProtection.AUTHENTICATION.name().toLowerCase(Locale.ROOT)),
268        getChannelHeaderBytes(authMethod),
269        new SaslClientHandler.SaslExceptionHandler() {
270          @Override
271          public void handle(int retryCount, Random random, Throwable cause) {
272            try {
273              // Handle Sasl failure. Try to potentially get new credentials
274              handleSaslConnectionFailure(retryCount, cause, realTicket);
275
276              retryOrClose(bootstrap, failureCounter++, random.nextInt(reloginMaxBackoff) + 1,
277                cause);
278            } catch (IOException | InterruptedException e) {
279              close(e);
280            }
281          }
282        }, new SaslClientHandler.SaslSuccessfulConnectHandler() {
283          @Override
284          public void onSuccess(Channel channel) {
285            startHBaseConnection(channel);
286          }
287
288          @Override
289          public void onSaslProtectionSucess(Channel channel) {
290            startConnectionWithEncryption(channel);
291          }
292        });
293  }

295  /**
296   * Retry to connect or close
297   * @param bootstrap to connect with
298   * @param failureCount failure count
299   * @param e exception of fail
300   */
301  private void retryOrClose(final Bootstrap bootstrap, int failureCount, long timeout,
302      Throwable e) {
303    if (failureCount < client.maxRetries) {
304      client.newTimeout(new TimerTask() {
305        @Override
306        public void run(Timeout timeout) throws Exception {
307          connect(bootstrap);
308        }
309      }, timeout, TimeUnit.MILLISECONDS);
310    } else {
311      client.failedServers.addToFailedServers(address);
312      close(e);
313    }
314  }

316  /**
317   * Calls method on channel
318   * @param method to call
319   * @param request to send
320   * @param cellScanner with cells to send
321   * @param responsePrototype to construct response with
322   * @param rpcTimeout timeout for request
323   * @param priority for request
324   * @return Promise for the response Message
325   */
326  public <R extends Message, O> io.netty.util.concurrent.Promise<O> callMethod(
327      final Descriptors.MethodDescriptor method,
328      final Message request,final CellScanner cellScanner,
329      R responsePrototype, MessageConverter<R, O> messageConverter, IOExceptionConverter
330      exceptionConverter, long rpcTimeout, int priority) {
331    final AsyncCall<R, O> call = new AsyncCall<>(this, client.callIdCnt.getAndIncrement(),
332        method, request, cellScanner, responsePrototype, messageConverter, exceptionConverter,
333        rpcTimeout, priority, client.metrics);
334
335    synchronized (pendingCalls) {
336      if (closed) {
337        call.setFailure(new ConnectException());
338        return call;
339      }
340      pendingCalls.put(call.id, call);
341      // Add timeout for cleanup if none is present
342      if (cleanupTimer == null && call.getRpcTimeout() > 0) {
343        cleanupTimer = client.newTimeout(timeoutTask, call.getRpcTimeout(), TimeUnit.MILLISECONDS);
344      }
345      if (!connected) {
346        return call;
347      }
348    }
349    writeRequest(call);
350    return call;
351  }

363  /**
364   * Write the channel header
365   * @param channel to write to
366   * @return future of write
367   * @throws java.io.IOException on failure to write
368   */
369  private ChannelFuture writeChannelHeader(Channel channel) throws IOException {
370    RPCProtos.ConnectionHeader header = getChannelHeader(authMethod);
371    int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(header);
372    ByteBuf b = channel.alloc().directBuffer(totalSize);
373
374    b.writeInt(header.getSerializedSize());
375    b.writeBytes(header.toByteArray());
376
377    return channel.writeAndFlush(b);
378  }
379
380  private byte[] getChannelHeaderBytes(AuthMethod authMethod) {
381    RPCProtos.ConnectionHeader header = getChannelHeader(authMethod);
382    ByteBuffer b = ByteBuffer.allocate(header.getSerializedSize() + 4);
383    b.putInt(header.getSerializedSize());
384    b.put(header.toByteArray());
385    return b.array();
386  }
387
388  private RPCProtos.ConnectionHeader getChannelHeader(AuthMethod authMethod) {
389    RPCProtos.ConnectionHeader.Builder headerBuilder = RPCProtos.ConnectionHeader.newBuilder()
390        .setServiceName(serviceName);
391
392    RPCProtos.UserInformation userInfoPB = buildUserInfo(ticket.getUGI(), authMethod);
393    if (userInfoPB != null) {
394      headerBuilder.setUserInfo(userInfoPB);
395    }
396
397    if (client.codec != null) {
398      headerBuilder.setCellBlockCodecClass(client.codec.getClass().getCanonicalName());
399    }
400    if (client.compressor != null) {
401      headerBuilder.setCellBlockCompressorClass(client.compressor.getClass().getCanonicalName());
402    }
403
404    headerBuilder.setVersionInfo(ProtobufUtil.getVersionInfo());
405    return headerBuilder.build();
406  }

408  /**
409   * Write request to channel
410   * @param call to write
411   */
412  private void writeRequest(final AsyncCall call) {
413    try {
414      final RPCProtos.RequestHeader.Builder requestHeaderBuilder = RPCProtos.RequestHeader
415          .newBuilder();
416      requestHeaderBuilder.setCallId(call.id).setMethodName(call.method.getName())
417          .setRequestParam(call.param != null);
418
419      if (Trace.isTracing()) {
420        Span s = Trace.currentSpan();
421        requestHeaderBuilder.setTraceInfo(TracingProtos.RPCTInfo.newBuilder()
422            .setParentId(s.getSpanId()).setTraceId(s.getTraceId()));
423      }
424
425      ByteBuffer cellBlock = client.buildCellBlock(call.cellScanner());
426      if (cellBlock != null) {
427        final RPCProtos.CellBlockMeta.Builder cellBlockBuilder = RPCProtos.CellBlockMeta
428            .newBuilder();
429        cellBlockBuilder.setLength(cellBlock.limit());
430        requestHeaderBuilder.setCellBlockMeta(cellBlockBuilder.build());
431      }
432      // Only pass priority if there one. Let zero be same as no priority.
433      if (call.getPriority() != HBaseRpcController.PRIORITY_UNSET) {
434        requestHeaderBuilder.setPriority(call.getPriority());
435      }
436      requestHeaderBuilder.setTimeout(call.rpcTimeout > Integer.MAX_VALUE ?
437          Integer.MAX_VALUE : (int)call.rpcTimeout);
438
439      RPCProtos.RequestHeader rh = requestHeaderBuilder.build();
440
441      int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(rh, call.param);
442      if (cellBlock != null) {
443        totalSize += cellBlock.remaining();
444      }
445
446      ByteBuf b = channel.alloc().directBuffer(4 + totalSize);
447      try (ByteBufOutputStream out = new ByteBufOutputStream(b)) {
448        call.callStats.setRequestSizeBytes(IPCUtil.write(out, rh, call.param, cellBlock));
449      }
450
451      channel.writeAndFlush(b).addListener(new CallWriteListener(this, call.id));
452    } catch (IOException e) {
453      close(e);
454    }
455  }

457  /**
458   * Set up server authorization
459   * @throws java.io.IOException if auth setup failed
460   */
461  private void setupAuthorization() throws IOException {
462    SecurityInfo securityInfo = SecurityInfo.getInfo(serviceName);
463    this.useSasl = client.userProvider.isHBaseSecurityEnabled();
464
465    this.token = null;
466    if (useSasl && securityInfo != null) {
467      AuthenticationProtos.TokenIdentifier.Kind tokenKind = securityInfo.getTokenKind();
468      if (tokenKind != null) {
469        TokenSelector<? extends TokenIdentifier> tokenSelector = TOKEN_HANDDLERS.get(tokenKind);
470        if (tokenSelector != null) {
471          token = tokenSelector.selectToken(new Text(client.clusterId),
472            ticket.getUGI().getTokens());
473        } else if (LOG.isDebugEnabled()) {
474          LOG.debug("No token selector found for type " + tokenKind);
475        }
476      }
477      String serverKey = securityInfo.getServerPrincipal();
478      if (serverKey == null) {
479        throw new IOException("Can't obtain server Kerberos config key from SecurityInfo");
480      }
481      this.serverPrincipal = SecurityUtil.getServerPrincipal(client.conf.get(serverKey),
482        address.getAddress().getCanonicalHostName().toLowerCase(Locale.ROOT));
483      if (LOG.isDebugEnabled()) {
484        LOG.debug("RPC Server Kerberos principal name for service=" + serviceName + " is "
485            + serverPrincipal);
486      }
487    }
488
489    if (!useSasl) {
490      authMethod = AuthMethod.SIMPLE;
491    } else if (token != null) {
492      authMethod = AuthMethod.DIGEST;
493    } else {
494      authMethod = AuthMethod.KERBEROS;
495    }
496
497    if (LOG.isDebugEnabled()) {
498      LOG.debug(
499        "Use " + authMethod + " authentication for service " + serviceName + ", sasl=" + useSasl);
500    }
501    reloginMaxBackoff = client.conf.getInt("hbase.security.relogin.maxbackoff", 5000);
502  }

504  /**
505   * Build the user information
506   * @param ugi User Group Information
507   * @param authMethod Authorization method
508   * @return UserInformation protobuf
509   */
510  private RPCProtos.UserInformation buildUserInfo(UserGroupInformation ugi, AuthMethod authMethod) {
511    if (ugi == null || authMethod == AuthMethod.DIGEST) {
512      // Don't send user for token auth
513      return null;
514    }
515    RPCProtos.UserInformation.Builder userInfoPB = RPCProtos.UserInformation.newBuilder();
516    if (authMethod == AuthMethod.KERBEROS) {
517      // Send effective user for Kerberos auth
518      userInfoPB.setEffectiveUser(ugi.getUserName());
519    } else if (authMethod == AuthMethod.SIMPLE) {
520      // Send both effective user and real user for simple auth
521      userInfoPB.setEffectiveUser(ugi.getUserName());
522      if (ugi.getRealUser() != null) {
523        userInfoPB.setRealUser(ugi.getRealUser().getUserName());
524      }
525    }
526    return userInfoPB.build();
527  }

529  /**
530   * Create connection preamble
531   * @param byteBuf to write to
532   * @param authMethod to write
533   */
534  private void createPreamble(ByteBuf byteBuf, AuthMethod authMethod) {
535    byteBuf.writeBytes(HConstants.RPC_HEADER);
536    byteBuf.writeByte(HConstants.RPC_CURRENT_VERSION);
537    byteBuf.writeByte(authMethod.code);
538  }

540  private void close0(Throwable e) {
541    List<AsyncCall> toCleanup;
542    synchronized (pendingCalls) {
543      if (closed) {
544        return;
545      }
546      closed = true;
547      toCleanup = new ArrayList<AsyncCall>(pendingCalls.values());
548      pendingCalls.clear();
549    }
550    IOException closeException = null;
551    if (e != null) {
552      if (e instanceof IOException) {
553        closeException = (IOException) e;
554      } else {
555        closeException = new IOException(e);
556      }
557    }
558    // log the info
559    if (LOG.isDebugEnabled() && closeException != null) {
560      LOG.debug(name + ": closing ipc connection to " + address, closeException);
561    }
562    if (cleanupTimer != null) {
563      cleanupTimer.cancel();
564      cleanupTimer = null;
565    }
566    for (AsyncCall call : toCleanup) {
567      call.setFailed(closeException != null ? closeException
568          : new ConnectionClosingException(
569              "Call id=" + call.id + " on server " + address + " aborted: connection is closing"));
570    }
571    channel.disconnect().addListener(ChannelFutureListener.CLOSE);
572    if (LOG.isDebugEnabled()) {
573      LOG.debug(name + ": closed");
574    }
575  }

577  /**
578   * Close connection
579   * @param e exception on close
580   */
581  public void close(final Throwable e) {
582    client.removeConnection(this);
583
584    // Move closing from the requesting thread to the channel thread
585    if (channel.eventLoop().inEventLoop()) {
586      close0(e);
587    } else {
588      channel.eventLoop().execute(new Runnable() {
589        @Override
590        public void run() {
591          close0(e);
592        }
593      });
594    }
595  }

597  /**
598   * Clean up calls.
599   */
600  private void cleanupCalls() {
601    List<AsyncCall> toCleanup = new ArrayList<AsyncCall>();
602    long currentTime = EnvironmentEdgeManager.currentTime();
603    long nextCleanupTaskDelay = -1L;
604    synchronized (pendingCalls) {
605      for (Iterator<AsyncCall> iter = pendingCalls.values().iterator(); iter.hasNext();) {
606        AsyncCall call = iter.next();
607        long timeout = call.getRpcTimeout();
608        if (timeout > 0) {
609          if (currentTime - call.getStartTime() >= timeout) {
610            iter.remove();
611            toCleanup.add(call);
612          } else {
613            if (nextCleanupTaskDelay < 0 || timeout < nextCleanupTaskDelay) {
614              nextCleanupTaskDelay = timeout;
615            }
616          }
617        }
618      }
619      if (nextCleanupTaskDelay > 0) {
620        cleanupTimer = client.newTimeout(timeoutTask, nextCleanupTaskDelay, TimeUnit.MILLISECONDS);
621      } else {
622        cleanupTimer = null;
623      }
624    }
625    for (AsyncCall call : toCleanup) {
626      call.setFailed(new CallTimeoutException("Call id=" + call.id + ", waitTime="
627          + (currentTime - call.getStartTime()) + ", rpcTimeout=" + call.getRpcTimeout()));
628    }
629  }

643  /**
644   * Check if user should authenticate over Kerberos
645   * @return true if should be authenticated over Kerberos
646   * @throws java.io.IOException on failure of check
647   */
648  private synchronized boolean shouldAuthenticateOverKrb() throws IOException {
649    UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
650    UserGroupInformation currentUser = UserGroupInformation.getCurrentUser();
651    UserGroupInformation realUser = currentUser.getRealUser();
652    return authMethod == AuthMethod.KERBEROS && loginUser != null &&
653      // Make sure user logged in using Kerberos either keytab or TGT
654      loginUser.hasKerberosCredentials() &&
655      // relogin only in case it is the login user (e.g. JT)
656      // or superuser (like oozie).
657      (loginUser.equals(currentUser) || loginUser.equals(realUser));
658  }

660  /**
661   * If multiple clients with the same principal try to connect to the same server at the same time,
662   * the server assumes a replay attack is in progress. This is a feature of kerberos. In order to
663   * work around this, what is done is that the client backs off randomly and tries to initiate the
664   * connection again. The other problem is to do with ticket expiry. To handle that, a relogin is
665   * attempted.
666   * <p>
667   * The retry logic is governed by the {@link #shouldAuthenticateOverKrb} method. In case when the
668   * user doesn't have valid credentials, we don't need to retry (from cache or ticket). In such
669   * cases, it is prudent to throw a runtime exception when we receive a SaslException from the
670   * underlying authentication implementation, so there is no retry from other high level (for eg,
671   * HCM or HBaseAdmin).
672   * </p>
673   * @param currRetries retry count
674   * @param ex exception describing fail
675   * @param user which is trying to connect
676   * @throws java.io.IOException if IO fail
677   * @throws InterruptedException if thread is interrupted
678   */
679  private void handleSaslConnectionFailure(final int currRetries, final Throwable ex,
680      final UserGroupInformation user) throws IOException, InterruptedException {
681    user.doAs(new PrivilegedExceptionAction<Void>() {
682      @Override
683      public Void run() throws IOException, InterruptedException {
684        if (shouldAuthenticateOverKrb()) {
685          if (currRetries < MAX_SASL_RETRIES) {
686            LOG.debug("Exception encountered while connecting to the server : " + ex);
687            // try re-login
688            if (UserGroupInformation.isLoginKeytabBased()) {
689              UserGroupInformation.getLoginUser().reloginFromKeytab();
690            } else {
691              UserGroupInformation.getLoginUser().reloginFromTicketCache();
692            }
693
694            // Should reconnect
695            return null;
696          } else {
697            String msg = "Couldn't setup connection for "
698                + UserGroupInformation.getLoginUser().getUserName() + " to " + serverPrincipal;
699            LOG.warn(msg, ex);
700            throw (IOException) new IOException(msg).initCause(ex);
701          }
702        } else {
703          LOG.warn("Exception encountered while connecting to " + "the server : " + ex);
704        }
705        if (ex instanceof RemoteException) {
706          throw (RemoteException) ex;
707        }
708        if (ex instanceof SaslException) {
709          String msg = "SASL authentication failed."
710              + " The most likely cause is missing or invalid credentials." + " Consider 'kinit'.";
711          LOG.fatal(msg, ex);
712          throw new RuntimeException(msg, ex);
713        }
714        throw new IOException(ex);
715      }
716    });
717  }

742  /**
743   * Listens to call writes and fails if write failed
744   */
745  private static final class CallWriteListener implements ChannelFutureListener {
746    private final AsyncRpcChannel rpcChannel;
747    private final int id;
748
749    public CallWriteListener(AsyncRpcChannel asyncRpcChannelImpl, int id) {
750      this.rpcChannel = asyncRpcChannelImpl;
751      this.id = id;
752    }
753
754    @Override
755    public void operationComplete(ChannelFuture future) throws Exception {
756      if (!future.isSuccess()) {
757        AsyncCall call = rpcChannel.removePendingCall(id);
758        if (call != null) {
759          if (future.cause() instanceof IOException) {
760            call.setFailed((IOException) future.cause());
761          } else {
762            call.setFailed(new IOException(future.cause()));
763          }
764        }
765      }
766    }
767  }
