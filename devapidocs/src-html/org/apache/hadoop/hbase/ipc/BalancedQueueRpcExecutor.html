018package org.apache.hadoop.hbase.ipc;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.concurrent.BlockingQueue;
023import java.util.concurrent.LinkedBlockingQueue;
024
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.Abortable;
027import org.apache.hadoop.hbase.HBaseInterfaceAudience;
028import org.apache.hadoop.hbase.classification.InterfaceAudience;
029import org.apache.hadoop.hbase.classification.InterfaceStability;
030import org.apache.hadoop.hbase.util.ReflectionUtils;
031
032/**
033 * An {@link RpcExecutor} that will balance requests evenly across all its queues, but still remains
034 * efficient with a single queue via an inlinable queue balancing mechanism.
035 */
036@InterfaceAudience.LimitedPrivate({ HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX })
037@InterfaceStability.Evolving
038public class BalancedQueueRpcExecutor extends RpcExecutor {
039
040  protected final List<BlockingQueue<CallRunner>> queues;
041  private final QueueBalancer balancer;
042
043  public BalancedQueueRpcExecutor(final String name, final int handlerCount, final int numQueues,
044      final int maxQueueLength) {
045    this(name, handlerCount, numQueues, maxQueueLength, null, null);
046  }
047
048  public BalancedQueueRpcExecutor(final String name, final int handlerCount, final int numQueues,
049      final int maxQueueLength, final Configuration conf, final Abortable abortable) {
050    this(name, handlerCount, numQueues, conf, abortable, LinkedBlockingQueue.class, maxQueueLength);
051  }
052
053  public BalancedQueueRpcExecutor(final String name, final int handlerCount, final int numQueues,
054      final Class<? extends BlockingQueue> queueClass, Object... initargs) {
055    this(name, handlerCount, numQueues, null, null,  queueClass, initargs);
056  }
057
058  public BalancedQueueRpcExecutor(final String name, final int handlerCount, final int numQueues,
059      final Configuration conf, final Abortable abortable,
060      final Class<? extends BlockingQueue> queueClass, Object... initargs) {
061    super(name, Math.max(handlerCount, numQueues), conf, abortable);
062    queues = new ArrayList<BlockingQueue<CallRunner>>(numQueues);
063    this.balancer = getBalancer(numQueues);
064    initializeQueues(numQueues, queueClass, initargs);
065  }
066
067  protected void initializeQueues(final int numQueues,
068      final Class<? extends BlockingQueue> queueClass, Object... initargs) {
069    if (initargs.length > 0) {
070      currentQueueLimit = (int) initargs[0];
071      initargs[0] = Math.max((int) initargs[0], DEFAULT_CALL_QUEUE_SIZE_HARD_LIMIT);
072    }
073    for (int i = 0; i < numQueues; ++i) {
074      queues.add((BlockingQueue<CallRunner>) ReflectionUtils.newInstance(queueClass, initargs));
075    }
076  }
077
078  @Override
079  public boolean dispatch(final CallRunner callTask) throws InterruptedException {
080    int queueIndex = balancer.getNextQueue();
081    BlockingQueue<CallRunner> queue = queues.get(queueIndex);
082    // that means we can overflow by at most <num reader> size (5), that's ok
083    if (queue.size() >= currentQueueLimit) {
084      return false;
085    }
086    return queue.offer(callTask);
087  }
088
089  @Override
090  public int getQueueLength() {
091    int length = 0;
092    for (final BlockingQueue<CallRunner> queue : queues) {
093      length += queue.size();
094    }
095    return length;
096  }
097
098  @Override
099  public List<BlockingQueue<CallRunner>> getQueues() {
100    return queues;
101  }
102}
