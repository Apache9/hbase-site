103  @VisibleForTesting
104  static Pair<EventLoopGroup, Class<? extends Channel>> GLOBAL_EVENT_LOOP_GROUP;
105
106  synchronized static Pair<EventLoopGroup, Class<? extends Channel>>
107      getGlobalEventLoopGroup(Configuration conf) {
108    if (GLOBAL_EVENT_LOOP_GROUP == null) {
109      GLOBAL_EVENT_LOOP_GROUP = createEventLoopGroup(conf);
110      if (LOG.isDebugEnabled()) {
111        LOG.debug("Create global event loop group "
112            + GLOBAL_EVENT_LOOP_GROUP.getFirst().getClass().getSimpleName());
113      }
114    }
115    return GLOBAL_EVENT_LOOP_GROUP;
116  }
117
118  private static Pair<EventLoopGroup, Class<? extends Channel>> createEventLoopGroup(
119      Configuration conf) {
120    // Max amount of threads to use. 0 lets Netty decide based on amount of cores
121    int maxThreads = conf.getInt(CLIENT_MAX_THREADS, 0);
122
123    // Config to enable native transport. Does not seem to be stable at time of implementation
124    // although it is not extensively tested.
125    boolean epollEnabled = conf.getBoolean(USE_NATIVE_TRANSPORT, false);
126
127    // Use the faster native epoll transport mechanism on linux if enabled
128    if (epollEnabled && JVM.isLinux() && JVM.isAmd64()) {
129      if (LOG.isDebugEnabled()) {
130        LOG.debug("Create EpollEventLoopGroup with maxThreads = " + maxThreads);
131      }
132      return new Pair<EventLoopGroup, Class<? extends Channel>>(new EpollEventLoopGroup(maxThreads,
133          Threads.newDaemonThreadFactory("AsyncRpcChannel")), EpollSocketChannel.class);
134    } else {
135      if (LOG.isDebugEnabled()) {
136        LOG.debug("Create NioEventLoopGroup with maxThreads = " + maxThreads);
137      }
138      return new Pair<EventLoopGroup, Class<? extends Channel>>(new NioEventLoopGroup(maxThreads,
139          Threads.newDaemonThreadFactory("AsyncRpcChannel")), NioSocketChannel.class);
140    }
141  }

143  /**
144   * Constructor for tests
145   *
146   * @param configuration      to HBase
147   * @param clusterId          for the cluster
148   * @param localAddress       local address to connect to
149   * @param metrics            the connection metrics
150   * @param channelInitializer for custom channel handlers
151   */
152  protected AsyncRpcClient(Configuration configuration, String clusterId,
153      SocketAddress localAddress, MetricsConnection metrics,
154      ChannelInitializer<SocketChannel> channelInitializer) {
155    super(configuration, clusterId, localAddress, metrics);
156
157    if (LOG.isDebugEnabled()) {
158      LOG.debug("Starting async Hbase RPC client");
159    }
160
161    Pair<EventLoopGroup, Class<? extends Channel>> eventLoopGroupAndChannelClass;
162    this.useGlobalEventLoopGroup = conf.getBoolean(USE_GLOBAL_EVENT_LOOP_GROUP, true);
163    if (useGlobalEventLoopGroup) {
164      eventLoopGroupAndChannelClass = getGlobalEventLoopGroup(configuration);
165    } else {
166      eventLoopGroupAndChannelClass = createEventLoopGroup(configuration);
167    }
168    if (LOG.isDebugEnabled()) {
169      LOG.debug("Use " + (useGlobalEventLoopGroup ? "global" : "individual") + " event loop group "
170          + eventLoopGroupAndChannelClass.getFirst().getClass().getSimpleName());
171    }
172
173    this.connections = new PoolMap<>(getPoolType(configuration), getPoolSize(configuration));
174    this.failedServers = new FailedServers(configuration);
175
176    int operationTimeout = configuration.getInt(HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
177        HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
178
179    // Configure the default bootstrap.
180    this.bootstrap = new Bootstrap();
181    bootstrap.group(eventLoopGroupAndChannelClass.getFirst())
182        .channel(eventLoopGroupAndChannelClass.getSecond())
183        .option(ChannelOption.TCP_NODELAY, tcpNoDelay)
184        .option(ChannelOption.SO_KEEPALIVE, tcpKeepAlive)
185        .option(ChannelOption.CONNECT_TIMEOUT_MILLIS, operationTimeout);
186    if (channelInitializer == null) {
187      channelInitializer = DEFAULT_CHANNEL_INITIALIZER;
188    }
189    bootstrap.handler(channelInitializer);
190    if (localAddress != null) {
191      bootstrap.localAddress(localAddress);
192    }
193  }

219  /**
220   * Make a call, passing <code>param</code>, to the IPC server running at
221   * <code>address</code> which is servicing the <code>protocol</code> protocol,
222   * with the <code>ticket</code> credentials, returning the value.
223   * Throws exceptions if there are network problems or if the remote code
224   * threw an exception.
225   *
226   * @param ticket Be careful which ticket you pass. A new user will mean a new Connection.
227   *               {@link org.apache.hadoop.hbase.security.UserProvider#getCurrent()} makes a new
228   *               instance of User each time so will be a new Connection each time.
229   * @return A pair with the Message response and the Cell data (if any).
230   * @throws InterruptedException if call is interrupted
231   * @throws java.io.IOException  if a connection failure is encountered
232   */
233  @Override
234  protected Pair<Message, CellScanner> call(PayloadCarryingRpcController pcrc,
235      Descriptors.MethodDescriptor md, Message param, Message returnType, User ticket,
236      InetSocketAddress addr, MetricsConnection.CallStats callStats)
237      throws IOException, InterruptedException {
238    if (pcrc == null) {
239      pcrc = new PayloadCarryingRpcController();
240    }
241    final AsyncRpcChannel connection = createRpcChannel(md.getService().getName(), addr, ticket);
242
243    final Promise<Message> promise = connection.callMethod(md, param, pcrc.cellScanner(), returnType,
244        getMessageConverterWithRpcController(pcrc), null, pcrc.getCallTimeout(),
245        pcrc.getPriority());
246
247    pcrc.notifyOnCancel(new RpcCallback<Object>() {
248      @Override
249      public void run(Object parameter) {
250        // Will automatically fail the promise with CancellationException
251        promise.cancel(true);
252      }
253    });
254
255    long timeout = pcrc.hasCallTimeout() ? pcrc.getCallTimeout() : 0;
256    try {
257      Message response = timeout > 0 ? promise.get(timeout, TimeUnit.MILLISECONDS) : promise.get();
258      return new Pair<>(response, pcrc.cellScanner());
259    } catch (ExecutionException e) {
260      if (e.getCause() instanceof IOException) {
261        throw (IOException) e.getCause();
262      } else {
263        throw IPCUtil.wrapException(addr, (Exception) e.getCause());
264      }
265    } catch (TimeoutException e) {
266      CallTimeoutException cte = new CallTimeoutException(promise.toString());
267      throw IPCUtil.wrapException(addr, cte);
268    }
269  }
270
271  private MessageConverter<Message, Message> getMessageConverterWithRpcController(
272      final PayloadCarryingRpcController pcrc) {
273    return new
274      MessageConverter<Message, Message>() {
275        @Override
276        public Message convert(Message msg, CellScanner cellScanner) {
277          pcrc.setCellScanner(cellScanner);
278          return msg;
279        }
280      };
281  }

283  /**
284   * Call method async
285   */
286  private void callMethod(final Descriptors.MethodDescriptor md,
287      final PayloadCarryingRpcController pcrc, final Message param, Message returnType, User ticket,
288      InetSocketAddress addr, final RpcCallback<Message> done) {
289    final AsyncRpcChannel connection;
290    try {
291      connection = createRpcChannel(md.getService().getName(), addr, ticket);
292
293      FutureListener<Message> listener =
294        new FutureListener<Message>() {
295          @Override
296          public void operationComplete(Future<Message> future) throws Exception {
297            if (!future.isSuccess()) {
298              Throwable cause = future.cause();
299              if (cause instanceof IOException) {
300                pcrc.setFailed((IOException) cause);
301              } else {
302                pcrc.setFailed(new IOException(cause));
303              }
304            } else {
305              try {
306                done.run(future.get());
307              } catch (ExecutionException e) {
308                Throwable cause = e.getCause();
309                if (cause instanceof IOException) {
310                  pcrc.setFailed((IOException) cause);
311                } else {
312                  pcrc.setFailed(new IOException(cause));
313                }
314              } catch (InterruptedException e) {
315                pcrc.setFailed(new IOException(e));
316              }
317            }
318          }
319        };
320      connection.callMethod(md, param, pcrc.cellScanner(), returnType,
321          getMessageConverterWithRpcController(pcrc), null,
322          pcrc.getCallTimeout(), pcrc.getPriority())
323          .addListener(listener);
324    } catch (StoppedRpcClientException|FailedServerException e) {
325      pcrc.setFailed(e);
326    }
327  }

329  private boolean closed = false;
330
331  /**
332   * Close netty
333   */
334  public void close() {
335    if (LOG.isDebugEnabled()) {
336      LOG.debug("Stopping async HBase RPC client");
337    }
338
339    synchronized (connections) {
340      if (closed) {
341        return;
342      }
343      closed = true;
344      for (AsyncRpcChannel conn : connections.values()) {
345        conn.close(null);
346      }
347    }
348    // do not close global EventLoopGroup.
349    if (!useGlobalEventLoopGroup) {
350      bootstrap.config().group().shutdownGracefully();
351    }
352  }

376  /**
377   * Creates an RPC client
378   *
379   * @param serviceName    name of service
380   * @param location       to connect to
381   * @param ticket         for current user
382   * @return new RpcChannel
383   * @throws StoppedRpcClientException when Rpc client is stopped
384   * @throws FailedServerException if server failed
385   */
386  private AsyncRpcChannel createRpcChannel(String serviceName, InetSocketAddress location,
387      User ticket) throws StoppedRpcClientException, FailedServerException {
388    // Check if server is failed
389    if (this.failedServers.isFailedServer(location)) {
390      if (LOG.isDebugEnabled()) {
391        LOG.debug("Not trying to connect to " + location +
392            " this server is in the failed servers list");
393      }
394      throw new FailedServerException(
395          "This server is in the failed servers list: " + location);
396    }
397
398    int hashCode = ConnectionId.hashCode(ticket,serviceName,location);
399
400    AsyncRpcChannel rpcChannel;
401    synchronized (connections) {
402      if (closed) {
403        throw new StoppedRpcClientException();
404      }
405      rpcChannel = connections.get(hashCode);
406      if (rpcChannel != null && !rpcChannel.isAlive()) {
407        LOG.debug("Removing dead channel from server="+rpcChannel.getAddress().toString());
408        connections.remove(hashCode);
409      }
410      if (rpcChannel == null) {
411        rpcChannel = new AsyncRpcChannel(this.bootstrap, this, ticket, serviceName, location);
412        connections.put(hashCode, rpcChannel);
413      }
414    }
415
416    return rpcChannel;
417  }

419  /**
420   * Interrupt the connections to the given ip:port server. This should be called if the server
421   * is known as actually dead. This will not prevent current operation to be retried, and,
422   * depending on their own behavior, they may retry on the same server. This can be a feature,
423   * for example at startup. In any case, they're likely to get connection refused (if the
424   * process died) or no route to host: i.e. there next retries should be faster and with a
425   * safe exception.
426   *
427   * @param sn server to cancel connections for
428   */
429  @Override
430  public void cancelConnections(ServerName sn) {
431    synchronized (connections) {
432      for (AsyncRpcChannel rpcChannel : connections.values()) {
433        if (rpcChannel.isAlive() &&
434            rpcChannel.getAddress().getPort() == sn.getPort() &&
435            rpcChannel.getAddress().getHostName().contentEquals(sn.getHostname())) {
436          LOG.info("The server on " + sn.toString() +
437              " is dead - stopping the connection " + rpcChannel.toString());
438          rpcChannel.close(null);
439        }
440      }
441    }
442  }

444  /**
445   * Remove connection from pool
446   * @param connection to remove
447   */
448  public void removeConnection(AsyncRpcChannel connection) {
449    int connectionHashCode = connection.hashCode();
450    synchronized (connections) {
451      // we use address as cache key, so we should check here to prevent removing the
452      // wrong connection
453      AsyncRpcChannel connectionInPool = this.connections.get(connectionHashCode);
454      if (connectionInPool != null && connectionInPool.equals(connection)) {
455        this.connections.remove(connectionHashCode);
456      } else if (LOG.isDebugEnabled()) {
457        LOG.debug(String.format("%s already removed, expected instance %08x, actual %08x",
458          connection.toString(), System.identityHashCode(connection),
459          System.identityHashCode(connectionInPool)));
460      }
461    }
462  }

469  /**
470   * Blocking rpc channel that goes via hbase rpc.
471   */
472  @VisibleForTesting
473  public static class RpcChannelImplementation implements RpcChannel {
474    private final InetSocketAddress isa;
475    private final AsyncRpcClient rpcClient;
476    private final User ticket;
477    private final int channelOperationTimeout;
478
479    /**
480     * @param channelOperationTimeout - the default timeout when no timeout is given
481     */
482    protected RpcChannelImplementation(final AsyncRpcClient rpcClient,
483        final ServerName sn, final User ticket, int channelOperationTimeout) {
484      this.isa = new InetSocketAddress(sn.getHostname(), sn.getPort());
485      this.rpcClient = rpcClient;
486      this.ticket = ticket;
487      this.channelOperationTimeout = channelOperationTimeout;
488    }
489
490    @Override
491    public void callMethod(Descriptors.MethodDescriptor md, RpcController controller,
492        Message param, Message returnType, RpcCallback<Message> done) {
493      PayloadCarryingRpcController pcrc =
494          configurePayloadCarryingRpcController(controller, channelOperationTimeout);
495
496      this.rpcClient.callMethod(md, pcrc, param, returnType, this.ticket, this.isa, done);
497    }
498  }
