403    /**
404     * Set up cipher for rpc encryption with Apache Commons Crypto
405     *
406     * @throws FatalConnectionException
407     */
408    protected void setupCryptoCipher(final ConnectionHeader header,
409        RPCProtos.ConnectionHeaderResponse.Builder chrBuilder)
410        throws FatalConnectionException {
411      // If simple auth, return
412      if (saslServer == null) return;
413      // check if rpc encryption with Crypto AES
414      String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
415      boolean isEncryption = SaslUtil.QualityOfProtection.PRIVACY
416          .getSaslQop().equalsIgnoreCase(qop);
417      boolean isCryptoAesEncryption = isEncryption && conf.getBoolean(
418          "hbase.rpc.crypto.encryption.aes.enabled", false);
419      if (!isCryptoAesEncryption) return;
420      if (!header.hasRpcCryptoCipherTransformation()) return;
421      String transformation = header.getRpcCryptoCipherTransformation();
422      if (transformation == null || transformation.length() == 0) return;
423       // Negotiates AES based on complete saslServer.
424       // The Crypto metadata need to be encrypted and send to client.
425      Properties properties = new Properties();
426      // the property for SecureRandomFactory
427      properties.setProperty(CryptoRandomFactory.CLASSES_KEY,
428          conf.get("hbase.crypto.sasl.encryption.aes.crypto.random",
429              "org.apache.commons.crypto.random.JavaCryptoRandom"));
430      // the property for cipher class
431      properties.setProperty(CryptoCipherFactory.CLASSES_KEY,
432          conf.get("hbase.rpc.crypto.encryption.aes.cipher.class",
433              "org.apache.commons.crypto.cipher.JceCipher"));
434
435      int cipherKeyBits = conf.getInt(
436          "hbase.rpc.crypto.encryption.aes.cipher.keySizeBits", 128);
437      // generate key and iv
438      if (cipherKeyBits % 8 != 0) {
439        throw new IllegalArgumentException("The AES cipher key size in bits" +
440            " should be a multiple of byte");
441      }
442      int len = cipherKeyBits / 8;
443      byte[] inKey = new byte[len];
444      byte[] outKey = new byte[len];
445      byte[] inIv = new byte[len];
446      byte[] outIv = new byte[len];
447
448      try {
449        // generate the cipher meta data with SecureRandom
450        CryptoRandom secureRandom = CryptoRandomFactory.getCryptoRandom(properties);
451        secureRandom.nextBytes(inKey);
452        secureRandom.nextBytes(outKey);
453        secureRandom.nextBytes(inIv);
454        secureRandom.nextBytes(outIv);
455
456        // create CryptoAES for server
457        cryptoAES = new CryptoAES(transformation, properties,
458            inKey, outKey, inIv, outIv);
459        // create SaslCipherMeta and send to client,
460        //  for client, the [inKey, outKey], [inIv, outIv] should be reversed
461        RPCProtos.CryptoCipherMeta.Builder ccmBuilder = RPCProtos.CryptoCipherMeta.newBuilder();
462        ccmBuilder.setTransformation(transformation);
463        ccmBuilder.setInIv(getByteString(outIv));
464        ccmBuilder.setInKey(getByteString(outKey));
465        ccmBuilder.setOutIv(getByteString(inIv));
466        ccmBuilder.setOutKey(getByteString(inKey));
467        chrBuilder.setCryptoCipherMeta(ccmBuilder);
468        useCryptoAesWrap = true;
469      } catch (GeneralSecurityException | IOException ex) {
470        throw new UnsupportedCryptoException(ex.getMessage(), ex);
471      }
472    }

520    protected void doRawSaslReply(SaslStatus status, Writable rv,
521        String errorClass, String error) throws IOException {
522      ByteBufferOutputStream saslResponse = null;
523      DataOutputStream out = null;
524      try {
525        // In my testing, have noticed that sasl messages are usually
526        // in the ballpark of 100-200. That's why the initial capacity is 256.
527        saslResponse = new ByteBufferOutputStream(256);
528        out = new DataOutputStream(saslResponse);
529        out.writeInt(status.state); // write status
530        if (status == SaslStatus.SUCCESS) {
531          rv.write(out);
532        } else {
533          WritableUtils.writeString(out, errorClass);
534          WritableUtils.writeString(out, error);
535        }
536        saslCall.setSaslTokenResponse(saslResponse.getByteBuffer());
537        saslCall.sendResponseIfReady();
538      } finally {
539        if (saslResponse != null) {
540          saslResponse.close();
541        }
542        if (out != null) {
543          out.close();
544        }
545      }

548    public void saslReadAndProcess(ByteBuff saslToken) throws IOException,
549        InterruptedException {
550      if (saslContextEstablished) {
551        if (LOG.isTraceEnabled())
552          LOG.trace("Have read input token of size " + saslToken.limit()
553              + " for processing by saslServer.unwrap()");
554
555        if (!useWrap) {
556          processOneRpc(saslToken);
557        } else {
558          byte[] b = saslToken.hasArray() ? saslToken.array() : saslToken.toBytes();
559          byte [] plaintextData;
560          if (useCryptoAesWrap) {
561            // unwrap with CryptoAES
562            plaintextData = cryptoAES.unwrap(b, 0, b.length);
563          } else {
564            plaintextData = saslServer.unwrap(b, 0, b.length);
565          }
566          processUnwrappedData(plaintextData);
567        }
568      } else {
569        byte[] replyToken;
570        try {
571          if (saslServer == null) {
572            switch (authMethod) {
573            case DIGEST:
574              if (secretManager == null) {
575                throw new AccessDeniedException(
576                    "Server is not configured to do DIGEST authentication.");
577              }
578              saslServer = Sasl.createSaslServer(AuthMethod.DIGEST
579                  .getMechanismName(), null, SaslUtil.SASL_DEFAULT_REALM,
580                  HBaseSaslRpcServer.getSaslProps(), new SaslDigestCallbackHandler(
581                      secretManager, this));
582              break;
583            default:
584              UserGroupInformation current = UserGroupInformation.getCurrentUser();
585              String fullName = current.getUserName();
586              if (LOG.isDebugEnabled()) {
587                LOG.debug("Kerberos principal name is " + fullName);
588              }
589              final String names[] = SaslUtil.splitKerberosName(fullName);
590              if (names.length != 3) {
591                throw new AccessDeniedException(
592                    "Kerberos principal name does NOT have the expected "
593                        + "hostname part: " + fullName);
594              }
595              current.doAs(new PrivilegedExceptionAction<Object>() {
596                @Override
597                public Object run() throws SaslException {
598                  saslServer = Sasl.createSaslServer(AuthMethod.KERBEROS
599                      .getMechanismName(), names[0], names[1],
600                      HBaseSaslRpcServer.getSaslProps(), new SaslGssCallbackHandler());
601                  return null;
602                }
603              });
604            }
605            if (saslServer == null)
606              throw new AccessDeniedException(
607                  "Unable to find SASL server implementation for "
608                      + authMethod.getMechanismName());
609            if (LOG.isDebugEnabled()) {
610              LOG.debug("Created SASL server with mechanism = " + authMethod.getMechanismName());
611            }
612          }
613          if (LOG.isDebugEnabled()) {
614            LOG.debug("Have read input token of size " + saslToken.limit()
615                + " for processing by saslServer.evaluateResponse()");
616          }
617          replyToken = saslServer
618              .evaluateResponse(saslToken.hasArray() ? saslToken.array() : saslToken.toBytes());
619        } catch (IOException e) {
620          IOException sendToClient = e;
621          Throwable cause = e;
622          while (cause != null) {
623            if (cause instanceof InvalidToken) {
624              sendToClient = (InvalidToken) cause;
625              break;
626            }
627            cause = cause.getCause();
628          }
629          doRawSaslReply(SaslStatus.ERROR, null, sendToClient.getClass().getName(),
630            sendToClient.getLocalizedMessage());
631          metrics.authenticationFailure();
632          String clientIP = this.toString();
633          // attempting user could be null
634          AUDITLOG.warn(AUTH_FAILED_FOR + clientIP + ":" + attemptingUser);
635          throw e;
636        }
637        if (replyToken != null) {
638          if (LOG.isDebugEnabled()) {
639            LOG.debug("Will send token of size " + replyToken.length
640                + " from saslServer.");
641          }
642          doRawSaslReply(SaslStatus.SUCCESS, new BytesWritable(replyToken), null,
643              null);
644        }
645        if (saslServer.isComplete()) {
646          String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
647          useWrap = qop != null && !"auth".equalsIgnoreCase(qop);
648          ugi = getAuthorizedUgi(saslServer.getAuthorizationID());
649          if (LOG.isDebugEnabled()) {
650            LOG.debug("SASL server context established. Authenticated client: "
651              + ugi + ". Negotiated QoP is "
652              + saslServer.getNegotiatedProperty(Sasl.QOP));
653          }
654          metrics.authenticationSuccess();
655          AUDITLOG.info(AUTH_SUCCESSFUL_FOR + ugi);
656          saslContextEstablished = true;
657        }
658      }
659    }

661    private void processUnwrappedData(byte[] inBuf) throws IOException,
662    InterruptedException {
663      ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(inBuf));
664      // Read all RPCs contained in the inBuf, even partial ones
665      while (true) {
666        int count;
667        if (unwrappedDataLengthBuffer.remaining() > 0) {
668          count = channelRead(ch, unwrappedDataLengthBuffer);
669          if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0)
670            return;
671        }
672
673        if (unwrappedData == null) {
674          unwrappedDataLengthBuffer.flip();
675          int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
676
677          if (unwrappedDataLength == RpcClient.PING_CALL_ID) {
678            if (LOG.isDebugEnabled())
679              LOG.debug("Received ping message");
680            unwrappedDataLengthBuffer.clear();
681            continue; // ping message
682          }
683          unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
684        }
685
686        count = channelRead(ch, unwrappedData);
687        if (count <= 0 || unwrappedData.remaining() > 0)
688          return;
689
690        if (unwrappedData.remaining() == 0) {
691          unwrappedDataLengthBuffer.clear();
692          unwrappedData.flip();
693          processOneRpc(new SingleByteBuff(unwrappedData));
694          unwrappedData = null;
695        }
696      }
697    }

699    public void processOneRpc(ByteBuff buf) throws IOException,
700        InterruptedException {
701      if (connectionHeaderRead) {
702        processRequest(buf);
703      } else {
704        processConnectionHeader(buf);
705        this.connectionHeaderRead = true;
706        if (!authorizeConnection()) {
707          // Throw FatalConnectionException wrapping ACE so client does right thing and closes
708          // down the connection instead of trying to read non-existent retun.
709          throw new AccessDeniedException("Connection from " + this + " for service " +
710            connectionHeader.getServiceName() + " is unauthorized for user: " + ugi);
711        }
712        this.user = userProvider.create(this.ugi);
713      }
714    }

716    protected boolean authorizeConnection() throws IOException {
717      try {
718        // If auth method is DIGEST, the token was obtained by the
719        // real user for the effective user, therefore not required to
720        // authorize real user. doAs is allowed only for simple or kerberos
721        // authentication
722        if (ugi != null && ugi.getRealUser() != null
723            && (authMethod != AuthMethod.DIGEST)) {
724          ProxyUsers.authorize(ugi, this.getHostAddress(), conf);
725        }
726        authorize(ugi, connectionHeader, getHostInetAddress());
727        metrics.authorizationSuccess();
728      } catch (AuthorizationException ae) {
729        if (LOG.isDebugEnabled()) {
730          LOG.debug("Connection authorization failed: " + ae.getMessage(), ae);
731        }
732        metrics.authorizationFailure();
733        setupResponse(authFailedResponse, authFailedCall,
734          new AccessDeniedException(ae), ae.getMessage());
735        authFailedCall.sendResponseIfReady();
736        return false;
737      }
738      return true;
739    }

741    // Reads the connection header following version
742    protected void processConnectionHeader(ByteBuff buf) throws IOException {
743      if (buf.hasArray()) {
744        this.connectionHeader = ConnectionHeader.parseFrom(buf.array());
745      } else {
746        CodedInputStream cis = UnsafeByteOperations
747            .unsafeWrap(new ByteBuffByteInput(buf, 0, buf.limit()), 0, buf.limit()).newCodedInput();
748        cis.enableAliasing(true);
749        this.connectionHeader = ConnectionHeader.parseFrom(cis);
750      }
751      String serviceName = connectionHeader.getServiceName();
752      if (serviceName == null) throw new EmptyServiceNameException();
753      this.service = getService(services, serviceName);
754      if (this.service == null) throw new UnknownServiceException(serviceName);
755      setupCellBlockCodecs(this.connectionHeader);
756      RPCProtos.ConnectionHeaderResponse.Builder chrBuilder =
757          RPCProtos.ConnectionHeaderResponse.newBuilder();
758      setupCryptoCipher(this.connectionHeader, chrBuilder);
759      responseConnectionHeader(chrBuilder);
760      UserGroupInformation protocolUser = createUser(connectionHeader);
761      if (!useSasl) {
762        ugi = protocolUser;
763        if (ugi != null) {
764          ugi.setAuthenticationMethod(AuthMethod.SIMPLE.authenticationMethod);
765        }
766        // audit logging for SASL authenticated users happens in saslReadAndProcess()
767        if (authenticatedWithFallback) {
768          LOG.warn("Allowed fallback to SIMPLE auth for " + ugi
769              + " connecting from " + getHostAddress());
770        }
771        AUDITLOG.info(AUTH_SUCCESSFUL_FOR + ugi);
772      } else {
773        // user is authenticated
774        ugi.setAuthenticationMethod(authMethod.authenticationMethod);
775        //Now we check if this is a proxy user case. If the protocol user is
776        //different from the 'user', it is a proxy user scenario. However,
777        //this is not allowed if user authenticated with DIGEST.
778        if ((protocolUser != null)
779            && (!protocolUser.getUserName().equals(ugi.getUserName()))) {
780          if (authMethod == AuthMethod.DIGEST) {
781            // Not allowed to doAs if token authentication is used
782            throw new AccessDeniedException("Authenticated user (" + ugi
783                + ") doesn't match what the client claims to be ("
784                + protocolUser + ")");
785          } else {
786            // Effective user can be different from authenticated user
787            // for simple auth or kerberos auth
788            // The user is the real user. Now we create a proxy user
789            UserGroupInformation realUser = ugi;
790            ugi = UserGroupInformation.createProxyUser(protocolUser
791                .getUserName(), realUser);
792            // Now the user is a proxy user, set Authentication method Proxy.
793            ugi.setAuthenticationMethod(AuthenticationMethod.PROXY);
794          }
795        }
796      }
797      if (connectionHeader.hasVersionInfo()) {
798        // see if this connection will support RetryImmediatelyException
799        retryImmediatelySupported = VersionInfoUtil.hasMinimumVersion(getVersionInfo(), 1, 2);
800
801        AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
802            + " with version info: "
803            + TextFormat.shortDebugString(connectionHeader.getVersionInfo()));
804      } else {
805        AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
806            + " with unknown version info");
807      }
808    }

810    private void responseConnectionHeader(RPCProtos.ConnectionHeaderResponse.Builder chrBuilder)
811        throws FatalConnectionException {
812      // Response the connection header if Crypto AES is enabled
813      if (!chrBuilder.hasCryptoCipherMeta()) return;
814      try {
815        byte[] connectionHeaderResBytes = chrBuilder.build().toByteArray();
816        // encrypt the Crypto AES cipher meta data with sasl server, and send to client
817        byte[] unwrapped = new byte[connectionHeaderResBytes.length + 4];
818        Bytes.putBytes(unwrapped, 0, Bytes.toBytes(connectionHeaderResBytes.length), 0, 4);
819        Bytes.putBytes(unwrapped, 4, connectionHeaderResBytes, 0, connectionHeaderResBytes.length);
820
821        doConnectionHeaderResponse(saslServer.wrap(unwrapped, 0, unwrapped.length));
822      } catch (IOException ex) {
823        throw new UnsupportedCryptoException(ex.getMessage(), ex);
824      }
825    }

853    /**
854     * @param buf
855     *          Has the request header and the request param and optionally
856     *          encoded data buffer all in this one array.
857     * @throws IOException
858     * @throws InterruptedException
859     */
860    protected void processRequest(ByteBuff buf) throws IOException,
861        InterruptedException {
862      long totalRequestSize = buf.limit();
863      int offset = 0;
864      // Here we read in the header. We avoid having pb
865      // do its default 4k allocation for CodedInputStream. We force it to use
866      // backing array.
867      CodedInputStream cis;
868      if (buf.hasArray()) {
869        cis = UnsafeByteOperations.unsafeWrap(buf.array(), 0, buf.limit())
870            .newCodedInput();
871      } else {
872        cis = UnsafeByteOperations.unsafeWrap(
873            new ByteBuffByteInput(buf, 0, buf.limit()), 0, buf.limit())
874            .newCodedInput();
875      }
876      cis.enableAliasing(true);
877      int headerSize = cis.readRawVarint32();
878      offset = cis.getTotalBytesRead();
879      Message.Builder builder = RequestHeader.newBuilder();
880      ProtobufUtil.mergeFrom(builder, cis, headerSize);
881      RequestHeader header = (RequestHeader) builder.build();
882      offset += headerSize;
883      int id = header.getCallId();
884      if (LOG.isTraceEnabled()) {
885        LOG.trace("RequestHeader " + TextFormat.shortDebugString(header)
886            + " totalRequestSize: " + totalRequestSize + " bytes");
887      }
888      // Enforcing the call queue size, this triggers a retry in the client
889      // This is a bit late to be doing this check - we have already read in the
890      // total request.
891      if ((totalRequestSize + callQueueSizeInBytes.sum()) > maxQueueSizeInBytes) {
892        final ServerCall callTooBig = createCall(id, this.service, null,
893            null, null, null, this, totalRequestSize, null, null, 0,
894            this.callCleanup);
895        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
896        metrics.exception(CALL_QUEUE_TOO_BIG_EXCEPTION);
897        setupResponse(responseBuffer, callTooBig, CALL_QUEUE_TOO_BIG_EXCEPTION,
898            "Call queue is full on " + server.getServerName()
899                + ", is hbase.ipc.server.max.callqueue.size too small?");
900        callTooBig.sendResponseIfReady();
901        return;
902      }
903      MethodDescriptor md = null;
904      Message param = null;
905      CellScanner cellScanner = null;
906      try {
907        if (header.hasRequestParam() && header.getRequestParam()) {
908          md = this.service.getDescriptorForType().findMethodByName(
909              header.getMethodName());
910          if (md == null)
911            throw new UnsupportedOperationException(header.getMethodName());
912          builder = this.service.getRequestPrototype(md).newBuilderForType();
913          cis.resetSizeCounter();
914          int paramSize = cis.readRawVarint32();
915          offset += cis.getTotalBytesRead();
916          if (builder != null) {
917            ProtobufUtil.mergeFrom(builder, cis, paramSize);
918            param = builder.build();
919          }
920          offset += paramSize;
921        } else {
922          // currently header must have request param, so we directly throw
923          // exception here
924          String msg = "Invalid request header: "
925              + TextFormat.shortDebugString(header)
926              + ", should have param set in it";
927          LOG.warn(msg);
928          throw new DoNotRetryIOException(msg);
929        }
930        if (header.hasCellBlockMeta()) {
931          buf.position(offset);
932          ByteBuff dup = buf.duplicate();
933          dup.limit(offset + header.getCellBlockMeta().getLength());
934          cellScanner = cellBlockBuilder.createCellScannerReusingBuffers(
935              this.codec, this.compressionCodec, dup);
936        }
937      } catch (Throwable t) {
938        InetSocketAddress address = getListenerAddress();
939        String msg = (address != null ? address : "(channel closed)")
940            + " is unable to read call parameter from client "
941            + getHostAddress();
942        LOG.warn(msg, t);
943
944        metrics.exception(t);
945
946        // probably the hbase hadoop version does not match the running hadoop
947        // version
948        if (t instanceof LinkageError) {
949          t = new DoNotRetryIOException(t);
950        }
951        // If the method is not present on the server, do not retry.
952        if (t instanceof UnsupportedOperationException) {
953          t = new DoNotRetryIOException(t);
954        }
955
956        final ServerCall readParamsFailedCall = createCall(id,
957            this.service, null, null, null, null, this, totalRequestSize, null,
958            null, 0, this.callCleanup);
959        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
960        setupResponse(responseBuffer, readParamsFailedCall, t,
961            msg + "; " + t.getMessage());
962        readParamsFailedCall.sendResponseIfReady();
963        return;
964      }
965
966      TraceInfo traceInfo = header.hasTraceInfo() ? new TraceInfo(header
967          .getTraceInfo().getTraceId(), header.getTraceInfo().getParentId())
968          : null;
969      int timeout = 0;
970      if (header.hasTimeout() && header.getTimeout() > 0) {
971        timeout = Math.max(minClientRequestTimeout, header.getTimeout());
972      }
973      ServerCall call = createCall(id, this.service, md, header, param,
974          cellScanner, this, totalRequestSize, traceInfo, this.addr, timeout,
975          this.callCleanup);
976
977      if (!scheduler.dispatch(new CallRunner(RpcServer.this, call))) {
978        callQueueSizeInBytes.add(-1 * call.getSize());
979
980        ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
981        metrics.exception(CALL_QUEUE_TOO_BIG_EXCEPTION);
982        setupResponse(responseBuffer, call, CALL_QUEUE_TOO_BIG_EXCEPTION,
983            "Call queue is full on " + server.getServerName()
984                + ", too many items queued ?");
985        call.sendResponseIfReady();
986      }
987    }

1020  /**
1021   * Constructs a server listening on the named port and address.
1022   * @param server hosting instance of {@link Server}. We will do authentications if an
1023   * instance else pass null for no authentication check.
1024   * @param name Used keying this rpc servers' metrics and for naming the Listener thread.
1025   * @param services A list of services.
1026   * @param bindAddress Where to listen
1027   * @param conf
1028   * @param scheduler
1029   */
1030  public RpcServer(final Server server, final String name,
1031      final List<BlockingServiceAndInterface> services,
1032      final InetSocketAddress bindAddress, Configuration conf,
1033      RpcScheduler scheduler)
1034      throws IOException {
1035    if (conf.getBoolean("hbase.ipc.server.reservoir.enabled", true)) {
1036      int poolBufSize = conf.getInt(ByteBufferPool.BUFFER_SIZE_KEY,
1037          ByteBufferPool.DEFAULT_BUFFER_SIZE);
1038      // The max number of buffers to be pooled in the ByteBufferPool. The default value been
1039      // selected based on the #handlers configured. When it is read request, 2 MB is the max size
1040      // at which we will send back one RPC request. Means max we need 2 MB for creating the
1041      // response cell block. (Well it might be much lesser than this because in 2 MB size calc, we
1042      // include the heap size overhead of each cells also.) Considering 2 MB, we will need
1043      // (2 * 1024 * 1024) / poolBufSize buffers to make the response cell block. Pool buffer size
1044      // is by default 64 KB.
1045      // In case of read request, at the end of the handler process, we will make the response
1046      // cellblock and add the Call to connection's response Q and a single Responder thread takes
1047      // connections and responses from that one by one and do the socket write. So there is chances
1048      // that by the time a handler originated response is actually done writing to socket and so
1049      // released the BBs it used, the handler might have processed one more read req. On an avg 2x
1050      // we consider and consider that also for the max buffers to pool
1051      int bufsForTwoMB = (2 * 1024 * 1024) / poolBufSize;
1052      int maxPoolSize = conf.getInt(ByteBufferPool.MAX_POOL_SIZE_KEY,
1053          conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT,
1054              HConstants.DEFAULT_REGION_SERVER_HANDLER_COUNT) * bufsForTwoMB * 2);
1055      this.reservoir = new ByteBufferPool(poolBufSize, maxPoolSize);
1056      this.minSizeForReservoirUse = getMinSizeForReservoirUse(this.reservoir);
1057    } else {
1058      reservoir = null;
1059      this.minSizeForReservoirUse = Integer.MAX_VALUE;// reservoir itself not in place.
1060    }
1061    this.server = server;
1062    this.services = services;
1063    this.bindAddress = bindAddress;
1064    this.conf = conf;
1065    // See declaration above for documentation on what this size is.
1066    this.maxQueueSizeInBytes =
1067      this.conf.getLong("hbase.ipc.server.max.callqueue.size", DEFAULT_MAX_CALLQUEUE_SIZE);
1068
1069    this.warnResponseTime = conf.getInt(WARN_RESPONSE_TIME, DEFAULT_WARN_RESPONSE_TIME);
1070    this.warnResponseSize = conf.getInt(WARN_RESPONSE_SIZE, DEFAULT_WARN_RESPONSE_SIZE);
1071    this.minClientRequestTimeout = conf.getInt(MIN_CLIENT_REQUEST_TIMEOUT,
1072        DEFAULT_MIN_CLIENT_REQUEST_TIMEOUT);
1073    this.maxRequestSize = conf.getInt(MAX_REQUEST_SIZE, DEFAULT_MAX_REQUEST_SIZE);
1074
1075    this.metrics = new MetricsHBaseServer(name, new MetricsHBaseServerWrapperImpl(this));
1076    this.tcpNoDelay = conf.getBoolean("hbase.ipc.server.tcpnodelay", true);
1077    this.tcpKeepAlive = conf.getBoolean("hbase.ipc.server.tcpkeepalive", true);
1078
1079    this.cellBlockBuilder = new CellBlockBuilder(conf);
1080
1081    this.authorize = conf.getBoolean(HADOOP_SECURITY_AUTHORIZATION, false);
1082    this.userProvider = UserProvider.instantiate(conf);
1083    this.isSecurityEnabled = userProvider.isHBaseSecurityEnabled();
1084    if (isSecurityEnabled) {
1085      HBaseSaslRpcServer.init(conf);
1086    }
1087
1088    this.scheduler = scheduler;
1089  }

1168  /**
1169   * This is a server side method, which is invoked over RPC. On success
1170   * the return response has protobuf response payload. On failure, the
1171   * exception name and the stack trace are returned in the protobuf response.
1172   */
1173  @Override
1174  public Pair<Message, CellScanner> call(RpcCall call,
1175      MonitoredRPCHandler status) throws IOException {
1176    try {
1177      MethodDescriptor md = call.getMethod();
1178      Message param = call.getParam();
1179      status.setRPC(md.getName(), new Object[]{param},
1180        call.getReceiveTime());
1181      // TODO: Review after we add in encoded data blocks.
1182      status.setRPCPacket(param);
1183      status.resume("Servicing call");
1184      //get an instance of the method arg type
1185      HBaseRpcController controller = new HBaseRpcControllerImpl(call.getCellScanner());
1186      controller.setCallTimeout(call.getTimeout());
1187      Message result = call.getService().callBlockingMethod(md, controller, param);
1188      long receiveTime = call.getReceiveTime();
1189      long startTime = call.getStartTime();
1190      long endTime = System.currentTimeMillis();
1191      int processingTime = (int) (endTime - startTime);
1192      int qTime = (int) (startTime - receiveTime);
1193      int totalTime = (int) (endTime - receiveTime);
1194      if (LOG.isTraceEnabled()) {
1195        LOG.trace(CurCall.get().toString() +
1196            ", response " + TextFormat.shortDebugString(result) +
1197            " queueTime: " + qTime +
1198            " processingTime: " + processingTime +
1199            " totalTime: " + totalTime);
1200      }
1201      // Use the raw request call size for now.
1202      long requestSize = call.getSize();
1203      long responseSize = result.getSerializedSize();
1204      if (call.isClientCellBlockSupported()) {
1205        // Include the payload size in HBaseRpcController
1206        responseSize += call.getResponseCellSize();
1207      }
1208
1209      metrics.dequeuedCall(qTime);
1210      metrics.processedCall(processingTime);
1211      metrics.totalCall(totalTime);
1212      metrics.receivedRequest(requestSize);
1213      metrics.sentResponse(responseSize);
1214      // log any RPC responses that are slower than the configured warn
1215      // response time or larger than configured warning size
1216      boolean tooSlow = (processingTime > warnResponseTime && warnResponseTime > -1);
1217      boolean tooLarge = (responseSize > warnResponseSize && warnResponseSize > -1);
1218      if (tooSlow || tooLarge) {
1219        // when tagging, we let TooLarge trump TooSmall to keep output simple
1220        // note that large responses will often also be slow.
1221        logResponse(param,
1222            md.getName(), md.getName() + "(" + param.getClass().getName() + ")",
1223            (tooLarge ? "TooLarge" : "TooSlow"),
1224            status.getClient(), startTime, processingTime, qTime,
1225            responseSize);
1226      }
1227      return new Pair<>(result, controller.cellScanner());
1228    } catch (Throwable e) {
1229      // The above callBlockingMethod will always return a SE.  Strip the SE wrapper before
1230      // putting it on the wire.  Its needed to adhere to the pb Service Interface but we don't
1231      // need to pass it over the wire.
1232      if (e instanceof ServiceException) {
1233        if (e.getCause() == null) {
1234          LOG.debug("Caught a ServiceException with null cause", e);
1235        } else {
1236          e = e.getCause();
1237        }
1238      }
1239
1240      // increment the number of requests that were exceptions.
1241      metrics.exception(e);
1242
1243      if (e instanceof LinkageError) throw new DoNotRetryIOException(e);
1244      if (e instanceof IOException) throw (IOException)e;
1245      LOG.error("Unexpected throwable object ", e);
1246      throw new IOException(e.getMessage(), e);
1247    }
1248  }

1336  /**
1337   * When the read or write buffer size is larger than this limit, i/o will be
1338   * done in chunks of this size. Most RPC requests and responses would be
1339   * be smaller.
1340   */
1341  protected static final int NIO_BUFFER_LIMIT = 64 * 1024; //should not be more than 64KB.
1342
1343  /**
1344   * This is a wrapper around {@link java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)}.
1345   * If the amount of data is large, it writes to channel in smaller chunks.
1346   * This is to avoid jdk from creating many direct buffers as the size of
1347   * ByteBuffer increases. There should not be any performance degredation.
1348   *
1349   * @param channel writable byte channel to write on
1350   * @param buffer buffer to write
1351   * @return number of bytes written
1352   * @throws java.io.IOException e
1353   * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
1354   */
1355  protected int channelRead(ReadableByteChannel channel,
1356                                   ByteBuffer buffer) throws IOException {
1357
1358    int count = (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
1359           channel.read(buffer) : channelIO(channel, null, buffer);
1360    if (count > 0) {
1361      metrics.receivedBytes(count);
1362    }
1363    return count;
1364  }

1366  /**
1367   * Helper for {@link #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)}
1368   * and {@link #channelWrite(GatheringByteChannel, BufferChain)}. Only
1369   * one of readCh or writeCh should be non-null.
1370   *
1371   * @param readCh read channel
1372   * @param writeCh write channel
1373   * @param buf buffer to read or write into/out of
1374   * @return bytes written
1375   * @throws java.io.IOException e
1376   * @see #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)
1377   * @see #channelWrite(GatheringByteChannel, BufferChain)
1378   */
1379  private static int channelIO(ReadableByteChannel readCh,
1380                               WritableByteChannel writeCh,
1381                               ByteBuffer buf) throws IOException {
1382
1383    int originalLimit = buf.limit();
1384    int initialRemaining = buf.remaining();
1385    int ret = 0;
1386
1387    while (buf.remaining() > 0) {
1388      try {
1389        int ioSize = Math.min(buf.remaining(), NIO_BUFFER_LIMIT);
1390        buf.limit(buf.position() + ioSize);
1391
1392        ret = (readCh == null) ? writeCh.write(buf) : readCh.read(buf);
1393
1394        if (ret < ioSize) {
1395          break;
1396        }
1397
1398      } finally {
1399        buf.limit(originalLimit);
1400      }
1401    }
1402
1403    int nBytes = initialRemaining - buf.remaining();
1404    return (nBytes > 0) ? nBytes : ret;
1405  }

1407  /**
1408   * This is extracted to a static method for better unit testing. We try to get buffer(s) from pool
1409   * as much as possible.
1410   *
1411   * @param pool The ByteBufferPool to use
1412   * @param minSizeForPoolUse Only for buffer size above this, we will try to use pool. Any buffer
1413   *           need of size below this, create on heap ByteBuffer.
1414   * @param reqLen Bytes count in request
1415   */
1416  @VisibleForTesting
1417  static Pair<ByteBuff, CallCleanup> allocateByteBuffToReadInto(ByteBufferPool pool,
1418      int minSizeForPoolUse, int reqLen) {
1419    ByteBuff resultBuf;
1420    List<ByteBuffer> bbs = new ArrayList<>((reqLen / pool.getBufferSize()) + 1);
1421    int remain = reqLen;
1422    ByteBuffer buf = null;
1423    while (remain >= minSizeForPoolUse && (buf = pool.getBuffer()) != null) {
1424      bbs.add(buf);
1425      remain -= pool.getBufferSize();
1426    }
1427    ByteBuffer[] bufsFromPool = null;
1428    if (bbs.size() > 0) {
1429      bufsFromPool = new ByteBuffer[bbs.size()];
1430      bbs.toArray(bufsFromPool);
1431    }
1432    if (remain > 0) {
1433      bbs.add(ByteBuffer.allocate(remain));
1434    }
1435    if (bbs.size() > 1) {
1436      ByteBuffer[] items = new ByteBuffer[bbs.size()];
1437      bbs.toArray(items);
1438      resultBuf = new MultiByteBuff(items);
1439    } else {
1440      // We are backed by single BB
1441      resultBuf = new SingleByteBuff(bbs.get(0));
1442    }
1443    resultBuf.limit(reqLen);
1444    if (bufsFromPool != null) {
1445      final ByteBuffer[] bufsFromPoolFinal = bufsFromPool;
1446      return new Pair<>(resultBuf, () -> {
1447        // Return back all the BBs to pool
1448        for (int i = 0; i < bufsFromPoolFinal.length; i++) {
1449          pool.putbackBuffer(bufsFromPoolFinal[i]);
1450        }
1451      });
1452    }
1453    return new Pair<>(resultBuf, null);
1454  }
