020package org.apache.hadoop.hbase.ipc;
021
022import org.apache.hadoop.hbase.classification.InterfaceAudience;
023import org.apache.hadoop.hbase.metrics.BaseSourceImpl;
024import org.apache.hadoop.metrics2.MetricsCollector;
025import org.apache.hadoop.metrics2.MetricsRecordBuilder;
026import org.apache.hadoop.metrics2.lib.Interns;
027import org.apache.hadoop.metrics2.lib.MutableCounterLong;
028import org.apache.hadoop.metrics2.lib.MutableHistogram;
029
030@InterfaceAudience.Private
031public class MetricsHBaseServerSourceImpl extends BaseSourceImpl
032    implements MetricsHBaseServerSource {
033
034  private final MetricsHBaseServerWrapper wrapper;
035  private final MutableCounterLong authorizationSuccesses;
036  private final MutableCounterLong authorizationFailures;
037  private final MutableCounterLong authenticationSuccesses;
038  private final MutableCounterLong authenticationFailures;
039  private final MutableCounterLong authenticationFallbacks;
040  private final MutableCounterLong sentBytes;
041  private final MutableCounterLong receivedBytes;
042
043  private final MutableCounterLong exceptions;
044  private final MutableCounterLong exceptionsOOO;
045  private final MutableCounterLong exceptionsBusy;
046  private final MutableCounterLong exceptionsUnknown;
047  private final MutableCounterLong exceptionsSanity;
048  private final MutableCounterLong exceptionsNSRE;
049  private final MutableCounterLong exceptionsMoved;
050
051
052  private MutableHistogram queueCallTime;
053  private MutableHistogram processCallTime;
054  private MutableHistogram totalCallTime;
055  private MutableHistogram requestSize;
056  private MutableHistogram responseSize;
057
058  public MetricsHBaseServerSourceImpl(String metricsName,
059                                      String metricsDescription,
060                                      String metricsContext,
061                                      String metricsJmxContext,
062                                      MetricsHBaseServerWrapper wrapper) {
063    super(metricsName, metricsDescription, metricsContext, metricsJmxContext);
064    this.wrapper = wrapper;
065
066    this.authorizationSuccesses = this.getMetricsRegistry().newCounter(AUTHORIZATION_SUCCESSES_NAME,
067        AUTHORIZATION_SUCCESSES_DESC, 0L);
068    this.authorizationFailures = this.getMetricsRegistry().newCounter(AUTHORIZATION_FAILURES_NAME,
069        AUTHORIZATION_FAILURES_DESC, 0L);
070
071    this.exceptions = this.getMetricsRegistry().newCounter(EXCEPTIONS_NAME, EXCEPTIONS_DESC, 0L);
072    this.exceptionsOOO = this.getMetricsRegistry()
073        .newCounter(EXCEPTIONS_OOO_NAME, EXCEPTIONS_TYPE_DESC, 0L);
074    this.exceptionsBusy = this.getMetricsRegistry()
075        .newCounter(EXCEPTIONS_BUSY_NAME, EXCEPTIONS_TYPE_DESC, 0L);
076    this.exceptionsUnknown = this.getMetricsRegistry()
077        .newCounter(EXCEPTIONS_UNKNOWN_NAME, EXCEPTIONS_TYPE_DESC, 0L);
078    this.exceptionsSanity = this.getMetricsRegistry()
079        .newCounter(EXCEPTIONS_SANITY_NAME, EXCEPTIONS_TYPE_DESC, 0L);
080    this.exceptionsMoved = this.getMetricsRegistry()
081        .newCounter(EXCEPTIONS_MOVED_NAME, EXCEPTIONS_TYPE_DESC, 0L);
082    this.exceptionsNSRE = this.getMetricsRegistry()
083        .newCounter(EXCEPTIONS_NSRE_NAME, EXCEPTIONS_TYPE_DESC, 0L);
084
085    this.authenticationSuccesses = this.getMetricsRegistry().newCounter(
086        AUTHENTICATION_SUCCESSES_NAME, AUTHENTICATION_SUCCESSES_DESC, 0L);
087    this.authenticationFailures = this.getMetricsRegistry().newCounter(AUTHENTICATION_FAILURES_NAME,
088        AUTHENTICATION_FAILURES_DESC, 0L);
089    this.authenticationFallbacks = this.getMetricsRegistry().newCounter(
090        AUTHENTICATION_FALLBACKS_NAME, AUTHENTICATION_FALLBACKS_DESC, 0L);
091    this.sentBytes = this.getMetricsRegistry().newCounter(SENT_BYTES_NAME,
092        SENT_BYTES_DESC, 0L);
093    this.receivedBytes = this.getMetricsRegistry().newCounter(RECEIVED_BYTES_NAME,
094        RECEIVED_BYTES_DESC, 0L);
095    this.queueCallTime = this.getMetricsRegistry().newHistogram(QUEUE_CALL_TIME_NAME,
096        QUEUE_CALL_TIME_DESC);
097    this.processCallTime = this.getMetricsRegistry().newHistogram(PROCESS_CALL_TIME_NAME,
098        PROCESS_CALL_TIME_DESC);
099    this.totalCallTime = this.getMetricsRegistry().newHistogram(TOTAL_CALL_TIME_NAME,
100        TOTAL_CALL_TIME_DESC);
101    this.requestSize = this.getMetricsRegistry().newHistogram(REQUEST_SIZE_NAME,
102        REQUEST_SIZE_DESC);
103    this.responseSize = this.getMetricsRegistry().newHistogram(RESPONSE_SIZE_NAME,
104              RESPONSE_SIZE_DESC);
105  }
106
107  @Override
108  public void authorizationSuccess() {
109    authorizationSuccesses.incr();
110  }
111
112  @Override
113  public void authorizationFailure() {
114    authorizationFailures.incr();
115  }
116
117  @Override
118  public void authenticationFailure() {
119    authenticationFailures.incr();
120  }
121
122  @Override
123  public void authenticationFallback() {
124    authenticationFallbacks.incr();
125  }
126
127  @Override
128  public void exception() {
129    exceptions.incr();
130  }
131
132  @Override
133  public void outOfOrderException() {
134    exceptionsOOO.incr();
135  }
136
137  @Override
138  public void failedSanityException() {
139    exceptionsSanity.incr();
140  }
141
142  @Override
143  public void movedRegionException() {
144    exceptionsMoved.incr();
145  }
146
147  @Override
148  public void notServingRegionException() {
149    exceptionsNSRE.incr();
150  }
151
152  @Override
153  public void unknownScannerException() {
154    exceptionsUnknown.incr();
155  }
156
157  @Override
158  public void tooBusyException() {
159    exceptionsBusy.incr();
160  }
161
162  @Override
163  public void authenticationSuccess() {
164    authenticationSuccesses.incr();
165  }
166
167  @Override
168  public void sentBytes(long count) {
169    this.sentBytes.incr(count);
170  }
171
172  @Override
173  public void receivedBytes(int count) {
174    this.receivedBytes.incr(count);
175  }
176
177  @Override
178  public void sentResponse(long count) { this.responseSize.add(count); }
179
180  @Override
181  public void receivedRequest(long count) { this.requestSize.add(count); }
182
183  @Override
184  public void dequeuedCall(int qTime) {
185    queueCallTime.add(qTime);
186  }
187
188  @Override
189  public void processedCall(int processingTime) {
190    processCallTime.add(processingTime);
191  }
192
193  @Override
194  public void queuedAndProcessedCall(int totalTime) {
195    totalCallTime.add(totalTime);
196  }
197
198  @Override
199  public void getMetrics(MetricsCollector metricsCollector, boolean all) {
200    MetricsRecordBuilder mrb = metricsCollector.addRecord(metricsName);
201
202    if (wrapper != null) {
203      mrb.addGauge(Interns.info(QUEUE_SIZE_NAME, QUEUE_SIZE_DESC), wrapper.getTotalQueueSize())
204          .addGauge(Interns.info(GENERAL_QUEUE_NAME, GENERAL_QUEUE_DESC),
205              wrapper.getGeneralQueueLength())
206          .addGauge(Interns.info(REPLICATION_QUEUE_NAME,
207              REPLICATION_QUEUE_DESC), wrapper.getReplicationQueueLength())
208          .addGauge(Interns.info(PRIORITY_QUEUE_NAME, PRIORITY_QUEUE_DESC),
209              wrapper.getPriorityQueueLength())
210          .addGauge(Interns.info(NUM_OPEN_CONNECTIONS_NAME,
211              NUM_OPEN_CONNECTIONS_DESC), wrapper.getNumOpenConnections())
212          .addGauge(Interns.info(NUM_ACTIVE_HANDLER_NAME,
213              NUM_ACTIVE_HANDLER_DESC), wrapper.getActiveRpcHandlerCount());
214    }
215
216    metricsRegistry.snapshot(mrb, all);
217  }
218}
