018package org.apache.hadoop.hbase.ipc;
019
020import com.google.common.base.Preconditions;
021import com.google.protobuf.CodedOutputStream;
022import com.google.protobuf.Message;
023
024import java.io.IOException;
025import java.io.OutputStream;
026import java.net.ConnectException;
027import java.net.InetSocketAddress;
028import java.net.SocketTimeoutException;
029import java.nio.ByteBuffer;
030
031import org.apache.hadoop.hbase.classification.InterfaceAudience;
032import org.apache.hadoop.hbase.exceptions.ConnectionClosingException;
033import org.apache.hadoop.hbase.protobuf.generated.RPCProtos.ExceptionResponse;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.apache.hadoop.ipc.RemoteException;
036
037/**
038 * Utility to help ipc'ing.
039 */
040@InterfaceAudience.Private
041public class IPCUtil {
042
043  /**
044   * Write out header, param, and cell block if there is one.
045   * @param dos Stream to write into
046   * @param header to write
047   * @param param to write
048   * @param cellBlock to write
049   * @return Total number of bytes written.
050   * @throws IOException if write action fails
051   */
052  public static int write(final OutputStream dos, final Message header, final Message param,
053      final ByteBuffer cellBlock) throws IOException {
054    // Must calculate total size and write that first so other side can read it all in in one
055    // swoop. This is dictated by how the server is currently written. Server needs to change
056    // if we are to be able to write without the length prefixing.
057    int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(header, param);
058    if (cellBlock != null) {
059      totalSize += cellBlock.remaining();
060    }
061    return write(dos, header, param, cellBlock, totalSize);
062  }
063
064  private static int write(final OutputStream dos, final Message header, final Message param,
065      final ByteBuffer cellBlock, final int totalSize) throws IOException {
066    // I confirmed toBytes does same as DataOutputStream#writeInt.
067    dos.write(Bytes.toBytes(totalSize));
068    // This allocates a buffer that is the size of the message internally.
069    header.writeDelimitedTo(dos);
070    if (param != null) {
071      param.writeDelimitedTo(dos);
072    }
073    if (cellBlock != null) {
074      dos.write(cellBlock.array(), 0, cellBlock.remaining());
075    }
076    dos.flush();
077    return totalSize;
078  }
079
080  /**
081   * @return Size on the wire when the two messages are written with writeDelimitedTo
082   */
083  public static int getTotalSizeWhenWrittenDelimited(Message... messages) {
084    int totalSize = 0;
085    for (Message m : messages) {
086      if (m == null) {
087        continue;
088      }
089      totalSize += m.getSerializedSize();
090      totalSize += CodedOutputStream.computeRawVarint32Size(m.getSerializedSize());
091    }
092    Preconditions.checkArgument(totalSize < Integer.MAX_VALUE);
093    return totalSize;
094  }
095
096  /**
097   * @return True if the exception is a fatal connection exception.
098   */
099  public static boolean isFatalConnectionException(final ExceptionResponse e) {
100    return e.getExceptionClassName().equals(FatalConnectionException.class.getName());
101  }
102
103  /**
104   * @param e exception to be wrapped
105   * @return RemoteException made from passed <code>e</code>
106   */
107  public static RemoteException createRemoteException(final ExceptionResponse e) {
108    String innerExceptionClassName = e.getExceptionClassName();
109    boolean doNotRetry = e.getDoNotRetry();
110    return e.hasHostname() ?
111    // If a hostname then add it to the RemoteWithExtrasException
112        new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), e.getHostname(),
113            e.getPort(), doNotRetry)
114        : new RemoteWithExtrasException(innerExceptionClassName, e.getStackTrace(), doNotRetry);
115  }
116
117  /**
118   * Takes an Exception and the address we were trying to connect to and return an IOException with
119   * the input exception as the cause. The new exception provides the stack trace of the place where
120   * the exception is thrown and some extra diagnostics information. If the exception is
121   * ConnectException or SocketTimeoutException, return a new one of the same type; Otherwise return
122   * an IOException.
123   * @param addr target address
124   * @param exception the relevant exception
125   * @return an exception to throw
126   */
127  public static IOException wrapException(InetSocketAddress addr, Exception exception) {
128    if (exception instanceof ConnectException) {
129      // connection refused; include the host:port in the error
130      return (ConnectException) new ConnectException(
131          "Call to " + addr + " failed on connection exception: " + exception).initCause(exception);
132    } else if (exception instanceof SocketTimeoutException) {
133      return (SocketTimeoutException) new SocketTimeoutException(
134          "Call to " + addr + " failed because " + exception).initCause(exception);
135    } else if (exception instanceof ConnectionClosingException) {
136      return (ConnectionClosingException) new ConnectionClosingException(
137          "Call to " + addr + " failed on local exception: " + exception).initCause(exception);
138    } else {
139      return (IOException) new IOException(
140          "Call to " + addr + " failed on local exception: " + exception).initCause(exception);
141    }
142  }
143}
