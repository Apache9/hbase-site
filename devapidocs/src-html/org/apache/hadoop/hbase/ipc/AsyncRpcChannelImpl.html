150  /**
151   * Connect to channel
152   * @param bootstrap to connect to
153   * @return future of connection
154   */
155  private ChannelFuture connect(final Bootstrap bootstrap) {
156    return bootstrap.remoteAddress(address).connect()
157        .addListener(new GenericFutureListener<ChannelFuture>() {
158          @Override
159          public void operationComplete(final ChannelFuture f) throws Exception {
160            if (!f.isSuccess()) {
161              retryOrClose(bootstrap, failureCounter++, client.failureSleep, f.cause());
162              return;
163            }
164            channel = f.channel();
165
166            setupAuthorization();
167
168            ByteBuf b = channel.alloc().directBuffer(6);
169            createPreamble(b, authMethod);
170            channel.writeAndFlush(b).addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
171            if (useSasl) {
172              UserGroupInformation ticket = AsyncRpcChannelImpl.this.ticket.getUGI();
173              if (authMethod == AuthMethod.KERBEROS) {
174                if (ticket != null && ticket.getRealUser() != null) {
175                  ticket = ticket.getRealUser();
176                }
177              }
178              SaslClientHandler saslHandler;
179              if (ticket == null) {
180                throw new FatalConnectionException("ticket/user is null");
181              }
182              final UserGroupInformation realTicket = ticket;
183              saslHandler = ticket.doAs(new PrivilegedExceptionAction<SaslClientHandler>() {
184                @Override
185                public SaslClientHandler run() throws IOException {
186                  return getSaslHandler(realTicket, bootstrap);
187                }
188              });
189              if (saslHandler != null) {
190                // Sasl connect is successful. Let's set up Sasl channel handler
191                channel.pipeline().addFirst(saslHandler);
192              } else {
193                // fall back to simple auth because server told us so.
194                authMethod = AuthMethod.SIMPLE;
195                useSasl = false;
196              }
197            } else {
198              startHBaseConnection(f.channel());
199            }
200          }
201        });
202  }

204  /**
205   * Start HBase connection
206   * @param ch channel to start connection on
207   */
208  private void startHBaseConnection(Channel ch) {
209    ch.pipeline().addLast("frameDecoder",
210      new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 4));
211    ch.pipeline().addLast(new AsyncServerResponseHandler(this));
212    try {
213      writeChannelHeader(ch).addListener(new GenericFutureListener<ChannelFuture>() {
214        @Override
215        public void operationComplete(ChannelFuture future) throws Exception {
216          if (!future.isSuccess()) {
217            close(future.cause());
218            return;
219          }
220          List<AsyncCall> callsToWrite;
221          synchronized (pendingCalls) {
222            connected = true;
223            callsToWrite = new ArrayList<AsyncCall>(pendingCalls.values());
224          }
225          for (AsyncCall call : callsToWrite) {
226            writeRequest(call);
227          }
228        }
229      });
230    } catch (IOException e) {
231      close(e);
232    }
233  }
234
235  private void startConnectionWithEncryption(Channel ch) {
236    // for rpc encryption, the order of ChannelInboundHandler should be:
237    // LengthFieldBasedFrameDecoder->SaslClientHandler->LengthFieldBasedFrameDecoder
238    // Don't skip the first 4 bytes for length in beforeUnwrapDecoder,
239    // SaslClientHandler will handler this
240    ch.pipeline().addFirst("beforeUnwrapDecoder",
241        new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 0));
242    ch.pipeline().addLast("afterUnwrapDecoder",
243        new LengthFieldBasedFrameDecoder(Integer.MAX_VALUE, 0, 4, 0, 4));
244    ch.pipeline().addLast(new AsyncServerResponseHandler(this));
245    List<AsyncCall> callsToWrite;
246    synchronized (pendingCalls) {
247      connected = true;
248      callsToWrite = new ArrayList<AsyncCall>(pendingCalls.values());
249    }
250    for (AsyncCall call : callsToWrite) {
251      writeRequest(call);
252    }
253  }

255  /**
256   * Get SASL handler
257   * @param bootstrap to reconnect to
258   * @return new SASL handler
259   * @throws java.io.IOException if handler failed to create
260   */
261  private SaslClientHandler getSaslHandler(final UserGroupInformation realTicket,
262      final Bootstrap bootstrap) throws IOException {
263    return new SaslClientHandler(realTicket, authMethod, token, serverPrincipal,
264        client.fallbackAllowed,
265        client.conf.get("hbase.rpc.protection",
266          SaslUtil.QualityOfProtection.AUTHENTICATION.name().toLowerCase(Locale.ROOT)),
267        getChannelHeaderBytes(authMethod),
268        new SaslClientHandler.SaslExceptionHandler() {
269          @Override
270          public void handle(int retryCount, Random random, Throwable cause) {
271            try {
272              // Handle Sasl failure. Try to potentially get new credentials
273              handleSaslConnectionFailure(retryCount, cause, realTicket);
274
275              retryOrClose(bootstrap, failureCounter++, random.nextInt(reloginMaxBackoff) + 1,
276                cause);
277            } catch (IOException | InterruptedException e) {
278              close(e);
279            }
280          }
281        }, new SaslClientHandler.SaslSuccessfulConnectHandler() {
282          @Override
283          public void onSuccess(Channel channel) {
284            startHBaseConnection(channel);
285          }
286
287          @Override
288          public void onSaslProtectionSucess(Channel channel) {
289            startConnectionWithEncryption(channel);
290          }
291        });
292  }

294  /**
295   * Retry to connect or close
296   * @param bootstrap to connect with
297   * @param failureCount failure count
298   * @param e exception of fail
299   */
300  private void retryOrClose(final Bootstrap bootstrap, int failureCount, long timeout,
301      Throwable e) {
302    if (failureCount < client.maxRetries) {
303      client.newTimeout(new TimerTask() {
304        @Override
305        public void run(Timeout timeout) throws Exception {
306          connect(bootstrap);
307        }
308      }, timeout, TimeUnit.MILLISECONDS);
309    } else {
310      client.failedServers.addToFailedServers(address);
311      close(e);
312    }
313  }

315  /**
316   * Calls method on channel
317   * @param method to call
318   * @param request to send
319   * @param cellScanner with cells to send
320   * @param responsePrototype to construct response with
321   * @param rpcTimeout timeout for request
322   * @param priority for request
323   * @return Promise for the response Message
324   */
325  @Override
326  public <R extends Message, O> Future<O> callMethod(
327      final Descriptors.MethodDescriptor method,
328      final Message request,final CellScanner cellScanner,
329      R responsePrototype, MessageConverter<R, O> messageConverter, IOExceptionConverter
330      exceptionConverter, long rpcTimeout, int priority) {
331    final AsyncCall<R, O> call = new AsyncCall<>(this, client.callIdCnt.getAndIncrement(),
332        method, request, cellScanner, responsePrototype, messageConverter, exceptionConverter,
333        rpcTimeout, priority, client.metrics);
334
335    synchronized (pendingCalls) {
336      if (closed) {
337        call.setFailure(new ConnectException());
338        return call;
339      }
340      pendingCalls.put(call.id, call);
341      // Add timeout for cleanup if none is present
342      if (cleanupTimer == null && call.getRpcTimeout() > 0) {
343        cleanupTimer = client.newTimeout(timeoutTask, call.getRpcTimeout(), TimeUnit.MILLISECONDS);
344      }
345      if (!connected) {
346        return call;
347      }
348    }
349    writeRequest(call);
350    return call;
351  }

364  /**
365   * Write the channel header
366   * @param channel to write to
367   * @return future of write
368   * @throws java.io.IOException on failure to write
369   */
370  private ChannelFuture writeChannelHeader(Channel channel) throws IOException {
371    RPCProtos.ConnectionHeader header = getChannelHeader(authMethod);
372    int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(header);
373    ByteBuf b = channel.alloc().directBuffer(totalSize);
374
375    b.writeInt(header.getSerializedSize());
376    b.writeBytes(header.toByteArray());
377
378    return channel.writeAndFlush(b);
379  }
380
381  private byte[] getChannelHeaderBytes(AuthMethod authMethod) {
382    RPCProtos.ConnectionHeader header = getChannelHeader(authMethod);
383    ByteBuffer b = ByteBuffer.allocate(header.getSerializedSize() + 4);
384    b.putInt(header.getSerializedSize());
385    b.put(header.toByteArray());
386    return b.array();
387  }
388
389  private RPCProtos.ConnectionHeader getChannelHeader(AuthMethod authMethod) {
390    RPCProtos.ConnectionHeader.Builder headerBuilder = RPCProtos.ConnectionHeader.newBuilder()
391        .setServiceName(serviceName);
392
393    RPCProtos.UserInformation userInfoPB = buildUserInfo(ticket.getUGI(), authMethod);
394    if (userInfoPB != null) {
395      headerBuilder.setUserInfo(userInfoPB);
396    }
397
398    if (client.codec != null) {
399      headerBuilder.setCellBlockCodecClass(client.codec.getClass().getCanonicalName());
400    }
401    if (client.compressor != null) {
402      headerBuilder.setCellBlockCompressorClass(client.compressor.getClass().getCanonicalName());
403    }
404
405    headerBuilder.setVersionInfo(ProtobufUtil.getVersionInfo());
406    return headerBuilder.build();
407  }

409  /**
410   * Write request to channel
411   * @param call to write
412   */
413  private void writeRequest(final AsyncCall call) {
414    try {
415      final RPCProtos.RequestHeader.Builder requestHeaderBuilder = RPCProtos.RequestHeader
416          .newBuilder();
417      requestHeaderBuilder.setCallId(call.id).setMethodName(call.method.getName())
418          .setRequestParam(call.param != null);
419
420      if (Trace.isTracing()) {
421        Span s = Trace.currentSpan();
422        requestHeaderBuilder.setTraceInfo(TracingProtos.RPCTInfo.newBuilder()
423            .setParentId(s.getSpanId()).setTraceId(s.getTraceId()));
424      }
425
426      ByteBuffer cellBlock = client.buildCellBlock(call.cellScanner());
427      if (cellBlock != null) {
428        final RPCProtos.CellBlockMeta.Builder cellBlockBuilder = RPCProtos.CellBlockMeta
429            .newBuilder();
430        cellBlockBuilder.setLength(cellBlock.limit());
431        requestHeaderBuilder.setCellBlockMeta(cellBlockBuilder.build());
432      }
433      // Only pass priority if there one. Let zero be same as no priority.
434      if (call.getPriority() != PayloadCarryingRpcController.PRIORITY_UNSET) {
435        requestHeaderBuilder.setPriority(call.getPriority());
436      }
437      requestHeaderBuilder.setTimeout(call.rpcTimeout > Integer.MAX_VALUE ?
438          Integer.MAX_VALUE : (int)call.rpcTimeout);
439
440      RPCProtos.RequestHeader rh = requestHeaderBuilder.build();
441
442      int totalSize = IPCUtil.getTotalSizeWhenWrittenDelimited(rh, call.param);
443      if (cellBlock != null) {
444        totalSize += cellBlock.remaining();
445      }
446
447      ByteBuf b = channel.alloc().directBuffer(4 + totalSize);
448      try (ByteBufOutputStream out = new ByteBufOutputStream(b)) {
449        call.callStats.setRequestSizeBytes(IPCUtil.write(out, rh, call.param, cellBlock));
450      }
451
452      channel.writeAndFlush(b).addListener(new CallWriteListener(this, call.id));
453    } catch (IOException e) {
454      close(e);
455    }
456  }

458  /**
459   * Set up server authorization
460   * @throws java.io.IOException if auth setup failed
461   */
462  private void setupAuthorization() throws IOException {
463    SecurityInfo securityInfo = SecurityInfo.getInfo(serviceName);
464    this.useSasl = client.userProvider.isHBaseSecurityEnabled();
465
466    this.token = null;
467    if (useSasl && securityInfo != null) {
468      AuthenticationProtos.TokenIdentifier.Kind tokenKind = securityInfo.getTokenKind();
469      if (tokenKind != null) {
470        TokenSelector<? extends TokenIdentifier> tokenSelector = TOKEN_HANDDLERS.get(tokenKind);
471        if (tokenSelector != null) {
472          token = tokenSelector.selectToken(new Text(client.clusterId),
473            ticket.getUGI().getTokens());
474        } else if (LOG.isDebugEnabled()) {
475          LOG.debug("No token selector found for type " + tokenKind);
476        }
477      }
478      String serverKey = securityInfo.getServerPrincipal();
479      if (serverKey == null) {
480        throw new IOException("Can't obtain server Kerberos config key from SecurityInfo");
481      }
482      this.serverPrincipal = SecurityUtil.getServerPrincipal(client.conf.get(serverKey),
483        address.getAddress().getCanonicalHostName().toLowerCase(Locale.ROOT));
484      if (LOG.isDebugEnabled()) {
485        LOG.debug("RPC Server Kerberos principal name for service=" + serviceName + " is "
486            + serverPrincipal);
487      }
488    }
489
490    if (!useSasl) {
491      authMethod = AuthMethod.SIMPLE;
492    } else if (token != null) {
493      authMethod = AuthMethod.DIGEST;
494    } else {
495      authMethod = AuthMethod.KERBEROS;
496    }
497
498    if (LOG.isDebugEnabled()) {
499      LOG.debug(
500        "Use " + authMethod + " authentication for service " + serviceName + ", sasl=" + useSasl);
501    }
502    reloginMaxBackoff = client.conf.getInt("hbase.security.relogin.maxbackoff", 5000);
503  }

505  /**
506   * Build the user information
507   * @param ugi User Group Information
508   * @param authMethod Authorization method
509   * @return UserInformation protobuf
510   */
511  private RPCProtos.UserInformation buildUserInfo(UserGroupInformation ugi, AuthMethod authMethod) {
512    if (ugi == null || authMethod == AuthMethod.DIGEST) {
513      // Don't send user for token auth
514      return null;
515    }
516    RPCProtos.UserInformation.Builder userInfoPB = RPCProtos.UserInformation.newBuilder();
517    if (authMethod == AuthMethod.KERBEROS) {
518      // Send effective user for Kerberos auth
519      userInfoPB.setEffectiveUser(ugi.getUserName());
520    } else if (authMethod == AuthMethod.SIMPLE) {
521      // Send both effective user and real user for simple auth
522      userInfoPB.setEffectiveUser(ugi.getUserName());
523      if (ugi.getRealUser() != null) {
524        userInfoPB.setRealUser(ugi.getRealUser().getUserName());
525      }
526    }
527    return userInfoPB.build();
528  }

530  /**
531   * Create connection preamble
532   * @param byteBuf to write to
533   * @param authMethod to write
534   */
535  private void createPreamble(ByteBuf byteBuf, AuthMethod authMethod) {
536    byteBuf.writeBytes(HConstants.RPC_HEADER);
537    byteBuf.writeByte(HConstants.RPC_CURRENT_VERSION);
538    byteBuf.writeByte(authMethod.code);
539  }

541  private void close0(Throwable e) {
542    List<AsyncCall> toCleanup;
543    synchronized (pendingCalls) {
544      if (closed) {
545        return;
546      }
547      closed = true;
548      toCleanup = new ArrayList<AsyncCall>(pendingCalls.values());
549      pendingCalls.clear();
550    }
551    IOException closeException = null;
552    if (e != null) {
553      if (e instanceof IOException) {
554        closeException = (IOException) e;
555      } else {
556        closeException = new IOException(e);
557      }
558    }
559    // log the info
560    if (LOG.isDebugEnabled() && closeException != null) {
561      LOG.debug(name + ": closing ipc connection to " + address, closeException);
562    }
563    if (cleanupTimer != null) {
564      cleanupTimer.cancel();
565      cleanupTimer = null;
566    }
567    for (AsyncCall call : toCleanup) {
568      call.setFailed(closeException != null ? closeException
569          : new ConnectionClosingException(
570              "Call id=" + call.id + " on server " + address + " aborted: connection is closing"));
571    }
572    channel.disconnect().addListener(ChannelFutureListener.CLOSE);
573    if (LOG.isDebugEnabled()) {
574      LOG.debug(name + ": closed");
575    }
576  }

578  /**
579   * Close connection
580   * @param e exception on close
581   */
582  public void close(final Throwable e) {
583    client.removeConnection(this);
584
585    // Move closing from the requesting thread to the channel thread
586    if (channel.eventLoop().inEventLoop()) {
587      close0(e);
588    } else {
589      channel.eventLoop().execute(new Runnable() {
590        @Override
591        public void run() {
592          close0(e);
593        }
594      });
595    }
596  }

598  /**
599   * Clean up calls.
600   */
601  private void cleanupCalls() {
602    List<AsyncCall> toCleanup = new ArrayList<AsyncCall>();
603    long currentTime = EnvironmentEdgeManager.currentTime();
604    long nextCleanupTaskDelay = -1L;
605    synchronized (pendingCalls) {
606      for (Iterator<AsyncCall> iter = pendingCalls.values().iterator(); iter.hasNext();) {
607        AsyncCall call = iter.next();
608        long timeout = call.getRpcTimeout();
609        if (timeout > 0) {
610          if (currentTime - call.getStartTime() >= timeout) {
611            iter.remove();
612            toCleanup.add(call);
613          } else {
614            if (nextCleanupTaskDelay < 0 || timeout < nextCleanupTaskDelay) {
615              nextCleanupTaskDelay = timeout;
616            }
617          }
618        }
619      }
620      if (nextCleanupTaskDelay > 0) {
621        cleanupTimer = client.newTimeout(timeoutTask, nextCleanupTaskDelay, TimeUnit.MILLISECONDS);
622      } else {
623        cleanupTimer = null;
624      }
625    }
626    for (AsyncCall call : toCleanup) {
627      call.setFailed(new CallTimeoutException("Call id=" + call.id + ", waitTime="
628          + (currentTime - call.getStartTime()) + ", rpcTimeout=" + call.getRpcTimeout()));
629    }
630  }

645  /**
646   * Check if user should authenticate over Kerberos
647   * @return true if should be authenticated over Kerberos
648   * @throws java.io.IOException on failure of check
649   */
650  private synchronized boolean shouldAuthenticateOverKrb() throws IOException {
651    UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
652    UserGroupInformation currentUser = UserGroupInformation.getCurrentUser();
653    UserGroupInformation realUser = currentUser.getRealUser();
654    return authMethod == AuthMethod.KERBEROS && loginUser != null &&
655      // Make sure user logged in using Kerberos either keytab or TGT
656      loginUser.hasKerberosCredentials() &&
657      // relogin only in case it is the login user (e.g. JT)
658      // or superuser (like oozie).
659      (loginUser.equals(currentUser) || loginUser.equals(realUser));
660  }

662  /**
663   * If multiple clients with the same principal try to connect to the same server at the same time,
664   * the server assumes a replay attack is in progress. This is a feature of kerberos. In order to
665   * work around this, what is done is that the client backs off randomly and tries to initiate the
666   * connection again. The other problem is to do with ticket expiry. To handle that, a relogin is
667   * attempted.
668   * <p>
669   * The retry logic is governed by the {@link #shouldAuthenticateOverKrb} method. In case when the
670   * user doesn't have valid credentials, we don't need to retry (from cache or ticket). In such
671   * cases, it is prudent to throw a runtime exception when we receive a SaslException from the
672   * underlying authentication implementation, so there is no retry from other high level (for eg,
673   * HCM or HBaseAdmin).
674   * </p>
675   * @param currRetries retry count
676   * @param ex exception describing fail
677   * @param user which is trying to connect
678   * @throws java.io.IOException if IO fail
679   * @throws InterruptedException if thread is interrupted
680   */
681  private void handleSaslConnectionFailure(final int currRetries, final Throwable ex,
682      final UserGroupInformation user) throws IOException, InterruptedException {
683    user.doAs(new PrivilegedExceptionAction<Void>() {
684      @Override
685      public Void run() throws IOException, InterruptedException {
686        if (shouldAuthenticateOverKrb()) {
687          if (currRetries < MAX_SASL_RETRIES) {
688            LOG.debug("Exception encountered while connecting to the server : " + ex);
689            // try re-login
690            if (UserGroupInformation.isLoginKeytabBased()) {
691              UserGroupInformation.getLoginUser().reloginFromKeytab();
692            } else {
693              UserGroupInformation.getLoginUser().reloginFromTicketCache();
694            }
695
696            // Should reconnect
697            return null;
698          } else {
699            String msg = "Couldn't setup connection for "
700                + UserGroupInformation.getLoginUser().getUserName() + " to " + serverPrincipal;
701            LOG.warn(msg, ex);
702            throw (IOException) new IOException(msg).initCause(ex);
703          }
704        } else {
705          LOG.warn("Exception encountered while connecting to " + "the server : " + ex);
706        }
707        if (ex instanceof RemoteException) {
708          throw (RemoteException) ex;
709        }
710        if (ex instanceof SaslException) {
711          String msg = "SASL authentication failed."
712              + " The most likely cause is missing or invalid credentials." + " Consider 'kinit'.";
713          LOG.fatal(msg, ex);
714          throw new RuntimeException(msg, ex);
715        }
716        throw new IOException(ex);
717      }
718    });
719  }

744  /**
745   * Listens to call writes and fails if write failed
746   */
747  private static final class CallWriteListener implements ChannelFutureListener {
748    private final AsyncRpcChannelImpl rpcChannel;
749    private final int id;
750
751    public CallWriteListener(AsyncRpcChannelImpl asyncRpcChannelImpl, int id) {
752      this.rpcChannel = asyncRpcChannelImpl;
753      this.id = id;
754    }
755
756    @Override
757    public void operationComplete(ChannelFuture future) throws Exception {
758      if (!future.isSuccess()) {
759        AsyncCall call = rpcChannel.removePendingCall(id);
760        if (call != null) {
761          if (future.cause() instanceof IOException) {
762            call.setFailed((IOException) future.cause());
763          } else {
764            call.setFailed(new IOException(future.cause()));
765          }
766        }
767      }
768    }
769  }
