018package org.apache.hadoop.hbase.ipc;
019
020import com.google.protobuf.Descriptors;
021import com.google.protobuf.Message;
022
023import io.netty.util.concurrent.DefaultPromise;
024
025import java.io.IOException;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.apache.hadoop.hbase.CellScanner;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.client.MetricsConnection;
032import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
033import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
034import org.apache.hadoop.hbase.util.ExceptionUtil;
035import org.apache.hadoop.ipc.RemoteException;
036
037/**
038 * Represents an Async Hbase call and its response.
039 *
040 * Responses are passed on to its given doneHandler and failures to the rpcController
041 *
042 * @param <T> Type of message returned
043 * @param <M> Message returned in communication to be converted
044 */
045@InterfaceAudience.Private
046public class AsyncCall<M extends Message, T> extends DefaultPromise<T> {
047  private static final Log LOG = LogFactory.getLog(AsyncCall.class.getName());
048
049  final int id;
050
051  private final AsyncRpcChannel channel;
052
053  final Descriptors.MethodDescriptor method;
054  final Message param;
055  final Message responseDefaultType;
056
057  private final MessageConverter<M,T> messageConverter;
058  private final IOExceptionConverter exceptionConverter;
059
060  final long rpcTimeout;
061
062  // For only the request
063  private final CellScanner cellScanner;
064  private final int priority;
065
066  final MetricsConnection clientMetrics;
067  final MetricsConnection.CallStats callStats;
068
069  /**
070   * Constructor
071   *
072   * @param channel             which initiated call
073   * @param connectId           connection id
074   * @param md                  the method descriptor
075   * @param param               parameters to send to Server
076   * @param cellScanner         cellScanner containing cells to send as request
077   * @param responseDefaultType the default response type
078   * @param messageConverter    converts the messages to what is the expected output
079   * @param exceptionConverter  converts exceptions to expected format. Can be null
080   * @param rpcTimeout          timeout for this call in ms
081   * @param priority            for this request
082   * @param metrics             MetricsConnection to which the metrics are stored for this request
083   */
084  public AsyncCall(AsyncRpcChannel channel, int connectId, Descriptors.MethodDescriptor
085        md, Message param, CellScanner cellScanner, M responseDefaultType, MessageConverter<M, T>
086        messageConverter, IOExceptionConverter exceptionConverter, long rpcTimeout, int priority,
087      MetricsConnection metrics) {
088    super(channel.getEventExecutor());
089    this.channel = channel;
090
091    this.id = connectId;
092
093    this.method = md;
094    this.param = param;
095    this.responseDefaultType = responseDefaultType;
096
097    this.messageConverter = messageConverter;
098    this.exceptionConverter = exceptionConverter;
099
100    this.rpcTimeout = rpcTimeout;
101
102    this.priority = priority;
103    this.cellScanner = cellScanner;
104
105    this.callStats = MetricsConnection.newCallStats();
106    callStats.setStartTime(EnvironmentEdgeManager.currentTime());
107
108    this.clientMetrics = metrics;
109  }
110
111  /**
112   * Get the start time
113   *
114   * @return start time for the call
115   */
116  public long getStartTime() {
117    return this.callStats.getStartTime();
118  }
119
120  @Override
121  public String toString() {
122    return "callId=" + this.id + ", method=" + this.method.getName() +
123      ", rpcTimeout=" + this.rpcTimeout + ", param {" +
124      (this.param != null ? ProtobufUtil.getShortTextFormat(this.param) : "") + "}";
125  }
126
127  /**
128   * Set success with a cellBlockScanner
129   *
130   * @param value            to set
131   * @param cellBlockScanner to set
132   */
133  public void setSuccess(M value, CellScanner cellBlockScanner) {
134    callStats.setCallTimeMs(EnvironmentEdgeManager.currentTime() - callStats.getStartTime());
135
136    if (LOG.isTraceEnabled()) {
137      LOG.trace("Call: " + method.getName() + ", callTime: " + callStats.getCallTimeMs() + "ms");
138    }
139
140    if (clientMetrics != null) {
141      clientMetrics.updateRpc(method, param, callStats);
142    }
143
144    try {
145      this.setSuccess(
146          this.messageConverter.convert(value, cellBlockScanner)
147      );
148    } catch (IOException e) {
149      this.setFailed(e);
150    }
151  }
152
153  /**
154   * Set failed
155   *
156   * @param exception to set
157   */
158  public void setFailed(IOException exception) {
159    if (ExceptionUtil.isInterrupt(exception)) {
160      exception = ExceptionUtil.asInterrupt(exception);
161    }
162    if (exception instanceof RemoteException) {
163      exception = ((RemoteException) exception).unwrapRemoteException();
164    }
165
166    if (this.exceptionConverter != null) {
167      exception = this.exceptionConverter.convert(exception);
168    }
169
170    this.setFailure(exception);
171  }
172
173  /**
174   * Get the rpc timeout
175   *
176   * @return current timeout for this call
177   */
178  public long getRpcTimeout() {
179    return rpcTimeout;
180  }
181
182
183  /**
184   * @return Priority for this call
185   */
186  public int getPriority() {
187    return priority;
188  }
189
190  /**
191   * Get the cellScanner for this request.
192   * @return CellScanner
193   */
194  public CellScanner cellScanner() {
195    return cellScanner;
196  }
197
198  @Override
199  public boolean cancel(boolean mayInterupt){
200    this.channel.removePendingCall(this.id);
201    return super.cancel(mayInterupt);
202  }
203
204}
