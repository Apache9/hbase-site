036/**
037 * A scheduler that maintains isolated handler pools for general,
038 * high-priority, and replication requests.
039 */
040@InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX})
041@InterfaceStability.Evolving
042public class SimpleRpcScheduler extends RpcScheduler implements ConfigurationObserver {
043  private static final Log LOG = LogFactory.getLog(SimpleRpcScheduler.class);
044
045  public static final String CALL_QUEUE_READ_SHARE_CONF_KEY =
046      "hbase.ipc.server.callqueue.read.ratio";
047  public static final String CALL_QUEUE_SCAN_SHARE_CONF_KEY =
048      "hbase.ipc.server.callqueue.scan.ratio";
049  public static final String CALL_QUEUE_HANDLER_FACTOR_CONF_KEY =
050      "hbase.ipc.server.callqueue.handler.factor";
051
052  /** If set to 'deadline', uses a priority queue and deprioritize long-running scans */
053  public static final String CALL_QUEUE_TYPE_CONF_KEY = "hbase.ipc.server.callqueue.type";
054  public static final String CALL_QUEUE_TYPE_CODEL_CONF_VALUE = "codel";
055  public static final String CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE = "deadline";
056  public static final String CALL_QUEUE_TYPE_FIFO_CONF_VALUE = "fifo";
057
058  /** max delay in msec used to bound the deprioritized requests */
059  public static final String QUEUE_MAX_CALL_DELAY_CONF_KEY
060      = "hbase.ipc.server.queue.max.call.delay";
061
062  // These 3 are only used by Codel executor
063  public static final String CALL_QUEUE_CODEL_TARGET_DELAY =
064    "hbase.ipc.server.callqueue.codel.target.delay";
065  public static final String CALL_QUEUE_CODEL_INTERVAL =
066    "hbase.ipc.server.callqueue.codel.interval";
067  public static final String CALL_QUEUE_CODEL_LIFO_THRESHOLD =
068    "hbase.ipc.server.callqueue.codel.lifo.threshold";
069
070  public static final int CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY = 5;
071  public static final int CALL_QUEUE_CODEL_DEFAULT_INTERVAL = 100;
072  public static final double CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD = 0.8;
073
074  private AtomicLong numGeneralCallsDropped = new AtomicLong();
075  private AtomicLong numLifoModeSwitches = new AtomicLong();
076
077  /**
078   * Resize call queues;
079   * @param conf new configuration
080   */
081  @Override
082  public void onConfigurationChange(Configuration conf) {
083    callExecutor.resizeQueues(conf);
084    if (priorityExecutor != null) {
085      priorityExecutor.resizeQueues(conf);
086    }
087    if (replicationExecutor != null) {
088      replicationExecutor.resizeQueues(conf);
089    }
090
091    String callQueueType = conf.get(CALL_QUEUE_TYPE_CONF_KEY,
092      CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE);
093
094    if (callQueueType.equals(CALL_QUEUE_TYPE_CODEL_CONF_VALUE)) {
095      // update CoDel Scheduler tunables
096      int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
097        CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
098      int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL,
099        CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
100      double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
101        CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
102
103      for (BlockingQueue<CallRunner> queue : callExecutor.getQueues()) {
104        if (queue instanceof AdaptiveLifoCoDelCallQueue) {
105          ((AdaptiveLifoCoDelCallQueue) queue).updateTunables(codelTargetDelay,
106            codelInterval, codelLifoThreshold);
107        }
108      }
109    }
110  }
111
112  /**
113   * Comparator used by the "normal callQueue" if DEADLINE_CALL_QUEUE_CONF_KEY is set to true.
114   * It uses the calculated "deadline" e.g. to deprioritize long-running job
115   *
116   * If multiple requests have the same deadline BoundedPriorityBlockingQueue will order them in
117   * FIFO (first-in-first-out) manner.
118   */
119  private static class CallPriorityComparator implements Comparator<CallRunner> {
120    private final static int DEFAULT_MAX_CALL_DELAY = 5000;
121
122    private final PriorityFunction priority;
123    private final int maxDelay;
124
125    public CallPriorityComparator(final Configuration conf, final PriorityFunction priority) {
126      this.priority = priority;
127      this.maxDelay = conf.getInt(QUEUE_MAX_CALL_DELAY_CONF_KEY, DEFAULT_MAX_CALL_DELAY);
128    }
129
130    @Override
131    public int compare(CallRunner a, CallRunner b) {
132      RpcServer.Call callA = a.getCall();
133      RpcServer.Call callB = b.getCall();
134      long deadlineA = priority.getDeadline(callA.getHeader(), callA.param);
135      long deadlineB = priority.getDeadline(callB.getHeader(), callB.param);
136      deadlineA = callA.timestamp + Math.min(deadlineA, maxDelay);
137      deadlineB = callB.timestamp + Math.min(deadlineB, maxDelay);
138      return (int)(deadlineA - deadlineB);
139    }
140  }
141
142  private int port;
143  private final PriorityFunction priority;
144  private final RpcExecutor callExecutor;
145  private final RpcExecutor priorityExecutor;
146  private final RpcExecutor replicationExecutor;
147
148  /** What level a high priority call is at. */
149  private final int highPriorityLevel;
150
151  private Abortable abortable = null;
152
153  /**
154   * @param conf
155   * @param handlerCount the number of handler threads that will be used to process calls
156   * @param priorityHandlerCount How many threads for priority handling.
157   * @param replicationHandlerCount How many threads for replication handling.
158   * @param highPriorityLevel
159   * @param priority Function to extract request priority.
160   */
161  public SimpleRpcScheduler(
162      Configuration conf,
163      int handlerCount,
164      int priorityHandlerCount,
165      int replicationHandlerCount,
166      PriorityFunction priority,
167      Abortable server,
168      int highPriorityLevel) {
169    int maxQueueLength = conf.getInt("hbase.ipc.server.max.callqueue.length",
170        handlerCount * RpcServer.DEFAULT_MAX_CALLQUEUE_LENGTH_PER_HANDLER);
171    this.priority = priority;
172    this.highPriorityLevel = highPriorityLevel;
173    this.abortable = server;
174
175    String callQueueType = conf.get(CALL_QUEUE_TYPE_CONF_KEY,
176      CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE);
177    float callqReadShare = conf.getFloat(CALL_QUEUE_READ_SHARE_CONF_KEY, 0);
178    float callqScanShare = conf.getFloat(CALL_QUEUE_SCAN_SHARE_CONF_KEY, 0);
179
180    int codelTargetDelay = conf.getInt(CALL_QUEUE_CODEL_TARGET_DELAY,
181      CALL_QUEUE_CODEL_DEFAULT_TARGET_DELAY);
182    int codelInterval = conf.getInt(CALL_QUEUE_CODEL_INTERVAL,
183      CALL_QUEUE_CODEL_DEFAULT_INTERVAL);
184    double codelLifoThreshold = conf.getDouble(CALL_QUEUE_CODEL_LIFO_THRESHOLD,
185      CALL_QUEUE_CODEL_DEFAULT_LIFO_THRESHOLD);
186
187    float callQueuesHandlersFactor = conf.getFloat(CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 0);
188    int numCallQueues = Math.max(1, (int)Math.round(handlerCount * callQueuesHandlersFactor));
189
190    LOG.info("Using " + callQueueType + " as user call queue, count=" + numCallQueues);
191
192    if (numCallQueues > 1 && callqReadShare > 0) {
193      // multiple read/write queues
194      if (callQueueType.equals(CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
195        CallPriorityComparator callPriority = new CallPriorityComparator(conf, this.priority);
196        callExecutor = new RWQueueRpcExecutor("RW.default", handlerCount, numCallQueues,
197            callqReadShare, callqScanShare, maxQueueLength, conf, abortable,
198            BoundedPriorityBlockingQueue.class, callPriority);
199      } else if (callQueueType.equals(CALL_QUEUE_TYPE_CODEL_CONF_VALUE)) {
200        Object[] callQueueInitArgs = {maxQueueLength, codelTargetDelay, codelInterval,
201          codelLifoThreshold, numGeneralCallsDropped, numLifoModeSwitches};
202        callExecutor = new RWQueueRpcExecutor("B.default", handlerCount,
203          numCallQueues, callqReadShare, callqScanShare,
204          AdaptiveLifoCoDelCallQueue.class, callQueueInitArgs,
205          AdaptiveLifoCoDelCallQueue.class, callQueueInitArgs);
206      } else {
207        callExecutor = new RWQueueRpcExecutor("RW.default", handlerCount, numCallQueues,
208          callqReadShare, callqScanShare, maxQueueLength, conf, abortable);
209      }
210    } else {
211      // multiple queues
212      if (callQueueType.equals(CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
213        CallPriorityComparator callPriority = new CallPriorityComparator(conf, this.priority);
214        callExecutor = new BalancedQueueRpcExecutor("B.default", handlerCount, numCallQueues,
215          conf, abortable, BoundedPriorityBlockingQueue.class, maxQueueLength, callPriority);
216      } else if (callQueueType.equals(CALL_QUEUE_TYPE_CODEL_CONF_VALUE)) {
217        callExecutor = new BalancedQueueRpcExecutor("B.default", handlerCount, numCallQueues,
218          conf, abortable, AdaptiveLifoCoDelCallQueue.class, maxQueueLength,
219          codelTargetDelay, codelInterval, codelLifoThreshold,
220          numGeneralCallsDropped, numLifoModeSwitches);
221      } else {
222        callExecutor = new BalancedQueueRpcExecutor("B.default", handlerCount,
223            numCallQueues, maxQueueLength, conf, abortable);
224      }
225    }
226
227    // Create 2 queues to help priorityExecutor be more scalable.
228    this.priorityExecutor = priorityHandlerCount > 0 ?
229        new BalancedQueueRpcExecutor("Priority", priorityHandlerCount, 2, maxQueueLength) : null;
230
231   this.replicationExecutor =
232     replicationHandlerCount > 0 ? new BalancedQueueRpcExecutor("Replication",
233       replicationHandlerCount, 1, maxQueueLength, conf, abortable) : null;
234  }
235
236  public SimpleRpcScheduler(
237        Configuration conf,
238        int handlerCount,
239        int priorityHandlerCount,
240        int replicationHandlerCount,
241        PriorityFunction priority,
242        int highPriorityLevel) {
243    this(conf, handlerCount, priorityHandlerCount, replicationHandlerCount, priority,
244      null, highPriorityLevel);
245  }
246
247  @Override
248  public void init(Context context) {
249    this.port = context.getListenerAddress().getPort();
250  }
251
252  @Override
253  public void start() {
254    callExecutor.start(port);
255    if (priorityExecutor != null) priorityExecutor.start(port);
256    if (replicationExecutor != null) replicationExecutor.start(port);
257  }
258
259  @Override
260  public void stop() {
261    callExecutor.stop();
262    if (priorityExecutor != null) priorityExecutor.stop();
263    if (replicationExecutor != null) replicationExecutor.stop();
264  }
265
266  @Override
267  public boolean dispatch(CallRunner callTask) throws InterruptedException {
268    RpcServer.Call call = callTask.getCall();
269    int level = priority.getPriority(call.getHeader(), call.param, call.getRequestUser());
270    if (priorityExecutor != null && level > highPriorityLevel) {
271      return priorityExecutor.dispatch(callTask);
272    } else if (replicationExecutor != null && level == HConstants.REPLICATION_QOS) {
273      return replicationExecutor.dispatch(callTask);
274    } else {
275      return callExecutor.dispatch(callTask);
276    }
277  }

280  public int getGeneralQueueLength() {
281    return callExecutor.getQueueLength();
282  }
283
284  @Override
285  public int getPriorityQueueLength() {
286    return priorityExecutor == null ? 0 : priorityExecutor.getQueueLength();
287  }
288
289  @Override
290  public int getReplicationQueueLength() {
291    return replicationExecutor == null ? 0 : replicationExecutor.getQueueLength();
292  }
293
294  @Override
295  public int getActiveRpcHandlerCount() {
296    return callExecutor.getActiveHandlerCount() +
297           (priorityExecutor == null ? 0 : priorityExecutor.getActiveHandlerCount()) +
298           (replicationExecutor == null ? 0 : replicationExecutor.getActiveHandlerCount());
299  }
300
301  @Override
302  public long getNumGeneralCallsDropped() {
303    return numGeneralCallsDropped.get();
304  }
305
306  @Override
307  public long getNumLifoModeSwitches() {
308    return numLifoModeSwitches.get();
309  }
