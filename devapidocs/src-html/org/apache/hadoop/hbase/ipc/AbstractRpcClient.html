052/**
053 * Provides the basics for a RpcClient implementation like configuration and Logging.
054 */
055@InterfaceAudience.Private
056public abstract class AbstractRpcClient implements RpcClient {
057  // Log level is being changed in tests
058  public static final Log LOG = LogFactory.getLog(AbstractRpcClient.class);
059
060  protected final Configuration conf;
061  protected String clusterId;
062  protected final SocketAddress localAddr;
063  protected final MetricsConnection metrics;
064
065  protected UserProvider userProvider;
066  protected final IPCUtil ipcUtil;
067
068  protected final int minIdleTimeBeforeClose; // if the connection is idle for more than this
069  // time (in ms), it will be closed at any moment.
070  protected final int maxRetries; //the max. no. of retries for socket connections
071  protected final long failureSleep; // Time to sleep before retry on failure.
072  protected final boolean tcpNoDelay; // if T then disable Nagle's Algorithm
073  protected final boolean tcpKeepAlive; // if T then use keepalives
074  protected final Codec codec;
075  protected final CompressionCodec compressor;
076  protected final boolean fallbackAllowed;
077
078  protected final int connectTO;
079  protected final int readTO;
080  protected final int writeTO;

082  /**
083   * Construct an IPC client for the cluster <code>clusterId</code>
084   *
085   * @param conf configuration
086   * @param clusterId the cluster id
087   * @param localAddr client socket bind address.
088   * @param metrics the connection metrics
089   */
090  public AbstractRpcClient(Configuration conf, String clusterId, SocketAddress localAddr,
091      MetricsConnection metrics) {
092    this.userProvider = UserProvider.instantiate(conf);
093    this.localAddr = localAddr;
094    this.tcpKeepAlive = conf.getBoolean("hbase.ipc.client.tcpkeepalive", true);
095    this.clusterId = clusterId != null ? clusterId : HConstants.CLUSTER_ID_DEFAULT;
096    this.failureSleep = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
097        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
098    this.maxRetries = conf.getInt("hbase.ipc.client.connect.max.retries", 0);
099    this.tcpNoDelay = conf.getBoolean("hbase.ipc.client.tcpnodelay", true);
100    this.ipcUtil = new IPCUtil(conf);
101
102    this.minIdleTimeBeforeClose = conf.getInt(IDLE_TIME, 120000); // 2 minutes
103    this.conf = conf;
104    this.codec = getCodec();
105    this.compressor = getCompressor(conf);
106    this.fallbackAllowed = conf.getBoolean(IPC_CLIENT_FALLBACK_TO_SIMPLE_AUTH_ALLOWED_KEY,
107        IPC_CLIENT_FALLBACK_TO_SIMPLE_AUTH_ALLOWED_DEFAULT);
108    this.connectTO = conf.getInt(SOCKET_TIMEOUT_CONNECT, DEFAULT_SOCKET_TIMEOUT_CONNECT);
109    this.readTO = conf.getInt(SOCKET_TIMEOUT_READ, DEFAULT_SOCKET_TIMEOUT_READ);
110    this.writeTO = conf.getInt(SOCKET_TIMEOUT_WRITE, DEFAULT_SOCKET_TIMEOUT_WRITE);
111    this.metrics = metrics;
112
113    // login the server principal (if using secure Hadoop)
114    if (LOG.isDebugEnabled()) {
115      LOG.debug("Codec=" + this.codec + ", compressor=" + this.compressor +
116          ", tcpKeepAlive=" + this.tcpKeepAlive +
117          ", tcpNoDelay=" + this.tcpNoDelay +
118          ", connectTO=" + this.connectTO +
119          ", readTO=" + this.readTO +
120          ", writeTO=" + this.writeTO +
121          ", minIdleTimeBeforeClose=" + this.minIdleTimeBeforeClose +
122          ", maxRetries=" + this.maxRetries +
123          ", fallbackAllowed=" + this.fallbackAllowed +
124          ", bind address=" + (this.localAddr != null ? this.localAddr : "null"));
125    }
126  }

128  @VisibleForTesting
129  public static String getDefaultCodec(final Configuration c) {
130    // If "hbase.client.default.rpc.codec" is empty string -- you can't set it to null because
131    // Configuration will complain -- then no default codec (and we'll pb everything).  Else
132    // default is KeyValueCodec
133    return c.get(DEFAULT_CODEC_CLASS, KeyValueCodec.class.getCanonicalName());
134  }
135
136  /**
137   * Encapsulate the ugly casting and RuntimeException conversion in private method.
138   * @return Codec to use on this client.
139   */
140  Codec getCodec() {
141    // For NO CODEC, "hbase.client.rpc.codec" must be configured with empty string AND
142    // "hbase.client.default.rpc.codec" also -- because default is to do cell block encoding.
143    String className = conf.get(HConstants.RPC_CODEC_CONF_KEY, getDefaultCodec(this.conf));
144    if (className == null || className.length() == 0) {
145      return null;
146    }
147    try {
148      return (Codec)Class.forName(className).newInstance();
149    } catch (Exception e) {
150      throw new RuntimeException("Failed getting codec " + className, e);
151    }
152  }
153
154  @Override
155  public boolean hasCellBlockSupport() {
156    return this.codec != null;
157  }

159  /**
160   * Encapsulate the ugly casting and RuntimeException conversion in private method.
161   * @param conf configuration
162   * @return The compressor to use on this client.
163   */
164  private static CompressionCodec getCompressor(final Configuration conf) {
165    String className = conf.get("hbase.client.rpc.compressor", null);
166    if (className == null || className.isEmpty()) {
167      return null;
168    }
169    try {
170      return (CompressionCodec)Class.forName(className).newInstance();
171    } catch (Exception e) {
172      throw new RuntimeException("Failed getting compressor " + className, e);
173    }
174  }

176  /**
177   * Return the pool type specified in the configuration, which must be set to
178   * either {@link org.apache.hadoop.hbase.util.PoolMap.PoolType#RoundRobin} or
179   * {@link org.apache.hadoop.hbase.util.PoolMap.PoolType#ThreadLocal},
180   * otherwise default to the former.
181   *
182   * For applications with many user threads, use a small round-robin pool. For
183   * applications with few user threads, you may want to try using a
184   * thread-local pool. In any case, the number of {@link org.apache.hadoop.hbase.ipc.RpcClient}
185   * instances should not exceed the operating system's hard limit on the number of
186   * connections.
187   *
188   * @param config configuration
189   * @return either a {@link org.apache.hadoop.hbase.util.PoolMap.PoolType#RoundRobin} or
190   *         {@link org.apache.hadoop.hbase.util.PoolMap.PoolType#ThreadLocal}
191   */
192  protected static PoolMap.PoolType getPoolType(Configuration config) {
193    return PoolMap.PoolType
194        .valueOf(config.get(HConstants.HBASE_CLIENT_IPC_POOL_TYPE), PoolMap.PoolType.RoundRobin,
195            PoolMap.PoolType.ThreadLocal);
196  }

198  /**
199   * Return the pool size specified in the configuration, which is applicable only if
200   * the pool type is {@link org.apache.hadoop.hbase.util.PoolMap.PoolType#RoundRobin}.
201   *
202   * @param config configuration
203   * @return the maximum pool size
204   */
205  protected static int getPoolSize(Configuration config) {
206    return config.getInt(HConstants.HBASE_CLIENT_IPC_POOL_SIZE, 1);
207  }

209  /**
210   * Make a blocking call. Throws exceptions if there are network problems or if the remote code
211   * threw an exception.
212   *
213   * @param ticket Be careful which ticket you pass. A new user will mean a new Connection.
214   *               {@link UserProvider#getCurrent()} makes a new instance of User each time so
215   *               will be a
216   *               new Connection each time.
217   * @return A pair with the Message response and the Cell data (if any).
218   */
219  Message callBlockingMethod(Descriptors.MethodDescriptor md, PayloadCarryingRpcController pcrc,
220      Message param, Message returnType, final User ticket, final InetSocketAddress isa)
221      throws ServiceException {
222    if (pcrc == null) {
223      pcrc = new PayloadCarryingRpcController();
224    }
225
226    Pair<Message, CellScanner> val;
227    try {
228      final MetricsConnection.CallStats cs = MetricsConnection.newCallStats();
229      cs.setStartTime(EnvironmentEdgeManager.currentTime());
230      val = call(pcrc, md, param, returnType, ticket, isa, cs);
231      // Shove the results into controller so can be carried across the proxy/pb service void.
232      pcrc.setCellScanner(val.getSecond());
233
234      cs.setCallTimeMs(EnvironmentEdgeManager.currentTime() - cs.getStartTime());
235      if (metrics != null) {
236        metrics.updateRpc(md, param, cs);
237      }
238      if (LOG.isTraceEnabled()) {
239        LOG.trace("Call: " + md.getName() + ", callTime: " + cs.getCallTimeMs() + "ms");
240      }
241      return val.getFirst();
242    } catch (Throwable e) {
243      throw new ServiceException(e);
244    }
245  }

247  /**
248   * Make a call, passing <code>param</code>, to the IPC server running at
249   * <code>address</code> which is servicing the <code>protocol</code> protocol,
250   * with the <code>ticket</code> credentials, returning the value.
251   * Throws exceptions if there are network problems or if the remote code
252   * threw an exception.
253   *
254   * @param ticket Be careful which ticket you pass. A new user will mean a new Connection.
255   *               {@link UserProvider#getCurrent()} makes a new instance of User each time so
256   *               will be a
257   *               new Connection each time.
258   * @return A pair with the Message response and the Cell data (if any).
259   * @throws InterruptedException if call is interrupted
260   * @throws java.io.IOException if transport failed
261   */
262  protected abstract Pair<Message, CellScanner> call(PayloadCarryingRpcController pcrc,
263      Descriptors.MethodDescriptor md, Message param, Message returnType, User ticket,
264      InetSocketAddress isa, MetricsConnection.CallStats callStats)
265      throws IOException, InterruptedException;

267  @Override
268  public BlockingRpcChannel createBlockingRpcChannel(final ServerName sn, final User ticket,
269      int defaultOperationTimeout) {
270    return new BlockingRpcChannelImplementation(this, sn, ticket, defaultOperationTimeout);
271  }

273  /**
274   * Configure a payload carrying controller
275   * @param controller to configure
276   * @param channelOperationTimeout timeout for operation
277   * @return configured payload controller
278   */
279  static PayloadCarryingRpcController configurePayloadCarryingRpcController(
280      RpcController controller, int channelOperationTimeout) {
281    PayloadCarryingRpcController pcrc;
282    if (controller != null && controller instanceof PayloadCarryingRpcController) {
283      pcrc = (PayloadCarryingRpcController) controller;
284      if (!pcrc.hasCallTimeout()) {
285        pcrc.setCallTimeout(channelOperationTimeout);
286      }
287    } else {
288      pcrc = new PayloadCarryingRpcController();
289      pcrc.setCallTimeout(channelOperationTimeout);
290    }
291    return pcrc;
292  }

294  /**
295   * Takes an Exception and the address we were trying to connect to and return an IOException with
296   * the input exception as the cause. The new exception provides the stack trace of the place where
297   * the exception is thrown and some extra diagnostics information. If the exception is
298   * ConnectException or SocketTimeoutException, return a new one of the same type; Otherwise return
299   * an IOException.
300   * @param addr target address
301   * @param exception the relevant exception
302   * @return an exception to throw
303   */
304  protected IOException wrapException(InetSocketAddress addr, Exception exception) {
305    if (exception instanceof ConnectException) {
306      // connection refused; include the host:port in the error
307      return (ConnectException) new ConnectException("Call to " + addr
308          + " failed on connection exception: " + exception).initCause(exception);
309    } else if (exception instanceof SocketTimeoutException) {
310      return (SocketTimeoutException) new SocketTimeoutException("Call to " + addr
311          + " failed because " + exception).initCause(exception);
312    } else if (exception instanceof ConnectionClosingException) {
313      return (ConnectionClosingException) new ConnectionClosingException("Call to " + addr
314          + " failed on local exception: " + exception).initCause(exception);
315    } else {
316      return (IOException) new IOException("Call to " + addr + " failed on local exception: "
317          + exception).initCause(exception);
318    }
319  }

321  /**
322   * Blocking rpc channel that goes via hbase rpc.
323   */
324  @VisibleForTesting
325  public static class BlockingRpcChannelImplementation implements BlockingRpcChannel {
326    private final InetSocketAddress isa;
327    private final AbstractRpcClient rpcClient;
328    private final User ticket;
329    private final int channelOperationTimeout;
330
331    /**
332     * @param channelOperationTimeout - the default timeout when no timeout is given
333     */
334    protected BlockingRpcChannelImplementation(final AbstractRpcClient rpcClient,
335        final ServerName sn, final User ticket, int channelOperationTimeout) {
336      this.isa = new InetSocketAddress(sn.getHostname(), sn.getPort());
337      this.rpcClient = rpcClient;
338      this.ticket = ticket;
339      this.channelOperationTimeout = channelOperationTimeout;
340    }
341
342    @Override
343    public Message callBlockingMethod(Descriptors.MethodDescriptor md, RpcController controller,
344        Message param, Message returnType) throws ServiceException {
345      PayloadCarryingRpcController pcrc = configurePayloadCarryingRpcController(
346          controller,
347          channelOperationTimeout);
348
349      return this.rpcClient.callBlockingMethod(md, pcrc, param, returnType, this.ticket, this.isa);
350    }
351  }
