185    /**
186     * If the client wants to interrupt its calls easily (i.e. call Thread#interrupt),
187     *  it gets into a java issue: an interruption during a write closes the socket/channel.
188     * A way to avoid this is to use a different thread for writing. This way, on interruptions,
189     *  we either cancel the writes or ignore the answer if the write is already done, but we
190     *  don't stop the write in the middle.
191     * This adds a thread per region server in the client, so it's kept as an option.
192     * <p>
193     * The implementation is simple: the client threads adds their call to the queue, and then
194     *  wait for an answer. The CallSender blocks on the queue, and writes the calls one
195     *  after the other. On interruption, the client cancels its call. The CallSender checks that
196     *  the call has not been canceled before writing it.
197     * </p>
198     * When the connection closes, all the calls not yet sent are dismissed. The client thread
199     *  is notified with an appropriate exception, as if the call was already sent but the answer
200     *  not yet received.
201     * </p>
202     */
203    private class CallSender extends Thread implements Closeable {
204      protected final BlockingQueue<CallFuture> callsToWrite;
205
206
207      public CallFuture sendCall(Call call, int priority, Span span)
208          throws InterruptedException, IOException {
209        CallFuture cts = new CallFuture(call, priority, span);
210        if (!callsToWrite.offer(cts)) {
211          throw new IOException("Can't add the call " + call.id +
212              " to the write queue. callsToWrite.size()=" + callsToWrite.size());
213        }
214        checkIsOpen(); // We check after the put, to be sure that the call we added won't stay
215                       //  in the list while the cleanup was already done.
216        return cts;
217      }
218
219      @Override
220      public void close(){
221        assert shouldCloseConnection.get();
222        callsToWrite.offer(CallFuture.DEATH_PILL);
223        // We don't care if we can't add the death pill to the queue: the writer
224        //  won't be blocked in the 'take', as its queue is full.
225      }
226
227      CallSender(String name, Configuration conf) {
228        int queueSize = conf.getInt("hbase.ipc.client.write.queueSize", 1000);
229        callsToWrite = new ArrayBlockingQueue<>(queueSize);
230        setDaemon(true);
231        setName(name + " - writer");
232      }
233
234      public void remove(CallFuture cts){
235        callsToWrite.remove(cts);
236
237        // By removing the call from the expected call list, we make the list smaller, but
238        //  it means as well that we don't know how many calls we cancelled.
239        calls.remove(cts.call.id);
240        cts.call.callComplete();
241      }
242
243      /**
244       * Reads the call from the queue, write them on the socket.
245       */
246      @Override
247      public void run() {
248        while (!shouldCloseConnection.get()) {
249          CallFuture cts = null;
250          try {
251            cts = callsToWrite.take();
252          } catch (InterruptedException e) {
253            markClosed(new InterruptedIOException());
254          }
255
256          if (cts == null || cts == CallFuture.DEATH_PILL) {
257            assert shouldCloseConnection.get();
258            break;
259          }
260
261          if (cts.call.done) {
262            continue;
263          }
264
265          if (cts.call.checkAndSetTimeout()) {
266            continue;
267          }
268
269          try {
270            Connection.this.tracedWriteRequest(cts.call, cts.priority, cts.span);
271          } catch (IOException e) {
272            if (LOG.isDebugEnabled()) {
273              LOG.debug("call write error for call #" + cts.call.id
274                + ", message =" + e.getMessage());
275            }
276            cts.call.setException(e);
277            markClosed(e);
278          }
279        }
280
281        cleanup();
282      }
283
284      /**
285       * Cleans the call not yet sent when we finish.
286       */
287      private void cleanup() {
288        assert shouldCloseConnection.get();
289
290        IOException ie = new ConnectionClosingException("Connection to " + server + " is closing.");
291        while (true) {
292          CallFuture cts = callsToWrite.poll();
293          if (cts == null) {
294            break;
295          }
296          if (cts.call != null && !cts.call.done) {
297            cts.call.setException(ie);
298          }
299        }
300      }
301    }

303    Connection(ConnectionId remoteId, final Codec codec, final CompressionCodec compressor)
304    throws IOException {
305      if (remoteId.getAddress().isUnresolved()) {
306        throw new UnknownHostException("unknown host: " + remoteId.getAddress().getHostName());
307      }
308      this.server = remoteId.getAddress();
309      this.codec = codec;
310      this.compressor = compressor;
311
312      UserGroupInformation ticket = remoteId.getTicket().getUGI();
313      SecurityInfo securityInfo = SecurityInfo.getInfo(remoteId.getServiceName());
314      this.useSasl = userProvider.isHBaseSecurityEnabled();
315      if (useSasl && securityInfo != null) {
316        AuthenticationProtos.TokenIdentifier.Kind tokenKind = securityInfo.getTokenKind();
317        if (tokenKind != null) {
318          TokenSelector<? extends TokenIdentifier> tokenSelector =
319              tokenHandlers.get(tokenKind);
320          if (tokenSelector != null) {
321            token = tokenSelector.selectToken(new Text(clusterId),
322                ticket.getTokens());
323          } else if (LOG.isDebugEnabled()) {
324            LOG.debug("No token selector found for type "+tokenKind);
325          }
326        }
327        String serverKey = securityInfo.getServerPrincipal();
328        if (serverKey == null) {
329          throw new IOException(
330              "Can't obtain server Kerberos config key from SecurityInfo");
331        }
332        serverPrincipal = SecurityUtil.getServerPrincipal(
333            conf.get(serverKey), server.getAddress().getCanonicalHostName().toLowerCase(Locale.ROOT));
334        if (LOG.isDebugEnabled()) {
335          LOG.debug("RPC Server Kerberos principal name for service="
336              + remoteId.getServiceName() + " is " + serverPrincipal);
337        }
338      }
339
340      if (!useSasl) {
341        authMethod = AuthMethod.SIMPLE;
342      } else if (token != null) {
343        authMethod = AuthMethod.DIGEST;
344      } else {
345        authMethod = AuthMethod.KERBEROS;
346      }
347
348      if (LOG.isDebugEnabled()) {
349        LOG.debug("Use " + authMethod + " authentication for service " + remoteId.serviceName +
350          ", sasl=" + useSasl);
351      }
352      reloginMaxBackoff = conf.getInt("hbase.security.relogin.maxbackoff", 5000);
353      this.remoteId = remoteId;
354
355      ConnectionHeader.Builder builder = ConnectionHeader.newBuilder();
356      builder.setServiceName(remoteId.getServiceName());
357      UserInformation userInfoPB = getUserInfo(ticket);
358      if (userInfoPB != null) {
359        builder.setUserInfo(userInfoPB);
360      }
361      if (this.codec != null) {
362        builder.setCellBlockCodecClass(this.codec.getClass().getCanonicalName());
363      }
364      if (this.compressor != null) {
365        builder.setCellBlockCompressorClass(this.compressor.getClass().getCanonicalName());
366      }
367      builder.setVersionInfo(ProtobufUtil.getVersionInfo());
368      this.header = builder.build();
369
370      this.setName("IPC Client (" + socketFactory.hashCode() +") connection to " +
371        remoteId.getAddress().toString() +
372        ((ticket==null)?" from an unknown user": (" from "
373        + ticket.getUserName())));
374      this.setDaemon(true);
375
376      if (conf.getBoolean(SPECIFIC_WRITE_THREAD, false)) {
377        callSender = new CallSender(getName(), conf);
378        callSender.start();
379      } else {
380        callSender = null;
381      }
382    }

384    private synchronized UserInformation getUserInfo(UserGroupInformation ugi) {
385      if (ugi == null || authMethod == AuthMethod.DIGEST) {
386        // Don't send user for token auth
387        return null;
388      }
389      UserInformation.Builder userInfoPB = UserInformation.newBuilder();
390      if (authMethod == AuthMethod.KERBEROS) {
391        // Send effective user for Kerberos auth
392        userInfoPB.setEffectiveUser(ugi.getUserName());
393      } else if (authMethod == AuthMethod.SIMPLE) {
394        //Send both effective user and real user for simple auth
395        userInfoPB.setEffectiveUser(ugi.getUserName());
396        if (ugi.getRealUser() != null) {
397          userInfoPB.setRealUser(ugi.getRealUser().getUserName());
398        }
399      }
400      return userInfoPB.build();
401    }

403    protected synchronized void setupConnection() throws IOException {
404      short ioFailures = 0;
405      short timeoutFailures = 0;
406      while (true) {
407        try {
408          this.socket = socketFactory.createSocket();
409          this.socket.setTcpNoDelay(tcpNoDelay);
410          this.socket.setKeepAlive(tcpKeepAlive);
411          if (localAddr != null) {
412            this.socket.bind(localAddr);
413          }
414          NetUtils.connect(this.socket, remoteId.getAddress(), connectTO);
415          this.socket.setSoTimeout(readTO);
416          return;
417        } catch (SocketTimeoutException toe) {
418          /* The max number of retries is 45,
419           * which amounts to 20s*45 = 15 minutes retries.
420           */
421          handleConnectionFailure(timeoutFailures++, maxRetries, toe);
422        } catch (IOException ie) {
423          handleConnectionFailure(ioFailures++, maxRetries, ie);
424        }
425      }
426    }

428    protected synchronized void closeConnection() {
429      if (socket == null) {
430        return;
431      }
432
433      // close the current connection
434      try {
435        if (socket.getOutputStream() != null) {
436          socket.getOutputStream().close();
437        }
438      } catch (IOException ignored) {  // Can happen if the socket is already closed
439        if (LOG.isTraceEnabled()){
440          LOG.trace("ignored", ignored);
441        }
442      }
443      try {
444        if (socket.getInputStream() != null) {
445          socket.getInputStream().close();
446        }
447      } catch (IOException ignored) {  // Can happen if the socket is already closed
448        if (LOG.isTraceEnabled()){
449          LOG.trace("ignored", ignored);
450        }
451      }
452      try {
453        if (socket.getChannel() != null) {
454          socket.getChannel().close();
455        }
456      } catch (IOException ignored) {  // Can happen if the socket is already closed
457        if (LOG.isTraceEnabled()){
458          LOG.trace("ignored", ignored);
459        }
460      }
461      try {
462        socket.close();
463      } catch (IOException e) {
464        LOG.warn("Not able to close a socket", e);
465      }
466
467      // set socket to null so that the next call to setupIOstreams
468      // can start the process of connect all over again.
469      socket = null;
470    }

472    /**
473     *  Handle connection failures
474     *
475     * If the current number of retries is equal to the max number of retries,
476     * stop retrying and throw the exception; Otherwise backoff N seconds and
477     * try connecting again.
478     *
479     * This Method is only called from inside setupIOstreams(), which is
480     * synchronized. Hence the sleep is synchronized; the locks will be retained.
481     *
482     * @param curRetries current number of retries
483     * @param maxRetries max number of retries allowed
484     * @param ioe failure reason
485     * @throws IOException if max number of retries is reached
486     */
487    private void handleConnectionFailure(int curRetries, int maxRetries, IOException ioe)
488    throws IOException {
489      closeConnection();
490
491      // throw the exception if the maximum number of retries is reached
492      if (curRetries >= maxRetries || ExceptionUtil.isInterrupt(ioe)) {
493        throw ioe;
494      }
495
496      // otherwise back off and retry
497      try {
498        Thread.sleep(failureSleep);
499      } catch (InterruptedException ie) {
500        ExceptionUtil.rethrowIfInterrupt(ie);
501      }
502
503      LOG.info("Retrying connect to server: " + remoteId.getAddress() +
504        " after sleeping " + failureSleep + "ms. Already tried " + curRetries +
505        " time(s).");
506    }

517    /* wait till someone signals us to start reading RPC response or
518     * it is idle too long, it is marked as to be closed,
519     * or the client is marked as not running.
520     *
521     * @return true if it is time to read a response; false otherwise.
522     */
523    protected synchronized boolean waitForWork() throws InterruptedException {
524      // beware of the concurrent access to the calls list: we can add calls, but as well
525      //  remove them.
526      long waitUntil = EnvironmentEdgeManager.currentTime() + minIdleTimeBeforeClose;
527
528      while (true) {
529        if (shouldCloseConnection.get()) {
530          return false;
531        }
532
533        if (!running.get()) {
534          markClosed(new IOException("stopped with " + calls.size() + " pending request(s)"));
535          return false;
536        }
537
538        if (!calls.isEmpty()) {
539          // shouldCloseConnection can be set to true by a parallel thread here. The caller
540          //  will need to check anyway.
541          return true;
542        }
543
544        if (EnvironmentEdgeManager.currentTime() >= waitUntil) {
545          // Connection is idle.
546          // We expect the number of calls to be zero here, but actually someone can
547          //  adds a call at the any moment, as there is no synchronization between this task
548          //  and adding new calls. It's not a big issue, but it will get an exception.
549          markClosed(new IOException(
550              "idle connection closed with " + calls.size() + " pending request(s)"));
551          return false;
552        }
553
554        wait(Math.min(minIdleTimeBeforeClose, 1000));
555      }
556    }

562    @Override
563    public void run() {
564      if (LOG.isTraceEnabled()) {
565        LOG.trace(getName() + ": starting, connections " + connections.size());
566      }
567
568      try {
569        while (waitForWork()) { // Wait here for work - read or close connection
570          readResponse();
571        }
572      } catch (InterruptedException t) {
573        if (LOG.isTraceEnabled()) {
574          LOG.trace(getName() + ": interrupted while waiting for call responses");
575        }
576        markClosed(ExceptionUtil.asInterrupt(t));
577      } catch (Throwable t) {
578        if (LOG.isDebugEnabled()) {
579          LOG.debug(getName() + ": unexpected throwable while waiting for call responses", t);
580        }
581        markClosed(new IOException("Unexpected throwable while waiting call responses", t));
582      }
583
584      close();
585
586      if (LOG.isTraceEnabled()) {
587        LOG.trace(getName() + ": stopped, connections " + connections.size());
588      }
589    }

602    private synchronized boolean shouldAuthenticateOverKrb() throws IOException {
603      UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
604      UserGroupInformation currentUser =
605        UserGroupInformation.getCurrentUser();
606      UserGroupInformation realUser = currentUser.getRealUser();
607      return authMethod == AuthMethod.KERBEROS &&
608          loginUser != null &&
609          //Make sure user logged in using Kerberos either keytab or TGT
610          loginUser.hasKerberosCredentials() &&
611          // relogin only in case it is the login user (e.g. JT)
612          // or superuser (like oozie).
613          (loginUser.equals(currentUser) || loginUser.equals(realUser));
614    }

624    /**
625     * If multiple clients with the same principal try to connect
626     * to the same server at the same time, the server assumes a
627     * replay attack is in progress. This is a feature of kerberos.
628     * In order to work around this, what is done is that the client
629     * backs off randomly and tries to initiate the connection
630     * again.
631     * The other problem is to do with ticket expiry. To handle that,
632     * a relogin is attempted.
633     * <p>
634     * The retry logic is governed by the {@link #shouldAuthenticateOverKrb}
635     * method. In case when the user doesn't have valid credentials, we don't
636     * need to retry (from cache or ticket). In such cases, it is prudent to
637     * throw a runtime exception when we receive a SaslException from the
638     * underlying authentication implementation, so there is no retry from
639     * other high level (for eg, HCM or HBaseAdmin).
640     * </p>
641     */
642    private synchronized void handleSaslConnectionFailure(
643        final int currRetries,
644        final int maxRetries, final Exception ex, final Random rand,
645        final UserGroupInformation user)
646    throws IOException, InterruptedException{
647      user.doAs(new PrivilegedExceptionAction<Object>() {
648        @Override
649        public Object run() throws IOException, InterruptedException {
650          closeConnection();
651          if (shouldAuthenticateOverKrb()) {
652            if (currRetries < maxRetries) {
653              if (LOG.isDebugEnabled()) {
654                LOG.debug("Exception encountered while connecting to " +
655                    "the server : " + ex);
656              }
657              //try re-login
658              if (UserGroupInformation.isLoginKeytabBased()) {
659                UserGroupInformation.getLoginUser().reloginFromKeytab();
660              } else {
661                UserGroupInformation.getLoginUser().reloginFromTicketCache();
662              }
663              disposeSasl();
664              //have granularity of milliseconds
665              //we are sleeping with the Connection lock held but since this
666              //connection instance is being used for connecting to the server
667              //in question, it is okay
668              Thread.sleep((rand.nextInt(reloginMaxBackoff) + 1));
669              return null;
670            } else {
671              String msg = "Couldn't setup connection for " +
672                UserGroupInformation.getLoginUser().getUserName() +
673                " to " + serverPrincipal;
674              LOG.warn(msg, ex);
675              throw (IOException) new IOException(msg).initCause(ex);
676            }
677          } else {
678            LOG.warn("Exception encountered while connecting to " +
679                "the server : " + ex);
680          }
681          if (ex instanceof RemoteException) {
682            throw (RemoteException)ex;
683          }
684          if (ex instanceof SaslException) {
685            String msg = "SASL authentication failed." +
686              " The most likely cause is missing or invalid credentials." +
687              " Consider 'kinit'.";
688            LOG.fatal(msg, ex);
689            throw new RuntimeException(msg, ex);
690          }
691          throw new IOException(ex);
692        }
693      });
694    }

696    protected synchronized void setupIOstreams() throws IOException {
697      if (socket != null) {
698        // The connection is already available. Perfect.
699        return;
700      }
701
702      if (shouldCloseConnection.get()){
703        throw new ConnectionClosingException("This connection is closing");
704      }
705
706      if (failedServers.isFailedServer(remoteId.getAddress())) {
707        if (LOG.isDebugEnabled()) {
708          LOG.debug("Not trying to connect to " + server +
709              " this server is in the failed servers list");
710        }
711        IOException e = new FailedServerException(
712            "This server is in the failed servers list: " + server);
713        markClosed(e);
714        close();
715        throw e;
716      }
717
718      try {
719        if (LOG.isDebugEnabled()) {
720          LOG.debug("Connecting to " + server);
721        }
722        short numRetries = 0;
723        final short MAX_RETRIES = 5;
724        Random rand = null;
725        while (true) {
726          setupConnection();
727          InputStream inStream = NetUtils.getInputStream(socket);
728          // This creates a socket with a write timeout. This timeout cannot be changed.
729          OutputStream outStream = NetUtils.getOutputStream(socket, writeTO);
730          // Write out the preamble -- MAGIC, version, and auth to use.
731          writeConnectionHeaderPreamble(outStream);
732          if (useSasl) {
733            final InputStream in2 = inStream;
734            final OutputStream out2 = outStream;
735            UserGroupInformation ticket = remoteId.getTicket().getUGI();
736            if (authMethod == AuthMethod.KERBEROS) {
737              if (ticket != null && ticket.getRealUser() != null) {
738                ticket = ticket.getRealUser();
739              }
740            }
741            boolean continueSasl;
742            if (ticket == null){
743              throw new FatalConnectionException("ticket/user is null");
744            }
745            try {
746              continueSasl = ticket.doAs(new PrivilegedExceptionAction<Boolean>() {
747                @Override
748                public Boolean run() throws IOException {
749                  return setupSaslConnection(in2, out2);
750                }
751              });
752            } catch (Exception ex) {
753              ExceptionUtil.rethrowIfInterrupt(ex);
754              if (rand == null) {
755                rand = new Random();
756              }
757              handleSaslConnectionFailure(numRetries++, MAX_RETRIES, ex, rand, ticket);
758              continue;
759            }
760            if (continueSasl) {
761              // Sasl connect is successful. Let's set up Sasl i/o streams.
762              inStream = saslRpcClient.getInputStream(inStream);
763              outStream = saslRpcClient.getOutputStream(outStream);
764            } else {
765              // fall back to simple auth because server told us so.
766              authMethod = AuthMethod.SIMPLE;
767              useSasl = false;
768            }
769          }
770          this.in = new DataInputStream(new BufferedInputStream(inStream));
771          synchronized (this.outLock) {
772            this.out = new DataOutputStream(new BufferedOutputStream(outStream));
773          }
774          // Now write out the connection header
775          writeConnectionHeader();
776
777          // start the receiver thread after the socket connection has been set up
778          start();
779          return;
780        }
781      } catch (Throwable t) {
782        IOException e = ExceptionUtil.asInterrupt(t);
783        if (e == null) {
784          failedServers.addToFailedServers(remoteId.address);
785          if (t instanceof LinkageError) {
786            // probably the hbase hadoop version does not match the running hadoop version
787            e = new DoNotRetryIOException(t);
788          } else if (t instanceof IOException) {
789            e = (IOException) t;
790          } else {
791            e = new IOException("Could not set up IO Streams to " + server, t);
792          }
793        }
794        markClosed(e);
795        close();
796        throw e;
797      }
798    }

800    /**
801     * Write the RPC header: {@code <MAGIC WORD -- 'HBas'> <ONEBYTE_VERSION> <ONEBYTE_AUTH_TYPE>}
802     */
803    private void writeConnectionHeaderPreamble(OutputStream outStream) throws IOException {
804      // Assemble the preamble up in a buffer first and then send it.  Writing individual elements,
805      // they are getting sent across piecemeal according to wireshark and then server is messing
806      // up the reading on occasion (the passed in stream is not buffered yet).
807
808      // Preamble is six bytes -- 'HBas' + VERSION + AUTH_CODE
809      int rpcHeaderLen = HConstants.RPC_HEADER.length;
810      byte [] preamble = new byte [rpcHeaderLen + 2];
811      System.arraycopy(HConstants.RPC_HEADER, 0, preamble, 0, rpcHeaderLen);
812      preamble[rpcHeaderLen] = HConstants.RPC_CURRENT_VERSION;
813      synchronized (this) {
814        preamble[rpcHeaderLen + 1] = authMethod.code;
815      }
816      outStream.write(preamble);
817      outStream.flush();
818    }

820    /**
821     * Write the connection header.
822     */
823    private synchronized void writeConnectionHeader() throws IOException {
824      synchronized (this.outLock) {
825        this.out.writeInt(this.header.getSerializedSize());
826        this.header.writeTo(this.out);
827        this.out.flush();
828      }
829    }

831    /** Close the connection. */
832    protected synchronized void close() {
833      if (!shouldCloseConnection.get()) {
834        LOG.error(getName() + ": the connection is not in the closed state");
835        return;
836      }
837
838      // release the resources
839      // first thing to do;take the connection out of the connection list
840      synchronized (connections) {
841        connections.removeValue(remoteId, this);
842      }
843
844      // close the streams and therefore the socket
845      synchronized(this.outLock) {
846        if (this.out != null) {
847          IOUtils.closeStream(out);
848          this.out = null;
849        }
850      }
851      IOUtils.closeStream(in);
852      this.in = null;
853      if (this.socket != null) {
854        try {
855          this.socket.close();
856          this.socket = null;
857        } catch (IOException e) {
858          LOG.error("Error while closing socket", e);
859        }
860      }
861
862      disposeSasl();
863
864      // log the info
865      if (LOG.isTraceEnabled()) {
866        LOG.trace(getName() + ": closing ipc connection to " + server);
867      }
868
869      cleanupCalls(true);
870
871      if (LOG.isTraceEnabled()) {
872        LOG.trace(getName() + ": ipc connection to " + server + " closed");
873      }
874    }

882    /**
883     * Initiates a call by sending the parameter to the remote server.
884     * Note: this is not called from the Connection thread, but by other
885     * threads.
886     * @see #readResponse()
887     */
888    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="IS2_INCONSISTENT_SYNC",
889        justification="Findbugs is misinterpreting locking missing fact that this.outLock is held")
890    private void writeRequest(Call call, final int priority, Span span) throws IOException {
891      RequestHeader.Builder builder = RequestHeader.newBuilder();
892      builder.setCallId(call.id);
893      if (span != null) {
894        builder.setTraceInfo(
895            RPCTInfo.newBuilder().setParentId(span.getSpanId()).setTraceId(span.getTraceId()));
896      }
897      builder.setMethodName(call.md.getName());
898      builder.setRequestParam(call.param != null);
899      ByteBuffer cellBlock = cellBlockBuilder.buildCellBlock(this.codec, this.compressor, call.cells);
900      if (cellBlock != null) {
901        CellBlockMeta.Builder cellBlockBuilder = CellBlockMeta.newBuilder();
902        cellBlockBuilder.setLength(cellBlock.limit());
903        builder.setCellBlockMeta(cellBlockBuilder.build());
904      }
905      // Only pass priority if there is one set.
906      if (priority != HBaseRpcController.PRIORITY_UNSET) {
907        builder.setPriority(priority);
908      }
909      builder.setTimeout(call.timeout);
910      RequestHeader requestHeader = builder.build();
911
912      setupIOstreams();
913
914      // Now we're going to write the call. We take the lock, then check that the connection
915      //  is still valid, and, if so we do the write to the socket. If the write fails, we don't
916      //  know where we stand, we have to close the connection.
917      checkIsOpen();
918      IOException writeException = null;
919      synchronized (this.outLock) {
920        if (Thread.interrupted()){
921          throw new InterruptedIOException();
922        }
923
924        calls.put(call.id, call); // We put first as we don't want the connection to become idle.
925        checkIsOpen(); // Now we're checking that it didn't became idle in between.
926
927        try {
928          call.callStats.setRequestSizeBytes(IPCUtil.write(this.out, requestHeader, call.param,
929              cellBlock));
930        } catch (IOException e) {
931          // We set the value inside the synchronized block, this way the next in line
932          //  won't even try to write. Otherwise we might miss a call in the calls map?
933          shouldCloseConnection.set(true);
934          writeException = e;
935          interrupt();
936        }
937      }
938
939      // call close outside of the synchronized (outLock) to prevent deadlock - HBASE-14474
940      if (writeException != null) {
941        markClosed(writeException);
942        close();
943      }
944
945      // We added a call, and may be started the connection close. In both cases, we
946      //  need to notify the reader.
947      doNotify();
948
949      // Now that we notified, we can rethrow the exception if any. Otherwise we're good.
950      if (writeException != null){
951        throw writeException;
952      }
953    }
954
955    @edu.umd.cs.findbugs.annotations.SuppressWarnings(value="NN_NAKED_NOTIFY",
956        justification="Presume notifyAll is because we are closing/shutting down")
957    private synchronized void doNotify() {
958      // Make a separate method so can do synchronize and add findbugs annotation; only one
959      // annotation at at time in source 1.7.
960      notifyAll(); // Findbugs: NN_NAKED_NOTIFY
961    }

963    /* Receive a response.
964     * Because only one receiver, so no synchronization on in.
965     */
966    protected void readResponse() {
967      if (shouldCloseConnection.get()){
968        return;
969      }
970      Call call = null;
971      boolean expectedCall = false;
972      try {
973        // See HBaseServer.Call.setResponse for where we write out the response.
974        // Total size of the response.  Unused.  But have to read it in anyways.
975        int totalSize = in.readInt();
976
977        // Read the header
978        ResponseHeader responseHeader = ResponseHeader.parseDelimitedFrom(in);
979        int id = responseHeader.getCallId();
980        call = calls.remove(id); // call.done have to be set before leaving this method
981        expectedCall = (call != null && !call.done);
982        if (!expectedCall) {
983          // So we got a response for which we have no corresponding 'call' here on the client-side.
984          // We probably timed out waiting, cleaned up all references, and now the server decides
985          // to return a response.  There is nothing we can do w/ the response at this stage. Clean
986          // out the wire of the response so its out of the way and we can get other responses on
987          // this connection.
988          int readSoFar = IPCUtil.getTotalSizeWhenWrittenDelimited(responseHeader);
989          int whatIsLeftToRead = totalSize - readSoFar;
990          IOUtils.skipFully(in, whatIsLeftToRead);
991          if (call != null) {
992            call.callStats.setResponseSizeBytes(totalSize);
993            call.callStats.setCallTimeMs(
994                EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
995          }
996          return;
997        }
998        if (responseHeader.hasException()) {
999          ExceptionResponse exceptionResponse = responseHeader.getException();
1000          RemoteException re = IPCUtil.createRemoteException(exceptionResponse);
1001          call.setException(re);
1002          call.callStats.setResponseSizeBytes(totalSize);
1003          call.callStats.setCallTimeMs(
1004              EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
1005          if (IPCUtil.isFatalConnectionException(exceptionResponse)) {
1006            markClosed(re);
1007          }
1008        } else {
1009          Message value = null;
1010          if (call.responseDefaultType != null) {
1011            Builder builder = call.responseDefaultType.newBuilderForType();
1012            ProtobufUtil.mergeDelimitedFrom(builder, in);
1013            value = builder.build();
1014          }
1015          CellScanner cellBlockScanner = null;
1016          if (responseHeader.hasCellBlockMeta()) {
1017            int size = responseHeader.getCellBlockMeta().getLength();
1018            byte [] cellBlock = new byte[size];
1019            IOUtils.readFully(this.in, cellBlock, 0, cellBlock.length);
1020            cellBlockScanner = cellBlockBuilder.createCellScanner(this.codec, this.compressor, cellBlock);
1021          }
1022          call.setResponse(value, cellBlockScanner);
1023          call.callStats.setResponseSizeBytes(totalSize);
1024          call.callStats.setCallTimeMs(
1025              EnvironmentEdgeManager.currentTime() - call.callStats.getStartTime());
1026        }
1027      } catch (IOException e) {
1028        if (expectedCall){
1029          call.setException(e);
1030        }
1031        if (e instanceof SocketTimeoutException) {
1032          // Clean up open calls but don't treat this as a fatal condition,
1033          // since we expect certain responses to not make it by the specified
1034          // {@link ConnectionId#rpcTimeout}.
1035          if (LOG.isTraceEnabled()){
1036            LOG.trace("ignored", e);
1037          }
1038        } else {
1039          // Treat this as a fatal condition and close this connection
1040          markClosed(e);
1041        }
1042      } finally {
1043        cleanupCalls(false);
1044      }
1045    }
1046
1047    protected synchronized boolean markClosed(IOException e) {
1048      if (e == null){
1049        throw new NullPointerException();
1050      }
1051
1052      boolean ret = shouldCloseConnection.compareAndSet(false, true);
1053      if (ret) {
1054        if (LOG.isTraceEnabled()) {
1055          LOG.trace(getName() + ": marking at should close, reason: " + e.getMessage());
1056        }
1057        if (callSender != null) {
1058          callSender.close();
1059        }
1060        notifyAll();
1061      }
1062      return ret;
1063    }

1066    /**
1067     * Cleanup the calls older than a given timeout, in milli seconds.
1068     * @param allCalls true for all calls, false for only the calls in timeout
1069     */
1070    protected synchronized void cleanupCalls(boolean allCalls) {
1071      Iterator<Entry<Integer, Call>> itor = calls.entrySet().iterator();
1072      while (itor.hasNext()) {
1073        Call c = itor.next().getValue();
1074        if (c.done) {
1075          // To catch the calls without timeout that were cancelled.
1076          itor.remove();
1077        } else if (allCalls) {
1078          long waitTime = EnvironmentEdgeManager.currentTime() - c.getStartTime();
1079          IOException ie = new ConnectionClosingException("Connection to " + getRemoteAddress()
1080              + " is closing. Call id=" + c.id + ", waitTime=" + waitTime);
1081          c.setException(ie);
1082          itor.remove();
1083        } else if (c.checkAndSetTimeout()) {
1084          itor.remove();
1085        } else {
1086          // We expect the call to be ordered by timeout. It may not be the case, but stopping
1087          //  at the first valid call allows to be sure that we still have something to do without
1088          //  spending too much time by reading the full list.
1089          break;
1090        }
1091      }
1092    }
1093  }

1148  /** Stop all threads related to this client.  No further calls may be made
1149   * using this client. */
1150  @Override
1151  public void close() {
1152    if (LOG.isDebugEnabled()){
1153      LOG.debug("Stopping rpc client");
1154    }
1155    if (!running.compareAndSet(true, false)){
1156      return;
1157    }
1158
1159    Set<Connection> connsToClose = null;
1160    // wake up all connections
1161    synchronized (connections) {
1162      for (Connection conn : connections.values()) {
1163        conn.interrupt();
1164        if (conn.callSender != null) {
1165          conn.callSender.interrupt();
1166        }
1167
1168        // In case the CallSender did not setupIOStreams() yet, the Connection may not be started
1169        // at all (if CallSender has a cancelled Call it can happen). See HBASE-13851
1170        if (!conn.isAlive()) {
1171          if (connsToClose == null) {
1172            connsToClose = new HashSet<>();
1173          }
1174          connsToClose.add(conn);
1175        }
1176      }
1177    }
1178    if (connsToClose != null) {
1179      for (Connection conn : connsToClose) {
1180        conn.markClosed(new InterruptedIOException("RpcClient is closing"));
1181        conn.close();
1182      }
1183    }
1184    // wait until all connections are closed
1185    while (!connections.isEmpty()) {
1186      try {
1187        Thread.sleep(10);
1188      } catch (InterruptedException e) {
1189        LOG.info("Interrupted while stopping the client. We still have " + connections.size() +
1190            " connections.");
1191        Thread.currentThread().interrupt();
1192        return;
1193      }
1194    }
1195  }

1225  /**
1226   * Make a call, passing <code>param</code>, to the IPC server running at
1227   * <code>address</code> which is servicing the <code>protocol</code> protocol,
1228   * with the <code>ticket</code> credentials, returning the value.
1229   * Throws exceptions if there are network problems or if the remote code
1230   * threw an exception.
1231   * @param ticket Be careful which ticket you pass. A new user will mean a new Connection.
1232   *          {@link UserProvider#getCurrent()} makes a new instance of User each time so will be a
1233   *          new Connection each time.
1234   * @return A Call
1235   * @throws InterruptedException if the call is interrupted
1236   * @throws IOException if something fails on the connection
1237   */
1238  private <R extends Message> Call call(MethodDescriptor method, Message request,
1239      R responsePrototype, HBaseRpcController pcrc, User ticket,
1240      InetSocketAddress addr, MetricsConnection.CallStats callStats)
1241      throws IOException, InterruptedException {
1242
1243    CellScanner cells = pcrc.cellScanner();
1244
1245    final Call call = new Call(callIdCnt.getAndIncrement(), method, request, cells,
1246        responsePrototype, pcrc.getCallTimeout(), callStats);
1247
1248    final Connection connection = getConnection(ticket, call, addr);
1249
1250    final CallFuture cts;
1251    if (connection.callSender != null) {
1252      cts = connection.callSender.sendCall(call, pcrc.getPriority(), Trace.currentSpan());
1253      pcrc.notifyOnCancel(new RpcCallback<Object>() {
1254        @Override
1255        public void run(Object parameter) {
1256          connection.callSender.remove(cts);
1257        }
1258      });
1259      if (pcrc.isCanceled()) {
1260        // To finish if the call was cancelled before we set the notification (race condition)
1261        call.callComplete();
1262        return call;
1263      }
1264    } else {
1265      cts = null;
1266      connection.tracedWriteRequest(call, pcrc.getPriority(), Trace.currentSpan());
1267    }
1268
1269    while (!call.done) {
1270      if (call.checkAndSetTimeout()) {
1271        if (cts != null){
1272          connection.callSender.remove(cts);
1273        }
1274        break;
1275      }
1276      if (connection.shouldCloseConnection.get()) {
1277        throw new ConnectionClosingException("Call id=" + call.id +
1278            " on server " + addr + " aborted: connection is closing");
1279      }
1280      try {
1281        synchronized (call) {
1282          if (call.done){
1283            break;
1284          }
1285          call.wait(Math.min(call.remainingTime(), 1000) + 1);
1286        }
1287      } catch (InterruptedException e) {
1288        call.setException(new InterruptedIOException());
1289        if (cts != null) {
1290          connection.callSender.remove(cts);
1291        }
1292        throw e;
1293      }
1294    }
1295
1296    if (call.error != null) {
1297      if (call.error instanceof RemoteException) {
1298        call.error.fillInStackTrace();
1299        throw call.error;
1300      }
1301      // local exception
1302      throw IPCUtil.wrapException(addr, call.error);
1303    }
1304
1305    return call;
1306  }

1314   * Interrupt the connections to the given ip:port server. This should be called if the server
1315   *  is known as actually dead. This will not prevent current operation to be retried, and,
1316   *  depending on their own behavior, they may retry on the same server. This can be a feature,
1317   *  for example at startup. In any case, they're likely to get connection refused (if the
1318   *  process died) or no route to host: i.e. their next retries should be faster and with a
1319   *  safe exception.
1320   */
1321  @Override
1322  public void cancelConnections(ServerName sn) {
1323    synchronized (connections) {
1324      for (Connection connection : connections.values()) {
1325        if (connection.isAlive() &&
1326            connection.getRemoteAddress().getPort() == sn.getPort() &&
1327            connection.getRemoteAddress().getHostName().equals(sn.getHostname())) {
1328          LOG.info("The server on " + sn.toString() +
1329              " is dead - stopping the connection " + connection.remoteId);
1330          connection.interrupt(); // We're interrupting a Reader. It means we want it to finish.
1331                                  // This will close the connection as well.
1332        }
1333      }
1334    }
1335  }
1336
1337  /**
1338   * Get a connection from the pool, or create a new one and add it to the
1339   * pool. Connections to a given host/port are reused.
1340   */
1341  protected Connection getConnection(User ticket, Call call, InetSocketAddress addr)
1342  throws IOException {
1343    if (!running.get()){
1344      throw new StoppedRpcClientException();
1345    }
1346    Connection connection;
1347    ConnectionId remoteId =
1348      new ConnectionId(ticket, call.md.getService().getName(), addr);
1349    synchronized (connections) {
1350      connection = connections.get(remoteId);
1351      if (connection == null) {
1352        connection = createConnection(remoteId, this.codec, this.compressor);
1353        connections.put(remoteId, connection);
1354      }
1355    }
1356
1357    return connection;
1358  }
