059/**
060 * Thread Pool that executes the submitted procedures.
061 * The executor has a ProcedureStore associated.
062 * Each operation is logged and on restart the pending procedures are resumed.
063 *
064 * Unless the Procedure code throws an error (e.g. invalid user input)
065 * the procedure will complete (at some point in time), On restart the pending
066 * procedures are resumed and the once failed will be rolledback.
067 *
068 * The user can add procedures to the executor via submitProcedure(proc)
069 * check for the finished state via isFinished(procId)
070 * and get the result via getResult(procId)
071 */
072@InterfaceAudience.Private
073@InterfaceStability.Evolving
074public class ProcedureExecutor<TEnvironment> {
075  private static final Log LOG = LogFactory.getLog(ProcedureExecutor.class);

113  /**
114   * Internal cleaner that removes the completed procedure results after a TTL.
115   * NOTE: This is a special case handled in timeoutLoop().
116   *
117   * Since the client code looks more or less like:
118   *   procId = master.doOperation()
119   *   while (master.getProcResult(procId) == ProcInProgress);
120   * The master should not throw away the proc result as soon as the procedure is done
121   * but should wait a result request from the client (see executor.removeResult(procId))
122   * The client will call something like master.isProcDone() or master.getProcResult()
123   * which will return the result/state to the client, and it will mark the completed
124   * proc as ready to delete. note that the client may not receive the response from
125   * the master (e.g. master failover) so, if we delay a bit the real deletion of
126   * the proc result the client will be able to get the result the next try.
127   */
128  private static class CompletedProcedureCleaner<TEnvironment>
129      extends ProcedureInMemoryChore<TEnvironment> {
130    private static final Log LOG = LogFactory.getLog(CompletedProcedureCleaner.class);
131
132    private static final String CLEANER_INTERVAL_CONF_KEY = "hbase.procedure.cleaner.interval";
133    private static final int DEFAULT_CLEANER_INTERVAL = 30 * 1000; // 30sec
134
135    private static final String EVICT_TTL_CONF_KEY = "hbase.procedure.cleaner.evict.ttl";
136    private static final int DEFAULT_EVICT_TTL = 15 * 60000; // 15min
137
138    private static final String EVICT_ACKED_TTL_CONF_KEY ="hbase.procedure.cleaner.acked.evict.ttl";
139    private static final int DEFAULT_ACKED_EVICT_TTL = 5 * 60000; // 5min
140
141    private final Map<Long, ProcedureInfo> completed;
142    private final Map<NonceKey, Long> nonceKeysToProcIdsMap;
143    private final ProcedureStore store;
144    private final Configuration conf;
145
146    public CompletedProcedureCleaner(final Configuration conf, final ProcedureStore store,
147        final Map<Long, ProcedureInfo> completedMap,
148        final Map<NonceKey, Long> nonceKeysToProcIdsMap) {
149      // set the timeout interval that triggers the periodic-procedure
150      super(conf.getInt(CLEANER_INTERVAL_CONF_KEY, DEFAULT_CLEANER_INTERVAL));
151      this.completed = completedMap;
152      this.nonceKeysToProcIdsMap = nonceKeysToProcIdsMap;
153      this.store = store;
154      this.conf = conf;
155    }
156
157    @Override
158    protected void periodicExecute(final TEnvironment env) {
159      if (completed.isEmpty()) {
160        if (LOG.isTraceEnabled()) {
161          LOG.trace("No completed procedures to cleanup.");
162        }
163        return;
164      }
165
166      final long evictTtl = conf.getInt(EVICT_TTL_CONF_KEY, DEFAULT_EVICT_TTL);
167      final long evictAckTtl = conf.getInt(EVICT_ACKED_TTL_CONF_KEY, DEFAULT_ACKED_EVICT_TTL);
168
169      final long now = EnvironmentEdgeManager.currentTime();
170      final Iterator<Map.Entry<Long, ProcedureInfo>> it = completed.entrySet().iterator();
171      final boolean isDebugEnabled = LOG.isDebugEnabled();
172      while (it.hasNext() && store.isRunning()) {
173        final Map.Entry<Long, ProcedureInfo> entry = it.next();
174        final ProcedureInfo procInfo = entry.getValue();
175
176        // TODO: Select TTL based on Procedure type
177        if ((procInfo.hasClientAckTime() && (now - procInfo.getClientAckTime()) >= evictAckTtl) ||
178            (now - procInfo.getLastUpdate()) >= evictTtl) {
179          if (isDebugEnabled) {
180            LOG.debug("Evict completed procedure: " + procInfo);
181          }
182          store.delete(entry.getKey());
183          it.remove();
184
185          NonceKey nonceKey = procInfo.getNonceKey();
186          if (nonceKey != null) {
187            nonceKeysToProcIdsMap.remove(nonceKey);
188          }
189        }
190      }
191    }
192  }

268  private void load(final boolean abortOnCorruption) throws IOException {
269    Preconditions.checkArgument(completed.isEmpty());
270    Preconditions.checkArgument(rollbackStack.isEmpty());
271    Preconditions.checkArgument(procedures.isEmpty());
272    Preconditions.checkArgument(waitingTimeout.isEmpty());
273    Preconditions.checkArgument(runnables.size() == 0);
274
275    store.load(new ProcedureStore.ProcedureLoader() {
276      @Override
277      public void setMaxProcId(long maxProcId) {
278        assert lastProcId.get() < 0 : "expected only one call to setMaxProcId()";
279        LOG.debug("load procedures maxProcId=" + maxProcId);
280        lastProcId.set(maxProcId);
281      }
282
283      @Override
284      public void load(ProcedureIterator procIter) throws IOException {
285        loadProcedures(procIter, abortOnCorruption);
286      }
287
288      @Override
289      public void handleCorrupted(ProcedureIterator procIter) throws IOException {
290        int corruptedCount = 0;
291        while (procIter.hasNext()) {
292          ProcedureInfo proc = procIter.nextAsProcedureInfo();
293          LOG.error("corrupted procedure: " + proc);
294          corruptedCount++;
295        }
296        if (abortOnCorruption && corruptedCount > 0) {
297          throw new IOException("found " + corruptedCount + " corrupted procedure(s) on replay");
298        }
299      }
300    });
301  }

303  private void loadProcedures(final ProcedureIterator procIter,
304      final boolean abortOnCorruption) throws IOException {
305    final boolean isDebugEnabled = LOG.isDebugEnabled();
306
307    // 1. Build the rollback stack
308    int runnablesCount = 0;
309    while (procIter.hasNext()) {
310      final NonceKey nonceKey;
311      final long procId;
312
313      if (procIter.isNextCompleted()) {
314        ProcedureInfo proc = procIter.nextAsProcedureInfo();
315        nonceKey = proc.getNonceKey();
316        procId = proc.getProcId();
317        completed.put(proc.getProcId(), proc);
318        if (isDebugEnabled) {
319          LOG.debug("The procedure is completed: " + proc);
320        }
321      } else {
322        Procedure proc = procIter.nextAsProcedure();
323        nonceKey = proc.getNonceKey();
324        procId = proc.getProcId();
325
326        if (!proc.hasParent()) {
327          assert !proc.isFinished() : "unexpected finished procedure";
328          rollbackStack.put(proc.getProcId(), new RootProcedureState());
329        }
330
331        // add the procedure to the map
332        proc.beforeReplay(getEnvironment());
333        procedures.put(proc.getProcId(), proc);
334
335        if (proc.getState() == ProcedureState.RUNNABLE) {
336          runnablesCount++;
337        }
338      }
339
340      // add the nonce to the map
341      if (nonceKey != null) {
342        nonceKeysToProcIdsMap.put(nonceKey, procId);
343      }
344    }
345
346    // 2. Initialize the stacks
347    ArrayList<Procedure> runnableList = new ArrayList(runnablesCount);
348    HashSet<Procedure> waitingSet = null;
349    procIter.reset();
350    while (procIter.hasNext()) {
351      if (procIter.isNextCompleted()) {
352        procIter.skipNext();
353        continue;
354      }
355
356      Procedure proc = procIter.nextAsProcedure();
357      assert !(proc.isFinished() && !proc.hasParent()) : "unexpected completed proc=" + proc;
358
359      if (isDebugEnabled) {
360        LOG.debug(String.format("Loading procedure state=%s isFailed=%s: %s",
361                    proc.getState(), proc.hasException(), proc));
362      }
363
364      Long rootProcId = getRootProcedureId(proc);
365      if (rootProcId == null) {
366        // The 'proc' was ready to run but the root procedure was rolledback?
367        runnables.addBack(proc);
368        continue;
369      }
370
371      if (proc.hasParent()) {
372        Procedure parent = procedures.get(proc.getParentProcId());
373        // corrupted procedures are handled later at step 3
374        if (parent != null && !proc.isFinished()) {
375          parent.incChildrenLatch();
376        }
377      }
378
379      RootProcedureState procStack = rollbackStack.get(rootProcId);
380      procStack.loadStack(proc);
381
382      switch (proc.getState()) {
383        case RUNNABLE:
384          runnableList.add(proc);
385          break;
386        case WAITING:
387          if (!proc.hasChildren()) {
388            runnableList.add(proc);
389          }
390          break;
391        case WAITING_TIMEOUT:
392          if (waitingSet == null) {
393            waitingSet = new HashSet<Procedure>();
394          }
395          waitingSet.add(proc);
396          break;
397        case FINISHED:
398          if (proc.hasException()) {
399            // add the proc to the runnables to perform the rollback
400            runnables.addBack(proc);
401          }
402          break;
403        case ROLLEDBACK:
404        case INITIALIZING:
405          String msg = "Unexpected " + proc.getState() + " state for " + proc;
406          LOG.error(msg);
407          throw new UnsupportedOperationException(msg);
408        default:
409          break;
410      }
411    }
412
413    // 3. Validate the stacks
414    int corruptedCount = 0;
415    Iterator<Map.Entry<Long, RootProcedureState>> itStack = rollbackStack.entrySet().iterator();
416    while (itStack.hasNext()) {
417      Map.Entry<Long, RootProcedureState> entry = itStack.next();
418      RootProcedureState procStack = entry.getValue();
419      if (procStack.isValid()) continue;
420
421      for (Procedure proc: procStack.getSubproceduresStack()) {
422        LOG.error("corrupted procedure: " + proc);
423        procedures.remove(proc.getProcId());
424        runnableList.remove(proc);
425        if (waitingSet != null) waitingSet.remove(proc);
426        corruptedCount++;
427      }
428      itStack.remove();
429    }
430
431    if (abortOnCorruption && corruptedCount > 0) {
432      throw new IOException("found " + corruptedCount + " procedures on replay");
433    }
434
435    // 4. Push the runnables
436    if (!runnableList.isEmpty()) {
437      // TODO: See ProcedureWALFormatReader#hasFastStartSupport
438      // some procedure may be started way before this stuff.
439      for (int i = runnableList.size() - 1; i >= 0; --i) {
440        Procedure proc = runnableList.get(i);
441        if (!proc.hasParent()) {
442          sendProcedureLoadedNotification(proc.getProcId());
443        }
444        if (proc.wasExecuted()) {
445          runnables.addFront(proc);
446        } else {
447          // if it was not in execution, it can wait.
448          runnables.addBack(proc);
449        }
450      }
451    }
452  }

454  /**
455   * Start the procedure executor.
456   * It calls ProcedureStore.recoverLease() and ProcedureStore.load() to
457   * recover the lease, and ensure a single executor, and start the procedure
458   * replay to resume and recover the previous pending and in-progress perocedures.
459   *
460   * @param numThreads number of threads available for procedure execution.
461   * @param abortOnCorruption true if you want to abort your service in case
462   *          a corrupted procedure is found on replay. otherwise false.
463   */
464  public void start(int numThreads, boolean abortOnCorruption) throws IOException {
465    if (running.getAndSet(true)) {
466      LOG.warn("Already running");
467      return;
468    }
469
470    // We have numThreads executor + one timer thread used for timing out
471    // procedures and triggering periodic procedures.
472    threads = new Thread[numThreads + 1];
473    LOG.info("Starting procedure executor threads=" + threads.length);
474
475    // Initialize procedures executor
476    for (int i = 0; i < numThreads; ++i) {
477      threads[i] = new Thread("ProcedureExecutor-" + i) {
478        @Override
479        public void run() {
480          execLoop();
481        }
482      };
483    }
484
485    // Initialize procedures timeout handler (this is the +1 thread)
486    threads[numThreads] = new Thread("ProcedureExecutorTimeoutThread") {
487      @Override
488      public void run() {
489        timeoutLoop();
490      }
491    };
492
493    long st, et;
494
495    // Acquire the store lease.
496    st = EnvironmentEdgeManager.currentTime();
497    store.recoverLease();
498    et = EnvironmentEdgeManager.currentTime();
499    LOG.info(String.format("recover procedure store (%s) lease: %s",
500      store.getClass().getSimpleName(), StringUtils.humanTimeDiff(et - st)));
501
502    // TODO: Split in two steps.
503    // TODO: Handle corrupted procedures (currently just a warn)
504    // The first one will make sure that we have the latest id,
505    // so we can start the threads and accept new procedures.
506    // The second step will do the actual load of old procedures.
507    st = EnvironmentEdgeManager.currentTime();
508    load(abortOnCorruption);
509    et = EnvironmentEdgeManager.currentTime();
510    LOG.info(String.format("load procedure store (%s): %s",
511      store.getClass().getSimpleName(), StringUtils.humanTimeDiff(et - st)));
512
513    // Start the executors. Here we must have the lastProcId set.
514    for (int i = 0; i < threads.length; ++i) {
515      threads[i].start();
516    }
517
518    // Add completed cleaner chore
519    addChore(new CompletedProcedureCleaner(conf, store, completed, nonceKeysToProcIdsMap));
520  }

522  public void stop() {
523    if (!running.getAndSet(false)) {
524      return;
525    }
526
527    LOG.info("Stopping the procedure executor");
528    runnables.signalAll();
529    waitingTimeout.signalAll();
530  }

532  public void join() {
533    boolean interrupted = false;
534
535    for (int i = 0; i < threads.length; ++i) {
536      try {
537        threads[i].join();
538      } catch (InterruptedException ex) {
539        interrupted = true;
540      }
541    }
542
543    if (interrupted) {
544      Thread.currentThread().interrupt();
545    }
546
547    completed.clear();
548    rollbackStack.clear();
549    procedures.clear();
550    nonceKeysToProcIdsMap.clear();
551    waitingTimeout.clear();
552    runnables.clear();
553    lastProcId.set(-1);
554  }

587  /**
588   * List procedures.
589   * @return the procedures in a list
590   */
591  public List<ProcedureInfo> listProcedures() {
592    List<ProcedureInfo> procedureLists =
593        new ArrayList<ProcedureInfo>(procedures.size() + completed.size());
594    for (java.util.Map.Entry<Long, Procedure> p: procedures.entrySet()) {
595      procedureLists.add(Procedure.createProcedureInfo(p.getValue(), null));
596    }
597    for (java.util.Map.Entry<Long, ProcedureInfo> e: completed.entrySet()) {
598      // Note: The procedure could show up twice in the list with different state, as
599      // it could complete after we walk through procedures list and insert into
600      // procedureList - it is ok, as we will use the information in the ProcedureInfo
601      // to figure it out; to prevent this would increase the complexity of the logic.
602      procedureLists.add(e.getValue());
603    }
604    return procedureLists;
605  }

632  /**
633   * Add a new root-procedure to the executor.
634   * @param proc the new procedure to execute.
635   * @param nonceGroup
636   * @param nonce
637   * @return the procedure id, that can be used to monitor the operation
638   */
639  public long submitProcedure(
640      final Procedure proc,
641      final long nonceGroup,
642      final long nonce) {
643    Preconditions.checkArgument(proc.getState() == ProcedureState.INITIALIZING);
644    Preconditions.checkArgument(isRunning());
645    Preconditions.checkArgument(lastProcId.get() >= 0);
646    Preconditions.checkArgument(!proc.hasParent());
647    if (this.checkOwnerSet) {
648      Preconditions.checkArgument(proc.hasOwner());
649    }
650
651    // Initialize the Procedure ID
652    long currentProcId = nextProcId();
653    proc.setProcId(currentProcId);
654
655    // Check whether the proc exists.  If exist, just return the proc id.
656    // This is to prevent the same proc to submit multiple times (it could happen
657    // when client could not talk to server and resubmit the same request).
658    if (nonce != HConstants.NO_NONCE) {
659      NonceKey noncekey = new NonceKey(nonceGroup, nonce);
660      proc.setNonceKey(noncekey);
661
662      Long oldProcId = nonceKeysToProcIdsMap.putIfAbsent(noncekey, currentProcId);
663      if (oldProcId != null) {
664        // Found the proc
665        return oldProcId.longValue();
666      }
667    }
668
669    // Commit the transaction
670    store.insert(proc, null);
671    if (LOG.isDebugEnabled()) {
672      LOG.debug("Procedure " + proc + " added to the store.");
673    }
674
675    // Create the rollback stack for the procedure
676    RootProcedureState stack = new RootProcedureState();
677    rollbackStack.put(currentProcId, stack);
678
679    // Submit the new subprocedures
680    assert !procedures.containsKey(currentProcId);
681    procedures.put(currentProcId, proc);
682    sendProcedureAddedNotification(currentProcId);
683    runnables.addBack(proc);

799  /**
800   * Execution loop (N threads)
801   * while the executor is in a running state,
802   * fetch a procedure from the runnables queue and start the execution.
803   */
804  private void execLoop() {
805    while (isRunning()) {
806      Procedure proc = runnables.poll();
807      if (proc == null) continue;
808
809      try {
810        activeExecutorCount.incrementAndGet();
811        execLoop(proc);
812      } finally {
813        activeExecutorCount.decrementAndGet();
814      }
815    }
816  }
817
818  private void execLoop(Procedure proc) {
819    if (LOG.isTraceEnabled()) {
820      LOG.trace("Trying to start the execution of " + proc);
821    }
822
823    Long rootProcId = getRootProcedureId(proc);
824    if (rootProcId == null) {
825      // The 'proc' was ready to run but the root procedure was rolledback
826      executeRollback(proc);
827      return;
828    }
829
830    RootProcedureState procStack = rollbackStack.get(rootProcId);
831    if (procStack == null) return;
832
833    do {
834      // Try to acquire the execution
835      if (!procStack.acquire(proc)) {
836        if (procStack.setRollback()) {
837          // we have the 'rollback-lock' we can start rollingback
838          if (!executeRollback(rootProcId, procStack)) {
839            procStack.unsetRollback();
840            runnables.yield(proc);
841          }
842        } else {
843          // if we can't rollback means that some child is still running.
844          // the rollback will be executed after all the children are done.
845          // If the procedure was never executed, remove and mark it as rolledback.
846          if (!proc.wasExecuted()) {
847            if (!executeRollback(proc)) {
848              runnables.yield(proc);
849            }
850          }
851        }
852        break;
853      }
854
855      // Execute the procedure
856      assert proc.getState() == ProcedureState.RUNNABLE : proc;
857      if (proc.acquireLock(getEnvironment())) {
858        execProcedure(procStack, proc);
859        proc.releaseLock(getEnvironment());
860      } else {
861        runnables.yield(proc);
862      }
863      procStack.release(proc);
864
865      // allows to kill the executor before something is stored to the wal.
866      // useful to test the procedure recovery.
867      if (testing != null && !isRunning()) {
868        break;
869      }
870
871      if (proc.isSuccess()) {
872        if (LOG.isDebugEnabled()) {
873          LOG.debug("Procedure completed in " +
874              StringUtils.humanTimeDiff(proc.elapsedTime()) + ": " + proc);
875        }
876        // Finalize the procedure state
877        if (proc.getProcId() == rootProcId) {
878          procedureFinished(proc);
879        }
880        break;
881      }
882    } while (procStack.isFailed());
883  }

885  private void timeoutLoop() {
886    while (isRunning()) {
887      Procedure proc = waitingTimeout.poll();
888      if (proc == null) continue;
889
890      if (proc.getTimeRemaining() > 100) {
891        // got an early wake, maybe a stop?
892        // re-enqueue the task in case was not a stop or just a signal
893        waitingTimeout.add(proc);
894        continue;
895      }
896
897      // ----------------------------------------------------------------------------
898      // TODO-MAYBE: Should we provide a notification to the store with the
899      // full set of procedures pending and completed to write a compacted
900      // version of the log (in case is a log)?
901      // In theory no, procedures are have a short life, so at some point the store
902      // will have the tracker saying everything is in the last log.
903      // ----------------------------------------------------------------------------
904
905      // The ProcedureInMemoryChore is a special case, and it acts as a chore.
906      // instead of bringing the Chore class in, we reuse this timeout thread for
907      // this special case.
908      if (proc instanceof ProcedureInMemoryChore) {
909        try {
910          ((ProcedureInMemoryChore)proc).periodicExecute(getEnvironment());
911        } catch (Throwable e) {
912          LOG.error("Ignoring CompletedProcedureCleaner exception: " + e.getMessage(), e);
913        }
914        proc.setStartTime(EnvironmentEdgeManager.currentTime());
915        waitingTimeout.add(proc);
916        continue;
917      }
918
919      // The procedure received an "abort-timeout", call abort() and
920      // add the procedure back in the queue for rollback.
921      if (proc.setTimeoutFailure()) {
922        long rootProcId = Procedure.getRootProcedureId(procedures, proc);
923        RootProcedureState procStack = rollbackStack.get(rootProcId);
924        procStack.abort();
925        store.update(proc);
926        runnables.addFront(proc);
927        continue;
928      }
929    }
930  }

932  /**
933   * Execute the rollback of the full procedure stack.
934   * Once the procedure is rolledback, the root-procedure will be visible as
935   * finished to user, and the result will be the fatal exception.
936   */
937  private boolean executeRollback(final long rootProcId, final RootProcedureState procStack) {
938    Procedure rootProc = procedures.get(rootProcId);
939    RemoteProcedureException exception = rootProc.getException();
940    if (exception == null) {
941      exception = procStack.getException();
942      rootProc.setFailure(exception);
943      store.update(rootProc);
944    }
945
946    List<Procedure> subprocStack = procStack.getSubproceduresStack();
947    assert subprocStack != null : "Called rollback with no steps executed rootProc=" + rootProc;
948
949    int stackTail = subprocStack.size();
950    boolean reuseLock = false;
951    while (stackTail --> 0) {
952      final Procedure proc = subprocStack.get(stackTail);
953
954      if (!reuseLock && !proc.acquireLock(getEnvironment())) {
955        // can't take a lock on the procedure, add the root-proc back on the
956        // queue waiting for the lock availability
957        return false;
958      }
959
960      boolean abortRollback = !executeRollback(proc);
961      abortRollback |= !isRunning() || !store.isRunning();
962
963      // If the next procedure is the same to this one
964      // (e.g. StateMachineProcedure reuse the same instance)
965      // we can avoid to lock/unlock each step
966      reuseLock = stackTail > 0 && (subprocStack.get(stackTail - 1) == proc) && !abortRollback;
967      if (!reuseLock) {
968        proc.releaseLock(getEnvironment());
969      }
970
971      // allows to kill the executor before something is stored to the wal.
972      // useful to test the procedure recovery.
973      if (abortRollback) {
974        return false;
975      }
976
977      subprocStack.remove(stackTail);
978
979      // if the procedure is kind enough to pass the slot to someone else, yield
980      if (proc.isYieldAfterExecutionStep(getEnvironment())) {
981        return false;
982      }
983    }
984
985    // Finalize the procedure state
986    LOG.info("Rolledback procedure " + rootProc +
987             " exec-time=" + StringUtils.humanTimeDiff(rootProc.elapsedTime()) +
988             " exception=" + exception.getMessage());
989    procedureFinished(rootProc);
990    return true;
991  }

993  /**
994   * Execute the rollback of the procedure step.
995   * It updates the store with the new state (stack index)
996   * or will remove completly the procedure in case it is a child.
997   */
998  private boolean executeRollback(final Procedure proc) {
999    try {
1000      proc.doRollback(getEnvironment());
1001    } catch (IOException e) {
1002      if (LOG.isDebugEnabled()) {
1003        LOG.debug("rollback attempt failed for " + proc, e);
1004      }
1005      return false;
1006    } catch (InterruptedException e) {
1007      handleInterruptedException(proc, e);
1008      return false;
1009    } catch (Throwable e) {
1010      // Catch NullPointerExceptions or similar errors...
1011      LOG.fatal("CODE-BUG: Uncatched runtime exception for procedure: " + proc, e);
1012    }
1013
1014    // allows to kill the executor before something is stored to the wal.
1015    // useful to test the procedure recovery.
1016    if (testing != null && testing.shouldKillBeforeStoreUpdate()) {
1017      LOG.debug("TESTING: Kill before store update");
1018      stop();
1019      return false;
1020    }
1021
1022    if (proc.removeStackIndex()) {
1023      proc.setState(ProcedureState.ROLLEDBACK);
1024      if (proc.hasParent()) {
1025        store.delete(proc.getProcId());
1026        procedures.remove(proc.getProcId());
1027      } else {
1028        final long[] childProcIds = rollbackStack.get(proc.getProcId()).getSubprocedureIds();
1029        if (childProcIds != null) {
1030          store.delete(proc, childProcIds);
1031        } else {
1032          store.update(proc);
1033        }
1034      }
1035    } else {
1036      store.update(proc);
1037    }
1038
1039    return true;
1040  }

1042  /**
1043   * Executes the specified procedure
1044   *  - calls the doExecute() of the procedure
1045   *  - if the procedure execution didn't fail (e.g. invalid user input)
1046   *     - ...and returned subprocedures
1047   *        - the subprocedures are initialized.
1048   *        - the subprocedures are added to the store
1049   *        - the subprocedures are added to the runnable queue
1050   *        - the procedure is now in a WAITING state, waiting for the subprocedures to complete
1051   *     - ...if there are no subprocedure
1052   *        - the procedure completed successfully
1053   *        - if there is a parent (WAITING)
1054   *            - the parent state will be set to RUNNABLE
1055   *  - in case of failure
1056   *    - the store is updated with the new state
1057   *    - the executor (caller of this method) will start the rollback of the procedure
1058   */
1059  private void execProcedure(final RootProcedureState procStack, final Procedure procedure) {
1060    Preconditions.checkArgument(procedure.getState() == ProcedureState.RUNNABLE);
1061
1062    // Execute the procedure
1063    boolean isSuspended = false;
1064    boolean reExecute = false;
1065    Procedure[] subprocs = null;
1066    do {
1067      reExecute = false;
1068      try {
1069        subprocs = procedure.doExecute(getEnvironment());
1070        if (subprocs != null && subprocs.length == 0) {
1071          subprocs = null;
1072        }
1073      } catch (ProcedureSuspendedException e) {
1074        isSuspended = true;
1075      } catch (ProcedureYieldException e) {
1076        if (LOG.isTraceEnabled()) {
1077          LOG.trace("Yield procedure: " + procedure + ": " + e.getMessage());
1078        }
1079        runnables.yield(procedure);
1080        return;
1081      } catch (InterruptedException e) {
1082        handleInterruptedException(procedure, e);
1083        runnables.yield(procedure);
1084        return;
1085      } catch (Throwable e) {
1086        // Catch NullPointerExceptions or similar errors...
1087        String msg = "CODE-BUG: Uncatched runtime exception for procedure: " + procedure;
1088        LOG.error(msg, e);
1089        procedure.setFailure(new RemoteProcedureException(msg, e));
1090      }
1091
1092      if (!procedure.isFailed()) {
1093        if (subprocs != null) {
1094          if (subprocs.length == 1 && subprocs[0] == procedure) {
1095            // quick-shortcut for a state machine like procedure
1096            subprocs = null;
1097            reExecute = true;
1098          } else {
1099            // yield the current procedure, and make the subprocedure runnable
1100            for (int i = 0; i < subprocs.length; ++i) {
1101              Procedure subproc = subprocs[i];
1102              if (subproc == null) {
1103                String msg = "subproc[" + i + "] is null, aborting the procedure";
1104                procedure.setFailure(new RemoteProcedureException(msg,
1105                  new IllegalArgumentIOException(msg)));
1106                subprocs = null;
1107                break;
1108              }
1109
1110              assert subproc.getState() == ProcedureState.INITIALIZING : subproc;
1111              subproc.setParentProcId(procedure.getProcId());
1112              subproc.setProcId(nextProcId());
1113              procStack.addSubProcedure(subproc);
1114            }
1115
1116            if (!procedure.isFailed()) {
1117              procedure.setChildrenLatch(subprocs.length);
1118              switch (procedure.getState()) {
1119                case RUNNABLE:
1120                  procedure.setState(ProcedureState.WAITING);
1121                  break;
1122                case WAITING_TIMEOUT:
1123                  waitingTimeout.add(procedure);
1124                  break;
1125                default:
1126                  break;
1127              }
1128            }
1129          }
1130        } else if (procedure.getState() == ProcedureState.WAITING_TIMEOUT) {
1131          waitingTimeout.add(procedure);
1132        } else if (!isSuspended) {
1133          // No subtask, so we are done
1134          procedure.setState(ProcedureState.FINISHED);
1135        }
1136      }
1137
1138      // Add the procedure to the stack
1139      procStack.addRollbackStep(procedure);
1140
1141      // allows to kill the executor before something is stored to the wal.
1142      // useful to test the procedure recovery.
1143      if (testing != null && testing.shouldKillBeforeStoreUpdate()) {
1144        LOG.debug("TESTING: Kill before store update");
1145        stop();
1146        return;
1147      }
1148
1149      // Commit the transaction
1150      updateStoreOnExec(procStack, procedure, subprocs);
1151
1152      // if the store is not running we are aborting
1153      if (!store.isRunning()) {
1154        return;
1155      }
1156
1157      // if the procedure is kind enough to pass the slot to someone else, yield
1158      if (procedure.getState() == ProcedureState.RUNNABLE &&
1159          procedure.isYieldAfterExecutionStep(getEnvironment())) {
1160        runnables.yield(procedure);
1161        return;
1162      }
1163
1164      assert (reExecute && subprocs == null) || !reExecute;
1165    } while (reExecute);
1166
1167    // Submit the new subprocedures
1168    if (subprocs != null && !procedure.isFailed()) {
1169      for (int i = 0; i < subprocs.length; ++i) {
1170        Procedure subproc = subprocs[i];
1171        assert !procedures.containsKey(subproc.getProcId());
1172        procedures.put(subproc.getProcId(), subproc);
1173        runnables.addFront(subproc);
1174      }
1175    }
1176
1177    if (procedure.isFinished() && procedure.hasParent()) {
1178      Procedure parent = procedures.get(procedure.getParentProcId());
1179      if (parent == null) {
1180        assert procStack.isRollingback();
1181        return;
1182      }
1183
1184      // If this procedure is the last child awake the parent procedure
1185      if (LOG.isTraceEnabled()) {
1186        LOG.trace(parent + " child is done: " + procedure);
1187      }
1188      if (parent.childrenCountDown() && parent.getState() == ProcedureState.WAITING) {
1189        parent.setState(ProcedureState.RUNNABLE);
1190        store.update(parent);
1191        runnables.addFront(parent);
1192        if (LOG.isTraceEnabled()) {
1193          LOG.trace(parent + " all the children finished their work, resume.");
1194        }
1195        return;
1196      }
1197    }
1198  }

1200  private void updateStoreOnExec(final RootProcedureState procStack,
1201      final Procedure procedure, final Procedure[] subprocs) {
1202    if (subprocs != null && !procedure.isFailed()) {
1203      if (LOG.isTraceEnabled()) {
1204        LOG.trace("Store add " + procedure + " children " + Arrays.toString(subprocs));
1205      }
1206      store.insert(procedure, subprocs);
1207    } else {
1208      if (LOG.isTraceEnabled()) {
1209        LOG.trace("Store update " + procedure);
1210      }
1211      if (procedure.isFinished() && !procedure.hasParent()) {
1212        // remove child procedures
1213        final long[] childProcIds = procStack.getSubprocedureIds();
1214        if (childProcIds != null) {
1215          store.delete(procedure, childProcIds);
1216          for (int i = 0; i < childProcIds.length; ++i) {
1217            procedures.remove(childProcIds[i]);
1218          }
1219        } else {
1220          store.update(procedure);
1221        }
1222      } else {
1223        store.update(procedure);
1224      }
1225    }
1226  }

1276  private long nextProcId() {
1277    long procId = lastProcId.incrementAndGet();
1278    if (procId < 0) {
1279      while (!lastProcId.compareAndSet(procId, 0)) {
1280        procId = lastProcId.get();
1281        if (procId >= 0)
1282          break;
1283      }
1284      while (procedures.containsKey(procId)) {
1285        procId = lastProcId.incrementAndGet();
1286      }
1287    }
1288    return procId;
1289  }

1300  private void procedureFinished(final Procedure proc) {
1301    // call the procedure completion cleanup handler
1302    try {
1303      proc.completionCleanup(getEnvironment());
1304    } catch (Throwable e) {
1305      // Catch NullPointerExceptions or similar errors...
1306      LOG.error("CODE-BUG: uncatched runtime exception for procedure: " + proc, e);
1307    }
1308
1309    // update the executor internal state maps
1310    ProcedureInfo procInfo = Procedure.createProcedureInfo(proc, proc.getNonceKey());
1311    if (!proc.shouldWaitClientAck(getEnvironment())) {
1312      procInfo.setClientAckTime(0);
1313    }
1314
1315    completed.put(procInfo.getProcId(), procInfo);
1316    rollbackStack.remove(proc.getProcId());
1317    procedures.remove(proc.getProcId());
1318
1319    // call the runnableSet completion cleanup handler
1320    try {
1321      runnables.completionCleanup(proc);
1322    } catch (Throwable e) {
1323      // Catch NullPointerExceptions or similar errors...
1324      LOG.error("CODE-BUG: uncatched runtime exception for runnableSet: " + runnables, e);
1325    }
1326
1327    // Notify the listeners
1328    sendProcedureFinishedNotification(proc.getProcId());
1329  }
