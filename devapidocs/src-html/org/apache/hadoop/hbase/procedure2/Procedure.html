043/**
044 * Base Procedure class responsible for Procedure Metadata;
045 * e.g. state, submittedTime, lastUpdate, stack-indexes, etc.
046 *
047 * <p>Procedures are run by a {@link ProcedureExecutor} instance. They are submitted and then
048 * the ProcedureExecutor keeps calling {@link #execute(Object)} until the Procedure is done.
049 * Execute may be called multiple times in the case of failure or a restart, so code must be
050 * idempotent. The return from an execute call is either: null to indicate we are done;
051 * ourself if there is more to do; or, a set of sub-procedures that need to
052 * be run to completion before the framework resumes our execution.
053 *
054 * <p>The ProcedureExecutor keeps its
055 * notion of Procedure State in the Procedure itself; e.g. it stamps the Procedure as INITIALIZING,
056 * RUNNABLE, SUCCESS, etc. Here are some of the States defined in the ProcedureState enum from
057 * protos:
058 *<ul>
059 * <li>{@link #isFailed()} A procedure has executed at least once and has failed. The procedure
060 * may or may not have rolled back yet. Any procedure in FAILED state will be eventually moved
061 * to ROLLEDBACK state.</li>
062 *
063 * <li>{@link #isSuccess()} A procedure is completed successfully without exception.</li>
064 *
065 * <li>{@link #isFinished()} As a procedure in FAILED state will be tried forever for rollback, only
066 * condition when scheduler/ executor will drop procedure from further processing is when procedure
067 * state is ROLLEDBACK or isSuccess() returns true. This is a terminal state of the procedure.</li>
068 *
069 * <li>{@link #isWaiting()} - Procedure is in one of the two waiting states
070 * ({@link ProcedureState#WAITING}, {@link ProcedureState#WAITING_TIMEOUT}).</li>
071 *</ul>
072 * NOTE: These states are of the ProcedureExecutor. Procedure implementations in turn can keep
073 * their own state. This can lead to confusion. Try to keep the two distinct.
074 *
075 * <p>rollback() is called when the procedure or one of the sub-procedures
076 * has failed. The rollback step is supposed to cleanup the resources created
077 * during the execute() step. In case of failure and restart, rollback() may be
078 * called multiple times, so again the code must be idempotent.
079 *
080 * <p>Procedure can be made respect a locking regime. It has acqure/release methods as
081 * well as an {@link #hasLock(Object)}. The lock implementation is up to the implementor.
082 * If an entity needs to be locked for the life of a procedure -- not just the calls to
083 * execute -- then implementations should say so with the {@link #holdLock(Object)}
084 * method.
085 *
086 * <p>There are hooks for collecting metrics on submit of the procedure and on finish.
087 * See {@link #updateMetricsOnSubmit(Object)} and
088 * {@link #updateMetricsOnFinish(Object, long, boolean)}.
089 */
090@InterfaceAudience.Private

181  /**
182   * The user should override this method if they need a lock on an Entity.
183   * A lock can be anything, and it is up to the implementor. The Procedure
184   * Framework will call this method just before it invokes {@link #execute(Object)}.
185   * It calls {@link #releaseLock(Object)} after the call to execute.
186   *
187   * <p>If you need to hold the lock for the life of the Procdure -- i.e. you do not
188   * want any other Procedure interfering while this Procedure is running, see
189   * {@link #holdLock(Object)}.
190   *
191   * <p>Example: in our Master we can execute request in parallel for different tables.
192   * We can create t1 and create t2 and these creates can be executed at the same time.
193   * Anything else on t1/t2 is queued waiting that specific table create to happen.
194   *
195   * <p>There are 3 LockState:
196   * <ul><li>LOCK_ACQUIRED should be returned when the proc has the lock and the proc is
197   * ready to execute.</li>
198   * <li>LOCK_YIELD_WAIT should be returned when the proc has not the lock and the framework
199   * should take care of readding the procedure back to the runnable set for retry</li>
200   * <li>LOCK_EVENT_WAIT should be returned when the proc has not the lock and someone will
201   * take care of readding the procedure back to the runnable set when the lock is available.
202   * </li></ul>
203   * @return the lock state as described above.
204   */
205  protected LockState acquireLock(final TEnvironment env) {
206    return LockState.LOCK_ACQUIRED;
207  }

293   * Override this method to provide procedure specific counters for submitted count, failed

302  /**
303   * This function will be called just when procedure is submitted for execution. Override this
304   * method to update the metrics at the beginning of the procedure. The default implementation
305   * updates submitted counter if {@link #getProcedureMetrics(Object)} returns non-null
306   * {@link ProcedureMetrics}.
307   */
308  protected void updateMetricsOnSubmit(final TEnvironment env) {
309    ProcedureMetrics metrics = getProcedureMetrics(env);
310    if (metrics == null) {
311      return;
312    }
313
314    Counter submittedCounter = metrics.getSubmittedCounter();
315    if (submittedCounter != null) {
316      submittedCounter.increment();
317    }
318  }
319
320  /**
321   * This function will be called just after procedure execution is finished. Override this method
322   * to update metrics at the end of the procedure. If {@link #getProcedureMetrics(Object)}
323   * returns non-null {@link ProcedureMetrics}, the default implementation adds runtime of a
324   * procedure to a time histogram for successfully completed procedures. Increments failed
325   * counter for failed procedures.
326   *
327   * TODO: As any of the sub-procedures on failure rolls back all procedures in the stack,
328   * including successfully finished siblings, this function may get called twice in certain
329   * cases for certain procedures. Explore further if this can be called once.
330   *
331   * @param env The environment passed to the procedure executor
332   * @param runtime Runtime of the procedure in milliseconds
333   * @param success true if procedure is completed successfully
334   */
335  protected void updateMetricsOnFinish(final TEnvironment env, final long runtime,
336                                       boolean success) {
337    ProcedureMetrics metrics = getProcedureMetrics(env);
338    if (metrics == null) {
339      return;
340    }
341
342    if (success) {
343      Histogram timeHisto = metrics.getTimeHisto();
344      if (timeHisto != null) {
345        timeHisto.update(runtime);
346      }
347    } else {
348      Counter failedCounter = metrics.getFailedCounter();
349      if (failedCounter != null) {
350        failedCounter.increment();
351      }
352    }
353  }

361  /**
362   * Build the StringBuilder for the simple form of
363   * procedure string.
364   * @return the StringBuilder
365   */
366  protected StringBuilder toStringSimpleSB() {
367    final StringBuilder sb = new StringBuilder();
368
369    sb.append("pid=");
370    sb.append(getProcId());
371
372    if (hasParent()) {
373      sb.append(", ppid=");
374      sb.append(getParentProcId());
375    }
376
377    /**
378     * TODO
379     * Enable later when this is being used.
380     * Currently owner not used.
381    if (hasOwner()) {
382      sb.append(", owner=");
383      sb.append(getOwner());
384    }*/
385
386    sb.append(", state="); // pState for Procedure State as opposed to any other kind.
387    toStringState(sb);
388
389    if (hasException()) {
390      sb.append(", exception=" + getException());
391    }
392
393    sb.append("; ");
394    toStringClassDetails(sb);
395
396    return sb;
397  }
398
399  /**
400   * Extend the toString() information with more procedure
401   * details
402   */
403  public String toStringDetails() {
404    final StringBuilder sb = toStringSimpleSB();
405
406    sb.append(" submittedTime=");
407    sb.append(getSubmittedTime());
408
409    sb.append(", lastUpdate=");
410    sb.append(getLastUpdate());
411
412    final int[] stackIndices = getStackIndexes();
413    if (stackIndices != null) {
414      sb.append("\n");
415      sb.append("stackIndexes=");
416      sb.append(Arrays.toString(stackIndices));
417    }
418
419    return sb.toString();
420  }

422  protected String toStringClass() {
423    StringBuilder sb = new StringBuilder();
424    toStringClassDetails(sb);
425    return sb.toString();
426  }

703  /**
704   * Called by the ProcedureExecutor when the timeout set by setTimeout() is expired.
705   * @return true to let the framework handle the timeout as abort,
706   *         false in case the procedure handled the timeout itself.
707   */
708  protected synchronized boolean setTimeoutFailure(final TEnvironment env) {
709    if (state == ProcedureState.WAITING_TIMEOUT) {
710      long timeDiff = EnvironmentEdgeManager.currentTime() - lastUpdate;
711      setFailure("ProcedureExecutor", new TimeoutIOException(
712        "Operation timed out after " + StringUtils.humanTimeDiff(timeDiff)));
713      return true;
714    }
715    return false;
716  }

726  /**
727   * Called by the ProcedureExecutor on procedure-load to restore the latch state
728   */
729  @InterfaceAudience.Private
730  protected synchronized void setChildrenLatch(final int numChildren) {
731    this.childrenLatch = numChildren;
732    if (LOG.isTraceEnabled()) {
733      LOG.trace("CHILD LATCH INCREMENT SET " +
734          this.childrenLatch, new Throwable(this.toString()));
735    }
736  }
737
738  /**
739   * Called by the ProcedureExecutor on procedure-load to restore the latch state
740   */
741  @InterfaceAudience.Private
742  protected synchronized void incChildrenLatch() {
743    // TODO: can this be inferred from the stack? I think so...
744    this.childrenLatch++;
745    if (LOG.isTraceEnabled()) {
746      LOG.trace("CHILD LATCH INCREMENT " + this.childrenLatch, new Throwable(this.toString()));
747    }
748  }
749
750  /**
751   * Called by the ProcedureExecutor to notify that one of the sub-procedures has completed.
752   */
753  @InterfaceAudience.Private
754  private synchronized boolean childrenCountDown() {
755    assert childrenLatch > 0: this;
756    boolean b = --childrenLatch == 0;
757    if (LOG.isTraceEnabled()) {
758      LOG.trace("CHILD LATCH DECREMENT " + childrenLatch, new Throwable(this.toString()));
759    }
760    return b;
761  }
762
763  /**
764   * Try to set this procedure into RUNNABLE state.
765   * Succeeds if all subprocedures/children are done.
766   * @return True if we were able to move procedure to RUNNABLE state.
767   */
768  synchronized boolean tryRunnable() {
769    // Don't use isWaiting in the below; it returns true for WAITING and WAITING_TIMEOUT
770    boolean b = getState() == ProcedureState.WAITING && childrenCountDown();
771    if (b) setState(ProcedureState.RUNNABLE);
772    return b;
773  }

775  @InterfaceAudience.Private
776  protected synchronized boolean hasChildren() {
777    return childrenLatch > 0;
778  }
779
780  @InterfaceAudience.Private
781  protected synchronized int getChildrenLatch() {
782    return childrenLatch;
783  }

785  /**
786   * Called by the RootProcedureState on procedure execution.
787   * Each procedure store its stack-index positions.
788   */
789  @InterfaceAudience.Private
790  protected synchronized void addStackIndex(final int index) {
791    if (stackIndexes == null) {
792      stackIndexes = new int[] { index };
793    } else {
794      int count = stackIndexes.length;
795      stackIndexes = Arrays.copyOf(stackIndexes, count + 1);
796      stackIndexes[count] = index;
797    }
798  }
799
800  @InterfaceAudience.Private
801  protected synchronized boolean removeStackIndex() {
802    if (stackIndexes != null && stackIndexes.length > 1) {
803      stackIndexes = Arrays.copyOf(stackIndexes, stackIndexes.length - 1);
804      return false;
805    } else {
806      stackIndexes = null;
807      return true;
808    }
809  }
810
811  /**
812   * Called on store load to initialize the Procedure internals after
813   * the creation/deserialization.
814   */
815  @InterfaceAudience.Private
816  protected synchronized void setStackIndexes(final List<Integer> stackIndexes) {
817    this.stackIndexes = new int[stackIndexes.size()];
818    for (int i = 0; i < this.stackIndexes.length; ++i) {
819      this.stackIndexes[i] = stackIndexes.get(i);
820    }
821  }

823  @InterfaceAudience.Private
824  protected synchronized boolean wasExecuted() {
825    return stackIndexes != null;
826  }
827
828  @InterfaceAudience.Private
829  protected synchronized int[] getStackIndexes() {
830    return stackIndexes;
831  }

838   * Internal method called by the ProcedureExecutor that starts the user-level code execute().
839   * @throws ProcedureSuspendedException This is used when procedure wants to halt processing and
840   * skip out without changing states or releasing any locks held.

842  @InterfaceAudience.Private

844      throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {

894   * @return the hashcode for the specified procId
895   */
896  public static long getProcIdHashCode(final long procId) {
897    long h = procId;
898    h ^= h >> 16;
899    h *= 0x85ebca6b;
900    h ^= h >> 13;
901    h *= 0xc2b2ae35;
902    h ^= h >> 16;
903    return h;
904  }

906  /*
907   * Helper to lookup the root Procedure ID given a specified procedure.
908   */
909  @InterfaceAudience.Private
910  protected static Long getRootProcedureId(final Map<Long, Procedure> procedures,
911      Procedure<?> proc) {
912    while (proc.hasParent()) {
913      proc = procedures.get(proc.getParentProcId());
914      if (proc == null) return null;
915    }
916    return proc.getProcId();
917  }
