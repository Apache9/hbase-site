039/**
040 * Base Procedure class responsible to handle the Procedure Metadata
041 * e.g. state, startTime, lastUpdate, stack-indexes, ...
042 *
043 * execute() is called each time the procedure is executed.
044 * it may be called multiple times in case of failure and restart, so the
045 * code must be idempotent.
046 * the return is a set of sub-procedures or null in case the procedure doesn't
047 * have sub-procedures. Once the sub-procedures are successfully completed
048 * the execute() method is called again, you should think at it as a stack:
049 *  -&gt; step 1
050 *  ---&gt; step 2
051 *  -&gt; step 1
052 *
053 * rollback() is called when the procedure or one of the sub-procedures is failed.
054 * the rollback step is supposed to cleanup the resources created during the
055 * execute() step. in case of failure and restart rollback() may be called
056 * multiple times, so the code must be idempotent.
057 */
058@InterfaceAudience.Private
059@InterfaceStability.Evolving
060public abstract class Procedure<TEnvironment> implements Comparable<Procedure> {
061  public static final long NO_PROC_ID = -1;
062  protected static final int NO_TIMEOUT = -1;
063
064  public enum LockState {
065    LOCK_ACQUIRED,       // lock acquired and ready to execute
066    LOCK_YIELD_WAIT,     // lock not acquired, framework needs to yield
067    LOCK_EVENT_WAIT,     // lock not acquired, an event will yield the procedure
068  }

089  /**
090   * The main code of the procedure. It must be idempotent since execute()
091   * may be called multiple time in case of machine failure in the middle
092   * of the execution.
093   * @param env the environment passed to the ProcedureExecutor
094   * @return a set of sub-procedures or null if there is nothing else to execute.
095   * @throws ProcedureYieldException the procedure will be added back to the queue and retried later
096   * @throws InterruptedException the procedure will be added back to the queue and retried later
097   */
098  protected abstract Procedure[] execute(TEnvironment env)
099    throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException;
100
101  /**
102   * The code to undo what done by the execute() code.
103   * It is called when the procedure or one of the sub-procedure failed or an
104   * abort was requested. It should cleanup all the resources created by
105   * the execute() call. The implementation must be idempotent since rollback()
106   * may be called multiple time in case of machine failure in the middle
107   * of the execution.
108   * @param env the environment passed to the ProcedureExecutor
109   * @throws IOException temporary failure, the rollback will retry later
110   * @throws InterruptedException the procedure will be added back to the queue and retried later
111   */
112  protected abstract void rollback(TEnvironment env)
113    throws IOException, InterruptedException;
114
115  /**
116   * The abort() call is asynchronous and each procedure must decide how to deal
117   * with that, if they want to be abortable. The simplest implementation
118   * is to have an AtomicBoolean set in the abort() method and then the execute()
119   * will check if the abort flag is set or not.
120   * abort() may be called multiple times from the client, so the implementation
121   * must be idempotent.
122   *
123   * NOTE: abort() is not like Thread.interrupt() it is just a notification
124   * that allows the procedure implementor where to abort to avoid leak and
125   * have a better control on what was executed and what not.
126   */
127  protected abstract boolean abort(TEnvironment env);

130   * The user-level code of the procedure may have some state to

145  /**
146   * The user should override this method, and try to take a lock if necessary.
147   * A lock can be anything, and it is up to the implementor.
148   *
149   * <p>Example: in our Master we can execute request in parallel for different tables.
150   * We can create t1 and create t2 and this can be executed at the same time.
151   * Anything else on t1/t2 is queued waiting that specific table create to happen.
152   *
153   * <p>There are 3 LockState:
154   * <ul><li>LOCK_ACQUIRED should be returned when the proc has the lock and the proc is
155   * ready to execute.</li>
156   * <li>LOCK_YIELD_WAIT should be returned when the proc has not the lock and the framework
157   * should take care of readding the procedure back to the runnable set for retry</li>
158   * <li>LOCK_EVENT_WAIT should be returned when the proc has not the lock and someone will
159   * take care of readding the procedure back to the runnable set when the lock is available.
160   * </li></ul>
161   * @return the lock state as described above.
162   */
163  protected LockState acquireLock(final TEnvironment env) {
164    return LockState.LOCK_ACQUIRED;
165  }
166
167  /**
168   * The user should override this method, and release lock if necessary.
169   */
170  protected void releaseLock(final TEnvironment env) {
171    // no-op
172  }

175   * Used to keep the procedure lock even when the procedure is yielding or suspended.

179    return false;

243  @Override
244  public String toString() {
245    // Return the simple String presentation of the procedure.
246    return toStringSimpleSB().toString();
247  }

249  /**
250   * Build the StringBuilder for the simple form of
251   * procedure string.
252   * @return the StringBuilder
253   */
254  protected StringBuilder toStringSimpleSB() {
255    final StringBuilder sb = new StringBuilder();
256    toStringClassDetails(sb);
257
258    sb.append(", procId=");
259    sb.append(getProcId());
260
261    if (hasParent()) {
262      sb.append(", parent=");
263      sb.append(getParentProcId());
264    }
265
266    if (hasOwner()) {
267      sb.append(", owner=");
268      sb.append(getOwner());
269    }
270
271    sb.append(", state=");
272    toStringState(sb);
273
274    if (hasException()) {
275      sb.append(", failed=" + getException());
276    }
277
278    return sb;
279  }
280
281  /**
282   * Extend the toString() information with more procedure
283   * details
284   */
285  public String toStringDetails() {
286    final StringBuilder sb = toStringSimpleSB();
287
288    sb.append(" startTime=");
289    sb.append(getStartTime());
290
291    sb.append(" lastUpdate=");
292    sb.append(getLastUpdate());
293
294    final int[] stackIndices = getStackIndexes();
295    if (stackIndices != null) {
296      sb.append("\n");
297      sb.append("stackIndexes=");
298      sb.append(Arrays.toString(stackIndices));
299    }
300
301    return sb.toString();
302  }
303
304  protected String toStringClass() {
305    StringBuilder sb = new StringBuilder();
306    toStringClassDetails(sb);
307    return sb.toString();
308  }

497  // ==============================================================================================
498  //  Runtime state, updated every operation by the ProcedureExecutor
499  //
500  //  There is always 1 thread at the time operating on the state of the procedure.
501  //  The ProcedureExecutor may check and set states, or some Procecedure may
502  //  update its own state. but no concurrent updates. we use synchronized here
503  //  just because the procedure can get scheduled on different executor threads on each step.
504  // ==============================================================================================

506  /**
507   * @return true if the procedure is in a RUNNABLE state.
508   */
509  protected synchronized boolean isRunnable() {
510    return state == ProcedureState.RUNNABLE;
511  }
512
513  public synchronized boolean isInitializing() {
514    return state == ProcedureState.INITIALIZING;
515  }

517  /**
518   * @return true if the procedure has failed.
519   *         true may mean failed but not yet rolledback or failed and rolledback.
520   */
521  public synchronized boolean isFailed() {
522    return exception != null || state == ProcedureState.ROLLEDBACK;
523  }
524
525  /**
526   * @return true if the procedure is finished successfully.
527   */
528  public synchronized boolean isSuccess() {
529    return state == ProcedureState.FINISHED && exception == null;
530  }
531
532  /**
533   * @return true if the procedure is finished. The Procedure may be completed
534   *         successfuly or failed and rolledback.
535   */
536  public synchronized boolean isFinished() {
537    switch (state) {
538      case ROLLEDBACK:
539        return true;
540      case FINISHED:
541        return exception == null;
542      default:
543        break;
544    }
545    return false;
546  }
547
548  /**
549   * @return true if the procedure is waiting for a child to finish or for an external event.
550   */
551  public synchronized boolean isWaiting() {
552    switch (state) {
553      case WAITING:
554      case WAITING_TIMEOUT:
555        return true;
556      default:
557        break;
558    }
559    return false;
560  }

589  /**
590   * Called by the ProcedureExecutor when the timeout set by setTimeout() is expired.
591   * @return true to let the framework handle the timeout as abort,
592   *         false in case the procedure handled the timeout itself.
593   */
594  protected synchronized boolean setTimeoutFailure(final TEnvironment env) {
595    if (state == ProcedureState.WAITING_TIMEOUT) {
596      long timeDiff = EnvironmentEdgeManager.currentTime() - lastUpdate;
597      setFailure("ProcedureExecutor", new TimeoutIOException(
598        "Operation timed out after " + StringUtils.humanTimeDiff(timeDiff)));
599      return true;
600    }
601    return false;
602  }

648  /**
649   * Called by the RootProcedureState on procedure execution.
650   * Each procedure store its stack-index positions.
651   */
652  @InterfaceAudience.Private
653  protected synchronized void addStackIndex(final int index) {
654    if (stackIndexes == null) {
655      stackIndexes = new int[] { index };
656    } else {
657      int count = stackIndexes.length;
658      stackIndexes = Arrays.copyOf(stackIndexes, count + 1);
659      stackIndexes[count] = index;
660    }
661  }
662
663  @InterfaceAudience.Private
664  protected synchronized boolean removeStackIndex() {
665    if (stackIndexes != null && stackIndexes.length > 1) {
666      stackIndexes = Arrays.copyOf(stackIndexes, stackIndexes.length - 1);
667      return false;
668    } else {
669      stackIndexes = null;
670      return true;
671    }
672  }
673
674  /**
675   * Called on store load to initialize the Procedure internals after
676   * the creation/deserialization.
677   */
678  @InterfaceAudience.Private
679  protected synchronized void setStackIndexes(final List<Integer> stackIndexes) {
680    this.stackIndexes = new int[stackIndexes.size()];
681    for (int i = 0; i < this.stackIndexes.length; ++i) {
682      this.stackIndexes[i] = stackIndexes.get(i);
683    }
684  }

688    return stackIndexes != null;
689  }

700  /**
701   * Internal method called by the ProcedureExecutor that starts the user-level code execute().
702   */
703  @InterfaceAudience.Private
704  protected Procedure[] doExecute(final TEnvironment env)
705      throws ProcedureYieldException, ProcedureSuspendedException, InterruptedException {
706    try {
707      updateTimestamp();
708      return execute(env);
709    } finally {
710      updateTimestamp();
711    }
712  }

714  /**
715   * Internal method called by the ProcedureExecutor that starts the user-level code rollback().
716   */
717  @InterfaceAudience.Private
718  protected void doRollback(final TEnvironment env)
719      throws IOException, InterruptedException {
720    try {
721      updateTimestamp();
722      rollback(env);
723    } finally {
724      updateTimestamp();
725    }
726  }

753  /**
754   * Get an hashcode for the specified Procedure ID
755   * @return the hashcode for the specified procId
756   */
757  public static long getProcIdHashCode(final long procId) {
758    long h = procId;
759    h ^= h >> 16;
760    h *= 0x85ebca6b;
761    h ^= h >> 13;
762    h *= 0xc2b2ae35;
763    h ^= h >> 16;
764    return h;
765  }

767  /*
768   * Helper to lookup the root Procedure ID given a specified procedure.
769   */
770  @InterfaceAudience.Private
771  protected static Long getRootProcedureId(final Map<Long, Procedure> procedures, Procedure proc) {
772    while (proc.hasParent()) {
773      proc = procedures.get(proc.getParentProcId());
774      if (proc == null) return null;
775    }
776    return proc.getProcId();
777  }

779  /**
780   * @param a the first procedure to be compared.
781   * @param b the second procedure to be compared.
782   * @return true if the two procedures have the same parent
783   */
784  public static boolean haveSameParent(final Procedure a, final Procedure b) {
785    if (a.hasParent() && b.hasParent()) {
786      return a.getParentProcId() == b.getParentProcId();
787    }
788    return false;
