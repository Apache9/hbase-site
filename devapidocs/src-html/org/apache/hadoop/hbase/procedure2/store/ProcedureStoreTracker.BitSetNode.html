031/**
032 * Keeps track of live procedures.
033 *
034 * It can be used by the ProcedureStore to identify which procedures are already
035 * deleted/completed to avoid the deserialization step on restart
036 */
037@InterfaceAudience.Private
038@InterfaceStability.Evolving
039public class ProcedureStoreTracker {
040  // Key is procedure id corresponding to first bit of the bitmap.
041  private final TreeMap<Long, BitSetNode> map = new TreeMap<>();
042
043  /**
044   * If true, do not remove bits corresponding to deleted procedures. Note that this can result
045   * in huge bitmaps overtime.
046   * Currently, it's set to true only when building tracker state from logs during recovery. During
047   * recovery, if we are sure that a procedure has been deleted, reading its old update entries
048   * can be skipped.
049   */
050  private boolean keepDeletes = false;
051  /**
052   * If true, it means tracker has incomplete information about the active/deleted procedures.
053   * It's set to true only when recovering from old logs. See {@link #isDeleted(long)} docs to
054   * understand it's real use.
055   */
056  private boolean partial = false;
057
058  private long minUpdatedProcId = Long.MAX_VALUE;
059  private long maxUpdatedProcId = Long.MIN_VALUE;
060
061  public enum DeleteState { YES, NO, MAYBE }
062
063  /**
064   * A bitmap which can grow/merge with other {@link BitSetNode} (if certain conditions are met).
065   * Boundaries of bitmap are aligned to multiples of {@link BitSetNode#BITS_PER_WORD}. So the
066   * range of a {@link BitSetNode} is from [x * K, y * K) where x and y are integers, y > x and K
067   * is BITS_PER_WORD.
068   */
069  public static class BitSetNode {
070    private final static long WORD_MASK = 0xffffffffffffffffL;
071    private final static int ADDRESS_BITS_PER_WORD = 6;
072    private final static int BITS_PER_WORD = 1 << ADDRESS_BITS_PER_WORD;
073    private final static int MAX_NODE_SIZE = 1 << ADDRESS_BITS_PER_WORD;
074
075    /**
076     * Mimics {@link ProcedureStoreTracker#partial}.
077     */
078    private final boolean partial;
079
080    /* ----------------------
081     * |  updated | deleted |  meaning
082     * |     0    |   0     |  proc exists, but hasn't been updated since last resetUpdates().
083     * |     1    |   0     |  proc was updated (but not deleted).
084     * |     1    |   1     |  proc was deleted.
085     * |     0    |   1     |  proc doesn't exist (maybe never created, maybe deleted in past).
086    /* ----------------------
087     */
088
089    /**
090     * Set of procedures which have been updated since last {@link #resetUpdates()}.
091     * Useful to track procedures which have been updated since last WAL write.
092     */
093    private long[] updated;
094    /**
095     * Keeps track of procedure ids which belong to this bitmap's range and have been deleted.
096     * This represents global state since it's not reset on WAL rolls.
097     */
098    private long[] deleted;
099    /**
100     * Offset of bitmap i.e. procedure id corresponding to first bit.
101     */
102    private long start;

125    public BitSetNode(final long procId, final boolean partial) {
126      start = alignDown(procId);
127
128      int count = 1;
129      updated = new long[count];
130      deleted = new long[count];
131      for (int i = 0; i < count; ++i) {
132        updated[i] = 0;
133        deleted[i] = partial ? 0 : WORD_MASK;
134      }
135
136      this.partial = partial;
137      updateState(procId, false);
138    }
139
140    protected BitSetNode(final long start, final long[] updated, final long[] deleted) {
141      this.start = start;
142      this.updated = updated;
143      this.deleted = deleted;
144      this.partial = false;
145    }

147    public BitSetNode(ProcedureProtos.ProcedureStoreTracker.TrackerNode data) {
148      start = data.getStartId();
149      int size = data.getUpdatedCount();
150      updated = new long[size];
151      deleted = new long[size];
152      for (int i = 0; i < size; ++i) {
153        updated[i] = data.getUpdated(i);
154        deleted[i] = data.getDeleted(i);
155      }
156      partial = false;
157    }

160      this.start = other.start;

162      this.updated = other.updated.clone();
163      this.deleted = other.deleted.clone();
164    }

174    public Long getStart() {
175      return start;
176    }

178    public Long getEnd() {
179      return start + (updated.length << ADDRESS_BITS_PER_WORD) - 1;
180    }

182    public boolean contains(final long procId) {
183      return start <= procId && procId <= getEnd();
184    }

186    public DeleteState isDeleted(final long procId) {
187      int bitmapIndex = getBitmapIndex(procId);
188      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
189      if (wordIndex >= deleted.length) {
190        return DeleteState.MAYBE;
191      }
192      return (deleted[wordIndex] & (1L << bitmapIndex)) != 0 ? DeleteState.YES : DeleteState.NO;
193    }
194
195    private boolean isUpdated(final long procId) {
196      int bitmapIndex = getBitmapIndex(procId);
197      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
198      if (wordIndex >= updated.length) {
199        return false;
200      }
201      return (updated[wordIndex] & (1L << bitmapIndex)) != 0;
202    }
203
204    public boolean isUpdated() {
205      // TODO: cache the value
206      for (int i = 0; i < updated.length; ++i) {
207        if ((updated[i] | deleted[i]) != WORD_MASK) {
208          return false;
209        }
210      }
211      return true;
212    }

214    /**
215     * @return true, if there are no active procedures in this BitSetNode, else false.
216     */
217    public boolean isEmpty() {
218      // TODO: cache the value
219      for (int i = 0; i < deleted.length; ++i) {
220        if (deleted[i] != WORD_MASK) {
221          return false;
222        }
223      }
224      return true;
225    }

242    public void unsetPartialFlag() {
243      for (int i = 0; i < updated.length; ++i) {
244        for (int j = 0; j < BITS_PER_WORD; ++j) {
245          if ((updated[i] & (1L << j)) == 0) {
246            deleted[i] |= (1L << j);
247          }
248        }
249      }
250    }

252    /**
253     * If an active (non-deleted) procedure in current BitSetNode has been updated in {@code other}
254     * BitSetNode, then delete it from current node.
255     * @return true if node changed, i.e. some procedure(s) from {@code other} was subtracted from
256     * current node.
257     */
258    public boolean subtract(BitSetNode other) {
259      // Assert that other node intersects with this node.
260      assert !(other.getEnd() < this.start) && !(this.getEnd() < other.start);
261      int thisOffset = 0, otherOffset = 0;
262      if (this.start < other.start) {
263        thisOffset = (int) (other.start - this.start) / BITS_PER_WORD;
264      } else {
265        otherOffset = (int) (this.start - other.start) / BITS_PER_WORD;
266      }
267      int size = Math.min(this.updated.length - thisOffset, other.updated.length - otherOffset);
268      boolean nonZeroIntersect = false;
269      for (int i = 0; i < size; i++) {
270        long intersect = ~this.deleted[thisOffset + i] & other.updated[otherOffset + i];
271        if (intersect != 0) {
272          this.deleted[thisOffset + i] |= intersect;
273          nonZeroIntersect = true;
274        }
275      }
276      return nonZeroIntersect;
277    }

282     * protobuf.

284    public ProcedureProtos.ProcedureStoreTracker.TrackerNode convert() {
285      ProcedureProtos.ProcedureStoreTracker.TrackerNode.Builder builder =
286        ProcedureProtos.ProcedureStoreTracker.TrackerNode.newBuilder();
287      builder.setStartId(start);
288      for (int i = 0; i < updated.length; ++i) {
289        builder.addUpdated(updated[i]);
290        builder.addDeleted(deleted[i]);
291      }
292      return builder.build();
293    }

299    public boolean canGrow(final long procId) {
300      return Math.abs(procId - start) < MAX_NODE_SIZE;
301    }
302
303    public boolean canMerge(final BitSetNode rightNode) {
304      // Can just compare 'starts' since boundaries are aligned to multiples of BITS_PER_WORD.
305      assert start < rightNode.start;
306      return (rightNode.getEnd() - start) < MAX_NODE_SIZE;
307    }
308
309    public void grow(final long procId) {
310      int delta, offset;
311
312      if (procId < start) {
313        // add to head
314        long newStart = alignDown(procId);
315        delta = (int)(start - newStart) >> ADDRESS_BITS_PER_WORD;
316        offset = delta;
317        start = newStart;
318      } else {
319        // Add to tail
320        long newEnd = alignUp(procId + 1);
321        delta = (int)(newEnd - getEnd()) >> ADDRESS_BITS_PER_WORD;
322        offset = 0;
323      }
324
325      long[] newBitmap;
326      int oldSize = updated.length;
327
328      newBitmap = new long[oldSize + delta];
329      for (int i = 0; i < newBitmap.length; ++i) {
330        newBitmap[i] = 0;
331      }
332      System.arraycopy(updated, 0, newBitmap, offset, oldSize);
333      updated = newBitmap;
334
335      newBitmap = new long[deleted.length + delta];
336      for (int i = 0; i < newBitmap.length; ++i) {
337        newBitmap[i] = partial ? 0 : WORD_MASK;
338      }
339      System.arraycopy(deleted, 0, newBitmap, offset, oldSize);
340      deleted = newBitmap;
341    }
342
343    public void merge(final BitSetNode rightNode) {
344      int delta = (int)(rightNode.getEnd() - getEnd()) >> ADDRESS_BITS_PER_WORD;
345
346      long[] newBitmap;
347      int oldSize = updated.length;
348      int newSize = (delta - rightNode.updated.length);
349      int offset = oldSize + newSize;
350
351      newBitmap = new long[oldSize + delta];
352      System.arraycopy(updated, 0, newBitmap, 0, oldSize);
353      System.arraycopy(rightNode.updated, 0, newBitmap, offset, rightNode.updated.length);
354      updated = newBitmap;
355
356      newBitmap = new long[oldSize + delta];
357      System.arraycopy(deleted, 0, newBitmap, 0, oldSize);
358      System.arraycopy(rightNode.deleted, 0, newBitmap, offset, rightNode.deleted.length);
359      deleted = newBitmap;
360
361      for (int i = 0; i < newSize; ++i) {
362        updated[offset + i] = 0;
363        deleted[offset + i] = partial ? 0 : WORD_MASK;
364      }
365    }

375    public long getActiveMinProcId() {
376      long minProcId = start;
377      for (int i = 0; i < deleted.length; ++i) {
378        if (deleted[i] == 0) {
379          return(minProcId);
380        }
381
382        if (deleted[i] != WORD_MASK) {
383          for (int j = 0; j < BITS_PER_WORD; ++j) {
384            if ((deleted[i] & (1L << j)) != 0) {
385              return minProcId + j;
386            }
387          }
388        }
389
390        minProcId += BITS_PER_WORD;
391      }
392      return minProcId;
393    }
394
395    public long getActiveMaxProcId() {
396      long maxProcId = getEnd();
397      for (int i = deleted.length - 1; i >= 0; --i) {
398        if (deleted[i] == 0) {
399          return maxProcId;
400        }
401
402        if (deleted[i] != WORD_MASK) {
403          for (int j = BITS_PER_WORD - 1; j >= 0; --j) {
404            if ((deleted[i] & (1L << j)) == 0) {
405              return maxProcId - (BITS_PER_WORD - 1 - j);
406            }
407          }
408        }
409        maxProcId -= BITS_PER_WORD;
410      }
411      return maxProcId;
412    }

414    // ========================================================================

416    // ========================================================================
417    private int getBitmapIndex(final long procId) {
418      return (int)(procId - start);
419    }
420
421    private void updateState(final long procId, final boolean isDeleted) {
422      int bitmapIndex = getBitmapIndex(procId);
423      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
424      long value = (1L << bitmapIndex);
425
426      updated[wordIndex] |= value;
427      if (isDeleted) {
428        deleted[wordIndex] |= value;
429      } else {
430        deleted[wordIndex] &= ~value;
431      }
432    }

438    /**
439     * @return upper boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
440     */
441    private static long alignUp(final long x) {
442      return (x + (BITS_PER_WORD - 1)) & -BITS_PER_WORD;
443    }

445    /**
446     * @return lower boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
447     */
448    private static long alignDown(final long x) {
449      return x & -BITS_PER_WORD;
450    }

487  public void update(long procId) {
488    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
489    assert entry != null : "expected node to update procId=" + procId;
490
491    BitSetNode node = entry.getValue();
492    assert node.contains(procId);
493    node.update(procId);
494    trackProcIds(procId);
495  }

497  public void delete(long procId) {
498    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
499    assert entry != null : "expected node to delete procId=" + procId;
500
501    BitSetNode node = entry.getValue();
502    assert node.contains(procId) : "expected procId in the node";
503    node.delete(procId);
504
505    if (!keepDeletes && node.isEmpty()) {
506      // TODO: RESET if (map.size() == 1)
507      map.remove(entry.getKey());
508    }
509
510    trackProcIds(procId);
511  }

554  /**
555   * If {@link #partial} is false, returns state from the bitmap. If no state is found for
556   * {@code procId}, returns YES.
557   * If partial is true, tracker doesn't have complete view of system state, so it returns MAYBE
558   * if there is no update for the procedure or if it doesn't have a state in bitmap. Otherwise,
559   * returns state from the bitmap.
560   */
561  public DeleteState isDeleted(long procId) {
562    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
563    if (entry != null && entry.getValue().contains(procId)) {
564      BitSetNode node = entry.getValue();
565      DeleteState state = node.isDeleted(procId);
566      return partial && !node.isUpdated(procId) ? DeleteState.MAYBE : state;
567    }
568    return partial ? DeleteState.MAYBE : DeleteState.YES;
569  }

577  public void setKeepDeletes(boolean keepDeletes) {
578    this.keepDeletes = keepDeletes;
579    // If not to keep deletes, remove the BitSetNodes which are empty (i.e. contains ids of deleted
580    // procedures).
581    if (!keepDeletes) {
582      Iterator<Map.Entry<Long, BitSetNode>> it = map.entrySet().iterator();
583      while (it.hasNext()) {
584        Map.Entry<Long, BitSetNode> entry = it.next();
585        if (entry.getValue().isEmpty()) {
586          it.remove();
587        }
588      }
589    }

652  private BitSetNode getOrCreateNode(final long procId) {
653    // If procId can fit in left node (directly or by growing it)
654    BitSetNode leftNode = null;
655    boolean leftCanGrow = false;
656    Map.Entry<Long, BitSetNode> leftEntry = map.floorEntry(procId);
657    if (leftEntry != null) {
658      leftNode = leftEntry.getValue();
659      if (leftNode.contains(procId)) {
660        return leftNode;
661      }
662      leftCanGrow = leftNode.canGrow(procId);
663    }
664
665    // If procId can fit in right node (directly or by growing it)
666    BitSetNode rightNode = null;
667    boolean rightCanGrow = false;
668    Map.Entry<Long, BitSetNode> rightEntry = map.ceilingEntry(procId);
669    if (rightEntry != null) {
670      rightNode = rightEntry.getValue();
671      rightCanGrow = rightNode.canGrow(procId);
672      if (leftNode != null) {
673        if (leftNode.canMerge(rightNode)) {
674          // merge left and right node
675          return mergeNodes(leftNode, rightNode);
676        }
677
678        // If left and right nodes can not merge, decide which one to grow.
679        if (leftCanGrow && rightCanGrow) {
680          if ((procId - leftNode.getEnd()) <= (rightNode.getStart() - procId)) {
681            return growNode(leftNode, procId);
682          }
683          return growNode(rightNode, procId);
684        }
685      }
686    }
687
688    // grow the left node
689    if (leftCanGrow) {
690      return growNode(leftNode, procId);
691    }
692
693    // grow the right node
694    if (rightCanGrow) {
695      return growNode(rightNode, procId);
696    }
697
698    // add new node if there are no left/right nodes which can be used.
699    BitSetNode node = new BitSetNode(procId, partial);
700    map.put(node.getStart(), node);
701    return node;
702  }

704  /**
705   * Grows {@code node} to contain {@code procId} and updates the map.
706   * @return {@link BitSetNode} instance which contains {@code procId}.
707   */
708  private BitSetNode growNode(BitSetNode node, long procId) {
709    map.remove(node.getStart());
710    node.grow(procId);
711    map.put(node.getStart(), node);
712    return node;
713  }

715  /**
716   * Merges {@code leftNode} & {@code rightNode} and updates the map.
717   */
718  private BitSetNode mergeNodes(BitSetNode leftNode, BitSetNode rightNode) {
719    assert leftNode.getStart() < rightNode.getStart();
720    leftNode.merge(rightNode);
721    map.remove(rightNode.getStart());
722    return leftNode;
723  }

734  /**
735   * Iterates over
736   * {@link BitSetNode}s in this.map and subtracts with corresponding ones from {@code other}
737   * tracker.
738   * @return true if tracker changed, i.e. some procedure from {@code other} were subtracted from
739   * current tracker.
740   */
741  public boolean subtract(ProcedureStoreTracker other) {
742    // Can not intersect partial bitmap.
743    assert !partial && !other.partial;
744    boolean nonZeroIntersect = false;
745    for (Map.Entry<Long, BitSetNode> currentEntry : map.entrySet()) {
746      BitSetNode currentBitSetNode = currentEntry.getValue();
747      Map.Entry<Long, BitSetNode> otherTrackerEntry = other.map.floorEntry(currentEntry.getKey());
748      if (otherTrackerEntry == null  // No node in other map with key <= currentEntry.getKey().
749          // First entry in other map doesn't intersect with currentEntry.
750          || otherTrackerEntry.getValue().getEnd() < currentEntry.getKey()) {
751        otherTrackerEntry = other.map.ceilingEntry(currentEntry.getKey());
752        if (otherTrackerEntry == null || !currentBitSetNode.contains(otherTrackerEntry.getKey())) {
753          // No node in other map intersects with currentBitSetNode's range.
754          continue;
755        }
756      }
757      do {
758        nonZeroIntersect |= currentEntry.getValue().subtract(otherTrackerEntry.getValue());
759        otherTrackerEntry = other.map.higherEntry(otherTrackerEntry.getKey());
760      } while (otherTrackerEntry != null && currentBitSetNode.contains(otherTrackerEntry.getKey()));
761    }
762    return nonZeroIntersect;
763  }
