032/**
033 * Keeps track of live procedures.
034 *
035 * It can be used by the ProcedureStore to identify which procedures are already
036 * deleted/completed to avoid the deserialization step on restart
037 */
038@InterfaceAudience.Private
039@InterfaceStability.Evolving
040public class ProcedureStoreTracker {
041  // Key is procedure id corresponding to first bit of the bitmap.
042  private final TreeMap<Long, BitSetNode> map = new TreeMap<Long, BitSetNode>();
043
044  /**
045   * If true, do not remove bits corresponding to deleted procedures. Note that this can result
046   * in huge bitmaps overtime.
047   * Currently, it's set to true only when building tracker state from logs during recovery. During
048   * recovery, if we are sure that a procedure has been deleted, reading its old update entries
049   * can be skipped.
050   */
051  private boolean keepDeletes = false;
052  /**
053   * If true, it means tracker has incomplete information about the active/deleted procedures.
054   * It's set to true only when recovering from old logs. See {@link #isDeleted(long)} docs to
055   * understand it's real use.
056   */
057  private boolean partial = false;
058
059  private long minUpdatedProcId = Long.MAX_VALUE;
060  private long maxUpdatedProcId = Long.MIN_VALUE;
061
062  public enum DeleteState { YES, NO, MAYBE }
063
064  /**
065   * A bitmap which can grow/merge with other {@link BitSetNode} (if certain conditions are met).
066   * Boundaries of bitmap are aligned to multiples of {@link BitSetNode#BITS_PER_WORD}. So the
067   * range of a {@link BitSetNode} is from [x * K, y * K) where x and y are integers, y > x and K
068   * is BITS_PER_WORD.
069   */
070  public static class BitSetNode {
071    private final static long WORD_MASK = 0xffffffffffffffffL;
072    private final static int ADDRESS_BITS_PER_WORD = 6;
073    private final static int BITS_PER_WORD = 1 << ADDRESS_BITS_PER_WORD;
074    private final static int MAX_NODE_SIZE = 1 << ADDRESS_BITS_PER_WORD;
075
076    /**
077     * Mimics {@link ProcedureStoreTracker#partial}.
078     */
079    private final boolean partial;
080    /**
081     * Set of procedures which have been updated since last {@link #resetUpdates()}.
082     * Useful to track procedures which have been updated since last WAL write.
083     */
084    private long[] updated;
085    /**
086     * Keeps track of procedure ids which belong to this bitmap's range and have been deleted.
087     */
088    private long[] deleted;
089    /**
090     * Offset of bitmap i.e. procedure id corresponding to first bit.
091     */
092    private long start;
093
094    public void dump() {
095      System.out.printf("%06d:%06d min=%d max=%d%n", getStart(), getEnd(),
096        getActiveMinProcId(), getActiveMaxProcId());

101        }

104      System.out.println();
105      System.out.println("Delete:");
106      for (int i = 0; i < deleted.length; ++i) {

108          System.out.print((deleted[i] & (1L << j)) != 0 ? "1" : "0");
109        }
110        System.out.println(" " + i);

115    public BitSetNode(final long procId, final boolean partial) {
116      start = alignDown(procId);
117
118      int count = 1;
119      updated = new long[count];
120      deleted = new long[count];
121      for (int i = 0; i < count; ++i) {
122        updated[i] = 0;
123        deleted[i] = partial ? 0 : WORD_MASK;
124      }
125
126      this.partial = partial;
127      updateState(procId, false);
128    }
129
130    protected BitSetNode(final long start, final long[] updated, final long[] deleted) {
131      this.start = start;
132      this.updated = updated;
133      this.deleted = deleted;
134      this.partial = false;
135    }
136
137    public void update(final long procId) {
138      updateState(procId, false);
139    }
140
141    public void delete(final long procId) {
142      updateState(procId, true);
143    }
144
145    public Long getStart() {
146      return start;
147    }
148
149    public Long getEnd() {
150      return start + (updated.length << ADDRESS_BITS_PER_WORD) - 1;
151    }
152
153    public boolean contains(final long procId) {
154      return start <= procId && procId <= getEnd();
155    }
156
157    public DeleteState isDeleted(final long procId) {
158      int bitmapIndex = getBitmapIndex(procId);
159      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
160      if (wordIndex >= deleted.length) {
161        return DeleteState.MAYBE;
162      }
163      return (deleted[wordIndex] & (1L << bitmapIndex)) != 0 ? DeleteState.YES : DeleteState.NO;
164    }
165
166    private boolean isUpdated(final long procId) {
167      int bitmapIndex = getBitmapIndex(procId);
168      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
169      if (wordIndex >= updated.length) {
170        return false;
171      }
172      return (updated[wordIndex] & (1L << bitmapIndex)) != 0;
173    }
174
175    public boolean isUpdated() {
176      // TODO: cache the value
177      for (int i = 0; i < updated.length; ++i) {
178        if ((updated[i] | deleted[i]) != WORD_MASK) {
179          return false;
180        }
181      }
182      return true;
183    }
184
185    /**
186     * @return true, if there are no active procedures in this BitSetNode, else false.
187     */
188    public boolean isEmpty() {
189      // TODO: cache the value
190      for (int i = 0; i < deleted.length; ++i) {
191        if (deleted[i] != WORD_MASK) {
192          return false;
193        }
194      }
195      return true;
196    }
197
198    public void resetUpdates() {
199      for (int i = 0; i < updated.length; ++i) {
200        updated[i] = 0;
201      }
202    }
203
204    /**
205     * Clears the {@link #deleted} bitmaps.
206     */
207    public void undeleteAll() {
208      for (int i = 0; i < updated.length; ++i) {
209        deleted[i] = 0;
210      }
211    }
212
213    public void unsetPartialFlag() {
214      for (int i = 0; i < updated.length; ++i) {
215        for (int j = 0; j < BITS_PER_WORD; ++j) {
216          if ((updated[i] & (1L << j)) == 0) {
217            deleted[i] |= (1L << j);
218          }
219        }
220      }
221    }

227    public ProcedureProtos.ProcedureStoreTracker.TrackerNode convert() {
228      ProcedureProtos.ProcedureStoreTracker.TrackerNode.Builder builder =
229        ProcedureProtos.ProcedureStoreTracker.TrackerNode.newBuilder();
230      builder.setStartId(start);
231      for (int i = 0; i < updated.length; ++i) {
232        builder.addUpdated(updated[i]);
233        builder.addDeleted(deleted[i]);
234      }
235      return builder.build();
236    }

238    public static BitSetNode convert(ProcedureProtos.ProcedureStoreTracker.TrackerNode data) {
239      long start = data.getStartId();
240      int size = data.getUpdatedCount();
241      long[] updated = new long[size];
242      long[] deleted = new long[size];
243      for (int i = 0; i < size; ++i) {
244        updated[i] = data.getUpdated(i);
245        deleted[i] = data.getDeleted(i);
246      }
247      return new BitSetNode(start, updated, deleted);
248    }

250    // ========================================================================
251    //  Grow/Merge Helpers
252    // ========================================================================
253    public boolean canGrow(final long procId) {
254      return Math.abs(procId - start) < MAX_NODE_SIZE;
255    }
256
257    public boolean canMerge(final BitSetNode rightNode) {
258      // Can just compare 'starts' since boundaries are aligned to multiples of BITS_PER_WORD.
259      assert start < rightNode.start;
260      return (rightNode.getEnd() - start) < MAX_NODE_SIZE;
261    }
262
263    public void grow(final long procId) {
264      int delta, offset;
265
266      if (procId < start) {
267        // add to head
268        long newStart = alignDown(procId);
269        delta = (int)(start - newStart) >> ADDRESS_BITS_PER_WORD;
270        offset = delta;
271        start = newStart;
272      } else {
273        // Add to tail
274        long newEnd = alignUp(procId + 1);
275        delta = (int)(newEnd - getEnd()) >> ADDRESS_BITS_PER_WORD;
276        offset = 0;
277      }
278
279      long[] newBitmap;
280      int oldSize = updated.length;
281
282      newBitmap = new long[oldSize + delta];
283      for (int i = 0; i < newBitmap.length; ++i) {
284        newBitmap[i] = 0;
285      }
286      System.arraycopy(updated, 0, newBitmap, offset, oldSize);
287      updated = newBitmap;
288
289      newBitmap = new long[deleted.length + delta];
290      for (int i = 0; i < newBitmap.length; ++i) {
291        newBitmap[i] = partial ? 0 : WORD_MASK;
292      }
293      System.arraycopy(deleted, 0, newBitmap, offset, oldSize);
294      deleted = newBitmap;
295    }
296
297    public void merge(final BitSetNode rightNode) {
298      int delta = (int)(rightNode.getEnd() - getEnd()) >> ADDRESS_BITS_PER_WORD;
299
300      long[] newBitmap;
301      int oldSize = updated.length;
302      int newSize = (delta - rightNode.updated.length);
303      int offset = oldSize + newSize;
304
305      newBitmap = new long[oldSize + delta];
306      System.arraycopy(updated, 0, newBitmap, 0, oldSize);
307      System.arraycopy(rightNode.updated, 0, newBitmap, offset, rightNode.updated.length);
308      updated = newBitmap;
309
310      newBitmap = new long[oldSize + delta];
311      System.arraycopy(deleted, 0, newBitmap, 0, oldSize);
312      System.arraycopy(rightNode.deleted, 0, newBitmap, offset, rightNode.deleted.length);
313      deleted = newBitmap;
314
315      for (int i = 0; i < newSize; ++i) {
316        updated[offset + i] = 0;
317        deleted[offset + i] = partial ? 0 : WORD_MASK;
318      }
319    }

327    //  Min/Max Helpers
328    // ========================================================================
329    public long getActiveMinProcId() {
330      long minProcId = start;
331      for (int i = 0; i < deleted.length; ++i) {
332        if (deleted[i] == 0) {
333          return(minProcId);
334        }
335
336        if (deleted[i] != WORD_MASK) {
337          for (int j = 0; j < BITS_PER_WORD; ++j) {
338            if ((deleted[i] & (1L << j)) != 0) {
339              return minProcId + j;
340            }
341          }
342        }
343
344        minProcId += BITS_PER_WORD;
345      }
346      return minProcId;
347    }
348
349    public long getActiveMaxProcId() {
350      long maxProcId = getEnd();
351      for (int i = deleted.length - 1; i >= 0; --i) {
352        if (deleted[i] == 0) {
353          return maxProcId;
354        }
355
356        if (deleted[i] != WORD_MASK) {
357          for (int j = BITS_PER_WORD - 1; j >= 0; --j) {
358            if ((deleted[i] & (1L << j)) == 0) {
359              return maxProcId - (BITS_PER_WORD - 1 - j);
360            }
361          }
362        }
363        maxProcId -= BITS_PER_WORD;
364      }
365      return maxProcId;
366    }

368    // ========================================================================
369    //  Bitmap Helpers
370    // ========================================================================
371    private int getBitmapIndex(final long procId) {
372      return (int)(procId - start);
373    }
374
375    private void updateState(final long procId, final boolean isDeleted) {
376      int bitmapIndex = getBitmapIndex(procId);
377      int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
378      long value = (1L << bitmapIndex);
379
380      if (isDeleted) {
381        updated[wordIndex] |= value;
382        deleted[wordIndex] |= value;
383      } else {
384        updated[wordIndex] |= value;
385        deleted[wordIndex] &= ~value;
386      }
387    }

392    /**
393     * @return upper boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
394     */
395    private static long alignUp(final long x) {
396      return (x + (BITS_PER_WORD - 1)) & -BITS_PER_WORD;
397    }
398
399    /**
400     * @return lower boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
401     */
402    private static long alignDown(final long x) {
403      return x & -BITS_PER_WORD;
404    }

420  public void update(long procId) {
421    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
422    assert entry != null : "expected node to update procId=" + procId;
423
424    BitSetNode node = entry.getValue();
425    assert node.contains(procId);
426    node.update(procId);
427    trackProcIds(procId);
428  }
429
430  public void delete(long procId) {
431    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
432    assert entry != null : "expected node to delete procId=" + procId;
433
434    BitSetNode node = entry.getValue();
435    assert node.contains(procId) : "expected procId in the node";
436    node.delete(procId);
437
438    if (!keepDeletes && node.isEmpty()) {
439      // TODO: RESET if (map.size() == 1)
440      map.remove(entry.getKey());
441    }
442
443    trackProcIds(procId);
444  }
445
446  private void trackProcIds(long procId) {
447    minUpdatedProcId = Math.min(minUpdatedProcId, procId);
448    maxUpdatedProcId = Math.max(maxUpdatedProcId, procId);
449  }

466  public void reset() {
467    this.keepDeletes = false;
468    this.partial = false;
469    this.map.clear();

473  /**
474   * If {@link #partial} is false, returns state from the bitmap. If no state is found for
475   * {@code procId}, returns YES.
476   * If partial is true, tracker doesn't have complete view of system state, so it returns MAYBE
477   * if there is no update for the procedure or if it doesn't have a state in bitmap. Otherwise,
478   * returns state from the bitmap.
479   */
480  public DeleteState isDeleted(long procId) {
481    Map.Entry<Long, BitSetNode> entry = map.floorEntry(procId);
482    if (entry != null && entry.getValue().contains(procId)) {
483      BitSetNode node = entry.getValue();
484      DeleteState state = node.isDeleted(procId);
485      return partial && !node.isUpdated(procId) ? DeleteState.MAYBE : state;
486    }
487    return partial ? DeleteState.MAYBE : DeleteState.YES;
488  }

490  public long getActiveMinProcId() {
491    // TODO: Cache?
492    Map.Entry<Long, BitSetNode> entry = map.firstEntry();
493    return entry == null ? 0 : entry.getValue().getActiveMinProcId();
494  }

496  public void setKeepDeletes(boolean keepDeletes) {
497    this.keepDeletes = keepDeletes;
498    // If not to keep deletes, remove the BitSetNodes which are empty (i.e. contains ids of deleted
499    // procedures).
500    if (!keepDeletes) {
501      Iterator<Map.Entry<Long, BitSetNode>> it = map.entrySet().iterator();
502      while (it.hasNext()) {
503        Map.Entry<Long, BitSetNode> entry = it.next();
504        if (entry.getValue().isEmpty()) {
505          it.remove();
506        }
507      }
508    }
509  }

511  public void setPartialFlag(boolean isPartial) {
512    if (this.partial && !isPartial) {
513      for (Map.Entry<Long, BitSetNode> entry : map.entrySet()) {
514        entry.getValue().unsetPartialFlag();
515      }
516    }

567  private BitSetNode getOrCreateNode(final long procId) {
568    // If procId can fit in left node (directly or by growing it)
569    BitSetNode leftNode = null;
570    boolean leftCanGrow = false;
571    Map.Entry<Long, BitSetNode> leftEntry = map.floorEntry(procId);
572    if (leftEntry != null) {
573      leftNode = leftEntry.getValue();
574      if (leftNode.contains(procId)) {
575        return leftNode;
576      }
577      leftCanGrow = leftNode.canGrow(procId);
578    }
579
580    // If procId can fit in right node (directly or by growing it)
581    BitSetNode rightNode = null;
582    boolean rightCanGrow = false;
583    Map.Entry<Long, BitSetNode> rightEntry = map.ceilingEntry(procId);
584    if (rightEntry != null) {
585      rightNode = rightEntry.getValue();
586      rightCanGrow = rightNode.canGrow(procId);
587      if (leftNode != null) {
588        if (leftNode.canMerge(rightNode)) {
589          // merge left and right node
590          return mergeNodes(leftNode, rightNode);
591        }
592
593        // If left and right nodes can not merge, decide which one to grow.
594        if (leftCanGrow && rightCanGrow) {
595          if ((procId - leftNode.getEnd()) <= (rightNode.getStart() - procId)) {
596            return growNode(leftNode, procId);
597          }
598          return growNode(rightNode, procId);
599        }
600      }
601    }
602
603    // grow the left node
604    if (leftCanGrow) {
605      return growNode(leftNode, procId);
606    }
607
608    // grow the right node
609    if (rightCanGrow) {
610      return growNode(rightNode, procId);
611    }
612
613    // add new node if there are no left/right nodes which can be used.
614    BitSetNode node = new BitSetNode(procId, partial);
615    map.put(node.getStart(), node);
616    return node;
617  }

619  /**
620   * Grows {@code node} to contain {@code procId} and updates the map.
621   * @return {@link BitSetNode} instance which contains {@code procId}.
622   */
623  private BitSetNode growNode(BitSetNode node, long procId) {
624    map.remove(node.getStart());
625    node.grow(procId);
626    map.put(node.getStart(), node);
627    return node;
628  }

630  /**
631   * Merges {@code leftNode} & {@code rightNode} and updates the map.
632   */
633  private BitSetNode mergeNodes(BitSetNode leftNode, BitSetNode rightNode) {
634    assert leftNode.getStart() < rightNode.getStart();
635    leftNode.merge(rightNode);
636    map.remove(rightNode.getStart());
637    return leftNode;
638  }

649  /**
650   * Builds
651   * {@link org.apache.hadoop.hbase.protobuf.generated.ProcedureProtos.ProcedureStoreTracker}
652   * protocol buffer from current state, serializes it and writes to the {@code stream}.
653   */
654  public void writeTo(final OutputStream stream) throws IOException {
655    ProcedureProtos.ProcedureStoreTracker.Builder builder =
656        ProcedureProtos.ProcedureStoreTracker.newBuilder();
657    for (Map.Entry<Long, BitSetNode> entry : map.entrySet()) {
658      builder.addNode(entry.getValue().convert());
659    }
660    builder.build().writeDelimitedTo(stream);
661  }

663  /**
664   * Reads serialized
665   * {@link org.apache.hadoop.hbase.protobuf.generated.ProcedureProtos.ProcedureStoreTracker}
666   * protocol buffer from the {@code stream}, and use it to build the state.
667   */
668  public void readFrom(final InputStream stream) throws IOException {
669    reset();
670    final ProcedureProtos.ProcedureStoreTracker data =
671        ProcedureProtos.ProcedureStoreTracker.parseDelimitedFrom(stream);
672    for (ProcedureProtos.ProcedureStoreTracker.TrackerNode protoNode: data.getNodeList()) {
673      final BitSetNode node = BitSetNode.convert(protoNode);
674      map.put(node.getStart(), node);
675    }
676  }
