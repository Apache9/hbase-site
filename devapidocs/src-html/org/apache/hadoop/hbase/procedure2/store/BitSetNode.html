018package org.apache.hadoop.hbase.procedure2.store;
019
020import java.util.Arrays;
021import org.apache.hadoop.hbase.procedure2.store.ProcedureStoreTracker.DeleteState;
022import org.apache.yetus.audience.InterfaceAudience;
023
024import org.apache.hadoop.hbase.shaded.protobuf.generated.ProcedureProtos;

026/**
027 * A bitmap which can grow/merge with other {@link BitSetNode} (if certain conditions are met).
028 * Boundaries of bitmap are aligned to multiples of {@link BitSetNode#BITS_PER_WORD}. So the range
029 * of a {@link BitSetNode} is from [x * K, y * K) where x and y are integers, y > x and K is
030 * BITS_PER_WORD.
031 * <p/>
032 * We have two main bit sets to describe the state of procedures, the meanings are:
033 *
034 * <pre>
035 *  ----------------------
036 * | modified | deleted |  meaning
037 * |     0    |   0     |  proc exists, but hasn't been updated since last resetUpdates().
038 * |     1    |   0     |  proc was updated (but not deleted).
039 * |     1    |   1     |  proc was deleted.
040 * |     0    |   1     |  proc doesn't exist (maybe never created, maybe deleted in past).
041 * ----------------------
042 * </pre>
043 *
044 * The meaning of modified is that, we have modified the state of the procedure, no matter insert,
045 * update, or delete. And if it is an insert or update, we will set the deleted to 0, if not we will
046 * set the delete to 1.
047 * <p/>
048 * For a non-partial BitSetNode, the initial modified value is 0 and deleted value is 1. For the
049 * partial one, the initial modified value is 0 and the initial deleted value is also 0. In
050 * {@link #unsetPartialFlag()} we will reset the deleted to 1 if it is not modified.
051 */
052@InterfaceAudience.Private
053class BitSetNode {
054  private static final long WORD_MASK = 0xffffffffffffffffL;
055  private static final int ADDRESS_BITS_PER_WORD = 6;
056  private static final int BITS_PER_WORD = 1 << ADDRESS_BITS_PER_WORD;
057  private static final int MAX_NODE_SIZE = 1 << ADDRESS_BITS_PER_WORD;
058
059  /**
060   * Mimics {@link ProcedureStoreTracker#partial}. It will effect how we fill the new deleted bits
061   * when growing.
062   */
063  private boolean partial;
064
065  /**
066   * Set of procedures which have been modified since last {@link #resetModified()}. Useful to track
067   * procedures which have been modified since last WAL write.
068   */
069  private long[] modified;
070
071  /**
072   * Keeps track of procedure ids which belong to this bitmap's range and have been deleted. This
073   * represents global state since it's not reset on WAL rolls.
074   */
075  private long[] deleted;
076  /**
077   * Offset of bitmap i.e. procedure id corresponding to first bit.
078   */
079  private long start;
080
081  public void dump() {
082    System.out.printf("%06d:%06d min=%d max=%d%n", getStart(), getEnd(), getActiveMinProcId(),
083      getActiveMaxProcId());
084    System.out.println("Modified:");
085    for (int i = 0; i < modified.length; ++i) {
086      for (int j = 0; j < BITS_PER_WORD; ++j) {
087        System.out.print((modified[i] & (1L << j)) != 0 ? "1" : "0");
088      }
089      System.out.println(" " + i);
090    }
091    System.out.println();
092    System.out.println("Delete:");
093    for (int i = 0; i < deleted.length; ++i) {
094      for (int j = 0; j < BITS_PER_WORD; ++j) {
095        System.out.print((deleted[i] & (1L << j)) != 0 ? "1" : "0");
096      }
097      System.out.println(" " + i);
098    }
099    System.out.println();
100  }
101
102  public BitSetNode(long procId, boolean partial) {
103    start = alignDown(procId);
104
105    int count = 1;
106    modified = new long[count];
107    deleted = new long[count];
108    if (!partial) {
109      Arrays.fill(deleted, WORD_MASK);
110    }
111
112    this.partial = partial;
113    updateState(procId, false);
114  }
115
116  public BitSetNode(ProcedureProtos.ProcedureStoreTracker.TrackerNode data) {
117    start = data.getStartId();
118    int size = data.getUpdatedCount();
119    assert size == data.getDeletedCount();
120    modified = new long[size];
121    deleted = new long[size];
122    for (int i = 0; i < size; ++i) {
123      modified[i] = data.getUpdated(i);
124      deleted[i] = data.getDeleted(i);
125    }
126    partial = false;
127  }
128
129  public BitSetNode(BitSetNode other, boolean resetDelete) {
130    this.start = other.start;
131    this.partial = other.partial;
132    this.modified = other.modified.clone();
133    // The resetDelete will be set to true when building cleanup tracker.
134    // The intention here is that, if a procedure is not modified in this tracker, then we do not
135    // need to take care of it, so we will set deleted to true for these bits, i.e, if modified is
136    // 0, then we set deleted to 1, otherwise keep it as is. So here, the equation is
137    // deleted |= ~modified, i.e,
138    if (resetDelete) {
139      this.deleted = new long[other.deleted.length];
140      for (int i = 0; i < this.deleted.length; ++i) {
141        this.deleted[i] |= ~(other.modified[i]);
142      }
143    } else {
144      this.deleted = other.deleted.clone();
145    }
146  }
147
148  public void insertOrUpdate(final long procId) {
149    updateState(procId, false);
150  }
151
152  public void delete(final long procId) {
153    updateState(procId, true);
154  }
155
156  public long getStart() {
157    return start;
158  }
159
160  public long getEnd() {
161    return start + (modified.length << ADDRESS_BITS_PER_WORD) - 1;
162  }
163
164  public boolean contains(final long procId) {
165    return start <= procId && procId <= getEnd();
166  }
167
168  public DeleteState isDeleted(final long procId) {
169    int bitmapIndex = getBitmapIndex(procId);
170    int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
171    if (wordIndex >= deleted.length) {
172      return DeleteState.MAYBE;
173    }
174    return (deleted[wordIndex] & (1L << bitmapIndex)) != 0 ? DeleteState.YES : DeleteState.NO;
175  }
176
177  public boolean isModified(long procId) {
178    int bitmapIndex = getBitmapIndex(procId);
179    int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
180    if (wordIndex >= modified.length) {
181      return false;
182    }
183    return (modified[wordIndex] & (1L << bitmapIndex)) != 0;
184  }
185
186  /**
187   * @return true, if all the procedures has been modified.
188   */
189  public boolean isAllModified() {
190    // TODO: cache the value
191    for (int i = 0; i < modified.length; ++i) {
192      if ((modified[i] | deleted[i]) != WORD_MASK) {
193        return false;
194      }
195    }
196    return true;
197  }
198
199  /**
200   * @return true, if there are no active procedures in this BitSetNode, else false.
201   */
202  public boolean isEmpty() {
203    // TODO: cache the value
204    for (int i = 0; i < deleted.length; ++i) {
205      if (deleted[i] != WORD_MASK) {
206        return false;
207      }
208    }
209    return true;
210  }
211
212  public void resetModified() {
213    Arrays.fill(modified, 0);
214  }
215
216  public void unsetPartialFlag() {
217    partial = false;
218    for (int i = 0; i < modified.length; ++i) {
219      for (int j = 0; j < BITS_PER_WORD; ++j) {
220        if ((modified[i] & (1L << j)) == 0) {
221          deleted[i] |= (1L << j);
222        }
223      }
224    }
225  }
226
227  /**
228   * Convert to
229   * org.apache.hadoop.hbase.protobuf.generated.ProcedureProtos.ProcedureStoreTracker.TrackerNode
230   * protobuf.
231   */
232  public ProcedureProtos.ProcedureStoreTracker.TrackerNode convert() {
233    ProcedureProtos.ProcedureStoreTracker.TrackerNode.Builder builder =
234      ProcedureProtos.ProcedureStoreTracker.TrackerNode.newBuilder();
235    builder.setStartId(start);
236    for (int i = 0; i < modified.length; ++i) {
237      builder.addUpdated(modified[i]);
238      builder.addDeleted(deleted[i]);
239    }
240    return builder.build();
241  }

243  // ========================================================================
244  // Grow/Merge Helpers
245  // ========================================================================
246  public boolean canGrow(final long procId) {
247    return Math.abs(procId - start) < MAX_NODE_SIZE;
248  }
249
250  public boolean canMerge(final BitSetNode rightNode) {
251    // Can just compare 'starts' since boundaries are aligned to multiples of BITS_PER_WORD.
252    assert start < rightNode.start;
253    return (rightNode.getEnd() - start) < MAX_NODE_SIZE;
254  }
255
256  public void grow(final long procId) {
257    int delta, offset;
258
259    if (procId < start) {
260      // add to head
261      long newStart = alignDown(procId);
262      delta = (int) (start - newStart) >> ADDRESS_BITS_PER_WORD;
263      offset = delta;
264      start = newStart;
265    } else {
266      // Add to tail
267      long newEnd = alignUp(procId + 1);
268      delta = (int) (newEnd - getEnd()) >> ADDRESS_BITS_PER_WORD;
269      offset = 0;
270    }
271
272    long[] newBitmap;
273    int oldSize = modified.length;
274
275    newBitmap = new long[oldSize + delta];
276    for (int i = 0; i < newBitmap.length; ++i) {
277      newBitmap[i] = 0;
278    }
279    System.arraycopy(modified, 0, newBitmap, offset, oldSize);
280    modified = newBitmap;
281
282    newBitmap = new long[deleted.length + delta];
283    for (int i = 0; i < newBitmap.length; ++i) {
284      newBitmap[i] = partial ? 0 : WORD_MASK;
285    }
286    System.arraycopy(deleted, 0, newBitmap, offset, oldSize);
287    deleted = newBitmap;
288  }
289
290  public void merge(final BitSetNode rightNode) {
291    int delta = (int) (rightNode.getEnd() - getEnd()) >> ADDRESS_BITS_PER_WORD;
292
293    long[] newBitmap;
294    int oldSize = modified.length;
295    int newSize = (delta - rightNode.modified.length);
296    int offset = oldSize + newSize;
297
298    newBitmap = new long[oldSize + delta];
299    System.arraycopy(modified, 0, newBitmap, 0, oldSize);
300    System.arraycopy(rightNode.modified, 0, newBitmap, offset, rightNode.modified.length);
301    modified = newBitmap;
302
303    newBitmap = new long[oldSize + delta];
304    System.arraycopy(deleted, 0, newBitmap, 0, oldSize);
305    System.arraycopy(rightNode.deleted, 0, newBitmap, offset, rightNode.deleted.length);
306    deleted = newBitmap;
307
308    for (int i = 0; i < newSize; ++i) {
309      modified[offset + i] = 0;
310      deleted[offset + i] = partial ? 0 : WORD_MASK;
311    }
312  }
313
314  @Override
315  public String toString() {
316    return "BitSetNode(" + getStart() + "-" + getEnd() + ")";
317  }

319  // ========================================================================
320  // Min/Max Helpers
321  // ========================================================================
322  public long getActiveMinProcId() {
323    long minProcId = start;
324    for (int i = 0; i < deleted.length; ++i) {
325      if (deleted[i] == 0) {
326        return (minProcId);
327      }
328
329      if (deleted[i] != WORD_MASK) {
330        for (int j = 0; j < BITS_PER_WORD; ++j) {
331          if ((deleted[i] & (1L << j)) != 0) {
332            return minProcId + j;
333          }
334        }
335      }
336
337      minProcId += BITS_PER_WORD;
338    }
339    return minProcId;
340  }
341
342  public long getActiveMaxProcId() {
343    long maxProcId = getEnd();
344    for (int i = deleted.length - 1; i >= 0; --i) {
345      if (deleted[i] == 0) {
346        return maxProcId;
347      }
348
349      if (deleted[i] != WORD_MASK) {
350        for (int j = BITS_PER_WORD - 1; j >= 0; --j) {
351          if ((deleted[i] & (1L << j)) == 0) {
352            return maxProcId - (BITS_PER_WORD - 1 - j);
353          }
354        }
355      }
356      maxProcId -= BITS_PER_WORD;
357    }
358    return maxProcId;
359  }
360
361  // ========================================================================
362  // Bitmap Helpers
363  // ========================================================================
364  private int getBitmapIndex(final long procId) {
365    return (int) (procId - start);
366  }
367
368  void updateState(long procId, boolean isDeleted) {
369    int bitmapIndex = getBitmapIndex(procId);
370    int wordIndex = bitmapIndex >> ADDRESS_BITS_PER_WORD;
371    long value = (1L << bitmapIndex);
372
373    modified[wordIndex] |= value;
374    if (isDeleted) {
375      deleted[wordIndex] |= value;
376    } else {
377      deleted[wordIndex] &= ~value;
378    }
379  }

381  // ========================================================================
382  // Helpers
383  // ========================================================================
384  /**
385   * @return upper boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
386   */
387  private static long alignUp(final long x) {
388    return (x + (BITS_PER_WORD - 1)) & -BITS_PER_WORD;
389  }
390
391  /**
392   * @return lower boundary (aligned to multiple of BITS_PER_WORD) of bitmap range x belongs to.
393   */
394  private static long alignDown(final long x) {
395    return x & -BITS_PER_WORD;
396  }
