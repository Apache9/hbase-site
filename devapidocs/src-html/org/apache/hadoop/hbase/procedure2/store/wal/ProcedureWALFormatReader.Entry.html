038/**
039 * Helper class that loads the procedures stored in a WAL
040 */
041@InterfaceAudience.Private
042@InterfaceStability.Evolving
043public class ProcedureWALFormatReader {
044  private static final Log LOG = LogFactory.getLog(ProcedureWALFormatReader.class);
045
046  // ==============================================================================================
047  //  We read the WALs in reverse order from the newest to the oldest.
048  //  We have different entry types:
049  //   - INIT: Procedure submitted by the user (also known as 'root procedure')
050  //   - INSERT: Children added to the procedure <parentId>:[<childId>, ...]
051  //   - UPDATE: The specified procedure was updated
052  //   - DELETE: The procedure was removed (finished/rolledback and result TTL expired)
053  //
054  // In the WAL we can find multiple times the same procedure as UPDATE or INSERT.
055  // We read the WAL from top to bottom, so every time we find an entry of the
056  // same procedure, that will be the "latest" update (Caveat: with multiple threads writing
057  // the store, this assumption does not hold).
058  //
059  // We keep two in-memory maps:
060  //  - localProcedureMap: is the map containing the entries in the WAL we are processing
061  //  - procedureMap: is the map containing all the procedures we found up to the WAL in process.
062  // localProcedureMap is merged with the procedureMap once we reach the WAL EOF.
063  //
064  // Since we are reading the WALs in reverse order (newest to oldest),
065  // if we find an entry related to a procedure we already have in 'procedureMap' we can discard it.
066  //
067  // The WAL is append-only so the last procedure in the WAL is the one that
068  // was in execution at the time we crashed/closed the server.
069  // Given that, the procedure replay order can be inferred by the WAL order.
070  //
071  // Example:
072  //    WAL-2: [A, B, A, C, D]
073  //    WAL-1: [F, G, A, F, B]
074  //    Replay-Order: [D, C, A, B, F, G]
075  //
076  // The "localProcedureMap" keeps a "replayOrder" list. Every time we add the
077  // record to the map that record is moved to the head of the "replayOrder" list.
078  // Using the example above:
079  //    WAL-2 localProcedureMap.replayOrder is [D, C, A, B]
080  //    WAL-1 localProcedureMap.replayOrder is [F, G]
081  //
082  // Each time we reach the WAL-EOF, the "replayOrder" list is merged/appended in 'procedureMap'
083  // so using the example above we end up with: [D, C, A, B] + [F, G] as replay order.
084  //
085  //  Fast Start: INIT/INSERT record and StackIDs
086  // ---------------------------------------------
087  // We have two special records, INIT and INSERT, that track the first time
088  // the procedure was added to the WAL. We can use this information to be able
089  // to start procedures before reaching the end of the WAL, or before reading all WALs.
090  // But in some cases, the WAL with that record can be already gone.
091  // As an alternative, we can use the stackIds on each procedure,
092  // to identify when a procedure is ready to start.
093  // If there are gaps in the sum of the stackIds we need to read more WALs.
094  //
095  // Example (all procs child of A):
096  //   WAL-2: [A, B]                   A stackIds = [0, 4], B stackIds = [1, 5]
097  //   WAL-1: [A, B, C, D]
098  //
099  // In the case above we need to read one more WAL to be able to consider
100  // the root procedure A and all children as ready.
101  // ==============================================================================================
102  private final WalProcedureMap localProcedureMap = new WalProcedureMap(1024);
103  private final WalProcedureMap procedureMap = new WalProcedureMap(1024);

138  public void read(final ProcedureWALFile log) throws IOException {
139    localTracker = log.getTracker().isPartial() ? log.getTracker() : null;
140    if (localTracker != null) {
141      LOG.info("Rebuilding tracker for " + log);
142    }
143
144    FSDataInputStream stream = log.getStream();
145    try {
146      boolean hasMore = true;
147      while (hasMore) {
148        ProcedureWALEntry entry = ProcedureWALFormat.readEntry(stream);
149        if (entry == null) {
150          LOG.warn("Nothing left to decode. Exiting with missing EOF, log=" + log);
151          break;
152        }
153        switch (entry.getType()) {
154          case PROCEDURE_WAL_INIT:
155            readInitEntry(entry);
156            break;
157          case PROCEDURE_WAL_INSERT:
158            readInsertEntry(entry);
159            break;
160          case PROCEDURE_WAL_UPDATE:
161          case PROCEDURE_WAL_COMPACT:
162            readUpdateEntry(entry);
163            break;
164          case PROCEDURE_WAL_DELETE:
165            readDeleteEntry(entry);
166            break;
167          case PROCEDURE_WAL_EOF:
168            hasMore = false;
169            break;
170          default:
171            throw new CorruptedWALProcedureStoreException("Invalid entry: " + entry);
172        }
173      }
174    } catch (InvalidProtocolBufferException e) {
175      LOG.error("While reading procedure from " + log, e);
176      loader.markCorruptedWAL(log, e);
177    }
178
179    if (localTracker != null) {
180      localTracker.setPartialFlag(false);
181    }
182    if (!localProcedureMap.isEmpty()) {
183      log.setProcIds(localProcedureMap.getMinProcId(), localProcedureMap.getMaxProcId());
184      procedureMap.mergeTail(localProcedureMap);
185
186      //if (hasFastStartSupport) {
187      // TODO: Some procedure may be already runnables (see readInitEntry())
188      //       (we can also check the "update map" in the log trackers)
189      // --------------------------------------------------
190      //EntryIterator iter = procedureMap.fetchReady();
191      //if (iter != null) loader.load(iter);
192      // --------------------------------------------------
193      //}
194    }
195  }
196
197  public void finish() throws IOException {
198    // notify the loader about the max proc ID
199    loader.setMaxProcId(maxProcId);
200
201    // fetch the procedure ready to run.
202    ProcedureIterator procIter = procedureMap.fetchReady();
203    if (procIter != null) loader.load(procIter);
204
205    // remaining procedures have missing link or dependencies
206    // consider them as corrupted, manual fix is probably required.
207    procIter = procedureMap.fetchAll();
208    if (procIter != null) loader.handleCorrupted(procIter);
209  }

211  private void loadProcedure(final ProcedureWALEntry entry, final ProcedureProtos.Procedure proc) {
212    maxProcId = Math.max(maxProcId, proc.getProcId());
213    if (isRequired(proc.getProcId())) {
214      if (LOG.isTraceEnabled()) {
215        LOG.trace("Read " + entry.getType() + " entry " + proc.getProcId());
216      }
217      localProcedureMap.add(proc);
218      if (tracker.isPartial()) {
219        tracker.insert(proc.getProcId());
220      }
221    }
222    if (localTracker != null) {
223      localTracker.insert(proc.getProcId());
224    }
225  }
226
227  private void readInitEntry(final ProcedureWALEntry entry)
228  throws IOException {
229    assert entry.getProcedureCount() == 1 : "Expected only one procedure";
230    loadProcedure(entry, entry.getProcedure(0));
231  }
232
233  private void readInsertEntry(final ProcedureWALEntry entry) throws IOException {
234    assert entry.getProcedureCount() >= 1 : "Expected one or more procedures";
235    loadProcedure(entry, entry.getProcedure(0));
236    for (int i = 1; i < entry.getProcedureCount(); ++i) {
237      loadProcedure(entry, entry.getProcedure(i));
238    }
239  }
240
241  private void readUpdateEntry(final ProcedureWALEntry entry) throws IOException {
242    assert entry.getProcedureCount() == 1 : "Expected only one procedure";
243    loadProcedure(entry, entry.getProcedure(0));
244  }
245
246  private void readDeleteEntry(final ProcedureWALEntry entry) throws IOException {
247    assert entry.hasProcId() : "expected ProcID";
248
249    if (entry.getChildIdCount() > 0) {
250      assert entry.getProcedureCount() == 1 : "Expected only one procedure";
251
252      // update the parent procedure
253      loadProcedure(entry, entry.getProcedure(0));
254
255      // remove the child procedures of entry.getProcId()
256      for (int i = 0, count = entry.getChildIdCount(); i < count; ++i) {
257        deleteEntry(entry.getChildId(i));
258      }
259    } else {
260      assert entry.getProcedureCount() == 0 : "Expected no procedures";
261
262      // delete the procedure
263      deleteEntry(entry.getProcId());
264    }
265  }
266
267  private void deleteEntry(final long procId) {
268    if (LOG.isTraceEnabled()) {
269      LOG.trace("delete entry " + procId);
270    }
271    maxProcId = Math.max(maxProcId, procId);
272    localProcedureMap.remove(procId);
273    assert !procedureMap.contains(procId);
274    if (tracker.isPartial()) {
275      tracker.setDeleted(procId, true);
276    }
277    if (localTracker != null) {
278      // In case there is only delete entry for this procedure in current log.
279      localTracker.setDeleted(procId, true);
280    }
281  }
282
283  private boolean isDeleted(final long procId) {
284    return tracker.isDeleted(procId) == ProcedureStoreTracker.DeleteState.YES;
285  }
286
287  private boolean isRequired(final long procId) {
288    return !isDeleted(procId) && !procedureMap.contains(procId);
289  }

291  // ==========================================================================
292  //  We keep an in-memory map of the procedures sorted by replay order.
293  //  (see the details in the beginning of the file)
294  //                      _______________________________________________
295  //      procedureMap = | A |   | E |   | C |   |   |   |   | G |   |   |
296  //                       D               B
297  //      replayOrderHead = C <-> B <-> E <-> D <-> A <-> G
298  //
299  //  We also have a lazy grouping by "root procedure", and a list of
300  //  unlinked procedures. If after reading all the WALs we have unlinked
301  //  procedures it means that we had a missing WAL or a corruption.
302  //      rootHead = A <-> D <-> G
303  //                 B     E
304  //                 C
305  //      unlinkFromLinkList = None
306  // ==========================================================================
307  private static class Entry {
308    // For bucketed linked lists in hash-table.
309    protected Entry hashNext;
310    // child head
311    protected Entry childHead;
312    // double-link for rootHead or childHead
313    protected Entry linkNext;
314    protected Entry linkPrev;
315    // replay double-linked-list
316    protected Entry replayNext;
317    protected Entry replayPrev;
318    // procedure-infos
319    protected Procedure procedure;
320    protected ProcedureProtos.Procedure proto;
321    protected boolean ready = false;
322
323    public Entry(Entry hashNext) {
324      this.hashNext = hashNext;
325    }
326
327    public long getProcId() {
328      return proto.getProcId();
329    }
330
331    public long getParentId() {
332      return proto.getParentId();
333    }
334
335    public boolean hasParent() {
336      return proto.hasParentId();
337    }
338
339    public boolean isReady() {
340      return ready;
341    }
342
343    public boolean isFinished() {
344      if (!hasParent()) {
345        // we only consider 'root' procedures. because for the user 'finished'
346        // means when everything up to the 'root' is finished.
347        switch (proto.getState()) {
348          case ROLLEDBACK:
349          case SUCCESS:
350            return true;
351          default:
352            break;
353        }
354      }
355      return false;
356    }
357
358    public Procedure convert() throws IOException {
359      if (procedure == null) {
360        procedure = ProcedureUtil.convertToProcedure(proto);
361      }
362      return procedure;
363    }
364
365    public ProcedureInfo convertToInfo() {
366      return ProcedureUtil.convertToProcedureInfo(proto);
367    }
368
369    @Override
370    public String toString() {
371      final StringBuilder sb = new StringBuilder();
372      sb.append("Entry(");
373      sb.append(getProcId());
374      sb.append(", parentId=");
375      sb.append(getParentId());
376      sb.append(", class=");
377      sb.append(proto.getClassName());
378      sb.append(")");
379      return sb.toString();
380    }
381  }

400    }
401

410    }
411

416      } finally {

424        return current.convertToInfo();

428    }

431  private static class WalProcedureMap {
432    // procedure hash table
433    private Entry[] procedureMap;
434
435    // replay-order double-linked-list
436    private Entry replayOrderHead;
437    private Entry replayOrderTail;
438
439    // root linked-list
440    private Entry rootHead;
441
442    // pending unlinked children (root not present yet)
443    private Entry childUnlinkedHead;
444
445    // Track ProcId range
446    private long minProcId = Long.MAX_VALUE;
447    private long maxProcId = Long.MIN_VALUE;
448
449    public WalProcedureMap(int size) {
450      procedureMap = new Entry[size];
451      replayOrderHead = null;
452      replayOrderTail = null;
453      rootHead = null;
454      childUnlinkedHead = null;
455    }
456
457    public void add(ProcedureProtos.Procedure procProto) {
458      trackProcIds(procProto.getProcId());
459      Entry entry = addToMap(procProto.getProcId(), procProto.hasParentId());
460      boolean newEntry = entry.proto == null;
461      // We have seen procedure WALs where the entries are out of order; see HBASE-18152.
462      // To compensate, only replace the Entry procedure if for sure this new procedure
463      // is indeed an entry that came later. TODO: Fix the writing of procedure info so
464      // it does not violate basic expectation, that WALs contain procedure changes going
465      // from start to finish in sequence.
466      if (newEntry || isIncreasing(entry.proto, procProto)) {
467        entry.proto = procProto;
468      }
469      addToReplayList(entry);
470      if(newEntry) {
471        if (procProto.hasParentId()) {
472          childUnlinkedHead = addToLinkList(entry, childUnlinkedHead);
473        } else {
474          rootHead = addToLinkList(entry, rootHead);
475        }
476      }
477    }
478
479    /**
480     * @return True if this new procedure is 'richer' than the current one else
481     * false and we log this incidence where it appears that the WAL has older entries
482     * appended after newer ones. See HBASE-18152.
483     */
484    private static boolean isIncreasing(ProcedureProtos.Procedure current,
485        ProcedureProtos.Procedure candidate) {
486      boolean increasing = current.getStackIdCount() <= candidate.getStackIdCount() &&
487        current.getLastUpdate() <= candidate.getLastUpdate();
488      if (!increasing) {
489        LOG.warn("NOT INCREASING! current=" + current + ", candidate=" + candidate);
490      }
491      return increasing;
492    }
493
494    public boolean remove(long procId) {
495      trackProcIds(procId);
496      Entry entry = removeFromMap(procId);
497      if (entry != null) {
498        unlinkFromReplayList(entry);
499        unlinkFromLinkList(entry);
500        return true;
501      }
502      return false;
503    }
504
505    private void trackProcIds(long procId) {
506      minProcId = Math.min(minProcId, procId);
507      maxProcId = Math.max(maxProcId, procId);
508    }
509
510    public long getMinProcId() {
511      return minProcId;
512    }
513
514    public long getMaxProcId() {
515      return maxProcId;
516    }
517
518    public boolean contains(long procId) {
519      return getProcedure(procId) != null;
520    }
521
522    public boolean isEmpty() {
523      return replayOrderHead == null;
524    }
525
526    public void clear() {
527      for (int i = 0; i < procedureMap.length; ++i) {
528        procedureMap[i] = null;
529      }
530      replayOrderHead = null;
531      replayOrderTail = null;
532      rootHead = null;
533      childUnlinkedHead = null;
534      minProcId = Long.MAX_VALUE;
535      maxProcId = Long.MIN_VALUE;
536    }
537
538    /*
539     * Merges two WalProcedureMap,
540     * the target is the "global" map, the source is the "local" map.
541     *  - The entries in the hashtables are guaranteed to be unique.
542     *    On replay we don't load procedures that already exist in the "global"
543     *    map (the one we are merging the "local" in to).
544     *  - The replayOrderList of the "local" nao will be appended to the "global"
545     *    map replay list.
546     *  - The "local" map will be cleared at the end of the operation.
547     */
548    public void mergeTail(WalProcedureMap other) {
549      for (Entry p = other.replayOrderHead; p != null; p = p.replayNext) {
550        int slotIndex = getMapSlot(p.getProcId());
551        p.hashNext = procedureMap[slotIndex];
552        procedureMap[slotIndex] = p;
553      }
554
555      if (replayOrderHead == null) {
556        replayOrderHead = other.replayOrderHead;
557        replayOrderTail = other.replayOrderTail;
558        rootHead = other.rootHead;
559        childUnlinkedHead = other.childUnlinkedHead;
560      } else {
561        // append replay list
562        assert replayOrderTail.replayNext == null;
563        assert other.replayOrderHead.replayPrev == null;
564        replayOrderTail.replayNext = other.replayOrderHead;
565        other.replayOrderHead.replayPrev = replayOrderTail;
566        replayOrderTail = other.replayOrderTail;
567
568        // merge rootHead
569        if (rootHead == null) {
570          rootHead = other.rootHead;
571        } else if (other.rootHead != null) {
572          Entry otherTail = findLinkListTail(other.rootHead);
573          otherTail.linkNext = rootHead;
574          rootHead.linkPrev = otherTail;
575          rootHead = other.rootHead;
576        }
577
578        // merge childUnlinkedHead
579        if (childUnlinkedHead == null) {
580          childUnlinkedHead = other.childUnlinkedHead;
581        } else if (other.childUnlinkedHead != null) {
582          Entry otherTail = findLinkListTail(other.childUnlinkedHead);
583          otherTail.linkNext = childUnlinkedHead;
584          childUnlinkedHead.linkPrev = otherTail;
585          childUnlinkedHead = other.childUnlinkedHead;
586        }
587      }
588      maxProcId = Math.max(maxProcId, other.maxProcId);
589      minProcId = Math.max(minProcId, other.minProcId);
590
591      other.clear();
592    }

594    /*
595     * Returns an EntryIterator with the list of procedures ready
596     * to be added to the executor.
597     * A Procedure is ready if its children and parent are ready.
598     */
599    public EntryIterator fetchReady() {
600      buildGraph();
601
602      Entry readyHead = null;
603      Entry readyTail = null;
604      Entry p = replayOrderHead;
605      while (p != null) {
606        Entry next = p.replayNext;
607        if (p.isReady()) {
608          unlinkFromReplayList(p);
609          if (readyTail != null) {
610            readyTail.replayNext = p;
611            p.replayPrev = readyTail;
612          } else {
613            p.replayPrev = null;
614            readyHead = p;
615          }
616          readyTail = p;
617          p.replayNext = null;
618        }
619        p = next;
620      }
621      // we need the hash-table lookups for parents, so this must be done
622      // out of the loop where we check isReadyToRun()
623      for (p = readyHead; p != null; p = p.replayNext) {
624        removeFromMap(p.getProcId());
625        unlinkFromLinkList(p);
626      }
627      return readyHead != null ? new EntryIterator(readyHead) : null;
628    }
629
630    /*
631     * Drain this map and return all procedures in it.
632     */
633    public EntryIterator fetchAll() {
634      Entry head = replayOrderHead;
635      for (Entry p = head; p != null; p = p.replayNext) {
636        removeFromMap(p.getProcId());
637      }
638      for (int i = 0; i < procedureMap.length; ++i) {
639        assert procedureMap[i] == null : "map not empty i=" + i;
640      }
641      replayOrderHead = null;
642      replayOrderTail = null;
643      childUnlinkedHead = null;
644      rootHead = null;
645      return head != null ? new EntryIterator(head) : null;
646    }

648    private void buildGraph() {
649      Entry p = childUnlinkedHead;
650      while (p != null) {
651        Entry next = p.linkNext;
652        Entry rootProc = getRootProcedure(p);
653        if (rootProc != null) {
654          rootProc.childHead = addToLinkList(p, rootProc.childHead);
655        }
656        p = next;
657      }
658
659      for (p = rootHead; p != null; p = p.linkNext) {
660        checkReadyToRun(p);
661      }
662    }

664    private Entry getRootProcedure(Entry entry) {
665      while (entry != null && entry.hasParent()) {
666        entry = getProcedure(entry.getParentId());
667      }
668      return entry;
669    }

671    /*
672     * (see the comprehensive explanation in the beginning of the file)
673     * A Procedure is ready when parent and children are ready.
674     * "ready" means that we all the information that we need in-memory.
675     *
676     * Example-1:
677     * We have two WALs, we start reading from the newest (wal-2)
678     *    wal-2 | C B |
679     *    wal-1 | A B C |
680     *
681     * If C and B don't depend on A (A is not the parent), we can start them
682     * before reading wal-1. If B is the only one with parent A we can start C.
683     * We have to read one more WAL before being able to start B.
684     *
685     * How do we know with the only information in B that we are not ready.
686     *  - easy case, the parent is missing from the global map
687     *  - more complex case we look at the Stack IDs.
688     *
689     * The Stack-IDs are added to the procedure order as an incremental index
690     * tracking how many times that procedure was executed, which is equivalent
691     * to the number of times we wrote the procedure to the WAL.
692     * In the example above:
693     *   wal-2: B has stackId = [1, 2]
694     *   wal-1: B has stackId = [1]
695     *   wal-1: A has stackId = [0]
696     *
697     * Since we know that the Stack-IDs are incremental for a Procedure,
698     * we notice that there is a gap in the stackIds of B, so something was
699     * executed before.
700     * To identify when a Procedure is ready we do the sum of the stackIds of
701     * the procedure and the parent. if the stackIdSum is equal to the
702     * sum of {1..maxStackId} then everything we need is available.
703     *
704     * Example-2
705     *    wal-2 | A |              A stackIds = [0, 2]
706     *    wal-1 | A B |            B stackIds = [1]
707     *
708     * There is a gap between A stackIds so something was executed in between.
709     */
710    private boolean checkReadyToRun(Entry rootEntry) {
711      assert !rootEntry.hasParent() : "expected root procedure, got " + rootEntry;
712
713      if (rootEntry.isFinished()) {
714        // If the root procedure is finished, sub-procedures should be gone
715        if (rootEntry.childHead != null) {
716          LOG.error("unexpected active children for root-procedure: " + rootEntry);
717          for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
718            LOG.error("unexpected active children: " + p);
719          }
720        }
721
722        assert rootEntry.childHead == null : "unexpected children on root completion. " + rootEntry;
723        rootEntry.ready = true;
724        return true;
725      }
726
727      int stackIdSum = 0;
728      int maxStackId = 0;
729      for (int i = 0; i < rootEntry.proto.getStackIdCount(); ++i) {
730        int stackId = 1 + rootEntry.proto.getStackId(i);
731        maxStackId  = Math.max(maxStackId, stackId);
732        stackIdSum += stackId;
733        if (LOG.isTraceEnabled()) {
734          LOG.trace("stackId=" + stackId + " stackIdSum=" + stackIdSum +
735          " maxStackid=" + maxStackId + " " + rootEntry);
736        }
737      }
738
739      for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
740        for (int i = 0; i < p.proto.getStackIdCount(); ++i) {
741          int stackId = 1 + p.proto.getStackId(i);
742          maxStackId  = Math.max(maxStackId, stackId);
743          stackIdSum += stackId;
744          if (LOG.isTraceEnabled()) {
745            LOG.trace("stackId=" + stackId + " stackIdSum=" + stackIdSum +
746              " maxStackid=" + maxStackId + " " + p);
747          }
748        }
749      }
750      // The cmpStackIdSum is this formula for finding the sum of a series of numbers:
751      // http://www.wikihow.com/Sum-the-Integers-from-1-to-N#/Image:Sum-the-Integers-from-1-to-N-Step-2-Version-3.jpg
752      final int cmpStackIdSum = (maxStackId * (maxStackId + 1) / 2);
753      if (cmpStackIdSum == stackIdSum) {
754        rootEntry.ready = true;
755        for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
756          p.ready = true;
757        }
758        return true;
759      }
760      return false;
761    }

763    private void unlinkFromReplayList(Entry entry) {
764      if (replayOrderHead == entry) {
765        replayOrderHead = entry.replayNext;
766      }
767      if (replayOrderTail == entry) {
768        replayOrderTail = entry.replayPrev;
769      }
770      if (entry.replayPrev != null) {
771        entry.replayPrev.replayNext = entry.replayNext;
772      }
773      if (entry.replayNext != null) {
774        entry.replayNext.replayPrev = entry.replayPrev;
775      }
776    }

778    private void addToReplayList(final Entry entry) {
779      unlinkFromReplayList(entry);
780      entry.replayNext = replayOrderHead;
781      entry.replayPrev = null;
782      if (replayOrderHead != null) {
783        replayOrderHead.replayPrev = entry;
784      } else {
785        replayOrderTail = entry;
786      }
787      replayOrderHead = entry;
788    }

790    private void unlinkFromLinkList(Entry entry) {
791      if (entry == rootHead) {
792        rootHead = entry.linkNext;
793      } else if (entry == childUnlinkedHead) {
794        childUnlinkedHead = entry.linkNext;
795      }
796      if (entry.linkPrev != null) {
797        entry.linkPrev.linkNext = entry.linkNext;
798      }
799      if (entry.linkNext != null) {
800        entry.linkNext.linkPrev = entry.linkPrev;
801      }
802    }
803
804    private Entry addToLinkList(Entry entry, Entry linkHead) {
805      unlinkFromLinkList(entry);
806      entry.linkNext = linkHead;
807      entry.linkPrev = null;
808      if (linkHead != null) {
809        linkHead.linkPrev = entry;
810      }
811      return entry;
812    }

814    private Entry findLinkListTail(Entry linkHead) {
815      Entry tail = linkHead;
816      while (tail.linkNext != null) {
817        tail = tail.linkNext;
818      }
819      return tail;
820    }

822    private Entry addToMap(final long procId, final boolean hasParent) {
823      int slotIndex = getMapSlot(procId);
824      Entry entry = getProcedure(slotIndex, procId);
825      if (entry != null) return entry;
826
827      entry = new Entry(procedureMap[slotIndex]);
828      procedureMap[slotIndex] = entry;
829      return entry;
830    }

832    private Entry removeFromMap(final long procId) {
833      int slotIndex = getMapSlot(procId);
834      Entry prev = null;
835      Entry entry = procedureMap[slotIndex];
836      while (entry != null) {
837        if (procId == entry.getProcId()) {
838          if (prev != null) {
839            prev.hashNext = entry.hashNext;
840          } else {
841            procedureMap[slotIndex] = entry.hashNext;
842          }
843          entry.hashNext = null;
844          return entry;
845        }
846        prev = entry;
847        entry = entry.hashNext;
848      }
849      return null;
850    }

852    private Entry getProcedure(final long procId) {
853      return getProcedure(getMapSlot(procId), procId);
854    }
855
856    private Entry getProcedure(final int slotIndex, final long procId) {
857      Entry entry = procedureMap[slotIndex];
858      while (entry != null) {
859        if (procId == entry.getProcId()) {
860          return entry;
861        }
862        entry = entry.hashNext;
863      }
864      return null;
865    }
866
867    private int getMapSlot(final long procId) {
868      return (int)(Procedure.getProcIdHashCode(procId) % procedureMap.length);
869    }
