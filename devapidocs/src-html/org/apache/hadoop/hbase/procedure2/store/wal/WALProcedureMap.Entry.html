029/**
030 * We keep an in-memory map of the procedures sorted by replay order. (see the details in the
031 * beginning of {@link ProcedureWALFormatReader}).
032 *
033 * <pre>
034 *      procedureMap = | A |   | E |   | C |   |   |   |   | G |   |   |
035 *                       D               B
036 *      replayOrderHead = C <-> B <-> E <-> D <-> A <-> G
037 *
038 *  We also have a lazy grouping by "root procedure", and a list of
039 *  unlinked procedures. If after reading all the WALs we have unlinked
040 *  procedures it means that we had a missing WAL or a corruption.
041 *      rootHead = A <-> D <-> G
042 *                 B     E
043 *                 C
044 *      unlinkFromLinkList = None
045 * </pre>
046 */
047class WALProcedureMap {
048
049  private static final Logger LOG = LoggerFactory.getLogger(WALProcedureMap.class);

051  private static class Entry {
052    // For bucketed linked lists in hash-table.
053    private Entry hashNext;
054    // child head
055    private Entry childHead;
056    // double-link for rootHead or childHead
057    private Entry linkNext;
058    private Entry linkPrev;
059    // replay double-linked-list
060    private Entry replayNext;
061    private Entry replayPrev;
062    // procedure-infos
063    private Procedure<?> procedure;
064    private ProcedureProtos.Procedure proto;
065    private boolean ready = false;
066
067    public Entry(Entry hashNext) {
068      this.hashNext = hashNext;
069    }
070
071    public long getProcId() {
072      return proto.getProcId();
073    }
074
075    public long getParentId() {
076      return proto.getParentId();
077    }
078
079    public boolean hasParent() {
080      return proto.hasParentId();
081    }
082
083    public boolean isReady() {
084      return ready;
085    }
086
087    public boolean isFinished() {
088      if (!hasParent()) {
089        // we only consider 'root' procedures. because for the user 'finished'
090        // means when everything up to the 'root' is finished.
091        switch (proto.getState()) {
092          case ROLLEDBACK:
093          case SUCCESS:
094            return true;
095          default:
096            break;
097        }
098      }
099      return false;
100    }
101
102    public Procedure<?> convert() throws IOException {
103      if (procedure == null) {
104        procedure = ProcedureUtil.convertToProcedure(proto);
105      }
106      return procedure;
107    }
108
109    @Override
110    public String toString() {
111      final StringBuilder sb = new StringBuilder();
112      sb.append("Entry(");
113      sb.append(getProcId());
114      sb.append(", parentId=");
115      sb.append(getParentId());
116      sb.append(", class=");
117      sb.append(proto.getClassName());
118      sb.append(")");
119      return sb.toString();
120    }
121  }

123  private static class EntryIterator implements ProcedureIterator {
124    private final Entry replayHead;
125    private Entry current;
126
127    public EntryIterator(Entry replayHead) {
128      this.replayHead = replayHead;
129      this.current = replayHead;
130    }
131
132    @Override
133    public void reset() {
134      this.current = replayHead;
135    }
136
137    @Override
138    public boolean hasNext() {
139      return current != null;
140    }
141
142    @Override
143    public boolean isNextFinished() {
144      return current != null && current.isFinished();
145    }
146
147    @Override
148    public void skipNext() {
149      current = current.replayNext;
150    }
151
152    @Override
153    public Procedure<?> next() throws IOException {
154      try {
155        return current.convert();
156      } finally {
157        current = current.replayNext;
158      }
159    }
160  }

162  // procedure hash table
163  private Entry[] procedureMap;
164
165  // replay-order double-linked-list
166  private Entry replayOrderHead;
167  private Entry replayOrderTail;
168
169  // root linked-list
170  private Entry rootHead;
171
172  // pending unlinked children (root not present yet)
173  private Entry childUnlinkedHead;

175  // Track ProcId range
176  private long minModifiedProcId = Long.MAX_VALUE;
177  private long maxModifiedProcId = Long.MIN_VALUE;
178
179  public WALProcedureMap(int size) {
180    procedureMap = new Entry[size];
181    replayOrderHead = null;
182    replayOrderTail = null;
183    rootHead = null;
184    childUnlinkedHead = null;
185  }

187  public void add(ProcedureProtos.Procedure procProto) {
188    trackProcIds(procProto.getProcId());
189    Entry entry = addToMap(procProto.getProcId(), procProto.hasParentId());
190    boolean newEntry = entry.proto == null;
191    // We have seen procedure WALs where the entries are out of order; see HBASE-18152.
192    // To compensate, only replace the Entry procedure if for sure this new procedure
193    // is indeed an entry that came later.
194    // TODO: Fix the writing of procedure info so it does not violate basic expectation, that WALs
195    // contain procedure changes goingfrom start to finish in sequence.
196    if (newEntry || isIncreasing(entry.proto, procProto)) {
197      entry.proto = procProto;
198    }
199    addToReplayList(entry);
200    if (newEntry) {
201      if (procProto.hasParentId()) {
202        childUnlinkedHead = addToLinkList(entry, childUnlinkedHead);
203      } else {
204        rootHead = addToLinkList(entry, rootHead);
205      }
206    }
207  }

209  /**
210   * @return True if this new procedure is 'richer' than the current one else false and we log this
211   *         incidence where it appears that the WAL has older entries appended after newer ones.
212   *         See HBASE-18152.
213   */
214  private static boolean isIncreasing(ProcedureProtos.Procedure current,
215      ProcedureProtos.Procedure candidate) {
216    // Check that the procedures we see are 'increasing'. We used to compare
217    // procedure id first and then update time but it can legitimately go backwards if the
218    // procedure is failed or rolled back so that was unreliable. Was going to compare
219    // state but lets see if comparing update time enough (unfortunately this issue only
220    // seen under load...)
221    boolean increasing = current.getLastUpdate() <= candidate.getLastUpdate();
222    if (!increasing) {
223      LOG.warn("NOT INCREASING! current=" + current + ", candidate=" + candidate);
224    }
225    return increasing;
226  }
227
228  public boolean remove(long procId) {
229    trackProcIds(procId);
230    Entry entry = removeFromMap(procId);
231    if (entry != null) {
232      unlinkFromReplayList(entry);
233      unlinkFromLinkList(entry);
234      return true;
235    }
236    return false;
237  }

239  private void trackProcIds(long procId) {
240    minModifiedProcId = Math.min(minModifiedProcId, procId);
241    maxModifiedProcId = Math.max(maxModifiedProcId, procId);
242  }

260  public void clear() {
261    for (int i = 0; i < procedureMap.length; ++i) {
262      procedureMap[i] = null;
263    }
264    replayOrderHead = null;
265    replayOrderTail = null;
266    rootHead = null;
267    childUnlinkedHead = null;
268    minModifiedProcId = Long.MAX_VALUE;
269    maxModifiedProcId = Long.MIN_VALUE;
270  }

272  /*
273   * Merges two WalProcedureMap, the target is the "global" map, the source is the "local" map. -
274   * The entries in the hashtables are guaranteed to be unique. On replay we don't load procedures
275   * that already exist in the "global" map (the one we are merging the "local" in to). - The
276   * replayOrderList of the "local" nao will be appended to the "global" map replay list. - The
277   * "local" map will be cleared at the end of the operation.
278   */
279  public void mergeTail(WALProcedureMap other) {
280    for (Entry p = other.replayOrderHead; p != null; p = p.replayNext) {
281      int slotIndex = getMapSlot(p.getProcId());
282      p.hashNext = procedureMap[slotIndex];
283      procedureMap[slotIndex] = p;
284    }
285
286    if (replayOrderHead == null) {
287      replayOrderHead = other.replayOrderHead;
288      replayOrderTail = other.replayOrderTail;
289      rootHead = other.rootHead;
290      childUnlinkedHead = other.childUnlinkedHead;
291    } else {
292      // append replay list
293      assert replayOrderTail.replayNext == null;
294      assert other.replayOrderHead.replayPrev == null;
295      replayOrderTail.replayNext = other.replayOrderHead;
296      other.replayOrderHead.replayPrev = replayOrderTail;
297      replayOrderTail = other.replayOrderTail;
298
299      // merge rootHead
300      if (rootHead == null) {
301        rootHead = other.rootHead;
302      } else if (other.rootHead != null) {
303        Entry otherTail = findLinkListTail(other.rootHead);
304        otherTail.linkNext = rootHead;
305        rootHead.linkPrev = otherTail;
306        rootHead = other.rootHead;
307      }
308
309      // merge childUnlinkedHead
310      if (childUnlinkedHead == null) {
311        childUnlinkedHead = other.childUnlinkedHead;
312      } else if (other.childUnlinkedHead != null) {
313        Entry otherTail = findLinkListTail(other.childUnlinkedHead);
314        otherTail.linkNext = childUnlinkedHead;
315        childUnlinkedHead.linkPrev = otherTail;
316        childUnlinkedHead = other.childUnlinkedHead;
317      }
318    }
319    maxModifiedProcId = Math.max(maxModifiedProcId, other.maxModifiedProcId);
320    minModifiedProcId = Math.max(minModifiedProcId, other.minModifiedProcId);
321
322    other.clear();
323  }

325  /**
326   * Returns an EntryIterator with the list of procedures ready to be added to the executor. A
327   * Procedure is ready if its children and parent are ready.
328   */
329  public ProcedureIterator fetchReady() {
330    buildGraph();
331
332    Entry readyHead = null;
333    Entry readyTail = null;
334    Entry p = replayOrderHead;
335    while (p != null) {
336      Entry next = p.replayNext;
337      if (p.isReady()) {
338        unlinkFromReplayList(p);
339        if (readyTail != null) {
340          readyTail.replayNext = p;
341          p.replayPrev = readyTail;
342        } else {
343          p.replayPrev = null;
344          readyHead = p;
345        }
346        readyTail = p;
347        p.replayNext = null;
348      }
349      p = next;
350    }
351    // we need the hash-table lookups for parents, so this must be done
352    // out of the loop where we check isReadyToRun()
353    for (p = readyHead; p != null; p = p.replayNext) {
354      removeFromMap(p.getProcId());
355      unlinkFromLinkList(p);
356    }
357    return readyHead != null ? new EntryIterator(readyHead) : null;
358  }
359
360  /**
361   * Drain this map and return all procedures in it.
362   */
363  public ProcedureIterator fetchAll() {
364    Entry head = replayOrderHead;
365    for (Entry p = head; p != null; p = p.replayNext) {
366      removeFromMap(p.getProcId());
367    }
368    for (int i = 0; i < procedureMap.length; ++i) {
369      assert procedureMap[i] == null : "map not empty i=" + i;
370    }
371    replayOrderHead = null;
372    replayOrderTail = null;
373    childUnlinkedHead = null;
374    rootHead = null;
375    return head != null ? new EntryIterator(head) : null;
376  }

378  private void buildGraph() {
379    Entry p = childUnlinkedHead;
380    while (p != null) {
381      Entry next = p.linkNext;
382      Entry rootProc = getRootProcedure(p);
383      if (rootProc != null) {
384        rootProc.childHead = addToLinkList(p, rootProc.childHead);
385      }
386      p = next;
387    }
388
389    for (p = rootHead; p != null; p = p.linkNext) {
390      checkReadyToRun(p);
391    }
392  }
393
394  private Entry getRootProcedure(Entry entry) {
395    while (entry != null && entry.hasParent()) {
396      entry = getProcedure(entry.getParentId());
397    }
398    return entry;
399  }

401  /**
402   * (see the comprehensive explanation in the beginning of {@link ProcedureWALFormatReader}). A
403   * Procedure is ready when parent and children are ready. "ready" means that we all the
404   * information that we need in-memory.
405   * <p/>
406   * Example-1:<br/>
407   * We have two WALs, we start reading from the newest (wal-2)
408   *
409   * <pre>
410   *    wal-2 | C B |
411   *    wal-1 | A B C |
412   * </pre>
413   *
414   * If C and B don't depend on A (A is not the parent), we can start them before reading wal-1. If
415   * B is the only one with parent A we can start C. We have to read one more WAL before being able
416   * to start B.
417   * <p/>
418   * How do we know with the only information in B that we are not ready.
419   * <ul>
420   * <li>easy case, the parent is missing from the global map</li>
421   * <li>more complex case we look at the Stack IDs.</li>
422   * </ul>
423   * The Stack-IDs are added to the procedure order as an incremental index tracking how many times
424   * that procedure was executed, which is equivalent to the number of times we wrote the procedure
425   * to the WAL. <br/>
426   * In the example above:
427   *
428   * <pre>
429   *   wal-2: B has stackId = [1, 2]
430   *   wal-1: B has stackId = [1]
431   *   wal-1: A has stackId = [0]
432   * </pre>
433   *
434   * Since we know that the Stack-IDs are incremental for a Procedure, we notice that there is a gap
435   * in the stackIds of B, so something was executed before.
436   * <p/>
437   * To identify when a Procedure is ready we do the sum of the stackIds of the procedure and the
438   * parent. if the stackIdSum is equal to the sum of {1..maxStackId} then everything we need is
439   * available.
440   * <p/>
441   * Example-2
442   *
443   * <pre>
444   *    wal-2 | A |              A stackIds = [0, 2]
445   *    wal-1 | A B |            B stackIds = [1]
446   * </pre>
447   *
448   * There is a gap between A stackIds so something was executed in between.
449   */
450  private boolean checkReadyToRun(Entry rootEntry) {
451    assert !rootEntry.hasParent() : "expected root procedure, got " + rootEntry;
452
453    if (rootEntry.isFinished()) {
454      // If the root procedure is finished, sub-procedures should be gone
455      if (rootEntry.childHead != null) {
456        LOG.error("unexpected active children for root-procedure: {}", rootEntry);
457        for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
458          LOG.error("unexpected active children: {}", p);
459        }
460      }
461
462      assert rootEntry.childHead == null : "unexpected children on root completion. " + rootEntry;
463      rootEntry.ready = true;
464      return true;
465    }
466
467    int stackIdSum = 0;
468    int maxStackId = 0;
469    for (int i = 0; i < rootEntry.proto.getStackIdCount(); ++i) {
470      int stackId = 1 + rootEntry.proto.getStackId(i);
471      maxStackId = Math.max(maxStackId, stackId);
472      stackIdSum += stackId;
473      LOG.trace("stackId={} stackIdSum={} maxStackid={} {}", stackId, stackIdSum, maxStackId,
474        rootEntry);
475    }
476
477    for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
478      for (int i = 0; i < p.proto.getStackIdCount(); ++i) {
479        int stackId = 1 + p.proto.getStackId(i);
480        maxStackId = Math.max(maxStackId, stackId);
481        stackIdSum += stackId;
482        LOG.trace("stackId={} stackIdSum={} maxStackid={} {}", stackId, stackIdSum, maxStackId, p);
483      }
484    }
485    // The cmpStackIdSum is this formula for finding the sum of a series of numbers:
486    // http://www.wikihow.com/Sum-the-Integers-from-1-to-N#/Image:Sum-the-Integers-from-1-to-N-Step-2-Version-3.jpg
487    final int cmpStackIdSum = (maxStackId * (maxStackId + 1) / 2);
488    if (cmpStackIdSum == stackIdSum) {
489      rootEntry.ready = true;
490      for (Entry p = rootEntry.childHead; p != null; p = p.linkNext) {
491        p.ready = true;
492      }
493      return true;
494    }
495    return false;
496  }

498  private void unlinkFromReplayList(Entry entry) {
499    if (replayOrderHead == entry) {
500      replayOrderHead = entry.replayNext;
501    }
502    if (replayOrderTail == entry) {
503      replayOrderTail = entry.replayPrev;
504    }
505    if (entry.replayPrev != null) {
506      entry.replayPrev.replayNext = entry.replayNext;
507    }
508    if (entry.replayNext != null) {
509      entry.replayNext.replayPrev = entry.replayPrev;
510    }
511  }
512
513  private void addToReplayList(final Entry entry) {
514    unlinkFromReplayList(entry);
515    entry.replayNext = replayOrderHead;
516    entry.replayPrev = null;
517    if (replayOrderHead != null) {
518      replayOrderHead.replayPrev = entry;
519    } else {
520      replayOrderTail = entry;
521    }
522    replayOrderHead = entry;
523  }
524
525  private void unlinkFromLinkList(Entry entry) {
526    if (entry == rootHead) {
527      rootHead = entry.linkNext;
528    } else if (entry == childUnlinkedHead) {
529      childUnlinkedHead = entry.linkNext;
530    }
531    if (entry.linkPrev != null) {
532      entry.linkPrev.linkNext = entry.linkNext;
533    }
534    if (entry.linkNext != null) {
535      entry.linkNext.linkPrev = entry.linkPrev;
536    }
537  }
538
539  private Entry addToLinkList(Entry entry, Entry linkHead) {
540    unlinkFromLinkList(entry);
541    entry.linkNext = linkHead;
542    entry.linkPrev = null;
543    if (linkHead != null) {
544      linkHead.linkPrev = entry;
545    }
546    return entry;
547  }
548
549  private Entry findLinkListTail(Entry linkHead) {
550    Entry tail = linkHead;
551    while (tail.linkNext != null) {
552      tail = tail.linkNext;
553    }
554    return tail;
555  }

557  private Entry addToMap(long procId, boolean hasParent) {
558    int slotIndex = getMapSlot(procId);
559    Entry entry = getProcedure(slotIndex, procId);
560    if (entry != null) {
561      return entry;
562    }
563
564    entry = new Entry(procedureMap[slotIndex]);
565    procedureMap[slotIndex] = entry;
566    return entry;
567  }
568
569  private Entry removeFromMap(final long procId) {
570    int slotIndex = getMapSlot(procId);
571    Entry prev = null;
572    Entry entry = procedureMap[slotIndex];
573    while (entry != null) {
574      if (procId == entry.getProcId()) {
575        if (prev != null) {
576          prev.hashNext = entry.hashNext;
577        } else {
578          procedureMap[slotIndex] = entry.hashNext;
579        }
580        entry.hashNext = null;
581        return entry;
582      }
583      prev = entry;
584      entry = entry.hashNext;
585    }
586    return null;
587  }
588
589  private Entry getProcedure(long procId) {
590    return getProcedure(getMapSlot(procId), procId);
591  }
592
593  private Entry getProcedure(int slotIndex, long procId) {
594    Entry entry = procedureMap[slotIndex];
595    while (entry != null) {
596      if (procId == entry.getProcId()) {
597        return entry;
598      }
599      entry = entry.hashNext;
600    }
601    return null;
602  }
603
604  private int getMapSlot(long procId) {
605    return (int) (Procedure.getProcIdHashCode(procId) % procedureMap.length);
606  }
