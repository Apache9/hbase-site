019package org.apache.hadoop.hbase.coprocessor;
020
021import org.apache.hadoop.hbase.CoprocessorEnvironment;
022import org.apache.hadoop.hbase.HBaseInterfaceAudience;
023import org.apache.hadoop.hbase.ServerName;
024import org.apache.hadoop.hbase.client.Connection;
025import org.apache.hadoop.hbase.metrics.MetricRegistry;
026import org.apache.hadoop.hbase.regionserver.OnlineRegions;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.yetus.audience.InterfaceStability;
029
030@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
031@InterfaceStability.Evolving
032public interface RegionServerCoprocessorEnvironment
033    extends CoprocessorEnvironment<RegionServerCoprocessor> {
034  /**
035   * @return Hosting Server's ServerName
036   */
037  ServerName getServerName();
038
039  /**
040   * @return Interface to Map of regions online on this RegionServer {@link #getServerName()}}.
041   */
042  OnlineRegions getOnlineRegions();
043
044  /**
045   * Be careful RPC'ing from a Coprocessor context.
046   * RPC's will fail, stall, retry, and/or crawl because the remote side is not online, is
047   * struggling or it is on the other side of a network partition. Any use of Connection from
048   * inside a Coprocessor must be able to handle all such hiccups.
049   *
050   * <p>Using a Connection to get at a local resource -- say a Region that is on the local
051   * Server or using Admin Interface from a Coprocessor hosted on the Master -- will result in a
052   * short-circuit of the RPC framework to make a direct invocation avoiding RPC (and
053   * protobuf marshalling/unmarshalling).
054   *
055   * @return The host's Connection to the Cluster.
056   */
057  Connection getConnection();
058
059  /**
060   * Returns a MetricRegistry that can be used to track metrics at the region server level.
061   *
062   * <p>See ExampleMasterObserverWithMetrics class in the hbase-examples modules for examples
063   * of how metrics can be instantiated and used.</p>
064   * @return A MetricRegistry for the coprocessor class to track and export metrics.
065   */
066  MetricRegistry getMetricRegistryForRegionServer();
067}
