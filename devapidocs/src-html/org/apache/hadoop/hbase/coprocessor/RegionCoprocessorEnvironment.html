020package org.apache.hadoop.hbase.coprocessor;
021
022import java.io.IOException;
023import java.util.concurrent.ConcurrentMap;
024
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.CoprocessorEnvironment;
027import org.apache.hadoop.hbase.HBaseInterfaceAudience;
028import org.apache.hadoop.hbase.ServerName;
029import org.apache.hadoop.hbase.client.Connection;
030import org.apache.hadoop.hbase.client.RegionInfo;
031import org.apache.hadoop.hbase.metrics.MetricRegistry;
032import org.apache.hadoop.hbase.regionserver.OnlineRegions;
033import org.apache.hadoop.hbase.regionserver.Region;
034import org.apache.yetus.audience.InterfaceAudience;
035import org.apache.yetus.audience.InterfaceStability;
036
037@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
038@InterfaceStability.Evolving
039public interface RegionCoprocessorEnvironment extends CoprocessorEnvironment<RegionCoprocessor> {
040  /** @return the region associated with this coprocessor */
041  Region getRegion();
042
043  /** @return region information for the region this coprocessor is running on */
044  RegionInfo getRegionInfo();
045
046  /**
047   * @return Interface to Map of regions online on this RegionServer {@link #getServerName()}}.
048   */
049  OnlineRegions getOnlineRegions();
050
051  /** @return shared data between all instances of this coprocessor */
052  ConcurrentMap<String, Object> getSharedData();
053
054  /**
055   * @return Hosting Server's ServerName
056   */
057  ServerName getServerName();
058
059  /**
060   * Be careful RPC'ing from a Coprocessor context.
061   * RPC's will fail, stall, retry, and/or crawl because the remote side is not online, is
062   * struggling or it is on the other side of a network partition. Any use of Connection from
063   * inside a Coprocessor must be able to handle all such hiccups.
064   *
065   * <p>Using a Connection to get at a local resource -- say a Region that is on the local
066   * Server or using Admin Interface from a Coprocessor hosted on the Master -- will result in a
067   * short-circuit of the RPC framework to make a direct invocation avoiding RPC.
068   *<p>
069   * Note: If you want to create Connection with your own Configuration and NOT use the RegionServer
070   * Connection (though its cache of locations will be warm, and its life-cycle is not the concern
071   * of the CP), see {@link #createConnection(Configuration)}.
072   * @return The host's Connection to the Cluster.
073   */
074  Connection getConnection();
075
076  /**
077   * Creates a cluster connection using the passed configuration.
078   * <p>Using this Connection to get at a local resource -- say a Region that is on the local
079   * Server or using Admin Interface from a Coprocessor hosted on the Master -- will result in a
080   * short-circuit of the RPC framework to make a direct invocation avoiding RPC.
081   * <p>
082   * Note: HBase will NOT cache/maintain this Connection. If Coprocessors need to cache and reuse
083   * this connection, it has to be done by Coprocessors. Also make sure to close it after use.
084   *
085   * @param conf configuration
086   * @return Connection created using the passed conf.
087   */
088  Connection createConnection(Configuration conf) throws IOException;
089
090  /**
091   * Returns a MetricRegistry that can be used to track metrics at the region server level. All
092   * metrics tracked at this level will be shared by all the coprocessor instances
093   * of the same class in the same region server process. Note that there will be one
094   * region coprocessor environment per region in the server, but all of these instances will share
095   * the same MetricRegistry. The metric instances (like Counter, Timer, etc) will also be shared
096   * among all of the region coprocessor instances.
097   *
098   * <p>See ExampleRegionObserverWithMetrics class in the hbase-examples modules to see examples of how
099   * metrics can be instantiated and used.</p>
100   * @return A MetricRegistry for the coprocessor class to track and export metrics.
101   */
102  // Note: we are not exposing getMetricRegistryForRegion(). per-region metrics are already costly
103  // so we do not want to allow coprocessors to export metrics at the region level. We can allow
104  // getMetricRegistryForTable() to allow coprocessors to track metrics per-table, per-regionserver.
105  MetricRegistry getMetricRegistryForRegionServer();
106}
