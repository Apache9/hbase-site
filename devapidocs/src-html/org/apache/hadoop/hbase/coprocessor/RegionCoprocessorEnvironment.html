020package org.apache.hadoop.hbase.coprocessor;
021
022import java.util.concurrent.ConcurrentMap;
023
024import org.apache.hadoop.hbase.CoprocessorEnvironment;
025import org.apache.hadoop.hbase.HBaseInterfaceAudience;
026import org.apache.hadoop.hbase.HRegionInfo;
027import org.apache.hadoop.hbase.classification.InterfaceAudience;
028import org.apache.hadoop.hbase.classification.InterfaceStability;
029import org.apache.hadoop.hbase.metrics.MetricRegistry;
030import org.apache.hadoop.hbase.regionserver.Region;
031import org.apache.hadoop.hbase.regionserver.RegionServerServices;
032
033@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
034@InterfaceStability.Evolving
035public interface RegionCoprocessorEnvironment extends CoprocessorEnvironment {
036  /** @return the region associated with this coprocessor */
037  Region getRegion();
038
039  /** @return region information for the region this coprocessor is running on */
040  HRegionInfo getRegionInfo();
041
042  /** @return reference to the region server services */
043  RegionServerServices getRegionServerServices();
044
045  /** @return shared data between all instances of this coprocessor */
046  ConcurrentMap<String, Object> getSharedData();
047
048  /**
049   * Returns a MetricRegistry that can be used to track metrics at the region server level. All
050   * metrics tracked at this level will be shared by all the coprocessor instances
051   * of the same class in the same region server process. Note that there will be one
052   * region coprocessor environment per region in the server, but all of these instances will share
053   * the same MetricRegistry. The metric instances (like Counter, Timer, etc) will also be shared
054   * among all of the region coprocessor instances.
055   *
056   * <p>See ExampleRegionObserverWithMetrics class in the hbase-examples modules to see examples of how
057   * metrics can be instantiated and used.</p>
058   * @return A MetricRegistry for the coprocessor class to track and export metrics.
059   */
060  // Note: we are not exposing getMetricRegistryForRegion(). per-region metrics are already costly
061  // so we do not want to allow coprocessors to export metrics at the region level. We can allow
062  // getMetricRegistryForTable() to allow coprocessors to track metrics per-table, per-regionserver.
063  MetricRegistry getMetricRegistryForRegionServer();
064
065
066}
