018package org.apache.hadoop.hbase.coprocessor;
019
020import java.util.Optional;
021
022import com.google.common.annotations.VisibleForTesting;
023import org.apache.hadoop.hbase.CoprocessorEnvironment;
024import org.apache.hadoop.hbase.HBaseInterfaceAudience;
025import org.apache.hadoop.hbase.ipc.RpcServer;
026import org.apache.hadoop.hbase.security.User;
027import org.apache.yetus.audience.InterfaceAudience;
028import org.apache.yetus.audience.InterfaceStability;
029
030/**
031 * Carries the execution state for a given invocation of an Observer coprocessor
032 * ({@link RegionObserver}, {@link MasterObserver}, or {@link WALObserver})
033 * method.  The same ObserverContext instance is passed sequentially to all loaded
034 * coprocessors for a given Observer method trigger, with the
035 * <code>CoprocessorEnvironment</code> reference swapped out for each
036 * coprocessor.
037 * @param <E> The {@link CoprocessorEnvironment} subclass applicable to the
038 *     revelant Observer interface.
039 */
040@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.COPROC)
041@InterfaceStability.Evolving
042public class ObserverContext<E extends CoprocessorEnvironment> {
043  private E env;
044  private boolean bypass;
045  private boolean complete;
046  private final User caller;
047
048  @InterfaceAudience.Private
049  public ObserverContext(User caller) {
050    this.caller = caller;
051  }
052
053  public E getEnvironment() {
054    return env;
055  }
056
057  @InterfaceAudience.Private
058  public void prepare(E env) {
059    this.env = env;
060  }
061
062  /**
063   * Call to indicate that the current coprocessor's return value should be
064   * used in place of the normal HBase obtained value.
065   */
066  public void bypass() {
067    bypass = true;
068  }
069
070  /**
071   * Call to indicate that additional coprocessors further down the execution
072   * chain do not need to be invoked.  Implies that this coprocessor's response
073   * is definitive.
074   */
075  public void complete() {
076    complete = true;
077  }
078
079  /**
080   * For use by the coprocessor framework.
081   * @return <code>true</code> if {@link ObserverContext#bypass()}
082   *     was called by one of the loaded coprocessors, <code>false</code> otherwise.
083   */
084  public boolean shouldBypass() {
085    boolean current = bypass;
086    bypass = false;
087    return current;
088  }
089
090  /**
091   * For use by the coprocessor framework.
092   * @return <code>true</code> if {@link ObserverContext#complete()}
093   *     was called by one of the loaded coprocessors, <code>false</code> otherwise.
094   */
095  public boolean shouldComplete() {
096    boolean current = complete;
097    complete = false;
098    return current;
099  }
100
101  /**
102   * Returns the active user for the coprocessor call. If an explicit {@code User} instance was
103   * provided to the constructor, that will be returned, otherwise if we are in the context of an
104   * RPC call, the remote user is used. May not be present if the execution is outside of an RPC
105   * context.
106   */
107  public Optional<User> getCaller() {
108    return Optional.ofNullable(caller);
109  }
110
111  /**
112   * Instantiates a new ObserverContext instance if the passed reference is <code>null</code> and
113   * sets the environment in the new or existing instance. This allows deferring the instantiation
114   * of a ObserverContext until it is actually needed.
115   * @param <E> The environment type for the context
116   * @param env The coprocessor environment to set
117   * @return An instance of <code>ObserverContext</code> with the environment set
118   */
119  @Deprecated
120  @InterfaceAudience.Private
121  @VisibleForTesting
122  // TODO: Remove this method, ObserverContext should not depend on RpcServer
123  public static <E extends CoprocessorEnvironment> ObserverContext<E> createAndPrepare(E env) {
124    ObserverContext<E> ctx = new ObserverContext<>(RpcServer.getRequestUser().orElse(null));
125    ctx.prepare(env);
126    return ctx;
127  }
128}
