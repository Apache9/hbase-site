018package org.apache.hadoop.hbase.client;
019
020import static org.apache.hadoop.hbase.client.ConnectionUtils.getPauseTime;
021import static org.apache.hadoop.hbase.client.ConnectionUtils.retries2Attempts;
022
023import io.netty.util.HashedWheelTimer;
024import io.netty.util.Timeout;
025import io.netty.util.TimerTask;
026
027import java.io.IOException;
028import java.lang.reflect.UndeclaredThrowableException;
029import java.util.ArrayList;
030import java.util.List;
031import java.util.concurrent.CompletableFuture;
032import java.util.concurrent.TimeUnit;
033import java.util.function.Consumer;
034import java.util.function.Supplier;
035
036import org.apache.commons.logging.Log;
037import org.apache.commons.logging.LogFactory;
038import org.apache.hadoop.hbase.DoNotRetryIOException;
039import org.apache.hadoop.hbase.HRegionLocation;
040import org.apache.hadoop.hbase.TableName;
041import org.apache.hadoop.hbase.classification.InterfaceAudience;
042import org.apache.hadoop.hbase.ipc.HBaseRpcController;
043import org.apache.hadoop.hbase.shaded.com.google.protobuf.ServiceException;
044import org.apache.hadoop.hbase.shaded.protobuf.generated.ClientProtos.ClientService;
045import org.apache.hadoop.hbase.util.Bytes;
046import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
047import org.apache.hadoop.ipc.RemoteException;
048
049/**
050 * Retry caller for a single request, such as get, put, delete, etc.
051 */
052@InterfaceAudience.Private
053class AsyncSingleRequestRpcRetryingCaller<T> {
054
055  private static final Log LOG = LogFactory.getLog(AsyncSingleRequestRpcRetryingCaller.class);
056
057  @FunctionalInterface
058  public interface Callable<T> {
059    CompletableFuture<T> call(HBaseRpcController controller, HRegionLocation loc,
060        ClientService.Interface stub);
061  }
062
063  private final HashedWheelTimer retryTimer;
064
065  private final AsyncConnectionImpl conn;
066
067  private final TableName tableName;
068
069  private final byte[] row;
070
071  private final Supplier<CompletableFuture<HRegionLocation>> locate;
072
073  private final Callable<T> callable;
074
075  private final long pauseNs;
076
077  private final int maxAttempts;
078
079  private final long operationTimeoutNs;
080
081  private final long rpcTimeoutNs;
082
083  private final int startLogErrorsCnt;
084
085  private final CompletableFuture<T> future;
086
087  private final HBaseRpcController controller;
088
089  private final List<RetriesExhaustedException.ThrowableWithExtraContext> exceptions;
090
091  private final long startNs;
092
093  public AsyncSingleRequestRpcRetryingCaller(HashedWheelTimer retryTimer, AsyncConnectionImpl conn,
094      TableName tableName, byte[] row, boolean locateToPreviousRegion, Callable<T> callable,
095      long pauseNs, int maxRetries, long operationTimeoutNs, long rpcTimeoutNs,
096      int startLogErrorsCnt) {
097    this.retryTimer = retryTimer;
098    this.conn = conn;
099    this.tableName = tableName;
100    this.row = row;
101    if (locateToPreviousRegion) {
102      this.locate = this::locatePrevious;
103    } else {
104      this.locate = this::locate;
105    }
106    this.callable = callable;
107    this.pauseNs = pauseNs;
108    this.maxAttempts = retries2Attempts(maxRetries);
109    this.operationTimeoutNs = operationTimeoutNs;
110    this.rpcTimeoutNs = rpcTimeoutNs;
111    this.startLogErrorsCnt = startLogErrorsCnt;
112    this.future = new CompletableFuture<>();
113    this.controller = conn.rpcControllerFactory.newController();
114    this.exceptions = new ArrayList<>();
115    this.startNs = System.nanoTime();
116  }
117
118  private int tries = 1;
119
120  private long elapsedMs() {
121    return TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startNs);
122  }
123
124  private static Throwable translateException(Throwable t) {
125    if (t instanceof UndeclaredThrowableException && t.getCause() != null) {
126      t = t.getCause();
127    }
128    if (t instanceof RemoteException) {
129      t = ((RemoteException) t).unwrapRemoteException();
130    }
131    if (t instanceof ServiceException && t.getCause() != null) {
132      t = translateException(t.getCause());
133    }
134    return t;
135  }
136
137  private void completeExceptionally() {
138    future.completeExceptionally(new RetriesExhaustedException(tries, exceptions));
139  }
140
141  private void onError(Throwable error, Supplier<String> errMsg,
142      Consumer<Throwable> updateCachedLocation) {
143    error = translateException(error);
144    if (tries > startLogErrorsCnt) {
145      LOG.warn(errMsg.get(), error);
146    }
147    RetriesExhaustedException.ThrowableWithExtraContext qt =
148        new RetriesExhaustedException.ThrowableWithExtraContext(error,
149            EnvironmentEdgeManager.currentTime(), "");
150    exceptions.add(qt);
151    if (error instanceof DoNotRetryIOException || tries >= maxAttempts) {
152      completeExceptionally();
153      return;
154    }
155    long delayNs;
156    if (operationTimeoutNs > 0) {
157      long maxDelayNs = operationTimeoutNs - (System.nanoTime() - startNs);
158      if (maxDelayNs <= 0) {
159        completeExceptionally();
160        return;
161      }
162      delayNs = Math.min(maxDelayNs, getPauseTime(pauseNs, tries - 1));
163    } else {
164      delayNs = getPauseTime(pauseNs, tries - 1);
165    }
166    updateCachedLocation.accept(error);
167    tries++;
168    retryTimer.newTimeout(new TimerTask() {
169
170      @Override
171      public void run(Timeout timeout) throws Exception {
172        // always restart from beginning.
173        locateThenCall();
174      }
175    }, delayNs, TimeUnit.NANOSECONDS);
176  }
177
178  private void resetController() {
179    controller.reset();
180    if (rpcTimeoutNs >= 0) {
181      controller.setCallTimeout(
182        (int) Math.min(Integer.MAX_VALUE, TimeUnit.NANOSECONDS.toMillis(rpcTimeoutNs)));
183    }
184  }
185
186  private void call(HRegionLocation loc) {
187    ClientService.Interface stub;
188    try {
189      stub = conn.getRegionServerStub(loc.getServerName());
190    } catch (IOException e) {
191      onError(e,
192        () -> "Get async stub to " + loc.getServerName() + " for '" + Bytes.toStringBinary(row)
193            + "' in " + loc.getRegionInfo().getEncodedName() + " of " + tableName
194            + " failed, tries = " + tries + ", maxAttempts = " + maxAttempts + ", timeout = "
195            + TimeUnit.NANOSECONDS.toMillis(operationTimeoutNs) + " ms, time elapsed = "
196            + elapsedMs() + " ms",
197        err -> conn.getLocator().updateCachedLocation(loc, err));
198      return;
199    }
200    resetController();
201    callable.call(controller, loc, stub).whenComplete((result, error) -> {
202      if (error != null) {
203        onError(error,
204          () -> "Call to " + loc.getServerName() + " for '" + Bytes.toStringBinary(row) + "' in "
205              + loc.getRegionInfo().getEncodedName() + " of " + tableName + " failed, tries = "
206              + tries + ", maxAttempts = " + maxAttempts + ", timeout = "
207              + TimeUnit.NANOSECONDS.toMillis(operationTimeoutNs) + " ms, time elapsed = "
208              + elapsedMs() + " ms",
209          err -> conn.getLocator().updateCachedLocation(loc, err));
210        return;
211      }
212      future.complete(result);
213    });
214  }
215
216  private void locateThenCall() {
217    locate.get().whenComplete((loc, error) -> {
218      if (error != null) {
219        onError(error,
220          () -> "Locate '" + Bytes.toStringBinary(row) + "' in " + tableName + " failed, tries = "
221              + tries + ", maxAttempts = " + maxAttempts + ", timeout = "
222              + TimeUnit.NANOSECONDS.toMillis(operationTimeoutNs) + " ms, time elapsed = "
223              + elapsedMs() + " ms",
224          err -> {
225          });
226        return;
227      }
228      call(loc);
229    });
230  }
231
232  private CompletableFuture<HRegionLocation> locate() {
233    return conn.getLocator().getRegionLocation(tableName, row);
234  }
235
236  private CompletableFuture<HRegionLocation> locatePrevious() {
237    return conn.getLocator().getPreviousRegionLocation(tableName, row);
238  }
239
240  public CompletableFuture<T> call() {
241    locateThenCall();
242    return future;
243  }
244}
