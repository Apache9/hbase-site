382  private interface TableOperator {
383    CompletableFuture<Void> operate(TableName table);
384  }

386  private CompletableFuture<List<TableDescriptor>> batchTableOperations(Pattern pattern,
387      TableOperator operator, String operationType) {
388    CompletableFuture<List<TableDescriptor>> future = new CompletableFuture<>();
389    List<TableDescriptor> failed = new LinkedList<>();
390    listTables(Optional.ofNullable(pattern), false).whenComplete(
391      (tables, error) -> {
392        if (error != null) {
393          future.completeExceptionally(error);
394          return;
395        }
396        CompletableFuture[] futures =
397            tables.stream()
398                .map((table) -> operator.operate(table.getTableName()).whenComplete((v, ex) -> {
399                  if (ex != null) {
400                    LOG.info("Failed to " + operationType + " table " + table.getTableName(), ex);
401                    failed.add(table);
402                  }
403                })).<CompletableFuture> toArray(size -> new CompletableFuture[size]);
404        CompletableFuture.allOf(futures).thenAccept((v) -> {
405          future.complete(failed);
406        });
407      });
408    return future;
409  }

577  public CompletableFuture<Boolean> isTableAvailable(TableName tableName, byte[][] splitKeys) {
578    CompletableFuture<Boolean> future = new CompletableFuture<>();
579    isTableEnabled(tableName).whenComplete(
580      (enabled, error) -> {
581        if (error != null) {
582          future.completeExceptionally(error);
583          return;
584        }
585        if (!enabled) {
586          future.complete(false);
587        } else {
588          AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName))
589              .whenComplete(
590                (locations, error1) -> {
591                  if (error1 != null) {
592                    future.completeExceptionally(error1);
593                    return;
594                  }
595                  int notDeployed = 0;
596                  int regionCount = 0;
597                  for (HRegionLocation location : locations) {
598                    HRegionInfo info = location.getRegionInfo();
599                    if (location.getServerName() == null) {
600                      if (LOG.isDebugEnabled()) {
601                        LOG.debug("Table " + tableName + " has not deployed region "
602                            + info.getEncodedName());
603                      }
604                      notDeployed++;
605                    } else if (splitKeys != null
606                        && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
607                      for (byte[] splitKey : splitKeys) {
608                        // Just check if the splitkey is available
609                        if (Bytes.equals(info.getStartKey(), splitKey)) {
610                          regionCount++;
611                          break;
612                        }
613                      }
614                    } else {
615                      // Always empty start row should be counted
616                      regionCount++;
617                    }
618                  }
619                  if (notDeployed > 0) {
620                    if (LOG.isDebugEnabled()) {
621                      LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
622                    }
623                    future.complete(false);
624                  } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
625                    if (LOG.isDebugEnabled()) {
626                      LOG.debug("Table " + tableName + " expected to have "
627                          + (splitKeys.length + 1) + " regions, but only " + regionCount
628                          + " available");
629                    }
630                    future.complete(false);
631                  } else {
632                    if (LOG.isDebugEnabled()) {
633                      LOG.debug("Table " + tableName + " should be available");
634                    }
635                    future.complete(true);
636                  }
637                });
638        }
639      });
640    return future;
641  }

938   * Compact column family of a table, Asynchronous operation even if CompletableFuture.get()
939   */
940  private CompletableFuture<Void> compact(final TableName tableName, Optional<byte[]> columnFamily,
941      final boolean major, CompactType compactType) {
942    if (CompactType.MOB.equals(compactType)) {
943      // TODO support MOB compact.
944      return failedFuture(new UnsupportedOperationException("MOB compact does not support"));
945    }
946    CompletableFuture<Void> future = new CompletableFuture<>();
947    getTableHRegionLocations(tableName).whenComplete((locations, err) -> {
948      if (err != null) {
949        future.completeExceptionally(err);
950        return;
951      }
952      List<CompletableFuture<Void>> compactFutures = new ArrayList<>();
953      for (HRegionLocation location : locations) {
954        if (location.getRegionInfo() == null || location.getRegionInfo().isOffline()) continue;
955        if (location.getServerName() == null) continue;
956        compactFutures
957            .add(compact(location.getServerName(), location.getRegionInfo(), major, columnFamily));
958      }
959      // future complete unless all of the compact futures are completed.
960      CompletableFuture
961          .allOf(compactFutures.toArray(new CompletableFuture<?>[compactFutures.size()]))
962          .whenComplete((ret, err2) -> {
963            if (err2 != null) {
964              future.completeExceptionally(err2);
965            } else {
966              future.complete(ret);
967            }
968          });
969    });
970    return future;
971  }

998      CompletableFuture<TableName> result) {
999    getRegionLocation(encodeRegionName).whenComplete(
1000      (location, err) -> {
1001        if (err != null) {
1002          result.completeExceptionally(err);
1003          return;
1004        }
1005        HRegionInfo regionInfo = location.getRegionInfo();
1006        if (regionInfo.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID) {
1007          result.completeExceptionally(new IllegalArgumentException(
1008              "Can't invoke merge on non-default regions directly"));
1009          return;
1010        }
1011        if (!tableName.compareAndSet(null, regionInfo.getTable())) {
1012          if (!tableName.get().equals(regionInfo.getTable())) {
1013            // tables of this two region should be same.
1014            result.completeExceptionally(new IllegalArgumentException(
1015                "Cannot merge regions from two different tables " + tableName.get() + " and "
1016                    + regionInfo.getTable()));
1017          } else {
1018            result.complete(tableName.get());
1019          }
1020        }
1021      });

1121      if (error != null) {

1565    SnapshotProtos.SnapshotDescription snapshot = ProtobufUtil
1566        .createHBaseProtosSnapshotDesc(snapshotDesc);
1567    try {
1568      ClientSnapshotDescriptionUtils.assertSnapshotRequestIsValid(snapshot);
1569    } catch (IllegalArgumentException e) {

1578            resp -> resp.getExpectedTimeout())).call().whenComplete((expectedTimeout, err) -> {
1579          if (err != null) {

1581            return;
1582          }
1583          TimerTask pollingTask = new TimerTask() {
1584            int tries = 0;
1585            long startTime = EnvironmentEdgeManager.currentTime();
1586            long endTime = startTime + expectedTimeout;
1587            long maxPauseTime = expectedTimeout / maxAttempts;
1588
1589            @Override

1591              if (EnvironmentEdgeManager.currentTime() < endTime) {
1592                isSnapshotFinished(snapshotDesc).whenComplete((done, err2) -> {
1593                  if (err2 != null) {
1594                    future.completeExceptionally(err2);
1595                  } else if (done) {
1596                    future.complete(null);
1597                  } else {
1598                    // retry again after pauseTime.
1599                  long pauseTime = ConnectionUtils.getPauseTime(
1600                    TimeUnit.NANOSECONDS.toMillis(pauseNs), ++tries);
1601                  pauseTime = Math.min(pauseTime, maxPauseTime);
1602                  AsyncConnectionImpl.RETRY_TIMER
1603                      .newTimeout(this, pauseTime, TimeUnit.MILLISECONDS);
1604                }
1605              } );
1606              } else {
1607                future.completeExceptionally(new SnapshotCreationException("Snapshot '"
1608                    + snapshot.getName() + "' wasn't completed in expectedTime:" + expectedTimeout
1609                    + " ms", snapshotDesc));
1610              }
1611            }
1612          };
1613          AsyncConnectionImpl.RETRY_TIMER.newTimeout(pollingTask, 1, TimeUnit.MILLISECONDS);
1614        });

1643      (snapshotDescriptions, err) -> {

1645          future.completeExceptionally(err);
1646          return;
1647        }

1649        if (snapshotDescriptions != null && !snapshotDescriptions.isEmpty()) {
1650          for (SnapshotDescription snap : snapshotDescriptions) {
1651            if (snap.getName().equals(snapshotName)) {

1654            }
1655          }
1656        }

1665              if (err2 != null) {
1666                future.completeExceptionally(err2);

1668                // if table does not exist, then just clone snapshot into new table.

1673                (disabled, err4) -> {
1674                  if (err4 != null) {
1675                    future.completeExceptionally(err4);
1676                  } else if (!disabled) {
1677                    future.completeExceptionally(new TableNotDisabledException(finalTableName));
1678                  } else {
1679                    completeConditionalOnFuture(future,
1680                      restoreSnapshot(snapshotName, finalTableName, takeFailSafeSnapshot));

1682                });
1683            }
1684          } );
1685      });

1689  private CompletableFuture<Void> restoreSnapshot(String snapshotName, TableName tableName,
1690      boolean takeFailSafeSnapshot) {
1691    if (takeFailSafeSnapshot) {
1692      CompletableFuture<Void> future = new CompletableFuture<>();
1693      // Step.1 Take a snapshot of the current state
1694      String failSafeSnapshotSnapshotNameFormat = this.connection.getConfiguration().get(
1695        HConstants.SNAPSHOT_RESTORE_FAILSAFE_NAME,
1696        HConstants.DEFAULT_SNAPSHOT_RESTORE_FAILSAFE_NAME);
1697      final String failSafeSnapshotSnapshotName = failSafeSnapshotSnapshotNameFormat
1698          .replace("{snapshot.name}", snapshotName)
1699          .replace("{table.name}", tableName.toString().replace(TableName.NAMESPACE_DELIM, '.'))
1700          .replace("{restore.timestamp}", String.valueOf(EnvironmentEdgeManager.currentTime()));
1701      LOG.info("Taking restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1702      snapshot(failSafeSnapshotSnapshotName, tableName).whenComplete((ret, err) -> {
1703        if (err != null) {
1704          future.completeExceptionally(err);
1705        } else {
1706          // Step.2 Restore snapshot
1707        internalRestoreSnapshot(snapshotName, tableName).whenComplete((void2, err2) -> {
1708          if (err2 != null) {
1709            // Step.3.a Something went wrong during the restore and try to rollback.
1710          internalRestoreSnapshot(failSafeSnapshotSnapshotName, tableName).whenComplete(
1711            (void3, err3) -> {
1712              if (err3 != null) {
1713                future.completeExceptionally(err3);
1714              } else {
1715                String msg = "Restore snapshot=" + snapshotName + " failed. Rollback to snapshot="
1716                    + failSafeSnapshotSnapshotName + " succeeded.";
1717                future.completeExceptionally(new RestoreSnapshotException(msg));
1718              }
1719            });
1720        } else {
1721          // Step.3.b If the restore is succeeded, delete the pre-restore snapshot.
1722          LOG.info("Deleting restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1723          deleteSnapshot(failSafeSnapshotSnapshotName).whenComplete(
1724            (ret3, err3) -> {
1725              if (err3 != null) {
1726                LOG.error(
1727                  "Unable to remove the failsafe snapshot: " + failSafeSnapshotSnapshotName, err3);
1728                future.completeExceptionally(err3);
1729              } else {
1730                future.complete(ret3);
1731              }
1732            });
1733        }
1734      } );
1735      }
1736    } );
1737      return future;
1738    } else {
1739      return internalRestoreSnapshot(snapshotName, tableName);
1740    }
1741  }

1892  public CompletableFuture<Void> execProcedure(String signature, String instance,
1893      Map<String, String> props) {
1894    CompletableFuture<Void> future = new CompletableFuture<>();
1895    ProcedureDescription procDesc =
1896        ProtobufUtil.buildProcedureDescription(signature, instance, props);
1897    this.<Long> newMasterCaller()
1898        .action((controller, stub) -> this.<ExecProcedureRequest, ExecProcedureResponse, Long> call(
1899          controller, stub, ExecProcedureRequest.newBuilder().setProcedure(procDesc).build(),
1900          (s, c, req, done) -> s.execProcedure(c, req, done), resp -> resp.getExpectedTimeout()))
1901        .call().whenComplete((expectedTimeout, err) -> {
1902          if (err != null) {
1903            future.completeExceptionally(err);
1904            return;
1905          }
1906          TimerTask pollingTask = new TimerTask() {
1907            int tries = 0;
1908            long startTime = EnvironmentEdgeManager.currentTime();
1909            long endTime = startTime + expectedTimeout;
1910            long maxPauseTime = expectedTimeout / maxAttempts;
1911
1912            @Override
1913            public void run(Timeout timeout) throws Exception {
1914              if (EnvironmentEdgeManager.currentTime() < endTime) {
1915                isProcedureFinished(signature, instance, props).whenComplete((done, err2) -> {
1916                  if (err2 != null) {
1917                    future.completeExceptionally(err2);
1918                    return;
1919                  }
1920                  if (done) {
1921                    future.complete(null);
1922                  } else {
1923                    // retry again after pauseTime.
1924                    long pauseTime = ConnectionUtils
1925                        .getPauseTime(TimeUnit.NANOSECONDS.toMillis(pauseNs), ++tries);
1926                    pauseTime = Math.min(pauseTime, maxPauseTime);
1927                    AsyncConnectionImpl.RETRY_TIMER.newTimeout(this, pauseTime,
1928                      TimeUnit.MICROSECONDS);
1929                  }
1930                });
1931              } else {
1932                future.completeExceptionally(new IOException("Procedure '" + signature + " : "
1933                    + instance + "' wasn't completed in expectedTime:" + expectedTimeout + " ms"));
1934              }
1935            }
1936          };
1937          // Queue the polling task into RETRY_TIMER to poll procedure state asynchronously.
1938          AsyncConnectionImpl.RETRY_TIMER.newTimeout(pollingTask, 1, TimeUnit.MILLISECONDS);
1939        });
1940    return future;
1941  }

2588    getTableHRegionLocations(tableName).whenComplete(
2589      (locations, err) -> {
2590        if (err != null) {

2592          return;
2593        }
2594        List<CompactionState> regionStates = new ArrayList<>();
2595        List<CompletableFuture<CompactionState>> futures = new ArrayList<>();
2596        locations.stream().filter(loc -> loc.getServerName() != null)
2597            .filter(loc -> loc.getRegionInfo() != null)
2598            .filter(loc -> !loc.getRegionInfo().isOffline())
2599            .map(loc -> loc.getRegionInfo().getRegionName()).forEach(region -> {
2600              futures.add(getCompactionStateForRegion(region).whenComplete((regionState, err2) -> {
2601                // If any region compaction state is MAJOR_AND_MINOR
2602                // the table compaction state is MAJOR_AND_MINOR, too.
2603                if (err2 != null) {
2604                  future.completeExceptionally(err2);
2605                } else if (regionState == CompactionState.MAJOR_AND_MINOR) {
2606
2607                  future.complete(regionState);
2608                } else {
2609                  regionStates.add(regionState);
2610                }
2611              }));
2612            });
2613        CompletableFuture.allOf(futures.toArray(new CompletableFuture<?>[futures.size()]))
2614            .whenComplete((ret, err3) -> {
2615              // If future not completed, check all regions's compaction state
2616              if (!future.isCompletedExceptionally() && !future.isDone()) {
2617                CompactionState state = CompactionState.NONE;
2618                for (CompactionState regionState : regionStates) {
2619                  switch (regionState) {
2620                  case MAJOR:
2621                    if (state == CompactionState.MINOR) {
2622                      future.complete(CompactionState.MAJOR_AND_MINOR);
2623                    } else {
2624                      state = CompactionState.MAJOR;

2626                    break;
2627                  case MINOR:
2628                    if (state == CompactionState.MAJOR) {
2629                      future.complete(CompactionState.MAJOR_AND_MINOR);
2630                    } else {
2631                      state = CompactionState.MINOR;
2632                    }
2633                    break;
2634                  case NONE:
2635                  default:
2636                  }
2637                  if (!future.isDone()) {
2638                    future.complete(state);
2639                  }
2640                }
2641              }
2642            });
