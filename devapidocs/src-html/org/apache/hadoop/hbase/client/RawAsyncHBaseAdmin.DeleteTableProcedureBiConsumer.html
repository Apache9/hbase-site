259  private <PREQ, PRESP, RESP> CompletableFuture<RESP> call(HBaseRpcController controller,
260      MasterService.Interface stub, PREQ preq, MasterRpcCall<PRESP, PREQ> rpcCall,
261      Converter<RESP, PRESP> respConverter) {
262    CompletableFuture<RESP> future = new CompletableFuture<>();
263    rpcCall.call(stub, controller, preq, new RpcCallback<PRESP>() {
264
265      @Override
266      public void run(PRESP resp) {
267        if (controller.failed()) {
268          future.completeExceptionally(controller.getFailed());
269        } else {
270          try {
271            future.complete(respConverter.convert(resp));
272          } catch (IOException e) {
273            future.completeExceptionally(e);
274          }
275        }
276      }
277    });
278    return future;
279  }

320  private CompletableFuture<List<TableDescriptor>> batchTableOperations(Pattern pattern,
321      TableOperator operator, String operationType) {
322    CompletableFuture<List<TableDescriptor>> future = new CompletableFuture<>();
323    List<TableDescriptor> failed = new LinkedList<>();
324    listTables(Optional.ofNullable(pattern), false).whenComplete(
325      (tables, error) -> {
326        if (error != null) {
327          future.completeExceptionally(error);
328          return;
329        }
330        CompletableFuture[] futures =
331            tables.stream()
332                .map((table) -> operator.operate(table.getTableName()).whenComplete((v, ex) -> {
333                  if (ex != null) {
334                    LOG.info("Failed to " + operationType + " table " + table.getTableName(), ex);
335                    failed.add(table);
336                  }
337                })).<CompletableFuture> toArray(size -> new CompletableFuture[size]);
338        CompletableFuture.allOf(futures).thenAccept((v) -> {
339          future.complete(failed);
340        });
341      });
342    return future;
343  }

510  @Override
511  public CompletableFuture<Boolean> isTableAvailable(TableName tableName, byte[][] splitKeys) {
512    CompletableFuture<Boolean> future = new CompletableFuture<>();
513    isTableEnabled(tableName).whenComplete(
514      (enabled, error) -> {
515        if (error != null) {
516          future.completeExceptionally(error);
517          return;
518        }
519        if (!enabled) {
520          future.complete(false);
521        } else {
522          AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName))
523              .whenComplete(
524                (locations, error1) -> {
525                  if (error1 != null) {
526                    future.completeExceptionally(error1);
527                    return;
528                  }
529                  int notDeployed = 0;
530                  int regionCount = 0;
531                  for (HRegionLocation location : locations) {
532                    HRegionInfo info = location.getRegionInfo();
533                    if (location.getServerName() == null) {
534                      if (LOG.isDebugEnabled()) {
535                        LOG.debug("Table " + tableName + " has not deployed region "
536                            + info.getEncodedName());
537                      }
538                      notDeployed++;
539                    } else if (splitKeys != null
540                        && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
541                      for (byte[] splitKey : splitKeys) {
542                        // Just check if the splitkey is available
543                        if (Bytes.equals(info.getStartKey(), splitKey)) {
544                          regionCount++;
545                          break;
546                        }
547                      }
548                    } else {
549                      // Always empty start row should be counted
550                      regionCount++;
551                    }
552                  }
553                  if (notDeployed > 0) {
554                    if (LOG.isDebugEnabled()) {
555                      LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
556                    }
557                    future.complete(false);
558                  } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
559                    if (LOG.isDebugEnabled()) {
560                      LOG.debug("Table " + tableName + " expected to have "
561                          + (splitKeys.length + 1) + " regions, but only " + regionCount
562                          + " available");
563                    }
564                    future.complete(false);
565                  } else {
566                    if (LOG.isDebugEnabled()) {
567                      LOG.debug("Table " + tableName + " should be available");
568                    }
569                    future.complete(true);
570                  }
571                });
572        }
573      });
574    return future;
575  }

938          .allOf(compactFutures.toArray(new CompletableFuture<?>[compactFutures.size()]))
939          .whenComplete((ret, err2) -> {

1050  public CompletableFuture<Void> split(TableName tableName) {
1051    CompletableFuture<Void> future = new CompletableFuture<>();
1052    tableExists(tableName).whenComplete((exist, error) -> {
1053      if (error != null) {
1054        future.completeExceptionally(error);
1055        return;
1056      }
1057      if (!exist) {
1058        future.completeExceptionally(new TableNotFoundException(tableName));
1059        return;
1060      }
1061      metaTable
1062          .scanAll(new Scan().setReadType(ReadType.PREAD).addFamily(HConstants.CATALOG_FAMILY)
1063              .withStartRow(MetaTableAccessor.getTableStartRowForMeta(tableName, QueryType.REGION))
1064              .withStopRow(MetaTableAccessor.getTableStopRowForMeta(tableName, QueryType.REGION)))
1065          .whenComplete((results, err2) -> {
1066            if (err2 != null) {
1067              future.completeExceptionally(err2);
1068              return;
1069            }
1070            if (results != null && !results.isEmpty()) {
1071              List<CompletableFuture<Void>> splitFutures = new ArrayList<>();
1072              for (Result r : results) {
1073                if (r.isEmpty() || MetaTableAccessor.getHRegionInfo(r) == null) continue;
1074                RegionLocations rl = MetaTableAccessor.getRegionLocations(r);
1075                if (rl != null) {
1076                  for (HRegionLocation h : rl.getRegionLocations()) {
1077                    if (h != null && h.getServerName() != null) {
1078                      HRegionInfo hri = h.getRegionInfo();
1079                      if (hri == null || hri.isSplitParent()
1080                          || hri.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID)
1081                        continue;
1082                      splitFutures.add(split(h.getServerName(), hri, Optional.empty()));
1083                    }
1084                  }
1085                }
1086              }
1087              CompletableFuture
1088                  .allOf(splitFutures.toArray(new CompletableFuture<?>[splitFutures.size()]))
1089                  .whenComplete((ret, exception) -> {
1090                    if (exception != null) {
1091                      future.completeExceptionally(exception);
1092                      return;
1093                    }
1094                    future.complete(ret);
1095                  });
1096            } else {
1097              future.complete(null);
1098            }
1099          });
1100    });
1101    return future;
1102  }

1593                  } else {
1594                    completeConditionalOnFuture(future,
1595                      restoreSnapshot(snapshotName, finalTableName, takeFailSafeSnapshot));

1597                });
1598            }
1599          } );
1600      });

1604  private CompletableFuture<Void> restoreSnapshot(String snapshotName, TableName tableName,
1605      boolean takeFailSafeSnapshot) {
1606    if (takeFailSafeSnapshot) {
1607      CompletableFuture<Void> future = new CompletableFuture<>();
1608      // Step.1 Take a snapshot of the current state
1609      String failSafeSnapshotSnapshotNameFormat = this.connection.getConfiguration().get(
1610        HConstants.SNAPSHOT_RESTORE_FAILSAFE_NAME,
1611        HConstants.DEFAULT_SNAPSHOT_RESTORE_FAILSAFE_NAME);
1612      final String failSafeSnapshotSnapshotName = failSafeSnapshotSnapshotNameFormat
1613          .replace("{snapshot.name}", snapshotName)
1614          .replace("{table.name}", tableName.toString().replace(TableName.NAMESPACE_DELIM, '.'))
1615          .replace("{restore.timestamp}", String.valueOf(EnvironmentEdgeManager.currentTime()));
1616      LOG.info("Taking restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1617      snapshot(failSafeSnapshotSnapshotName, tableName).whenComplete((ret, err) -> {
1618        if (err != null) {
1619          future.completeExceptionally(err);
1620        } else {
1621          // Step.2 Restore snapshot
1622        internalRestoreSnapshot(snapshotName, tableName).whenComplete((void2, err2) -> {
1623          if (err2 != null) {
1624            // Step.3.a Something went wrong during the restore and try to rollback.
1625          internalRestoreSnapshot(failSafeSnapshotSnapshotName, tableName).whenComplete(
1626            (void3, err3) -> {
1627              if (err3 != null) {
1628                future.completeExceptionally(err3);
1629              } else {
1630                String msg = "Restore snapshot=" + snapshotName + " failed. Rollback to snapshot="
1631                    + failSafeSnapshotSnapshotName + " succeeded.";
1632                future.completeExceptionally(new RestoreSnapshotException(msg));
1633              }
1634            });
1635        } else {
1636          // Step.3.b If the restore is succeeded, delete the pre-restore snapshot.
1637          LOG.info("Deleting restore-failsafe snapshot: " + failSafeSnapshotSnapshotName);
1638          deleteSnapshot(failSafeSnapshotSnapshotName).whenComplete(
1639            (ret3, err3) -> {
1640              if (err3 != null) {
1641                LOG.error(
1642                  "Unable to remove the failsafe snapshot: " + failSafeSnapshotSnapshotName, err3);
1643                future.completeExceptionally(err3);
1644              } else {
1645                future.complete(ret3);
1646              }
1647            });
1648        }
1649      } );
1650      }
1651    } );
1652      return future;
1653    } else {
1654      return internalRestoreSnapshot(snapshotName, tableName);
1655    }
1656  }

1806  @Override
1807  public CompletableFuture<Void> execProcedure(String signature, String instance,
1808      Map<String, String> props) {
1809    CompletableFuture<Void> future = new CompletableFuture<>();
1810    ProcedureDescription procDesc =
1811        ProtobufUtil.buildProcedureDescription(signature, instance, props);
1812    this.<Long> newMasterCaller()
1813        .action((controller, stub) -> this.<ExecProcedureRequest, ExecProcedureResponse, Long> call(
1814          controller, stub, ExecProcedureRequest.newBuilder().setProcedure(procDesc).build(),
1815          (s, c, req, done) -> s.execProcedure(c, req, done), resp -> resp.getExpectedTimeout()))
1816        .call().whenComplete((expectedTimeout, err) -> {
1817          if (err != null) {
1818            future.completeExceptionally(err);
1819            return;
1820          }
1821          TimerTask pollingTask = new TimerTask() {
1822            int tries = 0;
1823            long startTime = EnvironmentEdgeManager.currentTime();
1824            long endTime = startTime + expectedTimeout;
1825            long maxPauseTime = expectedTimeout / maxAttempts;
1826
1827            @Override
1828            public void run(Timeout timeout) throws Exception {
1829              if (EnvironmentEdgeManager.currentTime() < endTime) {
1830                isProcedureFinished(signature, instance, props).whenComplete((done, err2) -> {
1831                  if (err2 != null) {
1832                    future.completeExceptionally(err2);
1833                    return;
1834                  }
1835                  if (done) {
1836                    future.complete(null);
1837                  } else {
1838                    // retry again after pauseTime.
1839                    long pauseTime = ConnectionUtils
1840                        .getPauseTime(TimeUnit.NANOSECONDS.toMillis(pauseNs), ++tries);
1841                    pauseTime = Math.min(pauseTime, maxPauseTime);
1842                    AsyncConnectionImpl.RETRY_TIMER.newTimeout(this, pauseTime,
1843                      TimeUnit.MICROSECONDS);
1844                  }
1845                });
1846              } else {
1847                future.completeExceptionally(new IOException("Procedure '" + signature + " : "
1848                    + instance + "' wasn't completed in expectedTime:" + expectedTimeout + " ms"));
1849              }
1850            }
1851          };
1852          // Queue the polling task into RETRY_TIMER to poll procedure state asynchronously.
1853          AsyncConnectionImpl.RETRY_TIMER.newTimeout(pollingTask, 1, TimeUnit.MILLISECONDS);
1854        });
1855    return future;
1856  }

1980  private byte[][] getSplitKeys(byte[] startKey, byte[] endKey, int numRegions) {
1981    if (numRegions < 3) {
1982      throw new IllegalArgumentException("Must create at least three regions");
1983    } else if (Bytes.compareTo(startKey, endKey) >= 0) {
1984      throw new IllegalArgumentException("Start key must be smaller than end key");
1985    }
1986    if (numRegions == 3) {
1987      return new byte[][] { startKey, endKey };
1988    }
1989    byte[][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
1990    if (splitKeys == null || splitKeys.length != numRegions - 1) {
1991      throw new IllegalArgumentException("Unable to split key range into enough regions");
1992    }
1993    return splitKeys;
1994  }

1996  private void verifySplitKeys(byte[][] splitKeys) {
1997    Arrays.sort(splitKeys, Bytes.BYTES_COMPARATOR);
1998    // Verify there are no duplicate split keys
1999    byte[] lastKey = null;
2000    for (byte[] splitKey : splitKeys) {
2001      if (Bytes.compareTo(splitKey, HConstants.EMPTY_BYTE_ARRAY) == 0) {
2002        throw new IllegalArgumentException("Empty split key must not be passed in the split keys.");
2003      }
2004      if (lastKey != null && Bytes.equals(splitKey, lastKey)) {
2005        throw new IllegalArgumentException("All split keys must be unique, " + "found duplicate: "
2006            + Bytes.toStringBinary(splitKey) + ", " + Bytes.toStringBinary(lastKey));
2007      }
2008      lastKey = splitKey;
2009    }
