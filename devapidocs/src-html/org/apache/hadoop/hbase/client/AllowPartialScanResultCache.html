018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.util.Arrays;
022
023import org.apache.hadoop.hbase.Cell;
024import org.apache.hadoop.hbase.classification.InterfaceAudience;
025
026/**
027 * A ScanResultCache that may return partial result.
028 * <p>
029 * As we can only scan from the starting of a row when error, so here we also implement the logic
030 * that skips the cells that have already been returned.
031 */
032@InterfaceAudience.Private
033class AllowPartialScanResultCache implements ScanResultCache {
034
035  // used to filter out the cells that already returned to user as we always start from the
036  // beginning of a row when retry.
037  private Cell lastCell;
038
039  private Result filterCells(Result result) {
040    return lastCell == null ? result : ConnectionUtils.filterCells(result, lastCell);
041  }
042
043  private void updateLastCell(Result result) {
044    lastCell = result.rawCells()[result.rawCells().length - 1];
045  }
046
047  @Override
048  public Result[] addAndGet(Result[] results, boolean isHeartbeatMessage) throws IOException {
049    if (results.length == 0) {
050      return EMPTY_RESULT_ARRAY;
051    }
052    Result first = filterCells(results[0]);
053    if (results.length == 1) {
054      if (first == null) {
055        // do not update last cell if we filter out all cells
056        return EMPTY_RESULT_ARRAY;
057      }
058      updateLastCell(results[0]);
059      results[0] = first;
060      return results;
061    }
062    updateLastCell(results[results.length - 1]);
063    if (first == null) {
064      return Arrays.copyOfRange(results, 1, results.length);
065    }
066    results[0] = first;
067    return results;
068  }
069
070  @Override
071  public void clear() {
072    // we do not cache anything
073  }
074}
