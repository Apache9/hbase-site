052/**
053 * This class  allows a continuous flow of requests. It's written to be compatible with a
054 * synchronous caller such as HTable.
055 * <p>
056 * The caller sends a buffer of operation, by calling submit. This class extract from this list
057 * the operations it can send, i.e. the operations that are on region that are not considered
058 * as busy. The process is asynchronous, i.e. it returns immediately when if has finished to
059 * iterate on the list. If, and only if, the maximum number of current task is reached, the call
060 * to submit will block. Alternatively, the caller can call submitAll, in which case all the
061 * operations will be sent. Each call to submit returns a future-like object that can be used
062 * to track operation progress.
063 * </p>
064 * <p>
065 * The class manages internally the retries.
066 * </p>
067 * <p>
068 * The class can be constructed in regular mode, or "global error" mode. In global error mode,
069 * AP tracks errors across all calls (each "future" also has global view of all errors). That
070 * mode is necessary for backward compat with HTable behavior, where multiple submissions are
071 * made and the errors can propagate using any put/flush call, from previous calls.
072 * In "regular" mode, the errors are tracked inside the Future object that is returned.
073 * The results are always tracked inside the Future object and can be retrieved when the call
074 * has finished. Partial results can also be retrieved if some part of multi-request failed.
075 * </p>
076 * <p>
077 * This class is thread safe in regular mode; in global error code, submitting operations and
078 * retrieving errors from different threads may be not thread safe.
079 * Internally, the class is thread safe enough to manage simultaneously new submission and results
080 * arising from older operations.
081 * </p>
082 * <p>
083 * Internally, this class works with {@link Row}, this mean it could be theoretically used for
084 * gets as well.
085 * </p>
086 */
087@InterfaceAudience.Private
088@InterfaceStability.Evolving
089class AsyncProcess {
090  private static final Log LOG = LogFactory.getLog(AsyncProcess.class);
091  private static final AtomicLong COUNTER = new AtomicLong();

167  public static final String LOG_DETAILS_PERIOD = "hbase.client.log.detail.period.ms";
168  private static final int DEFAULT_LOG_DETAILS_PERIOD = 10000;
169  private final int periodToLog;
170  AsyncProcess(ClusterConnection hc, Configuration conf,
171      RpcRetryingCallerFactory rpcCaller, boolean useGlobalErrors,
172      RpcControllerFactory rpcFactory) {
173    if (hc == null) {
174      throw new IllegalArgumentException("ClusterConnection cannot be null.");
175    }
176
177    this.connection = hc;
178    this.globalErrors = useGlobalErrors ? new BatchErrors() : null;
179
180    this.id = COUNTER.incrementAndGet();
181
182    this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
183        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
184    long configuredPauseForCQTBE = conf.getLong(HConstants.HBASE_CLIENT_PAUSE_FOR_CQTBE, pause);
185    if (configuredPauseForCQTBE < pause) {
186      LOG.warn("The " + HConstants.HBASE_CLIENT_PAUSE_FOR_CQTBE + " setting: "
187          + configuredPauseForCQTBE + " is smaller than " + HConstants.HBASE_CLIENT_PAUSE
188          + ", will use " + pause + " instead.");
189      this.pauseForCQTBE = pause;
190    } else {
191      this.pauseForCQTBE = configuredPauseForCQTBE;
192    }
193    // how many times we could try in total, one more than retry number
194    this.numTries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
195        HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER) + 1;
196    this.primaryCallTimeoutMicroseconds = conf.getInt(PRIMARY_CALL_TIMEOUT_KEY, 10000);
197    this.startLogErrorsCnt =
198        conf.getInt(START_LOG_ERRORS_AFTER_COUNT_KEY, DEFAULT_START_LOG_ERRORS_AFTER_COUNT);
199    this.periodToLog = conf.getInt(LOG_DETAILS_PERIOD, DEFAULT_LOG_DETAILS_PERIOD);
200    // Server tracker allows us to do faster, and yet useful (hopefully), retries.
201    // However, if we are too useful, we might fail very quickly due to retry count limit.
202    // To avoid this, we are going to cheat for now (see HBASE-7659), and calculate maximum
203    // retry time if normal retries were used. Then we will retry until this time runs out.
204    // If we keep hitting one server, the net effect will be the incremental backoff, and
205    // essentially the same number of retries as planned. If we have to do faster retries,
206    // we will do more retries in aggregate, but the user will be none the wiser.
207    this.serverTrackerTimeout = 0L;
208    for (int i = 0; i < this.numTries; ++i) {
209      serverTrackerTimeout = serverTrackerTimeout + ConnectionUtils.getPauseTime(this.pause, i);
210    }
211
212    this.rpcCallerFactory = rpcCaller;
213    this.rpcFactory = rpcFactory;
214    this.logBatchErrorDetails = conf.getBoolean(LOG_DETAILS_FOR_BATCH_ERROR, false);
215
216    this.requestController = RequestControllerFactory.create(conf);
217  }

219  /**
220   * The submitted task may be not accomplished at all if there are too many running tasks or
221   * other limits.
222   * @param <CResult> The class to cast the result
223   * @param task The setting and data
224   * @return AsyncRequestFuture
225   */
226  public <CResult> AsyncRequestFuture submit(AsyncProcessTask<CResult> task) throws InterruptedIOException {
227    AsyncRequestFuture reqFuture = checkTask(task);
228    if (reqFuture != null) {
229      return reqFuture;
230    }
231    SubmittedRows submittedRows = task.getSubmittedRows() == null ? SubmittedRows.ALL : task.getSubmittedRows();
232    switch (submittedRows) {
233      case ALL:
234        return submitAll(task);
235      case AT_LEAST_ONE:
236        return submit(task, true);
237      default:
238        return submit(task, false);
239    }
240  }

242  /**
243   * Extract from the rows list what we can submit. The rows we can not submit are kept in the
244   * list. Does not send requests to replicas (not currently used for anything other
245   * than streaming puts anyway).
246   *
247   * @param task The setting and data
248   * @param atLeastOne true if we should submit at least a subset.
249   */
250  private <CResult> AsyncRequestFuture submit(AsyncProcessTask<CResult> task,
251    boolean atLeastOne) throws InterruptedIOException {
252    TableName tableName = task.getTableName();
253    RowAccess<? extends Row> rows = task.getRowAccess();
254    Map<ServerName, MultiAction> actionsByServer = new HashMap<>();
255    List<Action> retainedActions = new ArrayList<>(rows.size());
256
257    NonceGenerator ng = this.connection.getNonceGenerator();
258    long nonceGroup = ng.getNonceGroup(); // Currently, nonce group is per entire client.
259
260    // Location errors that happen before we decide what requests to take.
261    List<Exception> locationErrors = null;
262    List<Integer> locationErrorRows = null;
263    RequestController.Checker checker = requestController.newChecker();
264    boolean firstIter = true;
265    do {
266      // Wait until there is at least one slot for a new task.
267      requestController.waitForFreeSlot(id, periodToLog, getLogger(tableName, -1));
268      int posInList = -1;
269      if (!firstIter) {
270        checker.reset();
271      }
272      Iterator<? extends Row> it = rows.iterator();
273      while (it.hasNext()) {
274        Row r = it.next();
275        HRegionLocation loc;
276        try {
277          if (r == null) {
278            throw new IllegalArgumentException("#" + id + ", row cannot be null");
279          }
280          // Make sure we get 0-s replica.
281          RegionLocations locs = connection.locateRegion(
282              tableName, r.getRow(), true, true, RegionReplicaUtil.DEFAULT_REPLICA_ID);
283          if (locs == null || locs.isEmpty() || locs.getDefaultRegionLocation() == null) {
284            throw new IOException("#" + id + ", no location found, aborting submit for"
285                + " tableName=" + tableName + " rowkey=" + Bytes.toStringBinary(r.getRow()));
286          }
287          loc = locs.getDefaultRegionLocation();
288        } catch (IOException ex) {
289          locationErrors = new ArrayList<>(1);
290          locationErrorRows = new ArrayList<>(1);
291          LOG.error("Failed to get region location ", ex);
292          // This action failed before creating ars. Retain it, but do not add to submit list.
293          // We will then add it to ars in an already-failed state.
294
295          int priority = HConstants.NORMAL_QOS;
296          if (r instanceof Mutation) {
297            priority = ((Mutation) r).getPriority();
298          }
299          retainedActions.add(new Action(r, ++posInList, priority));
300          locationErrors.add(ex);
301          locationErrorRows.add(posInList);
302          it.remove();
303          break; // Backward compat: we stop considering actions on location error.
304        }
305        ReturnCode code = checker.canTakeRow(loc, r);
306        if (code == ReturnCode.END) {
307          break;
308        }
309        if (code == ReturnCode.INCLUDE) {
310          int priority = HConstants.NORMAL_QOS;
311          if (r instanceof Mutation) {
312            priority = ((Mutation) r).getPriority();
313          }
314          Action action = new Action(r, ++posInList, priority);
315          setNonce(ng, r, action);
316          retainedActions.add(action);
317          // TODO: replica-get is not supported on this path
318          byte[] regionName = loc.getRegionInfo().getRegionName();
319          addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
320          it.remove();
321        }
322      }
323      firstIter = false;
324    } while (retainedActions.isEmpty() && atLeastOne && (locationErrors == null));
325
326    if (retainedActions.isEmpty()) return NO_REQS_RESULT;
327
328    return submitMultiActions(task, retainedActions, nonceGroup,
329        locationErrors, locationErrorRows, actionsByServer);
330  }
331
332  <CResult> AsyncRequestFuture submitMultiActions(AsyncProcessTask task,
333      List<Action> retainedActions, long nonceGroup, List<Exception> locationErrors,
334      List<Integer> locationErrorRows, Map<ServerName, MultiAction> actionsByServer) {
335    AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(task, retainedActions, nonceGroup);
336    // Add location errors if any
337    if (locationErrors != null) {
338      for (int i = 0; i < locationErrors.size(); ++i) {
339        int originalIndex = locationErrorRows.get(i);
340        Row row = retainedActions.get(originalIndex).getAction();
341        ars.manageError(originalIndex, row,
342            AsyncRequestFutureImpl.Retry.NO_LOCATION_PROBLEM, locationErrors.get(i), null);
343      }
344    }
345    ars.sendMultiAction(actionsByServer, 1, null, false);
346    return ars;
347  }

349  /**
350   * Helper that is used when grouping the actions per region server.
351   *
352   * @param server - server
353   * @param regionName - regionName
354   * @param action - the action to add to the multiaction
355   * @param actionsByServer the multiaction per server
356   * @param nonceGroup Nonce group.
357   */
358  static void addAction(ServerName server, byte[] regionName, Action action,
359      Map<ServerName, MultiAction> actionsByServer, long nonceGroup) {
360    MultiAction multiAction = actionsByServer.get(server);
361    if (multiAction == null) {
362      multiAction = new MultiAction();
363      actionsByServer.put(server, multiAction);
364    }
365    if (action.hasNonce() && !multiAction.hasNonceGroup()) {
366      multiAction.setNonceGroup(nonceGroup);
367    }
368
369    multiAction.add(regionName, action);
370  }

372  /**
373   * Submit immediately the list of rows, whatever the server status. Kept for backward
374   * compatibility: it allows to be used with the batch interface that return an array of objects.
375   * @param task The setting and data
376   */
377  private <CResult> AsyncRequestFuture submitAll(AsyncProcessTask task) {
378    RowAccess<? extends Row> rows = task.getRowAccess();
379    List<Action> actions = new ArrayList<>(rows.size());
380
381    // The position will be used by the processBatch to match the object array returned.
382    int posInList = -1;
383    NonceGenerator ng = this.connection.getNonceGenerator();
384    int highestPriority = HConstants.PRIORITY_UNSET;
385    for (Row r : rows) {
386      posInList++;
387      if (r instanceof Put) {
388        Put put = (Put) r;
389        if (put.isEmpty()) {
390          throw new IllegalArgumentException("No columns to insert for #" + (posInList+1)+ " item");
391        }
392        highestPriority = Math.max(put.getPriority(), highestPriority);
393      }
394      Action action = new Action(r, posInList, highestPriority);
395      setNonce(ng, r, action);
396      actions.add(action);
397    }
398    AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(task, actions, ng.getNonceGroup());
399    ars.groupAndSendMultiAction(actions, 1);
400    return ars;
401  }

403  private <CResult> AsyncRequestFuture checkTask(AsyncProcessTask<CResult> task) {
404    if (task.getRowAccess() == null || task.getRowAccess().isEmpty()) {
405      return NO_REQS_RESULT;
406    }
407    Objects.requireNonNull(task.getPool(), "The pool can't be NULL");
408    checkOperationTimeout(task.getOperationTimeout());
409    checkRpcTimeout(task.getRpcTimeout());
410    return null;
411  }
412
413  private void setNonce(NonceGenerator ng, Row r, Action action) {
414    if (!(r instanceof Append) && !(r instanceof Increment)) return;
415    action.setNonce(ng.newNonce()); // Action handles NO_NONCE, so it's ok if ng is disabled.
416  }

418  private int checkTimeout(String name, int timeout) {
419    if (timeout < 0) {
420      throw new RuntimeException("The " + name + " must be bigger than zero,"
421        + "current value is" + timeout);
422    }
423    return timeout;
424  }
425  private int checkOperationTimeout(int operationTimeout) {
426    return checkTimeout("operation timeout", operationTimeout);
427  }
428
429  private int checkRpcTimeout(int rpcTimeout) {
430    return checkTimeout("rpc timeout", rpcTimeout);
431  }

439  /** Wait until the async does not have more than max tasks in progress. */
440  protected void waitForMaximumCurrentTasks(int max, TableName tableName)
441      throws InterruptedIOException {
442    requestController.waitForMaximumCurrentTasks(max, id, periodToLog,
443      getLogger(tableName, max));
444  }
445
446  private Consumer<Long> getLogger(TableName tableName, long max) {
447    return (currentInProgress) -> {
448      LOG.info("#" + id + (max < 0 ? ", waiting for any free slot"
449      : ", waiting for some tasks to finish. Expected max="
450      + max) + ", tasksInProgress=" + currentInProgress +
451      " hasError=" + hasError() + (tableName == null ? "" : ", tableName=" + tableName));
452    };
453  }

463  /**
464   * Only used w/useGlobalErrors ctor argument, for HTable backward compat.
465   * @return Whether there were any errors in any request since the last time
466   *          {@link #waitForAllPreviousOpsAndReset(List, TableName)} was called, or AP was created.
467   */
468  public boolean hasError() {
469    return globalErrors != null && globalErrors.hasErrors();
470  }
471
472  /**
473   * Only used w/useGlobalErrors ctor argument, for HTable backward compat.
474   * Waits for all previous operations to finish, and returns errors and (optionally)
475   * failed operations themselves.
476   * @param failedRows an optional list into which the rows that failed since the last time
477   *        {@link #waitForAllPreviousOpsAndReset(List, TableName)} was called, or AP was created, are saved.
478   * @param tableName name of the table
479   * @return all the errors since the last time {@link #waitForAllPreviousOpsAndReset(List, TableName)}
480   *          was called, or AP was created.
481   */
482  public RetriesExhaustedWithDetailsException waitForAllPreviousOpsAndReset(
483      List<Row> failedRows, TableName tableName) throws InterruptedIOException {
484    waitForMaximumCurrentTasks(0, tableName);
485    if (globalErrors == null || !globalErrors.hasErrors()) {
486      return null;
487    }
488    if (failedRows != null) {
489      failedRows.addAll(globalErrors.actions);
490    }
491    RetriesExhaustedWithDetailsException result = globalErrors.makeException(logBatchErrorDetails);
492    globalErrors.clear();
493    return result;
494  }

508   * Currently, to preserve the main assumption about current retries, and to work well with
