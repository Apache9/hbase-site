019package org.apache.hadoop.hbase.client;
020
021import java.io.IOException;
022
023import org.apache.hadoop.hbase.HRegionInfo;
024import org.apache.hadoop.hbase.TableName;
025import org.apache.hadoop.hbase.classification.InterfaceAudience;
026import org.apache.hadoop.hbase.ipc.PayloadCarryingRpcController;
027import org.apache.hadoop.hbase.ipc.RpcControllerFactory;
028import org.apache.hadoop.hbase.protobuf.ProtobufUtil;
029import org.apache.hadoop.hbase.protobuf.RequestConverter;
030import org.apache.hadoop.hbase.protobuf.generated.AdminProtos.FlushRegionRequest;
031import org.apache.hadoop.hbase.protobuf.generated.AdminProtos.FlushRegionResponse;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
034import org.mortbay.log.Log;
035
036import com.google.protobuf.ServiceException;
037
038/**
039 * A Callable for flushRegion() RPC.
040 */
041@InterfaceAudience.Private
042public class FlushRegionCallable extends RegionAdminServiceCallable<FlushRegionResponse> {
043
044  private final byte[] regionName;
045  private final boolean writeFlushWalMarker;
046  private boolean reload;
047
048  public FlushRegionCallable(ClusterConnection connection,
049      RpcControllerFactory rpcControllerFactory, TableName tableName, byte[] regionName,
050      byte[] regionStartKey, boolean writeFlushWalMarker) {
051    super(connection, rpcControllerFactory, tableName, regionStartKey);
052    this.regionName = regionName;
053    this.writeFlushWalMarker = writeFlushWalMarker;
054  }
055
056  public FlushRegionCallable(ClusterConnection connection,
057      RpcControllerFactory rpcControllerFactory, HRegionInfo regionInfo,
058      boolean writeFlushWalMarker) {
059    this(connection, rpcControllerFactory, regionInfo.getTable(), regionInfo.getRegionName(),
060      regionInfo.getStartKey(), writeFlushWalMarker);
061  }
062
063  @Override
064  public FlushRegionResponse call(int callTimeout) throws Exception {
065    return flushRegion();
066  }
067
068  @Override
069  public void prepare(boolean reload) throws IOException {
070    super.prepare(reload);
071    this.reload = reload;
072  }
073
074  private FlushRegionResponse flushRegion() throws IOException {
075    // check whether we should still do the flush to this region. If the regions are changed due
076    // to splits or merges, etc return success
077    if (!Bytes.equals(location.getRegionInfo().getRegionName(), regionName)) {
078      if (!reload) {
079        throw new IOException("Cached location seems to be different than requested region.");
080      }
081      Log.info("Skipping flush region, because the located region "
082          + Bytes.toStringBinary(location.getRegionInfo().getRegionName()) + " is different than "
083          + " requested region " + Bytes.toStringBinary(regionName));
084      return FlushRegionResponse.newBuilder()
085          .setLastFlushTime(EnvironmentEdgeManager.currentTime())
086          .setFlushed(false)
087          .setWroteFlushWalMarker(false)
088          .build();
089    }
090
091    FlushRegionRequest request =
092        RequestConverter.buildFlushRegionRequest(regionName, writeFlushWalMarker);
093
094    try {
095      PayloadCarryingRpcController controller = rpcControllerFactory.newController();
096      controller.setPriority(tableName);
097      return stub.flushRegion(controller, request);
098    } catch (ServiceException se) {
099      throw ProtobufUtil.getRemoteException(se);
100    }
101  }
102}
