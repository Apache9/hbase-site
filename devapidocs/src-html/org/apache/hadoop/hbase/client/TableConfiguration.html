012package org.apache.hadoop.hbase.client;
013
014import org.apache.hadoop.conf.Configuration;
015import org.apache.hadoop.hbase.HConstants;
016import org.apache.hadoop.hbase.classification.InterfaceAudience;
017
018import com.google.common.annotations.VisibleForTesting;
019
020/**
021 *
022 * Configuration is a heavy weight registry that does a lot of string operations and regex matching.
023 * Method calls into Configuration account for high CPU usage and have huge performance impact.
024 * This class caches the value in the TableConfiguration object to improve performance.
025 * see HBASE-12128
026 *
027 */
028@InterfaceAudience.Private
029public class TableConfiguration {
030
031  public static final String WRITE_BUFFER_SIZE_KEY = "hbase.client.write.buffer";
032  public static final long WRITE_BUFFER_SIZE_DEFAULT = 2097152;
033  public static final String MAX_KEYVALUE_SIZE_KEY = "hbase.client.keyvalue.maxsize";
034  public static final int MAX_KEYVALUE_SIZE_DEFAULT = -1;
035
036  private final long writeBufferSize;
037  private final int metaOperationTimeout;
038  private final int operationTimeout;
039  private final int scannerCaching;
040  private final long scannerMaxResultSize;
041  private final int primaryCallTimeoutMicroSecond;
042  private final int replicaCallTimeoutMicroSecondScan;
043  private final int retries;
044  private final int maxKeyValueSize;
045
046    // toggle for async/sync prefetch
047  private final boolean clientScannerAsyncPrefetch;
048
049    /**
050   * Constructor
051   * @param conf Configuration object
052   */
053  TableConfiguration(Configuration conf) {
054    this.writeBufferSize = conf.getLong(WRITE_BUFFER_SIZE_KEY, WRITE_BUFFER_SIZE_DEFAULT);
055
056    this.metaOperationTimeout = conf.getInt(
057      HConstants.HBASE_CLIENT_META_OPERATION_TIMEOUT,
058      HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
059
060    this.operationTimeout = conf.getInt(
061      HConstants.HBASE_CLIENT_OPERATION_TIMEOUT, HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
062
063    this.scannerCaching = conf.getInt(
064      HConstants.HBASE_CLIENT_SCANNER_CACHING, HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING);
065
066    this.scannerMaxResultSize =
067        conf.getLong(HConstants.HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE_KEY,
068          HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE);
069
070    this.primaryCallTimeoutMicroSecond =
071        conf.getInt("hbase.client.primaryCallTimeout.get", 10000); // 10ms
072
073    this.replicaCallTimeoutMicroSecondScan =
074        conf.getInt("hbase.client.replicaCallTimeout.scan", 1000000); // 1000 ms
075
076    this.retries = conf.getInt(
077       HConstants.HBASE_CLIENT_RETRIES_NUMBER, HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
078
079    this.clientScannerAsyncPrefetch = conf.getBoolean(
080       Scan.HBASE_CLIENT_SCANNER_ASYNC_PREFETCH, Scan.DEFAULT_HBASE_CLIENT_SCANNER_ASYNC_PREFETCH);
081
082    this.maxKeyValueSize = conf.getInt(MAX_KEYVALUE_SIZE_KEY, MAX_KEYVALUE_SIZE_DEFAULT);
083  }
084
085  /**
086   * Constructor
087   * This is for internal testing purpose (using the default value).
088   * In real usage, we should read the configuration from the Configuration object.
089   */
090  @VisibleForTesting
091  protected TableConfiguration() {
092    this.writeBufferSize = WRITE_BUFFER_SIZE_DEFAULT;
093    this.metaOperationTimeout = HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT;
094    this.operationTimeout = HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT;
095    this.scannerCaching = HConstants.DEFAULT_HBASE_CLIENT_SCANNER_CACHING;
096    this.scannerMaxResultSize = HConstants.DEFAULT_HBASE_CLIENT_SCANNER_MAX_RESULT_SIZE;
097    this.primaryCallTimeoutMicroSecond = 10000;
098    this.replicaCallTimeoutMicroSecondScan = 1000000;
099    this.retries = HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER;
100    this.clientScannerAsyncPrefetch = Scan.DEFAULT_HBASE_CLIENT_SCANNER_ASYNC_PREFETCH;
101    this.maxKeyValueSize = MAX_KEYVALUE_SIZE_DEFAULT;
102  }
103
104  public long getWriteBufferSize() {
105    return writeBufferSize;
106  }
107
108  public int getMetaOperationTimeout() {
109    return metaOperationTimeout;
110  }
111
112  public int getOperationTimeout() {
113    return operationTimeout;
114  }
115
116  public int getScannerCaching() {
117    return scannerCaching;
118  }
119
120  public int getPrimaryCallTimeoutMicroSecond() {
121    return primaryCallTimeoutMicroSecond;
122  }
123
124  public int getReplicaCallTimeoutMicroSecondScan() {
125    return replicaCallTimeoutMicroSecondScan;
126  }
127
128  public int getRetriesNumber() {
129    return retries;
130  }
131
132  public int getMaxKeyValueSize() {
133    return maxKeyValueSize;
134  }
135
136  public long getScannerMaxResultSize() {
137    return scannerMaxResultSize;
138  }
139
140  public boolean isClientScannerAsyncPrefetch() {
141    return clientScannerAsyncPrefetch;
142  }
143
144}
