070/**
071 * This class  allows a continuous flow of requests. It's written to be compatible with a
072 * synchronous caller such as HTable.
073 * <p>
074 * The caller sends a buffer of operation, by calling submit. This class extract from this list
075 * the operations it can send, i.e. the operations that are on region that are not considered
076 * as busy. The process is asynchronous, i.e. it returns immediately when if has finished to
077 * iterate on the list. If, and only if, the maximum number of current task is reached, the call
078 * to submit will block. Alternatively, the caller can call submitAll, in which case all the
079 * operations will be sent. Each call to submit returns a future-like object that can be used
080 * to track operation progress.
081 * </p>
082 * <p>
083 * The class manages internally the retries.
084 * </p>
085 * <p>
086 * The class can be constructed in regular mode, or "global error" mode. In global error mode,
087 * AP tracks errors across all calls (each "future" also has global view of all errors). That
088 * mode is necessary for backward compat with HTable behavior, where multiple submissions are
089 * made and the errors can propagate using any put/flush call, from previous calls.
090 * In "regular" mode, the errors are tracked inside the Future object that is returned.
091 * The results are always tracked inside the Future object and can be retrieved when the call
092 * has finished. Partial results can also be retrieved if some part of multi-request failed.
093 * </p>
094 * <p>
095 * This class is thread safe in regular mode; in global error code, submitting operations and
096 * retrieving errors from different threads may be not thread safe.
097 * Internally, the class is thread safe enough to manage simultaneously new submission and results
098 * arising from older operations.
099 * </p>
100 * <p>
101 * Internally, this class works with {@link Row}, this mean it could be theoretically used for
102 * gets as well.
103 * </p>
104 */

269    private final List<Throwable> throwables = new ArrayList<Throwable>();
270    private final List<Row> actions = new ArrayList<Row>();
271    private final List<String> addresses = new ArrayList<String>();

273    public synchronized void add(Throwable ex, Row row, ServerName serverName) {
274      if (row == null){
275        throw new IllegalArgumentException("row cannot be null. location=" + serverName);
276      }
277
278      throwables.add(ex);
279      actions.add(row);
280      addresses.add(serverName != null ? serverName.toString() : "null");
281    }
282
283    public boolean hasErrors() {
284      return !throwables.isEmpty();
285    }
286
287    private synchronized RetriesExhaustedWithDetailsException makeException(boolean logDetails) {
288      if (logDetails) {
289        LOG.error("Exception occurred! Exception details: " + throwables + ";\nActions: "
290            + actions);
291      }
292      return new RetriesExhaustedWithDetailsException(new ArrayList<Throwable>(throwables),
293          new ArrayList<Row>(actions), new ArrayList<String>(addresses));
294    }
295
296    public synchronized void clear() {
297      throwables.clear();
298      actions.clear();
299      addresses.clear();
300    }
301
302    public synchronized void merge(BatchErrors other) {
303      throwables.addAll(other.throwables);
304      actions.addAll(other.actions);
305      addresses.addAll(other.addresses);
306    }

308  public AsyncProcess(ClusterConnection hc, Configuration conf, ExecutorService pool,
309      RpcRetryingCallerFactory rpcCaller, boolean useGlobalErrors,
310      RpcControllerFactory rpcFactory, int rpcTimeout) {
311    if (hc == null) {
312      throw new IllegalArgumentException("ClusterConnection cannot be null.");
313    }
314
315    this.connection = hc;
316    this.pool = pool;
317    this.globalErrors = useGlobalErrors ? new BatchErrors() : null;
318
319    this.id = COUNTER.incrementAndGet();
320
321    this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
322        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
323    // how many times we could try in total, one more than retry number
324    this.numTries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
325        HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER) + 1;
326    this.timeout = rpcTimeout;
327    this.primaryCallTimeoutMicroseconds = conf.getInt(PRIMARY_CALL_TIMEOUT_KEY, 10000);
328
329    this.maxTotalConcurrentTasks = conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
330      HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS);
331    this.maxConcurrentTasksPerServer = conf.getInt(HConstants.HBASE_CLIENT_MAX_PERSERVER_TASKS,
332          HConstants.DEFAULT_HBASE_CLIENT_MAX_PERSERVER_TASKS);
333    this.maxConcurrentTasksPerRegion = conf.getInt(HConstants.HBASE_CLIENT_MAX_PERREGION_TASKS,
334          HConstants.DEFAULT_HBASE_CLIENT_MAX_PERREGION_TASKS);
335    this.maxHeapSizePerRequest = conf.getLong(HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE,
336          DEFAULT_HBASE_CLIENT_MAX_PERREQUEST_HEAPSIZE);
337    this.maxHeapSizeSubmit = conf.getLong(HBASE_CLIENT_MAX_SUBMIT_HEAPSIZE, DEFAULT_HBASE_CLIENT_MAX_SUBMIT_HEAPSIZE);
338    this.startLogErrorsCnt =
339        conf.getInt(START_LOG_ERRORS_AFTER_COUNT_KEY, DEFAULT_START_LOG_ERRORS_AFTER_COUNT);
340
341    if (this.maxTotalConcurrentTasks <= 0) {
342      throw new IllegalArgumentException("maxTotalConcurrentTasks=" + maxTotalConcurrentTasks);
343    }
344    if (this.maxConcurrentTasksPerServer <= 0) {
345      throw new IllegalArgumentException("maxConcurrentTasksPerServer=" +
346          maxConcurrentTasksPerServer);
347    }
348    if (this.maxConcurrentTasksPerRegion <= 0) {
349      throw new IllegalArgumentException("maxConcurrentTasksPerRegion=" +
350          maxConcurrentTasksPerRegion);
351    }
352    if (this.maxHeapSizePerRequest <= 0) {
353      throw new IllegalArgumentException("maxHeapSizePerServer=" +
354          maxHeapSizePerRequest);
355    }
356
357    if (this.maxHeapSizeSubmit <= 0) {
358      throw new IllegalArgumentException("maxHeapSizeSubmit=" +
359          maxHeapSizeSubmit);
360    }
361    // Server tracker allows us to do faster, and yet useful (hopefully), retries.
362    // However, if we are too useful, we might fail very quickly due to retry count limit.
363    // To avoid this, we are going to cheat for now (see HBASE-7659), and calculate maximum
364    // retry time if normal retries were used. Then we will retry until this time runs out.
365    // If we keep hitting one server, the net effect will be the incremental backoff, and
366    // essentially the same number of retries as planned. If we have to do faster retries,
367    // we will do more retries in aggregate, but the user will be none the wiser.
368    this.serverTrackerTimeout = 0;
369    for (int i = 0; i < this.numTries; ++i) {
370      serverTrackerTimeout += ConnectionUtils.getPauseTime(this.pause, i);
371    }
372
373    this.rpcCallerFactory = rpcCaller;
374    this.rpcFactory = rpcFactory;
375    this.logBatchErrorDetails = conf.getBoolean(LOG_DETAILS_FOR_BATCH_ERROR, false);
376
377    this.thresholdToLogUndoneTaskDetails =
378        conf.getInt(THRESHOLD_TO_LOG_UNDONE_TASK_DETAILS,
379          DEFAULT_THRESHOLD_TO_LOG_UNDONE_TASK_DETAILS);
380  }

424  /**
425   * Extract from the rows list what we can submit. The rows we can not submit are kept in the
426   * list. Does not send requests to replicas (not currently used for anything other
427   * than streaming puts anyway).
428   *
429   * @param pool ExecutorService to use.
430   * @param tableName The table for which this request is needed.
431   * @param callback Batch callback. Only called on success (94 behavior).
432   * @param needResults Whether results are needed, or can be discarded.
433   * @param rows - the submitted row. Modified by the method: we remove the rows we took.
434   * @param atLeastOne true if we should submit at least a subset.
435   */
436  public <CResult> AsyncRequestFuture submit(ExecutorService pool, TableName tableName,
437      RowAccess<? extends Row> rows, boolean atLeastOne, Batch.Callback<CResult> callback,
438      boolean needResults) throws InterruptedIOException {
439    if (rows.isEmpty()) {
440      return NO_REQS_RESULT;
441    }
442
443    Map<ServerName, MultiAction<Row>> actionsByServer =
444        new HashMap<ServerName, MultiAction<Row>>();
445    List<Action<Row>> retainedActions = new ArrayList<Action<Row>>(rows.size());
446
447    NonceGenerator ng = this.connection.getNonceGenerator();
448    long nonceGroup = ng.getNonceGroup(); // Currently, nonce group is per entire client.
449
450    // Location errors that happen before we decide what requests to take.
451    List<Exception> locationErrors = null;
452    List<Integer> locationErrorRows = null;
453    RowCheckerHost checker = createRowCheckerHost();
454    boolean firstIter = true;
455    do {
456      // Wait until there is at least one slot for a new task.
457      waitForMaximumCurrentTasks(maxTotalConcurrentTasks - 1, tableName.getNameAsString());
458      int posInList = -1;
459      if (!firstIter) {
460        checker.reset();
461      }
462      Iterator<? extends Row> it = rows.iterator();
463      while (it.hasNext()) {
464        Row r = it.next();
465        HRegionLocation loc;
466        try {
467          if (r == null) {
468            throw new IllegalArgumentException("#" + id + ", row cannot be null");
469          }
470          // Make sure we get 0-s replica.
471          RegionLocations locs = connection.locateRegion(
472              tableName, r.getRow(), true, true, RegionReplicaUtil.DEFAULT_REPLICA_ID);
473          if (locs == null || locs.isEmpty() || locs.getDefaultRegionLocation() == null) {
474            throw new IOException("#" + id + ", no location found, aborting submit for"
475                + " tableName=" + tableName + " rowkey=" + Bytes.toStringBinary(r.getRow()));
476          }
477          loc = locs.getDefaultRegionLocation();
478        } catch (IOException ex) {
479          locationErrors = new ArrayList<Exception>();
480          locationErrorRows = new ArrayList<Integer>();
481          LOG.error("Failed to get region location ", ex);
482          // This action failed before creating ars. Retain it, but do not add to submit list.
483          // We will then add it to ars in an already-failed state.
484          retainedActions.add(new Action<Row>(r, ++posInList));
485          locationErrors.add(ex);
486          locationErrorRows.add(posInList);
487          it.remove();
488          break; // Backward compat: we stop considering actions on location error.
489        }
490        long rowSize = (r instanceof Mutation) ? ((Mutation) r).heapSize() : 0;
491        ReturnCode code = checker.canTakeOperation(loc, rowSize);
492        if (code == ReturnCode.END) {
493          break;
494        }
495        if (code == ReturnCode.INCLUDE) {
496          Action<Row> action = new Action<Row>(r, ++posInList);
497          setNonce(ng, r, action);
498          retainedActions.add(action);
499          // TODO: replica-get is not supported on this path
500          byte[] regionName = loc.getRegionInfo().getRegionName();
501          addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
502          it.remove();
503        }
504      }
505      firstIter = false;
506    } while (retainedActions.isEmpty() && atLeastOne && (locationErrors == null));
507
508    if (retainedActions.isEmpty()) return NO_REQS_RESULT;
509
510    return submitMultiActions(tableName, retainedActions, nonceGroup, callback, null, needResults,
511        locationErrors, locationErrorRows, actionsByServer, pool);
512  }

514  private RowCheckerHost createRowCheckerHost() {
515    return new RowCheckerHost(Arrays.asList(
516        new TaskCountChecker(maxTotalConcurrentTasks,
517          maxConcurrentTasksPerServer,
518          maxConcurrentTasksPerRegion,
519          tasksInProgress,
520          taskCounterPerServer,
521          taskCounterPerRegion)
522        , new RequestSizeChecker(maxHeapSizePerRequest)
523        , new SubmittedSizeChecker(maxHeapSizeSubmit)
524    ));
525  }
526  <CResult> AsyncRequestFuture submitMultiActions(TableName tableName,
527      List<Action<Row>> retainedActions, long nonceGroup, Batch.Callback<CResult> callback,
528      Object[] results, boolean needResults, List<Exception> locationErrors,
529      List<Integer> locationErrorRows, Map<ServerName, MultiAction<Row>> actionsByServer,
530      ExecutorService pool) {
531    AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(
532      tableName, retainedActions, nonceGroup, pool, callback, results, needResults);
533    // Add location errors if any
534    if (locationErrors != null) {
535      for (int i = 0; i < locationErrors.size(); ++i) {
536        int originalIndex = locationErrorRows.get(i);
537        Row row = retainedActions.get(originalIndex).getAction();
538        ars.manageError(originalIndex, row,
539            Retry.NO_LOCATION_PROBLEM, locationErrors.get(i), null);
540      }
541    }
542    ars.sendMultiAction(actionsByServer, 1, null, false);
543    return ars;
544  }

546  /**
547   * Helper that is used when grouping the actions per region server.
548   *
549   * @param loc - the destination. Must not be null.
550   * @param action - the action to add to the multiaction
551   * @param actionsByServer the multiaction per server
552   * @param nonceGroup Nonce group.
553   */
554  private static void addAction(ServerName server, byte[] regionName, Action<Row> action,
555      Map<ServerName, MultiAction<Row>> actionsByServer, long nonceGroup) {
556    MultiAction<Row> multiAction = actionsByServer.get(server);
557    if (multiAction == null) {
558      multiAction = new MultiAction<Row>();
559      actionsByServer.put(server, multiAction);
560    }
561    if (action.hasNonce() && !multiAction.hasNonceGroup()) {
562      multiAction.setNonceGroup(nonceGroup);
563    }
564
565    multiAction.add(regionName, action);
566  }

590  public <CResult> AsyncRequestFuture submitAll(ExecutorService pool, TableName tableName,
591      List<? extends Row> rows, Batch.Callback<CResult> callback, Object[] results,
592      CancellableRegionServerCallable callable, int curTimeout) {
593    List<Action<Row>> actions = new ArrayList<Action<Row>>(rows.size());
594
595    // The position will be used by the processBatch to match the object array returned.
596    int posInList = -1;
597    NonceGenerator ng = this.connection.getNonceGenerator();
598    for (Row r : rows) {
599      posInList++;
600      if (r instanceof Put) {
601        Put put = (Put) r;
602        if (put.isEmpty()) {
603          throw new IllegalArgumentException("No columns to insert for #" + (posInList+1)+ " item");
604        }
605      }
606      Action<Row> action = new Action<Row>(r, posInList);
607      setNonce(ng, r, action);
608      actions.add(action);
609    }
610    AsyncRequestFutureImpl<CResult> ars = createAsyncRequestFuture(
611        tableName, actions, ng.getNonceGroup(), getPool(pool), callback, results, results != null,
612        callable, curTimeout);
613    ars.groupAndSendMultiAction(actions, 1);
614    return ars;
615  }

617  private static void setNonce(NonceGenerator ng, Row r, Action<Row> action) {
618    if (!(r instanceof Append) && !(r instanceof Increment)) return;
619    action.setNonce(ng.newNonce()); // Action handles NO_NONCE, so it's ok if ng is disabled.
620  }

622  /**
623   * The context, and return value, for a single submit/submitAll call.
624   * Note on how this class (one AP submit) works. Initially, all requests are split into groups
625   * by server; request is sent to each server in parallel; the RPC calls are not async so a
626   * thread per server is used. Every time some actions fail, regions/locations might have
627   * changed, so we re-group them by server and region again and send these groups in parallel
628   * too. The result, in case of retries, is a "tree" of threads, with parent exiting after
629   * scheduling children. This is why lots of code doesn't require any synchronization.
630   */

633    /**
634     * Runnable (that can be submitted to thread pool) that waits for when it's time
635     * to issue replica calls, finds region replicas, groups the requests by replica and
636     * issues the calls (on separate threads, via sendMultiAction).
637     * This is done on a separate thread because we don't want to wait on user thread for
638     * our asynchronous call, and usually we have to wait before making replica calls.
639     */
640    private final class ReplicaCallIssuingRunnable implements Runnable {
641      private final long startTime;
642      private final List<Action<Row>> initialActions;
643
644      public ReplicaCallIssuingRunnable(List<Action<Row>> initialActions, long startTime) {
645        this.initialActions = initialActions;
646        this.startTime = startTime;
647      }
648
649      @Override
650      public void run() {
651        boolean done = false;
652        if (primaryCallTimeoutMicroseconds > 0) {
653          try {
654            done = waitUntilDone(startTime * 1000L + primaryCallTimeoutMicroseconds);
655          } catch (InterruptedException ex) {
656            LOG.error("Replica thread was interrupted - no replica calls: " + ex.getMessage());
657            return;
658          }
659        }
660        if (done) return; // Done within primary timeout
661        Map<ServerName, MultiAction<Row>> actionsByServer =
662            new HashMap<ServerName, MultiAction<Row>>();
663        List<Action<Row>> unknownLocActions = new ArrayList<Action<Row>>();
664        if (replicaGetIndices == null) {
665          for (int i = 0; i < results.length; ++i) {
666            addReplicaActions(i, actionsByServer, unknownLocActions);
667          }
668        } else {
669          for (int replicaGetIndice : replicaGetIndices) {
670            addReplicaActions(replicaGetIndice, actionsByServer, unknownLocActions);
671          }
672        }
673        if (!actionsByServer.isEmpty()) {
674          sendMultiAction(actionsByServer, 1, null, unknownLocActions.isEmpty());
675        }
676        if (!unknownLocActions.isEmpty()) {
677          actionsByServer = new HashMap<ServerName, MultiAction<Row>>();
678          for (Action<Row> action : unknownLocActions) {
679            addReplicaActionsAgain(action, actionsByServer);
680          }
681          // Some actions may have completely failed, they are handled inside addAgain.
682          if (!actionsByServer.isEmpty()) {
683            sendMultiAction(actionsByServer, 1, null, true);
684          }
685        }
686      }
687
688      /**
689       * Add replica actions to action map by server.
690       * @param index Index of the original action.
691       * @param actionsByServer The map by server to add it to.
692       */
693      private void addReplicaActions(int index, Map<ServerName, MultiAction<Row>> actionsByServer,
694          List<Action<Row>> unknownReplicaActions) {
695        if (results[index] != null) return; // opportunistic. Never goes from non-null to null.
696        Action<Row> action = initialActions.get(index);
697        RegionLocations loc = findAllLocationsOrFail(action, true);
698        if (loc == null) return;
699        HRegionLocation[] locs = loc.getRegionLocations();
700        if (locs.length == 1) {
701          LOG.warn("No replicas found for " + action.getAction());
702          return;
703        }
704        synchronized (replicaResultLock) {
705          // Don't run replica calls if the original has finished. We could do it e.g. if
706          // original has already failed before first replica call (unlikely given retries),
707          // but that would require additional synchronization w.r.t. returning to caller.
708          if (results[index] != null) return;
709          // We set the number of calls here. After that any path must call setResult/setError.
710          // True even for replicas that are not found - if we refuse to send we MUST set error.
711          results[index] = new ReplicaResultState(locs.length);
712        }
713        for (int i = 1; i < locs.length; ++i) {
714          Action<Row> replicaAction = new Action<Row>(action, i);
715          if (locs[i] != null) {
716            addAction(locs[i].getServerName(), locs[i].getRegionInfo().getRegionName(),
717                replicaAction, actionsByServer, nonceGroup);
718          } else {
719            unknownReplicaActions.add(replicaAction);
720          }
721        }
722      }
723
724      private void addReplicaActionsAgain(
725          Action<Row> action, Map<ServerName, MultiAction<Row>> actionsByServer) {
726        if (action.getReplicaId() == RegionReplicaUtil.DEFAULT_REPLICA_ID) {
727          throw new AssertionError("Cannot have default replica here");
728        }
729        HRegionLocation loc = getReplicaLocationOrFail(action);
730        if (loc == null) return;
731        addAction(loc.getServerName(), loc.getRegionInfo().getRegionName(),
732            action, actionsByServer, nonceGroup);
733      }
734    }

737     * Runnable (that can be submitted to thread pool) that submits MultiAction to a
738     * single server. The server call is synchronous, therefore we do it on a thread pool.
739     */
740    private final class SingleServerRequestRunnable implements Runnable {
741      private final MultiAction<Row> multiAction;
742      private final int numAttempt;
743      private final ServerName server;
744      private final Set<CancellableRegionServerCallable> callsInProgress;
745      private Long heapSize = null;
746      private SingleServerRequestRunnable(
747          MultiAction<Row> multiAction, int numAttempt, ServerName server,
748          Set<CancellableRegionServerCallable> callsInProgress) {
749        this.multiAction = multiAction;
750        this.numAttempt = numAttempt;
751        this.server = server;
752        this.callsInProgress = callsInProgress;
753      }
754
755      @VisibleForTesting
756      long heapSize() {
757        if (heapSize != null) {
758          return heapSize;
759        }
760        heapSize = 0L;
761        for (Map.Entry<byte[], List<Action<Row>>> e: this.multiAction.actions.entrySet()) {
762          List<Action<Row>> actions = e.getValue();
763          for (Action<Row> action: actions) {
764            Row row = action.getAction();
765            if (row instanceof Mutation) {
766              heapSize += ((Mutation) row).heapSize();
767            }
768          }
769        }
770        return heapSize;
771      }
772
773      @Override
774      public void run() {
775        MultiResponse res;
776        CancellableRegionServerCallable callable = currentCallable;
777        try {
778          // setup the callable based on the actions, if we don't have one already from the request
779          if (callable == null) {
780            callable = createCallable(server, tableName, multiAction);
781          }
782          RpcRetryingCaller<MultiResponse> caller = createCaller(callable);
783          try {
784            if (callsInProgress != null) {
785              callsInProgress.add(callable);
786            }
787            res = caller.callWithoutRetries(callable, currentCallTotalTimeout);
788            if (res == null) {
789              // Cancelled
790              return;
791            }
792          } catch (IOException e) {
793            // The service itself failed . It may be an error coming from the communication
794            //   layer, but, as well, a functional error raised by the server.
795            receiveGlobalFailure(multiAction, server, numAttempt, e);
796            return;
797          } catch (Throwable t) {
798            // This should not happen. Let's log & retry anyway.
799            LOG.error("#" + id + ", Caught throwable while calling. This is unexpected." +
800                " Retrying. Server is " + server + ", tableName=" + tableName, t);
801            receiveGlobalFailure(multiAction, server, numAttempt, t);
802            return;
803          }
804
805          // Normal case: we received an answer from the server, and it's not an exception.
806          receiveMultiAction(multiAction, server, res, numAttempt);
807        } catch (Throwable t) {
808          // Something really bad happened. We are on the send thread that will now die.
809          LOG.error("Internal AsyncProcess #" + id + " error for "
810              + tableName + " processing for " + server, t);
811          throw new RuntimeException(t);
812        } finally {
813          decTaskCounters(multiAction.getRegions(), server);
814          if (callsInProgress != null && callable != null) {
815            callsInProgress.remove(callable);
816          }
817        }
818      }
819    }

855    public AsyncRequestFutureImpl(TableName tableName, List<Action<Row>> actions, long nonceGroup,
856        ExecutorService pool, boolean needResults, Object[] results,
857        Batch.Callback<CResult> callback, CancellableRegionServerCallable callable, int timeout) {
858      this.pool = pool;
859      this.callback = callback;
860      this.nonceGroup = nonceGroup;
861      this.tableName = tableName;
862      this.actionsInProgress.set(actions.size());
863      if (results != null) {
864        assert needResults;
865        if (results.length != actions.size()) {
866          throw new AssertionError("results.length");
867        }
868        this.results = results;
869        for (int i = 0; i != this.results.length; ++i) {
870          results[i] = null;
871        }
872      } else {
873        this.results = needResults ? new Object[actions.size()] : null;
874      }
875      List<Integer> replicaGetIndices = null;
876      boolean hasAnyReplicaGets = false;
877      if (needResults) {
878        // Check to see if any requests might require replica calls.
879        // We expect that many requests will consist of all or no multi-replica gets; in such
880        // cases we would just use a boolean (hasAnyReplicaGets). If there's a mix, we will
881        // store the list of action indexes for which replica gets are possible, and set
882        // hasAnyReplicaGets to true.
883        boolean hasAnyNonReplicaReqs = false;
884        int posInList = 0;
885        for (Action<Row> action : actions) {
886          boolean isReplicaGet = isReplicaGet(action.getAction());
887          if (isReplicaGet) {
888            hasAnyReplicaGets = true;
889            if (hasAnyNonReplicaReqs) { // Mixed case
890              if (replicaGetIndices == null) {
891                replicaGetIndices = new ArrayList<Integer>(actions.size() - 1);
892              }
893              replicaGetIndices.add(posInList);
894            }
895          } else if (!hasAnyNonReplicaReqs) {
896            // The first non-multi-replica request in the action list.
897            hasAnyNonReplicaReqs = true;
898            if (posInList > 0) {
899              // Add all the previous requests to the index lists. We know they are all
900              // replica-gets because this is the first non-multi-replica request in the list.
901              replicaGetIndices = new ArrayList<Integer>(actions.size() - 1);
902              for (int i = 0; i < posInList; ++i) {
903                replicaGetIndices.add(i);
904              }
905            }
906          }
907          ++posInList;
908        }
909      }
910      this.hasAnyReplicaGets = hasAnyReplicaGets;
911      if (replicaGetIndices != null) {
912        this.replicaGetIndices = new int[replicaGetIndices.size()];
913        int i = 0;
914        for (Integer el : replicaGetIndices) {
915          this.replicaGetIndices[i++] = el;
916        }
917      } else {
918        this.replicaGetIndices = null;
919      }
920      this.callsInProgress = !hasAnyReplicaGets ? null :
921          Collections.newSetFromMap(
922              new ConcurrentHashMap<CancellableRegionServerCallable, Boolean>());
923
924      this.errorsByServer = createServerErrorTracker();
925      this.errors = (globalErrors != null) ? globalErrors : new BatchErrors();
926      this.currentCallable = callable;
927      this.currentCallTotalTimeout = timeout;
928    }

949     * Group a list of actions per region servers, and send them.
950     *
951     * @param currentActions - the list of row to submit
952     * @param numAttempt - the current numAttempt (first attempt is 1)
953     */
954    private void groupAndSendMultiAction(List<Action<Row>> currentActions, int numAttempt) {
955      Map<ServerName, MultiAction<Row>> actionsByServer =
956          new HashMap<ServerName, MultiAction<Row>>();
957
958      boolean isReplica = false;
959      List<Action<Row>> unknownReplicaActions = null;
960      for (Action<Row> action : currentActions) {
961        RegionLocations locs = findAllLocationsOrFail(action, true);
962        if (locs == null) continue;
963        boolean isReplicaAction = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
964        if (isReplica && !isReplicaAction) {
965          // This is the property of the current implementation, not a requirement.
966          throw new AssertionError("Replica and non-replica actions in the same retry");
967        }
968        isReplica = isReplicaAction;
969        HRegionLocation loc = locs.getRegionLocation(action.getReplicaId());
970        if (loc == null || loc.getServerName() == null) {
971          if (isReplica) {
972            if (unknownReplicaActions == null) {
973              unknownReplicaActions = new ArrayList<Action<Row>>();
974            }
975            unknownReplicaActions.add(action);
976          } else {
977            // TODO: relies on primary location always being fetched
978            manageLocationError(action, null);
979          }
980        } else {
981          byte[] regionName = loc.getRegionInfo().getRegionName();
982          addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
983        }
984      }
985      boolean doStartReplica = (numAttempt == 1 && !isReplica && hasAnyReplicaGets);
986      boolean hasUnknown = unknownReplicaActions != null && !unknownReplicaActions.isEmpty();
987
988      if (!actionsByServer.isEmpty()) {
989        // If this is a first attempt to group and send, no replicas, we need replica thread.
990        sendMultiAction(actionsByServer, numAttempt, (doStartReplica && !hasUnknown)
991            ? currentActions : null, numAttempt > 1 && !hasUnknown);
992      }
993
994      if (hasUnknown) {
995        actionsByServer = new HashMap<ServerName, MultiAction<Row>>();
996        for (Action<Row> action : unknownReplicaActions) {
997          HRegionLocation loc = getReplicaLocationOrFail(action);
998          if (loc == null) continue;
999          byte[] regionName = loc.getRegionInfo().getRegionName();
1000          addAction(loc.getServerName(), regionName, action, actionsByServer, nonceGroup);
1001        }
1002        if (!actionsByServer.isEmpty()) {
1003          sendMultiAction(
1004              actionsByServer, numAttempt, doStartReplica ? currentActions : null, true);
1005        }
1006      }
1007    }

1009    private HRegionLocation getReplicaLocationOrFail(Action<Row> action) {
1010      // We are going to try get location once again. For each action, we'll do it once
1011      // from cache, because the previous calls in the loop might populate it.
1012      int replicaId = action.getReplicaId();
1013      RegionLocations locs = findAllLocationsOrFail(action, true);
1014      if (locs == null) return null; // manageError already called
1015      HRegionLocation loc = locs.getRegionLocation(replicaId);
1016      if (loc == null || loc.getServerName() == null) {
1017        locs = findAllLocationsOrFail(action, false);
1018        if (locs == null) return null; // manageError already called
1019        loc = locs.getRegionLocation(replicaId);
1020      }
1021      if (loc == null || loc.getServerName() == null) {
1022        manageLocationError(action, null);
1023        return null;
1024      }
1025      return loc;
1026    }

1052    /**
1053     * Send a multi action structure to the servers, after a delay depending on the attempt
1054     * number. Asynchronous.
1055     *
1056     * @param actionsByServer the actions structured by regions
1057     * @param numAttempt the attempt number.
1058     * @param actionsForReplicaThread original actions for replica thread; null on non-first call.
1059     */
1060    private void sendMultiAction(Map<ServerName, MultiAction<Row>> actionsByServer,
1061        int numAttempt, List<Action<Row>> actionsForReplicaThread, boolean reuseThread) {
1062      // Run the last item on the same thread if we are already on a send thread.
1063      // We hope most of the time it will be the only item, so we can cut down on threads.
1064      int actionsRemaining = actionsByServer.size();
1065      // This iteration is by server (the HRegionLocation comparator is by server portion only).
1066      for (Map.Entry<ServerName, MultiAction<Row>> e : actionsByServer.entrySet()) {
1067        ServerName server = e.getKey();
1068        MultiAction<Row> multiAction = e.getValue();
1069        Collection<? extends Runnable> runnables = getNewMultiActionRunnable(server, multiAction,
1070            numAttempt);
1071        // make sure we correctly count the number of runnables before we try to reuse the send
1072        // thread, in case we had to split the request into different runnables because of backoff
1073        if (runnables.size() > actionsRemaining) {
1074          actionsRemaining = runnables.size();
1075        }
1076
1077        // run all the runnables
1078        for (Runnable runnable : runnables) {
1079          if ((--actionsRemaining == 0) && reuseThread) {
1080            runnable.run();
1081          } else {
1082            try {
1083              pool.submit(runnable);
1084            } catch (Throwable t) {
1085              if (t instanceof RejectedExecutionException) {
1086                // This should never happen. But as the pool is provided by the end user,
1087               // let's secure this a little.
1088               LOG.warn("#" + id + ", the task was rejected by the pool. This is unexpected." +
1089                  " Server is " + server.getServerName(), t);
1090              } else {
1091                // see #HBASE-14359 for more details
1092                LOG.warn("Caught unexpected exception/error: ", t);
1093              }
1094              decTaskCounters(multiAction.getRegions(), server);
1095              // We're likely to fail again, but this will increment the attempt counter,
1096             // so it will finish.
1097              receiveGlobalFailure(multiAction, server, numAttempt, t);
1098            }
1099          }
1100        }
1101      }
1102
1103      if (actionsForReplicaThread != null) {
1104        startWaitingForReplicaCalls(actionsForReplicaThread);
1105      }
1106    }

1108    private Collection<? extends Runnable> getNewMultiActionRunnable(ServerName server,
1109        MultiAction<Row> multiAction,
1110        int numAttempt) {
1111      // no stats to manage, just do the standard action
1112      if (AsyncProcess.this.connection.getStatisticsTracker() == null) {
1113        if (connection.getConnectionMetrics() != null) {
1114          connection.getConnectionMetrics().incrNormalRunners();
1115        }
1116        incTaskCounters(multiAction.getRegions(), server);
1117        SingleServerRequestRunnable runnable = addSingleServerRequestHeapSize(server,
1118          new SingleServerRequestRunnable(multiAction, numAttempt, server, callsInProgress));
1119        return Collections.singletonList(Trace.wrap("AsyncProcess.sendMultiAction", runnable));
1120      }
1121
1122      // group the actions by the amount of delay
1123      Map<Long, DelayingRunner> actions = new HashMap<Long, DelayingRunner>(multiAction
1124          .size());
1125
1126      // split up the actions
1127      for (Map.Entry<byte[], List<Action<Row>>> e : multiAction.actions.entrySet()) {
1128        Long backoff = getBackoff(server, e.getKey());
1129        DelayingRunner runner = actions.get(backoff);
1130        if (runner == null) {
1131          actions.put(backoff, new DelayingRunner(backoff, e));
1132        } else {
1133          runner.add(e);
1134        }
1135      }
1136
1137      List<Runnable> toReturn = new ArrayList<Runnable>(actions.size());
1138      for (DelayingRunner runner : actions.values()) {
1139        incTaskCounters(runner.getActions().getRegions(), server);
1140        String traceText = "AsyncProcess.sendMultiAction";
1141        Runnable runnable = addSingleServerRequestHeapSize(server,
1142          new SingleServerRequestRunnable(runner.getActions(), numAttempt, server, callsInProgress));
1143        // use a delay runner only if we need to sleep for some time
1144        if (runner.getSleepTime() > 0) {
1145          runner.setRunner(runnable);
1146          traceText = "AsyncProcess.clientBackoff.sendMultiAction";
1147          runnable = runner;
1148          if (connection.getConnectionMetrics() != null) {
1149            connection.getConnectionMetrics().incrDelayRunners();
1150            connection.getConnectionMetrics().updateDelayInterval(runner.getSleepTime());
1151          }
1152        } else {
1153          if (connection.getConnectionMetrics() != null) {
1154            connection.getConnectionMetrics().incrNormalRunners();
1155          }
1156        }
1157        runnable = Trace.wrap(traceText, runnable);
1158        toReturn.add(runnable);
1159
1160      }
1161      return toReturn;

1199     * Check that we can retry acts accordingly: logs, set the error status.
1200     *
1201     * @param originalIndex the position in the list sent
1202     * @param row           the row
1203     * @param canRetry      if false, we won't retry whatever the settings.
1204     * @param throwable     the throwable, if any (can be null)
1205     * @param server        the location, if any (can be null)
1206     * @return true if the action can be retried, false otherwise.
1207     */
1208    public Retry manageError(int originalIndex, Row row, Retry canRetry,
1209                                Throwable throwable, ServerName server) {
1210      if (canRetry == Retry.YES
1211          && throwable != null && throwable instanceof DoNotRetryIOException) {
1212        canRetry = Retry.NO_NOT_RETRIABLE;
1213      }
1214
1215      if (canRetry != Retry.YES) {
1216        // Batch.Callback<Res> was not called on failure in 0.94. We keep this.
1217        setError(originalIndex, row, throwable, server);
1218      } else if (isActionComplete(originalIndex, row)) {
1219        canRetry = Retry.NO_OTHER_SUCCEEDED;
1220      }
1221      return canRetry;
1222    }

1224    /**
1225     * Resubmit all the actions from this multiaction after a failure.
1226     *
1227     * @param rsActions  the actions still to do from the initial list
1228     * @param server   the destination
1229     * @param numAttempt the number of attempts so far
1230     * @param t the throwable (if any) that caused the resubmit
1231     */
1232    private void receiveGlobalFailure(
1233        MultiAction<Row> rsActions, ServerName server, int numAttempt, Throwable t) {
1234      errorsByServer.reportServerError(server);
1235      Retry canRetry = errorsByServer.canTryMore(numAttempt)
1236          ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED;
1237
1238      if (tableName == null && ClientExceptionsUtil.isMetaClearingException(t)) {
1239        // tableName is null when we made a cross-table RPC call.
1240        connection.clearCaches(server);
1241      }
1242      int failed = 0, stopped = 0;
1243      List<Action<Row>> toReplay = new ArrayList<Action<Row>>();
1244      for (Map.Entry<byte[], List<Action<Row>>> e : rsActions.actions.entrySet()) {
1245        byte[] regionName = e.getKey();
1246        byte[] row = e.getValue().iterator().next().getAction().getRow();
1247        // Do not use the exception for updating cache because it might be coming from
1248        // any of the regions in the MultiAction.
1249        try {
1250          if (tableName != null) {
1251            connection.updateCachedLocations(tableName, regionName, row,
1252              ClientExceptionsUtil.isMetaClearingException(t) ? null : t, server);
1253          }
1254        } catch (Throwable ex) {
1255          // That should never happen, but if it did, we want to make sure
1256          // we still process errors
1257          LOG.error("Couldn't update cached region locations: " + ex);
1258        }
1259        for (Action<Row> action : e.getValue()) {
1260          Retry retry = manageError(
1261              action.getOriginalIndex(), action.getAction(), canRetry, t, server);
1262          if (retry == Retry.YES) {
1263            toReplay.add(action);
1264          } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
1265            ++stopped;
1266          } else {
1267            ++failed;
1268          }
1269        }
1270      }
1271
1272      if (toReplay.isEmpty()) {
1273        logNoResubmit(server, numAttempt, rsActions.size(), t, failed, stopped);
1274      } else {
1275        resubmit(server, toReplay, numAttempt, rsActions.size(), t);
1276      }
1277    }

1279    /**
1280     * Log as much info as possible, and, if there is something to replay,
1281     * submit it again after a back off sleep.
1282     */
1283    private void resubmit(ServerName oldServer, List<Action<Row>> toReplay,
1284        int numAttempt, int failureCount, Throwable throwable) {
1285      // We have something to replay. We're going to sleep a little before.
1286
1287      // We have two contradicting needs here:
1288      //  1) We want to get the new location after having slept, as it may change.
1289      //  2) We want to take into account the location when calculating the sleep time.
1290      //  3) If all this is just because the response needed to be chunked try again FAST.
1291      // It should be possible to have some heuristics to take the right decision. Short term,
1292      //  we go for one.
1293      boolean retryImmediately = throwable instanceof RetryImmediatelyException;
1294      int nextAttemptNumber = retryImmediately ? numAttempt : numAttempt + 1;
1295      long backOffTime = retryImmediately ? 0 :
1296          errorsByServer.calculateBackoffTime(oldServer, pause);
1297      if (numAttempt > startLogErrorsCnt) {
1298        // We use this value to have some logs when we have multiple failures, but not too many
1299        //  logs, as errors are to be expected when a region moves, splits and so on
1300        LOG.info(createLog(numAttempt, failureCount, toReplay.size(),
1301            oldServer, throwable, backOffTime, true, null, -1, -1));
1302      }
1303
1304      try {
1305        if (backOffTime > 0) {
1306          Thread.sleep(backOffTime);
1307        }
1308      } catch (InterruptedException e) {
1309        LOG.warn("#" + id + ", not sent: " + toReplay.size() + " operations, " + oldServer, e);
1310        Thread.currentThread().interrupt();
1311        return;
1312      }
1313
1314      groupAndSendMultiAction(toReplay, nextAttemptNumber);
1315    }

1332    /**
1333     * Called when we receive the result of a server query.
1334     *
1335     * @param multiAction    - the multiAction we sent
1336     * @param server       - the location. It's used as a server name.
1337     * @param responses      - the response, if any
1338     * @param numAttempt     - the attempt
1339     */
1340    private void receiveMultiAction(MultiAction<Row> multiAction,
1341        ServerName server, MultiResponse responses, int numAttempt) {
1342       assert responses != null;
1343
1344      // Success or partial success
1345      // Analyze detailed results. We can still have individual failures to be redo.
1346      // two specific throwables are managed:
1347      //  - DoNotRetryIOException: we continue to retry for other actions
1348      //  - RegionMovedException: we update the cache with the new region location
1349
1350      List<Action<Row>> toReplay = new ArrayList<Action<Row>>();
1351      Throwable throwable = null;
1352      int failureCount = 0;
1353      boolean canRetry = true;
1354
1355      Map<byte[], MultiResponse.RegionResult> results = responses.getResults();
1356      updateStats(server, results);
1357
1358      int failed = 0, stopped = 0;
1359      // Go by original action.
1360      for (Map.Entry<byte[], List<Action<Row>>> regionEntry : multiAction.actions.entrySet()) {
1361        byte[] regionName = regionEntry.getKey();
1362        Map<Integer, Object> regionResults = results.get(regionName) == null
1363            ?  null : results.get(regionName).result;
1364        if (regionResults == null) {
1365          if (!responses.getExceptions().containsKey(regionName)) {
1366            LOG.error("Server sent us neither results nor exceptions for "
1367                + Bytes.toStringBinary(regionName));
1368            responses.getExceptions().put(regionName, new RuntimeException("Invalid response"));
1369          }
1370          continue;
1371        }
1372        boolean regionFailureRegistered = false;
1373        for (Action<Row> sentAction : regionEntry.getValue()) {
1374          Object result = regionResults.get(sentAction.getOriginalIndex());
1375          // Failure: retry if it's make sense else update the errors lists
1376          if (result == null || result instanceof Throwable) {
1377            Row row = sentAction.getAction();
1378            throwable = ClientExceptionsUtil.findException(result);
1379            // Register corresponding failures once per server/once per region.
1380            if (!regionFailureRegistered) {
1381              regionFailureRegistered = true;
1382              try {
1383                connection.updateCachedLocations(
1384                  tableName, regionName, row.getRow(), result, server);
1385              } catch (Throwable ex) {
1386                // That should never happen, but if it did, we want to make sure
1387                // we still process errors
1388                LOG.error("Couldn't update cached region locations: " + ex);
1389              }
1390            }
1391            if (failureCount == 0) {
1392              errorsByServer.reportServerError(server);
1393              // We determine canRetry only once for all calls, after reporting server failure.
1394              canRetry = errorsByServer.canTryMore(numAttempt);
1395            }
1396            ++failureCount;
1397            Retry retry = manageError(sentAction.getOriginalIndex(), row,
1398                canRetry ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED, (Throwable) result, server);
1399            if (retry == Retry.YES) {
1400              toReplay.add(sentAction);
1401            } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
1402              ++stopped;
1403            } else {
1404              ++failed;
1405            }
1406          } else {
1407            if (callback != null) {
1408              try {
1409                //noinspection unchecked
1410                // TODO: would callback expect a replica region name if it gets one?
1411                this.callback.update(regionName, sentAction.getAction().getRow(), (CResult) result);
1412              } catch (Throwable t) {
1413                LOG.error("User callback threw an exception for "
1414                    + Bytes.toStringBinary(regionName) + ", ignoring", t);
1415              }
1416            }
1417            setResult(sentAction, result);
1418          }
1419        }
1420      }
1421
1422      // The failures global to a region. We will use for multiAction we sent previously to find the
1423      //   actions to replay.
1424      for (Map.Entry<byte[], Throwable> throwableEntry : responses.getExceptions().entrySet()) {
1425        throwable = throwableEntry.getValue();
1426        byte[] region = throwableEntry.getKey();
1427        List<Action<Row>> actions = multiAction.actions.get(region);
1428        if (actions == null || actions.isEmpty()) {
1429          throw new IllegalStateException("Wrong response for the region: " +
1430              HRegionInfo.encodeRegionName(region));
1431        }
1432
1433        if (failureCount == 0) {
1434          errorsByServer.reportServerError(server);
1435          canRetry = errorsByServer.canTryMore(numAttempt);
1436        }
1437        if (null == tableName && ClientExceptionsUtil.isMetaClearingException(throwable)) {
1438          // For multi-actions, we don't have a table name, but we want to make sure to clear the
1439          // cache in case there were location-related exceptions. We don't to clear the cache
1440          // for every possible exception that comes through, however.
1441          connection.clearCaches(server);
1442        } else {
1443          try {
1444            connection.updateCachedLocations(
1445              tableName, region, actions.get(0).getAction().getRow(), throwable, server);
1446          } catch (Throwable ex) {
1447            // That should never happen, but if it did, we want to make sure
1448            // we still process errors
1449            LOG.error("Couldn't update cached region locations: " + ex);
1450          }
1451        }
1452        failureCount += actions.size();
1453
1454        for (Action<Row> action : actions) {
1455          Row row = action.getAction();
1456          Retry retry = manageError(action.getOriginalIndex(), row,
1457              canRetry ? Retry.YES : Retry.NO_RETRIES_EXHAUSTED, throwable, server);
1458          if (retry == Retry.YES) {
1459            toReplay.add(action);
1460          } else if (retry == Retry.NO_OTHER_SUCCEEDED) {
1461            ++stopped;
1462          } else {
1463            ++failed;
1464          }
1465        }
1466      }
1467      if (toReplay.isEmpty()) {
1468        logNoResubmit(server, numAttempt, failureCount, throwable, failed, stopped);
1469      } else {
1470        resubmit(server, toReplay, numAttempt, failureCount, throwable);
1471      }
1472    }

1507    /**
1508     * Sets the non-error result from a particular action.
1509     * @param action Action (request) that the server responded to.
1510     * @param result The result.
1511     */
1512    private void setResult(Action<Row> action, Object result) {
1513      if (result == null) {
1514        throw new RuntimeException("Result cannot be null");
1515      }
1516      ReplicaResultState state = null;
1517      boolean isStale = !RegionReplicaUtil.isDefaultReplica(action.getReplicaId());
1518      int index = action.getOriginalIndex();
1519      if (results == null) {
1520        decActionCounter(index);
1521        return; // Simple case, no replica requests.
1522      }
1523      state = trySetResultSimple(index, action.getAction(), false, result, null, isStale);
1524      if (state == null) {
1525        return; // Simple case, no replica requests.
1526      }
1527      // At this point we know that state is set to replica tracking class.
1528      // It could be that someone else is also looking at it; however, we know there can
1529      // only be one state object, and only one thread can set callCount to 0. Other threads
1530      // will either see state with callCount 0 after locking it; or will not see state at all
1531      // we will replace it with the result.
1532      synchronized (state) {
1533        if (state.callCount == 0) {
1534          return; // someone already set the result
1535        }
1536        state.callCount = 0;
1537      }
1538      synchronized (replicaResultLock) {
1539        if (results[index] != state) {
1540          throw new AssertionError("We set the callCount but someone else replaced the result");
1541        }
1542        results[index] = result;
1543      }
1544
1545      decActionCounter(index);
1546    }

1548    /**
1549     * Sets the error from a particular action.
1550     * @param index Original action index.
1551     * @param row Original request.
1552     * @param throwable The resulting error.
1553     * @param server The source server.
1554     */
1555    private void setError(int index, Row row, Throwable throwable, ServerName server) {
1556      ReplicaResultState state = null;
1557      if (results == null) {
1558        // Note that we currently cannot have replica requests with null results. So it shouldn't
1559        // happen that multiple replica calls will call dAC for same actions with results == null.
1560        // Only one call per action should be present in this case.
1561        errors.add(throwable, row, server);
1562        decActionCounter(index);
1563        return; // Simple case, no replica requests.
1564      }
1565      state = trySetResultSimple(index, row, true, throwable, server, false);
1566      if (state == null) {
1567        return; // Simple case, no replica requests.
1568      }
1569      BatchErrors target = null; // Error will be added to final errors, or temp replica errors.
1570      boolean isActionDone = false;
1571      synchronized (state) {
1572        switch (state.callCount) {
1573          case 0: return; // someone already set the result
1574          case 1: { // All calls failed, we are the last error.
1575            target = errors;
1576            isActionDone = true;
1577            break;
1578          }
1579          default: {
1580            assert state.callCount > 1;
1581            if (state.replicaErrors == null) {
1582              state.replicaErrors = new BatchErrors();
1583            }
1584            target = state.replicaErrors;
1585            break;
1586          }
1587        }
1588        --state.callCount;
1589      }
1590      target.add(throwable, row, server);
1591      if (isActionDone) {
1592        if (state.replicaErrors != null) { // last call, no need to lock
1593          errors.merge(state.replicaErrors);
1594        }
1595        // See setResult for explanations.
1596        synchronized (replicaResultLock) {
1597          if (results[index] != state) {
1598            throw new AssertionError("We set the callCount but someone else replaced the result");
1599          }
1600          results[index] = throwable;
1601        }
1602        decActionCounter(index);
1603      }
1604    }

1620    /**
1621     * Tries to set the result or error for a particular action as if there were no replica calls.
1622     * @return null if successful; replica state if there were in fact replica calls.
1623     */
1624    private ReplicaResultState trySetResultSimple(int index, Row row, boolean isError,
1625        Object result, ServerName server, boolean isFromReplica) {
1626      Object resObj = null;
1627      if (!isReplicaGet(row)) {
1628        if (isFromReplica) {
1629          throw new AssertionError("Unexpected stale result for " + row);
1630        }
1631        results[index] = result;
1632      } else {
1633        synchronized (replicaResultLock) {
1634          resObj = results[index];
1635          if (resObj == null) {
1636            if (isFromReplica) {
1637              throw new AssertionError("Unexpected stale result for " + row);
1638            }
1639            results[index] = result;
1640          }
1641        }
1642      }
1643
1644      ReplicaResultState rrs =
1645          (resObj instanceof ReplicaResultState) ? (ReplicaResultState)resObj : null;
1646      if (rrs == null && isError) {
1647        // The resObj is not replica state (null or already set).
1648        errors.add((Throwable)result, row, server);
1649      }
1650
1651      if (resObj == null) {
1652        // resObj is null - no replica calls were made.
1653        decActionCounter(index);
1654        return null;
1655      }
1656      return rrs;
1657    }

1659    private void decActionCounter(int index) {
1660      long actionsRemaining = actionsInProgress.decrementAndGet();
1661      if (actionsRemaining < 0) {
1662        String error = buildDetailedErrorMsg("Incorrect actions in progress", index);
1663        throw new AssertionError(error);
1664      } else if (actionsRemaining == 0) {
1665        synchronized (actionsInProgress) {
1666          actionsInProgress.notifyAll();
1667        }
1668      }
1669    }

1707    private boolean waitUntilDone(long cutoff) throws InterruptedException {
1708      boolean hasWait = cutoff != Long.MAX_VALUE;
1709      long lastLog = EnvironmentEdgeManager.currentTime();
1710      long currentInProgress;
1711      while (0 != (currentInProgress = actionsInProgress.get())) {
1712        long now = EnvironmentEdgeManager.currentTime();
1713        if (hasWait && (now * 1000L) > cutoff) {
1714          return false;
1715        }
1716        if (!hasWait) { // Only log if wait is infinite.
1717          if (now > lastLog + 10000) {
1718            lastLog = now;
1719            LOG.info("#" + id + ", waiting for " + currentInProgress
1720                + "  actions to finish on table: " + tableName);
1721            if (currentInProgress <= thresholdToLogUndoneTaskDetails) {
1722              logDetailsOfUndoneTasks(currentInProgress);
1723            }
1724          }
1725        }
1726        synchronized (actionsInProgress) {
1727          if (actionsInProgress.get() == 0) break;
1728          if (!hasWait) {
1729            actionsInProgress.wait(10);
1730          } else {
1731            long waitMicroSecond = Math.min(100000L, (cutoff - now * 1000L));
1732            TimeUnit.MICROSECONDS.timedWait(actionsInProgress, waitMicroSecond);
1733          }
1734        }
1735      }
1736      return true;
1737    }

1827  @VisibleForTesting
1828  void waitForMaximumCurrentTasks(int max, final AtomicLong tasksInProgress, final long id,
1829      String tableName) throws InterruptedIOException {
1830    long lastLog = EnvironmentEdgeManager.currentTime();
1831    long currentInProgress, oldInProgress = Long.MAX_VALUE;
1832    while ((currentInProgress = tasksInProgress.get()) > max) {
1833      if (oldInProgress != currentInProgress) { // Wait for in progress to change.
1834        long now = EnvironmentEdgeManager.currentTime();
1835        if (now > lastLog + 10000) {
1836          lastLog = now;
1837          LOG.info("#" + id + ", waiting for some tasks to finish. Expected max="
1838              + max + ", tasksInProgress=" + currentInProgress +
1839              " hasError=" + hasError() + tableName == null ? "" : ", tableName=" + tableName);
1840          if (currentInProgress <= thresholdToLogUndoneTaskDetails) {
1841            logDetailsOfUndoneTasks(currentInProgress);
1842          }
1843        }
1844      }
1845      oldInProgress = currentInProgress;
1846      try {
1847        synchronized (tasksInProgress) {
1848          if (tasksInProgress.get() == oldInProgress) {
1849            tasksInProgress.wait(10);
1850          }
1851        }
1852      } catch (InterruptedException e) {
1853        throw new InterruptedIOException("#" + id + ", interrupted." +
1854            " currentNumberOfTask=" + currentInProgress);
1855      }
1856    }
1857  }

1860    ArrayList<ServerName> servers = new ArrayList<ServerName>();

1912   * increment the tasks counters for a given set of regions. MT safe.
1913   */
1914  protected void incTaskCounters(Collection<byte[]> regions, ServerName sn) {
1915    tasksInProgress.incrementAndGet();
1916
1917    AtomicInteger serverCnt = taskCounterPerServer.get(sn);
1918    if (serverCnt == null) {
1919      taskCounterPerServer.putIfAbsent(sn, new AtomicInteger());
1920      serverCnt = taskCounterPerServer.get(sn);
1921    }
1922    serverCnt.incrementAndGet();
1923
1924    for (byte[] regBytes : regions) {
1925      AtomicInteger regionCnt = taskCounterPerRegion.get(regBytes);
1926      if (regionCnt == null) {
1927        regionCnt = new AtomicInteger();
1928        AtomicInteger oldCnt = taskCounterPerRegion.putIfAbsent(regBytes, regionCnt);
1929        if (oldCnt != null) {
1930          regionCnt = oldCnt;
1931        }
1932      }
1933      regionCnt.incrementAndGet();
1934    }
1935  }

1984  @VisibleForTesting
1985  static class RowCheckerHost {

2022        }

2024          break;
2025        }
2026      }
2027      for (RowChecker checker : checkers) {
2028        checker.notifyFinal(code, loc, rowSize);

2098  /**
2099   * limit the max number of tasks in an AsyncProcess.
2100   */
2101  @VisibleForTesting
2102  static class TaskCountChecker implements RowChecker {
2103    private static final long MAX_WAITING_TIME = 1000; //ms
2104    private final Set<HRegionInfo> regionsIncluded = new HashSet<>();
2105    private final Set<ServerName> serversIncluded = new HashSet<>();
2106    private final int maxConcurrentTasksPerRegion;
2107    private final int maxTotalConcurrentTasks;
2108    private final int maxConcurrentTasksPerServer;
2109    private final Map<byte[], AtomicInteger> taskCounterPerRegion;
2110    private final Map<ServerName, AtomicInteger> taskCounterPerServer;
2111    private final Set<byte[]> busyRegions = new TreeSet<>(Bytes.BYTES_COMPARATOR);
2112    private final AtomicLong tasksInProgress;
2113    TaskCountChecker(final int maxTotalConcurrentTasks,
2114      final int maxConcurrentTasksPerServer,
2115      final int maxConcurrentTasksPerRegion,
2116      final AtomicLong tasksInProgress,
2117      final Map<ServerName, AtomicInteger> taskCounterPerServer,
2118      final Map<byte[], AtomicInteger> taskCounterPerRegion) {
2119      this.maxTotalConcurrentTasks = maxTotalConcurrentTasks;
2120      this.maxConcurrentTasksPerRegion = maxConcurrentTasksPerRegion;
2121      this.maxConcurrentTasksPerServer = maxConcurrentTasksPerServer;
2122      this.taskCounterPerRegion = taskCounterPerRegion;
2123      this.taskCounterPerServer = taskCounterPerServer;
2124      this.tasksInProgress = tasksInProgress;
2125    }
2126    @Override
2127    public void reset() throws InterruptedIOException {
2128      // prevent the busy-waiting
2129      waitForRegion();
2130      regionsIncluded.clear();
2131      serversIncluded.clear();
2132      busyRegions.clear();
2133    }
2134    private void waitForRegion() throws InterruptedIOException {
2135      if (busyRegions.isEmpty()) {
2136        return;
2137      }
2138      EnvironmentEdge ee = EnvironmentEdgeManager.getDelegate();
2139      final long start = ee.currentTime();
2140      while ((ee.currentTime() - start) <= MAX_WAITING_TIME) {
2141        for (byte[] region : busyRegions) {
2142          AtomicInteger count = taskCounterPerRegion.get(region);
2143          if (count == null || count.get() < maxConcurrentTasksPerRegion) {
2144            return;
2145          }
2146        }
2147        try {
2148          synchronized (tasksInProgress) {
2149            tasksInProgress.wait(10);
2150          }
2151        } catch (InterruptedException e) {
2152          throw new InterruptedIOException("Interrupted." +
2153              " tasksInProgress=" + tasksInProgress);
2154        }
2155      }
2156    }
2157    /**
2158     * 1) check the regions is allowed.
2159     * 2) check the concurrent tasks for regions.
2160     * 3) check the total concurrent tasks.
2161     * 4) check the concurrent tasks for server.
2162     * @param loc
2163     * @param rowSize
2164     * @return
2165     */
2166    @Override
2167    public ReturnCode canTakeOperation(HRegionLocation loc, long rowSize) {
2168
2169      HRegionInfo regionInfo = loc.getRegionInfo();
2170      if (regionsIncluded.contains(regionInfo)) {
2171        // We already know what to do with this region.
2172        return ReturnCode.INCLUDE;
2173      }
2174      AtomicInteger regionCnt = taskCounterPerRegion.get(loc.getRegionInfo().getRegionName());
2175      if (regionCnt != null && regionCnt.get() >= maxConcurrentTasksPerRegion) {
2176        // Too many tasks on this region already.
2177        return ReturnCode.SKIP;
2178      }
2179      int newServers = serversIncluded.size()
2180        + (serversIncluded.contains(loc.getServerName()) ? 0 : 1);
2181      if ((newServers + tasksInProgress.get()) > maxTotalConcurrentTasks) {
2182        // Too many tasks.
2183        return ReturnCode.SKIP;
2184      }
2185      AtomicInteger serverCnt = taskCounterPerServer.get(loc.getServerName());
2186      if (serverCnt != null && serverCnt.get() >= maxConcurrentTasksPerServer) {
2187        // Too many tasks for this individual server
2188        return ReturnCode.SKIP;
2189      }
2190      return ReturnCode.INCLUDE;
2191    }
2192
2193    @Override
2194    public void notifyFinal(ReturnCode code, HRegionLocation loc, long rowSize) {
2195      if (code == ReturnCode.INCLUDE) {
2196        regionsIncluded.add(loc.getRegionInfo());
2197        serversIncluded.add(loc.getServerName());
2198      }
2199      busyRegions.add(loc.getRegionInfo().getRegionName());
2200    }
2201  }
