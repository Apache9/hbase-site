054/**
055 * Retry caller for scanning a region.
056 * <p>
057 * We will modify the {@link Scan} object passed in directly. The upper layer should store the
058 * reference of this object and use it to open new single region scanners.
059 */
060@InterfaceAudience.Private
061class AsyncScanSingleRegionRpcRetryingCaller {
062
063  private static final Log LOG = LogFactory.getLog(AsyncScanSingleRegionRpcRetryingCaller.class);
064
065  private final HashedWheelTimer retryTimer;
066
067  private final Scan scan;
068
069  private final long scannerId;
070
071  private final ScanResultCache resultCache;
072
073  private final RawScanResultConsumer consumer;
074
075  private final ClientService.Interface stub;
076
077  private final HRegionLocation loc;
078
079  private final long pauseNs;
080
081  private final int maxAttempts;
082
083  private final long scanTimeoutNs;
084
085  private final long rpcTimeoutNs;
086
087  private final int startLogErrorsCnt;
088
089  private final Runnable completeWhenNoMoreResultsInRegion;
090
091  private final CompletableFuture<Boolean> future;
092
093  private final HBaseRpcController controller;
094
095  private byte[] nextStartRowWhenError;
096
097  private boolean includeNextStartRowWhenError;
098
099  private long nextCallStartNs;
100
101  private int tries = 1;
102
103  private final List<RetriesExhaustedException.ThrowableWithExtraContext> exceptions;
104
105  private long nextCallSeq = -1L;
106
107  public AsyncScanSingleRegionRpcRetryingCaller(HashedWheelTimer retryTimer,
108      AsyncConnectionImpl conn, Scan scan, long scannerId, ScanResultCache resultCache,
109      RawScanResultConsumer consumer, Interface stub, HRegionLocation loc, long pauseNs,
110      int maxAttempts, long scanTimeoutNs, long rpcTimeoutNs, int startLogErrorsCnt) {
111    this.retryTimer = retryTimer;
112    this.scan = scan;
113    this.scannerId = scannerId;
114    this.resultCache = resultCache;
115    this.consumer = consumer;
116    this.stub = stub;
117    this.loc = loc;
118    this.pauseNs = pauseNs;
119    this.maxAttempts = maxAttempts;
120    this.scanTimeoutNs = scanTimeoutNs;
121    this.rpcTimeoutNs = rpcTimeoutNs;
122    this.startLogErrorsCnt = startLogErrorsCnt;
123    if (scan.isReversed()) {
124      completeWhenNoMoreResultsInRegion = this::completeReversedWhenNoMoreResultsInRegion;
125    } else {
126      completeWhenNoMoreResultsInRegion = this::completeWhenNoMoreResultsInRegion;
127    }
128    this.future = new CompletableFuture<>();
129    this.controller = conn.rpcControllerFactory.newController();
130    this.exceptions = new ArrayList<>();
131  }
132
133  private long elapsedMs() {
134    return TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - nextCallStartNs);
135  }
136
137  private long remainingTimeNs() {
138    return scanTimeoutNs - (System.nanoTime() - nextCallStartNs);
139  }
140
141  private void closeScanner() {
142    resetController(controller, rpcTimeoutNs);
143    ScanRequest req = RequestConverter.buildScanRequest(this.scannerId, 0, true, false);
144    stub.scan(controller, req, resp -> {
145      if (controller.failed()) {
146        LOG.warn("Call to " + loc.getServerName() + " for closing scanner id = " + scannerId
147            + " for " + loc.getRegionInfo().getEncodedName() + " of "
148            + loc.getRegionInfo().getTable() + " failed, ignore, probably already closed",
149          controller.getFailed());
150      }
151    });
152  }
153
154  private void completeExceptionally(boolean closeScanner) {
155    resultCache.clear();
156    if (closeScanner) {
157      closeScanner();
158    }
159    future.completeExceptionally(new RetriesExhaustedException(tries - 1, exceptions));
160  }
161
162  private void completeNoMoreResults() {
163    future.complete(false);
164  }
165
166  private void completeWithNextStartRow(byte[] row, boolean inclusive) {
167    scan.withStartRow(row, inclusive);
168    future.complete(true);
169  }
170
171  private void completeWhenError(boolean closeScanner) {
172    resultCache.clear();
173    if (closeScanner) {
174      closeScanner();
175    }
176    if (nextStartRowWhenError != null) {
177      scan.withStartRow(nextStartRowWhenError, includeNextStartRowWhenError);
178    }
179    future.complete(true);
180  }
181
182  private void onError(Throwable error) {
183    error = translateException(error);
184    if (tries > startLogErrorsCnt) {
185      LOG.warn("Call to " + loc.getServerName() + " for scanner id = " + scannerId + " for "
186          + loc.getRegionInfo().getEncodedName() + " of " + loc.getRegionInfo().getTable()
187          + " failed, , tries = " + tries + ", maxAttempts = " + maxAttempts + ", timeout = "
188          + TimeUnit.NANOSECONDS.toMillis(scanTimeoutNs) + " ms, time elapsed = " + elapsedMs()
189          + " ms",
190        error);
191    }
192    boolean scannerClosed =
193        error instanceof UnknownScannerException || error instanceof NotServingRegionException
194            || error instanceof RegionServerStoppedException;
195    RetriesExhaustedException.ThrowableWithExtraContext qt =
196        new RetriesExhaustedException.ThrowableWithExtraContext(error,
197            EnvironmentEdgeManager.currentTime(), "");
198    exceptions.add(qt);
199    if (tries >= maxAttempts) {
200      completeExceptionally(!scannerClosed);
201      return;
202    }
203    long delayNs;
204    if (scanTimeoutNs > 0) {
205      long maxDelayNs = remainingTimeNs() - SLEEP_DELTA_NS;
206      if (maxDelayNs <= 0) {
207        completeExceptionally(!scannerClosed);
208        return;
209      }
210      delayNs = Math.min(maxDelayNs, getPauseTime(pauseNs, tries - 1));
211    } else {
212      delayNs = getPauseTime(pauseNs, tries - 1);
213    }
214    if (scannerClosed) {
215      completeWhenError(false);
216      return;
217    }
218    if (error instanceof OutOfOrderScannerNextException || error instanceof ScannerResetException) {
219      completeWhenError(true);
220      return;
221    }
222    if (error instanceof DoNotRetryIOException) {
223      completeExceptionally(true);
224      return;
225    }
226    tries++;
227    retryTimer.newTimeout(t -> call(), delayNs, TimeUnit.NANOSECONDS);
228  }
229
230  private void updateNextStartRowWhenError(Result result) {
231    nextStartRowWhenError = result.getRow();
232    includeNextStartRowWhenError = scan.getBatch() > 0 || result.isPartial();
233  }
234
235  private void completeWhenNoMoreResultsInRegion() {
236    if (noMoreResultsForScan(scan, loc.getRegionInfo())) {
237      completeNoMoreResults();
238    } else {
239      completeWithNextStartRow(loc.getRegionInfo().getEndKey(), true);
240    }
241  }
242
243  private void completeReversedWhenNoMoreResultsInRegion() {
244    if (noMoreResultsForReverseScan(scan, loc.getRegionInfo())) {
245      completeNoMoreResults();
246    } else {
247      completeWithNextStartRow(loc.getRegionInfo().getStartKey(), false);
248    }
249  }
250
251  private void onComplete(HBaseRpcController controller, ScanResponse resp) {
252    if (controller.failed()) {
253      onError(controller.getFailed());
254      return;
255    }
256    boolean isHeartbeatMessage = resp.hasHeartbeatMessage() && resp.getHeartbeatMessage();
257    Result[] results;
258    try {
259      results = resultCache.addAndGet(
260        Optional.ofNullable(ResponseConverter.getResults(controller.cellScanner(), resp))
261            .orElse(ScanResultCache.EMPTY_RESULT_ARRAY),
262        isHeartbeatMessage);
263    } catch (IOException e) {
264      // We can not retry here. The server has responded normally and the call sequence has been
265      // increased so a new scan with the same call sequence will cause an
266      // OutOfOrderScannerNextException. Let the upper layer open a new scanner.
267      LOG.warn("decode scan response failed", e);
268      completeWhenError(true);
269      return;
270    }
271
272    boolean stopByUser;
273    if (results.length == 0) {
274      // if we have nothing to return then this must be a heartbeat message.
275      stopByUser = !consumer.onHeartbeat();
276    } else {
277      updateNextStartRowWhenError(results[results.length - 1]);
278      stopByUser = !consumer.onNext(results);
279    }
280    if (resp.hasMoreResults() && !resp.getMoreResults()) {
281      // RS tells us there is no more data for the whole scan
282      completeNoMoreResults();
283      return;
284    }
285    if (stopByUser) {
286      if (resp.getMoreResultsInRegion()) {
287        // we have more results in region but user request to stop the scan, so we need to close the
288        // scanner explicitly.
289        closeScanner();
290      }
291      completeNoMoreResults();
292      return;
293    }
294    if (scan.getLimit() > 0) {
295      // The RS should have set the moreResults field in ScanResponse to false when we have reached
296      // the limit.
297      int limit = scan.getLimit() - results.length;
298      assert limit > 0;
299      scan.setLimit(limit);
300    }
301    // as in 2.0 this value will always be set
302    if (!resp.getMoreResultsInRegion()) {
303      completeWhenNoMoreResultsInRegion.run();
304      return;
305    }
306    next();
307  }
308
309  private void call() {
310    // As we have a call sequence for scan, it is useless to have a different rpc timeout which is
311    // less than the scan timeout. If the server does not respond in time(usually this will not
312    // happen as we have heartbeat now), we will get an OutOfOrderScannerNextException when
313    // resending the next request and the only way to fix this is to close the scanner and open a
314    // new one.
315    long callTimeoutNs;
316    if (scanTimeoutNs > 0) {
317      long remainingNs = scanTimeoutNs - (System.nanoTime() - nextCallStartNs);
318      if (remainingNs <= 0) {
319        completeExceptionally(true);
320        return;
321      }
322      callTimeoutNs = remainingNs;
323    } else {
324      callTimeoutNs = 0L;
325    }
326    resetController(controller, callTimeoutNs);
327    ScanRequest req = RequestConverter.buildScanRequest(scannerId, scan.getCaching(), false,
328      nextCallSeq, false, false, scan.getLimit());
329    stub.scan(controller, req, resp -> onComplete(controller, resp));
330  }
331
332  private void next() {
333    nextCallSeq++;
334    tries = 0;
335    exceptions.clear();
336    nextCallStartNs = System.nanoTime();
337    call();
338  }
339
340  /**
341   * Now we will also fetch some cells along with the scanner id when opening a scanner, so we also
342   * need to process the ScanResponse for the open scanner request. The HBaseRpcController for the
343   * open scanner request is also needed because we may have some data in the CellScanner which is
344   * contained in the controller.
345   * @return {@code true} if we should continue, otherwise {@code false}.
346   */
347  public CompletableFuture<Boolean> start(HBaseRpcController controller,
348      ScanResponse respWhenOpen) {
349    onComplete(controller, respWhenOpen);
350    return future;
351  }
