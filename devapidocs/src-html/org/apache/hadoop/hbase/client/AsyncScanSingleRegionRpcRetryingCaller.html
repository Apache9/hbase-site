059/**
060 * Retry caller for scanning a region.
061 * <p>
062 * We will modify the {@link Scan} object passed in directly. The upper layer should store the
063 * reference of this object and use it to open new single region scanners.
064 */
065@InterfaceAudience.Private
066class AsyncScanSingleRegionRpcRetryingCaller {
067
068  private static final Log LOG = LogFactory.getLog(AsyncScanSingleRegionRpcRetryingCaller.class);
069
070  private final HashedWheelTimer retryTimer;
071
072  private final Scan scan;
073
074  private final long scannerId;
075
076  private final ScanResultCache resultCache;
077
078  private final ScanResultConsumer consumer;
079
080  private final ClientService.Interface stub;
081
082  private final HRegionLocation loc;
083
084  private final long pauseNs;
085
086  private final int maxAttempts;
087
088  private final long scanTimeoutNs;
089
090  private final long rpcTimeoutNs;
091
092  private final int startLogErrorsCnt;
093
094  private final Supplier<byte[]> createNextStartRowWhenError;
095
096  private final Runnable completeWhenNoMoreResultsInRegion;
097
098  private final CompletableFuture<Boolean> future;
099
100  private final HBaseRpcController controller;
101
102  private byte[] nextStartRowWhenError;
103
104  private boolean includeNextStartRowWhenError;
105
106  private long nextCallStartNs;
107
108  private int tries = 1;
109
110  private final List<RetriesExhaustedException.ThrowableWithExtraContext> exceptions;
111
112  private long nextCallSeq = -1L;
113
114  public AsyncScanSingleRegionRpcRetryingCaller(HashedWheelTimer retryTimer,
115      AsyncConnectionImpl conn, Scan scan, long scannerId, ScanResultCache resultCache,
116      ScanResultConsumer consumer, Interface stub, HRegionLocation loc, long pauseNs,
117      int maxRetries, long scanTimeoutNs, long rpcTimeoutNs, int startLogErrorsCnt) {
118    this.retryTimer = retryTimer;
119    this.scan = scan;
120    this.scannerId = scannerId;
121    this.resultCache = resultCache;
122    this.consumer = consumer;
123    this.stub = stub;
124    this.loc = loc;
125    this.pauseNs = pauseNs;
126    this.maxAttempts = retries2Attempts(maxRetries);
127    this.scanTimeoutNs = scanTimeoutNs;
128    this.rpcTimeoutNs = rpcTimeoutNs;
129    this.startLogErrorsCnt = startLogErrorsCnt;
130    if (scan.isReversed()) {
131      createNextStartRowWhenError = this::createReversedNextStartRowWhenError;
132      completeWhenNoMoreResultsInRegion = this::completeReversedWhenNoMoreResultsInRegion;
133    } else {
134      createNextStartRowWhenError = this::createNextStartRowWhenError;
135      completeWhenNoMoreResultsInRegion = this::completeWhenNoMoreResultsInRegion;
136    }
137    this.future = new CompletableFuture<>();
138    this.controller = conn.rpcControllerFactory.newController();
139    this.exceptions = new ArrayList<>();
140  }
141
142  private long elapsedMs() {
143    return TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - nextCallStartNs);
144  }
145
146  private void closeScanner() {
147    resetController(controller, rpcTimeoutNs);
148    ScanRequest req = RequestConverter.buildScanRequest(this.scannerId, 0, true, false);
149    stub.scan(controller, req, resp -> {
150      if (controller.failed()) {
151        LOG.warn("Call to " + loc.getServerName() + " for closing scanner id = " + scannerId
152            + " for " + loc.getRegionInfo().getEncodedName() + " of "
153            + loc.getRegionInfo().getTable() + " failed, ignore, probably already closed",
154          controller.getFailed());
155      }
156    });
157  }
158
159  private void completeExceptionally(boolean closeScanner) {
160    resultCache.clear();
161    if (closeScanner) {
162      closeScanner();
163    }
164    future.completeExceptionally(new RetriesExhaustedException(tries, exceptions));
165  }
166
167  @edu.umd.cs.findbugs.annotations.SuppressWarnings(value = "NP_NONNULL_PARAM_VIOLATION",
168      justification = "https://github.com/findbugsproject/findbugs/issues/79")
169  private void completeNoMoreResults() {
170    future.complete(null);
171  }
172
173  private void completeWithNextStartRow(byte[] nextStartRow) {
174    scan.setStartRow(nextStartRow);
175    future.complete(scan.isReversed());
176  }
177
178  private byte[] createNextStartRowWhenError() {
179    return createClosestRowAfter(nextStartRowWhenError);
180  }
181
182  private byte[] createReversedNextStartRowWhenError() {
183    return createClosestRowBefore(nextStartRowWhenError);
184  }
185
186  private void completeWhenError(boolean closeScanner) {
187    resultCache.clear();
188    if (closeScanner) {
189      closeScanner();
190    }
191    if (nextStartRowWhenError != null) {
192      scan.setStartRow(
193        includeNextStartRowWhenError ? nextStartRowWhenError : createNextStartRowWhenError.get());
194    }
195    future.complete(
196      scan.isReversed() && Bytes.equals(scan.getStartRow(), loc.getRegionInfo().getEndKey()));
197  }
198
199  private void onError(Throwable error) {
200    error = translateException(error);
201    if (tries > startLogErrorsCnt) {
202      LOG.warn("Call to " + loc.getServerName() + " for scanner id = " + scannerId + " for "
203          + loc.getRegionInfo().getEncodedName() + " of " + loc.getRegionInfo().getTable()
204          + " failed, , tries = " + tries + ", maxAttempts = " + maxAttempts + ", timeout = "
205          + TimeUnit.NANOSECONDS.toMillis(scanTimeoutNs) + " ms, time elapsed = " + elapsedMs()
206          + " ms",
207        error);
208    }
209    boolean scannerClosed =
210        error instanceof UnknownScannerException || error instanceof NotServingRegionException
211            || error instanceof RegionServerStoppedException;
212    RetriesExhaustedException.ThrowableWithExtraContext qt =
213        new RetriesExhaustedException.ThrowableWithExtraContext(error,
214            EnvironmentEdgeManager.currentTime(), "");
215    exceptions.add(qt);
216    if (tries >= maxAttempts) {
217      completeExceptionally(!scannerClosed);
218      return;
219    }
220    long delayNs;
221    if (scanTimeoutNs > 0) {
222      long maxDelayNs = scanTimeoutNs - (System.nanoTime() - nextCallStartNs);
223      if (maxDelayNs <= 0) {
224        completeExceptionally(!scannerClosed);
225        return;
226      }
227      delayNs = Math.min(maxDelayNs, getPauseTime(pauseNs, tries - 1));
228    } else {
229      delayNs = getPauseTime(pauseNs, tries - 1);
230    }
231    if (scannerClosed) {
232      completeWhenError(false);
233      return;
234    }
235    if (error instanceof OutOfOrderScannerNextException || error instanceof ScannerResetException) {
236      completeWhenError(true);
237      return;
238    }
239    if (error instanceof DoNotRetryIOException) {
240      completeExceptionally(true);
241      return;
242    }
243    tries++;
244    retryTimer.newTimeout(t -> call(), delayNs, TimeUnit.NANOSECONDS);
245  }
246
247  private void updateNextStartRowWhenError(Result result) {
248    nextStartRowWhenError = result.getRow();
249    includeNextStartRowWhenError = result.isPartial();
250  }
251
252  private void completeWhenNoMoreResultsInRegion() {
253    if (isEmptyStopRow(scan.getStopRow())) {
254      if (isEmptyStopRow(loc.getRegionInfo().getEndKey())) {
255        completeNoMoreResults();
256      }
257    } else {
258      if (Bytes.compareTo(loc.getRegionInfo().getEndKey(), scan.getStopRow()) >= 0) {
259        completeNoMoreResults();
260      }
261    }
262    completeWithNextStartRow(loc.getRegionInfo().getEndKey());
263  }
264
265  private void completeReversedWhenNoMoreResultsInRegion() {
266    if (isEmptyStopRow(scan.getStopRow())) {
267      if (isEmptyStartRow(loc.getRegionInfo().getStartKey())) {
268        completeNoMoreResults();
269      }
270    } else {
271      if (Bytes.compareTo(loc.getRegionInfo().getStartKey(), scan.getStopRow()) <= 0) {
272        completeNoMoreResults();
273      }
274    }
275    completeWithNextStartRow(loc.getRegionInfo().getStartKey());
276  }
277
278  private void onComplete(ScanResponse resp) {
279    if (controller.failed()) {
280      onError(controller.getFailed());
281      return;
282    }
283    boolean isHeartbeatMessage = resp.hasHeartbeatMessage() && resp.getHeartbeatMessage();
284    Result[] results;
285    try {
286      results = resultCache.addAndGet(
287        Optional.ofNullable(ResponseConverter.getResults(controller.cellScanner(), resp))
288            .orElse(ScanResultCache.EMPTY_RESULT_ARRAY),
289        isHeartbeatMessage);
290    } catch (IOException e) {
291      // We can not retry here. The server has responded normally and the call sequence has been
292      // increased so a new scan with the same call sequence will cause an
293      // OutOfOrderScannerNextException. Let the upper layer open a new scanner.
294      LOG.warn("decode scan response failed", e);
295      completeWhenError(true);
296      return;
297    }
298
299    boolean stopByUser;
300    if (results.length == 0) {
301      // if we have nothing to return then this must be a heartbeat message.
302      stopByUser = !consumer.onHeartbeat();
303    } else {
304      updateNextStartRowWhenError(results[results.length - 1]);
305      stopByUser = !consumer.onNext(results);
306    }
307    if (resp.hasMoreResults() && !resp.getMoreResults()) {
308      // RS tells us there is no more data for the whole scan
309      completeNoMoreResults();
310      return;
311    }
312    if (stopByUser) {
313      if (resp.getMoreResultsInRegion()) {
314        // we have more results in region but user request to stop the scan, so we need to close the
315        // scanner explicitly.
316        closeScanner();
317      }
318      completeNoMoreResults();
319      return;
320    }
321    // as in 2.0 this value will always be set
322    if (!resp.getMoreResultsInRegion()) {
323      completeWhenNoMoreResultsInRegion.run();
324      return;
325    }
326    next();
327  }
328
329  private void call() {
330    resetController(controller, rpcTimeoutNs);
331    ScanRequest req = RequestConverter.buildScanRequest(scannerId, scan.getCaching(), false,
332      nextCallSeq, false, false);
333    stub.scan(controller, req, this::onComplete);
334  }
335
336  private void next() {
337    nextCallSeq++;
338    tries = 0;
339    exceptions.clear();
340    nextCallStartNs = System.nanoTime();
341    call();
342  }
343
344  /**
345   * @return return locate direction for next open scanner call, or null if we should stop.
346   */
347  public CompletableFuture<Boolean> start() {
348    next();
349    return future;
350  }
