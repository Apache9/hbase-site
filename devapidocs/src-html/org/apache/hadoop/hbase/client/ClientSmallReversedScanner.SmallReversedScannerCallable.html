045/**
046 * <p>
047 * Client scanner for small reversed scan. Generally, only one RPC is called to fetch the
048 * scan results, unless the results cross multiple regions or the row count of
049 * results exceed the caching.
050 * </p>
051 * For small scan, it will get better performance than {@link ReversedClientScanner}
052 */
053@InterfaceAudience.Private
054public class ClientSmallReversedScanner extends ReversedClientScanner {
055  private static final Log LOG = LogFactory.getLog(ClientSmallReversedScanner.class);
056  private ScannerCallableWithReplicas smallReversedScanCallable = null;
057  private SmallReversedScannerCallableFactory callableFactory;

125  /**
126   * Gets a scanner for following scan. Move to next region or continue from the last result or
127   * start from the start row.
128   *
129   * @param nbRows
130   * @param done
131   *          true if Server-side says we're done scanning.
132   * @param currentRegionDone
133   *          true if scan is over on current region
134   * @return true if has next scanner
135   * @throws IOException
136   */
137  private boolean nextScanner(int nbRows, final boolean done,
138                              boolean currentRegionDone) throws IOException {
139    // Where to start the next getter
140    byte[] localStartKey;
141    int cacheNum = nbRows;
142    boolean regionChanged = true;
143    boolean isFirstRegionToLocate = false;
144    // if we're at end of table, close and return false to stop iterating
145    if (this.currentRegion != null && currentRegionDone) {
146      byte[] startKey = this.currentRegion.getStartKey();
147      if (startKey == null
148          || Bytes.equals(startKey, HConstants.EMPTY_BYTE_ARRAY)
149          || checkScanStopRow(startKey) || done) {
150        close();
151        if (LOG.isDebugEnabled()) {
152          LOG.debug("Finished with small scan at " + this.currentRegion);
153        }
154        return false;
155      }
156      // We take the row just under to get to the previous region.
157      localStartKey = createClosestRowBefore(startKey);
158      if (LOG.isDebugEnabled()) {
159        LOG.debug("Finished with region " + this.currentRegion);
160      }
161    } else if (this.lastResult != null) {
162      regionChanged = false;
163      localStartKey = createClosestRowBefore(lastResult.getRow());
164    } else {
165      isFirstRegionToLocate = true;
166      localStartKey = this.scan.getStartRow();
167    }
168
169    if (!isFirstRegionToLocate
170        && (localStartKey == null || Bytes.equals(localStartKey, HConstants.EMPTY_BYTE_ARRAY))) {
171      // when non-firstRegion & localStartKey is empty bytes, no more rowKey should scan.
172      // otherwise, maybe infinity results with RowKey=0x00 will return.
173      return false;
174    }
175
176    if (LOG.isTraceEnabled()) {
177      LOG.trace("Advancing internal small scanner to startKey at '"
178          + Bytes.toStringBinary(localStartKey) + "'");
179    }
180
181    smallReversedScanCallable =
182        callableFactory.getCallable(getConnection(), getTable(), scan, getScanMetrics(),
183          localStartKey, cacheNum, rpcControllerFactory, getPool(), getPrimaryOperationTimeout(),
184          getRetries(), getScannerTimeout(), getConf(), caller, isFirstRegionToLocate);
185
186    if (this.scanMetrics != null && regionChanged) {
187      this.scanMetrics.countOfRegions.incrementAndGet();
188    }
189    return true;
190  }
191
192  @Override
193  public Result next() throws IOException {
194    // If the scanner is closed and there's nothing left in the cache, next is a
195    // no-op.
196    if (cache.size() == 0 && this.closed) {
197      return null;
198    }
199    if (cache.size() == 0) {
200      loadCache();
201    }
202
203    if (cache.size() > 0) {
204      return cache.poll();
205    }
206    // if we exhausted this scanner before calling close, write out the scan
207    // metrics
208    writeScanMetrics();
209    return null;
210  }
211
212  @Override
213  protected void loadCache() throws IOException {
214    Result[] values = null;
215    long remainingResultSize = maxScannerResultSize;
216    int countdown = this.caching;
217    boolean currentRegionDone = false;
218    // Values == null means server-side filter has determined we must STOP
219    while (remainingResultSize > 0 && countdown > 0
220        && nextScanner(countdown, values == null, currentRegionDone)) {
221      // Server returns a null values if scanning is to stop. Else,
222      // returns an empty array if scanning is to go on and we've just
223      // exhausted current region.
224      // callWithoutRetries is at this layer. Within the ScannerCallableWithReplicas,
225      // we do a callWithRetries
226      values = this.caller.callWithoutRetries(smallReversedScanCallable, scannerTimeout);
227      this.currentRegion = smallReversedScanCallable.getHRegionInfo();
228      long currentTime = System.currentTimeMillis();
229      if (this.scanMetrics != null) {
230        this.scanMetrics.sumOfMillisSecBetweenNexts.addAndGet(currentTime
231            - lastNext);
232      }
233      lastNext = currentTime;
234      if (values != null && values.length > 0) {
235        for (int i = 0; i < values.length; i++) {
236          Result rs = values[i];
237          cache.add(rs);
238          // We don't make Iterator here
239          for (Cell cell : rs.rawCells()) {
240            remainingResultSize -= CellUtil.estimatedHeapSizeOf(cell);
241          }
242          countdown--;
243          this.lastResult = rs;
244        }
245      }
246      if (smallReversedScanCallable.hasMoreResultsContext()) {
247        currentRegionDone = !smallReversedScanCallable.getServerHasMoreResults();
248      } else {
249        currentRegionDone = countdown > 0;
250      }
251    }
252  }

254  @Override
255  protected void initializeScannerInConstruction() throws IOException {
256    // No need to initialize the scanner when constructing instance, do it when
257    // calling next(). Do nothing here.
258  }

260  @Override
261  public void close() {
262    if (!scanMetricsPublished) writeScanMetrics();
263    closed = true;
264  }

266  /**
267   * A reversed ScannerCallable which supports backward small scanning.
268   */
269  static class SmallReversedScannerCallable extends ReversedScannerCallable {
270
271    public SmallReversedScannerCallable(ClusterConnection connection, TableName table, Scan scan,
272        ScanMetrics scanMetrics, byte[] locateStartRow, RpcControllerFactory controllerFactory,
273        int caching, int replicaId) {
274      super(connection, table, scan, scanMetrics, locateStartRow, controllerFactory, replicaId);
275      this.setCaching(caching);
276    }
277
278    @Override
279    protected Result[] rpcCall() throws Exception {
280      if (this.closed) return null;
281      if (Thread.interrupted()) {
282        throw new InterruptedIOException();
283      }
284      ClientProtos.ScanRequest request = RequestConverter.buildScanRequest(
285        getLocation().getRegionInfo().getRegionName(), getScan(), getCaching(), true);
286      ClientProtos.ScanResponse response = null;
287      response = getStub().scan(getRpcController(), request);
288      Result[] results = ResponseConverter.getResults(getRpcControllerCellScanner(), response);
289      if (response.hasMoreResultsInRegion()) {
290        setHasMoreResultsContext(true);
291        setServerHasMoreResults(response.getMoreResultsInRegion());
292      } else {
293        setHasMoreResultsContext(false);
294      }
295      // We need to update result metrics since we are overriding call()
296      updateResultsMetrics(results);
297      return results;
298    }
299
300    @Override
301    public ScannerCallable getScannerCallableForReplica(int id) {
302      return new SmallReversedScannerCallable(getConnection(), getTableName(), getScan(),
303          scanMetrics, locateStartRow, rpcControllerFactory, getCaching(), id);
304    }
305  }
306
307  @VisibleForTesting
308  protected void setScannerCallableFactory(SmallReversedScannerCallableFactory callableFactory) {
309    this.callableFactory = callableFactory;
310  }
311
312  protected static class SmallReversedScannerCallableFactory {
313
314    public ScannerCallableWithReplicas getCallable(ClusterConnection connection, TableName table,
315        Scan scan, ScanMetrics scanMetrics, byte[] localStartKey, int cacheNum,
316        RpcControllerFactory controllerFactory, ExecutorService pool, int primaryOperationTimeout,
317        int retries, int scannerTimeout, Configuration conf, RpcRetryingCaller<Result[]> caller,
318        boolean isFirstRegionToLocate) {
319      byte[] locateStartRow = null;
320      if (isFirstRegionToLocate
321          && (localStartKey == null || Bytes.equals(localStartKey, HConstants.EMPTY_BYTE_ARRAY))) {
322        // HBASE-16886: if not setting startRow, then we will use a range [MAX_BYTE_ARRAY, +oo) to
323        // locate a region list, and the last one in region list is the region where our scan start.
324        locateStartRow = ConnectionUtils.MAX_BYTE_ARRAY;
325      }
326
327      scan.setStartRow(localStartKey);
328      SmallReversedScannerCallable s = new SmallReversedScannerCallable(connection, table, scan,
329          scanMetrics, locateStartRow, controllerFactory, cacheNum, 0);
330      ScannerCallableWithReplicas scannerCallableWithReplicas =
331          new ScannerCallableWithReplicas(table, connection, s, pool, primaryOperationTimeout, scan,
332              retries, scannerTimeout, cacheNum, conf, caller);
333      return scannerCallableWithReplicas;
334    }
335  }
