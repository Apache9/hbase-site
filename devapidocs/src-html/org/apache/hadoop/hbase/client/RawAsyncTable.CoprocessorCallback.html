018package org.apache.hadoop.hbase.client;
019
020import com.google.protobuf.RpcCallback;
021import com.google.protobuf.RpcChannel;
022import com.google.protobuf.RpcController;
023
024import java.util.concurrent.CompletableFuture;
025import java.util.function.Function;
026
027import org.apache.hadoop.hbase.HRegionInfo;
028import org.apache.yetus.audience.InterfaceAudience;
029
030/**
031 * A low level asynchronous table.
032 * <p>
033 * The implementation is required to be thread safe.
034 * <p>
035 * The returned {@code CompletableFuture} will be finished directly in the rpc framework's callback
036 * thread, so typically you should not do any time consuming work inside these methods, otherwise
037 * you will be likely to block at least one connection to RS(even more if the rpc framework uses
038 * NIO).
039 * <p>
040 * So, only experts that want to build high performance service should use this interface directly,
041 * especially for the {@link #scan(Scan, RawScanResultConsumer)} below.
042 * <p>
043 * TODO: For now the only difference between this interface and {@link AsyncTable} is the scan
044 * method. The {@link RawScanResultConsumer} exposes the implementation details of a scan(heartbeat)
045 * so it is not suitable for a normal user. If it is still the only difference after we implement
046 * most features of AsyncTable, we can think about merge these two interfaces.
047 * @since 2.0.0
048 */
049@InterfaceAudience.Public
050public interface RawAsyncTable extends AsyncTableBase {
051
052  /**
053   * The basic scan API uses the observer pattern. All results that match the given scan object will
054   * be passed to the given {@code consumer} by calling {@code RawScanResultConsumer.onNext}.
055   * {@code RawScanResultConsumer.onComplete} means the scan is finished, and
056   * {@code RawScanResultConsumer.onError} means we hit an unrecoverable error and the scan is
057   * terminated. {@code RawScanResultConsumer.onHeartbeat} means the RS is still working but we can
058   * not get a valid result to call {@code RawScanResultConsumer.onNext}. This is usually because
059   * the matched results are too sparse, for example, a filter which almost filters out everything
060   * is specified.
061   * <p>
062   * Notice that, the methods of the given {@code consumer} will be called directly in the rpc
063   * framework's callback thread, so typically you should not do any time consuming work inside
064   * these methods, otherwise you will be likely to block at least one connection to RS(even more if
065   * the rpc framework uses NIO).
066   * @param scan A configured {@link Scan} object.
067   * @param consumer the consumer used to receive results.
068   */
069  void scan(Scan scan, RawScanResultConsumer consumer);
070
071  /**
072   * Delegate to a protobuf rpc call.
073   * <p>
074   * Usually, it is just a simple lambda expression, like:
075   *
076   * <pre>
077   * <code>
078   * (stub, controller, rpcCallback) -> {
079   *   XXXRequest request = ...; // prepare the request
080   *   stub.xxx(controller, request, rpcCallback);
081   * }
082   * </code>
083   * </pre>
084   *
085   * And if you can prepare the {@code request} before calling the coprocessorService method, the
086   * lambda expression will be:
087   *
088   * <pre>
089   * <code>
090   * (stub, controller, rpcCallback) -> stub.xxx(controller, request, rpcCallback)
091   * </code>
092   * </pre>
093   */
094  @InterfaceAudience.Public
095  @FunctionalInterface
096  interface CoprocessorCallable<S, R> {
097
098    /**
099     * Represent the actual protobuf rpc call.
100     * @param stub the asynchronous stub
101     * @param controller the rpc controller, has already been prepared for you
102     * @param rpcCallback the rpc callback, has already been prepared for you
103     */
104    void call(S stub, RpcController controller, RpcCallback<R> rpcCallback);
105  }
106
107  /**
108   * Execute the given coprocessor call on the region which contains the given {@code row}.
109   * <p>
110   * The {@code stubMaker} is just a delegation to the {@code newStub} call. Usually it is only a
111   * one line lambda expression, like:
112   *
113   * <pre>
114   * <code>
115   * channel -> xxxService.newStub(channel)
116   * </code>
117   * </pre>
118   *
119   * @param stubMaker a delegation to the actual {@code newStub} call.
120   * @param callable a delegation to the actual protobuf rpc call. See the comment of
121   *          {@link CoprocessorCallable} for more details.
122   * @param row The row key used to identify the remote region location
123   * @param <S> the type of the asynchronous stub
124   * @param <R> the type of the return value
125   * @return the return value of the protobuf rpc call, wrapped by a {@link CompletableFuture}.
126   * @see CoprocessorCallable
127   */
128  <S, R> CompletableFuture<R> coprocessorService(Function<RpcChannel, S> stubMaker,
129      CoprocessorCallable<S, R> callable, byte[] row);
130
131  /**
132   * The callback when we want to execute a coprocessor call on a range of regions.
133   * <p>
134   * As the locating itself also takes some time, the implementation may want to send rpc calls on
135   * the fly, which means we do not know how many regions we have when we get the return value of
136   * the rpc calls, so we need an {@link #onComplete()} which is used to tell you that we have
137   * passed all the return values to you(through the {@link #onRegionComplete(HRegionInfo, Object)}
138   * or {@link #onRegionError(HRegionInfo, Throwable)} calls), i.e, there will be no
139   * {@link #onRegionComplete(HRegionInfo, Object)} or
140   * {@link #onRegionError(HRegionInfo, Throwable)} calls in the future.
141   * <p>
142   * Here is a pseudo code to describe a typical implementation of a range coprocessor service
143   * method to help you better understand how the {@link CoprocessorCallback} will be called. The
144   * {@code callback} in the pseudo code is our {@link CoprocessorCallback}. And notice that the
145   * {@code whenComplete} is {@code CompletableFuture.whenComplete}.
146   *
147   * <pre>
148   * locateThenCall(byte[] row) {
149   *   locate(row).whenComplete((location, locateError) -> {
150   *     if (locateError != null) {
151   *       callback.onError(locateError);
152   *       return;
153   *     }
154   *     incPendingCall();
155   *     region = location.getRegion();
156   *     if (region.getEndKey() > endKey) {
157   *       locateEnd = true;
158   *     } else {
159   *       locateThenCall(region.getEndKey());
160   *     }
161   *     sendCall().whenComplete((resp, error) -> {
162   *       if (error != null) {
163   *         callback.onRegionError(region, error);
164   *       } else {
165   *         callback.onRegionComplete(region, resp);
166   *       }
167   *       if (locateEnd && decPendingCallAndGet() == 0) {
168   *         callback.onComplete();
169   *       }
170   *     });
171   *   });
172   * }
173   * </pre>
174   */
175  @InterfaceAudience.Public
176  interface CoprocessorCallback<R> {
177
178    /**
179     * @param region the region that the response belongs to
180     * @param resp the response of the coprocessor call
181     */
182    void onRegionComplete(HRegionInfo region, R resp);
183
184    /**
185     * @param region the region that the error belongs to
186     * @param error the response error of the coprocessor call
187     */
188    void onRegionError(HRegionInfo region, Throwable error);
189
190    /**
191     * Indicate that all responses of the regions have been notified by calling
192     * {@link #onRegionComplete(HRegionInfo, Object)} or
193     * {@link #onRegionError(HRegionInfo, Throwable)}.
194     */
195    void onComplete();
196
197    /**
198     * Indicate that we got an error which does not belong to any regions. Usually a locating error.
199     */
200    void onError(Throwable error);
201  }
202
203  /**
204   * Execute the given coprocessor call on the regions which are covered by the range from
205   * {@code startKey} inclusive and {@code endKey} exclusive. See the comment of
206   * {@link #coprocessorService(Function, CoprocessorCallable, byte[], boolean, byte[], boolean, CoprocessorCallback)}
207   * for more details.
208   * @see #coprocessorService(Function, CoprocessorCallable, byte[], boolean, byte[], boolean,
209   *      CoprocessorCallback)
210   */
211  default <S, R> void coprocessorService(Function<RpcChannel, S> stubMaker,
212      CoprocessorCallable<S, R> callable, byte[] startKey, byte[] endKey,
213      CoprocessorCallback<R> callback) {
214    coprocessorService(stubMaker, callable, startKey, true, endKey, false, callback);
215  }
216
217  /**
218   * Execute the given coprocessor call on the regions which are covered by the range from
219   * {@code startKey} and {@code endKey}. The inclusive of boundaries are specified by
220   * {@code startKeyInclusive} and {@code endKeyInclusive}. The {@code stubMaker} is just a
221   * delegation to the {@code xxxService.newStub} call. Usually it is only a one line lambda
222   * expression, like:
223   *
224   * <pre>
225   * <code>
226   * channel -> xxxService.newStub(channel)
227   * </code>
228   * </pre>
229   *
230   * @param stubMaker a delegation to the actual {@code newStub} call.
231   * @param callable a delegation to the actual protobuf rpc call. See the comment of
232   *          {@link CoprocessorCallable} for more details.
233   * @param startKey start region selection with region containing this row. If {@code null}, the
234   *          selection will start with the first table region.
235   * @param startKeyInclusive whether to include the startKey
236   * @param endKey select regions up to and including the region containing this row. If
237   *          {@code null}, selection will continue through the last table region.
238   * @param endKeyInclusive whether to include the endKey
239   * @param callback callback to get the response. See the comment of {@link CoprocessorCallback}
240   *          for more details.
241   * @param <S> the type of the asynchronous stub
242   * @param <R> the type of the return value
243   * @see CoprocessorCallable
244   * @see CoprocessorCallback
245   */
246  <S, R> void coprocessorService(Function<RpcChannel, S> stubMaker,
247      CoprocessorCallable<S, R> callable, byte[] startKey, boolean startKeyInclusive, byte[] endKey,
248      boolean endKeyInclusive, CoprocessorCallback<R> callback);
249}
