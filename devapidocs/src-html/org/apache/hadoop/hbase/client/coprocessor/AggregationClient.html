063/**
064 * This client class is for invoking the aggregate functions deployed on the
065 * Region Server side via the AggregateService. This class will implement the
066 * supporting functionality for summing/processing the individual results
067 * obtained from the AggregateService for each region.
068 * <p>
069 * This will serve as the client side handler for invoking the aggregate
070 * functions.
071 * For all aggregate functions,
072 * <ul>
073 * <li>start row &lt; end row is an essential condition (if they are not
074 * {@link HConstants#EMPTY_BYTE_ARRAY})
075 * <li>Column family can't be null. In case where multiple families are
076 * provided, an IOException will be thrown. An optional column qualifier can
077 * also be defined.</li>
078 * <li>For methods to find maximum, minimum, sum, rowcount, it returns the
079 * parameter type. For average and std, it returns a double value. For row
080 * count, it returns a long value.</li>
081 * </ul>
082 * <p>Call {@link #close()} when done.
083 */
084@InterfaceAudience.Private
085public class AggregationClient implements Closeable {

130  /**
131   * It gives the maximum value of a column for a given column family for the
132   * given range. In case qualifier is null, a max of all values for the given
133   * family is returned.
134   * @param table
135   * @param ci
136   * @param scan
137   * @return max val &lt;&gt;
138   * @throws Throwable
139   *           The caller is supposed to handle the exception as they are thrown
140   *           &amp; propagated to it.
141   */
142  public <R, S, P extends Message, Q extends Message, T extends Message>
143  R max(final Table table, final ColumnInterpreter<R, S, P, Q, T> ci,
144      final Scan scan) throws Throwable {
145    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
146    class MaxCallBack implements Batch.Callback<R> {
147      R max = null;
148
149      R getMax() {
150        return max;
151      }
152
153      @Override
154      public synchronized void update(byte[] region, byte[] row, R result) {
155        max = (max == null || (result != null && ci.compare(max, result) < 0)) ? result : max;
156      }
157    }
158    MaxCallBack aMaxCallBack = new MaxCallBack();
159    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
160        new Batch.Call<AggregateService, R>() {
161          @Override
162          public R call(AggregateService instance) throws IOException {
163            ServerRpcController controller = new ServerRpcController();
164            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
165                new CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse>();
166            instance.getMax(controller, requestArg, rpcCallback);
167            AggregateResponse response = rpcCallback.get();
168            if (controller.failedOnException()) {
169              throw controller.getFailedOn();
170            }
171            if (response.getFirstPartCount() > 0) {
172              ByteString b = response.getFirstPart(0);
173              Q q = getParsedGenericInstance(ci.getClass(), 3, b);
174              return ci.getCellValueFromProto(q);
175            }
176            return null;
177          }
178        }, aMaxCallBack);
179    return aMaxCallBack.getMax();
180  }

182  /*
183   * @param scan
184   * @param canFamilyBeAbsent whether column family can be absent in familyMap of scan
185   */
186  private void validateParameters(Scan scan, boolean canFamilyBeAbsent) throws IOException {
187    if (scan == null
188        || (Bytes.equals(scan.getStartRow(), scan.getStopRow()) && !Bytes.equals(
189          scan.getStartRow(), HConstants.EMPTY_START_ROW))
190        || ((Bytes.compareTo(scan.getStartRow(), scan.getStopRow()) > 0) && !Bytes.equals(
191          scan.getStopRow(), HConstants.EMPTY_END_ROW))) {
192      throw new IOException("Agg client Exception: Startrow should be smaller than Stoprow");
193    } else if (!canFamilyBeAbsent) {
194      if (scan.getFamilyMap().size() != 1) {
195        throw new IOException("There must be only one family.");
196      }
197    }
198  }

305  public <R, S, P extends Message, Q extends Message, T extends Message>
306  long rowCount(final Table table,
307      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
308    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, true);
309    class RowNumCallback implements Batch.Callback<Long> {

318        rowCountL.addAndGet(result.longValue());
319      }
320    }
321    RowNumCallback rowNum = new RowNumCallback();
322    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
323        new Batch.Call<AggregateService, Long>() {

325          public Long call(AggregateService instance) throws IOException {

327            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
328                new CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse>();
329            instance.getRowNum(controller, requestArg, rpcCallback);
330            AggregateResponse response = rpcCallback.get();
331            if (controller.failedOnException()) {
332              throw controller.getFailedOn();
333            }
334            byte[] bytes = getBytesFromResponse(response.getFirstPart(0));
335            ByteBuffer bb = ByteBuffer.allocate(8).put(bytes);

338          }
339        }, rowNum);

428  /**
429   * It computes average while fetching sum and row count from all the
430   * corresponding regions. Approach is to compute a global sum of region level
431   * sum and rowcount and then compute the average.
432   * @param table
433   * @param scan
434   * @throws Throwable
435   */
436  private <R, S, P extends Message, Q extends Message, T extends Message>
437  Pair<S, Long> getAvgArgs(final Table table,
438      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
439    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
440    class AvgCallBack implements Batch.Callback<Pair<S, Long>> {
441      S sum = null;
442      Long rowCount = 0l;
443
444      public synchronized Pair<S, Long> getAvgArgs() {
445        return new Pair<S, Long>(sum, rowCount);
446      }
447
448      @Override
449      public synchronized void update(byte[] region, byte[] row, Pair<S, Long> result) {
450        sum = ci.add(sum, result.getFirst());
451        rowCount += result.getSecond();
452      }
453    }
454    AvgCallBack avgCallBack = new AvgCallBack();
455    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
456        new Batch.Call<AggregateService, Pair<S, Long>>() {
457          @Override
458          public Pair<S, Long> call(AggregateService instance) throws IOException {
459            ServerRpcController controller = new ServerRpcController();
460            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
461                new CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse>();
462            instance.getAvg(controller, requestArg, rpcCallback);
463            AggregateResponse response = rpcCallback.get();
464            if (controller.failedOnException()) {
465              throw controller.getFailedOn();
466            }
467            Pair<S, Long> pair = new Pair<S, Long>(null, 0L);
468            if (response.getFirstPartCount() == 0) {
469              return pair;
470            }
471            ByteString b = response.getFirstPart(0);
472            T t = getParsedGenericInstance(ci.getClass(), 4, b);
473            S s = ci.getPromotedValueFromProto(t);
474            pair.setFirst(s);
475            ByteBuffer bb = ByteBuffer.allocate(8).put(
476                getBytesFromResponse(response.getSecondPart()));
477            bb.rewind();
478            pair.setSecond(bb.getLong());
479            return pair;
480          }
481        }, avgCallBack);
482    return avgCallBack.getAvgArgs();
483  }

522  /**
523   * It computes a global standard deviation for a given column and its value.
524   * Standard deviation is square root of (average of squares -
525   * average*average). From individual regions, it obtains sum, square sum and
526   * number of rows. With these, the above values are computed to get the global
527   * std.
528   * @param table
529   * @param scan
530   * @return standard deviations
531   * @throws Throwable
532   */
533  private <R, S, P extends Message, Q extends Message, T extends Message>
534  Pair<List<S>, Long> getStdArgs(final Table table,
535      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
536    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
537    class StdCallback implements Batch.Callback<Pair<List<S>, Long>> {
538      long rowCountVal = 0l;
539      S sumVal = null, sumSqVal = null;
540
541      public synchronized Pair<List<S>, Long> getStdParams() {
542        List<S> l = new ArrayList<S>();
543        l.add(sumVal);
544        l.add(sumSqVal);
545        Pair<List<S>, Long> p = new Pair<List<S>, Long>(l, rowCountVal);
546        return p;
547      }
548
549      @Override
550      public synchronized void update(byte[] region, byte[] row, Pair<List<S>, Long> result) {
551        if (result.getFirst().size() > 0) {
552          sumVal = ci.add(sumVal, result.getFirst().get(0));
553          sumSqVal = ci.add(sumSqVal, result.getFirst().get(1));
554          rowCountVal += result.getSecond();
555        }
556      }
557    }
558    StdCallback stdCallback = new StdCallback();
559    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
560        new Batch.Call<AggregateService, Pair<List<S>, Long>>() {
561          @Override
562          public Pair<List<S>, Long> call(AggregateService instance) throws IOException {
563            ServerRpcController controller = new ServerRpcController();
564            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
565                new CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse>();
566            instance.getStd(controller, requestArg, rpcCallback);
567            AggregateResponse response = rpcCallback.get();
568            if (controller.failedOnException()) {
569              throw controller.getFailedOn();
570            }
571            Pair<List<S>, Long> pair = new Pair<List<S>, Long>(new ArrayList<S>(), 0L);
572            if (response.getFirstPartCount() == 0) {
573              return pair;
574            }
575            List<S> list = new ArrayList<S>();
576            for (int i = 0; i < response.getFirstPartCount(); i++) {
577              ByteString b = response.getFirstPart(i);
578              T t = getParsedGenericInstance(ci.getClass(), 4, b);
579              S s = ci.getPromotedValueFromProto(t);
580              list.add(s);
581            }
582            pair.setFirst(list);
583            ByteBuffer bb = ByteBuffer.allocate(8).put(
584                getBytesFromResponse(response.getSecondPart()));
585            bb.rewind();
586            pair.setSecond(bb.getLong());
587            return pair;
588          }
589        }, stdCallback);
590    return stdCallback.getStdParams();
591  }

613  /**
614   * This is the client side interface/handle for calling the std method for a
615   * given cf-cq combination. It was necessary to add one more call stack as its
616   * return type should be a decimal value, irrespective of what
617   * columninterpreter says. So, this methods collects the necessary parameters
618   * to compute the std and returns the double value.
619   * @param table
620   * @param ci
621   * @param scan
622   * @return &lt;R, S&gt;
623   * @throws Throwable
624   */
625  public <R, S, P extends Message, Q extends Message, T extends Message> double std(
626      final Table table, ColumnInterpreter<R, S, P, Q, T> ci, Scan scan) throws Throwable {
627    Pair<List<S>, Long> p = getStdArgs(table, ci, scan);
628    double res = 0d;
629    double avg = ci.divideForAvg(p.getFirst().get(0), p.getSecond());
630    double avgOfSumSq = ci.divideForAvg(p.getFirst().get(1), p.getSecond());
631    res = avgOfSumSq - (avg) * (avg); // variance
632    res = Math.pow(res, 0.5);
633    return res;
634  }

636  /**
637   * It helps locate the region with median for a given column whose weight
638   * is specified in an optional column.
639   * From individual regions, it obtains sum of values and sum of weights.
640   * @param table
641   * @param ci
642   * @param scan
643   * @return pair whose first element is a map between start row of the region
644   *  and (sum of values, sum of weights) for the region, the second element is
645   *  (sum of values, sum of weights) for all the regions chosen
646   * @throws Throwable
647   */
648  private <R, S, P extends Message, Q extends Message, T extends Message>
649  Pair<NavigableMap<byte[], List<S>>, List<S>>
650  getMedianArgs(final Table table,
651      final ColumnInterpreter<R, S, P, Q, T> ci, final Scan scan) throws Throwable {
652    final AggregateRequest requestArg = validateArgAndGetPB(scan, ci, false);
653    final NavigableMap<byte[], List<S>> map =
654      new TreeMap<byte[], List<S>>(Bytes.BYTES_COMPARATOR);
655    class StdCallback implements Batch.Callback<List<S>> {
656      S sumVal = null, sumWeights = null;
657
658      public synchronized Pair<NavigableMap<byte[], List<S>>, List<S>> getMedianParams() {
659        List<S> l = new ArrayList<S>();
660        l.add(sumVal);
661        l.add(sumWeights);
662        Pair<NavigableMap<byte[], List<S>>, List<S>> p =
663          new Pair<NavigableMap<byte[], List<S>>, List<S>>(map, l);
664        return p;
665      }
666
667      @Override
668      public synchronized void update(byte[] region, byte[] row, List<S> result) {
669        map.put(row, result);
670        sumVal = ci.add(sumVal, result.get(0));
671        sumWeights = ci.add(sumWeights, result.get(1));
672      }
673    }
674    StdCallback stdCallback = new StdCallback();
675    table.coprocessorService(AggregateService.class, scan.getStartRow(), scan.getStopRow(),
676        new Batch.Call<AggregateService, List<S>>() {
677          @Override
678          public List<S> call(AggregateService instance) throws IOException {
679            ServerRpcController controller = new ServerRpcController();
680            CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse> rpcCallback =
681                new CoprocessorRpcUtils.BlockingRpcCallback<AggregateResponse>();
682            instance.getMedian(controller, requestArg, rpcCallback);
683            AggregateResponse response = rpcCallback.get();
684            if (controller.failedOnException()) {
685              throw controller.getFailedOn();
686            }
687
688            List<S> list = new ArrayList<S>();
689            for (int i = 0; i < response.getFirstPartCount(); i++) {
690              ByteString b = response.getFirstPart(i);
691              T t = getParsedGenericInstance(ci.getClass(), 4, b);
692              S s = ci.getPromotedValueFromProto(t);
693              list.add(s);
694            }
695            return list;
696          }
697
698        }, stdCallback);
699    return stdCallback.getMedianParams();
700  }

720  /**
721   * This is the client side interface/handler for calling the median method for a
722   * given cf-cq combination. This method collects the necessary parameters
723   * to compute the median and returns the median.
724   * @param table
725   * @param ci
726   * @param scan
727   * @return R the median
728   * @throws Throwable
729   */
730  public <R, S, P extends Message, Q extends Message, T extends Message>
731  R median(final Table table, ColumnInterpreter<R, S, P, Q, T> ci,
732      Scan scan) throws Throwable {
733    Pair<NavigableMap<byte[], List<S>>, List<S>> p = getMedianArgs(table, ci, scan);
734    byte[] startRow = null;
735    byte[] colFamily = scan.getFamilies()[0];
736    NavigableSet<byte[]> quals = scan.getFamilyMap().get(colFamily);
737    NavigableMap<byte[], List<S>> map = p.getFirst();
738    S sumVal = p.getSecond().get(0);
739    S sumWeights = p.getSecond().get(1);
740    double halfSumVal = ci.divideForAvg(sumVal, 2L);
741    double movingSumVal = 0;
742    boolean weighted = false;
743    if (quals.size() > 1) {
744      weighted = true;
745      halfSumVal = ci.divideForAvg(sumWeights, 2L);
746    }
747
748    for (Map.Entry<byte[], List<S>> entry : map.entrySet()) {
749      S s = weighted ? entry.getValue().get(1) : entry.getValue().get(0);
750      double newSumVal = movingSumVal + ci.divideForAvg(s, 1L);
751      if (newSumVal > halfSumVal) break;  // we found the region with the median
752      movingSumVal = newSumVal;
753      startRow = entry.getKey();
754    }
755    // scan the region with median and find it
756    Scan scan2 = new Scan(scan);
757    // inherit stop row from method parameter
758    if (startRow != null) scan2.setStartRow(startRow);
759    ResultScanner scanner = null;
760    try {
761      int cacheSize = scan2.getCaching();
762      if (!scan2.getCacheBlocks() || scan2.getCaching() < 2) {
763        scan2.setCacheBlocks(true);
764        cacheSize = 5;
765        scan2.setCaching(cacheSize);
766      }
767      scanner = table.getScanner(scan2);
768      Result[] results = null;
769      byte[] qualifier = quals.pollFirst();
770      // qualifier for the weight column
771      byte[] weightQualifier = weighted ? quals.pollLast() : qualifier;
772      R value = null;
773      do {
774        results = scanner.next(cacheSize);
775        if (results != null && results.length > 0) {
776          for (int i = 0; i < results.length; i++) {
777            Result r = results[i];
778            // retrieve weight
779            Cell kv = r.getColumnLatestCell(colFamily, weightQualifier);
780            R newValue = ci.getValue(colFamily, weightQualifier, kv);
781            S s = ci.castToReturnType(newValue);
782            double newSumVal = movingSumVal + ci.divideForAvg(s, 1L);
783            // see if we have moved past the median
784            if (newSumVal > halfSumVal) {
785              return value;
786            }
787            movingSumVal = newSumVal;
788            kv = r.getColumnLatestCell(colFamily, qualifier);
789            value = ci.getValue(colFamily, qualifier, kv);
790            }
791          }
792      } while (results != null && results.length > 0);
793    } finally {
794      if (scanner != null) {
795        scanner.close();
796      }
797    }
798    return null;
799  }

801  <R, S, P extends Message, Q extends Message, T extends Message> AggregateRequest
802  validateArgAndGetPB(Scan scan, ColumnInterpreter<R,S,P,Q,T> ci, boolean canFamilyBeAbsent)
803      throws IOException {
804    validateParameters(scan, canFamilyBeAbsent);
805    final AggregateRequest.Builder requestBuilder =
806        AggregateRequest.newBuilder();
807    requestBuilder.setInterpreterClassName(ci.getClass().getCanonicalName());

810       != null) {
811      requestBuilder.setInterpreterSpecificBytes(columnInterpreterSpecificData.toByteString());
812    }
813    requestBuilder.setScan(ProtobufUtil.toScan(scan));
814    return requestBuilder.build();
815  }

829  /**
830   * Get an instance of the argument type declared in a class's signature. The
831   * argument type is assumed to be a PB Message subclass, and the instance is
832   * created using parseFrom method on the passed ByteString.
833   * @param runtimeClass the runtime type of the class
834   * @param position the position of the argument in the class declaration
835   * @param b the ByteString which should be parsed to get the instance created
836   * @return the instance
837   * @throws IOException
838   */
839  @SuppressWarnings("unchecked")
840  // Used server-side too by Aggregation Coprocesor Endpoint. Undo this interdependence. TODO.
841  public static <T extends Message>
842  T getParsedGenericInstance(Class<?> runtimeClass, int position, ByteString b)
843      throws IOException {
844    Type type = runtimeClass.getGenericSuperclass();
845    Type argType = ((ParameterizedType)type).getActualTypeArguments()[position];
846    Class<T> classType = (Class<T>)argType;
847    T inst;
848    try {
849      Method m = classType.getMethod("parseFrom", ByteString.class);
850      inst = (T)m.invoke(null, b);
851      return inst;
852    } catch (SecurityException e) {
853      throw new IOException(e);
854    } catch (NoSuchMethodException e) {
855      throw new IOException(e);
856    } catch (IllegalArgumentException e) {
857      throw new IOException(e);
858    } catch (InvocationTargetException e) {
859      throw new IOException(e);
860    } catch (IllegalAccessException e) {
861      throw new IOException(e);
862    }
863  }
