070  /**
071   * Calculate pause time. Built on {@link HConstants#RETRY_BACKOFF}.
072   * @param pause time to pause
073   * @param tries amount of tries
074   * @return How long to wait after <code>tries</code> retries
075   */
076  public static long getPauseTime(final long pause, final int tries) {
077    int ntries = tries;
078    if (ntries >= HConstants.RETRY_BACKOFF.length) {
079      ntries = HConstants.RETRY_BACKOFF.length - 1;
080    }
081    if (ntries < 0) {
082      ntries = 0;
083    }
084
085    long normalPause = pause * HConstants.RETRY_BACKOFF[ntries];
086    // 1% possible jitter
087    long jitter = (long) (normalPause * ThreadLocalRandom.current().nextFloat() * 0.01f);
088    return normalPause + jitter;
089  }
090
091  /**
092   * Adds / subs an up to 50% jitter to a pause time. Minimum is 1.
093   * @param pause the expected pause.
094   * @param jitter the jitter ratio, between 0 and 1, exclusive.
095   */
096  public static long addJitter(final long pause, final float jitter) {
097    float lag = pause * (ThreadLocalRandom.current().nextFloat() - 0.5f) * jitter;
098    long newPause = pause + (long) lag;
099    if (newPause <= 0) {
100      return 1;
101    }
102    return newPause;
103  }

115  /**
116   * Changes the configuration to set the number of retries needed when using Connection internally,
117   * e.g. for updating catalog tables, etc. Call this method before we create any Connections.
118   * @param c The Configuration instance to set the retries into.
119   * @param log Used to log what we set in here.
120   */
121  public static void setServerSideHConnectionRetriesConfig(final Configuration c, final String sn,
122      final Log log) {
123    // TODO: Fix this. Not all connections from server side should have 10 times the retries.
124    int hcRetries = c.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
125      HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
126    // Go big. Multiply by 10. If we can't get to meta after this many retries
127    // then something seriously wrong.
128    int serversideMultiplier = c.getInt("hbase.client.serverside.retries.multiplier", 10);
129    int retries = hcRetries * serversideMultiplier;
130    c.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, retries);
131    log.info(sn + " server-side Connection retries=" + retries);
132  }

134  /**
135   * Creates a short-circuit connection that can bypass the RPC layer (serialization,
136   * deserialization, networking, etc..) when talking to a local server.
137   * @param conf the current configuration
138   * @param pool the thread pool to use for batch operations
139   * @param user the user the connection is for
140   * @param serverName the local server name
141   * @param admin the admin interface of the local server
142   * @param client the client interface of the local server
143   * @return an short-circuit connection.
144   * @throws IOException if IO failure occurred
145   */
146  public static ClusterConnection createShortCircuitConnection(final Configuration conf,
147      ExecutorService pool, User user, final ServerName serverName,
148      final AdminService.BlockingInterface admin, final ClientService.BlockingInterface client)
149      throws IOException {
150    if (user == null) {
151      user = UserProvider.instantiate(conf).getCurrent();
152    }
153    return new ConnectionImplementation(conf, pool, user) {
154      @Override
155      public AdminService.BlockingInterface getAdmin(ServerName sn) throws IOException {
156        return serverName.equals(sn) ? admin : super.getAdmin(sn);
157      }
158
159      @Override
160      public ClientService.BlockingInterface getClient(ServerName sn) throws IOException {
161        return serverName.equals(sn) ? client : super.getClient(sn);
162      }
163    };
164  }

198  /**
199   * Get a unique key for the rpc stub to the given server.
200   */
201  static String getStubKey(String serviceName, ServerName serverName, boolean hostnameCanChange) {
202    // Sometimes, servers go down and they come back up with the same hostname but a different
203    // IP address. Force a resolution of the rsHostname by trying to instantiate an
204    // InetSocketAddress, and this way we will rightfully get a new stubKey.
205    // Also, include the hostname in the key so as to take care of those cases where the
206    // DNS name is different but IP address remains the same.
207    String hostname = serverName.getHostname();
208    int port = serverName.getPort();
209    if (hostnameCanChange) {
210      try {
211        InetAddress ip = InetAddress.getByName(hostname);
212        return serviceName + "@" + hostname + "-" + ip.getHostAddress() + ":" + port;
213      } catch (UnknownHostException e) {
214        LOG.warn("Can not resolve " + hostname + ", please check your network", e);
215      }
216    }
217    return serviceName + "@" + hostname + ":" + port;
218  }

239  // A byte array in which all elements are the max byte, and it is used to
240  // construct closest front row
241  static final byte[] MAX_BYTE_ARRAY = Bytes.createMaxByteArray(9);
242
243  /**
244   * Create the closest row after the specified row
245   */
246  static byte[] createClosestRowAfter(byte[] row) {
247    return Arrays.copyOf(row, row.length + 1);
248  }
249
250  /**
251   * Create a row before the specified row and very close to the specified row.
252   */
253  static byte[] createCloseRowBefore(byte[] row) {
254    if (row.length == 0) {
255      return MAX_BYTE_ARRAY;
256    }
257    if (row[row.length - 1] == 0) {
258      return Arrays.copyOf(row, row.length - 1);
259    } else {
260      byte[] nextRow = new byte[row.length + MAX_BYTE_ARRAY.length];
261      System.arraycopy(row, 0, nextRow, 0, row.length - 1);
262      nextRow[row.length - 1] = (byte) ((row[row.length - 1] & 0xFF) - 1);
263      System.arraycopy(MAX_BYTE_ARRAY, 0, nextRow, row.length, MAX_BYTE_ARRAY.length);
264      return nextRow;
265    }
266  }

284  static Throwable translateException(Throwable t) {
285    if (t instanceof UndeclaredThrowableException && t.getCause() != null) {
286      t = t.getCause();
287    }
288    if (t instanceof RemoteException) {
289      t = ((RemoteException) t).unwrapRemoteException();
290    }
291    if (t instanceof ServiceException && t.getCause() != null) {
292      t = translateException(t.getCause());
293    }
294    return t;
295  }

297  static long calcEstimatedSize(Result rs) {
298    long estimatedHeapSizeOfResult = 0;
299    // We don't make Iterator here
300    for (Cell cell : rs.rawCells()) {
301      estimatedHeapSizeOfResult += CellUtil.estimatedHeapSizeOf(cell);
302    }
303    return estimatedHeapSizeOfResult;
304  }

306  static Result filterCells(Result result, Cell keepCellsAfter) {
307    // not the same row
308    if (!CellUtil.matchingRow(keepCellsAfter, result.getRow(), 0, result.getRow().length)) {
309      return result;
310    }
311    Cell[] rawCells = result.rawCells();
312    int index = Arrays.binarySearch(rawCells, keepCellsAfter, CellComparator::compareWithoutRow);
313    if (index < 0) {
314      index = -index - 1;
315    } else {
316      index++;
317    }
318    if (index == 0) {
319      return result;
320    }
321    if (index == rawCells.length) {
322      return null;
323    }
324    return Result.create(Arrays.copyOfRange(rawCells, index, rawCells.length), null,
325      result.isStale(), result.mayHaveMoreCellsInRow());
326  }

342  static RegionLocateType getLocateType(Scan scan) {
343    if (scan.isReversed()) {
344      if (isEmptyStartRow(scan.getStartRow())) {
345        return RegionLocateType.BEFORE;
346      } else {
347        return scan.includeStartRow() ? RegionLocateType.CURRENT : RegionLocateType.BEFORE;
348      }
349    } else {
350      return scan.includeStartRow() ? RegionLocateType.CURRENT : RegionLocateType.AFTER;
351    }
352  }

354  static boolean noMoreResultsForScan(Scan scan, HRegionInfo info) {
355    if (isEmptyStopRow(info.getEndKey())) {
356      return true;
357    }
358    if (isEmptyStopRow(scan.getStopRow())) {
359      return false;
360    }
361    int c = Bytes.compareTo(info.getEndKey(), scan.getStopRow());
362    // 1. if our stop row is less than the endKey of the region
363    // 2. if our stop row is equal to the endKey of the region and we do not include the stop row
364    // for scan.
365    return c > 0 || (c == 0 && !scan.includeStopRow());
366  }
367
368  static boolean noMoreResultsForReverseScan(Scan scan, HRegionInfo info) {
369    if (isEmptyStartRow(info.getStartKey())) {
370      return true;
371    }
372    if (isEmptyStopRow(scan.getStopRow())) {
373      return false;
374    }
375    // no need to test the inclusive of the stop row as the start key of a region is included in
376    // the region.
377    return Bytes.compareTo(info.getStartKey(), scan.getStopRow()) <= 0;
378  }

380  static <T> CompletableFuture<List<T>> allOf(List<CompletableFuture<T>> futures) {
381    return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
382        .thenApply(v -> futures.stream().map(f -> f.getNow(null)).collect(toList()));
383  }
