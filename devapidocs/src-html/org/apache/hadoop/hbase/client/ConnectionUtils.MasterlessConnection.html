061  /**
062   * Calculate pause time.
063   * Built on {@link HConstants#RETRY_BACKOFF}.
064   * @param pause time to pause
065   * @param tries amount of tries
066   * @return How long to wait after <code>tries</code> retries
067   */
068  public static long getPauseTime(final long pause, final int tries) {
069    int ntries = tries;
070    if (ntries >= HConstants.RETRY_BACKOFF.length) {
071      ntries = HConstants.RETRY_BACKOFF.length - 1;
072    }
073    if (ntries < 0) {
074      ntries = 0;
075    }
076
077    long normalPause = pause * HConstants.RETRY_BACKOFF[ntries];
078    // 1% possible jitter
079    long jitter = (long) (normalPause * ThreadLocalRandom.current().nextFloat() * 0.01f);
080    return normalPause + jitter;
081  }
082
083
084  /**
085   * Adds / subs an up to 50% jitter to a pause time. Minimum is 1.
086   * @param pause the expected pause.
087   * @param jitter the jitter ratio, between 0 and 1, exclusive.
088   */
089  public static long addJitter(final long pause, final float jitter) {
090    float lag = pause * (ThreadLocalRandom.current().nextFloat() - 0.5f) * jitter;
091    long newPause = pause + (long) lag;
092    if (newPause <= 0) {
093      return 1;
094    }
095    return newPause;
096  }

108  /**
109   * Changes the configuration to set the number of retries needed when using Connection
110   * internally, e.g. for  updating catalog tables, etc.
111   * Call this method before we create any Connections.
112   * @param c The Configuration instance to set the retries into.
113   * @param log Used to log what we set in here.
114   */
115  public static void setServerSideHConnectionRetriesConfig(
116      final Configuration c, final String sn, final Log log) {
117    // TODO: Fix this. Not all connections from server side should have 10 times the retries.
118    int hcRetries = c.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
119      HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
120    // Go big.  Multiply by 10.  If we can't get to meta after this many retries
121    // then something seriously wrong.
122    int serversideMultiplier = c.getInt("hbase.client.serverside.retries.multiplier", 10);
123    int retries = hcRetries * serversideMultiplier;
124    c.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, retries);
125    log.info(sn + " server-side Connection retries=" + retries);
126  }

128  /**
129   * Creates a short-circuit connection that can bypass the RPC layer (serialization,
130   * deserialization, networking, etc..) when talking to a local server.
131   * @param conf the current configuration
132   * @param pool the thread pool to use for batch operations
133   * @param user the user the connection is for
134   * @param serverName the local server name
135   * @param admin the admin interface of the local server
136   * @param client the client interface of the local server
137   * @return an short-circuit connection.
138   * @throws IOException if IO failure occurred
139   */
140  public static ClusterConnection createShortCircuitConnection(final Configuration conf,
141    ExecutorService pool, User user, final ServerName serverName,
142    final AdminService.BlockingInterface admin, final ClientService.BlockingInterface client)
143    throws IOException {
144    if (user == null) {
145      user = UserProvider.instantiate(conf).getCurrent();
146    }
147    return new ConnectionImplementation(conf, pool, user) {
148      @Override
149      public AdminService.BlockingInterface getAdmin(ServerName sn) throws IOException {
150        return serverName.equals(sn) ? admin : super.getAdmin(sn);
151      }
152
153      @Override
154      public ClientService.BlockingInterface getClient(ServerName sn) throws IOException {
155        return serverName.equals(sn) ? client : super.getClient(sn);
156      }
157    };
158  }

194  /**
195   * Get a unique key for the rpc stub to the given server.
196   */
197  static String getStubKey(String serviceName, ServerName serverName,
198      boolean hostnameCanChange) {
199    // Sometimes, servers go down and they come back up with the same hostname but a different
200    // IP address. Force a resolution of the rsHostname by trying to instantiate an
201    // InetSocketAddress, and this way we will rightfully get a new stubKey.
202    // Also, include the hostname in the key so as to take care of those cases where the
203    // DNS name is different but IP address remains the same.
204    String hostname = serverName.getHostname();
205    int port = serverName.getPort();
206    if (hostnameCanChange) {
207      try {
208        InetAddress ip = InetAddress.getByName(hostname);
209        return serviceName + "@" + hostname + "-" + ip.getHostAddress() + ":" + port;
210      } catch (UnknownHostException e) {
211        LOG.warn("Can not resolve " + hostname + ", please check your network", e);
212      }
213    }
214    return serviceName + "@" + hostname + ":" + port;
215  }

236  // A byte array in which all elements are the max byte, and it is used to
237  // construct closest front row
238  static final byte[] MAX_BYTE_ARRAY = Bytes.createMaxByteArray(9);
239
240  /**
241   * Create the closest row after the specified row
242   */
243  static byte[] createClosestRowAfter(byte[] row) {
244    return Arrays.copyOf(row, row.length + 1);
245  }
246
247  /**
248   * Create the closest row before the specified row
249   */
250  static byte[] createClosestRowBefore(byte[] row) {
251    if (row.length == 0) {
252      return MAX_BYTE_ARRAY;
253    }
254    if (row[row.length - 1] == 0) {
255      return Arrays.copyOf(row, row.length - 1);
256    } else {
257      byte[] nextRow = new byte[row.length + MAX_BYTE_ARRAY.length];
258      System.arraycopy(row, 0, nextRow, 0, row.length - 1);
259      nextRow[row.length - 1] = (byte) ((row[row.length - 1] & 0xFF) - 1);
260      System.arraycopy(MAX_BYTE_ARRAY, 0, nextRow, row.length, MAX_BYTE_ARRAY.length);
261      return nextRow;
262    }
263  }

273  static void resetController(HBaseRpcController controller, long timeoutNs) {
274    controller.reset();
275    if (timeoutNs >= 0) {
276      controller.setCallTimeout(
277        (int) Math.min(Integer.MAX_VALUE, TimeUnit.NANOSECONDS.toMillis(timeoutNs)));
278    }
279  }

281  static Throwable translateException(Throwable t) {
282    if (t instanceof UndeclaredThrowableException && t.getCause() != null) {
283      t = t.getCause();
284    }
285    if (t instanceof RemoteException) {
286      t = ((RemoteException) t).unwrapRemoteException();
287    }
288    if (t instanceof ServiceException && t.getCause() != null) {
289      t = translateException(t.getCause());
290    }
291    return t;
292  }
