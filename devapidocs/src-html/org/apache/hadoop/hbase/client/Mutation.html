154  /**
155   * Compile the column family (i.e. schema) information
156   * into a Map. Useful for parsing and aggregation by debugging,
157   * logging, and administration tools.
158   * @return Map
159   */
160  @Override
161  public Map<String, Object> getFingerprint() {
162    Map<String, Object> map = new HashMap<>();
163    List<String> families = new ArrayList<>(this.familyMap.entrySet().size());
164    // ideally, we would also include table information, but that information
165    // is not stored in each Operation instance.
166    map.put("families", families);
167    for (Map.Entry<byte [], List<Cell>> entry : this.familyMap.entrySet()) {
168      families.add(Bytes.toStringBinary(entry.getKey()));
169    }
170    return map;
171  }
172
173  /**
174   * Compile the details beyond the scope of getFingerprint (row, columns,
175   * timestamps, etc.) into a Map along with the fingerprinted information.
176   * Useful for debugging, logging, and administration tools.
177   * @param maxCols a limit on the number of columns output prior to truncation
178   * @return Map
179   */
180  @Override
181  public Map<String, Object> toMap(int maxCols) {
182    // we start with the fingerprint map and build on top of it.
183    Map<String, Object> map = getFingerprint();
184    // replace the fingerprint's simple list of families with a
185    // map from column families to lists of qualifiers and kv details
186    Map<String, List<Map<String, Object>>> columns = new HashMap<>();
187    map.put("families", columns);
188    map.put("row", Bytes.toStringBinary(this.row));
189    int colCount = 0;
190    // iterate through all column families affected
191    for (Map.Entry<byte [], List<Cell>> entry : this.familyMap.entrySet()) {
192      // map from this family to details for each cell affected within the family
193      List<Map<String, Object>> qualifierDetails = new ArrayList<>();
194      columns.put(Bytes.toStringBinary(entry.getKey()), qualifierDetails);
195      colCount += entry.getValue().size();
196      if (maxCols <= 0) {
197        continue;
198      }
199      // add details for each cell
200      for (Cell cell: entry.getValue()) {
201        if (--maxCols <= 0) {
202          continue;
203        }
204        Map<String, Object> cellMap = cellToStringMap(cell);
205        qualifierDetails.add(cellMap);
206      }
207    }
208    map.put("totalColumns", colCount);
209    // add the id if set
210    if (getId() != null) {
211      map.put("id", getId());
212    }
213    // Add the TTL if set
214    // Long.MAX_VALUE is the default, and is interpreted to mean this attribute
215    // has not been set.
216    if (getTTL() != Long.MAX_VALUE) {
217      map.put("ttl", getTTL());
218    }
219    return map;
220  }
221
222  private static Map<String, Object> cellToStringMap(Cell c) {
223    Map<String, Object> stringMap = new HashMap<>();
224    stringMap.put("qualifier", Bytes.toStringBinary(c.getQualifierArray(), c.getQualifierOffset(),
225                c.getQualifierLength()));
226    stringMap.put("timestamp", c.getTimestamp());
227    stringMap.put("vlen", c.getValueLength());
228    List<Tag> tags = CellUtil.getTags(c);
229    if (tags != null) {
230      List<String> tagsString = new ArrayList<>(tags.size());
231      for (Tag t : tags) {
232        tagsString.add((t.getType()) + ":" + Bytes.toStringBinary(TagUtil.cloneValue(t)));
233      }
234      stringMap.put("tag", tagsString);
235    }
236    return stringMap;
237  }

301  /**
302   * Marks that the clusters with the given clusterIds have consumed the mutation
303   * @param clusterIds of the clusters that have consumed the mutation
304   */
305  public Mutation setClusterIds(List<UUID> clusterIds) {
306    ByteArrayDataOutput out = ByteStreams.newDataOutput();
307    out.writeInt(clusterIds.size());
308    for (UUID clusterId : clusterIds) {
309      out.writeLong(clusterId.getMostSignificantBits());
310      out.writeLong(clusterId.getLeastSignificantBits());
311    }
312    setAttribute(CONSUMED_CLUSTER_IDS, out.toByteArray());
313    return this;
314  }
315
316  /**
317   * @return the set of clusterIds that have consumed the mutation
318   */
319  public List<UUID> getClusterIds() {
320    List<UUID> clusterIds = new ArrayList<>();
321    byte[] bytes = getAttribute(CONSUMED_CLUSTER_IDS);
322    if(bytes != null) {
323      ByteArrayDataInput in = ByteStreams.newDataInput(bytes);
324      int numClusters = in.readInt();
325      for(int i=0; i<numClusters; i++){
326        clusterIds.add(new UUID(in.readLong(), in.readLong()));
327      }
328    }
329    return clusterIds;
330  }

352  /**
353   * Create a protocol buffer CellVisibility based on a client CellVisibility.
354   *
355   * @param cellVisibility

357   */
358  static ClientProtos.CellVisibility toCellVisibility(CellVisibility cellVisibility) {
359    ClientProtos.CellVisibility.Builder builder = ClientProtos.CellVisibility.newBuilder();
360    builder.setExpression(cellVisibility.getExpression());
361    return builder.build();
362  }

364  /**
365   * Convert a protocol buffer CellVisibility to a client CellVisibility
366   *
367   * @param proto
368   * @return the converted client CellVisibility
369   */
370  private static CellVisibility toCellVisibility(ClientProtos.CellVisibility proto) {
371    if (proto == null) return null;
372    return new CellVisibility(proto.getExpression());
373  }
374
375  /**
376   * Convert a protocol buffer CellVisibility bytes to a client CellVisibility
377   *
378   * @param protoBytes
379   * @return the converted client CellVisibility
380   * @throws DeserializationException
381   */
382  private static CellVisibility toCellVisibility(byte[] protoBytes) throws DeserializationException {
383    if (protoBytes == null) return null;
384    ClientProtos.CellVisibility.Builder builder = ClientProtos.CellVisibility.newBuilder();
385    ClientProtos.CellVisibility proto = null;
386    try {
387      ProtobufUtil.mergeFrom(builder, protoBytes);
388      proto = builder.build();
389    } catch (IOException e) {
390      throw new DeserializationException(e);
391    }
392    return toCellVisibility(proto);
393  }

414  /**
415   * @return Calculate what Mutation adds to class heap size.
416   */
417  @Override
418  public long heapSize() {
419    long heapsize = MUTATION_OVERHEAD;
420    // Adding row
421    heapsize += ClassSize.align(ClassSize.ARRAY + this.row.length);
422
423    // Adding map overhead
424    heapsize +=
425      ClassSize.align(this.familyMap.size() * ClassSize.MAP_ENTRY);
426    for(Map.Entry<byte [], List<Cell>> entry : this.familyMap.entrySet()) {
427      //Adding key overhead
428      heapsize +=
429        ClassSize.align(ClassSize.ARRAY + entry.getKey().length);
430
431      //This part is kinds tricky since the JVM can reuse references if you
432      //store the same value, but have a good match with SizeOf at the moment
433      //Adding value overhead
434      heapsize += ClassSize.align(ClassSize.ARRAYLIST);
435      int size = entry.getValue().size();
436      heapsize += ClassSize.align(ClassSize.ARRAY +
437          size * ClassSize.REFERENCE);
438
439      for(Cell cell : entry.getValue()) {
440        heapsize += CellUtil.estimatedHeapSizeOf(cell);
441      }
442    }
443    heapsize += getAttributeSize();
444    heapsize += extraHeapSize();
445    return ClassSize.align(heapsize);
446  }

527  /**
528   * @param row Row to check
529   * @throws IllegalArgumentException Thrown if <code>row</code> is empty or null or
530   * &gt; {@link HConstants#MAX_ROW_LENGTH}
531   * @return <code>row</code>
532   */
533  static byte [] checkRow(final byte [] row) {
534    return checkRow(row, 0, row == null? 0: row.length);
535  }
536
537  /**
538   * @param row Row to check
539   * @param offset
540   * @param length
541   * @throws IllegalArgumentException Thrown if <code>row</code> is empty or null or
542   * &gt; {@link HConstants#MAX_ROW_LENGTH}
543   * @return <code>row</code>
544   */
545  static byte [] checkRow(final byte [] row, final int offset, final int length) {
546    if (row == null) {
547      throw new IllegalArgumentException("Row buffer is null");
548    }
549    if (length == 0) {
550      throw new IllegalArgumentException("Row length is 0");
551    }
552    if (length > HConstants.MAX_ROW_LENGTH) {
553      throw new IllegalArgumentException("Row length " + length + " is > " +
554        HConstants.MAX_ROW_LENGTH);
555    }
556    return row;
557  }
558
559  static void checkRow(ByteBuffer row) {
560    if (row == null) {
561      throw new IllegalArgumentException("Row buffer is null");
562    }
563    if (row.remaining() == 0) {
564      throw new IllegalArgumentException("Row length is 0");
565    }
566    if (row.remaining() > HConstants.MAX_ROW_LENGTH) {
567      throw new IllegalArgumentException("Row length " + row.remaining() + " is > " +
568          HConstants.MAX_ROW_LENGTH);
569    }
570  }
