054/**
055 * Scanner operations such as create, next, etc.
056 * Used by {@link ResultScanner}s made by {@link Table}. Passed to a retrying caller such as
057 * {@link RpcRetryingCaller} so fails are retried.
058 */
059@InterfaceAudience.Private
060public class ScannerCallable extends ClientServiceCallable<Result[]> {
061  public static final String LOG_SCANNER_LATENCY_CUTOFF
062    = "hbase.client.log.scanner.latency.cutoff";
063  public static final String LOG_SCANNER_ACTIVITY = "hbase.client.log.scanner.activity";
064
065  // Keeping LOG public as it is being used in TestScannerHeartbeatMessages
066  public static final Log LOG = LogFactory.getLog(ScannerCallable.class);
067  protected long scannerId = -1L;
068  protected boolean instantiated = false;
069  protected boolean closed = false;
070  protected boolean renew = false;
071  private Scan scan;
072  private int caching = 1;
073  protected ScanMetrics scanMetrics;
074  private boolean logScannerActivity = false;
075  private int logCutOffLatency = 1000;
076  private static String myAddress;
077  protected final int id;
078  protected boolean serverHasMoreResultsContext;
079  protected boolean serverHasMoreResults;

081  /**
082   * Saves whether or not the most recent response from the server was a heartbeat message.
083   * Heartbeat messages are identified by the flag {@link ScanResponse#getHeartbeatMessage()}
084   */
085  protected boolean heartbeatMessage = false;
086  static {
087    try {
088      myAddress = DNS.getDefaultHost("default", "default");
089    } catch (UnknownHostException uhe) {
090      LOG.error("cannot determine my address", uhe);
091    }

112  /**
113   *
114   * @param connection
115   * @param tableName
116   * @param scan
117   * @param scanMetrics
118   * @param id the replicaId
119   */
120  public ScannerCallable(ClusterConnection connection, TableName tableName, Scan scan,
121      ScanMetrics scanMetrics, RpcControllerFactory rpcControllerFactory, int id) {
122    super(connection, tableName, scan.getStartRow(), rpcControllerFactory.newController());
123    this.id = id;
124    this.scan = scan;
125    this.scanMetrics = scanMetrics;
126    Configuration conf = connection.getConfiguration();
127    logScannerActivity = conf.getBoolean(LOG_SCANNER_ACTIVITY, false);
128    logCutOffLatency = conf.getInt(LOG_SCANNER_LATENCY_CUTOFF, 1000);
129    this.rpcControllerFactory = rpcControllerFactory;
130  }

132  /**
133   * @param reload force reload of server location
134   * @throws IOException
135   */
136  @Override
137  public void prepare(boolean reload) throws IOException {
138    if (Thread.interrupted()) {
139      throw new InterruptedIOException();
140    }
141    RegionLocations rl = RpcRetryingCallerWithReadReplicas.getRegionLocations(!reload,
142        id, getConnection(), getTableName(), getRow());
143    location = id < rl.size() ? rl.getRegionLocation(id) : null;
144    if (location == null || location.getServerName() == null) {
145      // With this exception, there will be a retry. The location can be null for a replica
146      //  when the table is created or after a split.
147      throw new HBaseIOException("There is no location for replica id #" + id);
148    }
149    ServerName dest = location.getServerName();
150    setStub(super.getConnection().getClient(dest));
151    if (!instantiated || reload) {
152      checkIfRegionServerIsRemote();
153      instantiated = true;
154    }
155
156    // check how often we retry.
157    if (reload && this.scanMetrics != null) {
158      this.scanMetrics.countOfRPCRetries.incrementAndGet();
159      if (isRegionServerRemote) {
160        this.scanMetrics.countOfRemoteRPCRetries.incrementAndGet();
161      }
162    }
163  }
164
165  /**
166   * compare the local machine hostname with region server's hostname
167   * to decide if hbase client connects to a remote region server
168   */
169  protected void checkIfRegionServerIsRemote() {
170    if (getLocation().getHostname().equalsIgnoreCase(myAddress)) {
171      isRegionServerRemote = false;
172    } else {
173      isRegionServerRemote = true;
174    }
175  }

177  @Override
178  protected Result [] rpcCall() throws Exception {
179    if (Thread.interrupted()) {
180      throw new InterruptedIOException();
181    }
182    if (this.closed) {
183      if (this.scannerId != -1) {
184        close();
185      }
186    } else {
187      if (this.scannerId == -1L) {
188        this.scannerId = openScanner();
189      } else {
190        Result [] rrs = null;
191        ScanRequest request = null;
192        // Reset the heartbeat flag prior to each RPC in case an exception is thrown by the server
193        setHeartbeatMessage(false);
194        try {
195          incRPCcallsMetrics();
196          request = RequestConverter.buildScanRequest(scannerId, caching, false, nextCallSeq,
197                this.scanMetrics != null, renew);
198          ScanResponse response = null;
199          response = getStub().scan(getRpcController(), request);
200          // Client and RS maintain a nextCallSeq number during the scan. Every next() call
201          // from client to server will increment this number in both sides. Client passes this
202          // number along with the request and at RS side both the incoming nextCallSeq and its
203          // nextCallSeq will be matched. In case of a timeout this increment at the client side
204          // should not happen. If at the server side fetching of next batch of data was over,
205          // there will be mismatch in the nextCallSeq number. Server will throw
206          // OutOfOrderScannerNextException and then client will reopen the scanner with startrow
207          // as the last successfully retrieved row.
208          // See HBASE-5974
209          nextCallSeq++;
210          long timestamp = System.currentTimeMillis();
211          setHeartbeatMessage(response.hasHeartbeatMessage() && response.getHeartbeatMessage());
212          rrs = ResponseConverter.getResults(getRpcControllerCellScanner(), response);
213          if (logScannerActivity) {
214            long now = System.currentTimeMillis();
215            if (now - timestamp > logCutOffLatency) {
216              int rows = rrs == null ? 0 : rrs.length;
217              LOG.info("Took " + (now-timestamp) + "ms to fetch "
218                  + rows + " rows from scanner=" + scannerId);
219            }
220          }
221          updateServerSideMetrics(response);
222          // moreResults is only used for the case where a filter exhausts all elements
223          if (response.hasMoreResults() && !response.getMoreResults()) {
224            this.scannerId = -1L;
225            this.closed = true;
226            // Implied that no results were returned back, either.
227            return null;
228          }
229          // moreResultsInRegion explicitly defines when a RS may choose to terminate a batch due
230          // to size or quantity of results in the response.
231          if (response.hasMoreResultsInRegion()) {
232            // Set what the RS said
233            setHasMoreResultsContext(true);
234            setServerHasMoreResults(response.getMoreResultsInRegion());
235          } else {
236            // Server didn't respond whether it has more results or not.
237            setHasMoreResultsContext(false);
238          }
239          updateResultsMetrics(rrs);
240        } catch (IOException e) {
241          if (logScannerActivity) {
242            LOG.info("Got exception making request " + ProtobufUtil.toText(request) + " to " +
243                getLocation(), e);
244          }
245          IOException ioe = e;
246          if (e instanceof RemoteException) {
247            ioe = ((RemoteException) e).unwrapRemoteException();
248          }
249          if (logScannerActivity) {
250            if (ioe instanceof UnknownScannerException) {
251              try {
252                HRegionLocation location =
253                    getConnection().relocateRegion(getTableName(), scan.getStartRow());
254                LOG.info("Scanner=" + scannerId + " expired, current region location is " +
255                    location.toString());
256              } catch (Throwable t) {
257                LOG.info("Failed to relocate region", t);
258              }
259            } else if (ioe instanceof ScannerResetException) {
260              LOG.info("Scanner=" + scannerId + " has received an exception, and the server "
261                  + "asked us to reset the scanner state.", ioe);
262            }
263          }
264          // The below convertion of exceptions into DoNotRetryExceptions is a little strange.
265          // Why not just have these exceptions implment DNRIOE you ask?  Well, usually we want
266          // ServerCallable#withRetries to just retry when it gets these exceptions.  In here in
267          // a scan when doing a next in particular, we want to break out and get the scanner to
268          // reset itself up again.  Throwing a DNRIOE is how we signal this to happen (its ugly,
269          // yeah and hard to follow and in need of a refactor).
270          if (ioe instanceof NotServingRegionException) {
271            // Throw a DNRE so that we break out of cycle of calling NSRE
272            // when what we need is to open scanner against new location.
273            // Attach NSRE to signal client that it needs to re-setup scanner.
274            if (this.scanMetrics != null) {
275              this.scanMetrics.countOfNSRE.incrementAndGet();
276            }
277            throw new DoNotRetryIOException("Resetting the scanner -- see exception cause", ioe);
278          } else if (ioe instanceof RegionServerStoppedException) {
279            // Throw a DNRE so that we break out of cycle of the retries and instead go and
280            // open scanner against new location.
281            throw new DoNotRetryIOException("Resetting the scanner -- see exception cause", ioe);
282          } else {
283            // The outer layers will retry
284            throw ioe;
285          }
286        }
287        return rrs;
288      }
289    }
290    return null;
291  }

293  /**
294   * @return true when the most recent RPC response indicated that the response was a heartbeat
295   *         message. Heartbeat messages are sent back from the server when the processing of the
296   *         scan request exceeds a certain time threshold. Heartbeats allow the server to avoid
297   *         timeouts during long running scan operations.
298   */
299  protected boolean isHeartbeatMessage() {
300    return heartbeatMessage;
301  }
302
303  protected void setHeartbeatMessage(boolean heartbeatMessage) {
304    this.heartbeatMessage = heartbeatMessage;
305  }

307  private void incRPCcallsMetrics() {
308    if (this.scanMetrics == null) {
309      return;
310    }
311    this.scanMetrics.countOfRPCcalls.incrementAndGet();
312    if (isRegionServerRemote) {
313      this.scanMetrics.countOfRemoteRPCcalls.incrementAndGet();
314    }
315  }
316
317  protected void updateResultsMetrics(Result[] rrs) {
318    if (this.scanMetrics == null || rrs == null || rrs.length == 0) {
319      return;
320    }
321    long resultSize = 0;
322    for (Result rr : rrs) {
323      for (Cell cell : rr.rawCells()) {
324        resultSize += CellUtil.estimatedSerializedSizeOf(cell);
325      }
326    }
327    this.scanMetrics.countOfBytesInResults.addAndGet(resultSize);
328    if (isRegionServerRemote) {
329      this.scanMetrics.countOfBytesInRemoteResults.addAndGet(resultSize);
330    }
331  }
332
333  /**
334   * Use the scan metrics returned by the server to add to the identically named counters in the
335   * client side metrics. If a counter does not exist with the same name as the server side metric,
336   * the attempt to increase the counter will fail.
337   * @param response
338   */
339  private void updateServerSideMetrics(ScanResponse response) {
340    if (this.scanMetrics == null || response == null || !response.hasScanMetrics()) return;
341
342    Map<String, Long> serverMetrics = ResponseConverter.getScanMetrics(response);
343    for (Entry<String, Long> entry : serverMetrics.entrySet()) {
344      this.scanMetrics.addToCounter(entry.getKey(), entry.getValue());
345    }
346  }
347
348  private void close() {
349    if (this.scannerId == -1L) {
350      return;
351    }
352    try {
353      incRPCcallsMetrics();
354      ScanRequest request =
355          RequestConverter.buildScanRequest(this.scannerId, 0, true, this.scanMetrics != null);
356      try {
357        getStub().scan(getRpcController(), request);
358      } catch (Exception e) {
359        throw ProtobufUtil.handleRemoteException(e);
360      }
361    } catch (IOException e) {
362      LOG.warn("Ignore, probably already closed", e);
363    }
364    this.scannerId = -1L;
365  }
366
367  protected long openScanner() throws IOException {
368    incRPCcallsMetrics();
369    ScanRequest request = RequestConverter.buildScanRequest(
370        getLocation().getRegionInfo().getRegionName(), this.scan, 0, false);
371    try {
372      ScanResponse response = getStub().scan(getRpcController(), request);
373      long id = response.getScannerId();
374      if (logScannerActivity) {
375        LOG.info("Open scanner=" + id + " for scan=" + scan.toString()
376          + " on region " + getLocation().toString());
377      }
378      return id;
379    } catch (Exception e) {
380      throw ProtobufUtil.handleRemoteException(e);
381    }
382  }
