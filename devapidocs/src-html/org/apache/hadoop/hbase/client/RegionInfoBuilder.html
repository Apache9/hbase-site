103  public RegionInfo build() {
104    RegionInfo ri = new MutableRegionInfo(content);
105    // Run a late check that we are not creating default meta region.
106    if (ri.getTable().equals(TableName.META_TABLE_NAME) &&
107        ri.getReplicaId() == RegionInfo.DEFAULT_REPLICA_ID) {
108      throw new IllegalArgumentException("Cannot create the default meta region; " +
109        "use static define FIRST_META_REGIONINFO");
110    }
111    return new MutableRegionInfo(content);
112  }

114  /**
115   * An implementation of RegionInfo that adds mutable methods so can build a RegionInfo instance.

117  @InterfaceAudience.Private
118  static class MutableRegionInfo implements RegionInfo, Comparable<RegionInfo> {
119    /**
120     * The new format for a region name contains its encodedName at the end.
121     * The encoded name also serves as the directory name for the region
122     * in the filesystem.
123     *
124     * New region name format:
125     *    &lt;tablename>,,&lt;startkey>,&lt;regionIdTimestamp>.&lt;encodedName>.
126     * where,
127     *    &lt;encodedName> is a hex version of the MD5 hash of
128     *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
129     *
130     * The old region name format:
131     *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
132     * For region names in the old format, the encoded name is a 32-bit
133     * JenkinsHash integer value (in its decimal notation, string form).
134     *<p>
135     * **NOTE**
136     *
137     * The first hbase:meta region, and regions created by an older
138     * version of HBase (0.20 or prior) will continue to use the
139     * old region name format.
140     */
141
142    // This flag is in the parent of a split while the parent is still referenced
143    // by daughter regions.  We USED to set this flag when we disabled a table
144    // but now table state is kept up in zookeeper as of 0.90.0 HBase.
145    private boolean offLine = false;
146    private boolean split = false;
147    private long regionId = -1;
148    private int replicaId = RegionInfo.DEFAULT_REPLICA_ID;
149    private transient byte [] regionName = HConstants.EMPTY_BYTE_ARRAY;
150    private byte [] startKey = HConstants.EMPTY_BYTE_ARRAY;
151    private byte [] endKey = HConstants.EMPTY_BYTE_ARRAY;
152    private int hashCode = -1;
153    private String encodedName;
154    private byte [] encodedNameAsBytes;
155    // Current TableName
156    private TableName tableName;
157
158    private void setHashCode() {
159      int result = Arrays.hashCode(this.regionName);
160      result ^= this.regionId;
161      result ^= Arrays.hashCode(this.startKey);
162      result ^= Arrays.hashCode(this.endKey);
163      result ^= Boolean.valueOf(this.offLine).hashCode();
164      result ^= Arrays.hashCode(this.tableName.getName());
165      result ^= this.replicaId;
166      this.hashCode = result;
167    }

235    /**
236     * Construct MutableRegionInfo with explicit parameters
237     *
238     * @param tableName the table descriptor
239     * @param startKey first key in region
240     * @param endKey end of key range
241     * @param split true if this region has split and we have daughter regions
242     * regions that may or may not hold references to this region.
243     * @param regionid Region id to use.
244     * @param replicaId the replicaId to use
245     * @throws IllegalArgumentException
246     */
247    MutableRegionInfo(final TableName tableName, final byte[] startKey,
248                       final byte[] endKey, final boolean split, final long regionid,
249                       final int replicaId)
250      throws IllegalArgumentException {
251      super();
252      if (tableName == null) {
253        throw new IllegalArgumentException("TableName cannot be null");
254      }
255      this.tableName = tableName;
256      this.offLine = false;
257      this.regionId = regionid;
258      this.replicaId = replicaId;
259      if (this.replicaId > MAX_REPLICA_ID) {
260        throw new IllegalArgumentException("ReplicaId cannot be greater than" + MAX_REPLICA_ID);
261      }
262
263      this.regionName = RegionInfo.createRegionName(this.tableName, startKey, regionId, replicaId,
264        !this.tableName.equals(TableName.META_TABLE_NAME));
265
266      this.split = split;
267      this.endKey = endKey == null? HConstants.EMPTY_END_ROW: endKey.clone();
268      this.startKey = startKey == null?
269        HConstants.EMPTY_START_ROW: startKey.clone();
270      this.tableName = tableName;
271      setHashCode();
272    }
273
274    /**
275     * Construct MutableRegionInfo.
276     * Only for RegionInfoBuilder to use.
277     * @param other
278     */
279    MutableRegionInfo(MutableRegionInfo other, boolean isMetaRegion) {
280      super();
281      if (other.getTable() == null) {
282        throw new IllegalArgumentException("TableName cannot be null");
283      }
284      this.tableName = other.getTable();
285      this.offLine = other.isOffline();
286      this.regionId = other.getRegionId();
287      this.replicaId = other.getReplicaId();
288      if (this.replicaId > MAX_REPLICA_ID) {
289        throw new IllegalArgumentException("ReplicaId cannot be greater than" + MAX_REPLICA_ID);
290      }
291
292      if(isMetaRegion) {
293        // Note: First Meta region replicas names are in old format
294        this.regionName = RegionInfo.createRegionName(
295                other.getTable(), null, other.getRegionId(),
296                other.getReplicaId(), false);
297      } else {
298        this.regionName = RegionInfo.createRegionName(
299                other.getTable(), other.getStartKey(), other.getRegionId(),
300                other.getReplicaId(), true);
301      }
302
303      this.split = other.isSplit();
304      this.endKey = other.getEndKey() == null? HConstants.EMPTY_END_ROW: other.getEndKey().clone();
305      this.startKey = other.getStartKey() == null?
306        HConstants.EMPTY_START_ROW: other.getStartKey().clone();
307      this.tableName = other.getTable();
308      setHashCode();
309    }

326      this.tableName = regionInfo.getTable();
327      this.replicaId = regionInfo.getReplicaId();
328    }

452    /**
453     * Returns true if the given inclusive range of rows is fully contained
454     * by this region. For example, if the region is foo,a,g and this is
455     * passed ["b","c"] or ["a","c"] it will return true, but if this is passed
456     * ["b","z"] it will return false.
457     * @throws IllegalArgumentException if the range passed is invalid (ie. end &lt; start)
458     */
459    @Override
460    public boolean containsRange(byte[] rangeStartKey, byte[] rangeEndKey) {
461      if (Bytes.compareTo(rangeStartKey, rangeEndKey) > 0) {
462        throw new IllegalArgumentException(
463        "Invalid range: " + Bytes.toStringBinary(rangeStartKey) +
464        " > " + Bytes.toStringBinary(rangeEndKey));
465      }
466
467      boolean firstKeyInRange = Bytes.compareTo(rangeStartKey, startKey) >= 0;
468      boolean lastKeyInRange =
469        Bytes.compareTo(rangeEndKey, endKey) < 0 ||
470        Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY);
471      return firstKeyInRange && lastKeyInRange;
472    }
473
474    /**
475     * Return true if the given row falls in this region.
476     */
477    @Override
478    public boolean containsRow(byte[] row) {
479      return Bytes.compareTo(row, startKey) >= 0 &&
480        (Bytes.compareTo(row, endKey) < 0 ||
481         Bytes.equals(endKey, HConstants.EMPTY_BYTE_ARRAY));
482    }
