018package org.apache.hadoop.hbase.client;
019
020import java.util.List;
021import java.util.concurrent.CompletableFuture;
022import java.util.concurrent.ConcurrentLinkedQueue;
023import java.util.function.BiConsumer;
024import java.util.stream.Collectors;
025import org.apache.commons.lang3.mutable.MutableInt;
026import org.apache.hadoop.hbase.HConstants;
027import org.apache.hadoop.hbase.HRegionLocation;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.util.FutureUtils;
030import org.apache.yetus.audience.InterfaceAudience;
031
032import org.apache.hbase.thirdparty.com.google.common.annotations.VisibleForTesting;
033
034/**
035 * The implementation of AsyncRegionLocator.
036 */
037@InterfaceAudience.Private
038class AsyncTableRegionLocatorImpl implements AsyncTableRegionLocator {
039
040  private final TableName tableName;
041
042  private final AsyncRegionLocator locator;
043
044  public AsyncTableRegionLocatorImpl(TableName tableName, AsyncRegionLocator locator) {
045    this.tableName = tableName;
046    this.locator = locator;
047  }
048
049  @Override
050  public TableName getName() {
051    return tableName;
052  }
053
054  @Override
055  public CompletableFuture<HRegionLocation> getRegionLocation(byte[] row, int replicaId,
056      boolean reload) {
057    return locator.getRegionLocation(tableName, row, replicaId, RegionLocateType.CURRENT, reload,
058      -1L);
059  }
060
061  // this is used to prevent stack overflow if there are thousands of regions for the table. If the
062  // location is in cache, the CompletableFuture will be completed immediately inside the same
063  // thread, and then in the action we will call locate again, also in the same thread. If all the
064  // locations are in cache, and we do not use whenCompleteAsync to break the tie, the stack will be
065  // very very deep and cause stack overflow.
066  @VisibleForTesting
067  static final ThreadLocal<MutableInt> STACK_DEPTH = new ThreadLocal<MutableInt>() {
068
069    @Override
070    protected MutableInt initialValue() {
071      return new MutableInt(0);
072    }
073  };
074
075  @VisibleForTesting
076  static final int MAX_STACK_DEPTH = 16;
077
078  private void locate(CompletableFuture<List<HRegionLocation>> future,
079      ConcurrentLinkedQueue<HRegionLocation> result, byte[] row) {
080    BiConsumer<HRegionLocation, Throwable> listener = (loc, error) -> {
081      if (error != null) {
082        future.completeExceptionally(error);
083        return;
084      }
085      result.add(loc);
086      if (ConnectionUtils.isEmptyStartRow(loc.getRegion().getStartKey())) {
087        future.complete(result.stream()
088          .sorted((l1, l2) -> RegionInfo.COMPARATOR.compare(l1.getRegion(), l2.getRegion()))
089          .collect(Collectors.toList()));
090      } else {
091        locate(future, result, loc.getRegion().getStartKey());
092      }
093    };
094    MutableInt depth = STACK_DEPTH.get();
095    boolean async = depth.incrementAndGet() >= MAX_STACK_DEPTH;
096    try {
097      CompletableFuture<HRegionLocation> f =
098        locator.getRegionLocation(tableName, row, RegionLocateType.BEFORE, -1L);
099      if (async) {
100        FutureUtils.addListenerAsync(f, listener);
101      } else {
102        FutureUtils.addListener(f, listener);
103      }
104    } finally {
105      if (depth.decrementAndGet() == 0) {
106        STACK_DEPTH.remove();
107      }
108    }
109  }
110
111  @Override
112  public CompletableFuture<List<HRegionLocation>> getAllRegionLocations() {
113    ConcurrentLinkedQueue<HRegionLocation> result = new ConcurrentLinkedQueue<>();
114    CompletableFuture<List<HRegionLocation>> future = new CompletableFuture<>();
115    // start from end to start, as when locating we will do reverse scan, so we will prefetch the
116    // location of the regions before the current one.
117    locate(future, result, HConstants.EMPTY_END_ROW);
118    return future;
119  }
120}
