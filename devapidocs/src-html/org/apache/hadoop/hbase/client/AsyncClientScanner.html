018package org.apache.hadoop.hbase.client;
019
020import static org.apache.hadoop.hbase.HConstants.EMPTY_END_ROW;
021import static org.apache.hadoop.hbase.HConstants.EMPTY_START_ROW;
022import static org.apache.hadoop.hbase.client.ConnectionUtils.getLocateType;
023
024import java.io.IOException;
025import java.util.concurrent.CompletableFuture;
026import java.util.concurrent.TimeUnit;
027
028import org.apache.hadoop.hbase.HRegionLocation;
029import org.apache.hadoop.hbase.TableName;
030import org.apache.hadoop.hbase.classification.InterfaceAudience;
031import org.apache.hadoop.hbase.ipc.HBaseRpcController;
032import org.apache.hadoop.hbase.shaded.protobuf.RequestConverter;
033import org.apache.hadoop.hbase.shaded.protobuf.generated.ClientProtos.ClientService;
034import org.apache.hadoop.hbase.shaded.protobuf.generated.ClientProtos.ClientService.Interface;
035import org.apache.hadoop.hbase.shaded.protobuf.generated.ClientProtos.ScanRequest;
036import org.apache.hadoop.hbase.shaded.protobuf.generated.ClientProtos.ScanResponse;
037
038/**
039 * The asynchronous client scanner implementation.
040 * <p>
041 * Here we will call OpenScanner first and use the returned scannerId to create a
042 * {@link AsyncScanSingleRegionRpcRetryingCaller} to do the real scan operation. The return value of
043 * {@link AsyncScanSingleRegionRpcRetryingCaller} will tell us whether open a new scanner or finish
044 * scan.
045 */
046@InterfaceAudience.Private
047class AsyncClientScanner {
048
049  // We will use this scan object during the whole scan operation. The
050  // AsyncScanSingleRegionRpcRetryingCaller will modify this scan object directly.
051  private final Scan scan;
052
053  private final RawScanResultConsumer consumer;
054
055  private final TableName tableName;
056
057  private final AsyncConnectionImpl conn;
058
059  private final long pauseNs;
060
061  private final int maxAttempts;
062
063  private final long scanTimeoutNs;
064
065  private final long rpcTimeoutNs;
066
067  private final int startLogErrorsCnt;
068
069  private final ScanResultCache resultCache;
070
071  public AsyncClientScanner(Scan scan, RawScanResultConsumer consumer, TableName tableName,
072      AsyncConnectionImpl conn, long pauseNs, int maxAttempts, long scanTimeoutNs,
073      long rpcTimeoutNs, int startLogErrorsCnt) {
074    if (scan.getStartRow() == null) {
075      scan.withStartRow(EMPTY_START_ROW, scan.includeStartRow());
076    }
077    if (scan.getStopRow() == null) {
078      scan.withStopRow(EMPTY_END_ROW, scan.includeStopRow());
079    }
080    this.scan = scan;
081    this.consumer = consumer;
082    this.tableName = tableName;
083    this.conn = conn;
084    this.pauseNs = pauseNs;
085    this.maxAttempts = maxAttempts;
086    this.scanTimeoutNs = scanTimeoutNs;
087    this.rpcTimeoutNs = rpcTimeoutNs;
088    this.startLogErrorsCnt = startLogErrorsCnt;
089    this.resultCache = scan.getAllowPartialResults() || scan.getBatch() > 0
090        ? new AllowPartialScanResultCache() : new CompleteScanResultCache();
091  }
092
093  private static final class OpenScannerResponse {
094
095    public final HRegionLocation loc;
096
097    public final ClientService.Interface stub;
098
099    public final HBaseRpcController controller;
100
101    public final ScanResponse resp;
102
103    public OpenScannerResponse(HRegionLocation loc, Interface stub, HBaseRpcController controller,
104        ScanResponse resp) {
105      this.loc = loc;
106      this.stub = stub;
107      this.controller = controller;
108      this.resp = resp;
109    }
110  }
111
112  private CompletableFuture<OpenScannerResponse> callOpenScanner(HBaseRpcController controller,
113      HRegionLocation loc, ClientService.Interface stub) {
114    CompletableFuture<OpenScannerResponse> future = new CompletableFuture<>();
115    try {
116      ScanRequest request = RequestConverter.buildScanRequest(loc.getRegionInfo().getRegionName(),
117        scan, scan.getCaching(), false);
118      stub.scan(controller, request, resp -> {
119        if (controller.failed()) {
120          future.completeExceptionally(controller.getFailed());
121          return;
122        }
123        future.complete(new OpenScannerResponse(loc, stub, controller, resp));
124      });
125    } catch (IOException e) {
126      future.completeExceptionally(e);
127    }
128    return future;
129  }
130
131  private void startScan(OpenScannerResponse resp) {
132    conn.callerFactory.scanSingleRegion().id(resp.resp.getScannerId()).location(resp.loc)
133        .stub(resp.stub).setScan(scan).consumer(consumer).resultCache(resultCache)
134        .rpcTimeout(rpcTimeoutNs, TimeUnit.NANOSECONDS)
135        .scanTimeout(scanTimeoutNs, TimeUnit.NANOSECONDS).pause(pauseNs, TimeUnit.NANOSECONDS)
136        .maxAttempts(maxAttempts).startLogErrorsCnt(startLogErrorsCnt).start(resp.controller, resp.resp)
137        .whenComplete((hasMore, error) -> {
138          if (error != null) {
139            consumer.onError(error);
140            return;
141          }
142          if (hasMore) {
143            openScanner();
144          } else {
145            consumer.onComplete();
146          }
147        });
148  }
149
150  private void openScanner() {
151    conn.callerFactory.<OpenScannerResponse> single().table(tableName).row(scan.getStartRow())
152        .locateType(getLocateType(scan)).rpcTimeout(rpcTimeoutNs, TimeUnit.NANOSECONDS)
153        .operationTimeout(scanTimeoutNs, TimeUnit.NANOSECONDS).pause(pauseNs, TimeUnit.NANOSECONDS)
154        .maxAttempts(maxAttempts).startLogErrorsCnt(startLogErrorsCnt).action(this::callOpenScanner)
155        .call().whenComplete((resp, error) -> {
156          if (error != null) {
157            consumer.onError(error);
158            return;
159          }
160          startScan(resp);
161        });
162  }
163
164  public void start() {
165    openScanner();
166  }
167}
