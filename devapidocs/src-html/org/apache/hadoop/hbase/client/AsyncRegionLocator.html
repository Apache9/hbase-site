061  private static final Log LOG = LogFactory.getLog(AsyncRegionLocator.class);

063  private final AsyncConnectionImpl conn;

065  private final AtomicReference<HRegionLocation> metaRegionLocation = new AtomicReference<>();
066
067  private final AtomicReference<CompletableFuture<HRegionLocation>> metaRelocateFuture =
068      new AtomicReference<>();

070  private final ConcurrentMap<TableName, ConcurrentNavigableMap<byte[], HRegionLocation>> cache =
071      new ConcurrentHashMap<>();

073  AsyncRegionLocator(AsyncConnectionImpl conn) {
074    this.conn = conn;

077  private CompletableFuture<HRegionLocation> locateMetaRegion() {
078    for (;;) {
079      HRegionLocation metaRegionLocation = this.metaRegionLocation.get();
080      if (metaRegionLocation != null) {
081        return CompletableFuture.completedFuture(metaRegionLocation);
082      }
083      if (LOG.isTraceEnabled()) {
084        LOG.trace("Meta region location cache is null, try fetching from registry.");
085      }
086      if (metaRelocateFuture.compareAndSet(null, new CompletableFuture<>())) {
087        if (LOG.isDebugEnabled()) {
088          LOG.debug("Start fetching meta region location from registry.");
089        }
090        CompletableFuture<HRegionLocation> future = metaRelocateFuture.get();
091        conn.registry.getMetaRegionLocation().whenComplete((locs, error) -> {
092          if (error != null) {
093            if (LOG.isDebugEnabled()) {
094              LOG.debug("Failed to fetch meta region location from registry", error);
095            }
096            metaRelocateFuture.getAndSet(null).completeExceptionally(error);
097            return;
098          }
099          HRegionLocation loc = locs.getDefaultRegionLocation();
100          if (LOG.isDebugEnabled()) {
101            LOG.debug("The fetched meta region location is " + loc);
102          }
103          // Here we update cache before reset future, so it is possible that someone can get a
104          // stale value. Consider this:
105          // 1. update cache
106          // 2. someone clear the cache and relocate again
107          // 3. the metaRelocateFuture is not null so the old future is used.
108          // 4. we clear metaRelocateFuture and complete the future in it with the value being
109          // cleared in step 2.
110          // But we do not think it is a big deal as it rarely happens, and even if it happens, the
111          // caller will retry again later, no correctness problems.
112          this.metaRegionLocation.set(loc);
113          metaRelocateFuture.set(null);
114          future.complete(loc);
115        });
116      } else {
117        CompletableFuture<HRegionLocation> future = metaRelocateFuture.get();
118        if (future != null) {
119          return future;
120        }
121      }
122    }
123  }
124
125  private static ConcurrentNavigableMap<byte[], HRegionLocation> createTableCache() {
126    return new ConcurrentSkipListMap<>(Bytes.BYTES_COMPARATOR);
127  }
128
129  private void removeFromCache(HRegionLocation loc) {
130    ConcurrentNavigableMap<byte[], HRegionLocation> tableCache =
131        cache.get(loc.getRegionInfo().getTable());
132    if (tableCache == null) {
133      return;
134    }
135    tableCache.computeIfPresent(loc.getRegionInfo().getStartKey(), (k, oldLoc) -> {
136      if (oldLoc.getSeqNum() > loc.getSeqNum()
137          || !oldLoc.getServerName().equals(loc.getServerName())) {
138        return oldLoc;
139      }
140      return null;
141    });
142  }
143
144  private void addToCache(HRegionLocation loc) {
145    if (LOG.isTraceEnabled()) {
146      LOG.trace("Try adding " + loc + " to cache");
147    }
148    ConcurrentNavigableMap<byte[], HRegionLocation> tableCache = computeIfAbsent(cache,
149      loc.getRegionInfo().getTable(), AsyncRegionLocator::createTableCache);
150    byte[] startKey = loc.getRegionInfo().getStartKey();
151    HRegionLocation oldLoc = tableCache.putIfAbsent(startKey, loc);
152    if (oldLoc == null) {
153      return;
154    }
155    if (oldLoc.getSeqNum() > loc.getSeqNum()
156        || oldLoc.getServerName().equals(loc.getServerName())) {
157      if (LOG.isTraceEnabled()) {
158        LOG.trace("Will not add " + loc + " to cache because the old value " + oldLoc
159            + " is newer than us or has the same server name");
160      }
161      return;
162    }
163    tableCache.compute(startKey, (k, oldValue) -> {
164      if (oldValue == null || oldValue.getSeqNum() <= loc.getSeqNum()) {
165        return loc;
166      }
167      if (LOG.isTraceEnabled()) {
168        LOG.trace("Will not add " + loc + " to cache because the old value " + oldValue
169            + " is newer than us or has the same server name."
170            + " Maybe it is updated before we replace it");
171      }
172      return oldValue;
173    });
174  }
175
176  private HRegionLocation locateInCache(TableName tableName, byte[] row) {
177    ConcurrentNavigableMap<byte[], HRegionLocation> tableCache = cache.get(tableName);
178    if (tableCache == null) {
179      return null;
180    }
181    Map.Entry<byte[], HRegionLocation> entry = tableCache.floorEntry(row);
182    if (entry == null) {
183      return null;
184    }
185    HRegionLocation loc = entry.getValue();
186    byte[] endKey = loc.getRegionInfo().getEndKey();
187    if (isEmptyStopRow(endKey) || Bytes.compareTo(row, endKey) < 0) {
188      return loc;
189    } else {
190      return null;
191    }
192  }
193
194  private void onScanComplete(CompletableFuture<HRegionLocation> future, TableName tableName,
195      byte[] row, List<Result> results, Throwable error, String rowNameInErrorMsg,
196      Consumer<HRegionLocation> otherCheck) {
197    if (error != null) {
198      if (LOG.isDebugEnabled()) {
199        LOG.debug("Failed to fetch location of '" + tableName + "', " + rowNameInErrorMsg + "='"
200            + Bytes.toStringBinary(row) + "'",
201          error);
202      }
203      future.completeExceptionally(error);
204      return;
205    }
206    if (results.isEmpty()) {
207      future.completeExceptionally(new TableNotFoundException(tableName));
208      return;
209    }
210    RegionLocations locs = MetaTableAccessor.getRegionLocations(results.get(0));
211    if (LOG.isDebugEnabled()) {
212      LOG.debug("The fetched location of '" + tableName + "', " + rowNameInErrorMsg + "='"
213          + Bytes.toStringBinary(row) + "' is " + locs);
214    }
215    if (locs == null || locs.getDefaultRegionLocation() == null) {
216      future.completeExceptionally(
217        new IOException(String.format("No location found for '%s', %s='%s'", tableName,
218          rowNameInErrorMsg, Bytes.toStringBinary(row))));
219      return;
220    }
221    HRegionLocation loc = locs.getDefaultRegionLocation();
222    HRegionInfo info = loc.getRegionInfo();
223    if (info == null) {
224      future.completeExceptionally(
225        new IOException(String.format("HRegionInfo is null for '%s', %s='%s'", tableName,
226          rowNameInErrorMsg, Bytes.toStringBinary(row))));
227      return;
228    }
229    if (!info.getTable().equals(tableName)) {
230      future.completeExceptionally(new TableNotFoundException(
231          "Table '" + tableName + "' was not found, got: '" + info.getTable() + "'"));
232      return;
233    }
234    if (info.isSplit()) {
235      future.completeExceptionally(new RegionOfflineException(
236          "the only available region for the required row is a split parent,"
237              + " the daughters should be online soon: '" + info.getRegionNameAsString() + "'"));
238      return;
239    }
240    if (info.isOffline()) {
241      future.completeExceptionally(new RegionOfflineException("the region is offline, could"
242          + " be caused by a disable table call: '" + info.getRegionNameAsString() + "'"));
243      return;
244    }
245    if (loc.getServerName() == null) {
246      future.completeExceptionally(new NoServerForRegionException(
247          String.format("No server address listed for region '%s', %s='%s'",
248            info.getRegionNameAsString(), rowNameInErrorMsg, Bytes.toStringBinary(row))));
249      return;
250    }
251    otherCheck.accept(loc);
252    if (future.isDone()) {
253      return;
254    }
255    addToCache(loc);
256    future.complete(loc);
257  }

259  private CompletableFuture<HRegionLocation> locateInMeta(TableName tableName, byte[] row) {
260    if (LOG.isTraceEnabled()) {
261      LOG.trace("Try locate '" + tableName + "', row='" + Bytes.toStringBinary(row) + "' in meta");
262    }
263    CompletableFuture<HRegionLocation> future = new CompletableFuture<>();
264    byte[] metaKey = createRegionName(tableName, row, NINES, false);
265    conn.getRawTable(META_TABLE_NAME)
266        .smallScan(new Scan(metaKey).setReversed(true).setSmall(true).addFamily(CATALOG_FAMILY), 1)
267        .whenComplete(
268          (results, error) -> onScanComplete(future, tableName, row, results, error, "row", loc -> {
269          }));
270    return future;
271  }
272
273  private CompletableFuture<HRegionLocation> locateRegion(TableName tableName, byte[] row) {
274    HRegionLocation loc = locateInCache(tableName, row);
275    if (loc != null) {
276      if (LOG.isTraceEnabled()) {
277        LOG.trace("Found " + loc + " in cache for '" + tableName + "', row='"
278            + Bytes.toStringBinary(row) + "'");
279      }
280      return CompletableFuture.completedFuture(loc);
281    }
282    return locateInMeta(tableName, row);
283  }
284
285  CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row) {
286    if (tableName.equals(META_TABLE_NAME)) {
287      return locateMetaRegion();
288    } else {
289      return locateRegion(tableName, row);
290    }
291  }

293  private HRegionLocation locatePreviousInCache(TableName tableName,
294      byte[] startRowOfCurrentRegion) {
295    ConcurrentNavigableMap<byte[], HRegionLocation> tableCache = cache.get(tableName);
296    if (tableCache == null) {
297      return null;
298    }
299    Map.Entry<byte[], HRegionLocation> entry;
300    if (isEmptyStopRow(startRowOfCurrentRegion)) {
301      entry = tableCache.lastEntry();
302    } else {
303      entry = tableCache.lowerEntry(startRowOfCurrentRegion);
304    }
305    if (entry == null) {
306      return null;
307    }
308    HRegionLocation loc = entry.getValue();
309    if (Bytes.equals(loc.getRegionInfo().getEndKey(), startRowOfCurrentRegion)) {
310      return loc;
311    } else {
312      return null;
313    }
314  }

316  private CompletableFuture<HRegionLocation> locatePreviousInMeta(TableName tableName,
317      byte[] startRowOfCurrentRegion) {
318    if (LOG.isTraceEnabled()) {
319      LOG.trace("Try locate '" + tableName + "', startRowOfCurrentRegion='"
320          + Bytes.toStringBinary(startRowOfCurrentRegion) + "' in meta");
321    }
322    byte[] metaKey;
323    if (isEmptyStopRow(startRowOfCurrentRegion)) {
324      byte[] binaryTableName = tableName.getName();
325      metaKey = Arrays.copyOf(binaryTableName, binaryTableName.length + 1);
326    } else {
327      metaKey = createRegionName(tableName, startRowOfCurrentRegion, ZEROES, false);
328    }
329    CompletableFuture<HRegionLocation> future = new CompletableFuture<>();
330    conn.getRawTable(META_TABLE_NAME)
331        .smallScan(new Scan(metaKey).setReversed(true).setSmall(true).addFamily(CATALOG_FAMILY), 1)
332        .whenComplete((results, error) -> onScanComplete(future, tableName, startRowOfCurrentRegion,
333          results, error, "startRowOfCurrentRegion", loc -> {
334            HRegionInfo info = loc.getRegionInfo();
335            if (!Bytes.equals(info.getEndKey(), startRowOfCurrentRegion)) {
336              future.completeExceptionally(new IOException("The end key of '"
337                  + info.getRegionNameAsString() + "' is '" + Bytes.toStringBinary(info.getEndKey())
338                  + "', expected '" + Bytes.toStringBinary(startRowOfCurrentRegion) + "'"));
339            }
340          }));
341    return future;
342  }
343
344  private CompletableFuture<HRegionLocation> locatePreviousRegion(TableName tableName,
345      byte[] startRowOfCurrentRegion) {
346    HRegionLocation loc = locatePreviousInCache(tableName, startRowOfCurrentRegion);
347    if (loc != null) {
348      if (LOG.isTraceEnabled()) {
349        LOG.trace("Found " + loc + " in cache for '" + tableName + "', startRowOfCurrentRegion='"
350            + Bytes.toStringBinary(startRowOfCurrentRegion) + "'");
351      }
352      return CompletableFuture.completedFuture(loc);
353    }
354    return locatePreviousInMeta(tableName, startRowOfCurrentRegion);
355  }

357  /**
358   * Locate the previous region using the current regions start key. Used for reverse scan.
359   */
360  CompletableFuture<HRegionLocation> getPreviousRegionLocation(TableName tableName,
361      byte[] startRowOfCurrentRegion) {
362    if (tableName.equals(META_TABLE_NAME)) {
363      return locateMetaRegion();
364    } else {
365      return locatePreviousRegion(tableName, startRowOfCurrentRegion);
366    }
367  }

369  private boolean canUpdate(HRegionLocation loc, HRegionLocation oldLoc) {
370    // Do not need to update if no such location, or the location is newer, or the location is not
371    // same with us
372    return oldLoc != null && oldLoc.getSeqNum() <= loc.getSeqNum()
373        && oldLoc.getServerName().equals(loc.getServerName());
374  }
375
376  private void updateCachedLoation(HRegionLocation loc, Throwable exception,
377      Function<HRegionLocation, HRegionLocation> cachedLocationSupplier,
378      Consumer<HRegionLocation> addToCache, Consumer<HRegionLocation> removeFromCache) {
379    HRegionLocation oldLoc = cachedLocationSupplier.apply(loc);
380    if (LOG.isDebugEnabled()) {
381      LOG.debug("Try updating " + loc + ", the old value is " + oldLoc, exception);
382    }
383    if (!canUpdate(loc, oldLoc)) {
384      return;
385    }
386    Throwable cause = findException(exception);
387    if (LOG.isDebugEnabled()) {
388      LOG.debug("The actual exception when updating " + loc, cause);
389    }
390    if (cause == null || !isMetaClearingException(cause)) {
391      if (LOG.isDebugEnabled()) {
392        LOG.debug(
393          "Will not update " + loc + " because the exception is null or not the one we care about");
394      }
395      return;
396    }
397    if (cause instanceof RegionMovedException) {
398      RegionMovedException rme = (RegionMovedException) cause;
399      HRegionLocation newLoc =
400          new HRegionLocation(loc.getRegionInfo(), rme.getServerName(), rme.getLocationSeqNum());
401      if (LOG.isDebugEnabled()) {
402        LOG.debug(
403          "Try updating " + loc + " with the new location " + newLoc + " constructed by " + rme);
404      }
405      addToCache.accept(newLoc);
406    } else {
407      if (LOG.isDebugEnabled()) {
408        LOG.debug("Try removing " + loc + " from cache");
409      }
410      removeFromCache.accept(loc);
411    }
412  }
413
414  void updateCachedLocation(HRegionLocation loc, Throwable exception) {
415    if (loc.getRegionInfo().isMetaTable()) {
416      updateCachedLoation(loc, exception, l -> metaRegionLocation.get(), newLoc -> {
417        for (;;) {
418          HRegionLocation oldLoc = metaRegionLocation.get();
419          if (oldLoc != null && (oldLoc.getSeqNum() > newLoc.getSeqNum()
420              || oldLoc.getServerName().equals(newLoc.getServerName()))) {
421            return;
422          }
423          if (metaRegionLocation.compareAndSet(oldLoc, newLoc)) {
424            return;
425          }
426        }
427      }, l -> {
428        for (;;) {
429          HRegionLocation oldLoc = metaRegionLocation.get();
430          if (!canUpdate(l, oldLoc) || metaRegionLocation.compareAndSet(oldLoc, null)) {
431            return;
432          }
433        }
434      });
435    } else {
436      updateCachedLoation(loc, exception, l -> {
437        ConcurrentNavigableMap<byte[], HRegionLocation> tableCache =
438            cache.get(l.getRegionInfo().getTable());
439        if (tableCache == null) {
440          return null;
441        }
442        return tableCache.get(l.getRegionInfo().getStartKey());
443      }, this::addToCache, this::removeFromCache);
444    }
445  }
446
447  void clearCache(TableName tableName) {
448    if (LOG.isDebugEnabled()) {
449      LOG.debug("Clear meta cache for " + tableName);
450    }
451    cache.remove(tableName);
452  }
