267    Future<Boolean> future = abortProcedureAsync(procId, mayInterruptIfRunning);
268    try {
269      return future.get(syncWaitTimeout, TimeUnit.MILLISECONDS);
270    } catch (InterruptedException e) {

274    } catch (ExecutionException e) {
275      if (e.getCause() instanceof IOException) {
276        throw (IOException)e.getCause();
277      } else {
278        throw new IOException(e.getCause());
279      }
280    }

506  static HTableDescriptor getTableDescriptor(final TableName tableName,
507         HConnection connection, RpcRetryingCallerFactory rpcCallerFactory,
508         int operationTimeout) throws TableNotFoundException, IOException {
509
510      if (tableName == null) return null;
511      HTableDescriptor htd = executeCallable(new MasterCallable<HTableDescriptor>(connection) {
512        @Override
513        public HTableDescriptor call(int callTimeout) throws ServiceException {
514          GetTableDescriptorsResponse htds;
515          GetTableDescriptorsRequest req =
516                  RequestConverter.buildGetTableDescriptorsRequest(tableName);
517          htds = master.getTableDescriptors(null, req);
518
519          if (!htds.getTableSchemaList().isEmpty()) {
520            return HTableDescriptor.convert(htds.getTableSchemaList().get(0));
521          }
522          return null;
523        }
524      }, rpcCallerFactory, operationTimeout);
525      if (htd != null) {
526        return htd;
527      }
528      throw new TableNotFoundException(tableName.getNameAsString());
529  }

537    int triesCount = tries;

540    }
541    return this.pause * HConstants.RETRY_BACKOFF[triesCount];

592      throw new IllegalArgumentException("Must create at least three regions");
593    } else if(Bytes.compareTo(startKey, endKey) >= 0) {
594      throw new IllegalArgumentException("Start key must be smaller than end key");
595    }
596    if (numRegions == 3) {
597      createTable(desc, new byte[][]{startKey, endKey});
598      return;
599    }
600    byte [][] splitKeys = Bytes.split(startKey, endKey, numRegions - 3);
601    if(splitKeys == null || splitKeys.length != numRegions - 1) {
602      throw new IllegalArgumentException("Unable to split key range into enough regions");
603    }
604    createTable(desc, splitKeys);
605  }

676          throw new IllegalArgumentException("All split keys must be unique, " +
677            "found duplicate: " + Bytes.toStringBinary(splitKey) +

994   * Wait for the table to be enabled and available
995   * If enabling the table exceeds the retry period, an exception is thrown.
996   * @param tableName name of the table
997   * @throws IOException if a remote or network exception occurs or
998   *    table is not enabled after the retries period.
999   */
1000  private void waitUntilTableIsEnabled(final TableName tableName) throws IOException {
1001    boolean enabled = false;
1002    long start = EnvironmentEdgeManager.currentTime();
1003    for (int tries = 0; tries < (this.numRetries * this.retryLongerMultiplier); tries++) {
1004      try {
1005        enabled = isTableEnabled(tableName);
1006      } catch (TableNotFoundException tnfe) {
1007        // wait for table to be created
1008        enabled = false;
1009      }
1010      enabled = enabled && isTableAvailable(tableName);
1011      if (enabled) {
1012        break;
1013      }
1014      long sleep = getPauseTime(tries);
1015      if (LOG.isDebugEnabled()) {
1016        LOG.debug("Sleeping= " + sleep + "ms, waiting for all regions to be " +
1017          "enabled in " + tableName);
1018      }
1019      try {
1020        Thread.sleep(sleep);
1021      } catch (InterruptedException e) {
1022        // Do this conversion rather than let it out because do not want to
1023        // change the method signature.
1024        throw (InterruptedIOException)new InterruptedIOException("Interrupted").initCause(e);
1025      }
1026    }
1027    if (!enabled) {
1028      long msec = EnvironmentEdgeManager.currentTime() - start;
1029      throw new IOException("Table '" + tableName +
1030        "' not yet enabled, after " + msec + "ms.");
1031    }
1032  }

1917  /**
1918   * Compact a table.
1919   * Asynchronous operation.
1920   *
1921   * @param tableName table or region to compact
1922   * @param columnFamily column family within a table or region
1923   * @param major True if we are to do a major compaction.
1924   * @throws IOException if a remote or network exception occurs
1925   * @throws InterruptedException
1926   */
1927  private void compact(final TableName tableName, final byte[] columnFamily,final boolean major,
1928                       CompactType compactType) throws IOException {
1929    switch (compactType) {
1930      case MOB:
1931        ServerName master = getMasterAddress();
1932        compact(master, getMobRegionInfo(tableName), major, columnFamily);
1933        break;
1934      case NORMAL:
1935      default:
1936        ZooKeeperWatcher zookeeper = null;
1937        try {
1938          checkTableExists(tableName);
1939          zookeeper = new ZooKeeperWatcher(conf, ZK_IDENTIFIER_PREFIX + connection.toString(),
1940                  new ThrowableAbortable());
1941          List<Pair<HRegionInfo, ServerName>> pairs;
1942          if (TableName.META_TABLE_NAME.equals(tableName)) {
1943            pairs = new MetaTableLocator().getMetaRegionsAndLocations(zookeeper);
1944          } else {
1945            pairs = MetaTableAccessor.getTableRegionsAndLocations(connection, tableName);
1946          }
1947          for (Pair<HRegionInfo, ServerName> pair: pairs) {
1948            if (pair.getFirst().isOffline()) continue;
1949            if (pair.getSecond() == null) continue;
1950            try {
1951              compact(pair.getSecond(), pair.getFirst(), major, columnFamily);
1952            } catch (NotServingRegionException e) {
1953              if (LOG.isDebugEnabled()) {
1954                LOG.debug("Trying to" + (major ? " major" : "") + " compact " +
1955                        pair.getFirst() + ": " +
1956                        StringUtils.stringifyException(e));
1957              }
1958            }
1959          }
1960        } finally {
1961          if (zookeeper != null) {
1962            zookeeper.close();
1963          }
1964        }
1965        break;
1966    }
1967  }

2343  @Override
2344  public void split(final TableName tableName, final byte [] splitPoint)
2345  throws IOException {
2346    ZooKeeperWatcher zookeeper = null;
2347    try {
2348      checkTableExists(tableName);
2349      zookeeper = new ZooKeeperWatcher(conf, ZK_IDENTIFIER_PREFIX + connection.toString(),
2350        new ThrowableAbortable());
2351      List<Pair<HRegionInfo, ServerName>> pairs;
2352      if (TableName.META_TABLE_NAME.equals(tableName)) {
2353        pairs = new MetaTableLocator().getMetaRegionsAndLocations(zookeeper);
2354      } else {
2355        pairs = MetaTableAccessor.getTableRegionsAndLocations(connection, tableName);
2356      }
2357      for (Pair<HRegionInfo, ServerName> pair: pairs) {
2358        // May not be a server for a particular row
2359        if (pair.getSecond() == null) continue;
2360        HRegionInfo r = pair.getFirst();
2361        // check for parents
2362        if (r.isSplitParent()) continue;
2363        // if a split point given, only split that particular region
2364        if (r.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID ||
2365           (splitPoint != null && !r.containsRow(splitPoint))) continue;
2366        // call out to region server to do split now
2367        split(pair.getSecond(), pair.getFirst(), splitPoint);
2368      }
2369    } finally {
2370      if (zookeeper != null) {
2371        zookeeper.close();
2372      }
2373    }
2374  }

4142    CompactType compactType) throws IOException {

4144    checkTableExists(tableName);
4145    switch (compactType) {
4146      case MOB:

4331    private V waitProcedureResult(long procId, long deadlineTs)
4332        throws IOException, TimeoutException, InterruptedException {
4333      GetProcedureResultRequest request = GetProcedureResultRequest.newBuilder()
4334          .setProcId(procId)
4335          .build();
4336
4337      int tries = 0;
4338      IOException serviceEx = null;
4339      while (EnvironmentEdgeManager.currentTime() < deadlineTs) {
4340        GetProcedureResultResponse response = null;
4341        try {
4342          // Try to fetch the result
4343          response = getProcedureResult(request);
4344        } catch (IOException e) {
4345          serviceEx = unwrapException(e);
4346
4347          // the master may be down
4348          LOG.warn("failed to get the procedure result procId=" + procId, serviceEx);
4349
4350          // Not much to do, if we have a DoNotRetryIOException
4351          if (serviceEx instanceof DoNotRetryIOException) {
4352            // TODO: looks like there is no way to unwrap this exception and get the proper
4353            // UnsupportedOperationException aside from looking at the message.
4354            // anyway, if we fail here we just failover to the compatibility side
4355            // and that is always a valid solution.
4356            LOG.warn("Proc-v2 is unsupported on this master: " + serviceEx.getMessage(), serviceEx);
4357            procResultFound = false;
4358            return null;
4359          }
4360        }
4361
4362        // If the procedure is no longer running, we should have a result
4363        if (response != null && response.getState() != GetProcedureResultResponse.State.RUNNING) {
4364          procResultFound = response.getState() != GetProcedureResultResponse.State.NOT_FOUND;
4365          return convertResult(response);
4366        }
4367
4368        try {
4369          Thread.sleep(getAdmin().getPauseTime(tries++));
4370        } catch (InterruptedException e) {
4371          throw new InterruptedException(
4372            "Interrupted while waiting for the result of proc " + procId);
4373        }
4374      }
4375      if (serviceEx != null) {
4376        throw serviceEx;
4377      } else {
4378        throw new TimeoutException("The procedure " + procId + " is still running");
4379      }
4380    }

4598    protected void waitForAllRegionsOnline(final long deadlineTs, final byte[][] splitKeys)
4599        throws IOException, TimeoutException {
4600      final HTableDescriptor desc = getTableDescriptor();
4601      final AtomicInteger actualRegCount = new AtomicInteger(0);
4602      final MetaTableAccessor.Visitor visitor = new MetaTableAccessor.Visitor() {
4603        @Override
4604        public boolean visit(Result rowResult) throws IOException {
4605          RegionLocations list = MetaTableAccessor.getRegionLocations(rowResult);
4606          if (list == null) {
4607            LOG.warn("No serialized HRegionInfo in " + rowResult);
4608            return true;
4609          }
4610          HRegionLocation l = list.getRegionLocation();
4611          if (l == null) {
4612            return true;
4613          }
4614          if (!l.getRegionInfo().getTable().equals(desc.getTableName())) {
4615            return false;
4616          }
4617          if (l.getRegionInfo().isOffline() || l.getRegionInfo().isSplit()) return true;
4618          HRegionLocation[] locations = list.getRegionLocations();
4619          for (HRegionLocation location : locations) {
4620            if (location == null) continue;
4621            ServerName serverName = location.getServerName();
4622            // Make sure that regions are assigned to server
4623            if (serverName != null && serverName.getHostAndPort() != null) {
4624              actualRegCount.incrementAndGet();
4625            }
4626          }
4627          return true;
4628        }
4629      };
4630
4631      int tries = 0;
4632      int numRegs = (splitKeys == null ? 1 : splitKeys.length + 1) * desc.getRegionReplication();
4633      while (EnvironmentEdgeManager.currentTime() < deadlineTs) {
4634        actualRegCount.set(0);
4635        MetaTableAccessor.scanMetaForTableRegions(getAdmin().getConnection(), visitor,
4636          desc.getTableName());
4637        if (actualRegCount.get() == numRegs) {
4638          // all the regions are online
4639          return;
4640        }
4641
4642        try {
4643          Thread.sleep(getAdmin().getPauseTime(tries++));
4644        } catch (InterruptedException e) {
4645          throw new InterruptedIOException("Interrupted when opening" + " regions; "
4646              + actualRegCount.get() + " of " + numRegs + " regions processed so far");
4647        }
4648      }
4649      throw new TimeoutException("Only " + actualRegCount.get() + " of " + numRegs
4650          + " regions are online; retries exhausted.");
4651    }
