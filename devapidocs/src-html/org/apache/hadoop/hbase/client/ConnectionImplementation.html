204      HConstants.DEFAULT_USE_META_REPLICAS);

206    this.numTries = tableConfig.getRetriesNumber() + 1;
207    this.rpcTimeout = conf.getInt(
208        HConstants.HBASE_RPC_TIMEOUT_KEY,
209        HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
210    if (conf.getBoolean(CLIENT_NONCES_ENABLED_KEY, true)) {
211      synchronized (nonceGeneratorCreateLock) {
212        if (nonceGenerator == null) {
213          nonceGenerator = new PerClientRandomNonceGenerator();
214        }
215      }
216    } else {
217      nonceGenerator = new NoNonceGenerator();
218    }
219
220    this.stats = ServerStatisticTracker.create(conf);
221    this.interceptor = (new RetryingCallerInterceptorFactory(conf)).build();
222    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
223    this.rpcCallerFactory = RpcRetryingCallerFactory.instantiate(conf, interceptor, this.stats);
224    this.backoffPolicy = ClientBackoffPolicyFactory.create(conf);
225    this.asyncProcess = createAsyncProcess(this.conf);
226    if (conf.getBoolean(CLIENT_SIDE_METRICS_ENABLED_KEY, false)) {
227      this.metrics = new MetricsConnection(this);
228    } else {
229      this.metrics = null;
230    }
231    this.metaCache = new MetaCache(this.metrics);
232
233    boolean shouldListen = conf.getBoolean(HConstants.STATUS_PUBLISHED,
234        HConstants.STATUS_PUBLISHED_DEFAULT);
235    this.hostnamesCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
236    Class<? extends ClusterStatusListener.Listener> listenerClass =
237        conf.getClass(ClusterStatusListener.STATUS_LISTENER_CLASS,
238            ClusterStatusListener.DEFAULT_STATUS_LISTENER_CLASS,
239            ClusterStatusListener.Listener.class);
240
241    try {
242      this.registry = setupRegistry();
243      retrieveClusterId();
244
245      this.rpcClient = RpcClientFactory.createClient(this.conf, this.clusterId);
246
247      // Do we publish the status?
248      if (shouldListen) {
249        if (listenerClass == null) {
250          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
251              ClusterStatusListener.STATUS_LISTENER_CLASS + " is not set - not listening status");
252        } else {
253          clusterStatusListener = new ClusterStatusListener(
254              new ClusterStatusListener.DeadServerHandler() {
255                @Override
256                public void newDead(ServerName sn) {
257                  clearCaches(sn);
258                  rpcClient.cancelConnections(sn);
259                }
260              }, conf, listenerClass);
261        }
262      }
263    } catch (Throwable e) {
264      // avoid leaks: registry, rpcClient, ...
265      LOG.debug("connection construction failed", e);
266      close();
267      throw e;
268    }
269  }

287  /**
288   * Look for an exception we know in the remote exception:
289   * - hadoop.ipc wrapped exceptions
290   * - nested exceptions
291   *
292   * Looks for: RegionMovedException / RegionOpeningException / RegionTooBusyException /
293   *            ThrottlingException
294   * @return null if we didn't find the exception, the exception otherwise.
295   */
296  public static Throwable findException(Object exception) {
297    if (exception == null || !(exception instanceof Throwable)) {
298      return null;
299    }
300    Throwable cur = (Throwable) exception;
301    while (cur != null) {
302      if (cur instanceof RegionMovedException || cur instanceof RegionOpeningException
303          || cur instanceof RegionTooBusyException || cur instanceof ThrottlingException
304          || cur instanceof RetryImmediatelyException) {
305        return cur;
306      }
307      if (cur instanceof RemoteException) {
308        RemoteException re = (RemoteException) cur;
309        cur = re.unwrapRemoteException(
310            RegionOpeningException.class, RegionMovedException.class,
311            RegionTooBusyException.class);
312        if (cur == null) {
313          cur = re.unwrapRemoteException();
314        }
315        // unwrapRemoteException can return the exception given as a parameter when it cannot
316        //  unwrap it. In this case, there is no need to look further
317        // noinspection ObjectEquality
318        if (cur == re) {
319          return null;
320        }
321      } else {
322        cur = cur.getCause();
323      }
324    }
325
326    return null;
327  }

396  private ExecutorService getBatchPool() {
397    if (batchPool == null) {
398      synchronized (this) {
399        if (batchPool == null) {
400          this.batchPool = getThreadPool(conf.getInt("hbase.hconnection.threads.max", 256),
401              conf.getInt("hbase.hconnection.threads.core", 256), "-shared-", null);
402          this.cleanupPool = true;
403        }
404      }
405    }
406    return this.batchPool;
407  }

409  private ExecutorService getThreadPool(int maxThreads, int coreThreads, String nameHint,
410      BlockingQueue<Runnable> passedWorkQueue) {
411    // shared HTable thread executor not yet initialized
412    if (maxThreads == 0) {
413      maxThreads = Runtime.getRuntime().availableProcessors() * 8;
414    }
415    if (coreThreads == 0) {
416      coreThreads = Runtime.getRuntime().availableProcessors() * 8;
417    }
418    long keepAliveTime = conf.getLong("hbase.hconnection.threads.keepalivetime", 60);
419    BlockingQueue<Runnable> workQueue = passedWorkQueue;
420    if (workQueue == null) {
421      workQueue =
422        new LinkedBlockingQueue<Runnable>(maxThreads *
423            conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
424                HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS));
425    }
426    ThreadPoolExecutor tpe = new ThreadPoolExecutor(
427        coreThreads,
428        maxThreads,
429        keepAliveTime,
430        TimeUnit.SECONDS,
431        workQueue,
432        Threads.newDaemonThreadFactory(toString() + nameHint));
433    tpe.allowCoreThreadTimeOut(true);
434    return tpe;
435  }

437  private ExecutorService getMetaLookupPool() {
438    if (this.metaLookupPool == null) {
439      synchronized (this) {
440        if (this.metaLookupPool == null) {
441          //Some of the threads would be used for meta replicas
442          //To start with, threads.max.core threads can hit the meta (including replicas).
443          //After that, requests will get queued up in the passed queue, and only after
444          //the queue is full, a new thread will be started
445          this.metaLookupPool = getThreadPool(
446             conf.getInt("hbase.hconnection.meta.lookup.threads.max", 128),
447             conf.getInt("hbase.hconnection.meta.lookup.threads.core", 10),
448             "-metaLookup-shared-", new LinkedBlockingQueue<Runnable>());
449        }
450      }
451    }
452    return this.metaLookupPool;
453  }

463  private void shutdownPools() {
464    if (this.cleanupPool && this.batchPool != null && !this.batchPool.isShutdown()) {
465      shutdownBatchPool(this.batchPool);
466    }
467    if (this.metaLookupPool != null && !this.metaLookupPool.isShutdown()) {
468      shutdownBatchPool(this.metaLookupPool);
469    }
470  }

472  private void shutdownBatchPool(ExecutorService pool) {
473    pool.shutdown();
474    try {
475      if (!pool.awaitTermination(10, TimeUnit.SECONDS)) {
476        pool.shutdownNow();

525    String errorMsg;
526    try {
527      if (ZKUtil.checkExists(zkw, zkw.baseZNode) == -1) {
528        errorMsg = "The node " + zkw.baseZNode+" is not in ZooKeeper. "
529          + "It should have been written by the master. "
530          + "Check the value configured in 'zookeeper.znode.parent'. "
531          + "There could be a mismatch with the one configured in the master.";
532        LOG.error(errorMsg);
533        throw new MasterNotRunningException(errorMsg);
534      }
535    } catch (KeeperException e) {
536      errorMsg = "Can't get connection to ZooKeeper: " + e.getMessage();
537      LOG.error(errorMsg);
538      throw new MasterNotRunningException(errorMsg, e);
539    }

604  @Override
605  public boolean isTableAvailable(final TableName tableName, @Nullable final byte[][] splitKeys)
606      throws IOException {
607    if (this.closed) throw new IOException(toString() + " closed");
608    try {
609      if (!isTableEnabled(tableName)) {
610        LOG.debug("Table " + tableName + " not enabled");
611        return false;
612      }
613      List<Pair<HRegionInfo, ServerName>> locations =
614        MetaTableAccessor.getTableRegionsAndLocations(this, tableName, true);
615
616      int notDeployed = 0;
617      int regionCount = 0;
618      for (Pair<HRegionInfo, ServerName> pair : locations) {
619        HRegionInfo info = pair.getFirst();
620        if (pair.getSecond() == null) {
621          if (LOG.isDebugEnabled()) {
622            LOG.debug("Table " + tableName + " has not deployed region " + pair.getFirst()
623                .getEncodedName());
624          }
625          notDeployed++;
626        } else if (splitKeys != null
627            && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
628          for (byte[] splitKey : splitKeys) {
629            // Just check if the splitkey is available
630            if (Bytes.equals(info.getStartKey(), splitKey)) {
631              regionCount++;
632              break;
633            }
634          }
635        } else {
636          // Always empty start row should be counted
637          regionCount++;
638        }
639      }
640      if (notDeployed > 0) {
641        if (LOG.isDebugEnabled()) {
642          LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
643        }
644        return false;
645      } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
646        if (LOG.isDebugEnabled()) {
647          LOG.debug("Table " + tableName + " expected to have " + (splitKeys.length + 1)
648              + " regions, but only " + regionCount + " available");
649        }
650        return false;
651      } else {
652        if (LOG.isDebugEnabled()) {
653          LOG.debug("Table " + tableName + " should be available");
654        }
655        return true;
656      }

788  private RegionLocations locateMeta(final TableName tableName,
789      boolean useCache, int replicaId) throws IOException {
790    // HBASE-10785: We cache the location of the META itself, so that we are not overloading
791    // zookeeper with one request for every region lookup. We cache the META with empty row
792    // key in MetaCache.
793    byte[] metaCacheKey = HConstants.EMPTY_START_ROW; // use byte[0] as the row for meta
794    RegionLocations locations = null;
795    if (useCache) {
796      locations = getCachedLocation(tableName, metaCacheKey);
797      if (locations != null && locations.getRegionLocation(replicaId) != null) {
798        return locations;
799      }
800    }
801
802    // only one thread should do the lookup.
803    synchronized (metaRegionLock) {
804      // Check the cache again for a hit in case some other thread made the
805      // same query while we were waiting on the lock.
806      if (useCache) {
807        locations = getCachedLocation(tableName, metaCacheKey);
808        if (locations != null && locations.getRegionLocation(replicaId) != null) {
809          return locations;
810        }
811      }
812
813      // Look up from zookeeper
814      locations = this.registry.getMetaRegionLocation();
815      if (locations != null) {
816        cacheLocation(tableName, locations);
817      }
818    }
819    return locations;
820  }

822  /*
823    * Search the hbase:meta table for the HRegionLocation
824    * info that contains the table and row we're seeking.
825    */
826  private RegionLocations locateRegionInMeta(TableName tableName, byte[] row,
827                 boolean useCache, boolean retry, int replicaId) throws IOException {
828
829    // If we are supposed to be using the cache, look in the cache to see if
830    // we already have the region.
831    if (useCache) {
832      RegionLocations locations = getCachedLocation(tableName, row);
833      if (locations != null && locations.getRegionLocation(replicaId) != null) {
834        return locations;
835      }
836    }
837
838    // build the key of the meta region we should be looking for.
839    // the extra 9's on the end are necessary to allow "exact" matches
840    // without knowing the precise region names.
841    byte[] metaKey = HRegionInfo.createRegionName(tableName, row, HConstants.NINES, false);
842
843    Scan s = new Scan();
844    s.setReversed(true);
845    s.setStartRow(metaKey);
846    s.setSmall(true);
847    s.setCaching(1);
848    if (this.useMetaReplicas) {
849      s.setConsistency(Consistency.TIMELINE);
850    }
851
852    int maxAttempts = (retry ? numTries : 1);
853
854    for (int tries = 0; true; tries++) {
855      if (tries >= maxAttempts) {
856        throw new NoServerForRegionException("Unable to find region for "
857            + Bytes.toStringBinary(row) + " in " + tableName +
858            " after " + tries + " tries.");
859      }
860      if (useCache) {
861        RegionLocations locations = getCachedLocation(tableName, row);
862        if (locations != null && locations.getRegionLocation(replicaId) != null) {
863          return locations;
864        }
865      } else {
866        // If we are not supposed to be using the cache, delete any existing cached location
867        // so it won't interfere.
868        metaCache.clearCache(tableName, row);
869      }
870
871      // Query the meta region
872      try {
873        Result regionInfoRow = null;
874        ReversedClientScanner rcs = null;
875        try {
876          rcs = new ClientSmallReversedScanner(conf, s, TableName.META_TABLE_NAME, this,
877            rpcCallerFactory, rpcControllerFactory, getMetaLookupPool(), 0);
878          regionInfoRow = rcs.next();
879        } finally {
880          if (rcs != null) {
881            rcs.close();
882          }
883        }
884
885        if (regionInfoRow == null) {
886          throw new TableNotFoundException(tableName);
887        }
888
889        // convert the row result into the HRegionLocation we need!
890        RegionLocations locations = MetaTableAccessor.getRegionLocations(regionInfoRow);
891        if (locations == null || locations.getRegionLocation(replicaId) == null) {
892          throw new IOException("HRegionInfo was null in " +
893            tableName + ", row=" + regionInfoRow);
894        }
895        HRegionInfo regionInfo = locations.getRegionLocation(replicaId).getRegionInfo();
896        if (regionInfo == null) {
897          throw new IOException("HRegionInfo was null or empty in " +
898            TableName.META_TABLE_NAME + ", row=" + regionInfoRow);
899        }
900
901        // possible we got a region of a different table...
902        if (!regionInfo.getTable().equals(tableName)) {
903          throw new TableNotFoundException(
904                "Table '" + tableName + "' was not found, got: " +
905                regionInfo.getTable() + ".");
906        }
907        if (regionInfo.isSplit()) {
908          throw new RegionOfflineException("the only available region for" +
909            " the required row is a split parent," +
910            " the daughters should be online soon: " +
911            regionInfo.getRegionNameAsString());
912        }
913        if (regionInfo.isOffline()) {
914          throw new RegionOfflineException("the region is offline, could" +
915            " be caused by a disable table call: " +
916            regionInfo.getRegionNameAsString());
917        }
918
919        ServerName serverName = locations.getRegionLocation(replicaId).getServerName();
920        if (serverName == null) {
921          throw new NoServerForRegionException("No server address listed " +
922            "in " + TableName.META_TABLE_NAME + " for region " +
923            regionInfo.getRegionNameAsString() + " containing row " +
924            Bytes.toStringBinary(row));
925        }
926
927        if (isDeadServer(serverName)){
928          throw new RegionServerStoppedException("hbase:meta says the region "+
929              regionInfo.getRegionNameAsString()+" is managed by the server " + serverName +
930              ", but it is dead.");
931        }
932        // Instantiate the location
933        cacheLocation(tableName, locations);
934        return locations;
935      } catch (TableNotFoundException e) {
936        // if we got this error, probably means the table just plain doesn't
937        // exist. rethrow the error immediately. this should always be coming
938        // from the HTable constructor.
939        throw e;
940      } catch (IOException e) {
941        ExceptionUtil.rethrowIfInterrupt(e);
942
943        if (e instanceof RemoteException) {
944          e = ((RemoteException)e).unwrapRemoteException();
945        }
946        if (tries < maxAttempts - 1) {
947          if (LOG.isDebugEnabled()) {
948            LOG.debug("locateRegionInMeta parentTable=" +
949                TableName.META_TABLE_NAME + ", metaLocation=" +
950              ", attempt=" + tries + " of " +
951              maxAttempts + " failed; retrying after sleep of " +
952              ConnectionUtils.getPauseTime(this.pause, tries) + " because: " + e.getMessage());
953          }
954        } else {
955          throw e;
956        }
957        // Only relocate the parent region if necessary
958        if(!(e instanceof RegionOfflineException ||
959            e instanceof NoServerForRegionException)) {
960          relocateRegion(TableName.META_TABLE_NAME, metaKey, replicaId);
961        }
962      }
963      try{
964        Thread.sleep(ConnectionUtils.getPauseTime(this.pause, tries));
965      } catch (InterruptedException e) {
966        throw new InterruptedIOException("Giving up trying to location region in " +
967          "meta: thread is interrupted.");
968      }
969    }
970  }

1087  static class ServerErrorTracker {
1088    // We need a concurrent map here, as we could have multiple threads updating it in parallel.
1089    private final ConcurrentMap<ServerName, ServerErrors> errorsByServer =
1090        new ConcurrentHashMap<ServerName, ServerErrors>();
1091    private final long canRetryUntil;
1092    private final int maxTries;// max number to try
1093    private final long startTrackingTime;
1094
1095    /**
1096     * Constructor
1097     * @param timeout how long to wait before timeout, in unit of millisecond
1098     * @param maxTries how many times to try
1099     */
1100    public ServerErrorTracker(long timeout, int maxTries) {
1101      this.maxTries = maxTries;
1102      this.canRetryUntil = EnvironmentEdgeManager.currentTime() + timeout;
1103      this.startTrackingTime = new Date().getTime();
1104    }

1106    /**
1107     * We stop to retry when we have exhausted BOTH the number of tries and the time allocated.
1108     * @param numAttempt how many times we have tried by now
1109     */
1110    boolean canTryMore(int numAttempt) {
1111      // If there is a single try we must not take into account the time.
1112      return numAttempt < maxTries || (maxTries > 1 &&
1113          EnvironmentEdgeManager.currentTime() < this.canRetryUntil);
1114    }

1116    /**
1117     * Calculates the back-off time for a retrying request to a particular server.
1118     *
1119     * @param server    The server in question.
1120     * @param basePause The default hci pause.
1121     * @return The time to wait before sending next request.
1122     */
1123    long calculateBackoffTime(ServerName server, long basePause) {
1124      long result;
1125      ServerErrors errorStats = errorsByServer.get(server);
1126      if (errorStats != null) {
1127        result = ConnectionUtils.getPauseTime(basePause, errorStats.getCount());
1128      } else {
1129        result = 0; // yes, if the server is not in our list we don't wait before retrying.
1130      }
1131      return result;
1132    }

1147        }
1148      }

1195     * Create a stub. Try once only.  It is not typed because there is no common type to
1196     * protobuf services nor their interfaces.  Let the caller do appropriate casting.
1197     * @return A stub for master services.
1198     * @throws java.io.IOException
1199     * @throws org.apache.zookeeper.KeeperException
1200     * @throws com.google.protobuf.ServiceException
1201     */
1202    private Object makeStubNoRetries() throws IOException, KeeperException, ServiceException {
1203      ZooKeeperKeepAliveConnection zkw;
1204      try {
1205        zkw = getKeepAliveZooKeeperWatcher();
1206      } catch (IOException e) {
1207        ExceptionUtil.rethrowIfInterrupt(e);
1208        throw new ZooKeeperConnectionException("Can't connect to ZooKeeper", e);

1211        checkIfBaseNodeAvailable(zkw);

1214          String msg = "ZooKeeper available but no active master location found";

1217        }

1223            sn.getHostname(), sn.getPort(), hostnamesCanChange);
1224        connectionLock.putIfAbsent(key, key);
1225        Object stub = null;
1226        synchronized (connectionLock.get(key)) {

1231            isMasterRunning();
1232            stubs.put(key, stub);
1233          }
1234        }
1235        return stub;
1236      } finally {
1237        zkw.close();
1238      }

1345        // Just fail on first actual call rather than in here on setup.
1346        this.stubs.put(key, stub);

1352  static String getStubKey(final String serviceName,
1353                           final String rsHostname,
1354                           int port,
1355                           boolean resolveHostnames) {
1356    // Sometimes, servers go down and they come back up with the same hostname but a different
1357    // IP address. Force a resolution of the rsHostname by trying to instantiate an
1358    // InetSocketAddress, and this way we will rightfully get a new stubKey.
1359    // Also, include the hostname in the key so as to take care of those cases where the
1360    // DNS name is different but IP address remains the same.
1361    String address = rsHostname;
1362    if (resolveHostnames) {
1363      InetAddress i = new InetSocketAddress(rsHostname, port).getAddress();
1364      if (i != null) {
1365        address = i.getHostAddress() + "-" + rsHostname;
1366      }
1367    }
1368    return serviceName + "@" + address + ":" + port;
1369  }

1891  /**
1892   * Update the location with the new value (if the exception is a RegionMovedException)
1893   * or delete it from the cache. Does nothing if we can be sure from the exception that
1894   * the location is still accurate, or if the cache has already been updated.
1895   * @param exception an object (to simplify user code) on which we will try to find a nested
1896   *                  or wrapped or both RegionMovedException
1897   * @param source server that is the source of the location update.
1898   */
1899  @Override
1900  public void updateCachedLocations(final TableName tableName, byte[] regionName, byte[] rowkey,
1901    final Object exception, final ServerName source) {
1902    if (rowkey == null || tableName == null) {
1903      LOG.warn("Coding error, see method javadoc. row=" + (rowkey == null ? "null" : rowkey) +
1904          ", tableName=" + (tableName == null ? "null" : tableName));
1905      return;
1906    }
1907
1908    if (source == null) {
1909      // This should not happen, but let's secure ourselves.
1910      return;
1911    }
1912
1913    if (regionName == null) {
1914      // we do not know which region, so just remove the cache entry for the row and server
1915      metaCache.clearCache(tableName, rowkey, source);
1916      return;
1917    }
1918
1919    // Is it something we have already updated?
1920    final RegionLocations oldLocations = getCachedLocation(tableName, rowkey);
1921    HRegionLocation oldLocation = null;
1922    if (oldLocations != null) {
1923      oldLocation = oldLocations.getRegionLocationByRegionName(regionName);
1924    }
1925    if (oldLocation == null || !source.equals(oldLocation.getServerName())) {
1926      // There is no such location in the cache (it's been removed already) or
1927      // the cache has already been refreshed with a different location.  => nothing to do
1928      return;
1929    }
1930
1931    HRegionInfo regionInfo = oldLocation.getRegionInfo();
1932    Throwable cause = findException(exception);
1933    if (cause != null) {
1934      if (cause instanceof RegionTooBusyException || cause instanceof RegionOpeningException
1935          || cause instanceof ThrottlingException || cause instanceof MultiActionResultTooLarge) {
1936        // We know that the region is still on this region server
1937        return;
1938      }
1939
1940      if (cause instanceof RegionMovedException) {
1941        RegionMovedException rme = (RegionMovedException) cause;
1942        if (LOG.isTraceEnabled()) {
1943          LOG.trace("Region " + regionInfo.getRegionNameAsString() + " moved to " +
1944              rme.getHostname() + ":" + rme.getPort() +
1945              " according to " + source.getHostAndPort());
1946        }
1947        // We know that the region is not anymore on this region server, but we know
1948        //  the new location.
1949        updateCachedLocation(
1950            regionInfo, source, rme.getServerName(), rme.getLocationSeqNum());
1951        return;
1952      }
1953    }
1954
1955    // If we're here, it means that can cannot be sure about the location, so we remove it from
1956    // the cache. Do not send the source because source can be a new server in the same host:port
1957    metaCache.clearCache(regionInfo);
1958  }
