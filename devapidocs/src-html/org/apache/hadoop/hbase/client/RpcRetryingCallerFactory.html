018package org.apache.hadoop.hbase.client;
019
020import org.apache.hadoop.hbase.classification.InterfaceAudience;
021import org.apache.hadoop.conf.Configuration;
022import org.apache.hadoop.hbase.HConstants;
023import org.apache.hadoop.hbase.util.ReflectionUtils;
024
025/**
026 * Factory to create an {@link RpcRetryingCaller}
027 */
028@InterfaceAudience.Private
029public class RpcRetryingCallerFactory {
030
031  /** Configuration key for a custom {@link RpcRetryingCaller} */
032  public static final String CUSTOM_CALLER_CONF_KEY = "hbase.rpc.callerfactory.class";
033  protected final Configuration conf;
034  private final long pause;
035  private final int retries;
036  private final RetryingCallerInterceptor interceptor;
037  private final int startLogErrorsCnt;
038  private final boolean enableBackPressure;
039  private ServerStatisticTracker stats;
040
041  public RpcRetryingCallerFactory(Configuration conf) {
042    this(conf, RetryingCallerInterceptorFactory.NO_OP_INTERCEPTOR);
043  }
044  
045  public RpcRetryingCallerFactory(Configuration conf, RetryingCallerInterceptor interceptor) {
046    this.conf = conf;
047    pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
048        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
049    retries = conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
050        HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
051    startLogErrorsCnt = conf.getInt(AsyncProcess.START_LOG_ERRORS_AFTER_COUNT_KEY,
052        AsyncProcess.DEFAULT_START_LOG_ERRORS_AFTER_COUNT);
053    this.interceptor = interceptor;
054    enableBackPressure = conf.getBoolean(HConstants.ENABLE_CLIENT_BACKPRESSURE,
055        HConstants.DEFAULT_ENABLE_CLIENT_BACKPRESSURE);
056  }
057
058  /**
059   * Set the tracker that should be used for tracking statistics about the server
060   */
061  public void setStatisticTracker(ServerStatisticTracker statisticTracker) {
062    this.stats = statisticTracker;
063  }
064
065  public <T> RpcRetryingCaller<T> newCaller() {
066    // We store the values in the factory instance. This way, constructing new objects
067    //  is cheap as it does not require parsing a complex structure.
068    RpcRetryingCaller<T> caller = new RpcRetryingCallerImpl<T>(pause, retries, interceptor,
069        startLogErrorsCnt);
070
071    // wrap it with stats, if we are tracking them
072    if (enableBackPressure && this.stats != null) {
073      caller = new StatsTrackingRpcRetryingCaller<T>(caller, this.stats);
074    }
075
076    return caller;
077  }
078
079  public static RpcRetryingCallerFactory instantiate(Configuration configuration) {
080    return instantiate(configuration, RetryingCallerInterceptorFactory.NO_OP_INTERCEPTOR, null);
081  }
082
083  public static RpcRetryingCallerFactory instantiate(Configuration configuration,
084      ServerStatisticTracker stats) {
085    return instantiate(configuration, RetryingCallerInterceptorFactory.NO_OP_INTERCEPTOR, stats);
086  }
087
088  public static RpcRetryingCallerFactory instantiate(Configuration configuration,
089      RetryingCallerInterceptor interceptor, ServerStatisticTracker stats) {
090    String clazzName = RpcRetryingCallerFactory.class.getName();
091    String rpcCallerFactoryClazz =
092        configuration.get(RpcRetryingCallerFactory.CUSTOM_CALLER_CONF_KEY, clazzName);
093    RpcRetryingCallerFactory factory;
094    if (rpcCallerFactoryClazz.equals(clazzName)) {
095      factory = new RpcRetryingCallerFactory(configuration, interceptor);
096    } else {
097      factory = ReflectionUtils.instantiateWithCustomCtor(
098          rpcCallerFactoryClazz, new Class[] { Configuration.class },
099          new Object[] { configuration });
100    }
101
102    // setting for backwards compat with existing caller factories, rather than in the ctor
103    factory.setStatisticTracker(stats);
104    return factory;
105  }
106}
