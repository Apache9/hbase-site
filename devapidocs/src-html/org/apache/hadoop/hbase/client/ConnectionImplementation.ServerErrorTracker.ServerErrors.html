184  ConnectionImplementation(Configuration conf,
185                           ExecutorService pool, User user) throws IOException {
186    this.conf = conf;
187    this.user = user;
188    this.batchPool = pool;
189    this.connectionConfig = new ConnectionConfiguration(conf);
190    this.closed = false;
191    this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
192        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
193    this.useMetaReplicas = conf.getBoolean(HConstants.USE_META_REPLICAS,
194      HConstants.DEFAULT_USE_META_REPLICAS);
195    // how many times to try, one more than max *retry* time
196    this.numTries = connectionConfig.getRetriesNumber() + 1;
197    this.rpcTimeout = conf.getInt(
198        HConstants.HBASE_RPC_TIMEOUT_KEY,
199        HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
200    if (conf.getBoolean(CLIENT_NONCES_ENABLED_KEY, true)) {
201      synchronized (nonceGeneratorCreateLock) {
202        if (nonceGenerator == null) {
203          nonceGenerator = new PerClientRandomNonceGenerator();
204        }
205      }
206    } else {
207      nonceGenerator = new NoNonceGenerator();
208    }
209
210    this.stats = ServerStatisticTracker.create(conf);
211    this.interceptor = (new RetryingCallerInterceptorFactory(conf)).build();
212    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
213    this.rpcCallerFactory = RpcRetryingCallerFactory.instantiate(conf, interceptor, this.stats);
214    this.backoffPolicy = ClientBackoffPolicyFactory.create(conf);
215    this.asyncProcess = createAsyncProcess(this.conf);
216    if (conf.getBoolean(CLIENT_SIDE_METRICS_ENABLED_KEY, false)) {
217      this.metrics = new MetricsConnection(this);
218    } else {
219      this.metrics = null;
220    }
221    this.metaCache = new MetaCache(this.metrics);
222
223    boolean shouldListen = conf.getBoolean(HConstants.STATUS_PUBLISHED,
224        HConstants.STATUS_PUBLISHED_DEFAULT);
225    this.hostnamesCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
226    Class<? extends ClusterStatusListener.Listener> listenerClass =
227        conf.getClass(ClusterStatusListener.STATUS_LISTENER_CLASS,
228            ClusterStatusListener.DEFAULT_STATUS_LISTENER_CLASS,
229            ClusterStatusListener.Listener.class);
230
231    try {
232      this.registry = setupRegistry();
233      retrieveClusterId();
234
235      this.rpcClient = RpcClientFactory.createClient(this.conf, this.clusterId, this.metrics);
236
237      // Do we publish the status?
238      if (shouldListen) {
239        if (listenerClass == null) {
240          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
241              ClusterStatusListener.STATUS_LISTENER_CLASS + " is not set - not listening status");
242        } else {
243          clusterStatusListener = new ClusterStatusListener(
244              new ClusterStatusListener.DeadServerHandler() {
245                @Override
246                public void newDead(ServerName sn) {
247                  clearCaches(sn);
248                  rpcClient.cancelConnections(sn);
249                }
250              }, conf, listenerClass);
251        }
252      }
253    } catch (Throwable e) {
254      // avoid leaks: registry, rpcClient, ...
255      LOG.debug("connection construction failed", e);
256      close();
257      throw e;
258    }
259  }

325  private ExecutorService getBatchPool() {
326    if (batchPool == null) {
327      synchronized (this) {
328        if (batchPool == null) {
329          int threads = conf.getInt("hbase.hconnection.threads.max", 256);
330          this.batchPool = getThreadPool(threads, threads, "-shared", null);
331          this.cleanupPool = true;
332        }
333      }
334    }
335    return this.batchPool;
336  }
337
338  private ExecutorService getThreadPool(int maxThreads, int coreThreads, String nameHint,
339      BlockingQueue<Runnable> passedWorkQueue) {
340    // shared HTable thread executor not yet initialized
341    if (maxThreads == 0) {
342      maxThreads = Runtime.getRuntime().availableProcessors() * 8;
343    }
344    if (coreThreads == 0) {
345      coreThreads = Runtime.getRuntime().availableProcessors() * 8;
346    }
347    long keepAliveTime = conf.getLong("hbase.hconnection.threads.keepalivetime", 60);
348    BlockingQueue<Runnable> workQueue = passedWorkQueue;
349    if (workQueue == null) {
350      workQueue =
351        new LinkedBlockingQueue<Runnable>(maxThreads *
352            conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
353                HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS));
354      coreThreads = maxThreads;
355    }
356    ThreadPoolExecutor tpe = new ThreadPoolExecutor(
357        coreThreads,
358        maxThreads,
359        keepAliveTime,
360        TimeUnit.SECONDS,
361        workQueue,
362        Threads.newDaemonThreadFactory(toString() + nameHint));
363    tpe.allowCoreThreadTimeOut(true);
364    return tpe;
365  }

367  private ExecutorService getMetaLookupPool() {
368    if (this.metaLookupPool == null) {
369      synchronized (this) {
370        if (this.metaLookupPool == null) {
371          //Some of the threads would be used for meta replicas
372          //To start with, threads.max.core threads can hit the meta (including replicas).
373          //After that, requests will get queued up in the passed queue, and only after
374          //the queue is full, a new thread will be started
375          int threads = conf.getInt("hbase.hconnection.meta.lookup.threads.max", 128);
376          this.metaLookupPool = getThreadPool(
377             threads,
378             threads,
379             "-metaLookup-shared-", new LinkedBlockingQueue<Runnable>());
380        }
381      }
382    }
383    return this.metaLookupPool;
384  }

394  private void shutdownPools() {
395    if (this.cleanupPool && this.batchPool != null && !this.batchPool.isShutdown()) {
396      shutdownBatchPool(this.batchPool);
397    }
398    if (this.metaLookupPool != null && !this.metaLookupPool.isShutdown()) {
399      shutdownBatchPool(this.metaLookupPool);
400    }
401  }

403  private void shutdownBatchPool(ExecutorService pool) {
404    pool.shutdown();
405    try {
406      if (!pool.awaitTermination(10, TimeUnit.SECONDS)) {
407        pool.shutdownNow();
408      }
409    } catch (InterruptedException e) {
410      pool.shutdownNow();
411    }
412  }

455  private void checkIfBaseNodeAvailable(ZooKeeperWatcher zkw)
456    throws MasterNotRunningException {
457    String errorMsg;
458    try {
459      if (ZKUtil.checkExists(zkw, zkw.baseZNode) == -1) {
460        errorMsg = "The node " + zkw.baseZNode+" is not in ZooKeeper. "
461          + "It should have been written by the master. "
462          + "Check the value configured in 'zookeeper.znode.parent'. "
463          + "There could be a mismatch with the one configured in the master.";
464        LOG.error(errorMsg);
465        throw new MasterNotRunningException(errorMsg);
466      }
467    } catch (KeeperException e) {
468      errorMsg = "Can't get connection to ZooKeeper: " + e.getMessage();
469      LOG.error(errorMsg);
470      throw new MasterNotRunningException(errorMsg, e);
471    }
472  }

509  @Override
510  public boolean isTableAvailable(final TableName tableName, @Nullable final byte[][] splitKeys)
511      throws IOException {
512    if (this.closed) {
513      throw new IOException(toString() + " closed");
514    }
515    try {
516      if (!isTableEnabled(tableName)) {
517        LOG.debug("Table " + tableName + " not enabled");
518        return false;
519      }
520      List<Pair<HRegionInfo, ServerName>> locations =
521        MetaTableAccessor.getTableRegionsAndLocations(this, tableName, true);
522
523      int notDeployed = 0;
524      int regionCount = 0;
525      for (Pair<HRegionInfo, ServerName> pair : locations) {
526        HRegionInfo info = pair.getFirst();
527        if (pair.getSecond() == null) {
528          if (LOG.isDebugEnabled()) {
529            LOG.debug("Table " + tableName + " has not deployed region " + pair.getFirst()
530                .getEncodedName());
531          }
532          notDeployed++;
533        } else if (splitKeys != null
534            && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
535          for (byte[] splitKey : splitKeys) {
536            // Just check if the splitkey is available
537            if (Bytes.equals(info.getStartKey(), splitKey)) {
538              regionCount++;
539              break;
540            }
541          }
542        } else {
543          // Always empty start row should be counted
544          regionCount++;
545        }
546      }
547      if (notDeployed > 0) {
548        if (LOG.isDebugEnabled()) {
549          LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
550        }
551        return false;
552      } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
553        if (LOG.isDebugEnabled()) {
554          LOG.debug("Table " + tableName + " expected to have " + (splitKeys.length + 1)
555              + " regions, but only " + regionCount + " available");
556        }
557        return false;
558      } else {
559        if (LOG.isDebugEnabled()) {
560          LOG.debug("Table " + tableName + " should be available");
561        }
562        return true;
563      }
564    } catch (TableNotFoundException tnfe) {
565      LOG.warn("Table " + tableName + " not enabled, it is not exists");
566      return false;
567    }
568  }

647  @Override
648  public RegionLocations locateRegion(final TableName tableName,
649    final byte [] row, boolean useCache, boolean retry, int replicaId)
650  throws IOException {
651    if (this.closed) {
652      throw new IOException(toString() + " closed");
653    }
654    if (tableName== null || tableName.getName().length == 0) {
655      throw new IllegalArgumentException(
656          "table name cannot be null or zero length");
657    }
658    if (tableName.equals(TableName.META_TABLE_NAME)) {
659      return locateMeta(tableName, useCache, replicaId);
660    } else {
661      // Region not in the cache - have to go to the meta RS
662      return locateRegionInMeta(tableName, row, useCache, retry, replicaId);
663    }
664  }

666  private RegionLocations locateMeta(final TableName tableName,
667      boolean useCache, int replicaId) throws IOException {
668    // HBASE-10785: We cache the location of the META itself, so that we are not overloading
669    // zookeeper with one request for every region lookup. We cache the META with empty row
670    // key in MetaCache.
671    byte[] metaCacheKey = HConstants.EMPTY_START_ROW; // use byte[0] as the row for meta
672    RegionLocations locations = null;
673    if (useCache) {
674      locations = getCachedLocation(tableName, metaCacheKey);
675      if (locations != null && locations.getRegionLocation(replicaId) != null) {
676        return locations;
677      }
678    }
679
680    // only one thread should do the lookup.
681    synchronized (metaRegionLock) {
682      // Check the cache again for a hit in case some other thread made the
683      // same query while we were waiting on the lock.
684      if (useCache) {
685        locations = getCachedLocation(tableName, metaCacheKey);
686        if (locations != null && locations.getRegionLocation(replicaId) != null) {
687          return locations;
688        }
689      }
690
691      // Look up from zookeeper
692      locations = this.registry.getMetaRegionLocation();
693      if (locations != null) {
694        cacheLocation(tableName, locations);
695      }
696    }
697    return locations;
698  }

700  /*
701    * Search the hbase:meta table for the HRegionLocation
702    * info that contains the table and row we're seeking.
703    */
704  private RegionLocations locateRegionInMeta(TableName tableName, byte[] row,
705                 boolean useCache, boolean retry, int replicaId) throws IOException {
706
707    // If we are supposed to be using the cache, look in the cache to see if
708    // we already have the region.
709    if (useCache) {
710      RegionLocations locations = getCachedLocation(tableName, row);
711      if (locations != null && locations.getRegionLocation(replicaId) != null) {
712        return locations;
713      }
714    }
715
716    // build the key of the meta region we should be looking for.
717    // the extra 9's on the end are necessary to allow "exact" matches
718    // without knowing the precise region names.
719    byte[] metaKey = HRegionInfo.createRegionName(tableName, row, HConstants.NINES, false);
720
721    Scan s = new Scan();
722    s.setReversed(true);
723    s.setStartRow(metaKey);
724    s.setSmall(true);
725    s.setCaching(1);
726    if (this.useMetaReplicas) {
727      s.setConsistency(Consistency.TIMELINE);
728    }
729
730    int maxAttempts = (retry ? numTries : 1);
731
732    for (int tries = 0; true; tries++) {
733      if (tries >= maxAttempts) {
734        throw new NoServerForRegionException("Unable to find region for "
735            + Bytes.toStringBinary(row) + " in " + tableName +
736            " after " + tries + " tries.");
737      }
738      if (useCache) {
739        RegionLocations locations = getCachedLocation(tableName, row);
740        if (locations != null && locations.getRegionLocation(replicaId) != null) {
741          return locations;
742        }
743      } else {
744        // If we are not supposed to be using the cache, delete any existing cached location
745        // so it won't interfere.
746        metaCache.clearCache(tableName, row);
747      }
748
749      // Query the meta region
750      try {
751        Result regionInfoRow = null;
752        ReversedClientScanner rcs = null;
753        try {
754          rcs = new ClientSmallReversedScanner(conf, s, TableName.META_TABLE_NAME, this,
755            rpcCallerFactory, rpcControllerFactory, getMetaLookupPool(), 0);
756          regionInfoRow = rcs.next();
757        } finally {
758          if (rcs != null) {
759            rcs.close();
760          }
761        }
762
763        if (regionInfoRow == null) {
764          throw new TableNotFoundException(tableName);
765        }
766
767        // convert the row result into the HRegionLocation we need!
768        RegionLocations locations = MetaTableAccessor.getRegionLocations(regionInfoRow);
769        if (locations == null || locations.getRegionLocation(replicaId) == null) {
770          throw new IOException("HRegionInfo was null in " +
771            tableName + ", row=" + regionInfoRow);
772        }
773        HRegionInfo regionInfo = locations.getRegionLocation(replicaId).getRegionInfo();
774        if (regionInfo == null) {
775          throw new IOException("HRegionInfo was null or empty in " +
776            TableName.META_TABLE_NAME + ", row=" + regionInfoRow);
777        }
778
779        // possible we got a region of a different table...
780        if (!regionInfo.getTable().equals(tableName)) {
781          throw new TableNotFoundException(
782                "Table '" + tableName + "' was not found, got: " +
783                regionInfo.getTable() + ".");
784        }
785        if (regionInfo.isSplit()) {
786          throw new RegionOfflineException("the only available region for" +
787            " the required row is a split parent," +
788            " the daughters should be online soon: " +
789            regionInfo.getRegionNameAsString());
790        }
791        if (regionInfo.isOffline()) {
792          throw new RegionOfflineException("the region is offline, could" +
793            " be caused by a disable table call: " +
794            regionInfo.getRegionNameAsString());
795        }
796
797        ServerName serverName = locations.getRegionLocation(replicaId).getServerName();
798        if (serverName == null) {
799          throw new NoServerForRegionException("No server address listed " +
800            "in " + TableName.META_TABLE_NAME + " for region " +
801            regionInfo.getRegionNameAsString() + " containing row " +
802            Bytes.toStringBinary(row));
803        }
804
805        if (isDeadServer(serverName)){
806          throw new RegionServerStoppedException("hbase:meta says the region "+
807              regionInfo.getRegionNameAsString()+" is managed by the server " + serverName +
808              ", but it is dead.");
809        }
810        // Instantiate the location
811        cacheLocation(tableName, locations);
812        return locations;
813      } catch (TableNotFoundException e) {
814        // if we got this error, probably means the table just plain doesn't
815        // exist. rethrow the error immediately. this should always be coming
816        // from the HTable constructor.
817        throw e;
818      } catch (IOException e) {
819        ExceptionUtil.rethrowIfInterrupt(e);
820
821        if (e instanceof RemoteException) {
822          e = ((RemoteException)e).unwrapRemoteException();
823        }
824        if (tries < maxAttempts - 1) {
825          if (LOG.isDebugEnabled()) {
826            LOG.debug("locateRegionInMeta parentTable=" +
827                TableName.META_TABLE_NAME + ", metaLocation=" +
828              ", attempt=" + tries + " of " +
829              maxAttempts + " failed; retrying after sleep of " +
830              ConnectionUtils.getPauseTime(this.pause, tries) + " because: " + e.getMessage());
831          }
832        } else {
833          throw e;
834        }
835        // Only relocate the parent region if necessary
836        if(!(e instanceof RegionOfflineException ||
837            e instanceof NoServerForRegionException)) {
838          relocateRegion(TableName.META_TABLE_NAME, metaKey, replicaId);
839        }
840      }
841      try{
842        Thread.sleep(ConnectionUtils.getPauseTime(this.pause, tries));
843      } catch (InterruptedException e) {
844        throw new InterruptedIOException("Giving up trying to location region in " +
845          "meta: thread is interrupted.");
846      }
847    }
848  }

958  static class ServerErrorTracker {
959    // We need a concurrent map here, as we could have multiple threads updating it in parallel.
960    private final ConcurrentMap<ServerName, ServerErrors> errorsByServer =
961        new ConcurrentHashMap<ServerName, ServerErrors>();
962    private final long canRetryUntil;
963    private final int maxTries;// max number to try
964    private final long startTrackingTime;
965
966    /**
967     * Constructor
968     * @param timeout how long to wait before timeout, in unit of millisecond
969     * @param maxTries how many times to try
970     */
971    public ServerErrorTracker(long timeout, int maxTries) {
972      this.maxTries = maxTries;
973      this.canRetryUntil = EnvironmentEdgeManager.currentTime() + timeout;
974      this.startTrackingTime = new Date().getTime();
975    }
976
977    /**
978     * We stop to retry when we have exhausted BOTH the number of tries and the time allocated.
979     * @param numAttempt how many times we have tried by now
980     */
981    boolean canTryMore(int numAttempt) {
982      // If there is a single try we must not take into account the time.
983      return numAttempt < maxTries || (maxTries > 1 &&
984          EnvironmentEdgeManager.currentTime() < this.canRetryUntil);
985    }

987    /**
988     * Calculates the back-off time for a retrying request to a particular server.
989     *
990     * @param server    The server in question.
991     * @param basePause The default hci pause.
992     * @return The time to wait before sending next request.
993     */
994    long calculateBackoffTime(ServerName server, long basePause) {
995      long result;
996      ServerErrors errorStats = errorsByServer.get(server);
997      if (errorStats != null) {
998        result = ConnectionUtils.getPauseTime(basePause, errorStats.getCount());
999      } else {
1000        result = 0; // yes, if the server is not in our list we don't wait before retrying.
1001      }
1002      return result;
1003    }

1005    /**

1007     *
1008     * @param server The server in question.
1009     */
1010    void reportServerError(ServerName server) {
1011      ServerErrors errors = errorsByServer.get(server);
1012      if (errors != null) {
1013        errors.addError();
1014      } else {
1015        errors = errorsByServer.putIfAbsent(server, new ServerErrors());
1016        if (errors != null){
1017          errors.addError();
1018        }
1019      }
1020    }

1023      return startTrackingTime;
1024    }

1043   * Makes a client-side stub for master services. Sub-class to specialize.
1044   * Depends on hosting class so not static.  Exists so we avoid duplicating a bunch of code
1045   * when setting up the MasterMonitorService and MasterAdminService.
1046   */
1047  abstract class StubMaker {
1048    /**
1049     * Returns the name of the service stub being created.
1050     */
1051    protected abstract String getServiceName();
1052
1053    /**
1054     * Make stub and cache it internal so can be used later doing the isMasterRunning call.
1055     */
1056    protected abstract Object makeStub(final BlockingRpcChannel channel);
1057
1058    /**
1059     * Once setup, check it works by doing isMasterRunning check.
1060     */

1063    /**
1064     * Create a stub. Try once only.  It is not typed because there is no common type to
1065     * protobuf services nor their interfaces.  Let the caller do appropriate casting.
1066     * @return A stub for master services.
1067     */
1068    private Object makeStubNoRetries() throws IOException, KeeperException, ServiceException {
1069      ZooKeeperKeepAliveConnection zkw;
1070      try {
1071        zkw = getKeepAliveZooKeeperWatcher();
1072      } catch (IOException e) {
1073        ExceptionUtil.rethrowIfInterrupt(e);
1074        throw new ZooKeeperConnectionException("Can't connect to ZooKeeper", e);
1075      }
1076      try {
1077        checkIfBaseNodeAvailable(zkw);
1078        ServerName sn = MasterAddressTracker.getMasterAddress(zkw);
1079        if (sn == null) {
1080          String msg = "ZooKeeper available but no active master location found";
1081          LOG.info(msg);
1082          throw new MasterNotRunningException(msg);
1083        }
1084        if (isDeadServer(sn)) {
1085          throw new MasterNotRunningException(sn + " is dead.");
1086        }
1087        // Use the security info interface name as our stub key
1088        String key = getStubKey(getServiceName(),
1089            sn.getHostname(), sn.getPort(), hostnamesCanChange);
1090        connectionLock.putIfAbsent(key, key);
1091        Object stub = null;
1092        synchronized (connectionLock.get(key)) {
1093          stub = stubs.get(key);
1094          if (stub == null) {
1095            BlockingRpcChannel channel = rpcClient.createBlockingRpcChannel(sn, user, rpcTimeout);
1096            stub = makeStub(channel);
1097            isMasterRunning();
1098            stubs.put(key, stub);
1099          }
1100        }
1101        return stub;
1102      } finally {
1103        zkw.close();
1104      }
1105    }

1108     * Create a stub against the master.  Retry if necessary.
1109     * @return A stub to do <code>intf</code> against the master
1110     * @throws org.apache.hadoop.hbase.MasterNotRunningException if master is not running
1111     */
1112    Object makeStub() throws IOException {
1113      // The lock must be at the beginning to prevent multiple master creations
1114      //  (and leaks) in a multithread context
1115      synchronized (masterAndZKLock) {
1116        Exception exceptionCaught = null;
1117        if (!closed) {
1118          try {
1119            return makeStubNoRetries();
1120          } catch (IOException e) {
1121            exceptionCaught = e;
1122          } catch (KeeperException e) {
1123            exceptionCaught = e;
1124          } catch (ServiceException e) {
1125            exceptionCaught = e;
1126          }
1127
1128          throw new MasterNotRunningException(exceptionCaught);
1129        } else {
1130          throw new DoNotRetryIOException("Connection was closed while trying to get master");
1131        }
1132      }
1133    }

1211                           final String rsHostname,
1212                           int port,
1213                           boolean resolveHostnames) {
1214    // Sometimes, servers go down and they come back up with the same hostname but a different
1215    // IP address. Force a resolution of the rsHostname by trying to instantiate an
1216    // InetSocketAddress, and this way we will rightfully get a new stubKey.
1217    // Also, include the hostname in the key so as to take care of those cases where the
1218    // DNS name is different but IP address remains the same.
1219    String address = rsHostname;
1220    if (resolveHostnames) {
1221      InetAddress i = new InetSocketAddress(rsHostname, port).getAddress();
1222      if (i != null) {
1223        address = i.getHostAddress() + "-" + rsHostname;
1224      }
1225    }
1226    return serviceName + "@" + address + ":" + port;

1693  private boolean isKeepAliveMasterConnectedAndRunning(MasterServiceState mss) {
1694    if (mss.getStub() == null){
1695      return false;
1696    }
1697    try {
1698      return mss.isMasterRunning();
1699    } catch (UndeclaredThrowableException e) {
1700      // It's somehow messy, but we can receive exceptions such as
1701      //  java.net.ConnectException but they're not declared. So we catch it...
1702      LOG.info("Master connection is not running anymore", e.getUndeclaredThrowable());
1703      return false;
1704    } catch (ServiceException se) {
1705      LOG.warn("Checking master connection", se);
1706      return false;
1707    }
1708  }

1748  /**
1749   * Update the location with the new value (if the exception is a RegionMovedException)
1750   * or delete it from the cache. Does nothing if we can be sure from the exception that
1751   * the location is still accurate, or if the cache has already been updated.
1752   * @param exception an object (to simplify user code) on which we will try to find a nested
1753   *   or wrapped or both RegionMovedException
1754   * @param source server that is the source of the location update.
1755   */
1756  @Override
1757  public void updateCachedLocations(final TableName tableName, byte[] regionName, byte[] rowkey,
1758    final Object exception, final ServerName source) {
1759    if (rowkey == null || tableName == null) {
1760      LOG.warn("Coding error, see method javadoc. row=" + (rowkey == null ? "null" : rowkey) +
1761          ", tableName=" + (tableName == null ? "null" : tableName));
1762      return;
1763    }
1764
1765    if (source == null) {
1766      // This should not happen, but let's secure ourselves.
1767      return;
1768    }
1769
1770    if (regionName == null) {
1771      // we do not know which region, so just remove the cache entry for the row and server
1772      if (metrics != null) {
1773        metrics.incrCacheDroppingExceptions(exception);
1774      }
1775      metaCache.clearCache(tableName, rowkey, source);
1776      return;
1777    }
1778
1779    // Is it something we have already updated?
1780    final RegionLocations oldLocations = getCachedLocation(tableName, rowkey);
1781    HRegionLocation oldLocation = null;
1782    if (oldLocations != null) {
1783      oldLocation = oldLocations.getRegionLocationByRegionName(regionName);
1784    }
1785    if (oldLocation == null || !source.equals(oldLocation.getServerName())) {
1786      // There is no such location in the cache (it's been removed already) or
1787      // the cache has already been refreshed with a different location.  => nothing to do
1788      return;
1789    }
1790
1791    HRegionInfo regionInfo = oldLocation.getRegionInfo();
1792    Throwable cause = ClientExceptionsUtil.findException(exception);
1793    if (cause != null) {
1794      if (!ClientExceptionsUtil.isMetaClearingException(cause)) {
1795        // We know that the region is still on this region server
1796        return;
1797      }
1798
1799      if (cause instanceof RegionMovedException) {
1800        RegionMovedException rme = (RegionMovedException) cause;
1801        if (LOG.isTraceEnabled()) {
1802          LOG.trace("Region " + regionInfo.getRegionNameAsString() + " moved to " +
1803              rme.getHostname() + ":" + rme.getPort() +
1804              " according to " + source.getHostAndPort());
1805        }
1806        // We know that the region is not anymore on this region server, but we know
1807        //  the new location.
1808        updateCachedLocation(
1809            regionInfo, source, rme.getServerName(), rme.getLocationSeqNum());
1810        return;
1811      }
1812    }
1813
1814    if (metrics != null) {
1815      metrics.incrCacheDroppingExceptions(exception);
1816    }
1817
1818    // If we're here, it means that can cannot be sure about the location, so we remove it from
1819    // the cache. Do not send the source because source can be a new server in the same host:port
1820    metaCache.clearCache(regionInfo);
1821  }

1854  @Override
1855  public void abort(final String msg, Throwable t) {
1856    if (t instanceof KeeperException.SessionExpiredException
1857      && keepAliveZookeeper != null) {
1858      synchronized (masterAndZKLock) {
1859        if (keepAliveZookeeper != null) {
1860          LOG.warn("This client just lost it's session with ZooKeeper," +
1861            " closing it." +
1862            " It will be recreated next time someone needs it", t);
1863          closeZooKeeperWatcher();
1864        }
1865      }
1866    } else {
1867      if (t != null) {
1868        LOG.fatal(msg, t);
1869      } else {
1870        LOG.fatal(msg);
1871      }
1872      this.aborted = true;
1873      close();
1874      this.closed = true;
1875    }
1876  }

1909    if (rpcClient != null) {
1910      rpcClient.close();
1911    }
