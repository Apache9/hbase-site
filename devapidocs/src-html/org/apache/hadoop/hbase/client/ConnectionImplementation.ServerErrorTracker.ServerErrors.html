189  ConnectionImplementation(Configuration conf,
190                           ExecutorService pool, User user) throws IOException {
191    this.conf = conf;
192    this.user = user;
193    this.batchPool = pool;
194    this.connectionConfig = new ConnectionConfiguration(conf);
195    this.closed = false;
196    this.pause = conf.getLong(HConstants.HBASE_CLIENT_PAUSE,
197        HConstants.DEFAULT_HBASE_CLIENT_PAUSE);
198    this.useMetaReplicas = conf.getBoolean(HConstants.USE_META_REPLICAS,
199      HConstants.DEFAULT_USE_META_REPLICAS);
200    // how many times to try, one more than max *retry* time
201    this.numTries = connectionConfig.getRetriesNumber() + 1;
202    this.rpcTimeout = conf.getInt(
203        HConstants.HBASE_RPC_TIMEOUT_KEY,
204        HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
205    if (conf.getBoolean(CLIENT_NONCES_ENABLED_KEY, true)) {
206      synchronized (nonceGeneratorCreateLock) {
207        if (nonceGenerator == null) {
208          nonceGenerator = new PerClientRandomNonceGenerator();
209        }
210      }
211    } else {
212      nonceGenerator = new NoNonceGenerator();
213    }
214
215    this.stats = ServerStatisticTracker.create(conf);
216    this.interceptor = (new RetryingCallerInterceptorFactory(conf)).build();
217    this.rpcControllerFactory = RpcControllerFactory.instantiate(conf);
218    this.rpcCallerFactory = RpcRetryingCallerFactory.instantiate(conf, interceptor, this.stats);
219    this.backoffPolicy = ClientBackoffPolicyFactory.create(conf);
220    this.asyncProcess = createAsyncProcess(this.conf);
221    if (conf.getBoolean(CLIENT_SIDE_METRICS_ENABLED_KEY, false)) {
222      this.metrics = new MetricsConnection(this);
223    } else {
224      this.metrics = null;
225    }
226    this.metaCache = new MetaCache(this.metrics);
227
228    boolean shouldListen = conf.getBoolean(HConstants.STATUS_PUBLISHED,
229        HConstants.STATUS_PUBLISHED_DEFAULT);
230    this.hostnamesCanChange = conf.getBoolean(RESOLVE_HOSTNAME_ON_FAIL_KEY, true);
231    Class<? extends ClusterStatusListener.Listener> listenerClass =
232        conf.getClass(ClusterStatusListener.STATUS_LISTENER_CLASS,
233            ClusterStatusListener.DEFAULT_STATUS_LISTENER_CLASS,
234            ClusterStatusListener.Listener.class);
235
236    try {
237      this.registry = setupRegistry();
238      retrieveClusterId();
239
240      this.rpcClient = RpcClientFactory.createClient(this.conf, this.clusterId, this.metrics);
241
242      // Do we publish the status?
243      if (shouldListen) {
244        if (listenerClass == null) {
245          LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
246              ClusterStatusListener.STATUS_LISTENER_CLASS + " is not set - not listening status");
247        } else {
248          clusterStatusListener = new ClusterStatusListener(
249              new ClusterStatusListener.DeadServerHandler() {
250                @Override
251                public void newDead(ServerName sn) {
252                  clearCaches(sn);
253                  rpcClient.cancelConnections(sn);
254                }
255              }, conf, listenerClass);
256        }
257      }
258    } catch (Throwable e) {
259      // avoid leaks: registry, rpcClient, ...
260      LOG.debug("connection construction failed", e);
261      close();
262      throw e;
263    }
264  }

351  private ExecutorService getBatchPool() {
352    if (batchPool == null) {
353      synchronized (this) {
354        if (batchPool == null) {
355          this.batchPool = getThreadPool(conf.getInt("hbase.hconnection.threads.max", 256),
356              conf.getInt("hbase.hconnection.threads.core", 256), "-shared", null);
357          this.cleanupPool = true;
358        }
359      }
360    }
361    return this.batchPool;
362  }
363
364  private ExecutorService getThreadPool(int maxThreads, int coreThreads, String nameHint,
365      BlockingQueue<Runnable> passedWorkQueue) {
366    // shared HTable thread executor not yet initialized
367    if (maxThreads == 0) {
368      maxThreads = Runtime.getRuntime().availableProcessors() * 8;
369    }
370    if (coreThreads == 0) {
371      coreThreads = Runtime.getRuntime().availableProcessors() * 8;
372    }
373    long keepAliveTime = conf.getLong("hbase.hconnection.threads.keepalivetime", 60);
374    BlockingQueue<Runnable> workQueue = passedWorkQueue;
375    if (workQueue == null) {
376      workQueue =
377        new LinkedBlockingQueue<Runnable>(maxThreads *
378            conf.getInt(HConstants.HBASE_CLIENT_MAX_TOTAL_TASKS,
379                HConstants.DEFAULT_HBASE_CLIENT_MAX_TOTAL_TASKS));
380    }
381    ThreadPoolExecutor tpe = new ThreadPoolExecutor(
382        coreThreads,
383        maxThreads,
384        keepAliveTime,
385        TimeUnit.SECONDS,
386        workQueue,
387        Threads.newDaemonThreadFactory(toString() + nameHint));
388    tpe.allowCoreThreadTimeOut(true);
389    return tpe;
390  }

392  private ExecutorService getMetaLookupPool() {
393    if (this.metaLookupPool == null) {
394      synchronized (this) {
395        if (this.metaLookupPool == null) {
396          //Some of the threads would be used for meta replicas
397          //To start with, threads.max.core threads can hit the meta (including replicas).
398          //After that, requests will get queued up in the passed queue, and only after
399          //the queue is full, a new thread will be started
400          this.metaLookupPool = getThreadPool(
401             conf.getInt("hbase.hconnection.meta.lookup.threads.max", 128),
402             conf.getInt("hbase.hconnection.meta.lookup.threads.core", 10),
403             "-metaLookup-shared-", new LinkedBlockingQueue<Runnable>());
404        }
405      }
406    }
407    return this.metaLookupPool;
408  }

418  private void shutdownPools() {
419    if (this.cleanupPool && this.batchPool != null && !this.batchPool.isShutdown()) {
420      shutdownBatchPool(this.batchPool);
421    }
422    if (this.metaLookupPool != null && !this.metaLookupPool.isShutdown()) {
423      shutdownBatchPool(this.metaLookupPool);
424    }
425  }

427  private void shutdownBatchPool(ExecutorService pool) {
428    pool.shutdown();
429    try {
430      if (!pool.awaitTermination(10, TimeUnit.SECONDS)) {
431        pool.shutdownNow();
432      }
433    } catch (InterruptedException e) {
434      pool.shutdownNow();
435    }

477  private void checkIfBaseNodeAvailable(ZooKeeperWatcher zkw)
478    throws MasterNotRunningException {
479    String errorMsg;
480    try {
481      if (ZKUtil.checkExists(zkw, zkw.baseZNode) == -1) {
482        errorMsg = "The node " + zkw.baseZNode+" is not in ZooKeeper. "
483          + "It should have been written by the master. "
484          + "Check the value configured in 'zookeeper.znode.parent'. "
485          + "There could be a mismatch with the one configured in the master.";
486        LOG.error(errorMsg);
487        throw new MasterNotRunningException(errorMsg);
488      }
489    } catch (KeeperException e) {
490      errorMsg = "Can't get connection to ZooKeeper: " + e.getMessage();
491      LOG.error(errorMsg);
492      throw new MasterNotRunningException(errorMsg, e);
493    }
494  }

557  @Override
558  public boolean isTableAvailable(final TableName tableName, @Nullable final byte[][] splitKeys)
559      throws IOException {
560    if (this.closed) throw new IOException(toString() + " closed");
561    try {
562      if (!isTableEnabled(tableName)) {
563        LOG.debug("Table " + tableName + " not enabled");
564        return false;
565      }
566      List<Pair<HRegionInfo, ServerName>> locations =
567        MetaTableAccessor.getTableRegionsAndLocations(this, tableName, true);
568
569      int notDeployed = 0;
570      int regionCount = 0;
571      for (Pair<HRegionInfo, ServerName> pair : locations) {
572        HRegionInfo info = pair.getFirst();
573        if (pair.getSecond() == null) {
574          if (LOG.isDebugEnabled()) {
575            LOG.debug("Table " + tableName + " has not deployed region " + pair.getFirst()
576                .getEncodedName());
577          }
578          notDeployed++;
579        } else if (splitKeys != null
580            && !Bytes.equals(info.getStartKey(), HConstants.EMPTY_BYTE_ARRAY)) {
581          for (byte[] splitKey : splitKeys) {
582            // Just check if the splitkey is available
583            if (Bytes.equals(info.getStartKey(), splitKey)) {
584              regionCount++;
585              break;
586            }
587          }
588        } else {
589          // Always empty start row should be counted
590          regionCount++;
591        }
592      }
593      if (notDeployed > 0) {
594        if (LOG.isDebugEnabled()) {
595          LOG.debug("Table " + tableName + " has " + notDeployed + " regions");
596        }
597        return false;
598      } else if (splitKeys != null && regionCount != splitKeys.length + 1) {
599        if (LOG.isDebugEnabled()) {
600          LOG.debug("Table " + tableName + " expected to have " + (splitKeys.length + 1)
601              + " regions, but only " + regionCount + " available");
602        }
603        return false;
604      } else {
605        if (LOG.isDebugEnabled()) {
606          LOG.debug("Table " + tableName + " should be available");
607        }
608        return true;
609      }
610    } catch (TableNotFoundException tnfe) {
611      LOG.warn("Table " + tableName + " not enabled, it is not exists");
612      return false;
613    }
614  }

650  @Override
651  public List<HRegionLocation> locateRegions(final TableName tableName,
652      final boolean useCache, final boolean offlined) throws IOException {
653    List<HRegionInfo> regions = MetaTableAccessor
654        .getTableRegions(this, tableName, !offlined);
655    final List<HRegionLocation> locations = new ArrayList<HRegionLocation>();
656    for (HRegionInfo regionInfo : regions) {
657      RegionLocations list = locateRegion(tableName, regionInfo.getStartKey(), useCache, true);
658      if (list != null) {
659        for (HRegionLocation loc : list.getRegionLocations()) {
660          if (loc != null) {
661            locations.add(loc);

741  private RegionLocations locateMeta(final TableName tableName,
742      boolean useCache, int replicaId) throws IOException {
743    // HBASE-10785: We cache the location of the META itself, so that we are not overloading
744    // zookeeper with one request for every region lookup. We cache the META with empty row
745    // key in MetaCache.
746    byte[] metaCacheKey = HConstants.EMPTY_START_ROW; // use byte[0] as the row for meta
747    RegionLocations locations = null;
748    if (useCache) {
749      locations = getCachedLocation(tableName, metaCacheKey);
750      if (locations != null && locations.getRegionLocation(replicaId) != null) {
751        return locations;
752      }
753    }
754
755    // only one thread should do the lookup.
756    synchronized (metaRegionLock) {
757      // Check the cache again for a hit in case some other thread made the
758      // same query while we were waiting on the lock.
759      if (useCache) {
760        locations = getCachedLocation(tableName, metaCacheKey);
761        if (locations != null && locations.getRegionLocation(replicaId) != null) {
762          return locations;
763        }
764      }
765
766      // Look up from zookeeper
767      locations = this.registry.getMetaRegionLocation();
768      if (locations != null) {
769        cacheLocation(tableName, locations);
770      }
771    }
772    return locations;
773  }

775  /*
776    * Search the hbase:meta table for the HRegionLocation
777    * info that contains the table and row we're seeking.
778    */
779  private RegionLocations locateRegionInMeta(TableName tableName, byte[] row,
780                 boolean useCache, boolean retry, int replicaId) throws IOException {
781
782    // If we are supposed to be using the cache, look in the cache to see if
783    // we already have the region.
784    if (useCache) {
785      RegionLocations locations = getCachedLocation(tableName, row);
786      if (locations != null && locations.getRegionLocation(replicaId) != null) {
787        return locations;
788      }
789    }
790
791    // build the key of the meta region we should be looking for.
792    // the extra 9's on the end are necessary to allow "exact" matches
793    // without knowing the precise region names.
794    byte[] metaKey = HRegionInfo.createRegionName(tableName, row, HConstants.NINES, false);
795
796    Scan s = new Scan();
797    s.setReversed(true);
798    s.setStartRow(metaKey);
799    s.setSmall(true);
800    s.setCaching(1);
801    if (this.useMetaReplicas) {
802      s.setConsistency(Consistency.TIMELINE);
803    }
804
805    int maxAttempts = (retry ? numTries : 1);
806
807    for (int tries = 0; true; tries++) {
808      if (tries >= maxAttempts) {
809        throw new NoServerForRegionException("Unable to find region for "
810            + Bytes.toStringBinary(row) + " in " + tableName +
811            " after " + tries + " tries.");
812      }
813      if (useCache) {
814        RegionLocations locations = getCachedLocation(tableName, row);
815        if (locations != null && locations.getRegionLocation(replicaId) != null) {
816          return locations;
817        }
818      } else {
819        // If we are not supposed to be using the cache, delete any existing cached location
820        // so it won't interfere.
821        metaCache.clearCache(tableName, row);
822      }
823
824      // Query the meta region
825      try {
826        Result regionInfoRow = null;
827        ReversedClientScanner rcs = null;
828        try {
829          rcs = new ClientSmallReversedScanner(conf, s, TableName.META_TABLE_NAME, this,
830            rpcCallerFactory, rpcControllerFactory, getMetaLookupPool(), 0);
831          regionInfoRow = rcs.next();
832        } finally {
833          if (rcs != null) {
834            rcs.close();
835          }
836        }
837
838        if (regionInfoRow == null) {
839          throw new TableNotFoundException(tableName);
840        }
841
842        // convert the row result into the HRegionLocation we need!
843        RegionLocations locations = MetaTableAccessor.getRegionLocations(regionInfoRow);
844        if (locations == null || locations.getRegionLocation(replicaId) == null) {
845          throw new IOException("HRegionInfo was null in " +
846            tableName + ", row=" + regionInfoRow);
847        }
848        HRegionInfo regionInfo = locations.getRegionLocation(replicaId).getRegionInfo();
849        if (regionInfo == null) {
850          throw new IOException("HRegionInfo was null or empty in " +
851            TableName.META_TABLE_NAME + ", row=" + regionInfoRow);
852        }
853
854        // possible we got a region of a different table...
855        if (!regionInfo.getTable().equals(tableName)) {
856          throw new TableNotFoundException(
857                "Table '" + tableName + "' was not found, got: " +
858                regionInfo.getTable() + ".");
859        }
860        if (regionInfo.isSplit()) {
861          throw new RegionOfflineException("the only available region for" +
862            " the required row is a split parent," +
863            " the daughters should be online soon: " +
864            regionInfo.getRegionNameAsString());
865        }
866        if (regionInfo.isOffline()) {
867          throw new RegionOfflineException("the region is offline, could" +
868            " be caused by a disable table call: " +
869            regionInfo.getRegionNameAsString());
870        }
871
872        ServerName serverName = locations.getRegionLocation(replicaId).getServerName();
873        if (serverName == null) {
874          throw new NoServerForRegionException("No server address listed " +
875            "in " + TableName.META_TABLE_NAME + " for region " +
876            regionInfo.getRegionNameAsString() + " containing row " +
877            Bytes.toStringBinary(row));
878        }
879
880        if (isDeadServer(serverName)){
881          throw new RegionServerStoppedException("hbase:meta says the region "+
882              regionInfo.getRegionNameAsString()+" is managed by the server " + serverName +
883              ", but it is dead.");
884        }
885        // Instantiate the location
886        cacheLocation(tableName, locations);
887        return locations;
888      } catch (TableNotFoundException e) {
889        // if we got this error, probably means the table just plain doesn't
890        // exist. rethrow the error immediately. this should always be coming
891        // from the HTable constructor.
892        throw e;
893      } catch (IOException e) {
894        ExceptionUtil.rethrowIfInterrupt(e);
895
896        if (e instanceof RemoteException) {
897          e = ((RemoteException)e).unwrapRemoteException();
898        }
899        if (tries < maxAttempts - 1) {
900          if (LOG.isDebugEnabled()) {
901            LOG.debug("locateRegionInMeta parentTable=" +
902                TableName.META_TABLE_NAME + ", metaLocation=" +
903              ", attempt=" + tries + " of " +
904              maxAttempts + " failed; retrying after sleep of " +
905              ConnectionUtils.getPauseTime(this.pause, tries) + " because: " + e.getMessage());
906          }
907        } else {
908          throw e;
909        }
910        // Only relocate the parent region if necessary
911        if(!(e instanceof RegionOfflineException ||
912            e instanceof NoServerForRegionException)) {
913          relocateRegion(TableName.META_TABLE_NAME, metaKey, replicaId);
914        }
915      }
916      try{
917        Thread.sleep(ConnectionUtils.getPauseTime(this.pause, tries));
918      } catch (InterruptedException e) {
919        throw new InterruptedIOException("Giving up trying to location region in " +
920          "meta: thread is interrupted.");
921      }
922    }
923  }

1040    private final ConcurrentMap<ServerName, ServerErrors> errorsByServer =
1041        new ConcurrentHashMap<ServerName, ServerErrors>();
1042    private final long canRetryUntil;
1043    private final int maxTries;// max number to try
1044    private final long startTrackingTime;

1046    /**
1047     * Constructor
1048     * @param timeout how long to wait before timeout, in unit of millisecond
1049     * @param maxTries how many times to try
1050     */
1051    public ServerErrorTracker(long timeout, int maxTries) {
1052      this.maxTries = maxTries;
1053      this.canRetryUntil = EnvironmentEdgeManager.currentTime() + timeout;
1054      this.startTrackingTime = new Date().getTime();

1057    /**
1058     * We stop to retry when we have exhausted BOTH the number of tries and the time allocated.
1059     * @param numAttempt how many times we have tried by now
1060     */
1061    boolean canTryMore(int numAttempt) {
1062      // If there is a single try we must not take into account the time.
1063      return numAttempt < maxTries || (maxTries > 1 &&
1064          EnvironmentEdgeManager.currentTime() < this.canRetryUntil);
1065    }

1069     *

1071     * @param basePause The default hci pause.
1072     * @return The time to wait before sending next request.
1073     */
1074    long calculateBackoffTime(ServerName server, long basePause) {
1075      long result;
1076      ServerErrors errorStats = errorsByServer.get(server);
1077      if (errorStats != null) {
1078        result = ConnectionUtils.getPauseTime(basePause, errorStats.getCount());
1079      } else {
1080        result = 0; // yes, if the server is not in our list we don't wait before retrying.
1081      }
1082      return result;
1083    }

1096        if (errors != null){

1098        }
1099      }

1129     * Returns the name of the service stub being created.
1130     */

1132

1145     * protobuf services nor their interfaces.  Let the caller do appropriate casting.
1146     * @return A stub for master services.
1147     */
1148    private Object makeStubNoRetries() throws IOException, KeeperException, ServiceException {
1149      ZooKeeperKeepAliveConnection zkw;
1150      try {
1151        zkw = getKeepAliveZooKeeperWatcher();

1298  static String getStubKey(final String serviceName,
1299                           final String rsHostname,
1300                           int port,
1301                           boolean resolveHostnames) {
1302    // Sometimes, servers go down and they come back up with the same hostname but a different
1303    // IP address. Force a resolution of the rsHostname by trying to instantiate an
1304    // InetSocketAddress, and this way we will rightfully get a new stubKey.
1305    // Also, include the hostname in the key so as to take care of those cases where the
1306    // DNS name is different but IP address remains the same.
1307    String address = rsHostname;
1308    if (resolveHostnames) {
1309      InetAddress i = new InetSocketAddress(rsHostname, port).getAddress();
1310      if (i != null) {
1311        address = i.getHostAddress() + "-" + rsHostname;
1312      }
1313    }
1314    return serviceName + "@" + address + ":" + port;

1842  /**
1843   * Update the location with the new value (if the exception is a RegionMovedException)
1844   * or delete it from the cache. Does nothing if we can be sure from the exception that
1845   * the location is still accurate, or if the cache has already been updated.
1846   * @param exception an object (to simplify user code) on which we will try to find a nested
1847   *  or wrapped or both RegionMovedException
1848   * @param source server that is the source of the location update.
1849   */
1850  @Override
1851  public void updateCachedLocations(final TableName tableName, byte[] regionName, byte[] rowkey,
1852    final Object exception, final ServerName source) {
1853    if (rowkey == null || tableName == null) {
1854      LOG.warn("Coding error, see method javadoc. row=" + (rowkey == null ? "null" : rowkey) +
1855          ", tableName=" + (tableName == null ? "null" : tableName));
1856      return;
1857    }
1858
1859    if (source == null) {
1860      // This should not happen, but let's secure ourselves.
1861      return;
1862    }
1863
1864    if (regionName == null) {
1865      // we do not know which region, so just remove the cache entry for the row and server
1866      if (metrics != null) {
1867        metrics.incrCacheDroppingExceptions(exception);
1868      }
1869      metaCache.clearCache(tableName, rowkey, source);
1870      return;
1871    }
1872
1873    // Is it something we have already updated?
1874    final RegionLocations oldLocations = getCachedLocation(tableName, rowkey);
1875    HRegionLocation oldLocation = null;
1876    if (oldLocations != null) {
1877      oldLocation = oldLocations.getRegionLocationByRegionName(regionName);
1878    }
1879    if (oldLocation == null || !source.equals(oldLocation.getServerName())) {
1880      // There is no such location in the cache (it's been removed already) or
1881      // the cache has already been refreshed with a different location.  => nothing to do
1882      return;
1883    }
1884
1885    HRegionInfo regionInfo = oldLocation.getRegionInfo();
1886    Throwable cause = ClientExceptionsUtil.findException(exception);
1887    if (cause != null) {
1888      if (!ClientExceptionsUtil.isMetaClearingException(cause)) {
1889        // We know that the region is still on this region server
1890        return;
1891      }
1892
1893      if (cause instanceof RegionMovedException) {
1894        RegionMovedException rme = (RegionMovedException) cause;
1895        if (LOG.isTraceEnabled()) {
1896          LOG.trace("Region " + regionInfo.getRegionNameAsString() + " moved to " +
1897              rme.getHostname() + ":" + rme.getPort() +
1898              " according to " + source.getHostAndPort());
1899        }
1900        // We know that the region is not anymore on this region server, but we know
1901        //  the new location.
1902        updateCachedLocation(
1903            regionInfo, source, rme.getServerName(), rme.getLocationSeqNum());
1904        return;
1905      }
1906    }
1907
1908    if (metrics != null) {
1909      metrics.incrCacheDroppingExceptions(exception);
1910    }
1911
1912    // If we're here, it means that can cannot be sure about the location, so we remove it from
1913    // the cache. Do not send the source because source can be a new server in the same host:port
1914    metaCache.clearCache(regionInfo);
1915  }
