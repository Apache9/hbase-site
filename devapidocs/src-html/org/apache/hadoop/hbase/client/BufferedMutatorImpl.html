019import org.apache.commons.logging.Log;

027import java.io.IOException;
028import java.io.InterruptedIOException;
029import java.util.Arrays;
030import java.util.LinkedList;
031import java.util.List;
032import java.util.concurrent.ConcurrentLinkedQueue;
033import java.util.concurrent.ExecutorService;
034import java.util.concurrent.TimeUnit;
035import java.util.concurrent.atomic.AtomicLong;

037/**
038 * <p>
039 * Used to communicate with a single HBase table similar to {@link Table}
040 * but meant for batched, potentially asynchronous puts. Obtain an instance from
041 * a {@link Connection} and call {@link #close()} afterwards.
042 * </p>
043 *
044 * <p>
045 * While this can be used accross threads, great care should be used when doing so.
046 * Errors are global to the buffered mutator and the Exceptions can be thrown on any
047 * thread that causes the flush for requests.
048 * </p>
049 *
050 * @see ConnectionFactory
051 * @see Connection
052 * @since 1.0.0
053 */
054@InterfaceAudience.Private
055@InterfaceStability.Evolving
056public class BufferedMutatorImpl implements BufferedMutator {
057
058  private static final Log LOG = LogFactory.getLog(BufferedMutatorImpl.class);
059  
060  private final ExceptionListener listener;
061
062  protected ClusterConnection connection; // non-final so can be overridden in test
063  private final TableName tableName;
064  private volatile Configuration conf;
065
066  @VisibleForTesting
067  final ConcurrentLinkedQueue<Mutation> writeAsyncBuffer = new ConcurrentLinkedQueue<Mutation>();
068  @VisibleForTesting
069  AtomicLong currentWriteBufferSize = new AtomicLong(0);
070
071  private long writeBufferSize;
072  private final int maxKeyValueSize;
073  private boolean closed = false;
074  private final ExecutorService pool;
075
076  @VisibleForTesting
077  protected AsyncProcess ap; // non-final so can be overridden in test
078
079  BufferedMutatorImpl(ClusterConnection conn, RpcRetryingCallerFactory rpcCallerFactory,
080      RpcControllerFactory rpcFactory, BufferedMutatorParams params) {
081    if (conn == null || conn.isClosed()) {
082      throw new IllegalArgumentException("Connection is null or closed.");
083    }
084
085    this.tableName = params.getTableName();
086    this.connection = conn;
087    this.conf = connection.getConfiguration();
088    this.pool = params.getPool();
089    this.listener = params.getListener();
090
091    ConnectionConfiguration tableConf = new ConnectionConfiguration(conf);
092    this.writeBufferSize = params.getWriteBufferSize() != BufferedMutatorParams.UNSET ?
093        params.getWriteBufferSize() : tableConf.getWriteBufferSize();
094    this.maxKeyValueSize = params.getMaxKeyValueSize() != BufferedMutatorParams.UNSET ?
095        params.getMaxKeyValueSize() : tableConf.getMaxKeyValueSize();
096
097    // puts need to track errors globally due to how the APIs currently work.
098    ap = new AsyncProcess(connection, conf, pool, rpcCallerFactory, true, rpcFactory);
099  }
100
101  @Override
102  public TableName getName() {
103    return tableName;
104  }
105
106  @Override
107  public Configuration getConfiguration() {
108    return conf;
109  }
110
111  @Override
112  public void mutate(Mutation m) throws InterruptedIOException,
113      RetriesExhaustedWithDetailsException {
114    mutate(Arrays.asList(m));
115  }
116
117  @Override
118  public void mutate(List<? extends Mutation> ms) throws InterruptedIOException,
119      RetriesExhaustedWithDetailsException {
120
121    if (closed) {
122      throw new IllegalStateException("Cannot put when the BufferedMutator is closed.");
123    }
124
125    long toAddSize = 0;
126    for (Mutation m : ms) {
127      if (m instanceof Put) {
128        validatePut((Put) m);
129      }
130      toAddSize += m.heapSize();
131    }
132
133    // This behavior is highly non-intuitive... it does not protect us against
134    // 94-incompatible behavior, which is a timing issue because hasError, the below code
135    // and setter of hasError are not synchronized. Perhaps it should be removed.
136    if (ap.hasError()) {
137      currentWriteBufferSize.addAndGet(toAddSize);
138      writeAsyncBuffer.addAll(ms);
139      backgroundFlushCommits(true);
140    } else {
141      currentWriteBufferSize.addAndGet(toAddSize);
142      writeAsyncBuffer.addAll(ms);
143    }
144
145    // Now try and queue what needs to be queued.
146    while (currentWriteBufferSize.get() > writeBufferSize) {
147      backgroundFlushCommits(false);
148    }
149  }
150
151  // validate for well-formedness
152  public void validatePut(final Put put) throws IllegalArgumentException {
153    HTable.validatePut(put, maxKeyValueSize);
154  }
155
156  @Override
157  public synchronized void close() throws IOException {
158    try {
159      if (this.closed) {
160        return;
161      }
162      // As we can have an operation in progress even if the buffer is empty, we call
163      // backgroundFlushCommits at least one time.
164      backgroundFlushCommits(true);
165      this.pool.shutdown();
166      boolean terminated;
167      int loopCnt = 0;
168      do {
169        // wait until the pool has terminated
170        terminated = this.pool.awaitTermination(60, TimeUnit.SECONDS);
171        loopCnt += 1;
172        if (loopCnt >= 10) {
173          LOG.warn("close() failed to terminate pool after 10 minutes. Abandoning pool.");
174          break;
175        }
176      } while (!terminated);
177
178    } catch (InterruptedException e) {
179      LOG.warn("waitForTermination interrupted");
180
181    } finally {
182      this.closed = true;
183    }
184  }
185
186  @Override
187  public synchronized void flush() throws InterruptedIOException,
188      RetriesExhaustedWithDetailsException {
189    // As we can have an operation in progress even if the buffer is empty, we call
190    // backgroundFlushCommits at least one time.
191    backgroundFlushCommits(true);
192  }
193
194  /**
195   * Send the operations in the buffer to the servers. Does not wait for the server's answer. If
196   * the is an error (max retried reach from a previous flush or bad operation), it tries to send
197   * all operations in the buffer and sends an exception.
198   *
199   * @param synchronous - if true, sends all the writes and wait for all of them to finish before
200   *        returning.
201   */
202  private void backgroundFlushCommits(boolean synchronous) throws
203      InterruptedIOException,
204      RetriesExhaustedWithDetailsException {
205
206    LinkedList<Mutation> buffer = new LinkedList<>();
207    // Keep track of the size so that this thread doesn't spin forever
208    long dequeuedSize = 0;
209
210    try {
211      // Grab all of the available mutations.
212      Mutation m;
213
214      // If there's no buffer size drain everything. If there is a buffersize drain up to twice
215      // that amount. This should keep the loop from continually spinning if there are threads
216      // that keep adding more data to the buffer.
217      while (
218          (writeBufferSize <= 0 || dequeuedSize < (writeBufferSize * 2) || synchronous)
219              && (m = writeAsyncBuffer.poll()) != null) {
220        buffer.add(m);
221        long size = m.heapSize();
222        dequeuedSize += size;
223        currentWriteBufferSize.addAndGet(-size);
224      }
225
226      if (!synchronous && dequeuedSize == 0) {
227        return;
228      }
229
230      if (!synchronous) {
231        ap.submit(tableName, buffer, true, null, false);
232        if (ap.hasError()) {
233          LOG.debug(tableName + ": One or more of the operations have failed -"
234              + " waiting for all operation in progress to finish (successfully or not)");
235        }
236      }
237      if (synchronous || ap.hasError()) {
238        while (!buffer.isEmpty()) {
239          ap.submit(tableName, buffer, true, null, false);
240        }
241        RetriesExhaustedWithDetailsException error =
242            ap.waitForAllPreviousOpsAndReset(null, tableName.getNameAsString());
243        if (error != null) {
244          if (listener == null) {
245            throw error;
246          } else {
247            this.listener.onException(error, this);
248          }
249        }
250      }
251    } finally {
252      for (Mutation mut : buffer) {
253        long size = mut.heapSize();
254        currentWriteBufferSize.addAndGet(size);
255        dequeuedSize -= size;
256        writeAsyncBuffer.add(mut);
257      }
258    }
259  }
260
261  /**
262   * This is used for legacy purposes in {@link HTable#setWriteBufferSize(long)} only. This ought
263   * not be called for production uses.
264   * @deprecated Going away when we drop public support for {@link HTable}.
265   */
266  @Deprecated
267  public void setWriteBufferSize(long writeBufferSize) throws RetriesExhaustedWithDetailsException,
268      InterruptedIOException {
269    this.writeBufferSize = writeBufferSize;
270    if (currentWriteBufferSize.get() > writeBufferSize) {
271      flush();
272    }
273  }
274
275  /**
276   * {@inheritDoc}
277   */
278  @Override
279  public long getWriteBufferSize() {
280    return this.writeBufferSize;
281  }
