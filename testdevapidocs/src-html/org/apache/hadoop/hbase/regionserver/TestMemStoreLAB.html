068  /**
069   * Test a bunch of random allocations
070   */
071  @Test
072  public void testLABRandomAllocation() {
073    Random rand = new Random();
074    MemStoreLAB mslab = new MemStoreLABImpl();
075    int expectedOff = 0;
076    byte[] lastBuffer = null;
077    // 100K iterations by 0-1K alloc -> 50MB expected
078    // should be reasonable for unit test and also cover wraparound
079    // behavior
080    for (int i = 0; i < 100000; i++) {
081      int valSize = rand.nextInt(1000);
082      KeyValue kv = new KeyValue(rk, cf, q, new byte[valSize]);
083      int size = KeyValueUtil.length(kv);
084      KeyValue newKv = (KeyValue) mslab.copyCellInto(kv);
085      if (newKv.getBuffer() != lastBuffer) {
086        expectedOff = 0;
087        lastBuffer = newKv.getBuffer();
088      }
089      assertEquals(expectedOff, newKv.getOffset());
090      assertTrue("Allocation overruns buffer",
091          newKv.getOffset() + size <= newKv.getBuffer().length);
092      expectedOff += size;
093    }
094  }

096  @Test
097  public void testLABLargeAllocation() {
098    MemStoreLAB mslab = new MemStoreLABImpl();
099    KeyValue kv = new KeyValue(rk, cf, q, new byte[2 * 1024 * 1024]);
100    Cell newCell = mslab.copyCellInto(kv);
101    assertNull("2MB allocation shouldn't be satisfied by LAB.", newCell);
102  }

104  /**
105   * Test allocation from lots of threads, making sure the results don't
106   * overlap in any way
107   */
108  @Test
109  public void testLABThreading() throws Exception {
110    Configuration conf = new Configuration();
111    MultithreadedTestUtil.TestContext ctx =
112      new MultithreadedTestUtil.TestContext(conf);
113    
114    final AtomicInteger totalAllocated = new AtomicInteger();
115    
116    final MemStoreLAB mslab = new MemStoreLABImpl();
117    List<List<AllocRecord>> allocations = Lists.newArrayList();
118    
119    for (int i = 0; i < 10; i++) {
120      final List<AllocRecord> allocsByThisThread = Lists.newLinkedList();
121      allocations.add(allocsByThisThread);
122      
123      TestThread t = new MultithreadedTestUtil.RepeatingTestThread(ctx) {
124        private Random r = new Random();
125        @Override
126        public void doAnAction() throws Exception {
127          int valSize = r.nextInt(1000);
128          KeyValue kv = new KeyValue(rk, cf, q, new byte[valSize]);
129          int size = KeyValueUtil.length(kv);
130          KeyValue newKv = (KeyValue) mslab.copyCellInto(kv);
131          totalAllocated.addAndGet(size);
132          allocsByThisThread.add(new AllocRecord(newKv.getBuffer(), newKv.getOffset(), size));
133        }
134      };
135      ctx.addThread(t);
136    }
137    
138    ctx.startThreads();
139    while (totalAllocated.get() < 50*1024*1024 && ctx.shouldRun()) {
140      Thread.sleep(10);
141    }
142    ctx.stop();
143    
144    // Partition the allocations by the actual byte[] they point into,
145    // make sure offsets are unique for each chunk
146    Map<byte[], Map<Integer, AllocRecord>> mapsByChunk =
147      Maps.newHashMap();
148    
149    int sizeCounted = 0;
150    for (AllocRecord rec : Iterables.concat(allocations)) {
151      sizeCounted += rec.size;
152      if (rec.size == 0) continue;
153      
154      Map<Integer, AllocRecord> mapForThisByteArray =
155        mapsByChunk.get(rec.alloc);
156      if (mapForThisByteArray == null) {
157        mapForThisByteArray = Maps.newTreeMap();
158        mapsByChunk.put(rec.alloc, mapForThisByteArray);
159      }
160      AllocRecord oldVal = mapForThisByteArray.put(rec.offset, rec);
161      assertNull("Already had an entry " + oldVal + " for allocation " + rec,
162          oldVal);
163    }
164    assertEquals("Sanity check test", sizeCounted, totalAllocated.get());
165    
166    // Now check each byte array to make sure allocations don't overlap
167    for (Map<Integer, AllocRecord> allocsInChunk : mapsByChunk.values()) {
168      int expectedOff = 0;
169      for (AllocRecord alloc : allocsInChunk.values()) {
170        assertEquals(expectedOff, alloc.offset);
171        assertTrue("Allocation overruns buffer",
172            alloc.offset + alloc.size <= alloc.alloc.length);
173        expectedOff += alloc.size;
174      }
175    }
176
177  }

179  /**
180   * Test frequent chunk retirement with chunk pool triggered by lots of threads, making sure
181   * there's no memory leak (HBASE-16195)
182   * @throws Exception if any error occurred
183   */
184  @Test
185  public void testLABChunkQueue() throws Exception {
186    MemStoreLABImpl mslab = new MemStoreLABImpl();
187    // by default setting, there should be no chunks initialized in the pool
188    assertTrue(mslab.getPooledChunks().isEmpty());
189    // reset mslab with chunk pool
190    Configuration conf = HBaseConfiguration.create();
191    conf.setDouble(MemStoreLAB.CHUNK_POOL_MAXSIZE_KEY, 0.1);
192    // set chunk size to default max alloc size, so we could easily trigger chunk retirement
193    conf.setLong(MemStoreLABImpl.CHUNK_SIZE_KEY, MemStoreLABImpl.MAX_ALLOC_DEFAULT);
194    // reconstruct mslab
195    MemStoreChunkPool.clearDisableFlag();
196    mslab = new MemStoreLABImpl(conf);
197    // launch multiple threads to trigger frequent chunk retirement
198    List<Thread> threads = new ArrayList<Thread>();
199    final KeyValue kv = new KeyValue(Bytes.toBytes("r"), Bytes.toBytes("f"), Bytes.toBytes("q"),
200        new byte[MemStoreLABImpl.MAX_ALLOC_DEFAULT - 24]);
201    for (int i = 0; i < 10; i++) {
202      threads.add(getChunkQueueTestThread(mslab, "testLABChunkQueue-" + i, kv));
203    }
204    for (Thread thread : threads) {
205      thread.start();
206    }
207    // let it run for some time
208    Thread.sleep(1000);
209    for (Thread thread : threads) {
210      thread.interrupt();
211    }
212    boolean threadsRunning = true;
213    while (threadsRunning) {
214      for (Thread thread : threads) {
215        if (thread.isAlive()) {
216          threadsRunning = true;
217          break;
218        }
219      }
220      threadsRunning = false;
221    }
222    // close the mslab
223    mslab.close();
224    // make sure all chunks reclaimed or removed from chunk queue
225    int queueLength = mslab.getPooledChunks().size();
226    assertTrue("All chunks in chunk queue should be reclaimed or removed"
227        + " after mslab closed but actually: " + queueLength, queueLength == 0);
228  }

230  private Thread getChunkQueueTestThread(final MemStoreLABImpl mslab, String threadName,
231      Cell cellToCopyInto) {
232    Thread thread = new Thread() {
233      boolean stopped = false;
234
235      @Override
236      public void run() {
237        while (!stopped) {
238          // keep triggering chunk retirement
239          mslab.copyCellInto(cellToCopyInto);
240        }
241      }
242
243      @Override
244      public void interrupt() {
245        this.stopped = true;
246      }
247    };
248    thread.setName(threadName);
249    thread.setDaemon(true);
250    return thread;
251  }
252
253  private static class AllocRecord implements Comparable<AllocRecord>{
254    private final byte[] alloc;
255    private final int offset;
256    private final int size;
257
258    public AllocRecord(byte[] alloc, int offset, int size) {
259      super();
260      this.alloc = alloc;
261      this.offset = offset;
262      this.size = size;
263    }
264
265    @Override
266    public int compareTo(AllocRecord e) {
267      if (alloc != e.alloc) {
268        throw new RuntimeException("Can only compare within a particular array");
269      }
270      return Ints.compare(this.offset, e.offset);
271    }
272    
273    @Override
274    public String toString() {
275      return "AllocRecord(offset=" + this.offset + ", size=" + size + ")";
276    }
277  }
