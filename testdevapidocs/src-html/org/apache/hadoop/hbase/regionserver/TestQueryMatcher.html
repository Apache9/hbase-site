099  private void _testMatch_ExplicitColumns(Scan scan, List<MatchCode> expected) throws IOException {
100    long now = EnvironmentEdgeManager.currentTime();
101    // 2,4,5
102    ScanQueryMatcher qm = new ScanQueryMatcher(scan, new ScanInfo(this.conf, fam2,
103        0, 1, ttl, KeepDeletedCells.FALSE, 0, rowComparator), get.getFamilyMap().get(fam2),
104        now - ttl, now);
105
106    List<KeyValue> memstore = new ArrayList<KeyValue>();
107    memstore.add(new KeyValue(row1, fam2, col1, 1, data));
108    memstore.add(new KeyValue(row1, fam2, col2, 1, data));
109    memstore.add(new KeyValue(row1, fam2, col3, 1, data));
110    memstore.add(new KeyValue(row1, fam2, col4, 1, data));
111    memstore.add(new KeyValue(row1, fam2, col5, 1, data));
112
113    memstore.add(new KeyValue(row2, fam1, col1, data));
114
115    List<ScanQueryMatcher.MatchCode> actual = new ArrayList<ScanQueryMatcher.MatchCode>();
116    KeyValue k = memstore.get(0);
117    qm.setToNewRow(k);
118
119    for (KeyValue kv : memstore){
120      actual.add(qm.match(kv));
121    }
122
123    assertEquals(expected.size(), actual.size());
124    for(int i=0; i< expected.size(); i++){
125      assertEquals(expected.get(i), actual.get(i));
126      if(PRINT){
127        System.out.println("expected "+expected.get(i)+
128            ", actual " +actual.get(i));
129      }
130    }
131  }

309  @Test
310  public void testMatch_PartialRangeDropDeletes() throws Exception {
311    // Some ranges.
312    testDropDeletes(
313        row2, row3, new byte[][] { row1, row2, row2, row3 }, INCLUDE, SKIP, SKIP, INCLUDE);
314    testDropDeletes(row2, row3, new byte[][] { row1, row1, row2 }, INCLUDE, INCLUDE, SKIP);
315    testDropDeletes(row2, row3, new byte[][] { row2, row3, row3 }, SKIP, INCLUDE, INCLUDE);
316    testDropDeletes(row1, row3, new byte[][] { row1, row2, row3 }, SKIP, SKIP, INCLUDE);
317    // Open ranges.
318    testDropDeletes(HConstants.EMPTY_START_ROW, row3,
319        new byte[][] { row1, row2, row3 }, SKIP, SKIP, INCLUDE);
320    testDropDeletes(row2, HConstants.EMPTY_END_ROW,
321        new byte[][] { row1, row2, row3 }, INCLUDE, SKIP, SKIP);
322    testDropDeletes(HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW,
323        new byte[][] { row1, row2, row3, row3 }, SKIP, SKIP, SKIP, SKIP);
324
325    // No KVs in range.
326    testDropDeletes(row2, row3, new byte[][] { row1, row1, row3 }, INCLUDE, INCLUDE, INCLUDE);
327    testDropDeletes(row2, row3, new byte[][] { row3, row3 }, INCLUDE, INCLUDE);
328    testDropDeletes(row2, row3, new byte[][] { row1, row1 }, INCLUDE, INCLUDE);
329  }

331  private void testDropDeletes(
332      byte[] from, byte[] to, byte[][] rows, MatchCode... expected) throws IOException {
333    long now = EnvironmentEdgeManager.currentTime();
334    // Set time to purge deletes to negative value to avoid it ever happening.
335    ScanInfo scanInfo =
336      new ScanInfo(this.conf, fam2, 0, 1, ttl, KeepDeletedCells.FALSE, -1L, rowComparator);
337    NavigableSet<byte[]> cols = get.getFamilyMap().get(fam2);
338
339    ScanQueryMatcher qm = new ScanQueryMatcher(scan, scanInfo, cols, Long.MAX_VALUE,
340        HConstants.OLDEST_TIMESTAMP, HConstants.OLDEST_TIMESTAMP, now, from, to, null);
341    List<ScanQueryMatcher.MatchCode> actual =
342        new ArrayList<ScanQueryMatcher.MatchCode>(rows.length);
343    byte[] prevRow = null;
344    for (byte[] row : rows) {
345      if (prevRow == null || !Bytes.equals(prevRow, row)) {
346        qm.setToNewRow(KeyValueUtil.createFirstOnRow(row));
347        prevRow = row;
348      }
349      actual.add(qm.match(new KeyValue(row, fam2, null, now, Type.Delete)));
350    }
351
352    assertEquals(expected.length, actual.size());
353    for (int i = 0; i < expected.length; i++) {
354      if (PRINT) System.out.println("expected " + expected[i] + ", actual " + actual.get(i));
355      assertEquals(expected[i], actual.get(i));
356    }
357  }
