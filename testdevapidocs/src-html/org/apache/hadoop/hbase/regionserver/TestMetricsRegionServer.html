101  @Test
102  public void testSlowCount() {
103    for (int i=0; i < 12; i ++) {
104      rsm.updateAppend(12);
105      rsm.updateAppend(1002);
106    }
107    for (int i=0; i < 13; i ++) {
108      rsm.updateDelete(13);
109      rsm.updateDelete(1003);
110    }
111    for (int i=0; i < 14; i ++) {
112      rsm.updateGet(14);
113      rsm.updateGet(1004);
114    }
115    for (int i=0; i < 15; i ++) {
116      rsm.updateIncrement(15);
117      rsm.updateIncrement(1005);
118    }
119    for (int i=0; i < 16; i ++) {
120      rsm.updatePut(16);
121      rsm.updatePut(1006);
122    }
123
124    HELPER.assertCounter("appendNumOps", 24, serverSource);
125    HELPER.assertCounter("deleteNumOps", 26, serverSource);
126    HELPER.assertCounter("getNumOps", 28, serverSource);
127    HELPER.assertCounter("incrementNumOps", 30, serverSource);
128    HELPER.assertCounter("mutateNumOps", 32, serverSource);
129
130
131    HELPER.assertCounter("slowAppendCount", 12, serverSource);
132    HELPER.assertCounter("slowDeleteCount", 13, serverSource);
133    HELPER.assertCounter("slowGetCount", 14, serverSource);
134    HELPER.assertCounter("slowIncrementCount", 15, serverSource);
135    HELPER.assertCounter("slowPutCount", 16, serverSource);
136  }
