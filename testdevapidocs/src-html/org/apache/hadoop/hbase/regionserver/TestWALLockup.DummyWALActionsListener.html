110  /**
111   * Reproduce locking up that happens when we get an inopportune sync during setup for
112   * zigzaglatch wait. See HBASE-14317. If below is broken, we will see this test timeout because
113   * it is locked up.
114   * <p>First I need to set up some mocks for Server and RegionServerServices. I also need to
115   * set up a dodgy WAL that will throw an exception when we go to append to it.
116   */
117  @Test (timeout=20000)
118  public void testLockupWhenSyncInMiddleOfZigZagSetup() throws IOException {
119    // A WAL that we can have throw exceptions when a flag is set.
120    class DodgyFSLog extends FSHLog {
121      // Set this when want the WAL to start throwing exceptions.
122      volatile boolean throwException = false;
123
124      // Latch to hold up processing until after another operation has had time to run.
125      CountDownLatch latch = new CountDownLatch(1);
126
127      public DodgyFSLog(FileSystem fs, Path root, String logDir, Configuration conf)
128      throws IOException {
129        super(fs, root, logDir, conf);
130      }
131
132      @Override
133      protected void afterCreatingZigZagLatch() {
134        // If throwException set, then append will throw an exception causing the WAL to be
135        // rolled. We'll come in here. Hold up processing until a sync can get in before
136        // the zigzag has time to complete its setup and get its own sync in. This is what causes
137        // the lock up we've seen in production.
138        if (throwException) {
139          try {
140            LOG.info("LATCHED");
141            // So, timing can have it that the test can run and the bad flush below happens
142            // before we get here. In this case, we'll be stuck waiting on this latch but there
143            // is nothing in the WAL pipeline to get us to the below beforeWaitOnSafePoint...
144            // because all WALs have rolled. In this case, just give up on test.
145            if (!this.latch.await(5, TimeUnit.SECONDS)) {
146              LOG.warn("GIVE UP! Failed waiting on latch...Test is ABORTED!");
147            }
148          } catch (InterruptedException e) {
149            // TODO Auto-generated catch block
150            e.printStackTrace();
151          }
152        }
153      }
154
155      @Override
156      protected void beforeWaitOnSafePoint() {
157        if (throwException) {
158          LOG.info("COUNTDOWN");
159          // Don't countdown latch until someone waiting on it otherwise, the above
160          // afterCreatingZigZagLatch will get to the latch and no one will ever free it and we'll
161          // be stuck; test won't go down
162          while (this.latch.getCount() <= 0) Threads.sleep(1);
163          this.latch.countDown();
164        }
165      }
166
167      @Override
168      protected Writer createWriterInstance(Path path) throws IOException {
169        final Writer w = super.createWriterInstance(path);
170        return new Writer() {
171          @Override
172          public void close() throws IOException {
173            w.close();
174          }
175
176          @Override
177          public void sync() throws IOException {
178            if (throwException) {
179              throw new IOException("FAKE! Failed to replace a bad datanode...SYNC");
180            }
181            w.sync();
182          }
183
184          @Override
185          public void append(Entry entry) throws IOException {
186            if (throwException) {
187              throw new IOException("FAKE! Failed to replace a bad datanode...APPEND");
188            }
189            w.append(entry);
190          }
191
192          @Override
193          public long getLength() {
194            return w.getLength();
195          }
196        };
197      }
198    }
199
200    // Mocked up server and regionserver services. Needed below.
201    Server server = Mockito.mock(Server.class);
202    Mockito.when(server.getConfiguration()).thenReturn(CONF);
203    Mockito.when(server.isStopped()).thenReturn(false);
204    Mockito.when(server.isAborted()).thenReturn(false);
205    RegionServerServices services = Mockito.mock(RegionServerServices.class);
206
207    // OK. Now I have my mocked up Server & RegionServerServices and dodgy WAL, go ahead with test.
208    FileSystem fs = FileSystem.get(CONF);
209    Path rootDir = new Path(dir + getName());
210    DodgyFSLog dodgyWAL = new DodgyFSLog(fs, rootDir, getName(), CONF);
211    Path originalWAL = dodgyWAL.getCurrentFileName();
212    // I need a log roller running.
213    LogRoller logRoller = new LogRoller(server, services);
214    logRoller.addWAL(dodgyWAL);
215    // There is no 'stop' once a logRoller is running.. it just dies.
216    logRoller.start();
217    // Now get a region and start adding in edits.
218    HTableDescriptor htd = new HTableDescriptor(TableName.META_TABLE_NAME);
219    final HRegion region = initHRegion(tableName, null, null, dodgyWAL);
220    byte [] bytes = Bytes.toBytes(getName());
221    NavigableMap<byte[], Integer> scopes = new TreeMap<>(
222        Bytes.BYTES_COMPARATOR);
223    scopes.put(COLUMN_FAMILY_BYTES, 0);
224    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
225    try {
226      // First get something into memstore. Make a Put and then pull the Cell out of it. Will
227      // manage append and sync carefully in below to manufacture hang. We keep adding same
228      // edit. WAL subsystem doesn't care.
229      Put put = new Put(bytes);
230      put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("1"), bytes);
231      WALKey key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(), htd.getTableName(),
232          System.currentTimeMillis(), mvcc, scopes);
233      WALEdit edit = new WALEdit();
234      CellScanner CellScanner = put.cellScanner();
235      assertTrue(CellScanner.advance());
236      edit.add(CellScanner.current());
237      // Put something in memstore and out in the WAL. Do a big number of appends so we push
238      // out other side of the ringbuffer. If small numbers, stuff doesn't make it to WAL
239      for (int i = 0; i < 1000; i++) {
240        region.put(put);
241      }
242      // Set it so we start throwing exceptions.
243      LOG.info("SET throwing of exception on append");
244      dodgyWAL.throwException = true;
245      // This append provokes a WAL roll request
246      dodgyWAL.append(region.getRegionInfo(), key, edit, true);
247      boolean exception = false;
248      try {
249        dodgyWAL.sync();
250      } catch (Exception e) {
251        exception = true;
252      }
253      assertTrue("Did not get sync exception", exception);
254
255      // Get a memstore flush going too so we have same hung profile as up in the issue over
256      // in HBASE-14317. Flush hangs trying to get sequenceid because the ringbuffer is held up
257      // by the zigzaglatch waiting on syncs to come home.
258      Thread t = new Thread ("Flusher") {
259        public void run() {
260          try {
261            if (region.getMemStoreSize() <= 0) {
262              throw new IOException("memstore size=" + region.getMemStoreSize());
263            }
264            region.flush(false);
265          } catch (IOException e) {
266            // Can fail trying to flush in middle of a roll. Not a failure. Will succeed later
267            // when roll completes.
268            LOG.info("In flush", e);
269          }
270          LOG.info("Exiting");
271        };
272      };
273      t.setDaemon(true);
274      t.start();
275      // Wait until
276      while (dodgyWAL.latch.getCount() > 0) Threads.sleep(1);
277      // Now assert I got a new WAL file put in place even though loads of errors above.
278      assertTrue(originalWAL != dodgyWAL.getCurrentFileName());
279      // Can I append to it?
280      dodgyWAL.throwException = false;
281      try {
282        region.put(put);
283      } catch (Exception e) {
284        LOG.info("In the put", e);
285      }
286    } finally {
287      // To stop logRoller, its server has to say it is stopped.
288      Mockito.when(server.isStopped()).thenReturn(true);
289      if (logRoller != null) logRoller.close();
290      try {
291        if (region != null) region.close();
292        if (dodgyWAL != null) dodgyWAL.close();
293      } catch (Exception e) {
294        LOG.info("On way out", e);
295      }
296    }
297  }

299  /**
300   * Reproduce locking up that happens when there's no further syncs after
301   * append fails, and causing an isolated sync then infinite wait. See
302   * HBASE-16960. If below is broken, we will see this test timeout because it
303   * is locked up.
304   * <p/>
305   * Steps for reproduce:<br/>
306   * 1. Trigger server abort through dodgyWAL1<br/>
307   * 2. Add a {@link DummyWALActionsListener} to dodgyWAL2 to cause ringbuffer
308   * event handler thread sleep for a while thus keeping {@code endOfBatch}
309   * false<br/>
310   * 3. Publish a sync then an append which will throw exception, check whether
311   * the sync could return
312   */
313  @Test(timeout = 20000)
314  public void testLockup16960() throws IOException {
315    // A WAL that we can have throw exceptions when a flag is set.
316    class DodgyFSLog extends FSHLog {
317      // Set this when want the WAL to start throwing exceptions.
318      volatile boolean throwException = false;
319
320      public DodgyFSLog(FileSystem fs, Path root, String logDir,
321          Configuration conf) throws IOException {
322        super(fs, root, logDir, conf);
323      }
324
325      @Override
326      protected Writer createWriterInstance(Path path) throws IOException {
327        final Writer w = super.createWriterInstance(path);
328        return new Writer() {
329          @Override
330          public void close() throws IOException {
331            w.close();
332          }
333
334          @Override
335          public void sync() throws IOException {
336            if (throwException) {
337              throw new IOException(
338                  "FAKE! Failed to replace a bad datanode...SYNC");
339            }
340            w.sync();
341          }
342
343          @Override
344          public void append(Entry entry) throws IOException {
345            if (throwException) {
346              throw new IOException(
347                  "FAKE! Failed to replace a bad datanode...APPEND");
348            }
349            w.append(entry);
350          }
351
352          @Override
353          public long getLength() {
354            return w.getLength();
355          }
356        };
357      }
358
359      @Override
360      protected long doReplaceWriter(Path oldPath, Path newPath,
361          Writer nextWriter) throws IOException {
362        if (throwException) {
363          throw new FailedLogCloseException("oldPath=" + oldPath + ", newPath="
364              + newPath);
365        }
366        long oldFileLen = 0L;
367        oldFileLen = super.doReplaceWriter(oldPath, newPath, nextWriter);
368        return oldFileLen;
369      }
370    }
371
372    // Mocked up server and regionserver services. Needed below.
373    Server server = new DummyServer(CONF, ServerName.valueOf(
374        "hostname1.example.org", 1234, 1L).toString());
375    RegionServerServices services = Mockito.mock(RegionServerServices.class);
376
377    CONF.setLong("hbase.regionserver.hlog.sync.timeout", 10000);
378
379    // OK. Now I have my mocked up Server & RegionServerServices and dodgy WAL,
380    // go ahead with test.
381    FileSystem fs = FileSystem.get(CONF);
382    Path rootDir = new Path(dir + getName());
383    DodgyFSLog dodgyWAL1 = new DodgyFSLog(fs, rootDir, getName(), CONF);
384
385    Path rootDir2 = new Path(dir + getName() + "2");
386    final DodgyFSLog dodgyWAL2 = new DodgyFSLog(fs, rootDir2, getName() + "2",
387        CONF);
388    // Add a listener to force ringbuffer event handler sleep for a while
389    dodgyWAL2.registerWALActionsListener(new DummyWALActionsListener());
390
391    // I need a log roller running.
392    LogRoller logRoller = new LogRoller(server, services);
393    logRoller.addWAL(dodgyWAL1);
394    logRoller.addWAL(dodgyWAL2);
395    // There is no 'stop' once a logRoller is running.. it just dies.
396    logRoller.start();
397    // Now get a region and start adding in edits.
398    HTableDescriptor htd = new HTableDescriptor(TableName.META_TABLE_NAME);
399    final HRegion region = initHRegion(tableName, null, null, dodgyWAL1);
400    byte[] bytes = Bytes.toBytes(getName());
401    NavigableMap<byte[], Integer> scopes = new TreeMap<>(
402        Bytes.BYTES_COMPARATOR);
403    scopes.put(COLUMN_FAMILY_BYTES, 0);
404    MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl();
405    try {
406      Put put = new Put(bytes);
407      put.addColumn(COLUMN_FAMILY_BYTES, Bytes.toBytes("1"), bytes);
408      WALKey key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(), htd.getTableName(),
409          System.currentTimeMillis(), mvcc, scopes);
410      WALEdit edit = new WALEdit();
411      CellScanner CellScanner = put.cellScanner();
412      assertTrue(CellScanner.advance());
413      edit.add(CellScanner.current());
414
415      LOG.info("SET throwing of exception on append");
416      dodgyWAL1.throwException = true;
417      // This append provokes a WAL roll request
418      dodgyWAL1.append(region.getRegionInfo(), key, edit, true);
419      boolean exception = false;
420      try {
421        dodgyWAL1.sync();
422      } catch (Exception e) {
423        exception = true;
424      }
425      assertTrue("Did not get sync exception", exception);
426
427      // LogRoller call dodgyWAL1.rollWriter get FailedLogCloseException and
428      // cause server abort.
429      try {
430        // wait LogRoller exit.
431        Thread.sleep(50);
432      } catch (InterruptedException e) {
433        e.printStackTrace();
434      }
435
436      final CountDownLatch latch = new CountDownLatch(1);
437
438      // make RingBufferEventHandler sleep 1s, so the following sync
439      // endOfBatch=false
440      key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(), TableName.valueOf("sleep"),
441          System.currentTimeMillis(), mvcc, scopes);
442      dodgyWAL2.append(region.getRegionInfo(), key, edit, true);
443
444      Thread t = new Thread("Sync") {
445        public void run() {
446          try {
447            dodgyWAL2.sync();
448          } catch (IOException e) {
449            LOG.info("In sync", e);
450          }
451          latch.countDown();
452          LOG.info("Sync exiting");
453        };
454      };
455      t.setDaemon(true);
456      t.start();
457      try {
458        // make sure sync have published.
459        Thread.sleep(100);
460      } catch (InterruptedException e1) {
461        e1.printStackTrace();
462      }
463      // make append throw DamagedWALException
464      key = new WALKey(region.getRegionInfo().getEncodedNameAsBytes(),
465          TableName.valueOf("DamagedWALException"), System.currentTimeMillis(), mvcc, scopes);
466      dodgyWAL2.append(region.getRegionInfo(), key, edit, true);
467
468      while (latch.getCount() > 0) {
469        Threads.sleep(100);
470      }
471      assertTrue(server.isAborted());
472    } finally {
473      if (logRoller != null) {
474        logRoller.close();
475      }
476      try {
477        if (region != null) {
478          region.close();
479        }
480        if (dodgyWAL1 != null) {
481          dodgyWAL1.close();
482        }
483        if (dodgyWAL2 != null) {
484          dodgyWAL2.close();
485        }
486      } catch (Exception e) {
487        LOG.info("On way out", e);
488      }
489    }
490  }

574  static class DummyWALActionsListener extends WALActionsListener.Base {
575
576    @Override
577    public void visitLogEntryBeforeWrite(WALKey logKey, WALEdit logEdit)
578        throws IOException {
579      if (logKey.getTablename().getNameAsString().equalsIgnoreCase("sleep")) {
580        try {
581          Thread.sleep(1000);
582        } catch (InterruptedException e) {
583          e.printStackTrace();
584        }
585      }
586      if (logKey.getTablename().getNameAsString()
587          .equalsIgnoreCase("DamagedWALException")) {
588        throw new DamagedWALException("Failed appending");
589      }
590    }
