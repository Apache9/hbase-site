046@Category(MediumTests.class)
047public class TestFlushWithThroughputController {
048
049  private static final Log LOG = LogFactory.getLog(TestFlushWithThroughputController.class);
050
051  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
052
053  private static final double EPSILON = 1E-6;
054
055  private final TableName tableName = TableName.valueOf(getClass().getSimpleName());
056
057  private final byte[] family = Bytes.toBytes("f");
058
059  private final byte[] qualifier = Bytes.toBytes("q");
060
061  private Store getStoreWithName(TableName tableName) {
062    MiniHBaseCluster cluster = TEST_UTIL.getMiniHBaseCluster();
063    List<JVMClusterUtil.RegionServerThread> rsts = cluster.getRegionServerThreads();
064    for (int i = 0; i < cluster.getRegionServerThreads().size(); i++) {
065      HRegionServer hrs = rsts.get(i).getRegionServer();
066      for (Region region : hrs.getOnlineRegions(tableName)) {
067        return region.getStores().iterator().next();
068      }
069    }
070    return null;
071  }
072
073  private Store generateAndFlushData() throws IOException {
074    HBaseAdmin admin = TEST_UTIL.getHBaseAdmin();
075    if (admin.tableExists(tableName)) {
076      admin.disableTable(tableName);
077      admin.deleteTable(tableName);
078    }
079    HTable table = TEST_UTIL.createTable(tableName, family);
080    Random rand = new Random();
081    for (int i = 0; i < 10; i++) {
082      for (int j = 0; j < 10; j++) {
083        byte[] value = new byte[256 * 1024];
084        rand.nextBytes(value);
085        table.put(new Put(Bytes.toBytes(i * 10 + j)).addColumn(family, qualifier, value));
086      }
087      admin.flush(tableName);
088    }
089    return getStoreWithName(tableName);
090  }
091
092  private long testFlushWithThroughputLimit() throws Exception {
093    long throughputLimit = 1L * 1024 * 1024;
094    Configuration conf = TEST_UTIL.getConfiguration();
095    conf.set(FlushThroughputControllerFactory.HBASE_FLUSH_THROUGHPUT_CONTROLLER_KEY,
096      PressureAwareFlushThroughputController.class.getName());
097    conf.setLong(
098      PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_MAX_THROUGHPUT_LOWER_BOUND,
099      throughputLimit);
100    conf.setLong(
101      PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_MAX_THROUGHPUT_UPPER_BOUND,
102      throughputLimit);
103    conf.setLong(
104      PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_THROUGHPUT_CONTROL_CHECK_INTERVAL,
105      throughputLimit);
106    TEST_UTIL.startMiniCluster(1);
107    try {
108      long startTime = System.nanoTime();
109      Store store = generateAndFlushData();
110      assertEquals(10, store.getStorefilesCount());
111      long duration = System.nanoTime() - startTime;
112      double throughput = (double) store.getStorefilesSize() / duration * 1000 * 1000 * 1000;
113      LOG.debug("Throughput is: " + (throughput / 1024 / 1024) + " MB/s");
114      // confirm that the speed limit work properly(not too fast, and also not too slow)
115      // 20% is the max acceptable error rate.
116      assertTrue(throughput < throughputLimit * 1.2);
117      assertTrue(throughput > throughputLimit * 0.8);
118      return duration;
119    } finally {
120      TEST_UTIL.shutdownMiniCluster();
121    }
122  }
123
124  private long testFlushWithoutThroughputLimit() throws Exception {
125    Configuration conf = TEST_UTIL.getConfiguration();
126    conf.set(FlushThroughputControllerFactory.HBASE_FLUSH_THROUGHPUT_CONTROLLER_KEY,
127      NoLimitThroughputController.class.getName());
128    TEST_UTIL.startMiniCluster(1);
129    try {
130      long startTime = System.nanoTime();
131      Store store = generateAndFlushData();
132      assertEquals(10, store.getStorefilesCount());
133      long duration = System.nanoTime() - startTime;
134      double throughput = (double) store.getStorefilesSize() / duration * 1000 * 1000 * 1000;
135      LOG.debug("Throughput w/o limit is: " + (throughput / 1024 / 1024) + " MB/s");
136      return duration;
137    } finally {
138      TEST_UTIL.shutdownMiniCluster();
139    }
140  }
141
142  @Test
143  public void testFlushControl() throws Exception {
144    long limitTime = testFlushWithThroughputLimit();
145    long noLimitTime = testFlushWithoutThroughputLimit();
146    LOG.info("With 1M/s limit, flush use " + (limitTime / 1000000)
147        + "ms; without limit, flush use " + (noLimitTime / 1000000) + "ms");
148    // Commonly if multiple region flush at the same time, the throughput could be very high
149    // but flush in this test is in serial, so we use a weak assumption.
150    assertTrue(limitTime > 2 * noLimitTime);
151  }

153  /**
154   * Test the tuning task of {@link PressureAwareFlushThroughputController}
155   */
156  @Test
157  public void testFlushThroughputTuning() throws Exception {
158    Configuration conf = TEST_UTIL.getConfiguration();
159    conf.set(StoreEngine.STORE_ENGINE_CLASS_KEY, DefaultStoreEngine.class.getName());
160    conf.setLong(
161      PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_MAX_THROUGHPUT_UPPER_BOUND,
162      20L * 1024 * 1024);
163    conf.setLong(
164      PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_MAX_THROUGHPUT_LOWER_BOUND,
165      10L * 1024 * 1024);
166    conf.set(FlushThroughputControllerFactory.HBASE_FLUSH_THROUGHPUT_CONTROLLER_KEY,
167      PressureAwareFlushThroughputController.class.getName());
168    conf.setInt(PressureAwareFlushThroughputController.HBASE_HSTORE_FLUSH_THROUGHPUT_TUNE_PERIOD,
169      3000);
170    TEST_UTIL.startMiniCluster(1);
171    Connection conn = ConnectionFactory.createConnection(conf);
172    try {
173      HTableDescriptor htd = new HTableDescriptor(tableName);
174      htd.addFamily(new HColumnDescriptor(family));
175      htd.setCompactionEnabled(false);
176      TEST_UTIL.getHBaseAdmin().createTable(htd);
177      TEST_UTIL.waitTableAvailable(tableName);
178      HRegionServer regionServer = TEST_UTIL.getRSForFirstRegionInTable(tableName);
179      PressureAwareFlushThroughputController throughputController =
180          (PressureAwareFlushThroughputController) regionServer.getFlushThroughputController();
181      for (Region region : regionServer.getOnlineRegions()) {
182        region.flush(true);
183      }
184      assertEquals(0.0, regionServer.getFlushPressure(), EPSILON);
185      Thread.sleep(5000);
186      assertEquals(10L * 1024 * 1024, throughputController.getMaxThroughput(), EPSILON);
187      Table table = conn.getTable(tableName);
188      Random rand = new Random();
189      for (int i = 0; i < 10; i++) {
190        for (int j = 0; j < 10; j++) {
191          byte[] value = new byte[256 * 1024];
192          rand.nextBytes(value);
193          table.put(new Put(Bytes.toBytes(i * 10 + j)).addColumn(family, qualifier, value));
194        }
195      }
196      Thread.sleep(5000);
197      double expectedThroughPut = 10L * 1024 * 1024 * (1 + regionServer.getFlushPressure());
198      assertEquals(expectedThroughPut, throughputController.getMaxThroughput(), EPSILON);
199
200      conf.set(FlushThroughputControllerFactory.HBASE_FLUSH_THROUGHPUT_CONTROLLER_KEY,
201        NoLimitThroughputController.class.getName());
202      regionServer.onConfigurationChange(conf);
203      assertTrue(throughputController.isStopped());
204      assertTrue(regionServer.getFlushThroughputController() instanceof NoLimitThroughputController);
205    } finally {
206      conn.close();
207      TEST_UTIL.shutdownMiniCluster();
208    }
209  }
