046/**
047 * A test similar to TestHRegion, but with in-memory flush families.
048 * Also checks wal truncation after in-memory compaction.
049 */
050@Category({VerySlowRegionServerTests.class, LargeTests.class})
051@SuppressWarnings("deprecation")
052public class TestHRegionWithInMemoryFlush extends TestHRegion{
053  // Do not spin up clusters in here. If you need to spin up a cluster, do it
054  // over in TestHRegionOnCluster.
055  private static final Log LOG = LogFactory.getLog(TestHRegionWithInMemoryFlush.class);
056  @ClassRule
057  public static final TestRule timeout =
058      CategoryBasedTimeout.forClass(TestHRegionWithInMemoryFlush.class);
059
060  /**
061   * @return A region on which you must call
062   *         {@link HBaseTestingUtility#closeRegionAndWAL(HRegion)} when done.
063   */
064  @Override
065  public HRegion initHRegion(TableName tableName, byte[] startKey, byte[] stopKey,
066      boolean isReadOnly, Durability durability, WAL wal, byte[]... families) throws IOException {
067    boolean[] inMemory = new boolean[families.length];
068    for(int i = 0; i < inMemory.length; i++) {
069      inMemory[i] = true;
070    }
071    return TEST_UTIL.createLocalHRegionWithInMemoryFlags(tableName, startKey, stopKey,
072        isReadOnly, durability, wal, inMemory, families);
073  }
074
075  /**
076   * Splits twice and verifies getting from each of the split regions.
077   *
078   * @throws Exception
079   */
080  @Override
081  public void testBasicSplit() throws Exception {
082    byte[][] families = { fam1, fam2, fam3 };
083
084    Configuration hc = initSplit();
085    // Setting up region
086    this.region = initHRegion(tableName, method, hc, families);
087
088    try {
089      LOG.info("" + HBaseTestCase.addContent(region, fam3));
090      region.flush(true);
091      region.compactStores();
092      byte[] splitRow = region.checkSplit();
093      assertNotNull(splitRow);
094      LOG.info("SplitRow: " + Bytes.toString(splitRow));
095      HRegion[] regions = splitRegion(region, splitRow);
096      try {
097        // Need to open the regions.
098        // TODO: Add an 'open' to HRegion... don't do open by constructing
099        // instance.
100        for (int i = 0; i < regions.length; i++) {
101          regions[i] = HRegion.openHRegion(regions[i], null);
102        }
103        // Assert can get rows out of new regions. Should be able to get first
104        // row from first region and the midkey from second region.
105        assertGet(regions[0], fam3, Bytes.toBytes(START_KEY));
106        assertGet(regions[1], fam3, splitRow);
107        // Test I can get scanner and that it starts at right place.
108        assertScan(regions[0], fam3, Bytes.toBytes(START_KEY));
109        assertScan(regions[1], fam3, splitRow);
110        // Now prove can't split regions that have references.
111        for (int i = 0; i < regions.length; i++) {
112          // Add so much data to this region, we create a store file that is >
113          // than one of our unsplitable references. it will.
114          for (int j = 0; j < 2; j++) {
115            HBaseTestCase.addContent(regions[i], fam3);
116          }
117          HBaseTestCase.addContent(regions[i], fam2);
118          HBaseTestCase.addContent(regions[i], fam1);
119          regions[i].flush(true);
120        }
121
122        byte[][] midkeys = new byte[regions.length][];
123        // To make regions splitable force compaction.
124        for (int i = 0; i < regions.length; i++) {
125          regions[i].compactStores();
126          midkeys[i] = regions[i].checkSplit();
127        }
128
129        TreeMap<String, HRegion> sortedMap = new TreeMap<String, HRegion>();
130        // Split these two daughter regions so then I'll have 4 regions. Will
131        // split because added data above.
132        for (int i = 0; i < regions.length; i++) {
133          HRegion[] rs = null;
134          if (midkeys[i] != null) {
135            rs = splitRegion(regions[i], midkeys[i]);
136            for (int j = 0; j < rs.length; j++) {
137              sortedMap.put(Bytes.toString(rs[j].getRegionInfo().getRegionName()),
138                  HRegion.openHRegion(rs[j], null));
139            }
140          }
141        }
142        LOG.info("Made 4 regions");
143      } finally {
144        for (int i = 0; i < regions.length; i++) {
145          try {
146            regions[i].close();
147          } catch (IOException e) {
148            // Ignore.
149          }
150        }
151      }
152    } finally {
153      HBaseTestingUtility.closeRegionAndWAL(this.region);
154      this.region = null;
155    }
156  }
