041  ArrayList<StoreFile> sfCreate(long[] minTimestamps, long[] maxTimestamps, long[] sizes)
042      throws IOException {
043    ManualEnvironmentEdge timeMachine = new ManualEnvironmentEdge();
044    EnvironmentEdgeManager.injectEdge(timeMachine);
045    // Has to be  > 0 and < now.
046    timeMachine.setValue(1);
047    ArrayList<Long> ageInDisk = new ArrayList<Long>();
048    for (int i = 0; i < sizes.length; i++) {
049      ageInDisk.add(0L);
050    }
051
052    ArrayList<StoreFile> ret = Lists.newArrayList();
053    for (int i = 0; i < sizes.length; i++) {
054      MockStoreFile msf =
055          new MockStoreFile(TEST_UTIL, TEST_FILE, sizes[i], ageInDisk.get(i), false, i);
056      msf.setTimeRangeTracker(new TimeRangeTracker(minTimestamps[i], maxTimestamps[i]));
057      ret.add(msf);
058    }
059    return ret;
060  }

062  @Override
063  protected void config() {
064    super.config();
065
066    // Set up policy
067    conf.set(StoreEngine.STORE_ENGINE_CLASS_KEY,
068      "org.apache.hadoop.hbase.regionserver.DateTieredStoreEngine");
069    conf.setLong(CompactionConfiguration.MAX_AGE_MILLIS_KEY, 100);
070    conf.setLong(CompactionConfiguration.INCOMING_WINDOW_MIN_KEY, 3);
071    conf.setLong(CompactionConfiguration.BASE_WINDOW_MILLIS_KEY, 6);
072    conf.setInt(CompactionConfiguration.WINDOWS_PER_TIER_KEY, 4);
073    conf.setBoolean(CompactionConfiguration.SINGLE_OUTPUT_FOR_MINOR_COMPACTION_KEY, false);
074
075    // Special settings for compaction policy per window
076    this.conf.setInt(CompactionConfiguration.HBASE_HSTORE_COMPACTION_MIN_KEY, 2);
077    this.conf.setInt(CompactionConfiguration.HBASE_HSTORE_COMPACTION_MAX_KEY, 12);
078    this.conf.setFloat(CompactionConfiguration.HBASE_HSTORE_COMPACTION_RATIO_KEY, 1.2F);
079
080    conf.setInt(HStore.BLOCKING_STOREFILES_KEY, 20);
081    conf.setLong(HConstants.MAJOR_COMPACTION_PERIOD, 10);
082  }

084  void compactEquals(long now, ArrayList<StoreFile> candidates, long[] expectedFileSizes,
085      long[] expectedBoundaries, boolean isMajor, boolean toCompact) throws IOException {
086    ManualEnvironmentEdge timeMachine = new ManualEnvironmentEdge();
087    EnvironmentEdgeManager.injectEdge(timeMachine);
088    timeMachine.setValue(now);
089    DateTieredCompactionRequest request;
090    if (isMajor) {
091      for (StoreFile file : candidates) {
092        ((MockStoreFile)file).setIsMajor(true);
093      }
094      Assert.assertEquals(toCompact, ((DateTieredCompactionPolicy) store.storeEngine.getCompactionPolicy())
095        .shouldPerformMajorCompaction(candidates));
096      request = (DateTieredCompactionRequest) ((DateTieredCompactionPolicy) store.storeEngine
097          .getCompactionPolicy()).selectMajorCompaction(candidates);
098    } else {
099      Assert.assertEquals(toCompact, ((DateTieredCompactionPolicy) store.storeEngine.getCompactionPolicy())
100          .needsCompaction(candidates, ImmutableList.<StoreFile> of()));
101      request = (DateTieredCompactionRequest) ((DateTieredCompactionPolicy) store.storeEngine
102          .getCompactionPolicy()).selectMinorCompaction(candidates, false, false);
103    }
104    List<StoreFile> actual = Lists.newArrayList(request.getFiles());
105    Assert.assertEquals(Arrays.toString(expectedFileSizes), Arrays.toString(getSizes(actual)));
106    Assert.assertEquals(Arrays.toString(expectedBoundaries),
107    Arrays.toString(request.getBoundaries().toArray()));
108  }
