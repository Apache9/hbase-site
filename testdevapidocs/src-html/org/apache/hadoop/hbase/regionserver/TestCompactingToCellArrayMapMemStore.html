076  public void testCompaction1Bucket() throws IOException {
077    int counter = 0;
078    String[] keys1 = { "A", "A", "B", "C" }; //A1, A2, B3, C4
079
080    // test 1 bucket
081    long totalCellsLen = addRowsByKeys(memstore, keys1);
082    long totalHeapOverhead = 4 * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
083    assertEquals(totalCellsLen, regionServicesForStores.getMemstoreSize());
084    assertEquals(totalHeapOverhead, ((CompactingMemStore)memstore).heapOverhead());
085
086    assertEquals(4, memstore.getActive().getCellsCount());
087    MemstoreSize size = memstore.getFlushableSize();
088    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
089    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
090      Threads.sleep(10);
091    }
092    assertEquals(0, memstore.getSnapshot().getCellsCount());
093    // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
094    // totalCellsLen
095    totalCellsLen = (totalCellsLen * 3) / 4;
096    assertEquals(totalCellsLen, regionServicesForStores.getMemstoreSize());
097    totalHeapOverhead = 3 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY);
098    assertEquals(totalHeapOverhead, ((CompactingMemStore)memstore).heapOverhead());
099    for ( Segment s : memstore.getSegments()) {
100      counter += s.getCellsCount();
101    }
102    assertEquals(3, counter);
103    size = memstore.getFlushableSize();
104    MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
105    region.decrMemstoreSize(size);  // simulate flusher
106    ImmutableSegment s = memstore.getSnapshot();
107    assertEquals(3, s.getCellsCount());
108    assertEquals(0, regionServicesForStores.getMemstoreSize());
109
110    memstore.clearSnapshot(snapshot.getId());
111  }

113  public void testCompaction2Buckets() throws IOException {
114
115    String[] keys1 = { "A", "A", "B", "C" };
116    String[] keys2 = { "A", "B", "D" };
117
118    long totalCellsLen1 = addRowsByKeys(memstore, keys1);
119    long totalHeapOverhead1 = 4
120        * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
121    assertEquals(totalCellsLen1, regionServicesForStores.getMemstoreSize());
122    assertEquals(totalHeapOverhead1, ((CompactingMemStore) memstore).heapOverhead());
123    MemstoreSize size = memstore.getFlushableSize();
124
125    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
126    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
127      Threads.sleep(1000);
128    }
129    int counter = 0;
130    for ( Segment s : memstore.getSegments()) {
131      counter += s.getCellsCount();
132    }
133    assertEquals(3,counter);
134    assertEquals(0, memstore.getSnapshot().getCellsCount());
135    // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
136    // totalCellsLen
137    totalCellsLen1 = (totalCellsLen1 * 3) / 4;
138    totalHeapOverhead1 = 3 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY);
139    assertEquals(totalCellsLen1, regionServicesForStores.getMemstoreSize());
140    assertEquals(totalHeapOverhead1, ((CompactingMemStore) memstore).heapOverhead());
141
142    long totalCellsLen2 = addRowsByKeys(memstore, keys2);
143    long totalHeapOverhead2 = 3
144        * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
145    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemstoreSize());
146    assertEquals(totalHeapOverhead1 + totalHeapOverhead2,
147        ((CompactingMemStore) memstore).heapOverhead());
148

162      counter += s.getCellsCount();

166    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemstoreSize());
167    totalHeapOverhead2 = 1 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY);

169        ((CompactingMemStore) memstore).heapOverhead());
170
171    size = memstore.getFlushableSize();
172    MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
173    region.decrMemstoreSize(size);  // simulate flusher
174    ImmutableSegment s = memstore.getSnapshot();
175    assertEquals(4, s.getCellsCount());
176    assertEquals(0, regionServicesForStores.getMemstoreSize());
177
178    memstore.clearSnapshot(snapshot.getId());

181  public void testCompaction3Buckets() throws IOException {
182
183    String[] keys1 = { "A", "A", "B", "C" };
184    String[] keys2 = { "A", "B", "D" };
185    String[] keys3 = { "D", "B", "B" };
186
187    long totalCellsLen1 = addRowsByKeys(memstore, keys1);
188    long totalHeapOverhead1 = 4
189        * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
190    assertEquals(totalCellsLen1, region.getMemstoreSize());
191    assertEquals(totalHeapOverhead1, ((CompactingMemStore) memstore).heapOverhead());
192
193    MemstoreSize size = memstore.getFlushableSize();
194    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
195
196    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
197      Threads.sleep(10);
198    }
199    assertEquals(0, memstore.getSnapshot().getCellsCount());
200    // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
201    // totalCellsLen
202    totalCellsLen1 = (totalCellsLen1 * 3) / 4;
203    totalHeapOverhead1 = 3 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY);
204    assertEquals(totalCellsLen1, regionServicesForStores.getMemstoreSize());
205    assertEquals(totalHeapOverhead1, ((CompactingMemStore) memstore).heapOverhead());
206
207    long totalCellsLen2 = addRowsByKeys(memstore, keys2);
208    long totalHeapOverhead2 = 3
209        * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
210
211    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemstoreSize());
212    assertEquals(totalHeapOverhead1 + totalHeapOverhead2,
213        ((CompactingMemStore) memstore).heapOverhead());
214
215    ((CompactingMemStore) memstore).disableCompaction();
216    size = memstore.getFlushableSize();
217    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline without compaction
218    assertEquals(0, memstore.getSnapshot().getCellsCount());
219    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemstoreSize());
220    assertEquals(totalHeapOverhead1 + totalHeapOverhead2,
221        ((CompactingMemStore) memstore).heapOverhead());
222
223    long totalCellsLen3 = addRowsByKeys(memstore, keys3);
224    long totalHeapOverhead3 = 3
225        * (KeyValue.FIXED_OVERHEAD + ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY);
226    assertEquals(totalCellsLen1 + totalCellsLen2 + totalCellsLen3,
227        regionServicesForStores.getMemstoreSize());
228    assertEquals(totalHeapOverhead1 + totalHeapOverhead2 + totalHeapOverhead3,
229        ((CompactingMemStore) memstore).heapOverhead());
230
231    ((CompactingMemStore) memstore).enableCompaction();
232    size = memstore.getFlushableSize();
233    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
234    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
235      Threads.sleep(10);
236    }
237    assertEquals(0, memstore.getSnapshot().getCellsCount());
238    // active flushed to pipeline and all 3 segments compacted. Will get rid of duplicated cells.
239    // Out of total 10, only 4 cells are unique
240    totalCellsLen2 = totalCellsLen2 / 3;// 2 out of 3 cells are duplicated
241    totalCellsLen3 = 0;// All duplicated cells.
242    assertEquals(totalCellsLen1 + totalCellsLen2 + totalCellsLen3,
243        regionServicesForStores.getMemstoreSize());
244    // Only 4 unique cells left
245    assertEquals(4 * (KeyValue.FIXED_OVERHEAD + ClassSize.CELL_ARRAY_MAP_ENTRY),
246        ((CompactingMemStore) memstore).heapOverhead());
247
248    size = memstore.getFlushableSize();
249    MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
250    region.decrMemstoreSize(size);  // simulate flusher
251    ImmutableSegment s = memstore.getSnapshot();
252    assertEquals(4, s.getCellsCount());
253    assertEquals(0, regionServicesForStores.getMemstoreSize());
254
255    memstore.clearSnapshot(snapshot.getId());
256
257  }

276      Threads.sleep(10);
277    }
278    assertEquals(0, memstore.getSnapshot().getCellsCount());
279

281
282    int counter2 = 0;
283    for ( Segment s : memstore.getSegments()) {
284      counter2 += s.getCellsCount();
285    }
286    assertEquals(12, counter2);
287

289
290    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline without flattening
291    assertEquals(0, memstore.getSnapshot().getCellsCount());
292
293    int counter3 = 0;
294    for ( Segment s : memstore.getSegments()) {
295      counter3 += s.getCellsCount();
296    }
297    assertEquals(12, counter3);
298
299    addRowsByKeys(memstore, keys3);
300
301    int counter4 = 0;
302    for ( Segment s : memstore.getSegments()) {
303      counter4 += s.getCellsCount();
304    }
305    assertEquals(16, counter4);
306
307    ((CompactingMemStore) memstore).enableCompaction();
308
309
310    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact

315

317    for ( Segment s : memstore.getSegments()) {
318      counter += s.getCellsCount();

320    assertEquals(16,counter);
321

323    ImmutableSegment s = memstore.getSnapshot();
324    memstore.clearSnapshot(snapshot.getId());

328  public void testCountOfCellsAfterFlatteningByScan() throws IOException {
329    String[] keys1 = { "A", "B", "C" }; // A, B, C
330    addRowsByKeysWith50Cols(memstore, keys1);
331    // this should only flatten as there are no duplicates
332    ((CompactingMemStore) memstore).flushInMemory();
333    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
334      Threads.sleep(10);
335    }
336    List<KeyValueScanner> scanners = memstore.getScanners(Long.MAX_VALUE);
337    // seek
338    scanners.get(0).seek(KeyValue.LOWESTKEY);
339    int count = 0;
340    while (scanners.get(0).next() != null) {
341      count++;
342    }
343    assertEquals("the count should be ", count, 150);
344    scanners.get(0).close();
345  }

372  private void addRowsByKeysWith50Cols(AbstractMemStore hmc, String[] keys) {
373    byte[] fam = Bytes.toBytes("testfamily");
374    for (int i = 0; i < keys.length; i++) {
375      long timestamp = System.currentTimeMillis();
376      Threads.sleep(1); // to make sure each kv gets a different ts
377      byte[] row = Bytes.toBytes(keys[i]);
378      for(int  j =0 ;j < 50; j++) {
379        byte[] qf = Bytes.toBytes("testqualifier"+j);
380        byte[] val = Bytes.toBytes(keys[i] + j);
381        KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
382        hmc.add(kv, null);
383      }
384    }
385  }

387  private long addRowsByKeys(final AbstractMemStore hmc, String[] keys) {
388    byte[] fam = Bytes.toBytes("testfamily");
389    byte[] qf = Bytes.toBytes("testqualifier");
390    MemstoreSize memstoreSize = new MemstoreSize();
391    for (int i = 0; i < keys.length; i++) {
392      long timestamp = System.currentTimeMillis();
393      Threads.sleep(1); // to make sure each kv gets a different ts
394      byte[] row = Bytes.toBytes(keys[i]);
395      byte[] val = Bytes.toBytes(keys[i] + i);
396      KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
397      hmc.add(kv, memstoreSize);
398      LOG.debug("added kv: " + kv.getKeyString() + ", timestamp" + kv.getTimestamp());
399    }
400    regionServicesForStores.addMemstoreSize(memstoreSize);
401    return memstoreSize.getDataSize();
402  }
