055/**
056 * This test verifies the correctness of the Per Column Family flushing strategy
057 * when part of the memstores are compacted memstores
058 */
059@Category({ RegionServerTests.class, LargeTests.class })
060public class TestWalAndCompactingMemStoreFlush {
061
062  private static final Log LOG = LogFactory.getLog(TestWalAndCompactingMemStoreFlush.class);
063  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
064  private static final Path DIR = TEST_UTIL.getDataTestDir("TestHRegion");
065  public static final TableName TABLENAME = TableName.valueOf("TestWalAndCompactingMemStoreFlush",
066      "t1");
067
068  public static final byte[][] FAMILIES = { Bytes.toBytes("f1"), Bytes.toBytes("f2"),
069      Bytes.toBytes("f3"), Bytes.toBytes("f4"), Bytes.toBytes("f5") };
070
071  public static final byte[] FAMILY1 = FAMILIES[0];
072  public static final byte[] FAMILY2 = FAMILIES[1];
073  public static final byte[] FAMILY3 = FAMILIES[2];
074
075  private HRegion initHRegion(String callingMethod, Configuration conf) throws IOException {
076    int i=0;
077    HTableDescriptor htd = new HTableDescriptor(TABLENAME);
078    for (byte[] family : FAMILIES) {
079      HColumnDescriptor hcd = new HColumnDescriptor(family);
080      // even column families are going to have compacted memstore
081      if(i%2 == 0) hcd.setInMemoryCompaction(true);
082      htd.addFamily(hcd);
083      i++;
084    }
085
086    HRegionInfo info = new HRegionInfo(TABLENAME, null, null, false);
087    Path path = new Path(DIR, callingMethod);
088    return HBaseTestingUtility.createRegionAndWAL(info, path, conf, htd);
089  }
090
091  // A helper function to create puts.
092  private Put createPut(int familyNum, int putNum) {
093    byte[] qf  = Bytes.toBytes("q" + familyNum);
094    byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
095    byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
096    Put p = new Put(row);
097    p.addColumn(FAMILIES[familyNum - 1], qf, val);
098    return p;
099  }
100
101  // A helper function to create double puts, so something can be compacted later.
102  private Put createDoublePut(int familyNum, int putNum) {
103    byte[] qf  = Bytes.toBytes("q" + familyNum);
104    byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
105    byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
106    Put p = new Put(row);
107    // add twice with different timestamps
108    p.addColumn(FAMILIES[familyNum - 1], qf, 10, val);
109    p.addColumn(FAMILIES[familyNum - 1], qf, 20, val);
110    return p;
111  }
112
113  // A helper function to create gets.
114  private Get createGet(int familyNum, int putNum) {
115    byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
116    return new Get(row);
117  }
118
119  // A helper function to verify edits.
120  void verifyEdit(int familyNum, int putNum, Table table) throws IOException {
121    Result r = table.get(createGet(familyNum, putNum));
122    byte[] family = FAMILIES[familyNum - 1];
123    byte[] qf = Bytes.toBytes("q" + familyNum);
124    byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
125    assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum), r.getFamilyMap(family));
126    assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum),
127      r.getFamilyMap(family).get(qf));
128    assertTrue(("Incorrect value for Put#" + putNum + " for CF# " + familyNum),
129      Arrays.equals(r.getFamilyMap(family).get(qf), val));
130  }

132  @Test(timeout = 180000)
133  public void testSelectiveFlushWhenEnabled() throws IOException {
134
135    // Set up the configuration
136    Configuration conf = HBaseConfiguration.create();
137    conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 600 * 1024);
138    conf.set(FlushPolicyFactory.HBASE_FLUSH_POLICY_KEY, FlushNonSloppyStoresFirstPolicy.class
139        .getName());
140    conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 200 *
141        1024);
142    conf.setDouble(CompactingMemStore.IN_MEMORY_FLUSH_THRESHOLD_FACTOR_KEY, 0.5);
143
144    // Intialize the region
145    Region region = initHRegion("testSelectiveFlushWhenEnabled", conf);
146
147    // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
148    for (int i = 1; i <= 1200; i++) {
149      region.put(createPut(1, i));        // compacted memstore
150
151      if (i <= 100) {
152        region.put(createPut(2, i));
153        if (i <= 50) {
154          region.put(createDoublePut(3, i)); // subject for in-memory compaction
155        }
156      }
157    }
158
159    // Now add more puts for CF2, so that we only flush CF2 to disk
160    for (int i = 100; i < 2000; i++) {
161      region.put(createPut(2, i));
162    }
163
164    long totalMemstoreSize = region.getMemstoreSize();
165
166    // Find the smallest LSNs for edits wrt to each CF.
167    long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
168    long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
169    long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
170
171    // Find the sizes of the memstores of each CF.
172    long cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getMemStoreSize();
173    long cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getMemStoreSize();
174    long cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getMemStoreSize();
175
176    // Get the overall smallest LSN in the region's memstores.
177    long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
178        .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
179
180    String s = "\n\n----------------------------------\n"
181        + "Upon initial insert and before any flush, size of CF1 is:"
182        + cf1MemstoreSizePhaseI + ", is CF1 compacted memstore?:"
183        + region.getStore(FAMILY1).getMemStore().isSloppy() + ". Size of CF2 is:"
184        + cf2MemstoreSizePhaseI + ", is CF2 compacted memstore?:"
185        + region.getStore(FAMILY2).getMemStore().isSloppy() + ". Size of CF3 is:"
186        + cf3MemstoreSizePhaseI + ", is CF3 compacted memstore?:"
187        + region.getStore(FAMILY3).getMemStore().isSloppy() + "\n";
188
189    // The overall smallest LSN in the region's memstores should be the same as
190    // the LSN of the smallest edit in CF1
191    assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
192
193    // Some other sanity checks.
194    assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
195    assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
196    assertTrue(cf1MemstoreSizePhaseI > 0);
197    assertTrue(cf2MemstoreSizePhaseI > 0);
198    assertTrue(cf3MemstoreSizePhaseI > 0);
199
200    // The total memstore size should be the same as the sum of the sizes of
201    // memstores of CF1, CF2 and CF3.
202    String msg = "totalMemstoreSize="+totalMemstoreSize +
203        " DefaultMemStore.DEEP_OVERHEAD="+DefaultMemStore.DEEP_OVERHEAD +
204        " DEEP_OVERHEAD_PER_PIPELINE_ITEM="+CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM +
205        " cf1MemstoreSizePhaseI="+cf1MemstoreSizePhaseI +
206        " cf2MemstoreSizePhaseI="+cf2MemstoreSizePhaseI +
207        " cf3MemstoreSizePhaseI="+cf3MemstoreSizePhaseI ;
208    assertEquals(msg,totalMemstoreSize + 3 * DefaultMemStore.DEEP_OVERHEAD,
209        cf1MemstoreSizePhaseI + cf2MemstoreSizePhaseI + cf3MemstoreSizePhaseI);
210
211    // Flush!!!!!!!!!!!!!!!!!!!!!!
212    // We have big compacting memstore CF1 and two small memstores:
213    // CF2 (not compacted) and CF3 (compacting)
214    // All together they are above the flush size lower bound.
215    // Since CF1 and CF3 should be flushed to memory (not to disk),
216    // CF2 is going to be flushed to disk.
217    // CF1 - nothing to compact, CF3 - should be twice compacted
218    ((CompactingMemStore) region.getStore(FAMILY1).getMemStore()).flushInMemory();
219    ((CompactingMemStore) region.getStore(FAMILY3).getMemStore()).flushInMemory();
220    region.flush(false);
221
222    // CF3 should be compacted so wait here to be sure the compaction is done
223    while (((CompactingMemStore) region.getStore(FAMILY3).getMemStore())
224        .isMemStoreFlushingInMemory())
225      Threads.sleep(10);
226
227    // Recalculate everything
228    long cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getMemStoreSize();
229    long cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getMemStoreSize();
230    long cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getMemStoreSize();
231
232    long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
233        .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
234    // Find the smallest LSNs for edits wrt to each CF.
235    long smallestSeqCF1PhaseII = region.getOldestSeqIdOfStore(FAMILY1);
236    long smallestSeqCF2PhaseII = region.getOldestSeqIdOfStore(FAMILY2);
237    long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
238
239    s = s + "DefaultMemStore DEEP_OVERHEAD is:" + DefaultMemStore.DEEP_OVERHEAD
240        + ", CompactingMemStore DEEP_OVERHEAD is:" + CompactingMemStore.DEEP_OVERHEAD
241        + ", CompactingMemStore DEEP_OVERHEAD_PER_PIPELINE_ITEM is:" + CompactingMemStore
242        .DEEP_OVERHEAD_PER_PIPELINE_ITEM
243        + "\n----After first flush! CF1 should be flushed to memory, but not compacted.---\n"
244        + "Size of CF1 is:" + cf1MemstoreSizePhaseII + ", size of CF2 is:" + cf2MemstoreSizePhaseII
245        + ", size of CF3 is:" + cf3MemstoreSizePhaseII + "\n";
246
247    // CF1 was flushed to memory, but there is nothing to compact, should
248    // remain the same size plus renewed empty skip-list
249    assertEquals(s, cf1MemstoreSizePhaseII,
250        cf1MemstoreSizePhaseI + CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM);
251
252    // CF2 should become empty
253    assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseII);
254
255    // verify that CF3 was flushed to memory and was compacted (this is approximation check)
256    assertTrue(cf3MemstoreSizePhaseI/2+DefaultMemStore.DEEP_OVERHEAD +
257        CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM >
258        cf3MemstoreSizePhaseII);
259    assertTrue(cf3MemstoreSizePhaseI/2 < cf3MemstoreSizePhaseII);
260
261
262    // Now the smallest LSN in the region should be the same as the smallest
263    // LSN in the memstore of CF1.
264    assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
265
266    // Now add more puts for CF1, so that we also flush CF1 to disk instead of
267    // memory in next flush
268    for (int i = 1200; i < 2000; i++) {
269      region.put(createPut(1, i));
270    }
271
272    s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseII
273        + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseII + ", " +
274        "the smallest sequence in CF2:"
275        + smallestSeqCF2PhaseII +", the smallest sequence in CF3:" + smallestSeqCF3PhaseII + "\n";
276
277    // How much does the CF1 memstore occupy? Will be used later.
278    long cf1MemstoreSizePhaseIII = region.getStore(FAMILY1).getMemStoreSize();
279    long smallestSeqCF1PhaseIII = region.getOldestSeqIdOfStore(FAMILY1);
280
281    s = s + "----After more puts into CF1 its size is:" + cf1MemstoreSizePhaseIII
282        + ", and its sequence is:" + smallestSeqCF1PhaseIII + " ----\n" ;
283
284
285    // Flush!!!!!!!!!!!!!!!!!!!!!!
286    // Flush again, CF1 is flushed to disk
287    // CF2 is flushed to disk, because it is not in-memory compacted memstore
288    // CF3 is flushed empty to memory (actually nothing happens to CF3)
289    region.flush(false);
290
291    // Recalculate everything
292    long cf1MemstoreSizePhaseIV = region.getStore(FAMILY1).getMemStoreSize();
293    long cf2MemstoreSizePhaseIV = region.getStore(FAMILY2).getMemStoreSize();
294    long cf3MemstoreSizePhaseIV = region.getStore(FAMILY3).getMemStoreSize();
295
296    long smallestSeqInRegionCurrentMemstorePhaseIV = getWAL(region)
297        .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
298    long smallestSeqCF1PhaseIV = region.getOldestSeqIdOfStore(FAMILY1);
299    long smallestSeqCF2PhaseIV = region.getOldestSeqIdOfStore(FAMILY2);
300    long smallestSeqCF3PhaseIV = region.getOldestSeqIdOfStore(FAMILY3);
301
302    s = s + "----After SECOND FLUSH, CF1 size is:" + cf1MemstoreSizePhaseIV + ", CF2 size is:"
303        + cf2MemstoreSizePhaseIV + " and CF3 size is:" + cf3MemstoreSizePhaseIV
304        + "\n";
305
306    s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseIV
307        + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseIV + ", " +
308        "the smallest sequence in CF2:"
309        + smallestSeqCF2PhaseIV +", the smallest sequence in CF3:" + smallestSeqCF3PhaseIV
310        + "\n";
311
312    // CF1's pipeline component (inserted before first flush) should be flushed to disk
313    // CF2 should be flushed to disk
314    assertEquals(cf1MemstoreSizePhaseIII - cf1MemstoreSizePhaseI + DefaultMemStore.DEEP_OVERHEAD,
315        cf1MemstoreSizePhaseIV);
316    assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseIV);
317
318    // CF3 shouldn't have been touched.
319    assertEquals(cf3MemstoreSizePhaseIV, cf3MemstoreSizePhaseII);
320
321    // the smallest LSN of CF3 shouldn't change
322    assertEquals(smallestSeqCF3PhaseII, smallestSeqCF3PhaseIV);
323
324    // CF3 should be bottleneck for WAL
325     assertEquals(s, smallestSeqInRegionCurrentMemstorePhaseIV, smallestSeqCF3PhaseIV);
326
327    // Flush!!!!!!!!!!!!!!!!!!!!!!
328    // Clearing the existing memstores, CF2 all flushed to disk. The single
329    // memstore segment in the compaction pipeline of CF1 and CF3 should be flushed to disk.
330    // Note that active sets of CF1 and CF3 are empty
331    region.flush(true);
332
333    // Recalculate everything
334    long cf1MemstoreSizePhaseV = region.getStore(FAMILY1).getMemStoreSize();
335    long cf2MemstoreSizePhaseV = region.getStore(FAMILY2).getMemStoreSize();
336    long cf3MemstoreSizePhaseV = region.getStore(FAMILY3).getMemStoreSize();
337    long smallestSeqInRegionCurrentMemstorePhaseV = getWAL(region)
338        .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
339
340    assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf1MemstoreSizePhaseV);
341    assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseV);
342    assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf3MemstoreSizePhaseV);
343
344    // Because there is nothing in any memstore the WAL's LSN should be -1
345    assertEquals(smallestSeqInRegionCurrentMemstorePhaseV, HConstants.NO_SEQNUM);
346
347    // What happens when we hit the memstore limit, but we are not able to find
348    // any Column Family above the threshold?
349    // In that case, we should flush all the CFs.
350
351    // The memstore limit is 200*1024 and the column family flush threshold is
352    // around 50*1024. We try to just hit the memstore limit with each CF's
353    // memstore being below the CF flush threshold.
354    for (int i = 1; i <= 300; i++) {
355      region.put(createPut(1, i));
356      region.put(createPut(2, i));
357      region.put(createPut(3, i));
358      region.put(createPut(4, i));
359      region.put(createPut(5, i));
360    }
361
362    region.flush(false);
363
364    s = s + "----AFTER THIRD AND FORTH FLUSH, The smallest sequence in region WAL is: "
365        + smallestSeqInRegionCurrentMemstorePhaseV
366        + ". After additional inserts and last flush, the entire region size is:" + region
367        .getMemstoreSize()
368        + "\n----------------------------------\n";
369
370    // Since we won't find any CF above the threshold, and hence no specific
371    // store to flush, we should flush all the memstores
372    // Also compacted memstores are flushed to disk.
373    assertEquals(0, region.getMemstoreSize());
374    System.out.println(s);
375    HBaseTestingUtility.closeRegionAndWAL(region);
376  }

498        region.getWAL().getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
499    long smallestSeqCF1PhaseIV = region.getOldestSeqIdOfStore(FAMILY1);
500    long smallestSeqCF2PhaseIV = region.getOldestSeqIdOfStore(FAMILY2);
501    long smallestSeqCF3PhaseIV = region.getOldestSeqIdOfStore(FAMILY3);
502
503    s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseIV
504        + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseIV + ", " +
505        "the smallest sequence in CF2:"
506        + smallestSeqCF2PhaseIV +", the smallest sequence in CF3:" + smallestSeqCF3PhaseIV + "\n";
507
508    // now check that the LSN of the entire WAL, of CF1 and of CF3 has progressed due to compaction
509    assertTrue(s, smallestSeqInRegionCurrentMemstorePhaseIV >
510        smallestSeqInRegionCurrentMemstorePhaseIII);
511    assertTrue(smallestSeqCF1PhaseIV > smallestSeqCF1PhaseIII);
512    assertTrue(smallestSeqCF3PhaseIV > smallestSeqCF3PhaseIII);
513
514    HBaseTestingUtility.closeRegionAndWAL(region);
515  }
