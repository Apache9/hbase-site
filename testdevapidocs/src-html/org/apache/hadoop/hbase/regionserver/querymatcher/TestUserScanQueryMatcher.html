047  /**
048   * This is a cryptic test. It is checking that we don't include a fake cell, one that has a
049   * timestamp of {@link HConstants#OLDEST_TIMESTAMP}. See HBASE-16074 for background.
050   * @throws IOException
051   */
052  @Test
053  public void testNeverIncludeFakeCell() throws IOException {
054    long now = EnvironmentEdgeManager.currentTime();
055    // Do with fam2 which has a col2 qualifier.
056    UserScanQueryMatcher qm = UserScanQueryMatcher.create(scan,
057      new ScanInfo(this.conf, fam2, 10, 1, ttl, KeepDeletedCells.FALSE,
058          HConstants.DEFAULT_BLOCKSIZE, 0, rowComparator),
059      get.getFamilyMap().get(fam2), now - ttl, now, null);
060    Cell kv = new KeyValue(row1, fam2, col2, 1, data);
061    Cell cell = CellUtil.createLastOnRowCol(kv);
062    qm.setToNewRow(kv);
063    MatchCode code = qm.match(cell);
064    assertFalse(code.compareTo(MatchCode.SEEK_NEXT_COL) != 0);
065  }

067  @Test
068  public void testMatchExplicitColumns() throws IOException {
069    // Moving up from the Tracker by using Gets and List<KeyValue> instead
070    // of just byte []
071
072    // Expected result
073    List<MatchCode> expected = new ArrayList<>(6);
074    expected.add(ScanQueryMatcher.MatchCode.SEEK_NEXT_COL);
075    expected.add(ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_COL);
076    expected.add(ScanQueryMatcher.MatchCode.SEEK_NEXT_COL);
077    expected.add(ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_COL);
078    expected.add(ScanQueryMatcher.MatchCode.INCLUDE_AND_SEEK_NEXT_ROW);
079    expected.add(ScanQueryMatcher.MatchCode.DONE);
080
081    long now = EnvironmentEdgeManager.currentTime();
082    // 2,4,5
083    UserScanQueryMatcher qm = UserScanQueryMatcher.create(
084      scan, new ScanInfo(this.conf, fam2, 0, 1, ttl, KeepDeletedCells.FALSE,
085          HConstants.DEFAULT_BLOCKSIZE, 0, rowComparator),
086      get.getFamilyMap().get(fam2), now - ttl, now, null);
087
088    List<KeyValue> memstore = new ArrayList<>(6);
089    memstore.add(new KeyValue(row1, fam2, col1, 1, data));
090    memstore.add(new KeyValue(row1, fam2, col2, 1, data));
091    memstore.add(new KeyValue(row1, fam2, col3, 1, data));
092    memstore.add(new KeyValue(row1, fam2, col4, 1, data));
093    memstore.add(new KeyValue(row1, fam2, col5, 1, data));
094
095    memstore.add(new KeyValue(row2, fam1, col1, data));
096
097    List<ScanQueryMatcher.MatchCode> actual = new ArrayList<>(memstore.size());
098    KeyValue k = memstore.get(0);
099    qm.setToNewRow(k);
100
101    for (KeyValue kv : memstore) {
102      actual.add(qm.match(kv));
103    }
104
105    assertEquals(expected.size(), actual.size());
106    for (int i = 0; i < expected.size(); i++) {
107      LOG.debug("expected " + expected.get(i) + ", actual " + actual.get(i));
108      assertEquals(expected.get(i), actual.get(i));
109    }
110  }
