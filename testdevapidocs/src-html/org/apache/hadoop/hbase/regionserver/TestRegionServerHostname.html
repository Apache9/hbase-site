018package org.apache.hadoop.hbase.regionserver;
019
020import static org.junit.Assert.assertTrue;
021
022import java.io.IOException;
023import java.net.InetAddress;
024import java.net.NetworkInterface;
025import java.util.Enumeration;
026import java.util.List;
027import java.util.Locale;
028
029import org.apache.commons.logging.Log;
030import org.apache.commons.logging.LogFactory;
031import org.apache.hadoop.hbase.HBaseTestingUtility;
032import org.apache.hadoop.hbase.testclassification.MediumTests;
033import org.apache.hadoop.hbase.testclassification.RegionServerTests;
034import org.apache.hadoop.hbase.zookeeper.ZKUtil;
035import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
036import org.junit.Test;
037import org.junit.experimental.categories.Category;
038
039/**
040 * Tests for the hostname specification by region server
041 */
042@Category({RegionServerTests.class, MediumTests.class})
043public class TestRegionServerHostname {
044  private static final Log LOG = LogFactory.getLog(TestRegionServerHostname.class);
045  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
046
047  @Test (timeout=30000)
048  public void testInvalidRegionServerHostnameAbortsServer() throws Exception {
049    final int NUM_MASTERS = 1;
050    final int NUM_RS = 1;
051    String invalidHostname = "hostAddr.invalid";
052    TEST_UTIL.getConfiguration().set(HRegionServer.RS_HOSTNAME_KEY, invalidHostname);
053    try {
054      TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
055    } catch (IOException ioe) {
056      Throwable t1 = ioe.getCause();
057      Throwable t2 = t1.getCause();
058      assertTrue(t1.getMessage() + " - " + t2.getMessage(),
059        t2.getMessage().contains("Failed resolve of " + invalidHostname) ||
060        t2.getMessage().contains("Problem binding to " + invalidHostname));
061      return;
062    } finally {
063      TEST_UTIL.shutdownMiniCluster();
064    }
065    assertTrue("Failed to validate against invalid hostname", false);
066  }
067
068  @Test(timeout=120000)
069  public void testRegionServerHostname() throws Exception {
070    final int NUM_MASTERS = 1;
071    final int NUM_RS = 1;
072    Enumeration<NetworkInterface> netInterfaceList = NetworkInterface.getNetworkInterfaces();
073
074    while (netInterfaceList.hasMoreElements()) {
075      NetworkInterface ni = netInterfaceList.nextElement();
076      Enumeration<InetAddress> addrList = ni.getInetAddresses();
077      // iterate through host addresses and use each as hostname
078      while (addrList.hasMoreElements()) {
079        InetAddress addr = addrList.nextElement();
080        if (addr.isLoopbackAddress() || addr.isLinkLocalAddress() || addr.isMulticastAddress()) {
081          continue;
082        }
083        String hostName = addr.getHostName();
084        LOG.info("Found " + hostName + " on " + ni);
085        
086        TEST_UTIL.getConfiguration().set(HRegionServer.MASTER_HOSTNAME_KEY, hostName);
087        TEST_UTIL.getConfiguration().set(HRegionServer.RS_HOSTNAME_KEY, hostName);
088        TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
089        try {
090          ZooKeeperWatcher zkw = TEST_UTIL.getZooKeeperWatcher();
091          List<String> servers = ZKUtil.listChildrenNoWatch(zkw, zkw.rsZNode);
092          // there would be NUM_RS+1 children - one for the master
093          assertTrue(servers.size() == NUM_RS+1);
094          for (String server : servers) {
095            assertTrue("From zookeeper: " + server + " hostname: " + hostName,
096              server.startsWith(hostName.toLowerCase(Locale.ROOT)+","));
097          }
098          zkw.close();
099        } finally {
100          TEST_UTIL.shutdownMiniCluster();
101        }
102      }
103    }
104  }
105}
