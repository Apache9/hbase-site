052/**
053 * Tests to read old ROOT, Meta edits.
054 */
055@Category({RegionServerTests.class, MediumTests.class})
056
057public class TestReadOldRootAndMetaEdits {
058
059  private final static Log LOG = LogFactory.getLog(TestReadOldRootAndMetaEdits.class);
060  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
061  private static Configuration conf;
062  private static FileSystem fs;
063  private static Path dir;
064
065  @BeforeClass
066  public static void setupBeforeClass() throws Exception {
067    conf = TEST_UTIL.getConfiguration();
068    conf.setClass("hbase.regionserver.hlog.writer.impl",
069      SequenceFileLogWriter.class, WALProvider.Writer.class);
070    fs = TEST_UTIL.getTestFileSystem();
071    dir = new Path(TEST_UTIL.createRootDir(), "testReadOldRootAndMetaEdits");
072    fs.mkdirs(dir);
073
074  }
075  @AfterClass
076  public static void tearDownAfterClass() throws Exception {
077  }
078
079  /**
080   * Inserts three waledits in the wal file, and reads them back. The first edit is of a regular
081   * table, second waledit is for the ROOT table (it will be ignored while reading),
082   * and last waledit is for the hbase:meta table, which will be linked to the new system:meta table.
083   * @throws IOException
084   */
085  @Test
086  public void testReadOldRootAndMetaEdits() throws IOException {
087    LOG.debug("testReadOldRootAndMetaEdits");
088    // kv list to be used for all WALEdits.
089    byte[] row = Bytes.toBytes("row");
090    KeyValue kv = new KeyValue(row, row, row, row);
091    List<KeyValue> kvs = new ArrayList<KeyValue>();
092    kvs.add(kv);
093
094    WALProvider.Writer writer = null;
095    WAL.Reader reader = null;
096    // a regular table
097    TableName t = TableName.valueOf("t");
098    HRegionInfo tRegionInfo = null;
099    int logCount = 0;
100    long timestamp = System.currentTimeMillis();
101    Path path = new Path(dir, "t");
102    try {
103      tRegionInfo = new HRegionInfo(t, HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
104      WAL.Entry tEntry = createAEntry(new HLogKey(tRegionInfo.getEncodedNameAsBytes(), t,
105          ++logCount, timestamp, HConstants.DEFAULT_CLUSTER_ID), kvs);
106
107      // create a old root edit (-ROOT-).
108      WAL.Entry rootEntry = createAEntry(new HLogKey(Bytes.toBytes(TableName.OLD_ROOT_STR),
109          TableName.OLD_ROOT_TABLE_NAME, ++logCount, timestamp,
110          HConstants.DEFAULT_CLUSTER_ID), kvs);
111
112      // create a old meta edit (hbase:meta).
113      WAL.Entry oldMetaEntry = createAEntry(new HLogKey(Bytes.toBytes(TableName.OLD_META_STR),
114          TableName.OLD_META_TABLE_NAME, ++logCount, timestamp,
115          HConstants.DEFAULT_CLUSTER_ID), kvs);
116
117      // write above entries
118      writer = WALFactory.createWALWriter(fs, path, conf);
119      writer.append(tEntry);
120      writer.append(rootEntry);
121      writer.append(oldMetaEntry);
122
123      // sync/close the writer
124      writer.sync();
125      writer.close();
126
127      // read the log and see things are okay.
128      reader = WALFactory.createReader(fs, path, conf);
129      WAL.Entry entry = reader.next();
130      assertNotNull(entry);
131      assertTrue(entry.getKey().getTablename().equals(t));
132      assertEquals(Bytes.toString(entry.getKey().getEncodedRegionName()),
133        Bytes.toString(tRegionInfo.getEncodedNameAsBytes()));
134
135      // read the ROOT waledit, but that will be ignored, and hbase:meta waledit will be read instead.
136      entry = reader.next();
137      assertEquals(entry.getKey().getTablename(), TableName.META_TABLE_NAME);
138      // should reach end of log
139      assertNull(reader.next());
140    } finally {
141      if (writer != null) {
142        writer.close();
143      }
144      if (reader != null) {
145        reader.close();
146      }
147    }
148}
149  /**
150   * Creates a WALEdit for the passed KeyValues and returns a WALProvider.Entry instance composed of
151   * the WALEdit and passed WALKey.
152   * @return WAL.Entry instance for the passed WALKey and KeyValues
153   */
154  private WAL.Entry createAEntry(WALKey walKey, List<KeyValue> kvs) {
155    WALEdit edit = new WALEdit();
156    for (KeyValue kv : kvs )
157    edit.add(kv);
158    return new WAL.Entry(walKey, edit);
159  }
