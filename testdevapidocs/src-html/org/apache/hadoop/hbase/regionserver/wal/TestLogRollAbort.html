062/**
063 * Tests for conditions that should trigger RegionServer aborts when
064 * rolling the current WAL fails.
065 */
066@Category({RegionServerTests.class, MediumTests.class})
067public class TestLogRollAbort {
068  private static final Log LOG = LogFactory.getLog(TestLogRolling.class);
069  private static MiniDFSCluster dfsCluster;
070  private static Admin admin;
071  private static MiniHBaseCluster cluster;
072  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
073
074  /* For the split-then-roll test */
075  private static final Path HBASEDIR = new Path("/hbase");
076  private static final Path OLDLOGDIR = new Path(HBASEDIR, HConstants.HREGION_OLDLOGDIR_NAME);
077
078  // Need to override this setup so we can edit the config before it gets sent
079  // to the HDFS & HBase cluster startup.
080  @BeforeClass
081  public static void setUpBeforeClass() throws Exception {
082    // Tweak default timeout values down for faster recovery
083    TEST_UTIL.getConfiguration().setInt(
084        "hbase.regionserver.logroll.errors.tolerated", 2);
085    TEST_UTIL.getConfiguration().setInt("hbase.rpc.timeout", 10 * 1000);
086
087    // Increase the amount of time between client retries
088    TEST_UTIL.getConfiguration().setLong("hbase.client.pause", 5 * 1000);
089
090    // make sure log.hflush() calls syncFs() to open a pipeline
091    TEST_UTIL.getConfiguration().setBoolean("dfs.support.append", true);
092    // lower the namenode & datanode heartbeat so the namenode
093    // quickly detects datanode failures
094    TEST_UTIL.getConfiguration().setInt("dfs.namenode.heartbeat.recheck-interval", 5000);
095    TEST_UTIL.getConfiguration().setInt("dfs.heartbeat.interval", 1);
096    // the namenode might still try to choose the recently-dead datanode
097    // for a pipeline, so try to a new pipeline multiple times
098    TEST_UTIL.getConfiguration().setInt("dfs.client.block.write.retries", 10);
099  }

102  private FileSystem fs;

124  /**
125   * Tests that RegionServer aborts if we hit an error closing the WAL when
126   * there are unsynced WAL edits.  See HBASE-4282.
127   */
128  @Test
129  public void testRSAbortWithUnflushedEdits() throws Exception {
130    LOG.info("Starting testRSAbortWithUnflushedEdits()");
131
132    // When the hbase:meta table can be opened, the region servers are running
133    TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME).close();
134
135    // Create the test table and open it
136    TableName tableName = TableName.valueOf(this.getClass().getSimpleName());
137    HTableDescriptor desc = new HTableDescriptor(tableName);
138    desc.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
139
140    admin.createTable(desc);
141    Table table = TEST_UTIL.getConnection().getTable(desc.getTableName());
142    try {
143      HRegionServer server = TEST_UTIL.getRSForFirstRegionInTable(tableName);
144      WAL log = server.getWAL(null);
145
146      // don't run this test without append support (HDFS-200 & HDFS-142)
147      assertTrue("Need append support for this test",
148        FSUtils.isAppendSupported(TEST_UTIL.getConfiguration()));
149
150      Put p = new Put(Bytes.toBytes("row2001"));
151      p.addColumn(HConstants.CATALOG_FAMILY, Bytes.toBytes("col"), Bytes.toBytes(2001));
152      table.put(p);
153
154      log.sync();
155
156      p = new Put(Bytes.toBytes("row2002"));
157      p.addColumn(HConstants.CATALOG_FAMILY, Bytes.toBytes("col"), Bytes.toBytes(2002));
158      table.put(p);
159
160      dfsCluster.restartDataNodes();
161      LOG.info("Restarted datanodes");
162
163      try {
164        log.rollWriter(true);
165      } catch (FailedLogCloseException flce) {
166        // Expected exception.  We used to expect that there would be unsynced appends but this
167        // not reliable now that sync plays a roll in wall rolling.  The above puts also now call
168        // sync.
169      } catch (Throwable t) {
170        LOG.fatal("FAILED TEST: Got wrong exception", t);
171      }
172    } finally {
173      table.close();
174    }
175  }

177  /**
178   * Tests the case where a RegionServer enters a GC pause,
179   * comes back online after the master declared it dead and started to split.
180   * Want log rolling after a master split to fail. See HBASE-2312.
181   */
182  @Test (timeout=300000)
183  public void testLogRollAfterSplitStart() throws IOException {
184    LOG.info("Verify wal roll after split starts will fail.");
185    String logName = "testLogRollAfterSplitStart";
186    Path thisTestsDir = new Path(HBASEDIR, DefaultWALProvider.getWALDirectoryName(logName));
187    final WALFactory wals = new WALFactory(conf, null, logName);
188
189    try {
190      // put some entries in an WAL
191      TableName tableName =
192          TableName.valueOf(this.getClass().getName());
193      HRegionInfo regioninfo = new HRegionInfo(tableName,
194          HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
195      final WAL log = wals.getWAL(regioninfo.getEncodedNameAsBytes(),
196          regioninfo.getTable().getNamespace());
197      MultiVersionConcurrencyControl mvcc = new MultiVersionConcurrencyControl(1);
198
199      final int total = 20;
200      for (int i = 0; i < total; i++) {
201        WALEdit kvs = new WALEdit();
202        kvs.add(new KeyValue(Bytes.toBytes(i), tableName.getName(), tableName.getName()));
203        HTableDescriptor htd = new HTableDescriptor(tableName);
204        htd.addFamily(new HColumnDescriptor("column"));
205        NavigableMap<byte[], Integer> scopes = new TreeMap<byte[], Integer>(
206            Bytes.BYTES_COMPARATOR);
207        for(byte[] fam : htd.getFamiliesKeys()) {
208          scopes.put(fam, 0);
209        }
210        log.append(regioninfo, new WALKey(regioninfo.getEncodedNameAsBytes(), tableName,
211            System.currentTimeMillis(), mvcc, scopes), kvs, true);
212      }
213      // Send the data to HDFS datanodes and close the HDFS writer
214      log.sync();
215      ((FSHLog) log).replaceWriter(((FSHLog)log).getOldPath(), null, null, null);
216
217      /* code taken from MasterFileSystem.getLogDirs(), which is called from MasterFileSystem.splitLog()
218       * handles RS shutdowns (as observed by the splitting process)
219       */
220      // rename the directory so a rogue RS doesn't create more WALs
221      Path rsSplitDir = thisTestsDir.suffix(DefaultWALProvider.SPLITTING_EXT);
222      if (!fs.rename(thisTestsDir, rsSplitDir)) {
223        throw new IOException("Failed fs.rename for log split: " + thisTestsDir);
224      }
225      LOG.debug("Renamed region directory: " + rsSplitDir);
226
227      LOG.debug("Processing the old log files.");
228      WALSplitter.split(HBASEDIR, rsSplitDir, OLDLOGDIR, fs, conf, wals);
229
230      LOG.debug("Trying to roll the WAL.");
231      try {
232        log.rollWriter();
233        Assert.fail("rollWriter() did not throw any exception.");
234      } catch (IOException ioe) {
235        if (ioe.getCause() instanceof FileNotFoundException) {
236          LOG.info("Got the expected exception: ", ioe.getCause());
237        } else {
238          Assert.fail("Unexpected exception: " + ioe);
239        }
240      }
241    } finally {
242      wals.close();
243      if (fs.exists(thisTestsDir)) {
244        fs.delete(thisTestsDir, true);
245      }
246    }
247  }
