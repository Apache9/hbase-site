022import java.io.IOException;
023import java.lang.reflect.Field;
024import java.lang.reflect.InvocationTargetException;
025import java.util.TreeMap;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.apache.hadoop.hbase.classification.InterfaceAudience;
030import org.apache.hadoop.hbase.io.util.LRUDictionary;
031import org.apache.hadoop.conf.Configuration;
032import org.apache.hadoop.fs.FSDataOutputStream;
033import org.apache.hadoop.fs.FileSystem;
034import org.apache.hadoop.fs.Path;
035import org.apache.hadoop.hbase.HBaseInterfaceAudience;
036import org.apache.hadoop.hbase.HConstants;
037import org.apache.hadoop.hbase.util.FSUtils;
038import org.apache.hadoop.hbase.wal.DefaultWALProvider;
039import org.apache.hadoop.hbase.wal.WAL;
040import org.apache.hadoop.io.SequenceFile;
041import org.apache.hadoop.io.SequenceFile.CompressionType;
042import org.apache.hadoop.io.SequenceFile.Metadata;
043import org.apache.hadoop.io.Text;
044import org.apache.hadoop.io.compress.CompressionCodec;
045import org.apache.hadoop.io.compress.DefaultCodec;

047/**
048 * Implementation of {@link WALProvider.Writer} that delegates to
049 * SequenceFile.Writer. Legacy implementation only used for compat tests.
050 *
051 * Note that because this class writes to the legacy hadoop-specific SequenceFile
052 * format, users of it must write {@link HLogKey} keys and not arbitrary
053 * {@link WALKey}s because the latter are not Writables (nor made to work with
054 * Hadoop serialization).
055 */
056@InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.CONFIG)
057public class SequenceFileLogWriter implements DefaultWALProvider.Writer {
058  private static final Log LOG = LogFactory.getLog(SequenceFileLogWriter.class);
059  // The sequence file we delegate to.
060  private SequenceFile.Writer writer;
061  // This is the FSDataOutputStream instance that is the 'out' instance
062  // in the SequenceFile.Writer 'writer' instance above.
063  private FSDataOutputStream writer_out;
064
065  private CompressionContext compressionContext;
066
067  // Legacy stuff from pre-PB WAL metadata.
068  private static final Text WAL_VERSION_KEY = new Text("version");
069  private static final Text WAL_COMPRESSION_TYPE_KEY = new Text("compression.type");
070  private static final Text DICTIONARY_COMPRESSION_TYPE = new Text("dictionary");
071
072  /**
073   * Default constructor.
074   */
075  public SequenceFileLogWriter() {
076    super();
077  }
078  /**
079   * Create sequence file Metadata for our WAL file with version and compression
080   * type (if any).
081   * @param conf
082   * @param compress
083   * @return Metadata instance.
084   */
085  private static Metadata createMetadata(final Configuration conf,
086      final boolean compress) {
087    TreeMap<Text, Text> metaMap = new TreeMap<Text, Text>();
088    metaMap.put(WAL_VERSION_KEY, new Text("1"));
089    if (compress) {
090      // Currently we only do one compression type.
091      metaMap.put(WAL_COMPRESSION_TYPE_KEY, DICTIONARY_COMPRESSION_TYPE);
092    }
093    return new Metadata(metaMap);
094  }
095
096  private boolean initializeCompressionContext(Configuration conf, Path path) throws IOException {
097    boolean doCompress = conf.getBoolean(HConstants.ENABLE_WAL_COMPRESSION, false);
098    if (doCompress) {
099      try {
100        this.compressionContext = new CompressionContext(LRUDictionary.class,
101            FSUtils.isRecoveredEdits(path), conf.getBoolean(
102                CompressionContext.ENABLE_WAL_TAGS_COMPRESSION, true));
103      } catch (Exception e) {
104        throw new IOException("Failed to initiate CompressionContext", e);
105      }
106    }
107    return doCompress;
108  }
109
110  @Override
111  public void init(FileSystem fs, Path path, Configuration conf, boolean overwritable)
112  throws IOException {
113    boolean compress = initializeCompressionContext(conf, path);
114    // Create a SF.Writer instance.
115    try {
116      // reflection for a version of SequenceFile.createWriter that doesn't
117      // automatically create the parent directory (see HBASE-2312)
118      this.writer = (SequenceFile.Writer) SequenceFile.class
119        .getMethod("createWriter", new Class[] {FileSystem.class,
120            Configuration.class, Path.class, Class.class, Class.class,
121            Integer.TYPE, Short.TYPE, Long.TYPE, Boolean.TYPE,
122            CompressionType.class, CompressionCodec.class, Metadata.class})
123        .invoke(null, new Object[] {fs, conf, path, HLogKey.class, WALEdit.class,
124            Integer.valueOf(FSUtils.getDefaultBufferSize(fs)),
125            Short.valueOf((short)
126              conf.getInt("hbase.regionserver.hlog.replication",
127              FSUtils.getDefaultReplication(fs, path))),
128            Long.valueOf(conf.getLong("hbase.regionserver.hlog.blocksize",
129                FSUtils.getDefaultBlockSize(fs, path))),
130            Boolean.valueOf(false) /*createParent*/,
131            SequenceFile.CompressionType.NONE, new DefaultCodec(),
132            createMetadata(conf, compress)
133            });
134    } catch (InvocationTargetException ite) {
135      // function was properly called, but threw it's own exception
136      throw new IOException(ite.getCause());
137    } catch (Exception e) {
138      // ignore all other exceptions. related to reflection failure
139    }
140
141    // if reflection failed, use the old createWriter
142    if (this.writer == null) {
143      LOG.debug("new createWriter -- HADOOP-6840 -- not available");
144      this.writer = SequenceFile.createWriter(fs, conf, path,
145        HLogKey.class, WALEdit.class,
146        FSUtils.getDefaultBufferSize(fs),
147        (short) conf.getInt("hbase.regionserver.hlog.replication",
148          FSUtils.getDefaultReplication(fs, path)),
149        conf.getLong("hbase.regionserver.hlog.blocksize",
150          FSUtils.getDefaultBlockSize(fs, path)),
151        SequenceFile.CompressionType.NONE,
152        new DefaultCodec(),
153        null,
154        createMetadata(conf, compress));
155    } else {
156      if (LOG.isTraceEnabled()) LOG.trace("Using new createWriter -- HADOOP-6840");
157    }
158    
159    this.writer_out = getSequenceFilePrivateFSDataOutputStreamAccessible();
160    if (LOG.isTraceEnabled()) LOG.trace("Path=" + path + ", compression=" + compress);
161  }
162
163  // Get at the private FSDataOutputStream inside in SequenceFile so we can
164  // call sync on it.  Make it accessible.
165  private FSDataOutputStream getSequenceFilePrivateFSDataOutputStreamAccessible()
166  throws IOException {
167    FSDataOutputStream out = null;
168    final Field fields [] = this.writer.getClass().getDeclaredFields();
169    final String fieldName = "out";
170    for (int i = 0; i < fields.length; ++i) {
171      if (fieldName.equals(fields[i].getName())) {
172        try {
173          // Make the 'out' field up in SF.Writer accessible.
174          fields[i].setAccessible(true);
175          out = (FSDataOutputStream)fields[i].get(this.writer);
176          break;
177        } catch (IllegalAccessException ex) {
178          throw new IOException("Accessing " + fieldName, ex);
179        } catch (SecurityException e) {
180          LOG.warn("Does not have access to out field from FSDataOutputStream",
181              e);
182        }
183      }
184    }
185    return out;
186  }
187
188  @Override
189  public void append(WAL.Entry entry) throws IOException {
190    entry.setCompressionContext(compressionContext);
191    try {
192      this.writer.append(entry.getKey(), entry.getEdit());
193    } catch (NullPointerException npe) {
194      // Concurrent close...
195      throw new IOException(npe);
196    }
197  }
198
199  @Override
200  public void close() throws IOException {
201    if (this.writer != null) {
202      try {
203        this.writer.close();
204      } catch (NullPointerException npe) {
205        // Can get a NPE coming up from down in DFSClient$DFSOutputStream#close
206        LOG.warn(npe);
207      }
208      this.writer = null;
209    }
210  }
211
212  @Override
213  public void sync() throws IOException {
214    try {
215      this.writer.syncFs();
216    } catch (NullPointerException npe) {
217      // Concurrent close...
218      throw new IOException(npe);
219    }
220  }
221
222  @Override
223  public long getLength() throws IOException {
224    try {
225      return this.writer.getLength();
226    } catch (NullPointerException npe) {
227      // Concurrent close...
228      throw new IOException(npe);
229    }
230  }
231
232  /**
233   * @return The dfsclient out stream up inside SF.Writer made accessible, or
234   * null if not available.
235   */
236  public FSDataOutputStream getWriterFSDataOutputStream() {
237    return this.writer_out;
238  }
