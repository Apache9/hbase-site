056/**
057 * Tests a race condition between archiving of compacted files in CompactedHFilesDischarger chore
058 * and HRegion.close();
059 */
060@Category({RegionServerTests.class, MediumTests.class})
061public class TestCompactionArchiveConcurrentClose {
062  public HBaseTestingUtility testUtil;
063
064  private Path testDir;
065  private AtomicBoolean archived = new AtomicBoolean();
066
067  @Rule
068  public TestName name = new TestName();
069
070  @Before
071  public void setup() throws Exception {
072    testUtil = HBaseTestingUtility.createLocalHTU();
073    testDir = testUtil.getDataTestDir("TestStoreFileRefresherChore");
074    FSUtils.setRootDir(testUtil.getConfiguration(), testDir);
075  }
076
077  @After
078  public void tearDown() throws Exception {
079    testUtil.cleanupTestDir();
080  }
081
082  @Test
083  public void testStoreCloseAndDischargeRunningInParallel() throws Exception {
084    byte[] fam = Bytes.toBytes("f");
085    byte[] col = Bytes.toBytes("c");
086    byte[] val = Bytes.toBytes("val");
087
088    final TableName tableName = TableName.valueOf(name.getMethodName());
089    HTableDescriptor htd = new HTableDescriptor(tableName);
090    htd.addFamily(new HColumnDescriptor(fam));
091    HRegionInfo info = new HRegionInfo(tableName, null, null, false);
092    Region region = initHRegion(htd, info);
093    RegionServerServices rss = mock(RegionServerServices.class);
094    List<Region> regions = new ArrayList<Region>();
095    regions.add(region);
096    when(rss.getOnlineRegions()).thenReturn(regions);
097
098    // Create the cleaner object
099    CompactedHFilesDischarger cleaner =
100        new CompactedHFilesDischarger(1000, (Stoppable) null, rss, false);
101    // Add some data to the region and do some flushes
102    int batchSize = 10;
103    int fileCount = 10;
104    for (int f = 0; f < fileCount; f++) {
105      int start = f * batchSize;
106      for (int i = start; i < start + batchSize; i++) {
107        Put p = new Put(Bytes.toBytes("row" + i));
108        p.addColumn(fam, col, val);
109        region.put(p);
110      }
111      // flush them
112      region.flush(true);
113    }
114
115    Store store = region.getStore(fam);
116    assertEquals(fileCount, store.getStorefilesCount());
117
118    Collection<StoreFile> storefiles = store.getStorefiles();
119    // None of the files should be in compacted state.
120    for (StoreFile file : storefiles) {
121      assertFalse(file.isCompactedAway());
122    }
123    // Do compaction
124    region.compact(true);
125
126    // now run the cleaner with a concurrent close
127    Thread cleanerThread = new Thread() {
128      public void run() {
129        cleaner.chore();
130      }
131    };
132    cleanerThread.start();
133    // wait for cleaner to pause
134    synchronized (archived) {
135      if (!archived.get()) {
136        archived.wait();
137      }
138    }
139    final AtomicReference<Exception> closeException = new AtomicReference<>();
140    Thread closeThread = new Thread() {
141      public void run() {
142        // wait for the chore to complete and call close
143        try {
144          ((HRegion) region).close();
145        } catch (IOException e) {
146          closeException.set(e);
147        }
148      }
149    };
150    closeThread.start();
151    // no error should occur after the execution of the test
152    closeThread.join();
153    cleanerThread.join();
154
155    if (closeException.get() != null) {
156      throw closeException.get();
157    }
158  }
159
160  private Region initHRegion(HTableDescriptor htd, HRegionInfo info)
161      throws IOException {
162    Configuration conf = testUtil.getConfiguration();
163    Path tableDir = FSUtils.getTableDir(testDir, htd.getTableName());
164
165    HRegionFileSystem fs = new WaitingHRegionFileSystem(conf, tableDir.getFileSystem(conf),
166        tableDir, info);
167    final Configuration walConf = new Configuration(conf);
168    FSUtils.setRootDir(walConf, tableDir);
169    final WALFactory wals = new WALFactory(walConf, null, "log_" + info.getEncodedName());
170    HRegion region =
171        new HRegion(fs, wals.getWAL(info.getEncodedNameAsBytes(), info.getTable().getNamespace()),
172            conf, htd, null);
173
174    region.initialize();
175
176    return region;
177  }
178
179  private class WaitingHRegionFileSystem extends HRegionFileSystem {
180
181    public WaitingHRegionFileSystem(final Configuration conf, final FileSystem fs,
182        final Path tableDir, final HRegionInfo regionInfo) {
183      super(conf, fs, tableDir, regionInfo);
184    }
185
186    @Override
187    public void removeStoreFiles(String familyName, Collection<StoreFile> storeFiles)
188        throws IOException {
189      super.removeStoreFiles(familyName, storeFiles);
190      archived.set(true);
191      synchronized (archived) {
192        archived.notifyAll();
193      }
194      try {
195        // unfortunately we can't use a stronger barrier here as the fix synchronizing
196        // the race condition will then block
197        Thread.sleep(100);
198      } catch (InterruptedException ie) {
199        throw new InterruptedIOException("Interrupted waiting for latch");
200      }
201    }
202  }
