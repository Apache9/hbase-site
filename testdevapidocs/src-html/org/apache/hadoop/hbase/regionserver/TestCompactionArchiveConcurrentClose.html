054/**
055 * Tests a race condition between archiving of compacted files in CompactedHFilesDischarger chore
056 * and HRegion.close();
057 */
058@Category({RegionServerTests.class, MediumTests.class})
059public class TestCompactionArchiveConcurrentClose {
060  public HBaseTestingUtility testUtil;
061
062  private Path testDir;
063  private AtomicBoolean archived = new AtomicBoolean();
064
065  @Before
066  public void setup() throws Exception {
067    testUtil = HBaseTestingUtility.createLocalHTU();
068    testDir = testUtil.getDataTestDir("TestStoreFileRefresherChore");
069    FSUtils.setRootDir(testUtil.getConfiguration(), testDir);
070  }
071
072  @After
073  public void tearDown() throws Exception {
074    testUtil.cleanupTestDir();
075  }
076
077  @Test
078  public void testStoreCloseAndDischargeRunningInParallel() throws Exception {
079    byte[] fam = Bytes.toBytes("f");
080    byte[] col = Bytes.toBytes("c");
081    byte[] val = Bytes.toBytes("val");
082
083    TableName tableName = TableName.valueOf(getClass().getSimpleName());
084    HTableDescriptor htd = new HTableDescriptor(tableName);
085    htd.addFamily(new HColumnDescriptor(fam));
086    HRegionInfo info = new HRegionInfo(tableName, null, null, false);
087    Region region = initHRegion(htd, info);
088    RegionServerServices rss = mock(RegionServerServices.class);
089    List<Region> regions = new ArrayList<Region>();
090    regions.add(region);
091    when(rss.getOnlineRegions()).thenReturn(regions);
092
093    // Create the cleaner object
094    CompactedHFilesDischarger cleaner =
095        new CompactedHFilesDischarger(1000, (Stoppable) null, rss, false);
096    // Add some data to the region and do some flushes
097    int batchSize = 10;
098    int fileCount = 10;
099    for (int f = 0; f < fileCount; f++) {
100      int start = f * batchSize;
101      for (int i = start; i < start + batchSize; i++) {
102        Put p = new Put(Bytes.toBytes("row" + i));
103        p.addColumn(fam, col, val);
104        region.put(p);
105      }
106      // flush them
107      region.flush(true);
108    }
109
110    Store store = region.getStore(fam);
111    assertEquals(fileCount, store.getStorefilesCount());
112
113    Collection<StoreFile> storefiles = store.getStorefiles();
114    // None of the files should be in compacted state.
115    for (StoreFile file : storefiles) {
116      assertFalse(file.isCompactedAway());
117    }
118    // Do compaction
119    region.compact(true);
120
121    // now run the cleaner with a concurrent close
122    Thread cleanerThread = new Thread() {
123      public void run() {
124        cleaner.chore();
125      }
126    };
127    cleanerThread.start();
128    // wait for cleaner to pause
129    synchronized (archived) {
130      if (!archived.get()) {
131        archived.wait();
132      }
133    }
134    final AtomicReference<Exception> closeException = new AtomicReference<>();
135    Thread closeThread = new Thread() {
136      public void run() {
137        // wait for the chore to complete and call close
138        try {
139          ((HRegion) region).close();
140        } catch (IOException e) {
141          closeException.set(e);
142        }
143      }
144    };
145    closeThread.start();
146    // no error should occur after the execution of the test
147    closeThread.join();
148    cleanerThread.join();
149
150    if (closeException.get() != null) {
151      throw closeException.get();
152    }
153  }
154
155  private Region initHRegion(HTableDescriptor htd, HRegionInfo info)
156      throws IOException {
157    Configuration conf = testUtil.getConfiguration();
158    Path tableDir = FSUtils.getTableDir(testDir, htd.getTableName());
159
160    HRegionFileSystem fs = new WaitingHRegionFileSystem(conf, tableDir.getFileSystem(conf),
161        tableDir, info);
162    final Configuration walConf = new Configuration(conf);
163    FSUtils.setRootDir(walConf, tableDir);
164    final WALFactory wals = new WALFactory(walConf, null, "log_" + info.getEncodedName());
165    HRegion region =
166        new HRegion(fs, wals.getWAL(info.getEncodedNameAsBytes(), info.getTable().getNamespace()),
167            conf, htd, null);
168
169    region.initialize();
170
171    return region;
172  }
173
174  private class WaitingHRegionFileSystem extends HRegionFileSystem {
175
176    public WaitingHRegionFileSystem(final Configuration conf, final FileSystem fs,
177        final Path tableDir, final HRegionInfo regionInfo) {
178      super(conf, fs, tableDir, regionInfo);
179    }
180
181    @Override
182    public void removeStoreFiles(String familyName, Collection<StoreFile> storeFiles)
183        throws IOException {
184      super.removeStoreFiles(familyName, storeFiles);
185      archived.set(true);
186      synchronized (archived) {
187        archived.notifyAll();
188      }
189      try {
190        // unfortunately we can't use a stronger barrier here as the fix synchronizing
191        // the race condition will then block
192        Thread.sleep(100);
193      } catch (InterruptedException ie) {
194        throw new InterruptedIOException("Interrupted waiting for latch");
195      }
196    }
197  }
