164  @SuppressWarnings("deprecation")
165  private Store init(String methodName, Configuration conf, HTableDescriptor htd,
166      HColumnDescriptor hcd) throws IOException {
167    //Setting up a Store
168    Path basedir = new Path(DIR+methodName);
169    Path tableDir = FSUtils.getTableDir(basedir, htd.getTableName());
170    final Path logdir = new Path(basedir, AbstractFSWALProvider.getWALDirectoryName(methodName));
171
172    FileSystem fs = FileSystem.get(conf);
173
174    fs.delete(logdir, true);
175
176    if (htd.hasFamily(hcd.getName())) {
177      htd.modifyFamily(hcd);
178    } else {
179      htd.addFamily(hcd);
180    }
181    ChunkCreator.initialize(MemStoreLABImpl.CHUNK_SIZE_DEFAULT, false, 0, 0, 0, null);
182    HRegionInfo info = new HRegionInfo(htd.getTableName(), null, null, false);
183    final Configuration walConf = new Configuration(conf);
184    FSUtils.setRootDir(walConf, basedir);
185    final WALFactory wals = new WALFactory(walConf, null, methodName);
186    HRegion region = new HRegion(tableDir, wals.getWAL(info.getEncodedNameAsBytes(),
187            info.getTable().getNamespace()), fs, conf, info, htd, null);
188
189    store = new HStore(region, hcd, conf);
190    return store;
191  }

194   * Test we do not lose data if we fail a flush and then close.
195   * Part of HBase-10466

198  @Test
199  public void testFlushSizeAccounting() throws Exception {
200    LOG.info("Setting up a faulty file system that cannot write in " +
201      this.name.getMethodName());
202    final Configuration conf = HBaseConfiguration.create();
203    // Only retry once.
204    conf.setInt("hbase.hstore.flush.retries.number", 1);
205    User user = User.createUserForTesting(conf, this.name.getMethodName(),
206      new String[]{"foo"});
207    // Inject our faulty LocalFileSystem
208    conf.setClass("fs.file.impl", FaultyFileSystem.class, FileSystem.class);
209    user.runAs(new PrivilegedExceptionAction<Object>() {
210      @Override
211      public Object run() throws Exception {
212        // Make sure it worked (above is sensitive to caching details in hadoop core)
213        FileSystem fs = FileSystem.get(conf);
214        Assert.assertEquals(FaultyFileSystem.class, fs.getClass());
215        FaultyFileSystem ffs = (FaultyFileSystem)fs;
216
217        // Initialize region
218        init(name.getMethodName(), conf);
219
220        MemstoreSize size = store.memstore.getFlushableSize();
221        Assert.assertEquals(0, size.getDataSize());
222        LOG.info("Adding some data");
223        MemstoreSize kvSize = new MemstoreSize();
224        store.add(new KeyValue(row, family, qf1, 1, (byte[])null), kvSize);
225        size = store.memstore.getFlushableSize();
226        Assert.assertEquals(kvSize, size);
227        // Flush.  Bug #1 from HBASE-10466.  Make sure size calculation on failed flush is right.
228        try {
229          LOG.info("Flushing");
230          flushStore(store, id++);
231          Assert.fail("Didn't bubble up IOE!");
232        } catch (IOException ioe) {
233          Assert.assertTrue(ioe.getMessage().contains("Fault injected"));
234        }
235        size = store.memstore.getFlushableSize();
236        Assert.assertEquals(kvSize, size);
237        MemstoreSize kvSize2 = new MemstoreSize();
238        store.add(new KeyValue(row, family, qf2, 2, (byte[])null), kvSize2);
239        // Even though we add a new kv, we expect the flushable size to be 'same' since we have
240        // not yet cleared the snapshot -- the above flush failed.
241        Assert.assertEquals(kvSize, size);
242        ffs.fault.set(false);
243        flushStore(store, id++);
244        size = store.memstore.getFlushableSize();
245        // Size should be the foreground kv size.
246        Assert.assertEquals(kvSize2, size);
247        flushStore(store, id++);
248        size = store.memstore.getFlushableSize();
249        assertEquals(0, size.getDataSize());
250        assertEquals(0, size.getHeapSize());
251        return null;
252      }
253    });
254  }

292  /*
293   * @param minVersions the MIN_VERSIONS for the column family
294   */
295  public void testDeleteExpiredStoreFiles(int minVersions) throws Exception {
296    int storeFileNum = 4;
297    int ttl = 4;
298    IncrementingEnvironmentEdge edge = new IncrementingEnvironmentEdge();
299    EnvironmentEdgeManagerTestHelper.injectEdge(edge);
300
301    Configuration conf = HBaseConfiguration.create();
302    // Enable the expired store file deletion
303    conf.setBoolean("hbase.store.delete.expired.storefile", true);
304    // Set the compaction threshold higher to avoid normal compactions.
305    conf.setInt(CompactionConfiguration.HBASE_HSTORE_COMPACTION_MIN_KEY, 5);
306
307    HColumnDescriptor hcd = new HColumnDescriptor(family);
308    hcd.setMinVersions(minVersions);
309    hcd.setTimeToLive(ttl);
310    init(name.getMethodName() + "-" + minVersions, conf, hcd);
311
312    long storeTtl = this.store.getScanInfo().getTtl();
313    long sleepTime = storeTtl / storeFileNum;
314    long timeStamp;
315    // There are 4 store files and the max time stamp difference among these
316    // store files will be (this.store.ttl / storeFileNum)
317    for (int i = 1; i <= storeFileNum; i++) {
318      LOG.info("Adding some data for the store file #" + i);
319      timeStamp = EnvironmentEdgeManager.currentTime();
320      this.store.add(new KeyValue(row, family, qf1, timeStamp, (byte[]) null), null);
321      this.store.add(new KeyValue(row, family, qf2, timeStamp, (byte[]) null), null);
322      this.store.add(new KeyValue(row, family, qf3, timeStamp, (byte[]) null), null);
323      flush(i);
324      edge.incrementTime(sleepTime);
325    }
326
327    // Verify the total number of store files
328    Assert.assertEquals(storeFileNum, this.store.getStorefiles().size());
329
330     // Each call will find one expired store file and delete it before compaction happens.
331     // There will be no compaction due to threshold above. Last file will not be replaced.
332    for (int i = 1; i <= storeFileNum - 1; i++) {
333      // verify the expired store file.
334      assertNull(this.store.requestCompaction());
335      Collection<StoreFile> sfs = this.store.getStorefiles();
336      // Ensure i files are gone.
337      if (minVersions == 0) {
338        assertEquals(storeFileNum - i, sfs.size());
339        // Ensure only non-expired files remain.
340        for (StoreFile sf : sfs) {
341          assertTrue(sf.getReader().getMaxTimestamp() >= (edge.currentTime() - storeTtl));
342        }
343      } else {
344        assertEquals(storeFileNum, sfs.size());
345      }
346      // Let the next store file expired.
347      edge.incrementTime(sleepTime);
348    }
349    assertNull(this.store.requestCompaction());
350
351    Collection<StoreFile> sfs = this.store.getStorefiles();
352    // Assert the last expired file is not removed.
353    if (minVersions == 0) {
354      assertEquals(1, sfs.size());
355    }
356    long ts = sfs.iterator().next().getReader().getMaxTimestamp();
357    assertTrue(ts < (edge.currentTime() - storeTtl));
358
359    for (StoreFile sf : sfs) {
360      sf.closeReader(true);
361    }
362  }

365  public void testLowestModificationTime() throws Exception {
366    Configuration conf = HBaseConfiguration.create();
367    FileSystem fs = FileSystem.get(conf);
368    // Initialize region
369    init(name.getMethodName(), conf);
370
371    int storeFileNum = 4;
372    for (int i = 1; i <= storeFileNum; i++) {
373      LOG.info("Adding some data for the store file #"+i);
374      this.store.add(new KeyValue(row, family, qf1, i, (byte[])null), null);
375      this.store.add(new KeyValue(row, family, qf2, i, (byte[])null), null);
376      this.store.add(new KeyValue(row, family, qf3, i, (byte[])null), null);
377      flush(i);
378    }
379    // after flush; check the lowest time stamp
380    long lowestTimeStampFromManager = StoreUtils.getLowestTimestamp(store.getStorefiles());
381    long lowestTimeStampFromFS = getLowestTimeStampFromFS(fs, store.getStorefiles());
382    Assert.assertEquals(lowestTimeStampFromManager,lowestTimeStampFromFS);
383
384    // after compact; check the lowest time stamp
385    store.compact(store.requestCompaction(), NoLimitThroughputController.INSTANCE, null);
386    lowestTimeStampFromManager = StoreUtils.getLowestTimestamp(store.getStorefiles());
387    lowestTimeStampFromFS = getLowestTimeStampFromFS(fs, store.getStorefiles());
388    Assert.assertEquals(lowestTimeStampFromManager, lowestTimeStampFromFS);
389  }
390
391  private static long getLowestTimeStampFromFS(FileSystem fs,
392      final Collection<StoreFile> candidates) throws IOException {
393    long minTs = Long.MAX_VALUE;
394    if (candidates.isEmpty()) {
395      return minTs;
396    }
397    Path[] p = new Path[candidates.size()];
398    int i = 0;
399    for (StoreFile sf : candidates) {
400      p[i] = sf.getPath();
401      ++i;
402    }
403
404    FileStatus[] stats = fs.listStatus(p);
405    if (stats == null || stats.length == 0) {
406      return minTs;
407    }
408    for (FileStatus s : stats) {
409      minTs = Math.min(minTs, s.getModificationTime());
410    }
411    return minTs;
412  }

576
577    final Configuration conf = HBaseConfiguration.create();
578    User user = User.createUserForTesting(conf,

590        // Initialize region
591        init(name.getMethodName(), conf);

593        LOG.info("Adding some data");
594        store.add(new KeyValue(row, family, qf1, 1, (byte[])null), null);
595        store.add(new KeyValue(row, family, qf2, 1, (byte[])null), null);
596        store.add(new KeyValue(row, family, qf3, 1, (byte[])null), null);

661    volatile long faultPos = Long.MAX_VALUE;
662    private final AtomicBoolean fault;
663
664    public FaultyOutputStream(FSDataOutputStream out, long faultPos, final AtomicBoolean fault)
665    throws IOException {
666      super(out, null);
667      this.faultPos = faultPos;
668      this.fault = fault;
669    }
670
671    @Override
672    public void write(byte[] buf, int offset, int length) throws IOException {
673      System.err.println("faulty stream write at pos " + getPos());
674      injectFault();
675      super.write(buf, offset, length);
676    }
677
678    private void injectFault() throws IOException {
679      if (this.fault.get() && getPos() >= faultPos) {
680        throw new IOException("Fault injected");
681      }
682    }
