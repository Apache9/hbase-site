055  @Test
056  public void testOnDiskRegionCreation() throws IOException {
057    Path rootDir = TEST_UTIL.getDataTestDirOnTestFS("testOnDiskRegionCreation");
058    FileSystem fs = TEST_UTIL.getTestFileSystem();
059    Configuration conf = TEST_UTIL.getConfiguration();
060
061    // Create a Region
062    HRegionInfo hri = new HRegionInfo(TableName.valueOf("TestTable"));
063    HRegionFileSystem regionFs = HRegionFileSystem.createRegionOnFileSystem(conf, fs,
064        FSUtils.getTableDir(rootDir, hri.getTable()), hri);
065
066    // Verify if the region is on disk
067    Path regionDir = regionFs.getRegionDir();
068    assertTrue("The region folder should be created", fs.exists(regionDir));
069
070    // Verify the .regioninfo
071    HRegionInfo hriVerify = HRegionFileSystem.loadRegionInfoFileContent(fs, regionDir);
072    assertEquals(hri, hriVerify);
073
074    // Open the region
075    regionFs = HRegionFileSystem.openRegionFromFileSystem(conf, fs,
076        FSUtils.getTableDir(rootDir, hri.getTable()), hri, false);
077    assertEquals(regionDir, regionFs.getRegionDir());
078
079    // Delete the region
080    HRegionFileSystem.deleteRegionFromFileSystem(conf, fs,
081        FSUtils.getTableDir(rootDir, hri.getTable()), hri);
082    assertFalse("The region folder should be removed", fs.exists(regionDir));
083
084    fs.delete(rootDir, true);
085  }

087  @Test
088  public void testNonIdempotentOpsWithRetries() throws IOException {
089    Path rootDir = TEST_UTIL.getDataTestDirOnTestFS("testOnDiskRegionCreation");
090    FileSystem fs = TEST_UTIL.getTestFileSystem();
091    Configuration conf = TEST_UTIL.getConfiguration();
092
093    // Create a Region
094    HRegionInfo hri = new HRegionInfo(TableName.valueOf("TestTable"));
095    HRegionFileSystem regionFs = HRegionFileSystem.createRegionOnFileSystem(conf, fs, rootDir, hri);
096    assertTrue(fs.exists(regionFs.getRegionDir()));
097
098    regionFs = new HRegionFileSystem(conf, new MockFileSystemForCreate(),
099        null, null);
100    // HRegionFileSystem.createRegionOnFileSystem(conf, new MockFileSystemForCreate(), rootDir,
101    // hri);
102    boolean result = regionFs.createDir(new Path("/foo/bar"));
103    assertTrue("Couldn't create the directory", result);
104
105
106    regionFs = new HRegionFileSystem(conf, new MockFileSystem(), null, null);
107    result = regionFs.rename(new Path("/foo/bar"), new Path("/foo/bar2"));
108    assertTrue("Couldn't rename the directory", result);
109
110    regionFs = new HRegionFileSystem(conf, new MockFileSystem(), null, null);
111    result = regionFs.deleteDir(new Path("/foo/bar"));
112    assertTrue("Couldn't delete the directory", result);
113    fs.delete(rootDir, true);
114  }

211  @Test
212  public void testTempAndCommit() throws IOException {
213    Path rootDir = TEST_UTIL.getDataTestDirOnTestFS("testTempAndCommit");
214    FileSystem fs = TEST_UTIL.getTestFileSystem();
215    Configuration conf = TEST_UTIL.getConfiguration();
216
217    // Create a Region
218    String familyName = "cf";
219    HRegionInfo hri = new HRegionInfo(TableName.valueOf("TestTable"));
220    HRegionFileSystem regionFs = HRegionFileSystem.createRegionOnFileSystem(conf, fs, rootDir, hri);
221
222    // New region, no store files
223    Collection<StoreFileInfo> storeFiles = regionFs.getStoreFiles(familyName);
224    assertEquals(0, storeFiles != null ? storeFiles.size() : 0);
225
226    // Create a new file in temp (no files in the family)
227    Path buildPath = regionFs.createTempName();
228    fs.createNewFile(buildPath);
229    storeFiles = regionFs.getStoreFiles(familyName);
230    assertEquals(0, storeFiles != null ? storeFiles.size() : 0);
231
232    // commit the file
233    Path dstPath = regionFs.commitStoreFile(familyName, buildPath);
234    storeFiles = regionFs.getStoreFiles(familyName);
235    assertEquals(0, storeFiles != null ? storeFiles.size() : 0);
236    assertFalse(fs.exists(buildPath));
237
238    fs.delete(rootDir, true);
239  }
