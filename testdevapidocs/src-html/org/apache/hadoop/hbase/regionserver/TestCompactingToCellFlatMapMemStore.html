086  public void testCompaction1Bucket() throws IOException {
087    int counter = 0;
088    String[] keys1 = { "A", "A", "B", "C" }; //A1, A2, B3, C4
089    if (toCellChunkMap) {
090      // set memstore to flat into CellChunkMap
091      conf.set(CompactingMemStore.COMPACTING_MEMSTORE_INDEX_KEY,
092          String.valueOf(CompactingMemStore.IndexType.CHUNK_MAP));
093      ((CompactingMemStore)memstore).setIndexType();
094    }
095
096    // test 1 bucket
097    long totalCellsLen = addRowsByKeys(memstore, keys1);
098    long cellBeforeFlushSize = cellBeforeFlushSize();
099    long cellAfterFlushSize  = cellAfterFlushSize();
100    long totalHeapSize = MutableSegment.DEEP_OVERHEAD + 4 * cellBeforeFlushSize;
101
102    assertEquals(totalCellsLen, regionServicesForStores.getMemStoreSize());
103    assertEquals(totalHeapSize, ((CompactingMemStore)memstore).heapSize());
104
105    assertEquals(4, memstore.getActive().getCellsCount());
106    ((CompactingMemStore) memstore).flushInMemory();    // push keys to pipeline and compact
107    assertEquals(0, memstore.getSnapshot().getCellsCount());
108    // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
109    // totalCellsLen
110    totalCellsLen = (totalCellsLen * 3) / 4;
111    assertEquals(totalCellsLen, regionServicesForStores.getMemStoreSize());
112
113    totalHeapSize =
114        3 * cellAfterFlushSize + MutableSegment.DEEP_OVERHEAD
115            + (toCellChunkMap ?
116            CellChunkImmutableSegment.DEEP_OVERHEAD_CCM :
117            CellArrayImmutableSegment.DEEP_OVERHEAD_CAM);
118    assertEquals(totalHeapSize, ((CompactingMemStore)memstore).heapSize());
119    for ( Segment s : memstore.getSegments()) {
120      counter += s.getCellsCount();
121    }
122    assertEquals(3, counter);
123    MemStoreSize size = memstore.getFlushableSize();
124    MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
125    region.decrMemStoreSize(size);  // simulate flusher
126    ImmutableSegment s = memstore.getSnapshot();
127    assertEquals(3, s.getCellsCount());
128    assertEquals(0, regionServicesForStores.getMemStoreSize());
129
130    memstore.clearSnapshot(snapshot.getId());
131  }

195    if (toCellChunkMap) {
196      // set memstore to flat into CellChunkMap
197      conf.set(CompactingMemStore.COMPACTING_MEMSTORE_INDEX_KEY,
198          String.valueOf(CompactingMemStore.IndexType.CHUNK_MAP));
199      ((CompactingMemStore)memstore).setIndexType();
200    }
201    String[] keys1 = { "A", "A", "B", "C" };
202    String[] keys2 = { "A", "B", "D" };
203    String[] keys3 = { "D", "B", "B" };
204
205    long totalCellsLen1 = addRowsByKeys(memstore, keys1);
206    long cellBeforeFlushSize = cellBeforeFlushSize();
207    long cellAfterFlushSize = cellAfterFlushSize();
208    long totalHeapSize1 = MutableSegment.DEEP_OVERHEAD + 4 * cellBeforeFlushSize;
209    assertEquals(totalCellsLen1, region.getMemStoreSize());
210    assertEquals(totalHeapSize1, ((CompactingMemStore) memstore).heapSize());
211
212    MemStoreSize size = memstore.getFlushableSize();
213    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
214
215    assertEquals(0, memstore.getSnapshot().getCellsCount());
216    // One cell is duplicated and the compaction will remove it. All cells of same size so adjusting
217    // totalCellsLen
218    totalCellsLen1 = (totalCellsLen1 * 3) / 4;
219    totalHeapSize1 = 3 * cellAfterFlushSize + MutableSegment.DEEP_OVERHEAD
220        + (toCellChunkMap ?
221        CellChunkImmutableSegment.DEEP_OVERHEAD_CCM :
222        CellArrayImmutableSegment.DEEP_OVERHEAD_CAM);
223    assertEquals(totalCellsLen1, regionServicesForStores.getMemStoreSize());
224    assertEquals(totalHeapSize1, ((CompactingMemStore) memstore).heapSize());
225
226    long totalCellsLen2 = addRowsByKeys(memstore, keys2);
227    long totalHeapSize2 = 3 * cellBeforeFlushSize;
228
229    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemStoreSize());
230    assertEquals(totalHeapSize1 + totalHeapSize2, ((CompactingMemStore) memstore).heapSize());
231
232    ((CompactingMemStore) memstore).disableCompaction();
233    size = memstore.getFlushableSize();
234    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline without compaction
235    totalHeapSize2 = totalHeapSize2 + CSLMImmutableSegment.DEEP_OVERHEAD_CSLM;
236    assertEquals(0, memstore.getSnapshot().getCellsCount());
237    assertEquals(totalCellsLen1 + totalCellsLen2, regionServicesForStores.getMemStoreSize());
238    assertEquals(totalHeapSize1 + totalHeapSize2, ((CompactingMemStore) memstore).heapSize());
239
240    long totalCellsLen3 = addRowsByKeys(memstore, keys3);
241    long totalHeapSize3 = 3 * cellBeforeFlushSize;
242    assertEquals(totalCellsLen1 + totalCellsLen2 + totalCellsLen3,
243        regionServicesForStores.getMemStoreSize());
244    assertEquals(totalHeapSize1 + totalHeapSize2 + totalHeapSize3,
245        ((CompactingMemStore) memstore).heapSize());
246
247    ((CompactingMemStore) memstore).enableCompaction();
248    size = memstore.getFlushableSize();
249    ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
250    while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
251      Threads.sleep(10);
252    }
253    assertEquals(0, memstore.getSnapshot().getCellsCount());
254    // active flushed to pipeline and all 3 segments compacted. Will get rid of duplicated cells.
255    // Out of total 10, only 4 cells are unique
256    totalCellsLen2 = totalCellsLen2 / 3;// 2 out of 3 cells are duplicated
257    totalCellsLen3 = 0;// All duplicated cells.
258    assertEquals(totalCellsLen1 + totalCellsLen2 + totalCellsLen3,
259        regionServicesForStores.getMemStoreSize());
260    // Only 4 unique cells left
261    long totalHeapSize4 = 4 * cellAfterFlushSize + MutableSegment.DEEP_OVERHEAD
262        + (toCellChunkMap ?
263        CellChunkImmutableSegment.DEEP_OVERHEAD_CCM :
264        CellArrayImmutableSegment.DEEP_OVERHEAD_CAM);
265    assertEquals(totalHeapSize4, ((CompactingMemStore) memstore).heapSize());
266
267    size = memstore.getFlushableSize();
268    MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
269    region.decrMemStoreSize(size);  // simulate flusher
270    ImmutableSegment s = memstore.getSnapshot();
271    assertEquals(4, s.getCellsCount());
272    assertEquals(0, regionServicesForStores.getMemStoreSize());
273
274    memstore.clearSnapshot(snapshot.getId());
275

369      }
370    }
371    assertEquals("the count should be ", count, 150);
372    for(int i = 0; i < scanners.size(); i++) {
373      scanners.get(i).close();
374    }

417  @Test
418  public void testPuttingBackChunksWithOpeningScanner() throws IOException {
419    byte[] row = Bytes.toBytes("testrow");
420    byte[] fam = Bytes.toBytes("testfamily");
421    byte[] qf1 = Bytes.toBytes("testqualifier1");
422    byte[] qf2 = Bytes.toBytes("testqualifier2");
423    byte[] qf3 = Bytes.toBytes("testqualifier3");
424    byte[] qf4 = Bytes.toBytes("testqualifier4");
425    byte[] qf5 = Bytes.toBytes("testqualifier5");
426    byte[] qf6 = Bytes.toBytes("testqualifier6");
427    byte[] qf7 = Bytes.toBytes("testqualifier7");
428    byte[] val = Bytes.toBytes("testval");
429
430    // Setting up memstore
431    memstore.add(new KeyValue(row, fam, qf1, val), null);
432    memstore.add(new KeyValue(row, fam, qf2, val), null);
433    memstore.add(new KeyValue(row, fam, qf3, val), null);
434
435    // Creating a snapshot
436    MemStoreSnapshot snapshot = memstore.snapshot();
437    assertEquals(3, memstore.getSnapshot().getCellsCount());
438
439    // Adding value to "new" memstore
440    assertEquals(0, memstore.getActive().getCellsCount());
441    memstore.add(new KeyValue(row, fam, qf4, val), null);
442    memstore.add(new KeyValue(row, fam, qf5, val), null);
443    assertEquals(2, memstore.getActive().getCellsCount());
444
445    // opening scanner before clear the snapshot
446    List<KeyValueScanner> scanners = memstore.getScanners(0);
447    // Shouldn't putting back the chunks to pool,since some scanners are opening
448    // based on their data
449    // close the scanners
450    for(KeyValueScanner scanner : snapshot.getScanners()) {
451      scanner.close();
452    }
453    memstore.clearSnapshot(snapshot.getId());
454
455    assertTrue(chunkCreator.getPoolSize() == 0);
456
457    // Chunks will be put back to pool after close scanners;
458    for (KeyValueScanner scanner : scanners) {
459      scanner.close();
460    }
461    assertTrue(chunkCreator.getPoolSize() > 0);
462
463    // clear chunks
464    chunkCreator.clearChunksInPool();
465
466    // Creating another snapshot
467
468    snapshot = memstore.snapshot();
469    // Adding more value
470    memstore.add(new KeyValue(row, fam, qf6, val), null);
471    memstore.add(new KeyValue(row, fam, qf7, val), null);
472    // opening scanners
473    scanners = memstore.getScanners(0);
474    // close scanners before clear the snapshot
475    for (KeyValueScanner scanner : scanners) {
476      scanner.close();
477    }
478    // Since no opening scanner, the chunks of snapshot should be put back to
479    // pool
480    // close the scanners
481    for(KeyValueScanner scanner : snapshot.getScanners()) {
482      scanner.close();
483    }
484    memstore.clearSnapshot(snapshot.getId());
485    assertTrue(chunkCreator.getPoolSize() > 0);
486  }

524  private long addRowsByKeys(final AbstractMemStore hmc, String[] keys) {
525    byte[] fam = Bytes.toBytes("testfamily");
526    byte[] qf = Bytes.toBytes("testqualifier");
527    MemStoreSize memstoreSize = new MemStoreSize();
528    for (int i = 0; i < keys.length; i++) {
529      long timestamp = System.currentTimeMillis();
530      Threads.sleep(1); // to make sure each kv gets a different ts
531      byte[] row = Bytes.toBytes(keys[i]);
532      byte[] val = Bytes.toBytes(keys[i] + i);
533      KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
534      hmc.add(kv, memstoreSize);
535      LOG.debug("added kv: " + kv.getKeyString() + ", timestamp" + kv.getTimestamp());
536    }
537    regionServicesForStores.addMemStoreSize(memstoreSize);
538    return memstoreSize.getDataSize();
539  }

541  private long cellBeforeFlushSize() {
542    // make one cell
543    byte[] row = Bytes.toBytes("A");
544    byte[] val = Bytes.toBytes("A" + 0);
545    KeyValue kv =
546        new KeyValue(row, Bytes.toBytes("testfamily"), Bytes.toBytes("testqualifier"),
547            System.currentTimeMillis(), val);
548    return ClassSize.align(
549        ClassSize.CONCURRENT_SKIPLISTMAP_ENTRY + KeyValue.FIXED_OVERHEAD + KeyValueUtil.length(kv));
550  }

552  private long cellAfterFlushSize() {
553    // make one cell
554    byte[] row = Bytes.toBytes("A");
555    byte[] val = Bytes.toBytes("A" + 0);
556    KeyValue kv =
557        new KeyValue(row, Bytes.toBytes("testfamily"), Bytes.toBytes("testqualifier"),
558            System.currentTimeMillis(), val);
559
560    return toCellChunkMap ?
561        ClassSize.align(
562        ClassSize.CELL_CHUNK_MAP_ENTRY + KeyValueUtil.length(kv)) :
563        ClassSize.align(
564        ClassSize.CELL_ARRAY_MAP_ENTRY + KeyValue.FIXED_OVERHEAD + KeyValueUtil.length(kv));
