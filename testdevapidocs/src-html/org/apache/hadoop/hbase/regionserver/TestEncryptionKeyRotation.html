061@Category({RegionServerTests.class, MediumTests.class})
062public class TestEncryptionKeyRotation {
063  private static final Logger LOG = LoggerFactory.getLogger(TestEncryptionKeyRotation.class);
064  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
065  private static final Configuration conf = TEST_UTIL.getConfiguration();
066  private static final Key initialCFKey;
067  private static final Key secondCFKey;
068
069  @Rule
070  public TestName name = new TestName();
071
072  static {
073    // Create the test encryption keys
074    SecureRandom rng = new SecureRandom();
075    byte[] keyBytes = new byte[AES.KEY_LENGTH];
076    rng.nextBytes(keyBytes);
077    String algorithm =
078        conf.get(HConstants.CRYPTO_KEY_ALGORITHM_CONF_KEY, HConstants.CIPHER_AES);
079    initialCFKey = new SecretKeySpec(keyBytes, algorithm);
080    rng.nextBytes(keyBytes);
081    secondCFKey = new SecretKeySpec(keyBytes, algorithm);
082  }
083
084  @BeforeClass
085  public static void setUp() throws Exception {
086    conf.setInt("hfile.format.version", 3);
087    conf.set(HConstants.CRYPTO_KEYPROVIDER_CONF_KEY, KeyProviderForTesting.class.getName());
088    conf.set(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, "hbase");
089
090    // Start the minicluster
091    TEST_UTIL.startMiniCluster(1);
092  }
093
094  @AfterClass
095  public static void tearDown() throws Exception {
096    TEST_UTIL.shutdownMiniCluster();
097  }

099  @Test
100  public void testCFKeyRotation() throws Exception {
101    // Create the table schema
102    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("default", name.getMethodName()));
103    HColumnDescriptor hcd = new HColumnDescriptor("cf");
104    String algorithm =
105        conf.get(HConstants.CRYPTO_KEY_ALGORITHM_CONF_KEY, HConstants.CIPHER_AES);
106    hcd.setEncryptionType(algorithm);
107    hcd.setEncryptionKey(EncryptionUtil.wrapKey(conf, "hbase", initialCFKey));
108    htd.addFamily(hcd);
109
110    // Create the table and some on disk files
111    createTableAndFlush(htd);
112
113    // Verify we have store file(s) with the initial key
114    final List<Path> initialPaths = findStorefilePaths(htd.getTableName());
115    assertTrue(initialPaths.size() > 0);
116    for (Path path: initialPaths) {
117      assertTrue("Store file " + path + " has incorrect key",
118        Bytes.equals(initialCFKey.getEncoded(), extractHFileKey(path)));
119    }
120
121    // Update the schema with a new encryption key
122    hcd = htd.getFamily(Bytes.toBytes("cf"));
123    hcd.setEncryptionKey(EncryptionUtil.wrapKey(conf,
124      conf.get(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, User.getCurrent().getShortName()),
125      secondCFKey));
126    TEST_UTIL.getAdmin().modifyColumnFamily(htd.getTableName(), hcd);
127    Thread.sleep(5000); // Need a predicate for online schema change
128
129    // And major compact
130    TEST_UTIL.getAdmin().majorCompact(htd.getTableName());
131    final List<Path> updatePaths = findCompactedStorefilePaths(htd.getTableName());
132    TEST_UTIL.waitFor(30000, 1000, true, new Predicate<Exception>() {
133      @Override
134      public boolean evaluate() throws Exception {
135        // When compaction has finished, all of the original files will be
136        // gone
137        boolean found = false;
138        for (Path path: updatePaths) {
139          found = TEST_UTIL.getTestFileSystem().exists(path);
140          if (found) {
141            LOG.info("Found " + path);
142            break;
143          }
144        }
145        return !found;
146      }
147    });
148
149    // Verify we have store file(s) with only the new key
150    Thread.sleep(1000);
151    waitForCompaction(htd.getTableName());
152    List<Path> pathsAfterCompaction = findStorefilePaths(htd.getTableName());
153    assertTrue(pathsAfterCompaction.size() > 0);
154    for (Path path: pathsAfterCompaction) {
155      assertTrue("Store file " + path + " has incorrect key",
156        Bytes.equals(secondCFKey.getEncoded(), extractHFileKey(path)));
157    }
158    List<Path> compactedPaths = findCompactedStorefilePaths(htd.getTableName());
159    assertTrue(compactedPaths.size() > 0);
160    for (Path path: compactedPaths) {
161      assertTrue("Store file " + path + " retains initial key",
162        Bytes.equals(initialCFKey.getEncoded(), extractHFileKey(path)));
163    }
164  }

166  @Test
167  public void testMasterKeyRotation() throws Exception {
168    // Create the table schema
169    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("default", name.getMethodName()));
170    HColumnDescriptor hcd = new HColumnDescriptor("cf");
171    String algorithm =
172        conf.get(HConstants.CRYPTO_KEY_ALGORITHM_CONF_KEY, HConstants.CIPHER_AES);
173    hcd.setEncryptionType(algorithm);
174    hcd.setEncryptionKey(EncryptionUtil.wrapKey(conf, "hbase", initialCFKey));
175    htd.addFamily(hcd);
176
177    // Create the table and some on disk files
178    createTableAndFlush(htd);
179
180    // Verify we have store file(s) with the initial key
181    List<Path> storeFilePaths = findStorefilePaths(htd.getTableName());
182    assertTrue(storeFilePaths.size() > 0);
183    for (Path path: storeFilePaths) {
184      assertTrue("Store file " + path + " has incorrect key",
185        Bytes.equals(initialCFKey.getEncoded(), extractHFileKey(path)));
186    }
187
188    // Now shut down the HBase cluster
189    TEST_UTIL.shutdownMiniHBaseCluster();
190
191    // "Rotate" the master key
192    conf.set(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, "other");
193    conf.set(HConstants.CRYPTO_MASTERKEY_ALTERNATE_NAME_CONF_KEY, "hbase");
194
195    // Start the cluster back up
196    TEST_UTIL.startMiniHBaseCluster(1, 1);
197    // Verify the table can still be loaded
198    TEST_UTIL.waitTableAvailable(htd.getTableName(), 5000);
199    // Double check that the store file keys can be unwrapped
200    storeFilePaths = findStorefilePaths(htd.getTableName());
201    assertTrue(storeFilePaths.size() > 0);
202    for (Path path: storeFilePaths) {
203      assertTrue("Store file " + path + " has incorrect key",
204        Bytes.equals(initialCFKey.getEncoded(), extractHFileKey(path)));
205    }
206  }

208  private static void waitForCompaction(TableName tableName)
209      throws IOException, InterruptedException {
210    boolean compacted = false;
211    for (Region region : TEST_UTIL.getRSForFirstRegionInTable(tableName)
212        .getRegions(tableName)) {
213      for (HStore store : ((HRegion) region).getStores()) {
214        compacted = false;
215        while (!compacted) {
216          if (store.getStorefiles() != null) {
217            while (store.getStorefilesCount() != 1) {
218              Thread.sleep(100);
219            }
220            for (HStoreFile storefile : store.getStorefiles()) {
221              if (!storefile.isCompactedAway()) {
222                compacted = true;
223                break;
224              }
225              Thread.sleep(100);
226            }
227          } else {
228            break;
229          }
230        }
231      }
232    }
233  }
234
235  private static List<Path> findStorefilePaths(TableName tableName) throws Exception {
236    List<Path> paths = new ArrayList<>();
237    for (Region region : TEST_UTIL.getRSForFirstRegionInTable(tableName)
238        .getRegions(tableName)) {
239      for (HStore store : ((HRegion) region).getStores()) {
240        for (HStoreFile storefile : store.getStorefiles()) {
241          paths.add(storefile.getPath());
242        }
243      }
244    }
245    return paths;
246  }
247
248  private static List<Path> findCompactedStorefilePaths(TableName tableName) throws Exception {
249    List<Path> paths = new ArrayList<>();
250    for (Region region : TEST_UTIL.getRSForFirstRegionInTable(tableName)
251        .getRegions(tableName)) {
252      for (HStore store : ((HRegion) region).getStores()) {
253        Collection<HStoreFile> compactedfiles =
254            store.getStoreEngine().getStoreFileManager().getCompactedfiles();
255        if (compactedfiles != null) {
256          for (HStoreFile storefile : compactedfiles) {
257            paths.add(storefile.getPath());
258          }
259        }
260      }
261    }
262    return paths;
263  }
264
265  private void createTableAndFlush(HTableDescriptor htd) throws Exception {
266    HColumnDescriptor hcd = htd.getFamilies().iterator().next();
267    // Create the test table
268    TEST_UTIL.getAdmin().createTable(htd);
269    TEST_UTIL.waitTableAvailable(htd.getTableName(), 5000);
270    // Create a store file
271    Table table = TEST_UTIL.getConnection().getTable(htd.getTableName());
272    try {
273      table.put(new Put(Bytes.toBytes("testrow"))
274              .addColumn(hcd.getName(), Bytes.toBytes("q"), Bytes.toBytes("value")));
275    } finally {
276      table.close();
277    }
278    TEST_UTIL.getAdmin().flush(htd.getTableName());
279  }
280
281  private static byte[] extractHFileKey(Path path) throws Exception {
282    HFile.Reader reader = HFile.createReader(TEST_UTIL.getTestFileSystem(), path,
283      new CacheConfig(conf), true, conf);
284    try {
285      reader.loadFileInfo();
286      Encryption.Context cryptoContext = reader.getFileContext().getEncryptionContext();
287      assertNotNull("Reader has a null crypto context", cryptoContext);
288      Key key = cryptoContext.getKey();
289      assertNotNull("Crypto context has no key", key);
290      return key.getEncoded();
291    } finally {
292      reader.close();
293    }
294  }
