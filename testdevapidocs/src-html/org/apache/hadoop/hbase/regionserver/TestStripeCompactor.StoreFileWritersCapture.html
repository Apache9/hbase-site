092  @Test
093  public void testBoundaryCompactions() throws Exception {
094    // General verification
095    verifyBoundaryCompaction(a(KV_A, KV_A, KV_B, KV_B, KV_C, KV_D),
096        a(OPEN_KEY, KEY_B, KEY_D, OPEN_KEY), a(a(KV_A, KV_A), a(KV_B, KV_B, KV_C), a(KV_D)));
097    verifyBoundaryCompaction(a(KV_B, KV_C), a(KEY_B, KEY_C, KEY_D), a(a(KV_B), a(KV_C)));
098    verifyBoundaryCompaction(a(KV_B, KV_C), a(KEY_B, KEY_D), new KeyValue[][] { a(KV_B, KV_C) });
099  }

101  @Test
102  public void testBoundaryCompactionEmptyFiles() throws Exception {
103    // No empty file if there're already files.
104    verifyBoundaryCompaction(
105        a(KV_B), a(KEY_B, KEY_C, KEY_D, OPEN_KEY), a(a(KV_B), null, null), null, null, false);
106    verifyBoundaryCompaction(a(KV_A, KV_C),
107        a(OPEN_KEY, KEY_B, KEY_C, KEY_D), a(a(KV_A), null, a(KV_C)), null, null, false);
108    // But should be created if there are no file.
109    verifyBoundaryCompaction(
110        e(), a(OPEN_KEY, KEY_B, KEY_C, OPEN_KEY), a(null, null, e()), null, null, false);
111    // In major range if there's major range.
112    verifyBoundaryCompaction(
113        e(), a(OPEN_KEY, KEY_B, KEY_C, OPEN_KEY), a(null, e(), null), KEY_B, KEY_C, false);
114    verifyBoundaryCompaction(
115        e(), a(OPEN_KEY, KEY_B, KEY_C, OPEN_KEY), a(e(), e(), null), OPEN_KEY, KEY_C, false);
116    // Major range should have files regardless of KVs.
117    verifyBoundaryCompaction(a(KV_A), a(OPEN_KEY, KEY_B, KEY_C, KEY_D, OPEN_KEY),
118        a(a(KV_A), e(), e(), null), KEY_B, KEY_D, false);
119    verifyBoundaryCompaction(a(KV_C), a(OPEN_KEY, KEY_B, KEY_C, KEY_D, OPEN_KEY),
120        a(null, null, a(KV_C), e()), KEY_C, OPEN_KEY, false);
121
122  }

124  public static void verifyBoundaryCompaction(
125      KeyValue[] input, byte[][] boundaries, KeyValue[][] output) throws Exception {
126    verifyBoundaryCompaction(input, boundaries, output, null, null, true);
127  }

129  public static void verifyBoundaryCompaction(KeyValue[] input, byte[][] boundaries,
130      KeyValue[][] output, byte[] majorFrom, byte[] majorTo, boolean allFiles)
131          throws Exception {
132    StoreFileWritersCapture writers = new StoreFileWritersCapture();
133    StripeCompactor sc = createCompactor(writers, input);
134    List<Path> paths =
135        sc.compact(createDummyRequest(), Arrays.asList(boundaries), majorFrom, majorTo,
136          NoLimitCompactionThroughputController.INSTANCE);
137    writers.verifyKvs(output, allFiles, true);
138    if (allFiles) {
139      assertEquals(output.length, paths.size());
140      writers.verifyBoundaries(boundaries);
141    }
142  }

144  @Test
145  public void testSizeCompactions() throws Exception {
146    // General verification with different sizes.
147    verifySizeCompaction(a(KV_A, KV_A, KV_B, KV_C, KV_D), 3, 2, OPEN_KEY, OPEN_KEY,
148        a(a(KV_A, KV_A), a(KV_B, KV_C), a(KV_D)));
149    verifySizeCompaction(a(KV_A, KV_B, KV_C, KV_D), 4, 1, OPEN_KEY, OPEN_KEY,
150        a(a(KV_A), a(KV_B), a(KV_C), a(KV_D)));
151    verifySizeCompaction(a(KV_B, KV_C), 2, 1, KEY_B, KEY_D, a(a(KV_B), a(KV_C)));
152    // Verify row boundaries are preserved.
153    verifySizeCompaction(a(KV_A, KV_A, KV_A, KV_C, KV_D), 3, 2, OPEN_KEY, OPEN_KEY,
154        a(a(KV_A, KV_A, KV_A), a(KV_C, KV_D)));
155    verifySizeCompaction(a(KV_A, KV_B, KV_B, KV_C), 3, 1, OPEN_KEY, OPEN_KEY,
156        a(a(KV_A), a(KV_B, KV_B), a(KV_C)));
157    // Too much data, count limits the number of files.
158    verifySizeCompaction(a(KV_A, KV_B, KV_C, KV_D), 2, 1, OPEN_KEY, OPEN_KEY,
159        a(a(KV_A), a(KV_B, KV_C, KV_D)));
160    verifySizeCompaction(a(KV_A, KV_B, KV_C), 1, Long.MAX_VALUE, OPEN_KEY, KEY_D,
161        new KeyValue[][] { a(KV_A, KV_B, KV_C) });
162    // Too little data/large count, no extra files.
163    verifySizeCompaction(a(KV_A, KV_B, KV_C, KV_D), Integer.MAX_VALUE, 2, OPEN_KEY, OPEN_KEY,
164        a(a(KV_A, KV_B), a(KV_C, KV_D)));
165  }

167  public static void verifySizeCompaction(KeyValue[] input, int targetCount, long targetSize,
168      byte[] left, byte[] right, KeyValue[][] output) throws Exception {
169    StoreFileWritersCapture writers = new StoreFileWritersCapture();
170    StripeCompactor sc = createCompactor(writers, input);
171    List<Path> paths =
172        sc.compact(createDummyRequest(), targetCount, targetSize, left, right, null, null,
173          NoLimitCompactionThroughputController.INSTANCE);
174    assertEquals(output.length, paths.size());
175    writers.verifyKvs(output, true, true);
176    List<byte[]> boundaries = new ArrayList<byte[]>();
177    boundaries.add(left);
178    for (int i = 1; i < output.length; ++i) {
179      boundaries.add(CellUtil.cloneRow(output[i][0]));
180    }
181    boundaries.add(right);
182    writers.verifyBoundaries(boundaries.toArray(new byte[][] {}));
183  }

185  private static StripeCompactor createCompactor(
186      StoreFileWritersCapture writers, KeyValue[] input) throws Exception {
187    Configuration conf = HBaseConfiguration.create();
188    final Scanner scanner = new Scanner(input);
189
190    // Create store mock that is satisfactory for compactor.
191    HColumnDescriptor col = new HColumnDescriptor(NAME_OF_THINGS);
192    ScanInfo si = new ScanInfo(conf, col, Long.MAX_VALUE, 0, CellComparator.COMPARATOR);
193    Store store = mock(Store.class);
194    when(store.getFamily()).thenReturn(col);
195    when(store.getScanInfo()).thenReturn(si);
196    when(store.areWritesEnabled()).thenReturn(true);
197    when(store.getFileSystem()).thenReturn(mock(FileSystem.class));
198    when(store.getRegionInfo()).thenReturn(new HRegionInfo(TABLE_NAME));
199    when(store.createWriterInTmp(anyLong(), any(Compression.Algorithm.class),
200        anyBoolean(), anyBoolean(), anyBoolean(), anyBoolean())).thenAnswer(writers);
201    when(store.getComparator()).thenReturn(CellComparator.COMPARATOR);
202
203    return new StripeCompactor(conf, store) {
204      @Override
205      protected InternalScanner createScanner(Store store, List<StoreFileScanner> scanners,
206          long smallestReadPoint, long earliestPutTs, byte[] dropDeletesFromRow,
207          byte[] dropDeletesToRow) throws IOException {
208        return scanner;
209      }
210
211      @Override
212      protected InternalScanner createScanner(Store store, List<StoreFileScanner> scanners,
213          ScanType scanType, long smallestReadPoint, long earliestPutTs) throws IOException {
214        return scanner;
215      }
216    };
217  }

219  private static CompactionRequest createDummyRequest() throws Exception {
220    // "Files" are totally unused, it's Scanner class below that gives compactor fake KVs.
221    // But compaction depends on everything under the sun, so stub everything with dummies.
222    StoreFile sf = mock(StoreFile.class);
223    StoreFile.Reader r = mock(StoreFile.Reader.class);
224    when(r.length()).thenReturn(1L);
225    when(r.getBloomFilterType()).thenReturn(BloomType.NONE);
226    when(r.getHFileReader()).thenReturn(mock(HFile.Reader.class));
227    when(r.getStoreFileScanner(anyBoolean(), anyBoolean(), anyBoolean(), anyLong()))
228      .thenReturn(mock(StoreFileScanner.class));
229    when(sf.getReader()).thenReturn(r);
230    when(sf.createReader()).thenReturn(r);
231    when(sf.createReader(anyBoolean())).thenReturn(r);
232    return new CompactionRequest(Arrays.asList(sf));
233  }

235  private static class Scanner implements InternalScanner {
236    private final ArrayList<KeyValue> kvs;
237    public Scanner(KeyValue... kvs) {
238      this.kvs = new ArrayList<KeyValue>(Arrays.asList(kvs));
239    }
240
241    @Override
242    public boolean next(List<Cell> results) throws IOException {
243      if (kvs.isEmpty()) return false;
244      results.add(kvs.remove(0));
245      return !kvs.isEmpty();
246    }
247
248    @Override
249    public boolean next(List<Cell> result, ScannerContext scannerContext)
250        throws IOException {
251      return next(result);
252    }
253
254    @Override
255    public void close() throws IOException {}
256  }

258  // StoreFile.Writer has private ctor and is unwieldy, so this has to be convoluted.
259  public static class StoreFileWritersCapture implements
260    Answer<StoreFile.Writer>, StripeMultiFileWriter.WriterFactory {
261    public static class Writer {
262      public ArrayList<KeyValue> kvs = new ArrayList<KeyValue>();
263      public TreeMap<byte[], byte[]> data = new TreeMap<byte[], byte[]>(Bytes.BYTES_COMPARATOR);
264    }
265
266    private List<Writer> writers = new ArrayList<Writer>();
267
268    @Override
269    public StoreFile.Writer createWriter() throws IOException {
270      final Writer realWriter = new Writer();
271      writers.add(realWriter);
272      StoreFile.Writer writer = mock(StoreFile.Writer.class);
273      doAnswer(new Answer<Object>() {
274        public Object answer(InvocationOnMock invocation) {
275          return realWriter.kvs.add((KeyValue)invocation.getArguments()[0]);
276        }}).when(writer).append(any(KeyValue.class));
277      doAnswer(new Answer<Object>() {
278        public Object answer(InvocationOnMock invocation) {
279          Object[] args = invocation.getArguments();
280          return realWriter.data.put((byte[])args[0], (byte[])args[1]);
281        }}).when(writer).appendFileInfo(any(byte[].class), any(byte[].class));
282      return writer;
283    }
284
285    @Override
286    public StoreFile.Writer answer(InvocationOnMock invocation) throws Throwable {
287      return createWriter();
288    }
289
290    public void verifyKvs(KeyValue[][] kvss, boolean allFiles, boolean requireMetadata) {
291      if (allFiles) {
292        assertEquals(kvss.length, writers.size());
293      }
294      int skippedWriters = 0;
295      for (int i = 0; i < kvss.length; ++i) {
296        KeyValue[] kvs = kvss[i];
297        if (kvs != null) {
298          Writer w = writers.get(i - skippedWriters);
299          if (requireMetadata) {
300            assertNotNull(w.data.get(STRIPE_START_KEY));
301            assertNotNull(w.data.get(STRIPE_END_KEY));
302          } else {
303            assertNull(w.data.get(STRIPE_START_KEY));
304            assertNull(w.data.get(STRIPE_END_KEY));
305          }
306          assertEquals(kvs.length, w.kvs.size());
307          for (int j = 0; j < kvs.length; ++j) {
308            assertEquals(kvs[j], w.kvs.get(j));
309          }
310        } else {
311          assertFalse(allFiles);
312          ++skippedWriters;
313        }
314      }
315    }
316
317    public void verifyBoundaries(byte[][] boundaries) {
318      assertEquals(boundaries.length - 1, writers.size());
319      for (int i = 0; i < writers.size(); ++i) {
320        assertArrayEquals("i = " + i, boundaries[i], writers.get(i).data.get(STRIPE_START_KEY));
321        assertArrayEquals("i = " + i, boundaries[i + 1], writers.get(i).data.get(STRIPE_END_KEY));
322      }
323    }
