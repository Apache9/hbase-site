092  /**
093   * Run a bunch of threads against a single TimeRangeTracker and ensure we arrive
094   * at right range.  Here we do ten threads each incrementing over 100k at an offset
095   * of the thread index; max is 10 * 10k and min is 0.
096   * @throws InterruptedException
097   */
098  @Test
099  public void testArriveAtRightAnswer() throws InterruptedException {
100    final TimeRangeTracker trr = new TimeRangeTracker();
101    final int threadCount = 10;
102    final int calls = 1000 * 1000;
103    Thread [] threads = new Thread[threadCount];
104    for (int i = 0; i < threads.length; i++) {
105      Thread t = new Thread("" + i) {
106        @Override
107        public void run() {
108          int offset = Integer.parseInt(getName());
109          boolean even = offset % 2 == 0;
110          if (even) {
111            for (int i = (offset * calls); i < calls; i++) trr.includeTimestamp(i);
112          } else {
113            int base = offset * calls;
114            for (int i = base + calls; i >= base; i--) trr.includeTimestamp(i);
115          }
116        }
117      };
118      t.start();
119      threads[i] = t;
120    }
121    for (int i = 0; i < threads.length; i++) {
122      threads[i].join();
123    }
124
125    assertTrue(trr.getMax() == calls * threadCount);
126    assertTrue(trr.getMin() == 0);
127  }

162  final static int NUM_OF_THREADS = 20;
163
164  class RandomTestData {
165    private long[] keys = new long[NUM_KEYS];
166    private long min = Long.MAX_VALUE;
167    private long max = 0;
168
169    public RandomTestData() {
170      if (ThreadLocalRandom.current().nextInt(NUM_OF_THREADS) % 2 == 0) {
171        for (int i = 0; i < NUM_KEYS; i++) {
172          keys[i] = i + ThreadLocalRandom.current().nextLong(NUM_OF_THREADS);
173          if (keys[i] < min) min = keys[i];
174          if (keys[i] > max) max = keys[i];
175        }
176      } else {
177        for (int i = NUM_KEYS - 1; i >= 0; i--) {
178          keys[i] = i + ThreadLocalRandom.current().nextLong(NUM_OF_THREADS);
179          if (keys[i] < min) min = keys[i];
180          if (keys[i] > max) max = keys[i];
181        }
182      }
183    }
184
185    public long getMax() {
186      return this.max;
187    }
188
189    public long getMin() {
190      return this.min;
191    }
192  }
193
194  class TrtUpdateRunnable implements Runnable {
195
196    private TimeRangeTracker trt;
197    private RandomTestData data;
198    public TrtUpdateRunnable(final TimeRangeTracker trt, final RandomTestData data) {
199      this.trt = trt;
200      this.data = data;
201    }
202
203    public void run() {
204      for (long key : data.keys) {
205        trt.includeTimestamp(key);
206      }
207    }
208  }

210  /**
211   * Run a bunch of threads against a single TimeRangeTracker and ensure we arrive
212   * at right range.  The data chosen is going to ensure that there are lots collisions, i.e,
213   * some other threads may already update the value while one tries to update min/max value.
214   */
215  @Test
216  public void testConcurrentIncludeTimestampCorrectness() {
217    RandomTestData[] testData = new RandomTestData[NUM_OF_THREADS];
218    long min = Long.MAX_VALUE, max = 0;
219    for (int i = 0; i < NUM_OF_THREADS; i ++) {
220      testData[i] = new RandomTestData();
221      if (testData[i].getMin() < min) {
222        min = testData[i].getMin();
223      }
224      if (testData[i].getMax() > max) {
225        max = testData[i].getMax();
226      }
227    }
228
229    TimeRangeTracker trt = new TimeRangeTracker();
230
231    Thread[] t = new Thread[NUM_OF_THREADS];
232    for (int i = 0; i < NUM_OF_THREADS; i++) {
233      t[i] = new Thread(new TrtUpdateRunnable(trt, testData[i]));
234      t[i].start();
235    }
236
237    for (Thread thread : t) {
238      try {
239        thread.join();
240      } catch (InterruptedException e) {
241        e.printStackTrace();
242      }
243    }
244
245    assertTrue(min == trt.getMin());
246    assertTrue(max == trt.getMax());
247  }

254  public static void main(String[] args) throws InterruptedException {
255    long start = System.currentTimeMillis();
256    final TimeRangeTracker trr = new TimeRangeTracker();
257    final int threadCount = 5;
258    final int calls = 1024 * 1024 * 128;
259    Thread [] threads = new Thread[threadCount];
260    for (int i = 0; i < threads.length; i++) {
261      Thread t = new Thread("" + i) {
262        @Override
263        public void run() {
264          for (int i = 0; i < calls; i++) trr.includeTimestamp(i);
265        }
266      };
267      t.start();
268      threads[i] = t;
269    }
270    for (int i = 0; i < threads.length; i++) {
271      threads[i].join();
272    }
273    System.out.println(trr.getMin() + " " + trr.getMax() + " " +
274      (System.currentTimeMillis() - start));
