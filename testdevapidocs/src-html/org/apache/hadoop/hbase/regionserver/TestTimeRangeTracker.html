032@Category({RegionServerTests.class, SmallTests.class})
033public class TestTimeRangeTracker {
034  @Test
035  public void testAlwaysDecrementingSetsMaximum() {
036    TimeRangeTracker trr = new TimeRangeTracker();
037    trr.includeTimestamp(3);
038    trr.includeTimestamp(2);
039    trr.includeTimestamp(1);
040    assertTrue(trr.getMin() != TimeRangeTracker.INITIAL_MIN_TIMESTAMP);
041    assertTrue(trr.getMax() != -1 /*The initial max value*/);
042  }
043
044  @Test
045  public void testSimpleInRange() {
046    TimeRangeTracker trr = new TimeRangeTracker();
047    trr.includeTimestamp(0);
048    trr.includeTimestamp(2);
049    assertTrue(trr.includesTimeRange(new TimeRange(1)));
050  }
051
052  /**
053   * Run a bunch of threads against a single TimeRangeTracker and ensure we arrive
054   * at right range.  Here we do ten threads each incrementing over 100k at an offset
055   * of the thread index; max is 10 * 10k and min is 0.
056   * @throws InterruptedException
057   */
058  @Test
059  public void testArriveAtRightAnswer() throws InterruptedException {
060    final TimeRangeTracker trr = new TimeRangeTracker();
061    final int threadCount = 10;
062    final int calls = 1000 * 1000;
063    Thread [] threads = new Thread[threadCount];
064    for (int i = 0; i < threads.length; i++) {
065      Thread t = new Thread("" + i) {
066        @Override
067        public void run() {
068          int offset = Integer.parseInt(getName());
069          boolean even = offset % 2 == 0;
070          if (even) {
071            for (int i = (offset * calls); i < calls; i++) trr.includeTimestamp(i);
072          } else {
073            int base = offset * calls;
074            for (int i = base + calls; i >= base; i--) trr.includeTimestamp(i);
075          }
076        }
077      };
078      t.start();
079      threads[i] = t;
080    }
081    for (int i = 0; i < threads.length; i++) {
082      threads[i].join();
083    }
084    assertTrue(trr.getMax() == calls * threadCount);
085    assertTrue(trr.getMin() == 0);
086  }
087
088  @Test
089  public void testRangeConstruction() throws IOException {
090    TimeRange defaultRange = new TimeRange();
091    assertEquals(0L, defaultRange.getMin());
092    assertEquals(Long.MAX_VALUE, defaultRange.getMax());
093    assertTrue(defaultRange.isAllTime());
094
095    TimeRange oneArgRange = new TimeRange(0L);
096    assertEquals(0L, oneArgRange.getMin());
097    assertEquals(Long.MAX_VALUE, oneArgRange.getMax());
098    assertTrue(oneArgRange.isAllTime());
099
100    TimeRange oneArgRange2 = new TimeRange(1);
101    assertEquals(1, oneArgRange2.getMin());
102    assertEquals(Long.MAX_VALUE, oneArgRange2.getMax());
103    assertFalse(oneArgRange2.isAllTime());
104
105    TimeRange twoArgRange = new TimeRange(0L, Long.MAX_VALUE);
106    assertEquals(0L, twoArgRange.getMin());
107    assertEquals(Long.MAX_VALUE, twoArgRange.getMax());
108    assertTrue(twoArgRange.isAllTime());
109
110    TimeRange twoArgRange2 = new TimeRange(0L, Long.MAX_VALUE - 1);
111    assertEquals(0L, twoArgRange2.getMin());
112    assertEquals(Long.MAX_VALUE - 1, twoArgRange2.getMax());
113    assertFalse(twoArgRange2.isAllTime());
114
115    TimeRange twoArgRange3 = new TimeRange(1, Long.MAX_VALUE);
116    assertEquals(1, twoArgRange3.getMin());
117    assertEquals(Long.MAX_VALUE, twoArgRange3.getMax());
118    assertFalse(twoArgRange3.isAllTime());
119  }

121  /**
122   * Bit of code to test concurrent access on this class.
123   * @param args
124   * @throws InterruptedException
125   */
126  public static void main(String[] args) throws InterruptedException {
127    long start = System.currentTimeMillis();
128    final TimeRangeTracker trr = new TimeRangeTracker();
129    final int threadCount = 5;
130    final int calls = 1024 * 1024 * 128;
131    Thread [] threads = new Thread[threadCount];
132    for (int i = 0; i < threads.length; i++) {
133      Thread t = new Thread("" + i) {
134        @Override
135        public void run() {
136          for (int i = 0; i < calls; i++) trr.includeTimestamp(i);
137        }
138      };
139      t.start();
140      threads[i] = t;
141    }
142    for (int i = 0; i < threads.length; i++) {
143      threads[i].join();
144    }
145    System.out.println(trr.getMin() + " " + trr.getMax() + " " +
146      (System.currentTimeMillis() - start));
147  }
