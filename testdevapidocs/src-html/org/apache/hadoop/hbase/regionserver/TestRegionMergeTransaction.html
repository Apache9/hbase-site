082  @Before
083  public void setup() throws IOException {
084    this.fs = FileSystem.get(TEST_UTIL.getConfiguration());

089    this.region_a = createRegion(this.testdir, this.wals, STARTROW_A, STARTROW_B);
090    this.region_b = createRegion(this.testdir, this.wals, STARTROW_B, STARTROW_C);
091    this.region_c = createRegion(this.testdir, this.wals, STARTROW_C, ENDROW);

096  @After
097  public void teardown() throws IOException {
098    for (HRegion region : new HRegion[] { region_a, region_b, region_c }) {
099      if (region != null && !region.isClosed()) region.close();
100      if (this.fs.exists(region.getRegionFileSystem().getRegionDir())
101          && !this.fs.delete(region.getRegionFileSystem().getRegionDir(), true)) {
102        throw new IOException("Failed deleting of "
103            + region.getRegionFileSystem().getRegionDir());
104      }
105    }
106    if (this.wals != null) {
107      this.wals.close();
108    }
109    this.fs.delete(this.testdir, true);
110  }

214  @Test
215  public void testWholesomeMerge() throws IOException, InterruptedException {
216    final int rowCountOfRegionA = loadRegion(this.region_a, CF, true);
217    final int rowCountOfRegionB = loadRegion(this.region_b, CF, true);
218    assertTrue(rowCountOfRegionA > 0 && rowCountOfRegionB > 0);
219    assertEquals(rowCountOfRegionA, countRows(this.region_a));
220    assertEquals(rowCountOfRegionB, countRows(this.region_b));
221
222    // Start transaction.
223    RegionMergeTransactionImpl mt = prepareOnGoodRegions();
224
225    // Run the execute. Look at what it returns.
226    TEST_UTIL.getConfiguration().setInt(HConstants.REGIONSERVER_PORT, 0);
227    CoordinatedStateManager cp = CoordinatedStateManagerFactory.getCoordinatedStateManager(
228      TEST_UTIL.getConfiguration());
229    Server mockServer = new HRegionServer(TEST_UTIL.getConfiguration(), cp);
230    HRegion mergedRegion = (HRegion)mt.execute(mockServer, null);
231    // Do some assertions about execution.
232    assertTrue(this.fs.exists(mt.getMergesDir()));
233    // Assert region_a and region_b is closed.
234    assertTrue(region_a.isClosed());
235    assertTrue(region_b.isClosed());
236
237    // Assert mergedir is empty -- because its content will have been moved out
238    // to be under the merged region dirs.
239    assertEquals(0, this.fs.listStatus(mt.getMergesDir()).length);
240    // Check merged region have correct key span.
241    assertTrue(Bytes.equals(this.region_a.getRegionInfo().getStartKey(),
242        mergedRegion.getRegionInfo().getStartKey()));
243    assertTrue(Bytes.equals(this.region_b.getRegionInfo().getEndKey(),
244        mergedRegion.getRegionInfo().getEndKey()));
245    // Count rows. merged region are already open
246    try {
247      int mergedRegionRowCount = countRows(mergedRegion);
248      assertEquals((rowCountOfRegionA + rowCountOfRegionB),
249          mergedRegionRowCount);
250    } finally {
251      HBaseTestingUtility.closeRegionAndWAL(mergedRegion);
252    }
253    // Assert the write lock is no longer held on region_a and region_b
254    assertTrue(!this.region_a.lock.writeLock().isHeldByCurrentThread());
255    assertTrue(!this.region_b.lock.writeLock().isHeldByCurrentThread());
256  }

258  @Test
259  public void testRollback() throws IOException, InterruptedException {
260    final int rowCountOfRegionA = loadRegion(this.region_a, CF, true);
261    final int rowCountOfRegionB = loadRegion(this.region_b, CF, true);
262    assertTrue(rowCountOfRegionA > 0 && rowCountOfRegionB > 0);
263    assertEquals(rowCountOfRegionA, countRows(this.region_a));
264    assertEquals(rowCountOfRegionB, countRows(this.region_b));
265
266    // Start transaction.
267    RegionMergeTransactionImpl mt = prepareOnGoodRegions();
268
269    when(mt.createMergedRegionFromMerges(region_a, region_b,
270        mt.getMergedRegionInfo())).thenThrow(
271        new MockedFailedMergedRegionCreation());
272
273    // Run the execute. Look at what it returns.
274    boolean expectedException = false;
275    TEST_UTIL.getConfiguration().setInt(HConstants.REGIONSERVER_PORT, 0);
276    CoordinatedStateManager cp = CoordinatedStateManagerFactory.getCoordinatedStateManager(
277      TEST_UTIL.getConfiguration());
278    Server mockServer = new HRegionServer(TEST_UTIL.getConfiguration(), cp);
279    try {
280      mt.execute(mockServer, null);
281    } catch (MockedFailedMergedRegionCreation e) {
282      expectedException = true;
283    }
284    assertTrue(expectedException);
285    // Run rollback
286    assertTrue(mt.rollback(null, null));
287
288    // Assert I can scan region_a and region_b.
289    int rowCountOfRegionA2 = countRows(this.region_a);
290    assertEquals(rowCountOfRegionA, rowCountOfRegionA2);
291    int rowCountOfRegionB2 = countRows(this.region_b);
292    assertEquals(rowCountOfRegionB, rowCountOfRegionB2);
293
294    // Assert rollback cleaned up stuff in fs
295    assertTrue(!this.fs.exists(HRegion.getRegionDir(this.testdir,
296        mt.getMergedRegionInfo())));
297
298    assertTrue(!this.region_a.lock.writeLock().isHeldByCurrentThread());
299    assertTrue(!this.region_b.lock.writeLock().isHeldByCurrentThread());
300
301    // Now retry the merge but do not throw an exception this time.
302    assertTrue(mt.prepare(null));
303    HRegion mergedRegion = (HRegion)mt.execute(mockServer, null);
304    // Count rows. daughters are already open
305    // Count rows. merged region are already open
306    try {
307      int mergedRegionRowCount = countRows(mergedRegion);
308      assertEquals((rowCountOfRegionA + rowCountOfRegionB),
309          mergedRegionRowCount);
310    } finally {
311      HBaseTestingUtility.closeRegionAndWAL(mergedRegion);
312    }
313    // Assert the write lock is no longer held on region_a and region_b
314    assertTrue(!this.region_a.lock.writeLock().isHeldByCurrentThread());
315    assertTrue(!this.region_b.lock.writeLock().isHeldByCurrentThread());
316  }

318  @Test
319  public void testFailAfterPONR() throws IOException, KeeperException, InterruptedException {
320    final int rowCountOfRegionA = loadRegion(this.region_a, CF, true);
321    final int rowCountOfRegionB = loadRegion(this.region_b, CF, true);
322    assertTrue(rowCountOfRegionA > 0 && rowCountOfRegionB > 0);
323    assertEquals(rowCountOfRegionA, countRows(this.region_a));
324    assertEquals(rowCountOfRegionB, countRows(this.region_b));
325
326    // Start transaction.
327    RegionMergeTransactionImpl mt = prepareOnGoodRegions();
328    Mockito.doThrow(new MockedFailedMergedRegionOpen())
329        .when(mt)
330        .openMergedRegion((Server) Mockito.anyObject(),
331            (RegionServerServices) Mockito.anyObject(),
332            (HRegion) Mockito.anyObject());
333
334    // Run the execute. Look at what it returns.
335    boolean expectedException = false;
336    TEST_UTIL.getConfiguration().setInt(HConstants.REGIONSERVER_PORT, 0);
337    CoordinatedStateManager cp = CoordinatedStateManagerFactory.getCoordinatedStateManager(
338      TEST_UTIL.getConfiguration());
339    Server mockServer = new HRegionServer(TEST_UTIL.getConfiguration(), cp);
340    try {
341      mt.execute(mockServer, null);
342    } catch (MockedFailedMergedRegionOpen e) {
343      expectedException = true;
344    }
345    assertTrue(expectedException);
346    // Run rollback returns false that we should restart.
347    assertFalse(mt.rollback(null, null));
348    // Make sure that merged region is still in the filesystem, that
349    // they have not been removed; this is supposed to be the case if we go
350    // past point of no return.
351    Path tableDir = this.region_a.getRegionFileSystem().getRegionDir()
352        .getParent();
353    Path mergedRegionDir = new Path(tableDir, mt.getMergedRegionInfo()
354        .getEncodedName());
355    assertTrue(TEST_UTIL.getTestFileSystem().exists(mergedRegionDir));
356  }

358  @Test
359  public void testMeregedRegionBoundary() {
360    TableName tableName =
361        TableName.valueOf("testMeregedRegionBoundary");
362    byte[] a = Bytes.toBytes("a");
363    byte[] b = Bytes.toBytes("b");
364    byte[] z = Bytes.toBytes("z");
365    HRegionInfo r1 = new HRegionInfo(tableName);
366    HRegionInfo r2 = new HRegionInfo(tableName, a, z);
367    HRegionInfo m = RegionMergeTransactionImpl.getMergedRegionInfo(r1, r2);
368    assertTrue(Bytes.equals(m.getStartKey(), r1.getStartKey())
369        && Bytes.equals(m.getEndKey(), r1.getEndKey()));
370
371    r1 = new HRegionInfo(tableName, null, a);
372    r2 = new HRegionInfo(tableName, a, z);
373    m = RegionMergeTransactionImpl.getMergedRegionInfo(r1, r2);
374    assertTrue(Bytes.equals(m.getStartKey(), r1.getStartKey())
375        && Bytes.equals(m.getEndKey(), r2.getEndKey()));
376
377    r1 = new HRegionInfo(tableName, null, a);
378    r2 = new HRegionInfo(tableName, z, null);
379    m = RegionMergeTransactionImpl.getMergedRegionInfo(r1, r2);
380    assertTrue(Bytes.equals(m.getStartKey(), r1.getStartKey())
381        && Bytes.equals(m.getEndKey(), r2.getEndKey()));
382
383    r1 = new HRegionInfo(tableName, a, z);
384    r2 = new HRegionInfo(tableName, z, null);
385    m = RegionMergeTransactionImpl.getMergedRegionInfo(r1, r2);
386    assertTrue(Bytes.equals(m.getStartKey(), r1.getStartKey())
387      && Bytes.equals(m.getEndKey(), r2.getEndKey()));

407  private HRegion createRegion(final Path testdir, final WALFactory wals,
408      final byte[] startrow, final byte[] endrow)
409      throws IOException {
410    // Make a region with start and end keys.
411    HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("table"));
412    HColumnDescriptor hcd = new HColumnDescriptor(CF);
413    htd.addFamily(hcd);
414    HRegionInfo hri = new HRegionInfo(htd.getTableName(), startrow, endrow);
415    HRegion a = HBaseTestingUtility.createRegionAndWAL(hri, testdir,
416        TEST_UTIL.getConfiguration(), htd);
417    HBaseTestingUtility.closeRegionAndWAL(a);
418    return HRegion.openHRegion(testdir, hri, htd,
419      wals.getWAL(hri.getEncodedNameAsBytes(), hri.getTable().getNamespace()),
420      TEST_UTIL.getConfiguration());
421  }

423  private int countRows(final HRegion r) throws IOException {
424    int rowcount = 0;
425    InternalScanner scanner = r.getScanner(new Scan());
426    try {
427      List<Cell> kvs = new ArrayList<Cell>();
428      boolean hasNext = true;
429      while (hasNext) {
430        hasNext = scanner.next(kvs);
431        if (!kvs.isEmpty())
432          rowcount++;
433      }
434    } finally {
435      scanner.close();
436    }
437    return rowcount;
438  }

440  /**
441   * Load region with rows from 'aaa' to 'zzz', skip the rows which are out of
442   * range of the region
443   * @param r Region
444   * @param f Family
445   * @param flush flush the cache if true
446   * @return Count of rows loaded.
447   * @throws IOException
448   */
449  private int loadRegion(final HRegion r, final byte[] f, final boolean flush)
450      throws IOException {
451    byte[] k = new byte[3];
452    int rowCount = 0;
453    for (byte b1 = 'a'; b1 <= 'z'; b1++) {
454      for (byte b2 = 'a'; b2 <= 'z'; b2++) {
455        for (byte b3 = 'a'; b3 <= 'z'; b3++) {
456          k[0] = b1;
457          k[1] = b2;
458          k[2] = b3;
459          if (!HRegion.rowIsInRange(r.getRegionInfo(), k)) {
460            continue;
461          }
462          Put put = new Put(k);
463          put.addColumn(f, null, k);
464          if (r.getWAL() == null)
465            put.setDurability(Durability.SKIP_WAL);
466          r.put(put);
467          rowCount++;
468        }
469      }
470      if (flush) {
471        r.flush(true);
472      }
473    }
474    return rowCount;
475  }
