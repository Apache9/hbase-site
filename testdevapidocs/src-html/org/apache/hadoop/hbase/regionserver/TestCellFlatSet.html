039@Category({RegionServerTests.class, SmallTests.class})
040public class TestCellFlatSet extends TestCase {
041  private static final int NUM_OF_CELLS = 4;
042  private Cell ascCells[];
043  private CellArrayMap ascCbOnHeap;
044  private Cell descCells[];
045  private CellArrayMap descCbOnHeap;
046  private final static Configuration CONF = new Configuration();
047  private KeyValue lowerOuterCell;
048  private KeyValue upperOuterCell;
049
050  @Override
051  protected void setUp() throws Exception {
052    super.setUp();
053
054    // create array of Cells to bass to the CellFlatMap under CellSet
055    final byte[] one = Bytes.toBytes(15);
056    final byte[] two = Bytes.toBytes(25);
057    final byte[] three = Bytes.toBytes(35);
058    final byte[] four = Bytes.toBytes(45);
059
060    final byte[] f = Bytes.toBytes("f");
061    final byte[] q = Bytes.toBytes("q");
062    final byte[] v = Bytes.toBytes(4);
063
064    final KeyValue kv1 = new KeyValue(one, f, q, 10, v);
065    final KeyValue kv2 = new KeyValue(two, f, q, 20, v);
066    final KeyValue kv3 = new KeyValue(three, f, q, 30, v);
067    final KeyValue kv4 = new KeyValue(four, f, q, 40, v);
068    lowerOuterCell = new KeyValue(Bytes.toBytes(10), f, q, 10, v);
069    upperOuterCell = new KeyValue(Bytes.toBytes(50), f, q, 10, v);
070    ascCells = new Cell[] {kv1,kv2,kv3,kv4};
071    ascCbOnHeap = new CellArrayMap(CellComparator.COMPARATOR,ascCells,0,NUM_OF_CELLS,false);
072    descCells = new Cell[] {kv4,kv3,kv2,kv1};
073    descCbOnHeap = new CellArrayMap(CellComparator.COMPARATOR,descCells,0,NUM_OF_CELLS,true);
074    CONF.setBoolean(MemStoreLAB.USEMSLAB_KEY, true);
075    CONF.setFloat(MemStoreLAB.CHUNK_POOL_MAXSIZE_KEY, 0.2f);
076    MemStoreChunkPool.chunkPoolDisabled = false;
077  }

079  /* Create and test CellSet based on CellArrayMap */
080  public void testCellBlocksOnHeap() throws Exception {
081    CellSet cs = new CellSet(ascCbOnHeap);
082    testCellBlocks(cs);
083    testIterators(cs);
084  }
085  @Test
086  public void testAsc() throws Exception {
087    CellSet ascCs = new CellSet(ascCbOnHeap);
088    assertEquals(NUM_OF_CELLS, ascCs.size());
089    testSubSet(ascCs);
090  }
091  @Test
092  public void testDesc() throws Exception {
093    CellSet descCs = new CellSet(descCbOnHeap);
094    assertEquals(NUM_OF_CELLS, descCs.size());
095    testSubSet(descCs);
096  }

098  private void testSubSet(CellSet cs) throws Exception {
099    for (int i = 0; i != ascCells.length; ++i) {
100      NavigableSet<Cell> excludeTail = cs.tailSet(ascCells[i], false);
101      NavigableSet<Cell> includeTail = cs.tailSet(ascCells[i], true);
102      assertEquals(ascCells.length - 1 - i, excludeTail.size());
103      assertEquals(ascCells.length - i, includeTail.size());
104      Iterator<Cell> excludeIter = excludeTail.iterator();
105      Iterator<Cell> includeIter = includeTail.iterator();
106      for (int j = 1 + i; j != ascCells.length; ++j) {
107        assertEquals(true, CellUtil.equals(excludeIter.next(), ascCells[j]));
108      }
109      for (int j = i; j != ascCells.length; ++j) {
110        assertEquals(true, CellUtil.equals(includeIter.next(), ascCells[j]));
111      }
112    }
113    assertEquals(NUM_OF_CELLS, cs.tailSet(lowerOuterCell, false).size());
114    assertEquals(0, cs.tailSet(upperOuterCell, false).size());
115    for (int i = 0; i != ascCells.length; ++i) {
116      NavigableSet<Cell> excludeHead = cs.headSet(ascCells[i], false);
117      NavigableSet<Cell> includeHead = cs.headSet(ascCells[i], true);
118      assertEquals(i, excludeHead.size());
119      assertEquals(i + 1, includeHead.size());
120      Iterator<Cell> excludeIter = excludeHead.iterator();
121      Iterator<Cell> includeIter = includeHead.iterator();
122      for (int j = 0; j != i; ++j) {
123        assertEquals(true, CellUtil.equals(excludeIter.next(), ascCells[j]));
124      }
125      for (int j = 0; j != i + 1; ++j) {
126        assertEquals(true, CellUtil.equals(includeIter.next(), ascCells[j]));
127      }
128    }
129    assertEquals(0, cs.headSet(lowerOuterCell, false).size());
130    assertEquals(NUM_OF_CELLS, cs.headSet(upperOuterCell, false).size());
131
132    NavigableMap<Cell, Cell> sub = cs.getDelegatee().subMap(lowerOuterCell, true, upperOuterCell, true);
133    assertEquals(NUM_OF_CELLS, sub.size());
134    Iterator<Cell> iter = sub.values().iterator();
135    for (int i = 0; i != ascCells.length; ++i) {
136      assertEquals(true, CellUtil.equals(iter.next(), ascCells[i]));
137    }
138  }

140  /* Generic basic test for immutable CellSet */
141  private void testCellBlocks(CellSet cs) throws Exception {
142    final byte[] oneAndHalf = Bytes.toBytes(20);
143    final byte[] f = Bytes.toBytes("f");
144    final byte[] q = Bytes.toBytes("q");
145    final byte[] v = Bytes.toBytes(4);
146    final KeyValue outerCell = new KeyValue(oneAndHalf, f, q, 10, v);
147
148    assertEquals(NUM_OF_CELLS, cs.size());          // check size
149    assertFalse(cs.contains(outerCell));            // check outer cell
150
151    assertTrue(cs.contains(ascCells[0]));              // check existence of the first
152    Cell first = cs.first();
153    assertTrue(ascCells[0].equals(first));
154
155    assertTrue(cs.contains(ascCells[NUM_OF_CELLS - 1]));  // check last
156    Cell last = cs.last();
157    assertTrue(ascCells[NUM_OF_CELLS - 1].equals(last));
158
159    SortedSet<Cell> tail = cs.tailSet(ascCells[1]);    // check tail abd head sizes
160    assertEquals(NUM_OF_CELLS - 1, tail.size());
161    SortedSet<Cell> head = cs.headSet(ascCells[1]);
162    assertEquals(1, head.size());
163
164    SortedSet<Cell> tailOuter = cs.tailSet(outerCell);  // check tail starting from outer cell
165    assertEquals(NUM_OF_CELLS - 1, tailOuter.size());
166
167    Cell tailFirst = tail.first();
168    assertTrue(ascCells[1].equals(tailFirst));
169    Cell tailLast = tail.last();
170    assertTrue(ascCells[NUM_OF_CELLS - 1].equals(tailLast));
171
172    Cell headFirst = head.first();
173    assertTrue(ascCells[0].equals(headFirst));
174    Cell headLast = head.last();
175    assertTrue(ascCells[0].equals(headLast));
176  }

178  /* Generic iterators test for immutable CellSet */
179  private void testIterators(CellSet cs) throws Exception {
180
181    // Assert that we have NUM_OF_CELLS values and that they are in order
182    int count = 0;
183    for (Cell kv: cs) {
184      assertEquals("\n\n-------------------------------------------------------------------\n"
185              + "Comparing iteration number " + (count + 1) + " the returned cell: " + kv
186              + ", the first Cell in the CellBlocksMap: " + ascCells[count]
187              + ", and the same transformed to String: " + ascCells[count].toString()
188              + "\n-------------------------------------------------------------------\n",
189              ascCells[count], kv);
190      count++;
191    }
192    assertEquals(NUM_OF_CELLS, count);
193
194    // Test descending iterator
195    count = 0;
196    for (Iterator<Cell> i = cs.descendingIterator(); i.hasNext();) {
197      Cell kv = i.next();
198      assertEquals(ascCells[NUM_OF_CELLS - (count + 1)], kv);
199      count++;
200    }
201    assertEquals(NUM_OF_CELLS, count);
202  }
