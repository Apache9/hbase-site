019package org.apache.hadoop.hbase.regionserver;
020
021import junit.framework.TestCase;
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.*;
024import org.apache.hadoop.hbase.testclassification.RegionServerTests;
025import org.apache.hadoop.hbase.testclassification.SmallTests;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.junit.experimental.categories.Category;
028
029import java.util.Iterator;
030import java.util.NavigableMap;
031import java.util.SortedSet;
032import static org.junit.Assert.assertTrue;
033
034@Category({RegionServerTests.class, SmallTests.class})
035public class TestCellFlatSet extends TestCase {
036
037  private static final int NUM_OF_CELLS = 4;
038
039  private Cell cells[];
040  private CellArrayMap cbOnHeap;
041
042  private final static Configuration conf = new Configuration();
043  private HeapMemStoreLAB mslab;
044
045
046  protected void setUp() throws Exception {
047    super.setUp();
048
049    // create array of Cells to bass to the CellFlatMap under CellSet
050    final byte[] one = Bytes.toBytes(15);
051    final byte[] two = Bytes.toBytes(25);
052    final byte[] three = Bytes.toBytes(35);
053    final byte[] four = Bytes.toBytes(45);
054
055    final byte[] f = Bytes.toBytes("f");
056    final byte[] q = Bytes.toBytes("q");
057    final byte[] v = Bytes.toBytes(4);
058
059    final KeyValue kv1 = new KeyValue(one, f, q, 10, v);
060    final KeyValue kv2 = new KeyValue(two, f, q, 20, v);
061    final KeyValue kv3 = new KeyValue(three, f, q, 30, v);
062    final KeyValue kv4 = new KeyValue(four, f, q, 40, v);
063
064    cells = new Cell[] {kv1,kv2,kv3,kv4};
065    cbOnHeap = new CellArrayMap(CellComparator.COMPARATOR,cells,0,NUM_OF_CELLS,false);
066
067    conf.setBoolean(SegmentFactory.USEMSLAB_KEY, true);
068    conf.setFloat(MemStoreChunkPool.CHUNK_POOL_MAXSIZE_KEY, 0.2f);
069    MemStoreChunkPool.chunkPoolDisabled = false;
070    mslab = new HeapMemStoreLAB(conf);
071  }
072
073  /* Create and test CellSet based on CellArrayMap */
074  public void testCellBlocksOnHeap() throws Exception {
075    CellSet cs = new CellSet(cbOnHeap);
076    testCellBlocks(cs);
077    testIterators(cs);
078  }
079
080  /* Generic basic test for immutable CellSet */
081  private void testCellBlocks(CellSet cs) throws Exception {
082    final byte[] oneAndHalf = Bytes.toBytes(20);
083    final byte[] f = Bytes.toBytes("f");
084    final byte[] q = Bytes.toBytes("q");
085    final byte[] v = Bytes.toBytes(4);
086    final KeyValue outerCell = new KeyValue(oneAndHalf, f, q, 10, v);
087
088    assertEquals(NUM_OF_CELLS, cs.size());          // check size
089    assertFalse(cs.contains(outerCell));            // check outer cell
090
091    assertTrue(cs.contains(cells[0]));              // check existence of the first
092    Cell first = cs.first();
093    assertTrue(cells[0].equals(first));
094
095    assertTrue(cs.contains(cells[NUM_OF_CELLS - 1]));  // check last
096    Cell last = cs.last();
097    assertTrue(cells[NUM_OF_CELLS - 1].equals(last));
098
099    SortedSet<Cell> tail = cs.tailSet(cells[1]);    // check tail abd head sizes
100    assertEquals(NUM_OF_CELLS - 1, tail.size());
101    SortedSet<Cell> head = cs.headSet(cells[1]);
102    assertEquals(1, head.size());
103
104    SortedSet<Cell> tailOuter = cs.tailSet(outerCell);  // check tail starting from outer cell
105    assertEquals(NUM_OF_CELLS - 1, tailOuter.size());
106
107    Cell tailFirst = tail.first();
108    assertTrue(cells[1].equals(tailFirst));
109    Cell tailLast = tail.last();
110    assertTrue(cells[NUM_OF_CELLS - 1].equals(tailLast));
111
112    Cell headFirst = head.first();
113    assertTrue(cells[0].equals(headFirst));
114    Cell headLast = head.last();
115    assertTrue(cells[0].equals(headLast));
116  }
117
118  /* Generic iterators test for immutable CellSet */
119  private void testIterators(CellSet cs) throws Exception {
120
121    // Assert that we have NUM_OF_CELLS values and that they are in order
122    int count = 0;
123    for (Cell kv: cs) {
124      assertEquals("\n\n-------------------------------------------------------------------\n"
125              + "Comparing iteration number " + (count + 1) + " the returned cell: " + kv
126              + ", the first Cell in the CellBlocksMap: " + cells[count]
127              + ", and the same transformed to String: " + cells[count].toString()
128              + "\n-------------------------------------------------------------------\n",
129              cells[count], kv);
130      count++;
131    }
132    assertEquals(NUM_OF_CELLS, count);
133
134    // Test descending iterator
135    count = 0;
136    for (Iterator<Cell> i = cs.descendingIterator(); i.hasNext();) {
137      Cell kv = i.next();
138      assertEquals(cells[NUM_OF_CELLS - (count + 1)], kv);
139      count++;
140    }
141    assertEquals(NUM_OF_CELLS, count);
142  }
143}
