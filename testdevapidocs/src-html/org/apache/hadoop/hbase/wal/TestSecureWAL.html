055@Category({RegionServerTests.class, MediumTests.class})
056public class TestSecureWAL {
057  private static final Log LOG = LogFactory.getLog(TestSecureWAL.class);
058  static {
059    ((Log4JLogger)LogFactory.getLog("org.apache.hadoop.hbase.regionserver.wal"))
060      .getLogger().setLevel(Level.ALL);
061  };
062  static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
063
064  @BeforeClass
065  public static void setUpBeforeClass() throws Exception {
066    Configuration conf = TEST_UTIL.getConfiguration();
067    conf.set(HConstants.CRYPTO_KEYPROVIDER_CONF_KEY, KeyProviderForTesting.class.getName());
068    conf.set(HConstants.CRYPTO_MASTERKEY_NAME_CONF_KEY, "hbase");
069    conf.setClass("hbase.regionserver.hlog.reader.impl", SecureProtobufLogReader.class,
070      WAL.Reader.class);
071    conf.setClass("hbase.regionserver.hlog.writer.impl", SecureProtobufLogWriter.class,
072      WALProvider.Writer.class);
073    conf.setBoolean(HConstants.ENABLE_WAL_ENCRYPTION, true);
074    FSUtils.setRootDir(conf, TEST_UTIL.getDataTestDir());
075  }
076
077  @Test
078  public void testSecureWAL() throws Exception {
079    TableName tableName = TableName.valueOf("TestSecureWAL");
080    HTableDescriptor htd = new HTableDescriptor(tableName);
081    htd.addFamily(new HColumnDescriptor(tableName.getName()));
082    HRegionInfo regioninfo = new HRegionInfo(tableName,
083      HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false);
084    final int total = 10;
085    final byte[] row = Bytes.toBytes("row");
086    final byte[] family = Bytes.toBytes("family");
087    final byte[] value = Bytes.toBytes("Test value");
088    FileSystem fs = TEST_UTIL.getTestFileSystem();
089    final WALFactory wals = new WALFactory(TEST_UTIL.getConfiguration(), null, "TestSecureWAL");
090
091    // Write the WAL
092    final WAL wal =
093        wals.getWAL(regioninfo.getEncodedNameAsBytes(), regioninfo.getTable().getNamespace());
094
095    for (int i = 0; i < total; i++) {
096      WALEdit kvs = new WALEdit();
097      kvs.add(new KeyValue(row, family, Bytes.toBytes(i), value));
098      wal.append(htd, regioninfo, new WALKey(regioninfo.getEncodedNameAsBytes(), tableName,
099          System.currentTimeMillis()), kvs, true);
100    }
101    wal.sync();
102    final Path walPath = DefaultWALProvider.getCurrentFileName(wal);
103    wals.shutdown();
104
105    // Insure edits are not plaintext
106    long length = fs.getFileStatus(walPath).getLen();
107    FSDataInputStream in = fs.open(walPath);
108    byte[] fileData = new byte[(int)length];
109    IOUtils.readFully(in, fileData);
110    in.close();
111    assertFalse("Cells appear to be plaintext", Bytes.contains(fileData, value));
112
113    // Confirm the WAL can be read back
114    WAL.Reader reader = wals.createReader(TEST_UTIL.getTestFileSystem(), walPath);
115    int count = 0;
116    WAL.Entry entry = new WAL.Entry();
117    while (reader.next(entry) != null) {
118      count++;
119      List<Cell> cells = entry.getEdit().getCells();
120      assertTrue("Should be one KV per WALEdit", cells.size() == 1);
121      for (Cell cell: cells) {
122        assertTrue("Incorrect row", Bytes.equals(cell.getRowArray(), cell.getRowOffset(),
123          cell.getRowLength(), row, 0, row.length));
124        assertTrue("Incorrect family", Bytes.equals(cell.getFamilyArray(), cell.getFamilyOffset(),
125          cell.getFamilyLength(), family, 0, family.length));
126        assertTrue("Incorrect value", Bytes.equals(cell.getValueArray(), cell.getValueOffset(),
127          cell.getValueLength(), value, 0, value.length));
128      }
129    }
130    assertEquals("Should have read back as many KVs as written", total, count);
131    reader.close();
132  }
