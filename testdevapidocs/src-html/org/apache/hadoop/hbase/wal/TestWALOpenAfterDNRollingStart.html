019package org.apache.hadoop.hbase.wal;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.fs.Path;
024import org.apache.hadoop.hbase.HBaseTestingUtility;
025import org.apache.hadoop.hbase.HConstants;
026import org.apache.hadoop.hbase.regionserver.HRegionServer;
027import org.apache.hadoop.hbase.regionserver.wal.FSHLog;
028import org.apache.hadoop.hbase.testclassification.MediumTests;
029import org.apache.hadoop.hbase.util.FSUtils;
030import org.junit.BeforeClass;
031import org.junit.Test;
032import org.junit.experimental.categories.Category;
033
034
035@Category(MediumTests.class)
036public class TestWALOpenAfterDNRollingStart {
037  final Log LOG = LogFactory.getLog(getClass());
038  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
039  private static long DataNodeRestartInterval;
040
041  @BeforeClass
042  public static void setUpBeforeClass() throws Exception {
043    // Sleep time before restart next dn, we need to wait the current dn to finish start up
044    DataNodeRestartInterval = 15000;
045    // interval of checking low replication. The sleep time must smaller than DataNodeRestartInterval
046    // so a low replication case will be detected and the wal will be rolled
047    long checkLowReplicationInterval = 10000;
048    //don't let hdfs client to choose a new replica when dn down
049    TEST_UTIL.getConfiguration().setBoolean("dfs.client.block.write.replace-datanode-on-failure.enable",
050        false);
051    TEST_UTIL.getConfiguration().setLong("hbase.regionserver.hlog.check.lowreplication.interval",
052        checkLowReplicationInterval);
053    TEST_UTIL.startMiniDFSCluster(3);
054    TEST_UTIL.startMiniCluster(1);
055
056  }
057
058  /**
059   * see HBASE-18132
060   * This is a test case of failing open a wal(for replication for example) after all datanode
061   * restarted (rolling upgrade, for example).
062   * Before this patch, low replication detection is only used when syncing wal.
063   * But if the wal haven't had any entry whiten, it will never know all the replica of the wal
064   * is broken(because of dn restarting). And this wal can never be open
065   * @throws Exception
066   */
067  @Test(timeout = 300000)
068  public void test() throws Exception {
069    HRegionServer server = TEST_UTIL.getHBaseCluster().getRegionServer(0);
070    FSHLog hlog = (FSHLog)server.getWAL(null);
071    Path currentFile = hlog.getCurrentFileName();
072    //restart every dn to simulate a dn rolling upgrade
073    for(int i = 0; i < TEST_UTIL.getDFSCluster().getDataNodes().size(); i++) {
074      //This is NOT a bug, when restart dn in miniDFSCluster, it will remove the stopped dn from
075      //the dn list and then add to the tail of this list, we need to always restart the first one
076      //to simulate rolling upgrade of every dn.
077      TEST_UTIL.getDFSCluster().restartDataNode(0);
078      //sleep enough time so log roller can detect the pipeline break and roll log
079      Thread.sleep(DataNodeRestartInterval);
080    }
081
082    if(!server.getFileSystem().exists(currentFile)) {
083      Path walRootDir = FSUtils.getWALRootDir(TEST_UTIL.getConfiguration());
084      final Path oldLogDir = new Path(walRootDir, HConstants.HREGION_OLDLOGDIR_NAME);
085      currentFile = new Path(oldLogDir, currentFile.getName());
086    }
087    //if the log is not rolled, then we can never open this wal forever.
088    WAL.Reader reader = WALFactory
089        .createReader(TEST_UTIL.getTestFileSystem(), currentFile, TEST_UTIL.getConfiguration());
090
091  }
