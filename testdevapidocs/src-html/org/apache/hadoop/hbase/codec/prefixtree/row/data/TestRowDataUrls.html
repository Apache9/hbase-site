034/*
035 * test different timestamps
036 * 
037 * http://pastebin.com/7ks8kzJ2
038 * http://pastebin.com/MPn03nsK
039 */
040public class TestRowDataUrls extends BaseTestRowData{
041
042  static List<ByteRange> rows;
043  static{
044    List<String> rowStrings = new ArrayList<>(16);
045    rowStrings.add("com.edsBlog/directoryAa/pageAaa");
046    rowStrings.add("com.edsBlog/directoryAa/pageBbb");
047    rowStrings.add("com.edsBlog/directoryAa/pageCcc");
048    rowStrings.add("com.edsBlog/directoryAa/pageDdd");
049    rowStrings.add("com.edsBlog/directoryBb/pageEee");
050    rowStrings.add("com.edsBlog/directoryBb/pageFff");
051    rowStrings.add("com.edsBlog/directoryBb/pageGgg");
052    rowStrings.add("com.edsBlog/directoryBb/pageHhh");
053    rowStrings.add("com.isabellasBlog/directoryAa/pageAaa");
054    rowStrings.add("com.isabellasBlog/directoryAa/pageBbb");
055    rowStrings.add("com.isabellasBlog/directoryAa/pageCcc");
056    rowStrings.add("com.isabellasBlog/directoryAa/pageDdd");
057    rowStrings.add("com.isabellasBlog/directoryBb/pageEee");
058    rowStrings.add("com.isabellasBlog/directoryBb/pageFff");
059    rowStrings.add("com.isabellasBlog/directoryBb/pageGgg");
060    rowStrings.add("com.isabellasBlog/directoryBb/pageHhh");
061    ByteRangeTreeSet ba = new ByteRangeTreeSet();
062    for (String row : rowStrings) {
063      ba.add(new SimpleMutableByteRange(Bytes.toBytes(row)));
064    }
065    rows = ba.compile().getSortedRanges();
066  }
067
068  static List<String> cols = Lists.newArrayList();
069  static {
070    cols.add("Chrome");
071    cols.add("Chromeb");
072    cols.add("Firefox");
073    cols.add("InternetExplorer");
074    cols.add("Opera");
075    cols.add("Safari");
076  }
077
078  static long ts = 1234567890;
079
080  static int MAX_VALUE = 50;
081
082  static List<KeyValue> kvs = Lists.newArrayList();
083  static {
084    for (ByteRange row : rows) {
085      for (String col : cols) {
086        KeyValue kv = new KeyValue(row.deepCopyToNewArray(), PrefixTreeTestConstants.TEST_CF,
087            Bytes.toBytes(col), ts, KeyValue.Type.Put, Bytes.toBytes("VALUE"));
088        kvs.add(kv);
089        // System.out.println("TestRows5:"+kv);
090      }
091    }
092  }
093
094  @Override
095  public List<KeyValue> getInputs() {
096    return kvs;
097  }
098
099}
