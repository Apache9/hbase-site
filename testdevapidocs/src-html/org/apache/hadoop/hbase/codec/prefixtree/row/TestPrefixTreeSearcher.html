062  protected ByteBuff block;
063
064  public TestPrefixTreeSearcher(TestRowData testRows) throws IOException {
065    this.rows = testRows;
066    ByteArrayOutputStream os = new ByteArrayOutputStream(1 << 20);
067    PrefixTreeEncoder kvBuilder = new PrefixTreeEncoder(os, true);
068    for (KeyValue kv : rows.getInputs()) {
069      kvBuilder.write(kv);
070    }
071    kvBuilder.flush();
072    byte[] outputBytes = os.toByteArray();
073    ByteBuffer out = ByteBuffer.allocateDirect(outputBytes.length);
074    ByteBufferUtils.copyFromArrayToBuffer(out, outputBytes, 0, outputBytes.length);
075    out.position(0);
076    this.block = new SingleByteBuff(out);
077  }
078
079  @Test
080  public void testScanForwards() throws IOException {
081    CellSearcher searcher = null;
082    try {
083      searcher = DecoderFactory.checkOut(block, true);
084
085      int i = -1;
086      while (searcher.advance()) {
087        ++i;
088        KeyValue inputCell = rows.getInputs().get(i);
089        Cell outputCell = searcher.current();
090
091        // check all 3 permutations of equals()
092        Assert.assertEquals(inputCell, outputCell);
093        Assert.assertEquals(outputCell, inputCell);
094        Assert.assertTrue(CellUtil.equals(inputCell, outputCell));
095      }
096      Assert.assertEquals(rows.getInputs().size(), i + 1);
097    } finally {
098      DecoderFactory.checkIn(searcher);
099    }
100  }

103  @Test
104  public void testScanBackwards() throws IOException {
105    CellSearcher searcher = null;
106    try {
107      searcher = DecoderFactory.checkOut(block, true);
108      searcher.positionAfterLastCell();
109      int i = -1;
110      while (searcher.previous()) {
111        ++i;
112        int oppositeIndex = rows.getInputs().size() - i - 1;
113        KeyValue inputKv = rows.getInputs().get(oppositeIndex);
114        KeyValue outputKv = KeyValueUtil.copyToNewKeyValue(searcher.current());
115        Assert.assertEquals(inputKv, outputKv);
116      }
117      Assert.assertEquals(rows.getInputs().size(), i + 1);
118    } finally {
119      DecoderFactory.checkIn(searcher);
120    }
121  }

140  @Test
141  public void testRandomSeekMisses() throws IOException {
142    CellSearcher searcher = null;
143    List<Integer> rowStartIndexes = rows.getRowStartIndexes();
144    try {
145      searcher = DecoderFactory.checkOut(block, true);
146
147      //test both the positionAtOrBefore and positionAtOrAfter methods
148      for(boolean beforeVsAfterOnMiss : new boolean[]{true, false}){
149        for (int i=0; i < rows.getInputs().size(); ++i) {
150          KeyValue kv = rows.getInputs().get(i);
151
152          //nextRow
153          Cell inputNextRow = CellUtil.createFirstOnNextRow(kv);
154
155          CellScannerPosition position = beforeVsAfterOnMiss
156              ? searcher.positionAtOrBefore(inputNextRow)
157              : searcher.positionAtOrAfter(inputNextRow);
158          boolean isFirstInRow = rowStartIndexes.contains(i);
159          if(isFirstInRow){
160            int rowIndex = rowStartIndexes.indexOf(i);
161            if(rowIndex < rowStartIndexes.size() - 1){
162              if(beforeVsAfterOnMiss){
163                Assert.assertEquals(CellScannerPosition.BEFORE, position);
164              }else{
165                Assert.assertEquals(CellScannerPosition.AFTER, position);
166              }
167
168              int expectedInputIndex = beforeVsAfterOnMiss
169                  ? rowStartIndexes.get(rowIndex + 1) - 1
170                  : rowStartIndexes.get(rowIndex + 1);
171              Assert.assertEquals(rows.getInputs().get(expectedInputIndex), searcher.current());
172            }
173          }
174
175          //previous KV
176          KeyValue inputPreviousKv = KeyValueUtil.previousKey(kv);
177          boolean hit = searcher.positionAt(inputPreviousKv);
178          Assert.assertFalse(hit);
179          position = searcher.positionAtOrAfter(inputPreviousKv);
180          if(CollectionUtils.isLastIndex(rows.getInputs(), i)){
181            Assert.assertTrue(CellScannerPosition.AFTER_LAST == position);
182          }else{
183            Assert.assertTrue(CellScannerPosition.AFTER == position);
184            /*
185             * TODO: why i+1 instead of i?
186             */
187            Assert.assertEquals(rows.getInputs().get(i+1), searcher.current());
188          }
189        }
190      }
191    } finally {
192      DecoderFactory.checkIn(searcher);
193    }
194  }

208  @Test
209  public void testSeekWithPrefix() throws IOException {
210    if (!(rows instanceof TestRowDataSearchWithPrefix)) {
211      return;
212    }
213    CellSearcher searcher = null;
214    try {
215      searcher = DecoderFactory.checkOut(block, true);
216      // seek with half bytes of second row key, should return second row
217      KeyValue kv = rows.getInputs().get(1);
218      KeyValue firstKVOnRow = KeyValueUtil.createFirstOnRow(Arrays.copyOfRange(
219          kv.getRowArray(), kv.getRowOffset(),
220          kv.getRowOffset() + kv.getRowLength() / 2));
221      CellScannerPosition position = searcher.positionAtOrAfter(firstKVOnRow);
222      Assert.assertEquals(CellScannerPosition.AFTER, position);
223      Assert.assertEquals(kv, searcher.current());
224    } finally {
225      DecoderFactory.checkIn(searcher);
226    }
227  }
