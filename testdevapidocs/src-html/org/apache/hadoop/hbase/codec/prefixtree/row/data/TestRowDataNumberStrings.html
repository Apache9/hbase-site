019package org.apache.hadoop.hbase.codec.prefixtree.row.data;
020
021import java.util.Collections;
022import java.util.List;
023
024import org.apache.hadoop.hbase.CellComparatorImpl;
025import org.apache.hadoop.hbase.KeyValue;
026import org.apache.hadoop.hbase.KeyValue.Type;
027import org.apache.hadoop.hbase.codec.prefixtree.row.BaseTestRowData;
028import org.apache.hadoop.hbase.util.Bytes;
029
030import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
031
032public class TestRowDataNumberStrings extends BaseTestRowData{
033
034  static List<KeyValue> d = Lists.newArrayList();
035  static {
036
037  /**
038   * Test a string-encoded list of numbers.  0, 1, 10, 11 will sort as 0, 1, 10, 11 if strings
039   * <p/>
040   * This helped catch a bug with reverse scanning where it was jumping from the last leaf cell to
041   * the previous nub.  It should do 11->10, but it was incorrectly doing 11->1
042   */
043    List<Integer> problematicSeries = Lists.newArrayList(0, 1, 10, 11);//sort this at the end
044    for(Integer i : problematicSeries){
045//    for(int i=0; i < 13; ++i){
046      byte[] row = Bytes.toBytes(""+i);
047      byte[] family = Bytes.toBytes("F");
048      byte[] column = Bytes.toBytes("C");
049      byte[] value = Bytes.toBytes("V");
050
051      d.add(new KeyValue(row, family, column, 0L, Type.Put, value));
052    }
053    Collections.sort(d, CellComparatorImpl.COMPARATOR);
054  }
055
056  @Override
057  public List<KeyValue> getInputs() {
058    return d;
059  }
060
061}
