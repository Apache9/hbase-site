019package org.apache.hadoop.hbase.codec.prefixtree.row.data;
020
021import java.io.IOException;
022import java.util.List;
023
024import org.apache.hadoop.hbase.Cell;
025import org.apache.hadoop.hbase.CellUtil;
026import org.apache.hadoop.hbase.KeyValue;
027import org.apache.hadoop.hbase.codec.prefixtree.row.BaseTestRowData;
028import org.apache.hadoop.hbase.codec.prefixtree.scanner.CellScannerPosition;
029import org.apache.hadoop.hbase.codec.prefixtree.scanner.CellSearcher;
030import org.apache.hadoop.hbase.util.Bytes;
031import org.apache.hadoop.hbase.util.CollectionUtils;
032import org.junit.Assert;
033
034import com.google.common.collect.Lists;
035
036public class TestRowDataSimple extends BaseTestRowData {
037
038  static byte[]
039  // don't let the rows share any common prefix bytes
040      rowA = Bytes.toBytes("Arow"),
041      rowB = Bytes.toBytes("Brow"), cf = Bytes.toBytes("fam"),
042      cq0 = Bytes.toBytes("cq0"),
043      cq1 = Bytes.toBytes("cq1tail"),// make sure tail does not come back as liat
044      cq2 = Bytes.toBytes("dcq2"),// start with a different character
045      v0 = Bytes.toBytes("v0");
046
047  static long ts = 55L;
048
049  static List<KeyValue> d = Lists.newArrayList();
050  static {
051    d.add(new KeyValue(rowA, cf, cq0, ts, v0));
052    d.add(new KeyValue(rowA, cf, cq1, ts, v0));
053    d.add(new KeyValue(rowA, cf, cq2, ts, v0));
054    d.add(new KeyValue(rowB, cf, cq0, ts, v0));
055    d.add(new KeyValue(rowB, cf, cq1, ts, v0));
056    d.add(new KeyValue(rowB, cf, cq2, ts, v0));
057  }
058
059  @Override
060  public List<KeyValue> getInputs() {
061    return d;
062  }
063
064  @Override
065  public void individualSearcherAssertions(CellSearcher searcher) {
066    CellScannerPosition p;// reuse
067    searcher.resetToBeforeFirstEntry();
068
069    // test first cell
070    try {
071      searcher.advance();
072    } catch (IOException e) {
073      throw new RuntimeException(e);
074    }
075    Cell first = searcher.current();
076    Assert.assertTrue(CellUtil.equals(d.get(0), first));
077
078    // test first cell in second row
079    Assert.assertTrue(searcher.positionAt(d.get(3)));
080    Assert.assertTrue(CellUtil.equals(d.get(3), searcher.current()));
081
082    Cell between4And5 = new KeyValue(rowB, cf, cq1, ts - 2, v0);
083
084    // test exact
085    Assert.assertFalse(searcher.positionAt(between4And5));
086
087    // test atOrBefore
088    p = searcher.positionAtOrBefore(between4And5);
089    Assert.assertEquals(CellScannerPosition.BEFORE, p);
090    Assert.assertTrue(CellUtil.equals(searcher.current(), d.get(4)));
091
092    // test atOrAfter
093    p = searcher.positionAtOrAfter(between4And5);
094    Assert.assertEquals(CellScannerPosition.AFTER, p);
095    Assert.assertTrue(CellUtil.equals(searcher.current(), d.get(5)));
096
097    // test when key falls before first key in block
098    Cell beforeFirst = new KeyValue(Bytes.toBytes("A"), cf, cq0, ts, v0);
099    Assert.assertFalse(searcher.positionAt(beforeFirst));
100    p = searcher.positionAtOrBefore(beforeFirst);
101    Assert.assertEquals(CellScannerPosition.BEFORE_FIRST, p);
102    p = searcher.positionAtOrAfter(beforeFirst);
103    Assert.assertEquals(CellScannerPosition.AFTER, p);
104    Assert.assertTrue(CellUtil.equals(searcher.current(), d.get(0)));
105    Assert.assertEquals(d.get(0), searcher.current());
106
107    // test when key falls after last key in block
108    Cell afterLast = new KeyValue(Bytes.toBytes("z"), cf, cq0, ts, v0);// must be lower case z
109    Assert.assertFalse(searcher.positionAt(afterLast));
110    p = searcher.positionAtOrAfter(afterLast);
111    Assert.assertEquals(CellScannerPosition.AFTER_LAST, p);
112    p = searcher.positionAtOrBefore(afterLast);
113    Assert.assertEquals(CellScannerPosition.BEFORE, p);
114    Assert.assertTrue(CellUtil.equals(searcher.current(), CollectionUtils.getLast(d)));
115  }
116
117}
