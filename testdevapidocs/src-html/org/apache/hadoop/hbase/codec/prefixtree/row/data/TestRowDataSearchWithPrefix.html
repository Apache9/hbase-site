018package org.apache.hadoop.hbase.codec.prefixtree.row.data;
019
020import java.io.ByteArrayOutputStream;
021import java.io.DataOutputStream;
022import java.io.IOException;
023import java.util.List;
024
025import org.apache.hadoop.hbase.KeyValue;
026import org.apache.hadoop.hbase.codec.prefixtree.row.BaseTestRowData;
027import org.apache.hadoop.hbase.util.Bytes;
028
029import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
030
031public class TestRowDataSearchWithPrefix extends BaseTestRowData {
032
033  static byte[] cf = Bytes.toBytes("cf");
034
035  static byte[] cq = Bytes.toBytes("cq");
036
037  static byte[] v = Bytes.toBytes("v");
038
039  static List<KeyValue> d = Lists.newArrayList();
040
041  static long ts = 55L;
042
043  static byte[] createRowKey(int keyPart1, int keyPart2) {
044    ByteArrayOutputStream bos = new ByteArrayOutputStream(16);
045    DataOutputStream dos = new DataOutputStream(bos);
046    try {
047      dos.writeInt(keyPart1);
048      dos.writeInt(keyPart2);
049    } catch (IOException e) {
050      // should not happen
051      throw new RuntimeException(e);
052    }
053
054    return bos.toByteArray();
055  }
056
057  static {
058    d.add(new KeyValue(createRowKey(1, 12345), cf, cq, ts, v));
059    d.add(new KeyValue(createRowKey(12345, 0x01000000), cf, cq, ts, v));
060    d.add(new KeyValue(createRowKey(12345, 0x01010000), cf, cq, ts, v));
061    d.add(new KeyValue(createRowKey(12345, 0x02000000), cf, cq, ts, v));
062    d.add(new KeyValue(createRowKey(12345, 0x02020000), cf, cq, ts, v));
063    d.add(new KeyValue(createRowKey(12345, 0x03000000), cf, cq, ts, v));
064    d.add(new KeyValue(createRowKey(12345, 0x03030000), cf, cq, ts, v));
065    d.add(new KeyValue(createRowKey(12345, 0x04000000), cf, cq, ts, v));
066    d.add(new KeyValue(createRowKey(12345, 0x04040000), cf, cq, ts, v));
067  }
068
069  @Override
070  public List<KeyValue> getInputs() {
071    return d;
072  }
073
074}
