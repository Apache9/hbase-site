019package org.apache.hadoop.hbase.codec.prefixtree.row.data;
020
021import java.util.List;
022
023import org.apache.hadoop.hbase.KeyValue;
024import org.apache.hadoop.hbase.codec.prefixtree.PrefixTreeBlockMeta;
025import org.apache.hadoop.hbase.codec.prefixtree.row.BaseTestRowData;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.junit.Assert;
028
029import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
030
031/*
032 * test different timestamps
033 */
034public class TestRowDataDifferentTimestamps extends BaseTestRowData{
035
036  static byte[]
037    Arow = Bytes.toBytes("Arow"),
038    Brow = Bytes.toBytes("Brow"),
039    cf = Bytes.toBytes("fammy"),
040    cq0 = Bytes.toBytes("cq0"),
041    cq1 = Bytes.toBytes("cq1"),
042    v0 = Bytes.toBytes("v0");
043
044  static List<KeyValue> d = Lists.newArrayList();
045  static{
046    KeyValue kv0 = new KeyValue(Arow, cf, cq0, 0L, v0);
047    kv0.setSequenceId(123456789L);
048    d.add(kv0);
049
050    KeyValue kv1 = new KeyValue(Arow, cf, cq1, 1L, v0);
051    kv1.setSequenceId(3L);
052    d.add(kv1);
053
054    KeyValue kv2 = new KeyValue(Brow, cf, cq0, 12345678L, v0);
055    kv2.setSequenceId(65537L);
056    d.add(kv2);
057
058    //watch out... Long.MAX_VALUE comes back as 1332221664203, even with other encoders
059    //d.add(new KeyValue(Brow, cf, cq1, Long.MAX_VALUE, v0));
060    KeyValue kv3 = new KeyValue(Brow, cf, cq1, Long.MAX_VALUE-1, v0);
061    kv3.setSequenceId(1L);
062    d.add(kv3);
063
064    KeyValue kv4 = new KeyValue(Brow, cf, cq1, 999999999, v0);
065    //don't set memstoreTS
066    d.add(kv4);
067
068    KeyValue kv5 = new KeyValue(Brow, cf, cq1, 12345, v0);
069    kv5.setSequenceId(0L);
070    d.add(kv5);
071  }
072
073  @Override
074  public List<KeyValue> getInputs() {
075    return d;
076  }
077
078  @Override
079  public void individualBlockMetaAssertions(PrefixTreeBlockMeta blockMeta) {
080    Assert.assertTrue(blockMeta.getNumMvccVersionBytes() > 0);
081    Assert.assertEquals(12, blockMeta.getNumValueBytes());
082
083    Assert.assertFalse(blockMeta.isAllSameTimestamp());
084    Assert.assertNotNull(blockMeta.getMinTimestamp());
085    Assert.assertTrue(blockMeta.getTimestampIndexWidth() > 0);
086    Assert.assertTrue(blockMeta.getTimestampDeltaWidth() > 0);
087
088    Assert.assertFalse(blockMeta.isAllSameMvccVersion());
089    Assert.assertNotNull(blockMeta.getMinMvccVersion());
090    Assert.assertTrue(blockMeta.getMvccVersionIndexWidth() > 0);
091    Assert.assertTrue(blockMeta.getMvccVersionDeltaWidth() > 0);
092  }
093
094}
