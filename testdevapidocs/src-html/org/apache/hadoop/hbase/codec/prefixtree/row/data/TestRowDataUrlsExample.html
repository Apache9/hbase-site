025import org.apache.hadoop.hbase.KeyValue;

033
034import com.google.common.collect.Lists;
035
036/*
037 * test different timestamps
038 * 
039 * http://pastebin.com/7ks8kzJ2
040 * http://pastebin.com/MPn03nsK
041 */
042public class TestRowDataUrlsExample extends BaseTestRowData{
043
044  static String TENANT_ID = Integer.toString(95322);
045  static String APP_ID = Integer.toString(12);
046  static List<String> URLS = Lists.newArrayList(
047      "com.dablog/2011/10/04/boating", 
048      "com.dablog/2011/10/09/lasers", 
049      "com.jamiesrecipes", //this nub helped find a bug
050      "com.jamiesrecipes/eggs");
051  static String FAMILY = "hits";
052  static List<String> BROWSERS = Lists.newArrayList(
053      "Chrome", "IE8", "IE9beta");//, "Opera", "Safari");
054  static long TIMESTAMP = 1234567890;
055
056  static int MAX_VALUE = 50;
057
058  static List<KeyValue> kvs = Lists.newArrayList();
059  static{
060    for(String rowKey : URLS){
061      for(String qualifier : BROWSERS){
062        KeyValue kv = new KeyValue(
063            Bytes.toBytes(rowKey), 
064            Bytes.toBytes(FAMILY), 
065            Bytes.toBytes(qualifier), 
066            TIMESTAMP, 
067            KeyValue.Type.Put, 
068            Bytes.toBytes("VvvV"));
069        kvs.add(kv);
070      }
071    }
072  }
073
074  /**
075   * Used for generating docs.
076   */
077  public static void main(String... args) throws IOException{
078    System.out.println("-- inputs --");
079    System.out.println(KeyValueTestUtil.toStringWithPadding(kvs, true));
080    ByteArrayOutputStream os = new ByteArrayOutputStream(1<<20);
081    PrefixTreeEncoder encoder = new PrefixTreeEncoder(os, false);
082
083    for(KeyValue kv : kvs){
084      encoder.write(kv);
085    }
086    encoder.flush();
087
088    System.out.println("-- qualifier SortedPtBuilderNodes --");
089    for(TokenizerNode tokenizer : encoder.getQualifierWriter().getNonLeaves()){
090      System.out.println(tokenizer);
091    }
092    for(TokenizerNode tokenizerNode : encoder.getQualifierWriter().getLeaves()){
093      System.out.println(tokenizerNode);
094    }
095
096    System.out.println("-- qualifier PtColumnNodeWriters --");
097    for(ColumnNodeWriter writer : encoder.getQualifierWriter().getColumnNodeWriters()){
098      System.out.println(writer);
099    }
100
101    System.out.println("-- rowKey SortedPtBuilderNodes --");
102    for(TokenizerNode tokenizerNode : encoder.getRowWriter().getNonLeaves()){
103      System.out.println(tokenizerNode);
104    }
105    for(TokenizerNode tokenizerNode : encoder.getRowWriter().getLeaves()){
106      System.out.println(tokenizerNode);
107    }
108
109    System.out.println("-- row PtRowNodeWriters --");
110    for(RowNodeWriter writer : encoder.getRowWriter().getNonLeafWriters()){
111      System.out.println(writer);
112    }
113    for(RowNodeWriter writer : encoder.getRowWriter().getLeafWriters()){
114      System.out.println(writer);
115    }
116
117    System.out.println("-- concatenated values --");
118    System.out.println(Bytes.toStringBinary(encoder.getValueByteRange().deepCopyToNewArray()));
119  }
120
121  @Override
122  public List<KeyValue> getInputs() {
123    return kvs;
124  }
125
126}
