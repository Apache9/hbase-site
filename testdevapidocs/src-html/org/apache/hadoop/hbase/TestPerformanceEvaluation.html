054  @Test
055  public void testSerialization()
056  throws JsonGenerationException, JsonMappingException, IOException {
057    PerformanceEvaluation.TestOptions options = new PerformanceEvaluation.TestOptions();
058    assertTrue(!options.isAutoFlush());
059    options.setAutoFlush(true);
060    ObjectMapper mapper = new ObjectMapper();
061    String optionsString = mapper.writeValueAsString(options);
062    PerformanceEvaluation.TestOptions optionsDeserialized =
063        mapper.readValue(optionsString, PerformanceEvaluation.TestOptions.class);
064    assertTrue(optionsDeserialized.isAutoFlush());
065  }

067  /**
068   * Exercise the mr spec writing.  Simple assertions to make sure it is basically working.
069   * @throws IOException
070   */
071  @Ignore @Test
072  public void testWriteInputFile() throws IOException {
073    TestOptions opts = new PerformanceEvaluation.TestOptions();
074    final int clients = 10;
075    opts.setNumClientThreads(clients);
076    opts.setPerClientRunRows(10);
077    Path dir =
078      PerformanceEvaluation.writeInputFile(HTU.getConfiguration(), opts, HTU.getDataTestDir());
079    FileSystem fs = FileSystem.get(HTU.getConfiguration());
080    Path p = new Path(dir, PerformanceEvaluation.JOB_INPUT_FILENAME);
081    long len = fs.getFileStatus(p).getLen();
082    assertTrue(len > 0);
083    byte [] content = new byte[(int)len];
084    FSDataInputStream dis = fs.open(p);
085    try {
086      dis.readFully(content);
087      BufferedReader br =
088        new BufferedReader(new InputStreamReader(new ByteArrayInputStream(content)));
089      int count = 0;
090      while (br.readLine() != null) {
091        count++;
092      }
093      assertEquals(clients * PerformanceEvaluation.TASKS_PER_CLIENT, count);
094    } finally {
095      dis.close();
096    }
097  }

099  @Test
100  public void testSizeCalculation() {
101    TestOptions opts = new PerformanceEvaluation.TestOptions();
102    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
103    int rows = opts.getPerClientRunRows();
104    // Default row count
105    final int defaultPerClientRunRows = 1024 * 1024;
106    assertEquals(defaultPerClientRunRows, rows);
107    // If size is 2G, then twice the row count.
108    opts.setSize(2.0f);
109    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
110    assertEquals(defaultPerClientRunRows * 2, opts.getPerClientRunRows());
111    // If two clients, then they get half the rows each.
112    opts.setNumClientThreads(2);
113    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
114    assertEquals(defaultPerClientRunRows, opts.getPerClientRunRows());
115    // What if valueSize is 'random'? Then half of the valueSize so twice the rows.
116    opts.valueRandom = true;
117    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
118    assertEquals(defaultPerClientRunRows * 2, opts.getPerClientRunRows());
119  }

121  @Test
122  public void testZipfian()
123  throws NoSuchMethodException, SecurityException, InstantiationException, IllegalAccessException,
124      IllegalArgumentException, InvocationTargetException {
125    TestOptions opts = new PerformanceEvaluation.TestOptions();
126    opts.setValueZipf(true);
127    final int valueSize = 1024;
128    opts.setValueSize(valueSize);
129    RandomReadTest rrt = new RandomReadTest(null, opts, null);
130    Constructor<?> ctor =
131      Histogram.class.getDeclaredConstructor(com.codahale.metrics.Reservoir.class);
132    ctor.setAccessible(true);
133    Histogram histogram = (Histogram)ctor.newInstance(new UniformReservoir(1024 * 500));
134    for (int i = 0; i < 100; i++) {
135      histogram.update(rrt.getValueLength(null));
136    }
137    Snapshot snapshot = histogram.getSnapshot();
138    double stddev = snapshot.getStdDev();
139    assertTrue(stddev != 0 && stddev != 1.0);
140    assertTrue(snapshot.getStdDev() != 0);
141    double median = snapshot.getMedian();
142    assertTrue(median != 0 && median != 1 && median != valueSize);
143  }

145  @Test
146  public void testParseOptsWithThreads() {
147    Queue<String> opts = new LinkedList<>();
148    String cmdName = "sequentialWrite";
149    int threads = 1;
150    opts.offer(cmdName);
151    opts.offer(String.valueOf(threads));
152    PerformanceEvaluation.TestOptions options = PerformanceEvaluation.parseOpts(opts);
153    assertNotNull(options);
154    assertNotNull(options.getCmdName());
155    assertEquals(cmdName, options.getCmdName());
156    assertEquals(threads, options.getNumClientThreads());
157  }
