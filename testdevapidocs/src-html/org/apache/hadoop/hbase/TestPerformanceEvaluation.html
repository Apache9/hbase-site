055  @Test
056  public void testSerialization()
057  throws JsonGenerationException, JsonMappingException, IOException {
058    PerformanceEvaluation.TestOptions options = new PerformanceEvaluation.TestOptions();
059    assertTrue(!options.isAutoFlush());
060    options.setAutoFlush(true);
061    ObjectMapper mapper = new ObjectMapper();
062    String optionsString = mapper.writeValueAsString(options);
063    PerformanceEvaluation.TestOptions optionsDeserialized =
064        mapper.readValue(optionsString, PerformanceEvaluation.TestOptions.class);
065    assertTrue(optionsDeserialized.isAutoFlush());
066  }

072  @Ignore @Test
073  public void testWriteInputFile() throws IOException {
074    TestOptions opts = new PerformanceEvaluation.TestOptions();
075    final int clients = 10;
076    opts.setNumClientThreads(clients);
077    opts.setPerClientRunRows(10);
078    Path dir =
079      PerformanceEvaluation.writeInputFile(HTU.getConfiguration(), opts, HTU.getDataTestDir());
080    FileSystem fs = FileSystem.get(HTU.getConfiguration());
081    Path p = new Path(dir, PerformanceEvaluation.JOB_INPUT_FILENAME);
082    long len = fs.getFileStatus(p).getLen();
083    assertTrue(len > 0);
084    byte [] content = new byte[(int)len];
085    FSDataInputStream dis = fs.open(p);
086    try {
087      dis.readFully(content);
088      BufferedReader br =
089        new BufferedReader(new InputStreamReader(new ByteArrayInputStream(content)));
090      int count = 0;
091      while (br.readLine() != null) {
092        count++;
093      }
094      assertEquals(clients, count);
095    } finally {
096      dis.close();
097    }
098  }

100  @Test
101  public void testSizeCalculation() {
102    TestOptions opts = new PerformanceEvaluation.TestOptions();
103    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
104    int rows = opts.getPerClientRunRows();
105    // Default row count
106    final int defaultPerClientRunRows = 1024 * 1024;
107    assertEquals(defaultPerClientRunRows, rows);
108    // If size is 2G, then twice the row count.
109    opts.setSize(2.0f);
110    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
111    assertEquals(defaultPerClientRunRows * 2, opts.getPerClientRunRows());
112    // If two clients, then they get half the rows each.
113    opts.setNumClientThreads(2);
114    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
115    assertEquals(defaultPerClientRunRows, opts.getPerClientRunRows());
116    // What if valueSize is 'random'? Then half of the valueSize so twice the rows.
117    opts.valueRandom = true;
118    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
119    assertEquals(defaultPerClientRunRows * 2, opts.getPerClientRunRows());
120  }

122  @Test
123  public void testRandomReadCalculation() {
124    TestOptions opts = new PerformanceEvaluation.TestOptions();
125    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
126    int rows = opts.getPerClientRunRows();
127    // Default row count
128    final int defaultPerClientRunRows = 1024 * 1024;
129    assertEquals(defaultPerClientRunRows, rows);
130    // If size is 2G, then twice the row count.
131    opts.setSize(2.0f);
132    opts.setPerClientRunRows(1000);
133    opts.setCmdName(PerformanceEvaluation.RANDOM_READ);
134    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
135    assertEquals(1000, opts.getPerClientRunRows());
136    // If two clients, then they get half the rows each.
137    opts.setNumClientThreads(2);
138    opts = PerformanceEvaluation.calculateRowsAndSize(opts);
139    assertEquals(1000, opts.getPerClientRunRows());
140    Random random = new Random();
141    // assuming we will get one before this loop expires
142    boolean foundValue = false;
143    for (int i = 0; i < 10000000; i++) {
144      int randomRow = PerformanceEvaluation.generateRandomRow(random, opts.totalRows);
145      if (randomRow > 1000) {
146        foundValue = true;
147        break;
148      }
149    }
150    assertTrue("We need to get a value more than 1000", foundValue);
151  }

153  @Test
154  public void testZipfian()
155  throws NoSuchMethodException, SecurityException, InstantiationException, IllegalAccessException,
156      IllegalArgumentException, InvocationTargetException {
157    TestOptions opts = new PerformanceEvaluation.TestOptions();
158    opts.setValueZipf(true);
159    final int valueSize = 1024;
160    opts.setValueSize(valueSize);
161    RandomReadTest rrt = new RandomReadTest(null, opts, null);
162    Constructor<?> ctor =
163      Histogram.class.getDeclaredConstructor(com.codahale.metrics.Reservoir.class);
164    ctor.setAccessible(true);
165    Histogram histogram = (Histogram)ctor.newInstance(new UniformReservoir(1024 * 500));
166    for (int i = 0; i < 100; i++) {
167      histogram.update(rrt.getValueLength(null));
168    }
169    Snapshot snapshot = histogram.getSnapshot();
170    double stddev = snapshot.getStdDev();
171    assertTrue(stddev != 0 && stddev != 1.0);
172    assertTrue(snapshot.getStdDev() != 0);
173    double median = snapshot.getMedian();
174    assertTrue(median != 0 && median != 1 && median != valueSize);
175  }

177  @Test
178  public void testParseOptsWithThreads() {
179    Queue<String> opts = new LinkedList<>();
180    String cmdName = "sequentialWrite";
181    int threads = 1;
182    opts.offer(cmdName);
183    opts.offer(String.valueOf(threads));
184    PerformanceEvaluation.TestOptions options = PerformanceEvaluation.parseOpts(opts);
185    assertNotNull(options);
186    assertNotNull(options.getCmdName());
187    assertEquals(cmdName, options.getCmdName());
188    assertEquals(threads, options.getNumClientThreads());
189  }
