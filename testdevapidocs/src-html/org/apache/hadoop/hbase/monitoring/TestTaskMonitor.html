019package org.apache.hadoop.hbase.monitoring;
020
021import static org.junit.Assert.*;
022
023import java.util.concurrent.atomic.AtomicBoolean;
024
025import org.apache.hadoop.hbase.testclassification.MiscTests;
026import org.apache.hadoop.hbase.testclassification.SmallTests;
027import org.junit.Test;
028import org.junit.experimental.categories.Category;
029
030@Category({MiscTests.class, SmallTests.class})
031public class TestTaskMonitor {
032
033  @Test
034  public void testTaskMonitorBasics() {
035    TaskMonitor tm = new TaskMonitor();
036    assertTrue("Task monitor should start empty",
037        tm.getTasks().isEmpty());
038    
039    // Make a task and fetch it back out
040    MonitoredTask task = tm.createStatus("Test task");
041    MonitoredTask taskFromTm = tm.getTasks().get(0);
042    
043    // Make sure the state is reasonable.
044    assertEquals(task.getDescription(), taskFromTm.getDescription());
045    assertEquals(-1, taskFromTm.getCompletionTimestamp());
046    assertEquals(MonitoredTask.State.RUNNING, taskFromTm.getState());
047    
048    // Mark it as finished
049    task.markComplete("Finished!");
050    assertEquals(MonitoredTask.State.COMPLETE, task.getState());
051    
052    // It should still show up in the TaskMonitor list
053    assertEquals(1, tm.getTasks().size());
054    
055    // If we mark its completion time back a few minutes, it should get gced
056    task.expireNow();
057    assertEquals(0, tm.getTasks().size());
058  }
059  
060  @Test
061  public void testTasksGetAbortedOnLeak() throws InterruptedException {
062    final TaskMonitor tm = new TaskMonitor();
063    assertTrue("Task monitor should start empty",
064        tm.getTasks().isEmpty());
065    
066    final AtomicBoolean threadSuccess = new AtomicBoolean(false);
067    // Make a task in some other thread and leak it
068    Thread t = new Thread() {
069      @Override
070      public void run() {
071        MonitoredTask task = tm.createStatus("Test task");    
072        assertEquals(MonitoredTask.State.RUNNING, task.getState());
073        threadSuccess.set(true);
074      }
075    };
076    t.start();
077    t.join();
078    // Make sure the thread saw the correct state
079    assertTrue(threadSuccess.get());
080    
081    // Make sure the leaked reference gets cleared
082    System.gc();
083    System.gc();
084    System.gc();
085    
086    // Now it should be aborted 
087    MonitoredTask taskFromTm = tm.getTasks().get(0);
088    assertEquals(MonitoredTask.State.ABORTED, taskFromTm.getState());
089  }
090  
091  @Test
092  public void testTaskLimit() throws Exception {
093    TaskMonitor tm = new TaskMonitor();
094    for (int i = 0; i < TaskMonitor.MAX_TASKS + 10; i++) {
095      tm.createStatus("task " + i);
096    }
097    // Make sure it was limited correctly
098    assertEquals(TaskMonitor.MAX_TASKS, tm.getTasks().size());
099    // Make sure we culled the earlier tasks, not later
100    // (i.e. tasks 0 through 9 should have been deleted)
101    assertEquals("task 10", tm.getTasks().get(0).getDescription());
102  }
