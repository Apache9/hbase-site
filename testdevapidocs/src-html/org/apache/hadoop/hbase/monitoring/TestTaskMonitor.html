041  @Test
042  public void testTaskMonitorBasics() {
043    TaskMonitor tm = new TaskMonitor(new Configuration());
044    assertTrue("Task monitor should start empty",
045        tm.getTasks().isEmpty());
046    
047    // Make a task and fetch it back out
048    MonitoredTask task = tm.createStatus("Test task");
049    MonitoredTask taskFromTm = tm.getTasks().get(0);
050    
051    // Make sure the state is reasonable.
052    assertEquals(task.getDescription(), taskFromTm.getDescription());
053    assertEquals(-1, taskFromTm.getCompletionTimestamp());
054    assertEquals(MonitoredTask.State.RUNNING, taskFromTm.getState());
055    
056    // Mark it as finished
057    task.markComplete("Finished!");
058    assertEquals(MonitoredTask.State.COMPLETE, task.getState());
059    
060    // It should still show up in the TaskMonitor list
061    assertEquals(1, tm.getTasks().size());
062    
063    // If we mark its completion time back a few minutes, it should get gced
064    task.expireNow();
065    assertEquals(0, tm.getTasks().size());
066
067    tm.shutdown();
068  }

070  @Test
071  public void testTasksGetAbortedOnLeak() throws InterruptedException {
072    final TaskMonitor tm = new TaskMonitor(new Configuration());
073    assertTrue("Task monitor should start empty",
074        tm.getTasks().isEmpty());
075    
076    final AtomicBoolean threadSuccess = new AtomicBoolean(false);
077    // Make a task in some other thread and leak it
078    Thread t = new Thread() {
079      @Override
080      public void run() {
081        MonitoredTask task = tm.createStatus("Test task");    
082        assertEquals(MonitoredTask.State.RUNNING, task.getState());
083        threadSuccess.set(true);
084      }
085    };
086    t.start();
087    t.join();
088    // Make sure the thread saw the correct state
089    assertTrue(threadSuccess.get());
090    
091    // Make sure the leaked reference gets cleared
092    System.gc();
093    System.gc();
094    System.gc();
095    
096    // Now it should be aborted 
097    MonitoredTask taskFromTm = tm.getTasks().get(0);
098    assertEquals(MonitoredTask.State.ABORTED, taskFromTm.getState());
099
100    tm.shutdown();
101  }

103  @Test
104  public void testTaskLimit() throws Exception {
105    TaskMonitor tm = new TaskMonitor(new Configuration());
106    for (int i = 0; i < TaskMonitor.DEFAULT_MAX_TASKS + 10; i++) {
107      tm.createStatus("task " + i);
108    }
109    // Make sure it was limited correctly
110    assertEquals(TaskMonitor.DEFAULT_MAX_TASKS, tm.getTasks().size());
111    // Make sure we culled the earlier tasks, not later
112    // (i.e. tasks 0 through 9 should have been deleted)
113    assertEquals("task 10", tm.getTasks().get(0).getDescription());
114    tm.shutdown();
115  }

117  @Test
118  public void testDoNotPurgeRPCTask() throws Exception {
119    int RPCTaskNums = 10;
120    TaskMonitor tm = TaskMonitor.get();
121    for(int i = 0; i < RPCTaskNums; i++) {
122      tm.createRPCStatus("PRCTask" + i);
123    }
124    for(int i = 0; i < TaskMonitor.DEFAULT_MAX_TASKS; i++) {
125      tm.createStatus("otherTask" + i);
126    }
127    int remainRPCTask = 0;
128    for(MonitoredTask task: tm.getTasks()) {
129      if(task instanceof MonitoredRPCHandler) {
130        remainRPCTask++;
131      }
132    }
133    assertEquals("RPC Tasks have been purged!", RPCTaskNums, remainRPCTask);
134    tm.shutdown();
135  }

137  @Test
138  public void testWarnStuckTasks() throws Exception {
139    final int INTERVAL = 1000;
140    Configuration conf = new Configuration();
141    conf.setLong(TaskMonitor.RPC_WARN_TIME_KEY, INTERVAL);
142    conf.setLong(TaskMonitor.MONITOR_INTERVAL_KEY, INTERVAL);
143    final TaskMonitor tm = new TaskMonitor(conf);
144    MonitoredRPCHandler t = tm.createRPCStatus("test task");
145    long then = EnvironmentEdgeManager.currentTime();
146    t.setRPC("testMethod", new Object[0], then);
147    Thread.sleep(INTERVAL * 2);
148    assertTrue("We did not warn", t.getWarnTime() > then);
149    tm.shutdown();
150  }

152  @Test
153  public void testGetTasksWithFilter() throws Exception {
154    TaskMonitor tm = new TaskMonitor(new Configuration());
155    assertTrue("Task monitor should start empty", tm.getTasks().isEmpty());
156    // Create 5 general tasks
157    tm.createStatus("General task1");
158    tm.createStatus("General task2");
159    tm.createStatus("General task3");
160    tm.createStatus("General task4");
161    tm.createStatus("General task5");
162    // Create 5 rpc tasks, and mark 1 completed
163    int length = 5;
164    ArrayList<MonitoredRPCHandler> rpcHandlers = new ArrayList<>(length);
165    for (int i = 0; i < length; i++) {
166      MonitoredRPCHandler rpcHandler = tm.createRPCStatus("Rpc task" + i);
167      rpcHandlers.add(rpcHandler);
168    }
169    // Create rpc opertions
170    byte[] row = new byte[] { 0x01 };
171    Mutation m = new Put(row);
172    Query q = new Scan();
173    String notOperation = "for test";
174    rpcHandlers.get(0).setRPC("operations", new Object[]{ m, q }, 3000);
175    rpcHandlers.get(1).setRPC("operations", new Object[]{ m, q }, 3000);
176    rpcHandlers.get(2).setRPC("operations", new Object[]{ m, q }, 3000);
177    rpcHandlers.get(3).setRPC("operations", new Object[]{ notOperation }, 3000);
178    rpcHandlers.get(4).setRPC("operations", new Object[]{ m, q }, 3000);
179    MonitoredRPCHandler completed = rpcHandlers.get(4);
180    completed.markComplete("Completed!");
181    // Test get tasks with filter
182    List<MonitoredTask> generalTasks = tm.getTasks("general");
183    assertEquals(5, generalTasks.size());
184    List<MonitoredTask> handlerTasks = tm.getTasks("handler");
185    assertEquals(5, handlerTasks.size());
186    List<MonitoredTask> rpcTasks = tm.getTasks("rpc");
187    // The last rpc handler is stopped
188    assertEquals(4, rpcTasks.size());
189    List<MonitoredTask> operationTasks = tm.getTasks("operation");
190    // Handler 3 doesn't handle Operation.
191    assertEquals(3, operationTasks.size());
192    tm.shutdown();
193  }
