017package org.apache.hadoop.hbase;
018
019import static org.junit.Assert.assertTrue;
020
021import org.apache.hadoop.hbase.io.HeapSize;
022import org.apache.hadoop.hbase.testclassification.SmallTests;
023import org.apache.hadoop.hbase.util.Bytes;
024import org.junit.Test;
025import org.junit.experimental.categories.Category;
026
027@Category(SmallTests.class)
028public class TestTagRewriteCell {
029
030  @Test
031  public void testHeapSize() {
032    Cell originalCell = CellUtil.createCell(Bytes.toBytes("row"), Bytes.toBytes("value"));
033    final int fakeTagArrayLength = 10;
034    Cell trCell = CellUtil.createCell(originalCell, new byte[fakeTagArrayLength]);
035
036    // Get the heapSize before the internal tags array in trCell are nuked
037    long trCellHeapSize = ((HeapSize)trCell).heapSize();
038
039    // Make another TagRewriteCell with the original TagRewriteCell
040    // This happens on systems with more than one RegionObserver/Coproc loaded (such as
041    // VisibilityController and AccessController)
042    Cell trCell2 = CellUtil.createCell(trCell, new byte[fakeTagArrayLength]);
043
044    assertTrue("TagRewriteCell containing a TagRewriteCell's heapsize should be larger than a " +
045        "single TagRewriteCell's heapsize", trCellHeapSize < ((HeapSize)trCell2).heapSize());
046    assertTrue("TagRewriteCell should have had nulled out tags array", ((HeapSize)trCell).heapSize() <
047        trCellHeapSize);
048  }
049}
