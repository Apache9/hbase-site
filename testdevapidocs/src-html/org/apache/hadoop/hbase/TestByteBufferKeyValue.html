034@Category({ MiscTests.class, SmallTests.class })
035public class TestByteBufferKeyValue {
036  private static final String QUAL2 = "qual2";
037  private static final String FAM2 = "fam2";
038  private static final String QUAL1 = "qual1";
039  private static final String FAM1 = "fam1";
040  private static final String ROW1 = "row1";
041  private static final byte[] row1 = Bytes.toBytes(ROW1);
042  private static final byte[] fam1 = Bytes.toBytes(FAM1);
043  private static final byte[] fam2 = Bytes.toBytes(FAM2);
044  private static final byte[] qual1 = Bytes.toBytes(QUAL1);
045  private static final byte[] qual2 = Bytes.toBytes(QUAL2);
046  private static final Tag t1 = new ArrayBackedTag((byte) 1, Bytes.toBytes("TAG1"));
047  private static final Tag t2 = new ArrayBackedTag((byte) 2, Bytes.toBytes("TAG2"));
048  private static final ArrayList<Tag> tags = new ArrayList<Tag>();
049  static {
050    tags.add(t1);
051    tags.add(t2);
052  }
053
054  @Test
055  public void testByteBufferBackedKeyValue() throws Exception {
056    KeyValue kvCell = new KeyValue(row1, fam1, qual1, 0L, Type.Put, row1);
057    ByteBuffer buf = ByteBuffer.allocateDirect(kvCell.getBuffer().length);
058    ByteBufferUtils.copyFromArrayToBuffer(buf, kvCell.getBuffer(), 0, kvCell.getBuffer().length);
059    ByteBufferCell offheapKV = new ByteBufferKeyValue(buf, 0, buf.capacity(), 0L);
060    assertEquals(
061      ROW1,
062      ByteBufferUtils.toStringBinary(offheapKV.getRowByteBuffer(),
063        offheapKV.getRowPosition(), offheapKV.getRowLength()));
064    assertEquals(
065      FAM1,
066      ByteBufferUtils.toStringBinary(offheapKV.getFamilyByteBuffer(),
067        offheapKV.getFamilyPosition(), offheapKV.getFamilyLength()));
068    assertEquals(
069      QUAL1,
070      ByteBufferUtils.toStringBinary(offheapKV.getQualifierByteBuffer(),
071        offheapKV.getQualifierPosition(), offheapKV.getQualifierLength()));
072    assertEquals(
073      ROW1,
074      ByteBufferUtils.toStringBinary(offheapKV.getValueByteBuffer(),
075        offheapKV.getValuePosition(), offheapKV.getValueLength()));
076    assertEquals(0L, offheapKV.getTimestamp());
077    assertEquals(Type.Put.getCode(), offheapKV.getTypeByte());
078
079    // Use the array() APIs
080    assertEquals(
081      ROW1,
082      Bytes.toStringBinary(offheapKV.getRowArray(),
083        offheapKV.getRowOffset(), offheapKV.getRowLength()));
084    assertEquals(
085      FAM1,
086      Bytes.toStringBinary(offheapKV.getFamilyArray(),
087        offheapKV.getFamilyOffset(), offheapKV.getFamilyLength()));
088    assertEquals(
089      QUAL1,
090      Bytes.toStringBinary(offheapKV.getQualifierArray(),
091        offheapKV.getQualifierOffset(), offheapKV.getQualifierLength()));
092    assertEquals(
093      ROW1,
094      Bytes.toStringBinary(offheapKV.getValueArray(),
095        offheapKV.getValueOffset(), offheapKV.getValueLength()));
096    assertEquals(0L, offheapKV.getTimestamp());
097    assertEquals(Type.Put.getCode(), offheapKV.getTypeByte());
098
099    kvCell = new KeyValue(row1, fam2, qual2, 0L, Type.Put, row1);
100    buf = ByteBuffer.allocateDirect(kvCell.getBuffer().length);
101    ByteBufferUtils.copyFromArrayToBuffer(buf, kvCell.getBuffer(), 0, kvCell.getBuffer().length);
102    offheapKV = new ByteBufferKeyValue(buf, 0, buf.capacity(), 0L);
103    assertEquals(
104      FAM2,
105      ByteBufferUtils.toStringBinary(offheapKV.getFamilyByteBuffer(),
106        offheapKV.getFamilyPosition(), offheapKV.getFamilyLength()));
107    assertEquals(
108      QUAL2,
109      ByteBufferUtils.toStringBinary(offheapKV.getQualifierByteBuffer(),
110        offheapKV.getQualifierPosition(), offheapKV.getQualifierLength()));
111    byte[] nullQualifier = new byte[0];
112    kvCell = new KeyValue(row1, fam1, nullQualifier, 0L, Type.Put, row1);
113    buf = ByteBuffer.allocateDirect(kvCell.getBuffer().length);
114    ByteBufferUtils.copyFromArrayToBuffer(buf, kvCell.getBuffer(), 0, kvCell.getBuffer().length);
115    offheapKV = new ByteBufferKeyValue(buf, 0, buf.capacity(), 0L);
116    assertEquals(
117      ROW1,
118      ByteBufferUtils.toStringBinary(offheapKV.getRowByteBuffer(),
119        offheapKV.getRowPosition(), offheapKV.getRowLength()));
120    assertEquals(
121      FAM1,
122      ByteBufferUtils.toStringBinary(offheapKV.getFamilyByteBuffer(),
123        offheapKV.getFamilyPosition(), offheapKV.getFamilyLength()));
124    assertEquals(
125      "",
126      ByteBufferUtils.toStringBinary(offheapKV.getQualifierByteBuffer(),
127        offheapKV.getQualifierPosition(), offheapKV.getQualifierLength()));
128    assertEquals(
129      ROW1,
130      ByteBufferUtils.toStringBinary(offheapKV.getValueByteBuffer(),
131        offheapKV.getValuePosition(), offheapKV.getValueLength()));
132    assertEquals(0L, offheapKV.getTimestamp());
133    assertEquals(Type.Put.getCode(), offheapKV.getTypeByte());
134  }

136  @Test
137  public void testByteBufferBackedKeyValueWithTags() throws Exception {
138    KeyValue kvCell = new KeyValue(row1, fam1, qual1, 0L, Type.Put, row1, tags);
139    ByteBuffer buf = ByteBuffer.allocateDirect(kvCell.getBuffer().length);
140    ByteBufferUtils.copyFromArrayToBuffer(buf, kvCell.getBuffer(), 0, kvCell.getBuffer().length);
141    ByteBufferKeyValue offheapKV = new ByteBufferKeyValue(buf, 0, buf.capacity(), 0L);
142    assertEquals(
143      ROW1,
144      ByteBufferUtils.toStringBinary(offheapKV.getRowByteBuffer(),
145        offheapKV.getRowPosition(), offheapKV.getRowLength()));
146    assertEquals(
147      FAM1,
148      ByteBufferUtils.toStringBinary(offheapKV.getFamilyByteBuffer(),
149        offheapKV.getFamilyPosition(), offheapKV.getFamilyLength()));
150    assertEquals(
151      QUAL1,
152      ByteBufferUtils.toStringBinary(offheapKV.getQualifierByteBuffer(),
153        offheapKV.getQualifierPosition(), offheapKV.getQualifierLength()));
154    assertEquals(
155      ROW1,
156      ByteBufferUtils.toStringBinary(offheapKV.getValueByteBuffer(),
157        offheapKV.getValuePosition(), offheapKV.getValueLength()));
158    assertEquals(0L, offheapKV.getTimestamp());
159    assertEquals(Type.Put.getCode(), offheapKV.getTypeByte());
160    // change tags to handle both onheap and offheap stuff
161    List<Tag> resTags = offheapKV.getTags();
162    Tag tag1 = resTags.get(0);
163    assertEquals(t1.getType(), tag1.getType());
164    assertEquals(Tag.getValueAsString(t1),
165      Tag.getValueAsString(tag1));
166    Tag tag2 = resTags.get(1);
167    assertEquals(tag2.getType(), tag2.getType());
168    assertEquals(Tag.getValueAsString(t2),
169      Tag.getValueAsString(tag2));
170    Tag res = PrivateCellUtil.getTag(offheapKV, (byte) 2).get();
171    assertEquals(Tag.getValueAsString(t2),
172      Tag.getValueAsString(tag2));
173    assertFalse(PrivateCellUtil.getTag(offheapKV, (byte) 3).isPresent());
174  }
