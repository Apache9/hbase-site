157   * Make sure we can use the cluster
158   * @throws Exception
159   */
160  private void testSanity(final String testName) throws Exception{
161    String tableName = testName + "_" + System.currentTimeMillis();
162    HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(tableName));
163    HColumnDescriptor family = new HColumnDescriptor("fam");
164    desc.addFamily(family);
165    LOG.info("Creating table " + tableName);
166    Admin admin = TEST_UTIL.getAdmin();
167    try {
168      admin.createTable(desc);
169    } finally {
170      admin.close();
171    }
172
173    Table table = TEST_UTIL.getConnection().getTable(desc.getTableName());
174    Put put = new Put(Bytes.toBytes("testrow"));
175    put.addColumn(Bytes.toBytes("fam"), Bytes.toBytes("col"), Bytes.toBytes("testdata"));
176    LOG.info("Putting table " + tableName);
177    table.put(put);
178    table.close();
179  }

207  @Test

211      TestZooKeeper.class.getName(), null);
212    ZKUtil.createWithParents(zkw, "/l1/l2/l3/l4");

216    } catch (KeeperException ex) {
217      assertNotNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2/l3/l4", null));
218    }
219    ZKUtil.deleteNodeRecursively(zkw, "/l1/l2");
220    // make sure it really is deleted
221    assertNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2/l3/l4", null));

223    // do the same delete again and make sure it doesn't crash
224    ZKUtil.deleteNodeRecursively(zkw, "/l1/l2");
225
226    ZKUtil.deleteNode(zkw, "/l1");
227    assertNull(ZKUtil.getDataNoWatch(zkw, "/l1/l2", null));

230  /**
231   * A test for HBASE-3238
232   * @throws IOException A connection attempt to zk failed
233   * @throws InterruptedException One of the non ZKUtil actions was interrupted
234   * @throws KeeperException Any of the zookeeper connections had a
235   * KeeperException
236   */
237  @Test
238  public void testCreateSilentIsReallySilent() throws InterruptedException,
239      KeeperException, IOException {
240    Configuration c = TEST_UTIL.getConfiguration();
241
242    String aclZnode = "/aclRoot";
243    String quorumServers = ZKConfig.getZKQuorumServersString(c);
244    int sessionTimeout = 5 * 1000; // 5 seconds
245    ZooKeeper zk = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
246    zk.addAuthInfo("digest", "hbase:rox".getBytes());
247
248    // Assumes the  root of the ZooKeeper space is writable as it creates a node
249    // wherever the cluster home is defined.
250    ZooKeeperWatcher zk2 = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(),
251      "testCreateSilentIsReallySilent", null);
252
253    // Save the previous ACL
254    Stat s =  null;
255    List<ACL> oldACL = null;
256    while (true) {
257      try {
258        s = new Stat();
259        oldACL = zk.getACL("/", s);
260        break;
261      } catch (KeeperException e) {
262        switch (e.code()) {
263          case CONNECTIONLOSS:
264          case SESSIONEXPIRED:
265          case OPERATIONTIMEOUT:
266            LOG.warn("Possibly transient ZooKeeper exception", e);
267            Threads.sleep(100);
268            break;
269         default:
270            throw e;
271        }
272      }
273    }
274
275    // I set this acl after the attempted creation of the cluster home node.
276    // Add retries in case of retryable zk exceptions.
277    while (true) {
278      try {
279        zk.setACL("/", ZooDefs.Ids.CREATOR_ALL_ACL, -1);
280        break;
281      } catch (KeeperException e) {
282        switch (e.code()) {
283          case CONNECTIONLOSS:
284          case SESSIONEXPIRED:
285          case OPERATIONTIMEOUT:
286            LOG.warn("Possibly transient ZooKeeper exception: " + e);
287            Threads.sleep(100);
288            break;
289         default:
290            throw e;
291        }
292      }
293    }
294
295    while (true) {
296      try {
297        zk.create(aclZnode, null, ZooDefs.Ids.CREATOR_ALL_ACL, CreateMode.PERSISTENT);
298        break;
299      } catch (KeeperException e) {
300        switch (e.code()) {
301          case CONNECTIONLOSS:
302          case SESSIONEXPIRED:
303          case OPERATIONTIMEOUT:
304            LOG.warn("Possibly transient ZooKeeper exception: " + e);
305            Threads.sleep(100);
306            break;
307         default:
308            throw e;
309        }
310      }
311    }
312    zk.close();
313    ZKUtil.createAndFailSilent(zk2, aclZnode);
314
315    // Restore the ACL
316    ZooKeeper zk3 = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
317    zk3.addAuthInfo("digest", "hbase:rox".getBytes());
318    try {
319      zk3.setACL("/", oldACL, -1);
320    } finally {
321      zk3.close();
322    }
323 }

337  /**
338   * Tests that the master does not call retainAssignment after recovery from expired zookeeper
339   * session. Without the HBASE-6046 fix master always tries to assign all the user regions by
340   * calling retainAssignment.
341   */
342  @Test(timeout = 300000)
343  public void testRegionAssignmentAfterMasterRecoveryDueToZKExpiry() throws Exception {
344    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
345    cluster.startRegionServer();
346    cluster.waitForActiveAndReadyMaster(10000);
347    HMaster m = cluster.getMaster();
348    final ZooKeeperWatcher zkw = m.getZooKeeper();
349    // now the cluster is up. So assign some regions.
350    try (Admin admin = TEST_UTIL.getAdmin()) {
351      byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("a"), Bytes.toBytes("b"),
352          Bytes.toBytes("c"), Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
353          Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i"), Bytes.toBytes("j") };
354      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
355      htd.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
356      admin.createTable(htd, SPLIT_KEYS);
357      TEST_UTIL.waitUntilNoRegionsInTransition(60000);
358      m.getZooKeeper().close();
359      MockLoadBalancer.retainAssignCalled = false;
360      final int expectedNumOfListeners = countPermanentListeners(zkw);
361      m.abort("Test recovery from zk session expired",
362          new KeeperException.SessionExpiredException());
363      assertTrue(m.isStopped()); // Master doesn't recover any more
364      // The recovered master should not call retainAssignment, as it is not a
365      // clean startup.
366      assertFalse("Retain assignment should not be called", MockLoadBalancer.retainAssignCalled);
367      // number of listeners should be same as the value before master aborted
368      // wait for new master is initialized
369      cluster.waitForActiveAndReadyMaster(120000);
370      final HMaster newMaster = cluster.getMasterThread().getMaster();
371      assertEquals(expectedNumOfListeners, countPermanentListeners(newMaster.getZooKeeper()));
372    }
373  }
374
375  /**
376   * Count listeners in zkw excluding listeners, that belongs to workers or other
377   * temporary processes.
378   */
379  private int countPermanentListeners(ZooKeeperWatcher watcher) {
380    return countListeners(watcher, ZkSplitLogWorkerCoordination.class);
381  }
382
383  /**
384   * Count listeners in zkw excluding provided classes
385   */
386  private int countListeners(ZooKeeperWatcher watcher, Class<?>... exclude) {
387    int cnt = 0;
388    for (Object o : watcher.getListeners()) {
389      boolean skip = false;
390      for (Class<?> aClass : exclude) {
391        if (aClass.isAssignableFrom(o.getClass())) {
392          skip = true;
393          break;
394        }
395      }
396      if (!skip) {
397        cnt += 1;
398      }
399    }
400    return cnt;
401  }

403  /**
404   * Tests whether the logs are split when master recovers from a expired zookeeper session and an
405   * RS goes down.
406   */
407  @Test(timeout = 300000)
408  public void testLogSplittingAfterMasterRecoveryDueToZKExpiry() throws Exception {
409    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
410    cluster.startRegionServer();
411    HMaster m = cluster.getMaster();
412    // now the cluster is up. So assign some regions.
413    Admin admin = TEST_UTIL.getAdmin();
414    Table table = null;
415    try {
416      byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("1"), Bytes.toBytes("2"),
417        Bytes.toBytes("3"), Bytes.toBytes("4"), Bytes.toBytes("5") };
418
419      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
420      HColumnDescriptor hcd = new HColumnDescriptor("col");
421      htd.addFamily(hcd);
422      admin.createTable(htd, SPLIT_KEYS);
423      TEST_UTIL.waitUntilNoRegionsInTransition(60000);
424      table = TEST_UTIL.getConnection().getTable(htd.getTableName());
425      Put p;
426      int numberOfPuts;
427      for (numberOfPuts = 0; numberOfPuts < 6; numberOfPuts++) {
428        p = new Put(Bytes.toBytes(numberOfPuts));
429        p.addColumn(Bytes.toBytes("col"), Bytes.toBytes("ql"),
430                Bytes.toBytes("value" + numberOfPuts));
431        table.put(p);
432      }
433      m.getZooKeeper().close();
434      m.abort("Test recovery from zk session expired",
435        new KeeperException.SessionExpiredException());
436      assertTrue(m.isStopped()); // Master doesn't recover any more
437      cluster.getRegionServer(0).abort("Aborting");
438      // Without patch for HBASE-6046 this test case will always timeout
439      // with patch the test case should pass.
440      Scan scan = new Scan();
441      int numberOfRows = 0;
442      ResultScanner scanner = table.getScanner(scan);
443      Result[] result = scanner.next(1);
444      while (result != null && result.length > 0) {
445        numberOfRows++;
446        result = scanner.next(1);
447      }
448      assertEquals("Number of rows should be equal to number of puts.", numberOfPuts,
449        numberOfRows);
450    } finally {
451      if (table != null) table.close();
452      admin.close();
453    }
454  }
