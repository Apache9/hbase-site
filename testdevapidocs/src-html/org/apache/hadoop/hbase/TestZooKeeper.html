133  private ZooKeeperWatcher getZooKeeperWatcher(Connection c)
134  throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
135    Method getterZK = c.getClass().getDeclaredMethod("getKeepAliveZooKeeperWatcher");

277    Table localMeta = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME);

284    final byte [] row = new byte [] {'r'};
285    localMeta.exists(new Get(row));
286    ipMeta.exists(new Get(row));
287

289    ZooKeeperWatcher z1 =
290      getZooKeeperWatcher(ConnectionFactory.createConnection(localMeta.getConfiguration()));

292      getZooKeeperWatcher(ConnectionFactory.createConnection(otherConf));

294    assertFalse(z1.getQuorum().equals(z2.getQuorum()));
295
296    localMeta.close();
297    ipMeta.close();

350  /**
351   * A test for HBASE-3238
352   * @throws IOException A connection attempt to zk failed
353   * @throws InterruptedException One of the non ZKUtil actions was interrupted
354   * @throws KeeperException Any of the zookeeper connections had a
355   * KeeperException
356   */
357  @Test
358  public void testCreateSilentIsReallySilent() throws InterruptedException,
359      KeeperException, IOException {
360    Configuration c = TEST_UTIL.getConfiguration();
361
362    String aclZnode = "/aclRoot";
363    String quorumServers = ZKConfig.getZKQuorumServersString(c);
364    int sessionTimeout = 5 * 1000; // 5 seconds
365    ZooKeeper zk = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
366    zk.addAuthInfo("digest", "hbase:rox".getBytes());
367
368    // Assumes the  root of the ZooKeeper space is writable as it creates a node
369    // wherever the cluster home is defined.
370    ZooKeeperWatcher zk2 = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(),
371      "testCreateSilentIsReallySilent", null);
372
373    // Save the previous ACL
374    Stat s =  null;
375    List<ACL> oldACL = null;
376    while (true) {
377      try {
378        s = new Stat();
379        oldACL = zk.getACL("/", s);
380        break;
381      } catch (KeeperException e) {
382        switch (e.code()) {
383          case CONNECTIONLOSS:
384          case SESSIONEXPIRED:
385          case OPERATIONTIMEOUT:
386            LOG.warn("Possibly transient ZooKeeper exception", e);
387            Threads.sleep(100);
388            break;
389         default:
390            throw e;
391        }
392      }
393    }
394
395    // I set this acl after the attempted creation of the cluster home node.
396    // Add retries in case of retryable zk exceptions.
397    while (true) {
398      try {
399        zk.setACL("/", ZooDefs.Ids.CREATOR_ALL_ACL, -1);
400        break;
401      } catch (KeeperException e) {
402        switch (e.code()) {
403          case CONNECTIONLOSS:
404          case SESSIONEXPIRED:
405          case OPERATIONTIMEOUT:
406            LOG.warn("Possibly transient ZooKeeper exception: " + e);
407            Threads.sleep(100);
408            break;
409         default:
410            throw e;
411        }
412      }
413    }
414
415    while (true) {
416      try {
417        zk.create(aclZnode, null, ZooDefs.Ids.CREATOR_ALL_ACL, CreateMode.PERSISTENT);
418        break;
419      } catch (KeeperException e) {
420        switch (e.code()) {
421          case CONNECTIONLOSS:
422          case SESSIONEXPIRED:
423          case OPERATIONTIMEOUT:
424            LOG.warn("Possibly transient ZooKeeper exception: " + e);
425            Threads.sleep(100);
426            break;
427         default:
428            throw e;
429        }
430      }
431    }
432    zk.close();
433    ZKUtil.createAndFailSilent(zk2, aclZnode);
434
435    // Restore the ACL
436    ZooKeeper zk3 = new ZooKeeper(quorumServers, sessionTimeout, EmptyWatcher.instance);
437    zk3.addAuthInfo("digest", "hbase:rox".getBytes());
438    try {
439      zk3.setACL("/", oldACL, -1);
440    } finally {
441      zk3.close();
442    }
443 }

457  /**
458   * Tests that the master does not call retainAssignment after recovery from expired zookeeper
459   * session. Without the HBASE-6046 fix master always tries to assign all the user regions by
460   * calling retainAssignment.
461   */
462  @Test(timeout = 300000)
463  public void testRegionAssignmentAfterMasterRecoveryDueToZKExpiry() throws Exception {
464    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
465    cluster.startRegionServer();
466    cluster.waitForActiveAndReadyMaster(10000);
467    HMaster m = cluster.getMaster();
468    final ZooKeeperWatcher zkw = m.getZooKeeper();
469    // now the cluster is up. So assign some regions.
470    try (Admin admin = TEST_UTIL.getAdmin()) {
471      byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("a"), Bytes.toBytes("b"),
472          Bytes.toBytes("c"), Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
473          Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i"), Bytes.toBytes("j") };
474      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
475      htd.addFamily(new HColumnDescriptor(HConstants.CATALOG_FAMILY));
476      admin.createTable(htd, SPLIT_KEYS);
477      TEST_UTIL.waitUntilNoRegionsInTransition(60000);
478      m.getZooKeeper().close();
479      MockLoadBalancer.retainAssignCalled = false;
480      final int expectedNumOfListeners = countPermanentListeners(zkw);
481      m.abort("Test recovery from zk session expired",
482          new KeeperException.SessionExpiredException());
483      assertTrue(m.isStopped()); // Master doesn't recover any more
484      // The recovered master should not call retainAssignment, as it is not a
485      // clean startup.
486      assertFalse("Retain assignment should not be called", MockLoadBalancer.retainAssignCalled);
487      // number of listeners should be same as the value before master aborted
488      // wait for new master is initialized
489      cluster.waitForActiveAndReadyMaster(120000);
490      final HMaster newMaster = cluster.getMasterThread().getMaster();
491      assertEquals(expectedNumOfListeners, countPermanentListeners(newMaster.getZooKeeper()));
492    }
493  }

495  /**
496   * Count listeners in zkw excluding listeners, that belongs to workers or other
497   * temporary processes.
498   */
499  private int countPermanentListeners(ZooKeeperWatcher watcher) {
500    return countListeners(watcher, ZkSplitLogWorkerCoordination.class);
501  }
502
503  /**
504   * Count listeners in zkw excluding provided classes
505   */
506  private int countListeners(ZooKeeperWatcher watcher, Class<?>... exclude) {
507    int cnt = 0;
508    for (Object o : watcher.getListeners()) {
509      boolean skip = false;
510      for (Class<?> aClass : exclude) {
511        if (aClass.isAssignableFrom(o.getClass())) {
512          skip = true;
513          break;
514        }
515      }
516      if (!skip) {
517        cnt += 1;
518      }
519    }
520    return cnt;
521  }

523  /**
524   * Tests whether the logs are split when master recovers from a expired zookeeper session and an
525   * RS goes down.
526   */
527  @Test(timeout = 300000)
528  public void testLogSplittingAfterMasterRecoveryDueToZKExpiry() throws Exception {
529    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
530    cluster.startRegionServer();
531    HMaster m = cluster.getMaster();
532    // now the cluster is up. So assign some regions.
533    Admin admin = TEST_UTIL.getAdmin();
534    Table table = null;
535    try {
536      byte[][] SPLIT_KEYS = new byte[][] { Bytes.toBytes("1"), Bytes.toBytes("2"),
537        Bytes.toBytes("3"), Bytes.toBytes("4"), Bytes.toBytes("5") };
538
539      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name.getMethodName()));
540      HColumnDescriptor hcd = new HColumnDescriptor("col");
541      htd.addFamily(hcd);
542      admin.createTable(htd, SPLIT_KEYS);
543      TEST_UTIL.waitUntilNoRegionsInTransition(60000);
544      table = TEST_UTIL.getConnection().getTable(htd.getTableName());
545      Put p;
546      int numberOfPuts;
547      for (numberOfPuts = 0; numberOfPuts < 6; numberOfPuts++) {
548        p = new Put(Bytes.toBytes(numberOfPuts));
549        p.addColumn(Bytes.toBytes("col"), Bytes.toBytes("ql"),
550                Bytes.toBytes("value" + numberOfPuts));
551        table.put(p);
552      }
553      m.getZooKeeper().close();
554      m.abort("Test recovery from zk session expired",
555        new KeeperException.SessionExpiredException());
556      assertTrue(m.isStopped()); // Master doesn't recover any more
557      cluster.getRegionServer(0).abort("Aborting");
558      // Without patch for HBASE-6046 this test case will always timeout
559      // with patch the test case should pass.
560      Scan scan = new Scan();
561      int numberOfRows = 0;
562      ResultScanner scanner = table.getScanner(scan);
563      Result[] result = scanner.next(1);
564      while (result != null && result.length > 0) {
565        numberOfRows++;
566        result = scanner.next(1);
567      }
568      assertEquals("Number of rows should be equal to number of puts.", numberOfPuts,
569        numberOfRows);
570    } finally {
571      if (table != null) table.close();
572      admin.close();
573    }
574  }
