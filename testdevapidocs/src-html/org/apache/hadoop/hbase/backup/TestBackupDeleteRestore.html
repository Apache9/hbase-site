029
030import org.apache.hadoop.hbase.shaded.com.google.common.collect.Lists;
031
032@Category(MediumTests.class)
033public class TestBackupDeleteRestore extends TestBackupBase {
034
035  private static final Log LOG = LogFactory.getLog(TestBackupDeleteRestore.class);
036
037  /**
038   * Verify that load data- backup - delete some data - restore works as expected - deleted data get
039   * restored.
040   * @throws Exception
041   */
042  @Test
043  public void testBackupDeleteRestore() throws Exception {
044
045    LOG.info("test full restore on a single table empty table");
046
047    List<TableName> tables = Lists.newArrayList(table1);
048    String backupId = fullTableBackup(tables);
049    assertTrue(checkSucceeded(backupId));
050    LOG.info("backup complete");
051    int numRows = TEST_UTIL.countRows(table1);
052    HBaseAdmin hba = TEST_UTIL.getHBaseAdmin();
053    // delete row
054    try (Table table = TEST_UTIL.getConnection().getTable(table1)) {
055      Delete delete = new Delete("row0".getBytes());
056      table.delete(delete);
057      hba.flush(table1);
058    }
059
060    TableName[] tableset = new TableName[] { table1 };
061    TableName[] tablemap = null;// new TableName[] { table1_restore };
062    BackupAdmin client = getBackupAdmin();
063    client.restore(BackupUtils.createRestoreRequest(BACKUP_ROOT_DIR, backupId, false,
064      tableset, tablemap, true));
065
066    int numRowsAfterRestore = TEST_UTIL.countRows(table1);
067    assertEquals(numRows, numRowsAfterRestore);
068    hba.close();
069  }
070}
