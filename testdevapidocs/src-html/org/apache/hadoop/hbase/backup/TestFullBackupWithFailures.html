050@Category(LargeTests.class)
051public class TestFullBackupWithFailures extends TestBackupBase {
052
053  private static final Log LOG = LogFactory.getLog(TestFullBackupWithFailures.class);
054
055  static class FullTableBackupClientForTest extends FullTableBackupClient
056  {
057    public static final String BACKUP_TEST_MODE_STAGE = "backup.test.mode.stage";
058
059    public FullTableBackupClientForTest() {
060    }
061
062    public FullTableBackupClientForTest(Connection conn, String backupId, BackupRequest request)
063        throws IOException {
064      super(conn, backupId, request);
065    }
066
067    @Override
068    public void execute() throws IOException
069    {
070      // Get the stage ID to fail on
071      try (Admin admin = conn.getAdmin();) {
072        // Begin BACKUP
073        beginBackup(backupManager, backupInfo);
074        failStageIf(0);
075        String savedStartCode = null;
076        boolean firstBackup = false;
077        // do snapshot for full table backup
078        savedStartCode = backupManager.readBackupStartCode();
079        firstBackup = savedStartCode == null || Long.parseLong(savedStartCode) == 0L;
080        if (firstBackup) {
081          // This is our first backup. Let's put some marker to system table so that we can hold the logs
082          // while we do the backup.
083          backupManager.writeBackupStartCode(0L);
084        }
085        failStageIf(1);
086        // We roll log here before we do the snapshot. It is possible there is duplicate data
087        // in the log that is already in the snapshot. But if we do it after the snapshot, we
088        // could have data loss.
089        // A better approach is to do the roll log on each RS in the same global procedure as
090        // the snapshot.
091        LOG.info("Execute roll log procedure for full backup ...");
092
093        Map<String, String> props = new HashMap<String, String>();
094        props.put("backupRoot", backupInfo.getBackupRootDir());
095        admin.execProcedure(LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_SIGNATURE,
096          LogRollMasterProcedureManager.ROLLLOG_PROCEDURE_NAME, props);
097        failStageIf(2);
098        newTimestamps = backupManager.readRegionServerLastLogRollResult();
099        if (firstBackup) {
100          // Updates registered log files
101          // We record ALL old WAL files as registered, because
102          // this is a first full backup in the system and these
103          // files are not needed for next incremental backup
104          List<String> logFiles = BackupUtils.getWALFilesOlderThan(conf, newTimestamps);
105          backupManager.recordWALFiles(logFiles);
106        }
107
108        // SNAPSHOT_TABLES:
109        backupInfo.setPhase(BackupPhase.SNAPSHOT);
110        for (TableName tableName : tableList) {
111          String snapshotName =
112              "snapshot_" + Long.toString(EnvironmentEdgeManager.currentTime()) + "_"
113                  + tableName.getNamespaceAsString() + "_" + tableName.getQualifierAsString();
114
115          snapshotTable(admin, tableName, snapshotName);
116          backupInfo.setSnapshotName(tableName, snapshotName);
117        }
118        failStageIf(3);
119        // SNAPSHOT_COPY:
120        // do snapshot copy
121        LOG.debug("snapshot copy for " + backupId);
122        snapshotCopy(backupInfo);
123        // Updates incremental backup table set
124        backupManager.addIncrementalBackupTableSet(backupInfo.getTables());
125
126        // BACKUP_COMPLETE:
127        // set overall backup status: complete. Here we make sure to complete the backup.
128        // After this checkpoint, even if entering cancel process, will let the backup finished
129        backupInfo.setState(BackupState.COMPLETE);
130        // The table list in backupInfo is good for both full backup and incremental backup.
131        // For incremental backup, it contains the incremental backup table set.
132        backupManager.writeRegionServerLogTimestamp(backupInfo.getTables(), newTimestamps);
133
134        HashMap<TableName, HashMap<String, Long>> newTableSetTimestampMap =
135            backupManager.readLogTimestampMap();
136
137        Long newStartCode =
138            BackupUtils.getMinValue(BackupUtils
139                .getRSLogTimestampMins(newTableSetTimestampMap));
140        backupManager.writeBackupStartCode(newStartCode);
141        failStageIf(4);
142        // backup complete
143        completeBackup(conn, backupInfo, backupManager, BackupType.FULL, conf);
144
145      } catch (Exception e) {
146        failBackup(conn, backupInfo, backupManager, e, "Unexpected BackupException : ",
147          BackupType.FULL, conf);
148        throw new IOException(e);
149      }
150
151    }
152
153
154
155    @VisibleForTesting
156    protected int getTestStageId() {
157      return conf.getInt(BACKUP_TEST_MODE_STAGE, 0);
158    }
159
160    @VisibleForTesting
161
162    protected void failStageIf(int stage) throws IOException {
163      int current = getTestStageId();
164      if (current == stage) {
165        throw new IOException("Failed stage " + stage+" in testing");
166      }
167    }
168
169  }
170
171  @Test
172  public void testFullBackupWithFailures() throws Exception {
173    conf1.set(TableBackupClient.BACKUP_CLIENT_IMPL_CLASS,
174      FullTableBackupClientForTest.class.getName());
175    int stage = (new Random()).nextInt(5);
176    // Fail random stage between 0 and 4 inclusive
177    LOG.info("Running stage " + stage);
178    runBackupAndFailAtStage(stage);
179  }
180
181  public void runBackupAndFailAtStage(int stage) throws Exception {
182
183    conf1.setInt(FullTableBackupClientForTest.BACKUP_TEST_MODE_STAGE, stage);
184    try (BackupSystemTable table = new BackupSystemTable(TEST_UTIL.getConnection())) {
185      int before = table.getBackupHistory().size();
186      String[] args =
187          new String[] { "create", "full", BACKUP_ROOT_DIR, "-t",
188              table1.getNameAsString() + "," + table2.getNameAsString() };
189      // Run backup
190      int ret = ToolRunner.run(conf1, new BackupDriver(), args);
191      assertFalse(ret == 0);
192      List<BackupInfo> backups = table.getBackupHistory();
193      int after = table.getBackupHistory().size();
194
195      assertTrue(after ==  before +1);
196      for (BackupInfo data : backups) {
197        String backupId = data.getBackupId();
198        assertFalse(checkSucceeded(backupId));
199      }
200      Set<TableName> tables = table.getIncrementalBackupTableSet(BACKUP_ROOT_DIR);
201      assertTrue(tables.size() == 0);
202    }
203  }
