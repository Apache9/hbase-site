066  @Test
067  public void testSeqInsertAndDelete() {
068    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
069    assertTrue(tracker.isEmpty());
070
071    final int MIN_PROC = 1;
072    final int MAX_PROC = 1 << 10;
073
074    // sequential insert
075    for (int i = MIN_PROC; i < MAX_PROC; ++i) {
076      tracker.insert(i);
077
078      // All the proc that we inserted should not be deleted
079      for (int j = MIN_PROC; j <= i; ++j) {
080        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
081      }
082      // All the proc that are not yet inserted should be result as deleted
083      for (int j = i + 1; j < MAX_PROC; ++j) {
084        assertTrue(tracker.isDeleted(j) != ProcedureStoreTracker.DeleteState.NO);
085      }
086    }
087
088    // sequential delete
089    for (int i = MIN_PROC; i < MAX_PROC; ++i) {
090      tracker.delete(i);
091
092      // All the proc that we deleted should be deleted
093      for (int j = MIN_PROC; j <= i; ++j) {
094        assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(j));
095      }
096      // All the proc that are not yet deleted should be result as not deleted
097      for (int j = i + 1; j < MAX_PROC; ++j) {
098        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
099      }
100    }
101    assertTrue(tracker.isEmpty());
102  }

104  @Test
105  public void testPartialTracker() {
106    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
107    tracker.setPartialFlag(true);
108
109    // nothing in the tracker, the state is unknown
110    assertTrue(tracker.isEmpty());
111    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(1));
112    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
113
114    // Mark 1 as deleted, now that is a known state
115    tracker.setDeleted(1, true);
116    tracker.dump();
117    assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
118    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
119    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
120
121    // Mark 579 as non-deleted, now that is a known state
122    tracker.setDeleted(579, false);
123    assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
124    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
125    assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(579));
126    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(577));
127    assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(580));
128
129    tracker.setDeleted(579, true);
130    tracker.setPartialFlag(false);
131    assertTrue(tracker.isEmpty());
132  }

134  @Test
135  public void testIsTracking() {
136    long[][] procIds = new long[][] {{4, 7}, {1024, 1027}, {8192, 8194}};
137    long[][] checkIds = new long[][] {{2, 8}, {1023, 1025}, {8193, 8191}};
138
139    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
140    for (int i = 0; i < procIds.length; ++i) {
141      long[] seq = procIds[i];
142      tracker.insert(seq[0]);
143      tracker.insert(seq[1]);
144    }
145
146    for (int i = 0; i < procIds.length; ++i) {
147      long[] check = checkIds[i];
148      long[] seq = procIds[i];
149      assertTrue(tracker.isTracking(seq[0], seq[1]));
150      assertTrue(tracker.isTracking(check[0], check[1]));
151      tracker.delete(seq[0]);
152      tracker.delete(seq[1]);
153      assertFalse(tracker.isTracking(seq[0], seq[1]));
154      assertFalse(tracker.isTracking(check[0], check[1]));
155    }
156
157    assertTrue(tracker.isEmpty());
158  }

160  @Test
161  public void testBasicCRUD() {
162    ProcedureStoreTracker tracker = new ProcedureStoreTracker();
163    assertTrue(tracker.isEmpty());
164
165    Procedure[] procs = new TestProcedure[] {
166      new TestProcedure(1), new TestProcedure(2), new TestProcedure(3),
167      new TestProcedure(4), new TestProcedure(5), new TestProcedure(6),
168    };
169
170    tracker.insert(procs[0], null);
171    tracker.insert(procs[1], new Procedure[] { procs[2], procs[3], procs[4] });
172    assertFalse(tracker.isEmpty());
173    assertTrue(tracker.isUpdated());
174
175    tracker.resetUpdates();
176    assertFalse(tracker.isUpdated());
177
178    for (int i = 0; i < 4; ++i) {
179      tracker.update(procs[i]);
180      assertFalse(tracker.isEmpty());
181      assertFalse(tracker.isUpdated());
182    }
183
184    tracker.update(procs[4]);
185    assertFalse(tracker.isEmpty());
186    assertTrue(tracker.isUpdated());
187
188    tracker.update(procs[5]);
189    assertFalse(tracker.isEmpty());
190    assertTrue(tracker.isUpdated());
191
192    for (int i = 0; i < 5; ++i) {
193      tracker.delete(procs[i].getProcId());
194      assertFalse(tracker.isEmpty());
195      assertTrue(tracker.isUpdated());
196    }
197    tracker.delete(procs[5].getProcId());
198    assertTrue(tracker.isEmpty());
199  }

201  @Test
202  public void testRandLoad() {
203    final int NPROCEDURES = 2500;
204    final int NRUNS = 5000;
205
206    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
207
208    Random rand = new Random(1);
209    for (int i = 0; i < NRUNS; ++i) {
210      assertTrue(tracker.isEmpty());
211
212      int count = 0;
213      while (count < NPROCEDURES) {
214        long procId = rand.nextLong();
215        if (procId < 1) continue;
216
217        tracker.setDeleted(procId, i % 2 == 0);
218        count++;
219      }
220
221      tracker.clear();
222    }
223  }

225  @Test
226  public void testLoad() {
227    final int MAX_PROCS = 1000;
228    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
229    for (int numProcs = 1; numProcs < MAX_PROCS; ++numProcs) {
230      for (int start = 1; start <= numProcs; ++start) {
231        assertTrue(tracker.isEmpty());
232
233        LOG.debug("loading " + numProcs + " procs from start=" + start);
234        for (int i = start; i <= numProcs; ++i) {
235          tracker.setDeleted(i, false);
236        }
237        for (int i = 1; i < start; ++i) {
238          tracker.setDeleted(i, false);
239        }
240
241        tracker.clear();
242      }
243    }
244  }

246  @Test
247  public void testDelete() {
248    final ProcedureStoreTracker tracker = new ProcedureStoreTracker();
249
250    long[] procIds = new long[] { 65, 1, 193 };
251    for (int i = 0; i < procIds.length; ++i) {
252      tracker.insert(procIds[i]);
253      tracker.dump();
254    }
255
256    for (int i = 0; i < (64 * 4); ++i) {
257      boolean hasProc = false;
258      for (int j = 0; j < procIds.length; ++j) {
259        if (procIds[j] == i) {
260          hasProc = true;
261          break;
262        }
263      }
264      if (hasProc) {
265        assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(i));
266      } else {
267        assertEquals("procId=" + i, ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(i));
268      }
269    }
