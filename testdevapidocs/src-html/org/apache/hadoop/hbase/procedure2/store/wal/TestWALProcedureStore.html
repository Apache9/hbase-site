120  @Test
121  public void testLoad() throws Exception {
122    Set<Long> procIds = new HashSet<>();
123
124    // Insert something in the log
125    Procedure proc1 = new TestSequentialProcedure();
126    procIds.add(proc1.getProcId());
127    procStore.insert(proc1, null);
128
129    Procedure proc2 = new TestSequentialProcedure();
130    Procedure[] child2 = new Procedure[2];
131    child2[0] = new TestSequentialProcedure();
132    child2[1] = new TestSequentialProcedure();
133
134    procIds.add(proc2.getProcId());
135    procIds.add(child2[0].getProcId());
136    procIds.add(child2[1].getProcId());
137    procStore.insert(proc2, child2);
138
139    // Verify that everything is there
140    verifyProcIdsOnRestart(procIds);
141
142    // Update and delete something
143    procStore.update(proc1);
144    procStore.update(child2[1]);
145    procStore.delete(child2[1].getProcId());
146    procIds.remove(child2[1].getProcId());
147
148    // Verify that everything is there
149    verifyProcIdsOnRestart(procIds);
150
151    // Remove 4 byte from the trailers
152    procStore.stop(false);
153    FileStatus[] logs = fs.listStatus(logDir);
154    assertEquals(3, logs.length);
155    for (int i = 0; i < logs.length; ++i) {
156      corruptLog(logs[i], 4);
157    }
158    verifyProcIdsOnRestart(procIds);
159  }

206  @Test
207  public void testCorruptedProcedures() throws Exception {
208    // Insert root-procedures
209    TestProcedure[] rootProcs = new TestProcedure[10];
210    for (int i = 1; i <= rootProcs.length; i++) {
211      rootProcs[i-1] = new TestProcedure(i, 0);
212      procStore.insert(rootProcs[i-1], null);
213      rootProcs[i-1].addStackId(0);
214      procStore.update(rootProcs[i-1]);
215    }
216    // insert root-child txn
217    procStore.rollWriter();
218    for (int i = 1; i <= rootProcs.length; i++) {
219      TestProcedure b = new TestProcedure(rootProcs.length + i, i);
220      rootProcs[i-1].addStackId(1);
221      procStore.insert(rootProcs[i-1], new Procedure[] { b });
222    }
223    // insert child updates
224    procStore.rollWriter();
225    for (int i = 1; i <= rootProcs.length; i++) {
226      procStore.update(new TestProcedure(rootProcs.length + i, i));
227    }
228
229    // Stop the store
230    procStore.stop(false);
231
232    // Remove 4 byte from the trailer
233    FileStatus[] logs = fs.listStatus(logDir);
234    assertEquals(3, logs.length);
235    Arrays.sort(logs, new Comparator<FileStatus>() {
236      @Override
237      public int compare(FileStatus o1, FileStatus o2) {
238        return o1.getPath().getName().compareTo(o2.getPath().getName());
239      }
240    });
241
242    // Remove the first log, we have insert-txn and updates in the others so everything is fine.
243    fs.delete(logs[0].getPath(), false);
244    LoadCounter loader = new LoadCounter();
245    storeRestart(loader);
246    assertEquals(rootProcs.length * 2, loader.getLoadedCount());
247    assertEquals(0, loader.getCorruptedCount());
248
249    // Remove the second log, we have lost any root/parent references
250    fs.delete(logs[1].getPath(), false);
251    loader.reset();
252    storeRestart(loader);
253    assertEquals(0, loader.getLoadedCount());
254    assertEquals(rootProcs.length, loader.getCorruptedCount());
255    for (Procedure proc: loader.getCorrupted()) {
256      assertTrue(proc.toString(), proc.getParentProcId() <= rootProcs.length);
257      assertTrue(proc.toString(),
258                  proc.getProcId() > rootProcs.length &&
259                  proc.getProcId() <= (rootProcs.length * 2));
260    }
261  }

369  @Test
370  public void testInsertUpdateDelete() throws Exception {
371    final int NTHREAD = 2;
372
373    procStore.stop(false);
374    fs.delete(logDir, true);
375
376    org.apache.hadoop.conf.Configuration conf =
377      new org.apache.hadoop.conf.Configuration(htu.getConfiguration());
378    conf.setBoolean("hbase.procedure.store.wal.use.hsync", false);
379    conf.setInt("hbase.procedure.store.wal.periodic.roll.msec", 10000);
380    conf.setInt("hbase.procedure.store.wal.roll.threshold", 128 * 1024);
381
382    fs.mkdirs(logDir);
383    procStore = ProcedureTestingUtility.createWalStore(conf, fs, logDir);
384    procStore.start(NTHREAD);
385    procStore.recoverLease();
386
387    LoadCounter loader = new LoadCounter();
388    procStore.load(loader);
389    assertEquals(0, loader.getMaxProcId());
390    assertEquals(0, loader.getLoadedCount());
391    assertEquals(0, loader.getCorruptedCount());
392
393    final long LAST_PROC_ID = 9999;
394    final Thread[] thread = new Thread[NTHREAD];
395    final AtomicLong procCounter = new AtomicLong((long)Math.round(Math.random() * 100));
396    for (int i = 0; i < thread.length; ++i) {
397      thread[i] = new Thread() {
398        @Override
399        public void run() {
400          Random rand = new Random();
401          TestProcedure proc;
402          do {
403            proc = new TestProcedure(procCounter.addAndGet(1));
404            // Insert
405            procStore.insert(proc, null);
406            // Update
407            for (int i = 0, nupdates = rand.nextInt(10); i <= nupdates; ++i) {
408              try { Thread.sleep(0, rand.nextInt(15)); } catch (InterruptedException e) {}
409              procStore.update(proc);
410            }
411            // Delete
412            procStore.delete(proc.getProcId());
413          } while (proc.getProcId() < LAST_PROC_ID);
414        }
415      };
416      thread[i].start();
417    }
418
419    for (int i = 0; i < thread.length; ++i) {
420      thread[i].join();
421    }
422
423    procStore.getStoreTracker().dump();
424    assertTrue(procCounter.get() >= LAST_PROC_ID);
425    assertTrue(procStore.getStoreTracker().isEmpty());
426    assertEquals(1, procStore.getActiveLogs().size());
427  }

429  private void corruptLog(final FileStatus logFile, final long dropBytes)
430      throws IOException {
431    assertTrue(logFile.getLen() > dropBytes);
432    LOG.debug("corrupt log " + logFile.getPath() +
433              " size=" + logFile.getLen() + " drop=" + dropBytes);
434    Path tmpPath = new Path(testDir, "corrupted.log");
435    InputStream in = fs.open(logFile.getPath());
436    OutputStream out =  fs.create(tmpPath);
437    IOUtils.copyBytes(in, out, logFile.getLen() - dropBytes, true);
438    if (!fs.rename(tmpPath, logFile.getPath())) {
439      throw new IOException("Unable to rename");
440    }
441  }

462  public static class TestSequentialProcedure extends SequentialProcedure<Void> {
463    private static long seqid = 0;
464
465    public TestSequentialProcedure() {
466      setProcId(++seqid);
467    }
468
469    @Override
470    protected Procedure[] execute(Void env) { return null; }
471
472    @Override
473    protected void rollback(Void env) { }
474
475    @Override
476    protected boolean abort(Void env) { return false; }
477
478    @Override
479    protected void serializeStateData(final OutputStream stream) throws IOException {
480      long procId = getProcId();
481      if (procId % 2 == 0) {
482        stream.write(Bytes.toBytes(procId));
483      }
484    }
485
486    @Override
487    protected void deserializeStateData(InputStream stream) throws IOException {
488      long procId = getProcId();
489      if (procId % 2 == 0) {
490        byte[] bProcId = new byte[8];
491        assertEquals(8, stream.read(bProcId));
492        assertEquals(procId, Bytes.toLong(bProcId));
493      } else {
494        assertEquals(0, stream.available());
495      }
496    }
497  }

499  private class LoadCounter implements ProcedureStore.ProcedureLoader {
500    private final ArrayList<Procedure> corrupted = new ArrayList<Procedure>();
501    private final ArrayList<Procedure> loaded = new ArrayList<Procedure>();
502
503    private Set<Long> procIds;
504    private long maxProcId = 0;
505
506    public LoadCounter() {
507      this(null);
508    }
509
510    public LoadCounter(final Set<Long> procIds) {
511      this.procIds = procIds;
512    }
513
514    public void reset() {
515      reset(null);
516    }
517
518    public void reset(final Set<Long> procIds) {
519      corrupted.clear();
520      loaded.clear();
521      this.procIds = procIds;
522      this.maxProcId = 0;
523    }
524
525    public long getMaxProcId() {
526      return maxProcId;
527    }
528
529    public ArrayList<Procedure> getLoaded() {
530      return loaded;
531    }
532
533    public int getLoadedCount() {
534      return loaded.size();
535    }
536
537    public ArrayList<Procedure> getCorrupted() {
538      return corrupted;
539    }
540
541    public int getCorruptedCount() {
542      return corrupted.size();
543    }
544
545    @Override
546    public void setMaxProcId(long maxProcId) {
547      maxProcId = maxProcId;
548    }
549
550    @Override
551    public void load(ProcedureIterator procIter) throws IOException {
552      while (procIter.hasNext()) {
553        Procedure proc = procIter.next();
554        LOG.debug("loading procId=" + proc.getProcId() + ": " + proc);
555        if (procIds != null) {
556          assertTrue("procId=" + proc.getProcId() + " unexpected",
557                     procIds.contains(proc.getProcId()));
558        }
559        loaded.add(proc);
560      }
561    }
562
563    @Override
564    public void handleCorrupted(ProcedureIterator procIter) throws IOException {
565      while (procIter.hasNext()) {
566        Procedure proc = procIter.next();
567        LOG.debug("corrupted procId=" + proc.getProcId() + ": " + proc);
568        corrupted.add(proc);
569      }
570    }
571  }
