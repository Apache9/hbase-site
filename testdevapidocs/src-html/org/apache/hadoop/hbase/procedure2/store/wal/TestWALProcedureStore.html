117  @Test
118  public void testLoad() throws Exception {
119    Set<Long> procIds = new HashSet<>();
120
121    // Insert something in the log
122    Procedure proc1 = new TestSequentialProcedure();
123    procIds.add(proc1.getProcId());
124    procStore.insert(proc1, null);
125
126    Procedure proc2 = new TestSequentialProcedure();
127    Procedure[] child2 = new Procedure[2];
128    child2[0] = new TestSequentialProcedure();
129    child2[1] = new TestSequentialProcedure();
130
131    procIds.add(proc2.getProcId());
132    procIds.add(child2[0].getProcId());
133    procIds.add(child2[1].getProcId());
134    procStore.insert(proc2, child2);
135
136    // Verify that everything is there
137    verifyProcIdsOnRestart(procIds);
138
139    // Update and delete something
140    procStore.update(proc1);
141    procStore.update(child2[1]);
142    procStore.delete(child2[1].getProcId());
143    procIds.remove(child2[1].getProcId());
144
145    // Verify that everything is there
146    verifyProcIdsOnRestart(procIds);
147
148    // Remove 4 byte from the trailers
149    procStore.stop(false);
150    FileStatus[] logs = fs.listStatus(logDir);
151    assertEquals(3, logs.length);
152    for (int i = 0; i < logs.length; ++i) {
153      corruptLog(logs[i], 4);
154    }
155    verifyProcIdsOnRestart(procIds);
156  }

160    // log-0001: proc 0, 1 and 2 are inserted
161    Procedure proc0 = new TestSequentialProcedure();
162    procStore.insert(proc0, null);
163    Procedure proc1 = new TestSequentialProcedure();
164    procStore.insert(proc1, null);
165    Procedure proc2 = new TestSequentialProcedure();
166    procStore.insert(proc2, null);
167    procStore.rollWriterForTesting();
168
169    // log-0002: proc 1 deleted
170    procStore.delete(proc1.getProcId());
171    procStore.rollWriterForTesting();
172
173    // log-0003: proc 2 is update
174    procStore.update(proc2);
175    procStore.rollWriterForTesting();
176
177    // log-0004: proc 2 deleted
178    procStore.delete(proc2.getProcId());
179
180    // stop the store and remove the trailer
181    procStore.stop(false);
182    FileStatus[] logs = fs.listStatus(logDir);
183    assertEquals(4, logs.length);
184    for (int i = 0; i < logs.length; ++i) {
185      corruptLog(logs[i], 4);
186    }
187
188    // Test Load 1
189    LoadCounter loader = new LoadCounter();
190    storeRestart(loader);
191    assertEquals(1, loader.getLoadedCount());
192    assertEquals(0, loader.getCorruptedCount());
193
194    // Test Load 2
195    assertEquals(5, fs.listStatus(logDir).length);
196    loader = new LoadCounter();
197    storeRestart(loader);
198    assertEquals(1, loader.getLoadedCount());
199    assertEquals(0, loader.getCorruptedCount());
200
201    // remove proc-0
202    procStore.delete(proc0.getProcId());
203    procStore.periodicRollForTesting();
204    assertEquals(1, fs.listStatus(logDir).length);
205    storeRestart(loader);
206  }

279  @Test
280  public void testCorruptedProcedures() throws Exception {
281    // Insert root-procedures
282    TestProcedure[] rootProcs = new TestProcedure[10];
283    for (int i = 1; i <= rootProcs.length; i++) {
284      rootProcs[i-1] = new TestProcedure(i, 0);
285      procStore.insert(rootProcs[i-1], null);
286      rootProcs[i-1].addStackId(0);
287      procStore.update(rootProcs[i-1]);
288    }
289    // insert root-child txn
290    procStore.rollWriterForTesting();
291    for (int i = 1; i <= rootProcs.length; i++) {
292      TestProcedure b = new TestProcedure(rootProcs.length + i, i);
293      rootProcs[i-1].addStackId(1);
294      procStore.insert(rootProcs[i-1], new Procedure[] { b });
295    }
296    // insert child updates
297    procStore.rollWriterForTesting();
298    for (int i = 1; i <= rootProcs.length; i++) {
299      procStore.update(new TestProcedure(rootProcs.length + i, i));
300    }
301
302    // Stop the store
303    procStore.stop(false);
304
305    // the first log was removed,
306    // we have insert-txn and updates in the others so everything is fine
307    FileStatus[] logs = fs.listStatus(logDir);
308    assertEquals(Arrays.toString(logs), 2, logs.length);
309    Arrays.sort(logs, new Comparator<FileStatus>() {
310      @Override
311      public int compare(FileStatus o1, FileStatus o2) {
312        return o1.getPath().getName().compareTo(o2.getPath().getName());
313      }
314    });
315
316    LoadCounter loader = new LoadCounter();
317    storeRestart(loader);
318    assertEquals(rootProcs.length * 2, loader.getLoadedCount());
319    assertEquals(0, loader.getCorruptedCount());
320
321    // Remove the second log, we have lost all the root/parent references
322    fs.delete(logs[0].getPath(), false);
323    loader.reset();
324    storeRestart(loader);
325    assertEquals(0, loader.getLoadedCount());
326    assertEquals(rootProcs.length, loader.getCorruptedCount());
327    for (Procedure proc: loader.getCorrupted()) {
328      assertTrue(proc.toString(), proc.getParentProcId() <= rootProcs.length);
329      assertTrue(proc.toString(),
330                  proc.getProcId() > rootProcs.length &&
331                  proc.getProcId() <= (rootProcs.length * 2));
332    }
333  }

441  @Test
442  public void testRollAndRemove() throws IOException {
443    // Insert something in the log
444    Procedure proc1 = new TestSequentialProcedure();
445    procStore.insert(proc1, null);
446
447    Procedure proc2 = new TestSequentialProcedure();
448    procStore.insert(proc2, null);
449
450    // roll the log, now we have 2
451    procStore.rollWriterForTesting();
452    assertEquals(2, procStore.getActiveLogs().size());
453
454    // everything will be up to date in the second log
455    // so we can remove the first one
456    procStore.update(proc1);
457    procStore.update(proc2);
458    assertEquals(1, procStore.getActiveLogs().size());
459
460    // roll the log, now we have 2
461    procStore.rollWriterForTesting();
462    assertEquals(2, procStore.getActiveLogs().size());
463
464    // remove everything active
465    // so we can remove all the logs
466    procStore.delete(proc1.getProcId());
467    procStore.delete(proc2.getProcId());
468    assertEquals(1, procStore.getActiveLogs().size());
469  }

471  private void corruptLog(final FileStatus logFile, final long dropBytes)
472      throws IOException {
473    assertTrue(logFile.getLen() > dropBytes);
474    LOG.debug("corrupt log " + logFile.getPath() +
475              " size=" + logFile.getLen() + " drop=" + dropBytes);
476    Path tmpPath = new Path(testDir, "corrupted.log");
477    InputStream in = fs.open(logFile.getPath());
478    OutputStream out =  fs.create(tmpPath);
479    IOUtils.copyBytes(in, out, logFile.getLen() - dropBytes, true);
480    if (!fs.rename(tmpPath, logFile.getPath())) {
481      throw new IOException("Unable to rename");
482    }
483  }

504  public static class TestSequentialProcedure extends SequentialProcedure<Void> {
505    private static long seqid = 0;
506
507    public TestSequentialProcedure() {
508      setProcId(++seqid);
509    }
510
511    @Override
512    protected Procedure[] execute(Void env) { return null; }
513
514    @Override
515    protected void rollback(Void env) { }
516
517    @Override
518    protected boolean abort(Void env) { return false; }
519
520    @Override
521    protected void serializeStateData(final OutputStream stream) throws IOException {
522      long procId = getProcId();
523      if (procId % 2 == 0) {
524        stream.write(Bytes.toBytes(procId));
525      }
526    }
527
528    @Override
529    protected void deserializeStateData(InputStream stream) throws IOException {
530      long procId = getProcId();
531      if (procId % 2 == 0) {
532        byte[] bProcId = new byte[8];
533        assertEquals(8, stream.read(bProcId));
534        assertEquals(procId, Bytes.toLong(bProcId));
535      } else {
536        assertEquals(0, stream.available());
537      }
538    }
539  }
