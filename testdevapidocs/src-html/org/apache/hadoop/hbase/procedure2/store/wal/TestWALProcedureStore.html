122
123    // Insert something in the log
124    Procedure proc1 = new TestSequentialProcedure();
125    procIds.add(proc1.getProcId());
126    procStore.insert(proc1, null);

141    // Update and delete something

154    for (int i = 0; i < logs.length; ++i) {
155      corruptLog(logs[i], 4);
156    }
157    verifyProcIdsOnRestart(procIds);

255  @Test
256  public void testCorruptedProcedures() throws Exception {
257    // Insert root-procedures
258    TestProcedure[] rootProcs = new TestProcedure[10];
259    for (int i = 1; i <= rootProcs.length; i++) {
260      rootProcs[i-1] = new TestProcedure(i, 0);
261      procStore.insert(rootProcs[i-1], null);
262      rootProcs[i-1].addStackId(0);
263      procStore.update(rootProcs[i-1]);
264    }
265    // insert root-child txn
266    procStore.rollWriterForTesting();
267    for (int i = 1; i <= rootProcs.length; i++) {
268      TestProcedure b = new TestProcedure(rootProcs.length + i, i);
269      rootProcs[i-1].addStackId(1);
270      procStore.insert(rootProcs[i-1], new Procedure[] { b });
271    }
272    // insert child updates
273    procStore.rollWriterForTesting();
274    for (int i = 1; i <= rootProcs.length; i++) {
275      procStore.update(new TestProcedure(rootProcs.length + i, i));
276    }
277
278    // Stop the store
279    procStore.stop(false);
280
281    // the first log was removed,
282    // we have insert-txn and updates in the others so everything is fine
283    FileStatus[] logs = fs.listStatus(logDir);
284    assertEquals(Arrays.toString(logs), 2, logs.length);
285    Arrays.sort(logs, new Comparator<FileStatus>() {
286      @Override
287      public int compare(FileStatus o1, FileStatus o2) {
288        return o1.getPath().getName().compareTo(o2.getPath().getName());
289      }
290    });
291
292    LoadCounter loader = new LoadCounter();
293    storeRestart(loader);
294    assertEquals(rootProcs.length * 2, loader.getLoadedCount());
295    assertEquals(0, loader.getCorruptedCount());
296
297    // Remove the second log, we have lost all the root/parent references
298    fs.delete(logs[0].getPath(), false);
299    loader.reset();
300    storeRestart(loader);
301    assertEquals(0, loader.getLoadedCount());
302    assertEquals(rootProcs.length, loader.getCorruptedCount());
303    for (Procedure proc: loader.getCorrupted()) {
304      assertTrue(proc.toString(), proc.getParentProcId() <= rootProcs.length);
305      assertTrue(proc.toString(),
306                  proc.getProcId() > rootProcs.length &&
307                  proc.getProcId() <= (rootProcs.length * 2));
308    }
309  }

417  @Test
418  public void testInsertUpdateDelete() throws Exception {
419    final int NTHREAD = 2;
420
421    procStore.stop(false);
422    fs.delete(logDir, true);
423
424    org.apache.hadoop.conf.Configuration conf =
425      new org.apache.hadoop.conf.Configuration(htu.getConfiguration());
426    conf.setBoolean("hbase.procedure.store.wal.use.hsync", false);
427    conf.setInt("hbase.procedure.store.wal.periodic.roll.msec", 10000);
428    conf.setInt("hbase.procedure.store.wal.roll.threshold", 128 * 1024);
429
430    fs.mkdirs(logDir);
431    procStore = ProcedureTestingUtility.createWalStore(conf, fs, logDir);
432    procStore.start(NTHREAD);
433    procStore.recoverLease();
434
435    LoadCounter loader = new LoadCounter();
436    procStore.load(loader);
437    assertEquals(0, loader.getMaxProcId());
438    assertEquals(0, loader.getLoadedCount());
439    assertEquals(0, loader.getCorruptedCount());
440
441    final long LAST_PROC_ID = 9999;
442    final Thread[] thread = new Thread[NTHREAD];
443    final AtomicLong procCounter = new AtomicLong((long)Math.round(Math.random() * 100));
444    for (int i = 0; i < thread.length; ++i) {
445      thread[i] = new Thread() {
446        @Override
447        public void run() {
448          Random rand = new Random();
449          TestProcedure proc;
450          do {
451            proc = new TestProcedure(procCounter.addAndGet(1));
452            // Insert
453            procStore.insert(proc, null);
454            // Update
455            for (int i = 0, nupdates = rand.nextInt(10); i <= nupdates; ++i) {
456              try { Thread.sleep(0, rand.nextInt(15)); } catch (InterruptedException e) {}
457              procStore.update(proc);
458            }
459            // Delete
460            procStore.delete(proc.getProcId());
461          } while (proc.getProcId() < LAST_PROC_ID);
462        }
463      };
464      thread[i].start();
465    }
466
467    for (int i = 0; i < thread.length; ++i) {
468      thread[i].join();
469    }
470
471    procStore.getStoreTracker().dump();
472    assertTrue(procCounter.get() >= LAST_PROC_ID);
473    assertTrue(procStore.getStoreTracker().isEmpty());
474    assertEquals(1, procStore.getActiveLogs().size());
475  }

478  public void testRollAndRemove() throws IOException {
479    // Insert something in the log
480    Procedure proc1 = new TestSequentialProcedure();
481    procStore.insert(proc1, null);
482
483    Procedure proc2 = new TestSequentialProcedure();
484    procStore.insert(proc2, null);

488    assertEquals(2, procStore.getActiveLogs().size());
489

491    // so we can remove the first one
492    procStore.update(proc1);
493    procStore.update(proc2);
494    assertEquals(1, procStore.getActiveLogs().size());
495
496    // roll the log, now we have 2
497    procStore.rollWriterForTesting();
498    assertEquals(2, procStore.getActiveLogs().size());
499
500    // remove everything active
501    // so we can remove all the logs
502    procStore.delete(proc1.getProcId());
503    procStore.delete(proc2.getProcId());
504    assertEquals(1, procStore.getActiveLogs().size());
505  }

507  private void corruptLog(final FileStatus logFile, final long dropBytes)
508      throws IOException {
509    assertTrue(logFile.getLen() > dropBytes);
510    LOG.debug("corrupt log " + logFile.getPath() +
511              " size=" + logFile.getLen() + " drop=" + dropBytes);
512    Path tmpPath = new Path(testDir, "corrupted.log");
513    InputStream in = fs.open(logFile.getPath());
514    OutputStream out =  fs.create(tmpPath);
515    IOUtils.copyBytes(in, out, logFile.getLen() - dropBytes, true);
516    if (!fs.rename(tmpPath, logFile.getPath())) {
517      throw new IOException("Unable to rename");
518    }
519  }

540  public static class TestSequentialProcedure extends SequentialProcedure<Void> {
541    private static long seqid = 0;
542
543    public TestSequentialProcedure() {
544      setProcId(++seqid);
545    }
546
547    @Override
548    protected Procedure[] execute(Void env) { return null; }
549
550    @Override
551    protected void rollback(Void env) { }
552
553    @Override
554    protected boolean abort(Void env) { return false; }
555
556    @Override
557    protected void serializeStateData(final OutputStream stream) throws IOException {
558      long procId = getProcId();
559      if (procId % 2 == 0) {
560        stream.write(Bytes.toBytes(procId));
561      }
562    }
563
564    @Override
565    protected void deserializeStateData(InputStream stream) throws IOException {
566      long procId = getProcId();
567      if (procId % 2 == 0) {
568        byte[] bProcId = new byte[8];
569        assertEquals(8, stream.read(bProcId));
570        assertEquals(procId, Bytes.toLong(bProcId));
571      } else {
572        assertEquals(0, stream.available());
573      }
574    }
575  }

577  private class LoadCounter implements ProcedureStore.ProcedureLoader {
578    private final ArrayList<Procedure> corrupted = new ArrayList<Procedure>();
579    private final ArrayList<Procedure> loaded = new ArrayList<Procedure>();
580
581    private Set<Long> procIds;
582    private long maxProcId = 0;
583
584    public LoadCounter() {
585      this(null);
586    }
587
588    public LoadCounter(final Set<Long> procIds) {
589      this.procIds = procIds;
590    }
591
592    public void reset() {
593      reset(null);
594    }
595
596    public void reset(final Set<Long> procIds) {
597      corrupted.clear();
598      loaded.clear();
599      this.procIds = procIds;
600      this.maxProcId = 0;
601    }
602
603    public long getMaxProcId() {
604      return maxProcId;
605    }
606
607    public ArrayList<Procedure> getLoaded() {
608      return loaded;
609    }
610
611    public int getLoadedCount() {
612      return loaded.size();
613    }
614
615    public ArrayList<Procedure> getCorrupted() {
616      return corrupted;
617    }
618
619    public int getCorruptedCount() {
620      return corrupted.size();
621    }
622
623    @Override
624    public void setMaxProcId(long maxProcId) {
625      maxProcId = maxProcId;
626    }
627
628    @Override
629    public void load(ProcedureIterator procIter) throws IOException {
630      while (procIter.hasNext()) {
631        Procedure proc = procIter.nextAsProcedure();
632        LOG.debug("loading procId=" + proc.getProcId() + ": " + proc);
633        if (procIds != null) {
634          assertTrue("procId=" + proc.getProcId() + " unexpected",
635                     procIds.contains(proc.getProcId()));
636        }
637        loaded.add(proc);
638      }
639    }
640
641    @Override
642    public void handleCorrupted(ProcedureIterator procIter) throws IOException {
643      while (procIter.hasNext()) {
644        Procedure proc = procIter.nextAsProcedure();
645        LOG.debug("corrupted procId=" + proc.getProcId() + ": " + proc);
646        corrupted.add(proc);
647      }
648    }
649  }
