064  private void createProcExecutor(String dir) throws Exception {
065    logDir = new Path(testDir, dir);
066    procStore = ProcedureTestingUtility.createWalStore(htu.getConfiguration(), logDir);
067    procExecutor = new ProcedureExecutor<>(htu.getConfiguration(), procEnv,
068        procStore);
069    procStore.start(PROCEDURE_EXECUTOR_SLOTS);
070    ProcedureTestingUtility
071        .initAndStartWorkers(procExecutor, PROCEDURE_EXECUTOR_SLOTS, true, true);
072  }

087  @Test
088  public void testProcedureShouldNotCleanOnLoad() throws Exception {
089    createProcExecutor("testProcedureShouldNotCleanOnLoad");
090    final RootProcedure proc = new RootProcedure();
091    long rootProc = procExecutor.submitProcedure(proc);
092    LOG.info("Begin to execute " + rootProc);
093    // wait until the child procedure arrival
094    while(procExecutor.getProcedures().size() < 2) {
095      Thread.sleep(100);
096    }
097    SuspendProcedure suspendProcedure = (SuspendProcedure) procExecutor
098        .getProcedures().get(1);
099    // wait until the suspendProcedure executed
100    suspendProcedure.latch.countDown();
101    Thread.sleep(100);
102    // roll the procedure log
103    LOG.info("Begin to roll log ");
104    procStore.rollWriterForTesting();
105    LOG.info("finish to roll log ");
106    Thread.sleep(500);
107    LOG.info("begin to restart1 ");
108    ProcedureTestingUtility.restart(procExecutor, true);
109    LOG.info("finish to restart1 ");
110    Assert.assertTrue(procExecutor.getProcedure(rootProc) != null);
111    Thread.sleep(500);
112    LOG.info("begin to restart2 ");
113    ProcedureTestingUtility.restart(procExecutor, true);
114    LOG.info("finish to restart2 ");
115    Assert.assertTrue(procExecutor.getProcedure(rootProc) != null);
116  }

118  @Test
119  public void testProcedureUpdatedShouldClean() throws Exception {
120    createProcExecutor("testProcedureUpdatedShouldClean");
121    SuspendProcedure suspendProcedure = new SuspendProcedure();
122    long suspendProc = procExecutor.submitProcedure(suspendProcedure);
123    LOG.info("Begin to execute " + suspendProc);
124    suspendProcedure.latch.countDown();
125    Thread.sleep(500);
126    LOG.info("begin to restart1 ");
127    ProcedureTestingUtility.restart(procExecutor, true);
128    LOG.info("finish to restart1 ");
129    while(procExecutor.getProcedure(suspendProc) == null) {
130      Thread.sleep(100);
131    }
132    // Wait until the suspendProc executed after restart
133    suspendProcedure = (SuspendProcedure) procExecutor.getProcedure(suspendProc);
134    suspendProcedure.latch.countDown();
135    Thread.sleep(500);
136    // Should be 1 log since the suspendProcedure is updated in the new log
137    Assert.assertTrue(procStore.getActiveLogs().size() == 1);
138    // restart procExecutor
139    LOG.info("begin to restart2");
140    // Restart the executor but do not start the workers.
141    // Otherwise, the suspendProcedure will soon be executed and the oldest log
142    // will be cleaned, leaving only the newest log.
143    ProcedureTestingUtility.restart(procExecutor, true, false);
144    LOG.info("finish to restart2");
145    // There should be two active logs
146    Assert.assertTrue(procStore.getActiveLogs().size() == 2);
147    procExecutor.startWorkers();
148
149  }

151  @Test
152  public void testProcedureDeletedShouldClean() throws Exception {
153    createProcExecutor("testProcedureDeletedShouldClean");
154    WaitProcedure waitProcedure = new WaitProcedure();
155    long waitProce = procExecutor.submitProcedure(waitProcedure);
156    LOG.info("Begin to execute " + waitProce);
157    Thread.sleep(500);
158    LOG.info("begin to restart1 ");
159    ProcedureTestingUtility.restart(procExecutor, true);
160    LOG.info("finish to restart1 ");
161    while(procExecutor.getProcedure(waitProce) == null) {
162      Thread.sleep(100);
163    }
164    // Wait until the suspendProc executed after restart
165    waitProcedure = (WaitProcedure) procExecutor.getProcedure(waitProce);
166    waitProcedure.latch.countDown();
167    Thread.sleep(500);
168    // Should be 1 log since the suspendProcedure is updated in the new log
169    Assert.assertTrue(procStore.getActiveLogs().size() == 1);
170    // restart procExecutor
171    LOG.info("begin to restart2");
172    // Restart the executor but do not start the workers.
173    // Otherwise, the suspendProcedure will soon be executed and the oldest log
174    // will be cleaned, leaving only the newest log.
175    ProcedureTestingUtility.restart(procExecutor, true, false);
176    LOG.info("finish to restart2");
177    // There should be two active logs
178    Assert.assertTrue(procStore.getActiveLogs().size() == 2);
179    procExecutor.startWorkers();
180  }

182  public static class WaitProcedure
183      extends ProcedureTestingUtility.NoopProcedure<TestProcEnv> {
184    public WaitProcedure() {
185      super();
186    }
187
188    private CountDownLatch latch = new CountDownLatch(1);
189
190    @Override
191    protected Procedure[] execute(final TestProcEnv env)
192        throws ProcedureSuspendedException {
193      // Always wait here
194      LOG.info("wait here");
195      try {
196        latch.await();
197      } catch (Throwable t) {
198
199      }
200      LOG.info("finished");
201      return null;
202    }
203  }
204
205  public static class SuspendProcedure extends ProcedureTestingUtility.NoopProcedure<TestProcEnv> {
206    public SuspendProcedure() {
207      super();
208    }
209
210    private CountDownLatch latch = new CountDownLatch(1);
211
212    @Override
213    protected Procedure[] execute(final TestProcEnv env)
214        throws ProcedureSuspendedException {
215      // Always suspend the procedure
216      LOG.info("suspend here");
217      latch.countDown();
218      throw new ProcedureSuspendedException();
219    }
220  }
221
222  public static class RootProcedure extends ProcedureTestingUtility.NoopProcedure<TestProcEnv> {
223    private boolean childSpwaned = false;
224
225    public RootProcedure() {
226      super();
227    }
228
229    @Override
230    protected Procedure[] execute(final TestProcEnv env)
231        throws ProcedureSuspendedException {
232      if (!childSpwaned) {
233        childSpwaned = true;
234        return new Procedure[] {new SuspendProcedure()};
235      } else {
236        return null;
237      }
238    }
239  }
