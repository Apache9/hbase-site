068  public static <TEnv> void restart(ProcedureExecutor<TEnv> procExecutor)
069      throws Exception {
070    restart(procExecutor, null, true);
071  }

073  public static <TEnv> void restart(ProcedureExecutor<TEnv> procExecutor,
074      Runnable beforeStartAction, boolean failOnCorrupted) throws Exception {
075    ProcedureStore procStore = procExecutor.getStore();
076    int storeThreads = procExecutor.getNumThreads();
077    int execThreads = procExecutor.getNumThreads();
078    // stop
079    procExecutor.stop();
080    procExecutor.join();
081    procStore.stop(false);
082    // nothing running...
083    if (beforeStartAction != null) {
084      beforeStartAction.run();
085    }
086    // re-start
087    procStore.start(storeThreads);
088    procExecutor.start(execThreads, failOnCorrupted);
089  }

108  public static <TEnv> void toggleKillBeforeStoreUpdate(ProcedureExecutor<TEnv> procExecutor) {
109    if (procExecutor.testing == null) {

122  public static <TEnv> long submitAndWait(Configuration conf, TEnv env, Procedure<TEnv> proc)
123      throws IOException {
124    NoopProcedureStore procStore = new NoopProcedureStore();
125    ProcedureExecutor<TEnv> procExecutor = new ProcedureExecutor<TEnv>(conf, env, procStore);
126    procStore.start(1);
127    procExecutor.start(1, false);
128    try {
129      return submitAndWait(procExecutor, proc, HConstants.NO_NONCE, HConstants.NO_NONCE);
130    } finally {
131      procStore.stop(false);
132      procExecutor.stop();
133    }
134  }
135
136  public static <TEnv> long submitAndWait(ProcedureExecutor<TEnv> procExecutor, Procedure proc) {
137    return submitAndWait(procExecutor, proc, HConstants.NO_NONCE, HConstants.NO_NONCE);
138  }

140  public static <TEnv> long submitAndWait(ProcedureExecutor<TEnv> procExecutor, Procedure proc,
141      final long nonceGroup,
142      final long nonce) {
143    long procId = procExecutor.submitProcedure(proc, nonceGroup, nonce);
144    waitProcedure(procExecutor, procId);
145    return procId;
146  }
147
148  public static <TEnv> void waitProcedure(ProcedureExecutor<TEnv> procExecutor, long procId) {
149    while (!procExecutor.isFinished(procId) && procExecutor.isRunning()) {
150      Threads.sleepWithoutInterrupt(250);
151    }
152  }
153
154  public static <TEnv> void waitNoProcedureRunning(ProcedureExecutor<TEnv> procExecutor) {
155    int stableRuns = 0;
156    while (stableRuns < 10) {
157      if (procExecutor.getActiveExecutorCount() > 0 || procExecutor.getRunnableSet().size() > 0) {
158        stableRuns = 0;
159        Threads.sleepWithoutInterrupt(100);
160      } else {
161        stableRuns++;
162        Threads.sleepWithoutInterrupt(25);
163      }
164    }
165  }

167  public static <TEnv> void assertProcNotYetCompleted(ProcedureExecutor<TEnv> procExecutor,
168      long procId) {
169    assertFalse("expected a running proc", procExecutor.isFinished(procId));
170    assertEquals(null, procExecutor.getResult(procId));
171  }

173  public static <TEnv> void assertProcNotFailed(ProcedureExecutor<TEnv> procExecutor,
174      long procId) {
175    ProcedureInfo result = procExecutor.getResult(procId);
176    assertTrue("expected procedure result", result != null);
177    assertProcNotFailed(result);
178  }
179
180  public static void assertProcNotFailed(final ProcedureInfo result) {
181    if (result.getForeignExceptionMessage() != null) {
182      ForeignExceptionMessage exception =
183          result.getForeignExceptionMessage().getForeignExchangeMessage();
184      String msg = exception != null ? result.getExceptionFullMessage() : "no exception found";
185      assertFalse(msg, result.isFailed());
186    }
187  }
188
189  public static void assertIsAbortException(final ProcedureInfo result) {
190    assertEquals(true, result.isFailed());
191    LOG.info(result.getExceptionFullMessage());
192    Throwable cause = getExceptionCause(result);
193    assertTrue("expected abort exception, got "+ cause, cause instanceof ProcedureAbortedException);
194  }
195
196  public static void assertIsTimeoutException(final ProcedureInfo result) {
197    assertEquals(true, result.isFailed());
198    LOG.info(result.getExceptionFullMessage());
199    Throwable cause = getExceptionCause(result);
200    assertTrue("expected TimeoutIOException, got " + cause, cause instanceof TimeoutIOException);
201  }
202
203  public static void assertIsIllegalArgumentException(final ProcedureInfo result) {
204    assertEquals(true, result.isFailed());
205    LOG.info(result.getExceptionFullMessage());
206    Throwable cause = ProcedureTestingUtility.getExceptionCause(result);
207    assertTrue("expected IllegalArgumentIOException, got " + cause,
208      cause instanceof IllegalArgumentIOException);
209  }
210
211  public static Throwable getExceptionCause(final ProcedureInfo procInfo) {
212    assert procInfo.getForeignExceptionMessage() != null;
213    return RemoteProcedureException
214        .fromProto(procInfo.getForeignExceptionMessage().getForeignExchangeMessage()).getCause();
215  }

251  public static class LoadCounter implements ProcedureStore.ProcedureLoader {
252    private final ArrayList<Procedure> corrupted = new ArrayList<Procedure>();
253    private final ArrayList<Procedure> loaded = new ArrayList<Procedure>();
254
255    private Set<Long> procIds;
256    private long maxProcId = 0;
257
258    public LoadCounter() {
259      this(null);
260    }
261
262    public LoadCounter(final Set<Long> procIds) {
263      this.procIds = procIds;
264    }
265
266    public void reset() {
267      reset(null);
268    }
269
270    public void reset(final Set<Long> procIds) {
271      corrupted.clear();
272      loaded.clear();
273      this.procIds = procIds;
274      this.maxProcId = 0;
275    }
276
277    public long getMaxProcId() {
278      return maxProcId;
279    }
280
281    public ArrayList<Procedure> getLoaded() {
282      return loaded;
283    }
284
285    public int getLoadedCount() {
286      return loaded.size();
287    }
288
289    public ArrayList<Procedure> getCorrupted() {
290      return corrupted;
291    }
292
293    public int getCorruptedCount() {
294      return corrupted.size();
295    }
296
297    @Override
298    public void setMaxProcId(long maxProcId) {
299      maxProcId = maxProcId;
300    }
301
302    @Override
303    public void load(ProcedureIterator procIter) throws IOException {
304      while (procIter.hasNext()) {
305        Procedure proc = procIter.nextAsProcedure();
306        LOG.debug("loading procId=" + proc.getProcId() + ": " + proc);
307        if (procIds != null) {
308          assertTrue("procId=" + proc.getProcId() + " unexpected",
309                     procIds.contains(proc.getProcId()));
310        }
311        loaded.add(proc);
312      }
313    }
314
315    @Override
316    public void handleCorrupted(ProcedureIterator procIter) throws IOException {
317      while (procIter.hasNext()) {
318        Procedure proc = procIter.nextAsProcedure();
319        LOG.debug("corrupted procId=" + proc.getProcId() + ": " + proc);
320        corrupted.add(proc);
321      }
322    }
323  }
