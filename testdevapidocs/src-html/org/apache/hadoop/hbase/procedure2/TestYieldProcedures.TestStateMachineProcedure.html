082  @Test
083  public void testYieldEachExecutionStep() throws Exception {
084    final int NUM_STATES = 3;
085
086    TestStateMachineProcedure[] procs = new TestStateMachineProcedure[3];
087    for (int i = 0; i < procs.length; ++i) {
088      procs[i] = new TestStateMachineProcedure(true, false);
089      procExecutor.submitProcedure(procs[i]);
090    }
091    ProcedureTestingUtility.waitNoProcedureRunning(procExecutor);
092
093    // verify yield during execute()
094    long prevTimestamp = 0;
095    for (int execStep = 0; execStep < NUM_STATES; ++execStep) {
096      for (int i = 0; i < procs.length; ++i) {
097        assertEquals(NUM_STATES * 2, procs[i].getExecutionInfo().size());
098        TestStateMachineProcedure.ExecutionInfo info = procs[i].getExecutionInfo().get(execStep);
099        LOG.info("i=" + i + " execStep=" + execStep + " timestamp=" + info.getTimestamp());
100        assertEquals(false, info.isRollback());
101        assertEquals(execStep, info.getStep().ordinal());
102        assertEquals(prevTimestamp + 1, info.getTimestamp());
103        prevTimestamp++;
104      }
105    }
106
107    // verify yield during rollback()
108    int count = NUM_STATES;
109    for (int execStep = NUM_STATES - 1; execStep >= 0; --execStep) {
110      for (int i = 0; i < procs.length; ++i) {
111        assertEquals(NUM_STATES * 2, procs[i].getExecutionInfo().size());
112        TestStateMachineProcedure.ExecutionInfo info = procs[i].getExecutionInfo().get(count);
113        LOG.info("i=" + i + " execStep=" + execStep + " timestamp=" + info.getTimestamp());
114        assertEquals(true, info.isRollback());
115        assertEquals(execStep, info.getStep().ordinal());
116        assertEquals(prevTimestamp + 1, info.getTimestamp());
117        prevTimestamp++;
118      }
119      count++;
120    }
121  }

123  @Test
124  public void testYieldOnInterrupt() throws Exception {
125    final int NUM_STATES = 3;
126    int count = 0;
127
128    TestStateMachineProcedure proc = new TestStateMachineProcedure(true, true);
129    ProcedureTestingUtility.submitAndWait(procExecutor, proc);
130
131    // test execute (we execute steps twice, one has the IE the other completes)
132    assertEquals(NUM_STATES * 4, proc.getExecutionInfo().size());
133    for (int i = 0; i < NUM_STATES; ++i) {
134      TestStateMachineProcedure.ExecutionInfo info = proc.getExecutionInfo().get(count++);
135      assertEquals(false, info.isRollback());
136      assertEquals(i, info.getStep().ordinal());
137
138      info = proc.getExecutionInfo().get(count++);
139      assertEquals(false, info.isRollback());
140      assertEquals(i, info.getStep().ordinal());
141    }
142
143    // test rollback (we execute steps twice, one has the IE the other completes)
144    for (int i = NUM_STATES - 1; i >= 0; --i) {
145      TestStateMachineProcedure.ExecutionInfo info = proc.getExecutionInfo().get(count++);
146      assertEquals(true, info.isRollback());
147      assertEquals(i, info.getStep().ordinal());
148
149      info = proc.getExecutionInfo().get(count++);
150      assertEquals(true, info.isRollback());
151      assertEquals(i, info.getStep().ordinal());
152    }
153  }

155  private static class TestProcEnv {
156    public final AtomicLong timestamp = new AtomicLong(0);
157
158    public long nextTimestamp() {
159      return timestamp.incrementAndGet();
160    }
161  }
162
163  public static class TestStateMachineProcedure
164      extends StateMachineProcedure<TestProcEnv, TestStateMachineProcedure.State> {
165    enum State { STATE_1, STATE_2, STATE_3 }
166
167    public class ExecutionInfo {
168      private final boolean rollback;
169      private final long timestamp;
170      private final State step;
171
172      public ExecutionInfo(long timestamp, State step, boolean isRollback) {
173        this.timestamp = timestamp;
174        this.step = step;
175        this.rollback = isRollback;
176      }
177
178      public State getStep() { return step; }
179      public long getTimestamp() { return timestamp; }
180      public boolean isRollback() { return rollback; }
181    }
182
183    private final ArrayList<ExecutionInfo> executionInfo = new ArrayList<ExecutionInfo>();
184    private final AtomicBoolean aborted = new AtomicBoolean(false);
185    private final boolean throwInterruptOnceOnEachStep;
186    private final boolean abortOnFinalStep;
187
188    public TestStateMachineProcedure() {
189      this(false, false);
190    }
191
192    public TestStateMachineProcedure(boolean abortOnFinalStep,
193        boolean throwInterruptOnceOnEachStep) {
194      this.abortOnFinalStep = abortOnFinalStep;
195      this.throwInterruptOnceOnEachStep = throwInterruptOnceOnEachStep;
196    }
197
198    public ArrayList<ExecutionInfo> getExecutionInfo() {
199      return executionInfo;
200    }
201
202    @Override
203    protected StateMachineProcedure.Flow executeFromState(TestProcEnv env, State state)
204        throws InterruptedException {
205      LOG.info("execute step " + state);
206      executionInfo.add(new ExecutionInfo(env.nextTimestamp(), state, false));
207      Thread.sleep(150);
208
209      if (throwInterruptOnceOnEachStep && ((executionInfo.size() - 1) % 2) == 0) {
210        LOG.debug("THROW INTERRUPT");
211        throw new InterruptedException("test interrupt");
212      }
213
214      switch (state) {
215        case STATE_1:
216          setNextState(State.STATE_2);
217          break;
218        case STATE_2:
219          setNextState(State.STATE_3);
220          break;
221        case STATE_3:
222          if (abortOnFinalStep) {
223            setFailure("test", new IOException("Requested abort on final step"));
224          }
225          return Flow.NO_MORE_STATE;
226        default:
227          throw new UnsupportedOperationException();
228      }
229      return Flow.HAS_MORE_STATE;
230    }
231
232    @Override
233    protected void rollbackState(TestProcEnv env, final State state)
234        throws InterruptedException {
235      LOG.debug("rollback state " + state);
236      executionInfo.add(new ExecutionInfo(env.nextTimestamp(), state, true));
237      Thread.sleep(150);
238
239      if (throwInterruptOnceOnEachStep && ((executionInfo.size() - 1) % 2) == 0) {
240        LOG.debug("THROW INTERRUPT");
241        throw new InterruptedException("test interrupt");
242      }
243
244      switch (state) {
245        case STATE_1:
246          break;
247        case STATE_2:
248          break;
249        case STATE_3:
250          break;
251        default:
252          throw new UnsupportedOperationException();
253      }
254    }
255
256    @Override
257    protected State getState(final int stateId) {
258      return State.values()[stateId];
259    }
260
261    @Override
262    protected int getStateId(final State state) {
263      return state.ordinal();
264    }
265
266    @Override
267    protected State getInitialState() {
268      return State.STATE_1;
269    }
270
271    @Override
272    protected boolean isYieldBeforeExecuteFromState(TestProcEnv env, State state) {
273      return true;
274    }
275
276    @Override
277    protected boolean abort(TestProcEnv env) {
278      aborted.set(true);
279      return true;
280    }
281  }
