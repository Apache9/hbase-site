043  @Test(timeout=60000)
044  public void testQuotaStateBypass() {
045    QuotaState quotaInfo = new QuotaState();
046    assertTrue(quotaInfo.isBypass());
047    assertNoopLimiter(quotaInfo.getGlobalLimiter());
048
049    UserQuotaState userQuotaState = new UserQuotaState();
050    assertTrue(userQuotaState.isBypass());
051    assertNoopLimiter(userQuotaState.getTableLimiter(UNKNOWN_TABLE_NAME));
052  }

054  @Test(timeout=60000)
055  public void testSimpleQuotaStateOperation() {
056    final TableName table = TableName.valueOf("testSimpleQuotaStateOperationTable");
057    final int NUM_GLOBAL_THROTTLE = 3;
058    final int NUM_TABLE_THROTTLE = 2;
059
060    UserQuotaState quotaInfo = new UserQuotaState();
061    assertTrue(quotaInfo.isBypass());
062
063    // Set global quota
064    quotaInfo.setQuotas(buildReqNumThrottle(NUM_GLOBAL_THROTTLE));
065    assertFalse(quotaInfo.isBypass());
066
067    // Set table quota
068    quotaInfo.setQuotas(table, buildReqNumThrottle(NUM_TABLE_THROTTLE));
069    assertFalse(quotaInfo.isBypass());
070    assertTrue(quotaInfo.getGlobalLimiter() == quotaInfo.getTableLimiter(UNKNOWN_TABLE_NAME));
071    assertThrottleException(quotaInfo.getTableLimiter(UNKNOWN_TABLE_NAME), NUM_GLOBAL_THROTTLE);
072    assertThrottleException(quotaInfo.getTableLimiter(table), NUM_TABLE_THROTTLE);
073  }

075  @Test(timeout=60000)
076  public void testQuotaStateUpdateBypassThrottle() {
077    final long LAST_UPDATE = 10;
078
079    UserQuotaState quotaInfo = new UserQuotaState();
080    assertEquals(0, quotaInfo.getLastUpdate());
081    assertTrue(quotaInfo.isBypass());
082
083    UserQuotaState otherQuotaState = new UserQuotaState(LAST_UPDATE);

091    assertNoopLimiter(quotaInfo.getTableLimiter(UNKNOWN_TABLE_NAME));
092  }

094  @Test(timeout=60000)
095  public void testQuotaStateUpdateGlobalThrottle() {
096    final int NUM_GLOBAL_THROTTLE_1 = 3;
097    final int NUM_GLOBAL_THROTTLE_2 = 11;
098    final long LAST_UPDATE_1 = 10;
099    final long LAST_UPDATE_2 = 20;
100    final long LAST_UPDATE_3 = 30;
101
102    QuotaState quotaInfo = new QuotaState();
103    assertEquals(0, quotaInfo.getLastUpdate());
104    assertTrue(quotaInfo.isBypass());
105
106    // Add global throttle
107    QuotaState otherQuotaState = new QuotaState(LAST_UPDATE_1);
108    otherQuotaState.setQuotas(buildReqNumThrottle(NUM_GLOBAL_THROTTLE_1));
109    assertEquals(LAST_UPDATE_1, otherQuotaState.getLastUpdate());
110    assertFalse(otherQuotaState.isBypass());
111
112    quotaInfo.update(otherQuotaState);
113    assertEquals(LAST_UPDATE_1, quotaInfo.getLastUpdate());
114    assertFalse(quotaInfo.isBypass());
115    assertThrottleException(quotaInfo.getGlobalLimiter(), NUM_GLOBAL_THROTTLE_1);
116
117    // Update global Throttle
118    otherQuotaState = new QuotaState(LAST_UPDATE_2);
119    otherQuotaState.setQuotas(buildReqNumThrottle(NUM_GLOBAL_THROTTLE_2));
120    assertEquals(LAST_UPDATE_2, otherQuotaState.getLastUpdate());
121    assertFalse(otherQuotaState.isBypass());
122
123    quotaInfo.update(otherQuotaState);
124    assertEquals(LAST_UPDATE_2, quotaInfo.getLastUpdate());
125    assertFalse(quotaInfo.isBypass());
126    assertThrottleException(quotaInfo.getGlobalLimiter(),
127        NUM_GLOBAL_THROTTLE_2 - NUM_GLOBAL_THROTTLE_1);
128
129    // Remove global throttle
130    otherQuotaState = new QuotaState(LAST_UPDATE_3);
131    assertEquals(LAST_UPDATE_3, otherQuotaState.getLastUpdate());
132    assertTrue(otherQuotaState.isBypass());
133
134    quotaInfo.update(otherQuotaState);
135    assertEquals(LAST_UPDATE_3, quotaInfo.getLastUpdate());
136    assertTrue(quotaInfo.isBypass());
137    assertNoopLimiter(quotaInfo.getGlobalLimiter());
138  }

140  @Test(timeout=60000)
141  public void testQuotaStateUpdateTableThrottle() {
142    final TableName TABLE_A = TableName.valueOf("TableA");
143    final TableName TABLE_B = TableName.valueOf("TableB");
144    final TableName TABLE_C = TableName.valueOf("TableC");
145    final int TABLE_A_THROTTLE_1 = 3;
146    final int TABLE_A_THROTTLE_2 = 11;
147    final int TABLE_B_THROTTLE = 4;
148    final int TABLE_C_THROTTLE = 5;
149    final long LAST_UPDATE_1 = 10;
150    final long LAST_UPDATE_2 = 20;
151    final long LAST_UPDATE_3 = 30;
152
153    UserQuotaState quotaInfo = new UserQuotaState();
154    assertEquals(0, quotaInfo.getLastUpdate());
155    assertTrue(quotaInfo.isBypass());
156
157    // Add A B table limiters
158    UserQuotaState otherQuotaState = new UserQuotaState(LAST_UPDATE_1);
159    otherQuotaState.setQuotas(TABLE_A, buildReqNumThrottle(TABLE_A_THROTTLE_1));
160    otherQuotaState.setQuotas(TABLE_B, buildReqNumThrottle(TABLE_B_THROTTLE));
161    assertEquals(LAST_UPDATE_1, otherQuotaState.getLastUpdate());
162    assertFalse(otherQuotaState.isBypass());
163
164    quotaInfo.update(otherQuotaState);
165    assertEquals(LAST_UPDATE_1, quotaInfo.getLastUpdate());
166    assertFalse(quotaInfo.isBypass());
167    assertThrottleException(quotaInfo.getTableLimiter(TABLE_A), TABLE_A_THROTTLE_1);
168    assertThrottleException(quotaInfo.getTableLimiter(TABLE_B), TABLE_B_THROTTLE);
169    assertNoopLimiter(quotaInfo.getTableLimiter(TABLE_C));
170
171    // Add C, Remove B, Update A table limiters
172    otherQuotaState = new UserQuotaState(LAST_UPDATE_2);
173    otherQuotaState.setQuotas(TABLE_A, buildReqNumThrottle(TABLE_A_THROTTLE_2));
174    otherQuotaState.setQuotas(TABLE_C, buildReqNumThrottle(TABLE_C_THROTTLE));
175    assertEquals(LAST_UPDATE_2, otherQuotaState.getLastUpdate());
176    assertFalse(otherQuotaState.isBypass());
177
178    quotaInfo.update(otherQuotaState);
179    assertEquals(LAST_UPDATE_2, quotaInfo.getLastUpdate());
180    assertFalse(quotaInfo.isBypass());
181    assertThrottleException(quotaInfo.getTableLimiter(TABLE_A),
182        TABLE_A_THROTTLE_2 - TABLE_A_THROTTLE_1);
183    assertThrottleException(quotaInfo.getTableLimiter(TABLE_C), TABLE_C_THROTTLE);
184    assertNoopLimiter(quotaInfo.getTableLimiter(TABLE_B));
185
186    // Remove table limiters
187    otherQuotaState = new UserQuotaState(LAST_UPDATE_3);
188    assertEquals(LAST_UPDATE_3, otherQuotaState.getLastUpdate());
189    assertTrue(otherQuotaState.isBypass());
190
191    quotaInfo.update(otherQuotaState);
192    assertEquals(LAST_UPDATE_3, quotaInfo.getLastUpdate());
193    assertTrue(quotaInfo.isBypass());
194    assertNoopLimiter(quotaInfo.getTableLimiter(UNKNOWN_TABLE_NAME));
195  }

197  private Quotas buildReqNumThrottle(final long limit) {
198    return Quotas.newBuilder()
199            .setThrottle(Throttle.newBuilder()
200              .setReqNum(ProtobufUtil.toTimedQuota(limit, TimeUnit.MINUTES, QuotaScope.MACHINE))
201              .build())
202            .build();
203  }
204
205  private void assertThrottleException(final QuotaLimiter limiter, final int availReqs) {
206    assertNoThrottleException(limiter, availReqs);
207    try {
208      limiter.checkQuota(1, 1);
209      fail("Should have thrown ThrottlingException");
210    } catch (ThrottlingException e) {
211      // expected
212    }
213  }
214
215  private void assertNoThrottleException(final QuotaLimiter limiter, final int availReqs) {
216    for (int i = 0; i < availReqs; ++i) {
217      try {
218        limiter.checkQuota(1, 1);
219      } catch (ThrottlingException e) {
220        fail("Unexpected ThrottlingException after " + i + " requests. limit=" + availReqs);
221      }
222      limiter.grabQuota(1, 1);
223    }
224  }
225
226  private void assertNoopLimiter(final QuotaLimiter limiter) {
227    assertTrue(limiter == NoopQuotaLimiter.get());
228    assertNoThrottleException(limiter, 100);
229  }
