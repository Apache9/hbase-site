068  @Test
069  public void testThrottleType() throws Exception {
070    Admin admin = TEST_UTIL.getHBaseAdmin();
071    String userName = User.getCurrent().getShortName();
072
073    admin.setQuota(QuotaSettingsFactory
074      .throttleUser(userName, ThrottleType.READ_NUMBER, 6, TimeUnit.MINUTES));
075    admin.setQuota(QuotaSettingsFactory
076      .throttleUser(userName, ThrottleType.WRITE_NUMBER, 12, TimeUnit.MINUTES));
077    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, true));
078
079    try (QuotaRetriever scanner = QuotaRetriever.open(TEST_UTIL.getConfiguration())) {
080      int countThrottle = 0;
081      int countGlobalBypass = 0;
082      for (QuotaSettings settings: scanner) {
083        switch (settings.getQuotaType()) {
084          case THROTTLE:
085            ThrottleSettings throttle = (ThrottleSettings)settings;
086            if (throttle.getSoftLimit() == 6) {
087              assertEquals(ThrottleType.READ_NUMBER, throttle.getThrottleType());
088            } else if (throttle.getSoftLimit() == 12) {
089              assertEquals(ThrottleType.WRITE_NUMBER, throttle.getThrottleType());
090            } else {
091              fail("should not come here, because don't set quota with this limit");
092            }
093            assertEquals(userName, throttle.getUserName());
094            assertEquals(null, throttle.getTableName());
095            assertEquals(null, throttle.getNamespace());
096            assertEquals(TimeUnit.MINUTES, throttle.getTimeUnit());
097            countThrottle++;
098            break;
099          case GLOBAL_BYPASS:
100            countGlobalBypass++;
101            break;
102          default:
103            fail("unexpected settings type: " + settings.getQuotaType());
104        }
105      }
106      assertEquals(2, countThrottle);
107      assertEquals(1, countGlobalBypass);
108    }
109
110    admin.setQuota(QuotaSettingsFactory.unthrottleUser(userName));
111    assertNumResults(1, null);
112    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, false));
113    assertNumResults(0, null);
114  }

157  @Test
158  public void testQuotaRetrieverFilter() throws Exception {
159    Admin admin = TEST_UTIL.getHBaseAdmin();
160    TableName[] tables = new TableName[] {
161      TableName.valueOf("T0"), TableName.valueOf("T01"), TableName.valueOf("NS0:T2"),
162    };
163    String[] namespaces = new String[] { "NS0", "NS01", "NS2" };
164    String[] users = new String[] { "User0", "User01", "User2" };
165
166    for (String user: users) {
167      admin.setQuota(QuotaSettingsFactory
168        .throttleUser(user, ThrottleType.REQUEST_NUMBER, 1, TimeUnit.MINUTES));
169
170      for (TableName table: tables) {
171        admin.setQuota(QuotaSettingsFactory
172          .throttleUser(user, table, ThrottleType.REQUEST_NUMBER, 2, TimeUnit.MINUTES));
173      }
174
175      for (String ns: namespaces) {
176        admin.setQuota(QuotaSettingsFactory
177          .throttleUser(user, ns, ThrottleType.REQUEST_NUMBER, 3, TimeUnit.MINUTES));
178      }
179    }
180    assertNumResults(21, null);
181
182    for (TableName table: tables) {
183      admin.setQuota(QuotaSettingsFactory
184        .throttleTable(table, ThrottleType.REQUEST_NUMBER, 4, TimeUnit.MINUTES));
185    }
186    assertNumResults(24, null);
187
188    for (String ns: namespaces) {
189      admin.setQuota(QuotaSettingsFactory
190        .throttleNamespace(ns, ThrottleType.REQUEST_NUMBER, 5, TimeUnit.MINUTES));
191    }
192    assertNumResults(27, null);
193
194    assertNumResults(7, new QuotaFilter().setUserFilter("User0"));
195    assertNumResults(0, new QuotaFilter().setUserFilter("User"));
196    assertNumResults(21, new QuotaFilter().setUserFilter("User.*"));
197    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setTableFilter("T0"));
198    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setTableFilter("NS.*"));
199    assertNumResults(0, new QuotaFilter().setUserFilter("User.*").setTableFilter("T"));
200    assertNumResults(6, new QuotaFilter().setUserFilter("User.*").setTableFilter("T.*"));
201    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS0"));
202    assertNumResults(0, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS"));
203    assertNumResults(9, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS.*"));
204    assertNumResults(6, new QuotaFilter().setUserFilter("User.*")
205                                            .setTableFilter("T0").setNamespaceFilter("NS0"));
206    assertNumResults(1, new QuotaFilter().setTableFilter("T0"));
207    assertNumResults(0, new QuotaFilter().setTableFilter("T"));
208    assertNumResults(2, new QuotaFilter().setTableFilter("T.*"));
209    assertNumResults(3, new QuotaFilter().setTableFilter(".*T.*"));
210    assertNumResults(1, new QuotaFilter().setNamespaceFilter("NS0"));
211    assertNumResults(0, new QuotaFilter().setNamespaceFilter("NS"));
212    assertNumResults(3, new QuotaFilter().setNamespaceFilter("NS.*"));
213
214    for (String user: users) {
215      admin.setQuota(QuotaSettingsFactory.unthrottleUser(user));
216      for (TableName table: tables) {
217        admin.setQuota(QuotaSettingsFactory.unthrottleUser(user, table));
218      }
219      for (String ns: namespaces) {
220        admin.setQuota(QuotaSettingsFactory.unthrottleUser(user, ns));
221      }
222    }
223    assertNumResults(6, null);
224
225    for (TableName table: tables) {
226      admin.setQuota(QuotaSettingsFactory.unthrottleTable(table));
227    }
228    assertNumResults(3, null);
229
230    for (String ns: namespaces) {
231      admin.setQuota(QuotaSettingsFactory.unthrottleNamespace(ns));
232    }
233    assertNumResults(0, null);
234  }
235
236  private void assertNumResults(int expected, final QuotaFilter filter) throws Exception {
237    assertEquals(expected, countResults(filter));
238  }
239
240  private int countResults(final QuotaFilter filter) throws Exception {
241    QuotaRetriever scanner = QuotaRetriever.open(TEST_UTIL.getConfiguration(), filter);
242    try {
243      int count = 0;
244      for (QuotaSettings settings: scanner) {
245        LOG.debug(settings);
246        count++;
247      }
248      return count;
249    } finally {
250      scanner.close();
251    }
252  }
