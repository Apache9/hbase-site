467    final String NAMESPACE = "default";

474    admin.setQuota(QuotaSettingsFactory

476    triggerNamespaceCacheRefresh(false, TABLE_NAMES[1]);
477
478    // should execute at max 6 requests on table[0] and (13 - 6) on table[1]
479    assertEquals(6, doPuts(100, tables[0]));
480    assertEquals(7, doGets(100, tables[1]));

482
483    // Set the global bypass for the user
484    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, true));
485    admin.setQuota(QuotaSettingsFactory
486      .throttleUser(userName, TABLE_NAMES[2], ThrottleType.REQUEST_NUMBER, 6, TimeUnit.MINUTES));
487    triggerUserCacheRefresh(false, TABLE_NAMES[2]);

489    assertEquals(30, doGets(30, tables[1]));
490    waitMinuteQuota();
491
492    // Remove the global bypass
493    // should execute at max 6 requests on table[0] and (13 - 6) on table[1]
494    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, false));
495    admin.setQuota(QuotaSettingsFactory.unthrottleUser(userName, TABLE_NAMES[2]));
496    triggerUserCacheRefresh(true, TABLE_NAMES[2]);
497    assertEquals(6, doPuts(100, tables[0]));
498    assertEquals(7, doGets(100, tables[1]));
499
500    // unset throttle
501    admin.setQuota(QuotaSettingsFactory.unthrottleTable(TABLE_NAMES[0]));
502    admin.setQuota(QuotaSettingsFactory.unthrottleNamespace(NAMESPACE));
503    waitMinuteQuota();
504    triggerTableCacheRefresh(true, TABLE_NAMES[0]);
505    triggerNamespaceCacheRefresh(true, TABLE_NAMES[1]);
506    assertEquals(30, doGets(30, tables[0]));
507    assertEquals(30, doGets(30, tables[1]));
508  }

510  private int doPuts(int maxOps, final Table... tables) throws Exception {
511    int count = 0;
512    try {
513      while (count < maxOps) {
514        Put put = new Put(Bytes.toBytes("row-" + count));
515        put.addColumn(FAMILY, QUALIFIER, Bytes.toBytes("data-" + count));
516        for (final Table table: tables) {
517          table.put(put);
518        }
519        count += tables.length;
520      }
521    } catch (ThrottlingException e) {
522      LOG.error("put failed after nRetries=" + count, e);
523    }
524    return count;
525  }
526
527  private long doGets(int maxOps, final Table... tables) throws Exception {
528    int count = 0;
529    try {
530      while (count < maxOps) {
531        Get get = new Get(Bytes.toBytes("row-" + count));
532        for (final Table table: tables) {
533          table.get(get);
534        }
535        count += tables.length;
536      }
537    } catch (ThrottlingException e) {
538      LOG.error("get failed after nRetries=" + count, e);
539    }
540    return count;
541  }

543  private void triggerUserCacheRefresh(boolean bypass, TableName... tables) throws Exception {
544    triggerCacheRefresh(bypass, true, false, false, tables);
545  }
546
547  private void triggerTableCacheRefresh(boolean bypass, TableName... tables) throws Exception {

549  }

551  private void triggerNamespaceCacheRefresh(boolean bypass, TableName... tables) throws Exception {
552    triggerCacheRefresh(bypass, false, false, true, tables);

554
555  private void triggerCacheRefresh(boolean bypass, boolean userLimiter, boolean tableLimiter,
556      boolean nsLimiter, final TableName... tables) throws Exception {
557    envEdge.incValue(2 * REFRESH_TIME);
558    for (RegionServerThread rst: TEST_UTIL.getMiniHBaseCluster().getRegionServerThreads()) {
559      RegionServerRpcQuotaManager quotaManager = rst.getRegionServer().getRegionServerRpcQuotaManager();
560      QuotaCache quotaCache = quotaManager.getQuotaCache();
561
562      quotaCache.triggerCacheRefresh();
563      // sleep for cache update
564      Thread.sleep(250);
565
566      for (TableName table: tables) {
567        quotaCache.getTableLimiter(table);
568      }
569
570      boolean isUpdated = false;
571      while (!isUpdated) {
572        quotaCache.triggerCacheRefresh();
573        isUpdated = true;
574        for (TableName table: tables) {
575          boolean isBypass = true;
576          if (userLimiter) {
577            isBypass &= quotaCache.getUserLimiter(User.getCurrent().getUGI(), table).isBypass();
578          }
579          if (tableLimiter) {
580            isBypass &= quotaCache.getTableLimiter(table).isBypass();
581          }
582          if (nsLimiter) {
583            isBypass &= quotaCache.getNamespaceLimiter(table.getNamespaceAsString()).isBypass();
584          }
585          if (isBypass != bypass) {
586            envEdge.incValue(100);
587            isUpdated = false;
588            break;
589          }
590        }
591      }
592
593      LOG.debug("QuotaCache");
594      LOG.debug(Objects.toString(quotaCache.getNamespaceQuotaCache()));
595      LOG.debug(Objects.toString(quotaCache.getTableQuotaCache()));
596      LOG.debug(Objects.toString(quotaCache.getUserQuotaCache()));
597    }
598  }
