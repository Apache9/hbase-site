019package org.apache.hadoop.hbase;
020
021import java.io.IOException;
022import java.util.concurrent.CountDownLatch;
023import java.util.concurrent.TimeUnit;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.client.Admin;
028import org.apache.hadoop.hbase.testclassification.IntegrationTests;
029import org.apache.hadoop.hbase.util.RegionSplitter;
030import org.apache.hadoop.hbase.util.RegionSplitter.SplitAlgorithm;
031import org.junit.After;
032import org.junit.Before;
033import org.junit.Test;
034import org.junit.experimental.categories.Category;
035
036import static org.junit.Assert.assertTrue;
037import static org.junit.Assert.fail;
038
039/**
040 * An integration test to detect regressions in HBASE-7220. Create
041 * a table with many regions and verify it completes within a
042 * reasonable amount of time.
043 * @see <a href="https://issues.apache.org/jira/browse/HBASE-7220">HBASE-7220</a>
044 */
045@Category(IntegrationTests.class)
046public class IntegrationTestManyRegions {
047
048  private static final String CLASS_NAME
049    = IntegrationTestManyRegions.class.getSimpleName();
050
051  protected static final Log LOG
052    = LogFactory.getLog(IntegrationTestManyRegions.class);
053  protected static final TableName TABLE_NAME = TableName.valueOf(CLASS_NAME);
054  protected static final String COLUMN_NAME = "f";
055  protected static final String REGION_COUNT_KEY
056    = String.format("hbase.%s.regions", CLASS_NAME);
057  protected static final String REGIONSERVER_COUNT_KEY
058    = String.format("hbase.%s.regionServers", CLASS_NAME);
059  protected static final String TIMEOUT_MINUTES_KEY
060    = String.format("hbase.%s.timeoutMinutes", CLASS_NAME);
061
062  protected static final int DEFAULT_REGION_COUNT = 1000;
063  protected static final int DEFAULT_REGIONSERVER_COUNT = 1;
064  // running this test on my laptop consistently takes about 2.5
065  // minutes. A timeout of 4 minutes should be reasonably safe.
066  protected static final int DEFAULT_TIMEOUT_MINUTES = 4;
067  protected static final IntegrationTestingUtility util
068    = new IntegrationTestingUtility();
069
070  protected static final int REGION_COUNT = util.getConfiguration()
071    .getInt(REGION_COUNT_KEY, DEFAULT_REGION_COUNT);
072  protected static final int REGION_SERVER_COUNT = util.getConfiguration()
073    .getInt(REGIONSERVER_COUNT_KEY, DEFAULT_REGIONSERVER_COUNT);
074  protected static final int TIMEOUT_MINUTES = util.getConfiguration()
075    .getInt(TIMEOUT_MINUTES_KEY, DEFAULT_TIMEOUT_MINUTES);
076
077  @Before
078  public void setUp() throws Exception {
079    LOG.info(String.format("Initializing cluster with %d region servers.",
080      REGION_SERVER_COUNT));
081    util.initializeCluster(REGION_SERVER_COUNT);
082    LOG.info("Cluster initialized");
083
084    Admin admin = util.getHBaseAdmin();
085    if (admin.tableExists(TABLE_NAME)) {
086      LOG.info(String.format("Deleting existing table %s.", TABLE_NAME));
087      if (admin.isTableEnabled(TABLE_NAME)) admin.disableTable(TABLE_NAME);
088      admin.deleteTable(TABLE_NAME);
089      LOG.info(String.format("Existing table %s deleted.", TABLE_NAME));
090    }
091    LOG.info("Cluster ready");
092  }
093
094  @After
095  public void tearDown() throws IOException {
096    LOG.info("Cleaning up after test.");
097    Admin admin = util.getHBaseAdmin();
098    if (admin.tableExists(TABLE_NAME)) {
099      if (admin.isTableEnabled(TABLE_NAME)) admin.disableTable(TABLE_NAME);
100      admin.deleteTable(TABLE_NAME);
101    }
102    LOG.info("Restoring cluster.");
103    util.restoreCluster();
104    LOG.info("Cluster restored.");
105  }
106
107  @Test
108  public void testCreateTableWithRegions() throws Exception {
109    CountDownLatch doneSignal = new CountDownLatch(1);
110    Worker worker = new Worker(doneSignal, util.getHBaseAdmin());
111    Thread t = new Thread(worker);
112
113    LOG.info("Launching worker thread to create the table.");
114    t.start();
115    boolean workerComplete = false;
116    workerComplete = doneSignal.await(TIMEOUT_MINUTES, TimeUnit.MINUTES);
117    if (!workerComplete) {
118      t.interrupt();
119      fail("Timeout limit expired.");
120    }
121    assertTrue("Table creation failed.", worker.isSuccess());
122  }
123
124  private static class Worker implements Runnable {
125    private final CountDownLatch doneSignal;
126    private final Admin admin;
127    private boolean success = false;
128
129    public Worker(final CountDownLatch doneSignal, final Admin admin) {
130      this.doneSignal = doneSignal;
131      this.admin = admin;
132    }
133
134    public boolean isSuccess() {
135      return this.success;
136    }
137
138    @Override
139    public void run() {
140      long startTime, endTime;
141      HTableDescriptor desc = new HTableDescriptor(TABLE_NAME);
142      desc.addFamily(new HColumnDescriptor(COLUMN_NAME));
143      SplitAlgorithm algo = new RegionSplitter.HexStringSplit();
144      byte[][] splits = algo.split(REGION_COUNT);
145
146      LOG.info(String.format("Creating table %s with %d splits.",
147        TABLE_NAME, REGION_COUNT));
148      startTime = System.currentTimeMillis();
149      try {
150        admin.createTable(desc, splits);
151        endTime = System.currentTimeMillis();
152        success = true;
153        LOG.info(String.format("Pre-split table created successfully in %dms.",
154          (endTime - startTime)));
155      } catch (IOException e) {
156        LOG.error("Failed to create table", e);
157      } finally {
158        doneSignal.countDown();
159      }
160    }
161  }
162}
