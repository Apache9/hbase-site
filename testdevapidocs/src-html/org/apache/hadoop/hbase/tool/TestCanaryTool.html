048import static org.junit.Assert.*;

050import static org.mockito.Matchers.isA;

057@RunWith(MockitoJUnitRunner.class)
058@Category({MediumTests.class})
059public class TestCanaryTool {
060
061  private HBaseTestingUtility testingUtility;
062  private static final byte[] FAMILY = Bytes.toBytes("f");
063  private static final byte[] COLUMN = Bytes.toBytes("col");
064
065  @Before
066  public void setUp() throws Exception {
067    testingUtility = new HBaseTestingUtility();
068    testingUtility.startMiniCluster();
069    LogManager.getRootLogger().addAppender(mockAppender);
070  }
071
072  @After
073  public void tearDown() throws Exception {
074    testingUtility.shutdownMiniCluster();
075    LogManager.getRootLogger().removeAppender(mockAppender);
076  }
077
078  @Mock
079  Appender mockAppender;
080
081  @Test
082  public void testBasicCanaryWorks() throws Exception {
083    TableName tableName = TableName.valueOf("testTable");
084    Table table = testingUtility.createTable(tableName, new byte[][] { FAMILY });
085    // insert some test rows
086    for (int i=0; i<1000; i++) {
087      byte[] iBytes = Bytes.toBytes(i);
088      Put p = new Put(iBytes);
089      p.addColumn(FAMILY, COLUMN, iBytes);
090      table.put(p);
091    }
092    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
093    Canary.RegionServerStdOutSink sink = spy(new Canary.RegionServerStdOutSink());
094    Canary canary = new Canary(executor, sink);
095    String[] args = { "-t", "10000", "testTable" };
096    ToolRunner.run(testingUtility.getConfiguration(), canary, args);
097    verify(sink, atLeastOnce())
098        .publishReadTiming(isA(HRegionInfo.class), isA(HColumnDescriptor.class), anyLong());
099  }
100
101  //no table created, so there should be no regions
102  @Test
103  public void testRegionserverNoRegions() throws Exception {
104    runRegionserverCanary();
105    verify(mockAppender).doAppend(argThat(new ArgumentMatcher<LoggingEvent>() {
106      @Override
107      public boolean matches(Object argument) {
108        return ((LoggingEvent) argument).getRenderedMessage().contains("Regionserver not serving any regions");
109      }
110    }));
111  }
112
113  //by creating a table, there shouldn't be any region servers not serving any regions
114  @Test
115  public void testRegionserverWithRegions() throws Exception {
116    TableName tableName = TableName.valueOf("testTable");
117    testingUtility.createTable(tableName, new byte[][] { FAMILY });
118    runRegionserverCanary();
119    verify(mockAppender, never()).doAppend(argThat(new ArgumentMatcher<LoggingEvent>() {
120      @Override
121      public boolean matches(Object argument) {
122        return ((LoggingEvent) argument).getRenderedMessage().contains("Regionserver not serving any regions");
123      }
124    }));
125  }
126
127  private void runRegionserverCanary() throws Exception {
128    ExecutorService executor = new ScheduledThreadPoolExecutor(1);
129    Canary canary = new Canary(executor, new Canary.RegionServerStdOutSink());
130    String[] args = { "-t", "10000", "-regionserver"};
131    ToolRunner.run(testingUtility.getConfiguration(), canary, args);
132  }
