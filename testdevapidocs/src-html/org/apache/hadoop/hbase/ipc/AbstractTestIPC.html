092  // We are using the test TestRpcServiceProtos generated classes and Service because they are
093  // available and basic with methods like 'echo', and ping. Below we make a blocking service
094  // by passing in implementation of blocking interface. We use this service in all tests that
095  // follow.
096  static final BlockingService SERVICE =
097      TestRpcServiceProtos.TestProtobufRpcProto
098          .newReflectiveBlockingService(new TestRpcServiceProtos.TestProtobufRpcProto.BlockingInterface() {
099
100            @Override
101            public EmptyResponseProto ping(RpcController controller, EmptyRequestProto request)
102                throws ServiceException {
103              return null;
104            }
105
106            @Override
107            public EmptyResponseProto error(RpcController controller, EmptyRequestProto request)
108                throws ServiceException {
109              return null;
110            }
111
112            @Override
113            public EchoResponseProto echo(RpcController controller, EchoRequestProto request)
114                throws ServiceException {
115              if (controller instanceof PayloadCarryingRpcController) {
116                PayloadCarryingRpcController pcrc = (PayloadCarryingRpcController) controller;
117                // If cells, scan them to check we are able to iterate what we were given and since
118                // this is
119                // an echo, just put them back on the controller creating a new block. Tests our
120                // block
121                // building.
122                CellScanner cellScanner = pcrc.cellScanner();
123                List<Cell> list = null;
124                if (cellScanner != null) {
125                  list = new ArrayList<>();
126                  try {
127                    while (cellScanner.advance()) {
128                      list.add(cellScanner.current());
129                    }
130                  } catch (IOException e) {
131                    throw new ServiceException(e);
132                  }
133                }
134                cellScanner = CellUtil.createCellScanner(list);
135                ((PayloadCarryingRpcController) controller).setCellScanner(cellScanner);
136              }
137              return EchoResponseProto.newBuilder().setMessage(request.getMessage()).build();
138            }
139          });

532  private Message setupAsyncConnection(TestRpcServer rpcServer, RpcClient client)
533      throws IOException, InterruptedException, ExecutionException,
534      java.util.concurrent.TimeoutException {
535    InetSocketAddress address = rpcServer.getListenerAddress();
536    if (address == null) {
537      throw new IOException("Listener channel is closed");
538    }
539    MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
540    EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
541
542    ServerName serverName =
543        ServerName.valueOf(address.getHostName(), address.getPort(), System.currentTimeMillis());
544
545    AsyncRpcChannel channel =
546        client.createRpcChannel(md.getService().getName(), serverName, User.getCurrent());
547
548    final Future<Message> f = channel
549        .callMethod(md, param, null, md.getOutputType().toProto(), MessageConverter.NO_CONVERTER,
550            null, 1000, HConstants.NORMAL_QOS);
551
552    return f.get(1, TimeUnit.SECONDS);
