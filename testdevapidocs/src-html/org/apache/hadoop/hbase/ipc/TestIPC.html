064@Category({ RPCTests.class, SmallTests.class })
065public class TestIPC extends AbstractTestIPC {
066
067  private static final Log LOG = LogFactory.getLog(TestIPC.class);
068
069  @Override
070  protected RpcClientImpl createRpcClientNoCodec(Configuration conf) {
071    return new RpcClientImpl(conf, HConstants.CLUSTER_ID_DEFAULT) {
072      @Override
073      Codec getCodec() {
074        return null;
075      }
076    };
077  }
078
079  @Override
080  protected RpcClientImpl createRpcClient(Configuration conf) {
081    return new RpcClientImpl(conf, HConstants.CLUSTER_ID_DEFAULT);
082  }
083
084  @Override
085  protected RpcClientImpl createRpcClientRTEDuringConnectionSetup(Configuration conf)
086      throws IOException {
087    SocketFactory spyFactory = spy(NetUtils.getDefaultSocketFactory(conf));
088    Mockito.doAnswer(new Answer<Socket>() {
089      @Override
090      public Socket answer(InvocationOnMock invocation) throws Throwable {
091        Socket s = spy((Socket) invocation.callRealMethod());
092        doThrow(new RuntimeException("Injected fault")).when(s).setSoTimeout(anyInt());
093        return s;
094      }
095    }).when(spyFactory).createSocket();
096
097    return new RpcClientImpl(conf, HConstants.CLUSTER_ID_DEFAULT, spyFactory);
098  }
099
100  public static void main(String[] args) throws IOException, SecurityException,
101      NoSuchMethodException, InterruptedException {
102    if (args.length != 2) {
103      System.out.println("Usage: TestIPC <CYCLES> <CELLS_PER_CYCLE>");
104      return;
105    }
106    // ((Log4JLogger)HBaseServer.LOG).getLogger().setLevel(Level.INFO);
107    // ((Log4JLogger)HBaseClient.LOG).getLogger().setLevel(Level.INFO);
108    int cycles = Integer.parseInt(args[0]);
109    int cellcount = Integer.parseInt(args[1]);
110    Configuration conf = HBaseConfiguration.create();
111    TestRpcServer rpcServer = new TestRpcServer();
112    MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
113    EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
114    RpcClientImpl client = new RpcClientImpl(conf, HConstants.CLUSTER_ID_DEFAULT);
115    KeyValue kv = BIG_CELL;
116    Put p = new Put(CellUtil.cloneRow(kv));
117    for (int i = 0; i < cellcount; i++) {
118      p.add(kv);
119    }
120    RowMutations rm = new RowMutations(CellUtil.cloneRow(kv));
121    rm.add(p);
122    try {
123      rpcServer.start();
124      long startTime = System.currentTimeMillis();
125      User user = User.getCurrent();
126      InetSocketAddress address = rpcServer.getListenerAddress();
127      if (address == null) {
128        throw new IOException("Listener channel is closed");
129      }
130      for (int i = 0; i < cycles; i++) {
131        List<CellScannable> cells = new ArrayList<>();
132        // Message param = RequestConverter.buildMultiRequest(HConstants.EMPTY_BYTE_ARRAY, rm);
133        ClientProtos.RegionAction.Builder builder =
134            RequestConverter.buildNoDataRegionAction(HConstants.EMPTY_BYTE_ARRAY, rm, cells,
135              RegionAction.newBuilder(), ClientProtos.Action.newBuilder(),
136              MutationProto.newBuilder());
137        builder.setRegion(RegionSpecifier
138            .newBuilder()
139            .setType(RegionSpecifierType.REGION_NAME)
140            .setValue(
141              ByteString.copyFrom(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes())));
142        if (i % 100000 == 0) {
143          LOG.info("" + i);
144          // Uncomment this for a thread dump every so often.
145          // ReflectionUtils.printThreadInfo(new PrintWriter(System.out),
146          // "Thread dump " + Thread.currentThread().getName());
147        }
148        PayloadCarryingRpcController pcrc =
149            new PayloadCarryingRpcController(CellUtil.createCellScanner(cells));
150        // Pair<Message, CellScanner> response =
151        client.call(pcrc, md, builder.build(), param, user, address,
152            new MetricsConnection.CallStats());
153        /*
154         * int count = 0; while (p.getSecond().advance()) { count++; } assertEquals(cells.size(),
155         * count);
156         */
157      }
158      LOG.info("Cycled " + cycles + " time(s) with " + cellcount + " cell(s) in "
159          + (System.currentTimeMillis() - startTime) + "ms");
160    } finally {
161      client.close();
162      rpcServer.stop();
163    }
164  }
