064@RunWith(Parameterized.class)
065@Category({ RPCTests.class, SmallTests.class })
066public class TestAsyncIPC extends AbstractTestIPC {
067
068  private static final Log LOG = LogFactory.getLog(TestAsyncIPC.class);
069
070  @Parameters
071  public static Collection<Object[]> parameters() {
072    List<Object[]> paramList = new ArrayList<>();
073    paramList.add(new Object[] { false, false });
074    paramList.add(new Object[] { false, true });
075    paramList.add(new Object[] { true, false });
076    paramList.add(new Object[] { true, true });
077    return paramList;
078  }
079
080  private final boolean useNativeTransport;
081
082  private final boolean useGlobalEventLoopGroup;
083
084  public TestAsyncIPC(boolean useNativeTransport, boolean useGlobalEventLoopGroup) {
085    this.useNativeTransport = useNativeTransport;
086    this.useGlobalEventLoopGroup = useGlobalEventLoopGroup;
087  }
088
089  private void setConf(Configuration conf) {
090    conf.setBoolean(AsyncRpcClient.USE_NATIVE_TRANSPORT, useNativeTransport);
091    conf.setBoolean(AsyncRpcClient.USE_GLOBAL_EVENT_LOOP_GROUP, useGlobalEventLoopGroup);
092    if (useGlobalEventLoopGroup && AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP != null) {
093      if (useNativeTransport
094          && !(AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst() instanceof EpollEventLoopGroup)
095          || (!useNativeTransport
096          && !(AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst() instanceof NioEventLoopGroup))) {
097        AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst().shutdownGracefully();
098        AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP = null;
099      }
100    }
101  }
102
103  @Override
104  protected AsyncRpcClient createRpcClientNoCodec(Configuration conf) {
105    setConf(conf);
106    return new AsyncRpcClient(conf) {
107
108      @Override
109      Codec getCodec() {
110        return null;
111      }
112
113    };
114  }
115
116  @Override
117  protected AsyncRpcClient createRpcClient(Configuration conf) {
118    setConf(conf);
119    return new AsyncRpcClient(conf);
120  }
121
122  @Override
123  protected AsyncRpcClient createRpcClientRTEDuringConnectionSetup(Configuration conf) {
124    setConf(conf);
125    return new AsyncRpcClient(conf, new ChannelInitializer<SocketChannel>() {
126          @Override
127          protected void initChannel(SocketChannel ch) throws Exception {
128            ch.pipeline().addFirst(new ChannelOutboundHandlerAdapter() {
129              @Override
130              public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
131                  throws Exception {
132                promise.setFailure(new RuntimeException("Injected fault"));
133              }
134            });
135          }
136        });
137  }
138
139  public static void main(String[] args) throws IOException, SecurityException,
140      NoSuchMethodException, InterruptedException {
141    if (args.length != 2) {
142      System.out.println("Usage: TestAsyncIPC <CYCLES> <CELLS_PER_CYCLE>");
143      return;
144    }
145    // ((Log4JLogger)HBaseServer.LOG).getLogger().setLevel(Level.INFO);
146    // ((Log4JLogger)HBaseClient.LOG).getLogger().setLevel(Level.INFO);
147    int cycles = Integer.parseInt(args[0]);
148    int cellcount = Integer.parseInt(args[1]);
149    Configuration conf = HBaseConfiguration.create();
150    TestRpcServer rpcServer = new TestRpcServer();
151    MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
152    EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
153    KeyValue kv = BIG_CELL;
154    Put p = new Put(CellUtil.cloneRow(kv));
155    for (int i = 0; i < cellcount; i++) {
156      p.add(kv);
157    }
158    RowMutations rm = new RowMutations(CellUtil.cloneRow(kv));
159    rm.add(p);
160    try (AsyncRpcClient client = new AsyncRpcClient(conf)) {
161      rpcServer.start();
162      InetSocketAddress address = rpcServer.getListenerAddress();
163      if (address == null) {
164        throw new IOException("Listener channel is closed");
165      }
166      long startTime = System.currentTimeMillis();
167      User user = User.getCurrent();
168      for (int i = 0; i < cycles; i++) {
169        List<CellScannable> cells = new ArrayList<>();
170        // Message param = RequestConverter.buildMultiRequest(HConstants.EMPTY_BYTE_ARRAY, rm);
171        ClientProtos.RegionAction.Builder builder =
172            RequestConverter.buildNoDataRegionAction(HConstants.EMPTY_BYTE_ARRAY, rm, cells,
173                RegionAction.newBuilder(), ClientProtos.Action.newBuilder(),
174                MutationProto.newBuilder());
175        builder.setRegion(RegionSpecifier
176            .newBuilder()
177            .setType(RegionSpecifierType.REGION_NAME)
178            .setValue(
179                ByteString.copyFrom(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes())));
180        if (i % 100000 == 0) {
181          LOG.info("" + i);
182          // Uncomment this for a thread dump every so often.
183          // ReflectionUtils.printThreadInfo(new PrintWriter(System.out),
184          // "Thread dump " + Thread.currentThread().getName());
185        }
186        PayloadCarryingRpcController pcrc =
187            new PayloadCarryingRpcController(CellUtil.createCellScanner(cells));
188        // Pair<Message, CellScanner> response =
189        client.call(pcrc, md, builder.build(), param, user, address,
190            new MetricsConnection.CallStats());
191        /*
192         * int count = 0; while (p.getSecond().advance()) { count++; } assertEquals(cells.size(),
193         * count);
194         */
195      }
196      LOG.info("Cycled " + cycles + " time(s) with " + cellcount + " cell(s) in "
197          + (System.currentTimeMillis() - startTime) + "ms");
198    } finally {
199      rpcServer.stop();
200    }
201  }
