098  public TestAsyncIPC(boolean useNativeTransport, boolean useGlobalEventLoopGroup) {

103  private void setConf(Configuration conf) {
104    conf.setBoolean(AsyncRpcClient.USE_NATIVE_TRANSPORT, useNativeTransport);
105    conf.setBoolean(AsyncRpcClient.USE_GLOBAL_EVENT_LOOP_GROUP, useGlobalEventLoopGroup);
106    if (useGlobalEventLoopGroup && AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP != null) {
107      if (useNativeTransport
108          && !(AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst() instanceof EpollEventLoopGroup)
109          || (!useNativeTransport
110          && !(AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst() instanceof NioEventLoopGroup))) {
111        AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP.getFirst().shutdownGracefully();
112        AsyncRpcClient.GLOBAL_EVENT_LOOP_GROUP = null;
113      }
114    }
115  }

136  @Override

138    setConf(conf);
139    return new AsyncRpcClient(conf, new ChannelInitializer<SocketChannel>() {

141          protected void initChannel(SocketChannel ch) throws Exception {
142            ch.pipeline().addFirst(new ChannelOutboundHandlerAdapter() {
143              @Override
144              public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)

153  @Test
154  public void testAsyncConnectionSetup() throws Exception {
155    TestRpcServer rpcServer = new TestRpcServer();
156    AsyncRpcClient client = createRpcClient(CONF);
157    try {
158      rpcServer.start();
159      InetSocketAddress address = rpcServer.getListenerAddress();
160      if (address == null) {
161        throw new IOException("Listener channel is closed");
162      }
163      MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
164      EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
165
166      RpcChannel channel =
167          client.createRpcChannel(ServerName.valueOf(address.getHostName(), address.getPort(),
168            System.currentTimeMillis()), User.getCurrent(), 0);
169
170      final AtomicBoolean done = new AtomicBoolean(false);
171
172      channel.callMethod(md, new PayloadCarryingRpcController(), param, md.getOutputType()
173          .toProto(), new RpcCallback<Message>() {
174        @Override
175        public void run(Message parameter) {
176          done.set(true);
177        }
178      });
179
180      TEST_UTIL.waitFor(1000, new Waiter.Predicate<Exception>() {
181        @Override
182        public boolean evaluate() throws Exception {
183          return done.get();
184        }
185      });
186    } finally {
187      client.close();
188      rpcServer.stop();
189    }
190  }

192  @Test
193  public void testRTEDuringAsyncConnectionSetup() throws Exception {
194    TestRpcServer rpcServer = new TestRpcServer();
195    AsyncRpcClient client = createRpcClientRTEDuringConnectionSetup(CONF);
196    try {
197      rpcServer.start();
198      InetSocketAddress address = rpcServer.getListenerAddress();
199      if (address == null) {
200        throw new IOException("Listener channel is closed");
201      }
202      MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
203      EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
204
205      RpcChannel channel =
206          client.createRpcChannel(ServerName.valueOf(address.getHostName(), address.getPort(),
207            System.currentTimeMillis()), User.getCurrent(), 0);
208
209      final AtomicBoolean done = new AtomicBoolean(false);
210
211      PayloadCarryingRpcController controller = new PayloadCarryingRpcController();
212      controller.notifyOnFail(new RpcCallback<IOException>() {
213        @Override
214        public void run(IOException e) {
215          done.set(true);
216          LOG.info("Caught expected exception: " + e.toString());
217          assertTrue(StringUtils.stringifyException(e).contains("Injected fault"));
218        }
219      });
220
221      channel.callMethod(md, controller, param, md.getOutputType().toProto(),
222        new RpcCallback<Message>() {
223          @Override
224          public void run(Message parameter) {
225            done.set(true);
226            fail("Expected an exception to have been thrown!");
227          }
228        });
229
230      TEST_UTIL.waitFor(1000, new Waiter.Predicate<Exception>() {
231        @Override
232        public boolean evaluate() throws Exception {
233          return done.get();
234        }
235      });
236    } finally {
237      client.close();
238      rpcServer.stop();
239    }
240  }

242  public static void main(String[] args) throws IOException, SecurityException,
243      NoSuchMethodException, InterruptedException {
244    if (args.length != 2) {
245      System.out.println("Usage: TestAsyncIPC <CYCLES> <CELLS_PER_CYCLE>");
246      return;
247    }
248    // ((Log4JLogger)HBaseServer.LOG).getLogger().setLevel(Level.INFO);
249    // ((Log4JLogger)HBaseClient.LOG).getLogger().setLevel(Level.INFO);
250    int cycles = Integer.parseInt(args[0]);
251    int cellcount = Integer.parseInt(args[1]);
252    Configuration conf = HBaseConfiguration.create();
253    TestRpcServer rpcServer = new TestRpcServer();
254    MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
255    EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
256    AsyncRpcClient client = new AsyncRpcClient(conf);
257    KeyValue kv = BIG_CELL;
258    Put p = new Put(CellUtil.cloneRow(kv));
259    for (int i = 0; i < cellcount; i++) {
260      p.add(kv);
261    }
262    RowMutations rm = new RowMutations(CellUtil.cloneRow(kv));
263    rm.add(p);
264    try {
265      rpcServer.start();
266      InetSocketAddress address = rpcServer.getListenerAddress();
267      if (address == null) {
268        throw new IOException("Listener channel is closed");
269      }
270      long startTime = System.currentTimeMillis();
271      User user = User.getCurrent();
272      for (int i = 0; i < cycles; i++) {
273        List<CellScannable> cells = new ArrayList<CellScannable>();
274        // Message param = RequestConverter.buildMultiRequest(HConstants.EMPTY_BYTE_ARRAY, rm);
275        ClientProtos.RegionAction.Builder builder =
276            RequestConverter.buildNoDataRegionAction(HConstants.EMPTY_BYTE_ARRAY, rm, cells,
277              RegionAction.newBuilder(), ClientProtos.Action.newBuilder(),
278              MutationProto.newBuilder());
279        builder.setRegion(RegionSpecifier
280            .newBuilder()
281            .setType(RegionSpecifierType.REGION_NAME)
282            .setValue(
283              ByteString.copyFrom(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes())));
284        if (i % 100000 == 0) {
285          LOG.info("" + i);
286          // Uncomment this for a thread dump every so often.
287          // ReflectionUtils.printThreadInfo(new PrintWriter(System.out),
288          // "Thread dump " + Thread.currentThread().getName());
289        }
290        PayloadCarryingRpcController pcrc =
291            new PayloadCarryingRpcController(CellUtil.createCellScanner(cells));
292        // Pair<Message, CellScanner> response =
293        client.call(pcrc, md, builder.build(), param, user, address,
294            new MetricsConnection.CallStats());
295        /*
296         * int count = 0; while (p.getSecond().advance()) { count++; } assertEquals(cells.size(),
297         * count);
298         */
299      }
300      LOG.info("Cycled " + cycles + " time(s) with " + cellcount + " cell(s) in "
301          + (System.currentTimeMillis() - startTime) + "ms");
302    } finally {
303      client.close();
304      rpcServer.stop();
305    }
306  }
