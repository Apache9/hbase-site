115  @Test
116  public void testHandlerIsolation() throws IOException, InterruptedException {
117
118    CallRunner generalTask = createMockTask();
119    CallRunner priorityTask = createMockTask();
120    CallRunner replicationTask = createMockTask();
121    List<CallRunner> tasks = ImmutableList.of(
122        generalTask,
123        priorityTask,
124        replicationTask);
125    Map<CallRunner, Integer> qos = ImmutableMap.of(
126        generalTask, 0,
127        priorityTask, HConstants.HIGH_QOS + 1,
128        replicationTask, HConstants.REPLICATION_QOS);
129    PriorityFunction qosFunction = mock(PriorityFunction.class);
130    final Map<CallRunner, Thread> handlerThreads = Maps.newHashMap();
131    final CountDownLatch countDownLatch = new CountDownLatch(tasks.size());
132    Answer<Void> answerToRun = new Answer<Void>() {
133      @Override
134      public Void answer(InvocationOnMock invocationOnMock) throws Throwable {
135        synchronized (handlerThreads) {
136          handlerThreads.put(
137              (CallRunner) invocationOnMock.getMock(),
138              Thread.currentThread());
139        }
140        countDownLatch.countDown();
141        return null;
142      }
143    };
144    for (CallRunner task : tasks) {
145      task.setStatus(new MonitoredRPCHandlerImpl());
146      doAnswer(answerToRun).when(task).run();
147    }
148
149    RpcScheduler scheduler = new SimpleRpcScheduler(
150        conf, 1, 1 ,1, qosFunction, HConstants.HIGH_QOS);
151    scheduler.init(CONTEXT);
152    scheduler.start();
153    for (CallRunner task : tasks) {
154      when(qosFunction.getPriority((RPCProtos.RequestHeader) anyObject(),
155        (Message) anyObject(), (User) anyObject()))
156          .thenReturn(qos.get(task));
157      scheduler.dispatch(task);
158    }
159    for (CallRunner task : tasks) {
160      verify(task, timeout(1000)).run();
161    }
162    scheduler.stop();
163
164    // Tests that these requests are handled by three distinct threads.
165    countDownLatch.await();
166    assertEquals(3, ImmutableSet.copyOf(handlerThreads.values()).size());
167  }

183  private void testRpcScheduler(final String queueType) throws Exception {
184    Configuration schedConf = HBaseConfiguration.create();
185    schedConf.set(RpcExecutor.CALL_QUEUE_TYPE_CONF_KEY, queueType);
186
187    PriorityFunction priority = mock(PriorityFunction.class);
188    when(priority.getPriority(any(RequestHeader.class),
189      any(Message.class), any(User.class)))
190      .thenReturn(HConstants.NORMAL_QOS);
191
192    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 1, 1, 1, priority,
193                                                    HConstants.QOS_THRESHOLD);
194    try {
195      scheduler.start();
196
197      CallRunner smallCallTask = mock(CallRunner.class);
198      RpcServer.Call smallCall = mock(RpcServer.Call.class);
199      RequestHeader smallHead = RequestHeader.newBuilder().setCallId(1).build();
200      when(smallCallTask.getRpcCall()).thenReturn(smallCall);
201      when(smallCall.getHeader()).thenReturn(smallHead);
202
203      CallRunner largeCallTask = mock(CallRunner.class);
204      RpcServer.Call largeCall = mock(RpcServer.Call.class);
205      RequestHeader largeHead = RequestHeader.newBuilder().setCallId(50).build();
206      when(largeCallTask.getRpcCall()).thenReturn(largeCall);
207      when(largeCall.getHeader()).thenReturn(largeHead);
208
209      CallRunner hugeCallTask = mock(CallRunner.class);
210      RpcServer.Call hugeCall = mock(RpcServer.Call.class);
211      RequestHeader hugeHead = RequestHeader.newBuilder().setCallId(100).build();
212      when(hugeCallTask.getRpcCall()).thenReturn(hugeCall);
213      when(hugeCall.getHeader()).thenReturn(hugeHead);
214
215      when(priority.getDeadline(eq(smallHead), any(Message.class))).thenReturn(0L);
216      when(priority.getDeadline(eq(largeHead), any(Message.class))).thenReturn(50L);
217      when(priority.getDeadline(eq(hugeHead), any(Message.class))).thenReturn(100L);
218
219      final ArrayList<Integer> work = new ArrayList<>();
220      doAnswerTaskExecution(smallCallTask, work, 10, 250);
221      doAnswerTaskExecution(largeCallTask, work, 50, 250);
222      doAnswerTaskExecution(hugeCallTask, work, 100, 250);
223
224      scheduler.dispatch(smallCallTask);
225      scheduler.dispatch(smallCallTask);
226      scheduler.dispatch(smallCallTask);
227      scheduler.dispatch(hugeCallTask);
228      scheduler.dispatch(smallCallTask);
229      scheduler.dispatch(largeCallTask);
230      scheduler.dispatch(smallCallTask);
231      scheduler.dispatch(smallCallTask);
232
233      while (work.size() < 8) {
234        Thread.sleep(100);
235      }
236
237      int seqSum = 0;
238      int totalTime = 0;
239      for (int i = 0; i < work.size(); ++i) {
240        LOG.debug("Request i=" + i + " value=" + work.get(i));
241        seqSum += work.get(i);
242        totalTime += seqSum;
243      }
244      LOG.debug("Total Time: " + totalTime);
245
246      // -> [small small small huge small large small small]
247      // -> NO REORDER   [10 10 10 100 10 50 10 10] -> 930 (FIFO Queue)
248      // -> WITH REORDER [10 10 10 10 10 10 50 100] -> 530 (Deadline Queue)
249      if (queueType.equals(RpcExecutor.CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
250        assertEquals(530, totalTime);
251      } else if (queueType.equals(RpcExecutor.CALL_QUEUE_TYPE_FIFO_CONF_VALUE)) {
252        assertEquals(930, totalTime);
253      }
254    } finally {
255      scheduler.stop();
256    }
257  }

276  @Test
277  public void testScanQueues() throws Exception {
278    Configuration schedConf = HBaseConfiguration.create();
279    schedConf.setFloat(RpcExecutor.CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 1.0f);
280    schedConf.setFloat(RWQueueRpcExecutor.CALL_QUEUE_READ_SHARE_CONF_KEY, 0.7f);
281    schedConf.setFloat(RWQueueRpcExecutor.CALL_QUEUE_SCAN_SHARE_CONF_KEY, 0.5f);
282
283    PriorityFunction priority = mock(PriorityFunction.class);
284    when(priority.getPriority(any(RPCProtos.RequestHeader.class), any(Message.class),
285      any(User.class))).thenReturn(HConstants.NORMAL_QOS);
286
287    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 3, 1, 1, priority,
288                                                    HConstants.QOS_THRESHOLD);
289    try {
290      scheduler.start();
291
292      CallRunner putCallTask = mock(CallRunner.class);
293      RpcServer.Call putCall = mock(RpcServer.Call.class);
294      putCall.param = RequestConverter.buildMutateRequest(
295          Bytes.toBytes("abc"), new Put(Bytes.toBytes("row")));
296      RequestHeader putHead = RequestHeader.newBuilder().setMethodName("mutate").build();
297      when(putCallTask.getRpcCall()).thenReturn(putCall);
298      when(putCall.getHeader()).thenReturn(putHead);
299      when(putCall.getParam()).thenReturn(putCall.param);
300
301      CallRunner getCallTask = mock(CallRunner.class);
302      RpcServer.Call getCall = mock(RpcServer.Call.class);
303      RequestHeader getHead = RequestHeader.newBuilder().setMethodName("get").build();
304      when(getCallTask.getRpcCall()).thenReturn(getCall);
305      when(getCall.getHeader()).thenReturn(getHead);
306
307      CallRunner scanCallTask = mock(CallRunner.class);
308      RpcServer.Call scanCall = mock(RpcServer.Call.class);
309      scanCall.param = ScanRequest.newBuilder().setScannerId(1).build();
310      RequestHeader scanHead = RequestHeader.newBuilder().setMethodName("scan").build();
311      when(scanCallTask.getRpcCall()).thenReturn(scanCall);
312      when(scanCall.getHeader()).thenReturn(scanHead);
313      when(scanCall.getParam()).thenReturn(scanCall.param);
314
315      ArrayList<Integer> work = new ArrayList<>();
316      doAnswerTaskExecution(putCallTask, work, 1, 1000);
317      doAnswerTaskExecution(getCallTask, work, 2, 1000);
318      doAnswerTaskExecution(scanCallTask, work, 3, 1000);
319
320      // There are 3 queues: [puts], [gets], [scans]
321      // so the calls will be interleaved
322      scheduler.dispatch(putCallTask);
323      scheduler.dispatch(putCallTask);
324      scheduler.dispatch(putCallTask);
325      scheduler.dispatch(getCallTask);
326      scheduler.dispatch(getCallTask);
327      scheduler.dispatch(getCallTask);
328      scheduler.dispatch(scanCallTask);
329      scheduler.dispatch(scanCallTask);
330      scheduler.dispatch(scanCallTask);
331
332      while (work.size() < 6) {
333        Thread.sleep(100);
334      }
335
336      for (int i = 0; i < work.size() - 2; i += 3) {
337        assertNotEquals(work.get(i + 0), work.get(i + 1));
338        assertNotEquals(work.get(i + 0), work.get(i + 2));
339        assertNotEquals(work.get(i + 1), work.get(i + 2));
340      }
341    } finally {
342      scheduler.stop();
343    }
344  }

346  private void doAnswerTaskExecution(final CallRunner callTask,
347      final ArrayList<Integer> results, final int value, final int sleepInterval) {
348    callTask.setStatus(new MonitoredRPCHandlerImpl());
349    doAnswer(new Answer<Object>() {
350      @Override
351      public Object answer(InvocationOnMock invocation) {
352        synchronized (results) {
353          results.add(value);
354        }
355        Threads.sleepWithoutInterrupt(sleepInterval);
356        return null;
357      }
358    }).when(callTask).run();
359  }

406  private static final class CoDelEnvironmentEdge implements EnvironmentEdge {
407
408    private final BlockingQueue<Long> timeQ = new LinkedBlockingQueue<>();
409
410    private long offset;
411
412    private final Set<String> threadNamePrefixs = new HashSet<>();
413
414    @Override
415    public long currentTime() {
416      for (String threadNamePrefix : threadNamePrefixs) {
417        String threadName = Thread.currentThread().getName();
418        if (threadName.startsWith(threadNamePrefix)) {
419          return timeQ.poll().longValue() + offset; 
420        }
421      }
422      return System.currentTimeMillis();
423    }
424  }

503    }

509    final RpcServer.Call putCall = mock(RpcServer.Call.class);
510
511    putCall.timestamp = timestamp;
512    putCall.param = RequestConverter.buildMutateRequest(
513        Bytes.toBytes("abc"), new Put(Bytes.toBytes("row")));
514
515    RPCProtos.RequestHeader putHead = RPCProtos.RequestHeader.newBuilder()
516                                                             .setMethodName("mutate")
517                                                             .build();
518    when(putCall.getSize()).thenReturn(9L);
519    when(putCall.getHeader()).thenReturn(putHead);
520    when(putCall.getReceiveTime()).thenReturn(putCall.timestamp);
521    when(putCall.getParam()).thenReturn(putCall.param);
522
523    CallRunner cr = new CallRunner(null, putCall) {
524      public void run() {
525        if (sleepTime <= 0) return;
526        try {
527          LOG.warn("Sleeping for " + sleepTime);
528          Thread.sleep(sleepTime);
529          LOG.warn("Done Sleeping for " + sleepTime);
530        } catch (InterruptedException e) {
531        }
532      }
533      public RpcCall getRpcCall() {
534        return putCall;
535      }
536
537      public void drop() {}
538    };
539
540    return cr;
