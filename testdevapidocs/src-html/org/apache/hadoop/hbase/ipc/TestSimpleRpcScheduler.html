111  @Test
112  public void testHandlerIsolation() throws IOException, InterruptedException {
113    CallRunner generalTask = createMockTask();
114    CallRunner priorityTask = createMockTask();
115    CallRunner replicationTask = createMockTask();
116    List<CallRunner> tasks = ImmutableList.of(
117        generalTask,
118        priorityTask,
119        replicationTask);
120    Map<CallRunner, Integer> qos = ImmutableMap.of(
121        generalTask, 0,
122        priorityTask, HConstants.HIGH_QOS + 1,
123        replicationTask, HConstants.REPLICATION_QOS);
124    PriorityFunction qosFunction = mock(PriorityFunction.class);
125    final Map<CallRunner, Thread> handlerThreads = Maps.newHashMap();
126    final CountDownLatch countDownLatch = new CountDownLatch(tasks.size());
127    Answer<Void> answerToRun = new Answer<Void>() {
128      @Override
129      public Void answer(InvocationOnMock invocationOnMock) throws Throwable {
130        synchronized (handlerThreads) {
131          handlerThreads.put(
132              (CallRunner) invocationOnMock.getMock(),
133              Thread.currentThread());
134        }
135        countDownLatch.countDown();
136        return null;
137      }
138    };
139    for (CallRunner task : tasks) {
140      task.setStatus(new MonitoredRPCHandlerImpl());
141      doAnswer(answerToRun).when(task).run();
142    }
143
144    RpcScheduler scheduler = new SimpleRpcScheduler(
145        conf, 1, 1 ,1, qosFunction, HConstants.HIGH_QOS);
146    scheduler.init(CONTEXT);
147    scheduler.start();
148    for (CallRunner task : tasks) {
149      when(qosFunction.getPriority((RPCProtos.RequestHeader) anyObject(),
150        (Message) anyObject(), (User) anyObject()))
151          .thenReturn(qos.get(task));
152      scheduler.dispatch(task);
153    }
154    for (CallRunner task : tasks) {
155      verify(task, timeout(1000)).run();
156    }
157    scheduler.stop();
158
159    // Tests that these requests are handled by three distinct threads.
160    countDownLatch.await();
161    assertEquals(3, ImmutableSet.copyOf(handlerThreads.values()).size());
162  }

177  private void testRpcScheduler(final String queueType) throws Exception {
178    Configuration schedConf = HBaseConfiguration.create();
179    schedConf.set(SimpleRpcScheduler.CALL_QUEUE_TYPE_CONF_KEY, queueType);
180
181    PriorityFunction priority = mock(PriorityFunction.class);
182    when(priority.getPriority(any(RequestHeader.class),
183      any(Message.class), any(User.class)))
184      .thenReturn(HConstants.NORMAL_QOS);
185
186    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 1, 1, 1, priority,
187                                                    HConstants.QOS_THRESHOLD);
188    try {
189      scheduler.start();
190
191      CallRunner smallCallTask = mock(CallRunner.class);
192      RpcServer.Call smallCall = mock(RpcServer.Call.class);
193      RequestHeader smallHead = RequestHeader.newBuilder().setCallId(1).build();
194      when(smallCallTask.getCall()).thenReturn(smallCall);
195      when(smallCall.getHeader()).thenReturn(smallHead);
196
197      CallRunner largeCallTask = mock(CallRunner.class);
198      RpcServer.Call largeCall = mock(RpcServer.Call.class);
199      RequestHeader largeHead = RequestHeader.newBuilder().setCallId(50).build();
200      when(largeCallTask.getCall()).thenReturn(largeCall);
201      when(largeCall.getHeader()).thenReturn(largeHead);
202
203      CallRunner hugeCallTask = mock(CallRunner.class);
204      RpcServer.Call hugeCall = mock(RpcServer.Call.class);
205      RequestHeader hugeHead = RequestHeader.newBuilder().setCallId(100).build();
206      when(hugeCallTask.getCall()).thenReturn(hugeCall);
207      when(hugeCall.getHeader()).thenReturn(hugeHead);
208
209      when(priority.getDeadline(eq(smallHead), any(Message.class))).thenReturn(0L);
210      when(priority.getDeadline(eq(largeHead), any(Message.class))).thenReturn(50L);
211      when(priority.getDeadline(eq(hugeHead), any(Message.class))).thenReturn(100L);
212
213      final ArrayList<Integer> work = new ArrayList<Integer>();
214      doAnswerTaskExecution(smallCallTask, work, 10, 250);
215      doAnswerTaskExecution(largeCallTask, work, 50, 250);
216      doAnswerTaskExecution(hugeCallTask, work, 100, 250);
217
218      scheduler.dispatch(smallCallTask);
219      scheduler.dispatch(smallCallTask);
220      scheduler.dispatch(smallCallTask);
221      scheduler.dispatch(hugeCallTask);
222      scheduler.dispatch(smallCallTask);
223      scheduler.dispatch(largeCallTask);
224      scheduler.dispatch(smallCallTask);
225      scheduler.dispatch(smallCallTask);
226
227      while (work.size() < 8) {
228        Thread.sleep(100);
229      }
230
231      int seqSum = 0;
232      int totalTime = 0;
233      for (int i = 0; i < work.size(); ++i) {
234        LOG.debug("Request i=" + i + " value=" + work.get(i));
235        seqSum += work.get(i);
236        totalTime += seqSum;
237      }
238      LOG.debug("Total Time: " + totalTime);
239
240      // -> [small small small huge small large small small]
241      // -> NO REORDER   [10 10 10 100 10 50 10 10] -> 930 (FIFO Queue)
242      // -> WITH REORDER [10 10 10 10 10 10 50 100] -> 530 (Deadline Queue)
243      if (queueType.equals(SimpleRpcScheduler.CALL_QUEUE_TYPE_DEADLINE_CONF_VALUE)) {
244        assertEquals(530, totalTime);
245      } else if (queueType.equals(SimpleRpcScheduler.CALL_QUEUE_TYPE_FIFO_CONF_VALUE)) {
246        assertEquals(930, totalTime);
247      }
248    } finally {
249      scheduler.stop();
250    }
251  }

269  @Test
270  public void testScanQueues() throws Exception {
271    Configuration schedConf = HBaseConfiguration.create();
272    schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_HANDLER_FACTOR_CONF_KEY, 1.0f);
273    schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_READ_SHARE_CONF_KEY, 0.7f);
274    schedConf.setFloat(SimpleRpcScheduler.CALL_QUEUE_SCAN_SHARE_CONF_KEY, 0.5f);
275
276    PriorityFunction priority = mock(PriorityFunction.class);
277    when(priority.getPriority(any(RequestHeader.class), any(Message.class),
278      any(User.class))).thenReturn(HConstants.NORMAL_QOS);
279
280    RpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 3, 1, 1, priority,
281                                                    HConstants.QOS_THRESHOLD);
282    try {
283      scheduler.start();
284
285      CallRunner putCallTask = mock(CallRunner.class);
286      RpcServer.Call putCall = mock(RpcServer.Call.class);
287      putCall.param = RequestConverter.buildMutateRequest(
288          Bytes.toBytes("abc"), new Put(Bytes.toBytes("row")));
289      RequestHeader putHead = RequestHeader.newBuilder().setMethodName("mutate").build();
290      when(putCallTask.getCall()).thenReturn(putCall);
291      when(putCall.getHeader()).thenReturn(putHead);
292
293      CallRunner getCallTask = mock(CallRunner.class);
294      RpcServer.Call getCall = mock(RpcServer.Call.class);
295      RequestHeader getHead = RequestHeader.newBuilder().setMethodName("get").build();
296      when(getCallTask.getCall()).thenReturn(getCall);
297      when(getCall.getHeader()).thenReturn(getHead);
298
299      CallRunner scanCallTask = mock(CallRunner.class);
300      RpcServer.Call scanCall = mock(RpcServer.Call.class);
301      scanCall.param = ScanRequest.newBuilder().setScannerId(1).build();
302      RequestHeader scanHead = RequestHeader.newBuilder().setMethodName("scan").build();
303      when(scanCallTask.getCall()).thenReturn(scanCall);
304      when(scanCall.getHeader()).thenReturn(scanHead);
305
306      ArrayList<Integer> work = new ArrayList<Integer>();
307      doAnswerTaskExecution(putCallTask, work, 1, 1000);
308      doAnswerTaskExecution(getCallTask, work, 2, 1000);
309      doAnswerTaskExecution(scanCallTask, work, 3, 1000);
310
311      // There are 3 queues: [puts], [gets], [scans]
312      // so the calls will be interleaved
313      scheduler.dispatch(putCallTask);
314      scheduler.dispatch(putCallTask);
315      scheduler.dispatch(putCallTask);
316      scheduler.dispatch(getCallTask);
317      scheduler.dispatch(getCallTask);
318      scheduler.dispatch(getCallTask);
319      scheduler.dispatch(scanCallTask);
320      scheduler.dispatch(scanCallTask);
321      scheduler.dispatch(scanCallTask);
322
323      while (work.size() < 6) {
324        Thread.sleep(100);
325      }
326
327      for (int i = 0; i < work.size() - 2; i += 3) {
328        assertNotEquals(work.get(i + 0), work.get(i + 1));
329        assertNotEquals(work.get(i + 0), work.get(i + 2));
330        assertNotEquals(work.get(i + 1), work.get(i + 2));
331      }
332    } finally {
333      scheduler.stop();
334    }
335  }

337  private void doAnswerTaskExecution(final CallRunner callTask,
338      final ArrayList<Integer> results, final int value, final int sleepInterval) {
339    callTask.setStatus(new MonitoredRPCHandlerImpl());
340    doAnswer(new Answer<Object>() {
341      @Override
342      public Object answer(InvocationOnMock invocation) {
343        synchronized (results) {
344          results.add(value);
345        }
346        Threads.sleepWithoutInterrupt(sleepInterval);
347        return null;
348      }
349    }).when(callTask).run();
350  }

396  private static final class CoDelEnvironmentEdge implements EnvironmentEdge {
397
398    private final BlockingQueue<Long> timeQ = new LinkedBlockingQueue<>();
399
400    private long offset;
401
402    private final Set<String> threadNamePrefixs = new HashSet<>();
403
404    @Override
405    public long currentTime() {
406      for (String threadNamePrefix : threadNamePrefixs) {
407        if (Thread.currentThread().getName().startsWith(threadNamePrefix)) {
408          return timeQ.poll().longValue() + offset;
409        }
410      }
411      return System.currentTimeMillis();
412    }
413  }

415  @Test
416  public void testCoDelScheduling() throws Exception {
417    CoDelEnvironmentEdge envEdge = new CoDelEnvironmentEdge();
418    envEdge.threadNamePrefixs.add("RW.default");
419    envEdge.threadNamePrefixs.add("B.default");
420    Configuration schedConf = HBaseConfiguration.create();
421
422    schedConf.set(SimpleRpcScheduler.CALL_QUEUE_TYPE_CONF_KEY,
423      SimpleRpcScheduler.CALL_QUEUE_TYPE_CODEL_CONF_VALUE);
424
425    PriorityFunction priority = mock(PriorityFunction.class);
426    when(priority.getPriority(any(RPCProtos.RequestHeader.class), any(Message.class),
427      any(User.class))).thenReturn(HConstants.NORMAL_QOS);
428    SimpleRpcScheduler scheduler = new SimpleRpcScheduler(schedConf, 1, 1, 1, priority,
429      HConstants.QOS_THRESHOLD);
430    try {
431      scheduler.start();
432      EnvironmentEdgeManager.injectEdge(envEdge);
433      envEdge.offset = 5;
434      // calls faster than min delay
435      for (int i = 0; i < 100; i++) {
436        long time = System.currentTimeMillis();
437        envEdge.timeQ.put(time);
438        CallRunner cr = getMockedCallRunner(time);
439        Thread.sleep(5);
440        scheduler.dispatch(cr);
441      }
442      // make sure fast calls are handled
443      waitUntilQueueEmpty(scheduler);
444      Thread.sleep(100);
445      assertEquals("None of these calls should have been discarded", 0,
446        scheduler.getNumGeneralCallsDropped());
447
448      envEdge.offset = 6;
449      // calls slower than min delay, but not individually slow enough to be dropped
450      for (int i = 0; i < 20; i++) {
451        long time = System.currentTimeMillis();
452        envEdge.timeQ.put(time);
453        CallRunner cr = getMockedCallRunner(time);
454        Thread.sleep(6);
455        scheduler.dispatch(cr);
456      }
457
458      // make sure somewhat slow calls are handled
459      waitUntilQueueEmpty(scheduler);
460      Thread.sleep(100);
461      assertEquals("None of these calls should have been discarded", 0,
462        scheduler.getNumGeneralCallsDropped());
463
464      envEdge.offset = 12;
465      // now slow calls and the ones to be dropped
466      for (int i = 0; i < 20; i++) {
467        long time = System.currentTimeMillis();
468        envEdge.timeQ.put(time);
469        CallRunner cr = getMockedCallRunner(time);
470        Thread.sleep(12);
471        scheduler.dispatch(cr);
472      }
473
474      // make sure somewhat slow calls are handled
475      waitUntilQueueEmpty(scheduler);
476      Thread.sleep(100);
477      assertTrue("There should have been at least 12 calls dropped",
478        scheduler.getNumGeneralCallsDropped() > 12);
479    } finally {
480      scheduler.stop();
481    }
482  }
