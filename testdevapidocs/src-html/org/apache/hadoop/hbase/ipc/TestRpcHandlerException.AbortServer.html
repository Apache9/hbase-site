070  // We are using the test TestRpcServiceProtos generated classes and Service because they are
071  // available and basic with methods like 'echo', and ping. Below we make a blocking service
072  // by passing in implementation of blocking interface. We use this service in all tests that
073  // follow.
074  private static final BlockingService SERVICE =
075      TestRpcServiceProtos.TestProtobufRpcProto
076      .newReflectiveBlockingService(new TestRpcServiceProtos
077      .TestProtobufRpcProto.BlockingInterface() {
078
079        @Override
080        public EmptyResponseProto ping(RpcController controller, EmptyRequestProto request)
081            throws ServiceException {
082          return null;
083        }
084
085        @Override
086        public EmptyResponseProto error(RpcController controller, EmptyRequestProto request)
087            throws ServiceException {
088          return null;
089        }
090
091        @Override
092        public EchoResponseProto echo(RpcController controller, EchoRequestProto request)
093            throws Error, RuntimeException {
094          if (controller instanceof PayloadCarryingRpcController) {
095            PayloadCarryingRpcController pcrc = (PayloadCarryingRpcController) controller;
096            // If cells, scan them to check we are able to iterate what we were given and since
097            // this is
098            // an echo, just put them back on the controller creating a new block. Tests our
099            // block
100            // building.
101            CellScanner cellScanner = pcrc.cellScanner();
102            List<Cell> list = null;
103            if (cellScanner != null) {
104    list = new ArrayList<Cell>();
105    try {
106      while (cellScanner.advance()) {
107        list.add(cellScanner.current());
108        throw new StackOverflowError();
109      }
110    } catch (StackOverflowError e) {
111      throw e;
112    } catch (IOException e) {
113      throw new RuntimeException(e);
114    }
115            }
116            cellScanner = CellUtil.createCellScanner(list);
117            ((PayloadCarryingRpcController) controller).setCellScanner(cellScanner);
118          }
119          return EchoResponseProto.newBuilder().setMessage(request.getMessage()).build();
120        }
121      });

147  /** Tests that the rpc scheduler is called when requests arrive.
148   *  When Rpc handler thread dies, the client will hang and the test will fail.
149   *  The test is meant to be a unit test to test the behavior.
150   *
151   * */
152  private class AbortServer implements Abortable {
153    private boolean aborted = false;
154
155    @Override
156    public void abort(String why, Throwable e) {
157      aborted = true;
158    }
159
160    @Override
161    public boolean isAborted() {
162      return aborted;
163    }
164  }
165
166  /* This is a unit test to make sure to abort region server when the number of Rpc handler thread
167   * caught errors exceeds the threshold. Client will hang when RS aborts.
168   */
169  @Ignore
170  @Test
171  public void testRpcScheduler() throws IOException, InterruptedException {
172    PriorityFunction qosFunction = mock(PriorityFunction.class);
173    Abortable abortable = new AbortServer();
174    RpcScheduler scheduler = new SimpleRpcScheduler(CONF, 2, 0, 0, qosFunction, abortable, 0);
175    RpcServer rpcServer = new TestRpcServer(scheduler);
176    RpcClientImpl client = new RpcClientImpl(CONF, HConstants.CLUSTER_ID_DEFAULT);
177    try {
178      rpcServer.start();
179      MethodDescriptor md = SERVICE.getDescriptorForType().findMethodByName("echo");
180      EchoRequestProto param = EchoRequestProto.newBuilder().setMessage("hello").build();
181      PayloadCarryingRpcController controller =
182          new PayloadCarryingRpcController(CellUtil.createCellScanner(ImmutableList.of(CELL)));
183      InetSocketAddress address = rpcServer.getListenerAddress();
184      if (address == null) {
185        throw new IOException("Listener channel is closed");
186      }
187      client.call(controller, md, param, md.getOutputType().toProto(), User.getCurrent(),
188          address, new MetricsConnection.CallStats());
189    } catch (Throwable e) {
190      assert(abortable.isAborted() == true);
191    } finally {
192      rpcServer.stop();
193    }
194  }
