131  /**
132   * Tests the bulk assignment used during cluster startup.
133   *
134   * Round-robin. Should yield a balanced cluster so same invariant as the load
135   * balancer holds, all servers holding either floor(avg) or ceiling(avg).
136   *
137   * @throws Exception
138   */
139  @Test (timeout=180000)
140  public void testBulkAssignment() throws Exception {
141    List<ServerName> tmp = getListOfServerNames(randomServers(5, 0));
142    List<HRegionInfo> hris = randomRegions(20);
143    hris.add(HRegionInfo.FIRST_META_REGIONINFO);
144    tmp.add(master);
145    Map<ServerName, List<HRegionInfo>> plans = loadBalancer.roundRobinAssignment(hris, tmp);
146    assertTrue(plans.get(master).contains(HRegionInfo.FIRST_META_REGIONINFO));
147    assertEquals(1, plans.get(master).size());
148    int totalRegion = 0;
149    for (List<HRegionInfo> regions: plans.values()) {
150      totalRegion += regions.size();
151    }
152    assertEquals(hris.size(), totalRegion);
153    for (int[] mock : regionsAndServersMocks) {
154      LOG.debug("testBulkAssignment with " + mock[0] + " regions and " + mock[1] + " servers");
155      List<HRegionInfo> regions = randomRegions(mock[0]);
156      List<ServerAndLoad> servers = randomServers(mock[1], 0);
157      List<ServerName> list = getListOfServerNames(servers);
158      Map<ServerName, List<HRegionInfo>> assignments =
159          loadBalancer.roundRobinAssignment(regions, list);
160      float average = (float) regions.size() / servers.size();
161      int min = (int) Math.floor(average);
162      int max = (int) Math.ceil(average);
163      if (assignments != null && !assignments.isEmpty()) {
164        for (List<HRegionInfo> regionList : assignments.values()) {
165          assertTrue(regionList.size() == min || regionList.size() == max);
166        }
167      }
168      returnRegions(regions);
169      returnServers(list);
170    }
171  }

174   * Test the cluster startup bulk assignment which attempts to retain
175   * assignment info.
176   * @throws Exception
177   */
178  @Test (timeout=180000)
179  public void testRetainAssignment() throws Exception {
180    // Test simple case where all same servers are there
181    List<ServerAndLoad> servers = randomServers(10, 10);
182    List<HRegionInfo> regions = randomRegions(100);
183    Map<HRegionInfo, ServerName> existing = new TreeMap<HRegionInfo, ServerName>();
184    for (int i = 0; i < regions.size(); i++) {
185      ServerName sn = servers.get(i % servers.size()).getServerName();
186      // The old server would have had same host and port, but different
187      // start code!
188      ServerName snWithOldStartCode =
189          ServerName.valueOf(sn.getHostname(), sn.getPort(), sn.getStartcode() - 10);
190      existing.put(regions.get(i), snWithOldStartCode);
191    }
192    List<ServerName> listOfServerNames = getListOfServerNames(servers);
193    Map<ServerName, List<HRegionInfo>> assignment =
194        loadBalancer.retainAssignment(existing, listOfServerNames);
195    assertRetainedAssignment(existing, listOfServerNames, assignment);
196
197    // Include two new servers that were not there before
198    List<ServerAndLoad> servers2 = new ArrayList<ServerAndLoad>(servers);
199    servers2.add(randomServer(10));
200    servers2.add(randomServer(10));
201    listOfServerNames = getListOfServerNames(servers2);
202    assignment = loadBalancer.retainAssignment(existing, listOfServerNames);
203    assertRetainedAssignment(existing, listOfServerNames, assignment);
204
205    // Remove two of the servers that were previously there
206    List<ServerAndLoad> servers3 = new ArrayList<ServerAndLoad>(servers);
207    servers3.remove(0);
208    servers3.remove(0);
209    listOfServerNames = getListOfServerNames(servers3);
210    assignment = loadBalancer.retainAssignment(existing, listOfServerNames);
211    assertRetainedAssignment(existing, listOfServerNames, assignment);
212  }

214  @Test (timeout=180000)
215  public void testRegionAvailability() throws Exception {
216    // Create a cluster with a few servers, assign them to specific racks
217    // then assign some regions. The tests should check whether moving a
218    // replica from one node to a specific other node or rack lowers the
219    // availability of the region or not
220
221    List<HRegionInfo> list0 = new ArrayList<HRegionInfo>();
222    List<HRegionInfo> list1 = new ArrayList<HRegionInfo>();
223    List<HRegionInfo> list2 = new ArrayList<HRegionInfo>();
224    // create a region (region1)
225    HRegionInfo hri1 = new HRegionInfo(
226        TableName.valueOf("table"), "key1".getBytes(), "key2".getBytes(),
227        false, 100);
228    // create a replica of the region (replica_of_region1)
229    HRegionInfo hri2 = RegionReplicaUtil.getRegionInfoForReplica(hri1, 1);
230    // create a second region (region2)
231    HRegionInfo hri3 = new HRegionInfo(
232        TableName.valueOf("table"), "key2".getBytes(), "key3".getBytes(),
233        false, 101);
234    list0.add(hri1); //only region1
235    list1.add(hri2); //only replica_of_region1
236    list2.add(hri3); //only region2
237    Map<ServerName, List<HRegionInfo>> clusterState =
238        new LinkedHashMap<ServerName, List<HRegionInfo>>();
239    clusterState.put(servers[0], list0); //servers[0] hosts region1
240    clusterState.put(servers[1], list1); //servers[1] hosts replica_of_region1
241    clusterState.put(servers[2], list2); //servers[2] hosts region2
242    // create a cluster with the above clusterState. The way in which the
243    // cluster is created (constructor code) would make sure the indices of
244    // the servers are in the order in which it is inserted in the clusterState
245    // map (linkedhashmap is important). A similar thing applies to the region lists
246    Cluster cluster = new Cluster(clusterState, null, null, rackManager);
247    // check whether a move of region1 from servers[0] to servers[1] would lower
248    // the availability of region1
249    assertTrue(cluster.wouldLowerAvailability(hri1, servers[1]));
250    // check whether a move of region1 from servers[0] to servers[2] would lower
251    // the availability of region1
252    assertTrue(!cluster.wouldLowerAvailability(hri1, servers[2]));
253    // check whether a move of replica_of_region1 from servers[0] to servers[2] would lower
254    // the availability of replica_of_region1
255    assertTrue(!cluster.wouldLowerAvailability(hri2, servers[2]));
256    // check whether a move of region2 from servers[0] to servers[1] would lower
257    // the availability of region2
258    assertTrue(!cluster.wouldLowerAvailability(hri3, servers[1]));
259
260    // now lets have servers[1] host replica_of_region2
261    list1.add(RegionReplicaUtil.getRegionInfoForReplica(hri3, 1));
262    // create a new clusterState with the above change
263    cluster = new Cluster(clusterState, null, null, rackManager);
264    // now check whether a move of a replica from servers[0] to servers[1] would lower
265    // the availability of region2
266    assertTrue(cluster.wouldLowerAvailability(hri3, servers[1]));
267
268    // start over again
269    clusterState.clear();
270    clusterState.put(servers[0], list0); //servers[0], rack1 hosts region1
271    clusterState.put(servers[5], list1); //servers[5], rack2 hosts replica_of_region1 and replica_of_region2
272    clusterState.put(servers[6], list2); //servers[6], rack2 hosts region2
273    clusterState.put(servers[10], new ArrayList<HRegionInfo>()); //servers[10], rack3 hosts no region
274    // create a cluster with the above clusterState
275    cluster = new Cluster(clusterState, null, null, rackManager);
276    // check whether a move of region1 from servers[0],rack1 to servers[6],rack2 would
277    // lower the availability
278
279    assertTrue(cluster.wouldLowerAvailability(hri1, servers[0]));
280
281    // now create a cluster without the rack manager
282    cluster = new Cluster(clusterState, null, null, null);
283    // now repeat check whether a move of region1 from servers[0] to servers[6] would
284    // lower the availability
285    assertTrue(!cluster.wouldLowerAvailability(hri1, servers[6]));
286  }

289  public void testRegionAvailabilityWithRegionMoves() throws Exception {
290    List<HRegionInfo> list0 = new ArrayList<HRegionInfo>();
291    List<HRegionInfo> list1 = new ArrayList<HRegionInfo>();
292    List<HRegionInfo> list2 = new ArrayList<HRegionInfo>();
293    // create a region (region1)
294    HRegionInfo hri1 = new HRegionInfo(
295        TableName.valueOf("table"), "key1".getBytes(), "key2".getBytes(),
296        false, 100);
297    // create a replica of the region (replica_of_region1)
298    HRegionInfo hri2 = RegionReplicaUtil.getRegionInfoForReplica(hri1, 1);
299    // create a second region (region2)
300    HRegionInfo hri3 = new HRegionInfo(
301        TableName.valueOf("table"), "key2".getBytes(), "key3".getBytes(),
302        false, 101);
303    list0.add(hri1); //only region1
304    list1.add(hri2); //only replica_of_region1
305    list2.add(hri3); //only region2
306    Map<ServerName, List<HRegionInfo>> clusterState =
307        new LinkedHashMap<ServerName, List<HRegionInfo>>();
308    clusterState.put(servers[0], list0); //servers[0] hosts region1
309    clusterState.put(servers[1], list1); //servers[1] hosts replica_of_region1
310    clusterState.put(servers[2], list2); //servers[2] hosts region2
311    // create a cluster with the above clusterState. The way in which the
312    // cluster is created (constructor code) would make sure the indices of
313    // the servers are in the order in which it is inserted in the clusterState
314    // map (linkedhashmap is important).
315    Cluster cluster = new Cluster(clusterState, null, null, rackManager);
316    // check whether moving region1 from servers[1] to servers[2] would lower availability
317    assertTrue(!cluster.wouldLowerAvailability(hri1, servers[2]));
318
319    // now move region1 from servers[0] to servers[2]
320    cluster.doAction(new MoveRegionAction(0, 0, 2));
321    // now repeat check whether moving region1 from servers[1] to servers[2]
322    // would lower availability
323    assertTrue(cluster.wouldLowerAvailability(hri1, servers[2]));
324
325    // start over again
326    clusterState.clear();
327    List<HRegionInfo> list3 = new ArrayList<HRegionInfo>();
328    HRegionInfo hri4 = RegionReplicaUtil.getRegionInfoForReplica(hri3, 1);
329    list3.add(hri4);
330    clusterState.put(servers[0], list0); //servers[0], rack1 hosts region1
331    clusterState.put(servers[5], list1); //servers[5], rack2 hosts replica_of_region1
332    clusterState.put(servers[6], list2); //servers[6], rack2 hosts region2
333    clusterState.put(servers[12], list3); //servers[12], rack3 hosts replica_of_region2
334    // create a cluster with the above clusterState
335    cluster = new Cluster(clusterState, null, null, rackManager);
336    // check whether a move of replica_of_region2 from servers[12],rack3 to servers[0],rack1 would
337    // lower the availability
338    assertTrue(!cluster.wouldLowerAvailability(hri4, servers[0]));
339    // now move region2 from servers[6],rack2 to servers[0],rack1
340    cluster.doAction(new MoveRegionAction(2, 2, 0));
341    // now repeat check if replica_of_region2 from servers[12],rack3 to servers[0],rack1 would
342    // lower the availability
343    assertTrue(cluster.wouldLowerAvailability(hri3, servers[0]));
344  }

354  /**
355   * Asserts a valid retained assignment plan.
356   * <p>
357   * Must meet the following conditions:
358   * <ul>
359   * <li>Every input region has an assignment, and to an online server
360   * <li>If a region had an existing assignment to a server with the same
361   * address a a currently online server, it will be assigned to it
362   * </ul>
363   * @param existing
364   * @param servers
365   * @param assignment
366   */
367  private void assertRetainedAssignment(Map<HRegionInfo, ServerName> existing,
368      List<ServerName> servers, Map<ServerName, List<HRegionInfo>> assignment) {
369    // Verify condition 1, every region assigned, and to online server
370    Set<ServerName> onlineServerSet = new TreeSet<ServerName>(servers);
371    Set<HRegionInfo> assignedRegions = new TreeSet<HRegionInfo>();
372    for (Map.Entry<ServerName, List<HRegionInfo>> a : assignment.entrySet()) {
373      assertTrue("Region assigned to server that was not listed as online",
374        onlineServerSet.contains(a.getKey()));
375      for (HRegionInfo r : a.getValue())
376        assignedRegions.add(r);
377    }
378    assertEquals(existing.size(), assignedRegions.size());
379
380    // Verify condition 2, if server had existing assignment, must have same
381    Set<String> onlineHostNames = new TreeSet<String>();
382    for (ServerName s : servers) {
383      onlineHostNames.add(s.getHostname());
384    }
385
386    for (Map.Entry<ServerName, List<HRegionInfo>> a : assignment.entrySet()) {
387      ServerName assignedTo = a.getKey();
388      for (HRegionInfo r : a.getValue()) {
389        ServerName address = existing.get(r);
390        if (address != null && onlineHostNames.contains(address.getHostname())) {
391          // this region was prevously assigned somewhere, and that
392          // host is still around, then it should be re-assigned on the
393          // same host
394          assertEquals(address.getHostname(), assignedTo.getHostname());
395        }
396      }
397    }
398  }

402    // tests whether the BaseLoadBalancer.Cluster can be constructed with servers

418    assignRegions(regions, oldServers, clusterState);

421    BaseLoadBalancer.Cluster cluster = new Cluster(clusterState, null, null, null);
422    assertEquals(101 + 9, cluster.numRegions);
423    assertEquals(10, cluster.numServers); // only 10 servers because they share the same host + port
424  }

426  private void assignRegions(List<HRegionInfo> regions, List<ServerName> servers,
427      Map<ServerName, List<HRegionInfo>> clusterState) {
428    for (int i = 0; i < regions.size(); i++) {
429      ServerName sn = servers.get(i % servers.size());
430      List<HRegionInfo> regionsOfServer = clusterState.get(sn);
431      if (regionsOfServer == null) {
432        regionsOfServer = new ArrayList<HRegionInfo>(10);
433        clusterState.put(sn, regionsOfServer);
434      }
435
436      regionsOfServer.add(regions.get(i));
437    }
438  }

443    List<ServerName> servers = getListOfServerNames(randomServers(10, 10));

451    // block locality: region:0   => {server:0}
452    //                 region:1   => {server:0, server:1}

460    when(locationFinder.getTopBlockLocations(regions.get(43))).thenReturn(
461        Lists.newArrayList(ServerName.valueOf("foo", 0, 0))); // this server does not exists in clusterStatus
462
463    BaseLoadBalancer.Cluster cluster = new Cluster(clusterState, null, locationFinder, null);
464
465    int r0 = ArrayUtils.indexOf(cluster.regions, regions.get(0)); // this is ok, it is just a test
466    int r1 = ArrayUtils.indexOf(cluster.regions, regions.get(1));
467    int r10 = ArrayUtils.indexOf(cluster.regions, regions.get(10));
468    int r42 = ArrayUtils.indexOf(cluster.regions, regions.get(42));
469    int r43 = ArrayUtils.indexOf(cluster.regions, regions.get(43));
470
471    int s0 = cluster.serversToIndex.get(servers.get(0).getHostAndPort());
472    int s1 = cluster.serversToIndex.get(servers.get(1).getHostAndPort());
473    int s4 = cluster.serversToIndex.get(servers.get(4).getHostAndPort());
474    int s5 = cluster.serversToIndex.get(servers.get(5).getHostAndPort());
475    int s9 = cluster.serversToIndex.get(servers.get(9).getHostAndPort());
476
477    // region 0 locations
478    assertEquals(1, cluster.regionLocations[r0].length);
479    assertEquals(s0, cluster.regionLocations[r0][0]);
480
481    // region 1 locations
482    assertEquals(2, cluster.regionLocations[r1].length);
483    assertEquals(s0, cluster.regionLocations[r1][0]);
484    assertEquals(s1, cluster.regionLocations[r1][1]);
485
486    // region 10 locations
487    assertEquals(0, cluster.regionLocations[r10].length);
488
489    // region 42 locations
490    assertEquals(3, cluster.regionLocations[r42].length);
491    assertEquals(s4, cluster.regionLocations[r42][0]);
492    assertEquals(s9, cluster.regionLocations[r42][1]);
493    assertEquals(s5, cluster.regionLocations[r42][2]);
494
495    // region 43 locations
496    assertEquals(1, cluster.regionLocations[r43].length);
497    assertEquals(-1, cluster.regionLocations[r43][0]);
