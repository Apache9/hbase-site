063  @Test
064  public void testKeepRegionLoad() throws Exception {
065
066    ServerName sn = ServerName.valueOf("test:8080", 100);
067    int numClusterStatusToAdd = 20000;
068    for (int i = 0; i < numClusterStatusToAdd; i++) {
069      ServerLoad sl = mock(ServerLoad.class);
070
071      RegionLoad rl = mock(RegionLoad.class);
072      when(rl.getStores()).thenReturn(i);
073
074      Map<byte[], RegionLoad> regionLoadMap =
075          new TreeMap<byte[], RegionLoad>(Bytes.BYTES_COMPARATOR);
076      regionLoadMap.put(Bytes.toBytes(REGION_KEY), rl);
077      when(sl.getRegionsLoad()).thenReturn(regionLoadMap);
078
079      ClusterStatus clusterStatus = mock(ClusterStatus.class);
080      when(clusterStatus.getServers()).thenReturn(Arrays.asList(sn));
081      when(clusterStatus.getLoad(sn)).thenReturn(sl);
082
083      loadBalancer.setClusterStatus(clusterStatus);
084    }
085    assertTrue(loadBalancer.loads.get(REGION_KEY) != null);
086    assertTrue(loadBalancer.loads.get(REGION_KEY).size() == 15);
087
088    Queue<RegionLoad> loads = loadBalancer.loads.get(REGION_KEY);
089    int i = 0;
090    while(loads.size() > 0) {
091      RegionLoad rl = loads.remove();
092      assertEquals(i + (numClusterStatusToAdd - 15), rl.getStores());
093      i ++;
094    }
095  }

306  @Test
307  public void testReplicaCostForReplicas() {
308    Configuration conf = HBaseConfiguration.create();
309    StochasticLoadBalancer.CostFunction
310        costFunction = new StochasticLoadBalancer.RegionReplicaHostCostFunction(conf);
311
312    int [] servers = new int[] {3,3,3,3,3};
313    TreeMap<ServerName, List<HRegionInfo>> clusterState = mockClusterServers(servers);
314
315    BaseLoadBalancer.Cluster cluster;
316
317    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
318    costFunction.init(cluster);
319    double costWithoutReplicas = costFunction.cost();
320    assertEquals(0, costWithoutReplicas, 0);
321
322    // replicate the region from first server to the last server
323    HRegionInfo replica1 = RegionReplicaUtil.getRegionInfoForReplica(
324      clusterState.firstEntry().getValue().get(0),1);
325    clusterState.lastEntry().getValue().add(replica1);
326
327    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
328    costFunction.init(cluster);
329    double costWith1ReplicaDifferentServer = costFunction.cost();
330
331    assertEquals(0, costWith1ReplicaDifferentServer, 0);
332
333    // add a third replica to the last server
334    HRegionInfo replica2 = RegionReplicaUtil.getRegionInfoForReplica(replica1, 2);
335    clusterState.lastEntry().getValue().add(replica2);
336
337    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
338    costFunction.init(cluster);
339    double costWith1ReplicaSameServer = costFunction.cost();
340
341    assertTrue(costWith1ReplicaDifferentServer < costWith1ReplicaSameServer);
342
343    // test with replication = 4 for following:
344
345    HRegionInfo replica3;
346    Iterator<Entry<ServerName, List<HRegionInfo>>> it;
347    Entry<ServerName, List<HRegionInfo>> entry;
348
349    clusterState = mockClusterServers(servers);
350    it = clusterState.entrySet().iterator();
351    entry = it.next(); //first server
352    HRegionInfo hri = entry.getValue().get(0);
353    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
354    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
355    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
356    entry.getValue().add(replica1);
357    entry.getValue().add(replica2);
358    it.next().getValue().add(replica3); //2nd server
359
360    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
361    costFunction.init(cluster);
362    double costWith3ReplicasSameServer = costFunction.cost();
363
364    clusterState = mockClusterServers(servers);
365    hri = clusterState.firstEntry().getValue().get(0);
366    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
367    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
368    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
369
370    clusterState.firstEntry().getValue().add(replica1);
371    clusterState.lastEntry().getValue().add(replica2);
372    clusterState.lastEntry().getValue().add(replica3);
373
374    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
375    costFunction.init(cluster);
376    double costWith2ReplicasOnTwoServers = costFunction.cost();
377
378    assertTrue(costWith2ReplicasOnTwoServers < costWith3ReplicasSameServer);
379  }

381  @Test
382  public void testNeedsBalanceForColocatedReplicas() {
383    // check for the case where there are two hosts and with one rack, and where
384    // both the replicas are hosted on the same server
385    List<HRegionInfo> regions = randomRegions(1);
386    ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
387    ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
388    Map<ServerName, List<HRegionInfo>> map = new HashMap<ServerName, List<HRegionInfo>>();
389    map.put(s1, regions);
390    regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
391    // until the step above s1 holds two replicas of a region
392    regions = randomRegions(1);
393    map.put(s2, regions);
394    assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null, null)));
395    // check for the case where there are two hosts on the same rack and there are two racks
396    // and both the replicas are on the same rack
397    map.clear();
398    regions = randomRegions(1);
399    List<HRegionInfo> regionsOnS2 = new ArrayList<HRegionInfo>(1);
400    regionsOnS2.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
401    map.put(s1, regions);
402    map.put(s2, regionsOnS2);
403    // add another server so that the cluster has some host on another rack
404    map.put(ServerName.valueOf("host2", 1000, 11111), randomRegions(1));
405    assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null,
406        new ForTestRackManagerOne())));
407  }
