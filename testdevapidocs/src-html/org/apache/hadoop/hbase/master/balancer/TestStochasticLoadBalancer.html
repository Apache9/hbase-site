062  @Test
063  public void testKeepRegionLoad() throws Exception {
064
065    ServerName sn = ServerName.valueOf("test:8080", 100);
066    int numClusterStatusToAdd = 20000;
067    for (int i = 0; i < numClusterStatusToAdd; i++) {
068      ServerLoad sl = mock(ServerLoad.class);
069
070      RegionLoad rl = mock(RegionLoad.class);
071      when(rl.getStores()).thenReturn(i);
072
073      Map<byte[], RegionLoad> regionLoadMap =
074          new TreeMap<byte[], RegionLoad>(Bytes.BYTES_COMPARATOR);
075      regionLoadMap.put(Bytes.toBytes(REGION_KEY), rl);
076      when(sl.getRegionsLoad()).thenReturn(regionLoadMap);
077
078      ClusterStatus clusterStatus = mock(ClusterStatus.class);
079      when(clusterStatus.getServers()).thenReturn(Arrays.asList(sn));
080      when(clusterStatus.getLoad(sn)).thenReturn(sl);
081
082      loadBalancer.setClusterStatus(clusterStatus);
083    }
084    assertTrue(loadBalancer.loads.get(REGION_KEY) != null);
085    assertTrue(loadBalancer.loads.get(REGION_KEY).size() == 15);
086
087    Queue<RegionLoad> loads = loadBalancer.loads.get(REGION_KEY);
088    int i = 0;
089    while(loads.size() > 0) {
090      RegionLoad rl = loads.remove();
091      assertEquals(i + (numClusterStatusToAdd - 15), rl.getStores());
092      i ++;
093    }
094  }

126  public void testMoveCost() throws Exception {
127    Configuration conf = HBaseConfiguration.create();
128    StochasticLoadBalancer.CostFunction
129        costFunction = new StochasticLoadBalancer.MoveCostFunction(conf);
130    for (int[] mockCluster : clusterStateMocks) {
131      BaseLoadBalancer.Cluster cluster = mockCluster(mockCluster);
132      costFunction.init(cluster);
133      double cost = costFunction.cost();
134      assertEquals(0.0f, cost, 0.001);
135
136      // cluster region number is smaller than maxMoves=600
137      cluster.setNumRegions(200);
138      cluster.setNumMovedRegions(10);
139      cost = costFunction.cost();
140      assertEquals(0.05f, cost, 0.001);
141      cluster.setNumMovedRegions(100);
142      cost = costFunction.cost();
143      assertEquals(0.5f, cost, 0.001);
144      cluster.setNumMovedRegions(200);
145      cost = costFunction.cost();
146      assertEquals(1.0f, cost, 0.001);
147
148
149      // cluster region number is bigger than maxMoves=2500
150      cluster.setNumRegions(10000);
151      cluster.setNumMovedRegions(250);
152      cost = costFunction.cost();
153      assertEquals(0.1f, cost, 0.001);
154      cluster.setNumMovedRegions(1250);
155      cost = costFunction.cost();
156      assertEquals(0.5f, cost, 0.001);
157      cluster.setNumMovedRegions(2500);
158      cost = costFunction.cost();
159      assertEquals(1.0f, cost, 0.01);
160    }

184    assertEquals(0, costFunction.cost(), 0.01);
185    costFunction.init(mockCluster(new int[]{10000, 0, 0, 0, 0}));
186    assertEquals(1, costFunction.cost(), 0.01);

225      statThree[i] = (0);
226      statThree[i+100] = 100;

276  @Test
277  public void testReplicaCostForReplicas() {
278    Configuration conf = HBaseConfiguration.create();
279    StochasticLoadBalancer.CostFunction
280        costFunction = new StochasticLoadBalancer.RegionReplicaHostCostFunction(conf);
281
282    int [] servers = new int[] {3,3,3,3,3};
283    TreeMap<ServerName, List<HRegionInfo>> clusterState = mockClusterServers(servers);
284
285    BaseLoadBalancer.Cluster cluster;
286
287    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
288    costFunction.init(cluster);
289    double costWithoutReplicas = costFunction.cost();
290    assertEquals(0, costWithoutReplicas, 0);
291
292    // replicate the region from first server to the last server
293    HRegionInfo replica1 = RegionReplicaUtil.getRegionInfoForReplica(
294      clusterState.firstEntry().getValue().get(0),1);
295    clusterState.lastEntry().getValue().add(replica1);
296
297    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
298    costFunction.init(cluster);
299    double costWith1ReplicaDifferentServer = costFunction.cost();
300
301    assertEquals(0, costWith1ReplicaDifferentServer, 0);
302
303    // add a third replica to the last server
304    HRegionInfo replica2 = RegionReplicaUtil.getRegionInfoForReplica(replica1, 2);
305    clusterState.lastEntry().getValue().add(replica2);
306
307    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
308    costFunction.init(cluster);
309    double costWith1ReplicaSameServer = costFunction.cost();
310
311    assertTrue(costWith1ReplicaDifferentServer < costWith1ReplicaSameServer);
312
313    // test with replication = 4 for following:
314
315    HRegionInfo replica3;
316    Iterator<Entry<ServerName, List<HRegionInfo>>> it;
317    Entry<ServerName, List<HRegionInfo>> entry;
318
319    clusterState = mockClusterServers(servers);
320    it = clusterState.entrySet().iterator();
321    entry = it.next(); //first server
322    HRegionInfo hri = entry.getValue().get(0);
323    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
324    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
325    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
326    entry.getValue().add(replica1);
327    entry.getValue().add(replica2);
328    it.next().getValue().add(replica3); //2nd server
329
330    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
331    costFunction.init(cluster);
332    double costWith3ReplicasSameServer = costFunction.cost();
333
334    clusterState = mockClusterServers(servers);
335    hri = clusterState.firstEntry().getValue().get(0);
336    replica1 = RegionReplicaUtil.getRegionInfoForReplica(hri, 1);
337    replica2 = RegionReplicaUtil.getRegionInfoForReplica(hri, 2);
338    replica3 = RegionReplicaUtil.getRegionInfoForReplica(hri, 3);
339
340    clusterState.firstEntry().getValue().add(replica1);
341    clusterState.lastEntry().getValue().add(replica2);
342    clusterState.lastEntry().getValue().add(replica3);
343
344    cluster = new BaseLoadBalancer.Cluster(clusterState, null, null, null);
345    costFunction.init(cluster);
346    double costWith2ReplicasOnTwoServers = costFunction.cost();
347
348    assertTrue(costWith2ReplicasOnTwoServers < costWith3ReplicasSameServer);
349  }

351  @Test
352  public void testNeedsBalanceForColocatedReplicas() {
353    // check for the case where there are two hosts and with one rack, and where
354    // both the replicas are hosted on the same server
355    List<HRegionInfo> regions = randomRegions(1);
356    ServerName s1 = ServerName.valueOf("host1", 1000, 11111);
357    ServerName s2 = ServerName.valueOf("host11", 1000, 11111);
358    Map<ServerName, List<HRegionInfo>> map = new HashMap<ServerName, List<HRegionInfo>>();
359    map.put(s1, regions);
360    regions.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
361    // until the step above s1 holds two replicas of a region
362    regions = randomRegions(1);
363    map.put(s2, regions);
364    assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null, null)));
365    // check for the case where there are two hosts on the same rack and there are two racks
366    // and both the replicas are on the same rack
367    map.clear();
368    regions = randomRegions(1);
369    List<HRegionInfo> regionsOnS2 = new ArrayList<HRegionInfo>(1);
370    regionsOnS2.add(RegionReplicaUtil.getRegionInfoForReplica(regions.get(0), 1));
371    map.put(s1, regions);
372    map.put(s2, regionsOnS2);
373    // add another server so that the cluster has some host on another rack
374    map.put(ServerName.valueOf("host2", 1000, 11111), randomRegions(1));
375    assertTrue(loadBalancer.needsBalance(new Cluster(map, null, null,
376        new ForTestRackManagerOne())));
377  }
