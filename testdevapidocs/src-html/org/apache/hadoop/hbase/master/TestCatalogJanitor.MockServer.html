532  @Test
533  public void testCleanParent() throws IOException, InterruptedException {
534    HBaseTestingUtility htu = new HBaseTestingUtility();
535    setRootDirAndCleanIt(htu, "testCleanParent");
536    Server server = new MockServer(htu);
537    try {
538      MasterServices services = new MockMasterServices(server);
539      CatalogJanitor janitor = new CatalogJanitor(server, services);
540      // Create regions.
541      HTableDescriptor htd = new HTableDescriptor(TableName.valueOf("table"));
542      htd.addFamily(new HColumnDescriptor("f"));
543      HRegionInfo parent =
544        new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
545            Bytes.toBytes("eee"));
546      HRegionInfo splita =
547        new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
548            Bytes.toBytes("ccc"));
549      HRegionInfo splitb =
550        new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
551            Bytes.toBytes("eee"));
552      // Test that when both daughter regions are in place, that we do not
553      // remove the parent.
554      Result r = createResult(parent, splita, splitb);
555      // Add a reference under splitA directory so we don't clear out the parent.
556      Path rootdir = services.getMasterFileSystem().getRootDir();
557      Path tabledir =
558        FSUtils.getTableDir(rootdir, htd.getTableName());
559      Path storedir = HStore.getStoreHomedir(tabledir, splita,
560          htd.getColumnFamilies()[0].getName());
561      Reference ref = Reference.createTopReference(Bytes.toBytes("ccc"));
562      long now = System.currentTimeMillis();
563      // Reference name has this format: StoreFile#REF_NAME_PARSER
564      Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
565      FileSystem fs = services.getMasterFileSystem().getFileSystem();
566      Path path = ref.write(fs, p);
567      assertTrue(fs.exists(path));
568      assertFalse(janitor.cleanParent(parent, r));
569      // Remove the reference file and try again.
570      assertTrue(fs.delete(p, true));
571      assertTrue(janitor.cleanParent(parent, r));
572    } finally {
573      server.stop("shutdown");
574    }
575  }

601  /**
602   * Make sure parent with specified end key gets cleaned up even if daughter is cleaned up before it.
603   *
604   * @param rootDir the test case name, used as the HBase testing utility root
605   * @param lastEndKey the end key of the split parent
606   * @throws IOException
607   * @throws InterruptedException
608   */
609  private void parentWithSpecifiedEndKeyCleanedEvenIfDaughterGoneFirst(
610  final String rootDir, final byte[] lastEndKey)
611  throws IOException, InterruptedException {
612    HBaseTestingUtility htu = new HBaseTestingUtility();
613    setRootDirAndCleanIt(htu, rootDir);
614    Server server = new MockServer(htu);
615    MasterServices services = new MockMasterServices(server);
616    CatalogJanitor janitor = new CatalogJanitor(server, services);
617    final HTableDescriptor htd = createHTableDescriptor();
618
619    // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
620
621    // Parent
622    HRegionInfo parent = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
623      lastEndKey);
624    // Sleep a second else the encoded name on these regions comes out
625    // same for all with same start key and made in same second.
626    Thread.sleep(1001);
627
628    // Daughter a
629    HRegionInfo splita = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
630      Bytes.toBytes("ccc"));
631    Thread.sleep(1001);
632    // Make daughters of daughter a; splitaa and splitab.
633    HRegionInfo splitaa = new HRegionInfo(htd.getTableName(), Bytes.toBytes("aaa"),
634      Bytes.toBytes("bbb"));
635    HRegionInfo splitab = new HRegionInfo(htd.getTableName(), Bytes.toBytes("bbb"),
636      Bytes.toBytes("ccc"));
637
638    // Daughter b
639    HRegionInfo splitb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
640      lastEndKey);
641    Thread.sleep(1001);
642    // Make Daughters of daughterb; splitba and splitbb.
643    HRegionInfo splitba = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ccc"),
644      Bytes.toBytes("ddd"));
645    HRegionInfo splitbb = new HRegionInfo(htd.getTableName(), Bytes.toBytes("ddd"),
646    lastEndKey);
647
648    // First test that our Comparator works right up in CatalogJanitor.
649    // Just fo kicks.
650    SortedMap<HRegionInfo, Result> regions =
651      new TreeMap<HRegionInfo, Result>(new CatalogJanitor.SplitParentFirstComparator());
652    // Now make sure that this regions map sorts as we expect it to.
653    regions.put(parent, createResult(parent, splita, splitb));
654    regions.put(splitb, createResult(splitb, splitba, splitbb));
655    regions.put(splita, createResult(splita, splitaa, splitab));
656    // Assert its properly sorted.
657    int index = 0;
658    for (Map.Entry<HRegionInfo, Result> e: regions.entrySet()) {
659      if (index == 0) {
660        assertTrue(e.getKey().getEncodedName().equals(parent.getEncodedName()));
661      } else if (index == 1) {
662        assertTrue(e.getKey().getEncodedName().equals(splita.getEncodedName()));
663      } else if (index == 2) {
664        assertTrue(e.getKey().getEncodedName().equals(splitb.getEncodedName()));
665      }
666      index++;
667    }
668
669    // Now play around with the cleanParent function.  Create a ref from splita
670    // up to the parent.
671    Path splitaRef =
672      createReferences(services, htd, parent, splita, Bytes.toBytes("ccc"), false);
673    // Make sure actual super parent sticks around because splita has a ref.
674    assertFalse(janitor.cleanParent(parent, regions.get(parent)));
675
676    //splitba, and split bb, do not have dirs in fs.  That means that if
677    // we test splitb, it should get cleaned up.
678    assertTrue(janitor.cleanParent(splitb, regions.get(splitb)));
679
680    // Now remove ref from splita to parent... so parent can be let go and so
681    // the daughter splita can be split (can't split if still references).
682    // BUT make the timing such that the daughter gets cleaned up before we
683    // can get a chance to let go of the parent.
684    FileSystem fs = FileSystem.get(htu.getConfiguration());
685    assertTrue(fs.delete(splitaRef, true));
686    // Create the refs from daughters of splita.
687    Path splitaaRef =
688      createReferences(services, htd, splita, splitaa, Bytes.toBytes("bbb"), false);
689    Path splitabRef =
690      createReferences(services, htd, splita, splitab, Bytes.toBytes("bbb"), true);
691
692    // Test splita.  It should stick around because references from splitab, etc.
693    assertFalse(janitor.cleanParent(splita, regions.get(splita)));
694
695    // Now clean up parent daughter first.  Remove references from its daughters.
696    assertTrue(fs.delete(splitaaRef, true));
697    assertTrue(fs.delete(splitabRef, true));
698    assertTrue(janitor.cleanParent(splita, regions.get(splita)));
699
700    // Super parent should get cleaned up now both splita and splitb are gone.
701    assertTrue(janitor.cleanParent(parent, regions.get(parent)));
702
703    services.stop("test finished");
704    janitor.cancel(true);
705  }

1058   * @param top True if we are to write a 'top' reference.

1062  private Path createReferences(final MasterServices services,
1063      final HTableDescriptor htd, final HRegionInfo parent,
1064      final HRegionInfo daughter, final byte [] midkey, final boolean top)
1065  throws IOException {
1066    Path rootdir = services.getMasterFileSystem().getRootDir();

1068    Path storedir = HStore.getStoreHomedir(tabledir, daughter,
1069      htd.getColumnFamilies()[0].getName());
1070    Reference ref =
1071      top? Reference.createTopReference(midkey): Reference.createBottomReference(midkey);
1072    long now = System.currentTimeMillis();
1073    // Reference name has this format: StoreFile#REF_NAME_PARSER
1074    Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
1075    FileSystem fs = services.getMasterFileSystem().getFileSystem();
1076    ref.write(fs, p);
1077    return p;
1078  }

1096  private MultiResponse buildMultiResponse(MultiRequest req) {
1097    MultiResponse.Builder builder = MultiResponse.newBuilder();
1098    RegionActionResult.Builder regionActionResultBuilder =
1099        RegionActionResult.newBuilder();
1100    ResultOrException.Builder roeBuilder = ResultOrException.newBuilder();
1101    for (RegionAction regionAction: req.getRegionActionList()) {
1102      regionActionResultBuilder.clear();
1103      for (ClientProtos.Action action: regionAction.getActionList()) {
1104        roeBuilder.clear();
1105        roeBuilder.setResult(ClientProtos.Result.getDefaultInstance());
1106        roeBuilder.setIndex(action.getIndex());
1107        regionActionResultBuilder.addResultOrException(roeBuilder.build());
1108      }
1109      builder.addRegionActionResult(regionActionResultBuilder.build());
1110    }
1111    return builder.build();
1112  }
