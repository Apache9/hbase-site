044public class TestCreateTableProcedure extends TestTableDDLProcedureBase {

066  }

068  private void testSimpleCreate(final TableName tableName, byte[][] splitKeys) throws Exception {
069    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
070      getMasterProcedureExecutor(), tableName, splitKeys, F1, F2);
071    MasterProcedureTestingUtility.validateTableCreation(getMaster(), tableName, regions, F1, F2);
072  }

074  @Test(timeout=60000)
075  public void testCreateWithoutColumnFamily() throws Exception {
076    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
077    final TableName tableName = TableName.valueOf(name.getMethodName());
078    // create table with 0 families will fail
079    final HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(tableName);
080
081    // disable sanity check
082    htd.setConfiguration("hbase.table.sanity.checks", Boolean.FALSE.toString());
083    final HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, null);
084
085    long procId =
086        ProcedureTestingUtility.submitAndWait(procExec,
087            new CreateTableProcedure(procExec.getEnvironment(), htd, regions));
088    final ProcedureInfo result = procExec.getResult(procId);
089    assertEquals(true, result.isFailed());
090    Throwable cause = ProcedureTestingUtility.getExceptionCause(result);
091    assertTrue("expected DoNotRetryIOException, got " + cause,
092        cause instanceof DoNotRetryIOException);
093  }
094
095  @Test(timeout=60000, expected=TableExistsException.class)
096  public void testCreateExisting() throws Exception {
097    final TableName tableName = TableName.valueOf(name.getMethodName());
098    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
099    final HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(tableName, "f");
100    final HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, null);
101
102    // create the table
103    long procId1 = procExec.submitProcedure(
104      new CreateTableProcedure(procExec.getEnvironment(), htd, regions));
105
106    // create another with the same name
107    ProcedurePrepareLatch latch2 = new ProcedurePrepareLatch.CompatibilityLatch();
108    long procId2 = procExec.submitProcedure(
109      new CreateTableProcedure(procExec.getEnvironment(), htd, regions, latch2));
110
111    ProcedureTestingUtility.waitProcedure(procExec, procId1);
112    ProcedureTestingUtility.assertProcNotFailed(procExec.getResult(procId1));
113
114    ProcedureTestingUtility.waitProcedure(procExec, procId2);
115    latch2.await();
116  }

118  @Test(timeout=60000)
119  public void testRecoveryAndDoubleExecution() throws Exception {
120    final TableName tableName = TableName.valueOf(name.getMethodName());
121
122    // create the table
123    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
124    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
125
126    // Start the Create procedure && kill the executor
127    byte[][] splitKeys = null;
128    HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(tableName, "f1", "f2");
129    HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, splitKeys);
130    long procId = procExec.submitProcedure(
131      new CreateTableProcedure(procExec.getEnvironment(), htd, regions));
132
133    // Restart the executor and execute the step twice
134    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
135    MasterProcedureTestingUtility.validateTableCreation(getMaster(), tableName, regions, F1, F2);
136  }

144  @Test(timeout=90000)
145  public void testRollbackAndDoubleExecutionOnMobTable() throws Exception {
146    final TableName tableName = TableName.valueOf(name.getMethodName());
147    HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(tableName, F1, F2);
148    htd.getFamily(Bytes.toBytes(F1)).setMobEnabled(true);
149    testRollbackAndDoubleExecution(htd);
150  }
151
152  private void testRollbackAndDoubleExecution(HTableDescriptor htd) throws Exception {
153    // create the table
154    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
155    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
156
157    // Start the Create procedure && kill the executor
158    final byte[][] splitKeys = new byte[][] {
159      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
160    };
161    htd.setRegionReplication(3);
162    HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, splitKeys);
163    long procId = procExec.submitProcedure(
164      new CreateTableProcedure(procExec.getEnvironment(), htd, regions));
165
166    int numberOfSteps = 0; // failing at pre operation
167    MasterProcedureTestingUtility.testRollbackAndDoubleExecution(procExec, procId, numberOfSteps);
168
169    TableName tableName = htd.getTableName();
170    MasterProcedureTestingUtility.validateTableDeletion(getMaster(), tableName);
171
172    // are we able to create the table after a rollback?
173    resetProcExecutorTestingKillFlag();
174    testSimpleCreate(tableName, splitKeys);
175  }

189  }
