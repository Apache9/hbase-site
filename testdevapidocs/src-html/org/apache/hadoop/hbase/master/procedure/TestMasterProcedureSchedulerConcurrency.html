071  @Test(timeout=60000)
072  public void testConcurrentCreateDelete() throws Exception {
073    final MasterProcedureScheduler procQueue = queue;
074    final TableName table = TableName.valueOf("testtb");
075    final AtomicBoolean running = new AtomicBoolean(true);
076    final AtomicBoolean failure = new AtomicBoolean(false);
077    Thread createThread = new Thread() {
078      @Override
079      public void run() {
080        try {
081          TestTableProcedure proc = new TestTableProcedure(1, table,
082              TableProcedureInterface.TableOperationType.CREATE);
083          while (running.get() && !failure.get()) {
084            if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
085              procQueue.releaseTableExclusiveLock(proc, table);
086            }
087          }
088        } catch (Throwable e) {
089          LOG.error("create failed", e);
090          failure.set(true);
091        }
092      }
093    };
094
095    Thread deleteThread = new Thread() {
096      @Override
097      public void run() {
098        try {
099          TestTableProcedure proc = new TestTableProcedure(2, table,
100              TableProcedureInterface.TableOperationType.DELETE);
101          while (running.get() && !failure.get()) {
102            if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
103              procQueue.releaseTableExclusiveLock(proc, table);
104            }
105            procQueue.markTableAsDeleted(table, proc);
106          }
107        } catch (Throwable e) {
108          LOG.error("delete failed", e);
109          failure.set(true);
110        }
111      }
112    };
113
114    createThread.start();
115    deleteThread.start();
116    for (int i = 0; i < 100 && running.get() && !failure.get(); ++i) {
117      Thread.sleep(100);
118    }
119    running.set(false);
120    createThread.join();
121    deleteThread.join();
122    assertEquals(false, failure.get());
123  }

125  /**
126   * Verify that "write" operations for a single table are serialized,
127   * but different tables can be executed in parallel.
128   */
129  @Test(timeout=60000)
130  public void testConcurrentWriteOps() throws Exception {
131    final TestTableProcSet procSet = new TestTableProcSet(queue);
132
133    final int NUM_ITEMS = 10;
134    final int NUM_TABLES = 4;
135    final AtomicInteger opsCount = new AtomicInteger(0);
136    for (int i = 0; i < NUM_TABLES; ++i) {
137      TableName tableName = TableName.valueOf(String.format("testtb-%04d", i));
138      for (int j = 1; j < NUM_ITEMS; ++j) {
139        procSet.addBack(new TestTableProcedure(i * 100 + j, tableName,
140          TableProcedureInterface.TableOperationType.EDIT));
141        opsCount.incrementAndGet();
142      }
143    }
144    assertEquals(opsCount.get(), queue.size());
145
146    final Thread[] threads = new Thread[NUM_TABLES * 2];
147    final HashSet<TableName> concurrentTables = new HashSet<TableName>();
148    final ArrayList<String> failures = new ArrayList<String>();
149    final AtomicInteger concurrentCount = new AtomicInteger(0);
150    for (int i = 0; i < threads.length; ++i) {
151      threads[i] = new Thread() {
152        @Override
153        public void run() {
154          while (opsCount.get() > 0) {
155            try {
156              Procedure proc = procSet.acquire();
157              if (proc == null) {
158                queue.signalAll();
159                if (opsCount.get() > 0) {
160                  continue;
161                }
162                break;
163              }
164
165              TableName tableId = procSet.getTableName(proc);
166              synchronized (concurrentTables) {
167                assertTrue("unexpected concurrency on " + tableId, concurrentTables.add(tableId));
168              }
169              assertTrue(opsCount.decrementAndGet() >= 0);
170              try {
171                long procId = proc.getProcId();
172                int concurrent = concurrentCount.incrementAndGet();
173                assertTrue("inc-concurrent="+ concurrent +" 1 <= concurrent <= "+ NUM_TABLES,
174                  concurrent >= 1 && concurrent <= NUM_TABLES);
175                LOG.debug("[S] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
176                Thread.sleep(2000);
177                concurrent = concurrentCount.decrementAndGet();
178                LOG.debug("[E] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
179                assertTrue("dec-concurrent=" + concurrent, concurrent < NUM_TABLES);
180              } finally {
181                synchronized (concurrentTables) {
182                  assertTrue(concurrentTables.remove(tableId));
183                }
184                procSet.release(proc);
185              }
186            } catch (Throwable e) {
187              LOG.error("Failed " + e.getMessage(), e);
188              synchronized (failures) {
189                failures.add(e.getMessage());
190              }
191            } finally {
192              queue.signalAll();
193            }
194          }
195        }
196      };
197      threads[i].start();
198    }
199    for (int i = 0; i < threads.length; ++i) {
200      threads[i].join();
201    }
202    assertTrue(failures.toString(), failures.isEmpty());
203    assertEquals(0, opsCount.get());
204    assertEquals(0, queue.size());
205
206    for (int i = 1; i <= NUM_TABLES; ++i) {
207      final TableName table = TableName.valueOf(String.format("testtb-%04d", i));
208      final TestTableProcedure dummyProc = new TestTableProcedure(100, table,
209        TableProcedureInterface.TableOperationType.DELETE);
210      assertTrue("queue should be deleted, table=" + table,
211        queue.markTableAsDeleted(table, dummyProc));
212    }
213  }

225  private void testConcurrentWaitWake(final boolean useWakeBatch) throws Exception {
226    final TableName tableName = TableName.valueOf("testtb");
227
228    final int NPROCS = 20;
229    final int NRUNS = 100;
230
231    for (long i = 0; i < NPROCS; ++i) {
232      queue.addBack(new TestTableProcedureWithEvent(i, tableName,
233          TableProcedureInterface.TableOperationType.READ));
234    }
235
236    final Thread[] threads = new Thread[4];
237    final AtomicInteger waitCount = new AtomicInteger(0);
238    final AtomicInteger wakeCount = new AtomicInteger(0);
239
240    final ConcurrentSkipListSet<TestTableProcedureWithEvent> waitQueue =
241      new ConcurrentSkipListSet<TestTableProcedureWithEvent>();
242    threads[0] = new Thread() {
243      @Override
244      public void run() {
245        while (true) {
246          if (useWakeBatch) {
247            ProcedureEvent[] ev = new ProcedureEvent[waitQueue.size()];
248            for (int i = 0; i < ev.length; ++i) {
249              ev[i] = waitQueue.pollFirst().getEvent();
250              LOG.debug("WAKE " + ev[i] + " total=" + wakeCount.get());
251            }
252            queue.wakeEvents(ev, ev.length);
253            wakeCount.addAndGet(ev.length);
254          } else {
255            int size = waitQueue.size();
256            while (size-- > 0) {
257              ProcedureEvent ev = waitQueue.pollFirst().getEvent();
258              queue.wakeEvent(ev);
259              LOG.debug("WAKE " + ev + " total=" + wakeCount.get());
260              wakeCount.incrementAndGet();
261            }
262          }
263          if (wakeCount.get() >= NRUNS) {
264            break;
265          }
266          Threads.sleepWithoutInterrupt(25);
267        }
268      }
269    };
270
271    for (int i = 1; i < threads.length; ++i) {
272      threads[i] = new Thread() {
273        @Override
274        public void run() {
275          while (true) {
276            TestTableProcedureWithEvent proc = (TestTableProcedureWithEvent)queue.poll();
277            if (proc == null) continue;
278
279            waitQueue.add(proc);
280            queue.suspendEvent(proc.getEvent());
281            queue.waitEvent(proc.getEvent(), proc);
282            LOG.debug("WAIT " + proc.getEvent());
283            if (waitCount.incrementAndGet() >= NRUNS) {
284              break;
285            }
286          }
287        }
288      };
289    }
290
291    for (int i = 0; i < threads.length; ++i) {
292      threads[i].start();
293    }
294    for (int i = 0; i < threads.length; ++i) {
295      threads[i].join();
296    }
297
298    queue.clear();
299  }

301  public static class TestTableProcSet {
302    private final MasterProcedureScheduler queue;
303
304    public TestTableProcSet(final MasterProcedureScheduler queue) {
305      this.queue = queue;
306    }
307
308    public void addBack(Procedure proc) {
309      queue.addBack(proc);
310    }
311
312    public void addFront(Procedure proc) {
313      queue.addFront(proc);
314    }
315
316    public Procedure acquire() {
317      Procedure proc = null;
318      boolean avail = false;
319      while (!avail) {
320        proc = queue.poll();
321        if (proc == null) break;
322        switch (getTableOperationType(proc)) {
323          case CREATE:
324          case DELETE:
325          case EDIT:
326            avail = queue.tryAcquireTableExclusiveLock(proc, getTableName(proc));
327            break;
328          case READ:
329            avail = queue.tryAcquireTableSharedLock(proc, getTableName(proc));
330            break;
331          default:
332            throw new UnsupportedOperationException();
333        }
334        if (!avail) {
335          addFront(proc);
336          LOG.debug("yield procId=" + proc);
337        }
338      }
339      return proc;
340    }
341
342    public void release(Procedure proc) {
343      switch (getTableOperationType(proc)) {
344        case CREATE:
345        case DELETE:
346        case EDIT:
347          queue.releaseTableExclusiveLock(proc, getTableName(proc));
348          break;
349        case READ:
350          queue.releaseTableSharedLock(proc, getTableName(proc));
351          break;
352      }
353    }
354
355    public TableName getTableName(Procedure proc) {
356      return ((TableProcedureInterface)proc).getTableName();
357    }
358
359    public TableProcedureInterface.TableOperationType getTableOperationType(Procedure proc) {
360      return ((TableProcedureInterface)proc).getTableOperationType();
361    }
362  }
