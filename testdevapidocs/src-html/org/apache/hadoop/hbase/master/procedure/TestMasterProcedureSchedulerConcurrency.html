070  /**
071   * Verify that "write" operations for a single table are serialized,
072   * but different tables can be executed in parallel.
073   */
074  @Test(timeout=60000)
075  public void testConcurrentWriteOps() throws Exception {
076    final TestTableProcSet procSet = new TestTableProcSet(queue);
077
078    final int NUM_ITEMS = 10;
079    final int NUM_TABLES = 4;
080    final AtomicInteger opsCount = new AtomicInteger(0);
081    for (int i = 0; i < NUM_TABLES; ++i) {
082      TableName tableName = TableName.valueOf(String.format("testtb-%04d", i));
083      for (int j = 1; j < NUM_ITEMS; ++j) {
084        procSet.addBack(new TestTableProcedure(i * 100 + j, tableName,
085          TableProcedureInterface.TableOperationType.EDIT));
086        opsCount.incrementAndGet();
087      }
088    }
089    assertEquals(opsCount.get(), queue.size());
090
091    final Thread[] threads = new Thread[NUM_TABLES * 2];
092    final HashSet<TableName> concurrentTables = new HashSet<TableName>();
093    final ArrayList<String> failures = new ArrayList<String>();
094    final AtomicInteger concurrentCount = new AtomicInteger(0);
095    for (int i = 0; i < threads.length; ++i) {
096      threads[i] = new Thread() {
097        @Override
098        public void run() {
099          while (opsCount.get() > 0) {
100            try {
101              Procedure proc = procSet.acquire();
102              if (proc == null) {
103                queue.signalAll();
104                if (opsCount.get() > 0) {
105                  continue;
106                }
107                break;
108              }
109
110              TableName tableId = procSet.getTableName(proc);
111              synchronized (concurrentTables) {
112                assertTrue("unexpected concurrency on " + tableId, concurrentTables.add(tableId));
113              }
114              assertTrue(opsCount.decrementAndGet() >= 0);
115              try {
116                long procId = proc.getProcId();
117                int concurrent = concurrentCount.incrementAndGet();
118                assertTrue("inc-concurrent="+ concurrent +" 1 <= concurrent <= "+ NUM_TABLES,
119                  concurrent >= 1 && concurrent <= NUM_TABLES);
120                LOG.debug("[S] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
121                Thread.sleep(2000);
122                concurrent = concurrentCount.decrementAndGet();
123                LOG.debug("[E] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
124                assertTrue("dec-concurrent=" + concurrent, concurrent < NUM_TABLES);
125              } finally {
126                synchronized (concurrentTables) {
127                  assertTrue(concurrentTables.remove(tableId));
128                }
129                procSet.release(proc);
130              }
131            } catch (Throwable e) {
132              LOG.error("Failed " + e.getMessage(), e);
133              synchronized (failures) {
134                failures.add(e.getMessage());
135              }
136            } finally {
137              queue.signalAll();
138            }
139          }
140        }
141      };
142      threads[i].start();
143    }
144    for (int i = 0; i < threads.length; ++i) {
145      threads[i].join();
146    }
147    assertTrue(failures.toString(), failures.isEmpty());
148    assertEquals(0, opsCount.get());
149    assertEquals(0, queue.size());
150
151    for (int i = 1; i <= NUM_TABLES; ++i) {
152      final TableName table = TableName.valueOf(String.format("testtb-%04d", i));
153      final TestTableProcedure dummyProc = new TestTableProcedure(100, table,
154        TableProcedureInterface.TableOperationType.DELETE);
155      assertTrue("queue should be deleted, table=" + table,
156        queue.markTableAsDeleted(table, dummyProc));
157    }
158  }

168  public static class TestTableProcSet {
169    private final MasterProcedureScheduler queue;
170
171    public TestTableProcSet(final MasterProcedureScheduler queue) {
172      this.queue = queue;
173    }
174
175    public void addBack(Procedure proc) {
176      queue.addBack(proc);
177    }
178
179    public void addFront(Procedure proc) {
180      queue.addFront(proc);
181    }
182
183    public Procedure acquire() {
184      Procedure proc = null;
185      boolean waiting = true;
186      while (waiting && queue.size() > 0) {
187        proc = queue.poll(100000000L);
188        if (proc == null) continue;
189        switch (getTableOperationType(proc)) {
190          case CREATE:
191          case DELETE:
192          case EDIT:
193            waiting = queue.waitTableExclusiveLock(proc, getTableName(proc));
194            break;
195          case READ:
196            waiting = queue.waitTableSharedLock(proc, getTableName(proc));
197            break;
198          default:
199            throw new UnsupportedOperationException();
200        }
201      }
202      return proc;
203    }
204
205    public void release(Procedure proc) {
206      switch (getTableOperationType(proc)) {
207        case CREATE:
208        case DELETE:
209        case EDIT:
210          queue.wakeTableExclusiveLock(proc, getTableName(proc));
211          break;
212        case READ:
213          queue.wakeTableSharedLock(proc, getTableName(proc));
214          break;
215      }
216    }
217
218    public TableName getTableName(Procedure proc) {
219      return ((TableProcedureInterface)proc).getTableName();
220    }
221
222    public TableProcedureInterface.TableOperationType getTableOperationType(Procedure proc) {
223      return ((TableProcedureInterface)proc).getTableOperationType();
224    }
225  }
