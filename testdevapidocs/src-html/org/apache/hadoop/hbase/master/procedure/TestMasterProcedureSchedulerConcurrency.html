047@Category({MasterTests.class, MediumTests.class})

066  @Test(timeout=60000)
067  public void testConcurrentCreateDelete() throws Exception {
068    final MasterProcedureScheduler procQueue = queue;
069    final TableName table = TableName.valueOf("testtb");
070    final AtomicBoolean running = new AtomicBoolean(true);
071    final AtomicBoolean failure = new AtomicBoolean(false);
072    Thread createThread = new Thread() {
073      @Override
074      public void run() {
075        try {
076          TestTableProcedure proc = new TestTableProcedure(1, table,
077              TableProcedureInterface.TableOperationType.CREATE);
078          while (running.get() && !failure.get()) {
079            if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
080              procQueue.releaseTableExclusiveLock(proc, table);
081            }
082          }
083        } catch (Throwable e) {
084          LOG.error("create failed", e);
085          failure.set(true);
086        }
087      }
088    };
089
090    Thread deleteThread = new Thread() {
091      @Override
092      public void run() {
093        try {
094          TestTableProcedure proc = new TestTableProcedure(2, table,
095              TableProcedureInterface.TableOperationType.DELETE);
096          while (running.get() && !failure.get()) {
097            if (procQueue.tryAcquireTableExclusiveLock(proc, table)) {
098              procQueue.releaseTableExclusiveLock(proc, table);
099            }
100            procQueue.markTableAsDeleted(table, proc);
101          }
102        } catch (Throwable e) {
103          LOG.error("delete failed", e);
104          failure.set(true);
105        }
106      }
107    };
108
109    createThread.start();
110    deleteThread.start();
111    for (int i = 0; i < 100 && running.get() && !failure.get(); ++i) {
112      Thread.sleep(100);
113    }
114    running.set(false);
115    createThread.join();
116    deleteThread.join();
117    assertEquals(false, failure.get());
118  }

120  /**
121   * Verify that "write" operations for a single table are serialized,
122   * but different tables can be executed in parallel.
123   */
124  @Test(timeout=60000)
125  public void testConcurrentWriteOps() throws Exception {
126    final TestTableProcSet procSet = new TestTableProcSet(queue);
127
128    final int NUM_ITEMS = 10;
129    final int NUM_TABLES = 4;
130    final AtomicInteger opsCount = new AtomicInteger(0);
131    for (int i = 0; i < NUM_TABLES; ++i) {
132      TableName tableName = TableName.valueOf(String.format("testtb-%04d", i));
133      for (int j = 1; j < NUM_ITEMS; ++j) {
134        procSet.addBack(new TestTableProcedure(i * 100 + j, tableName,
135          TableProcedureInterface.TableOperationType.EDIT));
136        opsCount.incrementAndGet();
137      }
138    }
139    assertEquals(opsCount.get(), queue.size());
140
141    final Thread[] threads = new Thread[NUM_TABLES * 2];
142    final HashSet<TableName> concurrentTables = new HashSet<TableName>();
143    final ArrayList<String> failures = new ArrayList<String>();
144    final AtomicInteger concurrentCount = new AtomicInteger(0);
145    for (int i = 0; i < threads.length; ++i) {
146      threads[i] = new Thread() {
147        @Override
148        public void run() {
149          while (opsCount.get() > 0) {
150            try {
151              Procedure proc = procSet.acquire();
152              if (proc == null) {
153                queue.signalAll();
154                if (opsCount.get() > 0) {
155                  continue;
156                }
157                break;
158              }
159
160              TableName tableId = procSet.getTableName(proc);
161              synchronized (concurrentTables) {
162                assertTrue("unexpected concurrency on " + tableId, concurrentTables.add(tableId));
163              }
164              assertTrue(opsCount.decrementAndGet() >= 0);
165              try {
166                long procId = proc.getProcId();
167                int concurrent = concurrentCount.incrementAndGet();
168                assertTrue("inc-concurrent="+ concurrent +" 1 <= concurrent <= "+ NUM_TABLES,
169                  concurrent >= 1 && concurrent <= NUM_TABLES);
170                LOG.debug("[S] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
171                Thread.sleep(2000);
172                concurrent = concurrentCount.decrementAndGet();
173                LOG.debug("[E] tableId="+ tableId +" procId="+ procId +" concurrent="+ concurrent);
174                assertTrue("dec-concurrent=" + concurrent, concurrent < NUM_TABLES);
175              } finally {
176                synchronized (concurrentTables) {
177                  assertTrue(concurrentTables.remove(tableId));
178                }
179                procSet.release(proc);
180              }
181            } catch (Throwable e) {
182              LOG.error("Failed " + e.getMessage(), e);
183              synchronized (failures) {
184                failures.add(e.getMessage());
185              }
186            } finally {
187              queue.signalAll();
188            }
189          }
190        }
191      };
192      threads[i].start();
193    }
194    for (int i = 0; i < threads.length; ++i) {
195      threads[i].join();
196    }
197    assertTrue(failures.toString(), failures.isEmpty());
198    assertEquals(0, opsCount.get());
199    assertEquals(0, queue.size());
200
201    for (int i = 1; i <= NUM_TABLES; ++i) {
202      final TableName table = TableName.valueOf(String.format("testtb-%04d", i));
203      final TestTableProcedure dummyProc = new TestTableProcedure(100, table,
204        TableProcedureInterface.TableOperationType.DELETE);
205      assertTrue("queue should be deleted, table=" + table,
206        queue.markTableAsDeleted(table, dummyProc));
207    }
208  }

210  public static class TestTableProcSet {
211    private final MasterProcedureScheduler queue;
212
213    public TestTableProcSet(final MasterProcedureScheduler queue) {
214      this.queue = queue;
215    }
216
217    public void addBack(Procedure proc) {
218      queue.addBack(proc);
219    }
220
221    public void addFront(Procedure proc) {
222      queue.addFront(proc);
223    }
224
225    public Procedure acquire() {
226      Procedure proc = null;
227      boolean avail = false;
228      while (!avail) {
229        proc = queue.poll();
230        if (proc == null) break;
231        switch (getTableOperationType(proc)) {
232          case CREATE:
233          case DELETE:
234          case EDIT:
235            avail = queue.tryAcquireTableExclusiveLock(proc, getTableName(proc));
236            break;
237          case READ:
238            avail = queue.tryAcquireTableSharedLock(proc, getTableName(proc));
239            break;
240          default:
241            throw new UnsupportedOperationException();
242        }
243        if (!avail) {
244          addFront(proc);
245          LOG.debug("yield procId=" + proc);
246        }
247      }
248      return proc;
249    }
250
251    public void release(Procedure proc) {
252      switch (getTableOperationType(proc)) {
253        case CREATE:
254        case DELETE:
255        case EDIT:
256          queue.releaseTableExclusiveLock(proc, getTableName(proc));
257          break;
258        case READ:
259          queue.releaseTableSharedLock(proc, getTableName(proc));
260          break;
261      }
262    }
263
264    public TableName getTableName(Procedure proc) {
265      return ((TableProcedureInterface)proc).getTableName();
266    }
267
268    public TableProcedureInterface.TableOperationType getTableOperationType(Procedure proc) {
269      return ((TableProcedureInterface)proc).getTableOperationType();
270    }
271  }
