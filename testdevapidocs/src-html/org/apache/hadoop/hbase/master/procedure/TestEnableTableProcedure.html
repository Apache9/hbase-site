042  @Test(timeout = 60000)
043  public void testEnableTable() throws Exception {
044    final TableName tableName = TableName.valueOf("testEnableTable");
045    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
046
047    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1", "f2");
048    UTIL.getHBaseAdmin().disableTable(tableName);
049
050    // Enable the table
051    long procId = procExec.submitProcedure(
052      new EnableTableProcedure(procExec.getEnvironment(), tableName, false), nonceGroup, nonce);
053    // Wait the completion
054    ProcedureTestingUtility.waitProcedure(procExec, procId);
055    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
056    MasterProcedureTestingUtility.validateTableIsEnabled(UTIL.getHBaseCluster().getMaster(),
057      tableName);
058  }

060  @Test(timeout = 60000)
061  public void testEnableTableTwiceWithSameNonce() throws Exception {
062    final TableName tableName = TableName.valueOf("testEnableTableTwiceWithSameNonce");
063    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
064
065    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1", "f2");
066    UTIL.getHBaseAdmin().disableTable(tableName);
067
068    // Enable the table
069    long procId1 = procExec.submitProcedure(
070      new EnableTableProcedure(procExec.getEnvironment(), tableName, false), nonceGroup, nonce);
071    long procId2 = procExec.submitProcedure(
072      new EnableTableProcedure(procExec.getEnvironment(), tableName, false), nonceGroup, nonce);
073
074    // Wait the completion
075    ProcedureTestingUtility.waitProcedure(procExec, procId1);
076    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
077    // The second proc should succeed too - because it is the same proc.
078    ProcedureTestingUtility.waitProcedure(procExec, procId2);
079    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
080    assertTrue(procId1 == procId2);
081  }

083  @Test(timeout=60000, expected=TableNotDisabledException.class)
084  public void testEnableNonDisabledTable() throws Exception {
085    final TableName tableName = TableName.valueOf("testEnableNonExistingTable");
086    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
087
088    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f1", "f2");
089
090    // Enable the table - expect failure
091    long procId1 = procExec.submitProcedure(
092        new EnableTableProcedure(procExec.getEnvironment(), tableName, false), nonceGroup, nonce);
093    ProcedureTestingUtility.waitProcedure(procExec, procId1);
094
095    ProcedureInfo result = procExec.getResult(procId1);
096    assertTrue(result.isFailed());
097    LOG.debug("Enable failed with exception: " + result.getExceptionFullMessage());
098    assertTrue(
099      ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotDisabledException);
100
101    // Enable the table with skipping table state check flag (simulate recovery scenario)
102    long procId2 = procExec.submitProcedure(
103        new EnableTableProcedure(procExec.getEnvironment(), tableName, true),
104        nonceGroup + 1,
105        nonce + 1);
106    // Wait the completion
107    ProcedureTestingUtility.waitProcedure(procExec, procId2);
108    ProcedureTestingUtility.assertProcNotFailed(procExec, procId2);
109
110    // Enable the table - expect failure from ProcedurePrepareLatch
111    final ProcedurePrepareLatch prepareLatch = new ProcedurePrepareLatch.CompatibilityLatch();
112    long procId3 = procExec.submitProcedure(
113        new EnableTableProcedure(procExec.getEnvironment(), tableName, false, prepareLatch),
114        nonceGroup + 2,
115        nonce + 2);
116    prepareLatch.await();
117    Assert.fail("Enable should throw exception through latch.");
118  }

120  @Test(timeout = 60000)
121  public void testRecoveryAndDoubleExecution() throws Exception {
122    final TableName tableName = TableName.valueOf("testRecoveryAndDoubleExecution");
123    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
124
125    final byte[][] splitKeys = new byte[][] {
126      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
127    };
128    MasterProcedureTestingUtility.createTable(procExec, tableName, splitKeys, "f1", "f2");
129    UTIL.getHBaseAdmin().disableTable(tableName);
130    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
131    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
132
133    // Start the Enable procedure && kill the executor
134    long procId = procExec.submitProcedure(
135        new EnableTableProcedure(procExec.getEnvironment(), tableName, false), nonceGroup, nonce);
136
137    // Restart the executor and execute the step twice
138    int numberOfSteps = EnableTableState.values().length;
139    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId, numberOfSteps);
140    MasterProcedureTestingUtility.validateTableIsEnabled(UTIL.getHBaseCluster().getMaster(),
141      tableName);
142  }
