054  @Test
055  public void testTruncateNotExistentTable() throws Exception {
056    final TableName tableName = TableName.valueOf(name.getMethodName());
057
058    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
059    long procId = ProcedureTestingUtility.submitAndWait(procExec,
060        new TruncateTableProcedure(procExec.getEnvironment(), tableName, true));
061
062    // Second delete should fail with TableNotFound
063    Procedure<?> result = procExec.getResult(procId);
064    assertTrue(result.isFailed());
065    LOG.debug("Truncate failed with exception: " + result.getException());
066    assertTrue(ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotFoundException);
067  }

099  private void testSimpleTruncate(final TableName tableName, final boolean preserveSplits)
100      throws Exception {
101    final String[] families = new String[] { "f1", "f2" };
102    final byte[][] splitKeys = new byte[][] {
103      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
104    };
105
106    RegionInfo[] regions = MasterProcedureTestingUtility.createTable(
107      getMasterProcedureExecutor(), tableName, splitKeys, families);
108    // load and verify that there are rows in the table
109    MasterProcedureTestingUtility.loadData(
110      UTIL.getConnection(), tableName, 100, splitKeys, families);
111    assertEquals(100, UTIL.countRows(tableName));
112    // disable the table
113    UTIL.getAdmin().disableTable(tableName);
114
115    // truncate the table
116    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
117    long procId = ProcedureTestingUtility.submitAndWait(procExec,
118      new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
119    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
120
121    // If truncate procedure completed successfully, it means all regions were assigned correctly
122    // and table is enabled now.
123    UTIL.waitUntilAllRegionsAssigned(tableName);
124
125    // validate the table regions and layout
126    regions = UTIL.getAdmin().getTableRegions(tableName).toArray(new RegionInfo[0]);
127    if (preserveSplits) {
128      assertEquals(1 + splitKeys.length, regions.length);
129    } else {
130      assertEquals(1, regions.length);
131    }
132    MasterProcedureTestingUtility.validateTableCreation(
133      UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
134
135    // verify that there are no rows in the table
136    assertEquals(0, UTIL.countRows(tableName));
137
138    // verify that the table is read/writable
139    MasterProcedureTestingUtility.loadData(
140      UTIL.getConnection(), tableName, 50, splitKeys, families);
141    assertEquals(50, UTIL.countRows(tableName));
142  }

156  private void testRecoveryAndDoubleExecution(final TableName tableName,
157      final boolean preserveSplits) throws Exception {
158    final String[] families = new String[] { "f1", "f2" };
159
160    // create the table
161    final byte[][] splitKeys = new byte[][] {
162      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
163    };
164    RegionInfo[] regions = MasterProcedureTestingUtility.createTable(
165      getMasterProcedureExecutor(), tableName, splitKeys, families);
166    // load and verify that there are rows in the table
167    MasterProcedureTestingUtility.loadData(
168      UTIL.getConnection(), tableName, 100, splitKeys, families);
169    assertEquals(100, UTIL.countRows(tableName));
170    // disable the table
171    UTIL.getAdmin().disableTable(tableName);
172
173    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
174    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
175    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
176
177    // Start the Truncate procedure && kill the executor
178    long procId = procExec.submitProcedure(
179      new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
180
181    // Restart the executor and execute the step twice
182    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
183
184    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
185    UTIL.waitUntilAllRegionsAssigned(tableName);
186
187    // validate the table regions and layout
188    regions = UTIL.getAdmin().getTableRegions(tableName).toArray(new RegionInfo[0]);
189    if (preserveSplits) {
190      assertEquals(1 + splitKeys.length, regions.length);
191    } else {
192      assertEquals(1, regions.length);
193    }
194    MasterProcedureTestingUtility.validateTableCreation(
195      UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
196
197    // verify that there are no rows in the table
198    assertEquals(0, UTIL.countRows(tableName));
199
200    // verify that the table is read/writable
201    MasterProcedureTestingUtility.loadData(
202      UTIL.getConnection(), tableName, 50, splitKeys, families);
203    assertEquals(50, UTIL.countRows(tableName));
204  }
