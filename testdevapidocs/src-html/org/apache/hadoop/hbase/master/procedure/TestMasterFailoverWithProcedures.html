100  @Test
101  public void testWalRecoverLease() throws Exception {
102    final ProcedureStore masterStore = getMasterProcedureExecutor().getStore();
103    assertTrue("expected WALStore for this test", masterStore instanceof WALProcedureStore);
104
105    HMaster firstMaster = UTIL.getHBaseCluster().getMaster();
106    // Abort Latch for the master store
107    final CountDownLatch masterStoreAbort = new CountDownLatch(1);
108    masterStore.registerListener(new ProcedureStore.ProcedureStoreListener() {
109      @Override
110      public void postSync() {}
111
112      @Override
113      public void abortProcess() {
114        LOG.debug("Abort store of Master");
115        masterStoreAbort.countDown();
116      }
117    });
118
119    // startup a fake master the new WAL store will take the lease
120    // and the active master should abort.
121    HMaster backupMaster3 = Mockito.mock(HMaster.class);
122    Mockito.doReturn(firstMaster.getConfiguration()).when(backupMaster3).getConfiguration();
123    Mockito.doReturn(true).when(backupMaster3).isActiveMaster();
124    final WALProcedureStore backupStore3 = new WALProcedureStore(firstMaster.getConfiguration(),
125        firstMaster.getMasterFileSystem().getFileSystem(),
126        ((WALProcedureStore)masterStore).getLogDir(),
127        new MasterProcedureEnv.WALStoreLeaseRecovery(backupMaster3));
128    // Abort Latch for the test store
129    final CountDownLatch backupStore3Abort = new CountDownLatch(1);
130    backupStore3.registerListener(new ProcedureStore.ProcedureStoreListener() {
131      @Override
132      public void postSync() {}
133
134      @Override
135      public void abortProcess() {
136        LOG.debug("Abort store of backupMaster3");
137        backupStore3Abort.countDown();
138        backupStore3.stop(true);
139      }
140    });
141    backupStore3.start(1);
142    backupStore3.recoverLease();
143
144    // Try to trigger a command on the master (WAL lease expired on the active one)
145    HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(TableName.valueOf("mtb"), "f");
146    HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, null);
147    LOG.debug("submit proc");
148    try {
149      getMasterProcedureExecutor().submitProcedure(
150          new CreateTableProcedure(getMasterProcedureExecutor().getEnvironment(), htd, regions));
151      fail("expected RuntimeException 'sync aborted'");
152    } catch (RuntimeException e) {
153      LOG.info("got " + e.getMessage());
154    }
155    LOG.debug("wait master store abort");
156    masterStoreAbort.await();
157
158    // Now the real backup master should start up
159    LOG.debug("wait backup master to startup");
160    waitBackupMaster(UTIL, firstMaster);
161    assertEquals(true, firstMaster.isStopped());
162
163    // wait the store in here to abort (the test will fail due to timeout if it doesn't)
164    LOG.debug("wait the store to abort");
165    backupStore3.getStoreTracker().setDeleted(1, false);
166    try {
167      backupStore3.delete(1);
168      fail("expected RuntimeException 'sync aborted'");
169    } catch (RuntimeException e) {
170      LOG.info("got " + e.getMessage());
171    }
172    backupStore3Abort.await();
173  }

192  public void testWALfencing(boolean walRolls) throws IOException {
193    final ProcedureStore procStore = getMasterProcedureExecutor().getStore();
194    assertTrue("expected WALStore for this test", procStore instanceof WALProcedureStore);
195
196    HMaster firstMaster = UTIL.getHBaseCluster().getMaster();
197
198    // cause WAL rolling after a delete in WAL:
199    firstMaster.getConfiguration().setLong(WALProcedureStore.ROLL_THRESHOLD_CONF_KEY, 1);
200
201    HMaster backupMaster3 = Mockito.mock(HMaster.class);
202    Mockito.doReturn(firstMaster.getConfiguration()).when(backupMaster3).getConfiguration();
203    Mockito.doReturn(true).when(backupMaster3).isActiveMaster();
204    final WALProcedureStore procStore2 = new WALProcedureStore(firstMaster.getConfiguration(),
205        firstMaster.getMasterFileSystem().getFileSystem(),
206        ((WALProcedureStore)procStore).getLogDir(),
207        new MasterProcedureEnv.WALStoreLeaseRecovery(backupMaster3));
208
209    // start a second store which should fence the first one out
210    LOG.info("Starting new WALProcedureStore");
211    procStore2.start(1);
212    procStore2.recoverLease();
213
214    // before writing back to the WAL store, optionally do a couple of WAL rolls (which causes
215    // to delete the old WAL files).
216    if (walRolls) {
217      LOG.info("Inserting into second WALProcedureStore, causing WAL rolls");
218      for (int i = 0; i < 512; i++) {
219        // insert something to the second store then delete it, causing a WAL roll(s)
220        Procedure proc2 = new TestProcedure(i);
221        procStore2.insert(proc2, null);
222        procStore2.delete(proc2.getProcId()); // delete the procedure so that the WAL is removed later
223      }
224    }
225
226    // Now, insert something to the first store, should fail.
227    // If the store does a WAL roll and continue with another logId without checking higher logIds
228    // it will incorrectly succeed.
229    LOG.info("Inserting into first WALProcedureStore");
230    try {
231      procStore.insert(new TestProcedure(11), null);
232      fail("Inserting into Procedure Store should have failed");
233    } catch (Exception ex) {
234      LOG.info("Received expected exception", ex);
235    }
236  }

322  private void testTruncateWithFailoverAtStep(final boolean preserveSplits, final int step)
323      throws Exception {
324    final TableName tableName = TableName.valueOf("testTruncateWithFailoverAtStep" + step);
325
326    // create the table
327    final String[] families = new String[] { "f1", "f2" };
328    final byte[][] splitKeys = new byte[][] {
329        Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
330    };
331    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
332        getMasterProcedureExecutor(), tableName, splitKeys, families);
333    // load and verify that there are rows in the table
334    MasterProcedureTestingUtility.loadData(
335        UTIL.getConnection(), tableName, 100, splitKeys, families);
336    assertEquals(100, UTIL.countRows(tableName));
337    // disable the table
338    UTIL.getHBaseAdmin().disableTable(tableName);
339
340    ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
341    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
342
343    // Start the Truncate procedure && kill the executor
344    long procId = procExec.submitProcedure(
345        new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
346    testRecoveryAndDoubleExecution(UTIL, procId, step, TruncateTableState.values());
347
348    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
349    UTIL.waitUntilAllRegionsAssigned(tableName);
350
351    // validate the table regions and layout
352    if (preserveSplits) {
353      assertEquals(1 + splitKeys.length, UTIL.getHBaseAdmin().getTableRegions(tableName).size());
354    } else {
355      regions = UTIL.getHBaseAdmin().getTableRegions(tableName).toArray(new HRegionInfo[1]);
356      assertEquals(1, regions.length);
357    }
358    MasterProcedureTestingUtility.validateTableCreation(
359        UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
360
361    // verify that there are no rows in the table
362    assertEquals(0, UTIL.countRows(tableName));
363
364    // verify that the table is read/writable
365    MasterProcedureTestingUtility.loadData(
366        UTIL.getConnection(), tableName, 50, splitKeys, families);
367    assertEquals(50, UTIL.countRows(tableName));
368  }

444  //  Test Helpers
445  // ==========================================================================
446  public static <TState> void testRecoveryAndDoubleExecution(final HBaseTestingUtility testUtil,
447      final long procId, final int lastStepBeforeFailover, TState[] states) throws Exception {
448    ProcedureExecutor<MasterProcedureEnv> procExec =
449        testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
450    ProcedureTestingUtility.waitProcedure(procExec, procId);
451
452    for (int i = 0; i < lastStepBeforeFailover; ++i) {
453      LOG.info("Restart "+ i +" exec state: " + states[i]);
454      ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
455      ProcedureTestingUtility.restart(procExec);
456      ProcedureTestingUtility.waitProcedure(procExec, procId);
457    }
458    ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
459
460    LOG.info("Trigger master failover");
461    masterFailover(testUtil);
462
463    procExec = testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
464    ProcedureTestingUtility.waitProcedure(procExec, procId);
465    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
466  }

468  // ==========================================================================
469  //  Master failover utils
470  // ==========================================================================
471  public static void masterFailover(final HBaseTestingUtility testUtil)
472      throws Exception {
473    MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
474
475    // Kill the master
476    HMaster oldMaster = cluster.getMaster();
477    cluster.killMaster(cluster.getMaster().getServerName());
478
479    // Wait the secondary
480    waitBackupMaster(testUtil, oldMaster);
481  }
482
483  public static void waitBackupMaster(final HBaseTestingUtility testUtil,
484      final HMaster oldMaster) throws Exception {
485    MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
486
487    HMaster newMaster = cluster.getMaster();
488    while (newMaster == null || newMaster == oldMaster) {
489      Thread.sleep(250);
490      newMaster = cluster.getMaster();
491    }
492
493    while (!(newMaster.isActiveMaster() && newMaster.isInitialized())) {
494      Thread.sleep(250);
495    }
496  }
