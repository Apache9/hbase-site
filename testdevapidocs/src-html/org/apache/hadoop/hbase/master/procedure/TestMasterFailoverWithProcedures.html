102  @Test
103  public void testWalRecoverLease() throws Exception {
104    final ProcedureStore masterStore = getMasterProcedureExecutor().getStore();
105    assertTrue("expected WALStore for this test", masterStore instanceof WALProcedureStore);
106
107    HMaster firstMaster = UTIL.getHBaseCluster().getMaster();
108    // Abort Latch for the master store
109    final CountDownLatch masterStoreAbort = new CountDownLatch(1);
110    masterStore.registerListener(new ProcedureStore.ProcedureStoreListener() {
111      @Override
112      public void postSync() {}
113
114      @Override
115      public void abortProcess() {
116        LOG.debug("Abort store of Master");
117        masterStoreAbort.countDown();
118      }
119    });
120
121    // startup a fake master the new WAL store will take the lease
122    // and the active master should abort.
123    HMaster backupMaster3 = Mockito.mock(HMaster.class);
124    Mockito.doReturn(firstMaster.getConfiguration()).when(backupMaster3).getConfiguration();
125    Mockito.doReturn(true).when(backupMaster3).isActiveMaster();
126    final WALProcedureStore backupStore3 = new WALProcedureStore(firstMaster.getConfiguration(),
127        firstMaster.getMasterFileSystem().getFileSystem(),
128        ((WALProcedureStore)masterStore).getLogDir(),
129        new MasterProcedureEnv.WALStoreLeaseRecovery(backupMaster3));
130    // Abort Latch for the test store
131    final CountDownLatch backupStore3Abort = new CountDownLatch(1);
132    backupStore3.registerListener(new ProcedureStore.ProcedureStoreListener() {
133      @Override
134      public void postSync() {}
135
136      @Override
137      public void abortProcess() {
138        LOG.debug("Abort store of backupMaster3");
139        backupStore3Abort.countDown();
140        backupStore3.stop(true);
141      }
142    });
143    backupStore3.start(1);
144    backupStore3.recoverLease();
145
146    // Try to trigger a command on the master (WAL lease expired on the active one)
147    HTableDescriptor htd = MasterProcedureTestingUtility.createHTD(TableName.valueOf("mtb"), "f");
148    HRegionInfo[] regions = ModifyRegionUtils.createHRegionInfos(htd, null);
149    LOG.debug("submit proc");
150    try {
151      getMasterProcedureExecutor().submitProcedure(
152          new CreateTableProcedure(getMasterProcedureExecutor().getEnvironment(), htd, regions));
153      fail("expected RuntimeException 'sync aborted'");
154    } catch (RuntimeException e) {
155      LOG.info("got " + e.getMessage());
156    }
157    LOG.debug("wait master store abort");
158    masterStoreAbort.await();
159
160    // Now the real backup master should start up
161    LOG.debug("wait backup master to startup");
162    waitBackupMaster(UTIL, firstMaster);
163    assertEquals(true, firstMaster.isStopped());
164
165    // wait the store in here to abort (the test will fail due to timeout if it doesn't)
166    LOG.debug("wait the store to abort");
167    backupStore3.getStoreTracker().setDeleted(1, false);
168    try {
169      backupStore3.delete(1);
170      fail("expected RuntimeException 'sync aborted'");
171    } catch (RuntimeException e) {
172      LOG.info("got " + e.getMessage());
173    }
174    backupStore3Abort.await();
175  }

194  public void testWALfencing(boolean walRolls) throws IOException {
195    final ProcedureStore procStore = getMasterProcedureExecutor().getStore();
196    assertTrue("expected WALStore for this test", procStore instanceof WALProcedureStore);
197
198    HMaster firstMaster = UTIL.getHBaseCluster().getMaster();
199
200    // cause WAL rolling after a delete in WAL:
201    firstMaster.getConfiguration().setLong(WALProcedureStore.ROLL_THRESHOLD_CONF_KEY, 1);
202
203    HMaster backupMaster3 = Mockito.mock(HMaster.class);
204    Mockito.doReturn(firstMaster.getConfiguration()).when(backupMaster3).getConfiguration();
205    Mockito.doReturn(true).when(backupMaster3).isActiveMaster();
206    final WALProcedureStore procStore2 = new WALProcedureStore(firstMaster.getConfiguration(),
207        firstMaster.getMasterFileSystem().getFileSystem(),
208        ((WALProcedureStore)procStore).getLogDir(),
209        new MasterProcedureEnv.WALStoreLeaseRecovery(backupMaster3));
210
211    // start a second store which should fence the first one out
212    LOG.info("Starting new WALProcedureStore");
213    procStore2.start(1);
214    procStore2.recoverLease();
215
216    // before writing back to the WAL store, optionally do a couple of WAL rolls (which causes
217    // to delete the old WAL files).
218    if (walRolls) {
219      LOG.info("Inserting into second WALProcedureStore, causing WAL rolls");
220      for (int i = 0; i < 512; i++) {
221        // insert something to the second store then delete it, causing a WAL roll(s)
222        Procedure proc2 = new TestProcedure(i);
223        procStore2.insert(proc2, null);
224        procStore2.delete(proc2.getProcId()); // delete the procedure so that the WAL is removed later
225      }
226    }
227
228    // Now, insert something to the first store, should fail.
229    // If the store does a WAL roll and continue with another logId without checking higher logIds
230    // it will incorrectly succeed.
231    LOG.info("Inserting into first WALProcedureStore");
232    try {
233      procStore.insert(new TestProcedure(11), null);
234      fail("Inserting into Procedure Store should have failed");
235    } catch (Exception ex) {
236      LOG.info("Received expected exception", ex);
237    }
238  }

324  private void testTruncateWithFailoverAtStep(final boolean preserveSplits, final int step)
325      throws Exception {
326    final TableName tableName = TableName.valueOf("testTruncateWithFailoverAtStep" + step);
327
328    // create the table
329    final String[] families = new String[] { "f1", "f2" };
330    final byte[][] splitKeys = new byte[][] {
331        Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
332    };
333    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
334        getMasterProcedureExecutor(), tableName, splitKeys, families);
335    // load and verify that there are rows in the table
336    MasterProcedureTestingUtility.loadData(
337        UTIL.getConnection(), tableName, 100, splitKeys, families);
338    assertEquals(100, UTIL.countRows(tableName));
339    // disable the table
340    UTIL.getHBaseAdmin().disableTable(tableName);
341
342    ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
343    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
344
345    // Start the Truncate procedure && kill the executor
346    long procId = procExec.submitProcedure(
347        new TruncateTableProcedure(procExec.getEnvironment(), tableName, preserveSplits));
348    testRecoveryAndDoubleExecution(UTIL, procId, step, TruncateTableState.values());
349
350    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, false);
351    UTIL.waitUntilAllRegionsAssigned(tableName);
352
353    // validate the table regions and layout
354    if (preserveSplits) {
355      assertEquals(1 + splitKeys.length, UTIL.getHBaseAdmin().getTableRegions(tableName).size());
356    } else {
357      regions = UTIL.getHBaseAdmin().getTableRegions(tableName).toArray(new HRegionInfo[1]);
358      assertEquals(1, regions.length);
359    }
360    MasterProcedureTestingUtility.validateTableCreation(
361        UTIL.getHBaseCluster().getMaster(), tableName, regions, families);
362
363    // verify that there are no rows in the table
364    assertEquals(0, UTIL.countRows(tableName));
365
366    // verify that the table is read/writable
367    MasterProcedureTestingUtility.loadData(
368        UTIL.getConnection(), tableName, 50, splitKeys, families);
369    assertEquals(50, UTIL.countRows(tableName));
370  }

446  //  Test Helpers
447  // ==========================================================================
448  public static <TState> void testRecoveryAndDoubleExecution(final HBaseTestingUtility testUtil,
449      final long procId, final int lastStepBeforeFailover, TState[] states) throws Exception {
450    ProcedureExecutor<MasterProcedureEnv> procExec =
451        testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
452    ProcedureTestingUtility.waitProcedure(procExec, procId);
453
454    for (int i = 0; i < lastStepBeforeFailover; ++i) {
455      LOG.info("Restart "+ i +" exec state: " + states[i]);
456      ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
457      ProcedureTestingUtility.restart(procExec);
458      ProcedureTestingUtility.waitProcedure(procExec, procId);
459    }
460    ProcedureTestingUtility.assertProcNotYetCompleted(procExec, procId);
461
462    LOG.info("Trigger master failover");
463    masterFailover(testUtil);
464
465    procExec = testUtil.getHBaseCluster().getMaster().getMasterProcedureExecutor();
466    ProcedureTestingUtility.waitProcedure(procExec, procId);
467    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
468  }

470  // ==========================================================================
471  //  Master failover utils
472  // ==========================================================================
473  public static void masterFailover(final HBaseTestingUtility testUtil)
474      throws Exception {
475    MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
476
477    // Kill the master
478    HMaster oldMaster = cluster.getMaster();
479    cluster.killMaster(cluster.getMaster().getServerName());
480
481    // Wait the secondary
482    waitBackupMaster(testUtil, oldMaster);
483  }
484
485  public static void waitBackupMaster(final HBaseTestingUtility testUtil,
486      final HMaster oldMaster) throws Exception {
487    MiniHBaseCluster cluster = testUtil.getMiniHBaseCluster();
488
489    HMaster newMaster = cluster.getMaster();
490    while (newMaster == null || newMaster == oldMaster) {
491      Thread.sleep(250);
492      newMaster = cluster.getMaster();
493    }
494
495    while (!(newMaster.isActiveMaster() && newMaster.isInitialized())) {
496      Thread.sleep(250);
497    }
498  }
