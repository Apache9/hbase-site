127  private long getSCPProcId(ProcedureExecutor<?> procExec) {
128    util.waitFor(30000, () -> !procExec.getProcedures().isEmpty());
129    return procExec.getActiveProcIds().stream().mapToLong(Long::longValue).min().getAsLong();
130  }

132  /**
133   * Run server crash procedure steps twice to test idempotency and that we are persisting all
134   * needed state.
135   */
136  private void testRecoveryAndDoubleExecution(boolean carryingMeta, boolean doubleExecution)
137      throws Exception {
138    final TableName tableName = TableName.valueOf("testRecoveryAndDoubleExecution-carryingMeta-"
139        + carryingMeta + "-doubleExecution-" + doubleExecution);
140    try (Table t = createTable(tableName)) {
141      // Load the table with a bit of data so some logs to split and some edits in each region.
142      this.util.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
143      final int count = util.countRows(t);
144      assertTrue("expected some rows", count > 0);
145      final String checksum = util.checksumRows(t);
146      // Run the procedure executor outside the master so we can mess with it. Need to disable
147      // Master's running of the server crash processing.
148      final HMaster master = this.util.getHBaseCluster().getMaster();
149      final ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
150      // find the first server that match the request and executes the test
151      ServerName rsToKill = null;
152      for (RegionInfo hri : util.getAdmin().getRegions(tableName)) {
153        final ServerName serverName = AssignmentTestingUtil.getServerHoldingRegion(util, hri);
154        if (AssignmentTestingUtil.isServerHoldingMeta(util, serverName) == carryingMeta) {
155          rsToKill = serverName;
156          break;
157        }
158      }
159      // Enable test flags and then queue the crash procedure.
160      ProcedureTestingUtility.waitNoProcedureRunning(procExec);
161      if (doubleExecution) {
162        // For SCP, if you enable this then we will enter an infinite loop, as we will crash between
163        // queue and open for TRSP, and then going back to queue, as we will use the crash rs as the
164        // target server since it is recored in hbase:meta.
165        ProcedureTestingUtility.setKillIfHasParent(procExec, false);
166        ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
167        // kill the RS
168        AssignmentTestingUtil.killRs(util, rsToKill);
169        long procId = getSCPProcId(procExec);
170        // Now run through the procedure twice crashing the executor on each step...
171        MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
172      } else {
173        // kill the RS
174        AssignmentTestingUtil.killRs(util, rsToKill);
175        long procId = getSCPProcId(procExec);
176        ProcedureTestingUtility.waitProcedure(procExec, procId);
177      }
178      assertReplicaDistributed(t);
179      assertEquals(count, util.countRows(t));
180      assertEquals(checksum, util.checksumRows(t));
181    } catch (Throwable throwable) {
182      LOG.error("Test failed!", throwable);
183      throw throwable;
184    }
185  }

187  @Test
188  public void testConcurrentSCPForSameServer() throws Exception {
189    final TableName tableName =
190        TableName.valueOf("testConcurrentSCPForSameServer-" + splitWALCoordinatedByZK);
191    try (Table t = createTable(tableName)) {
192      // Load the table with a bit of data so some logs to split and some edits in each region.
193      this.util.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
194      final int count = util.countRows(t);
195      assertTrue("expected some rows", count > 0);
196      // find the first server that match the request and executes the test
197      ServerName rsToKill = null;
198      for (RegionInfo hri : util.getAdmin().getRegions(tableName)) {
199        final ServerName serverName = AssignmentTestingUtil.getServerHoldingRegion(util, hri);
200        if (AssignmentTestingUtil.isServerHoldingMeta(util, serverName) == true) {
201          rsToKill = serverName;
202          break;
203        }
204      }
205      HMaster master = util.getHBaseCluster().getMaster();
206      final ProcedureExecutor<MasterProcedureEnv> pExecutor = master.getMasterProcedureExecutor();
207      ServerCrashProcedure procB =
208          new ServerCrashProcedure(pExecutor.getEnvironment(), rsToKill, false, false);
209      AssignmentTestingUtil.killRs(util, rsToKill);
210      long procId = getSCPProcId(pExecutor);
211      Procedure procA = pExecutor.getProcedure(procId);
212      LOG.info("submit SCP procedureA");
213      util.waitFor(5000, () -> procA.hasLock());
214      LOG.info("procedureA acquired the lock");
215      assertEquals(Procedure.LockState.LOCK_EVENT_WAIT,
216          procB.acquireLock(pExecutor.getEnvironment()));
217      LOG.info("procedureB should not be able to get the lock");
218      util.waitFor(60000,
219        () -> procB.acquireLock(pExecutor.getEnvironment()) == Procedure.LockState.LOCK_ACQUIRED);
220      LOG.info("when procedure B get the lock, procedure A should be finished");
221      assertTrue(procA.isFinished());
222    }
223  }
