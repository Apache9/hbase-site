047/**
048 * It used to first run with DLS and then DLR but HBASE-12751 broke DLR so we disabled it here.
049 */
050@Category(LargeTests.class)
051@RunWith(Parameterized.class)
052public class TestServerCrashProcedure {
053  // Ugly junit parameterization. I just want to pass false and then true but seems like needs
054  // to return sequences of two-element arrays.
055  @Parameters(name = "{index}: setting={0}")
056  public static Collection<Object []> data() {
057    return Arrays.asList(new Object[] [] {{Boolean.FALSE, -1}});
058  }
059
060  private final HBaseTestingUtility util = new HBaseTestingUtility();
061
062  @Before
063  public void setup() throws Exception {
064    this.util.startMiniCluster(3);
065    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(
066      this.util.getHBaseCluster().getMaster().getMasterProcedureExecutor(), false);
067  }
068
069  @After
070  public void tearDown() throws Exception {
071    MiniHBaseCluster cluster = this.util.getHBaseCluster();
072    HMaster master = cluster == null? null: cluster.getMaster();
073    if (master != null && master.getMasterProcedureExecutor() != null) {
074      ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(master.getMasterProcedureExecutor(),
075        false);
076    }
077    this.util.shutdownMiniCluster();
078  }
079
080  public TestServerCrashProcedure(final Boolean b, final int ignore) {
081    this.util.getConfiguration().setBoolean("hbase.master.distributed.log.replay", b);
082    this.util.getConfiguration().setInt(MasterProcedureConstants.MASTER_PROCEDURE_THREADS, 1);
083  }
084
085  /**
086   * Run server crash procedure steps twice to test idempotency and that we are persisting all
087   * needed state.
088   * @throws Exception
089   */
090  @Test(timeout = 300000)
091  public void testRecoveryAndDoubleExecutionOnline() throws Exception {
092    final TableName tableName = TableName.valueOf("testRecoveryAndDoubleExecutionOnline");
093    this.util.createTable(tableName, HBaseTestingUtility.COLUMNS,
094      HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
095    try (Table t = this.util.getConnection().getTable(tableName)) {
096      // Load the table with a bit of data so some logs to split and some edits in each region.
097      this.util.loadTable(t, HBaseTestingUtility.COLUMNS[0]);
098      int count = util.countRows(t);
099      // Run the procedure executor outside the master so we can mess with it. Need to disable
100      // Master's running of the server crash processing.
101      HMaster master = this.util.getHBaseCluster().getMaster();
102      final ProcedureExecutor<MasterProcedureEnv> procExec = master.getMasterProcedureExecutor();
103      master.setServerCrashProcessingEnabled(false);
104      // Kill a server. Master will notice but do nothing other than add it to list of dead servers.
105      HRegionServer hrs = this.util.getHBaseCluster().getRegionServer(0);
106      boolean carryingMeta = master.getAssignmentManager().isCarryingMeta(hrs.getServerName());
107      this.util.getHBaseCluster().killRegionServer(hrs.getServerName());
108      hrs.join();
109      // Wait until the expiration of the server has arrived at the master. We won't process it
110      // by queuing a ServerCrashProcedure because we have disabled crash processing... but wait
111      // here so ServerManager gets notice and adds expired server to appropriate queues.
112      while (!master.getServerManager().isServerDead(hrs.getServerName())) Threads.sleep(10);
113      // Now, reenable processing else we can't get a lock on the ServerCrashProcedure.
114      master.setServerCrashProcessingEnabled(true);
115      // Do some of the master processing of dead servers so when SCP runs, it has expected 'state'.
116      master.getServerManager().moveFromOnlineToDeadServers(hrs.getServerName());
117      // Enable test flags and then queue the crash procedure.
118      ProcedureTestingUtility.waitNoProcedureRunning(procExec);
119      ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
120      long procId =
121        procExec.submitProcedure(new ServerCrashProcedure(
122          procExec.getEnvironment(), hrs.getServerName(), true, carryingMeta));
123      // Now run through the procedure twice crashing the executor on each step...
124      MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
125      // Assert all data came back.
126      assertEquals(count, util.countRows(t));
127    }
128  }
