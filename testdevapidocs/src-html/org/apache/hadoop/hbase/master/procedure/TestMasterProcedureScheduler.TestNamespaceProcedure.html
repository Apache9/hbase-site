073  @Test
074  public void testSimpleTableOpsQueues() throws Exception {
075    final int NUM_TABLES = 10;
076    final int NUM_ITEMS = 10;
077
078    int count = 0;
079    for (int i = 1; i <= NUM_TABLES; ++i) {
080      TableName tableName = TableName.valueOf(String.format("test-%04d", i));
081      // insert items
082      for (int j = 1; j <= NUM_ITEMS; ++j) {
083        queue.addBack(new TestTableProcedure(i * 1000 + j, tableName,
084          TableProcedureInterface.TableOperationType.EDIT));
085        assertEquals(++count, queue.size());
086      }
087    }
088    assertEquals(NUM_TABLES * NUM_ITEMS, queue.size());
089
090    for (int j = 1; j <= NUM_ITEMS; ++j) {
091      for (int i = 1; i <= NUM_TABLES; ++i) {
092        Procedure proc = queue.poll();
093        assertTrue(proc != null);
094        TableName tableName = ((TestTableProcedure)proc).getTableName();
095        queue.tryAcquireTableExclusiveLock(proc, tableName);
096        queue.releaseTableExclusiveLock(proc, tableName);
097        queue.completionCleanup(proc);
098        assertEquals(--count, queue.size());
099        assertEquals(i * 1000 + j, proc.getProcId());
100      }
101    }
102    assertEquals(0, queue.size());
103
104    for (int i = 1; i <= NUM_TABLES; ++i) {
105      final TableName tableName = TableName.valueOf(String.format("test-%04d", i));
106      final TestTableProcedure dummyProc = new TestTableProcedure(100, tableName,
107        TableProcedureInterface.TableOperationType.DELETE);
108      // complete the table deletion
109      assertTrue(queue.markTableAsDeleted(tableName, dummyProc));
110    }

117  @Test

120

144  /**
145   * Check that the table queue is not deletable until every procedure
146   * in-progress is completed (this is a special case for read-locks).
147   */
148  @Test
149  public void testCreateDeleteTableOperationsWithReadLock() throws Exception {
150    final TableName tableName = TableName.valueOf("testtb");
151    final int nitems = 2;
152
153    final TestTableProcedure dummyProc = new TestTableProcedure(100, tableName,
154        TableProcedureInterface.TableOperationType.DELETE);
155
156    for (int i = 1; i <= nitems; ++i) {
157      queue.addBack(new TestTableProcedure(i, tableName,
158            TableProcedureInterface.TableOperationType.READ));
159    }
160
161    // table can't be deleted because one item is in the queue
162    assertFalse(queue.markTableAsDeleted(tableName, dummyProc));
163
164    Procedure[] procs = new Procedure[nitems];
165    for (int i = 0; i < nitems; ++i) {
166      // fetch item and take a lock
167      Procedure proc = procs[i] = queue.poll();
168      assertEquals(i + 1, proc.getProcId());
169      // take the rlock
170      assertTrue(queue.tryAcquireTableSharedLock(proc, tableName));
171      // table can't be deleted because we have locks and/or items in the queue
172      assertFalse(queue.markTableAsDeleted(tableName, dummyProc));
173    }
174
175    for (int i = 0; i < nitems; ++i) {
176      // table can't be deleted because we have locks
177      assertFalse(queue.markTableAsDeleted(tableName, dummyProc));
178      // release the rlock
179      queue.releaseTableSharedLock(procs[i], tableName);
180    }
181
182    // there are no items and no lock in the queeu
183    assertEquals(0, queue.size());
184    // complete the table deletion
185    assertTrue(queue.markTableAsDeleted(tableName, dummyProc));
186  }

189   * Verify the correct logic of RWLocks on the queue
190   */
191  @Test
192  public void testVerifyRwLocks() throws Exception {
193    TableName tableName = TableName.valueOf("testtb");
194    queue.addBack(new TestTableProcedure(1, tableName,
195          TableProcedureInterface.TableOperationType.EDIT));
196    queue.addBack(new TestTableProcedure(2, tableName,
197          TableProcedureInterface.TableOperationType.READ));
198    queue.addBack(new TestTableProcedure(3, tableName,
199          TableProcedureInterface.TableOperationType.EDIT));
200    queue.addBack(new TestTableProcedure(4, tableName,
201          TableProcedureInterface.TableOperationType.READ));
202    queue.addBack(new TestTableProcedure(5, tableName,
203          TableProcedureInterface.TableOperationType.READ));
204
205    // Fetch the 1st item and take the write lock
206    Procedure proc = queue.poll();
207    assertEquals(1, proc.getProcId());
208    assertEquals(true, queue.tryAcquireTableExclusiveLock(proc, tableName));
209
210    // Fetch the 2nd item and verify that the lock can't be acquired
211    assertEquals(null, queue.poll(0));
212
213    // Release the write lock and acquire the read lock
214    queue.releaseTableExclusiveLock(proc, tableName);
215
216    // Fetch the 2nd item and take the read lock
217    Procedure rdProc = queue.poll();
218    assertEquals(2, rdProc.getProcId());
219    assertEquals(true, queue.tryAcquireTableSharedLock(rdProc, tableName));
220
221    // Fetch the 3rd item and verify that the lock can't be acquired
222    assertEquals(null, queue.poll(0));
223
224    // release the rdlock of item 2 and take the wrlock for the 3d item
225    queue.releaseTableSharedLock(rdProc, tableName);
226
227    // Fetch the 3rd item and take the write lock
228    Procedure wrProc = queue.poll();
229    assertEquals(true, queue.tryAcquireTableExclusiveLock(wrProc, tableName));
230
231    // Fetch 4th item and verify that the lock can't be acquired
232    assertEquals(null, queue.poll(0));
233
234    // Release the write lock and acquire the read lock
235    queue.releaseTableExclusiveLock(wrProc, tableName);
236
237    // Fetch the 4th item and take the read lock
238    rdProc = queue.poll();
239    assertEquals(4, rdProc.getProcId());
240    assertEquals(true, queue.tryAcquireTableSharedLock(rdProc, tableName));
241
242    // Fetch the 4th item and take the read lock
243    Procedure rdProc2 = queue.poll();
244    assertEquals(5, rdProc2.getProcId());
245    assertEquals(true, queue.tryAcquireTableSharedLock(rdProc2, tableName));
246
247    // Release 4th and 5th read-lock
248    queue.releaseTableSharedLock(rdProc, tableName);
249    queue.releaseTableSharedLock(rdProc2, tableName);
250
251    // remove table queue
252    assertEquals(0, queue.size());
253    assertTrue("queue should be deleted", queue.markTableAsDeleted(tableName, wrProc));
254  }

257  public void testVerifyNamespaceRwLocks() throws Exception {
258    String nsName1 = "ns1";
259    String nsName2 = "ns2";
260    TableName tableName1 = TableName.valueOf(nsName1, "testtb");
261    TableName tableName2 = TableName.valueOf(nsName2, "testtb");
262    queue.addBack(new TestNamespaceProcedure(1, nsName1,
263          TableProcedureInterface.TableOperationType.EDIT));
264    queue.addBack(new TestTableProcedure(2, tableName1,
265          TableProcedureInterface.TableOperationType.EDIT));
266    queue.addBack(new TestTableProcedure(3, tableName2,
267          TableProcedureInterface.TableOperationType.EDIT));
268    queue.addBack(new TestNamespaceProcedure(4, nsName2,
269          TableProcedureInterface.TableOperationType.EDIT));
270
271    // Fetch the 1st item and take the write lock
272    Procedure procNs1 = queue.poll();
273    assertEquals(1, procNs1.getProcId());
274    assertEquals(true, queue.tryAcquireNamespaceExclusiveLock(procNs1, nsName1));
275
276    // System tables have 2 as default priority
277    Procedure procNs2 = queue.poll();
278    assertEquals(4, procNs2.getProcId());
279    assertEquals(true, queue.tryAcquireNamespaceExclusiveLock(procNs2, nsName2));
280    queue.releaseNamespaceExclusiveLock(procNs2, nsName2);
281    queue.yield(procNs2);
282
283    // table on ns1 is locked, so we get table on ns2
284    procNs2 = queue.poll();
285    assertEquals(3, procNs2.getProcId());
286    assertEquals(true, queue.tryAcquireTableExclusiveLock(procNs2, tableName2));
287
288    // ns2 is not available (TODO we may avoid this one)
289    Procedure procNs2b = queue.poll();
290    assertEquals(4, procNs2b.getProcId());
291    assertEquals(false, queue.tryAcquireNamespaceExclusiveLock(procNs2b, nsName2));
292    queue.yield(procNs2b);
293
294    // release the ns1 lock
295    queue.releaseNamespaceExclusiveLock(procNs1, nsName1);
296
297    // we are now able to execute table of ns1
298    long procId = queue.poll().getProcId();
299    assertEquals(2, procId);
300
301    queue.releaseTableExclusiveLock(procNs2, tableName2);
302
303    // we are now able to execute ns2
304    procId = queue.poll().getProcId();
305    assertEquals(4, procId);
306  }

410  @Test
411  public void testVerifyRegionLocks() throws Exception {
412    final TableName tableName = TableName.valueOf("testtb");
413    final HRegionInfo regionA = new HRegionInfo(tableName, Bytes.toBytes("a"), Bytes.toBytes("b"));
414    final HRegionInfo regionB = new HRegionInfo(tableName, Bytes.toBytes("b"), Bytes.toBytes("c"));
415    final HRegionInfo regionC = new HRegionInfo(tableName, Bytes.toBytes("c"), Bytes.toBytes("d"));
416
417    queue.addBack(new TestTableProcedure(1, tableName,
418          TableProcedureInterface.TableOperationType.EDIT));
419    queue.addBack(new TestRegionProcedure(2, tableName,
420        TableProcedureInterface.TableOperationType.MERGE, regionA, regionB));
421    queue.addBack(new TestRegionProcedure(3, tableName,
422        TableProcedureInterface.TableOperationType.SPLIT, regionA));
423    queue.addBack(new TestRegionProcedure(4, tableName,
424        TableProcedureInterface.TableOperationType.SPLIT, regionB));
425    queue.addBack(new TestRegionProcedure(5, tableName,
426        TableProcedureInterface.TableOperationType.UNASSIGN, regionC));
427
428    // Fetch the 1st item and take the write lock
429    Procedure proc = queue.poll();
430    assertEquals(1, proc.getProcId());
431    assertEquals(true, queue.tryAcquireTableExclusiveLock(proc, tableName));
432
433    // everything is locked by the table operation
434    assertEquals(null, queue.poll(0));
435
436    // release the table lock
437    queue.releaseTableExclusiveLock(proc, tableName);
438
439    // Fetch the 2nd item and the the lock on regionA and regionB
440    Procedure mergeProc = queue.poll();
441    assertEquals(2, mergeProc.getProcId());
442    assertEquals(false, queue.waitRegions(mergeProc, tableName, regionA, regionB));
443
444    // Fetch the 3rd item and the try to lock region A which will fail
445    // because already locked. this procedure will go in waiting.
446    // (this stuff will be explicit until we get rid of the zk-lock)
447    Procedure procA = queue.poll();
448    assertEquals(3, procA.getProcId());
449    assertEquals(true, queue.waitRegions(procA, tableName, regionA));
450
451    // Fetch the 4th item, same story as the 3rd
452    Procedure procB = queue.poll();
453    assertEquals(4, procB.getProcId());
454    assertEquals(true, queue.waitRegions(procB, tableName, regionB));
455
456    // Fetch the 5th item, since it is a non-locked region we are able to execute it
457    Procedure procC = queue.poll();
458    assertEquals(5, procC.getProcId());
459    assertEquals(false, queue.waitRegions(procC, tableName, regionC));
460
461    // 3rd and 4th are in the region suspended queue
462    assertEquals(null, queue.poll(0));
463
464    // Release region A-B from merge operation (procId=2)
465    queue.wakeRegions(mergeProc, tableName, regionA, regionB);
466
467    // Fetch the 3rd item, now the lock on the region is available
468    procA = queue.poll();
469    assertEquals(3, procA.getProcId());
470    assertEquals(false, queue.waitRegions(procA, tableName, regionA));
471
472    // Fetch the 4th item, now the lock on the region is available
473    procB = queue.poll();
474    assertEquals(4, procB.getProcId());
475    assertEquals(false, queue.waitRegions(procB, tableName, regionB));
476
477    // release the locks on the regions
478    queue.wakeRegions(procA, tableName, regionA);
479    queue.wakeRegions(procB, tableName, regionB);
480    queue.wakeRegions(procC, tableName, regionC);
481  }

483  @Test
484  public void testVerifySubProcRegionLocks() throws Exception {
485    final TableName tableName = TableName.valueOf("testVerifySubProcRegionLocks");
486    final HRegionInfo regionA = new HRegionInfo(tableName, Bytes.toBytes("a"), Bytes.toBytes("b"));
487    final HRegionInfo regionB = new HRegionInfo(tableName, Bytes.toBytes("b"), Bytes.toBytes("c"));
488    final HRegionInfo regionC = new HRegionInfo(tableName, Bytes.toBytes("c"), Bytes.toBytes("d"));
489
490    queue.addBack(new TestTableProcedure(1, tableName,
491        TableProcedureInterface.TableOperationType.ENABLE));
492
493    // Fetch the 1st item from the queue, "the root procedure" and take the table lock
494    Procedure rootProc = queue.poll();
495    assertEquals(1, rootProc.getProcId());
496    assertEquals(true, queue.tryAcquireTableExclusiveLock(rootProc, tableName));
497    assertEquals(null, queue.poll(0));
498
499    // Execute the 1st step of the root-proc.
500    // we should get 3 sub-proc back, one for each region.
501    // (this step is done by the executor/rootProc, we are simulating it)
502    Procedure[] subProcs = new Procedure[] {
503      new TestRegionProcedure(1, 2, tableName,
504        TableProcedureInterface.TableOperationType.REGION_EDIT, regionA),
505      new TestRegionProcedure(1, 3, tableName,
506        TableProcedureInterface.TableOperationType.REGION_EDIT, regionB),
507      new TestRegionProcedure(1, 4, tableName,
508        TableProcedureInterface.TableOperationType.REGION_EDIT, regionC),
509    };
510
511    // at this point the rootProc is going in a waiting state
512    // and the sub-procedures will be added in the queue.
513    // (this step is done by the executor, we are simulating it)
514    for (int i = subProcs.length - 1; i >= 0; --i) {
515      queue.addFront(subProcs[i]);
516    }
517    assertEquals(subProcs.length, queue.size());
518
519    // we should be able to fetch and execute all the sub-procs,
520    // since they are operating on different regions
521    for (int i = 0; i < subProcs.length; ++i) {
522      TestRegionProcedure regionProc = (TestRegionProcedure)queue.poll(0);
523      assertEquals(subProcs[i].getProcId(), regionProc.getProcId());
524      assertEquals(false, queue.waitRegions(regionProc, tableName, regionProc.getRegionInfo()));
525    }
526
527    // nothing else in the queue
528    assertEquals(null, queue.poll(0));
529
530    // release all the region locks
531    for (int i = 0; i < subProcs.length; ++i) {
532      TestRegionProcedure regionProc = (TestRegionProcedure)subProcs[i];
533      queue.wakeRegions(regionProc, tableName, regionProc.getRegionInfo());
534    }
535
536    // nothing else in the queue
537    assertEquals(null, queue.poll(0));
538
539    // release the table lock (for the root procedure)
540    queue.releaseTableExclusiveLock(rootProc, tableName);
541  }

598    // fetch and execute child
599    for (int i = 0; i < regions.length; ++i) {
600      final int regionIdx = regions.length - i - 1;
601      Procedure childProc = queue.poll();
602      LOG.debug("fetch children " + childProc);
603      assertEquals(1 + regionIdx, childProc.getProcId());
604      assertEquals(false, queue.waitRegion(childProc, regions[regionIdx]));
605      queue.wakeRegion(childProc, regions[regionIdx]);
606    }
607
608    // nothing available, until xlock release
609    assertEquals(null, queue.poll(0));
610
611    // release xlock
612    queue.releaseTableExclusiveLock(parentProc, tableName);
613
614    // fetch the other xlock proc
615    Procedure proc = queue.poll();
616    assertEquals(100, proc.getProcId());
617    assertTrue(queue.tryAcquireTableExclusiveLock(proc, tableName));
618    queue.releaseTableExclusiveLock(proc, tableName);
619  }
