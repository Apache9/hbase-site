042@Category({MasterTests.class, MediumTests.class})
043public class TestDeleteTableProcedure extends TestTableDDLProcedureBase {
044  private static final Log LOG = LogFactory.getLog(TestDeleteTableProcedure.class);
045  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().withTimeout(this.getClass()).
046      withLookingForStuckThread(true).build();
047  @Rule public TestName name = new TestName();
048
049  @Test(timeout=60000, expected=TableNotFoundException.class)
050  public void testDeleteNotExistentTable() throws Exception {
051    final TableName tableName = TableName.valueOf(name.getMethodName());
052
053    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
054    ProcedurePrepareLatch latch = new ProcedurePrepareLatch.CompatibilityLatch();
055    long procId = ProcedureTestingUtility.submitAndWait(procExec,
056        new DeleteTableProcedure(procExec.getEnvironment(), tableName, latch));
057    latch.await();
058  }
059
060  @Test(timeout=60000, expected=TableNotDisabledException.class)
061  public void testDeleteNotDisabledTable() throws Exception {
062    final TableName tableName = TableName.valueOf(name.getMethodName());
063
064    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
065    MasterProcedureTestingUtility.createTable(procExec, tableName, null, "f");
066
067    ProcedurePrepareLatch latch = new ProcedurePrepareLatch.CompatibilityLatch();
068    long procId = ProcedureTestingUtility.submitAndWait(procExec,
069        new DeleteTableProcedure(procExec.getEnvironment(), tableName, latch));
070    latch.await();
071  }
072
073  @Test(timeout=60000)
074  public void testDeleteDeletedTable() throws Exception {
075    final TableName tableName = TableName.valueOf(name.getMethodName());
076    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
077
078    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
079      procExec, tableName, null, "f");
080    UTIL.getAdmin().disableTable(tableName);
081
082    // delete the table (that exists)
083    long procId1 = procExec.submitProcedure(
084        new DeleteTableProcedure(procExec.getEnvironment(), tableName));
085    // delete the table (that will no longer exist)
086    long procId2 = procExec.submitProcedure(
087        new DeleteTableProcedure(procExec.getEnvironment(), tableName));
088
089    // Wait the completion
090    ProcedureTestingUtility.waitProcedure(procExec, procId1);
091    ProcedureTestingUtility.waitProcedure(procExec, procId2);
092
093    // First delete should succeed
094    ProcedureTestingUtility.assertProcNotFailed(procExec, procId1);
095    MasterProcedureTestingUtility.validateTableDeletion(getMaster(), tableName);
096
097    // Second delete should fail with TableNotFound
098    Procedure<?> result = procExec.getResult(procId2);
099    assertTrue(result.isFailed());
100    LOG.debug("Delete failed with exception: " + result.getException());
101    assertTrue(ProcedureTestingUtility.getExceptionCause(result) instanceof TableNotFoundException);
102  }
103
104  @Test(timeout=60000)
105  public void testSimpleDelete() throws Exception {
106    final TableName tableName = TableName.valueOf(name.getMethodName());
107    final byte[][] splitKeys = null;
108    testSimpleDelete(tableName, splitKeys);
109  }
110
111  @Test(timeout=60000)
112  public void testSimpleDeleteWithSplits() throws Exception {
113    final TableName tableName = TableName.valueOf(name.getMethodName());
114    final byte[][] splitKeys = new byte[][] {
115      Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c")
116    };
117    testSimpleDelete(tableName, splitKeys);
118  }
119
120  private void testSimpleDelete(final TableName tableName, byte[][] splitKeys) throws Exception {
121    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
122      getMasterProcedureExecutor(), tableName, splitKeys, "f1", "f2");
123    UTIL.getAdmin().disableTable(tableName);
124
125    // delete the table
126    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
127    long procId = ProcedureTestingUtility.submitAndWait(procExec,
128      new DeleteTableProcedure(procExec.getEnvironment(), tableName));
129    ProcedureTestingUtility.assertProcNotFailed(procExec, procId);
130    MasterProcedureTestingUtility.validateTableDeletion(getMaster(), tableName);
131  }
132
133  @Test(timeout=60000)
134  public void testRecoveryAndDoubleExecution() throws Exception {
135    final TableName tableName = TableName.valueOf(name.getMethodName());
136
137    // create the table
138    byte[][] splitKeys = null;
139    HRegionInfo[] regions = MasterProcedureTestingUtility.createTable(
140      getMasterProcedureExecutor(), tableName, splitKeys, "f1", "f2");
141    UTIL.getAdmin().disableTable(tableName);
142
143    final ProcedureExecutor<MasterProcedureEnv> procExec = getMasterProcedureExecutor();
144    ProcedureTestingUtility.waitNoProcedureRunning(procExec);
145    ProcedureTestingUtility.setKillAndToggleBeforeStoreUpdate(procExec, true);
146
147    // Start the Delete procedure && kill the executor
148    long procId = procExec.submitProcedure(
149      new DeleteTableProcedure(procExec.getEnvironment(), tableName));
150
151    // Restart the executor and execute the step twice
152    MasterProcedureTestingUtility.testRecoveryAndDoubleExecution(procExec, procId);
153
154    MasterProcedureTestingUtility.validateTableDeletion(getMaster(), tableName);
155  }
