055  @Test (timeout=10000)
056  public void testCanMakeProgressThoughMetaIsDown()
057  throws IOException, InterruptedException, BrokenBarrierException {
058    MasterServices server = mock(MasterServices.class);
059    when(server.getServerName()).thenReturn(ServerName.valueOf("master,1,1"));
060    Connection connection = mock(ClusterConnection.class);
061    // Set up a table that gets 'stuck' when we try to fetch a row from the meta table.
062    // It is stuck on a CyclicBarrier latch. We use CyclicBarrier because it will tell us when
063    // thread is waiting on latch.
064    Table metaTable = Mockito.mock(Table.class);
065    final CyclicBarrier latch = new CyclicBarrier(2);
066    when(metaTable.get((Get)Mockito.any())).thenAnswer(new Answer<Result>() {
067      @Override
068      public Result answer(InvocationOnMock invocation) throws Throwable {
069        latch.await();
070        throw new java.net.ConnectException("Connection refused");
071      }
072    });
073    when(connection.getTable(TableName.META_TABLE_NAME)).thenReturn(metaTable);
074    when(server.getConnection()).thenReturn((ClusterConnection)connection);
075    Configuration configuration = mock(Configuration.class);
076    when(server.getConfiguration()).thenReturn(configuration);
077    TableStateManager tsm = mock(TableStateManager.class);
078    ServerManager sm = mock(ServerManager.class);
079    when(sm.isServerOnline(isA(ServerName.class))).thenReturn(true);
080
081    RegionStateStore rss = mock(RegionStateStore.class);
082    final RegionStates regionStates = new RegionStates(server, tsm, sm, rss);
083    final ServerName sn = mockServer("one", 1);
084    regionStates.updateRegionState(HRegionInfo.FIRST_META_REGIONINFO, State.SPLITTING_NEW, sn);
085    Thread backgroundThread = new Thread("Get stuck setting server offline") {
086      @Override
087      public void run() {
088        regionStates.serverOffline(sn);
089      }
090    };
091    assertTrue(latch.getNumberWaiting() == 0);
092    backgroundThread.start();
093    while (latch.getNumberWaiting() == 0);
094    // Verify I can do stuff with synchronized RegionStates methods, that I am not locked out.
095    // Below is a call that is synchronized.  Can I do it and not block?
096    regionStates.getRegionServerOfRegion(HRegionInfo.FIRST_META_REGIONINFO);
097    // Done. Trip the barrier on the background thread.
098    latch.await();
099  }

101  @Test
102  public void testWeDontReturnDrainingServersForOurBalancePlans() throws Exception {
103    MasterServices server = mock(MasterServices.class);
104    when(server.getServerName()).thenReturn(ServerName.valueOf("master,1,1"));
105    Configuration configuration = mock(Configuration.class);
106    when(server.getConfiguration()).thenReturn(configuration);
107    TableStateManager tsm = mock(TableStateManager.class);
108    ServerManager sm = mock(ServerManager.class);
109    when(sm.isServerOnline(isA(ServerName.class))).thenReturn(true);
110
111    RegionStateStore rss = mock(RegionStateStore.class);
112    RegionStates regionStates = new RegionStates(server, tsm, sm, rss);
113
114    ServerName one = mockServer("one", 1);
115    ServerName two = mockServer("two", 1);
116    ServerName three = mockServer("three", 1);
117
118    when(sm.getDrainingServersList()).thenReturn(Arrays.asList(three));
119
120    regionStates.regionOnline(createFakeRegion(), one);
121    regionStates.regionOnline(createFakeRegion(), two);
122    regionStates.regionOnline(createFakeRegion(), three);
123
124
125    Map<TableName, Map<ServerName, List<HRegionInfo>>> result =
126        regionStates.getAssignmentsByTable();
127    for (Map<ServerName, List<HRegionInfo>> map : result.values()) {
128      assertFalse(map.keySet().contains(three));
129    }
130  }
131
132  private HRegionInfo createFakeRegion() {
133    HRegionInfo info = mock(HRegionInfo.class);
134    when(info.getEncodedName()).thenReturn(UUID.randomUUID().toString());
135    return info;
136  }
137
138  private ServerName mockServer(String fakeHost, int fakePort) {
139    ServerName serverName = mock(ServerName.class);
140    when(serverName.getHostname()).thenReturn(fakeHost);
141    when(serverName.getPort()).thenReturn(fakePort);
142    return serverName;
143  }
