084    public void awaitModifications(int count) throws InterruptedException {

091  static class DummyAssignmentListener extends DummyListener implements AssignmentListener {
092    private AtomicInteger closeCount = new AtomicInteger(0);
093    private AtomicInteger openCount = new AtomicInteger(0);
094
095    public DummyAssignmentListener() {
096    }
097
098    @Override
099    public void regionOpened(final HRegionInfo regionInfo, final ServerName serverName) {
100      LOG.info("Assignment open region=" + regionInfo + " server=" + serverName);
101      openCount.incrementAndGet();
102      modified.incrementAndGet();
103    }
104
105    @Override
106    public void regionClosed(final HRegionInfo regionInfo) {
107      LOG.info("Assignment close region=" + regionInfo);
108      closeCount.incrementAndGet();
109      modified.incrementAndGet();
110    }
111
112    public void reset() {
113      openCount.set(0);
114      closeCount.set(0);
115    }
116
117    public int getLoadCount() {
118      return openCount.get();
119    }
120
121    public int getCloseCount() {
122      return closeCount.get();
123    }
124  }

171  @Test(timeout=60000)
172  public void testServerListener() throws IOException, InterruptedException {
173    ServerManager serverManager = TEST_UTIL.getHBaseCluster().getMaster().getServerManager();
174
175    DummyServerListener listener = new DummyServerListener();
176    serverManager.registerListener(listener);
177    try {
178      MiniHBaseCluster miniCluster = TEST_UTIL.getMiniHBaseCluster();
179
180      // Start a new Region Server
181      miniCluster.startRegionServer();
182      listener.awaitModifications(1);
183      assertEquals(1, listener.getAddedCount());
184      assertEquals(0, listener.getRemovedCount());
185
186      // Start another Region Server
187      listener.reset();
188      miniCluster.startRegionServer();
189      listener.awaitModifications(1);
190      assertEquals(1, listener.getAddedCount());
191      assertEquals(0, listener.getRemovedCount());
192
193      int nrs = miniCluster.getRegionServerThreads().size();
194
195      // Stop a Region Server
196      listener.reset();
197      miniCluster.stopRegionServer(nrs - 1);
198      listener.awaitModifications(1);
199      assertEquals(0, listener.getAddedCount());
200      assertEquals(1, listener.getRemovedCount());
201
202      // Stop another Region Server
203      listener.reset();
204      miniCluster.stopRegionServer(nrs - 2);
205      listener.awaitModifications(1);
206      assertEquals(0, listener.getAddedCount());
207      assertEquals(1, listener.getRemovedCount());
208    } finally {
209      serverManager.unregisterListener(listener);
210    }
211  }

213  @Test(timeout=60000)
214  public void testAssignmentListener() throws IOException, InterruptedException {
215    AssignmentManager am = TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager();
216    Admin admin = TEST_UTIL.getAdmin();
217
218    DummyAssignmentListener listener = new DummyAssignmentListener();
219    am.registerListener(listener);
220    try {
221      final TableName tableName = TableName.valueOf(name.getMethodName());
222      final byte[] FAMILY = Bytes.toBytes("cf");
223
224      // Create a new table, with a single region
225      LOG.info("Create Table");
226      TEST_UTIL.createTable(tableName, FAMILY);
227      listener.awaitModifications(1);
228      assertEquals(1, listener.getLoadCount());
229      assertEquals(0, listener.getCloseCount());
230
231      // Add some data
232      Table table = TEST_UTIL.getConnection().getTable(tableName);
233      try {
234        for (int i = 0; i < 10; ++i) {
235          byte[] key = Bytes.toBytes("row-" + i);
236          Put put = new Put(key);
237          put.addColumn(FAMILY, null, key);
238          table.put(put);
239        }
240      } finally {
241        table.close();
242      }
243
244      // Split the table in two
245      LOG.info("Split Table");
246      listener.reset();
247      admin.split(tableName, Bytes.toBytes("row-3"));
248      listener.awaitModifications(3);
249      assertEquals(2, listener.getLoadCount());     // daughters added
250      assertEquals(1, listener.getCloseCount());    // parent removed
251
252      // Wait for the Regions to be mergeable
253      MiniHBaseCluster miniCluster = TEST_UTIL.getMiniHBaseCluster();
254      int mergeable = 0;
255      while (mergeable < 2) {
256        Thread.sleep(100);
257        admin.majorCompact(tableName);
258        mergeable = 0;
259        for (JVMClusterUtil.RegionServerThread regionThread: miniCluster.getRegionServerThreads()) {
260          for (Region region: regionThread.getRegionServer().getOnlineRegions(tableName)) {
261            mergeable += ((HRegion)region).isMergeable() ? 1 : 0;
262          }
263        }
264      }
265
266      // Merge the two regions
267      LOG.info("Merge Regions");
268      listener.reset();
269      List<HRegionInfo> regions = admin.getTableRegions(tableName);
270      assertEquals(2, regions.size());
271      boolean sameServer = areAllRegionsLocatedOnSameServer(tableName);
272      // If the regions are located by different server, we need to move
273      // regions to same server before merging. So the expected modifications
274      // will increaes to 5. (open + close)
275      final int expectedModifications = sameServer ? 3 : 5;
276      final int expectedLoadCount = sameServer ? 1 : 2;
277      final int expectedCloseCount = sameServer ? 2 : 3;
278      admin.mergeRegionsAsync(regions.get(0).getEncodedNameAsBytes(),
279        regions.get(1).getEncodedNameAsBytes(), true);
280      listener.awaitModifications(expectedModifications);
281      assertEquals(1, admin.getTableRegions(tableName).size());
282      assertEquals(expectedLoadCount, listener.getLoadCount());     // new merged region added
283      assertEquals(expectedCloseCount, listener.getCloseCount());    // daughters removed
284
285      // Delete the table
286      LOG.info("Drop Table");
287      listener.reset();
288      TEST_UTIL.deleteTable(tableName);
289      listener.awaitModifications(1);
290      assertEquals(0, listener.getLoadCount());
291      assertEquals(1, listener.getCloseCount());
292    } finally {
293      am.unregisterListener(listener);
294    }
295  }

297  private boolean areAllRegionsLocatedOnSameServer(TableName TABLE_NAME) {
298    MiniHBaseCluster miniCluster = TEST_UTIL.getMiniHBaseCluster();
299    int serverCount = 0;
300    for (JVMClusterUtil.RegionServerThread regionThread: miniCluster.getRegionServerThreads()) {
301      if (!regionThread.getRegionServer().getOnlineRegions(TABLE_NAME).isEmpty()) {
302        ++serverCount;
303      }
304      if (serverCount > 1) {
305        return false;
306      }
307    }
308    return serverCount == 1;
309  }

311  @Test
312  public void testAddNewServerThatExistsInDraining() throws Exception {
313    // Under certain circumstances, such as when we failover to the Backup
314    // HMaster, the DrainingServerTracker is started with existing servers in
315    // draining before all of the Region Servers register with the
316    // ServerManager as "online".  This test is to ensure that Region Servers
317    // are properly added to the ServerManager.drainingServers when they
318    // register with the ServerManager under these circumstances.
319    Configuration conf = TEST_UTIL.getConfiguration();
320    ZooKeeperWatcher zooKeeper = new ZooKeeperWatcher(conf,
321        "zkWatcher-NewServerDrainTest", abortable, true);
322    String baseZNode = conf.get(HConstants.ZOOKEEPER_ZNODE_PARENT,
323        HConstants.DEFAULT_ZOOKEEPER_ZNODE_PARENT);
324    String drainingZNode = ZKUtil.joinZNode(baseZNode,
325        conf.get("zookeeper.znode.draining.rs", "draining"));
326
327    HMaster master = Mockito.mock(HMaster.class);
328    Mockito.when(master.getConfiguration()).thenReturn(conf);
329
330    ServerName SERVERNAME_A = ServerName.valueOf("mockserverbulk_a.org", 1000, 8000);
331    ServerName SERVERNAME_B = ServerName.valueOf("mockserverbulk_b.org", 1001, 8000);
332    ServerName SERVERNAME_C = ServerName.valueOf("mockserverbulk_c.org", 1002, 8000);
333
334    // We'll start with 2 servers in draining that existed before the
335    // HMaster started.
336    ArrayList<ServerName> drainingServers = new ArrayList<ServerName>();
337    drainingServers.add(SERVERNAME_A);
338    drainingServers.add(SERVERNAME_B);
339
340    // We'll have 2 servers that come online AFTER the DrainingServerTracker
341    // is started (just as we see when we failover to the Backup HMaster).
342    // One of these will already be a draining server.
343    HashMap<ServerName, ServerLoad> onlineServers = new HashMap<ServerName, ServerLoad>();
344    onlineServers.put(SERVERNAME_A, ServerLoad.EMPTY_SERVERLOAD);
345    onlineServers.put(SERVERNAME_C, ServerLoad.EMPTY_SERVERLOAD);
346
347    // Create draining znodes for the draining servers, which would have been
348    // performed when the previous HMaster was running.
349    for (ServerName sn : drainingServers) {
350      String znode = ZKUtil.joinZNode(drainingZNode, sn.getServerName());
351      ZKUtil.createAndFailSilent(zooKeeper, znode);
352    }
353
354    // Now, we follow the same order of steps that the HMaster does to setup
355    // the ServerManager, RegionServerTracker, and DrainingServerTracker.
356    ServerManager serverManager = new ServerManager(master);
357
358    RegionServerTracker regionServerTracker = new RegionServerTracker(
359        zooKeeper, master, serverManager);
360    regionServerTracker.start();
361
362    DrainingServerTracker drainingServerTracker = new DrainingServerTracker(
363        zooKeeper, master, serverManager);
364    drainingServerTracker.start();
365
366    // Confirm our ServerManager lists are empty.
367    Assert.assertEquals(serverManager.getOnlineServers(),
368        new HashMap<ServerName, ServerLoad>());
369    Assert.assertEquals(serverManager.getDrainingServersList(),
370        new ArrayList<ServerName>());
371
372    // checkAndRecordNewServer() is how servers are added to the ServerManager.
373    ArrayList<ServerName> onlineDrainingServers = new ArrayList<ServerName>();
374    for (ServerName sn : onlineServers.keySet()){
375      // Here's the actual test.
376      serverManager.checkAndRecordNewServer(sn, onlineServers.get(sn));
377      if (drainingServers.contains(sn)){
378        onlineDrainingServers.add(sn);  // keeping track for later verification
379      }
380    }
381
382    // Verify the ServerManager lists are correctly updated.
383    Assert.assertEquals(serverManager.getOnlineServers(), onlineServers);
384    Assert.assertEquals(serverManager.getDrainingServersList(),
385        onlineDrainingServers);
386  }
