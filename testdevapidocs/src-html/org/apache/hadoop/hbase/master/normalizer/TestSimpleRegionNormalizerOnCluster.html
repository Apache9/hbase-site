051/**
052 * Testing {@link SimpleRegionNormalizer} on minicluster.
053 */
054@Category({MasterTests.class, MediumTests.class})
055public class TestSimpleRegionNormalizerOnCluster {
056  private static final Log LOG = LogFactory.getLog(TestSimpleRegionNormalizerOnCluster.class);
057  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
058  private static final byte[] FAMILYNAME = Bytes.toBytes("fam");
059  private static Admin admin;
060
061  @BeforeClass
062  public static void beforeAllTests() throws Exception {
063    // we will retry operations when PleaseHoldException is thrown
064    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 3);
065    TEST_UTIL.getConfiguration().setBoolean(HConstants.HBASE_NORMALIZER_ENABLED, true);
066
067    // Start a cluster of two regionservers.
068    TEST_UTIL.startMiniCluster(1);
069    admin = TEST_UTIL.getHBaseAdmin();
070  }
071
072  @AfterClass
073  public static void afterAllTests() throws Exception {
074    TEST_UTIL.shutdownMiniCluster();
075  }
076
077  @Test(timeout = 60000)
078  @SuppressWarnings("deprecation")
079  public void testRegionNormalizationSplitOnCluster() throws Exception {
080    final TableName TABLENAME =
081      TableName.valueOf("testRegionNormalizationSplitOnCluster");
082    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
083    HMaster m = cluster.getMaster();
084
085    try (HTable ht = TEST_UTIL.createMultiRegionTable(TABLENAME, FAMILYNAME, 5)) {
086      // Need to get sorted list of regions here
087      List<HRegion> generatedRegions = TEST_UTIL.getHBaseCluster().getRegions(TABLENAME);
088      Collections.sort(generatedRegions, new Comparator<HRegion>() {
089        @Override
090        public int compare(HRegion o1, HRegion o2) {
091          return o1.getRegionInfo().compareTo(o2.getRegionInfo());
092        }
093      });
094
095      HRegion region = generatedRegions.get(0);
096      generateTestData(region, 1);
097      region.flush(true);
098
099      region = generatedRegions.get(1);
100      generateTestData(region, 1);
101      region.flush(true);
102
103      region = generatedRegions.get(2);
104      generateTestData(region, 2);
105      region.flush(true);
106
107      region = generatedRegions.get(3);
108      generateTestData(region, 2);
109      region.flush(true);
110
111      region = generatedRegions.get(4);
112      generateTestData(region, 5);
113      region.flush(true);
114    }
115
116    HTableDescriptor htd = admin.getTableDescriptor(TABLENAME);
117    htd.setNormalizationEnabled(true);
118    admin.modifyTable(TABLENAME, htd);
119
120    admin.flush(TABLENAME);
121
122    System.out.println(admin.getTableDescriptor(TABLENAME));
123
124    assertEquals(5, MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME));
125
126    // Now trigger a split and stop when the split is in progress
127    Thread.sleep(5000); // to let region load to update
128    m.normalizeRegions();
129
130    while (MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME) < 6) {
131      LOG.info("Waiting for normalization split to complete");
132      Thread.sleep(100);
133    }
134
135    assertEquals(6, MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME));
136
137    admin.disableTable(TABLENAME);
138    admin.deleteTable(TABLENAME);
139  }

141  @Test(timeout = 60000)
142  @SuppressWarnings("deprecation")
143  public void testRegionNormalizationMergeOnCluster() throws Exception {
144    final TableName TABLENAME =
145      TableName.valueOf("testRegionNormalizationMergeOnCluster");
146    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
147    HMaster m = cluster.getMaster();
148
149    // create 5 regions with sizes to trigger merge of small regions
150    try (HTable ht = TEST_UTIL.createMultiRegionTable(TABLENAME, FAMILYNAME, 5)) {
151      // Need to get sorted list of regions here
152      List<HRegion> generatedRegions = TEST_UTIL.getHBaseCluster().getRegions(TABLENAME);
153      Collections.sort(generatedRegions, new Comparator<HRegion>() {
154        @Override
155        public int compare(HRegion o1, HRegion o2) {
156          return o1.getRegionInfo().compareTo(o2.getRegionInfo());
157        }
158      });
159
160      HRegion region = generatedRegions.get(0);
161      generateTestData(region, 1);
162      region.flush(true);
163
164      region = generatedRegions.get(1);
165      generateTestData(region, 1);
166      region.flush(true);
167
168      region = generatedRegions.get(2);
169      generateTestData(region, 3);
170      region.flush(true);
171
172      region = generatedRegions.get(3);
173      generateTestData(region, 3);
174      region.flush(true);
175
176      region = generatedRegions.get(4);
177      generateTestData(region, 5);
178      region.flush(true);
179    }
180
181    HTableDescriptor htd = admin.getTableDescriptor(TABLENAME);
182    htd.setNormalizationEnabled(true);
183    admin.modifyTable(TABLENAME, htd);
184
185    admin.flush(TABLENAME);
186
187    assertEquals(5, MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME));
188
189    // Now trigger a merge and stop when the merge is in progress
190    Thread.sleep(5000); // to let region load to update
191    m.normalizeRegions();
192
193    while (MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME) > 4) {
194      LOG.info("Waiting for normalization merge to complete");
195      Thread.sleep(100);
196    }
197
198    assertEquals(4, MetaTableAccessor.getRegionCount(TEST_UTIL.getConnection(), TABLENAME));
199
200    admin.disableTable(TABLENAME);
201    admin.deleteTable(TABLENAME);
202  }
203
204  private void generateTestData(Region region, int numRows) throws IOException {
205    // generating 1Mb values
206    LoadTestKVGenerator dataGenerator = new LoadTestKVGenerator(1024 * 1024, 1024 * 1024);
207    for (int i = 0; i < numRows; ++i) {
208      byte[] key = Bytes.add(region.getRegionInfo().getStartKey(), Bytes.toBytes(i));
209      for (int j = 0; j < 1; ++j) {
210        Put put = new Put(key);
211        byte[] col = Bytes.toBytes(String.valueOf(j));
212        byte[] value = dataGenerator.generateRandomSizeValue(key, col);
213        put.addColumn(FAMILYNAME, col, value);
214        region.put(put);
215      }
216    }
217  }
