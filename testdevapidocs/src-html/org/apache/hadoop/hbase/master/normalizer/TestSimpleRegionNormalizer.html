142  public void testMergeOfSmallRegions(boolean mergeDesired) throws HBaseIOException {
143    TableName testTable = TableName.valueOf("testMergeOfSmallRegions");
144    List<HRegionInfo> hris = new ArrayList<>();
145    Map<byte[], Integer> regionSizes = new HashMap<>();
146
147    HRegionInfo hri1 = new HRegionInfo(testTable, Bytes.toBytes("aaa"), Bytes.toBytes("bbb"));
148    hris.add(hri1);
149    regionSizes.put(hri1.getRegionName(), 15);
150
151    HRegionInfo hri2 = new HRegionInfo(testTable, Bytes.toBytes("bbb"), Bytes.toBytes("ccc"));
152    hris.add(hri2);
153    regionSizes.put(hri2.getRegionName(), 5);
154
155    HRegionInfo hri3 = new HRegionInfo(testTable, Bytes.toBytes("ccc"), Bytes.toBytes("ddd"));
156    hris.add(hri3);
157    regionSizes.put(hri3.getRegionName(), 5);
158
159    HRegionInfo hri4 = new HRegionInfo(testTable, Bytes.toBytes("ddd"), Bytes.toBytes("eee"));
160    hris.add(hri4);
161    regionSizes.put(hri4.getRegionName(), 15);
162
163    HRegionInfo hri5 = new HRegionInfo(testTable, Bytes.toBytes("eee"), Bytes.toBytes("fff"));
164    hris.add(hri5);
165    regionSizes.put(hri5.getRegionName(), 16);
166
167    setupMocksForNormalizer(regionSizes, hris);
168    List<NormalizationPlan> plans = normalizer.computePlanForTable(testTable,
169      mergeDesired ? bothTypes : splitType);
170
171    if (mergeDesired) {
172      NormalizationPlan plan = plans.get(0);
173      assertTrue(plan instanceof MergeNormalizationPlan);
174      assertEquals(hri2, ((MergeNormalizationPlan) plan).getFirstRegion());
175      assertEquals(hri3, ((MergeNormalizationPlan) plan).getSecondRegion());
176    } else {
177      assertTrue(plans == null);
178    }
179  }

283  protected void setupMocksForNormalizer(Map<byte[], Integer> regionSizes,
284                                         List<HRegionInfo> hris) {
285    masterServices = Mockito.mock(MasterServices.class, RETURNS_DEEP_STUBS);
286
287    // for simplicity all regions are assumed to be on one server; doesn't matter to us
288    ServerName sn = ServerName.valueOf("localhost", -1, 1L);
289    when(masterServices.getAssignmentManager().getRegionStates().
290      getRegionsOfTable(any(TableName.class))).thenReturn(hris);
291    when(masterServices.getAssignmentManager().getRegionStates().
292      getRegionServerOfRegion(any(HRegionInfo.class))).thenReturn(sn);
293
294    for (Map.Entry<byte[], Integer> region : regionSizes.entrySet()) {
295      RegionLoad regionLoad = Mockito.mock(RegionLoad.class);
296      when(regionLoad.getName()).thenReturn(region.getKey());
297      when(regionLoad.getStorefileSizeMB()).thenReturn(region.getValue());
298
299      when(masterServices.getServerManager().getLoad(sn).
300        getRegionsLoad().get(region.getKey())).thenReturn(regionLoad);
301    }
302
303    normalizer.setMasterServices(masterServices);
304  }
