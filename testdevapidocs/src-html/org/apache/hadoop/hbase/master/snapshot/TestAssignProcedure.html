021import java.util.ArrayList;

023import java.util.List;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027import org.apache.hadoop.hbase.CategoryBasedTimeout;
028import org.apache.hadoop.hbase.TableName;
029import org.apache.hadoop.hbase.client.RegionInfo;
030import org.apache.hadoop.hbase.client.RegionInfoBuilder;
031import org.apache.hadoop.hbase.master.assignment.AssignProcedure;
032import org.apache.hadoop.hbase.testclassification.RegionServerTests;
033import org.apache.hadoop.hbase.testclassification.SmallTests;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.junit.Rule;
036import org.junit.Test;
037import org.junit.experimental.categories.Category;
038import org.junit.rules.TestName;
039import org.junit.rules.TestRule;
040
041import static junit.framework.TestCase.assertTrue;
042
043
044@Category({RegionServerTests.class, SmallTests.class})
045public class TestAssignProcedure {
046  private static final Log LOG = LogFactory.getLog(TestAssignProcedure.class);
047  @Rule public TestName name = new TestName();
048  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().
049      withTimeout(this.getClass()).
050      withLookingForStuckThread(true).
051      build();
052
053  @Test
054  public void testSimpleComparator() {
055    List<AssignProcedure> procedures = new ArrayList<AssignProcedure>();
056    RegionInfo user1 = RegionInfoBuilder.newBuilder(TableName.valueOf("user_space1")).build();
057    procedures.add(new AssignProcedure(user1));
058    RegionInfo user2 = RegionInfoBuilder.newBuilder(TableName.valueOf("user_space2")).build();
059    procedures.add(new AssignProcedure(RegionInfoBuilder.FIRST_META_REGIONINFO));
060    procedures.add(new AssignProcedure(user2));
061    RegionInfo system = RegionInfoBuilder.newBuilder(TableName.NAMESPACE_TABLE_NAME).build();
062    procedures.add(new AssignProcedure(system));
063    procedures.sort(AssignProcedure.COMPARATOR);
064    assertTrue(procedures.get(0).isMeta());
065    assertTrue(procedures.get(1).getRegionInfo().getTable().equals(TableName.NAMESPACE_TABLE_NAME));
066  }
067
068  @Test
069  public void testComparatorWithMetas() {
070    List<AssignProcedure> procedures = new ArrayList<AssignProcedure>();
071    RegionInfo user3 = RegionInfoBuilder.newBuilder(TableName.valueOf("user3")).build();
072    procedures.add(new AssignProcedure(user3));
073    RegionInfo system = RegionInfoBuilder.newBuilder(TableName.NAMESPACE_TABLE_NAME).build();
074    procedures.add(new AssignProcedure(system));
075    RegionInfo user1 = RegionInfoBuilder.newBuilder(TableName.valueOf("user_space1")).build();
076    RegionInfo user2 = RegionInfoBuilder.newBuilder(TableName.valueOf("user_space2")).build();
077    procedures.add(new AssignProcedure(user1));
078    RegionInfo meta2 = RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
079        setStartKey(Bytes.toBytes("002")).build();
080    procedures.add(new AssignProcedure(meta2));
081    procedures.add(new AssignProcedure(user2));
082    RegionInfo meta1 = RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
083        setStartKey(Bytes.toBytes("001")).build();
084    procedures.add(new AssignProcedure(meta1));
085    procedures.add(new AssignProcedure(RegionInfoBuilder.FIRST_META_REGIONINFO));
086    RegionInfo meta0 = RegionInfoBuilder.newBuilder(TableName.META_TABLE_NAME).
087        setStartKey(Bytes.toBytes("000")).build();
088    procedures.add(new AssignProcedure(meta0));
089    for (int i = 0; i < 10; i++) {
090      Collections.shuffle(procedures);
091      procedures.sort(AssignProcedure.COMPARATOR);
092      try {
093        assertTrue(procedures.get(0).getRegionInfo().equals(RegionInfoBuilder.FIRST_META_REGIONINFO));
094        assertTrue(procedures.get(1).getRegionInfo().equals(meta0));
095        assertTrue(procedures.get(2).getRegionInfo().equals(meta1));
096        assertTrue(procedures.get(3).getRegionInfo().equals(meta2));
097        assertTrue(procedures.get(4).getRegionInfo().getTable().equals(TableName.NAMESPACE_TABLE_NAME));
098        assertTrue(procedures.get(5).getRegionInfo().equals(user1));
099        assertTrue(procedures.get(6).getRegionInfo().equals(user2));
100        assertTrue(procedures.get(7).getRegionInfo().equals(user3));
101      } catch (Throwable t) {
102        for (AssignProcedure proc : procedures) {
103          LOG.debug(proc);
104        }
105        throw t;
106      }
107    }
108  }
109}
