062  /**
063   * Simple test of master failover.
064   * <p>
065   * Starts with three masters.  Kills a backup master.  Then kills the active
066   * master.  Ensures the final master becomes active and we can still contact
067   * the cluster.
068   */
069  @Test (timeout=240000)
070  public void testSimpleMasterFailover() throws Exception {
071
072    final int NUM_MASTERS = 3;
073    final int NUM_RS = 3;
074
075    // Start the cluster
076    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
077
078    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
079    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
080
081    // get all the master threads
082    List<MasterThread> masterThreads = cluster.getMasterThreads();
083
084    // wait for each to come online
085    for (MasterThread mt : masterThreads) {
086      assertTrue(mt.isAlive());
087    }
088
089    // verify only one is the active master and we have right number
090    int numActive = 0;
091    int activeIndex = -1;
092    ServerName activeName = null;
093    HMaster active = null;
094    for (int i = 0; i < masterThreads.size(); i++) {
095      if (masterThreads.get(i).getMaster().isActiveMaster()) {
096        numActive++;
097        activeIndex = i;
098        active = masterThreads.get(activeIndex).getMaster();
099        activeName = active.getServerName();
100      }
101    }
102    assertEquals(1, numActive);
103    assertEquals(NUM_MASTERS, masterThreads.size());
104    LOG.info("Active master " + activeName);
105
106    // Check that ClusterStatus reports the correct active and backup masters
107    assertNotNull(active);
108    ClusterStatus status = active.getClusterStatus();
109    assertTrue(status.getMaster().equals(activeName));
110    assertEquals(2, status.getBackupMastersSize());
111    assertEquals(2, status.getBackupMasters().size());
112
113    // attempt to stop one of the inactive masters
114    int backupIndex = (activeIndex == 0 ? 1 : activeIndex - 1);
115    HMaster master = cluster.getMaster(backupIndex);
116    LOG.debug("\n\nStopping a backup master: " + master.getServerName() + "\n");
117    cluster.stopMaster(backupIndex, false);
118    cluster.waitOnMaster(backupIndex);
119
120    // Verify still one active master and it's the same
121    for (int i = 0; i < masterThreads.size(); i++) {
122      if (masterThreads.get(i).getMaster().isActiveMaster()) {
123        assertTrue(activeName.equals(masterThreads.get(i).getMaster().getServerName()));
124        activeIndex = i;
125        active = masterThreads.get(activeIndex).getMaster();
126      }
127    }
128    assertEquals(1, numActive);
129    assertEquals(2, masterThreads.size());
130    int rsCount = masterThreads.get(activeIndex).getMaster().getClusterStatus().getServersSize();
131    LOG.info("Active master " + active.getServerName() + " managing " + rsCount +  " regions servers");
132    assertEquals(3, rsCount);
133
134    // Check that ClusterStatus reports the correct active and backup masters
135    assertNotNull(active);
136    status = active.getClusterStatus();
137    assertTrue(status.getMaster().equals(activeName));
138    assertEquals(1, status.getBackupMastersSize());
139    assertEquals(1, status.getBackupMasters().size());
140
141    // kill the active master
142    LOG.debug("\n\nStopping the active master " + active.getServerName() + "\n");
143    cluster.stopMaster(activeIndex, false);
144    cluster.waitOnMaster(activeIndex);
145
146    // wait for an active master to show up and be ready
147    assertTrue(cluster.waitForActiveAndReadyMaster());
148
149    LOG.debug("\n\nVerifying backup master is now active\n");
150    // should only have one master now
151    assertEquals(1, masterThreads.size());
152
153    // and he should be active
154    active = masterThreads.get(0).getMaster();
155    assertNotNull(active);
156    status = active.getClusterStatus();
157    ServerName mastername = status.getMaster();
158    assertTrue(mastername.equals(active.getServerName()));
159    assertTrue(active.isActiveMaster());
160    assertEquals(0, status.getBackupMastersSize());
161    assertEquals(0, status.getBackupMasters().size());
162    int rss = status.getServersSize();
163    LOG.info("Active master " + mastername.getServerName() + " managing " +
164      rss +  " region servers");
165    assertEquals(3, rss);
166
167    // Stop the cluster
168    TEST_UTIL.shutdownMiniCluster();
169  }

171  /**
172   * Test meta in transition when master failover
173   */
174  @Test(timeout = 180000)
175  public void testMetaInTransitionWhenMasterFailover() throws Exception {
176    final int NUM_MASTERS = 1;
177    final int NUM_RS = 1;
178
179    // Start the cluster
180    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
181    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
182    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
183    log("Cluster started");
184
185    log("Moving meta off the master");
186    HMaster activeMaster = cluster.getMaster();
187    HRegionServer rs = cluster.getRegionServer(0);
188    ServerName metaServerName = cluster.getLiveRegionServerThreads()
189      .get(0).getRegionServer().getServerName();
190    activeMaster.move(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes(),
191      Bytes.toBytes(metaServerName.getServerName()));
192    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
193    assertEquals("Meta should be assigned on expected regionserver",
194      metaServerName, activeMaster.getMetaTableLocator()
195        .getMetaRegionLocation(activeMaster.getZooKeeper()));
196
197    // Now kill master, meta should remain on rs, where we placed it before.
198    log("Aborting master");
199    activeMaster.abort("test-kill");
200    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
201    log("Master has aborted");
202
203    // meta should remain where it was
204    RegionState metaState =
205      MetaTableLocator.getMetaRegionState(rs.getZooKeeper());
206    assertEquals("hbase:meta should be online on RS",
207      metaState.getServerName(), rs.getServerName());
208    assertEquals("hbase:meta should be online on RS",
209      metaState.getState(), State.OPEN);
210
211    // Start up a new master
212    log("Starting up a new master");
213    activeMaster = cluster.startMaster().getMaster();
214    log("Waiting for master to be ready");
215    cluster.waitForActiveAndReadyMaster();
216    log("Master is ready");
217
218    // ensure meta is still deployed on RS
219    metaState =
220      MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
221    assertEquals("hbase:meta should be online on RS",
222      metaState.getServerName(), rs.getServerName());
223    assertEquals("hbase:meta should be online on RS",
224      metaState.getState(), State.OPEN);
225
226    // Update meta state as OPENING, then kill master
227    // that simulates, that RS successfully deployed, but
228    // RPC was lost right before failure.
229    // region server should expire (how it can be verified?)
230    MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
231      rs.getServerName(), State.OPENING);
232    Region meta = rs.getFromOnlineRegions(HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
233    rs.removeFromOnlineRegions(meta, null);
234    ((HRegion)meta).close();
235
236    log("Aborting master");
237    activeMaster.abort("test-kill");
238    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
239    log("Master has aborted");
240
241    // Start up a new master
242    log("Starting up a new master");
243    activeMaster = cluster.startMaster().getMaster();
244    log("Waiting for master to be ready");
245    cluster.waitForActiveAndReadyMaster();
246    log("Master is ready");
247
248    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
249    log("Meta was assigned");
250
251    metaState =
252      MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
253    assertEquals("hbase:meta should be online on RS",
254      metaState.getServerName(), rs.getServerName());
255    assertEquals("hbase:meta should be online on RS",
256      metaState.getState(), State.OPEN);
257
258    // Update meta state as CLOSING, then kill master
259    // that simulates, that RS successfully deployed, but
260    // RPC was lost right before failure.
261    // region server should expire (how it can be verified?)
262    MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
263      rs.getServerName(), State.CLOSING);
264
265    log("Aborting master");
266    activeMaster.abort("test-kill");
267    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
268    log("Master has aborted");
269
270    rs.getRSRpcServices().closeRegion(null, ProtobufUtil.buildCloseRegionRequest(
271      rs.getServerName(), HRegionInfo.FIRST_META_REGIONINFO.getRegionName()));
272
273    // Start up a new master
274    log("Starting up a new master");
275    activeMaster = cluster.startMaster().getMaster();
276    assertNotNull(activeMaster);
277    log("Waiting for master to be ready");
278    cluster.waitForActiveAndReadyMaster();
279    log("Master is ready");
280
281    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
282    log("Meta was assigned");
283
284    // Done, shutdown the cluster
285    TEST_UTIL.shutdownMiniCluster();
286  }
