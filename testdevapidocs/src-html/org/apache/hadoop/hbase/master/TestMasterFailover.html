069  HRegion createRegion(final HRegionInfo  hri, final Path rootdir, final Configuration c,
070      final HTableDescriptor htd)
071  throws IOException {
072    HRegion r = HBaseTestingUtility.createRegionAndWAL(hri, rootdir, c, htd);
073    // The above call to create a region will create an wal file.  Each
074    // log file create will also create a running thread to do syncing.  We need
075    // to close out this log else we will have a running thread trying to sync
076    // the file system continuously which is ugly when dfs is taken away at the
077    // end of the test.
078    HBaseTestingUtility.closeRegionAndWAL(r);
079    return r;
080  }

089  /**
090   * Simple test of master failover.
091   * <p>
092   * Starts with three masters.  Kills a backup master.  Then kills the active
093   * master.  Ensures the final master becomes active and we can still contact
094   * the cluster.
095   * @throws Exception
096   */
097  @Test (timeout=240000)
098  public void testSimpleMasterFailover() throws Exception {
099
100    final int NUM_MASTERS = 3;
101    final int NUM_RS = 3;
102
103    // Start the cluster
104    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
105
106    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
107    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
108
109    // get all the master threads
110    List<MasterThread> masterThreads = cluster.getMasterThreads();
111
112    // wait for each to come online
113    for (MasterThread mt : masterThreads) {
114      assertTrue(mt.isAlive());
115    }
116
117    // verify only one is the active master and we have right number
118    int numActive = 0;
119    int activeIndex = -1;
120    ServerName activeName = null;
121    HMaster active = null;
122    for (int i = 0; i < masterThreads.size(); i++) {
123      if (masterThreads.get(i).getMaster().isActiveMaster()) {
124        numActive++;
125        activeIndex = i;
126        active = masterThreads.get(activeIndex).getMaster();
127        activeName = active.getServerName();
128      }
129    }
130    assertEquals(1, numActive);
131    assertEquals(NUM_MASTERS, masterThreads.size());
132    LOG.info("Active master " + activeName);
133
134    // Check that ClusterStatus reports the correct active and backup masters
135    assertNotNull(active);
136    ClusterStatus status = active.getClusterStatus();
137    assertTrue(status.getMaster().equals(activeName));
138    assertEquals(2, status.getBackupMastersSize());
139    assertEquals(2, status.getBackupMasters().size());
140
141    // attempt to stop one of the inactive masters
142    int backupIndex = (activeIndex == 0 ? 1 : activeIndex - 1);
143    HMaster master = cluster.getMaster(backupIndex);
144    LOG.debug("\n\nStopping a backup master: " + master.getServerName() + "\n");
145    cluster.stopMaster(backupIndex, false);
146    cluster.waitOnMaster(backupIndex);
147
148    // Verify still one active master and it's the same
149    for (int i = 0; i < masterThreads.size(); i++) {
150      if (masterThreads.get(i).getMaster().isActiveMaster()) {
151        assertTrue(activeName.equals(masterThreads.get(i).getMaster().getServerName()));
152        activeIndex = i;
153        active = masterThreads.get(activeIndex).getMaster();
154      }
155    }
156    assertEquals(1, numActive);
157    assertEquals(2, masterThreads.size());
158    int rsCount = masterThreads.get(activeIndex).getMaster().getClusterStatus().getServersSize();
159    LOG.info("Active master " + active.getServerName() + " managing " + rsCount +  " regions servers");
160    assertEquals(4, rsCount);
161
162    // Check that ClusterStatus reports the correct active and backup masters
163    assertNotNull(active);
164    status = active.getClusterStatus();
165    assertTrue(status.getMaster().equals(activeName));
166    assertEquals(1, status.getBackupMastersSize());
167    assertEquals(1, status.getBackupMasters().size());
168
169    // kill the active master
170    LOG.debug("\n\nStopping the active master " + active.getServerName() + "\n");
171    cluster.stopMaster(activeIndex, false);
172    cluster.waitOnMaster(activeIndex);
173
174    // wait for an active master to show up and be ready
175    assertTrue(cluster.waitForActiveAndReadyMaster());
176
177    LOG.debug("\n\nVerifying backup master is now active\n");
178    // should only have one master now
179    assertEquals(1, masterThreads.size());
180
181    // and he should be active
182    active = masterThreads.get(0).getMaster();
183    assertNotNull(active);
184    status = active.getClusterStatus();
185    ServerName mastername = status.getMaster();
186    assertTrue(mastername.equals(active.getServerName()));
187    assertTrue(active.isActiveMaster());
188    assertEquals(0, status.getBackupMastersSize());
189    assertEquals(0, status.getBackupMasters().size());
190    int rss = status.getServersSize();
191    LOG.info("Active master " + mastername.getServerName() + " managing " +
192      rss +  " region servers");
193    assertEquals(4, rss);
194
195    // Stop the cluster
196    TEST_UTIL.shutdownMiniCluster();
197  }

199  /**
200   * Test region in pending_open/close when master failover
201   */
202  @Test (timeout=180000)
203  public void testPendingOpenOrCloseWhenMasterFailover() throws Exception {
204    final int NUM_MASTERS = 1;
205    final int NUM_RS = 1;
206
207    // Create config to use for this cluster
208    Configuration conf = HBaseConfiguration.create();
209
210    // Start the cluster
211    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
212    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
213    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
214    log("Cluster started");
215
216    // get all the master threads
217    List<MasterThread> masterThreads = cluster.getMasterThreads();
218    assertEquals(1, masterThreads.size());
219
220    // only one master thread, let's wait for it to be initialized
221    assertTrue(cluster.waitForActiveAndReadyMaster());
222    HMaster master = masterThreads.get(0).getMaster();
223    assertTrue(master.isActiveMaster());
224    assertTrue(master.isInitialized());
225
226    // Create a table with a region online
227    Table onlineTable = TEST_UTIL.createTable(TableName.valueOf("onlineTable"), "family");
228    onlineTable.close();
229    // Create a table in META, so it has a region offline
230    HTableDescriptor offlineTable = new HTableDescriptor(
231      TableName.valueOf(Bytes.toBytes("offlineTable")));
232    offlineTable.addFamily(new HColumnDescriptor(Bytes.toBytes("family")));
233
234    FileSystem filesystem = FileSystem.get(conf);
235    Path rootdir = FSUtils.getRootDir(conf);
236    FSTableDescriptors fstd = new FSTableDescriptors(conf, filesystem, rootdir);
237    fstd.createTableDescriptor(offlineTable);
238
239    HRegionInfo hriOffline = new HRegionInfo(offlineTable.getTableName(), null, null);
240    createRegion(hriOffline, rootdir, conf, offlineTable);
241    MetaTableAccessor.addRegionToMeta(master.getConnection(), hriOffline);
242
243    log("Regions in hbase:meta and namespace have been created");
244
245    // at this point we only expect 3 regions to be assigned out
246    // (catalogs and namespace, + 1 online region)
247    assertEquals(3, cluster.countServedRegions());
248    HRegionInfo hriOnline = null;
249    try (RegionLocator locator =
250        TEST_UTIL.getConnection().getRegionLocator(TableName.valueOf("onlineTable"))) {
251      hriOnline = locator.getRegionLocation(HConstants.EMPTY_START_ROW).getRegionInfo();
252    }
253    RegionStates regionStates = master.getAssignmentManager().getRegionStates();
254    RegionStateStore stateStore = master.getAssignmentManager().getRegionStateStore();
255
256    // Put the online region in pending_close. It is actually already opened.
257    // This is to simulate that the region close RPC is not sent out before failover
258    RegionState oldState = regionStates.getRegionState(hriOnline);
259    RegionState newState = new RegionState(hriOnline, State.CLOSING, oldState.getServerName());
260    stateStore.updateRegionState(HConstants.NO_SEQNUM, -1, newState, oldState);
261
262    // Put the offline region in pending_open. It is actually not opened yet.
263    // This is to simulate that the region open RPC is not sent out before failover
264    oldState = new RegionState(hriOffline, State.OFFLINE);
265    newState = new RegionState(hriOffline, State.OPENING, newState.getServerName());
266    stateStore.updateRegionState(HConstants.NO_SEQNUM, -1, newState, oldState);
267
268    HRegionInfo failedClose = new HRegionInfo(offlineTable.getTableName(), null, null);
269    createRegion(failedClose, rootdir, conf, offlineTable);
270    MetaTableAccessor.addRegionToMeta(master.getConnection(), failedClose);
271
272    oldState = new RegionState(failedClose, State.CLOSING);
273    newState = new RegionState(failedClose, State.FAILED_CLOSE, newState.getServerName());
274    stateStore.updateRegionState(HConstants.NO_SEQNUM, -1, newState, oldState);
275
276    HRegionInfo failedOpen = new HRegionInfo(offlineTable.getTableName(), null, null);
277    createRegion(failedOpen, rootdir, conf, offlineTable);
278    MetaTableAccessor.addRegionToMeta(master.getConnection(), failedOpen);
279
280    // Simulate a region transitioning to failed open when the region server reports the
281    // transition as FAILED_OPEN
282    oldState = new RegionState(failedOpen, State.OPENING);
283    newState = new RegionState(failedOpen, State.FAILED_OPEN, newState.getServerName());
284    stateStore.updateRegionState(HConstants.NO_SEQNUM, -1, newState, oldState);
285
286    HRegionInfo failedOpenNullServer = new HRegionInfo(offlineTable.getTableName(), null, null);
287    LOG.info("Failed open NUll server " + failedOpenNullServer.getEncodedName());
288    createRegion(failedOpenNullServer, rootdir, conf, offlineTable);
289    MetaTableAccessor.addRegionToMeta(master.getConnection(), failedOpenNullServer);
290
291    // Simulate a region transitioning to failed open when the master couldn't find a plan for
292    // the region
293    oldState = new RegionState(failedOpenNullServer, State.OFFLINE);
294    newState = new RegionState(failedOpenNullServer, State.FAILED_OPEN, null);
295    stateStore.updateRegionState(HConstants.NO_SEQNUM, -1, newState, oldState);
296
297    // Stop the master
298    log("Aborting master");
299    cluster.abortMaster(0);
300    cluster.waitOnMaster(0);
301    log("Master has aborted");
302
303    // Start up a new master
304    log("Starting up a new master");
305    master = cluster.startMaster().getMaster();
306    log("Waiting for master to be ready");
307    cluster.waitForActiveAndReadyMaster();
308    log("Master is ready");
309
310    // Wait till no region in transition any more
311    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
312
313    // Get new region states since master restarted
314    regionStates = master.getAssignmentManager().getRegionStates();
315
316    // Both pending_open (RPC sent/not yet) regions should be online
317    assertTrue(regionStates.isRegionOnline(hriOffline));
318    assertTrue(regionStates.isRegionOnline(hriOnline));
319    assertTrue(regionStates.isRegionOnline(failedClose));
320    assertTrue(regionStates.isRegionOnline(failedOpenNullServer));
321    assertTrue(regionStates.isRegionOnline(failedOpen));
322
323    log("Done with verification, shutting down cluster");
324
325    // Done, shutdown the cluster
326    TEST_UTIL.shutdownMiniCluster();
327  }

329  /**
330   * Test meta in transition when master failover
331   */
332  @Test(timeout = 180000)
333  public void testMetaInTransitionWhenMasterFailover() throws Exception {
334    final int NUM_MASTERS = 1;
335    final int NUM_RS = 1;
336
337    // Start the cluster
338    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
339    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
340    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
341    log("Cluster started");
342
343    log("Moving meta off the master");
344    HMaster activeMaster = cluster.getMaster();
345    HRegionServer rs = cluster.getRegionServer(0);
346    ServerName metaServerName = cluster.getLiveRegionServerThreads()
347      .get(0).getRegionServer().getServerName();
348    activeMaster.move(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes(),
349      Bytes.toBytes(metaServerName.getServerName()));
350    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
351    assertEquals("Meta should be assigned on expected regionserver",
352      metaServerName, activeMaster.getMetaTableLocator()
353        .getMetaRegionLocation(activeMaster.getZooKeeper()));
354
355    // Now kill master, meta should remain on rs, where we placed it before.
356    log("Aborting master");
357    activeMaster.abort("test-kill");
358    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
359    log("Master has aborted");
360
361    // meta should remain where it was
362    RegionState metaState =
363      MetaTableLocator.getMetaRegionState(rs.getZooKeeper());
364    assertEquals("hbase:meta should be onlined on RS",
365      metaState.getServerName(), rs.getServerName());
366    assertEquals("hbase:meta should be onlined on RS",
367      metaState.getState(), State.OPEN);
368
369    // Start up a new master
370    log("Starting up a new master");
371    activeMaster = cluster.startMaster().getMaster();
372    log("Waiting for master to be ready");
373    cluster.waitForActiveAndReadyMaster();
374    log("Master is ready");
375
376    // ensure meta is still deployed on RS
377    metaState =
378      MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
379    assertEquals("hbase:meta should be onlined on RS",
380      metaState.getServerName(), rs.getServerName());
381    assertEquals("hbase:meta should be onlined on RS",
382      metaState.getState(), State.OPEN);
383
384    // Update meta state as OPENING, then kill master
385    // that simulates, that RS successfully deployed, but
386    // RPC was lost right before failure.
387    // region server should expire (how it can be verified?)
388    MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
389      rs.getServerName(), State.OPENING);
390    Region meta = rs.getFromOnlineRegions(HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
391    rs.removeFromOnlineRegions(meta, null);
392    ((HRegion)meta).close();
393
394    log("Aborting master");
395    activeMaster.abort("test-kill");
396    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
397    log("Master has aborted");
398
399    // Start up a new master
400    log("Starting up a new master");
401    activeMaster = cluster.startMaster().getMaster();
402    log("Waiting for master to be ready");
403    cluster.waitForActiveAndReadyMaster();
404    log("Master is ready");
405
406    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
407    log("Meta was assigned");
408
409    metaState =
410      MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
411    assertEquals("hbase:meta should be onlined on RS",
412      metaState.getServerName(), rs.getServerName());
413    assertEquals("hbase:meta should be onlined on RS",
414      metaState.getState(), State.OPEN);
415
416    // Update meta state as CLOSING, then kill master
417    // that simulates, that RS successfully deployed, but
418    // RPC was lost right before failure.
419    // region server should expire (how it can be verified?)
420    MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
421      rs.getServerName(), State.CLOSING);
422
423    log("Aborting master");
424    activeMaster.abort("test-kill");
425    cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
426    log("Master has aborted");
427
428    rs.getRSRpcServices().closeRegion(null, ProtobufUtil.buildCloseRegionRequest(
429      rs.getServerName(), HRegionInfo.FIRST_META_REGIONINFO.getRegionName()));
430
431    // Start up a new master
432    log("Starting up a new master");
433    activeMaster = cluster.startMaster().getMaster();
434    log("Waiting for master to be ready");
435    cluster.waitForActiveAndReadyMaster();
436    log("Master is ready");
437
438    TEST_UTIL.waitUntilNoRegionsInTransition(60000);
439    log("Meta was assigned");
440
441    // Done, shutdown the cluster
442    TEST_UTIL.shutdownMiniCluster();
443  }
