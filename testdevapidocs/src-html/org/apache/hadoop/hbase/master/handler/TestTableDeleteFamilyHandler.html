053@Category({MasterTests.class, LargeTests.class})
054public class TestTableDeleteFamilyHandler {
055
056  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
057  private static final TableName TABLENAME =
058      TableName.valueOf("column_family_handlers");
059  private static final byte[][] FAMILIES = new byte[][] { Bytes.toBytes("cf1"),
060      Bytes.toBytes("cf2"), Bytes.toBytes("cf3") };
061
062  /**
063   * Start up a mini cluster and put a small table of empty regions into it.
064   * 
065   * @throws Exception
066   */
067  @BeforeClass
068  public static void beforeAllTests() throws Exception {
069    TEST_UTIL.getConfiguration().setBoolean("dfs.support.append", true);
070    TEST_UTIL.startMiniCluster(2);
071  }
072
073  @AfterClass
074  public static void afterAllTests() throws Exception {
075    TEST_UTIL.shutdownMiniCluster();
076  }
077
078  @Before
079  public void setup() throws IOException, InterruptedException {
080    // Create a table of three families. This will assign a region.
081    TEST_UTIL.createTable(TABLENAME, FAMILIES);
082    Table t = TEST_UTIL.getConnection().getTable(TABLENAME);
083    while(TEST_UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager()
084        .getRegionStates().getRegionsInTransition().size() > 0) {
085      Thread.sleep(100);
086    }
087    // Create multiple regions in all the three column families
088    while(TEST_UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager()
089        .getRegionStates().getRegionsInTransition().size() > 0) {
090      Thread.sleep(100);
091    }
092    // Load the table with data for all families
093    TEST_UTIL.loadTable(t, FAMILIES);
094
095    TEST_UTIL.flush();
096
097    t.close();
098
099    TEST_UTIL.ensureSomeRegionServersAvailable(2);
100  }
101
102  @After
103  public void cleanup() throws Exception {
104    TEST_UTIL.deleteTable(TABLENAME);
105  }

107  @Test
108  public void deleteColumnFamilyWithMultipleRegions() throws Exception {
109    Admin admin = TEST_UTIL.getHBaseAdmin();
110    HTableDescriptor beforehtd = admin.getTableDescriptor(TABLENAME);
111
112    FileSystem fs = TEST_UTIL.getDFSCluster().getFileSystem();
113
114    // 1 - Check if table exists in descriptor
115    assertTrue(admin.isTableAvailable(TABLENAME));
116
117    // 2 - Check if all three families exist in descriptor
118    assertEquals(3, beforehtd.getColumnFamilies().length);
119    HColumnDescriptor[] families = beforehtd.getColumnFamilies();
120    for (int i = 0; i < families.length; i++) {
121      assertTrue(families[i].getNameAsString().equals("cf" + (i + 1)));
122    }
123
124    // 3 - Check if table exists in FS
125    Path tableDir = FSUtils.getTableDir(TEST_UTIL.getDefaultRootDirPath(), TABLENAME);
126    assertTrue(fs.exists(tableDir));
127
128    // 4 - Check if all the 3 column families exist in FS
129    FileStatus[] fileStatus = fs.listStatus(tableDir);
130    for (int i = 0; i < fileStatus.length; i++) {
131      if (fileStatus[i].isDirectory() == true) {
132        FileStatus[] cf = fs.listStatus(fileStatus[i].getPath(), new PathFilter() {
133          @Override
134          public boolean accept(Path p) {
135            if (p.getName().contains(HConstants.RECOVERED_EDITS_DIR)) {
136              return false;
137            }
138            return true;
139          }
140        });
141        int k = 1;
142        for (int j = 0; j < cf.length; j++) {
143          if (cf[j].isDirectory() == true
144              && cf[j].getPath().getName().startsWith(".") == false) {
145            assertEquals(cf[j].getPath().getName(), "cf" + k);
146            k++;
147          }
148        }
149      }
150    }
151
152    // TEST - Disable and delete the column family
153    admin.disableTable(TABLENAME);
154    admin.deleteColumnFamily(TABLENAME, Bytes.toBytes("cf2"));
155
156    // 5 - Check if only 2 column families exist in the descriptor
157    HTableDescriptor afterhtd = admin.getTableDescriptor(TABLENAME);
158    assertEquals(2, afterhtd.getColumnFamilies().length);
159    HColumnDescriptor[] newFamilies = afterhtd.getColumnFamilies();
160    assertTrue(newFamilies[0].getNameAsString().equals("cf1"));
161    assertTrue(newFamilies[1].getNameAsString().equals("cf3"));
162
163    // 6 - Check if the second column family is gone from the FS
164    fileStatus = fs.listStatus(tableDir);
165    for (int i = 0; i < fileStatus.length; i++) {
166      if (fileStatus[i].isDirectory() == true) {
167        FileStatus[] cf = fs.listStatus(fileStatus[i].getPath(), new PathFilter() {
168          @Override
169          public boolean accept(Path p) {
170            if (WALSplitter.isSequenceIdFile(p)) {
171              return false;
172            }
173            return true;
174          }
175        });
176        for (int j = 0; j < cf.length; j++) {
177          if (cf[j].isDirectory() == true) {
178            assertFalse(cf[j].getPath().getName().equals("cf2"));
179          }
180        }
181      }
182    }
183  }

185  @Test
186  public void deleteColumnFamilyTwice() throws Exception {
187
188    Admin admin = TEST_UTIL.getHBaseAdmin();
189    HTableDescriptor beforehtd = admin.getTableDescriptor(TABLENAME);
190    String cfToDelete = "cf1";
191
192    FileSystem fs = TEST_UTIL.getDFSCluster().getFileSystem();
193
194    // 1 - Check if table exists in descriptor
195    assertTrue(admin.isTableAvailable(TABLENAME));
196
197    // 2 - Check if all the target column family exist in descriptor
198    HColumnDescriptor[] families = beforehtd.getColumnFamilies();
199    Boolean foundCF = false;
200    int i;
201    for (i = 0; i < families.length; i++) {
202      if (families[i].getNameAsString().equals(cfToDelete)) {
203        foundCF = true;
204        break;
205      }
206    }
207    assertTrue(foundCF);
208
209    // 3 - Check if table exists in FS
210    Path tableDir = FSUtils.getTableDir(TEST_UTIL.getDefaultRootDirPath(), TABLENAME);
211    assertTrue(fs.exists(tableDir));
212
213    // 4 - Check if all the target column family exist in FS
214    FileStatus[] fileStatus = fs.listStatus(tableDir);
215    foundCF = false;
216    for (i = 0; i < fileStatus.length; i++) {
217      if (fileStatus[i].isDirectory() == true) {
218        FileStatus[] cf = fs.listStatus(fileStatus[i].getPath(), new PathFilter() {
219          @Override
220          public boolean accept(Path p) {
221            if (p.getName().contains(HConstants.RECOVERED_EDITS_DIR)) {
222              return false;
223            }
224            return true;
225          }
226        });
227        for (int j = 0; j < cf.length; j++) {
228          if (cf[j].isDirectory() == true && cf[j].getPath().getName().equals(cfToDelete)) {
229            foundCF = true;
230            break;
231          }
232        }
233      }
234      if (foundCF) {
235        break;
236      }
237    }
238    assertTrue(foundCF);
239
240    // TEST - Disable and delete the column family
241    if (admin.isTableEnabled(TABLENAME)) {
242      admin.disableTable(TABLENAME);
243    }
244    admin.deleteColumnFamily(TABLENAME, Bytes.toBytes(cfToDelete));
245
246    // 5 - Check if the target column family is gone from the FS
247    fileStatus = fs.listStatus(tableDir);
248    for (i = 0; i < fileStatus.length; i++) {
249      if (fileStatus[i].isDirectory() == true) {
250        FileStatus[] cf = fs.listStatus(fileStatus[i].getPath(), new PathFilter() {
251          @Override
252          public boolean accept(Path p) {
253            if (WALSplitter.isSequenceIdFile(p)) {
254              return false;
255            }
256            return true;
257          }
258        });
259        for (int j = 0; j < cf.length; j++) {
260          if (cf[j].isDirectory() == true) {
261            assertFalse(cf[j].getPath().getName().equals(cfToDelete));
262          }
263        }
264      }
265    }
266
267    try {
268      // Test: delete again
269      admin.deleteColumnFamily(TABLENAME, Bytes.toBytes(cfToDelete));
270      Assert.fail("Delete a non-exist column family should fail");
271    } catch (InvalidFamilyOperationException e) {
272      // Expected.
273    }
274  }
