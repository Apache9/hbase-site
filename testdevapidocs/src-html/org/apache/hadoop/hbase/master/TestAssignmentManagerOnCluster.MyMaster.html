111  /**
112   * This tests restarting meta regionserver
113   */
114  @Test (timeout=180000)
115  public void testRestartMetaRegionServer() throws Exception {
116    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
117    boolean stoppedARegionServer = false;
118    try {
119      HMaster master = cluster.getMaster();
120      RegionStates regionStates = master.getAssignmentManager().getRegionStates();
121      ServerName metaServerName = regionStates.getRegionServerOfRegion(
122        HRegionInfo.FIRST_META_REGIONINFO);
123      if (master.getServerName().equals(metaServerName)) {
124        // Move meta off master
125        metaServerName = cluster.getLiveRegionServerThreads()
126          .get(0).getRegionServer().getServerName();
127        master.move(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes(),
128                Bytes.toBytes(metaServerName.getServerName()));
129        TEST_UTIL.waitUntilNoRegionsInTransition(60000);
130      }
131      RegionState metaState =
132        MetaTableLocator.getMetaRegionState(master.getZooKeeper());
133      assertEquals("Meta should be not in transition",
134          metaState.getState(), RegionState.State.OPEN);
135      assertNotEquals("Meta should be moved off master",
136        metaState.getServerName(), master.getServerName());
137      assertEquals("Meta should be on the meta server",
138        metaState.getServerName(), metaServerName);
139      cluster.killRegionServer(metaServerName);
140      stoppedARegionServer = true;
141      cluster.waitForRegionServerToStop(metaServerName, 60000);
142
143      // Wait for SSH to finish
144      final ServerName oldServerName = metaServerName;
145      final ServerManager serverManager = master.getServerManager();
146      TEST_UTIL.waitFor(120000, 200, new Waiter.Predicate<Exception>() {
147        @Override
148        public boolean evaluate() throws Exception {
149          return serverManager.isServerDead(oldServerName)
150            && !serverManager.areDeadServersInProgress();
151        }
152      });
153
154      TEST_UTIL.waitUntilNoRegionsInTransition(60000);
155      // Now, make sure meta is assigned
156      assertTrue("Meta should be assigned",
157        regionStates.isRegionOnline(HRegionInfo.FIRST_META_REGIONINFO));
158      // Now, make sure meta is registered in zk
159      metaState = MetaTableLocator.getMetaRegionState(master.getZooKeeper());
160      assertEquals("Meta should be not in transition",
161          metaState.getState(), RegionState.State.OPEN);
162      assertEquals("Meta should be assigned", metaState.getServerName(),
163        regionStates.getRegionServerOfRegion(HRegionInfo.FIRST_META_REGIONINFO));
164      assertNotEquals("Meta should be assigned on a different server",
165        metaState.getServerName(), metaServerName);
166    } finally {
167      if (stoppedARegionServer) {
168        cluster.startRegionServer();
169      }
170    }
171  }

216    TEST_UTIL.getMiniHBaseCluster().stopMaster(0);

219
220    ServerName deadServer = null;
221    HMaster master = null;
222    try {
223      HTableDescriptor desc = new HTableDescriptor(table);
224      desc.addFamily(new HColumnDescriptor(FAMILY));
225      admin.createTable(desc);
226
227      Table meta = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME);
228      final HRegionInfo hri = new HRegionInfo(
229        desc.getTableName(), Bytes.toBytes("A"), Bytes.toBytes("Z"));
230      MetaTableAccessor.addRegionToMeta(meta, hri);
231
232      master = TEST_UTIL.getHBaseCluster().getMaster();
233      Set<ServerName> onlineServers = master.serverManager.getOnlineServers().keySet();
234      assertFalse("There should be some servers online", onlineServers.isEmpty());
235
236      // Use the first server as the destination server
237      ServerName destServer = onlineServers.iterator().next();
238
239      // Created faked dead server that is still online in master
240      deadServer = ServerName.valueOf(destServer.getHostname(),
241          destServer.getPort(), destServer.getStartcode() - 100L);
242      master.serverManager.recordNewServerWithLock(deadServer, ServerLoad.EMPTY_SERVERLOAD);
243
244      final AssignmentManager am = master.getAssignmentManager();
245      RegionPlan plan = new RegionPlan(hri, null, deadServer);
246      am.addPlan(hri.getEncodedName(), plan);
247      master.assignRegion(hri);
248
249      TEST_UTIL.waitFor(60000, new Waiter.Predicate<Exception>() {
250        @Override
251        public boolean evaluate() throws Exception {
252          return ! am.getRegionStates().isRegionInTransition(hri);
253        }
254      });
255
256    assertFalse("Region should be assigned", am.getRegionStates().isRegionInTransition(hri));
257    } finally {
258      if (deadServer != null) {
259        master.serverManager.expireServer(deadServer);
260      }
261
262      TEST_UTIL.deleteTable(table);
263
264      // reset the value for other tests
265      TEST_UTIL.getMiniHBaseCluster().getConf().setInt("hbase.assignment.maximum.attempts", 3);
266      ServerName masterServerName = TEST_UTIL.getMiniHBaseCluster().getMaster().getServerName();
267      TEST_UTIL.getMiniHBaseCluster().stopMaster(masterServerName);
268      TEST_UTIL.getMiniHBaseCluster().startMaster();
269    }

395      am.balance(new RegionPlan(hri, serverName, destServerName));
396      assertFalse("The region should not be in transition",
397        regionStates.isRegionInTransition(hri));
398    } finally {

564  /**
565   * This tests region open failure which is not recoverable
566   */
567  @Test (timeout=60000)
568  public void testOpenFailedUnrecoverable() throws Exception {
569    TableName table =
570        TableName.valueOf("testOpenFailedUnrecoverable");
571    try {
572      HTableDescriptor desc = new HTableDescriptor(table);
573      desc.addFamily(new HColumnDescriptor(FAMILY));
574      admin.createTable(desc);
575
576      Table meta = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME);
577      HRegionInfo hri = new HRegionInfo(
578        desc.getTableName(), Bytes.toBytes("A"), Bytes.toBytes("Z"));
579      MetaTableAccessor.addRegionToMeta(meta, hri);
580
581      FileSystem fs = FileSystem.get(conf);
582      Path tableDir= FSUtils.getTableDir(FSUtils.getRootDir(conf), table);
583      Path regionDir = new Path(tableDir, hri.getEncodedName());
584      // create a file named the same as the region dir to
585      // mess up with region opening
586      fs.create(regionDir, true);
587
588      HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
589      master.assignRegion(hri);
590      AssignmentManager am = master.getAssignmentManager();
591      assertFalse(am.waitForAssignment(hri));
592
593      RegionState state = am.getRegionStates().getRegionState(hri);
594      assertEquals(RegionState.State.FAILED_OPEN, state.getState());
595      // Failed to open due to file system issue. Region state should
596      // carry the opening region server so that we can force close it
597      // later on before opening it again. See HBASE-9092.
598      assertNotNull(state.getServerName());
599
600      // remove the blocking file, so that region can be opened
601      fs.delete(regionDir, true);
602      master.assignRegion(hri);
603      assertTrue(am.waitForAssignment(hri));
604
605      ServerName serverName = master.getAssignmentManager().
606        getRegionStates().getRegionServerOfRegion(hri);

1170  static class MyLoadBalancer extends StochasticLoadBalancer {
1171    // For this region, if specified, always assign to nowhere
1172    static volatile String controledRegion = null;
1173
1174    static volatile Integer countRegionServers = null;
1175    static AtomicInteger counter = new AtomicInteger(0);
1176
1177    @Override
1178    public ServerName randomAssignment(HRegionInfo regionInfo,
1179        List<ServerName> servers) {
1180      if (regionInfo.getEncodedName().equals(controledRegion)) {
1181        return null;
1182      }
1183      return super.randomAssignment(regionInfo, servers);
1184    }
1185
1186    @Override
1187    public Map<ServerName, List<HRegionInfo>> roundRobinAssignment(
1188        List<HRegionInfo> regions, List<ServerName> servers) {
1189      if (countRegionServers != null && services != null) {
1190        int regionServers = services.getServerManager().countOfRegionServers();
1191        if (regionServers < countRegionServers.intValue()) {
1192          // Let's wait till more region servers join in.
1193          // Before that, fail region assignments.
1194          counter.incrementAndGet();
1195          return null;
1196        }
1197      }
1198      return super.roundRobinAssignment(regions, servers);
1199    }
1200  }

1251  public static class MyRegionObserver extends BaseRegionObserver {
1252    // If enabled, fail all preClose calls
1253    static AtomicBoolean preCloseEnabled = new AtomicBoolean(false);
1254
1255    // If enabled, stall postClose calls
1256    static AtomicBoolean postCloseEnabled = new AtomicBoolean(false);
1257
1258    // If enabled, stall postOpen calls
1259    static AtomicBoolean postOpenEnabled = new AtomicBoolean(false);
1260
1261    // A flag to track if postOpen is called
1262    static volatile boolean postOpenCalled = false;
1263
1264    @Override
1265    public void preClose(ObserverContext<RegionCoprocessorEnvironment> c,
1266        boolean abortRequested) throws IOException {
1267      if (preCloseEnabled.get()) throw new IOException("fail preClose from coprocessor");
1268    }
1269
1270    @Override
1271    public void postClose(ObserverContext<RegionCoprocessorEnvironment> c,
1272        boolean abortRequested) {
1273      stallOnFlag(postCloseEnabled);
1274    }
1275
1276    @Override
1277    public void postOpen(ObserverContext<RegionCoprocessorEnvironment> c) {
1278      postOpenCalled = true;
1279      stallOnFlag(postOpenEnabled);
1280    }
1281
1282    private void stallOnFlag(final AtomicBoolean flag) {
1283      try {
1284        // If enabled, stall
1285        while (flag.get()) {
1286          Thread.sleep(1000);
1287        }
1288      } catch (InterruptedException ie) {
1289        Thread.currentThread().interrupt();
1290      }
1291    }
