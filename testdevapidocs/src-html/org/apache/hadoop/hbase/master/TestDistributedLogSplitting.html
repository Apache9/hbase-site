208  @Ignore("DLR is broken by HBASE-12751") @Test (timeout=300000)
209  public void testRecoveredEdits() throws Exception {
210    LOG.info("testRecoveredEdits");
211    conf.setLong("hbase.regionserver.hlog.blocksize", 30 * 1024); // create more than one wal
212    conf.setBoolean(HConstants.DISTRIBUTED_LOG_REPLAY_KEY, false);
213    startCluster(NUM_RS);
214
215    final int NUM_LOG_LINES = 1000;
216    final SplitLogManager slm = master.getMasterWalManager().getSplitLogManager();
217    // turn off load balancing to prevent regions from moving around otherwise
218    // they will consume recovered.edits
219    master.balanceSwitch(false);
220    FileSystem fs = master.getMasterFileSystem().getFileSystem();
221
222    List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
223
224    Path rootdir = FSUtils.getRootDir(conf);
225
226    Table t = installTable(new ZooKeeperWatcher(conf, "table-creation", null),
227        "table", "family", 40);
228    try {
229      TableName table = t.getName();
230      List<HRegionInfo> regions = null;
231      HRegionServer hrs = null;
232      for (int i = 0; i < NUM_RS; i++) {
233        boolean foundRs = false;
234        hrs = rsts.get(i).getRegionServer();
235        regions = ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices());
236        for (HRegionInfo region : regions) {
237          if (region.getTable().getNameAsString().equalsIgnoreCase("table")) {
238            foundRs = true;
239            break;
240          }
241        }
242        if (foundRs) break;
243      }
244      final Path logDir = new Path(rootdir, AbstractFSWALProvider.getWALDirectoryName(hrs
245          .getServerName().toString()));
246
247      LOG.info("#regions = " + regions.size());
248      Iterator<HRegionInfo> it = regions.iterator();
249      while (it.hasNext()) {
250        HRegionInfo region = it.next();
251        if (region.getTable().getNamespaceAsString()
252            .equals(NamespaceDescriptor.SYSTEM_NAMESPACE_NAME_STR)) {
253          it.remove();
254        }
255      }
256
257      makeWAL(hrs, regions, "table", "family", NUM_LOG_LINES, 100);
258
259      slm.splitLogDistributed(logDir);
260
261      int count = 0;
262      for (HRegionInfo hri : regions) {
263
264        Path tdir = FSUtils.getTableDir(rootdir, table);
265        Path editsdir =
266            WALSplitter.getRegionDirRecoveredEditsDir(
267                HRegion.getRegionDir(tdir, hri.getEncodedName()));
268        LOG.debug("checking edits dir " + editsdir);
269        FileStatus[] files = fs.listStatus(editsdir, new PathFilter() {
270          @Override
271          public boolean accept(Path p) {
272            if (WALSplitter.isSequenceIdFile(p)) {
273              return false;
274            }
275            return true;
276          }
277        });
278        assertTrue(
279            "edits dir should have more than a single file in it. instead has " + files.length,
280            files.length > 1);
281        for (int i = 0; i < files.length; i++) {
282          int c = countWAL(files[i].getPath(), fs, conf);
283          count += c;
284        }
285        LOG.info(count + " edits in " + files.length + " recovered edits files.");
286      }
287
288      // check that the log file is moved
289      assertFalse(fs.exists(logDir));
290
291      assertEquals(NUM_LOG_LINES, count);
292    } finally {
293      if (t != null) t.close();
294    }
295  }

1562  public void makeWAL(HRegionServer hrs, List<HRegionInfo> regions, String tname, String fname,
1563      int num_edits, int edit_size, boolean cleanShutdown) throws IOException {
1564    TableName fullTName = TableName.valueOf(tname);
1565    // remove root and meta region
1566    regions.remove(HRegionInfo.FIRST_META_REGIONINFO);
1567    // using one sequenceId for edits across all regions is ok.
1568    final AtomicLong sequenceId = new AtomicLong(10);
1569
1570
1571    for(Iterator<HRegionInfo> iter = regions.iterator(); iter.hasNext(); ) {
1572      HRegionInfo regionInfo = iter.next();
1573      if(regionInfo.getTable().isSystemTable()) {
1574        iter.remove();
1575      }
1576    }
1577    HTableDescriptor htd = new HTableDescriptor(fullTName);
1578    byte[] family = Bytes.toBytes(fname);
1579    htd.addFamily(new HColumnDescriptor(family));
1580    byte[] value = new byte[edit_size];
1581
1582    List<HRegionInfo> hris = new ArrayList<HRegionInfo>();
1583    for (HRegionInfo region : regions) {
1584      if (!region.getTable().getNameAsString().equalsIgnoreCase(tname)) {
1585        continue;
1586      }
1587      hris.add(region);
1588    }
1589    LOG.info("Creating wal edits across " + hris.size() + " regions.");
1590    for (int i = 0; i < edit_size; i++) {
1591      value[i] = (byte) ('a' + (i % 26));
1592    }
1593    int n = hris.size();
1594    int[] counts = new int[n];
1595    // sync every ~30k to line up with desired wal rolls
1596    final int syncEvery = 30 * 1024 / edit_size;
1597    if (n > 0) {
1598      for (int i = 0; i < num_edits; i += 1) {
1599        WALEdit e = new WALEdit();
1600        HRegionInfo curRegionInfo = hris.get(i % n);
1601        final WAL log = hrs.getWAL(curRegionInfo);
1602        byte[] startRow = curRegionInfo.getStartKey();
1603        if (startRow == null || startRow.length == 0) {
1604          startRow = new byte[] { 0, 0, 0, 0, 1 };
1605        }
1606        byte[] row = Bytes.incrementBytes(startRow, counts[i % n]);
1607        row = Arrays.copyOfRange(row, 3, 8); // use last 5 bytes because
1608        // HBaseTestingUtility.createMultiRegions use 5 bytes
1609        // key
1610        byte[] qualifier = Bytes.toBytes("c" + Integer.toString(i));
1611        e.add(new KeyValue(row, family, qualifier, System.currentTimeMillis(), value));
1612        log.append(curRegionInfo,
1613            new HLogKey(curRegionInfo.getEncodedNameAsBytes(), fullTName,
1614                System.currentTimeMillis()), e, true);
1615        if (0 == i % syncEvery) {
1616          log.sync();
1617        }
1618        counts[i % n] += 1;
1619      }
1620    }
1621    // done as two passes because the regions might share logs. shutdown is idempotent, but sync
1622    // will cause errors if done after.
1623    for (HRegionInfo info : hris) {
1624      final WAL log = hrs.getWAL(info);
1625      log.sync();
1626    }
1627    if (cleanShutdown) {
1628      for (HRegionInfo info : hris) {
1629        final WAL log = hrs.getWAL(info);
1630        log.shutdown();
1631      }
1632    }
1633    for (int i = 0; i < n; i++) {
1634      LOG.info("region " + hris.get(i).getRegionNameAsString() + " has " + counts[i] + " edits");
1635    }
1636    return;
1637  }

1735  /**
1736   * Find a RS that has regions of a table.
1737   * @param hasMetaRegion when true, the returned RS has hbase:meta region as well
1738   * @param tableName
1739   * @return
1740   * @throws Exception
1741   */
1742  private HRegionServer findRSToKill(boolean hasMetaRegion, String tableName) throws Exception {
1743    List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
1744    List<HRegionInfo> regions = null;
1745    HRegionServer hrs = null;
1746
1747    for (RegionServerThread rst: rsts) {
1748      hrs = rst.getRegionServer();
1749      while (rst.isAlive() && !hrs.isOnline()) {
1750        Thread.sleep(100);
1751      }
1752      if (!rst.isAlive()) {
1753        continue;
1754      }
1755      boolean isCarryingMeta = false;
1756      boolean foundTableRegion = false;
1757      regions = ProtobufUtil.getOnlineRegions(hrs.getRSRpcServices());
1758      for (HRegionInfo region : regions) {
1759        if (region.isMetaRegion()) {
1760          isCarryingMeta = true;
1761        }
1762        if (tableName == null || region.getTable().getNameAsString().equals(tableName)) {
1763          foundTableRegion = true;
1764        }
1765        if (foundTableRegion && (isCarryingMeta || !hasMetaRegion)) {
1766          break;
1767        }
1768      }
1769      if (isCarryingMeta && hasMetaRegion) {
1770        // clients ask for a RS with META
1771        if (!foundTableRegion) {
1772          final HRegionServer destRS = hrs;
1773          // the RS doesn't have regions of the specified table so we need move one to this RS
1774          List<HRegionInfo> tableRegions =
1775              TEST_UTIL.getHBaseAdmin().getTableRegions(TableName.valueOf(tableName));
1776          final HRegionInfo hri = tableRegions.get(0);
1777          TEST_UTIL.getHBaseAdmin().move(hri.getEncodedNameAsBytes(),
1778              Bytes.toBytes(destRS.getServerName().getServerName()));
1779          // wait for region move completes
1780          final RegionStates regionStates =
1781              TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
1782          TEST_UTIL.waitFor(45000, 200, new Waiter.Predicate<Exception>() {
1783            @Override
1784            public boolean evaluate() throws Exception {
1785              ServerName sn = regionStates.getRegionServerOfRegion(hri);
1786              return (sn != null && sn.equals(destRS.getServerName()));
1787            }
1788          });
1789        }
1790        return hrs;
1791      } else if (hasMetaRegion || isCarryingMeta) {
1792        continue;
1793      }
1794      if (foundTableRegion) break;
1795    }
1796
1797    return hrs;
1798  }
