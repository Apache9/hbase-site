129  @Test
130  public void testIsFileDeletable() throws IOException, ReplicationException {
131    // 1. Create a file
132    Path file = new Path(root, "testIsFileDeletableWithNoHFileRefs");
133    fs.createNewFile(file);
134    // 2. Assert file is successfully created
135    assertTrue("Test file not created!", fs.exists(file));
136    ReplicationHFileCleaner cleaner = new ReplicationHFileCleaner();
137    cleaner.setConf(conf);
138    // 3. Assert that file as is should be deletable
139    assertTrue("Cleaner should allow to delete this file as there is no hfile reference node "
140        + "for it in the queue.",
141      cleaner.isFileDeletable(fs.getFileStatus(file)));
142
143    List<Pair<Path, Path>> files = new ArrayList<>(1);
144    files.add(new Pair<>(null, file));
145    // 4. Add the file to hfile-refs queue
146    rq.addHFileRefs(peerId, files);
147    // 5. Assert file should not be deletable
148    assertFalse("Cleaner should not allow to delete this file as there is a hfile reference node "
149        + "for it in the queue.",
150      cleaner.isFileDeletable(fs.getFileStatus(file)));
151  }

153  @Test
154  public void testGetDeletableFiles() throws Exception {
155    // 1. Create two files and assert that they do not exist
156    Path notDeletablefile = new Path(root, "testGetDeletableFiles_1");
157    fs.createNewFile(notDeletablefile);
158    assertTrue("Test file not created!", fs.exists(notDeletablefile));
159    Path deletablefile = new Path(root, "testGetDeletableFiles_2");
160    fs.createNewFile(deletablefile);
161    assertTrue("Test file not created!", fs.exists(deletablefile));
162
163    List<FileStatus> files = new ArrayList<>(2);
164    FileStatus f = new FileStatus();
165    f.setPath(deletablefile);
166    files.add(f);
167    f = new FileStatus();
168    f.setPath(notDeletablefile);
169    files.add(f);
170
171    List<Pair<Path, Path>> hfiles = new ArrayList<>(1);
172    hfiles.add(new Pair<>(null, notDeletablefile));
173    // 2. Add one file to hfile-refs queue
174    rq.addHFileRefs(peerId, hfiles);
175
176    ReplicationHFileCleaner cleaner = new ReplicationHFileCleaner();
177    cleaner.setConf(conf);
178    Iterator<FileStatus> deletableFilesIterator = cleaner.getDeletableFiles(files).iterator();
179    int i = 0;
180    while (deletableFilesIterator.hasNext() && i < 2) {
181      i++;
182    }
183    // 5. Assert one file should not be deletable and it is present in the list returned
184    if (i > 2) {
185      fail("File " + notDeletablefile
186          + " should not be deletable as its hfile reference node is not added.");
187    }
188    assertTrue(deletableFilesIterator.next().getPath().equals(deletablefile));
189  }

217  /**
218   * ReplicationHFileCleaner should be able to ride over ZooKeeper errors without aborting.
219   */
220  @Test
221  public void testZooKeeperAbort() throws Exception {
222    ReplicationHFileCleaner cleaner = new ReplicationHFileCleaner();
223
224    List<FileStatus> dummyFiles =
225        Lists.newArrayList(new FileStatus(100, false, 3, 100, System.currentTimeMillis(), new Path(
226            "hfile1")), new FileStatus(100, false, 3, 100, System.currentTimeMillis(), new Path(
227            "hfile2")));
228
229    FaultyZooKeeperWatcher faultyZK =
230        new FaultyZooKeeperWatcher(conf, "testZooKeeperAbort-faulty", null);
231    try {
232      faultyZK.init();
233      cleaner.setConf(conf, faultyZK);
234      // should keep all files due to a ConnectionLossException getting the queues znodes
235      Iterable<FileStatus> toDelete = cleaner.getDeletableFiles(dummyFiles);
236      assertFalse(toDelete.iterator().hasNext());
237      assertFalse(cleaner.isStopped());
238    } finally {
239      faultyZK.close();
240    }
241
242    // when zk is working both files should be returned
243    cleaner = new ReplicationHFileCleaner();
244    ZKWatcher zkw = new ZKWatcher(conf, "testZooKeeperAbort-normal", null);
245    try {
246      cleaner.setConf(conf, zkw);
247      Iterable<FileStatus> filesToDelete = cleaner.getDeletableFiles(dummyFiles);
248      Iterator<FileStatus> iter = filesToDelete.iterator();
249      assertTrue(iter.hasNext());
250      assertEquals(new Path("hfile1"), iter.next().getPath());
251      assertTrue(iter.hasNext());
252      assertEquals(new Path("hfile2"), iter.next().getPath());
253      assertFalse(iter.hasNext());
254    } finally {
255      zkw.close();
256    }
257  }

341  static class FaultyZooKeeperWatcher extends ZKWatcher {
342    private RecoverableZooKeeper zk;
343    public FaultyZooKeeperWatcher(Configuration conf, String identifier, Abortable abortable)
344        throws ZooKeeperConnectionException, IOException {
345      super(conf, identifier, abortable);
346    }
347
348    public void init() throws Exception {
349      this.zk = spy(super.getRecoverableZooKeeper());
350      doThrow(new KeeperException.ConnectionLossException())
351          .when(zk).getData("/hbase/replication/hfile-refs", null, new Stat());
352    }
353
354    public RecoverableZooKeeper getRecoverableZooKeeper() {
355      return zk;
356    }
