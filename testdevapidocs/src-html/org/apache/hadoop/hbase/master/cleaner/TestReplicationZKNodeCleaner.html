019package org.apache.hadoop.hbase.master.cleaner;
020
021import static org.junit.Assert.assertEquals;
022import static org.junit.Assert.assertTrue;
023
024import java.util.List;
025import java.util.Map;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.hbase.HBaseTestingUtility;
029import org.apache.hadoop.hbase.replication.ReplicationFactory;
030import org.apache.hadoop.hbase.replication.ReplicationQueues;
031import org.apache.hadoop.hbase.replication.ReplicationQueuesArguments;
032import org.apache.hadoop.hbase.replication.ReplicationQueuesZKImpl;
033import org.apache.hadoop.hbase.testclassification.MasterTests;
034import org.apache.hadoop.hbase.testclassification.MediumTests;
035import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
036import org.junit.AfterClass;
037import org.junit.BeforeClass;
038import org.junit.Test;
039import org.junit.experimental.categories.Category;
040
041@Category({ MasterTests.class, MediumTests.class })
042public class TestReplicationZKNodeCleaner {
043  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
044
045  private final String ID_ONE = "1";
046  private final String SERVER_ONE = "server1";
047  private final String ID_TWO = "2";
048  private final String SERVER_TWO = "server2";
049
050  private final Configuration conf;
051  private final ZooKeeperWatcher zkw;
052  private final ReplicationQueues repQueues;
053
054  public TestReplicationZKNodeCleaner() throws Exception {
055    conf = TEST_UTIL.getConfiguration();
056    zkw = new ZooKeeperWatcher(conf, "TestReplicationZKNodeCleaner", null);
057    repQueues = ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, null,
058        zkw));
059    assertTrue(repQueues instanceof ReplicationQueuesZKImpl);
060  }
061
062  @BeforeClass
063  public static void setUpBeforeClass() throws Exception {
064    TEST_UTIL.getConfiguration().setInt("hbase.master.cleaner.interval", 10000);
065    TEST_UTIL.startMiniCluster();
066  }
067
068  @AfterClass
069  public static void tearDownAfterClass() throws Exception {
070    TEST_UTIL.shutdownMiniCluster();
071  }
072
073  @Test
074  public void testReplicationZKNodeCleaner() throws Exception {
075    repQueues.init(SERVER_ONE);
076    // add queue for ID_ONE which isn't exist
077    repQueues.addLog(ID_ONE, "file1");
078
079    ReplicationZKNodeCleaner cleaner = new ReplicationZKNodeCleaner(conf, zkw, null);
080    Map<String, List<String>> undeletedQueues = cleaner.getUnDeletedQueues();
081    assertEquals(1, undeletedQueues.size());
082    assertTrue(undeletedQueues.containsKey(SERVER_ONE));
083    assertEquals(1, undeletedQueues.get(SERVER_ONE).size());
084    assertTrue(undeletedQueues.get(SERVER_ONE).contains(ID_ONE));
085
086    // add a recovery queue for ID_TWO which isn't exist
087    repQueues.addLog(ID_TWO + "-" + SERVER_TWO, "file2");
088
089    undeletedQueues = cleaner.getUnDeletedQueues();
090    assertEquals(1, undeletedQueues.size());
091    assertTrue(undeletedQueues.containsKey(SERVER_ONE));
092    assertEquals(2, undeletedQueues.get(SERVER_ONE).size());
093    assertTrue(undeletedQueues.get(SERVER_ONE).contains(ID_ONE));
094    assertTrue(undeletedQueues.get(SERVER_ONE).contains(ID_TWO + "-" + SERVER_TWO));
095
096    cleaner.removeQueues(undeletedQueues);
097    undeletedQueues = cleaner.getUnDeletedQueues();
098    assertEquals(0, undeletedQueues.size());
099  }
100
101  @Test
102  public void testReplicationZKNodeCleanerChore() throws Exception {
103    repQueues.init(SERVER_ONE);
104    // add queue for ID_ONE which isn't exist
105    repQueues.addLog(ID_ONE, "file1");
106    // add a recovery queue for ID_TWO which isn't exist
107    repQueues.addLog(ID_TWO + "-" + SERVER_TWO, "file2");
108
109    // Wait the cleaner chore to run
110    Thread.sleep(20000);
111
112    ReplicationZKNodeCleaner cleaner = new ReplicationZKNodeCleaner(conf, zkw, null);
113    assertEquals(0, cleaner.getUnDeletedQueues().size());
114  }
115}
