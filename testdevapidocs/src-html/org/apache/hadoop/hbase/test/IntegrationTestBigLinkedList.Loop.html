123/**
124 * This is an integration test borrowed from goraci, written by Keith Turner,
125 * which is in turn inspired by the Accumulo test called continous ingest (ci).
126 * The original source code can be found here:
127 * https://github.com/keith-turner/goraci
128 * https://github.com/enis/goraci/
129 *
130 * Apache Accumulo [0] has a simple test suite that verifies that data is not
131 * lost at scale. This test suite is called continuous ingest. This test runs
132 * many ingest clients that continually create linked lists containing 25
133 * million nodes. At some point the clients are stopped and a map reduce job is
134 * run to ensure no linked list has a hole. A hole indicates data was lost.??
135 *
136 * The nodes in the linked list are random. This causes each linked list to
137 * spread across the table. Therefore if one part of a table loses data, then it
138 * will be detected by references in another part of the table.
139 *
140 * THE ANATOMY OF THE TEST
141 *
142 * Below is rough sketch of how data is written. For specific details look at
143 * the Generator code.
144 *
145 * 1 Write out 1 million nodes? 2 Flush the client? 3 Write out 1 million that
146 * reference previous million? 4 If this is the 25th set of 1 million nodes,
147 * then update 1st set of million to point to last? 5 goto 1
148 *
149 * The key is that nodes only reference flushed nodes. Therefore a node should
150 * never reference a missing node, even if the ingest client is killed at any
151 * point in time.
152 *
153 * When running this test suite w/ Accumulo there is a script running in
154 * parallel called the Aggitator that randomly and continuously kills server
155 * processes.?? The outcome was that many data loss bugs were found in Accumulo
156 * by doing this.? This test suite can also help find bugs that impact uptime
157 * and stability when? run for days or weeks.??
158 *
159 * This test suite consists the following? - a few Java programs? - a little
160 * helper script to run the java programs - a maven script to build it.??
161 *
162 * When generating data, its best to have each map task generate a multiple of
163 * 25 million. The reason for this is that circular linked list are generated
164 * every 25M. Not generating a multiple in 25M will result in some nodes in the
165 * linked list not having references. The loss of an unreferenced node can not
166 * be detected.
167 *
168 *
169 * Below is a description of the Java programs
170 *
171 * Generator - A map only job that generates data. As stated previously,?
172 * its best to generate data in multiples of 25M.
173 *
174 * Verify - A map reduce job that looks for holes. Look at the counts after running. REFERENCED and
175 * UNREFERENCED are? ok, any UNDEFINED counts are bad. Do not run at the? same
176 * time as the Generator.
177 *
178 * Walker - A standalone program that start following a linked list? and emits timing info.??
179 *
180 * Print - A standalone program that prints nodes in the linked list
181 *
182 * Delete - A standalone program that deletes a single node
183 *
184 * This class can be run as a unit test, as an integration test, or from the command line
185 */

374    static class GeneratorMapper
375      extends Mapper<BytesWritable, NullWritable, NullWritable, NullWritable> {
376
377      byte[][] first = null;
378      byte[][] prev = null;
379      byte[][] current = null;
380      byte[] id;
381      long count = 0;
382      int i;
383      BufferedMutator mutator;
384      Connection connection;
385      long numNodes;
386      long wrap;
387      int width;
388      boolean multipleUnevenColumnFamilies;
389      byte[] tinyValue = new byte[] { 't' };
390      byte[] bigValue = null;
391
392      @Override
393      protected void setup(Context context) throws IOException, InterruptedException {
394        id = Bytes.toBytes("Job: "+context.getJobID() + " Task: " + context.getTaskAttemptID());
395        this.connection = ConnectionFactory.createConnection(context.getConfiguration());
396        instantiateHTable();
397        this.width = context.getConfiguration().getInt(GENERATOR_WIDTH_KEY, WIDTH_DEFAULT);
398        current = new byte[this.width][];
399        int wrapMultiplier = context.getConfiguration().getInt(GENERATOR_WRAP_KEY, WRAP_DEFAULT);
400        this.wrap = (long)wrapMultiplier * width;
401        this.numNodes = context.getConfiguration().getLong(
402            GENERATOR_NUM_ROWS_PER_MAP_KEY, (long)WIDTH_DEFAULT * WRAP_DEFAULT);
403        if (this.numNodes < this.wrap) {
404          this.wrap = this.numNodes;
405        }
406        this.multipleUnevenColumnFamilies = isMultiUnevenColumnFamilies(context.getConfiguration());
407      }
408
409      protected void instantiateHTable() throws IOException {
410        mutator = connection.getBufferedMutator(
411            new BufferedMutatorParams(getTableName(connection.getConfiguration()))
412                .writeBufferSize(4 * 1024 * 1024));
413      }
414
415      @Override
416      protected void cleanup(Context context) throws IOException ,InterruptedException {
417        mutator.close();
418        connection.close();
419      }
420
421      @Override
422      protected void map(BytesWritable key, NullWritable value, Context output) throws IOException {
423        current[i] = new byte[key.getLength()];
424        System.arraycopy(key.getBytes(), 0, current[i], 0, key.getLength());
425        if (++i == current.length) {
426          LOG.info("Persisting current.length=" + current.length + ", count=" + count + ", id=" +
427            Bytes.toStringBinary(id) + ", current=" + Bytes.toStringBinary(current[0]) +
428            ", i=" + i);
429          persist(output, count, prev, current, id);
430          i = 0;
431
432          if (first == null) {
433            first = current;
434          }
435          prev = current;
436          current = new byte[this.width][];
437
438          count += current.length;
439          output.setStatus("Count " + count);
440
441          if (count % wrap == 0) {
442            // this block of code turns the 1 million linked list of length 25 into one giant
443            //circular linked list of 25 million
444            circularLeftShift(first);
445
446            persist(output, -1, prev, first, null);
447
448            first = null;
449            prev = null;
450          }
451        }
452      }
453
454      private static <T> void circularLeftShift(T[] first) {
455        T ez = first[0];
456        System.arraycopy(first, 1, first, 0, first.length - 1);
457        first[first.length - 1] = ez;
458      }
459
460      protected void persist(Context output, long count, byte[][] prev, byte[][] current, byte[] id)
461          throws IOException {
462        for (int i = 0; i < current.length; i++) {
463          Put put = new Put(current[i]);
464          put.addColumn(FAMILY_NAME, COLUMN_PREV, prev == null ? NO_KEY : prev[i]);
465
466          if (count >= 0) {
467            put.addColumn(FAMILY_NAME, COLUMN_COUNT, Bytes.toBytes(count + i));
468          }
469          if (id != null) {
470            put.addColumn(FAMILY_NAME, COLUMN_CLIENT, id);
471          }
472          // See if we are to write multiple columns.
473          if (this.multipleUnevenColumnFamilies) {
474            // Use any column name.
475            put.addColumn(TINY_FAMILY_NAME, TINY_FAMILY_NAME, this.tinyValue);
476            // If we've not allocated bigValue, do it now. Reuse same value each time.
477            if (this.bigValue == null) {
478              this.bigValue = new byte[current[i].length * 10];
479              ThreadLocalRandom.current().nextBytes(this.bigValue);
480            }
481            // Use any column name.
482            put.addColumn(BIG_FAMILY_NAME, BIG_FAMILY_NAME, this.bigValue);
483          }
484          mutator.mutate(put);
485
486          if (i % 1000 == 0) {
487            // Tickle progress every so often else maprunner will think us hung
488            output.progress();
489          }
490        }
491
492        mutator.flush();
493      }

514    protected void createSchema() throws IOException {
515      Configuration conf = getConf();
516      TableName tableName = getTableName(conf);
517      try (Connection conn = ConnectionFactory.createConnection(conf);
518          Admin admin = conn.getAdmin()) {
519        if (!admin.tableExists(tableName)) {
520          HTableDescriptor htd = new HTableDescriptor(getTableName(getConf()));
521          htd.addFamily(new HColumnDescriptor(FAMILY_NAME));
522          // Always add these families. Just skip writing to them when we do not test per CF flush.
523          htd.addFamily(new HColumnDescriptor(BIG_FAMILY_NAME));
524          htd.addFamily(new HColumnDescriptor(TINY_FAMILY_NAME));
525          // if -DuseMob=true force all data through mob path.
526          if (conf.getBoolean("useMob", false)) {
527            for (HColumnDescriptor hcd : htd.getColumnFamilies() ) {
528              hcd.setMobEnabled(true);
529              hcd.setMobThreshold(4);
530            }
531          }
532
533          // If we want to pre-split compute how many splits.
534          if (conf.getBoolean(HBaseTestingUtility.PRESPLIT_TEST_TABLE_KEY,
535              HBaseTestingUtility.PRESPLIT_TEST_TABLE)) {
536            int numberOfServers = admin.getClusterStatus().getServers().size();
537            if (numberOfServers == 0) {
538              throw new IllegalStateException("No live regionservers");
539            }
540            int regionsPerServer = conf.getInt(HBaseTestingUtility.REGIONS_PER_SERVER_KEY,
541                HBaseTestingUtility.DEFAULT_REGIONS_PER_SERVER);
542            int totalNumberOfRegions = numberOfServers * regionsPerServer;
543            LOG.info("Number of live regionservers: " + numberOfServers + ", " +
544                "pre-splitting table into " + totalNumberOfRegions + " regions " +
545                "(default regions per server: " + regionsPerServer + ")");
546
547
548            byte[][] splits = new RegionSplitter.UniformSplit().split(totalNumberOfRegions);
549
550            admin.createTable(htd, splits);
551          } else {
552            // Looks like we're just letting things play out.
553            // Create a table with on region by default.
554            // This will make the splitting work hard.
555            admin.createTable(htd);
556          }
557        }
558      } catch (MasterNotRunningException e) {
559        LOG.error("Master not running", e);
560        throw new IOException(e);
561      }
562    }

755    static SortedSet<byte []> readKeysToSearch(final Configuration conf)
756    throws IOException, InterruptedException {
757      Path keysInputDir = new Path(conf.get(SEARCHER_INPUTDIR_KEY));
758      FileSystem fs = FileSystem.get(conf);
759      SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
760      if (!fs.exists(keysInputDir)) {
761        throw new FileNotFoundException(keysInputDir.toString());
762      }
763      if (!fs.isDirectory(keysInputDir)) {
764        throw new UnsupportedOperationException("TODO");
765      } else {
766        RemoteIterator<LocatedFileStatus> iterator = fs.listFiles(keysInputDir, false);
767        while(iterator.hasNext()) {
768          LocatedFileStatus keyFileStatus = iterator.next();
769          // Skip "_SUCCESS" file.
770          if (keyFileStatus.getPath().getName().startsWith("_")) continue;
771          result.addAll(readFileToSearch(conf, fs, keyFileStatus));
772        }
773      }
774      return result;

777    private static SortedSet<byte[]> readFileToSearch(final Configuration conf,
778        final FileSystem fs, final LocatedFileStatus keyFileStatus) throws IOException,
779        InterruptedException {
780      SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
781      // Return entries that are flagged Counts.UNDEFINED in the value. Return the row. This is
782      // what is missing.
783      TaskAttemptContext context = new TaskAttemptContextImpl(conf, new TaskAttemptID());
784      try (SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader rr =
785          new SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader()) {
786        InputSplit is =
787          new FileSplit(keyFileStatus.getPath(), 0, keyFileStatus.getLen(), new String [] {});
788        rr.initialize(is, context);
789        while (rr.nextKeyValue()) {
790          rr.getCurrentKey();
791          BytesWritable bw = rr.getCurrentValue();
792          if (Verify.VerifyReducer.whichType(bw.getBytes()) == Verify.Counts.UNDEFINED) {
793            byte[] key = new byte[rr.getCurrentKey().getLength()];
794            System.arraycopy(rr.getCurrentKey().getBytes(), 0, key, 0, rr.getCurrentKey()
795                .getLength());
796            result.add(key);
797          }
798        }
799      }
800      return result;
801    }

862     * subsequent investigative mapreduce jobs. Each emitted value is prefaced by a one byte flag
863     * saying what sort of emission it is. Flag is the Count enum ordinal as a short.

866        Reducer<BytesWritable, BytesWritable, BytesWritable, BytesWritable> {
867      private ArrayList<byte[]> refs = new ArrayList<byte[]>();
868      private final BytesWritable UNREF = new BytesWritable(addPrefixFlag(
869        Counts.UNREFERENCED.ordinal(), new byte[] {}));

871        Counts.LOST_FAMILIES.ordinal(), new byte[] {}));

878      throws IOException, InterruptedException {

893      /**
894       * @param ordinal
895       * @param r
896       * @return Return new byte array that has <code>ordinal</code> as prefix on front taking up
897       * Bytes.SIZEOF_SHORT bytes followed by <code>r</code>
898       */
899      public static byte[] addPrefixFlag(final int ordinal, final byte [] r) {
900        byte[] prefix = Bytes.toBytes((short)ordinal);
901        if (prefix.length != Bytes.SIZEOF_SHORT) {
902          throw new RuntimeException("Unexpected size: " + prefix.length);
903        }
904        byte[] result = new byte[prefix.length + r.length];
905        System.arraycopy(prefix, 0, result, 0, prefix.length);
906        System.arraycopy(r, 0, result, prefix.length, r.length);
907        return result;
908      }

913       * {@link #addPrefixFlag(int, byte[])}
914       */
915      public static Counts whichType(final byte [] bs) {
916        int ordinal = Bytes.toShort(bs, 0, Bytes.SIZEOF_SHORT);
917        return Counts.values()[ordinal];

930      @Override
931      public void reduce(BytesWritable key, Iterable<BytesWritable> values, Context context)
932          throws IOException, InterruptedException {
933        int defCount = 0;
934        boolean lostFamilies = false;
935        refs.clear();
936        for (BytesWritable type : values) {
937          if (type.getLength() == DEF.getLength()) {
938            defCount++;
939            if (type.getBytes()[0] == 1) {
940              lostFamilies = true;
941            }
942          } else {
943            byte[] bytes = new byte[type.getLength()];
944            System.arraycopy(type.getBytes(), 0, bytes, 0, type.getLength());
945            refs.add(bytes);
946          }
947        }
948
949        // TODO check for more than one def, should not happen
950        StringBuilder refsSb = null;
951        String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
952        if (defCount == 0 || refs.size() != 1) {
953          refsSb = dumpExtraInfoOnRefs(key, context, refs);
954          LOG.error("LinkedListError: key=" + keyString + ", reference(s)=" +
955            (refsSb != null? refsSb.toString(): ""));
956        }
957        if (lostFamilies) {
958          LOG.error("LinkedListError: key=" + keyString + ", lost big or tiny families");
959          context.getCounter(Counts.LOST_FAMILIES).increment(1);
960          context.write(key, LOSTFAM);
961        }
962
963        if (defCount == 0 && refs.size() > 0) {
964          // This is bad, found a node that is referenced but not defined. It must have been
965          // lost, emit some info about this node for debugging purposes.
966          // Write out a line per reference. If more than one, flag it.;
967          for (int i = 0; i < refs.size(); i++) {
968            byte[] bs = refs.get(i);
969            int ordinal;
970            if (i <= 0) {
971              ordinal = Counts.UNDEFINED.ordinal();
972              context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
973              context.getCounter(Counts.UNDEFINED).increment(1);
974            } else {
975              ordinal = Counts.EXTRA_UNDEF_REFERENCES.ordinal();
976              context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
977            }
978          }
979          if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
980            // Print out missing row; doing get on reference gives info on when the referencer
981            // was added which can help a little debugging. This info is only available in mapper
982            // output -- the 'Linked List error Key...' log message above. What we emit here is
983            // useless for debugging.
984            context.getCounter("undef", keyString).increment(1);
985          }
986        } else if (defCount > 0 && refs.size() == 0) {
987          // node is defined but not referenced
988          context.write(key, UNREF);
989          context.getCounter(Counts.UNREFERENCED).increment(1);
990          if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
991            context.getCounter("unref", keyString).increment(1);
992          }
993        } else {
994          if (refs.size() > 1) {
995            // Skip first reference.
996            for (int i = 1; i < refs.size(); i++) {
997              context.write(key,
998                new BytesWritable(addPrefixFlag(Counts.EXTRAREFERENCES.ordinal(), refs.get(i))));
999            }
1000            context.getCounter(Counts.EXTRAREFERENCES).increment(refs.size() - 1);
1001          }
1002          // node is defined and referenced
1003          context.getCounter(Counts.REFERENCED).increment(1);
1004        }
1005      }

1007      /**
1008       * Dump out extra info around references if there are any. Helps debugging.
1009       * @return StringBuilder filled with references if any.
1010       * @throws IOException
1011       */
1012      private StringBuilder dumpExtraInfoOnRefs(final BytesWritable key, final Context context,
1013          final List<byte []> refs)
1014      throws IOException {
1015        StringBuilder refsSb = null;
1016        if (refs.isEmpty()) return refsSb;
1017        refsSb = new StringBuilder();
1018        String comma = "";
1019        // If a row is a reference but has no define, print the content of the row that has
1020        // this row as a 'prev'; it will help debug.  The missing row was written just before
1021        // the row we are dumping out here.
1022        TableName tn = getTableName(context.getConfiguration());
1023        try (Table t = this.connection.getTable(tn)) {
1024          for (byte [] ref : refs) {
1025            Result r = t.get(new Get(ref));
1026            List<Cell> cells = r.listCells();
1027            String ts = (cells != null && !cells.isEmpty())?
1028                new java.util.Date(cells.get(0).getTimestamp()).toString(): "";
1029            byte [] b = r.getValue(FAMILY_NAME, COLUMN_CLIENT);
1030            String jobStr = (b != null && b.length > 0)? Bytes.toString(b): "";
1031            b = r.getValue(FAMILY_NAME, COLUMN_COUNT);
1032            long count = (b != null && b.length > 0)? Bytes.toLong(b): -1;
1033            b = r.getValue(FAMILY_NAME, COLUMN_PREV);
1034            String refRegionLocation = "";
1035            String keyRegionLocation = "";
1036            if (b != null && b.length > 0) {
1037              try (RegionLocator rl = this.connection.getRegionLocator(tn)) {
1038                HRegionLocation hrl = rl.getRegionLocation(b);
1039                if (hrl != null) refRegionLocation = hrl.toString();

1041                hrl = rl.getRegionLocation(key.getBytes());
1042                if (hrl != null) keyRegionLocation = hrl.toString();
1043              }
1044            }
1045            LOG.error("Extras on ref without a def, ref=" + Bytes.toStringBinary(ref) +
1046              ", refPrevEqualsKey=" +

1048                ", key=" + Bytes.toStringBinary(key.getBytes(), 0, key.getLength()) +
1049                ", ref row date=" + ts + ", jobStr=" + jobStr +
1050                ", ref row count=" + count +
1051                ", ref row regionLocation=" + refRegionLocation +
1052                ", key row regionLocation=" + keyRegionLocation);
1053            refsSb.append(comma);
1054            comma = ",";
1055            refsSb.append(Bytes.toStringBinary(ref));
1056          }
1057        }
1058        return refsSb;
1059      }

1163    protected boolean verifyExpectedValues(long expectedReferenced, Counters counters) {
1164      final Counter referenced = counters.findCounter(Counts.REFERENCED);
1165      final Counter unreferenced = counters.findCounter(Counts.UNREFERENCED);
1166      boolean success = true;

1172      }

1175        final Counter multiref = counters.findCounter(Counts.EXTRAREFERENCES);
1176        boolean couldBeMultiRef = (multiref.getValue() == unreferenced.getValue());

1178            + (couldBeMultiRef ? "; could be due to duplicate random numbers" : ""));

1190     * @return True if the "bad" counter objects are 0, false otherwise

1192    protected boolean verifyUnexpectedValues(Counters counters) {
1193      final Counter undefined = counters.findCounter(Counts.UNDEFINED);
1194      final Counter lostfamilies = counters.findCounter(Counts.LOST_FAMILIES);
1195      boolean success = true;

1403   * A stand alone program that follows a linked list created by {@link Generator} and prints
1404   * timing info.
1405   */
1406  private static class Walker extends Configured implements Tool {
1407    @Override
1408    public int run(String[] args) throws IOException {
1409      Options options = new Options();
1410      options.addOption("n", "num", true, "number of queries");
1411      options.addOption("s", "start", true, "key to start at, binary string");
1412      options.addOption("l", "logevery", true, "log every N queries");
1413
1414      GnuParser parser = new GnuParser();
1415      CommandLine cmd = null;
1416      try {
1417        cmd = parser.parse(options, args);
1418        if (cmd.getArgs().length != 0) {
1419          throw new ParseException("Command takes no arguments");
1420        }
1421      } catch (ParseException e) {
1422        System.err.println("Failed to parse command line " + e.getMessage());
1423        System.err.println();
1424        HelpFormatter formatter = new HelpFormatter();
1425        formatter.printHelp(getClass().getSimpleName(), options);
1426        System.exit(-1);
1427      }
1428
1429      long maxQueries = Long.MAX_VALUE;
1430      if (cmd.hasOption('n')) {
1431        maxQueries = Long.parseLong(cmd.getOptionValue("n"));
1432      }
1433      Random rand = new SecureRandom();
1434      boolean isSpecificStart = cmd.hasOption('s');
1435      byte[] startKey = isSpecificStart ? Bytes.toBytesBinary(cmd.getOptionValue('s')) : null;
1436      int logEvery = cmd.hasOption('l') ? Integer.parseInt(cmd.getOptionValue('l')) : 1;
1437
1438      Connection connection = ConnectionFactory.createConnection(getConf());
1439      Table table = connection.getTable(getTableName(getConf()));
1440      long numQueries = 0;
1441      // If isSpecificStart is set, only walk one list from that particular node.
1442      // Note that in case of circular (or P-shaped) list it will walk forever, as is
1443      // the case in normal run without startKey.
1444      while (numQueries < maxQueries && (numQueries == 0 || !isSpecificStart)) {
1445        if (!isSpecificStart) {
1446          startKey = new byte[ROWKEY_LENGTH];
1447          rand.nextBytes(startKey);
1448        }
1449        CINode node = findStartNode(table, startKey);
1450        if (node == null && isSpecificStart) {
1451          System.err.printf("Start node not found: %s \n", Bytes.toStringBinary(startKey));
1452        }
1453        numQueries++;
1454        while (node != null && node.prev.length != NO_KEY.length && numQueries < maxQueries) {
1455          byte[] prev = node.prev;
1456          long t1 = System.currentTimeMillis();
1457          node = getNode(prev, table, node);
1458          long t2 = System.currentTimeMillis();
1459          if (numQueries % logEvery == 0) {
1460            System.out.printf("CQ %d: %d %s \n", numQueries, t2 - t1, Bytes.toStringBinary(prev));
1461          }
1462          numQueries++;
1463          if (node == null) {
1464            System.err.printf("UNDEFINED NODE %s \n", Bytes.toStringBinary(prev));
1465          } else if (node.prev.length == NO_KEY.length) {
1466            System.err.printf("TERMINATING NODE %s \n", Bytes.toStringBinary(node.key));
1467          }
1468        }
1469      }
1470
1471      table.close();
1472      connection.close();
1473      return 0;
1474    }

1476    private static CINode findStartNode(Table table, byte[] startKey) throws IOException {
1477      Scan scan = new Scan();
1478      scan.setStartRow(startKey);
1479      scan.setBatch(1);
1480      scan.addColumn(FAMILY_NAME, COLUMN_PREV);
1481
1482      long t1 = System.currentTimeMillis();
1483      ResultScanner scanner = table.getScanner(scan);
1484      Result result = scanner.next();
1485      long t2 = System.currentTimeMillis();
1486      scanner.close();
1487
1488      if ( result != null) {
1489        CINode node = getCINode(result, new CINode());
1490        System.out.printf("FSR %d %s\n", t2 - t1, Bytes.toStringBinary(node.key));
1491        return node;
1492      }
1493
1494      System.out.println("FSR " + (t2 - t1));
1495
1496      return null;
1497    }
