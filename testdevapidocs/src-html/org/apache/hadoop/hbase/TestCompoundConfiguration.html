034@Category({MiscTests.class, SmallTests.class})
035public class TestCompoundConfiguration extends TestCase {
036  private Configuration baseConf;
037  private int baseConfSize;
038
039  @Override
040  protected void setUp() throws Exception {
041    baseConf = new Configuration();
042    baseConf.set("A", "1");
043    baseConf.setInt("B", 2);
044    baseConf.set("C", "3");
045    baseConfSize = baseConf.size();
046  }
047
048  @Test
049  public void testBasicFunctionality() throws ClassNotFoundException {
050    CompoundConfiguration compoundConf = new CompoundConfiguration()
051        .add(baseConf); 
052    assertEquals("1", compoundConf.get("A"));
053    assertEquals(2, compoundConf.getInt("B", 0));
054    assertEquals(3, compoundConf.getInt("C", 0));
055    assertEquals(0, compoundConf.getInt("D", 0));
056
057    assertEquals(CompoundConfiguration.class, compoundConf
058        .getClassByName(CompoundConfiguration.class.getName()));
059    try {
060      compoundConf.getClassByName("bad_class_name");
061      fail("Trying to load bad_class_name should throw an exception");
062    } catch (ClassNotFoundException e) {
063      // win!
064    }
065  }
066
067  @Test
068  public void testPut() {
069    CompoundConfiguration compoundConf = new CompoundConfiguration()
070      .add(baseConf);
071    assertEquals("1", compoundConf.get("A"));
072    assertEquals(2, compoundConf.getInt("B", 0));
073    assertEquals(3, compoundConf.getInt("C", 0));
074    assertEquals(0, compoundConf.getInt("D", 0));
075
076    compoundConf.set("A", "1337");
077    compoundConf.set("string", "stringvalue");
078    assertEquals(1337, compoundConf.getInt("A", 0));
079    assertEquals("stringvalue", compoundConf.get("string"));
080
081    // we didn't modify the base conf
082    assertEquals("1", baseConf.get("A"));
083    assertNull(baseConf.get("string"));
084
085    // adding to the base shows up in the compound
086    baseConf.set("setInParent", "fromParent");
087    assertEquals("fromParent", compoundConf.get("setInParent"));
088  }

090  @Test
091  public void testWithConfig() {
092    Configuration conf = new Configuration();
093    conf.set("B", "2b");
094    conf.set("C", "33");
095    conf.set("D", "4");
096
097    CompoundConfiguration compoundConf = new CompoundConfiguration()
098        .add(baseConf)
099        .add(conf);
100    assertEquals("1", compoundConf.get("A"));
101    assertEquals("2b", compoundConf.get("B"));
102    assertEquals(33, compoundConf.getInt("C", 0));
103    assertEquals("4", compoundConf.get("D"));
104    assertEquals(4, compoundConf.getInt("D", 0));
105    assertNull(compoundConf.get("E"));
106    assertEquals(6, compoundConf.getInt("F", 6));
107    
108    int cnt = 0;
109    for (Map.Entry<String,String> entry : compoundConf) {
110      cnt++;
111      if (entry.getKey().equals("B")) assertEquals("2b", entry.getValue());
112      else if (entry.getKey().equals("G")) assertEquals(null, entry.getValue());
113    }
114    // verify that entries from ImmutableConfigMap's are merged in the iterator's view
115    assertEquals(baseConfSize + 1, cnt);
116  }

118  private Bytes strToIb(String s) {
119    return new Bytes(Bytes.toBytes(s));
120  }
121
122  @Test
123  public void testWithIbwMap() {
124    Map<Bytes, Bytes> map = new HashMap<>();
125    map.put(strToIb("B"), strToIb("2b"));
126    map.put(strToIb("C"), strToIb("33"));
127    map.put(strToIb("D"), strToIb("4"));
128    // unlike config, note that IBW Maps can accept null values
129    map.put(strToIb("G"), null);
130
131    CompoundConfiguration compoundConf = new CompoundConfiguration()
132      .add(baseConf)
133      .addBytesMap(map);
134    assertEquals("1", compoundConf.get("A"));
135    assertEquals("2b", compoundConf.get("B"));
136    assertEquals(33, compoundConf.getInt("C", 0));
137    assertEquals("4", compoundConf.get("D"));
138    assertEquals(4, compoundConf.getInt("D", 0));
139    assertNull(compoundConf.get("E"));
140    assertEquals(6, compoundConf.getInt("F", 6));
141    assertNull(compoundConf.get("G"));
142    
143    int cnt = 0;
144    for (Map.Entry<String,String> entry : compoundConf) {
145      cnt++;
146      if (entry.getKey().equals("B")) assertEquals("2b", entry.getValue());
147      else if (entry.getKey().equals("G")) assertEquals(null, entry.getValue());
148    }
149    // verify that entries from ImmutableConfigMap's are merged in the iterator's view
150    assertEquals(baseConfSize + 2, cnt);
151
152    // Verify that adding map after compound configuration is modified overrides properly
153    CompoundConfiguration conf2 = new CompoundConfiguration();
154    conf2.set("X", "modification");
155    conf2.set("D", "not4");
156    assertEquals("modification", conf2.get("X"));
157    assertEquals("not4", conf2.get("D"));
158    conf2.addBytesMap(map);
159    assertEquals("4", conf2.get("D")); // map overrides
160  }

199  @Test
200  public void testLaterConfigsOverrideEarlier() {
201    Map<String, String> map1 = new HashMap<>();
202    map1.put("A", "2");
203    map1.put("D", "5");
204    Map<String, String> map2 = new HashMap<>();
205    String newValueForA = "3", newValueForB = "4";
206    map2.put("A", newValueForA);
207    map2.put("B", newValueForB);
208
209    CompoundConfiguration compoundConf = new CompoundConfiguration()
210      .addStringMap(map1).add(baseConf);
211    assertEquals("1", compoundConf.get("A"));
212    assertEquals("5", compoundConf.get("D"));
213    compoundConf.addStringMap(map2);
214    assertEquals(newValueForA, compoundConf.get("A"));
215    assertEquals(newValueForB, compoundConf.get("B"));
216    assertEquals("5", compoundConf.get("D"));
217
218    int cnt = 0;
219    for (Map.Entry<String,String> entry : compoundConf) {
220      cnt++;
221      if (entry.getKey().equals("A")) assertEquals(newValueForA, entry.getValue());
222      else if (entry.getKey().equals("B")) assertEquals(newValueForB, entry.getValue());
223    }
224    // verify that entries from ImmutableConfigMap's are merged in the iterator's view
225    assertEquals(baseConfSize + 1, cnt);
226  }
