041/**
042 * Test for{@link MultiHFileOutputFormat}. Sets up and runs a mapreduce job that output directories and
043 * writes hfiles.
044 */
045@Category(MediumTests.class)
046public class TestMultiHFileOutputFormat {
047    private static final Log LOG = LogFactory.getLog(TestMultiHFileOutputFormat.class);
048
049    private HBaseTestingUtility util = new HBaseTestingUtility();
050
051    private static int ROWSPERSPLIT = 10;
052
053    private static final int KEYLEN_DEFAULT = 10;
054    private static final String KEYLEN_CONF = "randomkv.key.length";
055
056    private static final int VALLEN_DEFAULT = 10;
057    private static final String VALLEN_CONF = "randomkv.val.length";
058
059    private static final byte[][] TABLES =
060        { Bytes.add(Bytes.toBytes(PerformanceEvaluation.TABLE_NAME), Bytes.toBytes("-1")),
061            Bytes.add(Bytes.toBytes(PerformanceEvaluation.TABLE_NAME), Bytes.toBytes("-2")) };
062
063    private static final byte[][] FAMILIES =
064        { Bytes.add(PerformanceEvaluation.FAMILY_NAME, Bytes.toBytes("-A")),
065            Bytes.add(PerformanceEvaluation.FAMILY_NAME, Bytes.toBytes("-B")) };
066
067    private static final byte[] QUALIFIER = Bytes.toBytes("data");
068
069    public static void main(String[] args) throws Exception {
070        new TestMultiHFileOutputFormat().testWritingDataIntoHFiles();
071    }
072
073    /**
074     * Run small MR job. this MR job will write HFile into
075     * testWritingDataIntoHFiles/tableNames/columFamilies/
076     */
077    @Test
078    public void testWritingDataIntoHFiles() throws Exception {
079        Configuration conf = util.getConfiguration();
080        util.startMiniCluster();
081        Path testDir = util.getDataTestDirOnTestFS("testWritingDataIntoHFiles");
082        FileSystem fs = testDir.getFileSystem(conf);
083        LOG.info("testWritingDataIntoHFiles dir writing to dir: " + testDir);
084
085        // Set down this value or we OOME in eclipse.
086        conf.setInt("mapreduce.task.io.sort.mb", 20);
087        // Write a few files by setting max file size.
088        conf.setLong(HConstants.HREGION_MAX_FILESIZE, 64 * 1024);
089
090        try {
091            Job job = Job.getInstance(conf, "testWritingDataIntoHFiles");
092
093            FileOutputFormat.setOutputPath(job, testDir);
094
095            job.setInputFormatClass(NMapInputFormat.class);
096            job.setMapperClass(Random_TableKV_GeneratingMapper.class);
097            job.setMapOutputKeyClass(ImmutableBytesWritable.class);
098            job.setMapOutputValueClass(KeyValue.class);
099            job.setReducerClass(Table_KeyValueSortReducer.class);
100            job.setOutputFormatClass(MultiHFileOutputFormat.class);
101            job.getConfiguration().setStrings("io.serializations", conf.get("io.serializations"),
102                MutationSerialization.class.getName(), ResultSerialization.class.getName(),
103                KeyValueSerialization.class.getName());
104
105            TableMapReduceUtil.addDependencyJars(job);
106            TableMapReduceUtil.initCredentials(job);
107            LOG.info("\nStarting test testWritingDataIntoHFiles\n");
108            assertTrue(job.waitForCompletion(true));
109            LOG.info("\nWaiting on checking MapReduce output\n");
110            assertTrue(checkMROutput(fs, testDir, 0));
111        } finally {
112            testDir.getFileSystem(conf).delete(testDir, true);
113            util.shutdownMiniCluster();
114        }
115    }
116
117    /**
118     * MR will output a 3 level directory, tableName->ColumnFamilyName->HFile this method to check the
119     * created directory is correct or not A recursion method, the testDir had better be small size
120     */
121    private boolean checkMROutput(FileSystem fs, Path testDir, int level)
122        throws FileNotFoundException, IOException {
123        if (level >= 3) {
124            return HFile.isHFileFormat(fs, testDir);
125        }
126        FileStatus[] fStats = fs.listStatus(testDir);
127        if (fStats == null || fStats.length <= 0) {
128            LOG.info("Created directory format is not correct");
129            return false;
130        }
131
132        for (FileStatus stats : fStats) {
133            // skip the _SUCCESS file created by MapReduce
134            if (level == 0 && stats.getPath().getName().endsWith(FileOutputCommitter.SUCCEEDED_FILE_NAME))
135                continue;
136            if (level < 2 && !stats.isDirectory()) {
137                LOG.info("Created directory format is not correct");
138                return false;
139            }
140            boolean flag = checkMROutput(fs, stats.getPath(), level + 1);
141            if (flag == false) return false;
142        }
143        return true;
144    }

146    /**
147     * Simple mapper that makes <TableName, KeyValue> output. With no input data
148     */
149    static class Random_TableKV_GeneratingMapper
150        extends Mapper<NullWritable, NullWritable, ImmutableBytesWritable, Cell> {
151
152        private int keyLength;
153        private int valLength;
154
155        @Override
156        protected void setup(Context context) throws IOException, InterruptedException {
157            super.setup(context);
158
159            Configuration conf = context.getConfiguration();
160            keyLength = conf.getInt(KEYLEN_CONF, KEYLEN_DEFAULT);
161            valLength = conf.getInt(VALLEN_CONF, VALLEN_DEFAULT);
162        }
163
164        @Override
165        protected void map(NullWritable n1, NullWritable n2,
166            Mapper<NullWritable, NullWritable, ImmutableBytesWritable, Cell>.Context context)
167            throws java.io.IOException, InterruptedException {
168
169            byte keyBytes[] = new byte[keyLength];
170            byte valBytes[] = new byte[valLength];
171
172            ArrayList<ImmutableBytesWritable> tables = new ArrayList<ImmutableBytesWritable>();
173            for (int i = 0; i < TABLES.length; i++) {
174                tables.add(new ImmutableBytesWritable(TABLES[i]));
175            }
176
177            int taskId = context.getTaskAttemptID().getTaskID().getId();
178            assert taskId < Byte.MAX_VALUE : "Unit tests dont support > 127 tasks!";
179            Random random = new Random();
180
181            for (int i = 0; i < ROWSPERSPLIT; i++) {
182                random.nextBytes(keyBytes);
183                // Ensure that unique tasks generate unique keys
184                keyBytes[keyLength - 1] = (byte) (taskId & 0xFF);
185                random.nextBytes(valBytes);
186
187                for (ImmutableBytesWritable table : tables) {
188                    for (byte[] family : FAMILIES) {
189                        Cell kv = new KeyValue(keyBytes, family, QUALIFIER, valBytes);
190                        context.write(table, kv);
191                    }
192                }
193            }
194        }
195    }

197    /**
198     * Simple Reducer that have input <TableName, KeyValue>, with KeyValues have no order. and output
199     * <TableName, KeyValue>, with KeyValues are ordered
200     */
201
202    static class Table_KeyValueSortReducer
203        extends Reducer<ImmutableBytesWritable, KeyValue, ImmutableBytesWritable, KeyValue> {
204        protected void reduce(ImmutableBytesWritable table, java.lang.Iterable<KeyValue> kvs,
205            org.apache.hadoop.mapreduce.Reducer<ImmutableBytesWritable, KeyValue, ImmutableBytesWritable, KeyValue>.Context context)
206            throws java.io.IOException, InterruptedException {
207            TreeSet<KeyValue> map = new TreeSet<KeyValue>(KeyValue.COMPARATOR);
208            for (KeyValue kv : kvs) {
209                try {
210                    map.add(kv.clone());
211                } catch (CloneNotSupportedException e) {
212                    throw new java.io.IOException(e);
213                }
214            }
215            context.setStatus("Read " + map.getClass());
216            int index = 0;
217            for (KeyValue kv : map) {
218                context.write(table, kv);
219                if (++index % 100 == 0) context.setStatus("Wrote " + index);
220            }
221        }
222    }
