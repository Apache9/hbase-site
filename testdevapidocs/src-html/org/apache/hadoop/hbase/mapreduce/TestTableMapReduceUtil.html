032/**
033 * Test different variants of initTableMapperJob method

035@Category({MapReduceTests.class, SmallTests.class})
036public class TestTableMapReduceUtil {
037
038  /*
039   * initTableSnapshotMapperJob is tested in {@link TestTableSnapshotInputFormat} because
040   * the method depends on an online cluster.
041   */
042
043  @Test
044  public void testInitTableMapperJob1() throws Exception {
045    Configuration configuration = new Configuration();
046    Job job = new Job(configuration, "tableName");
047    // test 
048    TableMapReduceUtil.initTableMapperJob("Table", new Scan(), Import.Importer.class, Text.class,
049        Text.class, job, false, WALInputFormat.class);
050    assertEquals(WALInputFormat.class, job.getInputFormatClass());
051    assertEquals(Import.Importer.class, job.getMapperClass());
052    assertEquals(LongWritable.class, job.getOutputKeyClass());
053    assertEquals(Text.class, job.getOutputValueClass());
054    assertNull(job.getCombinerClass());
055    assertEquals("Table", job.getConfiguration().get(TableInputFormat.INPUT_TABLE));
056  }
057
