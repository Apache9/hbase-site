091  @Test
092  public void testGetBestLocations() throws IOException {
093    TableSnapshotInputFormatImpl tsif = new TableSnapshotInputFormatImpl();
094    Configuration conf = UTIL.getConfiguration();
095
096    HDFSBlocksDistribution blockDistribution = new HDFSBlocksDistribution();
097    Assert.assertEquals(Lists.newArrayList(),
098      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
099
100    blockDistribution.addHostsAndBlockWeight(new String[] {"h1"}, 1);
101    Assert.assertEquals(Lists.newArrayList("h1"),
102      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
103
104    blockDistribution.addHostsAndBlockWeight(new String[] {"h1"}, 1);
105    Assert.assertEquals(Lists.newArrayList("h1"),
106      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
107
108    blockDistribution.addHostsAndBlockWeight(new String[] {"h2"}, 1);
109    Assert.assertEquals(Lists.newArrayList("h1"),
110      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
111
112    blockDistribution = new HDFSBlocksDistribution();
113    blockDistribution.addHostsAndBlockWeight(new String[] {"h1"}, 10);
114    blockDistribution.addHostsAndBlockWeight(new String[] {"h2"}, 7);
115    blockDistribution.addHostsAndBlockWeight(new String[] {"h3"}, 5);
116    blockDistribution.addHostsAndBlockWeight(new String[] {"h4"}, 1);
117    Assert.assertEquals(Lists.newArrayList("h1"),
118      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
119
120    blockDistribution.addHostsAndBlockWeight(new String[] {"h2"}, 2);
121    Assert.assertEquals(Lists.newArrayList("h1", "h2"),
122      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
123
124    blockDistribution.addHostsAndBlockWeight(new String[] {"h2"}, 3);
125    Assert.assertEquals(Lists.newArrayList("h2", "h1"),
126      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
127
128    blockDistribution.addHostsAndBlockWeight(new String[] {"h3"}, 6);
129    blockDistribution.addHostsAndBlockWeight(new String[] {"h4"}, 9);
130
131    Assert.assertEquals(Lists.newArrayList("h2", "h3", "h4", "h1"),
132      TableSnapshotInputFormatImpl.getBestLocations(conf, blockDistribution));
133  }

244  @Test
245  public void testNoDuplicateResultsWhenSplitting() throws Exception {
246    setupCluster();
247    TableName tableName = TableName.valueOf("testNoDuplicateResultsWhenSplitting");
248    String snapshotName = "testSnapshotBug";
249    try {
250      if (UTIL.getAdmin().tableExists(tableName)) {
251        UTIL.deleteTable(tableName);
252      }
253
254      UTIL.createTable(tableName, FAMILIES);
255      Admin admin = UTIL.getAdmin();
256
257      // put some stuff in the table
258      Table table = UTIL.getConnection().getTable(tableName);
259      UTIL.loadTable(table, FAMILIES);
260
261      // split to 2 regions
262      admin.split(tableName, Bytes.toBytes("eee"));
263      blockUntilSplitFinished(UTIL, tableName, 2);
264
265      Path rootDir = FSUtils.getRootDir(UTIL.getConfiguration());
266      FileSystem fs = rootDir.getFileSystem(UTIL.getConfiguration());
267
268      SnapshotTestingUtils.createSnapshotAndValidate(admin, tableName, Arrays.asList(FAMILIES),
269        null, snapshotName, rootDir, fs, true);
270
271      // load different values
272      byte[] value = Bytes.toBytes("after_snapshot_value");
273      UTIL.loadTable(table, FAMILIES, value);
274
275      // cause flush to create new files in the region
276      admin.flush(tableName);
277      table.close();
278
279      Job job = new Job(UTIL.getConfiguration());
280      Path tmpTableDir = UTIL.getDataTestDirOnTestFS(snapshotName);
281      // limit the scan
282      Scan scan = new Scan().withStartRow(getStartRow()).withStopRow(getEndRow());
283
284      TableMapReduceUtil.initTableSnapshotMapperJob(snapshotName, scan,
285        TestTableSnapshotMapper.class, ImmutableBytesWritable.class, NullWritable.class, job, false,
286        tmpTableDir);
287
288      verifyWithMockedMapReduce(job, 2, 2, getStartRow(), getEndRow());
289    } finally {
290      UTIL.getAdmin().deleteSnapshot(snapshotName);
291      UTIL.deleteTable(tableName);
292      tearDownCluster();
293    }
294  }

296  private void verifyWithMockedMapReduce(Job job, int numRegions, int expectedNumSplits,
297      byte[] startRow, byte[] stopRow)
298      throws IOException, InterruptedException {
299    TableSnapshotInputFormat tsif = new TableSnapshotInputFormat();
300    List<InputSplit> splits = tsif.getSplits(job);
301
302    Assert.assertEquals(expectedNumSplits, splits.size());
303
304    HBaseTestingUtility.SeenRowTracker rowTracker =
305        new HBaseTestingUtility.SeenRowTracker(startRow, stopRow);
306
307    for (int i = 0; i < splits.size(); i++) {
308      // validate input split
309      InputSplit split = splits.get(i);
310      Assert.assertTrue(split instanceof TableSnapshotRegionSplit);
311
312      // validate record reader
313      TaskAttemptContext taskAttemptContext = mock(TaskAttemptContext.class);
314      when(taskAttemptContext.getConfiguration()).thenReturn(job.getConfiguration());
315      RecordReader<ImmutableBytesWritable, Result> rr =
316          tsif.createRecordReader(split, taskAttemptContext);
317      rr.initialize(split, taskAttemptContext);
318
319      // validate we can read all the data back
320      while (rr.nextKeyValue()) {
321        byte[] row = rr.getCurrentKey().get();
322        verifyRowFromMap(rr.getCurrentKey(), rr.getCurrentValue());
323        rowTracker.addRow(row);
324      }
325
326      rr.close();
327    }
328
329    // validate all rows are seen
330    rowTracker.validate();
331  }

341  // this is also called by the IntegrationTestTableSnapshotInputFormat
342  public static void doTestWithMapReduce(HBaseTestingUtility util, TableName tableName,
343      String snapshotName, byte[] startRow, byte[] endRow, Path tableDir, int numRegions,
344      int expectedNumSplits, boolean shutdownCluster) throws Exception {
345
346    //create the table and snapshot
347    createTableAndSnapshot(util, tableName, snapshotName, startRow, endRow, numRegions);
348
349    if (shutdownCluster) {
350      util.shutdownMiniHBaseCluster();
351    }
352
353    try {
354      // create the job
355      Job job = new Job(util.getConfiguration());
356      Scan scan = new Scan(startRow, endRow); // limit the scan
357
358      job.setJarByClass(util.getClass());
359      TableMapReduceUtil.addDependencyJarsForClasses(job.getConfiguration(),
360        TestTableSnapshotInputFormat.class);
361
362      TableMapReduceUtil.initTableSnapshotMapperJob(snapshotName,
363        scan, TestTableSnapshotMapper.class, ImmutableBytesWritable.class,
364        NullWritable.class, job, true, tableDir);
365
366      job.setReducerClass(TestTableSnapshotInputFormat.TestTableSnapshotReducer.class);
367      job.setNumReduceTasks(1);
368      job.setOutputFormatClass(NullOutputFormat.class);
369
370      Assert.assertTrue(job.waitForCompletion(true));
371    } finally {
372      if (!shutdownCluster) {
373        util.getAdmin().deleteSnapshot(snapshotName);
374        util.deleteTable(tableName);
375      }
376    }
377  }
