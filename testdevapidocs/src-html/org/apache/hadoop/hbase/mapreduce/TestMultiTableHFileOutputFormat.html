089  /**
090   * Run small MR job. this MR job will write HFile into
091   * testWritingDataIntoHFiles/tableNames/columnFamilies/
092   */
093  @Test
094  public void testWritingDataIntoHFiles() throws Exception {
095    Configuration conf = util.getConfiguration();
096    util.startMiniCluster();
097    Path testDir = util.getDataTestDirOnTestFS("testWritingDataIntoHFiles");
098    FileSystem fs = testDir.getFileSystem(conf);
099    LOG.info("testWritingDataIntoHFiles dir writing to dir: " + testDir);
100
101    // Set down this value or we OOME in eclipse.
102    conf.setInt("mapreduce.task.io.sort.mb", 20);
103    // Write a few files by setting max file size.
104    conf.setLong(HConstants.HREGION_MAX_FILESIZE, 64 * 1024);
105
106    try {
107      Job job = Job.getInstance(conf, "testWritingDataIntoHFiles");
108
109      FileOutputFormat.setOutputPath(job, testDir);
110
111      job.setInputFormatClass(NMapInputFormat.class);
112      job.setMapperClass(Random_TableKV_GeneratingMapper.class);
113      job.setMapOutputKeyClass(ImmutableBytesWritable.class);
114      job.setMapOutputValueClass(KeyValue.class);
115      job.setReducerClass(Table_KeyValueSortReducer.class);
116      job.setOutputFormatClass(MultiTableHFileOutputFormat.class);
117      job.getConfiguration().setStrings("io.serializations", conf.get("io.serializations"),
118          MutationSerialization.class.getName(), ResultSerialization.class.getName(),
119          KeyValueSerialization.class.getName());
120
121      TableMapReduceUtil.addDependencyJars(job);
122      TableMapReduceUtil.initCredentials(job);
123      LOG.info("\nStarting test testWritingDataIntoHFiles\n");
124      assertTrue(job.waitForCompletion(true));
125      LOG.info("\nWaiting on checking MapReduce output\n");
126      assertTrue(checkMROutput(fs, testDir, 0));
127    } finally {
128      testDir.getFileSystem(conf).delete(testDir, true);
129      util.shutdownMiniCluster();
130    }
131  }

133  /**
134   * check whether create directory and hfiles as format designed in MultiHFilePartitioner
135   * and also check whether the output file has same related configuration as created table
136   */
137  @Test
138  public void testMultiHFilePartitioner() throws Exception {
139    Configuration conf = util.getConfiguration();
140    util.startMiniCluster();
141    Path testDir = util.getDataTestDirOnTestFS("testMultiHFilePartitioner");
142    FileSystem fs = testDir.getFileSystem(conf);
143    LOG.info("testMultiHFilePartitioner dir writing to : " + testDir);
144
145    // Set down this value or we OOME in eclipse.
146    conf.setInt("mapreduce.task.io.sort.mb", 20);
147    // Write a few files by setting max file size.
148    conf.setLong(HConstants.HREGION_MAX_FILESIZE, 64 * 1024);
149
150    // Create several tables for testing
151    List<TableName> tables = new ArrayList<TableName>();
152
153    // to store splitKeys for TABLE[0] for testing;
154    byte[][] testKeys = new byte[0][0];
155    for (int i = 0; i < TABLES.length; i++) {
156      TableName tableName = TableName.valueOf(TABLES[i]);
157      byte[][] splitKeys = generateRandomSplitKeys(3);
158      if (i == 0) {
159        testKeys = splitKeys;
160      }
161      HTableDescriptor tableDescriptor = new HTableDescriptor(tableName);
162      for (int j = 0; j < FAMILIES.length; j++) {
163        HColumnDescriptor familyDescriptor = new HColumnDescriptor(FAMILIES[j]);
164        //only set Tables[0] configuration, and specify compression type and DataBlockEncode
165        if (i == 0) {
166          familyDescriptor.setCompressionType(Compression.Algorithm.GZ);
167          familyDescriptor.setDataBlockEncoding(DataBlockEncoding.FAST_DIFF);
168        }
169        tableDescriptor.addFamily(familyDescriptor);
170      }
171      util.createTable(tableDescriptor, splitKeys, conf);
172      tables.add(tableName);
173    }
174    // set up for MapReduce job
175    try {
176      Job job = Job.getInstance(conf, "testMultiHFilePartitioner");
177      FileOutputFormat.setOutputPath(job, testDir);
178
179      job.setInputFormatClass(NMapInputFormat.class);
180      job.setMapperClass(Random_TableKV_GeneratingMapper.class);
181      job.setMapOutputKeyClass(ImmutableBytesWritable.class);
182      job.setMapOutputValueClass(KeyValue.class);
183
184      MultiTableHFileOutputFormat.configureIncrementalLoad(job, tables);
185
186      LOG.info("Starting test testWritingDataIntoHFiles");
187      assertTrue(job.waitForCompletion(true));
188      LOG.info("Waiting on checking MapReduce output");
189      assertTrue(checkMROutput(fs, testDir, 0));
190      assertTrue(checkFileConfAndSplitKeys(conf, fs, testDir, testKeys));
191    } finally {
192      for (int i = 0; i < TABLES.length; i++) {
193        TableName tName = TableName.valueOf(TABLES[i]);
194        util.deleteTable(tName);
195      }
196      fs.delete(testDir, true);
197      fs.close();
198      util.shutdownMiniCluster();
199    }
200  }

202  /**
203   * check the output hfile has same configuration as created test table
204   * and also check whether hfiles get split correctly
205   * only check TABLES[0]
206   */
207  private boolean checkFileConfAndSplitKeys(Configuration conf, FileSystem fs, Path testDir, byte[][] splitKeys) throws IOException {
208    FileStatus[] fStats = fs.listStatus(testDir);
209    for (FileStatus stats : fStats) {
210      if (stats.getPath().getName().equals(new String(TABLES[0]))) {
211        FileStatus[] cfStats = fs.listStatus(stats.getPath());
212        for (FileStatus cfstat : cfStats) {
213          FileStatus[] hfStats = fs.listStatus(cfstat.getPath());
214
215          List<byte[]> firsttKeys = new ArrayList<byte[]>();
216          List<byte[]> lastKeys = new ArrayList<byte[]>();
217          for (FileStatus hfstat : hfStats) {
218            if (HFile.isHFileFormat(fs, hfstat)) {
219              HFile.Reader hfr =
220                  HFile.createReader(fs, hfstat.getPath(), new CacheConfig(conf), true, conf);
221              if (!hfr.getDataBlockEncoding().equals(DataBlockEncoding.FAST_DIFF) || !hfr
222                  .getCompressionAlgorithm().equals(Compression.Algorithm.GZ)) return false;
223              firsttKeys.add(hfr.getFirstRowKey());
224              lastKeys.add(hfr.getLastRowKey());
225            }
226          }
227          if (checkFileSplit(splitKeys, firsttKeys, lastKeys) == false) {
228            return false;
229          }
230        }
231      }
232    }
233    return true;
234  }

236  /**
237   * Check whether the Hfile has been split by region boundaries
238   * @param splitKeys split keys for that table
239   * @param firstKeys first rowKey for hfiles
240   * @param lastKeys last rowKey for hfiles
241   */
242  private boolean checkFileSplit(byte[][] splitKeys, List<byte[]> firstKeys, List<byte[]> lastKeys) {
243    Collections.sort(firstKeys, Bytes.BYTES_RAWCOMPARATOR);
244    Collections.sort(lastKeys, Bytes.BYTES_RAWCOMPARATOR);
245    Arrays.sort(splitKeys, Bytes.BYTES_RAWCOMPARATOR);
246
247    int is = 0, il = 0;
248    for (byte[] key : lastKeys) {
249      while (is < splitKeys.length && Bytes.compareTo(key, splitKeys[is]) >= 0) is++;
250      if (is == splitKeys.length) {
251        break;
252      }
253      if (is > 0) {
254        if (Bytes.compareTo(firstKeys.get(il), splitKeys[is - 1]) < 0) return false;
255      }
256      il++;
257    }
258
259    if (is == splitKeys.length) {
260      return il == lastKeys.size() - 1;
261    }
262    return true;
263  }

266  /**
267   * MR will output a 3 level directory, tableName->ColumnFamilyName->HFile this method to check the
268   * created directory is correct or not A recursion method, the testDir had better be small size
269   */
270  private boolean checkMROutput(FileSystem fs, Path testDir, int level) throws IOException {
271    if (level >= 3) {
272      return HFile.isHFileFormat(fs, testDir);
273    }
274    FileStatus[] fStats = fs.listStatus(testDir);
275    if (fStats == null || fStats.length <= 0) {
276      LOG.info("Created directory format is not correct");
277      return false;
278    }
279
280    for (FileStatus stats : fStats) {
281      // skip the _SUCCESS file created by MapReduce
282      if (level == 0 && stats.getPath().getName().endsWith(FileOutputCommitter.SUCCEEDED_FILE_NAME))
283        continue;
284      if (level < 2 && !stats.isDirectory()) {
285        LOG.info("Created directory format is not correct");
286        return false;
287      }
288      boolean flag = checkMROutput(fs, stats.getPath(), level + 1);
289      if (flag == false) return false;
290    }
291    return true;
292  }

295  private byte[][] generateRandomSplitKeys(int numKeys) {
296    Random random = new Random();
297    byte[][] ret = new byte[numKeys][];
298    for (int i = 0; i < numKeys; i++) {
299      ret[i] = PerformanceEvaluation.generateData(random, KEYLEN_DEFAULT);
300    }
301    return ret;
302  }

305  /**
306   * Simple mapper that makes <TableName, KeyValue> output. With no input data
307   */
308  static class Random_TableKV_GeneratingMapper
309      extends Mapper<NullWritable, NullWritable, ImmutableBytesWritable, Cell> {
310
311    private int keyLength;
312    private int valLength;
313
314    @Override
315    protected void setup(Context context) throws IOException, InterruptedException {
316      super.setup(context);
317
318      Configuration conf = context.getConfiguration();
319      keyLength = conf.getInt(KEYLEN_CONF, KEYLEN_DEFAULT);
320      valLength = conf.getInt(VALLEN_CONF, VALLEN_DEFAULT);
321    }
322
323    @Override
324    protected void map(NullWritable n1, NullWritable n2,
325        Mapper<NullWritable, NullWritable, ImmutableBytesWritable, Cell>.Context context)
326        throws java.io.IOException, InterruptedException {
327
328      byte keyBytes[] = new byte[keyLength];
329      byte valBytes[] = new byte[valLength];
330
331      ArrayList<ImmutableBytesWritable> tables = new ArrayList<>();
332      for (int i = 0; i < TABLES.length; i++) {
333        tables.add(new ImmutableBytesWritable(TABLES[i]));
334      }
335
336      int taskId = context.getTaskAttemptID().getTaskID().getId();
337      assert taskId < Byte.MAX_VALUE : "Unit tests dont support > 127 tasks!";
338      Random random = new Random();
339
340      for (int i = 0; i < ROWSPERSPLIT; i++) {
341        random.nextBytes(keyBytes);
342        // Ensure that unique tasks generate unique keys
343        keyBytes[keyLength - 1] = (byte) (taskId & 0xFF);
344        random.nextBytes(valBytes);
345
346        for (ImmutableBytesWritable table : tables) {
347          for (byte[] family : FAMILIES) {
348            Cell kv = new KeyValue(keyBytes, family, QUALIFIER, valBytes);
349            context.write(table, kv);
350          }
351        }
352      }
353    }
354  }

356  /**
357   * Simple Reducer that have input <TableName, KeyValue>, with KeyValues have no order. and output
358   * <TableName, KeyValue>, with KeyValues are ordered
359   */
360
361  static class Table_KeyValueSortReducer
362      extends Reducer<ImmutableBytesWritable, KeyValue, ImmutableBytesWritable, KeyValue> {
363    protected void reduce(ImmutableBytesWritable table, java.lang.Iterable<KeyValue> kvs,
364        org.apache.hadoop.mapreduce.Reducer<ImmutableBytesWritable, KeyValue, ImmutableBytesWritable, KeyValue>.Context context)
365        throws java.io.IOException, InterruptedException {
366      TreeSet<KeyValue> map = new TreeSet<>(CellComparator.COMPARATOR);
367      for (KeyValue kv : kvs) {
368        try {
369          map.add(kv.clone());
370        } catch (CloneNotSupportedException e) {
371          throw new java.io.IOException(e);
372        }
373      }
374      context.setStatus("Read " + map.getClass());
375      int index = 0;
376      for (KeyValue kv : map) {
377        context.write(table, kv);
378        if (++index % 100 == 0) context.setStatus("Wrote " + index);
379      }
380    }
381  }
