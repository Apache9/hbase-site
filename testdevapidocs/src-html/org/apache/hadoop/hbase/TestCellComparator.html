018package org.apache.hadoop.hbase;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import java.nio.ByteBuffer;
024
025import org.apache.hadoop.hbase.KeyValue.Type;
026import org.apache.hadoop.hbase.TestCellUtil.ByteBufferCellImpl;
027import org.apache.hadoop.hbase.testclassification.MiscTests;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.junit.Test;
031import org.junit.experimental.categories.Category;
032@Category({MiscTests.class, SmallTests.class})
033public class TestCellComparator {
034
035  private CellComparator comparator = CellComparator.COMPARATOR;
036  byte[] row1 = Bytes.toBytes("row1");
037  byte[] row2 = Bytes.toBytes("row2");
038  byte[] row_1_0 = Bytes.toBytes("row10");
039
040  byte[] fam1 = Bytes.toBytes("fam1");
041  byte[] fam2 = Bytes.toBytes("fam2");
042  byte[] fam_1_2 = Bytes.toBytes("fam12");
043
044  byte[] qual1 = Bytes.toBytes("qual1");
045  byte[] qual2 = Bytes.toBytes("qual2");
046
047  byte[] val = Bytes.toBytes("val");
048
049  @Test
050  public void testCompareCells() {
051    KeyValue kv1 = new KeyValue(row1, fam1, qual1, val);
052    KeyValue kv2 = new KeyValue(row2, fam1, qual1, val);
053    assertTrue((comparator.compare(kv1, kv2)) < 0);
054
055    kv1 = new KeyValue(row1, fam2, qual1, val);
056    kv2 = new KeyValue(row1, fam1, qual1, val);
057    assertTrue((CellComparator.compareFamilies(kv1, kv2) > 0));
058
059    kv1 = new KeyValue(row1, fam1, qual1, 1l, val);
060    kv2 = new KeyValue(row1, fam1, qual1, 2l, val);
061    assertTrue((comparator.compare(kv1, kv2) > 0));
062
063    kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
064    kv2 = new KeyValue(row1, fam1, qual1, 1l, Type.Maximum);
065    assertTrue((comparator.compare(kv1, kv2) > 0));
066
067    kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
068    kv2 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
069    assertTrue((CellUtil.equals(kv1, kv2)));
070  }
071
072  @Test
073  public void testCompareCellWithKey() throws Exception {
074    KeyValue kv1 = new KeyValue(row1, fam1, qual1, val);
075    KeyValue kv2 = new KeyValue(row2, fam1, qual1, val);
076    assertTrue((comparator.compare(kv1, kv2.getKey(), 0, kv2.getKey().length)) < 0);
077
078    kv1 = new KeyValue(row1, fam2, qual1, val);
079    kv2 = new KeyValue(row1, fam1, qual1, val);
080    assertTrue((comparator.compare(kv1, kv2.getKey(), 0, kv2.getKey().length)) > 0);
081
082    kv1 = new KeyValue(row1, fam1, qual1, 1l, val);
083    kv2 = new KeyValue(row1, fam1, qual1, 2l, val);
084    assertTrue((comparator.compare(kv1, kv2.getKey(), 0, kv2.getKey().length)) > 0);
085
086    kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
087    kv2 = new KeyValue(row1, fam1, qual1, 1l, Type.Maximum);
088    assertTrue((comparator.compare(kv1, kv2.getKey(), 0, kv2.getKey().length)) > 0);
089
090    kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
091    kv2 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
092    assertTrue((comparator.compare(kv1, kv2.getKey(), 0, kv2.getKey().length)) == 0);
093  }
094
095  @Test
096  public void testCompareByteBufferedCell() {
097    byte[] r1 = Bytes.toBytes("row1");
098    byte[] r2 = Bytes.toBytes("row2");
099    byte[] f1 = Bytes.toBytes("cf1");
100    byte[] q1 = Bytes.toBytes("qual1");
101    byte[] q2 = Bytes.toBytes("qual2");
102    byte[] v = Bytes.toBytes("val1");
103    KeyValue kv = new KeyValue(r1, f1, q1, v);
104    ByteBuffer buffer = ByteBuffer.wrap(kv.getBuffer());
105    Cell bbCell1 = new ByteBufferCellImpl(buffer, 0, buffer.remaining());
106    kv = new KeyValue(r2, f1, q1, v);
107    buffer = ByteBuffer.wrap(kv.getBuffer());
108    Cell bbCell2 = new ByteBufferCellImpl(buffer, 0, buffer.remaining());
109    assertEquals(0, CellComparator.compareColumns(bbCell1, bbCell2));
110    assertEquals(0, CellComparator.compareColumns(bbCell1, kv));
111    kv = new KeyValue(r2, f1, q2, v);
112    buffer = ByteBuffer.wrap(kv.getBuffer());
113    Cell bbCell3 = new ByteBufferCellImpl(buffer, 0, buffer.remaining());
114    assertEquals(0, CellComparator.compareFamilies(bbCell2, bbCell3));
115    assertTrue(CellComparator.compareQualifiers(bbCell2, bbCell3) < 0);
116    assertTrue(CellComparator.compareColumns(bbCell2, bbCell3) < 0);
117
118    assertEquals(0, CellComparator.COMPARATOR.compareRows(bbCell2, bbCell3));
119    assertTrue(CellComparator.COMPARATOR.compareRows(bbCell1, bbCell2) < 0);
120  }
