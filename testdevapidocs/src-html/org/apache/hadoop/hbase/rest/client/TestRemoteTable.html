063  // Verify that invalid URL characters and arbitrary bytes are escaped when
064  // constructing REST URLs per HBASE-7621. RemoteHTable should support row keys
065  // and qualifiers containing any byte for all table operations.
066  private static final String INVALID_URL_CHARS_1 =
067      "|\"\\^{}\u0001\u0002\u0003\u0004\u0005\u0006\u0007\u0008\u0009\u000B\u000C";
068
069  // HColumnDescriptor prevents certain characters in column names.  The following
070  // are examples of characters are allowed in column names but are not valid in
071  // URLs.
072  private static final String INVALID_URL_CHARS_2 = "|^{}\u0242";
073
074  // Besides alphanumeric these characters can also be present in table names.
075  private static final String VALID_TABLE_NAME_CHARS = "_-.";
076
077  private static final TableName TABLE =
078      TableName.valueOf("TestRemoteTable" + VALID_TABLE_NAME_CHARS);
079
080  private static final byte[] ROW_1 = Bytes.toBytes("testrow1" + INVALID_URL_CHARS_1);
081  private static final byte[] ROW_2 = Bytes.toBytes("testrow2" + INVALID_URL_CHARS_1);
082  private static final byte[] ROW_3 = Bytes.toBytes("testrow3" + INVALID_URL_CHARS_1);
083  private static final byte[] ROW_4 = Bytes.toBytes("testrow4"+ INVALID_URL_CHARS_1);
084
085  private static final byte[] COLUMN_1 = Bytes.toBytes("a" + INVALID_URL_CHARS_2);
086  private static final byte[] COLUMN_2 = Bytes.toBytes("b" + INVALID_URL_CHARS_2);
087  private static final byte[] COLUMN_3 = Bytes.toBytes("c" + INVALID_URL_CHARS_2);
088
089  private static final byte[] QUALIFIER_1 = Bytes.toBytes("1" + INVALID_URL_CHARS_1);
090  private static final byte[] QUALIFIER_2 = Bytes.toBytes("2" + INVALID_URL_CHARS_1);
091  private static final byte[] VALUE_1 = Bytes.toBytes("testvalue1");
092  private static final byte[] VALUE_2 = Bytes.toBytes("testvalue2");

094  private static final long ONE_HOUR = 60 * 60 * 1000;
095  private static final long TS_2 = System.currentTimeMillis();
096  private static final long TS_1 = TS_2 - ONE_HOUR;

109  @Before
110  public void before() throws Exception  {
111    Admin admin = TEST_UTIL.getAdmin();
112    if (admin.tableExists(TABLE)) {
113      if (admin.isTableEnabled(TABLE)) admin.disableTable(TABLE);
114      admin.deleteTable(TABLE);
115    }
116    HTableDescriptor htd = new HTableDescriptor(TABLE);
117    htd.addFamily(new HColumnDescriptor(COLUMN_1).setMaxVersions(3));
118    htd.addFamily(new HColumnDescriptor(COLUMN_2).setMaxVersions(3));
119    htd.addFamily(new HColumnDescriptor(COLUMN_3).setMaxVersions(3));
120    admin.createTable(htd);
121    try (Table table = TEST_UTIL.getConnection().getTable(TABLE)) {
122      Put put = new Put(ROW_1);
123      put.addColumn(COLUMN_1, QUALIFIER_1, TS_2, VALUE_1);
124      table.put(put);
125      put = new Put(ROW_2);
126      put.addColumn(COLUMN_1, QUALIFIER_1, TS_1, VALUE_1);
127      put.addColumn(COLUMN_1, QUALIFIER_1, TS_2, VALUE_2);
128      put.addColumn(COLUMN_2, QUALIFIER_2, TS_2, VALUE_2);
129      table.put(put);
130    }
131    remoteTable = new RemoteHTable(
132      new Client(new Cluster().add("localhost", 
133          REST_TEST_UTIL.getServletPort())),
134        TEST_UTIL.getConfiguration(), TABLE.toBytes());
135  }

160  @Test
161  public void testGet() throws IOException {
162    Get get = new Get(ROW_1);
163    Result result = remoteTable.get(get);
164    byte[] value1 = result.getValue(COLUMN_1, QUALIFIER_1);
165    byte[] value2 = result.getValue(COLUMN_2, QUALIFIER_2);
166    assertNotNull(value1);
167    assertTrue(Bytes.equals(VALUE_1, value1));
168    assertNull(value2);
169
170    get = new Get(ROW_1);
171    get.addFamily(COLUMN_3);
172    result = remoteTable.get(get);
173    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
174    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
175    assertNull(value1);
176    assertNull(value2);
177
178    get = new Get(ROW_1);
179    get.addColumn(COLUMN_1, QUALIFIER_1);
180    get.addColumn(COLUMN_2, QUALIFIER_2);
181    result = remoteTable.get(get);
182    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
183    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
184    assertNotNull(value1);
185    assertTrue(Bytes.equals(VALUE_1, value1));
186    assertNull(value2);
187
188    get = new Get(ROW_2);
189    result = remoteTable.get(get);
190    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
191    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
192    assertNotNull(value1);
193    assertTrue(Bytes.equals(VALUE_2, value1)); // @TS_2
194    assertNotNull(value2);
195    assertTrue(Bytes.equals(VALUE_2, value2));
196
197    get = new Get(ROW_2);
198    get.addFamily(COLUMN_1);
199    result = remoteTable.get(get);
200    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
201    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
202    assertNotNull(value1);
203    assertTrue(Bytes.equals(VALUE_2, value1)); // @TS_2
204    assertNull(value2);
205
206    get = new Get(ROW_2);
207    get.addColumn(COLUMN_1, QUALIFIER_1);
208    get.addColumn(COLUMN_2, QUALIFIER_2);
209    result = remoteTable.get(get);
210    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
211    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
212    assertNotNull(value1);
213    assertTrue(Bytes.equals(VALUE_2, value1)); // @TS_2
214    assertNotNull(value2);
215    assertTrue(Bytes.equals(VALUE_2, value2));
216
217    // test timestamp
218
219    get = new Get(ROW_2);
220    get.addFamily(COLUMN_1);
221    get.addFamily(COLUMN_2);
222    get.setTimeStamp(TS_1);
223    result = remoteTable.get(get);
224    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
225    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
226    assertNotNull(value1);
227    assertTrue(Bytes.equals(VALUE_1, value1)); // @TS_1
228    assertNull(value2);
229
230    // test timerange
231
232    get = new Get(ROW_2);
233    get.addFamily(COLUMN_1);
234    get.addFamily(COLUMN_2);
235    get.setTimeRange(0, TS_1 + 1);
236    result = remoteTable.get(get);
237    value1 = result.getValue(COLUMN_1, QUALIFIER_1);
238    value2 = result.getValue(COLUMN_2, QUALIFIER_2);
239    assertNotNull(value1);
240    assertTrue(Bytes.equals(VALUE_1, value1)); // @TS_1
241    assertNull(value2);
242
243    // test maxVersions
244
245    get = new Get(ROW_2);
246    get.addFamily(COLUMN_1);
247    get.setMaxVersions(2);
248    result = remoteTable.get(get);
249    int count = 0;
250    for (Cell kv: result.listCells()) {
251      if (CellUtil.matchingFamily(kv, COLUMN_1) && TS_1 == kv.getTimestamp()) {
252        assertTrue(CellUtil.matchingValue(kv, VALUE_1)); // @TS_1
253        count++;
254      }
255      if (CellUtil.matchingFamily(kv, COLUMN_1) && TS_2 == kv.getTimestamp()) {
256        assertTrue(CellUtil.matchingValue(kv, VALUE_2)); // @TS_2
257        count++;
258      }
259    }
260    assertEquals(2, count);
261  }
