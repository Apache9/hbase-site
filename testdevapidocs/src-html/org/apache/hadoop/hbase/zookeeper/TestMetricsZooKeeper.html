019package org.apache.hadoop.hbase.zookeeper;
020
021import org.apache.hadoop.hbase.testclassification.SmallTests;
022import org.junit.Test;
023import org.junit.experimental.categories.Category;
024import static org.mockito.Mockito.*;
025
026@Category(SmallTests.class)
027public class TestMetricsZooKeeper {
028
029  @Test
030  public void testRegisterExceptions() {
031    MetricsZooKeeperSource zkSource = mock(MetricsZooKeeperSourceImpl.class);
032    MetricsZooKeeper metricsZK = new MetricsZooKeeper(zkSource);
033    metricsZK.registerAuthFailedException();
034    metricsZK.registerConnectionLossException();
035    metricsZK.registerConnectionLossException();
036    metricsZK.registerDataInconsistencyException();
037    metricsZK.registerInvalidACLException();
038    metricsZK.registerNoAuthException();
039    metricsZK.registerOperationTimeoutException();
040    metricsZK.registerOperationTimeoutException();
041    metricsZK.registerRuntimeInconsistencyException();
042    metricsZK.registerSessionExpiredException();
043    metricsZK.registerSystemErrorException();
044    metricsZK.registerSystemErrorException();
045    metricsZK.registerFailedZKCall();
046
047    verify(zkSource, times(1)).incrementAuthFailedCount();
048    // ConnectionLoss Exception was registered twice.
049    verify(zkSource, times(2)).incrementConnectionLossCount();
050    verify(zkSource, times(1)).incrementDataInconsistencyCount();
051    verify(zkSource, times(1)).incrementInvalidACLCount();
052    verify(zkSource, times(1)).incrementNoAuthCount();
053    // OperationTimeout Exception was registered twice.
054    verify(zkSource, times(2)).incrementOperationTimeoutCount();
055    verify(zkSource, times(1)).incrementRuntimeInconsistencyCount();
056    verify(zkSource, times(1)).incrementSessionExpiredCount();
057    // SystemError Exception was registered twice.
058    verify(zkSource, times(2)).incrementSystemErrorCount();
059    verify(zkSource, times(1)).incrementTotalFailedZKCalls();
060  }
061
062  @Test
063  public void testLatencyHistogramUpdates() {
064    MetricsZooKeeperSource zkSource = mock(MetricsZooKeeperSourceImpl.class);
065    MetricsZooKeeper metricsZK = new MetricsZooKeeper(zkSource);
066    long latency = 100;
067
068    metricsZK.registerReadOperationLatency(latency);
069    metricsZK.registerReadOperationLatency(latency);
070    metricsZK.registerWriteOperationLatency(latency);
071    metricsZK.registerSyncOperationLatency(latency);
072    // Read Operation Latency update was registered twice.
073    verify(zkSource, times(2)).recordReadOperationLatency(latency);
074    verify(zkSource, times(1)).recordWriteOperationLatency(latency);
075    verify(zkSource, times(1)).recordSyncOperationLatency(latency);
076  }
077}
