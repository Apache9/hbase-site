102  @Test(timeout = 30000)
103  public void testWriteLockExcludesWriters() throws Exception {
104    final String testName = "testWriteLockExcludesWriters";
105    final ZKInterProcessReadWriteLock readWriteLock =
106        getReadWriteLock(testName);
107    List<Future<Void>> results = Lists.newArrayList();
108    for (int i = 0; i < NUM_THREADS; ++i) {
109      final String threadDesc = testName + i;
110      results.add(executor.submit(new Callable<Void>() {
111        @Override
112        public Void call() throws IOException {
113          ZKInterProcessWriteLock writeLock =
114              readWriteLock.writeLock(Bytes.toBytes(threadDesc));
115          try {
116            writeLock.acquire();
117            try {
118              // No one else should hold the lock
119              assertTrue(isLockHeld.compareAndSet(false, true));
120              Thread.sleep(1000);
121              // No one else should have released the lock
122              assertTrue(isLockHeld.compareAndSet(true, false));
123            } finally {
124              isLockHeld.set(false);
125              writeLock.release();
126            }
127          } catch (InterruptedException e) {
128            LOG.warn(threadDesc + " interrupted", e);
129            Thread.currentThread().interrupt();
130            throw new InterruptedIOException();
131          }
132          return null;
133        }
134      }));
135
136    }
137    MultithreadedTestUtil.assertOnFutures(results);
138  }

140  @Test(timeout = 30000)
141  public void testReadLockDoesNotExcludeReaders() throws Exception {
142    final String testName = "testReadLockDoesNotExcludeReaders";
143    final ZKInterProcessReadWriteLock readWriteLock =
144        getReadWriteLock(testName);
145    final CountDownLatch locksAcquiredLatch = new CountDownLatch(NUM_THREADS);
146    final AtomicInteger locksHeld = new AtomicInteger(0);
147    List<Future<Void>> results = Lists.newArrayList();
148    for (int i = 0; i < NUM_THREADS; ++i) {
149      final String threadDesc = testName + i;
150      results.add(executor.submit(new Callable<Void>() {
151        @Override
152        public Void call() throws Exception {
153          ZKInterProcessReadLock readLock =
154              readWriteLock.readLock(Bytes.toBytes(threadDesc));
155          readLock.acquire();
156          try {
157            locksHeld.incrementAndGet();
158            locksAcquiredLatch.countDown();
159            Thread.sleep(1000);
160          } finally {
161            readLock.release();
162            locksHeld.decrementAndGet();
163          }
164          return null;
165        }
166      }));
167    }
168    locksAcquiredLatch.await();
169    assertEquals(locksHeld.get(), NUM_THREADS);
170    MultithreadedTestUtil.assertOnFutures(results);
171  }

173  @Test(timeout = 30000)
174  public void testReadLockExcludesWriters() throws Exception {
175    // Submit a read lock request first
176    // Submit a write lock request second
177    final String testName = "testReadLockExcludesWriters";
178    List<Future<Void>> results = Lists.newArrayList();
179    final CountDownLatch readLockAcquiredLatch = new CountDownLatch(1);
180    Callable<Void> acquireReadLock = new Callable<Void>() {
181      @Override
182      public Void call() throws Exception {
183        final String threadDesc = testName + "-acquireReadLock";
184        ZKInterProcessReadLock readLock =
185            getReadWriteLock(testName).readLock(Bytes.toBytes(threadDesc));
186        readLock.acquire();
187        try {
188          assertTrue(isLockHeld.compareAndSet(false, true));
189          readLockAcquiredLatch.countDown();
190          Thread.sleep(1000);
191        } finally {
192          isLockHeld.set(false);
193          readLock.release();
194        }
195        return null;
196      }
197    };
198    Callable<Void> acquireWriteLock = new Callable<Void>() {
199      @Override
200      public Void call() throws Exception {
201        final String threadDesc = testName + "-acquireWriteLock";
202        ZKInterProcessWriteLock writeLock =
203            getReadWriteLock(testName).writeLock(Bytes.toBytes(threadDesc));
204        readLockAcquiredLatch.await();
205        assertTrue(isLockHeld.get());
206        writeLock.acquire();
207        try {
208          assertFalse(isLockHeld.get());
209        } finally {
210          writeLock.release();
211        }
212        return null;
213      }
214    };
215    results.add(executor.submit(acquireReadLock));
216    results.add(executor.submit(acquireWriteLock));
217    MultithreadedTestUtil.assertOnFutures(results);
218  }

220  private static ZKInterProcessReadWriteLock getReadWriteLock(String testName)
221      throws IOException {
222    MetadataHandler handler = new MetadataHandler() {
223      @Override
224      public void handleMetadata(byte[] ownerMetadata) {
225        LOG.info("Lock info: " + Bytes.toString(ownerMetadata));
226      }
227    };
228    ZooKeeperWatcher zkWatcher = getZooKeeperWatcher(testName);
229    String znode = ZKUtil.joinZNode(zkWatcher.znodePaths.tableLockZNode, testName);
230
231    return new ZKInterProcessReadWriteLock(zkWatcher, znode, handler);
232  }

280  @Test(timeout = 60000)
281  public void testTimeout() throws Exception {
282    final String testName = "testTimeout";
283    final CountDownLatch lockAcquiredLatch = new CountDownLatch(1);
284    Callable<Void> shouldHog = new Callable<Void>() {
285      @Override
286      public Void call() throws Exception {
287        final String threadDesc = testName + "-shouldHog";
288        ZKInterProcessWriteLock lock =
289            getReadWriteLock(testName).writeLock(Bytes.toBytes(threadDesc));
290        lock.acquire();
291        lockAcquiredLatch.countDown();
292        Thread.sleep(10000);
293        lock.release();
294        return null;
295      }
296    };
297    Callable<Void> shouldTimeout = new Callable<Void>() {
298      @Override
299      public Void call() throws Exception {
300        final String threadDesc = testName + "-shouldTimeout";
301        ZKInterProcessWriteLock lock =
302            getReadWriteLock(testName).writeLock(Bytes.toBytes(threadDesc));
303        lockAcquiredLatch.await();
304        assertFalse(lock.tryAcquire(5000));
305        return null;
306      }
307    };
308    Callable<Void> shouldAcquireLock = new Callable<Void>() {
309      @Override
310      public Void call() throws Exception {
311        final String threadDesc = testName + "-shouldAcquireLock";
312        ZKInterProcessWriteLock lock =
313            getReadWriteLock(testName).writeLock(Bytes.toBytes(threadDesc));
314        lockAcquiredLatch.await();
315        assertTrue(lock.tryAcquire(30000));
316        lock.release();
317        return null;
318      }
319    };
320    List<Future<Void>> results = Lists.newArrayList();
321    results.add(executor.submit(shouldHog));
322    results.add(executor.submit(shouldTimeout));
323    results.add(executor.submit(shouldAcquireLock));
324    MultithreadedTestUtil.assertOnFutures(results);
325  }

327  @Test(timeout = 60000)
328  public void testMultipleClients() throws Exception {
329    //tests lock usage from multiple zookeeper clients with different sessions.
330    //acquire one read lock, then one write lock
331    final String testName = "testMultipleClients";
332
333    //different zookeeper sessions with separate identifiers
334    ZooKeeperWatcher zkWatcher1 = new ZooKeeperWatcher(conf, "testMultipleClients-1", null);
335    ZooKeeperWatcher zkWatcher2 = new ZooKeeperWatcher(conf, "testMultipleClients-2", null);
336
337    String znode = ZKUtil.joinZNode(zkWatcher1.znodePaths.tableLockZNode, testName);
338
339    ZKInterProcessReadWriteLock clientLock1
340      = new ZKInterProcessReadWriteLock(zkWatcher1, znode, null);
341    ZKInterProcessReadWriteLock clientLock2
342      = new ZKInterProcessReadWriteLock(zkWatcher2, znode, null);
343
344    InterProcessLock lock1 = clientLock1.readLock(Bytes.toBytes("client1"));
345    lock1.acquire();
346
347    //try to acquire, but it will timeout. We are testing whether this will cause any problems
348    //due to the read lock being from another client
349    InterProcessLock lock2 = clientLock2.writeLock(Bytes.toBytes("client2"));
350    assertFalse(lock2.tryAcquire(1000));
351
352    lock1.release();
353
354    //this time it will acquire
355    assertTrue(lock2.tryAcquire(5000));
356    lock2.release();
357    zkWatcher1.close();
358    zkWatcher2.close();
359  }
