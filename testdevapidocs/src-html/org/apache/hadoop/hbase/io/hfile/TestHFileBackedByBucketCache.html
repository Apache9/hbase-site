053/**
054 * Test for file-backed BucketCache.
055 */
056// This is marked a LargeTest so it runs in its own JVM. We do this because we are making use of
057// the cache and the cache is global. We don't want any other concurrent test polluting ours which
058// can happen if more than one test in a single JVM which can happen when tests are small.
059@Category({IOTests.class, LargeTests.class})
060public class TestHFileBackedByBucketCache {
061  private static final Log LOG = LogFactory.getLog(TestHFileBackedByBucketCache.class);
062  @Rule public TestName name = new TestName();
063  @Rule public final TestRule timeout = CategoryBasedTimeout.builder().withTimeout(this.getClass()).
064      withLookingForStuckThread(true).build();
065  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
066  private static final int ROW_LENGTH = 4;
067  private Configuration conf;
068  private FileSystem fs;
069
070  // MATH! SIZING FOR THE TEST!
071  // Set bucketcache to be smallest size possible which is 1MB. We do that in the test
072  // @Before <code>before</code> method. Into out 1MB cache, have it so only one bucket. If
073  // bucketsize is set to 125kb in size, we will have one bucket in our 1MB bucketcache. It is
074  // cryptic how this comes about but basically comes down to
075  // {@link BucketAllocator#FEWEST_ITEMS_IN_BUCKET} being '4'... so 4 * 125 = just over 500k or so
076  // which makes for one bucket only in 1M which you can see from TRACE logging:
077  //
078  // Cache totalSize=532480, buckets=1, bucket capacity=
079  //   532480=(4*133120)=(FEWEST_ITEMS_IN_BUCKET*(largest configured bucketcache size))
080  //
081  // Now into this one big bucket, we write hfileblocks....Each hfileblock has two keys because
082  // first is under the BLOCKSIZE of 64k and then the second puts us over the 64k...
083  // so two Cells per block...
084  /**
085   * Default size.
086   */
087  private static final int BLOCKSIZE = 64 * 1024;
088
089  /**
090   * Bucket sizes get multiplied by 4 for actual bucket size.
091   * See {@link BucketAllocator#FEWEST_ITEMS_IN_BUCKET}.
092   */
093  private static final int BUCKETSIZE = 125 * 1024;
094
095  /**
096   * Make it so one Cell is just under a BLOCKSIZE. The second Cell puts us over the BLOCKSIZE
097   * so we have two Cells per HFilBlock.
098   */
099  private static final int VALUE_SIZE = 33 * 1024;
100
101  @Before
102  public void before() throws IOException {
103    // Do setup of a bucketcache that has one bucket only. Enable trace-level logging for
104    // key classes.
105    this.conf = TEST_UTIL.getConfiguration();
106    this.fs = FileSystem.get(conf);
107
108    // Set BucketCache and HFileBlock to log at trace level.
109    setTraceLevel(BucketCache.class);
110    setTraceLevel(HFileBlock.class);
111    setTraceLevel(HFileReaderImpl.class);
112    setTraceLevel(BucketAllocator.class);
113  }
114
115  //  Assumes log4j logging.
116  private static void setTraceLevel(final Class<?> clazz) {
117    Log testlog = LogFactory.getLog(clazz.getName());
118    ((org.apache.commons.logging.impl.Log4JLogger)testlog).getLogger().
119      setLevel(org.apache.log4j.Level.TRACE);
120  }
121
122  /**
123   * Test that bucketcache is caching and that the persist of in-memory map works
124   * @throws IOException
125   */
126  @Test
127  public void testBucketCacheCachesAndPersists() throws IOException {
128    // Set up a bucket cache. Set up one that will persist by passing a
129    // hbase.bucketcache.persistent.path value to store the in-memory map of what is out in
130    // the file-backed bucketcache. Set bucketcache to have one size only, BUCKETSIZE.
131    // See "MATH! SIZING FOR THE TEST!" note above around declaration of BUCKETSIZE
132    String bucketCacheDataFile =
133      (new Path(TEST_UTIL.getDataTestDir(), "bucketcache.data")).toString();
134    (new File(bucketCacheDataFile)).getParentFile().mkdirs();
135    this.conf.set("hbase.bucketcache.ioengine", "file:" + bucketCacheDataFile);
136    this.conf.set("hbase.bucketcache.persistent.path", bucketCacheDataFile + ".map");
137    this.conf.setStrings("hbase.bucketcache.bucket.sizes", Integer.toString(BUCKETSIZE));
138    // This is minimum bucketcache size.... 1MB.
139    this.conf.setInt("hbase.bucketcache.size", 1);
140    // Write 8 entries which should make for four hfileBlocks.
141    final int count = 8;
142    final int hfileBlockCount = 4;
143    Path hfilePath = new Path(TEST_UTIL.getDataTestDir(), this.name.getMethodName());
144    // Clear out any existing global cache instance. Will pollute our tests below. Any concurrent
145    // running test will pollute our results below.
146    CacheConfig.GLOBAL_BLOCK_CACHE_INSTANCE = null;
147    CacheConfig cacheConfig = new CacheConfig(conf);
148    List<Cell> writtenCells = writeFile(hfilePath, Compression.Algorithm.NONE, cacheConfig, count);
149    CacheStats stats = cacheConfig.getBlockCache().getStats();
150    List<Cell> readCells = readFile(hfilePath, cacheConfig);
151    assertTrue(!writtenCells.isEmpty());
152    assertEquals(writtenCells.size(), readCells.size());
153    assertEquals(hfileBlockCount, stats.getMissCount());
154    assertEquals(1, stats.getHitCount()); // readFile will read first block is from cache.
155
156    // Now, close out the cache and then reopen and verify that cache still has our blocks.
157    // Assert that persistence works.
158    cacheConfig.getBlockCache().shutdown();
159    // Need to clear the global cache else the new CacheConfig won't create a bucketcache but
160    // just reuse the old one.
161    CacheConfig.GLOBAL_BLOCK_CACHE_INSTANCE = null;
162    cacheConfig = new CacheConfig(conf);
163    stats = cacheConfig.getBlockCache().getStats();
164    assertEquals(0, stats.getHitCachingCount());
165    readCells = readFile(hfilePath, cacheConfig);
166    // readFile will read all hfileblocs in the file, hfileBlockCount, and then one more, so + 1.
167    assertEquals(hfileBlockCount + 1, stats.getHitCachingCount());
168  }

170  /**
171   * Write a file with <code>count</code> entries.
172   * @return The Cells written to the file.
173   * @throws IOException
174   */
175  private List<Cell> writeFile(final Path hfilePath, final Compression.Algorithm compressAlgo,
176      final CacheConfig cacheConfig, final int count)
177  throws IOException {
178    List<Cell> cells = new ArrayList<Cell>(count);
179    HFileContext context =
180        new HFileContextBuilder().withBlockSize(BLOCKSIZE).withCompression(compressAlgo).build();
181    try (HFile.Writer writer = new HFile.WriterFactory(conf, cacheConfig).
182        withPath(fs, hfilePath).
183        withFileContext(context).
184        withComparator(CellComparator.COMPARATOR).
185        create()) {
186      byte [] valueBytes = new byte [VALUE_SIZE];
187      for (int i = 0; i < valueBytes.length; i++) valueBytes[i] = '0';
188      for (int i = 0; i < count; ++i) {
189        byte[] keyBytes = format(i);
190        KeyValue keyValue = new KeyValue(keyBytes, HConstants.CATALOG_FAMILY, keyBytes,
191            HConstants.LATEST_TIMESTAMP, valueBytes);
192        writer.append(keyValue);
193        cells.add(keyValue);
194      }
195    }
196    return cells;
197  }
198
199  /**
200   * Read the whole file, then read the first block so we get something from cache for sure.
201   * So... there are TOTAL_BLOCKS_IN_FILE read + 1. See math at head of this class.
202   * @return The Cells read from the file.
203   */
204  private List<Cell> readFile(final Path hfilePath, final CacheConfig cacheConfig)
205  throws IOException {
206    List<Cell> cells = new ArrayList<Cell>();
207    try (HFile.Reader reader = HFile.createReader(this.fs, hfilePath, cacheConfig, this.conf);
208        HFileScanner scanner = reader.getScanner(true, true)) {
209      scanner.seekTo();
210      do {
211        cells.add(scanner.getCell());
212        LOG.info(scanner.getKey());
213      } while (scanner.next());
214      // Do a random seek just so we see a block coming from cache.
215      scanner.seekTo(reader.getFirstKey());
216      scanner.next();
217      LOG.info(scanner.getCell());
218    }
219    return cells;
220  }

222  /*
223   * Format passed integer.
224   * @param number
225   * @return Returns zero-prefixed ROW_LENGTH-byte wide decimal version of passed
226   * number (Does absolute in case number is negative).
227   */
228  private static byte [] format(final int number) {
229    byte [] b = new byte[ROW_LENGTH];
230    int d = Math.abs(number);
231    for (int i = b.length - 1; i >= 0; i--) {
232      b[i] = (byte)((d % 10) + '0');
233      d /= 10;
234    }
235    return b;
236  }
