017package org.apache.hadoop.hbase.io.hfile;
018
019import java.util.ArrayList;
020import java.util.List;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.fs.FileSystem;
024import org.apache.hadoop.fs.Path;
025import org.apache.hadoop.hbase.CellUtil;
026import org.apache.hadoop.hbase.HBaseTestingUtility;
027import org.apache.hadoop.hbase.testclassification.IOTests;
028import org.apache.hadoop.hbase.testclassification.SmallTests;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.junit.Test;
031import org.junit.experimental.categories.Category;
032
033/**
034 * Test a case when an inline index chunk is converted to a root one. This reproduces the bug in
035 * HBASE-6871. We write a carefully selected number of relatively large keys so that we accumulate
036 * a leaf index chunk that only goes over the configured index chunk size after adding the last
037 * key/value. The bug is in that when we close the file, we convert that inline (leaf-level) chunk
038 * into a root chunk, but then look at the size of that root chunk, find that it is greater than
039 * the configured chunk size, and split it into a number of intermediate index blocks that should
040 * really be leaf-level blocks. If more keys were added, we would flush the leaf-level block, add
041 * another entry to the root-level block, and that would prevent us from upgrading the leaf-level
042 * chunk to the root chunk, thus not triggering the bug. 
043 */
044@Category({IOTests.class, SmallTests.class})
045public class TestHFileInlineToRootChunkConversion {
046  private final HBaseTestingUtility testUtil = new HBaseTestingUtility();
047  private final Configuration conf = testUtil.getConfiguration();
048  
049  @Test
050  public void testWriteHFile() throws Exception {
051    Path hfPath = new Path(testUtil.getDataTestDir(),
052        TestHFileInlineToRootChunkConversion.class.getSimpleName() + ".hfile");
053    int maxChunkSize = 1024;
054    FileSystem fs = FileSystem.get(conf);
055    CacheConfig cacheConf = new CacheConfig(conf);
056    conf.setInt(HFileBlockIndex.MAX_CHUNK_SIZE_KEY, maxChunkSize); 
057    HFileContext context = new HFileContextBuilder().withBlockSize(16).build();
058    HFile.Writer hfw = new HFile.WriterFactory(conf, cacheConf)
059            .withFileContext(context)
060            .withPath(fs, hfPath).create();
061    List<byte[]> keys = new ArrayList<>();
062    StringBuilder sb = new StringBuilder();
063
064    for (int i = 0; i < 4; ++i) {
065      sb.append("key" + String.format("%05d", i));
066      sb.append("_");
067      for (int j = 0; j < 100; ++j) {
068        sb.append('0' + j);
069      }
070      String keyStr = sb.toString();
071      sb.setLength(0);
072
073      byte[] k = Bytes.toBytes(keyStr);
074      keys.add(k);
075      byte[] v = Bytes.toBytes("value" + i);
076      hfw.append(CellUtil.createCell(k, v));
077    }
078    hfw.close();
079
080    HFile.Reader reader = HFile.createReader(fs, hfPath, cacheConf, true, conf);
081    // Scanner doesn't do Cells yet.  Fix.
082    HFileScanner scanner = reader.getScanner(true, true);
083    for (int i = 0; i < keys.size(); ++i) {
084      scanner.seekTo(CellUtil.createCell(keys.get(i)));
085    }
086    reader.close();
087  }
088}
