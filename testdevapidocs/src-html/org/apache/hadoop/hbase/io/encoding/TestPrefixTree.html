086  @Test
087  public void testHBASE11728() throws Exception {
088    Put put = new Put(Bytes.toBytes("a-b-0-0"));
089    put.addColumn(fam, qual1, Bytes.toBytes("c1-value"));
090    region.put(put);
091    put = new Put(row1_bytes);
092    put.addColumn(fam, qual1, Bytes.toBytes("c1-value"));
093    region.put(put);
094    put = new Put(row2_bytes);
095    put.addColumn(fam, qual2, Bytes.toBytes("c2-value"));
096    region.put(put);
097    put = new Put(row3_bytes);
098    put.addColumn(fam, qual2, Bytes.toBytes("c2-value-2"));
099    region.put(put);
100    put = new Put(row4_bytes);
101    put.addColumn(fam, qual2, Bytes.toBytes("c2-value-3"));
102    region.put(put);
103    region.flush(true);
104    String[] rows = new String[3];
105    rows[0] = row1;
106    rows[1] = row2;
107    rows[2] = row3;
108    byte[][] val = new byte[3][];
109    val[0] = Bytes.toBytes("c1-value");
110    val[1] = Bytes.toBytes("c2-value");
111    val[2] = Bytes.toBytes("c2-value-2");
112    Scan scan = new Scan();
113    scan.setStartRow(row1_bytes);
114    scan.setStopRow(Bytes.toBytes("a-b-A-1:"));
115
116    RegionScanner scanner = region.getScanner(scan);
117    List<Cell> cells = new ArrayList<Cell>();
118    for (int i = 0; i < 3; i++) {
119      assertEquals(i < 2, scanner.next(cells));
120      CellScanner cellScanner = Result.create(cells).cellScanner();
121      while (cellScanner.advance()) {
122        assertEquals(rows[i], Bytes.toString(cellScanner.current().getRowArray(), cellScanner
123            .current().getRowOffset(), cellScanner.current().getRowLength()));
124        assertEquals(Bytes.toString(val[i]), Bytes.toString(cellScanner.current().getValueArray(),
125          cellScanner.current().getValueOffset(), cellScanner.current().getValueLength()));
126      }
127      cells.clear();
128    }
129    scanner.close();
130
131    // Add column
132    scan = new Scan();
133    scan.addColumn(fam, qual2);
134    scan.setStartRow(row1_bytes);
135    scan.setStopRow(Bytes.toBytes("a-b-A-1:"));
136    scanner = region.getScanner(scan);
137    for (int i = 1; i < 3; i++) {
138      assertEquals(i < 2, scanner.next(cells));
139      CellScanner cellScanner = Result.create(cells).cellScanner();
140      while (cellScanner.advance()) {
141        assertEquals(rows[i], Bytes.toString(cellScanner.current().getRowArray(), cellScanner
142            .current().getRowOffset(), cellScanner.current().getRowLength()));
143      }
144      cells.clear();
145    }
146    scanner.close();
147
148    scan = new Scan();
149    scan.addColumn(fam, qual2);
150    scan.setStartRow(Bytes.toBytes("a-b-A-1-"));
151    scan.setStopRow(Bytes.toBytes("a-b-A-1:"));
152    scanner = region.getScanner(scan);
153    for (int i = 1; i < 3; i++) {
154      assertEquals(i < 2, scanner.next(cells));
155      CellScanner cellScanner = Result.create(cells).cellScanner();
156      while (cellScanner.advance()) {
157        assertEquals(rows[i], Bytes.toString(cellScanner.current().getRowArray(), cellScanner
158            .current().getRowOffset(), cellScanner.current().getRowLength()));
159      }
160      cells.clear();
161    }
162    scanner.close();
163
164    scan = new Scan();
165    scan.addColumn(fam, qual2);
166    scan.setStartRow(Bytes.toBytes("a-b-A-1-140239"));
167    scan.setStopRow(Bytes.toBytes("a-b-A-1:"));
168    scanner = region.getScanner(scan);
169    assertFalse(scanner.next(cells));
170    assertFalse(cells.isEmpty());
171    scanner.close();
172  }

174  @Test
175  public void testHBASE12817() throws IOException {
176    for (int i = 0; i < 100; i++) {
177      region
178          .put(new Put(Bytes.toBytes("obj" + (2900 + i))).addColumn(fam, qual1, Bytes.toBytes(i)));
179    }
180    region.put(new Put(Bytes.toBytes("obj299")).addColumn(fam, qual1, Bytes.toBytes("whatever")));
181    region.put(new Put(Bytes.toBytes("obj29")).addColumn(fam, qual1, Bytes.toBytes("whatever")));
182    region.put(new Put(Bytes.toBytes("obj2")).addColumn(fam, qual1, Bytes.toBytes("whatever")));
183    region.put(new Put(Bytes.toBytes("obj3")).addColumn(fam, qual1, Bytes.toBytes("whatever")));
184    region.flush(true);
185    Scan scan = new Scan(Bytes.toBytes("obj29995"));
186    RegionScanner scanner = region.getScanner(scan);
187    List<Cell> cells = new ArrayList<Cell>();
188    assertFalse(scanner.next(cells));
189    assertArrayEquals(Bytes.toBytes("obj3"), Result.create(cells).getRow());
190  }
