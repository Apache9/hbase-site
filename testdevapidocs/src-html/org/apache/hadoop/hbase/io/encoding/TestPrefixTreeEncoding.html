101  @Test
102  public void testSeekBeforeWithFixedData() throws Exception {
103    formatRowNum = true;
104    PrefixTreeCodec encoder = new PrefixTreeCodec();
105    int batchId = numBatchesWritten++;
106    HFileContext meta = new HFileContextBuilder()
107                        .withHBaseCheckSum(false)
108                        .withIncludesMvcc(false)
109                        .withIncludesTags(includesTag)
110                        .withCompression(Algorithm.NONE).build();
111    HFileBlockEncodingContext blkEncodingCtx = new HFileBlockDefaultEncodingContext(
112        DataBlockEncoding.PREFIX_TREE, new byte[0], meta);
113    ByteArrayOutputStream baosInMemory = new ByteArrayOutputStream();
114    DataOutputStream userDataStream = new DataOutputStream(baosInMemory);
115    generateFixedTestData(kvset, batchId, false, includesTag, encoder, blkEncodingCtx,
116        userDataStream);
117    EncodedSeeker seeker = encoder.createSeeker(CellComparatorImpl.COMPARATOR,
118        encoder.newDataBlockDecodingContext(meta));
119    byte[] onDiskBytes = baosInMemory.toByteArray();
120    ByteBuffer readBuffer = ByteBuffer.wrap(onDiskBytes, DataBlockEncoding.ID_SIZE,
121        onDiskBytes.length - DataBlockEncoding.ID_SIZE);
122    seeker.setCurrentBuffer(new SingleByteBuff(readBuffer));
123
124    // Seek before the first keyvalue;
125    Cell seekKey =
126        PrivateCellUtil.createFirstDeleteFamilyCellOnRow(getRowKey(batchId, 0), CF_BYTES);
127    seeker.seekToKeyInBlock(seekKey, true);
128    assertEquals(null, seeker.getCell());
129
130    // Seek before the middle keyvalue;
131    seekKey = PrivateCellUtil
132        .createFirstDeleteFamilyCellOnRow(getRowKey(batchId, NUM_ROWS_PER_BATCH / 3), CF_BYTES);
133    seeker.seekToKeyInBlock(seekKey, true);
134    assertNotNull(seeker.getCell());
135    assertArrayEquals(getRowKey(batchId, NUM_ROWS_PER_BATCH / 3 - 1),
136      CellUtil.cloneRow(seeker.getCell()));
137
138    // Seek before the last keyvalue;
139    seekKey = PrivateCellUtil.createFirstDeleteFamilyCellOnRow(Bytes.toBytes("zzzz"), CF_BYTES);
140    seeker.seekToKeyInBlock(seekKey, true);
141    assertNotNull(seeker.getCell());
142    assertArrayEquals(getRowKey(batchId, NUM_ROWS_PER_BATCH - 1),
143      CellUtil.cloneRow(seeker.getCell()));
144  }

150    DataOutputStream userDataStream = new DataOutputStream(baosInMemory);

158        DataBlockEncoding.PREFIX_TREE, new byte[0], meta);

160        userDataStream);
161    EncodedSeeker seeker = encoder.createSeeker(CellComparatorImpl.COMPARATOR,
162        encoder.newDataBlockDecodingContext(meta));
163    byte[] onDiskBytes = baosInMemory.toByteArray();
164    ByteBuffer readBuffer = ByteBuffer.wrap(onDiskBytes, DataBlockEncoding.ID_SIZE,
165        onDiskBytes.length - DataBlockEncoding.ID_SIZE);
166    seeker.setCurrentBuffer(new SingleByteBuff(readBuffer));
167    Cell previousKV = null;
168    do {
169      Cell currentKV = seeker.getCell();
170      System.out.println(currentKV);
171      if (previousKV != null && CellComparatorImpl.COMPARATOR.compare(currentKV, previousKV) < 0) {
172        dumpInputKVSet();
173        fail("Current kv " + currentKV + " is smaller than previous keyvalue " + previousKV);
174      }
175      if (!includesTag) {
176        assertFalse(currentKV.getTagsLength() > 0);
177      } else {
178        Assert.assertTrue(currentKV.getTagsLength() > 0);
179      }
180      previousKV = currentKV;
181    } while (seeker.next());
182  }

230  private void verifySeeking(EncodedSeeker encodeSeeker,
231      ByteBuffer encodedData, int batchId) {
232    List<KeyValue> kvList = new ArrayList<>();
233    for (int i = 0; i < NUM_ROWS_PER_BATCH; ++i) {
234      kvList.clear();
235      encodeSeeker.setCurrentBuffer(new SingleByteBuff(encodedData));
236      KeyValue firstOnRow = KeyValueUtil.createFirstOnRow(getRowKey(batchId, i));
237      encodeSeeker.seekToKeyInBlock(
238          new KeyValue.KeyOnlyKeyValue(firstOnRow.getBuffer(), firstOnRow.getKeyOffset(),
239              firstOnRow.getKeyLength()), false);
240      boolean hasMoreOfEncodeScanner = encodeSeeker.next();
241      CollectionBackedScanner collectionScanner = new CollectionBackedScanner(
242          this.kvset);
243      boolean hasMoreOfCollectionScanner = collectionScanner.seek(firstOnRow);
244      if (hasMoreOfEncodeScanner != hasMoreOfCollectionScanner) {
245        dumpInputKVSet();
246        fail("Get error result after seeking " + firstOnRow);
247      }
248      if (hasMoreOfEncodeScanner) {
249        if (CellComparatorImpl.COMPARATOR.compare(encodeSeeker.getCell(),
250            collectionScanner.peek()) != 0) {
251          dumpInputKVSet();
252          fail("Expected " + collectionScanner.peek() + " actual "
253              + encodeSeeker.getCell() + ", after seeking " + firstOnRow);
254        }
255      }
256    }
257  }
258
259  private void dumpInputKVSet() {
260    LOG.info("Dumping input keyvalue set in error case:");
261    for (Cell kv : kvset) {
262      System.out.println(kv);
263    }
264  }

266  private static void generateFixedTestData(ConcurrentSkipListSet<Cell> kvset, int batchId,
267      boolean useTags, PrefixTreeCodec encoder, HFileBlockEncodingContext blkEncodingCtx,
268      DataOutputStream userDataStream) throws Exception {
269    generateFixedTestData(kvset, batchId, true, useTags, encoder, blkEncodingCtx, userDataStream);
270  }
271
272  private static void generateFixedTestData(ConcurrentSkipListSet<Cell> kvset,
273      int batchId, boolean partial, boolean useTags, PrefixTreeCodec encoder,
274      HFileBlockEncodingContext blkEncodingCtx, DataOutputStream userDataStream) throws Exception {
275    for (int i = 0; i < NUM_ROWS_PER_BATCH; ++i) {
276      if (partial && i / 10 % 2 == 1)
277        continue;
278      for (int j = 0; j < NUM_COLS_PER_ROW; ++j) {
279        if (!useTags) {
280          KeyValue kv = new KeyValue(getRowKey(batchId, i), CF_BYTES, getQualifier(j), getValue(
281              batchId, i, j));
282          kvset.add(kv);
283        } else {
284          KeyValue kv = new KeyValue(getRowKey(batchId, i), CF_BYTES, getQualifier(j), 0l,
285              getValue(batchId, i, j), new Tag[] { new ArrayBackedTag((byte) 1, "metaValue1") });
286          kvset.add(kv);
287        }
288      }
289    }
290    encoder.startBlockEncoding(blkEncodingCtx, userDataStream);
291    for (Cell kv : kvset) {
292      encoder.encode(kv, blkEncodingCtx, userDataStream);
293    }
294    encoder.endBlockEncoding(blkEncodingCtx, userDataStream, null);
295  }

297  private static void generateRandomTestData(ConcurrentSkipListSet<Cell> kvset,
298      int batchId, boolean useTags, PrefixTreeCodec encoder,
299      HFileBlockEncodingContext blkEncodingCtx, DataOutputStream userDataStream) throws Exception {
300    Random random = new Random();
301    for (int i = 0; i < NUM_ROWS_PER_BATCH; ++i) {
302      if (random.nextInt(100) < 50)
303        continue;
304      for (int j = 0; j < NUM_COLS_PER_ROW; ++j) {
305        if (random.nextInt(100) < 50)
306          continue;
307        if (!useTags) {
308          KeyValue kv = new KeyValue(getRowKey(batchId, i), CF_BYTES, getQualifier(j), getValue(
309              batchId, i, j));
310          kvset.add(kv);
311        } else {
312          KeyValue kv = new KeyValue(getRowKey(batchId, i), CF_BYTES, getQualifier(j), 0l,
313              getValue(batchId, i, j), new Tag[] { new ArrayBackedTag((byte) 1, "metaValue1") });
314          kvset.add(kv);
315        }
316      }
317    }
318    encoder.startBlockEncoding(blkEncodingCtx, userDataStream);
319    for (Cell kv : kvset) {
320      encoder.encode(kv, blkEncodingCtx, userDataStream);
321    }
322    encoder.endBlockEncoding(blkEncodingCtx, userDataStream, null);
323  }

325  private static byte[] getRowKey(int batchId, int i) {
326    return Bytes
327        .toBytes("batch" + batchId + "_row" + (formatRowNum ? String.format("%04d", i) : i));
328  }
329
330  private static byte[] getQualifier(int j) {
331    return Bytes.toBytes("colfdfafhfhsdfhsdfh" + j);
332  }
333
334  private static byte[] getValue(int batchId, int i, int j) {
335    return Bytes.toBytes("value_for_" + Bytes.toString(getRowKey(batchId, i)) + "_col" + j);
336  }
