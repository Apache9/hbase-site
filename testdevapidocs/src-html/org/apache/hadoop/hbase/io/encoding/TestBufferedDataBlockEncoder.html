017package org.apache.hadoop.hbase.io.encoding;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertTrue;
021
022import java.nio.ByteBuffer;
023
024import org.apache.hadoop.hbase.KeyValue.Type;
025import org.apache.hadoop.hbase.testclassification.MediumTests;
026import org.apache.hadoop.hbase.testclassification.IOTests;
027import org.apache.hadoop.hbase.util.Bytes;
028import org.apache.hadoop.hbase.util.ObjectIntPair;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031
032import org.apache.hadoop.hbase.KeyValue;
033
034@Category({IOTests.class, MediumTests.class})
035public class TestBufferedDataBlockEncoder {
036
037  byte[] row1 = Bytes.toBytes("row1");
038  byte[] row2 = Bytes.toBytes("row2");
039  byte[] row_1_0 = Bytes.toBytes("row10");
040
041  byte[] fam1 = Bytes.toBytes("fam1");
042  byte[] fam2 = Bytes.toBytes("fam2");
043  byte[] fam_1_2 = Bytes.toBytes("fam12");
044
045  byte[] qual1 = Bytes.toBytes("qual1");
046  byte[] qual2 = Bytes.toBytes("qual2");
047
048  byte[] val = Bytes.toBytes("val");
049
050  @Test
051  public void testEnsureSpaceForKey() {
052    BufferedDataBlockEncoder.SeekerState state =
053        new BufferedDataBlockEncoder.SeekerState(new ObjectIntPair<ByteBuffer>(), false);
054    for (int i = 1; i <= 65536; ++i) {
055      state.keyLength = i;
056      state.ensureSpaceForKey();
057      state.keyBuffer[state.keyLength - 1] = (byte) ((i - 1) % 0xff);
058      for (int j = 0; j < i - 1; ++j) {
059        // Check that earlier bytes were preserved as the buffer grew.
060        assertEquals((byte) (j % 0xff), state.keyBuffer[j]);
061      }
062    }
063  }
064
065  @Test
066  public void testCommonPrefixComparators() {
067    KeyValue kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
068    KeyValue kv2 = new KeyValue(row1, fam_1_2, qual1, 1l, Type.Maximum);
069    assertTrue((BufferedDataBlockEncoder.compareCommonFamilyPrefix(kv1, kv2, 4) < 0));
070
071    kv1 = new KeyValue(row1, fam1, qual1, 1l, Type.Put);
072    kv2 = new KeyValue(row_1_0, fam_1_2, qual1, 1l, Type.Maximum);
073    assertTrue((BufferedDataBlockEncoder.compareCommonRowPrefix(kv1, kv2, 4) < 0));
074
075    kv1 = new KeyValue(row1, fam1, qual2, 1l, Type.Put);
076    kv2 = new KeyValue(row1, fam1, qual1, 1l, Type.Maximum);
077    assertTrue((BufferedDataBlockEncoder.compareCommonQualifierPrefix(kv1, kv2, 4) > 0));
078  }
079
080}
