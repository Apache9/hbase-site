018package org.apache.hadoop.hbase;
019
020import java.io.IOException;
021import java.util.HashSet;
022import java.util.Set;
023import java.util.concurrent.ExecutorService;
024import java.util.concurrent.Executors;
025import java.util.concurrent.Future;
026import java.util.concurrent.atomic.AtomicInteger;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.conf.Configuration;
031import org.apache.hadoop.hbase.client.Connection;
032import org.apache.hadoop.hbase.client.ConnectionFactory;
033import org.apache.hadoop.hbase.client.Table;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.apache.hadoop.util.Tool;
036import org.apache.hadoop.util.ToolRunner;
037
038import com.codahale.metrics.MetricRegistry;
039import com.codahale.metrics.Snapshot;
040import com.codahale.metrics.Timer;
041
042
043/**
044 * Simple Increments Performance Test. Run this from main. It is to go against a cluster.
045 * Presumption is the table exists already. Defaults are a zk ensemble of localhost:2181,
046 * a tableName of 'tableName', a column famly name of 'columnFamilyName', with 80 threads by
047 * default and 10000 increments per thread. To change any of these configs, pass -DNAME=VALUE as
048 * in -DtableName="newTableName". It prints out configuration it is running with at the start and
049 * on the end it prints out percentiles.
050 */
051public class IncrementPerformanceTest implements Tool {
052  private static final Log LOG = LogFactory.getLog(IncrementPerformanceTest.class);
053  private static final byte [] QUALIFIER = new byte [] {'q'};
054  private Configuration conf;
055  private final MetricRegistry metrics = new MetricRegistry();
056  private static final String TABLENAME = "tableName";
057  private static final String COLUMN_FAMILY = "columnFamilyName";
058  private static final String THREAD_COUNT = "threadCount";
059  private static final int DEFAULT_THREAD_COUNT = 80;
060  private static final String INCREMENT_COUNT = "incrementCount";
061  private static final int DEFAULT_INCREMENT_COUNT = 10000;
062
063  IncrementPerformanceTest() {}
064
065  public int run(final String [] args) throws Exception {
066    Configuration conf = getConf();
067    final TableName tableName = TableName.valueOf(conf.get(TABLENAME), TABLENAME);
068    final byte [] columnFamilyName = Bytes.toBytes(conf.get(COLUMN_FAMILY, COLUMN_FAMILY));
069    int threadCount = conf.getInt(THREAD_COUNT, DEFAULT_THREAD_COUNT);
070    final int incrementCount = conf.getInt(INCREMENT_COUNT, DEFAULT_INCREMENT_COUNT);
071    LOG.info("Running test with " + HConstants.ZOOKEEPER_QUORUM + "=" +
072      getConf().get(HConstants.ZOOKEEPER_QUORUM) + ", tableName=" + tableName +
073      ", columnFamilyName=" + columnFamilyName + ", threadCount=" + threadCount +
074      ", incrementCount=" + incrementCount);
075
076    ExecutorService service = Executors.newFixedThreadPool(threadCount);
077    Set<Future<?>> futures = new HashSet<Future<?>>();
078    final AtomicInteger integer = new AtomicInteger(0); // needed a simple "final" counter
079    while (integer.incrementAndGet() <= threadCount) {
080      futures.add(service.submit(new Runnable() {
081        @Override
082        public void run() {
083          try {
084            try (Connection connection = ConnectionFactory.createConnection(getConf())) {
085              try (Table table = connection.getTable(tableName)) {
086                Timer timer = metrics.timer("increments");
087                for (int i = 0; i < incrementCount; i++) {
088                  byte[] row = Bytes.toBytes(i);
089                  Timer.Context context = timer.time();
090                  try {
091                    table.incrementColumnValue(row, columnFamilyName, QUALIFIER, 1l);
092                  } catch (IOException e) {
093                    // swallow..it's a test.
094                  } finally {
095                    context.stop();
096                  }
097                }
098              }
099            }
100          } catch (IOException ioe) {
101            throw new RuntimeException(ioe);
102          }
103        }
104      }));
105    }
106
107    for(Future<?> future : futures) future.get();
108    service.shutdown();
109    Snapshot s = metrics.timer("increments").getSnapshot();
110    LOG.info(String.format("75th=%s, 95th=%s, 99th=%s", s.get75thPercentile(),
111        s.get95thPercentile(), s.get99thPercentile()));
112    return 0;
113  }
114
115  @Override
116  public Configuration getConf() {
117    return this.conf;
118  }
119
120  @Override
121  public void setConf(Configuration conf) {
122    this.conf = conf;
123  }
124
125  public static void main(String[] args) throws Exception {
126    System.exit(ToolRunner.run(HBaseConfiguration.create(), new IncrementPerformanceTest(), args));
127  }
128}
