053/**
054 * Tests invocation of the {@link org.apache.hadoop.hbase.coprocessor.RegionServerObserver}
055 * interface hooks at all appropriate times during normal HMaster operations.
056 */
057@Category({CoprocessorTests.class, MediumTests.class})
058public class TestRegionServerObserver {
059  private static final Log LOG = LogFactory.getLog(TestRegionServerObserver.class);
060
061  /**
062   * Test verifies the hooks in regions merge.
063   * @throws Exception
064   */
065  @Ignore
066  @Test
067  public void testCoprocessorHooksInRegionsMerge() throws Exception {
068    final int NUM_MASTERS = 1;
069    final int NUM_RS = 1;
070    final String TABLENAME = "testRegionServerObserver";
071    final String TABLENAME2 = "testRegionServerObserver_2";
072    final byte[] FAM = Bytes.toBytes("fam");
073
074    // Create config to use for this cluster
075    Configuration conf = HBaseConfiguration.create();
076    conf.setClass("hbase.coprocessor.regionserver.classes", CPRegionServerObserver.class,
077      RegionServerObserver.class);
078
079    // Start the cluster
080    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
081    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
082    Admin admin = TEST_UTIL.getHBaseAdmin();
083    try {
084      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
085      HRegionServer regionServer = cluster.getRegionServer(0);
086      RegionServerCoprocessorHost cpHost = regionServer.getRegionServerCoprocessorHost();
087      Coprocessor coprocessor = cpHost.findCoprocessor(CPRegionServerObserver.class.getName());
088      CPRegionServerObserver regionServerObserver = (CPRegionServerObserver) coprocessor;
089      HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(TABLENAME));
090      desc.addFamily(new HColumnDescriptor(FAM));
091      admin.createTable(desc, new byte[][] { Bytes.toBytes("row") });
092      desc = new HTableDescriptor(TableName.valueOf(TABLENAME2));
093      desc.addFamily(new HColumnDescriptor(FAM));
094      admin.createTable(desc, new byte[][] { Bytes.toBytes("row") });
095      assertFalse(regionServerObserver.wasRegionMergeCalled());
096      List<Region> regions = regionServer.getOnlineRegions(TableName.valueOf(TABLENAME));
097      admin.mergeRegionsAsync(
098        regions.get(0).getRegionInfo().getEncodedNameAsBytes(),
099        regions.get(1).getRegionInfo().getEncodedNameAsBytes(),
100        true);
101      int regionsCount = regionServer.getOnlineRegions(TableName.valueOf(TABLENAME)).size();
102      while (regionsCount != 1) {
103        regionsCount = regionServer.getOnlineRegions(TableName.valueOf(TABLENAME)).size();
104        Thread.sleep(1000);
105      }
106      assertTrue(regionServerObserver.wasRegionMergeCalled());
107      assertTrue(regionServerObserver.wasPreMergeCommit());
108      assertTrue(regionServerObserver.wasPostMergeCommit());
109      assertEquals(regionsCount, 1);
110      assertEquals(regionServer.getOnlineRegions(TableName.valueOf(TABLENAME2)).size(), 1);
111    } finally {
112      if (admin != null) admin.close();
113      TEST_UTIL.shutdownMiniCluster();
114    }
115  }
116
117  public static class CPRegionServerObserver extends BaseRegionServerObserver {
118    private RegionMergeTransactionImpl rmt = null;
119    private HRegion mergedRegion = null;
120
121    private boolean preMergeCalled;
122    private boolean preMergeBeforePONRCalled;
123    private boolean preMergeAfterPONRCalled;
124    private boolean preRollBackMergeCalled;
125    private boolean postRollBackMergeCalled;
126    private boolean postMergeCalled;
127
128    public void resetStates() {
129      preMergeCalled = false;
130      preMergeBeforePONRCalled = false;
131      preMergeAfterPONRCalled = false;
132      preRollBackMergeCalled = false;
133      postRollBackMergeCalled = false;
134      postMergeCalled = false;
135    }
136
137    @Override
138    public void preMerge(ObserverContext<RegionServerCoprocessorEnvironment> ctx, Region regionA,
139        Region regionB) throws IOException {
140      preMergeCalled = true;
141    }
142
143    @Override
144    public void preMergeCommit(ObserverContext<RegionServerCoprocessorEnvironment> ctx,
145        Region regionA, Region regionB, List<Mutation> metaEntries) throws IOException {
146      preMergeBeforePONRCalled = true;
147      RegionServerCoprocessorEnvironment environment = ctx.getEnvironment();
148      HRegionServer rs = (HRegionServer) environment.getRegionServerServices();
149      List<Region> onlineRegions =
150          rs.getOnlineRegions(TableName.valueOf("testRegionServerObserver_2"));
151      rmt = (RegionMergeTransactionImpl) new RegionMergeTransactionFactory(rs.getConfiguration())
152        .create(onlineRegions.get(0), onlineRegions.get(1), true);
153      if (!rmt.prepare(rs)) {
154        LOG.error("Prepare for the region merge of table "
155            + onlineRegions.get(0).getTableDesc().getNameAsString()
156            + " failed. So returning null. ");
157        ctx.bypass();
158        return;
159      }
160      mergedRegion = rmt.stepsBeforePONR(rs, rs, false);
161      rmt.prepareMutationsForMerge(mergedRegion.getRegionInfo(), regionA.getRegionInfo(),
162        regionB.getRegionInfo(), rs.getServerName(), metaEntries);
163      MetaTableAccessor.mutateMetaTable(rs.getConnection(), metaEntries);
164    }
165
166    @Override
167    public void postMergeCommit(ObserverContext<RegionServerCoprocessorEnvironment> ctx,
168        Region regionA, Region regionB, Region mr) throws IOException {
169      preMergeAfterPONRCalled = true;
170      RegionServerCoprocessorEnvironment environment = ctx.getEnvironment();
171      HRegionServer rs = (HRegionServer) environment.getRegionServerServices();
172      rmt.stepsAfterPONR(rs, rs, this.mergedRegion, null);
173    }
174
175    @Override
176    public void preRollBackMerge(ObserverContext<RegionServerCoprocessorEnvironment> ctx,
177        Region regionA, Region regionB) throws IOException {
178      preRollBackMergeCalled = true;
179    }
180
181    @Override
182    public void postRollBackMerge(ObserverContext<RegionServerCoprocessorEnvironment> ctx,
183        Region regionA, Region regionB) throws IOException {
184      postRollBackMergeCalled = true;
185    }
186
187    @Override
188    public void postMerge(ObserverContext<RegionServerCoprocessorEnvironment> c, Region regionA,
189        Region regionB, Region mergedRegion) throws IOException {
190      postMergeCalled = true;
191    }
192
193    public boolean wasPreMergeCalled() {
194      return this.preMergeCalled;
195    }
196
197    public boolean wasPostMergeCalled() {
198      return this.postMergeCalled;
199    }
200
201    public boolean wasPreMergeCommit() {
202      return this.preMergeBeforePONRCalled;
203    }
204
205    public boolean wasPostMergeCommit() {
206      return this.preMergeAfterPONRCalled;
207    }
208
209    public boolean wasPreRollBackMerge() {
210      return this.preRollBackMergeCalled;
211    }
212
213    public boolean wasPostRollBackMerge() {
214      return this.postRollBackMergeCalled;
215    }
216
217    public boolean wasRegionMergeCalled() {
218      return this.preMergeCalled && this.postMergeCalled;
219    }
220
221  }
