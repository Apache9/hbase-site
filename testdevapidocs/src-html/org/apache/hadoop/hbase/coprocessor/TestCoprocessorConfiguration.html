020package org.apache.hadoop.hbase.coprocessor;
021
022import java.io.IOException;
023import java.util.concurrent.atomic.AtomicBoolean;
024
025import static org.mockito.Mockito.*;
026import static org.junit.Assert.*;
027
028import org.apache.hadoop.conf.Configuration;
029import org.apache.hadoop.hbase.Coprocessor;
030import org.apache.hadoop.hbase.CoprocessorEnvironment;
031import org.apache.hadoop.hbase.HBaseConfiguration;
032import org.apache.hadoop.hbase.HRegionInfo;
033import org.apache.hadoop.hbase.HTableDescriptor;
034import org.apache.hadoop.hbase.TableName;
035import org.apache.hadoop.hbase.master.MasterCoprocessorHost;
036import org.apache.hadoop.hbase.master.MasterServices;
037import org.apache.hadoop.hbase.regionserver.HRegion;
038import org.apache.hadoop.hbase.regionserver.RegionCoprocessorHost;
039import org.apache.hadoop.hbase.regionserver.RegionServerCoprocessorHost;
040import org.apache.hadoop.hbase.regionserver.RegionServerServices;
041import org.apache.hadoop.hbase.testclassification.CoprocessorTests;
042import org.apache.hadoop.hbase.testclassification.SmallTests;
043import org.junit.Test;
044import org.junit.experimental.categories.Category;
045
046/**
047 * Tests for global coprocessor loading configuration
048 */
049@Category({CoprocessorTests.class, SmallTests.class})
050public class TestCoprocessorConfiguration {
051
052  private static final Configuration CONF = HBaseConfiguration.create();
053  static {
054    CONF.setStrings(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
055      SystemCoprocessor.class.getName());
056    CONF.setStrings(CoprocessorHost.REGIONSERVER_COPROCESSOR_CONF_KEY,
057      SystemCoprocessor.class.getName());
058    CONF.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
059      SystemCoprocessor.class.getName());
060  }
061  private static final TableName TABLENAME = TableName.valueOf("TestCoprocessorConfiguration");
062  private static final HRegionInfo REGIONINFO = new HRegionInfo(TABLENAME);
063  private static final HTableDescriptor TABLEDESC = new HTableDescriptor(TABLENAME);
064  static {
065    try {
066      TABLEDESC.addCoprocessor(TableCoprocessor.class.getName());
067    } catch (IOException e) {
068      throw new RuntimeException(e);
069    }
070  }
071
072  // use atomic types in case coprocessor loading is ever multithreaded, also
073  // so we can mutate them even though they are declared final here
074  private static final AtomicBoolean systemCoprocessorLoaded = new AtomicBoolean();
075  private static final AtomicBoolean tableCoprocessorLoaded = new AtomicBoolean();
076
077  public static class SystemCoprocessor implements Coprocessor {
078    @Override
079    public void start(CoprocessorEnvironment env) throws IOException {
080      systemCoprocessorLoaded.set(true);
081    }
082
083    @Override
084    public void stop(CoprocessorEnvironment env) throws IOException { }
085  }
086
087  public static class TableCoprocessor implements Coprocessor {
088    @Override
089    public void start(CoprocessorEnvironment env) throws IOException {
090      tableCoprocessorLoaded.set(true);
091    }
092
093    @Override
094    public void stop(CoprocessorEnvironment env) throws IOException { }
095  }
096
097  @Test
098  public void testRegionCoprocessorHostDefaults() throws Exception {
099    Configuration conf = new Configuration(CONF);
100    HRegion region = mock(HRegion.class);
101    when(region.getRegionInfo()).thenReturn(REGIONINFO);
102    when(region.getTableDesc()).thenReturn(TABLEDESC);
103    RegionServerServices rsServices = mock(RegionServerServices.class);
104    systemCoprocessorLoaded.set(false);
105    tableCoprocessorLoaded.set(false);
106    new RegionCoprocessorHost(region, rsServices, conf);
107    assertEquals("System coprocessors loading default was not honored",
108      systemCoprocessorLoaded.get(),
109      CoprocessorHost.DEFAULT_COPROCESSORS_ENABLED);
110    assertEquals("Table coprocessors loading default was not honored",
111      tableCoprocessorLoaded.get(), 
112      CoprocessorHost.DEFAULT_COPROCESSORS_ENABLED &&
113      CoprocessorHost.DEFAULT_USER_COPROCESSORS_ENABLED);
114  }
115
116  @Test
117  public void testRegionServerCoprocessorHostDefaults() throws Exception {
118    Configuration conf = new Configuration(CONF);
119    RegionServerServices rsServices = mock(RegionServerServices.class);
120    systemCoprocessorLoaded.set(false);
121    new RegionServerCoprocessorHost(rsServices, conf);
122    assertEquals("System coprocessors loading default was not honored",
123      systemCoprocessorLoaded.get(),
124      CoprocessorHost.DEFAULT_COPROCESSORS_ENABLED);
125  }
126
127  @Test
128  public void testMasterCoprocessorHostDefaults() throws Exception {
129    Configuration conf = new Configuration(CONF);
130    MasterServices masterServices = mock(MasterServices.class);
131    systemCoprocessorLoaded.set(false);
132    new MasterCoprocessorHost(masterServices, conf);
133    assertEquals("System coprocessors loading default was not honored",
134      systemCoprocessorLoaded.get(),
135      CoprocessorHost.DEFAULT_COPROCESSORS_ENABLED);
136  }
137
138  @Test
139  public void testRegionCoprocessorHostAllDisabled() throws Exception {
140    Configuration conf = new Configuration(CONF);
141    conf.setBoolean(CoprocessorHost.COPROCESSORS_ENABLED_CONF_KEY, false);
142    HRegion region = mock(HRegion.class);
143    when(region.getRegionInfo()).thenReturn(REGIONINFO);
144    when(region.getTableDesc()).thenReturn(TABLEDESC);
145    RegionServerServices rsServices = mock(RegionServerServices.class);
146    systemCoprocessorLoaded.set(false);
147    tableCoprocessorLoaded.set(false);
148    new RegionCoprocessorHost(region, rsServices, conf);
149    assertFalse("System coprocessors should not have been loaded",
150      systemCoprocessorLoaded.get());
151    assertFalse("Table coprocessors should not have been loaded",
152      tableCoprocessorLoaded.get());
153  }
154
155  @Test
156  public void testRegionCoprocessorHostTableLoadingDisabled() throws Exception {
157    Configuration conf = new Configuration(CONF);
158    conf.setBoolean(CoprocessorHost.COPROCESSORS_ENABLED_CONF_KEY, true); // if defaults change
159    conf.setBoolean(CoprocessorHost.USER_COPROCESSORS_ENABLED_CONF_KEY, false);
160    HRegion region = mock(HRegion.class);
161    when(region.getRegionInfo()).thenReturn(REGIONINFO);
162    when(region.getTableDesc()).thenReturn(TABLEDESC);
163    RegionServerServices rsServices = mock(RegionServerServices.class);
164    systemCoprocessorLoaded.set(false);
165    tableCoprocessorLoaded.set(false);
166    new RegionCoprocessorHost(region, rsServices, conf);
167    assertTrue("System coprocessors should have been loaded",
168      systemCoprocessorLoaded.get());
169    assertFalse("Table coprocessors should not have been loaded",
170      tableCoprocessorLoaded.get());
171  }
172}
