134  @Test
135  public void testHTableInterfaceMethods() throws Exception {
136    Configuration conf = util.getConfiguration();
137    MasterCoprocessorHost cpHost = util.getMiniHBaseCluster().getMaster().getMasterCoprocessorHost();
138    Class<?> implClazz = DummyRegionObserver.class;
139    cpHost.load(implClazz, Coprocessor.PRIORITY_HIGHEST, conf);
140    CoprocessorEnvironment env = cpHost.findCoprocessorEnvironment(implClazz.getName());
141    assertEquals(Coprocessor.VERSION, env.getVersion());
142    assertEquals(VersionInfo.getVersion(), env.getHBaseVersion());
143    hTableInterface = env.getTable(TEST_TABLE);
144    checkHTableInterfaceMethods();
145    cpHost.shutdown(env);
146  }

185  private void checkExists() throws IOException {
186    boolean ex = hTableInterface.exists(new Get(ROW_A).addColumn(TEST_FAMILY, qualifierCol1));
187    assertTrue(ex);
188
189    boolean[] exArray = hTableInterface.existsAll(Arrays.asList(new Get[]{
190      new Get(ROW_A).addColumn(TEST_FAMILY, qualifierCol1),
191      new Get(ROW_B).addColumn(TEST_FAMILY, qualifierCol1),
192      new Get(ROW_C).addColumn(TEST_FAMILY, qualifierCol1),
193      new Get(Bytes.toBytes("does not exist")).addColumn(TEST_FAMILY, qualifierCol1),}));
194    assertTrue(Arrays.equals(new boolean[]{true, true, true, false}, exArray));
195  }

197  private void checkAppend() throws IOException {
198    final byte[] appendValue = Bytes.toBytes("append");
199    Append append = new Append(qualifierCol1).addColumn(TEST_FAMILY, qualifierCol1, appendValue);
200    Result appendResult = hTableInterface.append(append);
201    byte[] appendedRow = appendResult.getRow();
202    checkRowValue(appendedRow, appendValue);
203  }

205  private void checkPutsAndDeletes() throws IOException {
206    // put:
207    Put putD = new Put(ROW_D).addColumn(TEST_FAMILY, qualifierCol1, bytes2);
208    hTableInterface.put(putD);
209    checkRowValue(ROW_D, bytes2);
210
211    // delete:
212    Delete delete = new Delete(ROW_D);
213    hTableInterface.delete(delete);
214    checkRowValue(ROW_D, null);
215
216    // multiple puts:
217    Put[] puts = new Put[] {new Put(ROW_D).addColumn(TEST_FAMILY, qualifierCol1, bytes2),
218            new Put(ROW_E).addColumn(TEST_FAMILY, qualifierCol1, bytes3)};
219    hTableInterface.put(Arrays.asList(puts));
220    checkRowsValues(new byte[][] { ROW_D, ROW_E }, new byte[][] { bytes2, bytes3 });
221
222    // multiple deletes:
223    Delete[] deletes = new Delete[] { new Delete(ROW_D), new Delete(ROW_E) };
224    hTableInterface.delete(new ArrayList<>(Arrays.asList(deletes)));
225    checkRowsValues(new byte[][] { ROW_D, ROW_E }, new byte[][] { null, null });
226  }

228  private void checkCheckAndPut() throws IOException {
229    Put putC = new Put(ROW_C).addColumn(TEST_FAMILY, qualifierCol1, bytes5);
230    assertFalse(hTableInterface.checkAndPut(ROW_C, TEST_FAMILY, qualifierCol1, /* expect */bytes4,
231      putC/* newValue */));
232    assertTrue(hTableInterface.checkAndPut(ROW_C, TEST_FAMILY, qualifierCol1, /* expect */bytes3,
233      putC/* newValue */));
234    checkRowValue(ROW_C, bytes5);
235  }

238    Delete delete = new Delete(ROW_C);
239    assertFalse(hTableInterface.checkAndDelete(ROW_C, TEST_FAMILY, qualifierCol1, bytes4, delete));
240    assertTrue(hTableInterface.checkAndDelete(ROW_C, TEST_FAMILY, qualifierCol1, bytes5, delete));
241    checkRowValue(ROW_C, null);
242  }

244  private void checkIncrementColumnValue() throws IOException {
245    hTableInterface.put(new Put(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, Bytes.toBytes(1L)));
246    checkRowValue(ROW_A, Bytes.toBytes(1L));
247
248    final long newVal = hTableInterface
249        .incrementColumnValue(ROW_A, TEST_FAMILY, qualifierCol1, 10L);
250    assertEquals(11L, newVal);
251    checkRowValue(ROW_A, Bytes.toBytes(11L));
252
253    final long newVal2 = hTableInterface.incrementColumnValue(ROW_A, TEST_FAMILY, qualifierCol1,
254        -10L, Durability.SYNC_WAL);
255    assertEquals(1L, newVal2);
256    checkRowValue(ROW_A, Bytes.toBytes(1L));
257  }

261    checkRowValue(ROW_A, Bytes.toBytes(-4L));
262  }

264  private void checkBatch() throws IOException, InterruptedException {
265    List<Row> actions =
266        Arrays.asList(new Row[] { new Increment(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, 2L),
267            new Increment(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, 2L) });
268    Object[] results3 = new Object[actions.size()];
269    Object[] results1 = results3;
270    hTableInterface.batch(actions, results1);
271    assertEquals(2, results1.length);
272    for (Object r2 : results1) {
273      assertTrue(r2 instanceof Result);
274    }
275    checkRowValue(ROW_A, Bytes.toBytes(0L));
276    Object[] results2 = new Result[2];
277    hTableInterface.batch(
278        actions, results2);
279    for (Object r2 : results2) {
280      assertTrue(r2 instanceof Result);
281    }
282    checkRowValue(ROW_A, Bytes.toBytes(4L));
283
284    // with callbacks:
285    final long[] updateCounter = new long[] { 0L };
286    hTableInterface.batchCallback(actions, results3, new Batch.Callback<Result>() {
287      @Override
288      public void update(byte[] region, byte[] row, Result result) {
289        updateCounter[0]++;
290      }
291    });
292    assertEquals(2, updateCounter[0]);
293    assertEquals(2, results3.length);
294    for (Object r3 : results3) {
295      assertTrue(r3 instanceof Result);
296    }
297    checkRowValue(ROW_A, Bytes.toBytes(8L));
298
299    Object[] results4 = new Result[2];
300    updateCounter[0] = 0L;
301    hTableInterface.batchCallback(
302        actions, results4,
303        new Batch.Callback<Result>() {
304          @Override
305          public void update(byte[] region, byte[] row, Result result) {
306            updateCounter[0]++;
307          }
308        });
309    assertEquals(2, updateCounter[0]);
310    for (Object r2 : results4) {
311      assertTrue(r2 instanceof Result);
312    }
313    checkRowValue(ROW_A, Bytes.toBytes(12L));
314  }

329  private void checkResultScanner() throws IOException {
330    ResultScanner resultScanner = hTableInterface.getScanner(TEST_FAMILY);
331    Result[] results = resultScanner.next(10);
332    assertEquals(3, results.length);
333
334    resultScanner = hTableInterface.getScanner(TEST_FAMILY, qualifierCol1);
335    results = resultScanner.next(10);
336    assertEquals(3, results.length);
337
338    resultScanner = hTableInterface.getScanner(new Scan(ROW_A, ROW_C));
339    results = resultScanner.next(10);
340    assertEquals(2, results.length);
341  }

343  private void checkRowValue(byte[] row, byte[] expectedValue) throws IOException {
344    Get get = new Get(row).addColumn(TEST_FAMILY, qualifierCol1);
345    Result result = hTableInterface.get(get);
346    byte[] actualValue = result.getValue(TEST_FAMILY, qualifierCol1);
347    assertArrayEquals(expectedValue, actualValue);
348  }
349
350  private void checkRowsValues(byte[][] rows, byte[][] expectedValues) throws IOException {
351    if (rows.length != expectedValues.length) {
352      throw new IllegalArgumentException();
353    }
354    Get[] gets = new Get[rows.length];
355    for (int i = 0; i < gets.length; i++) {
356      gets[i] = new Get(rows[i]).addColumn(TEST_FAMILY, qualifierCol1);
357    }
358    Result[] results = hTableInterface.get(Arrays.asList(gets));
359    for (int i = 0; i < expectedValues.length; i++) {
360      byte[] actualValue = results[i].getValue(TEST_FAMILY, qualifierCol1);
361      assertArrayEquals(expectedValues[i], actualValue);
362    }
363  }
