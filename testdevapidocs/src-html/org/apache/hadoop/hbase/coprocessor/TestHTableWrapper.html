139  @Test
140  public void testHTableInterfaceMethods() throws Exception {
141    Configuration conf = util.getConfiguration();
142    MasterCoprocessorHost cpHost = util.getMiniHBaseCluster().getMaster().getMasterCoprocessorHost();
143    Class<? extends MasterCoprocessor> implClazz = DummyRegionObserver.class;
144    cpHost.load(implClazz, Coprocessor.PRIORITY_HIGHEST, conf);
145    CoprocessorEnvironment env = cpHost.findCoprocessorEnvironment(implClazz.getName());
146    assertEquals(Coprocessor.VERSION, env.getVersion());
147    assertEquals(VersionInfo.getVersion(), env.getHBaseVersion());
148    hTableInterface = env.getTable(TEST_TABLE);
149    checkHTableInterfaceMethods();
150    cpHost.shutdown((MasterCoprocessorEnvironment) env);
151  }

182  private void checkExists() throws IOException {
183    boolean ex = hTableInterface.exists(new Get(ROW_A).addColumn(TEST_FAMILY, qualifierCol1));
184    assertTrue(ex);
185
186    boolean[] exArray = hTableInterface.existsAll(Arrays.asList(new Get[]{
187      new Get(ROW_A).addColumn(TEST_FAMILY, qualifierCol1),
188      new Get(ROW_B).addColumn(TEST_FAMILY, qualifierCol1),
189      new Get(ROW_C).addColumn(TEST_FAMILY, qualifierCol1),
190      new Get(Bytes.toBytes("does not exist")).addColumn(TEST_FAMILY, qualifierCol1),}));
191    assertTrue(Arrays.equals(new boolean[]{true, true, true, false}, exArray));
192  }

194  private void checkAppend() throws IOException {
195    final byte[] appendValue = Bytes.toBytes("append");
196    Append append = new Append(qualifierCol1).addColumn(TEST_FAMILY, qualifierCol1, appendValue);
197    Result appendResult = hTableInterface.append(append);
198    byte[] appendedRow = appendResult.getRow();
199    checkRowValue(appendedRow, appendValue);
200  }

202  private void checkPutsAndDeletes() throws IOException {
203    // put:
204    Put putD = new Put(ROW_D).addColumn(TEST_FAMILY, qualifierCol1, bytes2);
205    hTableInterface.put(putD);
206    checkRowValue(ROW_D, bytes2);
207
208    // delete:
209    Delete delete = new Delete(ROW_D);
210    hTableInterface.delete(delete);
211    checkRowValue(ROW_D, null);
212
213    // multiple puts:
214    Put[] puts = new Put[] {new Put(ROW_D).addColumn(TEST_FAMILY, qualifierCol1, bytes2),
215            new Put(ROW_E).addColumn(TEST_FAMILY, qualifierCol1, bytes3)};
216    hTableInterface.put(Arrays.asList(puts));
217    checkRowsValues(new byte[][] { ROW_D, ROW_E }, new byte[][] { bytes2, bytes3 });
218
219    // multiple deletes:
220    Delete[] deletes = new Delete[] { new Delete(ROW_D), new Delete(ROW_E) };
221    hTableInterface.delete(new ArrayList<>(Arrays.asList(deletes)));
222    checkRowsValues(new byte[][] { ROW_D, ROW_E }, new byte[][] { null, null });
223  }

225  private void checkCheckAndPut() throws IOException {
226    Put putC = new Put(ROW_C).addColumn(TEST_FAMILY, qualifierCol1, bytes5);
227    assertFalse(hTableInterface.checkAndPut(ROW_C, TEST_FAMILY, qualifierCol1, /* expect */bytes4,
228      putC/* newValue */));
229    assertTrue(hTableInterface.checkAndPut(ROW_C, TEST_FAMILY, qualifierCol1, /* expect */bytes3,
230      putC/* newValue */));
231    checkRowValue(ROW_C, bytes5);
232  }

234  private void checkCheckAndDelete() throws IOException {
235    Delete delete = new Delete(ROW_C);
236    assertFalse(hTableInterface.checkAndDelete(ROW_C, TEST_FAMILY, qualifierCol1, bytes4, delete));
237    assertTrue(hTableInterface.checkAndDelete(ROW_C, TEST_FAMILY, qualifierCol1, bytes5, delete));
238    checkRowValue(ROW_C, null);
239  }

241  private void checkIncrementColumnValue() throws IOException {
242    hTableInterface.put(new Put(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, Bytes.toBytes(1L)));
243    checkRowValue(ROW_A, Bytes.toBytes(1L));
244
245    final long newVal = hTableInterface
246        .incrementColumnValue(ROW_A, TEST_FAMILY, qualifierCol1, 10L);
247    assertEquals(11L, newVal);
248    checkRowValue(ROW_A, Bytes.toBytes(11L));
249
250    final long newVal2 = hTableInterface.incrementColumnValue(ROW_A, TEST_FAMILY, qualifierCol1,
251        -10L, Durability.SYNC_WAL);
252    assertEquals(1L, newVal2);
253    checkRowValue(ROW_A, Bytes.toBytes(1L));
254  }

258    checkRowValue(ROW_A, Bytes.toBytes(-4L));

261  private void checkBatch() throws IOException, InterruptedException {
262    List<Row> actions =
263        Arrays.asList(new Row[] { new Increment(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, 2L),
264            new Increment(ROW_A).addColumn(TEST_FAMILY, qualifierCol1, 2L) });
265    Object[] results3 = new Object[actions.size()];
266    Object[] results1 = results3;
267    hTableInterface.batch(actions, results1);
268    assertEquals(2, results1.length);
269    for (Object r2 : results1) {
270      assertTrue(r2 instanceof Result);
271    }
272    checkRowValue(ROW_A, Bytes.toBytes(0L));
273    Object[] results2 = new Result[2];
274    hTableInterface.batch(
275        actions, results2);
276    for (Object r2 : results2) {
277      assertTrue(r2 instanceof Result);
278    }
279    checkRowValue(ROW_A, Bytes.toBytes(4L));
280
281    // with callbacks:
282    final long[] updateCounter = new long[] { 0L };
283    hTableInterface.batchCallback(actions, results3, new Batch.Callback<Result>() {
284      @Override
285      public void update(byte[] region, byte[] row, Result result) {
286        updateCounter[0]++;
287      }
288    });
289    assertEquals(2, updateCounter[0]);
290    assertEquals(2, results3.length);
291    for (Object r3 : results3) {
292      assertTrue(r3 instanceof Result);
293    }
294    checkRowValue(ROW_A, Bytes.toBytes(8L));
295
296    Object[] results4 = new Result[2];
297    updateCounter[0] = 0L;
298    hTableInterface.batchCallback(
299        actions, results4,
300        new Batch.Callback<Result>() {
301          @Override
302          public void update(byte[] region, byte[] row, Result result) {
303            updateCounter[0]++;
304          }
305        });
306    assertEquals(2, updateCounter[0]);
307    for (Object r2 : results4) {
308      assertTrue(r2 instanceof Result);
309    }
310    checkRowValue(ROW_A, Bytes.toBytes(12L));
311  }

326  private void checkResultScanner() throws IOException {
327    ResultScanner resultScanner = hTableInterface.getScanner(TEST_FAMILY);
328    Result[] results = resultScanner.next(10);
329    assertEquals(3, results.length);
330
331    resultScanner = hTableInterface.getScanner(TEST_FAMILY, qualifierCol1);
332    results = resultScanner.next(10);
333    assertEquals(3, results.length);
334
335    resultScanner = hTableInterface.getScanner(new Scan(ROW_A, ROW_C));
336    results = resultScanner.next(10);
337    assertEquals(2, results.length);
338  }

340  private void checkRowValue(byte[] row, byte[] expectedValue) throws IOException {
341    Get get = new Get(row).addColumn(TEST_FAMILY, qualifierCol1);
342    Result result = hTableInterface.get(get);
343    byte[] actualValue = result.getValue(TEST_FAMILY, qualifierCol1);
344    assertArrayEquals(expectedValue, actualValue);
345  }
346
347  private void checkRowsValues(byte[][] rows, byte[][] expectedValues) throws IOException {
348    if (rows.length != expectedValues.length) {
349      throw new IllegalArgumentException();
350    }
351    Get[] gets = new Get[rows.length];
352    for (int i = 0; i < gets.length; i++) {
353      gets[i] = new Get(rows[i]).addColumn(TEST_FAMILY, qualifierCol1);
354    }
355    Result[] results = hTableInterface.get(Arrays.asList(gets));
356    for (int i = 0; i < expectedValues.length; i++) {
357      byte[] actualValue = results[i].getValue(TEST_FAMILY, qualifierCol1);
358      assertArrayEquals(expectedValues[i], actualValue);
359    }
360  }
