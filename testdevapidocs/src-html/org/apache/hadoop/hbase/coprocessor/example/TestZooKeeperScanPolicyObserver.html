021import static org.junit.Assert.assertEquals;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.TableName;

031import org.apache.hadoop.hbase.client.Put;

033import org.apache.hadoop.hbase.client.Table;
034import org.apache.hadoop.hbase.coprocessor.CoprocessorHost;
035import org.apache.hadoop.hbase.testclassification.CoprocessorTests;
036import org.apache.hadoop.hbase.testclassification.MediumTests;
037import org.apache.hadoop.hbase.util.Bytes;
038import org.apache.hadoop.hbase.util.EnvironmentEdgeManager;
039import org.apache.hadoop.hbase.zookeeper.ZKUtil;
040import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
041import org.apache.zookeeper.ZooKeeper;
042import org.junit.experimental.categories.Category;
043
044@Category({CoprocessorTests.class, MediumTests.class})
045public class TestZooKeeperScanPolicyObserver {
046  private static final Log LOG = LogFactory.getLog(TestZooKeeperScanPolicyObserver.class);
047  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
048  private static final byte[] F = Bytes.toBytes("fam");
049  private static final byte[] Q = Bytes.toBytes("qual");
050  private static final byte[] R = Bytes.toBytes("row");
051
052  // @BeforeClass
053  public static void setUpBeforeClass() throws Exception {
054    System.out.println("HERE!!!!!!!!");
055    // Test we can first start the ZK cluster by itself
056    Configuration conf = TEST_UTIL.getConfiguration();
057    conf.setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
058        ZooKeeperScanPolicyObserver.class.getName());
059    TEST_UTIL.startMiniZKCluster();
060    TEST_UTIL.startMiniCluster();
061  }
062
063  // @AfterClass
064  public static void tearDownAfterClass() throws Exception {
065    TEST_UTIL.shutdownMiniCluster();
066  }
067
068  // @Ignore @Test
069  public void testScanPolicyObserver() throws Exception {
070    TableName tableName =
071        TableName.valueOf("testScanPolicyObserver");
072    HTableDescriptor desc = new HTableDescriptor(tableName);
073    HColumnDescriptor hcd = new HColumnDescriptor(F)
074    .setMaxVersions(10)
075    .setTimeToLive(1);
076    desc.addFamily(hcd);
077    TEST_UTIL.getHBaseAdmin().createTable(desc);
078    Table t = TEST_UTIL.getConnection().getTable(tableName);
079    long now = EnvironmentEdgeManager.currentTime();
080
081    ZooKeeperWatcher zkw = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(), "test", null);
082    ZooKeeper zk = zkw.getRecoverableZooKeeper().getZooKeeper();
083    ZKUtil.createWithParents(zkw, ZooKeeperScanPolicyObserver.node);
084    // let's say test last backup was 1h ago
085    // using plain ZK here, because RecoverableZooKeeper add extra encoding to the data
086    zk.setData(ZooKeeperScanPolicyObserver.node, Bytes.toBytes(now - 3600*1000), -1);
087
088    LOG.debug("Set time: "+Bytes.toLong(Bytes.toBytes(now - 3600*1000)));
089
090    // sleep for 1s to give the ZK change a chance to reach the watcher in the observer.
091    // TODO: Better to wait for the data to be propagated
092    Thread.sleep(1000);
093
094    long ts = now - 2000;
095    Put p = new Put(R);
096    p.addColumn(F, Q, ts, Q);
097    t.put(p);
098    p = new Put(R);
099    p.addColumn(F, Q, ts + 1, Q);
100    t.put(p);
101
102    // these two should be expired but for the override
103    // (their ts was 2s in the past)
104    Get g = new Get(R);
105    g.setMaxVersions(10);
106    Result r = t.get(g);
107    // still there?
108    assertEquals(2, r.size());
109
110    TEST_UTIL.flush(tableName);
111    TEST_UTIL.compact(tableName, true);
112
113    g = new Get(R);
114    g.setMaxVersions(10);
115    r = t.get(g);
116    // still there?
117    assertEquals(2, r.size());
118    zk.setData(ZooKeeperScanPolicyObserver.node, Bytes.toBytes(now), -1);
119    LOG.debug("Set time: "+now);
120
121    TEST_UTIL.compact(tableName, true);
122
123    g = new Get(R);
124    g.setMaxVersions(10);
125    r = t.get(g);
126    // should be gone now
127    assertEquals(0, r.size());
128    t.close();
129  }
130}
