086  private static void setHdfsSecuredConfiguration(Configuration conf) throws Exception {
087    // change XXX_USER_NAME_KEY to XXX_KERBEROS_PRINCIPAL_KEY after we drop support for hadoop-2.4.1
088    conf.set(DFSConfigKeys.DFS_NAMENODE_USER_NAME_KEY, PRINCIPAL + "@" + KDC.getRealm());
089    conf.set(DFSConfigKeys.DFS_NAMENODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
090    conf.set(DFSConfigKeys.DFS_DATANODE_USER_NAME_KEY, PRINCIPAL + "@" + KDC.getRealm());
091    conf.set(DFSConfigKeys.DFS_DATANODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
092    conf.set(DFSConfigKeys.DFS_WEB_AUTHENTICATION_KERBEROS_PRINCIPAL_KEY, HTTP_PRINCIPAL + "@"
093        + KDC.getRealm());
094    conf.setBoolean(DFSConfigKeys.DFS_BLOCK_ACCESS_TOKEN_ENABLE_KEY, true);
095    conf.set(DFSConfigKeys.DFS_HTTP_POLICY_KEY, HttpConfig.Policy.HTTPS_ONLY.name());
096    conf.set(DFSConfigKeys.DFS_NAMENODE_HTTPS_ADDRESS_KEY, "localhost:0");
097    conf.set(DFSConfigKeys.DFS_DATANODE_HTTPS_ADDRESS_KEY, "localhost:0");
098
099    File keystoresDir = new File(TEST_UTIL.getDataTestDir("keystore").toUri().getPath());
100    keystoresDir.mkdirs();
101    String sslConfDir = KeyStoreTestUtil.getClasspathDir(TestGenerateDelegationToken.class);
102    KeyStoreTestUtil.setupSSLConfig(keystoresDir.getAbsolutePath(), sslConfDir, conf, false);
103
104    conf.setBoolean("ignore.secure.ports.for.testing", true);
105  }

107  @BeforeClass
108  public static void setUp() throws Exception {
109    Properties conf = MiniKdc.createConf();
110    conf.put(MiniKdc.DEBUG, true);
111    KDC = new MiniKdc(conf, new File(TEST_UTIL.getDataTestDir("kdc").toUri().getPath()));
112    KDC.start();
113    USERNAME = UserGroupInformation.getLoginUser().getShortUserName();
114    PRINCIPAL = USERNAME + "/" + HOST;
115    HTTP_PRINCIPAL = "HTTP/" + HOST;
116    KDC.createPrincipal(KEYTAB_FILE, PRINCIPAL, HTTP_PRINCIPAL);
117    TEST_UTIL.startMiniZKCluster();
118
119    HBaseKerberosUtils.setKeytabFileForTesting(KEYTAB_FILE.getAbsolutePath());
120    HBaseKerberosUtils.setPrincipalForTesting(PRINCIPAL + "@" + KDC.getRealm());
121    HBaseKerberosUtils.setSecuredConfiguration(TEST_UTIL.getConfiguration());
122    setHdfsSecuredConfiguration(TEST_UTIL.getConfiguration());
123    UserGroupInformation.setConfiguration(TEST_UTIL.getConfiguration());
124    TEST_UTIL.getConfiguration().setStrings(CoprocessorHost.REGION_COPROCESSOR_CONF_KEY,
125      TokenProvider.class.getName());
126    TEST_UTIL.startMiniDFSCluster(1);
127    Path rootdir = TEST_UTIL.getDataTestDirOnTestFS("TestGenerateDelegationToken");
128    FSUtils.setRootDir(TEST_UTIL.getConfiguration(), rootdir);
129    CLUSTER = new LocalHBaseCluster(TEST_UTIL.getConfiguration(), 1);
130    CLUSTER.startup();
131  }

145  private void testTokenAuth(Class<? extends RpcClient> rpcImplClass) throws IOException,
146      ServiceException {
147    TEST_UTIL.getConfiguration().set(RpcClientFactory.CUSTOM_RPC_CLIENT_IMPL_CONF_KEY,
148      rpcImplClass.getName());
149    try (Connection conn = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration());
150        Table table = conn.getTable(TableName.META_TABLE_NAME)) {
151      CoprocessorRpcChannel rpcChannel = table.coprocessorService(HConstants.EMPTY_START_ROW);
152      AuthenticationProtos.AuthenticationService.BlockingInterface service =
153          AuthenticationProtos.AuthenticationService.newBlockingStub(rpcChannel);
154      WhoAmIResponse response = service.whoAmI(null, WhoAmIRequest.getDefaultInstance());
155      assertEquals(USERNAME, response.getUsername());
156      assertEquals(AuthenticationMethod.TOKEN.name(), response.getAuthMethod());
157      try {
158        service.getAuthenticationToken(null, GetAuthenticationTokenRequest.getDefaultInstance());
159      } catch (ServiceException e) {
160        AccessDeniedException exc = (AccessDeniedException) ProtobufUtil.getRemoteException(e);
161        assertTrue(exc.getMessage().contains(
162          "Token generation only allowed for Kerberos authenticated clients"));
163      }
164    }
165  }
166
167  @Test
168  public void test() throws Exception {
169    try (Connection conn = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration())) {
170      Token<? extends TokenIdentifier> token = TokenUtil.obtainToken(conn);
171      UserGroupInformation.getCurrentUser().addToken(token);
172      testTokenAuth(RpcClientImpl.class);
173      testTokenAuth(AsyncRpcClient.class);
174    }
175
176  }
