119  private void testRpcCallWithEnabledKerberosSaslAuth(Class<? extends RpcClient> rpcImplClass)
120      throws Exception {
121    String krbKeytab = getKeytabFileForTesting();
122    String krbPrincipal = getPrincipalForTesting();
123
124    UserGroupInformation ugi = loginKerberosPrincipal(krbKeytab, krbPrincipal);
125    UserGroupInformation ugi2 = UserGroupInformation.getCurrentUser();
126
127    // check that the login user is okay:
128    assertSame(ugi, ugi2);
129    assertEquals(AuthenticationMethod.KERBEROS, ugi.getAuthenticationMethod());
130    assertEquals(krbPrincipal, ugi.getUserName());
131
132    Configuration clientConf = getSecuredConfiguration();
133    callRpcService(rpcImplClass, User.create(ugi2), clientConf, false);
134  }
135
136  private UserGroupInformation loginKerberosPrincipal(String krbKeytab, String krbPrincipal)
137      throws Exception {
138    Configuration cnf = new Configuration();
139    cnf.set(CommonConfigurationKeys.HADOOP_SECURITY_AUTHENTICATION, "kerberos");
140    UserGroupInformation.setConfiguration(cnf);
141    UserGroupInformation.loginUserFromKeytab(krbPrincipal, krbKeytab);
142    return UserGroupInformation.getLoginUser();
143  }
144
145  private void callRpcService(Class<? extends RpcClient> rpcImplClass, User clientUser,
146                              Configuration clientConf, boolean allowInsecureFallback)
147      throws Exception {
148    Configuration clientConfCopy = new Configuration(clientConf);
149    clientConfCopy.set(RpcClientFactory.CUSTOM_RPC_CLIENT_IMPL_CONF_KEY, rpcImplClass.getName());
150
151    Configuration conf = getSecuredConfiguration();
152    conf.setBoolean(RpcServer.FALLBACK_TO_INSECURE_CLIENT_AUTH, allowInsecureFallback);
153
154    SecurityInfo securityInfoMock = Mockito.mock(SecurityInfo.class);
155    Mockito.when(securityInfoMock.getServerPrincipal())
156        .thenReturn(HBaseKerberosUtils.KRB_PRINCIPAL);
157    SecurityInfo.addInfo("TestDelayedService", securityInfoMock);
158
159    boolean delayReturnValue = false;
160    InetSocketAddress isa = new InetSocketAddress(HOST, 0);
161    TestDelayedImplementation instance = new TestDelayedImplementation(delayReturnValue);
162    BlockingService service =
163        TestDelayedRpcProtos.TestDelayedService.newReflectiveBlockingService(instance);
164
165    RpcServerInterface rpcServer =
166        new RpcServer(null, "testSecuredDelayedRpc",
167            Lists.newArrayList(new RpcServer.BlockingServiceAndInterface(service, null)), isa,
168            conf, new FifoRpcScheduler(conf, 1));
169    rpcServer.start();
170    RpcClient rpcClient =
171        RpcClientFactory.createClient(clientConfCopy, HConstants.DEFAULT_CLUSTER_ID.toString());
172    try {
173      InetSocketAddress address = rpcServer.getListenerAddress();
174      if (address == null) {
175        throw new IOException("Listener channel is closed");
176      }
177      BlockingRpcChannel channel =
178          rpcClient.createBlockingRpcChannel(
179            ServerName.valueOf(address.getHostName(), address.getPort(),
180            System.currentTimeMillis()), clientUser, 5000);
181      TestDelayedRpcProtos.TestDelayedService.BlockingInterface stub =
182          TestDelayedRpcProtos.TestDelayedService.newBlockingStub(channel);
183      List<Integer> results = new ArrayList<Integer>();
184      TestThread th1 = new TestThread(stub, true, results);
185      th1.start();
186      th1.join();
187
188      assertEquals(0xDEADBEEF, results.get(0).intValue());
189    } finally {
190      rpcClient.close();
191      rpcServer.stop();
192    }
193  }

195  public void testRpcFallbackToSimpleAuth(Class<? extends RpcClient> rpcImplClass) throws Exception {
196    String krbKeytab = getKeytabFileForTesting();
197    String krbPrincipal = getPrincipalForTesting();
198
199    UserGroupInformation ugi = loginKerberosPrincipal(krbKeytab, krbPrincipal);
200    assertEquals(AuthenticationMethod.KERBEROS, ugi.getAuthenticationMethod());
201    assertEquals(krbPrincipal, ugi.getUserName());
202
203    String clientUsername = "testuser";
204    UserGroupInformation clientUgi = UserGroupInformation.createUserForTesting(clientUsername,
205        new String[]{clientUsername});
206
207    // check that the client user is insecure
208    assertNotSame(ugi, clientUgi);
209    assertEquals(AuthenticationMethod.SIMPLE, clientUgi.getAuthenticationMethod());
210    assertEquals(clientUsername, clientUgi.getUserName());
211
212    Configuration clientConf = new Configuration();
213    clientConf.set(User.HBASE_SECURITY_CONF_KEY, "simple");
214    callRpcService(rpcImplClass, User.create(clientUgi), clientConf, true);
215  }
