264  private static void setUpTableAndUserPermissions() throws Exception {
265    HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
266    HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
267    hcd.setMaxVersions(100);
268    htd.addFamily(hcd);
269    htd.setOwner(USER_OWNER);
270    createTable(TEST_UTIL, htd, new byte[][] { Bytes.toBytes("s") });
271
272    Region region = TEST_UTIL.getHBaseCluster().getRegions(TEST_TABLE).get(0);
273    RegionCoprocessorHost rcpHost = region.getCoprocessorHost();
274    RCP_ENV = rcpHost.createEnvironment(AccessController.class, ACCESS_CONTROLLER,
275      Coprocessor.PRIORITY_HIGHEST, 1, conf);
276
277    // Set up initial grants
278
279    grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(),
280      Permission.Action.ADMIN,
281      Permission.Action.CREATE,
282      Permission.Action.READ,
283      Permission.Action.WRITE);
284
285    grantOnTable(TEST_UTIL, USER_RW.getShortName(),
286      TEST_TABLE, TEST_FAMILY, null,
287      Permission.Action.READ,
288      Permission.Action.WRITE);
289
290    // USER_CREATE is USER_RW plus CREATE permissions
291    grantOnTable(TEST_UTIL, USER_CREATE.getShortName(),
292      TEST_TABLE, null, null,
293      Permission.Action.CREATE,
294      Permission.Action.READ,
295      Permission.Action.WRITE);
296
297    grantOnTable(TEST_UTIL, USER_RO.getShortName(),
298      TEST_TABLE, TEST_FAMILY, null,
299      Permission.Action.READ);
300
301    grantOnTable(TEST_UTIL, USER_ADMIN_CF.getShortName(),
302      TEST_TABLE, TEST_FAMILY,
303      null, Permission.Action.ADMIN, Permission.Action.CREATE);
304
305    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
306    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_CREATE), Permission.Action.CREATE);
307    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_READ), Permission.Action.READ);
308    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_WRITE), Permission.Action.WRITE);
309
310    assertEquals(5, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
311    try {
312      assertEquals(5, AccessControlClient.getUserPermissions(systemUserConnection,
313          TEST_TABLE.toString()).size());
314    } catch (Throwable e) {
315      LOG.error("error during call of AccessControlClient.getUserPermissions. ", e);
316    }
317  }

1082  public class BulkLoadHelper {
1083    private final FileSystem fs;
1084    private final Path loadPath;
1085    private final Configuration conf;
1086
1087    public BulkLoadHelper(Path loadPath) throws IOException {
1088      fs = TEST_UTIL.getTestFileSystem();
1089      conf = TEST_UTIL.getConfiguration();
1090      loadPath = loadPath.makeQualified(fs);
1091      this.loadPath = loadPath;
1092    }
1093
1094    private void createHFile(Path path,
1095        byte[] family, byte[] qualifier,
1096        byte[] startKey, byte[] endKey, int numRows) throws IOException {
1097
1098      HFile.Writer writer = null;
1099      long now = System.currentTimeMillis();
1100      try {
1101        HFileContext context = new HFileContextBuilder().build();
1102        writer = HFile.getWriterFactory(conf, new CacheConfig(conf))
1103            .withPath(fs, path)
1104            .withFileContext(context)
1105            .create();
1106        // subtract 2 since numRows doesn't include boundary keys
1107        for (byte[] key : Bytes.iterateOnSplits(startKey, endKey, true, numRows-2)) {
1108          KeyValue kv = new KeyValue(key, family, qualifier, now, key);
1109          writer.append(kv);
1110        }
1111      } finally {
1112        if(writer != null)
1113          writer.close();
1114      }
1115    }
1116
1117    private void bulkLoadHFile(
1118        TableName tableName,
1119        byte[] family,
1120        byte[] qualifier,
1121        byte[][][] hfileRanges,
1122        int numRowsPerRange) throws Exception {
1123
1124      Path familyDir = new Path(loadPath, Bytes.toString(family));
1125      fs.mkdirs(familyDir);
1126      int hfileIdx = 0;
1127      for (byte[][] range : hfileRanges) {
1128        byte[] from = range[0];
1129        byte[] to = range[1];
1130        createHFile(new Path(familyDir, "hfile_"+(hfileIdx++)),
1131            family, qualifier, from, to, numRowsPerRange);
1132      }
1133      //set global read so RegionServer can move it
1134      setPermission(loadPath, FsPermission.valueOf("-rwxrwxrwx"));
1135
1136
1137      try (Connection conn = ConnectionFactory.createConnection(conf);
1138           Admin admin = conn.getAdmin();
1139           RegionLocator locator = conn.getRegionLocator(tableName);
1140           Table table = conn.getTable(tableName)) {
1141        TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
1142        LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
1143        loader.doBulkLoad(loadPath, admin, table, locator);
1144      }
1145    }

1147    public void setPermission(Path dir, FsPermission perm) throws IOException {
1148      if(!fs.getFileStatus(dir).isDirectory()) {
1149        fs.setPermission(dir,perm);
1150      }
1151      else {
1152        for(FileStatus el : fs.listStatus(dir)) {
1153          fs.setPermission(el.getPath(), perm);
1154          setPermission(el.getPath() , perm);
1155        }
1156      }
1157    }
