263  private static void setUpTableAndUserPermissions() throws Exception {
264    HTableDescriptor htd = new HTableDescriptor(TEST_TABLE);
265    HColumnDescriptor hcd = new HColumnDescriptor(TEST_FAMILY);
266    hcd.setMaxVersions(100);
267    htd.addFamily(hcd);
268    htd.setOwner(USER_OWNER);
269    createTable(TEST_UTIL, htd, new byte[][] { Bytes.toBytes("s") });
270
271    Region region = TEST_UTIL.getHBaseCluster().getRegions(TEST_TABLE).get(0);
272    RegionCoprocessorHost rcpHost = region.getCoprocessorHost();
273    RCP_ENV = rcpHost.createEnvironment(AccessController.class, ACCESS_CONTROLLER,
274      Coprocessor.PRIORITY_HIGHEST, 1, conf);
275
276    // Set up initial grants
277
278    grantGlobal(TEST_UTIL, USER_ADMIN.getShortName(),
279      Permission.Action.ADMIN,
280      Permission.Action.CREATE,
281      Permission.Action.READ,
282      Permission.Action.WRITE);
283
284    grantOnTable(TEST_UTIL, USER_RW.getShortName(),
285      TEST_TABLE, TEST_FAMILY, null,
286      Permission.Action.READ,
287      Permission.Action.WRITE);
288
289    // USER_CREATE is USER_RW plus CREATE permissions
290    grantOnTable(TEST_UTIL, USER_CREATE.getShortName(),
291      TEST_TABLE, null, null,
292      Permission.Action.CREATE,
293      Permission.Action.READ,
294      Permission.Action.WRITE);
295
296    grantOnTable(TEST_UTIL, USER_RO.getShortName(),
297      TEST_TABLE, TEST_FAMILY, null,
298      Permission.Action.READ);
299
300    grantOnTable(TEST_UTIL, USER_ADMIN_CF.getShortName(),
301      TEST_TABLE, TEST_FAMILY,
302      null, Permission.Action.ADMIN, Permission.Action.CREATE);
303
304    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_ADMIN), Permission.Action.ADMIN);
305    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_CREATE), Permission.Action.CREATE);
306    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_READ), Permission.Action.READ);
307    grantGlobal(TEST_UTIL, toGroupEntry(GROUP_WRITE), Permission.Action.WRITE);
308
309    assertEquals(5, AccessControlLists.getTablePermissions(conf, TEST_TABLE).size());
310    try {
311      assertEquals(5, AccessControlClient.getUserPermissions(systemUserConnection,
312          TEST_TABLE.toString()).size());
313    } catch (Throwable e) {
314      LOG.error("error during call of AccessControlClient.getUserPermissions. ", e);
315    }
316  }

1054  public class BulkLoadHelper {
1055    private final FileSystem fs;
1056    private final Path loadPath;
1057    private final Configuration conf;
1058
1059    public BulkLoadHelper(Path loadPath) throws IOException {
1060      fs = TEST_UTIL.getTestFileSystem();
1061      conf = TEST_UTIL.getConfiguration();
1062      loadPath = loadPath.makeQualified(fs);
1063      this.loadPath = loadPath;
1064    }
1065
1066    private void createHFile(Path path,
1067        byte[] family, byte[] qualifier,
1068        byte[] startKey, byte[] endKey, int numRows) throws IOException {
1069
1070      HFile.Writer writer = null;
1071      long now = System.currentTimeMillis();
1072      try {
1073        HFileContext context = new HFileContextBuilder().build();
1074        writer = HFile.getWriterFactory(conf, new CacheConfig(conf))
1075            .withPath(fs, path)
1076            .withFileContext(context)
1077            .create();
1078        // subtract 2 since numRows doesn't include boundary keys
1079        for (byte[] key : Bytes.iterateOnSplits(startKey, endKey, true, numRows-2)) {
1080          KeyValue kv = new KeyValue(key, family, qualifier, now, key);
1081          writer.append(kv);
1082        }
1083      } finally {
1084        if(writer != null)
1085          writer.close();
1086      }
1087    }
1088
1089    private void bulkLoadHFile(
1090        TableName tableName,
1091        byte[] family,
1092        byte[] qualifier,
1093        byte[][][] hfileRanges,
1094        int numRowsPerRange) throws Exception {
1095
1096      Path familyDir = new Path(loadPath, Bytes.toString(family));
1097      fs.mkdirs(familyDir);
1098      int hfileIdx = 0;
1099      for (byte[][] range : hfileRanges) {
1100        byte[] from = range[0];
1101        byte[] to = range[1];
1102        createHFile(new Path(familyDir, "hfile_"+(hfileIdx++)),
1103            family, qualifier, from, to, numRowsPerRange);
1104      }
1105      //set global read so RegionServer can move it
1106      setPermission(loadPath, FsPermission.valueOf("-rwxrwxrwx"));
1107
1108
1109      try (Connection conn = ConnectionFactory.createConnection(conf);
1110           Admin admin = conn.getAdmin();
1111           RegionLocator locator = conn.getRegionLocator(tableName);
1112           Table table = conn.getTable(tableName)) {
1113        TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
1114        LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
1115        loader.doBulkLoad(loadPath, admin, table, locator);
1116      }
1117    }

1119    public void setPermission(Path dir, FsPermission perm) throws IOException {
1120      if(!fs.getFileStatus(dir).isDirectory()) {
1121        fs.setPermission(dir,perm);
1122      }
1123      else {
1124        for(FileStatus el : fs.listStatus(dir)) {
1125          fs.setPermission(el.getPath(), perm);
1126          setPermission(el.getPath() , perm);
1127        }
1128      }
1129    }
