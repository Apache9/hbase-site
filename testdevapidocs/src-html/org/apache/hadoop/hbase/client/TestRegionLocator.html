018package org.apache.hadoop.hbase.client;
019
020import static org.junit.Assert.assertArrayEquals;
021import static org.junit.Assert.assertEquals;
022import static org.junit.Assert.assertNull;
023
024import java.io.IOException;
025import java.util.Collections;
026import java.util.List;
027import org.apache.hadoop.hbase.HBaseClassTestRule;
028import org.apache.hadoop.hbase.HBaseTestingUtility;
029import org.apache.hadoop.hbase.HConstants;
030import org.apache.hadoop.hbase.HRegionLocation;
031import org.apache.hadoop.hbase.ServerName;
032import org.apache.hadoop.hbase.TableName;
033import org.apache.hadoop.hbase.regionserver.Region;
034import org.apache.hadoop.hbase.testclassification.ClientTests;
035import org.apache.hadoop.hbase.testclassification.MediumTests;
036import org.apache.hadoop.hbase.util.Bytes;
037import org.apache.hadoop.hbase.util.Pair;
038import org.junit.After;
039import org.junit.AfterClass;
040import org.junit.BeforeClass;
041import org.junit.ClassRule;
042import org.junit.Test;
043import org.junit.experimental.categories.Category;
044
045@Category({ MediumTests.class, ClientTests.class })
046public class TestRegionLocator {
047
048  @ClassRule
049  public static final HBaseClassTestRule CLASS_RULE =
050    HBaseClassTestRule.forClass(TestRegionLocator.class);
051
052  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
053
054  private static TableName TABLE_NAME = TableName.valueOf("Locator");
055
056  private static byte[] FAMILY = Bytes.toBytes("family");
057
058  private static int REGION_REPLICATION = 3;
059
060  private static byte[][] SPLIT_KEYS;
061
062  @BeforeClass
063  public static void setUp() throws Exception {
064    UTIL.startMiniCluster(3);
065    TableDescriptor td =
066      TableDescriptorBuilder.newBuilder(TABLE_NAME).setRegionReplication(REGION_REPLICATION)
067        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY)).build();
068    SPLIT_KEYS = new byte[9][];
069    for (int i = 0; i < 9; i++) {
070      SPLIT_KEYS[i] = Bytes.toBytes(Integer.toString(i + 1));
071    }
072    UTIL.getAdmin().createTable(td, SPLIT_KEYS);
073    UTIL.waitTableAvailable(TABLE_NAME);
074    UTIL.getAdmin().balancerSwitch(false, true);
075  }
076
077  @AfterClass
078  public static void tearDown() throws Exception {
079    UTIL.shutdownMiniCluster();
080  }
081
082  @After
083  public void tearDownAfterTest() throws IOException {
084    try (RegionLocator locator = UTIL.getConnection().getRegionLocator(TABLE_NAME)) {
085      locator.clearRegionLocationCache();
086    }
087  }
088
089  private byte[] getStartKey(int index) {
090    return index == 0 ? HConstants.EMPTY_START_ROW : SPLIT_KEYS[index - 1];
091  }
092
093  private byte[] getEndKey(int index) {
094    return index == SPLIT_KEYS.length ? HConstants.EMPTY_END_ROW : SPLIT_KEYS[index];
095  }
096
097  private void assertStartKeys(byte[][] startKeys) {
098    assertEquals(SPLIT_KEYS.length + 1, startKeys.length);
099    for (int i = 0; i < startKeys.length; i++) {
100      assertArrayEquals(getStartKey(i), startKeys[i]);
101    }
102  }
103
104  private void assertEndKeys(byte[][] endKeys) {
105    assertEquals(SPLIT_KEYS.length + 1, endKeys.length);
106    for (int i = 0; i < endKeys.length; i++) {
107      assertArrayEquals(getEndKey(i), endKeys[i]);
108    }
109  }
110
111  @Test
112  public void testStartEndKeys() throws IOException {
113    try (RegionLocator locator = UTIL.getConnection().getRegionLocator(TABLE_NAME)) {
114      assertStartKeys(locator.getStartKeys());
115      assertEndKeys(locator.getEndKeys());
116      Pair<byte[][], byte[][]> startEndKeys = locator.getStartEndKeys();
117      assertStartKeys(startEndKeys.getFirst());
118      assertEndKeys(startEndKeys.getSecond());
119    }
120  }
121
122  private void assertRegionLocation(HRegionLocation loc, int index, int replicaId) {
123    RegionInfo region = loc.getRegion();
124    byte[] startKey = getStartKey(index);
125    assertArrayEquals(startKey, region.getStartKey());
126    assertArrayEquals(getEndKey(index), region.getEndKey());
127    assertEquals(replicaId, region.getReplicaId());
128    ServerName expected =
129      UTIL.getMiniHBaseCluster().getRegionServerThreads().stream().map(t -> t.getRegionServer())
130        .filter(rs -> rs.getRegions(TABLE_NAME).stream().map(Region::getRegionInfo)
131          .anyMatch(r -> r.containsRow(startKey) && r.getReplicaId() == replicaId))
132        .findFirst().get().getServerName();
133    assertEquals(expected, loc.getServerName());
134  }
135
136  @Test
137  public void testGetRegionLocation() throws IOException {
138    try (RegionLocator locator = UTIL.getConnection().getRegionLocator(TABLE_NAME)) {
139      for (int i = 0; i <= SPLIT_KEYS.length; i++) {
140        for (int replicaId = 0; replicaId < REGION_REPLICATION; replicaId++) {
141          assertRegionLocation(locator.getRegionLocation(getStartKey(i), replicaId), i, replicaId);
142        }
143      }
144    }
145  }
146
147  @Test
148  public void testGetAllRegionLocations() throws IOException {
149    try (RegionLocator locator = UTIL.getConnection().getRegionLocator(TABLE_NAME)) {
150      List<HRegionLocation> locs = locator.getAllRegionLocations();
151      assertEquals(REGION_REPLICATION * (SPLIT_KEYS.length + 1), locs.size());
152      Collections.sort(locs, (l1, l2) -> {
153        int c = Bytes.compareTo(l1.getRegion().getStartKey(), l2.getRegion().getStartKey());
154        if (c != 0) {
155          return c;
156        }
157        return Integer.compare(l1.getRegion().getReplicaId(), l2.getRegion().getReplicaId());
158      });
159      for (int i = 0; i <= SPLIT_KEYS.length; i++) {
160        for (int replicaId = 0; replicaId < REGION_REPLICATION; replicaId++) {
161          assertRegionLocation(locs.get(i * REGION_REPLICATION + replicaId), i, replicaId);
162        }
163      }
164    }
165  }
