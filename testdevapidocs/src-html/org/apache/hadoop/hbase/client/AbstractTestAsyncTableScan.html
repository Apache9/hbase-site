018package org.apache.hadoop.hbase.client;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import java.io.IOException;
024import java.io.UncheckedIOException;
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.List;
028import java.util.concurrent.CompletableFuture;
029import java.util.stream.Collectors;
030import java.util.stream.IntStream;
031
032import org.apache.hadoop.hbase.HBaseTestingUtility;
033import org.apache.hadoop.hbase.TableName;
034import org.apache.hadoop.hbase.util.Bytes;
035import org.junit.AfterClass;
036import org.junit.BeforeClass;
037import org.junit.Test;
038
039public abstract class AbstractTestAsyncTableScan {
040
041  protected static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
042
043  protected static TableName TABLE_NAME = TableName.valueOf("async");
044
045  protected static byte[] FAMILY = Bytes.toBytes("cf");
046
047  protected static byte[] CQ1 = Bytes.toBytes("cq1");
048
049  protected static byte[] CQ2 = Bytes.toBytes("cq2");
050
051  protected static int COUNT = 1000;
052
053  protected static AsyncConnection ASYNC_CONN;
054
055  @BeforeClass
056  public static void setUp() throws Exception {
057    TEST_UTIL.startMiniCluster(3);
058    byte[][] splitKeys = new byte[8][];
059    for (int i = 111; i < 999; i += 111) {
060      splitKeys[i / 111 - 1] = Bytes.toBytes(String.format("%03d", i));
061    }
062    TEST_UTIL.createTable(TABLE_NAME, FAMILY, splitKeys);
063    TEST_UTIL.waitTableAvailable(TABLE_NAME);
064    ASYNC_CONN = ConnectionFactory.createAsyncConnection(TEST_UTIL.getConfiguration());
065    RawAsyncTable table = ASYNC_CONN.getRawTable(TABLE_NAME);
066    List<CompletableFuture<?>> futures = new ArrayList<>();
067    IntStream.range(0, COUNT).forEach(
068      i -> futures.add(table.put(new Put(Bytes.toBytes(String.format("%03d", i)))
069          .addColumn(FAMILY, CQ1, Bytes.toBytes(i)).addColumn(FAMILY, CQ2, Bytes.toBytes(i * i)))));
070    CompletableFuture.allOf(futures.toArray(new CompletableFuture<?>[0])).get();
071  }
072
073  @AfterClass
074  public static void tearDown() throws Exception {
075    ASYNC_CONN.close();
076    TEST_UTIL.shutdownMiniCluster();
077  }
078
079  protected abstract Scan createScan();
080
081  protected abstract List<Result> doScan(Scan scan) throws Exception;
082
083  private Result convertToPartial(Result result) {
084    return Result.create(result.rawCells(), result.getExists(), result.isStale(), true);
085  }
086
087  protected final List<Result> convertFromBatchResult(List<Result> results) {
088    assertTrue(results.size() % 2 == 0);
089    return IntStream.range(0, results.size() / 2).mapToObj(i -> {
090      try {
091        return Result.createCompleteResult(Arrays.asList(convertToPartial(results.get(2 * i)),
092          convertToPartial(results.get(2 * i + 1))));
093      } catch (IOException e) {
094        throw new UncheckedIOException(e);
095      }
096    }).collect(Collectors.toList());
097  }
098
099  @Test
100  public void testScanAll() throws Exception {
101    List<Result> results = doScan(createScan());
102    assertEquals(COUNT, results.size());
103    IntStream.range(0, COUNT).forEach(i -> {
104      Result result = results.get(i);
105      assertEquals(String.format("%03d", i), Bytes.toString(result.getRow()));
106      assertEquals(i, Bytes.toInt(result.getValue(FAMILY, CQ1)));
107    });
108  }
109
110  private void assertResultEquals(Result result, int i) {
111    assertEquals(String.format("%03d", i), Bytes.toString(result.getRow()));
112    assertEquals(i, Bytes.toInt(result.getValue(FAMILY, CQ1)));
113    assertEquals(i * i, Bytes.toInt(result.getValue(FAMILY, CQ2)));
114  }
115
116  @Test
117  public void testReversedScanAll() throws Exception {
118    List<Result> results = doScan(createScan().setReversed(true));
119    assertEquals(COUNT, results.size());
120    IntStream.range(0, COUNT).forEach(i -> assertResultEquals(results.get(i), COUNT - i - 1));
121  }
122
123  @Test
124  public void testScanNoStopKey() throws Exception {
125    int start = 345;
126    List<Result> results =
127        doScan(createScan().setStartRow(Bytes.toBytes(String.format("%03d", start))));
128    assertEquals(COUNT - start, results.size());
129    IntStream.range(0, COUNT - start).forEach(i -> assertResultEquals(results.get(i), start + i));
130  }
131
132  @Test
133  public void testReverseScanNoStopKey() throws Exception {
134    int start = 765;
135    List<Result> results = doScan(
136      createScan().setStartRow(Bytes.toBytes(String.format("%03d", start))).setReversed(true));
137    assertEquals(start + 1, results.size());
138    IntStream.range(0, start + 1).forEach(i -> assertResultEquals(results.get(i), start - i));
139  }
140
141  private void testScan(int start, int stop) throws Exception {
142    List<Result> results =
143        doScan(createScan().setStartRow(Bytes.toBytes(String.format("%03d", start)))
144            .setStopRow(Bytes.toBytes(String.format("%03d", stop))));
145    assertEquals(stop - start, results.size());
146    IntStream.range(0, stop - start).forEach(i -> assertResultEquals(results.get(i), start + i));
147  }
148
149  private void testReversedScan(int start, int stop) throws Exception {
150    List<Result> results =
151        doScan(createScan().setStartRow(Bytes.toBytes(String.format("%03d", start)))
152            .setStopRow(Bytes.toBytes(String.format("%03d", stop))).setReversed(true));
153    assertEquals(start - stop, results.size());
154    IntStream.range(0, start - stop).forEach(i -> assertResultEquals(results.get(i), start - i));
155  }
156
157  @Test
158  public void testScanWithStartKeyAndStopKey() throws Exception {
159    testScan(345, 567);
160  }
161
162  @Test
163  public void testReversedScanWithStartKeyAndStopKey() throws Exception {
164    testReversedScan(765, 543);
165  }
166
167  @Test
168  public void testScanAtRegionBoundary() throws Exception {
169    testScan(222, 333);
170  }
171
172  @Test
173  public void testReversedScanAtRegionBoundary() throws Exception {
174    testScan(222, 333);
175  }
