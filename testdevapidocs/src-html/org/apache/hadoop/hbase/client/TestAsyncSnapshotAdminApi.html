019package org.apache.hadoop.hbase.client;
020
021import org.apache.hadoop.hbase.TableName;
022import org.apache.hadoop.hbase.testclassification.ClientTests;
023import org.apache.hadoop.hbase.testclassification.MediumTests;
024import org.apache.hadoop.hbase.util.Bytes;
025import org.junit.Assert;
026import org.junit.Test;
027import org.junit.experimental.categories.Category;
028
029import java.util.Collections;
030import java.util.List;
031
032@Category({ MediumTests.class, ClientTests.class })
033public class TestAsyncSnapshotAdminApi extends TestAsyncAdminBase {
034
035  @Test
036  public void testTakeSnapshot() throws Exception {
037    String snapshotName1 = "snapshotName1";
038    String snapshotName2 = "snapshotName2";
039    TableName tableName = TableName.valueOf("testTakeSnapshot");
040    Admin syncAdmin = TEST_UTIL.getAdmin();
041
042    try {
043      Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
044      for (int i = 0; i < 3000; i++) {
045        table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
046          Bytes.toBytes(i)));
047      }
048
049      admin.snapshot(snapshotName1, tableName).get();
050      admin.snapshot(snapshotName2, tableName).get();
051      List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
052      Collections.sort(snapshots, (snap1, snap2) -> {
053        Assert.assertNotNull(snap1);
054        Assert.assertNotNull(snap1.getName());
055        Assert.assertNotNull(snap2);
056        Assert.assertNotNull(snap2.getName());
057        return snap1.getName().compareTo(snap2.getName());
058      });
059
060      Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
061      Assert.assertEquals(tableName, snapshots.get(0).getTableName());
062      Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
063      Assert.assertEquals(snapshotName2, snapshots.get(1).getName());
064      Assert.assertEquals(tableName, snapshots.get(1).getTableName());
065      Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(1).getType());
066    } finally {
067      syncAdmin.deleteSnapshot(snapshotName1);
068      syncAdmin.deleteSnapshot(snapshotName2);
069      TEST_UTIL.deleteTable(tableName);
070    }
071  }
072
073  @Test
074  public void testCloneSnapshot() throws Exception {
075    String snapshotName1 = "snapshotName1";
076    TableName tableName = TableName.valueOf("testCloneSnapshot");
077    TableName tableName2 = TableName.valueOf("testCloneSnapshot2");
078    Admin syncAdmin = TEST_UTIL.getAdmin();
079
080    try {
081      Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
082      for (int i = 0; i < 3000; i++) {
083        table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
084          Bytes.toBytes(i)));
085      }
086
087      admin.snapshot(snapshotName1, tableName).get();
088      List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
089      Assert.assertEquals(snapshots.size(), 1);
090      Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
091      Assert.assertEquals(tableName, snapshots.get(0).getTableName());
092      Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
093
094      // cloneSnapshot into a existed table.
095      boolean failed = false;
096      try {
097        admin.cloneSnapshot(snapshotName1, tableName).get();
098      } catch (Exception e) {
099        failed = true;
100      }
101      Assert.assertTrue(failed);
102
103      // cloneSnapshot into a new table.
104      Assert.assertTrue(!syncAdmin.tableExists(tableName2));
105      admin.cloneSnapshot(snapshotName1, tableName2).get();
106      syncAdmin.tableExists(tableName2);
107    } finally {
108      syncAdmin.deleteSnapshot(snapshotName1);
109      TEST_UTIL.deleteTable(tableName);
110    }
111  }
112}
