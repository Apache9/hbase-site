061  @Test
062  public void testTakeSnapshot() throws Exception {
063    Admin syncAdmin = TEST_UTIL.getAdmin();
064
065    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
066    for (int i = 0; i < 3000; i++) {
067      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
068        Bytes.toBytes(i)));
069    }
070
071    admin.snapshot(snapshotName1, tableName).get();
072    admin.snapshot(snapshotName2, tableName).get();
073    List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
074    Collections.sort(snapshots, (snap1, snap2) -> {
075      Assert.assertNotNull(snap1);
076      Assert.assertNotNull(snap1.getName());
077      Assert.assertNotNull(snap2);
078      Assert.assertNotNull(snap2.getName());
079      return snap1.getName().compareTo(snap2.getName());
080    });
081
082    Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
083    Assert.assertEquals(tableName, snapshots.get(0).getTableName());
084    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
085    Assert.assertEquals(snapshotName2, snapshots.get(1).getName());
086    Assert.assertEquals(tableName, snapshots.get(1).getTableName());
087    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(1).getType());
088  }

090  @Test
091  public void testCloneSnapshot() throws Exception {
092    TableName tableName2 = TableName.valueOf("testCloneSnapshot2");
093    Admin syncAdmin = TEST_UTIL.getAdmin();
094
095    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
096    for (int i = 0; i < 3000; i++) {
097      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
098        Bytes.toBytes(i)));
099    }
100
101    admin.snapshot(snapshotName1, tableName).get();
102    List<SnapshotDescription> snapshots = syncAdmin.listSnapshots();
103    Assert.assertEquals(snapshots.size(), 1);
104    Assert.assertEquals(snapshotName1, snapshots.get(0).getName());
105    Assert.assertEquals(tableName, snapshots.get(0).getTableName());
106    Assert.assertEquals(SnapshotType.FLUSH, snapshots.get(0).getType());
107
108    // cloneSnapshot into a existed table.
109    boolean failed = false;
110    try {
111      admin.cloneSnapshot(snapshotName1, tableName).get();
112    } catch (Exception e) {
113      failed = true;
114    }
115    Assert.assertTrue(failed);
116
117    // cloneSnapshot into a new table.
118    Assert.assertTrue(!syncAdmin.tableExists(tableName2));
119    admin.cloneSnapshot(snapshotName1, tableName2).get();
120    syncAdmin.tableExists(tableName2);
121  }

123  private void assertResult(TableName tableName, int expectedRowCount) throws IOException {
124    try (Table table = TEST_UTIL.getConnection().getTable(tableName)) {
125      Scan scan = new Scan();
126      try (ResultScanner scanner = table.getScanner(scan)) {
127        Result result;
128        int rowCount = 0;
129        while ((result = scanner.next()) != null) {
130          Assert.assertArrayEquals(result.getRow(), Bytes.toBytes(rowCount));
131          Assert.assertArrayEquals(result.getValue(Bytes.toBytes("f1"), Bytes.toBytes("cq")),
132            Bytes.toBytes(rowCount));
133          rowCount += 1;
134        }
135        Assert.assertEquals(rowCount, expectedRowCount);
136      }
137    }
138  }

140  @Test
141  public void testRestoreSnapshot() throws Exception {
142    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
143    for (int i = 0; i < 3000; i++) {
144      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
145        Bytes.toBytes(i)));
146    }
147    Assert.assertEquals(admin.listSnapshots().get().size(), 0);
148
149    admin.snapshot(snapshotName1, tableName).get();
150    admin.snapshot(snapshotName2, tableName).get();
151    Assert.assertEquals(admin.listSnapshots().get().size(), 2);
152
153    admin.disableTable(tableName).get();
154    admin.restoreSnapshot(snapshotName1, true).get();
155    admin.enableTable(tableName).get();
156    assertResult(tableName, 3000);
157
158    admin.disableTable(tableName).get();
159    admin.restoreSnapshot(snapshotName2, false).get();
160    admin.enableTable(tableName).get();
161    assertResult(tableName, 3000);
162  }

164  @Test
165  public void testListSnapshots() throws Exception {
166    Table table = TEST_UTIL.createTable(tableName, Bytes.toBytes("f1"));
167    for (int i = 0; i < 3000; i++) {
168      table.put(new Put(Bytes.toBytes(i)).addColumn(Bytes.toBytes("f1"), Bytes.toBytes("cq"),
169        Bytes.toBytes(i)));
170    }
171    Assert.assertEquals(admin.listSnapshots().get().size(), 0);
172
173    admin.snapshot(snapshotName1, tableName).get();
174    admin.snapshot(snapshotName2, tableName).get();
175    admin.snapshot(snapshotName3, tableName).get();
176    Assert.assertEquals(admin.listSnapshots().get().size(), 3);
177
178    Assert.assertEquals(admin.listSnapshots("(.*)").get().size(), 3);
179    Assert.assertEquals(admin.listSnapshots("snapshotName(\\d+)").get().size(), 3);
180    Assert.assertEquals(admin.listSnapshots("snapshotName[1|3]").get().size(), 2);
181    Assert.assertEquals(admin.listSnapshots(Pattern.compile("snapshot(.*)")).get().size(), 3);
182    Assert.assertEquals(admin.listTableSnapshots("testListSnapshots", "s(.*)").get().size(), 3);
183    Assert.assertEquals(admin.listTableSnapshots("fakeTableName", "snap(.*)").get().size(), 0);
184    Assert.assertEquals(admin.listTableSnapshots("test(.*)", "snap(.*)[1|3]").get().size(), 2);
185  }
