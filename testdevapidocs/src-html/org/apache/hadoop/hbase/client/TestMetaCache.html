120  @Test
121  public void testPreserveMetaCacheOnException() throws Exception {
122    Table table = conn.getTable(TABLE_NAME);
123    byte[] row = badRS.getOnlineRegions().get(0).getRegionInfo().getStartKey();
124
125    Put put = new Put(row);
126    put.addColumn(FAMILY, QUALIFIER, Bytes.toBytes(10));
127    Get get = new Get(row);
128    Append append = new Append(row);
129    append.add(FAMILY, QUALIFIER, Bytes.toBytes(11));
130    Increment increment = new Increment(row);
131    increment.addColumn(FAMILY, QUALIFIER, 10);
132    Delete delete = new Delete(row);
133    delete.addColumn(FAMILY, QUALIFIER);
134    RowMutations mutations = new RowMutations(row);
135    mutations.add(put);
136    mutations.add(delete);
137
138    Exception exp;
139    boolean success;
140    for (int i = 0; i < 50; i++) {
141      exp = null;
142      success =false;
143      try {
144        table.put(put);
145        // If at least one operation succeeded, we should have cached the region location.
146        success = true;
147        table.get(get);
148        table.append(append);
149        table.increment(increment);
150        table.delete(delete);
151        table.mutateRow(mutations);
152      } catch (IOException ex) {
153        // Only keep track of the last exception that updated the meta cache
154        if (ClientExceptionsUtil.isMetaClearingException(ex) || success) {
155          exp = ex;
156        }
157      }
158      // Do not test if we did not touch the meta cache in this iteration.
159      if(exp != null && ClientExceptionsUtil.isMetaClearingException(exp)) {
160        assertNull(conn.getCachedLocation(TABLE_NAME, row));
161      } else if (success) {
162        assertNotNull(conn.getCachedLocation(TABLE_NAME, row));
163      }
164    }
165  }

167  public static List<Throwable> metaCachePreservingExceptions() {
168    return new ArrayList<Throwable>() {{
169      add(new RegionOpeningException(" "));
170      add(new RegionTooBusyException());
171      add(new ThrottlingException(" "));
172      add(new MultiActionResultTooLarge(" "));
173      add(new RetryImmediatelyException(" "));
174      add(new CallQueueTooBigException());
175    }};
176  }

178  protected static class RegionServerWithFakeRpcServices extends HRegionServer {
179
180    public RegionServerWithFakeRpcServices(Configuration conf, CoordinatedStateManager cp)
181      throws IOException, InterruptedException {
182      super(conf, cp);
183    }
184
185    @Override
186    protected RSRpcServices createRpcServices() throws IOException {
187      return new FakeRSRpcServices(this);
188    }
189  }
190
191  protected static class FakeRSRpcServices extends RSRpcServices {
192
193    private int numReqs = -1;
194    private int expCount = -1;
195    private List<Throwable> metaCachePreservingExceptions = metaCachePreservingExceptions();
196
197    public FakeRSRpcServices(HRegionServer rs) throws IOException {
198      super(rs);
199    }
200
201    @Override
202    public GetResponse get(final RpcController controller,
203                           final ClientProtos.GetRequest request) throws ServiceException {
204      throwSomeExceptions();
205      return super.get(controller, request);
206    }
207
208    @Override
209    public ClientProtos.MutateResponse mutate(final RpcController controller,
210        final ClientProtos.MutateRequest request) throws ServiceException {
211      throwSomeExceptions();
212      return super.mutate(controller, request);
213    }
214
215    @Override
216    public ClientProtos.ScanResponse scan(final RpcController controller,
217        final ClientProtos.ScanRequest request) throws ServiceException {
218      throwSomeExceptions();
219      return super.scan(controller, request);
220    }
221
222    /**
223     * Throw some exceptions. Mostly throw exceptions which do not clear meta cache.
224     * Periodically throw NotSevingRegionException which clears the meta cache.
225     * @throws ServiceException
226     */
227    private void throwSomeExceptions() throws ServiceException {
228      numReqs++;
229      // Succeed every 5 request, throw cache clearing exceptions twice every 5 requests and throw
230      // meta cache preserving exceptions otherwise.
231      if (numReqs % 5 ==0) {
232        return;
233      } else if (numReqs % 5 == 1 || numReqs % 5 == 2) {
234        throw new ServiceException(new NotServingRegionException());
235      }
236      // Round robin between different special exceptions.
237      // This is not ideal since exception types are not tied to the operation performed here,
238      // But, we don't really care here if we throw MultiActionTooLargeException while doing
239      // single Gets.
240      expCount++;
241      Throwable t = metaCachePreservingExceptions.get(
242        expCount % metaCachePreservingExceptions.size());
243      throw new ServiceException(t);
244    }
245  }
