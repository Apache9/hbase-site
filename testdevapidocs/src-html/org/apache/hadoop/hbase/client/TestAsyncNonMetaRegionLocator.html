131  private void assertLocEquals(byte[] startKey, byte[] endKey, ServerName serverName,
132      HRegionLocation loc) {
133    HRegionInfo info = loc.getRegionInfo();
134    assertEquals(TABLE_NAME, info.getTable());
135    assertArrayEquals(startKey, info.getStartKey());
136    assertArrayEquals(endKey, info.getEndKey());
137    assertEquals(serverName, loc.getServerName());
138  }

142    createSingleRegionTable();
143    ServerName serverName = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME).getServerName();
144    for (RegionLocateType locateType : RegionLocateType.values()) {
145      assertLocEquals(EMPTY_START_ROW, EMPTY_END_ROW, serverName,

161  private static byte[][] getStartKeys() {
162    byte[][] startKeys = new byte[SPLIT_KEYS.length + 1][];
163    startKeys[0] = EMPTY_START_ROW;
164    System.arraycopy(SPLIT_KEYS, 0, startKeys, 1, SPLIT_KEYS.length);
165    return startKeys;
166  }
167
168  private static byte[][] getEndKeys() {
169    byte[][] endKeys = Arrays.copyOf(SPLIT_KEYS, SPLIT_KEYS.length + 1);
170    endKeys[endKeys.length - 1] = EMPTY_START_ROW;
171    return endKeys;
172  }
173
174  private ServerName[] getLocations(byte[][] startKeys) {
175    ServerName[] serverNames = new ServerName[startKeys.length];
176    TEST_UTIL.getHBaseCluster().getRegionServerThreads().stream().map(t -> t.getRegionServer())
177        .forEach(rs -> {
178          rs.getOnlineRegions(TABLE_NAME).forEach(r -> {
179            serverNames[Arrays.binarySearch(startKeys, r.getRegionInfo().getStartKey(),
180              Bytes::compareTo)] = rs.getServerName();
181          });
182        });
183    return serverNames;
184  }

186  @Test
187  public void testMultiRegionTable() throws IOException, InterruptedException {
188    createMultiRegionTable();
189    byte[][] startKeys = getStartKeys();
190    ServerName[] serverNames = getLocations(startKeys);
191    IntStream.range(0, 2).forEach(n -> IntStream.range(0, startKeys.length).forEach(i -> {
192      try {
193        assertLocEquals(startKeys[i], i == startKeys.length - 1 ? EMPTY_END_ROW : startKeys[i + 1],
194          serverNames[i],
195          LOCATOR.getRegionLocation(TABLE_NAME, startKeys[i], RegionLocateType.CURRENT).get());
196      } catch (InterruptedException | ExecutionException e) {
197        throw new RuntimeException(e);
198      }
199    }));
200
201    LOCATOR.clearCache(TABLE_NAME);
202    IntStream.range(0, 2).forEach(n -> IntStream.range(0, startKeys.length).forEach(i -> {
203      try {
204        assertLocEquals(startKeys[i], i == startKeys.length - 1 ? EMPTY_END_ROW : startKeys[i + 1],
205          serverNames[i],
206          LOCATOR.getRegionLocation(TABLE_NAME, startKeys[i], RegionLocateType.AFTER).get());
207      } catch (InterruptedException | ExecutionException e) {
208        throw new RuntimeException(e);
209      }
210    }));
211    LOCATOR.clearCache(TABLE_NAME);
212    byte[][] endKeys = getEndKeys();
213    IntStream.range(0, 2).forEach(
214      n -> IntStream.range(0, endKeys.length).map(i -> endKeys.length - 1 - i).forEach(i -> {
215        try {
216          assertLocEquals(i == 0 ? EMPTY_START_ROW : endKeys[i - 1], endKeys[i], serverNames[i],
217            LOCATOR.getRegionLocation(TABLE_NAME, endKeys[i], RegionLocateType.BEFORE).get());
218        } catch (InterruptedException | ExecutionException e) {
219          throw new RuntimeException(e);
220        }
221      }));
222  }

224  @Test
225  public void testRegionMove() throws IOException, InterruptedException, ExecutionException {
226    createSingleRegionTable();
227    ServerName serverName = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME).getServerName();
228    HRegionLocation loc =
229        LOCATOR.getRegionLocation(TABLE_NAME, EMPTY_START_ROW, RegionLocateType.CURRENT).get();
230    assertLocEquals(EMPTY_START_ROW, EMPTY_END_ROW, serverName, loc);
231    ServerName newServerName = TEST_UTIL.getHBaseCluster().getRegionServerThreads().stream()
232        .map(t -> t.getRegionServer().getServerName()).filter(sn -> !sn.equals(serverName))
233        .findAny().get();
234
235    TEST_UTIL.getAdmin().move(Bytes.toBytes(loc.getRegionInfo().getEncodedName()),
236      Bytes.toBytes(newServerName.getServerName()));
237    while (!TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME).getServerName()
238        .equals(newServerName)) {
239      Thread.sleep(100);
240    }
241    // Should be same as it is in cache
242    assertSame(loc,
243      LOCATOR.getRegionLocation(TABLE_NAME, EMPTY_START_ROW, RegionLocateType.CURRENT).get());
244    LOCATOR.updateCachedLocation(loc, null);
245    // null error will not trigger a cache cleanup
246    assertSame(loc,
247      LOCATOR.getRegionLocation(TABLE_NAME, EMPTY_START_ROW, RegionLocateType.CURRENT).get());
248    LOCATOR.updateCachedLocation(loc, new NotServingRegionException());
249    assertLocEquals(EMPTY_START_ROW, EMPTY_END_ROW, newServerName,
250      LOCATOR.getRegionLocation(TABLE_NAME, EMPTY_START_ROW, RegionLocateType.CURRENT).get());
251  }

253  // usually locate after will return the same result, so we add a test to make it return different
254  // result.
255  @Test
256  public void testLocateAfter() throws IOException, InterruptedException, ExecutionException {
257    byte[] row = Bytes.toBytes("1");
258    byte[] splitKey = Arrays.copyOf(row, 2);
259    TEST_UTIL.createTable(TABLE_NAME, FAMILY, new byte[][] { splitKey });
260    TEST_UTIL.waitTableAvailable(TABLE_NAME);
261    HRegionLocation currentLoc =
262        LOCATOR.getRegionLocation(TABLE_NAME, row, RegionLocateType.CURRENT).get();
263    ServerName currentServerName = TEST_UTIL.getRSForFirstRegionInTable(TABLE_NAME).getServerName();
264    assertLocEquals(EMPTY_START_ROW, splitKey, currentServerName, currentLoc);
265
266    HRegionLocation afterLoc =
267        LOCATOR.getRegionLocation(TABLE_NAME, row, RegionLocateType.AFTER).get();
268    ServerName afterServerName =
269        TEST_UTIL.getHBaseCluster().getRegionServerThreads().stream().map(t -> t.getRegionServer())
270            .filter(rs -> rs.getOnlineRegions(TABLE_NAME).stream()
271                .anyMatch(r -> Bytes.equals(splitKey, r.getRegionInfo().getStartKey())))
272            .findAny().get().getServerName();
273    assertLocEquals(splitKey, EMPTY_END_ROW, afterServerName, afterLoc);
274
275    assertSame(afterLoc, LOCATOR.getRegionLocation(TABLE_NAME, row, RegionLocateType.AFTER).get());
276  }

278  // For HBASE-17402
279  @Test
280  public void testConcurrentLocate() throws IOException, InterruptedException, ExecutionException {
281    createMultiRegionTable();
282    byte[][] startKeys = getStartKeys();
283    byte[][] endKeys = getEndKeys();
284    ServerName[] serverNames = getLocations(startKeys);
285    for (int i = 0; i < 100; i++) {
286      LOCATOR.clearCache(TABLE_NAME);
287      List<CompletableFuture<HRegionLocation>> futures = IntStream.range(0, 1000)
288          .mapToObj(n -> String.format("%03d", n)).map(s -> Bytes.toBytes(s))
289          .map(r -> LOCATOR.getRegionLocation(TABLE_NAME, r, RegionLocateType.CURRENT))
290          .collect(toList());
291      for (int j = 0; j < 1000; j++) {
292        int index = Math.min(8, j / 111);
293        assertLocEquals(startKeys[index], endKeys[index], serverNames[index], futures.get(j).get());
294      }
295    }
296  }
