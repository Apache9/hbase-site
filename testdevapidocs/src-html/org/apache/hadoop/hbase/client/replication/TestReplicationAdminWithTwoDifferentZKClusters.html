012package org.apache.hadoop.hbase.client.replication;
013
014import static org.junit.Assert.assertTrue;
015
016import org.apache.hadoop.conf.Configuration;
017import org.apache.hadoop.hbase.HBaseConfiguration;
018import org.apache.hadoop.hbase.HBaseTestingUtility;
019import org.apache.hadoop.hbase.HColumnDescriptor;
020import org.apache.hadoop.hbase.HConstants;
021import org.apache.hadoop.hbase.HTableDescriptor;
022import org.apache.hadoop.hbase.TableName;
023import org.apache.hadoop.hbase.replication.ReplicationPeerConfig;
024import org.apache.hadoop.hbase.testclassification.ClientTests;
025import org.apache.hadoop.hbase.testclassification.MediumTests;
026import org.apache.hadoop.hbase.util.Bytes;
027import org.junit.AfterClass;
028import org.junit.BeforeClass;
029import org.junit.Test;
030import org.junit.experimental.categories.Category;
031
032@Category({ MediumTests.class, ClientTests.class })
033public class TestReplicationAdminWithTwoDifferentZKClusters {
034
035  private static Configuration conf1 = HBaseConfiguration.create();
036  private static Configuration conf2;
037
038  private static HBaseTestingUtility utility1;
039  private static HBaseTestingUtility utility2;
040  private static ReplicationAdmin admin;
041
042  private static final TableName tableName = TableName.valueOf("test");
043  private static final byte[] famName = Bytes.toBytes("f");
044  private static final String peerId = "peer1";
045
046  @BeforeClass
047  public static void setUpBeforeClass() throws Exception {
048    conf1.setBoolean(HConstants.REPLICATION_ENABLE_KEY, HConstants.REPLICATION_ENABLE_DEFAULT);
049    utility1 = new HBaseTestingUtility(conf1);
050    utility1.startMiniCluster();
051    admin = new ReplicationAdmin(conf1);
052
053    conf2 = HBaseConfiguration.create(conf1);
054    conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
055    conf2.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, 2182);
056
057    utility2 = new HBaseTestingUtility(conf2);
058    utility2.startMiniCluster();
059
060    ReplicationPeerConfig config = new ReplicationPeerConfig();
061    config.setClusterKey(utility2.getClusterKey());
062    admin.addPeer(peerId, config, null);
063
064    HTableDescriptor table = new HTableDescriptor(tableName);
065    HColumnDescriptor fam = new HColumnDescriptor(famName);
066    fam.setScope(HConstants.REPLICATION_SCOPE_GLOBAL);
067    table.addFamily(fam);
068
069    utility1.getAdmin().createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
070    utility1.waitUntilAllRegionsAssigned(tableName);
071  }
072
073  @AfterClass
074  public static void tearDownAfterClass() throws Exception {
075    admin.removePeer(peerId);
076    admin.close();
077    utility1.deleteTable(tableName);
078    utility2.deleteTable(tableName);
079    utility2.shutdownMiniCluster();
080    utility1.shutdownMiniCluster();
081  }
082
083  /*
084   * Test for HBASE-15393
085   */
086  @Test
087  public void testEnableTableReplication() throws Exception {
088    admin.enableTableRep(tableName);
089    assertTrue(utility2.getAdmin().tableExists(tableName));
090  }
091
092  @Test
093  public void testDisableTableReplication() throws Exception {
094    admin.disableTableRep(tableName);
095    assertTrue(utility2.getAdmin().tableExists(tableName));
096  }
097}
