083  /**
084   * Simple testing of adding and removing peers, basically shows that
085   * all interactions with ZK work
086   * @throws Exception
087   */
088  @Test
089  public void testAddRemovePeer() throws Exception {
090    // Add a valid peer
091    admin.addPeer(ID_ONE, KEY_ONE);
092    // try adding the same (fails)
093    try {
094      admin.addPeer(ID_ONE, KEY_ONE);
095    } catch (IllegalArgumentException iae) {
096      // OK!
097    }
098    assertEquals(1, admin.getPeersCount());
099    // Try to remove an inexisting peer
100    try {
101      admin.removePeer(ID_SECOND);
102      fail();
103    } catch (IllegalArgumentException iae) {
104      // OK!
105    }
106    assertEquals(1, admin.getPeersCount());
107    // Add a second since multi-slave is supported
108    try {
109      admin.addPeer(ID_SECOND, KEY_SECOND);
110    } catch (IllegalStateException iae) {
111      fail();
112    }
113    assertEquals(2, admin.getPeersCount());
114    // Remove the first peer we added
115    admin.removePeer(ID_ONE);
116    assertEquals(1, admin.getPeersCount());
117    admin.removePeer(ID_SECOND);
118    assertEquals(0, admin.getPeersCount());
119  }

121  @Test
122  public void testAddPeerWithUnDeletedQueues() throws Exception {
123    Configuration conf = TEST_UTIL.getConfiguration();
124    ZooKeeperWatcher zkw = new ZooKeeperWatcher(conf, "Test HBaseAdmin", null);
125    ReplicationQueues repQueues =
126        ReplicationFactory.getReplicationQueues(zkw, conf, null);
127    repQueues.init("server1");
128
129    // add queue for ID_ONE
130    repQueues.addLog(ID_ONE, "file1");
131    try {
132      admin.addPeer(ID_ONE, KEY_ONE);
133      fail();
134    } catch (ReplicationException e) {
135      // OK!
136    }
137    repQueues.removeQueue(ID_ONE);
138    assertEquals(0, repQueues.getAllQueues().size());
139    
140    // add recovered queue for ID_ONE
141    repQueues.addLog(ID_ONE + "-server2", "file1");
142    try {
143      admin.addPeer(ID_ONE, KEY_ONE);
144      fail();
145    } catch (ReplicationException e) {
146      // OK!
147    }
148    repQueues.removeAllQueues();
149    zkw.close();
150  }

172  @Test
173  public void testGetTableCfsStr() {
174    // opposite of TestPerTableCFReplication#testParseTableCFsFromConfig()
175
176    Map<TableName, List<String>> tabCFsMap = null;
177
178    // 1. null or empty string, result should be null
179    assertEquals(null, ReplicationAdmin.getTableCfsStr(tabCFsMap));
180
181
182    // 2. single table: "tab1" / "tab2:cf1" / "tab3:cf1,cf3"
183    tabCFsMap = new TreeMap<TableName, List<String>>();
184    tabCFsMap.put(TableName.valueOf("tab1"), null);   // its table name is "tab1"
185    assertEquals("tab1", ReplicationAdmin.getTableCfsStr(tabCFsMap));
186
187    tabCFsMap = new TreeMap<TableName, List<String>>();
188    tabCFsMap.put(TableName.valueOf("tab1"), Lists.newArrayList("cf1"));
189    assertEquals("tab1:cf1", ReplicationAdmin.getTableCfsStr(tabCFsMap));
190
191    tabCFsMap = new TreeMap<TableName, List<String>>();
192    tabCFsMap.put(TableName.valueOf("tab1"), Lists.newArrayList("cf1", "cf3"));
193    assertEquals("tab1:cf1,cf3", ReplicationAdmin.getTableCfsStr(tabCFsMap));
194
195    // 3. multiple tables: "tab1 ; tab2:cf1 ; tab3:cf1,cf3"
196    tabCFsMap = new TreeMap<TableName, List<String>>();
197    tabCFsMap.put(TableName.valueOf("tab1"), null);
198    tabCFsMap.put(TableName.valueOf("tab2"), Lists.newArrayList("cf1"));
199    tabCFsMap.put(TableName.valueOf("tab3"), Lists.newArrayList("cf1", "cf3"));
200    assertEquals("tab1;tab2:cf1;tab3:cf1,cf3", ReplicationAdmin.getTableCfsStr(tabCFsMap));
201  }

203  @Test
204  public void testAppendPeerTableCFs() throws Exception {
205    // Add a valid peer
206    admin.addPeer(ID_ONE, KEY_ONE);
207
208    admin.appendPeerTableCFs(ID_ONE, "t1");
209    assertEquals("t1", admin.getPeerTableCFs(ID_ONE));
210
211    // append table t2 to replication
212    admin.appendPeerTableCFs(ID_ONE, "t2");
213    String peerTablesOne = admin.getPeerTableCFs(ID_ONE);
214
215    // Different jdk's return different sort order for the tables. ( Not sure on why exactly )
216    //
217    // So instead of asserting that the string is exactly we
218    // assert that the string contains all tables and the needed separator.
219    assertTrue("Should contain t1", peerTablesOne.contains("t1"));
220    assertTrue("Should contain t2", peerTablesOne.contains("t2"));
221    assertTrue("Should contain ; as the seperator", peerTablesOne.contains(";"));
222
223    // append table column family: f1 of t3 to replication
224    admin.appendPeerTableCFs(ID_ONE, "t3:f1");
225    String peerTablesTwo = admin.getPeerTableCFs(ID_ONE);
226    assertTrue("Should contain t1", peerTablesTwo.contains("t1"));
227    assertTrue("Should contain t2", peerTablesTwo.contains("t2"));
228    assertTrue("Should contain t3:f1", peerTablesTwo.contains("t3:f1"));
229    assertTrue("Should contain ; as the seperator", peerTablesTwo.contains(";"));
230    admin.removePeer(ID_ONE);
231  }

233  @Test
234  public void testRemovePeerTableCFs() throws Exception {
235    // Add a valid peer
236    admin.addPeer(ID_ONE, KEY_ONE);
237    try {
238      admin.removePeerTableCFs(ID_ONE, "t3");
239      assertTrue(false);
240    } catch (ReplicationException e) {
241    }
242    assertEquals("", admin.getPeerTableCFs(ID_ONE));
243
244    admin.setPeerTableCFs(ID_ONE, "t1;t2:cf1");
245    try {
246      admin.removePeerTableCFs(ID_ONE, "t3");
247      assertTrue(false);
248    } catch (ReplicationException e) {
249    }
250    assertEquals("t1;t2:cf1", admin.getPeerTableCFs(ID_ONE));
251
252    try {
253      admin.removePeerTableCFs(ID_ONE, "t1:f1");
254      assertTrue(false);
255    } catch (ReplicationException e) {
256    }
257    admin.removePeerTableCFs(ID_ONE, "t1");
258    assertEquals("t2:cf1", admin.getPeerTableCFs(ID_ONE));
259
260    try {
261      admin.removePeerTableCFs(ID_ONE, "t2");
262      assertTrue(false);
263    } catch (ReplicationException e) {
264    }
265    admin.removePeerTableCFs(ID_ONE, "t2:cf1");
266    assertEquals("", admin.getPeerTableCFs(ID_ONE));
267    admin.removePeer(ID_ONE);
268  }
