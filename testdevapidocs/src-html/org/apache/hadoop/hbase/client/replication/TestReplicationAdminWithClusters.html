034/**
035 * Unit testing of ReplicationAdmin with clusters
036 */
037@Category({ MediumTests.class, ClientTests.class })
038public class TestReplicationAdminWithClusters extends TestReplicationBase {
039
040  static Connection connection1;
041  static Connection connection2;
042  static Admin admin1;
043  static Admin admin2;
044
045  @BeforeClass
046  public static void setUpBeforeClass() throws Exception {
047    TestReplicationBase.setUpBeforeClass();
048    connection1 = ConnectionFactory.createConnection(conf1);
049    connection2 = ConnectionFactory.createConnection(conf2);
050    admin1 = connection1.getAdmin();
051    admin2 = connection2.getAdmin();
052  }
053
054  @AfterClass
055  public static void tearDownAfterClass() throws Exception {
056    admin1.close();
057    admin2.close();
058    connection1.close();
059    connection2.close();
060    TestReplicationBase.tearDownAfterClass();
061  }
062
063  @Test(timeout = 300000)
064  public void testEnableReplicationWhenSlaveClusterDoesntHaveTable() throws Exception {
065    admin2.disableTable(tableName);
066    admin2.deleteTable(tableName);
067    assertFalse(admin2.tableExists(tableName));
068    ReplicationAdmin adminExt = new ReplicationAdmin(conf1);
069    adminExt.enableTableRep(tableName);
070    assertTrue(admin2.tableExists(tableName));
071  }
072
073  @Test(timeout = 300000)
074  public void testEnableReplicationWhenReplicationNotEnabled() throws Exception {
075    HTableDescriptor table = admin1.getTableDescriptor(tableName);
076    for (HColumnDescriptor fam : table.getColumnFamilies()) {
077      fam.setScope(HConstants.REPLICATION_SCOPE_LOCAL);
078    }
079    admin1.disableTable(tableName);
080    admin1.modifyTable(tableName, table);
081    admin1.enableTable(tableName);
082
083    admin2.disableTable(tableName);
084    admin2.modifyTable(tableName, table);
085    admin2.enableTable(tableName);
086
087    ReplicationAdmin adminExt = new ReplicationAdmin(conf1);
088    adminExt.enableTableRep(tableName);
089    table = admin1.getTableDescriptor(tableName);
090    for (HColumnDescriptor fam : table.getColumnFamilies()) {
091      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
092    }
093  }
094
095  @Test(timeout = 300000)
096  public void testEnableReplicationWhenTableDescriptorIsNotSameInClusters() throws Exception {
097    HTableDescriptor table = admin2.getTableDescriptor(tableName);
098    HColumnDescriptor f = new HColumnDescriptor("newFamily");
099    table.addFamily(f);
100    admin2.disableTable(tableName);
101    admin2.modifyTable(tableName, table);
102    admin2.enableTable(tableName);
103
104    ReplicationAdmin adminExt = new ReplicationAdmin(conf1);
105    try {
106      adminExt.enableTableRep(tableName);
107      fail("Exception should be thrown if table descriptors in the clusters are not same.");
108    } catch (RuntimeException ignored) {
109
110    }
111    admin1.disableTable(tableName);
112    admin1.modifyTable(tableName, table);
113    admin1.enableTable(tableName);
114    adminExt.enableTableRep(tableName);
115    table = admin1.getTableDescriptor(tableName);
116    for (HColumnDescriptor fam : table.getColumnFamilies()) {
117      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
118    }
119  }
120
121  @Test(timeout = 300000)
122  public void testDisableAndEnableReplication() throws Exception {
123    ReplicationAdmin adminExt = new ReplicationAdmin(conf1);
124    adminExt.disableTableRep(tableName);
125    HTableDescriptor table = admin1.getTableDescriptor(tableName);
126    for (HColumnDescriptor fam : table.getColumnFamilies()) {
127      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_LOCAL);
128    }
129    table = admin2.getTableDescriptor(tableName);
130    for (HColumnDescriptor fam : table.getColumnFamilies()) {
131      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_LOCAL);
132    }
133    adminExt.enableTableRep(tableName);
134    table = admin1.getTableDescriptor(tableName);
135    for (HColumnDescriptor fam : table.getColumnFamilies()) {
136      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
137    }
138  }
139
