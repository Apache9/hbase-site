171  /*
172   * Test enable table replication should create table only in user explicit specified table-cfs.
173   * HBASE-14717
174   */
175  @Test(timeout = 300000)
176  public void testEnableReplicationForExplicitSetTableCfs() throws Exception {
177    final TableName tableName = TableName.valueOf(name.getMethodName());
178    String peerId = "2";
179    if (admin2.isTableAvailable(TestReplicationBase.tableName)) {
180      admin2.disableTable(TestReplicationBase.tableName);
181      admin2.deleteTable(TestReplicationBase.tableName);
182    }
183    assertFalse("Table should not exists in the peer cluster", admin2.isTableAvailable(TestReplicationBase.tableName));
184
185    Map<TableName, ? extends Collection<String>> tableCfs =
186        new HashMap<TableName, Collection<String>>();
187    tableCfs.put(tableName, null);
188    try {
189      adminExt.setPeerTableCFs(peerId, tableCfs);
190      admin1.enableTableReplication(TestReplicationBase.tableName);
191      assertFalse("Table should not be created if user has set table cfs explicitly for the "
192          + "peer and this is not part of that collection",
193        admin2.isTableAvailable(TestReplicationBase.tableName));
194
195      tableCfs.put(TestReplicationBase.tableName, null);
196      adminExt.setPeerTableCFs(peerId, tableCfs);
197      admin1.enableTableReplication(TestReplicationBase.tableName);
198      assertTrue(
199        "Table should be created if user has explicitly added table into table cfs collection",
200        admin2.isTableAvailable(TestReplicationBase.tableName));
201    } finally {
202      adminExt.removePeerTableCFs(peerId, adminExt.getPeerTableCFs(peerId));
203      admin1.disableTableReplication(TestReplicationBase.tableName);
204    }
205  }

207  @Test(timeout=300000)
208  public void testReplicationPeerConfigUpdateCallback() throws Exception {
209    String peerId = "1";
210    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
211    rpc.setClusterKey(utility2.getClusterKey());
212    rpc.setReplicationEndpointImpl(TestUpdatableReplicationEndpoint.class.getName());
213    rpc.getConfiguration().put("key1", "value1");
214
215    admin.addPeer(peerId, rpc);
216    admin.peerAdded(peerId);
217
218    rpc.getConfiguration().put("key1", "value2");
219    admin.updatePeerConfig(peerId, rpc);
220    if (!TestUpdatableReplicationEndpoint.hasCalledBack()) {
221      synchronized(TestUpdatableReplicationEndpoint.class) {
222        TestUpdatableReplicationEndpoint.class.wait(2000L);
223      }
224    }
225
226    assertEquals(true, TestUpdatableReplicationEndpoint.hasCalledBack());
227  }
