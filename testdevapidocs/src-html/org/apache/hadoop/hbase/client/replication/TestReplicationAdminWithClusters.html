070  @Test(timeout = 300000)
071  public void testEnableReplicationWhenSlaveClusterDoesntHaveTable() throws Exception {
072    admin2.disableTable(tableName);
073    admin2.deleteTable(tableName);
074    assertFalse(admin2.tableExists(tableName));
075    adminExt.enableTableRep(tableName);
076    assertTrue(admin2.tableExists(tableName));

079  @Test(timeout = 300000)
080  public void testEnableReplicationWhenReplicationNotEnabled() throws Exception {
081    HTableDescriptor table = admin1.getTableDescriptor(tableName);
082    for (HColumnDescriptor fam : table.getColumnFamilies()) {
083      fam.setScope(HConstants.REPLICATION_SCOPE_LOCAL);
084    }
085    admin1.disableTable(tableName);
086    admin1.modifyTable(tableName, table);
087    admin1.enableTable(tableName);
088
089    admin2.disableTable(tableName);
090    admin2.modifyTable(tableName, table);
091    admin2.enableTable(tableName);
092
093    adminExt.enableTableRep(tableName);
094    table = admin1.getTableDescriptor(tableName);
095    for (HColumnDescriptor fam : table.getColumnFamilies()) {
096      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
097    }
098  }
099
100  @Test(timeout = 300000)
101  public void testEnableReplicationWhenTableDescriptorIsNotSameInClusters() throws Exception {
102    HTableDescriptor table = admin2.getTableDescriptor(tableName);
103    HColumnDescriptor f = new HColumnDescriptor("newFamily");
104    table.addFamily(f);
105    admin2.disableTable(tableName);
106    admin2.modifyTable(tableName, table);
107    admin2.enableTable(tableName);
108
109    try {
110      adminExt.enableTableRep(tableName);
111      fail("Exception should be thrown if table descriptors in the clusters are not same.");
112    } catch (RuntimeException ignored) {
113
114    }
115    admin1.disableTable(tableName);
116    admin1.modifyTable(tableName, table);
117    admin1.enableTable(tableName);
118    adminExt.enableTableRep(tableName);
119    table = admin1.getTableDescriptor(tableName);
120    for (HColumnDescriptor fam : table.getColumnFamilies()) {
121      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
122    }
123  }

125  @Test(timeout = 300000)
126  public void testDisableAndEnableReplication() throws Exception {
127    adminExt.disableTableRep(tableName);
128    HTableDescriptor table = admin1.getTableDescriptor(tableName);
129    for (HColumnDescriptor fam : table.getColumnFamilies()) {
130      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_LOCAL);
131    }
132    table = admin2.getTableDescriptor(tableName);
133    for (HColumnDescriptor fam : table.getColumnFamilies()) {
134      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_LOCAL);
135    }
136    adminExt.enableTableRep(tableName);
137    table = admin1.getTableDescriptor(tableName);
138    for (HColumnDescriptor fam : table.getColumnFamilies()) {
139      assertEquals(fam.getScope(), HConstants.REPLICATION_SCOPE_GLOBAL);
140    }
141  }

163  /*
164   * Test enable table replication should create table only in user explicit specified table-cfs.
165   * HBASE-14717
166   */
167  @Test(timeout = 300000)
168  public void testEnableReplicationForExplicitSetTableCfs() throws Exception {
169    TableName tn = TableName.valueOf("testEnableReplicationForSetTableCfs");
170    String peerId = "2";
171    if (admin2.isTableAvailable(tableName)) {
172      admin2.disableTable(tableName);
173      admin2.deleteTable(tableName);
174    }
175    assertFalse("Table should not exists in the peer cluster", admin2.isTableAvailable(tableName));
176
177    Map<TableName, ? extends Collection<String>> tableCfs =
178        new HashMap<TableName, Collection<String>>();
179    tableCfs.put(tn, null);
180    try {
181      adminExt.setPeerTableCFs(peerId, tableCfs);
182      adminExt.enableTableRep(tableName);
183      assertFalse("Table should not be created if user has set table cfs explicitly for the "
184          + "peer and this is not part of that collection",
185        admin2.isTableAvailable(tableName));
186
187      tableCfs.put(tableName, null);
188      adminExt.setPeerTableCFs(peerId, tableCfs);
189      adminExt.enableTableRep(tableName);
190      assertTrue(
191        "Table should be created if user has explicitly added table into table cfs collection",
192        admin2.isTableAvailable(tableName));
193    } finally {
194      adminExt.removePeerTableCFs(peerId, adminExt.getPeerTableCFs(peerId));
195      adminExt.disableTableRep(tableName);
196    }
197  }
