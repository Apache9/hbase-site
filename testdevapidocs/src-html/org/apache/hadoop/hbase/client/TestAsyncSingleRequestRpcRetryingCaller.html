049@Category({ MediumTests.class, ClientTests.class })
050public class TestAsyncSingleRequestRpcRetryingCaller {
051
052  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
053
054  private static TableName TABLE_NAME = TableName.valueOf("async");
055
056  private static byte[] FAMILY = Bytes.toBytes("cf");
057
058  private static byte[] QUALIFIER = Bytes.toBytes("cq");
059
060  private static byte[] ROW = Bytes.toBytes("row");
061
062  private static byte[] VALUE = Bytes.toBytes("value");

064  private AsyncConnectionImpl asyncConn;

066  @BeforeClass
067  public static void setUpBeforeClass() throws Exception {
068    TEST_UTIL.startMiniCluster(2);
069    TEST_UTIL.getAdmin().setBalancerRunning(false, true);
070    TEST_UTIL.createTable(TABLE_NAME, FAMILY);
071    TEST_UTIL.waitTableAvailable(TABLE_NAME);
072  }

076    TEST_UTIL.shutdownMiniCluster();

082      asyncConn.close();
083      asyncConn = null;
084    }

087  private void initConn(int startLogErrorsCnt, long pauseMs, int maxRetires) throws IOException {
088    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
089    conf.setInt(AsyncProcess.START_LOG_ERRORS_AFTER_COUNT_KEY, startLogErrorsCnt);
090    conf.setLong(HConstants.HBASE_CLIENT_PAUSE, pauseMs);
091    conf.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, maxRetires);
092    asyncConn = new AsyncConnectionImpl(conf, User.getCurrent());
093  }
094
095  @Test
096  public void testRegionMove() throws InterruptedException, ExecutionException, IOException {
097    initConn(0, 100, 30);
098    // This will leave a cached entry in location cache
099    HRegionLocation loc = asyncConn.getRegionLocator(TABLE_NAME).getRegionLocation(ROW).get();
100    int index = TEST_UTIL.getHBaseCluster().getServerWith(loc.getRegionInfo().getRegionName());
101    TEST_UTIL.getAdmin().move(loc.getRegionInfo().getEncodedNameAsBytes(), Bytes.toBytes(
102      TEST_UTIL.getHBaseCluster().getRegionServer(1 - index).getServerName().getServerName()));
103    AsyncTable table = asyncConn.getTable(TABLE_NAME);
104    table.put(new Put(ROW).addColumn(FAMILY, QUALIFIER, VALUE)).get();
105
106    // move back
107    TEST_UTIL.getAdmin().move(loc.getRegionInfo().getEncodedNameAsBytes(),
108      Bytes.toBytes(loc.getServerName().getServerName()));
109    Result result = table.get(new Get(ROW).addColumn(FAMILY, QUALIFIER)).get();
110    assertArrayEquals(VALUE, result.getValue(FAMILY, QUALIFIER));
111  }
112
113  private <T> CompletableFuture<T> failedFuture() {
114    CompletableFuture<T> future = new CompletableFuture<>();
115    future.completeExceptionally(new RuntimeException("Inject error!"));
116    return future;
117  }
118
119  @Test
120  public void testMaxRetries() throws IOException, InterruptedException {
121    initConn(0, 10, 2);
122    try {
123      asyncConn.callerFactory.single().table(TABLE_NAME).row(ROW).operationTimeout(1, TimeUnit.DAYS)
124          .action((controller, loc, stub) -> failedFuture()).call().get();
125      fail();
126    } catch (ExecutionException e) {
127      assertThat(e.getCause(), instanceOf(RetriesExhaustedException.class));
128    }
129  }
130
131  @Test
132  public void testOperationTimeout() throws IOException, InterruptedException {
133    initConn(0, 100, Integer.MAX_VALUE);
134    long startNs = System.nanoTime();
135    try {
136      asyncConn.callerFactory.single().table(TABLE_NAME).row(ROW)
137          .operationTimeout(1, TimeUnit.SECONDS).action((controller, loc, stub) -> failedFuture())
138          .call().get();
139      fail();
140    } catch (ExecutionException e) {
141      assertThat(e.getCause(), instanceOf(RetriesExhaustedException.class));
142    }
143    long costNs = System.nanoTime() - startNs;
144    assertTrue(costNs >= TimeUnit.SECONDS.toNanos(1));
145    assertTrue(costNs < TimeUnit.SECONDS.toNanos(2));
146  }
147
148  @Test
149  public void testLocateError() throws IOException, InterruptedException, ExecutionException {
150    initConn(0, 100, 5);
151    AtomicBoolean errorTriggered = new AtomicBoolean(false);
152    AtomicInteger count = new AtomicInteger(0);
153    HRegionLocation loc = asyncConn.getRegionLocator(TABLE_NAME).getRegionLocation(ROW).get();
154
155    try (AsyncRegionLocator mockedLocator = new AsyncRegionLocator(asyncConn.getConfiguration()) {
156      @Override
157      CompletableFuture<HRegionLocation> getRegionLocation(TableName tableName, byte[] row,
158          boolean reload) {
159        if (tableName.equals(TABLE_NAME)) {
160          CompletableFuture<HRegionLocation> future = new CompletableFuture<>();
161          if (count.getAndIncrement() == 0) {
162            errorTriggered.set(true);
163            future.completeExceptionally(new RuntimeException("Inject error!"));
164          } else {
165            future.complete(loc);
166          }
167          return future;
168        } else {
169          return super.getRegionLocation(tableName, row, reload);
170        }
171      }
172
173      @Override
174      void updateCachedLocations(TableName tableName, byte[] regionName, byte[] row,
175          Object exception, ServerName source) {
176      }
177    };
178        AsyncConnectionImpl mockedConn = new AsyncConnectionImpl(asyncConn.getConfiguration(),
179            User.getCurrent()) {
180
181          @Override
182          AsyncRegionLocator getLocator() {
183            return mockedLocator;
184          }
185        }) {
186      AsyncTable table = new AsyncTableImpl(mockedConn, TABLE_NAME);
187      table.put(new Put(ROW).addColumn(FAMILY, QUALIFIER, VALUE)).get();
188      assertTrue(errorTriggered.get());
189      errorTriggered.set(false);
190      count.set(0);
191      Result result = table.get(new Get(ROW).addColumn(FAMILY, QUALIFIER)).get();
192      assertArrayEquals(VALUE, result.getValue(FAMILY, QUALIFIER));
193      assertTrue(errorTriggered.get());
194    }
195  }
