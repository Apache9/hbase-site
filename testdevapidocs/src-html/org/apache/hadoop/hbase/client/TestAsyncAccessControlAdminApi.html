012package org.apache.hadoop.hbase.client;
013
014import static org.junit.Assert.assertEquals;
015
016import java.util.List;
017import org.apache.hadoop.hbase.HBaseClassTestRule;
018import org.apache.hadoop.hbase.TableName;
019import org.apache.hadoop.hbase.security.access.AccessControlLists;
020import org.apache.hadoop.hbase.security.access.GetUserPermissionsRequest;
021import org.apache.hadoop.hbase.security.access.Permission;
022import org.apache.hadoop.hbase.security.access.SecureTestUtil;
023import org.apache.hadoop.hbase.security.access.UserPermission;
024import org.apache.hadoop.hbase.testclassification.ClientTests;
025import org.apache.hadoop.hbase.testclassification.SmallTests;
026import org.junit.BeforeClass;
027import org.junit.ClassRule;
028import org.junit.Test;
029import org.junit.experimental.categories.Category;
030import org.junit.runner.RunWith;
031import org.junit.runners.Parameterized;
032
033@RunWith(Parameterized.class)
034@Category({ ClientTests.class, SmallTests.class })
035public class TestAsyncAccessControlAdminApi extends TestAsyncAdminBase {
036
037  @ClassRule
038  public static final HBaseClassTestRule CLASS_RULE =
039      HBaseClassTestRule.forClass(TestAsyncAccessControlAdminApi.class);
040
041  @BeforeClass
042  public static void setUpBeforeClass() throws Exception {
043    SecureTestUtil.enableSecurity(TEST_UTIL.getConfiguration());
044    TEST_UTIL.startMiniCluster(1);
045    TEST_UTIL.waitTableAvailable(AccessControlLists.ACL_TABLE_NAME);
046    ASYNC_CONN = ConnectionFactory.createAsyncConnection(TEST_UTIL.getConfiguration()).get();
047  }
048
049  @Test
050  public void testGrant() throws Exception {
051    TableName tableName = TableName.valueOf("test-table");
052    String user = "test-user";
053    UserPermission userPermission = new UserPermission(user,
054        Permission.newBuilder(tableName).withActions(Permission.Action.READ).build());
055    // grant user table permission
056    admin.grant(userPermission, false).get();
057
058    // get table permissions
059    List<UserPermission> userPermissions =
060        admin.getUserPermissions(GetUserPermissionsRequest.newBuilder(tableName).build()).get();
061    assertEquals(1, userPermissions.size());
062    assertEquals(userPermission, userPermissions.get(0));
063
064    // get user table permissions
065    userPermissions = admin.getUserPermissions(
066      GetUserPermissionsRequest.newBuilder(tableName).withUserName(user).build()).get();
067    assertEquals(1, userPermissions.size());
068    assertEquals(userPermission, userPermissions.get(0));
069
070    userPermissions = admin.getUserPermissions(
071      GetUserPermissionsRequest.newBuilder(tableName).withUserName("u").build()).get();
072    assertEquals(0, userPermissions.size());
073  }
074}
