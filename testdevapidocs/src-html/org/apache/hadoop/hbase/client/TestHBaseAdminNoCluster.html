075  @Test
076  public void testMasterMonitorCallableRetries()
077  throws MasterNotRunningException, ZooKeeperConnectionException, IOException, ServiceException {
078    Configuration configuration = HBaseConfiguration.create();
079    // Set the pause and retry count way down.
080    configuration.setLong(HConstants.HBASE_CLIENT_PAUSE, 1);
081    final int count = 10;
082    configuration.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, count);
083    // Get mocked connection.   Getting the connection will register it so when HBaseAdmin is
084    // constructed with same configuration, it will find this mocked connection.
085    ClusterConnection connection = HConnectionTestingUtility.getMockedConnection(configuration);
086    // Mock so we get back the master interface.  Make it so when createTable is called, we throw
087    // the PleaseHoldException.
088    MasterKeepAliveConnection masterAdmin = Mockito.mock(MasterKeepAliveConnection.class);
089    Mockito.when(masterAdmin.createTable((RpcController)Mockito.any(),
090      (CreateTableRequest)Mockito.any())).
091        thenThrow(new ServiceException("Test fail").initCause(new PleaseHoldException("test")));
092    Mockito.when(connection.getKeepAliveMasterService()).thenReturn(masterAdmin);
093    Admin admin = new HBaseAdmin(connection);
094    try {
095      HTableDescriptor htd =
096        new HTableDescriptor(TableName.valueOf("testMasterMonitorCollableRetries"));
097      // Pass any old htable descriptor; not important
098      try {
099        admin.createTable(htd, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
100        fail();
101      } catch (RetriesExhaustedException e) {
102        Log.info("Expected fail", e);
103      }
104      // Assert we were called 'count' times.
105      Mockito.verify(masterAdmin, Mockito.atLeast(count)).createTable((RpcController)Mockito.any(),
106        (CreateTableRequest)Mockito.any());
107    } finally {
108      admin.close();
109      if (connection != null) connection.close();
110    }
111  }

284  private static interface MethodCaller {
285    void call(Admin admin) throws Exception;
286    void verify(MasterKeepAliveConnection masterAdmin, int count) throws Exception;
287  }

289  private void testMasterOperationIsRetried(MethodCaller caller) throws Exception {
290    Configuration configuration = HBaseConfiguration.create();
291    // Set the pause and retry count way down.
292    configuration.setLong(HConstants.HBASE_CLIENT_PAUSE, 1);
293    final int count = 10;
294    configuration.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, count);
295
296    ClusterConnection connection = mock(ClusterConnection.class);
297    when(connection.getConfiguration()).thenReturn(configuration);
298    MasterKeepAliveConnection masterAdmin =
299        Mockito.mock(MasterKeepAliveConnection.class, new Answer() {
300          @Override
301          public Object answer(InvocationOnMock invocation) throws Throwable {
302            if (invocation.getMethod().getName().equals("close")) {
303              return null;
304            }
305            throw new MasterNotRunningException(); // all methods will throw an exception
306          }
307        });
308    Mockito.when(connection.getKeepAliveMasterService()).thenReturn(masterAdmin);
309
310    Admin admin = null;
311    try {
312      admin = Mockito.spy(new HBaseAdmin(connection));
313      // mock the call to getRegion since in the absence of a cluster (which means the meta
314      // is not assigned), getRegion can't function
315      Mockito.doReturn(null).when(((HBaseAdmin)admin)).getRegion(Matchers.<byte[]>any());
316      try {
317        caller.call(admin); // invoke the HBaseAdmin method
318        fail();
319      } catch (RetriesExhaustedException e) {
320        Log.info("Expected fail", e);
321      }
322      // Assert we were called 'count' times.
323      caller.verify(masterAdmin, count);
324    } finally {
325      if (admin != null) {admin.close();}
326    }
327  }
