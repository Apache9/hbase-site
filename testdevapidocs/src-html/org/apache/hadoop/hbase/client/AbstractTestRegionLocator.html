018package org.apache.hadoop.hbase.client;
019
020import static org.junit.Assert.assertArrayEquals;
021import static org.junit.Assert.assertEquals;
022
023import java.io.IOException;
024import java.util.Collections;
025import java.util.List;
026import org.apache.hadoop.hbase.HBaseTestingUtility;
027import org.apache.hadoop.hbase.HConstants;
028import org.apache.hadoop.hbase.HRegionLocation;
029import org.apache.hadoop.hbase.ServerName;
030import org.apache.hadoop.hbase.TableName;
031import org.apache.hadoop.hbase.regionserver.Region;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.Pair;
034import org.junit.After;
035import org.junit.Test;
036
037public abstract class AbstractTestRegionLocator {
038
039  protected static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
040
041  protected static TableName TABLE_NAME = TableName.valueOf("Locator");
042
043  protected static byte[] FAMILY = Bytes.toBytes("family");
044
045  protected static int REGION_REPLICATION = 3;
046
047  protected static byte[][] SPLIT_KEYS;
048
049  protected static void startClusterAndCreateTable() throws Exception {
050    UTIL.startMiniCluster(3);
051    TableDescriptor td =
052      TableDescriptorBuilder.newBuilder(TABLE_NAME).setRegionReplication(REGION_REPLICATION)
053        .setColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY)).build();
054    SPLIT_KEYS = new byte[9][];
055    for (int i = 0; i < 9; i++) {
056      SPLIT_KEYS[i] = Bytes.toBytes(Integer.toString(i + 1));
057    }
058    UTIL.getAdmin().createTable(td, SPLIT_KEYS);
059    UTIL.waitTableAvailable(TABLE_NAME);
060    UTIL.getAdmin().balancerSwitch(false, true);
061  }
062
063  @After
064  public void tearDownAfterTest() throws IOException {
065    clearCache();
066  }
067
068  private byte[] getStartKey(int index) {
069    return index == 0 ? HConstants.EMPTY_START_ROW : SPLIT_KEYS[index - 1];
070  }
071
072  private byte[] getEndKey(int index) {
073    return index == SPLIT_KEYS.length ? HConstants.EMPTY_END_ROW : SPLIT_KEYS[index];
074  }
075
076  private void assertStartKeys(byte[][] startKeys) {
077    assertEquals(SPLIT_KEYS.length + 1, startKeys.length);
078    for (int i = 0; i < startKeys.length; i++) {
079      assertArrayEquals(getStartKey(i), startKeys[i]);
080    }
081  }
082
083  private void assertEndKeys(byte[][] endKeys) {
084    assertEquals(SPLIT_KEYS.length + 1, endKeys.length);
085    for (int i = 0; i < endKeys.length; i++) {
086      assertArrayEquals(getEndKey(i), endKeys[i]);
087    }
088  }
089
090  @Test
091  public void testStartEndKeys() throws IOException {
092    assertStartKeys(getStartKeys());
093    assertEndKeys(getEndKeys());
094    Pair<byte[][], byte[][]> startEndKeys = getStartEndKeys();
095    assertStartKeys(startEndKeys.getFirst());
096    assertEndKeys(startEndKeys.getSecond());
097  }
098
099  private void assertRegionLocation(HRegionLocation loc, int index, int replicaId) {
100    RegionInfo region = loc.getRegion();
101    byte[] startKey = getStartKey(index);
102    assertArrayEquals(startKey, region.getStartKey());
103    assertArrayEquals(getEndKey(index), region.getEndKey());
104    assertEquals(replicaId, region.getReplicaId());
105    ServerName expected =
106      UTIL.getMiniHBaseCluster().getRegionServerThreads().stream().map(t -> t.getRegionServer())
107        .filter(rs -> rs.getRegions(TABLE_NAME).stream().map(Region::getRegionInfo)
108          .anyMatch(r -> r.containsRow(startKey) && r.getReplicaId() == replicaId))
109        .findFirst().get().getServerName();
110    assertEquals(expected, loc.getServerName());
111  }
112
113  @Test
114  public void testGetRegionLocation() throws IOException {
115    for (int i = 0; i <= SPLIT_KEYS.length; i++) {
116      for (int replicaId = 0; replicaId < REGION_REPLICATION; replicaId++) {
117        assertRegionLocation(getRegionLocation(getStartKey(i), replicaId), i, replicaId);
118      }
119    }
120  }
121
122  @Test
123  public void testGetRegionLocations() throws IOException {
124    for (int i = 0; i <= SPLIT_KEYS.length; i++) {
125      List<HRegionLocation> locs = getRegionLocations(getStartKey(i));
126      assertEquals(REGION_REPLICATION, locs.size());
127      for (int replicaId = 0; replicaId < REGION_REPLICATION; replicaId++) {
128        assertRegionLocation(locs.get(replicaId), i, replicaId);
129      }
130    }
131  }
132
133  @Test
134  public void testGetAllRegionLocations() throws IOException {
135    List<HRegionLocation> locs = getAllRegionLocations();
136    assertEquals(REGION_REPLICATION * (SPLIT_KEYS.length + 1), locs.size());
137    Collections.sort(locs, (l1, l2) -> {
138      int c = Bytes.compareTo(l1.getRegion().getStartKey(), l2.getRegion().getStartKey());
139      if (c != 0) {
140        return c;
141      }
142      return Integer.compare(l1.getRegion().getReplicaId(), l2.getRegion().getReplicaId());
143    });
144    for (int i = 0; i <= SPLIT_KEYS.length; i++) {
145      for (int replicaId = 0; replicaId < REGION_REPLICATION; replicaId++) {
146        assertRegionLocation(locs.get(i * REGION_REPLICATION + replicaId), i, replicaId);
147      }
148    }
149  }
150
151  protected abstract byte[][] getStartKeys() throws IOException;
152
153  protected abstract byte[][] getEndKeys() throws IOException;
154
155  protected abstract Pair<byte[][], byte[][]> getStartEndKeys() throws IOException;
156
157  protected abstract HRegionLocation getRegionLocation(byte[] row, int replicaId)
158      throws IOException;
159
160  protected abstract List<HRegionLocation> getRegionLocations(byte[] row) throws IOException;
161
162  protected abstract List<HRegionLocation> getAllRegionLocations() throws IOException;
163
164  protected abstract void clearCache() throws IOException;
165}
