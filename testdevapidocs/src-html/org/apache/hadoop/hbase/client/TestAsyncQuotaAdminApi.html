081  @Test
082  public void testThrottleType() throws Exception {
083    String userName = User.getCurrent().getShortName();
084
085    admin.setQuota(
086      QuotaSettingsFactory.throttleUser(userName, ThrottleType.READ_NUMBER, 6, TimeUnit.MINUTES))
087        .get();
088    admin.setQuota(
089      QuotaSettingsFactory.throttleUser(userName, ThrottleType.WRITE_NUMBER, 12, TimeUnit.MINUTES))
090        .get();
091    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, true)).get();
092
093    int countThrottle = 0;
094    int countGlobalBypass = 0;
095    for (QuotaSettings settings : admin.getQuota(null).get()) {
096      switch (settings.getQuotaType()) {
097      case THROTTLE:
098        countThrottle++;
099        break;
100      case GLOBAL_BYPASS:
101        countGlobalBypass++;
102        break;
103      default:
104        fail("unexpected settings type: " + settings.getQuotaType());
105      }
106    }
107    assertEquals(2, countThrottle);
108    assertEquals(1, countGlobalBypass);
109
110    admin.setQuota(QuotaSettingsFactory.unthrottleUser(userName)).get();
111    assertNumResults(1, null);
112    admin.setQuota(QuotaSettingsFactory.bypassGlobals(userName, false)).get();
113    assertNumResults(0, null);
114  }

116  @Test
117  public void testQuotaRetrieverFilter() throws Exception {
118    TableName[] tables = new TableName[] { TableName.valueOf("T0"), TableName.valueOf("T01"),
119        TableName.valueOf("NS0:T2"), };
120    String[] namespaces = new String[] { "NS0", "NS01", "NS2" };
121    String[] users = new String[] { "User0", "User01", "User2" };
122
123    for (String user : users) {
124      admin.setQuota(
125        QuotaSettingsFactory.throttleUser(user, ThrottleType.REQUEST_NUMBER, 1, TimeUnit.MINUTES))
126          .get();
127
128      for (TableName table : tables) {
129        admin.setQuota(QuotaSettingsFactory.throttleUser(user, table, ThrottleType.REQUEST_NUMBER,
130          2, TimeUnit.MINUTES)).get();
131      }
132
133      for (String ns : namespaces) {
134        admin.setQuota(QuotaSettingsFactory.throttleUser(user, ns, ThrottleType.REQUEST_NUMBER, 3,
135          TimeUnit.MINUTES)).get();
136      }
137    }
138    assertNumResults(21, null);
139
140    for (TableName table : tables) {
141      admin.setQuota(
142        QuotaSettingsFactory.throttleTable(table, ThrottleType.REQUEST_NUMBER, 4, TimeUnit.MINUTES))
143          .get();
144    }
145    assertNumResults(24, null);
146
147    for (String ns : namespaces) {
148      admin.setQuota(QuotaSettingsFactory.throttleNamespace(ns, ThrottleType.REQUEST_NUMBER, 5,
149        TimeUnit.MINUTES)).get();
150    }
151    assertNumResults(27, null);
152
153    assertNumResults(7, new QuotaFilter().setUserFilter("User0"));
154    assertNumResults(0, new QuotaFilter().setUserFilter("User"));
155    assertNumResults(21, new QuotaFilter().setUserFilter("User.*"));
156    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setTableFilter("T0"));
157    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setTableFilter("NS.*"));
158    assertNumResults(0, new QuotaFilter().setUserFilter("User.*").setTableFilter("T"));
159    assertNumResults(6, new QuotaFilter().setUserFilter("User.*").setTableFilter("T.*"));
160    assertNumResults(3, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS0"));
161    assertNumResults(0, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS"));
162    assertNumResults(9, new QuotaFilter().setUserFilter("User.*").setNamespaceFilter("NS.*"));
163    assertNumResults(6,
164      new QuotaFilter().setUserFilter("User.*").setTableFilter("T0").setNamespaceFilter("NS0"));
165    assertNumResults(1, new QuotaFilter().setTableFilter("T0"));
166    assertNumResults(0, new QuotaFilter().setTableFilter("T"));
167    assertNumResults(2, new QuotaFilter().setTableFilter("T.*"));
168    assertNumResults(3, new QuotaFilter().setTableFilter(".*T.*"));
169    assertNumResults(1, new QuotaFilter().setNamespaceFilter("NS0"));
170    assertNumResults(0, new QuotaFilter().setNamespaceFilter("NS"));
171    assertNumResults(3, new QuotaFilter().setNamespaceFilter("NS.*"));
172
173    for (String user : users) {
174      admin.setQuota(QuotaSettingsFactory.unthrottleUser(user)).get();
175      for (TableName table : tables) {
176        admin.setQuota(QuotaSettingsFactory.unthrottleUser(user, table)).get();
177      }
178      for (String ns : namespaces) {
179        admin.setQuota(QuotaSettingsFactory.unthrottleUser(user, ns)).get();
180      }
181    }
182    assertNumResults(6, null);
183
184    for (TableName table : tables) {
185      admin.setQuota(QuotaSettingsFactory.unthrottleTable(table)).get();
186    }
187    assertNumResults(3, null);
188
189    for (String ns : namespaces) {
190      admin.setQuota(QuotaSettingsFactory.unthrottleNamespace(ns)).get();
191    }
192    assertNumResults(0, null);
193  }
194
195  private void assertNumResults(int expected, final QuotaFilter filter) throws Exception {
196    assertEquals(expected, countResults(filter));
197  }
198
199  private int countResults(final QuotaFilter filter) throws Exception {
200    int count = 0;
201    for (QuotaSettings settings : admin.getQuota(filter).get()) {
202      LOG.debug(settings);
203      count++;
204    }
205    return count;
206  }
