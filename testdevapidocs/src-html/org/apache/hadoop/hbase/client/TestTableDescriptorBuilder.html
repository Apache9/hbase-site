099  @Test
100  public void testPb() throws DeserializationException, IOException {
101    final int v = 123;
102    TableDescriptor htd
103      = TableDescriptorBuilder.newBuilder(TableName.META_TABLE_NAME)
104          .setMaxFileSize(v)
105          .setDurability(Durability.ASYNC_WAL)
106          .setReadOnly(true)
107          .setRegionReplication(2)
108          .build();
109
110    byte [] bytes = TableDescriptorBuilder.toByteArray(htd);

115    assertEquals(Durability.ASYNC_WAL, deserializedHtd.getDurability());

139   * Test cps in the table description
140   * @throws Exception
141   */
142  @Test
143  public void testSetListRemoveCP() throws Exception {
144    TableDescriptor desc
145      = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName())).build();
146    // Check that any coprocessor is present.
147    assertTrue(desc.getCoprocessors().isEmpty());
148
149    // simple CP
150    String className1 = "org.apache.hadoop.hbase.coprocessor.SimpleRegionObserver";
151    String className2 = "org.apache.hadoop.hbase.coprocessor.SampleRegionWALObserver";
152    desc = TableDescriptorBuilder.newBuilder(desc)
153            .addCoprocessor(className1) // Add the 1 coprocessor and check if present.
154            .build();
155    assertTrue(desc.getCoprocessors().size() == 1);
156    assertTrue(desc.getCoprocessors().contains(className1));
157
158    desc = TableDescriptorBuilder.newBuilder(desc)
159            // Add the 2nd coprocessor and check if present.
160            // remove it and check that it is gone
161            .addCoprocessor(className2)
162            .build();
163    assertTrue(desc.getCoprocessors().size() == 2);
164    assertTrue(desc.getCoprocessors().contains(className2));
165
166    desc = TableDescriptorBuilder.newBuilder(desc)
167            // Remove one and check
168            .removeCoprocessor(className1)
169            .build();
170    assertTrue(desc.getCoprocessors().size() == 1);
171    assertFalse(desc.getCoprocessors().contains(className1));
172    assertTrue(desc.getCoprocessors().contains(className2));
173
174    desc = TableDescriptorBuilder.newBuilder(desc)
175            // Remove the last and check
176            .removeCoprocessor(className2)
177            .build();
178    assertTrue(desc.getCoprocessors().isEmpty());
179    assertFalse(desc.getCoprocessors().contains(className1));
180    assertFalse(desc.getCoprocessors().contains(className2));
181  }

189    byte[] key = Bytes.toBytes("Some");
190    byte[] value = Bytes.toBytes("value");

193            .setValue(key, value)

195    assertTrue(Bytes.equals(value, desc.getValue(key)));

198            .build();
199    assertTrue(desc.getValue(key) == null);

279  @Test
280  public void testModifyFamily() {
281    byte[] familyName = Bytes.toBytes("cf");
282    ColumnFamilyDescriptor hcd = ColumnFamilyDescriptorBuilder.newBuilder(familyName)
283            .setBlocksize(1000)
284            .setDFSReplication((short) 3)
285            .build();
286    TableDescriptor htd
287      = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
288              .addColumnFamily(hcd)
289              .build();
290
291    assertEquals(1000, htd.getColumnFamily(familyName).getBlocksize());
292    assertEquals(3, htd.getColumnFamily(familyName).getDFSReplication());
293    hcd = ColumnFamilyDescriptorBuilder.newBuilder(familyName)
294            .setBlocksize(2000)
295            .setDFSReplication((short) 1)
296            .build();
297    htd = TableDescriptorBuilder.newBuilder(htd)
298              .modifyColumnFamily(hcd)
299              .build();
300    assertEquals(2000, htd.getColumnFamily(familyName).getBlocksize());
301    assertEquals(1, htd.getColumnFamily(familyName).getDFSReplication());
302  }

339  @Test
340  public void testSerialReplicationScope() {
341    HColumnDescriptor hcdWithScope = new HColumnDescriptor(Bytes.toBytes("cf0"));
342    hcdWithScope.setScope(HConstants.REPLICATION_SCOPE_SERIAL);
343    HColumnDescriptor hcdWithoutScope = new HColumnDescriptor(Bytes.toBytes("cf1"));
344    TableDescriptor htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
345            .addColumnFamily(hcdWithoutScope)
346            .build();
347    assertFalse(htd.hasSerialReplicationScope());
348
349    htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
350            .addColumnFamily(hcdWithScope)
351            .build();
352    assertTrue(htd.hasSerialReplicationScope());
353
354    htd = TableDescriptorBuilder.newBuilder(TableName.valueOf(name.getMethodName()))
355            .addColumnFamily(hcdWithScope)
356            .addColumnFamily(hcdWithoutScope)
357            .build();
358    assertTrue(htd.hasSerialReplicationScope());
359  }
