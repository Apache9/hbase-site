047@RunWith(Parameterized.class)
048@Category({ MiscTests.class, MediumTests.class })
049public class TestAsyncClusterAdminApi extends TestAsyncAdminBase {
050
051  @Test
052  public void testRegionLoad() throws Exception {
053    // Turn off the balancer
054    admin.setBalancerOn(false).join();
055    TableName[] tables =
056        new TableName[] { TableName.valueOf(tableName.getNameAsString() + "1"),
057            TableName.valueOf(tableName.getNameAsString() + "2"),
058            TableName.valueOf(tableName.getNameAsString() + "3") };
059    createAndLoadTable(tables);
060    // Check if regions match with the regionLoad from the server
061    Collection<ServerName> servers = admin.getRegionServers().get();
062    for (ServerName serverName : servers) {
063      List<HRegionInfo> regions = admin.getOnlineRegions(serverName).get();
064      checkRegionsAndRegionLoads(regions, admin.getRegionLoads(serverName).get());
065    }
066
067    // Check if regionLoad matches the table's regions and nothing is missed
068    for (TableName table : tables) {
069      List<HRegionInfo> tableRegions = admin.getTableRegions(table).get();
070      List<RegionLoad> regionLoads = Lists.newArrayList();
071      for (ServerName serverName : servers) {
072        regionLoads.addAll(admin.getRegionLoads(serverName, Optional.of(table)).get());
073      }
074      checkRegionsAndRegionLoads(tableRegions, regionLoads);
075    }
076
077    // Check RegionLoad matches the regionLoad from ClusterStatus
078    ClusterStatus clusterStatus = admin.getClusterStatus().get();
079    for (ServerName serverName : clusterStatus.getServers()) {
080      ServerLoad serverLoad = clusterStatus.getLoad(serverName);
081      compareRegionLoads(serverLoad.getRegionsLoad().values(), admin.getRegionLoads(serverName)
082          .get());
083    }
084  }
085
086  private void compareRegionLoads(Collection<RegionLoad> regionLoadCluster,
087      Collection<RegionLoad> regionLoads) {
088
089    assertEquals("No of regionLoads from clusterStatus and regionloads from RS doesn't match",
090      regionLoadCluster.size(), regionLoads.size());
091
092    for (RegionLoad loadCluster : regionLoadCluster) {
093      boolean matched = false;
094      for (RegionLoad load : regionLoads) {
095        if (Bytes.equals(loadCluster.getName(), load.getName())) {
096          matched = true;
097          continue;
098        }
099      }
100      assertTrue("The contents of region load from cluster and server should match", matched);
101    }
102  }
103
104  private void checkRegionsAndRegionLoads(Collection<HRegionInfo> regions,
105      Collection<RegionLoad> regionLoads) {
106
107    assertEquals("No of regions and regionloads doesn't match", regions.size(), regionLoads.size());
108
109    Map<byte[], RegionLoad> regionLoadMap = Maps.newTreeMap(Bytes.BYTES_COMPARATOR);
110    for (RegionLoad regionLoad : regionLoads) {
111      regionLoadMap.put(regionLoad.getName(), regionLoad);
112    }
113    for (HRegionInfo info : regions) {
114      assertTrue("Region not in regionLoadMap region:" + info.getRegionNameAsString()
115          + " regionMap: " + regionLoadMap, regionLoadMap.containsKey(info.getRegionName()));
116    }
117  }
118
119  private void createAndLoadTable(TableName[] tables) {
120    for (TableName table : tables) {
121      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(table);
122      builder.addColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY).build());
123      admin.createTable(builder.build(), Bytes.toBytes("aaaaa"), Bytes.toBytes("zzzzz"), 16).join();
124      RawAsyncTable asyncTable = ASYNC_CONN.getRawTable(table);
125      List<Put> puts = new ArrayList<>();
126      for (byte[] row : HBaseTestingUtility.ROWS) {
127        puts.add(new Put(row).addColumn(FAMILY, Bytes.toBytes("q"), Bytes.toBytes("v")));
128      }
129      asyncTable.putAll(puts).join();
130    }
131  }
