124  @Test
125  public void testGetRegionLoads() throws Exception {
126    // Turn off the balancer
127    admin.setBalancerOn(false).join();
128    TableName[] tables =
129        new TableName[] { TableName.valueOf(tableName.getNameAsString() + "1"),
130            TableName.valueOf(tableName.getNameAsString() + "2"),
131            TableName.valueOf(tableName.getNameAsString() + "3") };
132    createAndLoadTable(tables);
133    // Sleep to wait region server report
134    Thread.sleep(TEST_UTIL.getConfiguration().getInt("hbase.regionserver.msginterval", 3 * 1000) * 2);
135    // Check if regions match with the regionLoad from the server
136    Collection<ServerName> servers = admin.getRegionServers().get();
137    for (ServerName serverName : servers) {
138      List<HRegionInfo> regions = admin.getOnlineRegions(serverName).get();
139      checkRegionsAndRegionLoads(regions, admin.getRegionLoads(serverName).get());
140    }
141
142    // Check if regionLoad matches the table's regions and nothing is missed
143    for (TableName table : tables) {
144      List<HRegionInfo> tableRegions = admin.getTableRegions(table).get();
145      List<RegionLoad> regionLoads = Lists.newArrayList();
146      for (ServerName serverName : servers) {
147        regionLoads.addAll(admin.getRegionLoads(serverName, Optional.of(table)).get());
148      }
149      checkRegionsAndRegionLoads(tableRegions, regionLoads);
150    }
151
152    // Check RegionLoad matches the regionLoad from ClusterStatus
153    ClusterStatus clusterStatus = admin.getClusterStatus().get();
154    for (ServerName serverName : clusterStatus.getServers()) {
155      ServerLoad serverLoad = clusterStatus.getLoad(serverName);
156      compareRegionLoads(serverLoad.getRegionsLoad().values(), admin.getRegionLoads(serverName)
157          .get());
158    }
159  }
160
161  private void compareRegionLoads(Collection<RegionLoad> regionLoadCluster,
162      Collection<RegionLoad> regionLoads) {
163
164    assertEquals("No of regionLoads from clusterStatus and regionloads from RS doesn't match",
165      regionLoadCluster.size(), regionLoads.size());
166
167    for (RegionLoad loadCluster : regionLoadCluster) {
168      boolean matched = false;
169      for (RegionLoad load : regionLoads) {
170        if (Bytes.equals(loadCluster.getName(), load.getName())) {
171          matched = true;
172          continue;
173        }
174      }
175      assertTrue("The contents of region load from cluster and server should match", matched);
176    }
177  }
178
179  private void checkRegionsAndRegionLoads(Collection<HRegionInfo> regions,
180      Collection<RegionLoad> regionLoads) {
181
182    assertEquals("No of regions and regionloads doesn't match", regions.size(), regionLoads.size());
183
184    Map<byte[], RegionLoad> regionLoadMap = Maps.newTreeMap(Bytes.BYTES_COMPARATOR);
185    for (RegionLoad regionLoad : regionLoads) {
186      regionLoadMap.put(regionLoad.getName(), regionLoad);
187    }
188    for (HRegionInfo info : regions) {
189      assertTrue("Region not in regionLoadMap region:" + info.getRegionNameAsString()
190          + " regionMap: " + regionLoadMap, regionLoadMap.containsKey(info.getRegionName()));
191    }
192  }
193
194  private void createAndLoadTable(TableName[] tables) {
195    for (TableName table : tables) {
196      TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(table);
197      builder.addColumnFamily(ColumnFamilyDescriptorBuilder.newBuilder(FAMILY).build());
198      admin.createTable(builder.build(), Bytes.toBytes("aaaaa"), Bytes.toBytes("zzzzz"), 16).join();
199      RawAsyncTable asyncTable = ASYNC_CONN.getRawTable(table);
200      List<Put> puts = new ArrayList<>();
201      for (byte[] row : HBaseTestingUtility.ROWS) {
202        puts.add(new Put(row).addColumn(FAMILY, Bytes.toBytes("q"), Bytes.toBytes("v")));
203      }
204      asyncTable.putAll(puts).join();
205    }
206  }
