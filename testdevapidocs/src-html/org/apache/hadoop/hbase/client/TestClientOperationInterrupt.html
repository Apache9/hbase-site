050@Category({MediumTests.class, ClientTests.class})
051public class TestClientOperationInterrupt {
052  private static final Log LOG = LogFactory.getLog(TestClientOperationInterrupt.class);
053
054  private static HBaseTestingUtility util;
055  private static final TableName tableName = TableName.valueOf("test");
056  private static final byte[] dummy = Bytes.toBytes("dummy");
057  private static final byte[] row1 = Bytes.toBytes("r1");
058  private static final byte[] test = Bytes.toBytes("test");
059  private static Configuration conf;
060
061  public static class TestCoprocessor extends BaseRegionObserver {
062    @Override
063    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
064                         final Get get, final List<Cell> results) throws IOException {
065      Threads.sleep(2500);
066    }
067  }
068
069
070  @BeforeClass
071  public static void setUpBeforeClass() throws Exception {
072    conf = HBaseConfiguration.create();
073    conf.setStrings(CoprocessorHost.USER_REGION_COPROCESSOR_CONF_KEY,
074        TestCoprocessor.class.getName());
075    util = new HBaseTestingUtility(conf);
076    util.startMiniCluster();
077
078    Admin admin = util.getHBaseAdmin();
079    if (admin.tableExists(tableName)) {
080      if (admin.isTableEnabled(tableName)) {
081        admin.disableTable(tableName);
082      }
083      admin.deleteTable(tableName);
084    }
085    Table ht = util.createTable(tableName, new byte[][]{dummy, test});
086
087    Put p = new Put(row1);
088    p.addColumn(dummy, dummy, dummy);
089    ht.put(p);
090  }
091
092
093  @Test
094  public void testInterrupt50Percent() throws IOException, InterruptedException {
095    final AtomicInteger noEx = new AtomicInteger(0);
096    final AtomicInteger badEx = new AtomicInteger(0);
097    final AtomicInteger noInt = new AtomicInteger(0);
098    final AtomicInteger done = new AtomicInteger(0);
099    List<Thread> threads = new ArrayList<Thread>();
100
101    final int nbThread = 100;
102
103    for (int i = 0; i < nbThread; i++) {
104      Thread t = new Thread() {
105        @Override
106        public void run() {
107          try {
108            Table ht = util.getConnection().getTable(tableName);
109            Result r = ht.get(new Get(row1));
110            noEx.incrementAndGet();
111          } catch (IOException e) {
112            LOG.info("exception", e);
113            if (!(e instanceof InterruptedIOException) || (e instanceof SocketTimeoutException)) {
114              badEx.incrementAndGet();
115            } else {
116              if (Thread.currentThread().isInterrupted()) {
117                noInt.incrementAndGet();
118                LOG.info("The thread should NOT be with the 'interrupt' status.");
119              }
120            }
121          } finally {
122            done.incrementAndGet();
123          }
124        }
125      };
126      t.setName("TestClientOperationInterrupt #" + i);
127      threads.add(t);
128      t.start();
129    }
130
131    for (int i = 0; i < nbThread / 2; i++) {
132      threads.get(i).interrupt();
133    }
134
135
136    boolean stillAlive = true;
137    while (stillAlive) {
138      stillAlive = false;
139      for (Thread t : threads) {
140        if (t.isAlive()) {
141          stillAlive = true;
142        }
143      }
144      Threads.sleep(10);
145    }
146
147    Assert.assertFalse(Thread.currentThread().isInterrupted());
148
149    Assert.assertTrue(" noEx: " + noEx.get() + ", badEx=" + badEx.get() + ", noInt=" + noInt.get(),
150        noEx.get() == nbThread / 2 && badEx.get() == 0);
151
152    // The problem here is that we need the server to free its handlers to handle all operations
153    while (done.get() != nbThread){
154      Thread.sleep(1);
155    }
156
157    Table ht = util.getConnection().getTable(tableName);
158    Result r = ht.get(new Get(row1));
159    Assert.assertFalse(r.isEmpty());
160  }

162  @AfterClass

164    util.shutdownMiniCluster();
165  }
