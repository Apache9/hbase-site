209  public static class SleepLongerAtFirstCoprocessor implements RegionObserver {
210    public static final int SLEEP_TIME = 2000;
211    static final AtomicLong ct = new AtomicLong(0);
212    @Override
213    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
214        final Get get, final List<Cell> results) throws IOException {
215      // After first sleep, all requests are timeout except the last retry. If we handle
216      // all the following requests, finally the last request is also timeout. If we drop all
217      // timeout requests, we can handle the last request immediately and it will not timeout.
218      if (ct.incrementAndGet() <= 1) {
219        Threads.sleep(SLEEP_TIME * RPC_RETRY * 2);
220      } else {
221        Threads.sleep(SLEEP_TIME);
222      }
223    }
224  }

245    ThreadPoolExecutor otherPool = new ThreadPoolExecutor(1, 1,
246        5, TimeUnit.SECONDS,
247        new SynchronousQueue<>(),
248        Threads.newDaemonThreadFactory("test-hcm"));
249
250    Connection con1 = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration());
251    Connection con2 = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration(), otherPool);
252    // make sure the internally created ExecutorService is the one passed
253    assertTrue(otherPool == ((ConnectionImplementation) con2).getCurrentBatchPool());
254
255    final TableName tableName = TableName.valueOf(name.getMethodName());
256    TEST_UTIL.createTable(tableName, FAM_NAM).close();
257    Table table = con1.getTable(tableName, otherPool);
258
259    ExecutorService pool = null;
260
261    if(table instanceof HTable) {
262      HTable t = (HTable) table;
263      // make sure passing a pool to the getTable does not trigger creation of an internal pool
264      assertNull("Internal Thread pool should be null",
265        ((ConnectionImplementation) con1).getCurrentBatchPool());
266      // table should use the pool passed
267      assertTrue(otherPool == t.getPool());
268      t.close();
269
270      t = (HTable) con2.getTable(tableName);
271      // table should use the connectin's internal pool
272      assertTrue(otherPool == t.getPool());
273      t.close();
274
275      t = (HTable) con2.getTable(tableName);
276      // try other API too
277      assertTrue(otherPool == t.getPool());
278      t.close();
279
280      t = (HTable) con2.getTable(tableName);
281      // try other API too
282      assertTrue(otherPool == t.getPool());
283      t.close();
284
285      t = (HTable) con1.getTable(tableName);
286      pool = ((ConnectionImplementation) con1).getCurrentBatchPool();
287      // make sure an internal pool was created
288      assertNotNull("An internal Thread pool should have been created", pool);
289      // and that the table is using it
290      assertTrue(t.getPool() == pool);
291      t.close();
292
293      t = (HTable) con1.getTable(tableName);
294      // still using the *same* internal pool
295      assertTrue(t.getPool() == pool);
296      t.close();
297    } else {
298      table.close();
299    }
300
301    con1.close();
302
303    // if the pool was created on demand it should be closed upon connection close
304    if(pool != null) {
305      assertTrue(pool.isShutdown());
306    }
307
308    con2.close();
309    // if the pool is passed, it is not closed
310    assertFalse(otherPool.isShutdown());
311    otherPool.shutdownNow();
312  }

395   * Test that an operation can fail if we read the global operation timeout, even if the
396   * individual timeout is fine. We do that with:
397   * - client side: an operation timeout of 30 seconds
398   * - server side: we sleep 20 second at each attempt. The first work fails, the second one
399   * succeeds. But the client won't wait that much, because 20 + 20 > 30, so the client
400   * timeouted when the server answers.
401   */
402  @Test
403  public void testGetOperationTimeout() throws Exception {
404    HTableDescriptor hdt = TEST_UTIL.createTableDescriptor(TableName.valueOf(name.getMethodName()));
405    hdt.addCoprocessor(SleepAndFailFirstTime.class.getName());
406    Table table = TEST_UTIL.createTable(hdt, new byte[][]{FAM_NAM}, TEST_UTIL.getConfiguration());
407    table.setRpcTimeout(Integer.MAX_VALUE);
408    SleepAndFailFirstTime.ct.set(0);
409    // Check that it works if the timeout is big enough
410    table.setOperationTimeout(120 * 1000);
411    table.get(new Get(FAM_NAM));
412
413    // Resetting and retrying. Will fail this time, not enough time for the second try
414    SleepAndFailFirstTime.ct.set(0);
415    try {
416      table.setOperationTimeout(30 * 1000);
417      table.get(new Get(FAM_NAM));
418      Assert.fail("We expect an exception here");
419    } catch (SocketTimeoutException e) {
420      // The client has a CallTimeout class, but it's not shared.We're not very clean today,
421      //  in the general case you can expect the call to stop, but the exception may vary.
422      // In this test however, we're sure that it will be a socket timeout.
423      LOG.info("We received an exception, as expected ", e);
424    } catch (IOException e) {
425      Assert.fail("Wrong exception:" + e.getMessage());
426    } finally {
427      table.close();
428    }
429  }
