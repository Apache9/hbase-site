269  private void splitTests(TableName tableName, int rowCount, boolean isSplitRegion,
270      byte[] splitPoint) throws Exception {
271    int count = 0;
272    // create table
273    HColumnDescriptor cd = new HColumnDescriptor("d");
274    HTableDescriptor td = new HTableDescriptor(tableName);
275    td.addFamily(cd);
276    Table table = TEST_UTIL.createTable(td, null);
277    TEST_UTIL.waitTableAvailable(tableName);
278
279    List<HRegionInfo> regions = TEST_UTIL.getAdmin().getTableRegions(tableName);
280    assertEquals(regions.size(), 1);
281
282    List<Put> puts = new ArrayList<>();
283    for (int i = 0; i < rowCount; i++) {
284      Put put = new Put(Bytes.toBytes(i));
285      put.addColumn(Bytes.toBytes("d"), null, Bytes.toBytes("value" + i));
286      puts.add(put);
287    }
288    table.put(puts);
289
290    if (isSplitRegion) {
291      admin.splitRegion(regions.get(0).getRegionName(), splitPoint).get();
292    } else {
293      if (splitPoint == null) {
294        admin.split(tableName).get();
295      } else {
296        admin.split(tableName, splitPoint).get();
297      }
298    }
299
300    for (int i = 0; i < 45; i++) {
301      try {
302        List<HRegionInfo> hRegionInfos = TEST_UTIL.getAdmin().getTableRegions(tableName);
303        count = hRegionInfos.size();
304        if (count >= 2) {
305          break;
306        }
307        Thread.sleep(1000L);
308      } catch (Exception e) {
309        LOG.error(e);
310      }
311    }
312
313    assertEquals(count, 2);
314  }

320      // create test table
321      HTableDescriptor desc = new HTableDescriptor(tableName);
322      desc.addFamily(new HColumnDescriptor(FAMILY));
323      admin.createTable(desc).get();
324
325      // add region to meta.
326      Table meta = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME);
327      HRegionInfo hri =
328          new HRegionInfo(desc.getTableName(), Bytes.toBytes("A"), Bytes.toBytes("Z"));
329      MetaTableAccessor.addRegionToMeta(meta, hri);
330
331      // assign region.
332      HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
333      AssignmentManager am = master.getAssignmentManager();
334      admin.assign(hri.getRegionName()).get();
335      am.waitForAssignment(hri);
336
337      // assert region on server
338      RegionStates regionStates = am.getRegionStates();
339      ServerName serverName = regionStates.getRegionServerOfRegion(hri);

345      admin.assign(hri.getRegionName()).get();
346      am.waitForAssignment(hri);
347      assertTrue(regionStates.getRegionState(hri).isOpened());
348
349      // unassign region
350      admin.unassign(hri.getRegionName(), true).get();
351      am.waitForAssignment(hri);

561  private void compactionTest(final String tableName, final int flushes,
562      final CompactionState expectedState, boolean singleFamily) throws Exception {
563    // Create a table with regions
564    final TableName table = TableName.valueOf(tableName);
565    byte[] family = Bytes.toBytes("family");
566    byte[][] families =
567        { family, Bytes.add(family, Bytes.toBytes("2")), Bytes.add(family, Bytes.toBytes("3")) };
568    Table ht = null;
569    try {
570      ht = TEST_UTIL.createTable(table, families);
571      loadData(ht, families, 3000, flushes);
572      List<Region> regions = new ArrayList<>();
573      TEST_UTIL.getHBaseCluster().getLiveRegionServerThreads()
574          .forEach(rsThread -> regions.addAll(rsThread.getRegionServer().getOnlineRegions(table)));
575      Assert.assertEquals(regions.size(), 1);
576      int countBefore = countStoreFilesInFamilies(regions, families);
577      int countBeforeSingleFamily = countStoreFilesInFamily(regions, family);
578      assertTrue(countBefore > 0); // there should be some data files
579      if (expectedState == CompactionState.MINOR) {
580        if (singleFamily) {
581          admin.compact(table, family).get();
582        } else {
583          admin.compact(table).get();
584        }
585      } else {
586        if (singleFamily) {
587          admin.majorCompact(table, family).get();
588        } else {
589          admin.majorCompact(table).get();
590        }
591      }
592      long curt = System.currentTimeMillis();
593      long waitTime = 5000;
594      long endt = curt + waitTime;
595      CompactionState state = TEST_UTIL.getAdmin().getCompactionState(table);
596      while (state == CompactionState.NONE && curt < endt) {
597        Thread.sleep(10);
598        state = TEST_UTIL.getAdmin().getCompactionState(table);
599        curt = System.currentTimeMillis();
600      }
601      // Now, should have the right compaction state,
602      // otherwise, the compaction should have already been done
603      if (expectedState != state) {
604        for (Region region : regions) {
605          state = CompactionState.valueOf(region.getCompactionState().toString());
606          assertEquals(CompactionState.NONE, state);
607        }
608      } else {
609        // Wait until the compaction is done
610        state = TEST_UTIL.getAdmin().getCompactionState(table);
611        while (state != CompactionState.NONE && curt < endt) {
612          Thread.sleep(10);
613          state = TEST_UTIL.getAdmin().getCompactionState(table);
614        }
615        // Now, compaction should be done.
616        assertEquals(CompactionState.NONE, state);
617      }
618      int countAfter = countStoreFilesInFamilies(regions, families);
619      int countAfterSingleFamily = countStoreFilesInFamily(regions, family);
620      assertTrue(countAfter < countBefore);
621      if (!singleFamily) {
622        if (expectedState == CompactionState.MAJOR) assertTrue(families.length == countAfter);
623        else assertTrue(families.length < countAfter);
624      } else {
625        int singleFamDiff = countBeforeSingleFamily - countAfterSingleFamily;
626        // assert only change was to single column family
627        assertTrue(singleFamDiff == (countBefore - countAfter));
628        if (expectedState == CompactionState.MAJOR) {
629          assertTrue(1 == countAfterSingleFamily);
630        } else {
631          assertTrue(1 < countAfterSingleFamily);
632        }
633      }
634    } finally {
635      if (ht != null) {
636        TEST_UTIL.deleteTable(table);
637      }
638    }
639  }

641  private static int countStoreFilesInFamily(List<Region> regions, final byte[] family) {
642    return countStoreFilesInFamilies(regions, new byte[][] { family });
643  }
644
645  private static int countStoreFilesInFamilies(List<Region> regions, final byte[][] families) {
646    int count = 0;
647    for (Region region : regions) {
648      count += region.getStoreFileList(families).size();
649    }
650    return count;
651  }

653  private static void loadData(final Table ht, final byte[][] families, final int rows,
654      final int flushes) throws IOException {
655    List<Put> puts = new ArrayList<>(rows);
656    byte[] qualifier = Bytes.toBytes("val");
657    for (int i = 0; i < flushes; i++) {
658      for (int k = 0; k < rows; k++) {
659        byte[] row = Bytes.toBytes(RANDOM.nextLong());
660        Put p = new Put(row);
661        for (int j = 0; j < families.length; ++j) {
662          p.addColumn(families[j], qualifier, row);
663        }
664        puts.add(p);
665      }
666      ht.put(puts);
667      TEST_UTIL.flush();
668      puts.clear();
669    }
670  }
