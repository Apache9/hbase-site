258    HRegionInfo hri = createTableAndGetOneRegion(tableName);
259    RawAsyncHBaseAdmin rawAdmin = (RawAsyncHBaseAdmin) ASYNC_CONN.getAdmin();

280      ServerName sn = rawAdmin.getRegionLocation(hri.getRegionName()).get().getServerName();

283      }

397          throws Exception {
398    // create table
399    createTableWithDefaultConf(tableName);
400
401    RawAsyncTable metaTable = ASYNC_CONN.getRawTable(META_TABLE_NAME);
402    List<HRegionLocation> regionLocations =
403        AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName)).get();
404    assertEquals(1, regionLocations.size());
405
406    RawAsyncTable table = ASYNC_CONN.getRawTable(tableName);
407    List<Put> puts = new ArrayList<>();
408    for (int i = 0; i < rowCount; i++) {
409      Put put = new Put(Bytes.toBytes(i));
410      put.addColumn(FAMILY, null, Bytes.toBytes("value" + i));
411      puts.add(put);
412    }
413    table.putAll(puts).join();
414
415    if (isSplitRegion) {
416      admin.splitRegion(regionLocations.get(0).getRegionInfo().getRegionName(),
417        Optional.ofNullable(splitPoint)).get();
418    } else {
419      if (splitPoint == null) {
420        admin.split(tableName).get();
421      } else {
422        admin.split(tableName, splitPoint).get();
423      }
424    }
425
426    int count = 0;
427    for (int i = 0; i < 45; i++) {
428      try {
429        regionLocations =
430            AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName))
431                .get();
432        count = regionLocations.size();
433        if (count >= 2) {
434          break;
435        }
436        Thread.sleep(1000L);
437      } catch (Exception e) {
438        LOG.error(e);
439      }
440    }
441    assertEquals(count, 2);
442  }

482  private void compactionTest(final TableName tableName, final int flushes,
483      final CompactionState expectedState, boolean singleFamily) throws Exception {
484    // Create a table with regions
485    byte[] family = Bytes.toBytes("family");
486    byte[][] families =
487        { family, Bytes.add(family, Bytes.toBytes("2")), Bytes.add(family, Bytes.toBytes("3")) };
488    createTableWithDefaultConf(tableName, Optional.empty(), families);
489    loadData(tableName, families, 3000, flushes);
490
491    List<Region> regions = new ArrayList<>();
492    TEST_UTIL
493        .getHBaseCluster()
494        .getLiveRegionServerThreads()
495        .forEach(rsThread -> regions.addAll(rsThread.getRegionServer().getOnlineRegions(tableName)));
496    Assert.assertEquals(regions.size(), 1);
497
498    int countBefore = countStoreFilesInFamilies(regions, families);
499    int countBeforeSingleFamily = countStoreFilesInFamily(regions, family);
500    assertTrue(countBefore > 0); // there should be some data files
501    if (expectedState == CompactionState.MINOR) {
502      if (singleFamily) {
503        admin.compact(tableName, Optional.of(family)).get();
504      } else {
505        admin.compact(tableName, Optional.empty()).get();
506      }
507    } else {
508      if (singleFamily) {
509        admin.majorCompact(tableName, Optional.of(family)).get();
510      } else {
511        admin.majorCompact(tableName, Optional.empty()).get();
512      }
513    }
514
515    long curt = System.currentTimeMillis();
516    long waitTime = 5000;
517    long endt = curt + waitTime;
518    CompactionState state = admin.getCompactionState(tableName).get();
519    while (state == CompactionState.NONE && curt < endt) {
520      Thread.sleep(10);
521      state = admin.getCompactionState(tableName).get();
522      curt = System.currentTimeMillis();
523    }
524    // Now, should have the right compaction state,
525    // otherwise, the compaction should have already been done
526    if (expectedState != state) {
527      for (Region region : regions) {
528        state = CompactionState.valueOf(region.getCompactionState().toString());
529        assertEquals(CompactionState.NONE, state);
530      }
531    } else {
532      // Wait until the compaction is done
533      state = admin.getCompactionState(tableName).get();
534      while (state != CompactionState.NONE && curt < endt) {
535        Thread.sleep(10);
536        state = admin.getCompactionState(tableName).get();
537      }
538      // Now, compaction should be done.
539      assertEquals(CompactionState.NONE, state);
540    }
541
542    int countAfter = countStoreFilesInFamilies(regions, families);
543    int countAfterSingleFamily = countStoreFilesInFamily(regions, family);
544    assertTrue(countAfter < countBefore);
545    if (!singleFamily) {
546      if (expectedState == CompactionState.MAJOR) assertTrue(families.length == countAfter);
547      else assertTrue(families.length < countAfter);
548    } else {
549      int singleFamDiff = countBeforeSingleFamily - countAfterSingleFamily;
550      // assert only change was to single column family
551      assertTrue(singleFamDiff == (countBefore - countAfter));
552      if (expectedState == CompactionState.MAJOR) {
553        assertTrue(1 == countAfterSingleFamily);
554      } else {
555        assertTrue(1 < countAfterSingleFamily);
556      }
557    }
558  }

560  private static int countStoreFilesInFamily(List<Region> regions, final byte[] family) {
561    return countStoreFilesInFamilies(regions, new byte[][] { family });
562  }
563
564  private static int countStoreFilesInFamilies(List<Region> regions, final byte[][] families) {
565    int count = 0;
566    for (Region region : regions) {
567      count += region.getStoreFileList(families).size();
568    }
569    return count;
570  }

572  private static void loadData(final TableName tableName, final byte[][] families, final int rows,
573      final int flushes) throws IOException {
574    RawAsyncTable table = ASYNC_CONN.getRawTable(tableName);
575    List<Put> puts = new ArrayList<>(rows);
576    byte[] qualifier = Bytes.toBytes("val");
577    for (int i = 0; i < flushes; i++) {
578      for (int k = 0; k < rows; k++) {
579        byte[] row = Bytes.toBytes(RANDOM.nextLong());
580        Put p = new Put(row);
581        for (int j = 0; j < families.length; ++j) {
582          p.addColumn(families[j], qualifier, row);
583        }
584        puts.add(p);
585      }
586      table.putAll(puts).join();
587      TEST_UTIL.flush();
588      puts.clear();
589    }
590  }
