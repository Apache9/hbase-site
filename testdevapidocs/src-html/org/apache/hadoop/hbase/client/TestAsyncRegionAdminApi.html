380      splitTest(TableName tableName, int rowCount, boolean isSplitRegion, byte[] splitPoint)
381          throws Exception {
382    // create table
383    createTableWithDefaultConf(tableName);
384
385    AsyncTable<AdvancedScanResultConsumer> metaTable = ASYNC_CONN.getTable(META_TABLE_NAME);
386    List<HRegionLocation> regionLocations =
387        AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName)).get();
388    assertEquals(1, regionLocations.size());
389
390    AsyncTable<?> table = ASYNC_CONN.getTable(tableName);
391    List<Put> puts = new ArrayList<>();
392    for (int i = 0; i < rowCount; i++) {
393      Put put = new Put(Bytes.toBytes(i));
394      put.addColumn(FAMILY, null, Bytes.toBytes("value" + i));
395      puts.add(put);
396    }
397    table.putAll(puts).join();
398
399    if (isSplitRegion) {
400      if (splitPoint == null) {
401        admin.splitRegion(regionLocations.get(0).getRegion().getRegionName()).get();
402      } else {
403        admin.splitRegion(regionLocations.get(0).getRegion().getRegionName(), splitPoint).get();
404      }
405    } else {
406      if (splitPoint == null) {
407        admin.split(tableName).get();
408      } else {
409        admin.split(tableName, splitPoint).get();
410      }
411    }
412
413    int count = 0;
414    for (int i = 0; i < 45; i++) {
415      try {
416        regionLocations =
417            AsyncMetaTableAccessor.getTableHRegionLocations(metaTable, Optional.of(tableName))
418                .get();
419        count = regionLocations.size();
420        if (count >= 2) {
421          break;
422        }
423        Thread.sleep(1000L);
424      } catch (Exception e) {
425        LOG.error(e.toString(), e);
426      }
427    }
428    assertEquals(2, count);
429  }

505  private void compactionTest(final TableName tableName, final int flushes,
506      final CompactionState expectedState, boolean singleFamily) throws Exception {
507    // Create a table with regions
508    byte[] family = Bytes.toBytes("family");
509    byte[][] families =
510        { family, Bytes.add(family, Bytes.toBytes("2")), Bytes.add(family, Bytes.toBytes("3")) };
511    createTableWithDefaultConf(tableName, null, families);
512    loadData(tableName, families, 3000, flushes);
513
514    List<Region> regions = new ArrayList<>();
515    TEST_UTIL
516        .getHBaseCluster()
517        .getLiveRegionServerThreads()
518        .forEach(rsThread -> regions.addAll(rsThread.getRegionServer().getRegions(tableName)));
519    assertEquals(1, regions.size());
520
521    int countBefore = countStoreFilesInFamilies(regions, families);
522    int countBeforeSingleFamily = countStoreFilesInFamily(regions, family);
523    assertTrue(countBefore > 0); // there should be some data files
524    if (expectedState == CompactionState.MINOR) {
525      if (singleFamily) {
526        admin.compact(tableName, family).get();
527      } else {
528        admin.compact(tableName).get();
529      }
530    } else {
531      if (singleFamily) {
532        admin.majorCompact(tableName, family).get();
533      } else {
534        admin.majorCompact(tableName).get();
535      }
536    }
537
538    long curt = System.currentTimeMillis();
539    long waitTime = 5000;
540    long endt = curt + waitTime;
541    CompactionState state = admin.getCompactionState(tableName).get();
542    while (state == CompactionState.NONE && curt < endt) {
543      Thread.sleep(10);
544      state = admin.getCompactionState(tableName).get();
545      curt = System.currentTimeMillis();
546    }
547    // Now, should have the right compaction state,
548    // otherwise, the compaction should have already been done
549    if (expectedState != state) {
550      for (Region region : regions) {
551        state = CompactionState.valueOf(region.getCompactionState().toString());
552        assertEquals(CompactionState.NONE, state);
553      }
554    } else {
555      // Wait until the compaction is done
556      state = admin.getCompactionState(tableName).get();
557      while (state != CompactionState.NONE && curt < endt) {
558        Thread.sleep(10);
559        state = admin.getCompactionState(tableName).get();
560      }
561      // Now, compaction should be done.
562      assertEquals(CompactionState.NONE, state);
563    }
564
565    int countAfter = countStoreFilesInFamilies(regions, families);
566    int countAfterSingleFamily = countStoreFilesInFamily(regions, family);
567    assertTrue(countAfter < countBefore);
568    if (!singleFamily) {
569      if (expectedState == CompactionState.MAJOR) assertTrue(families.length == countAfter);
570      else assertTrue(families.length < countAfter);
571    } else {
572      int singleFamDiff = countBeforeSingleFamily - countAfterSingleFamily;
573      // assert only change was to single column family
574      assertTrue(singleFamDiff == (countBefore - countAfter));
575      if (expectedState == CompactionState.MAJOR) {
576        assertTrue(1 == countAfterSingleFamily);
577      } else {
578        assertTrue(1 < countAfterSingleFamily);
579      }
580    }
581  }

583  private static int countStoreFilesInFamily(List<Region> regions, final byte[] family) {
584    return countStoreFilesInFamilies(regions, new byte[][] { family });
585  }

587  private static int countStoreFilesInFamilies(List<Region> regions, final byte[][] families) {
588    int count = 0;
589    for (Region region : regions) {
590      count += region.getStoreFileList(families).size();
591    }
592    return count;
593  }

602    AsyncTable<?> table = ASYNC_CONN.getTable(tableName);
603    List<Put> puts = new ArrayList<>(rows);
604    byte[] qualifier = Bytes.toBytes("val");
605    for (int i = 0; i < flushes; i++) {
606      for (int k = 0; k < rows; k++) {
607        byte[] row = Bytes.add(Bytes.toBytes(k), Bytes.toBytes(i));
608        Put p = new Put(row);
609        for (int j = 0; j < families.length; ++j) {
610          p.addColumn(families[j], qualifier, row);
611        }
612        puts.add(p);
613      }
614      table.putAll(puts).join();
615      TEST_UTIL.flush();
616      puts.clear();
617    }
618  }
