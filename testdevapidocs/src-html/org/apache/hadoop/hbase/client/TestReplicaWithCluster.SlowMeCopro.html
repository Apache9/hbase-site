075  /**
076   * This copro is used to synchronize the tests.
077   */
078  public static class SlowMeCopro extends BaseRegionObserver {
079    static final AtomicLong sleepTime = new AtomicLong(0);
080    static final AtomicReference<CountDownLatch> cdl = new AtomicReference<>(new CountDownLatch(0));
081
082    public SlowMeCopro() {
083    }
084
085    @Override
086    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
087                         final Get get, final List<Cell> results) throws IOException {
088
089      if (e.getEnvironment().getRegion().getRegionInfo().getReplicaId() == 0) {
090        CountDownLatch latch = cdl.get();
091        try {
092          if (sleepTime.get() > 0) {
093            LOG.info("Sleeping for " + sleepTime.get() + " ms");
094            Thread.sleep(sleepTime.get());
095          } else if (latch.getCount() > 0) {
096            LOG.info("Waiting for the counterCountDownLatch");
097            latch.await(2, TimeUnit.MINUTES); // To help the tests to finish.
098            if (latch.getCount() > 0) {
099              throw new RuntimeException("Can't wait more");
100            }
101          }
102        } catch (InterruptedException e1) {
103          LOG.error(e1);
104        }
105      } else {
106        LOG.info("We're not the primary replicas.");
107      }
108    }
109  }

136  @Test (timeout=30000)
137  public void testCreateDeleteTable() throws IOException {
138    // Create table then get the single region for our new table.
139    HTableDescriptor hdt = HTU.createTableDescriptor("testCreateDeleteTable");
140    hdt.setRegionReplication(NB_SERVERS);
141    hdt.addCoprocessor(SlowMeCopro.class.getName());
142    Table table = HTU.createTable(hdt, new byte[][]{f}, null);
143
144    Put p = new Put(row);
145    p.addColumn(f, row, row);
146    table.put(p);
147
148    Get g = new Get(row);
149    Result r = table.get(g);
150    Assert.assertFalse(r.isStale());
151
152    try {
153      // But if we ask for stale we will get it
154      SlowMeCopro.cdl.set(new CountDownLatch(1));
155      g = new Get(row);
156      g.setConsistency(Consistency.TIMELINE);
157      r = table.get(g);
158      Assert.assertTrue(r.isStale());
159      SlowMeCopro.cdl.get().countDown();
160    } finally {
161      SlowMeCopro.cdl.get().countDown();
162      SlowMeCopro.sleepTime.set(0);
163    }
164
165    HTU.getHBaseAdmin().disableTable(hdt.getTableName());
166    HTU.deleteTable(hdt.getTableName());
167  }

173    hdt.addCoprocessor(SlowMeCopro.class.getName());

177    Put p = new Put(row);

179    table.put(p);
180
181    Get g = new Get(row);
182    Result r = table.get(g);
183    Assert.assertFalse(r.isStale());
184
185    // Add a CF, it should work.
186    HTableDescriptor bHdt = HTU.getHBaseAdmin().getTableDescriptor(hdt.getTableName());
187    HColumnDescriptor hcd = new HColumnDescriptor(row);
188    hdt.addFamily(hcd);
189    HTU.getHBaseAdmin().disableTable(hdt.getTableName());
190    HTU.getHBaseAdmin().modifyTable(hdt.getTableName(), hdt);
191    HTU.getHBaseAdmin().enableTable(hdt.getTableName());
192    HTableDescriptor nHdt = HTU.getHBaseAdmin().getTableDescriptor(hdt.getTableName());
193    Assert.assertEquals("fams=" + Arrays.toString(nHdt.getColumnFamilies()),
194        bHdt.getColumnFamilies().length + 1, nHdt.getColumnFamilies().length);
195
196    p = new Put(row);
197    p.addColumn(row, row, row);
198    table.put(p);
199
200    g = new Get(row);
201    r = table.get(g);
202    Assert.assertFalse(r.isStale());
203
204    try {
205      SlowMeCopro.cdl.set(new CountDownLatch(1));
206      g = new Get(row);
207      g.setConsistency(Consistency.TIMELINE);
208      r = table.get(g);
209      Assert.assertTrue(r.isStale());
210    } finally {
211      SlowMeCopro.cdl.get().countDown();
212      SlowMeCopro.sleepTime.set(0);
213    }
214
215    Admin admin = HTU.getHBaseAdmin();
216    nHdt =admin.getTableDescriptor(hdt.getTableName());

218        bHdt.getColumnFamilies().length + 1, nHdt.getColumnFamilies().length);
219
220    admin.disableTable(hdt.getTableName());
221    admin.deleteTable(hdt.getTableName());
222    admin.close();
223  }

226  @Test (timeout=300000)
227  public void testReplicaAndReplication() throws Exception {
228    HTableDescriptor hdt = HTU.createTableDescriptor("testReplicaAndReplication");
229    hdt.setRegionReplication(NB_SERVERS);
230
231    HColumnDescriptor fam = new HColumnDescriptor(row);
232    fam.setScope(HConstants.REPLICATION_SCOPE_GLOBAL);
233    hdt.addFamily(fam);
234
235    hdt.addCoprocessor(SlowMeCopro.class.getName());
236    HTU.getHBaseAdmin().createTable(hdt, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
237
238    Configuration conf2 = HBaseConfiguration.create(HTU.getConfiguration());
239    conf2.set(HConstants.HBASE_CLIENT_INSTANCE_ID, String.valueOf(-1));
240    conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
241    MiniZooKeeperCluster miniZK = HTU.getZkCluster();
242
243    HTU2 = new HBaseTestingUtility(conf2);
244    HTU2.setZkCluster(miniZK);
245    HTU2.startMiniCluster(NB_SERVERS);
246    LOG.info("Setup second Zk");
247    HTU2.getHBaseAdmin().createTable(hdt, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
248
249    ReplicationAdmin admin = new ReplicationAdmin(HTU.getConfiguration());
250
251    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
252    rpc.setClusterKey(HTU2.getClusterKey());
253    admin.addPeer("2", rpc, null);
254    admin.close();
255
256    Put p = new Put(row);
257    p.addColumn(row, row, row);
258    final Table table = HTU.getConnection().getTable(hdt.getTableName());
259    table.put(p);
260
261    HTU.getHBaseAdmin().flush(table.getName());
262    LOG.info("Put & flush done on the first cluster. Now doing a get on the same cluster.");
263
264    Waiter.waitFor(HTU.getConfiguration(), 1000, new Waiter.Predicate<Exception>() {
265      @Override
266      public boolean evaluate() throws Exception {
267        try {
268          SlowMeCopro.cdl.set(new CountDownLatch(1));
269          Get g = new Get(row);
270          g.setConsistency(Consistency.TIMELINE);
271          Result r = table.get(g);
272          Assert.assertTrue(r.isStale());
273          return  !r.isEmpty();
274        } finally {
275          SlowMeCopro.cdl.get().countDown();
276          SlowMeCopro.sleepTime.set(0);
277        }
278      }});
279    table.close();
280    LOG.info("stale get on the first cluster done. Now for the second.");
281
282    final Table table2 = HTU.getConnection().getTable(hdt.getTableName());
283    Waiter.waitFor(HTU.getConfiguration(), 1000, new Waiter.Predicate<Exception>() {
284      @Override
285      public boolean evaluate() throws Exception {
286        try {
287          SlowMeCopro.cdl.set(new CountDownLatch(1));
288          Get g = new Get(row);
289          g.setConsistency(Consistency.TIMELINE);
290          Result r = table2.get(g);
291          Assert.assertTrue(r.isStale());
292          return  !r.isEmpty();
293        } finally {
294          SlowMeCopro.cdl.get().countDown();
295          SlowMeCopro.sleepTime.set(0);
296        }
297      }});
298    table2.close();
299
300    HTU.getHBaseAdmin().disableTable(hdt.getTableName());
301    HTU.deleteTable(hdt.getTableName());
302
303    HTU2.getHBaseAdmin().disableTable(hdt.getTableName());
304    HTU2.deleteTable(hdt.getTableName());
305
306    // We shutdown HTU2 minicluster later, in afterClass(), as shutting down
307    // the minicluster has negative impact of deleting all HConnections in JVM.
308  }

310  @Test (timeout=30000)
311  public void testBulkLoad() throws IOException {
312    // Create table then get the single region for our new table.
313    LOG.debug("Creating test table");
314    HTableDescriptor hdt = HTU.createTableDescriptor("testBulkLoad");
315    hdt.setRegionReplication(NB_SERVERS);
316    hdt.addCoprocessor(SlowMeCopro.class.getName());
317    Table table = HTU.createTable(hdt, new byte[][]{f}, null);
318
319    // create hfiles to load.
320    LOG.debug("Creating test data");
321    Path dir = HTU.getDataTestDirOnTestFS("testBulkLoad");
322    final int numRows = 10;
323    final byte[] qual = Bytes.toBytes("qual");
324    final byte[] val  = Bytes.toBytes("val");
325    final List<Pair<byte[], String>> famPaths = new ArrayList<Pair<byte[], String>>();
326    for (HColumnDescriptor col : hdt.getColumnFamilies()) {
327      Path hfile = new Path(dir, col.getNameAsString());
328      TestHRegionServerBulkLoad.createHFile(HTU.getTestFileSystem(), hfile, col.getName(),
329        qual, val, numRows);
330      famPaths.add(new Pair<byte[], String>(col.getName(), hfile.toString()));
331    }
332
333    // bulk load HFiles
334    LOG.debug("Loading test data");
335    @SuppressWarnings("deprecation")
336    final ClusterConnection conn = (ClusterConnection) HTU.getAdmin().getConnection();
337    table = conn.getTable(hdt.getTableName());
338    final String bulkToken = new SecureBulkLoadClient(table).prepareBulkLoad(conn);
339    RegionServerCallable<Void> callable = new RegionServerCallable<Void>(
340      conn, hdt.getTableName(), TestHRegionServerBulkLoad.rowkey(0)) {
341        @Override
342        public Void call(int timeout) throws Exception {
343          LOG.debug("Going to connect to server " + getLocation() + " for row "
344            + Bytes.toStringBinary(getRow()));
345          SecureBulkLoadClient secureClient = null;
346          byte[] regionName = getLocation().getRegionInfo().getRegionName();
347          try (Table table = conn.getTable(getTableName())) {
348            secureClient = new SecureBulkLoadClient(table);
349            secureClient.secureBulkLoadHFiles(getStub(), famPaths, regionName,
350                  true, null, bulkToken);
351          }
352          return null;
353        }
354      };
355    RpcRetryingCallerFactory factory = new RpcRetryingCallerFactory(HTU.getConfiguration());
356    RpcRetryingCaller<Void> caller = factory.newCaller();
357    caller.callWithRetries(callable, 10000);
358
359    // verify we can read them from the primary
360    LOG.debug("Verifying data load");
361    for (int i = 0; i < numRows; i++) {
362      byte[] row = TestHRegionServerBulkLoad.rowkey(i);
363      Get g = new Get(row);
364      Result r = table.get(g);
365      Assert.assertFalse(r.isStale());
366    }
367
368    // verify we can read them from the replica
369    LOG.debug("Verifying replica queries");
370    try {
371      SlowMeCopro.cdl.set(new CountDownLatch(1));
372      for (int i = 0; i < numRows; i++) {
373        byte[] row = TestHRegionServerBulkLoad.rowkey(i);
374        Get g = new Get(row);
375        g.setConsistency(Consistency.TIMELINE);
376        Result r = table.get(g);
377        Assert.assertTrue(r.isStale());
378      }
379      SlowMeCopro.cdl.get().countDown();
380    } finally {
381      SlowMeCopro.cdl.get().countDown();
382      SlowMeCopro.sleepTime.set(0);
383    }
384
385    HTU.getHBaseAdmin().disableTable(hdt.getTableName());
386    HTU.deleteTable(hdt.getTableName());
387  }
