103  private void randomCFPuts(Table table, byte[] row, byte[] family, int nPuts)
104      throws Exception {
105    Put put = new Put(row);
106    for (int i = 0; i < nPuts; i++) {
107      byte[] qualifier = Bytes.toBytes(random.nextInt());
108      byte[] value = Bytes.toBytes(random.nextInt());
109      put.addColumn(family, qualifier, value);
110    }
111    table.put(put);
112  }

114  private void performMultiplePutAndFlush(HBaseAdmin admin, Table table,
115      byte[] row, byte[] family, int nFlushes, int nPuts)
116  throws Exception {
117
118    try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(table.getName())) {
119      // connection needed for poll-wait
120      HRegionLocation loc = locator.getRegionLocation(row, true);
121      AdminProtos.AdminService.BlockingInterface server =
122        ((ClusterConnection) admin.getConnection()).getAdmin(loc.getServerName());
123      byte[] regName = loc.getRegionInfo().getRegionName();
124
125      for (int i = 0; i < nFlushes; i++) {
126        randomCFPuts(table, row, family, nPuts);
127        List<String> sf = ProtobufUtil.getStoreFiles(server, regName, FAMILY);
128        int sfCount = sf.size();
129
130        admin.flush(table.getName());
131      }
132    }
133  }

135  // override the config settings at the CF level and ensure priority
136  @Test(timeout = 60000)
137  public void testAdvancedConfigOverride() throws Exception {
138    /*
139     * Overall idea: (1) create 3 store files and issue a compaction. config's
140     * compaction.min == 3, so should work. (2) Increase the compaction.min
141     * toggle in the HTD to 5 and modify table. If we use the HTD value instead
142     * of the default config value, adding 3 files and issuing a compaction
143     * SHOULD NOT work (3) Decrease the compaction.min toggle in the HCD to 2
144     * and modify table. The CF schema should override the Table schema and now
145     * cause a minor compaction.
146     */
147    TEST_UTIL.getConfiguration().setInt("hbase.hstore.compaction.min", 3);
148
149    TableName tableName = TableName.valueOf("testAdvancedConfigOverride");
150    Table hTable = TEST_UTIL.createTable(tableName, FAMILY, 10);
151    Admin admin = TEST_UTIL.getHBaseAdmin();
152    ClusterConnection connection = (ClusterConnection) TEST_UTIL.getConnection();
153
154    // Create 3 store files.
155    byte[] row = Bytes.toBytes(random.nextInt());
156    performMultiplePutAndFlush((HBaseAdmin) admin, hTable, row, FAMILY, 3, 100);
157
158    try (RegionLocator locator = TEST_UTIL.getConnection().getRegionLocator(tableName)) {
159      // Verify we have multiple store files.
160      HRegionLocation loc = locator.getRegionLocation(row, true);
161      byte[] regionName = loc.getRegionInfo().getRegionName();
162      AdminProtos.AdminService.BlockingInterface server = connection.getAdmin(loc.getServerName());
163      assertTrue(ProtobufUtil.getStoreFiles(server, regionName, FAMILY).size() > 1);
164
165      // Issue a compaction request
166      admin.compact(tableName);
167
168      // poll wait for the compactions to happen
169      for (int i = 0; i < 10 * 1000 / 40; ++i) {
170        // The number of store files after compaction should be lesser.
171        loc = locator.getRegionLocation(row, true);
172        if (!loc.getRegionInfo().isOffline()) {
173          regionName = loc.getRegionInfo().getRegionName();
174          server = connection.getAdmin(loc.getServerName());
175          if (ProtobufUtil.getStoreFiles(server, regionName, FAMILY).size() <= 1) {
176            break;
177          }
178        }
179        Thread.sleep(40);
180      }
181      // verify the compactions took place and that we didn't just time out
182      assertTrue(ProtobufUtil.getStoreFiles(server, regionName, FAMILY).size() <= 1);
183
184      // change the compaction.min config option for this table to 5
185      LOG.info("hbase.hstore.compaction.min should now be 5");
186      HTableDescriptor htd = new HTableDescriptor(hTable.getTableDescriptor());
187      htd.setValue("hbase.hstore.compaction.min", String.valueOf(5));
188      admin.modifyTable(tableName, htd);
189      Pair<Integer, Integer> st;
190      while (null != (st = admin.getAlterStatus(tableName)) && st.getFirst() > 0) {
191        LOG.debug(st.getFirst() + " regions left to update");
192        Thread.sleep(40);
193      }
194      LOG.info("alter status finished");
195
196      // Create 3 more store files.
197      performMultiplePutAndFlush((HBaseAdmin) admin, hTable, row, FAMILY, 3, 10);
198
199      // Issue a compaction request
200      admin.compact(tableName);
201
202      // This time, the compaction request should not happen
203      Thread.sleep(10 * 1000);
204      loc = locator.getRegionLocation(row, true);
205      regionName = loc.getRegionInfo().getRegionName();
206      server = connection.getAdmin(loc.getServerName());
207      int sfCount = ProtobufUtil.getStoreFiles(server, regionName, FAMILY).size();
208      assertTrue(sfCount > 1);
209
210      // change an individual CF's config option to 2 & online schema update
211      LOG.info("hbase.hstore.compaction.min should now be 2");
212      HColumnDescriptor hcd = new HColumnDescriptor(htd.getFamily(FAMILY));
213      hcd.setValue("hbase.hstore.compaction.min", String.valueOf(2));
214      htd.modifyFamily(hcd);
215      admin.modifyTable(tableName, htd);
216      while (null != (st = admin.getAlterStatus(tableName)) && st.getFirst() > 0) {
217        LOG.debug(st.getFirst() + " regions left to update");
218        Thread.sleep(40);
219      }
220      LOG.info("alter status finished");
221
222      // Issue a compaction request
223      admin.compact(tableName);
224
225      // poll wait for the compactions to happen
226      for (int i = 0; i < 10 * 1000 / 40; ++i) {
227        loc = locator.getRegionLocation(row, true);
228        regionName = loc.getRegionInfo().getRegionName();
229        try {
230          server = connection.getAdmin(loc.getServerName());
231          if (ProtobufUtil.getStoreFiles(server, regionName, FAMILY).size() < sfCount) {
232            break;
233          }
234        } catch (Exception e) {
235          LOG.debug("Waiting for region to come online: " + regionName);
236        }
237        Thread.sleep(40);
238      }
239
240      // verify the compaction took place and that we didn't just time out
241      assertTrue(ProtobufUtil.getStoreFiles(
242        server, regionName, FAMILY).size() < sfCount);
243
244      // Finally, ensure that we can remove a custom config value after we made it
245      LOG.info("Removing CF config value");
246      LOG.info("hbase.hstore.compaction.min should now be 5");
247      hcd = new HColumnDescriptor(htd.getFamily(FAMILY));
248      hcd.setValue("hbase.hstore.compaction.min", null);
249      htd.modifyFamily(hcd);
250      admin.modifyTable(tableName, htd);
251      while (null != (st = admin.getAlterStatus(tableName)) && st.getFirst() > 0) {
252        LOG.debug(st.getFirst() + " regions left to update");
253        Thread.sleep(40);
254      }
255      LOG.info("alter status finished");
256      assertNull(hTable.getTableDescriptor().getFamily(FAMILY).getValue(
257          "hbase.hstore.compaction.min"));
258    }
259  }

401    Table table = TEST_UTIL.createTable(

403      new byte[][] { FAMILY }, 1, new byte[] { 0x00 }, new byte[] { (byte) 0xff }, 255);

410    gets.add(new Get(Bytes.add(ROW, new byte[] { 0x00 })));

412    gets.add(new Get(Bytes.add(ANOTHERROW, new byte[] { 0x00 })));
413
414    LOG.info("Calling exists");
415    boolean[] results = table.existsAll(gets);
416    assertEquals(results[0], false);
417    assertEquals(results[1], false);
418    assertEquals(results[2], true);
419    assertEquals(results[3], false);
420
421    // Test with the first region.
422    put = new Put(new byte[] { 0x00 });
423    put.addColumn(FAMILY, QUALIFIER, VALUE);
424    table.put(put);
425
426    gets = new ArrayList<Get>();
427    gets.add(new Get(new byte[] { 0x00 }));
428    gets.add(new Get(new byte[] { 0x00, 0x00 }));
429    results = table.existsAll(gets);
430    assertEquals(results[0], true);
431    assertEquals(results[1], false);
432
433    // Test with the last region
434    put = new Put(new byte[] { (byte) 0xff, (byte) 0xff });
435    put.addColumn(FAMILY, QUALIFIER, VALUE);
436    table.put(put);
437
438    gets = new ArrayList<Get>();
439    gets.add(new Get(new byte[] { (byte) 0xff }));
440    gets.add(new Get(new byte[] { (byte) 0xff, (byte) 0xff }));
441    gets.add(new Get(new byte[] { (byte) 0xff, (byte) 0xff, (byte) 0xff }));
442    results = table.existsAll(gets);
443    assertEquals(results[0], false);
444    assertEquals(results[1], true);
445    assertEquals(results[2], false);

448  @Test
449  public void testGetEmptyRow() throws Exception {
450    //Create a table and put in 1 row
451    Admin admin = TEST_UTIL.getHBaseAdmin();
452    HTableDescriptor desc = new HTableDescriptor(TableName.valueOf(Bytes.toBytes("test")));
453    desc.addFamily(new HColumnDescriptor(FAMILY));
454    admin.createTable(desc);
455    Table table = TEST_UTIL.getConnection().getTable(desc.getTableName());
456
457    Put put = new Put(ROW_BYTES);
458    put.addColumn(FAMILY, COL_QUAL, VAL_BYTES);
459    table.put(put);
460
461    //Try getting the row with an empty row key
462    Result res = null;
463    try {
464      res = table.get(new Get(new byte[0]));
465      fail();
466    } catch (IllegalArgumentException e) {
467      // Expected.
468    }
469    assertTrue(res == null);
470    res = table.get(new Get(Bytes.toBytes("r1-not-exist")));
471    assertTrue(res.isEmpty() == true);
472    res = table.get(new Get(ROW_BYTES));
473    assertTrue(Arrays.equals(res.getValue(FAMILY, COL_QUAL), VAL_BYTES));
474    table.close();
475  }
