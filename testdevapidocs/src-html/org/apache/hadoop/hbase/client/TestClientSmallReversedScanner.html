083  /**
084   * Create a simple Answer which returns true the first time, and false every time after.
085   */
086  private Answer<Boolean> createTrueThenFalseAnswer() {
087    return new Answer<Boolean>() {
088      boolean first = true;
089
090      @Override
091      public Boolean answer(InvocationOnMock invocation) {
092        if (first) {
093          first = false;
094          return true;
095        }
096        return false;
097      }
098    };
099  }

111      }
112    };

115  @Test
116  public void testContextPresent() throws Exception {
117    final KeyValue kv1 = new KeyValue("row1".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
118        Type.Maximum), kv2 = new KeyValue("row2".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
119        Type.Maximum), kv3 = new KeyValue("row3".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
120        Type.Maximum);
121
122    ScannerCallableWithReplicas callableWithReplicas = Mockito
123        .mock(ScannerCallableWithReplicas.class);
124
125    // Mock out the RpcCaller
126    @SuppressWarnings("unchecked")
127    RpcRetryingCaller<Result[]> caller = Mockito.mock(RpcRetryingCaller.class);
128    // Return the mock from the factory
129    Mockito.when(rpcFactory.<Result[]> newCaller()).thenReturn(caller);
130
131    // Intentionally leave a "default" caching size in the Scan. No matter the value, we
132    // should continue based on the server context
133
134    SmallScannerCallableFactory factory = getFactory(callableWithReplicas);
135
136    try (ClientSmallReversedScanner csrs = new ClientSmallReversedScanner(conf, scan,
137        TableName.valueOf("table"), clusterConn, rpcFactory, controllerFactory, pool,
138        Integer.MAX_VALUE)) {
139
140      csrs.setScannerCallableFactory(factory);
141
142      // Return some data the first time, less the second, and none after that
143      Mockito.when(caller.callWithoutRetries(callableWithReplicas, csrs.getScannerTimeout()))
144          .thenAnswer(new Answer<Result[]>() {
145            int count = 0;
146
147            @Override
148            public Result[] answer(InvocationOnMock invocation) {
149              Result[] results;
150              if (0 == count) {
151                results = new Result[] {Result.create(new Cell[] {kv3}),
152                    Result.create(new Cell[] {kv2})};
153              } else if (1 == count) {
154                results = new Result[] {Result.create(new Cell[] {kv1})};
155              } else {
156                results = new Result[0];
157              }
158              count++;
159              return results;
160            }
161          });
162
163      // Pass back the context always
164      Mockito.when(callableWithReplicas.hasMoreResultsContext()).thenReturn(true);
165      // Only have more results the first time
166      Mockito.when(callableWithReplicas.getServerHasMoreResults()).thenAnswer(
167          createTrueThenFalseAnswer());
168
169      // A mocked HRegionInfo so ClientSmallScanner#nextScanner(...) works right
170      HRegionInfo regionInfo = Mockito.mock(HRegionInfo.class);
171      Mockito.when(callableWithReplicas.getHRegionInfo()).thenReturn(regionInfo);
172      // Trigger the "no more data" branch for #nextScanner(...)
173      Mockito.when(regionInfo.getEndKey()).thenReturn(HConstants.EMPTY_BYTE_ARRAY);
174
175      csrs.loadCache();
176
177      Queue<Result> results = csrs.cache;
178      Iterator<Result> iter = results.iterator();
179      assertEquals(3, results.size());
180      for (int i = 3; i >= 1 && iter.hasNext(); i--) {
181        Result result = iter.next();
182        byte[] row = result.getRow();
183        assertEquals("row" + i, new String(row, StandardCharsets.UTF_8));
184        assertEquals(1, result.getMap().size());
185      }
186      assertTrue(csrs.closed);
187    }
188  }

190  @Test
191  public void testNoContextFewerRecords() throws Exception {
192    final KeyValue kv1 = new KeyValue("row1".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
193        Type.Maximum), kv2 = new KeyValue("row2".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
194        Type.Maximum), kv3 = new KeyValue("row3".getBytes(), "cf".getBytes(), "cq".getBytes(), 1,
195        Type.Maximum);
196
197    ScannerCallableWithReplicas callableWithReplicas = Mockito
198        .mock(ScannerCallableWithReplicas.class);
199
200    // While the server returns 2 records per batch, we expect more records.
201    scan.setCaching(2);
202
203    SmallScannerCallableFactory factory = getFactory(callableWithReplicas);
204
205    try (ClientSmallReversedScanner csrs = new ClientSmallReversedScanner(conf, scan,
206        TableName.valueOf("table"), clusterConn, rpcFactory, controllerFactory, pool,
207        Integer.MAX_VALUE)) {
208
209      csrs.setScannerCallableFactory(factory);
210
211      // Return some data the first time, less the second, and none after that
212      Mockito.when(caller.callWithoutRetries(callableWithReplicas, csrs.getScannerTimeout()))
213          .thenAnswer(new Answer<Result[]>() {
214            int count = 0;
215
216            @Override
217            public Result[] answer(InvocationOnMock invocation) {
218              Result[] results;
219              if (0 == count) {
220                results = new Result[] {Result.create(new Cell[] {kv3}),
221                    Result.create(new Cell[] {kv2})};
222              } else if (1 == count) {
223                // Return fewer records than expected (2)
224                results = new Result[] {Result.create(new Cell[] {kv1})};
225              } else {
226                throw new RuntimeException("Should not fetch a third batch from the server");
227              }
228              count++;
229              return results;
230            }
231          });
232
233      // Server doesn't return the context
234      Mockito.when(callableWithReplicas.hasMoreResultsContext()).thenReturn(false);
235      // getServerHasMoreResults shouldn't be called when hasMoreResultsContext returns false
236      Mockito.when(callableWithReplicas.getServerHasMoreResults())
237          .thenThrow(new RuntimeException("Should not be called"));
238
239      // A mocked HRegionInfo so ClientSmallScanner#nextScanner(...) works right
240      HRegionInfo regionInfo = Mockito.mock(HRegionInfo.class);
241      Mockito.when(callableWithReplicas.getHRegionInfo()).thenReturn(regionInfo);
242      // Trigger the "no more data" branch for #nextScanner(...)
243      Mockito.when(regionInfo.getEndKey()).thenReturn(HConstants.EMPTY_BYTE_ARRAY);
244
245      csrs.loadCache();
246
247      Queue<Result> results = csrs.cache;
248      Iterator<Result> iter = results.iterator();
249      assertEquals(2, results.size());
250      for (int i = 3; i >= 2 && iter.hasNext(); i--) {
251        Result result = iter.next();
252        byte[] row = result.getRow();
253        assertEquals("row" + i, new String(row, StandardCharsets.UTF_8));
254        assertEquals(1, result.getMap().size());
255      }
256
257      // "consume" the Results
258      results.clear();
259
260      csrs.loadCache();
261
262      assertEquals(1, results.size());
263      Result result = results.peek();
264      assertEquals("row1", new String(result.getRow(), StandardCharsets.UTF_8));
265      assertEquals(1, result.getMap().size());
266
267      assertTrue(csrs.closed);
268    }
269  }
