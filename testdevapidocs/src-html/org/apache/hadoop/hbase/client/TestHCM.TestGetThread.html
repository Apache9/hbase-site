119  public static class SleepAndFailFirstTime extends BaseRegionObserver {
120    static final AtomicLong ct = new AtomicLong(0);
121    static final String SLEEP_TIME_CONF_KEY =
122        "hbase.coprocessor.SleepAndFailFirstTime.sleepTime";
123    static final long DEFAULT_SLEEP_TIME = 20000;
124    static final AtomicLong sleepTime = new AtomicLong(DEFAULT_SLEEP_TIME);
125
126    public SleepAndFailFirstTime() {
127    }
128
129    @Override
130    public void postOpen(ObserverContext<RegionCoprocessorEnvironment> c) {
131      RegionCoprocessorEnvironment env = c.getEnvironment();
132      Configuration conf = env.getConfiguration();
133      sleepTime.set(conf.getLong(SLEEP_TIME_CONF_KEY, DEFAULT_SLEEP_TIME));
134    }
135
136    @Override
137    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
138              final Get get, final List<Cell> results) throws IOException {
139      Threads.sleep(sleepTime.get());
140      if (ct.incrementAndGet() == 1){
141        throw new IOException("first call I fail");
142      }
143    }
144  }

171  public static class SleepLongerAtFirstCoprocessor extends BaseRegionObserver {
172    public static final int SLEEP_TIME = 2000;
173    static final AtomicLong ct = new AtomicLong(0);
174    @Override
175    public void preGetOp(final ObserverContext<RegionCoprocessorEnvironment> e,
176        final Get get, final List<Cell> results) throws IOException {
177      // After first sleep, all requests are timeout except the last retry. If we handle
178      // all the following requests, finally the last request is also timeout. If we drop all
179      // timeout requests, we can handle the last request immediately and it will not timeout.
180      if (ct.incrementAndGet() <= 1) {
181        Threads.sleep(SLEEP_TIME * RPC_RETRY * 2);
182      } else {
183        Threads.sleep(SLEEP_TIME);
184      }
185    }
186  }

205  @Test
206  public void testClusterConnection() throws IOException {
207    ThreadPoolExecutor otherPool = new ThreadPoolExecutor(1, 1,
208        5, TimeUnit.SECONDS,
209        new SynchronousQueue<Runnable>(),
210        Threads.newDaemonThreadFactory("test-hcm"));
211
212    Connection con1 = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration());
213    Connection con2 = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration(), otherPool);
214    // make sure the internally created ExecutorService is the one passed
215    assertTrue(otherPool == ((ConnectionImplementation) con2).getCurrentBatchPool());
216
217    TableName tableName = TableName.valueOf("testClusterConnection");
218    TEST_UTIL.createTable(tableName, FAM_NAM).close();
219    Table table = con1.getTable(tableName, otherPool);
220
221    ExecutorService pool = null;
222
223    if(table instanceof HTable) {
224      HTable t = (HTable) table;
225      // make sure passing a pool to the getTable does not trigger creation of an internal pool
226      assertNull("Internal Thread pool should be null",
227        ((ConnectionImplementation) con1).getCurrentBatchPool());
228      // table should use the pool passed
229      assertTrue(otherPool == t.getPool());
230      t.close();
231
232      t = (HTable) con2.getTable(tableName);
233      // table should use the connectin's internal pool
234      assertTrue(otherPool == t.getPool());
235      t.close();
236
237      t = (HTable) con2.getTable(tableName);
238      // try other API too
239      assertTrue(otherPool == t.getPool());
240      t.close();
241
242      t = (HTable) con2.getTable(tableName);
243      // try other API too
244      assertTrue(otherPool == t.getPool());
245      t.close();
246
247      t = (HTable) con1.getTable(tableName);
248      pool = ((ConnectionImplementation) con1).getCurrentBatchPool();
249      // make sure an internal pool was created
250      assertNotNull("An internal Thread pool should have been created", pool);
251      // and that the table is using it
252      assertTrue(t.getPool() == pool);
253      t.close();
254
255      t = (HTable) con1.getTable(tableName);
256      // still using the *same* internal pool
257      assertTrue(t.getPool() == pool);
258      t.close();
259    } else {
260      table.close();
261    }
262
263    con1.close();
264
265    // if the pool was created on demand it should be closed upon connection close
266    if(pool != null) {
267      assertTrue(pool.isShutdown());
268    }
269
270    con2.close();
271    // if the pool is passed, it is not closed
272    assertFalse(otherPool.isShutdown());
273    otherPool.shutdownNow();
274  }

358  /**
359   * Test that an operation can fail if we read the global operation timeout, even if the
360   * individual timeout is fine. We do that with:
361   * - client side: an operation timeout of 30 seconds
362   * - server side: we sleep 20 second at each attempt. The first work fails, the second one
363   * succeeds. But the client won't wait that much, because 20 + 20 > 30, so the client
364   * timeouted when the server answers.
365   */
366  @Test
367  public void testOperationTimeout() throws Exception {
368    HTableDescriptor hdt = TEST_UTIL.createTableDescriptor("HCM-testOperationTimeout");
369    hdt.addCoprocessor(SleepAndFailFirstTime.class.getName());
370    Table table = TEST_UTIL.createTable(hdt, new byte[][]{FAM_NAM});
371    table.setRpcTimeout(Integer.MAX_VALUE);
372    // Check that it works if the timeout is big enough
373    table.setOperationTimeout(120 * 1000);
374    table.get(new Get(FAM_NAM));
375
376    // Resetting and retrying. Will fail this time, not enough time for the second try
377    SleepAndFailFirstTime.ct.set(0);
378    try {
379      table.setOperationTimeout(30 * 1000);
380      table.get(new Get(FAM_NAM));
381      Assert.fail("We expect an exception here");
382    } catch (SocketTimeoutException e) {
383      // The client has a CallTimeout class, but it's not shared.We're not very clean today,
384      //  in the general case you can expect the call to stop, but the exception may vary.
385      // In this test however, we're sure that it will be a socket timeout.
386      LOG.info("We received an exception, as expected ", e);
387    } catch (IOException e) {
388      Assert.fail("Wrong exception:" + e.getMessage());
389    } finally {
390      table.close();
391    }
392  }

481    // Simulate the situation that the server is slow and client retries for several times because
482    // of timeout. When a request can be handled after waiting in the queue, we will drop it if
483    // it has been considered as timeout at client. If we don't drop it, the server will waste time
484    // on handling timeout requests and finally all requests timeout and client throws exception.

487    Configuration c = new Configuration(TEST_UTIL.getConfiguration());

489      t.setRpcTimeout(SleepLongerAtFirstCoprocessor.SLEEP_TIME * 2);
490      t.get(new Get(FAM_NAM));
491    }

1180      TEST_UTIL.waitUntilNoRegionsInTransition();

1190

1197

1199

1202      byte[] otherRow = null;
1203       for (Region region : regions) {
1204         if (!region.getRegionInfo().getEncodedName().equals(toMove.getRegionInfo().getEncodedName())
1205             && Bytes.BYTES_COMPARATOR.compare(region.getRegionInfo().getStartKey(), ROW_X) < 0) {

1209       }

1233

1250          .containsKey(encodedRegionNameBytes));

1268      table.batch(actions, null); // first should be a valid row,
1269      // second we get RegionMovedException.

1271      setNumTries(conn, prevNumRetriesVal);

1273      table.close();

1279    // TODO: This test would seem to presume hardcoded RETRY_BACKOFF which it should not.
1280    final long ANY_PAUSE = 100;
1281    ServerName location = ServerName.valueOf("127.0.0.1", 1, 0);

1283
1284    ManualEnvironmentEdge timeMachine = new ManualEnvironmentEdge();
1285    EnvironmentEdgeManager.injectEdge(timeMachine);
1286    try {
1287      long largeAmountOfTime = ANY_PAUSE * 1000;
1288      ConnectionImplementation.ServerErrorTracker tracker =
1289          new ConnectionImplementation.ServerErrorTracker(largeAmountOfTime, 100);
1290
1291      // The default backoff is 0.
1292      assertEquals(0, tracker.calculateBackoffTime(location, ANY_PAUSE));
1293
1294      // Check some backoff values from HConstants sequence.
1295      tracker.reportServerError(location);
1296      assertEqualsWithJitter(ANY_PAUSE * HConstants.RETRY_BACKOFF[0],
1297        tracker.calculateBackoffTime(location, ANY_PAUSE));
1298      tracker.reportServerError(location);
1299      tracker.reportServerError(location);
1300      tracker.reportServerError(location);
1301      assertEqualsWithJitter(ANY_PAUSE * HConstants.RETRY_BACKOFF[3],
1302        tracker.calculateBackoffTime(location, ANY_PAUSE));
1303
1304      // All of this shouldn't affect backoff for different location.
1305      assertEquals(0, tracker.calculateBackoffTime(diffLocation, ANY_PAUSE));
1306      tracker.reportServerError(diffLocation);
1307      assertEqualsWithJitter(ANY_PAUSE * HConstants.RETRY_BACKOFF[0],
1308        tracker.calculateBackoffTime(diffLocation, ANY_PAUSE));
1309
1310      // Check with different base.
1311      assertEqualsWithJitter(ANY_PAUSE * 2 * HConstants.RETRY_BACKOFF[3],
1312          tracker.calculateBackoffTime(location, ANY_PAUSE * 2));
1313    } finally {
1314      EnvironmentEdgeManager.reset();
1315    }
