100  /**
101   * Initialize the tests with a table filled with some data
102   * and two snapshots (snapshotName0, snapshotName1) of different states.
103   * The tableName, snapshotNames and the number of rows in the snapshot are initialized.
104   */
105  @Before
106  public void setup() throws Exception {
107    this.admin = TEST_UTIL.getAdmin();
108
109    long tid = System.currentTimeMillis();
110    tableName = TableName.valueOf(name.getMethodName() + tid);
111    emptySnapshot = Bytes.toBytes("emptySnaptb-" + tid);
112    snapshotName0 = Bytes.toBytes("snaptb0-" + tid);
113    snapshotName1 = Bytes.toBytes("snaptb1-" + tid);
114    snapshotName2 = Bytes.toBytes("snaptb2-" + tid);
115
116    createTableAndSnapshots();
117  }
118
119  protected void createTableAndSnapshots() throws Exception {
120    // create Table and disable it
121    SnapshotTestingUtils.createTable(TEST_UTIL, tableName, getNumReplicas(), FAMILY);
122    admin.disableTable(tableName);
123
124    // take an empty snapshot
125    admin.snapshot(emptySnapshot, tableName);
126
127    // enable table and insert data
128    admin.enableTable(tableName);
129    SnapshotTestingUtils.loadData(TEST_UTIL, tableName, 500, FAMILY);
130    try (Table table = TEST_UTIL.getConnection().getTable(tableName)){
131      snapshot0Rows = TEST_UTIL.countRows(table);
132    }
133    admin.disableTable(tableName);
134
135    // take a snapshot
136    admin.snapshot(snapshotName0, tableName);
137
138    // enable table and insert more data
139    admin.enableTable(tableName);
140    SnapshotTestingUtils.loadData(TEST_UTIL, tableName, 500, FAMILY);
141    try (Table table = TEST_UTIL.getConnection().getTable(tableName)){
142      snapshot1Rows = TEST_UTIL.countRows(table);
143    }
144    admin.disableTable(tableName);
145
146    // take a snapshot of the updated table
147    admin.snapshot(snapshotName1, tableName);
148
149    // re-enable table
150    admin.enableTable(tableName);
151  }
152
153  protected int getNumReplicas() {
154    return 1;
155  }

157  @After
158  public void tearDown() throws Exception {
159    if (admin.tableExists(tableName)) {
160      TEST_UTIL.deleteTable(tableName);
161    }
162    SnapshotTestingUtils.deleteAllSnapshots(admin);
163    SnapshotTestingUtils.deleteArchiveDirectory(TEST_UTIL);
164  }

180  @Test
181  public void testCloneSnapshot() throws IOException, InterruptedException {
182    final TableName clonedTableName = TableName.valueOf(name.getMethodName() + "-"
183      + System.currentTimeMillis());
184    testCloneSnapshot(clonedTableName, snapshotName0, snapshot0Rows);
185    testCloneSnapshot(clonedTableName, snapshotName1, snapshot1Rows);
186    testCloneSnapshot(clonedTableName, emptySnapshot, 0);
187  }
188
189  private void testCloneSnapshot(final TableName tableName, final byte[] snapshotName,
190      int snapshotRows) throws IOException, InterruptedException {
191    // create a new table from snapshot
192    admin.cloneSnapshot(snapshotName, tableName);
193    verifyRowCount(TEST_UTIL, tableName, snapshotRows);
194
195    verifyReplicasCameOnline(tableName);
196    TEST_UTIL.deleteTable(tableName);
197  }
198
199  protected void verifyReplicasCameOnline(TableName tableName) throws IOException {
200    SnapshotTestingUtils.verifyReplicasCameOnline(tableName, admin, getNumReplicas());
201  }

203  @Test
204  public void testCloneSnapshotCrossNamespace() throws IOException, InterruptedException {
205    String nsName = "testCloneSnapshotCrossNamespace";
206    admin.createNamespace(NamespaceDescriptor.create(nsName).build());
207    final TableName clonedTableName = TableName.valueOf(nsName, name.getMethodName()
208      + "-" + System.currentTimeMillis());
209    testCloneSnapshot(clonedTableName, snapshotName0, snapshot0Rows);
210    testCloneSnapshot(clonedTableName, snapshotName1, snapshot1Rows);
211    testCloneSnapshot(clonedTableName, emptySnapshot, 0);
212  }

214  /**
215   * Verify that tables created from the snapshot are still alive after source table deletion.
216   */
217  @Test
218  public void testCloneLinksAfterDelete() throws IOException, InterruptedException {
219    // Clone a table from the first snapshot
220    final TableName clonedTableName = TableName.valueOf(name.getMethodName() + "1-"
221      + System.currentTimeMillis());
222    admin.cloneSnapshot(snapshotName0, clonedTableName);
223    verifyRowCount(TEST_UTIL, clonedTableName, snapshot0Rows);
224
225    // Take a snapshot of this cloned table.
226    admin.disableTable(clonedTableName);
227    admin.snapshot(snapshotName2, clonedTableName);
228
229    // Clone the snapshot of the cloned table
230    final TableName clonedTableName2 = TableName.valueOf(name.getMethodName() + "2-"
231      + System.currentTimeMillis());
232    admin.cloneSnapshot(snapshotName2, clonedTableName2);
233    verifyRowCount(TEST_UTIL, clonedTableName2, snapshot0Rows);
234    admin.disableTable(clonedTableName2);
235
236    // Remove the original table
237    TEST_UTIL.deleteTable(tableName);
238    waitCleanerRun();
239
240    // Verify the first cloned table
241    admin.enableTable(clonedTableName);
242    verifyRowCount(TEST_UTIL, clonedTableName, snapshot0Rows);
243
244    // Verify the second cloned table
245    admin.enableTable(clonedTableName2);
246    verifyRowCount(TEST_UTIL, clonedTableName2, snapshot0Rows);
247    admin.disableTable(clonedTableName2);
248
249    // Delete the first cloned table
250    TEST_UTIL.deleteTable(clonedTableName);
251    waitCleanerRun();
252
253    // Verify the second cloned table
254    admin.enableTable(clonedTableName2);
255    verifyRowCount(TEST_UTIL, clonedTableName2, snapshot0Rows);
256
257    // Clone a new table from cloned
258    final TableName clonedTableName3 = TableName.valueOf(name.getMethodName() + "3-"
259      + System.currentTimeMillis());
260    admin.cloneSnapshot(snapshotName2, clonedTableName3);
261    verifyRowCount(TEST_UTIL, clonedTableName3, snapshot0Rows);
262
263    // Delete the cloned tables
264    TEST_UTIL.deleteTable(clonedTableName2);
265    TEST_UTIL.deleteTable(clonedTableName3);
266    admin.deleteSnapshot(snapshotName2);
267  }

269  @Test
270  public void testCloneSnapshotAfterSplittingRegion() throws IOException, InterruptedException {
271    // Turn off the CatalogJanitor
272    admin.catalogJanitorSwitch(false);
273
274    try {
275      List<RegionInfo> regionInfos = admin.getRegions(tableName);
276      RegionReplicaUtil.removeNonDefaultRegions(regionInfos);
277
278      // Split the first region
279      splitRegion(regionInfos.get(0));
280
281      // Take a snapshot
282      admin.snapshot(snapshotName2, tableName);
283
284      // Clone the snapshot to another table
285      TableName clonedTableName = TableName.valueOf(name.getMethodName() + "-"
286        + System.currentTimeMillis());
287      admin.cloneSnapshot(snapshotName2, clonedTableName);
288      SnapshotTestingUtils.waitForTableToBeOnline(TEST_UTIL, clonedTableName);
289
290      RegionStates regionStates =
291        TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
292
293      // The region count of the cloned table should be the same as the one of the original table
294      int openRegionCountOfOriginalTable =
295        regionStates.getRegionByStateOfTable(tableName).get(RegionState.State.OPEN).size();
296      int openRegionCountOfClonedTable =
297        regionStates.getRegionByStateOfTable(clonedTableName).get(RegionState.State.OPEN).size();
298      assertEquals(openRegionCountOfOriginalTable, openRegionCountOfClonedTable);
299
300      int splitRegionCountOfOriginalTable =
301        regionStates.getRegionByStateOfTable(tableName).get(RegionState.State.SPLIT).size();
302      int splitRegionCountOfClonedTable =
303        regionStates.getRegionByStateOfTable(clonedTableName).get(RegionState.State.SPLIT).size();
304      assertEquals(splitRegionCountOfOriginalTable, splitRegionCountOfClonedTable);
305
306      TEST_UTIL.deleteTable(clonedTableName);
307    } finally {
308      admin.catalogJanitorSwitch(true);
309    }
310  }

326    byte[][] splitPoints = Bytes.split(regionInfo.getStartKey(), regionInfo.getEndKey(), 1);
