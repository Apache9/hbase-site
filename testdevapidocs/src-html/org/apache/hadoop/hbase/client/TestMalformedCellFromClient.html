048/**
049 * The purpose of this test is to make sure the region exception won't corrupt the results
050 * of batch. The prescription is shown below.
051 * 1) honor the action result rather than region exception. If the action have both of true result
052 * and region exception, the action is fine as the exception is caused by other actions
053 * which are in the same region.
054 * 2) honor the action exception rather than region exception. If the action have both of action
055 * exception and region exception, we deal with the action exception only. If we also
056 * handle the region exception for the same action, it will introduce the negative count of
057 * actions in progress. The AsyncRequestFuture#waitUntilDone will block forever.
058 *
059 * The no-cluster test is in TestAsyncProcessWithRegionException.
060 */
061@Category({ MediumTests.class, ClientTests.class })
062public class TestMalformedCellFromClient {
063
064  @ClassRule
065  public static final HBaseClassTestRule CLASS_RULE =
066    HBaseClassTestRule.forClass(TestMalformedCellFromClient.class);
067
068  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
069  private static final byte[] FAMILY = Bytes.toBytes("testFamily");
070  private static final int CELL_SIZE = 100;
071  private static final TableName TABLE_NAME = TableName.valueOf("TestMalformedCellFromClient");
072
073  @BeforeClass
074  public static void setUpBeforeClass() throws Exception {
075    // disable the retry
076    TEST_UTIL.getConfiguration().setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 0);
077    TEST_UTIL.startMiniCluster(1);
078  }
079
080  @Before
081  public void before() throws Exception {
082    TableDescriptor desc = TableDescriptorBuilder.newBuilder(TABLE_NAME)
083      .addColumnFamily(ColumnFamilyDescriptorBuilder.of(FAMILY))
084      .setValue(HRegion.HBASE_MAX_CELL_SIZE_KEY, String.valueOf(CELL_SIZE)).build();
085    TEST_UTIL.getConnection().getAdmin().createTable(desc);
086  }
087
088  @After
089  public void tearDown() throws Exception {
090    for (TableDescriptor htd : TEST_UTIL.getAdmin().listTableDescriptors()) {
091      TEST_UTIL.deleteTable(htd.getTableName());
092    }
093  }
094
095  @AfterClass
096  public static void tearDownAfterClass() throws Exception {
097    TEST_UTIL.shutdownMiniCluster();
098  }
099
100  /**
101   * The purpose of this ut is to check the consistency between the exception and results.
102   * If the RetriesExhaustedWithDetailsException contains the whole batch,
103   * each result should be of IOE. Otherwise, the row operation which is not in the exception
104   * should have a true result.
105   */
106  @Test
107  public void testRegionException() throws InterruptedException, IOException {
108    List<Row> batches = new ArrayList<>();
109    batches.add(new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[10]));
110    // the rm is used to prompt the region exception.
111    // see RSRpcServices#multi
112    RowMutations rm = new RowMutations(Bytes.toBytes("fail"));
113    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[CELL_SIZE]));
114    batches.add(rm);
115    Object[] results = new Object[batches.size()];
116
117    try (Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME)) {
118      Throwable exceptionByCaught = null;
119      try {
120        table.batch(batches, results);
121        fail("Where is the exception? We put the malformed cells!!!");
122      } catch (RetriesExhaustedWithDetailsException e) {
123        for (Throwable throwable : e.getCauses()) {
124          assertNotNull(throwable);
125        }
126        assertEquals(1, e.getNumExceptions());
127        exceptionByCaught = e.getCause(0);
128      }
129      for (Object obj : results) {
130        assertNotNull(obj);
131      }
132      assertEquals(Result.class, results[0].getClass());
133      assertEquals(exceptionByCaught.getClass(), results[1].getClass());
134      Result result = table.get(new Get(Bytes.toBytes("good")));
135      assertEquals(1, result.size());
136      Cell cell = result.getColumnLatestCell(FAMILY, null);
137      assertTrue(Bytes.equals(CellUtil.cloneValue(cell), new byte[10]));
138    }
139  }
140
141  /**
142   * The purpose of this ut is to check the consistency between the exception and results.
143   * If the RetriesExhaustedWithDetailsException contains the whole batch,
144   * each result should be of IOE. Otherwise, the row operation which is not in the exception
145   * should have a true result.
146   */
147  @Test
148  public void testRegionExceptionByAsync() throws Exception {
149    List<Row> batches = new ArrayList<>();
150    batches.add(new Put(Bytes.toBytes("good")).addColumn(FAMILY, null, new byte[10]));
151    // the rm is used to prompt the region exception.
152    // see RSRpcServices#multi
153    RowMutations rm = new RowMutations(Bytes.toBytes("fail"));
154    rm.add(new Put(rm.getRow()).addColumn(FAMILY, null, new byte[CELL_SIZE]));
155    batches.add(rm);
156    try (AsyncConnection asyncConnection = ConnectionFactory
157      .createAsyncConnection(TEST_UTIL.getConfiguration()).get()) {
158      AsyncTable<AdvancedScanResultConsumer> table = asyncConnection.getTable(TABLE_NAME);
159      List<CompletableFuture<AdvancedScanResultConsumer>> results = table.batch(batches);
160      assertEquals(2, results.size());
161      try {
162        results.get(1).get();
163        fail("Where is the exception? We put the malformed cells!!!");
164      } catch (ExecutionException e) {
165        // pass
166      }
167      Result result = table.get(new Get(Bytes.toBytes("good"))).get();
168      assertEquals(1, result.size());
169      Cell cell = result.getColumnLatestCell(FAMILY, null);
170      assertTrue(Bytes.equals(CellUtil.cloneValue(cell), new byte[10]));
171    }
172  }
