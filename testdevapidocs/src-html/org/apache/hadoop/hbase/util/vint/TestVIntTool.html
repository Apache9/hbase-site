019package org.apache.hadoop.hbase.util.vint;
020
021import java.io.ByteArrayInputStream;
022import java.io.ByteArrayOutputStream;
023import java.io.IOException;
024import java.nio.ByteBuffer;
025import java.util.Random;
026
027import org.apache.hadoop.hbase.nio.SingleByteBuff;
028import org.apache.hadoop.hbase.testclassification.MiscTests;
029import org.apache.hadoop.hbase.testclassification.SmallTests;
030import org.junit.Assert;
031import org.junit.Test;
032import org.junit.experimental.categories.Category;
033
034@Category({MiscTests.class, SmallTests.class})
035public class TestVIntTool {
036
037  @Test
038  public void testNumBytes() {
039    Assert.assertEquals(1, UVIntTool.numBytes(0));
040    Assert.assertEquals(1, UVIntTool.numBytes(1));
041    Assert.assertEquals(1, UVIntTool.numBytes(100));
042    Assert.assertEquals(1, UVIntTool.numBytes(126));
043    Assert.assertEquals(1, UVIntTool.numBytes(127));
044    Assert.assertEquals(2, UVIntTool.numBytes(128));
045    Assert.assertEquals(2, UVIntTool.numBytes(129));
046    Assert.assertEquals(5, UVIntTool.numBytes(Integer.MAX_VALUE));
047  }
048
049  @Test
050  public void testWriteBytes() throws IOException {
051    Assert.assertArrayEquals(new byte[] { 0 }, bytesViaOutputStream(0));
052    Assert.assertArrayEquals(new byte[] { 1 }, bytesViaOutputStream(1));
053    Assert.assertArrayEquals(new byte[] { 63 }, bytesViaOutputStream(63));
054    Assert.assertArrayEquals(new byte[] { 127 }, bytesViaOutputStream(127));
055    Assert.assertArrayEquals(new byte[] { -128, 1 }, bytesViaOutputStream(128));
056    Assert.assertArrayEquals(new byte[] { -128 + 27, 1 }, bytesViaOutputStream(155));
057    Assert.assertArrayEquals(UVIntTool.MAX_VALUE_BYTES, bytesViaOutputStream(Integer.MAX_VALUE));
058  }
059
060  private byte[] bytesViaOutputStream(int value) throws IOException {
061    ByteArrayOutputStream os = new ByteArrayOutputStream();
062    UVIntTool.writeBytes(value, os);
063    return os.toByteArray();
064  }
065
066  @Test
067  public void testToBytes() {
068    Assert.assertArrayEquals(new byte[] { 0 }, UVIntTool.getBytes(0));
069    Assert.assertArrayEquals(new byte[] { 1 }, UVIntTool.getBytes(1));
070    Assert.assertArrayEquals(new byte[] { 63 }, UVIntTool.getBytes(63));
071    Assert.assertArrayEquals(new byte[] { 127 }, UVIntTool.getBytes(127));
072    Assert.assertArrayEquals(new byte[] { -128, 1 }, UVIntTool.getBytes(128));
073    Assert.assertArrayEquals(new byte[] { -128 + 27, 1 }, UVIntTool.getBytes(155));
074    Assert.assertArrayEquals(UVIntTool.MAX_VALUE_BYTES, UVIntTool.getBytes(Integer.MAX_VALUE));
075  }
076
077  @Test
078  public void testFromBytes() {
079    Assert.assertEquals(Integer.MAX_VALUE,
080      UVIntTool.getInt(new SingleByteBuff(ByteBuffer.wrap(UVIntTool.MAX_VALUE_BYTES)), 0));
081  }
082
083  @Test
084  public void testRoundTrips() {
085    Random random = new Random();
086    for (int i = 0; i < 10000; ++i) {
087      int value = random.nextInt(Integer.MAX_VALUE);
088      byte[] bytes = UVIntTool.getBytes(value);
089      int roundTripped = UVIntTool.getInt(new SingleByteBuff(ByteBuffer.wrap(bytes)), 0);
090      Assert.assertEquals(value, roundTripped);
091    }
092  }
093
094  @Test
095  public void testInputStreams() throws IOException {
096    ByteArrayInputStream is;
097    is = new ByteArrayInputStream(new byte[] { 0 });
098    Assert.assertEquals(0, UVIntTool.getInt(is));
099    is = new ByteArrayInputStream(new byte[] { 5 });
100    Assert.assertEquals(5, UVIntTool.getInt(is));
101    is = new ByteArrayInputStream(new byte[] { -128 + 27, 1 });
102    Assert.assertEquals(155, UVIntTool.getInt(is));
103  }
104
105}
