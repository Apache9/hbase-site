042/**
043 * Tests merging a normal table's regions
044 */
045@Category({MiscTests.class, MediumTests.class})
046public class TestMergeTable {
047  private static final Log LOG = LogFactory.getLog(TestMergeTable.class);
048  private final HBaseTestingUtility UTIL = new HBaseTestingUtility();
049  private static final byte [] COLUMN_NAME = Bytes.toBytes("contents");
050  private static final byte [] VALUE;
051  static {
052    // We will use the same value for the rows as that is not really important here
053    String partialValue = String.valueOf(System.currentTimeMillis());
054    StringBuilder val = new StringBuilder();
055    while (val.length() < 1024) {
056      val.append(partialValue);
057    }
058    VALUE = Bytes.toBytes(val.toString());
059  }
060
061  /**
062   * Test merge.
063   * Hand-makes regions of a mergeable size and adds the hand-made regions to
064   * hand-made meta.  The hand-made regions are created offline.  We then start
065   * up mini cluster, disables the hand-made table and starts in on merging.
066   * @throws Exception
067   */
068  @Test (timeout=300000) public void testMergeTable() throws Exception {
069    // Table we are manually creating offline.
070    HTableDescriptor desc = new HTableDescriptor(org.apache.hadoop.hbase.TableName.valueOf(Bytes.toBytes("test")));
071    desc.addFamily(new HColumnDescriptor(COLUMN_NAME));
072
073    // Set maximum regionsize down.
074    UTIL.getConfiguration().setLong(HConstants.HREGION_MAX_FILESIZE, 64L * 1024L * 1024L);
075    // Make it so we don't split.
076    UTIL.getConfiguration().setInt("hbase.regionserver.regionSplitLimit", 0);
077    // Startup hdfs.  Its in here we'll be putting our manually made regions.
078    UTIL.startMiniDFSCluster(1);
079    // Create hdfs hbase rootdir.
080    Path rootdir = UTIL.createRootDir();
081    FileSystem fs = FileSystem.get(UTIL.getConfiguration());
082    if (fs.exists(rootdir)) {
083      if (fs.delete(rootdir, true)) {
084        LOG.info("Cleaned up existing " + rootdir);
085      }
086    }
087
088    // Now create three data regions: The first is too large to merge since it
089    // will be > 64 MB in size. The second two will be smaller and will be
090    // selected for merging.
091
092    // To ensure that the first region is larger than 64MB we need to write at
093    // least 65536 rows. We will make certain by writing 70000
094    byte [] row_70001 = Bytes.toBytes("row_70001");
095    byte [] row_80001 = Bytes.toBytes("row_80001");
096
097    // Create regions and populate them at same time.  Create the tabledir
098    // for them first.
099    new FSTableDescriptors(UTIL.getConfiguration(), fs, rootdir).createTableDescriptor(desc);
100    HRegion [] regions = {
101      createRegion(desc, null, row_70001, 1, 70000, rootdir),
102      createRegion(desc, row_70001, row_80001, 70001, 10000, rootdir),
103      createRegion(desc, row_80001, null, 80001, 11000, rootdir)
104    };
105
106    // Now create the root and meta regions and insert the data regions
107    // created above into hbase:meta
108    setupMeta(rootdir, regions);
109    try {
110      LOG.info("Starting mini zk cluster");
111      UTIL.startMiniZKCluster();
112      LOG.info("Starting mini hbase cluster");
113      UTIL.startMiniHBaseCluster(1, 1);
114      Configuration c = new Configuration(UTIL.getConfiguration());
115      Connection connection = UTIL.getConnection();
116
117      List<HRegionInfo> originalTableRegions =
118        MetaTableAccessor.getTableRegions(connection, desc.getTableName());
119      LOG.info("originalTableRegions size=" + originalTableRegions.size() +
120        "; " + originalTableRegions);
121      Admin admin = connection.getAdmin();
122      admin.disableTable(desc.getTableName());
123      admin.close();
124      HMerge.merge(c, FileSystem.get(c), desc.getTableName());
125      List<HRegionInfo> postMergeTableRegions =
126        MetaTableAccessor.getTableRegions(connection, desc.getTableName());
127      LOG.info("postMergeTableRegions size=" + postMergeTableRegions.size() +
128        "; " + postMergeTableRegions);
129      assertTrue("originalTableRegions=" + originalTableRegions.size() +
130        ", postMergeTableRegions=" + postMergeTableRegions.size(),
131        postMergeTableRegions.size() < originalTableRegions.size());
132      LOG.info("Done with merge");
133    } finally {
134      UTIL.shutdownMiniCluster();
135      LOG.info("After cluster shutdown");
136    }
137  }
138
139  private HRegion createRegion(final HTableDescriptor desc,
140      byte [] startKey, byte [] endKey, int firstRow, int nrows, Path rootdir)
141  throws IOException {
142    HRegionInfo hri = new HRegionInfo(desc.getTableName(), startKey, endKey);
143    HRegion region = HBaseTestingUtility.createRegionAndWAL(hri, rootdir, UTIL.getConfiguration(),
144        desc);
145    LOG.info("Created region " + region.getRegionInfo().getRegionNameAsString());
146    for(int i = firstRow; i < firstRow + nrows; i++) {
147      Put put = new Put(Bytes.toBytes("row_" + String.format("%1$05d", i)));
148      put.setDurability(Durability.SKIP_WAL);
149      put.addColumn(COLUMN_NAME, null, VALUE);
150      region.put(put);
151      if (i % 10000 == 0) {
152        LOG.info("Flushing write #" + i);
153        region.flush(true);
154      }
155    }
156    HBaseTestingUtility.closeRegionAndWAL(region);
157    return region;
158  }
159
160  protected void setupMeta(Path rootdir, final HRegion [] regions)
161  throws IOException {
162    HRegion meta =
163      HBaseTestingUtility.createRegionAndWAL(HRegionInfo.FIRST_META_REGIONINFO, rootdir,
164          UTIL.getConfiguration(), UTIL.getMetaTableDescriptor());
165    for (HRegion r: regions) {
166      HRegion.addRegionToMETA(meta, r);
167    }
168    HBaseTestingUtility.closeRegionAndWAL(meta);
169  }
