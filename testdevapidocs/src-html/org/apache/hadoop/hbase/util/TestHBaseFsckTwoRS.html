129  @Test (timeout=180000)
130  public void testDupeStartKey() throws Exception {
131    TableName table =
132        TableName.valueOf("tableDupeStartKey");
133    try {
134      setupTable(table);
135      assertNoErrors(doFsck(conf, false));
136      assertEquals(ROWKEYS.length, countRows());
137
138      // Now let's mess it up, by adding a region with a duplicate startkey
139      HRegionInfo hriDupe =
140          createRegion(tbl.getTableDescriptor(), Bytes.toBytes("A"), Bytes.toBytes("A2"));
141      TEST_UTIL.getHBaseCluster().getMaster().assignRegion(hriDupe);
142      TEST_UTIL.getHBaseCluster().getMaster().getAssignmentManager()
143          .waitForAssignment(hriDupe);
144      ServerName server = regionStates.getRegionServerOfRegion(hriDupe);
145      TEST_UTIL.assertRegionOnServer(hriDupe, server, REGION_ONLINE_TIMEOUT);
146
147      HBaseFsck hbck = doFsck(conf, false);
148      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] { HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS, HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS });
149      assertEquals(2, hbck.getOverlapGroups(table).size());
150      assertEquals(ROWKEYS.length, countRows()); // seems like the "bigger" region won.
151
152      // fix the degenerate region.
153      doFsck(conf, true);
154
155      // check that the degenerate region is gone and no data loss
156      HBaseFsck hbck2 = doFsck(conf,false);
157      assertNoErrors(hbck2);
158      assertEquals(0, hbck2.getOverlapGroups(table).size());
159      assertEquals(ROWKEYS.length, countRows());
160    } finally {
161      cleanupTable(table);
162    }
163  }

303  /**
304   * This creates and fixes a bad table where an overlap group of
305   * 3 regions. Set HBaseFsck.maxMerge to 2 to trigger sideline overlapped
306   * region. Mess around the meta data so that closeRegion/offlineRegion
307   * throws exceptions.
308   */
309  @Test (timeout=180000)
310  public void testSidelineOverlapRegion() throws Exception {
311    TableName table =
312        TableName.valueOf("testSidelineOverlapRegion");
313    try {
314      setupTable(table);
315      assertEquals(ROWKEYS.length, countRows());
316
317      // Mess it up by creating an overlap
318      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
319      HMaster master = cluster.getMaster();
320      HRegionInfo hriOverlap1 =
321          createRegion(tbl.getTableDescriptor(), Bytes.toBytes("A"), Bytes.toBytes("AB"));
322      master.assignRegion(hriOverlap1);
323      master.getAssignmentManager().waitForAssignment(hriOverlap1);
324      HRegionInfo hriOverlap2 =
325          createRegion(tbl.getTableDescriptor(), Bytes.toBytes("AB"), Bytes.toBytes("B"));
326      master.assignRegion(hriOverlap2);
327      master.getAssignmentManager().waitForAssignment(hriOverlap2);
328
329      HBaseFsck hbck = doFsck(conf, false);
330      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] {HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS,
331          HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS, HBaseFsck.ErrorReporter.ERROR_CODE.OVERLAP_IN_REGION_CHAIN});
332      assertEquals(3, hbck.getOverlapGroups(table).size());
333      assertEquals(ROWKEYS.length, countRows());
334
335      // mess around the overlapped regions, to trigger NotServingRegionException
336      Multimap<byte[], HBaseFsck.HbckInfo> overlapGroups = hbck.getOverlapGroups(table);
337      ServerName serverName = null;
338      byte[] regionName = null;
339      for (HBaseFsck.HbckInfo hbi: overlapGroups.values()) {
340        if ("A".equals(Bytes.toString(hbi.getStartKey()))
341            && "B".equals(Bytes.toString(hbi.getEndKey()))) {
342          regionName = hbi.getRegionName();
343
344          // get an RS not serving the region to force bad assignment info in to META.
345          int k = cluster.getServerWith(regionName);
346          for (int i = 0; i < 3; i++) {
347            if (i != k) {
348              HRegionServer rs = cluster.getRegionServer(i);
349              serverName = rs.getServerName();
350              break;
351            }
352          }
353
354          HBaseFsckRepair.closeRegionSilentlyAndWait((HConnection) connection,
355              cluster.getRegionServer(k).getServerName(), hbi.getHdfsHRI());
356          admin.offline(regionName);
357          break;
358        }
359      }
360
361      assertNotNull(regionName);
362      assertNotNull(serverName);
363      try (Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService)) {
364        Put put = new Put(regionName);
365        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
366            Bytes.toBytes(serverName.getHostAndPort()));
367        meta.put(put);
368      }
369
370      // fix the problem.
371      HBaseFsck fsck = new HBaseFsck(conf, hbfsckExecutorService);
372      fsck.connect();
373      HBaseFsck.setDisplayFullReport(); // i.e. -details
374      fsck.setTimeLag(0);
375      fsck.setFixAssignments(true);
376      fsck.setFixMeta(true);
377      fsck.setFixHdfsHoles(true);
378      fsck.setFixHdfsOverlaps(true);
379      fsck.setFixHdfsOrphans(true);
380      fsck.setFixVersionFile(true);
381      fsck.setSidelineBigOverlaps(true);
382      fsck.setMaxMerge(2);
383      fsck.onlineHbck();
384      fsck.close();
385
386      // verify that overlaps are fixed, and there are less rows
387      // since one region is sidelined.
388      HBaseFsck hbck2 = doFsck(conf,false);
389      assertNoErrors(hbck2);
390      assertEquals(0, hbck2.getOverlapGroups(table).size());
391      assertTrue(ROWKEYS.length > countRows());
392    } finally {
393      cleanupTable(table);
394    }
395  }

397  @Test(timeout=180000)
398  public void testHBaseFsck() throws Exception {
399    assertNoErrors(doFsck(conf, false));
400    TableName table = TableName.valueOf("tableBadMetaAssign");
401    HTableDescriptor desc = new HTableDescriptor(table);
402    HColumnDescriptor hcd = new HColumnDescriptor(Bytes.toString(FAM));
403    desc.addFamily(hcd); // If a table has no CF's it doesn't get checked
404    createTable(TEST_UTIL, desc, null);
405
406    // We created 1 table, should be fine
407    assertNoErrors(doFsck(conf, false));
408
409    // Now let's mess it up and change the assignment in hbase:meta to
410    // point to a different region server
411    Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService);
412    Scan scan = new Scan();
413    scan.setStartRow(Bytes.toBytes(table+",,"));
414    ResultScanner scanner = meta.getScanner(scan);
415    HRegionInfo hri = null;
416
417    Result res = scanner.next();
418    ServerName currServer =
419        ServerName.parseFrom(res.getValue(HConstants.CATALOG_FAMILY,
420            HConstants.SERVER_QUALIFIER));
421    long startCode = Bytes.toLong(res.getValue(HConstants.CATALOG_FAMILY,
422        HConstants.STARTCODE_QUALIFIER));
423
424    for (JVMClusterUtil.RegionServerThread rs :
425        TEST_UTIL.getHBaseCluster().getRegionServerThreads()) {
426
427      ServerName sn = rs.getRegionServer().getServerName();
428
429      // When we find a diff RS, change the assignment and break
430      if (!currServer.getHostAndPort().equals(sn.getHostAndPort()) ||
431          startCode != sn.getStartcode()) {
432        Put put = new Put(res.getRow());
433        put.setDurability(Durability.SKIP_WAL);
434        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
435            Bytes.toBytes(sn.getHostAndPort()));
436        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.STARTCODE_QUALIFIER,
437            Bytes.toBytes(sn.getStartcode()));
438        meta.put(put);
439        hri = MetaTableAccessor.getHRegionInfo(res);
440        break;
441      }
442    }
443
444    // Try to fix the data
445    assertErrors(doFsck(conf, true), new HBaseFsck.ErrorReporter.ERROR_CODE[]{
446        HBaseFsck.ErrorReporter.ERROR_CODE.SERVER_DOES_NOT_MATCH_META});
447
448    TEST_UTIL.getHBaseCluster().getMaster()
449        .getAssignmentManager().waitForAssignment(hri);
450
451    // Should be fixed now
452    assertNoErrors(doFsck(conf, false));
453
454    // comment needed - what is the purpose of this line
455    Table t = connection.getTable(table, tableExecutorService);
456    ResultScanner s = t.getScanner(new Scan());
457    s.close();
458    t.close();
459
460    scanner.close();
461    meta.close();
462  }

464  /**
465   * This creates and fixes a bad table with a missing region -- hole in meta and data present but
466   * .regioninfo missing (an orphan hdfs region)in the fs. At last we check every row was present
467   * at the correct region.
468   */
469  @Test(timeout = 180000)
470  public void testHDFSRegioninfoMissingAndCheckRegionBoundary() throws Exception {
471    TableName table = TableName.valueOf("testHDFSRegioninfoMissingAndCheckRegionBoundary");
472    try {
473      setupTable(table);
474      assertEquals(ROWKEYS.length, countRows());
475
476      // Mess it up by leaving a hole in the meta data
477      admin.disableTable(table);
478      deleteRegion(conf, tbl.getTableDescriptor(), Bytes.toBytes("B"), Bytes.toBytes("C"), true,
479        true, false, true, HRegionInfo.DEFAULT_REPLICA_ID);
480      admin.enableTable(table);
481
482      HBaseFsck hbck = doFsck(conf, false);
483      assertErrors(hbck,
484        new HBaseFsck.ErrorReporter.ERROR_CODE[] {
485            HBaseFsck.ErrorReporter.ERROR_CODE.ORPHAN_HDFS_REGION,
486            HBaseFsck.ErrorReporter.ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
487            HBaseFsck.ErrorReporter.ERROR_CODE.HOLE_IN_REGION_CHAIN });
488      // holes are separate from overlap groups
489      assertEquals(0, hbck.getOverlapGroups(table).size());
490
491      // fix hole
492      doFsck(conf, true);
493
494      // check that hole fixed
495      assertNoErrors(doFsck(conf, false));
496
497      // check data belong to the correct region,every scan should get one row.
498      for (int i = 0; i < ROWKEYS.length; i++) {
499        if (i != ROWKEYS.length - 1) {
500          assertEquals(1, countRows(ROWKEYS[i], ROWKEYS[i + 1]));
501        } else {
502          assertEquals(1, countRows(ROWKEYS[i], null));
503        }
504      }
505
506    } finally {
507      cleanupTable(table);
508    }
