385  /**
386   * This creates and fixes a bad table where an overlap group of
387   * 3 regions. Set HBaseFsck.maxMerge to 2 to trigger sideline overlapped
388   * region. Mess around the meta data so that closeRegion/offlineRegion
389   * throws exceptions.
390   */
391  @Test (timeout=180000)
392  public void testSidelineOverlapRegion() throws Exception {
393    final TableName tableName = TableName.valueOf(name.getMethodName());
394    try {
395      setupTable(tableName);
396      assertEquals(ROWKEYS.length, countRows());
397
398      // Mess it up by creating an overlap
399      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
400      HRegionInfo hriOverlap1 =
401          createRegion(tbl.getTableDescriptor(), Bytes.toBytes("A"), Bytes.toBytes("AB"));
402      TEST_UTIL.assignRegion(hriOverlap1);
403
404      HRegionInfo hriOverlap2 =
405          createRegion(tbl.getTableDescriptor(), Bytes.toBytes("AB"), Bytes.toBytes("B"));
406      TEST_UTIL.assignRegion(hriOverlap2);
407
408      HBaseFsck hbck = doFsck(conf, false);
409      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] {HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS,
410          HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS, HBaseFsck.ErrorReporter.ERROR_CODE.OVERLAP_IN_REGION_CHAIN});
411      assertEquals(3, hbck.getOverlapGroups(tableName).size());
412      assertEquals(ROWKEYS.length, countRows());
413
414      // mess around the overlapped regions, to trigger NotServingRegionException
415      Multimap<byte[], HBaseFsck.HbckInfo> overlapGroups = hbck.getOverlapGroups(tableName);
416      ServerName serverName = null;
417      byte[] regionName = null;
418      for (HBaseFsck.HbckInfo hbi: overlapGroups.values()) {
419        if ("A".equals(Bytes.toString(hbi.getStartKey()))
420            && "B".equals(Bytes.toString(hbi.getEndKey()))) {
421          regionName = hbi.getRegionName();
422
423          // get an RS not serving the region to force bad assignment info in to META.
424          int k = cluster.getServerWith(regionName);
425          for (int i = 0; i < 3; i++) {
426            if (i != k) {
427              HRegionServer rs = cluster.getRegionServer(i);
428              serverName = rs.getServerName();
429              break;
430            }
431          }
432
433          HBaseFsckRepair.closeRegionSilentlyAndWait(connection,
434              cluster.getRegionServer(k).getServerName(), hbi.getHdfsHRI());
435          admin.offline(regionName);
436          break;
437        }
438      }
439
440      assertNotNull(regionName);
441      assertNotNull(serverName);
442      try (Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService)) {
443        Put put = new Put(regionName);
444        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
445            Bytes.toBytes(serverName.getAddress().toString()));
446        meta.put(put);
447      }
448
449      // fix the problem.
450      HBaseFsck fsck = new HBaseFsck(conf, hbfsckExecutorService);
451      fsck.connect();
452      HBaseFsck.setDisplayFullReport(); // i.e. -details
453      fsck.setTimeLag(0);
454      fsck.setFixAssignments(true);
455      fsck.setFixMeta(true);
456      fsck.setFixHdfsHoles(true);
457      fsck.setFixHdfsOverlaps(true);
458      fsck.setFixHdfsOrphans(true);
459      fsck.setFixVersionFile(true);
460      fsck.setSidelineBigOverlaps(true);
461      fsck.setMaxMerge(2);
462      fsck.onlineHbck();
463      fsck.close();
464
465      // verify that overlaps are fixed, and there are less rows
466      // since one region is sidelined.
467      HBaseFsck hbck2 = doFsck(conf,false);
468      assertNoErrors(hbck2);
469      assertEquals(0, hbck2.getOverlapGroups(tableName).size());
470      assertTrue(ROWKEYS.length > countRows());
471    } finally {
472      cleanupTable(tableName);
473    }
474  }

476  @Test(timeout=180000)
477  public void testHBaseFsck() throws Exception {
478    assertNoErrors(doFsck(conf, false));
479    final TableName tableName = TableName.valueOf(name.getMethodName());
480    HTableDescriptor desc = new HTableDescriptor(tableName);
481    HColumnDescriptor hcd = new HColumnDescriptor(Bytes.toString(FAM));
482    desc.addFamily(hcd); // If a tableName has no CF's it doesn't get checked
483    createTable(TEST_UTIL, desc, null);
484
485    // We created 1 table, should be fine
486    assertNoErrors(doFsck(conf, false));
487
488    // Now let's mess it up and change the assignment in hbase:meta to
489    // point to a different region server
490    Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService);
491    Scan scan = new Scan();
492    scan.setStartRow(Bytes.toBytes(tableName+",,"));
493    ResultScanner scanner = meta.getScanner(scan);
494    HRegionInfo hri = null;
495
496    Result res = scanner.next();
497    ServerName currServer =
498        ProtobufUtil.parseServerNameFrom(res.getValue(HConstants.CATALOG_FAMILY,
499            HConstants.SERVER_QUALIFIER));
500    long startCode = Bytes.toLong(res.getValue(HConstants.CATALOG_FAMILY,
501        HConstants.STARTCODE_QUALIFIER));
502
503    for (JVMClusterUtil.RegionServerThread rs :
504        TEST_UTIL.getHBaseCluster().getRegionServerThreads()) {
505
506      ServerName sn = rs.getRegionServer().getServerName();
507
508      // When we find a diff RS, change the assignment and break
509      if (!currServer.getHostAndPort().equals(sn.getHostAndPort()) ||
510          startCode != sn.getStartcode()) {
511        Put put = new Put(res.getRow());
512        put.setDurability(Durability.SKIP_WAL);
513        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER,
514            Bytes.toBytes(sn.getHostAndPort()));
515        put.addColumn(HConstants.CATALOG_FAMILY, HConstants.STARTCODE_QUALIFIER,
516            Bytes.toBytes(sn.getStartcode()));
517        meta.put(put);
518        hri = MetaTableAccessor.getHRegionInfo(res);
519        break;
520      }
521    }
522
523    // Try to fix the data
524    assertErrors(doFsck(conf, true), new HBaseFsck.ErrorReporter.ERROR_CODE[]{
525        HBaseFsck.ErrorReporter.ERROR_CODE.SERVER_DOES_NOT_MATCH_META});
526
527    TEST_UTIL.getHBaseCluster().getMaster()
528        .getAssignmentManager().waitForAssignment(hri);
529
530    // Should be fixed now
531    assertNoErrors(doFsck(conf, false));
532
533    // comment needed - what is the purpose of this line
534    Table t = connection.getTable(tableName, tableExecutorService);
535    ResultScanner s = t.getScanner(new Scan());
536    s.close();
537    t.close();
538
539    scanner.close();
540    meta.close();
541  }
