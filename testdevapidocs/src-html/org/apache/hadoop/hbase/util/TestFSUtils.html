093  @Test public void testcomputeHDFSBlocksDistribution() throws Exception {
094    final int DEFAULT_BLOCK_SIZE = 1024;
095    conf.setLong("dfs.blocksize", DEFAULT_BLOCK_SIZE);
096    MiniDFSCluster cluster = null;
097    Path testFile = null;
098
099    try {
100      // set up a cluster with 3 nodes
101      String hosts[] = new String[] { "host1", "host2", "host3" };
102      cluster = htu.startMiniDFSCluster(hosts);
103      cluster.waitActive();
104      FileSystem fs = cluster.getFileSystem();
105
106      // create a file with two blocks
107      testFile = new Path("/test1.txt");
108      WriteDataToHDFS(fs, testFile, 2*DEFAULT_BLOCK_SIZE);
109
110      // given the default replication factor is 3, the same as the number of
111      // datanodes; the locality index for each host should be 100%,
112      // or getWeight for each host should be the same as getUniqueBlocksWeights
113      final long maxTime = System.currentTimeMillis() + 2000;
114      boolean ok;
115      do {
116        ok = true;
117        FileStatus status = fs.getFileStatus(testFile);
118        HDFSBlocksDistribution blocksDistribution =
119          FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
120        long uniqueBlocksTotalWeight =
121          blocksDistribution.getUniqueBlocksTotalWeight();
122        for (String host : hosts) {
123          long weight = blocksDistribution.getWeight(host);
124          ok = (ok && uniqueBlocksTotalWeight == weight);
125        }
126      } while (!ok && System.currentTimeMillis() < maxTime);
127      assertTrue(ok);
128      } finally {
129      htu.shutdownMiniDFSCluster();
130    }
131
132
133    try {
134      // set up a cluster with 4 nodes
135      String hosts[] = new String[] { "host1", "host2", "host3", "host4" };
136      cluster = htu.startMiniDFSCluster(hosts);
137      cluster.waitActive();
138      FileSystem fs = cluster.getFileSystem();
139
140      // create a file with three blocks
141      testFile = new Path("/test2.txt");
142      WriteDataToHDFS(fs, testFile, 3*DEFAULT_BLOCK_SIZE);
143
144      // given the default replication factor is 3, we will have total of 9
145      // replica of blocks; thus the host with the highest weight should have
146      // weight == 3 * DEFAULT_BLOCK_SIZE
147      final long maxTime = System.currentTimeMillis() + 2000;
148      long weight;
149      long uniqueBlocksTotalWeight;
150      do {
151        FileStatus status = fs.getFileStatus(testFile);
152        HDFSBlocksDistribution blocksDistribution =
153          FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
154        uniqueBlocksTotalWeight = blocksDistribution.getUniqueBlocksTotalWeight();
155
156        String tophost = blocksDistribution.getTopHosts().get(0);
157        weight = blocksDistribution.getWeight(tophost);
158
159        // NameNode is informed asynchronously, so we may have a delay. See HBASE-6175
160      } while (uniqueBlocksTotalWeight != weight && System.currentTimeMillis() < maxTime);
161      assertTrue(uniqueBlocksTotalWeight == weight);
162
163    } finally {
164      htu.shutdownMiniDFSCluster();
165    }
166
167
168    try {
169      // set up a cluster with 4 nodes
170      String hosts[] = new String[] { "host1", "host2", "host3", "host4" };
171      cluster = htu.startMiniDFSCluster(hosts);
172      cluster.waitActive();
173      FileSystem fs = cluster.getFileSystem();
174
175      // create a file with one block
176      testFile = new Path("/test3.txt");
177      WriteDataToHDFS(fs, testFile, DEFAULT_BLOCK_SIZE);
178
179      // given the default replication factor is 3, we will have total of 3
180      // replica of blocks; thus there is one host without weight
181      final long maxTime = System.currentTimeMillis() + 2000;
182      HDFSBlocksDistribution blocksDistribution;
183      do {
184        FileStatus status = fs.getFileStatus(testFile);
185        blocksDistribution = FSUtils.computeHDFSBlocksDistribution(fs, status, 0, status.getLen());
186        // NameNode is informed asynchronously, so we may have a delay. See HBASE-6175
187      }
188      while (blocksDistribution.getTopHosts().size() != 3 && System.currentTimeMillis() < maxTime);
189      assertEquals("Wrong number of hosts distributing blocks.", 3,
190        blocksDistribution.getTopHosts().size());
191    } finally {
192      htu.shutdownMiniDFSCluster();
193    }
194  }

198    final Path rootdir = htu.getDataTestDir();
199    final FileSystem fs = rootdir.getFileSystem(conf);
200    assertNull(FSUtils.getVersion(fs, rootdir));
201    // Write out old format version file.  See if we can read it in and convert.
202    Path versionFile = new Path(rootdir, HConstants.VERSION_FILE_NAME);
203    FSDataOutputStream s = fs.create(versionFile);
204    final String version = HConstants.FILE_SYSTEM_VERSION;
205    s.writeUTF(version);
206    s.close();
207    assertTrue(fs.exists(versionFile));
208    FileStatus [] status = fs.listStatus(versionFile);
209    assertNotNull(status);
210    assertTrue(status.length > 0);
211    String newVersion = FSUtils.getVersion(fs, rootdir);
212    assertEquals(version.length(), newVersion.length());
213    assertEquals(version, newVersion);
214    // File will have been converted. Exercise the pb format
215    assertEquals(version, FSUtils.getVersion(fs, rootdir));
216    FSUtils.checkVersion(fs, rootdir, true);
217  }

219  @Test
220  public void testPermMask() throws Exception {
221    final Path rootdir = htu.getDataTestDir();
222    final FileSystem fs = rootdir.getFileSystem(conf);
223    // default fs permission
224    FsPermission defaultFsPerm = FSUtils.getFilePermissions(fs, conf,
225        HConstants.DATA_FILE_UMASK_KEY);
226    // 'hbase.data.umask.enable' is false. We will get default fs permission.
227    assertEquals(FsPermission.getFileDefault(), defaultFsPerm);
228
229    conf.setBoolean(HConstants.ENABLE_DATA_FILE_UMASK, true);
230    // first check that we don't crash if we don't have perms set
231    FsPermission defaultStartPerm = FSUtils.getFilePermissions(fs, conf,
232        HConstants.DATA_FILE_UMASK_KEY);
233    // default 'hbase.data.umask'is 000, and this umask will be used when
234    // 'hbase.data.umask.enable' is true.
235    // Therefore we will not get the real fs default in this case.
236    // Instead we will get the starting point FULL_RWX_PERMISSIONS
237    assertEquals(new FsPermission(FSUtils.FULL_RWX_PERMISSIONS), defaultStartPerm);
238
239    conf.setStrings(HConstants.DATA_FILE_UMASK_KEY, "077");
240    // now check that we get the right perms
241    FsPermission filePerm = FSUtils.getFilePermissions(fs, conf,
242        HConstants.DATA_FILE_UMASK_KEY);
243    assertEquals(new FsPermission("700"), filePerm);
244
245    // then that the correct file is created
246    Path p = new Path("target" + File.separator + UUID.randomUUID().toString());
247    try {
248      FSDataOutputStream out = FSUtils.create(conf, fs, p, filePerm, null);
249      out.close();
250      FileStatus stat = fs.getFileStatus(p);
251      assertEquals(new FsPermission("700"), stat.getPermission());
252      // and then cleanup
253    } finally {
254      fs.delete(p, true);
255    }
256  }

258  @Test
259  public void testDeleteAndExists() throws Exception {
260    final Path rootdir = htu.getDataTestDir();
261    final FileSystem fs = rootdir.getFileSystem(conf);
262    conf.setBoolean(HConstants.ENABLE_DATA_FILE_UMASK, true);
263    FsPermission perms = FSUtils.getFilePermissions(fs, conf, HConstants.DATA_FILE_UMASK_KEY);
264    // then that the correct file is created
265    String file = UUID.randomUUID().toString();
266    Path p = new Path(htu.getDataTestDir(), "temptarget" + File.separator + file);
267    Path p1 = new Path(htu.getDataTestDir(), "temppath" + File.separator + file);
268    try {
269      FSDataOutputStream out = FSUtils.create(conf, fs, p, perms, null);
270      out.close();
271      assertTrue("The created file should be present", FSUtils.isExists(fs, p));
272      // delete the file with recursion as false. Only the file will be deleted.
273      FSUtils.delete(fs, p, false);
274      // Create another file
275      FSDataOutputStream out1 = FSUtils.create(conf, fs, p1, perms, null);
276      out1.close();
277      // delete the file with recursion as false. Still the file only will be deleted
278      FSUtils.delete(fs, p1, true);
279      assertFalse("The created file should be present", FSUtils.isExists(fs, p1));
280      // and then cleanup
281    } finally {
282      FSUtils.delete(fs, p, true);
283      FSUtils.delete(fs, p1, true);
284    }
285  }

371  @Test public void testDFSHedgedReadMetrics() throws Exception {
372    // Enable hedged reads and set it so the threshold is really low.
373    // Most of this test is taken from HDFS, from TestPread.
374    conf.setInt(DFSConfigKeys.DFS_DFSCLIENT_HEDGED_READ_THREADPOOL_SIZE, 5);
375    conf.setLong(DFSConfigKeys.DFS_DFSCLIENT_HEDGED_READ_THRESHOLD_MILLIS, 0);
376    conf.setLong(DFSConfigKeys.DFS_BLOCK_SIZE_KEY, 4096);
377    conf.setLong(DFSConfigKeys.DFS_CLIENT_READ_PREFETCH_SIZE_KEY, 4096);
378    // Set short retry timeouts so this test runs faster
379    conf.setInt(DFSConfigKeys.DFS_CLIENT_RETRY_WINDOW_BASE, 0);
380    conf.setBoolean("dfs.datanode.transferTo.allowed", false);
381    MiniDFSCluster cluster = new MiniDFSCluster.Builder(conf).numDataNodes(3).build();
382    // Get the metrics.  Should be empty.
383    DFSHedgedReadMetrics metrics = FSUtils.getDFSHedgedReadMetrics(conf);
384    assertEquals(0, metrics.getHedgedReadOps());
385    FileSystem fileSys = cluster.getFileSystem();
386    try {
387      Path p = new Path("preadtest.dat");
388      // We need > 1 blocks to test out the hedged reads.
389      DFSTestUtil.createFile(fileSys, p, 12 * blockSize, 12 * blockSize,
390        blockSize, (short) 3, seed);
391      pReadFile(fileSys, p);
392      cleanupFile(fileSys, p);
393      assertTrue(metrics.getHedgedReadOps() > 0);
394    } finally {
395      fileSys.close();
396      cluster.shutdown();
397    }
398  }

400  // Below is taken from TestPread over in HDFS.
401  static final int blockSize = 4096;
402  static final long seed = 0xDEADBEEFL;
403
404  private void pReadFile(FileSystem fileSys, Path name) throws IOException {
405    FSDataInputStream stm = fileSys.open(name);
406    byte[] expected = new byte[12 * blockSize];
407    Random rand = new Random(seed);
408    rand.nextBytes(expected);
409    // do a sanity check. Read first 4K bytes
410    byte[] actual = new byte[4096];
411    stm.readFully(actual);
412    checkAndEraseData(actual, 0, expected, "Read Sanity Test");
413    // now do a pread for the first 8K bytes
414    actual = new byte[8192];
415    doPread(stm, 0L, actual, 0, 8192);
416    checkAndEraseData(actual, 0, expected, "Pread Test 1");
417    // Now check to see if the normal read returns 4K-8K byte range
418    actual = new byte[4096];
419    stm.readFully(actual);
420    checkAndEraseData(actual, 4096, expected, "Pread Test 2");
421    // Now see if we can cross a single block boundary successfully
422    // read 4K bytes from blockSize - 2K offset
423    stm.readFully(blockSize - 2048, actual, 0, 4096);
424    checkAndEraseData(actual, (blockSize - 2048), expected, "Pread Test 3");
425    // now see if we can cross two block boundaries successfully
426    // read blockSize + 4K bytes from blockSize - 2K offset
427    actual = new byte[blockSize + 4096];
428    stm.readFully(blockSize - 2048, actual);
429    checkAndEraseData(actual, (blockSize - 2048), expected, "Pread Test 4");
430    // now see if we can cross two block boundaries that are not cached
431    // read blockSize + 4K bytes from 10*blockSize - 2K offset
432    actual = new byte[blockSize + 4096];
433    stm.readFully(10 * blockSize - 2048, actual);
434    checkAndEraseData(actual, (10 * blockSize - 2048), expected, "Pread Test 5");
435    // now check that even after all these preads, we can still read
436    // bytes 8K-12K
437    actual = new byte[4096];
438    stm.readFully(actual);
439    checkAndEraseData(actual, 8192, expected, "Pread Test 6");
440    // done
441    stm.close();
442    // check block location caching
443    stm = fileSys.open(name);
444    stm.readFully(1, actual, 0, 4096);
445    stm.readFully(4*blockSize, actual, 0, 4096);
446    stm.readFully(7*blockSize, actual, 0, 4096);
447    actual = new byte[3*4096];
448    stm.readFully(0*blockSize, actual, 0, 3*4096);
449    checkAndEraseData(actual, 0, expected, "Pread Test 7");
450    actual = new byte[8*4096];
451    stm.readFully(3*blockSize, actual, 0, 8*4096);
452    checkAndEraseData(actual, 3*blockSize, expected, "Pread Test 8");
453    // read the tail
454    stm.readFully(11*blockSize+blockSize/2, actual, 0, blockSize/2);
455    IOException res = null;
456    try { // read beyond the end of the file
457      stm.readFully(11*blockSize+blockSize/2, actual, 0, blockSize);
458    } catch (IOException e) {
459      // should throw an exception
460      res = e;
461    }
462    assertTrue("Error reading beyond file boundary.", res != null);
463
464    stm.close();
465  }

467  private void checkAndEraseData(byte[] actual, int from, byte[] expected, String message) {
468    for (int idx = 0; idx < actual.length; idx++) {
469      assertEquals(message+" byte "+(from+idx)+" differs. expected "+
470                        expected[from+idx]+" actual "+actual[idx],
471                        actual[idx], expected[from+idx]);
472      actual[idx] = 0;
473    }
474  }
