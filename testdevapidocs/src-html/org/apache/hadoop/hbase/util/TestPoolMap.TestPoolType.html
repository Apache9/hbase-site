019package org.apache.hadoop.hbase.util;
020
021import java.util.ArrayList;
022import java.util.List;
023import java.util.Random;
024import java.util.concurrent.ExecutionException;
025import java.util.concurrent.atomic.AtomicBoolean;
026
027import junit.framework.TestCase;
028
029import org.apache.hadoop.hbase.testclassification.MiscTests;
030import org.apache.hadoop.hbase.testclassification.SmallTests;
031import org.apache.hadoop.hbase.util.PoolMap.PoolType;
032import org.junit.experimental.categories.Category;
033import org.junit.runner.RunWith;
034import org.junit.runners.Suite;
035
036@RunWith(Suite.class)
037@Suite.SuiteClasses({TestPoolMap.TestRoundRobinPoolType.class, TestPoolMap.TestThreadLocalPoolType.class,
038        TestPoolMap.TestReusablePoolType.class})
039@Category({MiscTests.class, SmallTests.class})
040public class TestPoolMap {
041  public abstract static class TestPoolType extends TestCase {
042    protected PoolMap<String, String> poolMap;
043    protected Random random = new Random();
044
045    protected static final int POOL_SIZE = 3;
046
047    @Override
048    protected void setUp() throws Exception {
049      this.poolMap = new PoolMap<String, String>(getPoolType(), POOL_SIZE);
050    }
051
052    protected abstract PoolType getPoolType();
053
054    @Override
055    protected void tearDown() throws Exception {
056      this.poolMap.clear();
057    }
058
059    protected void runThread(final String randomKey, final String randomValue,
060        final String expectedValue) throws InterruptedException {
061      final AtomicBoolean matchFound = new AtomicBoolean(false);
062      Thread thread = new Thread(new Runnable() {
063        @Override
064        public void run() {
065          poolMap.put(randomKey, randomValue);
066          String actualValue = poolMap.get(randomKey);
067          matchFound.set(expectedValue == null ? actualValue == null
068              : expectedValue.equals(actualValue));
069        }
070      });
071      thread.start();
072      thread.join();
073      assertTrue(matchFound.get());
074    }
075  }
076
077  @Category({MiscTests.class, SmallTests.class})
078  public static class TestRoundRobinPoolType extends TestPoolType {
079    @Override
080    protected PoolType getPoolType() {
081      return PoolType.RoundRobin;
082    }
083
084    public void testSingleThreadedClient() throws InterruptedException,
085        ExecutionException {
086      String randomKey = String.valueOf(random.nextInt());
087      String randomValue = String.valueOf(random.nextInt());
088      // As long as the pool is not full, we'll get null back.
089      // This forces the user to create new values that can be used to populate
090      // the pool.
091      runThread(randomKey, randomValue, null);
092      assertEquals(1, poolMap.size(randomKey));
093    }
094
095    public void testMultiThreadedClients() throws InterruptedException,
096        ExecutionException {
097      for (int i = 0; i < POOL_SIZE; i++) {
098        String randomKey = String.valueOf(random.nextInt());
099        String randomValue = String.valueOf(random.nextInt());
100        // As long as the pool is not full, we'll get null back
101        runThread(randomKey, randomValue, null);
102        // As long as we use distinct keys, each pool will have one value
103        assertEquals(1, poolMap.size(randomKey));
104      }
105      poolMap.clear();
106      String randomKey = String.valueOf(random.nextInt());
107      for (int i = 0; i < POOL_SIZE - 1; i++) {
108        String randomValue = String.valueOf(random.nextInt());
109        // As long as the pool is not full, we'll get null back
110        runThread(randomKey, randomValue, null);
111        // since we use the same key, the pool size should grow
112        assertEquals(i + 1, poolMap.size(randomKey));
113      }
114      // at the end of the day, there should be as many values as we put
115      assertEquals(POOL_SIZE - 1, poolMap.size(randomKey));
116    }
117
118    public void testPoolCap() throws InterruptedException, ExecutionException {
119      String randomKey = String.valueOf(random.nextInt());
120      List<String> randomValues = new ArrayList<String>();
121      for (int i = 0; i < POOL_SIZE * 2; i++) {
122        String randomValue = String.valueOf(random.nextInt());
123        randomValues.add(randomValue);
124        if (i < POOL_SIZE - 1) {
125          // As long as the pool is not full, we'll get null back
126          runThread(randomKey, randomValue, null);
127        } else {
128          // when the pool becomes full, we expect the value we get back to be
129          // what we put earlier, in round-robin order
130          runThread(randomKey, randomValue,
131              randomValues.get((i - POOL_SIZE + 1) % POOL_SIZE));
132        }
133      }
134      assertEquals(POOL_SIZE, poolMap.size(randomKey));
135    }
136
137  }

139  @Category({MiscTests.class, SmallTests.class})
140  public static class TestThreadLocalPoolType extends TestPoolType {
141    @Override
142    protected PoolType getPoolType() {
143      return PoolType.ThreadLocal;
144    }
145
146    public void testSingleThreadedClient() throws InterruptedException,
147        ExecutionException {
148      String randomKey = String.valueOf(random.nextInt());
149      String randomValue = String.valueOf(random.nextInt());
150      // As long as the pool is not full, we should get back what we put
151      runThread(randomKey, randomValue, randomValue);
152      assertEquals(1, poolMap.size(randomKey));
153    }
154
155    public void testMultiThreadedClients() throws InterruptedException,
156        ExecutionException {
157      // As long as the pool is not full, we should get back what we put
158      for (int i = 0; i < POOL_SIZE; i++) {
159        String randomKey = String.valueOf(random.nextInt());
160        String randomValue = String.valueOf(random.nextInt());
161        runThread(randomKey, randomValue, randomValue);
162        assertEquals(1, poolMap.size(randomKey));
163      }
164      String randomKey = String.valueOf(random.nextInt());
165      for (int i = 0; i < POOL_SIZE; i++) {
166        String randomValue = String.valueOf(random.nextInt());
167        runThread(randomKey, randomValue, randomValue);
168        assertEquals(i + 1, poolMap.size(randomKey));
169      }
170    }
171
172    public void testPoolCap() throws InterruptedException, ExecutionException {
173      String randomKey = String.valueOf(random.nextInt());
174      for (int i = 0; i < POOL_SIZE * 2; i++) {
175        String randomValue = String.valueOf(random.nextInt());
176        // as of HBASE-4150, pool limit is no longer used with ThreadLocalPool
177          runThread(randomKey, randomValue, randomValue);
178      }
179      assertEquals(POOL_SIZE * 2, poolMap.size(randomKey));
180    }

182  }

194      String randomValue = String.valueOf(random.nextInt());

198    }
199
200    public void testMultiThreadedClients() throws InterruptedException,
201        ExecutionException {
202      // As long as we poll values we put, the pool size should remain zero
203      for (int i = 0; i < POOL_SIZE; i++) {
204        String randomKey = String.valueOf(random.nextInt());
205        String randomValue = String.valueOf(random.nextInt());
206        runThread(randomKey, randomValue, randomValue);
207        assertEquals(0, poolMap.size(randomKey));
208      }
209      poolMap.clear();
210      String randomKey = String.valueOf(random.nextInt());
211      for (int i = 0; i < POOL_SIZE - 1; i++) {
212        String randomValue = String.valueOf(random.nextInt());
213        runThread(randomKey, randomValue, randomValue);
214        assertEquals(0, poolMap.size(randomKey));
215      }
216      assertEquals(0, poolMap.size(randomKey));
217    }

221      String randomKey = String.valueOf(random.nextInt());

223      for (int i = 0; i < POOL_SIZE * 2; i++) {
224        String randomValue = String.valueOf(random.nextInt());
225        randomValues.add(randomValue);

227      }
228      assertEquals(0, poolMap.size(randomKey));
229    }
