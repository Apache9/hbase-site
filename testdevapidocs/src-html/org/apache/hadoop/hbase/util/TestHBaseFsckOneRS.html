468   * @throws Exception
469   */
470  @Test(timeout=180000)
471  public void testParallelHbck() throws Exception {
472    final ExecutorService service;
473    final Future<HBaseFsck> hbck1,hbck2;
474
475    class RunHbck implements Callable<HBaseFsck> {
476      boolean fail = true;
477      @Override
478      public HBaseFsck call(){
479        Configuration c = new Configuration(conf);
480        c.setInt("hbase.hbck.lockfile.attempts", 1);
481        // HBASE-13574 found that in HADOOP-2.6 and later, the create file would internally retry.
482        // To avoid flakiness of the test, set low max wait time.
483        c.setInt("hbase.hbck.lockfile.maxwaittime", 3);
484        try{
485          return doFsck(c, true); // Exclusive hbck only when fixing
486        } catch(Exception e){
487          if (e.getMessage().contains("Duplicate hbck")) {
488            fail = false;
489          }
490        }
491        // If we reach here, then an exception was caught
492        if (fail) fail();
493        return null;
494      }
495    }
496    service = Executors.newFixedThreadPool(2);
497    hbck1 = service.submit(new RunHbck());
498    hbck2 = service.submit(new RunHbck());
499    service.shutdown();
500    //wait for 15 seconds, for both hbck calls finish
501    service.awaitTermination(15, TimeUnit.SECONDS);
502    HBaseFsck h1 = hbck1.get();
503    HBaseFsck h2 = hbck2.get();
504    // Make sure only one of the calls was successful
505    assert(h1 == null || h2 == null);
506    if (h1 != null) {
507      assert(h1.getRetCode() >= 0);
508    }
509    if (h2 != null) {
510      assert(h2.getRetCode() >= 0);
511    }

514  /**
515   * This test makes sure that with enough retries both parallel instances
516   * of hbck will be completed successfully.
517   *
518   * @throws Exception
519   */
520  @Test (timeout=180000)
521  public void testParallelWithRetriesHbck() throws Exception {
522    final ExecutorService service;
523    final Future<HBaseFsck> hbck1,hbck2;
524
525    // With the ExponentialBackoffPolicyWithLimit (starting with 200 milliseconds sleep time, and
526    // max sleep time of 5 seconds), we can retry around 15 times within 80 seconds before bail out.
527    //
528    // Note: the reason to use 80 seconds is that in HADOOP-2.6 and later, the create file would
529    // retry up to HdfsConstants.LEASE_SOFTLIMIT_PERIOD (60 seconds).  See HBASE-13574 for more
530    // details.
531    final int timeoutInSeconds = 80;
532    final int sleepIntervalInMilliseconds = 200;
533    final int maxSleepTimeInMilliseconds = 6000;
534    final int maxRetryAttempts = 15;
535
536    class RunHbck implements Callable<HBaseFsck>{
537
538      @Override
539      public HBaseFsck call() throws Exception {
540        // Increase retry attempts to make sure the non-active hbck doesn't get starved
541        Configuration c = new Configuration(conf);
542        c.setInt("hbase.hbck.lockfile.maxwaittime", timeoutInSeconds);
543        c.setInt("hbase.hbck.lockfile.attempt.sleep.interval", sleepIntervalInMilliseconds);
544        c.setInt("hbase.hbck.lockfile.attempt.maxsleeptime", maxSleepTimeInMilliseconds);
545        c.setInt("hbase.hbck.lockfile.attempts", maxRetryAttempts);
546        return doFsck(c, false);
547      }
548    }
549
550    service = Executors.newFixedThreadPool(2);
551    hbck1 = service.submit(new RunHbck());
552    hbck2 = service.submit(new RunHbck());
553    service.shutdown();
554    //wait for some time, for both hbck calls finish
555    service.awaitTermination(timeoutInSeconds * 2, TimeUnit.SECONDS);
556    HBaseFsck h1 = hbck1.get();
557    HBaseFsck h2 = hbck2.get();
558    // Both should be successful
559    assertNotNull(h1);
560    assertNotNull(h2);
561    assert(h1.getRetCode() >= 0);
562    assert(h2.getRetCode() >= 0);
563
564  }
