471  @Test(timeout=180000)
472  public void testParallelHbck() throws Exception {
473    final ExecutorService service;
474    final Future<HBaseFsck> hbck1,hbck2;
475
476    class RunHbck implements Callable<HBaseFsck> {
477      boolean fail = true;
478      @Override
479      public HBaseFsck call(){
480        Configuration c = new Configuration(conf);
481        c.setInt("hbase.hbck.lockfile.attempts", 1);
482        // HBASE-13574 found that in HADOOP-2.6 and later, the create file would internally retry.
483        // To avoid flakiness of the test, set low max wait time.
484        c.setInt("hbase.hbck.lockfile.maxwaittime", 3);
485        try{
486          return doFsck(c, true); // Exclusive hbck only when fixing

488          if (e.getMessage().contains("Duplicate hbck")) {
489            fail = false;
490          }
491        }
492        // If we reach here, then an exception was caught
493        if (fail) fail();
494        return null;
495      }
496    }
497    service = Executors.newFixedThreadPool(2);
498    hbck1 = service.submit(new RunHbck());
499    hbck2 = service.submit(new RunHbck());
500    service.shutdown();
501    //wait for 15 seconds, for both hbck calls finish
502    service.awaitTermination(15, TimeUnit.SECONDS);
503    HBaseFsck h1 = hbck1.get();
504    HBaseFsck h2 = hbck2.get();
505    // Make sure only one of the calls was successful
506    assert(h1 == null || h2 == null);
507    if (h1 != null) {
508      assert(h1.getRetCode() >= 0);
509    }
510    if (h2 != null) {
511      assert(h2.getRetCode() >= 0);
512    }

515  /**
516   * This test makes sure that with enough retries both parallel instances
517   * of hbck will be completed successfully.
518   *
519   * @throws Exception
520   */
521  @Test (timeout=180000)
522  public void testParallelWithRetriesHbck() throws Exception {
523    final ExecutorService service;
524    final Future<HBaseFsck> hbck1,hbck2;
525
526    // With the ExponentialBackoffPolicyWithLimit (starting with 200 milliseconds sleep time, and
527    // max sleep time of 5 seconds), we can retry around 15 times within 80 seconds before bail out.
528    //
529    // Note: the reason to use 80 seconds is that in HADOOP-2.6 and later, the create file would
530    // retry up to HdfsConstants.LEASE_SOFTLIMIT_PERIOD (60 seconds).  See HBASE-13574 for more
531    // details.
532    final int timeoutInSeconds = 80;
533    final int sleepIntervalInMilliseconds = 200;
534    final int maxSleepTimeInMilliseconds = 6000;
535    final int maxRetryAttempts = 15;
536
537    class RunHbck implements Callable<HBaseFsck>{
538
539      @Override
540      public HBaseFsck call() throws Exception {
541        // Increase retry attempts to make sure the non-active hbck doesn't get starved
542        Configuration c = new Configuration(conf);
543        c.setInt("hbase.hbck.lockfile.maxwaittime", timeoutInSeconds);
544        c.setInt("hbase.hbck.lockfile.attempt.sleep.interval", sleepIntervalInMilliseconds);
545        c.setInt("hbase.hbck.lockfile.attempt.maxsleeptime", maxSleepTimeInMilliseconds);
546        c.setInt("hbase.hbck.lockfile.attempts", maxRetryAttempts);
547        return doFsck(c, false);
548      }
549    }
550
551    service = Executors.newFixedThreadPool(2);
552    hbck1 = service.submit(new RunHbck());
553    hbck2 = service.submit(new RunHbck());
554    service.shutdown();
555    //wait for some time, for both hbck calls finish
556    service.awaitTermination(timeoutInSeconds * 2, TimeUnit.SECONDS);
557    HBaseFsck h1 = hbck1.get();
558    HBaseFsck h2 = hbck2.get();
559    // Both should be successful
560    assertNotNull(h1);
561    assertNotNull(h2);
562    assert(h1.getRetCode() >= 0);
563    assert(h2.getRetCode() >= 0);
564
565  }
