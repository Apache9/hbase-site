477   * This test makes sure that parallel instances of Hbck is disabled.
478   *
479   * @throws Exception
480   */
481  @Test(timeout=180000)
482  public void testParallelHbck() throws Exception {
483    final ExecutorService service;
484    final Future<HBaseFsck> hbck1,hbck2;
485
486    class RunHbck implements Callable<HBaseFsck> {
487      boolean fail = true;
488      @Override
489      public HBaseFsck call(){
490        Configuration c = new Configuration(conf);
491        c.setInt("hbase.hbck.lockfile.attempts", 1);
492        // HBASE-13574 found that in HADOOP-2.6 and later, the create file would internally retry.
493        // To avoid flakiness of the test, set low max wait time.
494        c.setInt("hbase.hbck.lockfile.maxwaittime", 3);
495        try{
496          return doFsck(c, true); // Exclusive hbck only when fixing
497        } catch(Exception e){
498          if (e.getMessage().contains("Duplicate hbck")) {
499            fail = false;
500          }
501        }
502        // If we reach here, then an exception was caught
503        if (fail) fail();
504        return null;
505      }
506    }
507    service = Executors.newFixedThreadPool(2);
508    hbck1 = service.submit(new RunHbck());
509    hbck2 = service.submit(new RunHbck());
510    service.shutdown();
511    //wait for 15 seconds, for both hbck calls finish
512    service.awaitTermination(15, TimeUnit.SECONDS);
513    HBaseFsck h1 = hbck1.get();
514    HBaseFsck h2 = hbck2.get();
515    // Make sure only one of the calls was successful
516    assert(h1 == null || h2 == null);
517    if (h1 != null) {
518      assert(h1.getRetCode() >= 0);
519    }
520    if (h2 != null) {
521      assert(h2.getRetCode() >= 0);
522    }
523  }

525  /**
526   * This test makes sure that with enough retries both parallel instances
527   * of hbck will be completed successfully.
528   *
529   * @throws Exception
530   */
531  @Test (timeout=180000)
532  public void testParallelWithRetriesHbck() throws Exception {
533    final ExecutorService service;
534    final Future<HBaseFsck> hbck1,hbck2;
535
536    // With the ExponentialBackoffPolicyWithLimit (starting with 200 milliseconds sleep time, and
537    // max sleep time of 5 seconds), we can retry around 15 times within 80 seconds before bail out.
538    //
539    // Note: the reason to use 80 seconds is that in HADOOP-2.6 and later, the create file would
540    // retry up to HdfsConstants.LEASE_SOFTLIMIT_PERIOD (60 seconds).  See HBASE-13574 for more
541    // details.
542    final int timeoutInSeconds = 80;
543    final int sleepIntervalInMilliseconds = 200;
544    final int maxSleepTimeInMilliseconds = 6000;
545    final int maxRetryAttempts = 15;
546
547    class RunHbck implements Callable<HBaseFsck>{
548
549      @Override
550      public HBaseFsck call() throws Exception {
551        // Increase retry attempts to make sure the non-active hbck doesn't get starved
552        Configuration c = new Configuration(conf);
553        c.setInt("hbase.hbck.lockfile.maxwaittime", timeoutInSeconds);
554        c.setInt("hbase.hbck.lockfile.attempt.sleep.interval", sleepIntervalInMilliseconds);
555        c.setInt("hbase.hbck.lockfile.attempt.maxsleeptime", maxSleepTimeInMilliseconds);
556        c.setInt("hbase.hbck.lockfile.attempts", maxRetryAttempts);
557        return doFsck(c, false);
558      }
559    }
560
561    service = Executors.newFixedThreadPool(2);
562    hbck1 = service.submit(new RunHbck());
563    hbck2 = service.submit(new RunHbck());
564    service.shutdown();
565    //wait for some time, for both hbck calls finish
566    service.awaitTermination(timeoutInSeconds * 2, TimeUnit.SECONDS);
567    HBaseFsck h1 = hbck1.get();
568    HBaseFsck h2 = hbck2.get();
569    // Both should be successful
570    assertNotNull(h1);
571    assertNotNull(h2);
572    assert(h1.getRetCode() >= 0);
573    assert(h2.getRetCode() >= 0);
574
575  }
