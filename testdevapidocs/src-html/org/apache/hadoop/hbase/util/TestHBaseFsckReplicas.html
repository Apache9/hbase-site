144  @Test (timeout=180000)
145  public void testHbckWithExcessReplica() throws Exception {
146    TableName table =
147        TableName.valueOf("testHbckWithExcessReplica");
148    try {
149      setupTableWithRegionReplica(table, 2);
150      admin.flush(table);
151      assertNoErrors(doFsck(conf, false));
152      assertEquals(ROWKEYS.length, countRows());
153      // the next few lines inject a location in meta for a replica, and then
154      // asks the master to assign the replica (the meta needs to be injected
155      // for the master to treat the request for assignment as valid; the master
156      // checks the region is valid either from its memory or meta)
157      Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService);
158      List<HRegionInfo> regions = admin.getTableRegions(table);
159      byte[] startKey = Bytes.toBytes("B");
160      byte[] endKey = Bytes.toBytes("C");
161      byte[] metaKey = null;
162      HRegionInfo newHri = null;
163      for (HRegionInfo h : regions) {
164        if (Bytes.compareTo(h.getStartKey(), startKey) == 0  &&
165            Bytes.compareTo(h.getEndKey(), endKey) == 0 &&
166            h.getReplicaId() == HRegionInfo.DEFAULT_REPLICA_ID) {
167          metaKey = h.getRegionName();
168          //create a hri with replicaId as 2 (since we already have replicas with replicaid 0 and 1)
169          newHri = RegionReplicaUtil.getRegionInfoForReplica(h, 2);
170          break;
171        }
172      }
173      Put put = new Put(metaKey);
174      Collection<ServerName> var = admin.getClusterStatus().getServers();
175      ServerName sn = var.toArray(new ServerName[var.size()])[0];
176      //add a location with replicaId as 2 (since we already have replicas with replicaid 0 and 1)
177      MetaTableAccessor.addLocation(put, sn, sn.getStartcode(), -1, 2);
178      meta.put(put);
179      // assign the new replica
180      HBaseFsckRepair.fixUnassigned(admin, newHri);
181      HBaseFsckRepair.waitUntilAssigned(admin, newHri);
182      // now reset the meta row to its original value
183      Delete delete = new Delete(metaKey);
184      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getServerColumn(2));
185      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getStartCodeColumn(2));
186      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getSeqNumColumn(2));
187      meta.delete(delete);
188      meta.close();
189      // check that problem exists
190      HBaseFsck hbck = doFsck(conf, false);
191      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[]{HBaseFsck.ErrorReporter.ERROR_CODE.NOT_IN_META});
192      // fix the problem
193      hbck = doFsck(conf, true);
194      // run hbck again to make sure we don't see any errors
195      hbck = doFsck(conf, false);
196      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[]{});
197    } finally {
198      cleanupTable(table);
199    }
200  }

202  /**
203   * This creates and fixes a bad table with a region that is in meta but has
204   * no deployment or data hdfs. The table has region_replication set to 2.
205   */
206  @Test (timeout=180000)
207  public void testNotInHdfsWithReplicas() throws Exception {
208    TableName table =
209        TableName.valueOf("tableNotInHdfs");
210    try {
211      HRegionInfo[] oldHris = new HRegionInfo[2];
212      setupTableWithRegionReplica(table, 2);
213      assertEquals(ROWKEYS.length, countRows());
214      NavigableMap<HRegionInfo, ServerName> map =
215          MetaTableAccessor.allTableRegions(TEST_UTIL.getConnection(),
216              tbl.getName());
217      int i = 0;
218      // store the HRIs of the regions we will mess up
219      for (Map.Entry<HRegionInfo, ServerName> m : map.entrySet()) {
220        if (m.getKey().getStartKey().length > 0 &&
221            m.getKey().getStartKey()[0] == Bytes.toBytes("B")[0]) {
222          LOG.debug("Initially server hosting " + m.getKey() + " is " + m.getValue());
223          oldHris[i++] = m.getKey();
224        }
225      }
226      // make sure data in regions
227      admin.flush(table);
228
229      // Mess it up by leaving a hole in the hdfs data
230      deleteRegion(conf, tbl.getTableDescriptor(), Bytes.toBytes("B"), Bytes.toBytes("C"), false,
231          false, true); // don't rm meta
232
233      HBaseFsck hbck = doFsck(conf, false);
234      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] { HBaseFsck.ErrorReporter.ERROR_CODE.NOT_IN_HDFS });
235
236      // fix hole
237      doFsck(conf, true);
238
239      // check that hole fixed
240      assertNoErrors(doFsck(conf, false));
241      assertEquals(ROWKEYS.length - 2, countRows());
242
243      // the following code checks whether the old primary/secondary has
244      // been unassigned and the new primary/secondary has been assigned
245      i = 0;
246      HRegionInfo[] newHris = new HRegionInfo[2];
247      // get all table's regions from meta
248      map = MetaTableAccessor.allTableRegions(TEST_UTIL.getConnection(), tbl.getName());
249      // get the HRIs of the new regions (hbck created new regions for fixing the hdfs mess-up)
250      for (Map.Entry<HRegionInfo, ServerName> m : map.entrySet()) {
251        if (m.getKey().getStartKey().length > 0 &&
252            m.getKey().getStartKey()[0] == Bytes.toBytes("B")[0]) {
253          newHris[i++] = m.getKey();
254        }
255      }
256      // get all the online regions in the regionservers
257      Collection<ServerName> servers = admin.getClusterStatus().getServers();
258      Set<HRegionInfo> onlineRegions = new HashSet<HRegionInfo>();
259      for (ServerName s : servers) {
260        List<HRegionInfo> list = admin.getOnlineRegions(s);
261        onlineRegions.addAll(list);
262      }
263      // the new HRIs must be a subset of the online regions
264      assertTrue(onlineRegions.containsAll(Arrays.asList(newHris)));
265      // the old HRIs must not be part of the set (removeAll would return false if
266      // the set didn't change)
267      assertFalse(onlineRegions.removeAll(Arrays.asList(oldHris)));
268    } finally {
269      cleanupTable(table);
270      admin.close();
271    }
272  }
