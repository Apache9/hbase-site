177  @Test (timeout=180000)
178  public void testHbckWithExcessReplica() throws Exception {
179    final TableName tableName = TableName.valueOf(name.getMethodName());
180    try {
181      setupTableWithRegionReplica(tableName, 2);
182      admin.flush(tableName);
183      assertNoErrors(doFsck(conf, false));
184      assertEquals(ROWKEYS.length, countRows());
185      // the next few lines inject a location in meta for a replica, and then
186      // asks the master to assign the replica (the meta needs to be injected
187      // for the master to treat the request for assignment as valid; the master
188      // checks the region is valid either from its memory or meta)
189      Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService);
190      List<HRegionInfo> regions = admin.getTableRegions(tableName);
191      byte[] startKey = Bytes.toBytes("B");
192      byte[] endKey = Bytes.toBytes("C");
193      byte[] metaKey = null;
194      HRegionInfo newHri = null;
195      for (HRegionInfo h : regions) {
196        if (Bytes.compareTo(h.getStartKey(), startKey) == 0  &&
197            Bytes.compareTo(h.getEndKey(), endKey) == 0 &&
198            h.getReplicaId() == HRegionInfo.DEFAULT_REPLICA_ID) {
199          metaKey = h.getRegionName();
200          //create a hri with replicaId as 2 (since we already have replicas with replicaid 0 and 1)
201          newHri = RegionReplicaUtil.getRegionInfoForReplica(h, 2);
202          break;
203        }
204      }
205      Put put = new Put(metaKey);
206      Collection<ServerName> var = admin.getClusterStatus().getServers();
207      ServerName sn = var.toArray(new ServerName[var.size()])[0];
208      //add a location with replicaId as 2 (since we already have replicas with replicaid 0 and 1)
209      MetaTableAccessor.addLocation(put, sn, sn.getStartcode(), -1, 2);
210      meta.put(put);
211      // assign the new replica
212      HBaseFsckRepair.fixUnassigned(admin, newHri);
213      HBaseFsckRepair.waitUntilAssigned(admin, newHri);
214      // now reset the meta row to its original value
215      Delete delete = new Delete(metaKey);
216      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getServerColumn(2));
217      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getStartCodeColumn(2));
218      delete.addColumns(HConstants.CATALOG_FAMILY, MetaTableAccessor.getSeqNumColumn(2));
219      meta.delete(delete);
220      meta.close();
221      // check that problem exists
222      HBaseFsck hbck = doFsck(conf, false);
223      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[]{HBaseFsck.ErrorReporter.ERROR_CODE.NOT_IN_META});
224      // fix the problem
225      hbck = doFsck(conf, true);
226      // run hbck again to make sure we don't see any errors
227      hbck = doFsck(conf, false);
228      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[]{});
229    } finally {
230      cleanupTable(tableName);
231    }
232  }

234  /**
235   * This creates and fixes a bad table with a region that is in meta but has
236   * no deployment or data hdfs. The table has region_replication set to 2.
237   */
238  @Test (timeout=180000)
239  public void testNotInHdfsWithReplicas() throws Exception {
240    final TableName tableName = TableName.valueOf(name.getMethodName());
241    try {
242      HRegionInfo[] oldHris = new HRegionInfo[2];
243      setupTableWithRegionReplica(tableName, 2);
244      assertEquals(ROWKEYS.length, countRows());
245      NavigableMap<HRegionInfo, ServerName> map =
246          MetaTableAccessor.allTableRegions(TEST_UTIL.getConnection(),
247              tbl.getName());
248      int i = 0;
249      // store the HRIs of the regions we will mess up
250      for (Map.Entry<HRegionInfo, ServerName> m : map.entrySet()) {
251        if (m.getKey().getStartKey().length > 0 &&
252            m.getKey().getStartKey()[0] == Bytes.toBytes("B")[0]) {
253          LOG.debug("Initially server hosting " + m.getKey() + " is " + m.getValue());
254          oldHris[i++] = m.getKey();
255        }
256      }
257      // make sure data in regions
258      admin.flush(tableName);
259
260      // Mess it up by leaving a hole in the hdfs data
261      deleteRegion(conf, tbl.getTableDescriptor(), Bytes.toBytes("B"), Bytes.toBytes("C"), false,
262          false, true); // don't rm meta
263
264      HBaseFsck hbck = doFsck(conf, false);
265      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] { HBaseFsck.ErrorReporter.ERROR_CODE.NOT_IN_HDFS });
266
267      // fix hole
268      doFsck(conf, true);
269
270      // check that hole fixed
271      assertNoErrors(doFsck(conf, false));
272      assertEquals(ROWKEYS.length - 2, countRows());
273
274      // the following code checks whether the old primary/secondary has
275      // been unassigned and the new primary/secondary has been assigned
276      i = 0;
277      HRegionInfo[] newHris = new HRegionInfo[2];
278      // get all table's regions from meta
279      map = MetaTableAccessor.allTableRegions(TEST_UTIL.getConnection(), tbl.getName());
280      // get the HRIs of the new regions (hbck created new regions for fixing the hdfs mess-up)
281      for (Map.Entry<HRegionInfo, ServerName> m : map.entrySet()) {
282        if (m.getKey().getStartKey().length > 0 &&
283            m.getKey().getStartKey()[0] == Bytes.toBytes("B")[0]) {
284          newHris[i++] = m.getKey();
285        }
286      }
287      // get all the online regions in the regionservers
288      Collection<ServerName> servers = admin.getClusterStatus().getServers();
289      Set<HRegionInfo> onlineRegions = new HashSet<>();
290      for (ServerName s : servers) {
291        List<HRegionInfo> list = admin.getOnlineRegions(s);
292        onlineRegions.addAll(list);
293      }
294      // the new HRIs must be a subset of the online regions
295      assertTrue(onlineRegions.containsAll(Arrays.asList(newHris)));
296      // the old HRIs must not be part of the set (removeAll would return false if
297      // the set didn't change)
298      assertFalse(onlineRegions.removeAll(Arrays.asList(oldHris)));
299    } finally {
300      cleanupTable(tableName);
301      admin.close();
302    }
303  }

305  /**
306   * Creates and fixes a bad table with a successful split that have a deployed
307   * start and end keys and region replicas enabled
308   */
309  @Test (timeout=180000)
310  public void testSplitAndDupeRegionWithRegionReplica() throws Exception {
311    TableName table =
312      TableName.valueOf("testSplitAndDupeRegionWithRegionReplica");
313    Table meta = null;
314
315    try {
316      setupTableWithRegionReplica(table, 2);
317
318      assertNoErrors(doFsck(conf, false));
319      assertEquals(ROWKEYS.length, countRows());
320
321      // No Catalog Janitor running
322      admin.enableCatalogJanitor(false);
323      meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService);
324      HRegionLocation loc = this.connection.getRegionLocation(table, SPLITS[0], false);
325      HRegionInfo hriParent = loc.getRegionInfo();
326
327      // Split Region A just before B
328      this.connection.getAdmin().split(table, Bytes.toBytes("A@"));
329      Thread.sleep(1000);
330
331      // We need to make sure the parent region is not in a split state, so we put it in CLOSED state.
332      regionStates.updateRegionState(hriParent, RegionState.State.CLOSED);
333      TEST_UTIL.assignRegion(hriParent);
334      MetaTableAccessor.addRegionToMeta(meta, hriParent);
335      ServerName server = regionStates.getRegionServerOfRegion(hriParent);
336
337      if (server != null)
338        TEST_UTIL.assertRegionOnServer(hriParent, server, REGION_ONLINE_TIMEOUT);
339
340      while (findDeployedHSI(getDeployedHRIs((HBaseAdmin) admin), hriParent) == null) {
341        Thread.sleep(250);
342      }
343
344      LOG.debug("Finished assignment of parent region");
345
346      // TODO why is dupe region different from dupe start keys?
347      HBaseFsck hbck = doFsck(conf, false);
348      assertErrors(hbck, new HBaseFsck.ErrorReporter.ERROR_CODE[] { HBaseFsck.ErrorReporter.ERROR_CODE.NOT_DEPLOYED,
349        HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS,
350        HBaseFsck.ErrorReporter.ERROR_CODE.DUPE_STARTKEYS, HBaseFsck.ErrorReporter.ERROR_CODE.OVERLAP_IN_REGION_CHAIN});
351      assertEquals(3, hbck.getOverlapGroups(table).size());
352
353      // fix the degenerate region.
354      hbck = new HBaseFsck(conf, hbfsckExecutorService);
355      hbck.setDisplayFullReport(); // i.e. -details
356      hbck.setTimeLag(0);
357      hbck.setFixHdfsOverlaps(true);
358      hbck.setRemoveParents(true);
359      hbck.setFixReferenceFiles(true);
360      hbck.setFixHFileLinks(true);
361      hbck.connect();
362      hbck.onlineHbck();
363      hbck.close();
364
365      hbck = doFsck(conf, false);
366
367      assertNoErrors(hbck);
368      assertEquals(0, hbck.getOverlapGroups(table).size());
369      assertEquals(ROWKEYS.length, countRows());
370    } finally {
371      cleanupTable(table);
372    }
373  }
