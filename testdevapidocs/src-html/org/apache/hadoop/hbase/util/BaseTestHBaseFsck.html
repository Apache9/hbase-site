156  /**
157   * This method is used to undeploy a region -- close it and attempt to
158   * remove its state from the Master.
159   */
160  protected void undeployRegion(Connection conn, ServerName sn,
161      HRegionInfo hri) throws IOException, InterruptedException {
162    try {
163      HBaseFsckRepair.closeRegionSilentlyAndWait((HConnection) conn, sn, hri);
164      if (!hri.isMetaTable()) {
165        admin.offline(hri.getRegionName());
166      }
167    } catch (IOException ioe) {
168      LOG.warn("Got exception when attempting to offline region "
169          + Bytes.toString(hri.getRegionName()), ioe);
170    }
171  }

185  /**
186   * Delete a region from assignments, meta, or completely from hdfs.
187   * @param unassign if true unassign region if assigned
188   * @param metaRow  if true remove region's row from META
189   * @param hdfs if true remove region's dir in HDFS
190   * @param regionInfoOnly if true remove a region dir's .regioninfo file
191   * @param replicaId replica id
192   */
193  protected void deleteRegion(Configuration conf, final HTableDescriptor htd,
194      byte[] startKey, byte[] endKey, boolean unassign, boolean metaRow,
195      boolean hdfs, boolean regionInfoOnly, int replicaId)
196          throws IOException, InterruptedException {
197    LOG.info("** Before delete:");
198    dumpMeta(htd.getTableName());
199
200    List<HRegionLocation> locations;
201    try(RegionLocator rl = connection.getRegionLocator(tbl.getName())) {
202      locations = rl.getAllRegionLocations();
203    }
204
205    for (HRegionLocation location : locations) {
206      HRegionInfo hri = location.getRegionInfo();
207      ServerName hsa = location.getServerName();
208      if (Bytes.compareTo(hri.getStartKey(), startKey) == 0
209          && Bytes.compareTo(hri.getEndKey(), endKey) == 0
210          && hri.getReplicaId() == replicaId) {
211
212        LOG.info("RegionName: " +hri.getRegionNameAsString());
213        byte[] deleteRow = hri.getRegionName();
214
215        if (unassign) {
216          LOG.info("Undeploying region " + hri + " from server " + hsa);
217          undeployRegion(connection, hsa, hri);
218        }
219
220        if (regionInfoOnly) {
221          LOG.info("deleting hdfs .regioninfo data: " + hri.toString() + hsa.toString());
222          Path rootDir = FSUtils.getRootDir(conf);
223          FileSystem fs = rootDir.getFileSystem(conf);
224          Path p = new Path(FSUtils.getTableDir(rootDir, htd.getTableName()),
225              hri.getEncodedName());
226          Path hriPath = new Path(p, HRegionFileSystem.REGION_INFO_FILE);
227          fs.delete(hriPath, true);
228        }
229
230        if (hdfs) {
231          LOG.info("deleting hdfs data: " + hri.toString() + hsa.toString());
232          Path rootDir = FSUtils.getRootDir(conf);
233          FileSystem fs = rootDir.getFileSystem(conf);
234          Path p = new Path(FSUtils.getTableDir(rootDir, htd.getTableName()),
235              hri.getEncodedName());
236          HBaseFsck.debugLsr(conf, p);
237          boolean success = fs.delete(p, true);
238          LOG.info("Deleted " + p + " sucessfully? " + success);
239          HBaseFsck.debugLsr(conf, p);
240        }
241
242        if (metaRow) {
243          try (Table meta = connection.getTable(TableName.META_TABLE_NAME, tableExecutorService)) {
244            Delete delete = new Delete(deleteRow);
245            meta.delete(delete);
246          }
247        }
248      }
249      LOG.info(hri.toString() + hsa.toString());
250    }
251
252    TEST_UTIL.getMetaTableRows(htd.getTableName());
253    LOG.info("*** After delete:");
254    dumpMeta(htd.getTableName());
255  }

277  void setupTableWithRegionReplica(TableName tablename, int replicaCount) throws Exception {
278    HTableDescriptor desc = new HTableDescriptor(tablename);
279    desc.setRegionReplication(replicaCount);
280    HColumnDescriptor hcd = new HColumnDescriptor(Bytes.toString(FAM));
281    desc.addFamily(hcd); // If a table has no CF's it doesn't get checked

283

285    List<Put> puts = new ArrayList<Put>();
286    for (byte[] row : ROWKEYS) {
287      Put p = new Put(row);
288      p.addColumn(FAM, Bytes.toBytes("val"), row);
289      puts.add(p);
290    }
291    tbl.put(puts);
292  }

347  /**
348   * Get region info from local cluster.
349   */
350  Map<ServerName, List<String>> getDeployedHRIs(final Admin admin) throws IOException {
351    ClusterStatus status = admin.getClusterStatus();
352    Collection<ServerName> regionServers = status.getServers();
353    Map<ServerName, List<String>> mm =
354        new HashMap<ServerName, List<String>>();
355    for (ServerName hsi : regionServers) {
356      AdminProtos.AdminService.BlockingInterface server = ((HConnection) connection).getAdmin(hsi);
357
358      // list all online regions from this region server
359      List<HRegionInfo> regions = ProtobufUtil.getOnlineRegions(server);
360      List<String> regionNames = new ArrayList<String>();
361      for (HRegionInfo hri : regions) {
362        regionNames.add(hri.getRegionNameAsString());
363      }
364      mm.put(hsi, regionNames);
365    }
366    return mm;
367  }

456  public void testCleanUpDaughtersNotInMetaAfterFailedSplit() throws Exception {
457    TableName table = TableName.valueOf("testCleanUpDaughtersNotInMetaAfterFailedSplit");
458    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
459    try {
460      HTableDescriptor desc = new HTableDescriptor(table);
461      desc.addFamily(new HColumnDescriptor(Bytes.toBytes("f")));
462      createTable(TEST_UTIL, desc, null);
463
464      tbl = connection.getTable(desc.getTableName());
465      for (int i = 0; i < 5; i++) {
466        Put p1 = new Put(("r" + i).getBytes());
467        p1.addColumn(Bytes.toBytes("f"), "q1".getBytes(), "v".getBytes());
468        tbl.put(p1);
469      }
470      admin.flush(desc.getTableName());
471      List<HRegion> regions = cluster.getRegions(desc.getTableName());
472      int serverWith = cluster.getServerWith(regions.get(0).getRegionInfo().getRegionName());
473      HRegionServer regionServer = cluster.getRegionServer(serverWith);
474      cluster.getServerWith(regions.get(0).getRegionInfo().getRegionName());
475      SplitTransactionImpl st = (SplitTransactionImpl)
476        new SplitTransactionFactory(TEST_UTIL.getConfiguration())
477          .create(regions.get(0), Bytes.toBytes("r3"));
478      st.prepare();
479      st.stepsBeforePONR(regionServer, regionServer, false);
480      AssignmentManager am = cluster.getMaster().getAssignmentManager();
481      Map<String, RegionState> regionsInTransition = am.getRegionStates().getRegionsInTransition();
482      for (RegionState state : regionsInTransition.values()) {
483        am.regionOffline(state.getRegion());
484      }
485      Map<HRegionInfo, ServerName> regionsMap = new HashMap<HRegionInfo, ServerName>();
486      regionsMap.put(regions.get(0).getRegionInfo(), regionServer.getServerName());
487      am.assign(regionsMap);
488      am.waitForAssignment(regions.get(0).getRegionInfo());
489      HBaseFsck hbck = doFsck(conf, false);
490      assertErrors(hbck, new ERROR_CODE[] { ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
491          ERROR_CODE.NOT_IN_META_OR_DEPLOYED });
492      // holes are separate from overlap groups
493      assertEquals(0, hbck.getOverlapGroups(table).size());
494
495      // fix hole
496      assertErrors(
497          doFsck(conf, false, true, false, false, false, false, false, false, false, false, false,
498            null),
499          new ERROR_CODE[] { ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
500              ERROR_CODE.NOT_IN_META_OR_DEPLOYED });
501
502      // check that hole fixed
503      assertNoErrors(doFsck(conf, false));
504      assertEquals(5, countRows());
505    } finally {
506      if (tbl != null) {
507        tbl.close();
508        tbl = null;
509      }
510      cleanupTable(table);
511    }
512  }

598  /**
599   * We don't have an easy way to verify that a flush completed, so we loop until we find a
600   * legitimate hfile and return it.
601   * @param fs
602   * @param table
603   * @return Path of a flushed hfile.
604   * @throws IOException
605   */
606  Path getFlushedHFile(FileSystem fs, TableName table) throws IOException {
607    Path tableDir= FSUtils.getTableDir(FSUtils.getRootDir(conf), table);
608    Path regionDir = FSUtils.getRegionDirs(fs, tableDir).get(0);
609    Path famDir = new Path(regionDir, FAM_STR);
610
611    // keep doing this until we get a legit hfile
612    while (true) {

668  protected void doQuarantineTest(TableName table, HBaseFsck hbck, int check,
669                                  int corrupt, int fail, int quar, int missing) throws Exception {
670    try {
671      setupTable(table);
672      assertEquals(ROWKEYS.length, countRows());
673      admin.flush(table); // flush is async.
674
675      // Mess it up by leaving a hole in the assignment, meta, and hdfs data
676      admin.disableTable(table);
677
678      String[] args = {"-sidelineCorruptHFiles", "-repairHoles", "-ignorePreCheckPermission",
679          table.getNameAsString()};
680      HBaseFsck res = hbck.exec(hbfsckExecutorService, args);
681
682      HFileCorruptionChecker hfcc = res.getHFilecorruptionChecker();
683      assertEquals(hfcc.getHFilesChecked(), check);
684      assertEquals(hfcc.getCorrupted().size(), corrupt);
685      assertEquals(hfcc.getFailures().size(), fail);
686      assertEquals(hfcc.getQuarantined().size(), quar);
687      assertEquals(hfcc.getMissing().size(), missing);
688
689      // its been fixed, verify that we can enable
690      admin.enableTableAsync(table);
691      while (!admin.isTableEnabled(table)) {
692        try {
693          Thread.sleep(250);
694        } catch (InterruptedException e) {
695          e.printStackTrace();
696          fail("Interrupted when trying to enable table " + table);
697        }
698      }
699    } finally {
700      cleanupTable(table);
701    }
702  }

847  protected void deleteMetaRegion(Configuration conf, boolean unassign, boolean hdfs,
848                                  boolean regionInfoOnly) throws IOException, InterruptedException {
849    HRegionLocation metaLocation = connection.getRegionLocator(TableName.META_TABLE_NAME)
850        .getRegionLocation(HConstants.EMPTY_START_ROW);
851    ServerName hsa = metaLocation.getServerName();
852    HRegionInfo hri = metaLocation.getRegionInfo();
853    if (unassign) {
854      LOG.info("Undeploying meta region " + hri + " from server " + hsa);
855      try (Connection unmanagedConnection = ConnectionFactory.createConnection(conf)) {
856        undeployRegion(unmanagedConnection, hsa, hri);
857      }
858    }
859
860    if (regionInfoOnly) {
861      LOG.info("deleting hdfs .regioninfo data: " + hri.toString() + hsa.toString());
862      Path rootDir = FSUtils.getRootDir(conf);
863      FileSystem fs = rootDir.getFileSystem(conf);
864      Path p = new Path(rootDir + "/" + TableName.META_TABLE_NAME.getNameAsString(),
865          hri.getEncodedName());
866      Path hriPath = new Path(p, HRegionFileSystem.REGION_INFO_FILE);
867      fs.delete(hriPath, true);
868    }
869
870    if (hdfs) {
871      LOG.info("deleting hdfs data: " + hri.toString() + hsa.toString());
872      Path rootDir = FSUtils.getRootDir(conf);
873      FileSystem fs = rootDir.getFileSystem(conf);
874      Path p = new Path(rootDir + "/" + TableName.META_TABLE_NAME.getNameAsString(),
875          hri.getEncodedName());
876      HBaseFsck.debugLsr(conf, p);
877      boolean success = fs.delete(p, true);
878      LOG.info("Deleted " + p + " sucessfully? " + success);
879      HBaseFsck.debugLsr(conf, p);
880    }
881  }

918  public static void createTable(HBaseTestingUtility testUtil, HTableDescriptor htd,
919    byte [][] splitKeys) throws Exception {
920    // NOTE: We need a latch because admin is not sync,
921    // so the postOp coprocessor method may be called after the admin operation returned.
922    MasterSyncObserver observer = (MasterSyncObserver)testUtil.getHBaseCluster().getMaster()
923      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class.getName());
924    observer.tableCreationLatch = new CountDownLatch(1);
925    if (splitKeys != null) {
926      admin.createTable(htd, splitKeys);
927    } else {
928      admin.createTable(htd);
929    }
930    observer.tableCreationLatch.await();
931    observer.tableCreationLatch = null;
932    testUtil.waitUntilAllRegionsAssigned(htd.getTableName());
933  }

935  public static void deleteTable(HBaseTestingUtility testUtil, TableName tableName)
936    throws Exception {
937    // NOTE: We need a latch because admin is not sync,
938    // so the postOp coprocessor method may be called after the admin operation returned.
939    MasterSyncObserver observer = (MasterSyncObserver)testUtil.getHBaseCluster().getMaster()
940      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class.getName());
941    observer.tableDeletionLatch = new CountDownLatch(1);
942    try {
943      admin.disableTable(tableName);
944    } catch (Exception e) {
945      LOG.debug("Table: " + tableName + " already disabled, so just deleting it.");
946    }
947    admin.deleteTable(tableName);
948    observer.tableDeletionLatch.await();
949    observer.tableDeletionLatch = null;
950  }
