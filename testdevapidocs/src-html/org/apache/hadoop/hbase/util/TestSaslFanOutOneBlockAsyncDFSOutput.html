065@RunWith(Parameterized.class)
066@Category({ MiscTests.class, MediumTests.class })
067public class TestSaslFanOutOneBlockAsyncDFSOutput {
068
069  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
070
071  private static DistributedFileSystem FS;
072
073  private static EventLoopGroup EVENT_LOOP_GROUP;
074
075  private static int READ_TIMEOUT_MS = 200000;
076
077  private static final File KEYTAB_FILE = new File(
078      TEST_UTIL.getDataTestDir("keytab").toUri().getPath());
079
080  private static MiniKdc KDC;
081
082  private static String HOST = "localhost";
083
084  private static String USERNAME;
085
086  private static String PRINCIPAL;
087
088  private static String HTTP_PRINCIPAL;
089  @Rule
090  public TestName name = new TestName();
091
092  @Parameter(0)
093  public String protection;
094
095  @Parameter(1)
096  public String encryptionAlgorithm;
097
098  @Parameters(name = "{index}: protection={0}, encryption={1}")
099  public static Iterable<Object[]> data() {
100    List<Object[]> params = new ArrayList<>();
101    for (String protection : Arrays.asList("authentication", "integrity", "privacy")) {
102      for (String encryptionAlgorithm : Arrays.asList("", "3des", "rc4")) {
103        params.add(new Object[] { protection, encryptionAlgorithm });
104      }
105    }
106    return params;
107  }
108
109  private static void setHdfsSecuredConfiguration(Configuration conf) throws Exception {
110    // change XXX_USER_NAME_KEY to XXX_KERBEROS_PRINCIPAL_KEY after we drop support for hadoop-2.4.1
111    conf.set(DFSConfigKeys.DFS_NAMENODE_USER_NAME_KEY, PRINCIPAL + "@" + KDC.getRealm());
112    conf.set(DFSConfigKeys.DFS_NAMENODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
113    conf.set(DFSConfigKeys.DFS_DATANODE_USER_NAME_KEY, PRINCIPAL + "@" + KDC.getRealm());
114    conf.set(DFSConfigKeys.DFS_DATANODE_KEYTAB_FILE_KEY, KEYTAB_FILE.getAbsolutePath());
115    conf.set(DFSConfigKeys.DFS_WEB_AUTHENTICATION_KERBEROS_PRINCIPAL_KEY,
116      HTTP_PRINCIPAL + "@" + KDC.getRealm());
117    conf.setBoolean(DFSConfigKeys.DFS_BLOCK_ACCESS_TOKEN_ENABLE_KEY, true);
118    conf.set(DFSConfigKeys.DFS_HTTP_POLICY_KEY, HttpConfig.Policy.HTTPS_ONLY.name());
119    conf.set(DFSConfigKeys.DFS_NAMENODE_HTTPS_ADDRESS_KEY, "localhost:0");
120    conf.set(DFSConfigKeys.DFS_DATANODE_HTTPS_ADDRESS_KEY, "localhost:0");
121
122    File keystoresDir = new File(TEST_UTIL.getDataTestDir("keystore").toUri().getPath());
123    keystoresDir.mkdirs();
124    String sslConfDir = KeyStoreTestUtil.getClasspathDir(TestGenerateDelegationToken.class);
125    KeyStoreTestUtil.setupSSLConfig(keystoresDir.getAbsolutePath(), sslConfDir, conf, false);
126
127    conf.setBoolean("ignore.secure.ports.for.testing", true);
128  }
129
130  @BeforeClass
131  public static void setUpBeforeClass() throws Exception {
132    Logger.getLogger("org.apache.hadoop.hdfs.StateChange").setLevel(Level.DEBUG);
133    Logger.getLogger("BlockStateChange").setLevel(Level.DEBUG);
134    EVENT_LOOP_GROUP = new NioEventLoopGroup();
135    TEST_UTIL.getConfiguration().setInt(DFS_CLIENT_SOCKET_TIMEOUT_KEY, READ_TIMEOUT_MS);
136    Properties conf = MiniKdc.createConf();
137    conf.put(MiniKdc.DEBUG, true);
138    KDC = new MiniKdc(conf, new File(TEST_UTIL.getDataTestDir("kdc").toUri().getPath()));
139    KDC.start();
140    USERNAME = UserGroupInformation.getLoginUser().getShortUserName();
141    PRINCIPAL = USERNAME + "/" + HOST;
142    HTTP_PRINCIPAL = "HTTP/" + HOST;
143    KDC.createPrincipal(KEYTAB_FILE, PRINCIPAL, HTTP_PRINCIPAL);
144    setHdfsSecuredConfiguration(TEST_UTIL.getConfiguration());
145    HBaseKerberosUtils.setKeytabFileForTesting(KEYTAB_FILE.getAbsolutePath());
146    HBaseKerberosUtils.setPrincipalForTesting(PRINCIPAL + "@" + KDC.getRealm());
147    HBaseKerberosUtils.setSecuredConfiguration(TEST_UTIL.getConfiguration());
148    UserGroupInformation.setConfiguration(TEST_UTIL.getConfiguration());
149  }
150
151  @AfterClass
152  public static void tearDownAfterClass() throws IOException, InterruptedException {
153    if (EVENT_LOOP_GROUP != null) {
154      EVENT_LOOP_GROUP.shutdownGracefully().sync();
155    }
156    if (KDC != null) {
157      KDC.stop();
158    }
159  }
160
161  @Before
162  public void setUp() throws Exception {
163    TEST_UTIL.getConfiguration().set("dfs.data.transfer.protection", protection);
164    if (StringUtils.isBlank(encryptionAlgorithm)) {
165      TEST_UTIL.getConfiguration().setBoolean(DFS_ENCRYPT_DATA_TRANSFER_KEY, false);
166      TEST_UTIL.getConfiguration().unset(DFS_DATA_ENCRYPTION_ALGORITHM_KEY);
167    } else {
168      TEST_UTIL.getConfiguration().setBoolean(DFS_ENCRYPT_DATA_TRANSFER_KEY, true);
169      TEST_UTIL.getConfiguration().set(DFS_DATA_ENCRYPTION_ALGORITHM_KEY, encryptionAlgorithm);
170    }
171    TEST_UTIL.startMiniDFSCluster(3);
172    FS = TEST_UTIL.getDFSCluster().getFileSystem();
173  }
174
175  @After
176  public void tearDown() throws IOException {
177    TEST_UTIL.shutdownMiniDFSCluster();
178  }
179
180  private Path getTestFile() {
181    return new Path("/" + name.getMethodName().replaceAll("[^0-9a-zA-Z]", "_"));
182  }
183
184  @Test
185  public void test() throws IOException, InterruptedException, ExecutionException {
186    Path f = getTestFile();
187    EventLoop eventLoop = EVENT_LOOP_GROUP.next();
188    final FanOutOneBlockAsyncDFSOutput out = FanOutOneBlockAsyncDFSOutputHelper.createOutput(FS, f,
189      true, false, (short) 1, FS.getDefaultBlockSize(), eventLoop);
190    TestFanOutOneBlockAsyncDFSOutput.writeAndVerify(eventLoop, FS, f, out);
191  }
