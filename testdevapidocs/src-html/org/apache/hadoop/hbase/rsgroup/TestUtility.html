093  @Before
094  public void before() throws IOException {
095    this.abortable = new Abortable() {
096      @Override
097      public void abort(String why, Throwable e) {

103        return false;

107      new ZKWatcher(UTIL.getConfiguration(), this.getClass().getSimpleName(), this.abortable, true);

110  @After
111  public void after() {

123  /**
124   * @param admin An {@link AdminProtos.AdminService.BlockingInterface} instance; you'll likely want
125   *          to pass a mocked HRS; can be null.
126   * @param client A mocked ClientProtocol instance, can be null
127   * @return Mock up a connection that returns a {@link Configuration} when
128   *         {@link org.apache.hadoop.hbase.client.ClusterConnection#getConfiguration()} is called,
129   *         a 'location' when
130   *         {@link org.apache.hadoop.hbase.client.RegionLocator#getRegionLocation(byte[], boolean)}
131   *         is called, and that returns the passed
132   *         {@link AdminProtos.AdminService.BlockingInterface} instance when
133   *         {@link org.apache.hadoop.hbase.client.ClusterConnection#getAdmin(ServerName)} is
134   *         called, returns the passed {@link ClientProtos.ClientService.BlockingInterface}
135   *         instance when
136   *         {@link org.apache.hadoop.hbase.client.ClusterConnection#getClient(ServerName)} is
137   *         called.
138   */
139  private ClusterConnection mockConnection(final AdminProtos.AdminService.BlockingInterface admin,
140      final ClientProtos.ClientService.BlockingInterface client) throws IOException {
141    ClusterConnection connection =
142      HConnectionTestingUtility.getMockedConnection(UTIL.getConfiguration());
143    Mockito.doNothing().when(connection).close();
144    // Make it so we return any old location when asked.
145    final HRegionLocation anyLocation =
146      new HRegionLocation(RegionInfoBuilder.FIRST_META_REGIONINFO, SN);
147    Mockito.when(connection.getRegionLocation((TableName) Mockito.any(), (byte[]) Mockito.any(),
148      Mockito.anyBoolean())).thenReturn(anyLocation);
149    Mockito.when(connection.locateRegion((TableName) Mockito.any(), (byte[]) Mockito.any()))
150      .thenReturn(anyLocation);
151    if (admin != null) {
152      // If a call to getHRegionConnection, return this implementation.
153      Mockito.when(connection.getAdmin(Mockito.any())).thenReturn(admin);
154    }
155    if (client != null) {
156      // If a call to getClient, return this implementation.
157      Mockito.when(connection.getClient(Mockito.any())).thenReturn(client);
158    }
159    return connection;
160  }
161
162  private void testVerifyMetaRegionLocationWithException(Exception ex)
163      throws IOException, InterruptedException, KeeperException, ServiceException {
164    // Mock an ClientProtocol.
165    final ClientProtos.ClientService.BlockingInterface implementation =
166      Mockito.mock(ClientProtos.ClientService.BlockingInterface.class);
167
168    ClusterConnection connection = mockConnection(null, implementation);
169
170    // If a 'get' is called on mocked interface, throw connection refused.
171    Mockito.when(implementation.get((RpcController) Mockito.any(), (GetRequest) Mockito.any()))
172      .thenThrow(new ServiceException(ex));
173
174    long timeout = UTIL.getConfiguration().getLong("hbase.catalog.verification.timeout", 1000);
175    MetaTableLocator.setMetaLocation(this.watcher, SN, RegionState.State.OPENING);
176    assertFalse(Utility.verifyMetaRegionLocation(connection, watcher, timeout));
177
178    MetaTableLocator.setMetaLocation(this.watcher, SN, RegionState.State.OPEN);
179    assertFalse(Utility.verifyMetaRegionLocation(connection, watcher, timeout));
180  }

182  /**
183   * Test get of meta region fails properly if nothing to connect to.
184   */
185  @Test
186  public void testVerifyMetaRegionLocationFails()
187      throws IOException, InterruptedException, KeeperException, ServiceException {
188    ClusterConnection connection = Mockito.mock(ClusterConnection.class);
189    ServiceException connectException =
190      new ServiceException(new ConnectException("Connection refused"));
191    final AdminProtos.AdminService.BlockingInterface implementation =
192      Mockito.mock(AdminProtos.AdminService.BlockingInterface.class);
193    Mockito.when(implementation.getRegionInfo((RpcController) Mockito.any(),
194      (GetRegionInfoRequest) Mockito.any())).thenThrow(connectException);
195    Mockito.when(connection.getAdmin(Mockito.any())).thenReturn(implementation);
196    RpcControllerFactory controllerFactory = Mockito.mock(RpcControllerFactory.class);
197    Mockito.when(controllerFactory.newController())
198      .thenReturn(Mockito.mock(HBaseRpcController.class));
199    Mockito.when(connection.getRpcControllerFactory()).thenReturn(controllerFactory);
200
201    ServerName sn = ServerName.valueOf("example.com", 1234, System.currentTimeMillis());
202    MetaTableLocator.setMetaLocation(this.watcher, sn, RegionState.State.OPENING);
203    assertFalse(Utility.verifyMetaRegionLocation(connection, watcher, 100));
204    MetaTableLocator.setMetaLocation(this.watcher, sn, RegionState.State.OPEN);
205    assertFalse(Utility.verifyMetaRegionLocation(connection, watcher, 100));
206  }
