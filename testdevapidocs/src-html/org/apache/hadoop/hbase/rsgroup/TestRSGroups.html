234  public void testNamespaceConstraint() throws Exception {
235    String nsName = tablePrefix+"_foo";
236    String groupName = tablePrefix+"_foo";
237    LOG.info("testNamespaceConstraint");
238    rsGroupAdmin.addRSGroup(groupName);
239    assertTrue(observer.preAddRSGroupCalled);
240    assertTrue(observer.postAddRSGroupCalled);
241
242    admin.createNamespace(NamespaceDescriptor.create(nsName)
243        .addConfiguration(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP, groupName)
244        .build());
245    //test removing a referenced group
246    try {
247      rsGroupAdmin.removeRSGroup(groupName);
248      fail("Expected a constraint exception");
249    } catch (IOException ex) {
250    }
251    //test modify group
252    //changing with the same name is fine
253    admin.modifyNamespace(
254        NamespaceDescriptor.create(nsName)
255          .addConfiguration(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP, groupName)
256          .build());
257    String anotherGroup = tablePrefix+"_anotherGroup";
258    rsGroupAdmin.addRSGroup(anotherGroup);
259    //test add non-existent group
260    admin.deleteNamespace(nsName);
261    rsGroupAdmin.removeRSGroup(groupName);
262    assertTrue(observer.preRemoveRSGroupCalled);
263    assertTrue(observer.postRemoveRSGroupCalled);
264    try {
265      admin.createNamespace(NamespaceDescriptor.create(nsName)
266          .addConfiguration(RSGroupInfo.NAMESPACE_DESC_PROP_GROUP, "foo")
267          .build());
268      fail("Expected a constraint exception");
269    } catch (IOException ex) {
270    }

547  @Test
548  public void testRSGroupListDoesNotContainFailedTableCreation() throws Exception {
549    toggleQuotaCheckAndRestartMiniCluster(true);
550    String nsp = "np1";
551    NamespaceDescriptor nspDesc =
552        NamespaceDescriptor.create(nsp).addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "5")
553            .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "2").build();
554    admin.createNamespace(nspDesc);
555    assertEquals(3, admin.listNamespaceDescriptors().length);
556    HColumnDescriptor fam1 = new HColumnDescriptor("fam1");
557    HTableDescriptor tableDescOne =
558        new HTableDescriptor(TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table1"));
559    tableDescOne.addFamily(fam1);
560    admin.createTable(tableDescOne);
561
562    HTableDescriptor tableDescTwo =
563        new HTableDescriptor(TableName.valueOf(nsp + TableName.NAMESPACE_DELIM + "table2"));
564    tableDescTwo.addFamily(fam1);
565    boolean constraintViolated = false;
566
567    try {
568      admin.createTable(tableDescTwo, Bytes.toBytes("AAA"), Bytes.toBytes("ZZZ"),
569          6);
570      Assert.fail("Creation table should fail because of quota violation.");
571    } catch (Exception exp) {
572      assertTrue(exp instanceof IOException);
573      constraintViolated = true;
574    } finally {
575      assertTrue("Constraint not violated for table " + tableDescTwo.getTableName(),
576          constraintViolated);
577    }
578    List<RSGroupInfo> rsGroupInfoList = rsGroupAdmin.listRSGroups();
579    boolean foundTable2 = false;
580    boolean foundTable1 = false;
581    for(int i = 0; i < rsGroupInfoList.size(); i++){
582      if(rsGroupInfoList.get(i).getTables().contains(tableDescTwo.getTableName())){
583        foundTable2 = true;
584      }
585      if(rsGroupInfoList.get(i).getTables().contains(tableDescOne.getTableName())){
586        foundTable1 = true;
587      }
588    }
589    assertFalse("Found table2 in rsgroup list.", foundTable2);
590    assertTrue("Did not find table1 in rsgroup list", foundTable1);
591
592    TEST_UTIL.deleteTable(tableDescOne.getTableName());
593    admin.deleteNamespace(nspDesc.getName());
594    toggleQuotaCheckAndRestartMiniCluster(false);
595
596  }
