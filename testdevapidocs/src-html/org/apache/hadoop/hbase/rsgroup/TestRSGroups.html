068  @BeforeClass
069  public static void setUp() throws Exception {
070    TEST_UTIL = new HBaseTestingUtility();
071    TEST_UTIL.getConfiguration().setFloat(
072            "hbase.master.balancer.stochastic.tableSkewCost", 6000);
073    TEST_UTIL.getConfiguration().set(
074        HConstants.HBASE_MASTER_LOADBALANCER_CLASS,
075        RSGroupBasedLoadBalancer.class.getName());
076    TEST_UTIL.getConfiguration().set(CoprocessorHost.MASTER_COPROCESSOR_CONF_KEY,
077        RSGroupAdminEndpoint.class.getName());
078    TEST_UTIL.startMiniCluster(NUM_SLAVES_BASE);
079    TEST_UTIL.getConfiguration().set(
080        ServerManager.WAIT_ON_REGIONSERVERS_MINTOSTART,
081        ""+NUM_SLAVES_BASE);
082
083    admin = TEST_UTIL.getHBaseAdmin();
084    cluster = TEST_UTIL.getHBaseCluster();
085    master = ((MiniHBaseCluster)cluster).getMaster();
086
087    //wait for balancer to come online
088    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
089      @Override
090      public boolean evaluate() throws Exception {
091        return master.isInitialized() &&
092            ((RSGroupBasedLoadBalancer) master.getLoadBalancer()).isOnline();
093      }
094    });
095    admin.setBalancerRunning(false,true);
096    rsGroupAdmin = new VerifyingRSGroupAdminClient(rsGroupAdmin.newClient(TEST_UTIL.getConnection()),
097        TEST_UTIL.getConfiguration());
098    RSGroupAdminEndpoint =
099        master.getMasterCoprocessorHost().findCoprocessors(RSGroupAdminEndpoint.class).get(0);
100  }

116  @After
117  public void afterMethod() throws Exception {
118    deleteTableIfNecessary();
119    deleteNamespaceIfNecessary();
120    deleteGroups();
121
122    int missing = NUM_SLAVES_BASE - getNumServers();
123    LOG.info("Restoring servers: "+missing);
124    for(int i=0; i<missing; i++) {
125      ((MiniHBaseCluster)cluster).startRegionServer();
126    }
127
128    rsGroupAdmin.addRSGroup("master");
129    ServerName masterServerName =
130        ((MiniHBaseCluster)cluster).getMaster().getServerName();
131
132    try {
133      rsGroupAdmin.moveServers(
134          Sets.newHashSet(masterServerName.getHostPort()),
135          "master");
136    } catch (Exception ex) {
137      // ignore
138    }
139    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
140      @Override
141      public boolean evaluate() throws Exception {
142        LOG.info("Waiting for cleanup to finish " + rsGroupAdmin.listRSGroups());
143        //Might be greater since moving servers back to default
144        //is after starting a server
145
146        return rsGroupAdmin.getRSGroupInfo(RSGroupInfo.DEFAULT_GROUP).getServers().size()
147            == NUM_SLAVES_BASE;
148      }
149    });
150  }

162  @Test
163  public void testNamespaceCreateAndAssign() throws Exception {
164    LOG.info("testNamespaceCreateAndAssign");
165    String nsName = tablePrefix+"_foo";
166    final TableName tableName = TableName.valueOf(nsName, tablePrefix + "_testCreateAndAssign");
167    RSGroupInfo appInfo = addGroup(rsGroupAdmin, "appInfo", 1);
168    admin.createNamespace(NamespaceDescriptor.create(nsName)
169        .addConfiguration(RSGroupInfo.NAMESPACEDESC_PROP_GROUP, "appInfo").build());
170    final HTableDescriptor desc = new HTableDescriptor(tableName);
171    desc.addFamily(new HColumnDescriptor("f"));
172    admin.createTable(desc);
173    //wait for created table to be assigned
174    TEST_UTIL.waitFor(WAIT_TIMEOUT, new Waiter.Predicate<Exception>() {
175      @Override
176      public boolean evaluate() throws Exception {
177        return getTableRegionMap().get(desc.getTableName()) != null;
178      }
179    });
180    ServerName targetServer =
181        ServerName.parseServerName(appInfo.getServers().iterator().next().toString());
182    AdminProtos.AdminService.BlockingInterface rs =
183      ((ClusterConnection) admin.getConnection()).getAdmin(targetServer);
184    //verify it was assigned to the right group
185    Assert.assertEquals(1, ProtobufUtil.getOnlineRegions(rs).size());
186  }

206  @Test
207  public void testNamespaceConstraint() throws Exception {
208    String nsName = tablePrefix+"_foo";
209    String groupName = tablePrefix+"_foo";
210    LOG.info("testNamespaceConstraint");
211    rsGroupAdmin.addRSGroup(groupName);
212    admin.createNamespace(NamespaceDescriptor.create(nsName)
213        .addConfiguration(RSGroupInfo.NAMESPACEDESC_PROP_GROUP, groupName)
214        .build());
215    //test removing a referenced group
216    try {
217      rsGroupAdmin.removeRSGroup(groupName);
218      fail("Expected a constraint exception");
219    } catch (IOException ex) {
220    }
221    //test modify group
222    //changing with the same name is fine
223    admin.modifyNamespace(
224        NamespaceDescriptor.create(nsName)
225          .addConfiguration(RSGroupInfo.NAMESPACEDESC_PROP_GROUP, groupName)
226          .build());
227    String anotherGroup = tablePrefix+"_anotherGroup";
228    rsGroupAdmin.addRSGroup(anotherGroup);
229    //test add non-existent group
230    admin.deleteNamespace(nsName);
231    rsGroupAdmin.removeRSGroup(groupName);
232    try {
233      admin.createNamespace(NamespaceDescriptor.create(nsName)
234          .addConfiguration(RSGroupInfo.NAMESPACEDESC_PROP_GROUP, "foo")
235          .build());
236      fail("Expected a constraint exception");
237    } catch (IOException ex) {
238    }
239  }

252  @Test
253  public void testMisplacedRegions() throws Exception {
254    final TableName tableName = TableName.valueOf(tablePrefix+"_testMisplacedRegions");
255    LOG.info("testMisplacedRegions");
256
257    final RSGroupInfo RSGroupInfo = addGroup(rsGroupAdmin, "testMisplacedRegions", 1);
258
259    TEST_UTIL.createMultiRegionTable(tableName, new byte[]{'f'}, 15);
260    TEST_UTIL.waitUntilAllRegionsAssigned(tableName);
261
262    RSGroupAdminEndpoint.getGroupInfoManager()
263        .moveTables(Sets.newHashSet(tableName), RSGroupInfo.getName());
264
265    assertTrue(rsGroupAdmin.balanceRSGroup(RSGroupInfo.getName()));
266
267    TEST_UTIL.waitFor(60000, new Predicate<Exception>() {
268      @Override
269      public boolean evaluate() throws Exception {
270        ServerName serverName =
271            ServerName.valueOf(RSGroupInfo.getServers().iterator().next().toString(), 1);
272        return admin.getConnection().getAdmin()
273            .getOnlineRegions(serverName).size() == 15;
274      }
275    });
