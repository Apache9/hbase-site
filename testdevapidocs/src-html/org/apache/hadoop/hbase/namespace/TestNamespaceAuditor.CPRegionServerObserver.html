338  @Test
339  public void testRegionMerge() throws Exception {
340    String nsp1 = prefix + "_regiontest";
341    NamespaceDescriptor nspDesc =
342        NamespaceDescriptor.create(nsp1)
343            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "3")
344            .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "2").build();
345    ADMIN.createNamespace(nspDesc);
346    final TableName tableTwo = TableName.valueOf(nsp1 + TableName.NAMESPACE_DELIM + "table2");
347    byte[] columnFamily = Bytes.toBytes("info");
348    HTableDescriptor tableDescOne = new HTableDescriptor(tableTwo);
349    tableDescOne.addFamily(new HColumnDescriptor(columnFamily));
350    final int initialRegions = 3;
351    ADMIN.createTable(tableDescOne, Bytes.toBytes("1"), Bytes.toBytes("2000"), initialRegions);
352    Connection connection = ConnectionFactory.createConnection(UTIL.getConfiguration());
353    try (Table table = connection.getTable(tableTwo)) {
354      UTIL.loadNumericRows(table, Bytes.toBytes("info"), 1000, 1999);
355    }
356    ADMIN.flush(tableTwo);
357    List<HRegionInfo> hris = ADMIN.getTableRegions(tableTwo);
358    Collections.sort(hris);
359    // merge the two regions
360    final Set<String> encodedRegionNamesToMerge =
361        Sets.newHashSet(hris.get(0).getEncodedName(), hris.get(1).getEncodedName());
362    ADMIN.mergeRegionsAsync(
363      hris.get(0).getEncodedNameAsBytes(),
364      hris.get(1).getEncodedNameAsBytes(),
365      false);
366    UTIL.waitFor(10000, 100, new Waiter.ExplainingPredicate<Exception>() {
367
368      @Override
369      public boolean evaluate() throws Exception {
370        RegionStates regionStates =
371            UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
372        for (HRegionInfo hri : ADMIN.getTableRegions(tableTwo)) {
373          if (encodedRegionNamesToMerge.contains(hri.getEncodedName())) {
374            return false;
375          }
376          if (!regionStates.isRegionInState(hri, RegionState.State.OPEN)) {
377            return false;
378          }
379        }
380        return true;
381      }
382
383      @Override
384      public String explainFailure() throws Exception {
385        RegionStates regionStates =
386            UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
387        for (HRegionInfo hri : ADMIN.getTableRegions(tableTwo)) {
388          if (encodedRegionNamesToMerge.contains(hri.getEncodedName())) {
389            return hri + " which is expected to be merged is still online";
390          }
391          if (!regionStates.isRegionInState(hri, RegionState.State.OPEN)) {
392            return hri + " is still in not opened";
393          }
394        }
395        return "Unknown";
396      }
397    });
398    hris = ADMIN.getTableRegions(tableTwo);
399    assertEquals(initialRegions - 1, hris.size());
400    Collections.sort(hris);
401
402    final HRegionInfo hriToSplit = hris.get(1);
403    ADMIN.split(tableTwo, Bytes.toBytes("500"));
404
405    UTIL.waitFor(10000, 100, new Waiter.ExplainingPredicate<Exception>() {
406
407      @Override
408      public boolean evaluate() throws Exception {
409        RegionStates regionStates =
410            UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
411        for (HRegionInfo hri : ADMIN.getTableRegions(tableTwo)) {
412          if (hri.getEncodedName().equals(hriToSplit.getEncodedName())) {
413            return false;
414          }
415          if (!regionStates.isRegionInState(hri, RegionState.State.OPEN)) {
416            return false;
417          }
418        }
419        return true;
420      }
421
422      @Override
423      public String explainFailure() throws Exception {
424        RegionStates regionStates =
425            UTIL.getMiniHBaseCluster().getMaster().getAssignmentManager().getRegionStates();
426        for (HRegionInfo hri : ADMIN.getTableRegions(tableTwo)) {
427          if (hri.getEncodedName().equals(hriToSplit.getEncodedName())) {
428            return hriToSplit + " which is expected to be split is still online";
429          }
430          if (!regionStates.isRegionInState(hri, RegionState.State.OPEN)) {
431            return hri + " is still in not opened";
432          }
433        }
434        return "Unknown";
435      }
436    });
437    hris = ADMIN.getTableRegions(tableTwo);
438    assertEquals(initialRegions, hris.size());
439    Collections.sort(hris);
440
441    // fail region merge through Coprocessor hook
442    MiniHBaseCluster cluster = UTIL.getHBaseCluster();
443    MasterCoprocessorHost cpHost = cluster.getMaster().getMasterCoprocessorHost();
444    Coprocessor coprocessor = cpHost.findCoprocessor(CPMasterObserver.class.getName());
445    CPMasterObserver masterObserver = (CPMasterObserver) coprocessor;
446    masterObserver.failMerge(true);
447    masterObserver.triggered = false;
448
449    ADMIN.mergeRegionsAsync(
450      hris.get(1).getEncodedNameAsBytes(),
451      hris.get(2).getEncodedNameAsBytes(),
452      false);
453    masterObserver.waitUtilTriggered();
454    hris = ADMIN.getTableRegions(tableTwo);
455    assertEquals(initialRegions, hris.size());
456    Collections.sort(hris);
457    // verify that we cannot split
458    HRegionInfo hriToSplit2 = hris.get(1);
459    ADMIN.split(tableTwo,
460      TableInputFormatBase.getSplitKey(hriToSplit2.getStartKey(), hriToSplit2.getEndKey(), true));
461    Thread.sleep(2000);
462    assertEquals(initialRegions, ADMIN.getTableRegions(tableTwo).size());
463  }

500    CustomObserver observer = (CustomObserver) actualRegion.getCoprocessorHost().findCoprocessor(
501        CustomObserver.class.getName());
502    assertNotNull(observer);
503
504    ADMIN.split(tableOne, Bytes.toBytes("500"));
505    observer.postSplit.await();
506    assertEquals(2, ADMIN.getTableRegions(tableOne).size());
507    actualRegion = UTIL.getHBaseCluster().getRegions(tableOne).get(0);
508    observer = (CustomObserver) actualRegion.getCoprocessorHost().findCoprocessor(
509      CustomObserver.class.getName());
510    assertNotNull(observer);
511
512    //Before we go on split, we should remove all reference store files.
513    ADMIN.compact(tableOne);
514    observer.postCompact.await();
515
516    ADMIN.split(tableOne, getSplitKey(actualRegion.getRegionInfo().getStartKey(),
517      actualRegion.getRegionInfo().getEndKey()));
518    observer.postSplit.await();
519    // Make sure no regions have been added.
520    List<HRegionInfo> hris = ADMIN.getTableRegions(tableOne);
521    assertEquals(2, hris.size());
522
523    htable.close();
524  }

527   * Create a table and make sure that the table creation fails after adding this table entry into
528   * namespace quota cache. Now correct the failure and recreate the table with same name.
529   * HBASE-13394
530   */
531  @Test
532  public void testRecreateTableWithSameNameAfterFirstTimeFailure() throws Exception {
533    String nsp1 = prefix + "_testRecreateTable";
534    NamespaceDescriptor nspDesc =
535        NamespaceDescriptor.create(nsp1)
536            .addConfiguration(TableNamespaceManager.KEY_MAX_REGIONS, "20")
537            .addConfiguration(TableNamespaceManager.KEY_MAX_TABLES, "1").build();
538    ADMIN.createNamespace(nspDesc);
539    final TableName tableOne = TableName.valueOf(nsp1 + TableName.NAMESPACE_DELIM + "table1");
540    byte[] columnFamily = Bytes.toBytes("info");
541    HTableDescriptor tableDescOne = new HTableDescriptor(tableOne);
542    tableDescOne.addFamily(new HColumnDescriptor(columnFamily));
543    MasterSyncObserver.throwExceptionInPreCreateTableAction = true;
544    try {
545      try {
546        ADMIN.createTable(tableDescOne);
547        fail("Table " + tableOne.toString() + "creation should fail.");
548      } catch (Exception exp) {
549        LOG.error(exp);
550      }
551      assertFalse(ADMIN.tableExists(tableOne));
552
553      NamespaceTableAndRegionInfo nstate = getNamespaceState(nsp1);
554      assertEquals("First table creation failed in namespace so number of tables in namespace "
555          + "should be 0.", 0, nstate.getTables().size());
556
557      MasterSyncObserver.throwExceptionInPreCreateTableAction = false;
558      try {
559        ADMIN.createTable(tableDescOne);
560      } catch (Exception e) {
561        fail("Table " + tableOne.toString() + "creation should succeed.");
562        LOG.error(e);
563      }
564      assertTrue(ADMIN.tableExists(tableOne));
565      nstate = getNamespaceState(nsp1);
566      assertEquals("First table was created successfully so table size in namespace should "
567          + "be one now.", 1, nstate.getTables().size());
568    } finally {
569      MasterSyncObserver.throwExceptionInPreCreateTableAction = false;
570      if (ADMIN.tableExists(tableOne)) {
571        ADMIN.disableTable(tableOne);
572        deleteTable(tableOne);
573      }
574      ADMIN.deleteNamespace(nsp1);
575    }
576  }

583  byte[] getSplitKey(byte[] startKey, byte[] endKey) {
584    String skey = Bytes.toString(startKey);
585    int key;
586    if (StringUtils.isBlank(skey)) {
587      key = Integer.parseInt(Bytes.toString(endKey))/2 ;
588    } else {
589      key = (int) (Integer.parseInt(skey) * 1.5);
590    }

653  public static void waitForQuotaEnabled(final HBaseTestingUtility util) throws Exception {
654    util.waitFor(60000, new Waiter.Predicate<Exception>() {
655      @Override
656      public boolean evaluate() throws Exception {
657        HMaster master = util.getHBaseCluster().getMaster();
658        if (master == null) {
659          return false;
660        }
661        MasterQuotaManager quotaManager = master.getMasterQuotaManager();
662        return quotaManager != null && quotaManager.isQuotaEnabled();
663      }
664    });
665  }

679  public static class MasterSyncObserver extends BaseMasterObserver {
680    volatile CountDownLatch tableDeletionLatch;
681    static boolean throwExceptionInPreCreateTableAction;
682
683    @Override
684    public void preDeleteTable(ObserverContext<MasterCoprocessorEnvironment> ctx,
685        TableName tableName) throws IOException {
686      tableDeletionLatch = new CountDownLatch(1);
687    }
688
689    @Override
690    public void postCompletedDeleteTableAction(
691        final ObserverContext<MasterCoprocessorEnvironment> ctx,
692        final TableName tableName) throws IOException {
693      tableDeletionLatch.countDown();
694    }
695
696    @Override
697    public void preCreateTableAction(ObserverContext<MasterCoprocessorEnvironment> ctx,
698        HTableDescriptor desc, HRegionInfo[] regions) throws IOException {
699      if (throwExceptionInPreCreateTableAction) {
700        throw new IOException("Throw exception as it is demanded.");
701      }
702    }
703  }

705  private void deleteTable(final TableName tableName) throws Exception {
706    // NOTE: We need a latch because admin is not sync,
707    // so the postOp coprocessor method may be called after the admin operation returned.
708    MasterSyncObserver observer = (MasterSyncObserver)UTIL.getHBaseCluster().getMaster()
709      .getMasterCoprocessorHost().findCoprocessor(MasterSyncObserver.class.getName());
710    ADMIN.deleteTable(tableName);
711    observer.tableDeletionLatch.await();
712  }
