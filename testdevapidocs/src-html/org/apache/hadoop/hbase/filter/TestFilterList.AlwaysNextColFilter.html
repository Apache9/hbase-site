099  private void mpOneTest(Filter filterMPONE) throws Exception {
100    /* Filter must do all below steps:
101     * <ul>
102     * <li>{@link #reset()}</li>
103     * <li>{@link #filterAllRemaining()} -> true indicates scan is over, false, keep going on.</li>
104     * <li>{@link #filterRowKey(byte[],int,int)} -> true to drop this row,
105     * if false, we will also call</li>
106     * <li>{@link #filterKeyValue(org.apache.hadoop.hbase.KeyValue)} -> true to drop this key/value</li>
107     * <li>{@link #filterRow()} -> last chance to drop entire row based on the sequence of
108     * filterValue() calls. Eg: filter a row if it doesn't contain a specified column.
109     * </li>
110     * </ul>
111    */
112    filterMPONE.reset();
113    assertFalse(filterMPONE.filterAllRemaining());
114
115    /* Will pass both */
116    byte [] rowkey = Bytes.toBytes("yyyyyyyyy");
117    for (int i = 0; i < MAX_PAGES - 1; i++) {
118      assertFalse(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
119      KeyValue kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(i),
120        Bytes.toBytes(i));
121      assertTrue(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
122      assertFalse(filterMPONE.filterRow());
123    }
124
125    /* Only pass PageFilter */
126    rowkey = Bytes.toBytes("z");
127    assertFalse(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
128    KeyValue kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(0),
129        Bytes.toBytes(0));
130    assertTrue(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
131    assertFalse(filterMPONE.filterRow());
132
133    /* reach MAX_PAGES already, should filter any rows */
134    rowkey = Bytes.toBytes("yyy");
135    assertTrue(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
136    kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(0),
137        Bytes.toBytes(0));
138    assertFalse(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
139    assertFalse(filterMPONE.filterRow());
140
141    /* We should filter any row */
142    rowkey = Bytes.toBytes("z");
143    assertTrue(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
144    assertTrue(filterMPONE.filterAllRemaining());
145  }

213  public void orderingTest(Filter filterMPONE) throws Exception {
214    /* Filter must do all below steps:
215     * <ul>
216     * <li>{@link #reset()}</li>
217     * <li>{@link #filterAllRemaining()} -> true indicates scan is over, false, keep going on.</li>
218     * <li>{@link #filterRowKey(byte[],int,int)} -> true to drop this row,
219     * if false, we will also call</li>
220     * <li>{@link #filterKeyValue(org.apache.hadoop.hbase.KeyValue)} -> true to drop this key/value</li>
221     * <li>{@link #filterRow()} -> last chance to drop entire row based on the sequence of
222     * filterValue() calls. Eg: filter a row if it doesn't contain a specified column.
223     * </li>
224     * </ul>
225    */
226    filterMPONE.reset();
227    assertFalse(filterMPONE.filterAllRemaining());
228
229    /* We should be able to fill MAX_PAGES without incrementing page counter */
230    byte [] rowkey = Bytes.toBytes("yyyyyyyy");
231    for (int i = 0; i < MAX_PAGES; i++) {
232      assertFalse(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
233      KeyValue kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(i),
234          Bytes.toBytes(i));
235        assertTrue(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
236      assertFalse(filterMPONE.filterRow());
237    }
238
239    /* Now let's fill the page filter */
240    rowkey = Bytes.toBytes("xxxxxxx");
241    for (int i = 0; i < MAX_PAGES; i++) {
242      assertFalse(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
243      KeyValue kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(i),
244          Bytes.toBytes(i));
245        assertTrue(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
246      assertFalse(filterMPONE.filterRow());
247    }
248
249    /* We should still be able to include even though page filter is at max */
250    rowkey = Bytes.toBytes("yyy");
251    for (int i = 0; i < MAX_PAGES; i++) {
252      assertFalse(filterMPONE.filterRowKey(KeyValueUtil.createFirstOnRow(rowkey)));
253      KeyValue kv = new KeyValue(rowkey, rowkey, Bytes.toBytes(i),
254          Bytes.toBytes(i));
255        assertTrue(Filter.ReturnCode.INCLUDE == filterMPONE.filterKeyValue(kv));
256      assertFalse(filterMPONE.filterRow());
257    }

262   * we expect to get the same result as the 'prefix' only result.

265  public void testFilterListTwoFiltersMustPassOne() throws Exception {
266    byte[] r1 = Bytes.toBytes("Row1");
267    byte[] r11 = Bytes.toBytes("Row11");
268    byte[] r2 = Bytes.toBytes("Row2");
269
270    FilterList flist = new FilterList(FilterList.Operator.MUST_PASS_ONE);
271    flist.addFilter(new PrefixFilter(r1));
272    flist.filterRowKey(KeyValueUtil.createFirstOnRow(r1));
273    assertEquals(flist.filterKeyValue(new KeyValue(r1,r1,r1)), ReturnCode.INCLUDE);
274    assertEquals(flist.filterKeyValue(new KeyValue(r11,r11,r11)), ReturnCode.INCLUDE);
275
276    flist.reset();
277    flist.filterRowKey(KeyValueUtil.createFirstOnRow(r2));
278    assertEquals(flist.filterKeyValue(new KeyValue(r2,r2,r2)), ReturnCode.SKIP);
279
280    flist = new FilterList(FilterList.Operator.MUST_PASS_ONE);
281    flist.addFilter(new AlwaysNextColFilter());
282    flist.addFilter(new PrefixFilter(r1));
283    flist.filterRowKey(KeyValueUtil.createFirstOnRow(r1));
284    assertEquals(flist.filterKeyValue(new KeyValue(r1,r1,r1)), ReturnCode.INCLUDE);
285    assertEquals(flist.filterKeyValue(new KeyValue(r11,r11,r11)), ReturnCode.INCLUDE);
286
287    flist.reset();
288    flist.filterRowKey(KeyValueUtil.createFirstOnRow(r2));
289    assertEquals(flist.filterKeyValue(new KeyValue(r2,r2,r2)), ReturnCode.SKIP);
290  }

353   * Test filterKeyValue logic.
354   * @throws Exception
355   */
356  public void testFilterKeyValue() throws Exception {
357    Filter includeFilter = new FilterBase() {
358      @Override
359      public Filter.ReturnCode filterKeyValue(Cell v) {
360        return Filter.ReturnCode.INCLUDE;
361      }
362    };
363
364    Filter alternateFilter = new FilterBase() {
365      boolean returnInclude = true;
366
367      @Override
368      public Filter.ReturnCode filterKeyValue(Cell v) {
369        Filter.ReturnCode returnCode = returnInclude ? Filter.ReturnCode.INCLUDE :
370                                                       Filter.ReturnCode.SKIP;
371        returnInclude = !returnInclude;
372        return returnCode;
373      }
374    };
375
376    Filter alternateIncludeFilter = new FilterBase() {
377      boolean returnIncludeOnly = false;
378
379      @Override
380      public Filter.ReturnCode filterKeyValue(Cell v) {
381        Filter.ReturnCode returnCode = returnIncludeOnly ? Filter.ReturnCode.INCLUDE :
382                                                           Filter.ReturnCode.INCLUDE_AND_NEXT_COL;
383        returnIncludeOnly = !returnIncludeOnly;
384        return returnCode;
385      }
386    };
387
388    // Check must pass one filter.
389    FilterList mpOnefilterList = new FilterList(Operator.MUST_PASS_ONE,
390        Arrays.asList(new Filter[] { includeFilter, alternateIncludeFilter, alternateFilter }));
391    // INCLUDE, INCLUDE, INCLUDE_AND_NEXT_COL.
392    assertEquals(Filter.ReturnCode.INCLUDE_AND_NEXT_COL, mpOnefilterList.filterKeyValue(null));
393    // INCLUDE, SKIP, INCLUDE.
394    assertEquals(Filter.ReturnCode.INCLUDE, mpOnefilterList.filterKeyValue(null));
395
396    // Check must pass all filter.
397    FilterList mpAllfilterList = new FilterList(Operator.MUST_PASS_ALL,
398        Arrays.asList(new Filter[] { includeFilter, alternateIncludeFilter, alternateFilter }));
399    // INCLUDE, INCLUDE, INCLUDE_AND_NEXT_COL.
400    assertEquals(Filter.ReturnCode.INCLUDE_AND_NEXT_COL, mpAllfilterList.filterKeyValue(null));
401    // INCLUDE, SKIP, INCLUDE.
402    assertEquals(Filter.ReturnCode.SKIP, mpAllfilterList.filterKeyValue(null));
403  }

408  @Test
409  public void testHintPassThru() throws Exception {
410
411    final KeyValue minKeyValue = new KeyValue(Bytes.toBytes(0L), null, null);
412    final KeyValue maxKeyValue = new KeyValue(Bytes.toBytes(Long.MAX_VALUE),
413        null, null);
414
415    Filter filterNoHint = new FilterBase() {
416      @Override
417      public byte [] toByteArray() {
418        return null;
419      }
420
421      @Override
422      public ReturnCode filterKeyValue(Cell ignored) throws IOException {
423        return ReturnCode.INCLUDE;
424      }
425    };
426
427    Filter filterMinHint = new FilterBase() {
428      @Override
429      public ReturnCode filterKeyValue(Cell ignored) {
430        return ReturnCode.SEEK_NEXT_USING_HINT;
431      }
432
433      @Override
434      public Cell getNextCellHint(Cell currentKV) {
435        return minKeyValue;
436      }
437
438      @Override
439      public byte [] toByteArray() {return null;}
440    };
441
442    Filter filterMaxHint = new FilterBase() {
443      @Override
444      public ReturnCode filterKeyValue(Cell ignored) {
445        return ReturnCode.SEEK_NEXT_USING_HINT;
446      }
447
448      @Override
449      public Cell getNextCellHint(Cell cell) {
450        return new KeyValue(Bytes.toBytes(Long.MAX_VALUE), null, null);
451      }
452
453      @Override
454      public byte [] toByteArray() {return null;}
455    };
456
457    // MUST PASS ONE
458
459    // Should take the min if given two hints
460    FilterList filterList = new FilterList(Operator.MUST_PASS_ONE,
461        Arrays.asList(new Filter [] { filterMinHint, filterMaxHint } ));
462    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
463        minKeyValue));
464
465    // Should have no hint if any filter has no hint
466    filterList = new FilterList(Operator.MUST_PASS_ONE,
467        Arrays.asList(
468            new Filter [] { filterMinHint, filterMaxHint, filterNoHint } ));
469    assertNull(filterList.getNextCellHint(null));
470    filterList = new FilterList(Operator.MUST_PASS_ONE,
471        Arrays.asList(new Filter [] { filterNoHint, filterMaxHint } ));
472    assertNull(filterList.getNextCellHint(null));
473
474    // Should give max hint if its the only one
475    filterList = new FilterList(Operator.MUST_PASS_ONE,
476        Arrays.asList(new Filter [] { filterMaxHint, filterMaxHint } ));
477    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
478        maxKeyValue));
479
480    // MUST PASS ALL
481
482    // Should take the first hint
483    filterList = new FilterList(Operator.MUST_PASS_ALL,
484        Arrays.asList(new Filter [] { filterMinHint, filterMaxHint } ));
485    filterList.filterKeyValue(null);
486    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
487        minKeyValue));
488
489    filterList = new FilterList(Operator.MUST_PASS_ALL,
490        Arrays.asList(new Filter [] { filterMaxHint, filterMinHint } ));
491    filterList.filterKeyValue(null);
492    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
493        maxKeyValue));
494
495    // Should have first hint even if a filter has no hint
496    filterList = new FilterList(Operator.MUST_PASS_ALL,
497        Arrays.asList(
498            new Filter [] { filterNoHint, filterMinHint, filterMaxHint } ));
499    filterList.filterKeyValue(null);
500    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
501        minKeyValue));
502    filterList = new FilterList(Operator.MUST_PASS_ALL,
503        Arrays.asList(new Filter [] { filterNoHint, filterMaxHint } ));
504    filterList.filterKeyValue(null);
505    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
506        maxKeyValue));
507    filterList = new FilterList(Operator.MUST_PASS_ALL,
508        Arrays.asList(new Filter [] { filterNoHint, filterMinHint } ));
509    filterList.filterKeyValue(null);
510    assertEquals(0, CellComparator.COMPARATOR.compare(filterList.getNextCellHint(null),
511        minKeyValue));
512  }

514  /**
515   * Tests the behavior of transform() in a hierarchical filter.
516   *
517   * transform() only applies after a filterKeyValue() whose return-code includes the KeyValue.
518   * Lazy evaluation of AND
519   */
520  @Test
521  public void testTransformMPO() throws Exception {
522    // Apply the following filter:
523    //     (family=fam AND qualifier=qual1 AND KeyOnlyFilter)
524    //  OR (family=fam AND qualifier=qual2)
525    final FilterList flist = new FilterList(Operator.MUST_PASS_ONE, Lists.<Filter>newArrayList(
526        new FilterList(Operator.MUST_PASS_ALL, Lists.<Filter>newArrayList(
527            new FamilyFilter(CompareOp.EQUAL, new BinaryComparator(Bytes.toBytes("fam"))),
528            new QualifierFilter(CompareOp.EQUAL, new BinaryComparator(Bytes.toBytes("qual1"))),
529            new KeyOnlyFilter())),
530        new FilterList(Operator.MUST_PASS_ALL, Lists.<Filter>newArrayList(
531            new FamilyFilter(CompareOp.EQUAL, new BinaryComparator(Bytes.toBytes("fam"))),
532            new QualifierFilter(CompareOp.EQUAL, new BinaryComparator(Bytes.toBytes("qual2")))))));
533
534    final KeyValue kvQual1 = new KeyValue(
535        Bytes.toBytes("row"), Bytes.toBytes("fam"), Bytes.toBytes("qual1"), Bytes.toBytes("value"));
536    final KeyValue kvQual2 = new KeyValue(
537        Bytes.toBytes("row"), Bytes.toBytes("fam"), Bytes.toBytes("qual2"), Bytes.toBytes("value"));
538    final KeyValue kvQual3 = new KeyValue(
539        Bytes.toBytes("row"), Bytes.toBytes("fam"), Bytes.toBytes("qual3"), Bytes.toBytes("value"));
540
541    // Value for fam:qual1 should be stripped:
542    assertEquals(Filter.ReturnCode.INCLUDE, flist.filterKeyValue(kvQual1));
543    final KeyValue transformedQual1 = KeyValueUtil.ensureKeyValue(flist.transformCell(kvQual1));
544    assertEquals(0, transformedQual1.getValueLength());
545
546    // Value for fam:qual2 should not be stripped:
547    assertEquals(Filter.ReturnCode.INCLUDE, flist.filterKeyValue(kvQual2));
548    final KeyValue transformedQual2 = KeyValueUtil.ensureKeyValue(flist.transformCell(kvQual2));
549    assertEquals("value", Bytes.toString(transformedQual2.getValueArray(),
550      transformedQual2.getValueOffset(), transformedQual2.getValueLength()));
551
552    // Other keys should be skipped:
553    assertEquals(Filter.ReturnCode.SKIP, flist.filterKeyValue(kvQual3));
554  }
