050@Category({ReplicationTests.class, MediumTests.class})
051public class TestReplicationStateZKImpl extends TestReplicationStateBasic {
052
053  private static final Log LOG = LogFactory.getLog(TestReplicationStateZKImpl.class);
054
055  private static Configuration conf;
056  private static HBaseTestingUtility utility;
057  private static ZooKeeperWatcher zkw;
058  private static String replicationZNode;
059  private ReplicationQueuesZKImpl rqZK;
060
061  @BeforeClass
062  public static void setUpBeforeClass() throws Exception {
063    utility = new HBaseTestingUtility();
064    utility.startMiniZKCluster();
065    conf = utility.getConfiguration();
066    zkw = HBaseTestingUtility.getZooKeeperWatcher(utility);
067    String replicationZNodeName = conf.get("zookeeper.znode.replication", "replication");
068    replicationZNode = ZKUtil.joinZNode(zkw.baseZNode, replicationZNodeName);
069    KEY_ONE = initPeerClusterState("/hbase1");
070    KEY_TWO = initPeerClusterState("/hbase2");
071  }
072
073  private static String initPeerClusterState(String baseZKNode)
074      throws IOException, KeeperException {
075    // Add a dummy region server and set up the cluster id
076    Configuration testConf = new Configuration(conf);
077    testConf.set(HConstants.ZOOKEEPER_ZNODE_PARENT, baseZKNode);
078    ZooKeeperWatcher zkw1 = new ZooKeeperWatcher(testConf, "test1", null);
079    String fakeRs = ZKUtil.joinZNode(zkw1.rsZNode, "hostname1.example.org:1234");
080    ZKUtil.createWithParents(zkw1, fakeRs);
081    ZKClusterId.setClusterId(zkw1, new ClusterId());
082    return ZKUtil.getZooKeeperClusterKey(testConf);
083  }
084
085  @Before
086  @Override
087  public void setUp() {
088    super.setUp();
089    DummyServer ds1 = new DummyServer(server1);
090    DummyServer ds2 = new DummyServer(server2);
091    DummyServer ds3 = new DummyServer(server3);
092    rq1 = ReplicationFactory.getReplicationQueues(zkw, conf, ds1);
093    rq2 = ReplicationFactory.getReplicationQueues(zkw, conf, ds2);
094    rq3 = ReplicationFactory.getReplicationQueues(zkw, conf, ds3);
095    rqc = ReplicationFactory.getReplicationQueuesClient(zkw, conf, ds1);
096    rp = ReplicationFactory.getReplicationPeers(zkw, conf, zkw);
097    OUR_KEY = ZKUtil.getZooKeeperClusterKey(conf);
098    rqZK = new ReplicationQueuesZKImpl(zkw, conf, ds1);
099  }
100
101  @After
102  public void tearDown() throws KeeperException, IOException {
103    ZKUtil.deleteNodeRecursively(zkw, replicationZNode);
104  }
105
106  @AfterClass
107  public static void tearDownAfterClass() throws Exception {
108    utility.shutdownMiniZKCluster();
109  }
110
111  @Test
112  public void testIsPeerPath_PathToParentOfPeerNode() {
113    assertFalse(rqZK.isPeerPath(rqZK.peersZNode));
114  }
115
116  @Test
117  public void testIsPeerPath_PathToChildOfPeerNode() {
118    String peerChild = ZKUtil.joinZNode(ZKUtil.joinZNode(rqZK.peersZNode, "1"), "child");
119    assertFalse(rqZK.isPeerPath(peerChild));
120  }
121
122  @Test
123  public void testIsPeerPath_ActualPeerPath() {
124    String peerPath = ZKUtil.joinZNode(rqZK.peersZNode, "1");
125    assertTrue(rqZK.isPeerPath(peerPath));
126  }
