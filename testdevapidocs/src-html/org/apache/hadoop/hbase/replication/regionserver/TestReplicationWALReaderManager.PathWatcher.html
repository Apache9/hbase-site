091  @Parameters
092  public static Collection<Object[]> parameters() {
093    // Try out different combinations of row count and KeyValue count
094    int[] NB_ROWS = { 1500, 60000 };
095    int[] NB_KVS = { 1, 100 };
096    // whether compression is used
097    Boolean[] BOOL_VALS = { false, true };
098    List<Object[]> parameters = new ArrayList<Object[]>();
099    for (int nbRows : NB_ROWS) {
100      for (int walEditKVs : NB_KVS) {
101        for (boolean b : BOOL_VALS) {
102          Object[] arr = new Object[3];
103          arr[0] = nbRows;
104          arr[1] = walEditKVs;
105          arr[2] = b;
106          parameters.add(arr);
107        }
108      }
109    }
110    return parameters;
111  }

156  @Test
157  public void test() throws Exception {
158    // Grab the path that was generated when the log rolled as part of its creation
159    Path path = pathWatcher.currentPath;
160
161    assertEquals(0, logManager.getPosition());
162
163    appendToLog();
164
165    // There's one edit in the log, read it. Reading past it needs to return nulls
166    assertNotNull(logManager.openReader(path));
167    logManager.seek();
168    WAL.Entry entry = logManager.readNextAndSetPosition();
169    assertNotNull(entry);
170    entry = logManager.readNextAndSetPosition();
171    assertNull(entry);
172    logManager.closeReader();
173    long oldPos = logManager.getPosition();
174
175    appendToLog();
176
177    // Read the newly added entry, make sure we made progress
178    assertNotNull(logManager.openReader(path));
179    logManager.seek();
180    entry = logManager.readNextAndSetPosition();
181    assertNotEquals(oldPos, logManager.getPosition());
182    assertNotNull(entry);
183    logManager.closeReader();
184    oldPos = logManager.getPosition();
185
186    log.rollWriter();
187
188    // We rolled but we still should see the end of the first log and not get data
189    assertNotNull(logManager.openReader(path));
190    logManager.seek();
191    entry = logManager.readNextAndSetPosition();
192    assertEquals(oldPos, logManager.getPosition());
193    assertNull(entry);
194    logManager.finishCurrentFile();
195
196    path = pathWatcher.currentPath;
197
198    for (int i = 0; i < nbRows; i++) { appendToLogPlus(walEditKVs); }
199    log.rollWriter();
200    logManager.openReader(path);
201    logManager.seek();
202    for (int i = 0; i < nbRows; i++) {
203      WAL.Entry e = logManager.readNextAndSetPosition();
204      if (e == null) {
205        fail("Should have enough entries");
206      }
207    }
208  }
209
210  private void appendToLog() throws IOException {
211    appendToLogPlus(1);
212  }
213
214  private void appendToLogPlus(int count) throws IOException {
215    final long txid = log.append(info, new WALKey(info.getEncodedNameAsBytes(), tableName,
216        System.currentTimeMillis(), mvcc, scopes), getWALEdits(count), true);
217    log.sync(txid);
218  }
219
220  private WALEdit getWALEdits(int count) {
221    WALEdit edit = new WALEdit();
222    for (int i = 0; i < count; i++) {
223      edit.add(new KeyValue(Bytes.toBytes(System.currentTimeMillis()), family, qualifier,
224        System.currentTimeMillis(), qualifier));
225    }
226    return edit;
227  }

235      currentPath = newPath;
236    }
