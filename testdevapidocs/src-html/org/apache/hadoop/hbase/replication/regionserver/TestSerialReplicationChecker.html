060@Category({ ReplicationTests.class, MediumTests.class })
061public class TestSerialReplicationChecker {
062
063  @ClassRule
064  public static final HBaseClassTestRule CLASS_RULE =
065    HBaseClassTestRule.forClass(TestSerialReplicationChecker.class);
066
067  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
068
069  private static String PEER_ID = "1";
070
071  private static ReplicationQueueStorage QUEUE_STORAGE;
072
073  private static String WAL_FILE_NAME = "test.wal";
074
075  private SerialReplicationChecker checker;
076
077  @Rule
078  public final TestName name = new TestName();
079
080  private TableName tableName;
081
082  @BeforeClass
083  public static void setUpBeforeClass() throws Exception {
084    UTIL.startMiniCluster(1);
085    QUEUE_STORAGE = ReplicationStorageFactory.getReplicationQueueStorage(UTIL.getZooKeeperWatcher(),
086      UTIL.getConfiguration());
087    QUEUE_STORAGE.addWAL(UTIL.getMiniHBaseCluster().getRegionServer(0).getServerName(), PEER_ID,
088      WAL_FILE_NAME);
089  }
090
091  @AfterClass
092  public static void tearDownAfterClass() throws Exception {
093    UTIL.shutdownMiniCluster();
094  }
095
096  @Before
097  public void setUp() throws IOException {
098    ReplicationSource source = mock(ReplicationSource.class);
099    when(source.getPeerId()).thenReturn(PEER_ID);
100    when(source.getQueueStorage()).thenReturn(QUEUE_STORAGE);
101    Server server = mock(Server.class);
102    when(server.getConnection()).thenReturn(UTIL.getConnection());
103    when(source.getServer()).thenReturn(server);
104    checker = new SerialReplicationChecker(UTIL.getConfiguration(), source);
105    tableName = TableName.valueOf(name.getMethodName());
106  }
107
108  private Entry createEntry(RegionInfo region, long seqId) {
109    WALKeyImpl key = mock(WALKeyImpl.class);
110    when(key.getTableName()).thenReturn(tableName);
111    when(key.getEncodedRegionName()).thenReturn(region.getEncodedNameAsBytes());
112    when(key.getSequenceId()).thenReturn(seqId);
113    Entry entry = mock(Entry.class);
114    when(entry.getKey()).thenReturn(key);
115    return entry;
116  }
117
118  private Cell createCell(RegionInfo region) {
119    return CellBuilderFactory.create(CellBuilderType.DEEP_COPY).setRow(region.getStartKey())
120      .setType(Type.Put).build();
121  }
122
123  @Test
124  public void testNoBarrierCanPush() throws IOException {
125    RegionInfo region = RegionInfoBuilder.newBuilder(tableName).build();
126    assertTrue(checker.canPush(createEntry(region, 100), createCell(region)));
127  }
128
129  private void addStateAndBarrier(RegionInfo region, RegionState.State state, long... barriers)
130      throws IOException {
131    Put put = new Put(region.getRegionName(), EnvironmentEdgeManager.currentTime());
132    put.addColumn(HConstants.CATALOG_FAMILY, HConstants.STATE_QUALIFIER,
133      Bytes.toBytes(state.name()));
134    for (int i = 0; i < barriers.length; i++) {
135      put.addColumn(HConstants.REPLICATION_BARRIER_FAMILY, HConstants.SEQNUM_QUALIFIER,
136        put.getTimeStamp() - i, Bytes.toBytes(barriers[i]));
137    }
138    try (Table table = UTIL.getConnection().getTable(TableName.META_TABLE_NAME)) {
139      table.put(put);
140    }
141  }
142
143  private void setState(RegionInfo region, RegionState.State state) throws IOException {
144    Put put = new Put(region.getRegionName(), EnvironmentEdgeManager.currentTime());
145    put.addColumn(HConstants.CATALOG_FAMILY, HConstants.STATE_QUALIFIER,
146      Bytes.toBytes(state.name()));
147    try (Table table = UTIL.getConnection().getTable(TableName.META_TABLE_NAME)) {
148      table.put(put);
149    }
150  }
151
152  private void updatePushedSeqId(RegionInfo region, long seqId) throws ReplicationException {
153    QUEUE_STORAGE.setWALPosition(UTIL.getMiniHBaseCluster().getRegionServer(0).getServerName(),
154      PEER_ID, WAL_FILE_NAME, 10, ImmutableMap.of(region.getEncodedName(), seqId));
155  }
156
157  @Test
158  public void testLastRegionAndOpeningCanNotPush() throws IOException, ReplicationException {
159    RegionInfo region = RegionInfoBuilder.newBuilder(tableName).build();
160    addStateAndBarrier(region, RegionState.State.OPEN, 10);
161    Cell cell = createCell(region);
162    // can push since we are in the first range
163    assertTrue(checker.canPush(createEntry(region, 100), cell));
164    setState(region, RegionState.State.OPENING);
165    // can not push since we are in the last range and the state is OPENING
166    assertFalse(checker.canPush(createEntry(region, 102), cell));
167    addStateAndBarrier(region, RegionState.State.OPEN, 50);
168    // can not push since the previous range has not been finished yet
169    assertFalse(checker.canPush(createEntry(region, 102), cell));
170    updatePushedSeqId(region, 49);
171    // can push since the previous range has been finished
172    assertTrue(checker.canPush(createEntry(region, 102), cell));
173    setState(region, RegionState.State.OPENING);
174    assertFalse(checker.canPush(createEntry(region, 104), cell));
175  }
