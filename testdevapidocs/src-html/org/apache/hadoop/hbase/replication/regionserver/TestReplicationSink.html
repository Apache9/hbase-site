126  /**
127   * Insert a whole batch of entries
128   * @throws Exception
129   */
130  @Test
131  public void testBatchSink() throws Exception {
132    List<WALEntry> entries = new ArrayList<WALEntry>(BATCH_SIZE);
133    List<Cell> cells = new ArrayList<Cell>();
134    for(int i = 0; i < BATCH_SIZE; i++) {
135      entries.add(createEntry(TABLE_NAME1, i, KeyValue.Type.Put, cells));
136    }
137    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
138    Scan scan = new Scan();
139    ResultScanner scanRes = table1.getScanner(scan);
140    assertEquals(BATCH_SIZE, scanRes.next(BATCH_SIZE).length);
141  }

143  /**
144   * Insert a mix of puts and deletes
145   * @throws Exception
146   */
147  @Test
148  public void testMixedPutDelete() throws Exception {
149    List<WALEntry> entries = new ArrayList<WALEntry>(BATCH_SIZE/2);
150    List<Cell> cells = new ArrayList<Cell>();
151    for(int i = 0; i < BATCH_SIZE/2; i++) {
152      entries.add(createEntry(TABLE_NAME1, i, KeyValue.Type.Put, cells));
153    }
154    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells));
155
156    entries = new ArrayList<WALEntry>(BATCH_SIZE);
157    cells = new ArrayList<Cell>();
158    for(int i = 0; i < BATCH_SIZE; i++) {
159      entries.add(createEntry(TABLE_NAME1, i,
160          i % 2 != 0 ? KeyValue.Type.Put: KeyValue.Type.DeleteColumn, cells));
161    }
162
163    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
164    Scan scan = new Scan();
165    ResultScanner scanRes = table1.getScanner(scan);
166    assertEquals(BATCH_SIZE/2, scanRes.next(BATCH_SIZE).length);
167  }

169  /**
170   * Insert to 2 different tables
171   * @throws Exception
172   */
173  @Test
174  public void testMixedPutTables() throws Exception {
175    List<WALEntry> entries = new ArrayList<WALEntry>(BATCH_SIZE/2);
176    List<Cell> cells = new ArrayList<Cell>();
177    for(int i = 0; i < BATCH_SIZE; i++) {
178      entries.add(createEntry( i % 2 == 0 ? TABLE_NAME2 : TABLE_NAME1,
179              i, KeyValue.Type.Put, cells));
180    }
181
182    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
183    Scan scan = new Scan();
184    ResultScanner scanRes = table2.getScanner(scan);
185    for(Result res : scanRes) {
186      assertTrue(Bytes.toInt(res.getRow()) % 2 == 0);
187    }
188  }

190  /**
191   * Insert then do different types of deletes
192   * @throws Exception
193   */
194  @Test
195  public void testMixedDeletes() throws Exception {
196    List<WALEntry> entries = new ArrayList<WALEntry>(3);
197    List<Cell> cells = new ArrayList<Cell>();
198    for(int i = 0; i < 3; i++) {
199      entries.add(createEntry(TABLE_NAME1, i, KeyValue.Type.Put, cells));
200    }
201    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
202    entries = new ArrayList<WALEntry>(3);
203    cells = new ArrayList<Cell>();
204    entries.add(createEntry(TABLE_NAME1, 0, KeyValue.Type.DeleteColumn, cells));
205    entries.add(createEntry(TABLE_NAME1, 1, KeyValue.Type.DeleteFamily, cells));
206    entries.add(createEntry(TABLE_NAME1, 2, KeyValue.Type.DeleteColumn, cells));
207
208    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
209
210    Scan scan = new Scan();
211    ResultScanner scanRes = table1.getScanner(scan);
212    assertEquals(0, scanRes.next(3).length);
213  }

215  /**
216   * Puts are buffered, but this tests when a delete (not-buffered) is applied
217   * before the actual Put that creates it.
218   * @throws Exception
219   */
220  @Test
221  public void testApplyDeleteBeforePut() throws Exception {
222    List<WALEntry> entries = new ArrayList<WALEntry>(5);
223    List<Cell> cells = new ArrayList<Cell>();
224    for(int i = 0; i < 2; i++) {
225      entries.add(createEntry(TABLE_NAME1, i, KeyValue.Type.Put, cells));
226    }
227    entries.add(createEntry(TABLE_NAME1, 1, KeyValue.Type.DeleteFamily, cells));
228    for(int i = 3; i < 5; i++) {
229      entries.add(createEntry(TABLE_NAME1, i, KeyValue.Type.Put, cells));
230    }
231    SINK.replicateEntries(entries, CellUtil.createCellScanner(cells.iterator()));
232    Get get = new Get(Bytes.toBytes(1));
233    Result res = table1.get(get);
234    assertEquals(0, res.size());
235  }

237  private WALEntry createEntry(TableName table, int row,  KeyValue.Type type, List<Cell> cells) {
238    byte[] fam = table.equals(TABLE_NAME1) ? FAM_NAME1 : FAM_NAME2;
239    byte[] rowBytes = Bytes.toBytes(row);
240    // Just make sure we don't get the same ts for two consecutive rows with
241    // same key
242    try {
243      Thread.sleep(1);
244    } catch (InterruptedException e) {
245      LOG.info("Was interrupted while sleep, meh", e);
246    }
247    final long now = System.currentTimeMillis();
248    KeyValue kv = null;
249    if(type.getCode() == KeyValue.Type.Put.getCode()) {
250      kv = new KeyValue(rowBytes, fam, fam, now,
251          KeyValue.Type.Put, Bytes.toBytes(row));
252    } else if (type.getCode() == KeyValue.Type.DeleteColumn.getCode()) {
253        kv = new KeyValue(rowBytes, fam, fam,
254            now, KeyValue.Type.DeleteColumn);
255    } else if (type.getCode() == KeyValue.Type.DeleteFamily.getCode()) {
256        kv = new KeyValue(rowBytes, fam, null,
257            now, KeyValue.Type.DeleteFamily);
258    }
259    WALEntry.Builder builder = WALEntry.newBuilder();
260    builder.setAssociatedCellCount(1);
261    WALKey.Builder keyBuilder = WALKey.newBuilder();
262    UUID.Builder uuidBuilder = UUID.newBuilder();
263    uuidBuilder.setLeastSigBits(HConstants.DEFAULT_CLUSTER_ID.getLeastSignificantBits());
264    uuidBuilder.setMostSigBits(HConstants.DEFAULT_CLUSTER_ID.getMostSignificantBits());
265    keyBuilder.setClusterId(uuidBuilder.build());
266    keyBuilder.setTableName(ByteStringer.wrap(table.getName()));
267    keyBuilder.setWriteTime(now);
268    keyBuilder.setEncodedRegionName(ByteStringer.wrap(HConstants.EMPTY_BYTE_ARRAY));
269    keyBuilder.setLogSequenceNumber(-1);
270    builder.setKey(keyBuilder.build());
271    cells.add(kv);
272
273    return builder.build();
274  }
