049/**
050 * This class is only a base for other integration-level replication tests.
051 * Do not add tests here.
052 * TestReplicationSmallTests is where tests that don't require bring machines up/down should go
053 * All other tests should have their own classes and extend this one
054 */
055public class TestReplicationBase {
056/*
057  {
058    ((Log4JLogger) ReplicationSource.LOG).getLogger().setLevel(Level.ALL);
059  }*/
060
061  private static final Log LOG = LogFactory.getLog(TestReplicationBase.class);
062
063  protected static Configuration conf1 = HBaseConfiguration.create();
064  protected static Configuration conf2;
065  protected static Configuration CONF_WITH_LOCALFS;
066
067  protected static ZooKeeperWatcher zkw1;
068  protected static ZooKeeperWatcher zkw2;
069
070  protected static ReplicationAdmin admin;
071  protected static Admin hbaseAdmin;
072
073  protected static Table htable1;
074  protected static Table htable2;
075  protected static NavigableMap<byte[], Integer> scopes;
076
077  protected static HBaseTestingUtility utility1;
078  protected static HBaseTestingUtility utility2;
079  protected static final int NB_ROWS_IN_BATCH = 100;
080  protected static final int NB_ROWS_IN_BIG_BATCH =
081      NB_ROWS_IN_BATCH * 10;
082  protected static final long SLEEP_TIME = 500;
083  protected static final int NB_RETRIES = 10;
084
085  protected static final TableName tableName = TableName.valueOf("test");
086  protected static final byte[] famName = Bytes.toBytes("f");
087  protected static final byte[] row = Bytes.toBytes("row");
088  protected static final byte[] noRepfamName = Bytes.toBytes("norep");
089
090  @Parameter
091  public static boolean seperateOldWALs;
092
093  @Parameters
094  public static List<Boolean> params() {
095    return Arrays.asList(false, true);
096  }
097
098  /**
099   * @throws java.lang.Exception
100   */
101  @BeforeClass
102  public static void setUpBeforeClass() throws Exception {
103    conf1.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/1");
104    // We don't want too many edits per batch sent to the ReplicationEndpoint to trigger
105    // sufficient number of events. But we don't want to go too low because
106    // HBaseInterClusterReplicationEndpoint partitions entries into batches and we want
107    // more than one batch sent to the peer cluster for better testing.
108    conf1.setInt("replication.source.size.capacity", 102400);
109    conf1.setLong("replication.source.sleepforretries", 100);
110    conf1.setInt("hbase.regionserver.maxlogs", 10);
111    conf1.setLong("hbase.master.logcleaner.ttl", 10);
112    conf1.setInt("zookeeper.recovery.retry", 1);
113    conf1.setInt("zookeeper.recovery.retry.intervalmill", 10);
114    conf1.setLong(HConstants.THREAD_WAKE_FREQUENCY, 100);
115    conf1.setInt("replication.stats.thread.period.seconds", 5);
116    conf1.setBoolean("hbase.tests.use.shortcircuit.reads", false);
117    conf1.setLong("replication.sleep.before.failover", 2000);
118    conf1.setInt("replication.source.maxretriesmultiplier", 10);
119    conf1.setFloat("replication.source.ratio", 1.0f);
120    conf1.setBoolean("replication.source.eof.autorecovery", true);
121
122    // Parameter config
123    conf1.setBoolean(AbstractFSWALProvider.SEPARATE_OLDLOGDIR, seperateOldWALs);
124
125    utility1 = new HBaseTestingUtility(conf1);
126    utility1.startMiniZKCluster();
127    MiniZooKeeperCluster miniZK = utility1.getZkCluster();
128    // Have to reget conf1 in case zk cluster location different
129    // than default
130    conf1 = utility1.getConfiguration();  
131    zkw1 = new ZooKeeperWatcher(conf1, "cluster1", null, true);
132    admin = new ReplicationAdmin(conf1);
133    LOG.info("Setup first Zk");
134
135    // Base conf2 on conf1 so it gets the right zk cluster.
136    conf2 = HBaseConfiguration.create(conf1);
137    conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
138    conf2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 6);
139    conf2.setBoolean("hbase.tests.use.shortcircuit.reads", false);
140
141    utility2 = new HBaseTestingUtility(conf2);
142    utility2.setZkCluster(miniZK);
143    zkw2 = new ZooKeeperWatcher(conf2, "cluster2", null, true);
144    LOG.info("Setup second Zk");
145
146    CONF_WITH_LOCALFS = HBaseConfiguration.create(conf1);
147    utility1.startMiniCluster(2);
148    // Have a bunch of slave servers, because inter-cluster shipping logic uses number of sinks
149    // as a component in deciding maximum number of parallel batches to send to the peer cluster.
150    utility2.startMiniCluster(4);
151
152    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
153    rpc.setClusterKey(utility2.getClusterKey());
154    hbaseAdmin = ConnectionFactory.createConnection(conf1).getAdmin();
155    hbaseAdmin.addReplicationPeer("2", rpc);
156
157    HTableDescriptor table = new HTableDescriptor(tableName);
158    HColumnDescriptor fam = new HColumnDescriptor(famName);
159    fam.setMaxVersions(100);
160    fam.setScope(HConstants.REPLICATION_SCOPE_GLOBAL);
161    table.addFamily(fam);
162    fam = new HColumnDescriptor(noRepfamName);
163    table.addFamily(fam);
164    scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
165    for(HColumnDescriptor f : table.getColumnFamilies()) {
166      scopes.put(f.getName(), f.getScope());
167    }
168    Connection connection1 = ConnectionFactory.createConnection(conf1);
169    Connection connection2 = ConnectionFactory.createConnection(conf2);
170    try (Admin admin1 = connection1.getAdmin()) {
171      admin1.createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
172    }
173    try (Admin admin2 = connection2.getAdmin()) {
174      admin2.createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
175    }
176    utility1.waitUntilAllRegionsAssigned(tableName);
177    utility2.waitUntilAllRegionsAssigned(tableName);
178    htable1 = connection1.getTable(tableName);
179    htable2 = connection2.getTable(tableName);
180  }
181
182  /**
183   * @throws java.lang.Exception
184   */
185  @AfterClass
186  public static void tearDownAfterClass() throws Exception {
187    htable2.close();
188    htable1.close();
189    admin.close();
190    utility2.shutdownMiniCluster();
191    utility1.shutdownMiniCluster();
192  }
