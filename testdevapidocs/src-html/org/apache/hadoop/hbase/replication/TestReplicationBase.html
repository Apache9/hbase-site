044/**
045 * This class is only a base for other integration-level replication tests.
046 * Do not add tests here.
047 * TestReplicationSmallTests is where tests that don't require bring machines up/down should go
048 * All other tests should have their own classes and extend this one
049 */
050public class TestReplicationBase {
051/*
052  {
053    ((Log4JLogger) ReplicationSource.LOG).getLogger().setLevel(Level.ALL);
054  }*/
055
056  private static final Log LOG = LogFactory.getLog(TestReplicationBase.class);
057
058  protected static Configuration conf1 = HBaseConfiguration.create();
059  protected static Configuration conf2;
060  protected static Configuration CONF_WITH_LOCALFS;
061
062  protected static ZooKeeperWatcher zkw1;
063  protected static ZooKeeperWatcher zkw2;
064
065  protected static ReplicationAdmin admin;
066  private static Admin hbaseAdmin;
067
068  protected static Table htable1;
069  protected static Table htable2;
070  protected static NavigableMap<byte[], Integer> scopes;
071
072  protected static HBaseTestingUtility utility1;
073  protected static HBaseTestingUtility utility2;
074  protected static final int NB_ROWS_IN_BATCH = 100;
075  protected static final int NB_ROWS_IN_BIG_BATCH =
076      NB_ROWS_IN_BATCH * 10;
077  protected static final long SLEEP_TIME = 500;
078  protected static final int NB_RETRIES = 10;
079
080  protected static final TableName tableName = TableName.valueOf("test");
081  protected static final byte[] famName = Bytes.toBytes("f");
082  protected static final byte[] row = Bytes.toBytes("row");
083  protected static final byte[] noRepfamName = Bytes.toBytes("norep");
084
085  /**
086   * @throws java.lang.Exception
087   */
088  @BeforeClass
089  public static void setUpBeforeClass() throws Exception {
090    conf1.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/1");
091    // We don't want too many edits per batch sent to the ReplicationEndpoint to trigger
092    // sufficient number of events. But we don't want to go too low because
093    // HBaseInterClusterReplicationEndpoint partitions entries into batches and we want
094    // more than one batch sent to the peer cluster for better testing.
095    conf1.setInt("replication.source.size.capacity", 102400);
096    conf1.setLong("replication.source.sleepforretries", 100);
097    conf1.setInt("hbase.regionserver.maxlogs", 10);
098    conf1.setLong("hbase.master.logcleaner.ttl", 10);
099    conf1.setInt("zookeeper.recovery.retry", 1);
100    conf1.setInt("zookeeper.recovery.retry.intervalmill", 10);
101    conf1.setLong(HConstants.THREAD_WAKE_FREQUENCY, 100);
102    conf1.setInt("replication.stats.thread.period.seconds", 5);
103    conf1.setBoolean("hbase.tests.use.shortcircuit.reads", false);
104    conf1.setLong("replication.sleep.before.failover", 2000);
105    conf1.setInt("replication.source.maxretriesmultiplier", 10);
106    conf1.setFloat("replication.source.ratio", 1.0f);
107
108    utility1 = new HBaseTestingUtility(conf1);
109    utility1.startMiniZKCluster();
110    MiniZooKeeperCluster miniZK = utility1.getZkCluster();
111    // Have to reget conf1 in case zk cluster location different
112    // than default
113    conf1 = utility1.getConfiguration();  
114    zkw1 = new ZooKeeperWatcher(conf1, "cluster1", null, true);
115    admin = new ReplicationAdmin(conf1);
116    LOG.info("Setup first Zk");
117
118    // Base conf2 on conf1 so it gets the right zk cluster.
119    conf2 = HBaseConfiguration.create(conf1);
120    conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
121    conf2.setInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER, 6);
122    conf2.setBoolean("hbase.tests.use.shortcircuit.reads", false);
123
124    utility2 = new HBaseTestingUtility(conf2);
125    utility2.setZkCluster(miniZK);
126    zkw2 = new ZooKeeperWatcher(conf2, "cluster2", null, true);
127    LOG.info("Setup second Zk");
128
129    CONF_WITH_LOCALFS = HBaseConfiguration.create(conf1);
130    utility1.startMiniCluster(2);
131    // Have a bunch of slave servers, because inter-cluster shipping logic uses number of sinks
132    // as a component in deciding maximum number of parallel batches to send to the peer cluster.
133    utility2.startMiniCluster(4);
134
135    ReplicationPeerConfig rpc = new ReplicationPeerConfig();
136    rpc.setClusterKey(utility2.getClusterKey());
137    hbaseAdmin = ConnectionFactory.createConnection(conf1).getAdmin();
138    hbaseAdmin.addReplicationPeer("2", rpc);
139
140    HTableDescriptor table = new HTableDescriptor(tableName);
141    HColumnDescriptor fam = new HColumnDescriptor(famName);
142    fam.setMaxVersions(100);
143    fam.setScope(HConstants.REPLICATION_SCOPE_GLOBAL);
144    table.addFamily(fam);
145    fam = new HColumnDescriptor(noRepfamName);
146    table.addFamily(fam);
147    scopes = new TreeMap<>(Bytes.BYTES_COMPARATOR);
148    for(HColumnDescriptor f : table.getColumnFamilies()) {
149      scopes.put(f.getName(), f.getScope());
150    }
151    Connection connection1 = ConnectionFactory.createConnection(conf1);
152    Connection connection2 = ConnectionFactory.createConnection(conf2);
153    try (Admin admin1 = connection1.getAdmin()) {
154      admin1.createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
155    }
156    try (Admin admin2 = connection2.getAdmin()) {
157      admin2.createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
158    }
159    utility1.waitUntilAllRegionsAssigned(tableName);
160    utility2.waitUntilAllRegionsAssigned(tableName);
161    htable1 = connection1.getTable(tableName);
162    htable1.setWriteBufferSize(1024);
163    htable2 = connection2.getTable(tableName);
164  }
165
166  /**
167   * @throws java.lang.Exception
168   */
169  @AfterClass
170  public static void tearDownAfterClass() throws Exception {
171    htable2.close();
172    htable1.close();
173    admin.close();
174    utility2.shutdownMiniCluster();
175    utility1.shutdownMiniCluster();
176  }
177}
