122  @Test
123  public void testRegionMoveAndFailover() throws Exception {
124    TableName tableName = TableName.valueOf("testRSFailover");
125    HTableDescriptor table = new HTableDescriptor(tableName);
126    HColumnDescriptor fam = new HColumnDescriptor(famName);
127    fam.setScope(HConstants.REPLICATION_SCOPE_SERIAL);
128    table.addFamily(fam);
129    utility1.getHBaseAdmin().createTable(table);
130    utility2.getHBaseAdmin().createTable(table);
131    try(Table t1 = utility1.getConnection().getTable(tableName);
132        Table t2 = utility2.getConnection().getTable(tableName)) {
133      LOG.info("move to 1");
134      moveRegion(t1, 1);
135      LOG.info("move to 0");
136      moveRegion(t1, 0);
137      for (int i = 10; i < 20; i++) {
138        Put put = new Put(ROWS[i]);
139        put.addColumn(famName, VALUE, VALUE);
140        t1.put(put);
141      }
142      LOG.info("move to 2");
143      moveRegion(t1, 2);
144      for (int i = 20; i < 30; i++) {
145        Put put = new Put(ROWS[i]);
146        put.addColumn(famName, VALUE, VALUE);
147        t1.put(put);
148      }
149      utility1.getHBaseCluster().abortRegionServer(2);
150      for (int i = 30; i < 40; i++) {
151        Put put = new Put(ROWS[i]);
152        put.addColumn(famName, VALUE, VALUE);
153        t1.put(put);
154      }
155
156      long start = EnvironmentEdgeManager.currentTime();
157      while (EnvironmentEdgeManager.currentTime() - start < 180000) {
158        Scan scan = new Scan();
159        scan.setCaching(100);
160        List<Cell> list = new ArrayList<>();
161        try (ResultScanner results = t2.getScanner(scan)) {
162          for (Result result : results) {
163            assertEquals(1, result.rawCells().length);
164            list.add(result.rawCells()[0]);
165          }
166        }
167        List<Integer> listOfNumbers = getRowNumbers(list);
168        LOG.info(Arrays.toString(listOfNumbers.toArray()));
169        assertIntegerList(listOfNumbers, 10, 1);
170        if (listOfNumbers.size() != 30) {
171          LOG.info("Waiting all logs pushed to slave. Expected 30 , actual " + list.size());
172          Thread.sleep(200);
173          continue;
174        }
175        return;
176      }
177      throw new Exception("Not all logs have been pushed");
178    }
179  }

183    TableName tableName = TableName.valueOf("testRegionSplit");
184    HTableDescriptor table = new HTableDescriptor(tableName);

186    fam.setScope(HConstants.REPLICATION_SCOPE_SERIAL);

190    try(Table t1 = utility1.getConnection().getTable(tableName);
191        Table t2 = utility2.getConnection().getTable(tableName)) {
192
193      for (int i = 10; i < 100; i += 10) {
194        Put put = new Put(ROWS[i]);
195        put.addColumn(famName, VALUE, VALUE);
196        t1.put(put);
197      }
198      utility1.getHBaseAdmin().split(tableName, ROWS[50]);
199      waitTableHasRightNumberOfRegions(tableName, 2);
200      for (int i = 11; i < 100; i += 10) {
201        Put put = new Put(ROWS[i]);
202        put.addColumn(famName, VALUE, VALUE);
203        t1.put(put);
204      }
205
206      long start = EnvironmentEdgeManager.currentTime();
207      while (EnvironmentEdgeManager.currentTime() - start < 180000) {
208        Scan scan = new Scan();
209        scan.setCaching(100);
210        List<Cell> list = new ArrayList<>();
211        try (ResultScanner results = t2.getScanner(scan)) {
212          for (Result result : results) {
213            assertEquals(1, result.rawCells().length);
214            list.add(result.rawCells()[0]);
215          }
216        }
217        List<Integer> listOfNumbers = getRowNumbers(list);
218        List<Integer> list1 = new ArrayList<>();
219        List<Integer> list21 = new ArrayList<>();
220        List<Integer> list22 = new ArrayList<>();
221        for (int num : listOfNumbers) {
222          if (num % 10 == 0) {
223            list1.add(num);
224          }else if (num < 50) { //num%10==1
225            list21.add(num);
226          } else { // num%10==1&&num>50
227            list22.add(num);
228          }
229        }
230
231        LOG.info(Arrays.toString(list1.toArray()));
232        LOG.info(Arrays.toString(list21.toArray()));
233        LOG.info(Arrays.toString(list22.toArray()));
234        assertIntegerList(list1, 10, 10);
235        assertIntegerList(list21, 11, 10);
236        assertIntegerList(list22, 51, 10);
237        if (!list21.isEmpty() || !list22.isEmpty()) {
238          assertEquals(9, list1.size());
239        }
240
241        if (list.size() == 18) {
242          return;
243        }
244        LOG.info("Waiting all logs pushed to slave. Expected 27 , actual " + list.size());
245        Thread.sleep(200);
246      }
247      throw new Exception("Not all logs have been pushed");
248    }

251  @Test
252  public void testRegionMerge() throws Exception {
253    TableName tableName = TableName.valueOf("testRegionMerge");
254    HTableDescriptor table = new HTableDescriptor(tableName);
255    HColumnDescriptor fam = new HColumnDescriptor(famName);
256    fam.setScope(HConstants.REPLICATION_SCOPE_SERIAL);
257    table.addFamily(fam);
258    utility1.getHBaseAdmin().createTable(table);
259    utility2.getHBaseAdmin().createTable(table);
260    Threads.sleep(5000);
261    utility1.getHBaseAdmin().split(tableName, ROWS[50]);
262    waitTableHasRightNumberOfRegions(tableName, 2);
263
264    try(Table t1 = utility1.getConnection().getTable(tableName);
265        Table t2 = utility2.getConnection().getTable(tableName)) {
266      for (int i = 10; i < 100; i += 10) {
267        Put put = new Put(ROWS[i]);
268        put.addColumn(famName, VALUE, VALUE);
269        t1.put(put);
270      }
271      List<Pair<HRegionInfo, ServerName>> regions =
272          MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), tableName);
273      utility1.getHBaseAdmin().mergeRegions(regions.get(0).getFirst().getRegionName(),
274          regions.get(1).getFirst().getRegionName(), true);
275      waitTableHasRightNumberOfRegions(tableName, 1);
276      for (int i = 11; i < 100; i += 10) {
277        Put put = new Put(ROWS[i]);
278        put.addColumn(famName, VALUE, VALUE);
279        t1.put(put);
280      }
281
282      long start = EnvironmentEdgeManager.currentTime();
283      while (EnvironmentEdgeManager.currentTime() - start < 180000) {
284        Scan scan = new Scan();
285        scan.setCaching(100);
286        List<Cell> list = new ArrayList<>();
287        try (ResultScanner results = t2.getScanner(scan)) {
288          for (Result result : results) {
289            assertEquals(1, result.rawCells().length);
290            list.add(result.rawCells()[0]);
291          }
292        }
293        List<Integer> listOfNumbers = getRowNumbers(list);
294        List<Integer> list0 = new ArrayList<>();
295        List<Integer> list1 = new ArrayList<>();
296        for (int num : listOfNumbers) {
297          if (num % 10 == 0) {
298            list0.add(num);
299          } else {
300            list1.add(num);
301          }
302        }
303        LOG.info(Arrays.toString(list0.toArray()));
304        LOG.info(Arrays.toString(list1.toArray()));
305        assertIntegerList(list1, 11, 10);
306        if (!list1.isEmpty()) {
307          assertEquals(9, list0.size());
308        }
309        if (list.size() == 18) {
310          return;
311        }
312        LOG.info("Waiting all logs pushed to slave. Expected 18 , actual " + list.size());
313        Thread.sleep(200);
314      }
315
316    }
317  }

319  private List<Integer> getRowNumbers(List<Cell> cells) {
320    List<Integer> listOfRowNumbers = new ArrayList<>();
321    for (Cell c : cells) {
322      listOfRowNumbers.add(Integer.parseInt(Bytes
323          .toString(c.getRowArray(), c.getRowOffset() + ROW.length,
324              c.getRowLength() - ROW.length)));
325    }
326    return listOfRowNumbers;
327  }

335  private void moveRegion(Table table, int index) throws IOException {
336    List<Pair<HRegionInfo, ServerName>> regions =
337        MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), table.getName());
338    assertEquals(1, regions.size());
339    HRegionInfo regionInfo = regions.get(0).getFirst();
340    ServerName name = utility1.getHBaseCluster().getRegionServer(index).getServerName();
341    utility1.getAdmin()
342        .move(regionInfo.getEncodedNameAsBytes(), Bytes.toBytes(name.getServerName()));
343    while (true) {
344      regions =
345          MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), table.getName());
346      if (regions.get(0).getSecond().equals(name)) {
347        break;
348      }
349      Threads.sleep(100);
350    }
351  }

353  private void balanceTwoRegions(Table table) throws Exception {
354    List<Pair<HRegionInfo, ServerName>> regions =
355        MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), table.getName());
356    assertEquals(2, regions.size());
357    HRegionInfo regionInfo1 = regions.get(0).getFirst();
358    ServerName name1 = utility1.getHBaseCluster().getRegionServer(0).getServerName();
359    HRegionInfo regionInfo2 = regions.get(1).getFirst();
360    ServerName name2 = utility1.getHBaseCluster().getRegionServer(1).getServerName();
361    utility1.getAdmin()
362        .move(regionInfo1.getEncodedNameAsBytes(), Bytes.toBytes(name1.getServerName()));
363    utility1.getAdmin()
364        .move(regionInfo2.getEncodedNameAsBytes(), Bytes.toBytes(name2.getServerName()));
365    while (true) {
366      regions =
367          MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), table.getName());
368      if (regions.get(0).getSecond().equals(name1) && regions.get(1).getSecond().equals(name2)) {
369        break;
370      }
371      Threads.sleep(100);
372    }
373  }

375  private void waitTableHasRightNumberOfRegions(TableName tableName, int num) throws IOException {
376    while (true) {
377      List<Pair<HRegionInfo, ServerName>> regions =
378          MetaTableAccessor.getTableRegionsAndLocations(utility1.getConnection(), tableName);
379      if (regions.size() == num) {
380        return;
381      }
382      Threads.sleep(100);
383    }
384
385  }

387  private void assertIntegerList(List<Integer> list, int start, int step) {
388    int size = list.size();
389    for (int i = 0; i < size; i++) {
390      assertEquals(start + step * i, list.get(i).intValue());
391    }
