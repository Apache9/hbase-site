077  @Before
078  public void setup() throws Exception {
079    ReplicationEndpointForTest.contructedCount.set(0);
080    ReplicationEndpointForTest.startedCount.set(0);
081    ReplicationEndpointForTest.replicateCount.set(0);
082    ReplicationEndpointReturningFalse.replicated.set(false);
083    ReplicationEndpointForTest.lastEntries = null;
084    final List<RegionServerThread> rsThreads =
085        utility1.getMiniHBaseCluster().getRegionServerThreads();
086    for (RegionServerThread rs : rsThreads) {
087      utility1.getHBaseAdmin().rollWALWriter(rs.getRegionServer().getServerName());
088    }
089    // Wait for  all log roll to finish
090    utility1.waitFor(3000, new Waiter.ExplainingPredicate<Exception>() {
091      @Override
092      public boolean evaluate() throws Exception {
093        for (RegionServerThread rs : rsThreads) {
094          if (!rs.getRegionServer().walRollRequestFinished()) {
095            return false;
096          }
097        }
098        return true;
099      }
100
101      @Override
102      public String explainFailure() throws Exception {
103        List<String> logRollInProgressRsList = new ArrayList<String>();
104        for (RegionServerThread rs : rsThreads) {
105          if (!rs.getRegionServer().walRollRequestFinished()) {
106            logRollInProgressRsList.add(rs.getRegionServer().toString());
107          }
108        }
109        return "Still waiting for log roll on regionservers: " + logRollInProgressRsList;
110      }
111    });
112  }

117    admin.addPeer("testCustomReplicationEndpoint",
118      new ReplicationPeerConfig().setClusterKey(ZKUtil.getZooKeeperClusterKey(conf1))

153  @Test (timeout=120000)
154  public void testReplicationEndpointReturnsFalseOnReplicate() throws Exception {
155    Assert.assertEquals(0, ReplicationEndpointForTest.replicateCount.get());
156    Assert.assertTrue(!ReplicationEndpointReturningFalse.replicated.get());
157    int peerCount = admin.getPeersCount();
158    final String id = "testReplicationEndpointReturnsFalseOnReplicate";
159    admin.addPeer(id,
160      new ReplicationPeerConfig().setClusterKey(ZKUtil.getZooKeeperClusterKey(conf1))
161        .setReplicationEndpointImpl(ReplicationEndpointReturningFalse.class.getName()), null);
162    // This test is flakey and then there is so much stuff flying around in here its, hard to
163    // debug.  Peer needs to be up for the edit to make it across. This wait on
164    // peer count seems to be a hack that has us not progress till peer is up.
165    if (admin.getPeersCount() <= peerCount) {
166      LOG.info("Waiting on peercount to go up from " + peerCount);
167      Threads.sleep(100);
168    }
169    // now replicate some data
170    doPut(row);
171
172    Waiter.waitFor(conf1, 60000, new Waiter.Predicate<Exception>() {
173      @Override
174      public boolean evaluate() throws Exception {
175        // Looks like replication endpoint returns false unless we put more than 10 edits. We
176        // only send over one edit.
177        int count = ReplicationEndpointForTest.replicateCount.get();
178        LOG.info("count=" + count);
179        return ReplicationEndpointReturningFalse.replicated.get();
180      }
181    });
182    if (ReplicationEndpointReturningFalse.ex.get() != null) {
183      throw ReplicationEndpointReturningFalse.ex.get();
184    }
185
186    admin.removePeer("testReplicationEndpointReturnsFalseOnReplicate");
187  }

189  @Test (timeout=120000)
190  public void testInterClusterReplication() throws Exception {
191    final String id = "testInterClusterReplication";
192
193    List<HRegion> regions = utility1.getHBaseCluster().getRegions(tableName);
194    int totEdits = 0;
195
196    // Make sure edits are spread across regions because we do region based batching
197    // before shipping edits.
198    for(HRegion region: regions) {
199      HRegionInfo hri = region.getRegionInfo();
200      byte[] row = hri.getStartKey();
201      for (int i = 0; i < 100; i++) {
202        if (row.length > 0) {
203          Put put = new Put(row);
204          put.addColumn(famName, row, row);
205          region.put(put);
206          totEdits++;
207        }
208      }
209    }
210
211    admin.addPeer(id,
212        new ReplicationPeerConfig().setClusterKey(ZKUtil.getZooKeeperClusterKey(conf2))
213            .setReplicationEndpointImpl(InterClusterReplicationEndpointForTest.class.getName()),
214        null);
215
216    final int numEdits = totEdits;
217    Waiter.waitFor(conf1, 30000, new Waiter.ExplainingPredicate<Exception>() {
218      @Override
219      public boolean evaluate() throws Exception {
220        return InterClusterReplicationEndpointForTest.replicateCount.get() == numEdits;
221      }
222      @Override
223      public String explainFailure() throws Exception {
224        String failure = "Failed to replicate all edits, expected = " + numEdits
225            + " replicated = " + InterClusterReplicationEndpointForTest.replicateCount.get();
226        return failure;
227      }
228    });
229
230    admin.removePeer("testInterClusterReplication");
231    utility1.deleteTableData(tableName);
232  }

258  private void doPut(byte[] row) throws IOException {
259    try (Connection connection = ConnectionFactory.createConnection(conf1)) {
260      doPut(connection, row);
261    }
262  }
263
264  private void doPut(final Connection connection, final byte [] row) throws IOException {
265    try (Table t = connection.getTable(tableName)) {
266      Put put = new Put(row);
267      put.addColumn(famName, row, row);
268      t.put(put);
269    }
270  }
271
272  private static void doAssert(byte[] row) throws Exception {
273    if (ReplicationEndpointForTest.lastEntries == null) {
274      return; // first call
275    }
276    Assert.assertEquals(1, ReplicationEndpointForTest.lastEntries.size());
277    List<Cell> cells = ReplicationEndpointForTest.lastEntries.get(0).getEdit().getCells();
278    Assert.assertEquals(1, cells.size());
279    Assert.assertTrue(Bytes.equals(cells.get(0).getRowArray(), cells.get(0).getRowOffset(),
280      cells.get(0).getRowLength(), row, 0, row.length));
281  }

283  public static class ReplicationEndpointForTest extends BaseReplicationEndpoint {
284    static UUID uuid = UUID.randomUUID();
285    static AtomicInteger contructedCount = new AtomicInteger();
286    static AtomicInteger startedCount = new AtomicInteger();
287    static AtomicInteger stoppedCount = new AtomicInteger();
288    static AtomicInteger replicateCount = new AtomicInteger();
289    static volatile List<Entry> lastEntries = null;
290
291    public ReplicationEndpointForTest() {
292      contructedCount.incrementAndGet();
293    }
294
295    @Override
296    public UUID getPeerUUID() {
297      return uuid;
298    }
299
300    @Override
301    public boolean replicate(ReplicateContext replicateContext) {
302      replicateCount.incrementAndGet();
303      lastEntries = replicateContext.entries;
304      return true;
305    }
306
307    @Override
308    protected void doStart() {
309      startedCount.incrementAndGet();
310      notifyStarted();
311    }
312
313    @Override
314    protected void doStop() {
315      stoppedCount.incrementAndGet();
316      notifyStopped();
317    }
318  }

320  public static class InterClusterReplicationEndpointForTest
321      extends HBaseInterClusterReplicationEndpoint {
322
323    static AtomicInteger replicateCount = new AtomicInteger();
324    static boolean failedOnce;
325
326    @Override
327    public boolean replicate(ReplicateContext replicateContext) {
328      boolean success = super.replicate(replicateContext);
329      if (success) {
330        replicateCount.addAndGet(replicateContext.entries.size());
331      }
332      return success;
333    }
334
335    @Override
336    protected Replicator createReplicator(List<Entry> entries, int ordinal) {
337      // Fail only once, we don't want to slow down the test.
338      if (failedOnce) {
339        return new DummyReplicator(entries, ordinal);
340      } else {
341        failedOnce = true;
342        return new FailingDummyReplicator(entries, ordinal);
343      }
344    }
345
346    protected class DummyReplicator extends Replicator {
347
348      private int ordinal;
349
350      public DummyReplicator(List<Entry> entries, int ordinal) {
351        super(entries, ordinal);
352        this.ordinal = ordinal;
353      }
354
355      @Override
356      public Integer call() throws IOException {
357        return ordinal;
358      }
359    }
360
361    protected class FailingDummyReplicator extends DummyReplicator {
362
363      public FailingDummyReplicator(List<Entry> entries, int ordinal) {
364        super(entries, ordinal);
365      }
366
367      @Override
368      public Integer call() throws IOException {
369        throw new IOException("Sample Exception: Failed to replicate.");
370      }
371    }
372  }

374  public static class ReplicationEndpointReturningFalse extends ReplicationEndpointForTest {
375    static int COUNT = 10;
376    static AtomicReference<Exception> ex = new AtomicReference<Exception>(null);
377    static AtomicBoolean replicated = new AtomicBoolean(false);
378    @Override
379    public boolean replicate(ReplicateContext replicateContext) {
380      try {
381        // check row
382        doAssert(row);
383      } catch (Exception e) {
384        ex.set(e);
385      }
386
387      super.replicate(replicateContext);
388      LOG.info("Replicated " + row + ", count=" + replicateCount.get());
389
390      replicated.set(replicateCount.get() > COUNT); // first 10 times, we return false
391      return replicated.get();
392    }
393  }

395  // return a WALEntry filter which only accepts "row", but not other rows
396  public static class ReplicationEndpointWithWALEntryFilter extends ReplicationEndpointForTest {
397    static AtomicReference<Exception> ex = new AtomicReference<Exception>(null);
398
399    @Override
400    public boolean replicate(ReplicateContext replicateContext) {
401      try {
402        super.replicate(replicateContext);
403        doAssert(row);
404      } catch (Exception e) {
405        ex.set(e);
406      }
407      return true;
408    }
409
410    @Override
411    public WALEntryFilter getWALEntryfilter() {
412      return new ChainWALEntryFilter(super.getWALEntryfilter(), new WALEntryFilter() {
413        @Override
414        public Entry filter(Entry entry) {
415          ArrayList<Cell> cells = entry.getEdit().getCells();
416          int size = cells.size();
417          for (int i = size-1; i >= 0; i--) {
418            Cell cell = cells.get(i);
419            if (!Bytes.equals(cell.getRowArray(), cell.getRowOffset(), cell.getRowLength(),
420              row, 0, row.length)) {
421              cells.remove(i);
422            }
423          }
424          return entry;
425        }
426      });
427    }
