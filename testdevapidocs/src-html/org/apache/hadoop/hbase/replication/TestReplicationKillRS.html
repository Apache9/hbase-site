019package org.apache.hadoop.hbase.replication;
020
021import org.apache.hadoop.hbase.HBaseTestingUtility;
022import org.apache.hadoop.hbase.UnknownScannerException;
023import org.apache.hadoop.hbase.client.Result;
024import org.apache.hadoop.hbase.client.ResultScanner;
025import org.apache.hadoop.hbase.client.Scan;
026import org.apache.hadoop.hbase.testclassification.LargeTests;
027import org.apache.hadoop.hbase.testclassification.ReplicationTests;
028import org.junit.experimental.categories.Category;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import static org.junit.Assert.fail;
033
034@Category({ReplicationTests.class, LargeTests.class})
035public class TestReplicationKillRS extends TestReplicationBase {
036
037  private static final Logger LOG = LoggerFactory.getLogger(TestReplicationKillRS.class);
038
039  /**
040   * Load up 1 tables over 2 region servers and kill a source during
041   * the upload. The failover happens internally.
042   *
043   * WARNING this test sometimes fails because of HBASE-3515
044   *
045   * @throws Exception
046   */
047  public void loadTableAndKillRS(HBaseTestingUtility util) throws Exception {
048    // killing the RS with hbase:meta can result into failed puts until we solve
049    // IO fencing
050    int rsToKill1 =
051        util.getHBaseCluster().getServerWithMeta() == 0 ? 1 : 0;
052
053    // Takes about 20 secs to run the full loading, kill around the middle
054    Thread killer = killARegionServer(util, 5000, rsToKill1);
055
056    LOG.info("Start loading table");
057    int initialCount = utility1.loadTable(htable1, famName);
058    LOG.info("Done loading table");
059    killer.join(5000);
060    LOG.info("Done waiting for threads");
061
062    Result[] res;
063    while (true) {
064      try {
065        Scan scan = new Scan();
066        ResultScanner scanner = htable1.getScanner(scan);
067        res = scanner.next(initialCount);
068        scanner.close();
069        break;
070      } catch (UnknownScannerException ex) {
071        LOG.info("Cluster wasn't ready yet, restarting scanner");
072      }
073    }
074    // Test we actually have all the rows, we may miss some because we
075    // don't have IO fencing.
076    if (res.length != initialCount) {
077      LOG.warn("We lost some rows on the master cluster!");
078      // We don't really expect the other cluster to have more rows
079      initialCount = res.length;
080    }
081
082    int lastCount = 0;
083
084    final long start = System.currentTimeMillis();
085    int i = 0;
086    while (true) {
087      if (i==NB_RETRIES-1) {
088        fail("Waited too much time for queueFailover replication. " +
089            "Waited "+(System.currentTimeMillis() - start)+"ms.");
090      }
091      Scan scan2 = new Scan();
092      ResultScanner scanner2 = htable2.getScanner(scan2);
093      Result[] res2 = scanner2.next(initialCount * 2);
094      scanner2.close();
095      if (res2.length < initialCount) {
096        if (lastCount < res2.length) {
097          i--; // Don't increment timeout if we make progress
098        } else {
099          i++;
100        }
101        lastCount = res2.length;
102        LOG.info("Only got " + lastCount + " rows instead of " +
103            initialCount + " current i=" + i);
104        Thread.sleep(SLEEP_TIME*2);
105      } else {
106        break;
107      }
108    }
109  }
110
111  private static Thread killARegionServer(final HBaseTestingUtility utility,
112                                          final long timeout, final int rs) {
113    Thread killer = new Thread() {
114      public void run() {
115        try {
116          Thread.sleep(timeout);
117          utility.getHBaseCluster().getRegionServer(rs).stop("Stopping as part of the test");
118        } catch (Exception e) {
119          LOG.error("Couldn't kill a region server", e);
120        }
121      }
122    };
123    killer.setDaemon(true);
124    killer.start();
125    return killer;
126  }
127}
