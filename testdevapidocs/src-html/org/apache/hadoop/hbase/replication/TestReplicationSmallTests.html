097    // Starting and stopping replication can make us miss new logs,
098    // rolling like this makes sure the most recent one gets added to the queue
099    for ( JVMClusterUtil.RegionServerThread r :
100        utility1.getHBaseCluster().getRegionServerThreads()) {
101      utility1.getAdmin().rollWALWriter(r.getRegionServer().getServerName());
102    }
103    int rowCount = utility1.countRows(tableName);
104    utility1.deleteTableData(tableName);
105    // truncating the table will send one Delete per row to the slave cluster
106    // in an async fashion, which is why we cannot just call deleteTableData on
107    // utility2 since late writes could make it to the slave in some way.
108    // Instead, we truncate the first table and wait for all the Deletes to
109    // make it to the slave.
110    Scan scan = new Scan();
111    int lastCount = 0;
112    for (int i = 0; i < NB_RETRIES; i++) {
113      if (i==NB_RETRIES-1) {
114        fail("Waited too much time for truncate");
115      }
116      ResultScanner scanner = htable2.getScanner(scan);
117      Result[] res = scanner.next(rowCount);
118      scanner.close();
119      if (res.length != 0) {
120        if (res.length < lastCount) {
121          i--; // Don't increment timeout if we make progress
122        }
123        lastCount = res.length;
124        LOG.info("Still got " + res.length + " rows");
125        Thread.sleep(SLEEP_TIME);
126      } else {
127        break;
128      }
129    }

133   * Verify that version and column delete marker types are replicated
134   * correctly.
135   * @throws Exception
136   */
137  @Test(timeout=300000)
138  public void testDeleteTypes() throws Exception {
139    LOG.info("testDeleteTypes");
140    final byte[] v1 = Bytes.toBytes("v1");
141    final byte[] v2 = Bytes.toBytes("v2");
142    final byte[] v3 = Bytes.toBytes("v3");
143    htable1 = utility1.getConnection().getTable(tableName);
144
145    long t = EnvironmentEdgeManager.currentTime();
146    // create three versions for "row"
147    Put put = new Put(row);
148    put.addColumn(famName, row, t, v1);
149    htable1.put(put);
150
151    put = new Put(row);
152    put.addColumn(famName, row, t + 1, v2);
153    htable1.put(put);
154
155    put = new Put(row);
156    put.addColumn(famName, row, t + 2, v3);
157    htable1.put(put);
158
159    Get get = new Get(row);
160    get.setMaxVersions();
161    for (int i = 0; i < NB_RETRIES; i++) {
162      if (i==NB_RETRIES-1) {
163        fail("Waited too much time for put replication");
164      }
165      Result res = htable2.get(get);
166      if (res.size() < 3) {
167        LOG.info("Rows not available");
168        Thread.sleep(SLEEP_TIME);
169      } else {
170        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[0]), v3);
171        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[1]), v2);
172        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[2]), v1);
173        break;
174      }
175    }
176    // place a version delete marker (delete last version)
177    Delete d = new Delete(row);
178    d.addColumn(famName, row, t);
179    htable1.delete(d);
180
181    get = new Get(row);
182    get.setMaxVersions();
183    for (int i = 0; i < NB_RETRIES; i++) {
184      if (i==NB_RETRIES-1) {
185        fail("Waited too much time for put replication");
186      }
187      Result res = htable2.get(get);
188      if (res.size() > 2) {
189        LOG.info("Version not deleted");
190        Thread.sleep(SLEEP_TIME);
191      } else {
192        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[0]), v3);
193        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[1]), v2);
194        break;
195      }
196    }
197
198    // place a column delete marker
199    d = new Delete(row);
200    d.addColumns(famName, row, t+2);
201    htable1.delete(d);
202
203    // now *both* of the remaining version should be deleted
204    // at the replica
205    get = new Get(row);
206    for (int i = 0; i < NB_RETRIES; i++) {
207      if (i==NB_RETRIES-1) {
208        fail("Waited too much time for del replication");
209      }
210      Result res = htable2.get(get);
211      if (res.size() >= 1) {
212        LOG.info("Rows not deleted");
213        Thread.sleep(SLEEP_TIME);
214      } else {
215        break;
216      }
217    }
218  }

286  private void waitForReplication(int expectedRows, int retries) throws IOException, InterruptedException {
287    Scan scan;
288    for (int i = 0; i < retries; i++) {
289      scan = new Scan();
290      if (i== retries -1) {
291        fail("Waited too much time for normal batch replication");
292      }
293      ResultScanner scanner = htable2.getScanner(scan);
294      Result[] res = scanner.next(expectedRows);
295      scanner.close();
296      if (res.length != expectedRows) {
297        LOG.info("Only got " + res.length + " rows");
298        Thread.sleep(SLEEP_TIME);
299      } else {
300        break;
301      }
302    }
303  }

416   * Do a more intense version testSmallBatch, one  that will trigger
417   * wal rolling and other non-trivial code paths
418   * @throws Exception
419   */
420  @Test(timeout=300000)
421  public void testLoading() throws Exception {
422    LOG.info("Writing out rows to table1 in testLoading");
423    List<Put> puts = new ArrayList<>(NB_ROWS_IN_BIG_BATCH);
424    for (int i = 0; i < NB_ROWS_IN_BIG_BATCH; i++) {
425      Put put = new Put(Bytes.toBytes(i));
426      put.addColumn(famName, row, row);
427      puts.add(put);
428    }
429    htable1.setWriteBufferSize(1024);
430    // The puts will be iterated through and flushed only when the buffer
431    // size is reached.
432    htable1.put(puts);
433
434    Scan scan = new Scan();
435
436    ResultScanner scanner = htable1.getScanner(scan);
437    Result[] res = scanner.next(NB_ROWS_IN_BIG_BATCH);
438    scanner.close();
439
440    assertEquals(NB_ROWS_IN_BIG_BATCH, res.length);
441
442    LOG.info("Looking in table2 for replicated rows in testLoading");
443    long start = System.currentTimeMillis();
444    // Retry more than NB_RETRIES.  As it was, retries were done in 5 seconds and we'd fail
445    // sometimes.
446    final long retries = NB_RETRIES * 10;
447    for (int i = 0; i < retries; i++) {
448      scan = new Scan();
449      scanner = htable2.getScanner(scan);
450      res = scanner.next(NB_ROWS_IN_BIG_BATCH);
451      scanner.close();
452      if (res.length != NB_ROWS_IN_BIG_BATCH) {
453        if (i == retries - 1) {
454          int lastRow = -1;
455          for (Result result : res) {
456            int currentRow = Bytes.toInt(result.getRow());
457            for (int row = lastRow+1; row < currentRow; row++) {
458              LOG.error("Row missing: " + row);
459            }
460            lastRow = currentRow;
461          }
462          LOG.error("Last row: " + lastRow);
463          fail("Waited too much time for normal batch replication, " +
464            res.length + " instead of " + NB_ROWS_IN_BIG_BATCH + "; waited=" +
465            (System.currentTimeMillis() - start) + "ms");
466        } else {
467          LOG.info("Only got " + res.length + " rows... retrying");
468          Thread.sleep(SLEEP_TIME);
469        }
470      } else {
471        break;
472      }
473    }
474  }

601  private void runVerifyReplication(String[] args, int expectedGoodRows, int expectedBadRows)
602      throws IOException, InterruptedException, ClassNotFoundException {
603    Job job = new VerifyReplication().createSubmittableJob(new Configuration(conf1), args);
604    if (job == null) {
605      fail("Job wasn't created, see the log");
606    }
607    if (!job.waitForCompletion(true)) {
608      fail("Job failed, see the log");
609    }
610    assertEquals(expectedGoodRows, job.getCounters().
611        findCounter(VerifyReplication.Verifier.Counters.GOODROWS).getValue());
612    assertEquals(expectedBadRows, job.getCounters().
613        findCounter(VerifyReplication.Verifier.Counters.BADROWS).getValue());
614  }

986  public void testEmptyWALRecovery() throws Exception {
987    final int numRs = utility1.getHBaseCluster().getRegionServerThreads().size();
988
989    // for each RS, create an empty wal with same walGroupId
990    final List<Path> emptyWalPaths = new ArrayList<>();
991    long ts = System.currentTimeMillis();
992    for (int i = 0; i < numRs; i++) {
993      HRegionInfo regionInfo =
994          utility1.getHBaseCluster().getRegions(htable1.getName()).get(0).getRegionInfo();
995      WAL wal = utility1.getHBaseCluster().getRegionServer(i).getWAL(regionInfo);
996      Path currentWalPath = AbstractFSWALProvider.getCurrentFileName(wal);
997      String walGroupId = AbstractFSWALProvider.getWALPrefixFromWALName(currentWalPath.getName());
998      Path emptyWalPath = new Path(utility1.getDataTestDir(), walGroupId + "." + ts);
999      utility1.getTestFileSystem().create(emptyWalPath).close();
1000      emptyWalPaths.add(emptyWalPath);
1001    }
1002
1003    // inject our empty wal into the replication queue
1004    for (int i = 0; i < numRs; i++) {
1005      Replication replicationService =
1006          (Replication) utility1.getHBaseCluster().getRegionServer(i).getReplicationSourceService();
1007      replicationService.preLogRoll(null, emptyWalPaths.get(i));
1008      replicationService.postLogRoll(null, emptyWalPaths.get(i));
1009    }
1010
1011    // wait for ReplicationSource to start reading from our empty wal
1012    waitForLogAdvance(numRs, emptyWalPaths, false);
1013
1014    // roll the original wal, which enqueues a new wal behind our empty wal
1015    for (int i = 0; i < numRs; i++) {
1016      HRegionInfo regionInfo =
1017          utility1.getHBaseCluster().getRegions(htable1.getName()).get(0).getRegionInfo();
1018      WAL wal = utility1.getHBaseCluster().getRegionServer(i).getWAL(regionInfo);
1019      wal.rollWriter(true);
1020    }
1021
1022    // ReplicationSource should advance past the empty wal, or else the test will fail
1023    waitForLogAdvance(numRs, emptyWalPaths, true);
1024
1025    // we're now writing to the new wal
1026    // if everything works, the source should've stopped reading from the empty wal, and start
1027    // replicating from the new wal
1028    testSimplePutDelete();
1029  }

1031  /**
1032   * Waits for the ReplicationSource to start reading from the given paths
1033   * @param numRs number of regionservers
1034   * @param emptyWalPaths path for each regionserver
1035   * @param invert if true, waits until ReplicationSource is NOT reading from the given paths
1036   */
1037  private void waitForLogAdvance(final int numRs, final List<Path> emptyWalPaths,
1038      final boolean invert) throws Exception {
1039    Waiter.waitFor(conf1, 10000, new Waiter.Predicate<Exception>() {
1040      @Override
1041      public boolean evaluate() throws Exception {
1042        for (int i = 0; i < numRs; i++) {
1043          Replication replicationService = (Replication) utility1.getHBaseCluster()
1044              .getRegionServer(i).getReplicationSourceService();
1045          for (ReplicationSourceInterface rsi : replicationService.getReplicationManager()
1046              .getSources()) {
1047            ReplicationSource source = (ReplicationSource) rsi;
1048            if (!invert && !emptyWalPaths.get(i).equals(source.getCurrentPath())) {
1049              return false;
1050            }
1051            if (invert && emptyWalPaths.get(i).equals(source.getCurrentPath())) {
1052              return false;
1053            }
1054          }
1055        }
1056        return true;
1057      }
1058    });
1059  }
