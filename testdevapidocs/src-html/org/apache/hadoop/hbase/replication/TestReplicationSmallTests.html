094    // rolling like this makes sure the most recent one gets added to the queue
095    for ( JVMClusterUtil.RegionServerThread r :
096        utility1.getHBaseCluster().getRegionServerThreads()) {

098    }
099    int rowCount = utility1.countRows(tableName);
100    utility1.deleteTableData(tableName);
101    // truncating the table will send one Delete per row to the slave cluster
102    // in an async fashion, which is why we cannot just call deleteTableData on
103    // utility2 since late writes could make it to the slave in some way.
104    // Instead, we truncate the first table and wait for all the Deletes to
105    // make it to the slave.
106    Scan scan = new Scan();
107    int lastCount = 0;
108    for (int i = 0; i < NB_RETRIES; i++) {
109      if (i==NB_RETRIES-1) {
110        fail("Waited too much time for truncate");
111      }
112      ResultScanner scanner = htable2.getScanner(scan);
113      Result[] res = scanner.next(rowCount);
114      scanner.close();
115      if (res.length != 0) {
116        if (res.length < lastCount) {
117          i--; // Don't increment timeout if we make progress
118        }
119        lastCount = res.length;
120        LOG.info("Still got " + res.length + " rows");
121        Thread.sleep(SLEEP_TIME);
122      } else {
123        break;
124      }
125    }

129   * Verify that version and column delete marker types are replicated
130   * correctly.
131   * @throws Exception
132   */
133  @Test(timeout=300000)
134  public void testDeleteTypes() throws Exception {
135    LOG.info("testDeleteTypes");
136    final byte[] v1 = Bytes.toBytes("v1");
137    final byte[] v2 = Bytes.toBytes("v2");
138    final byte[] v3 = Bytes.toBytes("v3");
139    htable1 = utility1.getConnection().getTable(tableName);
140
141    long t = EnvironmentEdgeManager.currentTime();
142    // create three versions for "row"
143    Put put = new Put(row);
144    put.addColumn(famName, row, t, v1);
145    htable1.put(put);
146
147    put = new Put(row);
148    put.addColumn(famName, row, t + 1, v2);
149    htable1.put(put);
150
151    put = new Put(row);
152    put.addColumn(famName, row, t + 2, v3);
153    htable1.put(put);
154
155    Get get = new Get(row);
156    get.setMaxVersions();
157    for (int i = 0; i < NB_RETRIES; i++) {
158      if (i==NB_RETRIES-1) {
159        fail("Waited too much time for put replication");
160      }
161      Result res = htable2.get(get);
162      if (res.size() < 3) {
163        LOG.info("Rows not available");
164        Thread.sleep(SLEEP_TIME);
165      } else {
166        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[0]), v3);
167        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[1]), v2);
168        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[2]), v1);
169        break;
170      }
171    }
172    // place a version delete marker (delete last version)
173    Delete d = new Delete(row);
174    d.addColumn(famName, row, t);
175    htable1.delete(d);
176
177    get = new Get(row);
178    get.setMaxVersions();
179    for (int i = 0; i < NB_RETRIES; i++) {
180      if (i==NB_RETRIES-1) {
181        fail("Waited too much time for put replication");
182      }
183      Result res = htable2.get(get);
184      if (res.size() > 2) {
185        LOG.info("Version not deleted");
186        Thread.sleep(SLEEP_TIME);
187      } else {
188        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[0]), v3);
189        assertArrayEquals(CellUtil.cloneValue(res.rawCells()[1]), v2);
190        break;
191      }
192    }
193
194    // place a column delete marker
195    d = new Delete(row);
196    d.addColumns(famName, row, t+2);
197    htable1.delete(d);
198
199    // now *both* of the remaining version should be deleted
200    // at the replica
201    get = new Get(row);
202    for (int i = 0; i < NB_RETRIES; i++) {
203      if (i==NB_RETRIES-1) {
204        fail("Waited too much time for del replication");
205      }
206      Result res = htable2.get(get);
207      if (res.size() >= 1) {
208        LOG.info("Rows not deleted");
209        Thread.sleep(SLEEP_TIME);
210      } else {
211        break;
212      }
213    }
214  }

282  private void waitForReplication(int expectedRows, int retries) throws IOException, InterruptedException {
283    Scan scan;
284    for (int i = 0; i < retries; i++) {
285      scan = new Scan();
286      if (i== retries -1) {
287        fail("Waited too much time for normal batch replication");
288      }
289      ResultScanner scanner = htable2.getScanner(scan);
290      Result[] res = scanner.next(expectedRows);
291      scanner.close();
292      if (res.length != expectedRows) {
293        LOG.info("Only got " + res.length + " rows");
294        Thread.sleep(SLEEP_TIME);
295      } else {
296        break;
297      }
298    }
299  }

412   * Do a more intense version testSmallBatch, one  that will trigger
413   * wal rolling and other non-trivial code paths
414   * @throws Exception

416  @Test(timeout=300000)
417  public void testLoading() throws Exception {
418    LOG.info("Writing out rows to table1 in testLoading");
419    List<Put> puts = new ArrayList<>(NB_ROWS_IN_BIG_BATCH);
420    for (int i = 0; i < NB_ROWS_IN_BIG_BATCH; i++) {
421      Put put = new Put(Bytes.toBytes(i));
422      put.addColumn(famName, row, row);
423      puts.add(put);
424    }
425    htable1.setWriteBufferSize(1024);
426    // The puts will be iterated through and flushed only when the buffer
427    // size is reached.
428    htable1.put(puts);
429
430    Scan scan = new Scan();
431
432    ResultScanner scanner = htable1.getScanner(scan);
433    Result[] res = scanner.next(NB_ROWS_IN_BIG_BATCH);
434    scanner.close();
435
436    assertEquals(NB_ROWS_IN_BIG_BATCH, res.length);
437
438    LOG.info("Looking in table2 for replicated rows in testLoading");
439    long start = System.currentTimeMillis();
440    // Retry more than NB_RETRIES.  As it was, retries were done in 5 seconds and we'd fail
441    // sometimes.
442    final long retries = NB_RETRIES * 10;
443    for (int i = 0; i < retries; i++) {
444      scan = new Scan();
445      scanner = htable2.getScanner(scan);
446      res = scanner.next(NB_ROWS_IN_BIG_BATCH);
447      scanner.close();
448      if (res.length != NB_ROWS_IN_BIG_BATCH) {
449        if (i == retries - 1) {
450          int lastRow = -1;
451          for (Result result : res) {
452            int currentRow = Bytes.toInt(result.getRow());
453            for (int row = lastRow+1; row < currentRow; row++) {
454              LOG.error("Row missing: " + row);
455            }
456            lastRow = currentRow;
457          }
458          LOG.error("Last row: " + lastRow);
459          fail("Waited too much time for normal batch replication, " +
460            res.length + " instead of " + NB_ROWS_IN_BIG_BATCH + "; waited=" +
461            (System.currentTimeMillis() - start) + "ms");
462        } else {
463          LOG.info("Only got " + res.length + " rows... retrying");
464          Thread.sleep(SLEEP_TIME);
465        }
466      } else {
467        break;
468      }
469    }
470  }

597  private void runVerifyReplication(String[] args, int expectedGoodRows, int expectedBadRows)
598      throws IOException, InterruptedException, ClassNotFoundException {
599    Job job = new VerifyReplication().createSubmittableJob(new Configuration(conf1), args);
600    if (job == null) {
601      fail("Job wasn't created, see the log");
602    }
603    if (!job.waitForCompletion(true)) {
604      fail("Job failed, see the log");
605    }
606    assertEquals(expectedGoodRows, job.getCounters().
607        findCounter(VerifyReplication.Verifier.Counters.GOODROWS).getValue());
608    assertEquals(expectedBadRows, job.getCounters().
609        findCounter(VerifyReplication.Verifier.Counters.BADROWS).getValue());
610  }
