080  private ServerName getServerName(int i) {
081    return ServerName.valueOf("127.0.0.1", 8000 + i, 10000 + i);
082  }

084  @Test
085  public void testReplicator() throws ReplicationException {
086    assertTrue(STORAGE.getListOfReplicators().isEmpty());
087    String queueId = "1";
088    for (int i = 0; i < 10; i++) {
089      STORAGE.addWAL(getServerName(i), queueId, "file" + i);
090    }
091    List<ServerName> replicators = STORAGE.getListOfReplicators();
092    assertEquals(10, replicators.size());
093    for (int i = 0; i < 10; i++) {
094      assertThat(replicators, hasItems(getServerName(i)));
095    }
096    for (int i = 0; i < 5; i++) {
097      STORAGE.removeQueue(getServerName(i), queueId);
098    }
099    for (int i = 0; i < 10; i++) {
100      STORAGE.removeReplicatorIfQueueIsEmpty(getServerName(i));
101    }
102    replicators = STORAGE.getListOfReplicators();
103    assertEquals(5, replicators.size());
104    for (int i = 5; i < 10; i++) {
105      assertThat(replicators, hasItems(getServerName(i)));
106    }
107  }

109  private String getFileName(String base, int i) {
110    return String.format(base + "-%04d", i);
111  }
112
113  @Test
114  public void testAddRemoveLog() throws ReplicationException {
115    ServerName serverName1 = ServerName.valueOf("127.0.0.1", 8000, 10000);
116    assertTrue(STORAGE.getAllQueues(serverName1).isEmpty());
117    String queue1 = "1";
118    String queue2 = "2";
119    for (int i = 0; i < 10; i++) {
120      STORAGE.addWAL(serverName1, queue1, getFileName("file1", i));
121      STORAGE.addWAL(serverName1, queue2, getFileName("file2", i));
122    }
123    List<String> queueIds = STORAGE.getAllQueues(serverName1);
124    assertEquals(2, queueIds.size());
125    assertThat(queueIds, hasItems("1", "2"));
126
127    List<String> wals1 = STORAGE.getWALsInQueue(serverName1, queue1);
128    List<String> wals2 = STORAGE.getWALsInQueue(serverName1, queue2);
129    assertEquals(10, wals1.size());
130    assertEquals(10, wals1.size());
131    for (int i = 0; i < 10; i++) {
132      assertThat(wals1, hasItems(getFileName("file1", i)));
133      assertThat(wals2, hasItems(getFileName("file2", i)));
134    }
135
136    for (int i = 0; i < 10; i++) {
137      assertEquals(0, STORAGE.getWALPosition(serverName1, queue1, getFileName("file1", i)));
138      assertEquals(0, STORAGE.getWALPosition(serverName1, queue2, getFileName("file2", i)));
139      STORAGE.setWALPosition(serverName1, queue1, getFileName("file1", i), (i + 1) * 100);
140      STORAGE.setWALPosition(serverName1, queue2, getFileName("file2", i), (i + 1) * 100 + 10);
141    }
142
143    for (int i = 0; i < 10; i++) {
144      assertEquals((i + 1) * 100,
145        STORAGE.getWALPosition(serverName1, queue1, getFileName("file1", i)));
146      assertEquals((i + 1) * 100 + 10,
147        STORAGE.getWALPosition(serverName1, queue2, getFileName("file2", i)));
148    }
149
150    for (int i = 0; i < 10; i++) {
151      if (i % 2 == 0) {
152        STORAGE.removeWAL(serverName1, queue1, getFileName("file1", i));
153      } else {
154        STORAGE.removeWAL(serverName1, queue2, getFileName("file2", i));
155      }
156    }
157
158    queueIds = STORAGE.getAllQueues(serverName1);
159    assertEquals(2, queueIds.size());
160    assertThat(queueIds, hasItems("1", "2"));
161
162    ServerName serverName2 = ServerName.valueOf("127.0.0.1", 8001, 10001);
163    Pair<String, SortedSet<String>> peer1 = STORAGE.claimQueue(serverName1, "1", serverName2);
164
165    assertEquals("1-" + serverName1.getServerName(), peer1.getFirst());
166    assertEquals(5, peer1.getSecond().size());
167    int i = 1;
168    for (String wal : peer1.getSecond()) {
169      assertEquals(getFileName("file1", i), wal);
170      assertEquals((i + 1) * 100,
171        STORAGE.getWALPosition(serverName2, peer1.getFirst(), getFileName("file1", i)));
172      i += 2;
173    }
174
175    queueIds = STORAGE.getAllQueues(serverName1);
176    assertEquals(1, queueIds.size());
177    assertThat(queueIds, hasItems("2"));
178    wals2 = STORAGE.getWALsInQueue(serverName1, queue2);
179    assertEquals(5, wals2.size());
180    for (i = 0; i < 10; i += 2) {
181      assertThat(wals2, hasItems(getFileName("file2", i)));
182    }
183
184    queueIds = STORAGE.getAllQueues(serverName2);
185    assertEquals(1, queueIds.size());
186    assertThat(queueIds, hasItems(peer1.getFirst()));
187    wals1 = STORAGE.getWALsInQueue(serverName2, peer1.getFirst());
188    assertEquals(5, wals1.size());
189    for (i = 1; i < 10; i += 2) {
190      assertThat(wals1, hasItems(getFileName("file1", i)));
191    }
192
193    Set<String> allWals = STORAGE.getAllWALs();
194    assertEquals(10, allWals.size());
195    for (i = 0; i < 10; i++) {
196      assertThat(allWals, hasItems(i % 2 == 0 ? getFileName("file2", i) : getFileName("file1", i)));
197    }
198  }

200  // For HBASE-12865
201  @Test
202  public void testClaimQueueChangeCversion() throws ReplicationException, KeeperException {
203    ServerName serverName1 = ServerName.valueOf("127.0.0.1", 8000, 10000);
204    STORAGE.addWAL(serverName1, "1", "file");
205
206    int v0 = STORAGE.getQueuesZNodeCversion();
207    ServerName serverName2 = ServerName.valueOf("127.0.0.1", 8001, 10001);
208    STORAGE.claimQueue(serverName1, "1", serverName2);
209    int v1 = STORAGE.getQueuesZNodeCversion();
210    // cversion should increase by 1 since a child node is deleted
211    assertEquals(1, v1 - v0);
212  }
213
214  private ZKReplicationQueueStorage createWithUnstableCversion() throws IOException {
215    return new ZKReplicationQueueStorage(UTIL.getZooKeeperWatcher(), UTIL.getConfiguration()) {
216
217      private int called = 0;
218
219      @Override
220      protected int getQueuesZNodeCversion() throws KeeperException {
221        if (called < 4) {
222          called++;
223        }
224        return called;
225      }
226    };
227  }

229  @Test
230  public void testGetAllWALsCversionChange() throws IOException, ReplicationException {
231    ZKReplicationQueueStorage storage = createWithUnstableCversion();
232    storage.addWAL(getServerName(0), "1", "file");
233    // This should return eventually when cversion stabilizes
234    Set<String> allWals = storage.getAllWALs();
235    assertEquals(1, allWals.size());
236    assertThat(allWals, hasItems("file"));
237  }
