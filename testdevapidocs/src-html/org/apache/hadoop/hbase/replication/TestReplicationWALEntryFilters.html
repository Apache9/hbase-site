079  @Test
080  public void testScopeWALEntryFilter() {
081    WALEntryFilter filter = new ChainWALEntryFilter(new ScopeWALEntryFilter());
082
083    Entry userEntry = createEntry(null, a, b);
084    Entry userEntryA = createEntry(null, a);
085    Entry userEntryB = createEntry(null, b);
086    Entry userEntryEmpty = createEntry(null);
087
088    // no scopes
089    assertEquals(null, filter.filter(userEntry));
090
091    // empty scopes
092    TreeMap<byte[], Integer> scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
093    userEntry = createEntry(scopes, a, b);
094    assertEquals(null, filter.filter(userEntry));
095
096    // different scope
097    scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
098    scopes.put(c, HConstants.REPLICATION_SCOPE_GLOBAL);
099    userEntry = createEntry(scopes, a, b);
100    // all kvs should be filtered
101    assertEquals(userEntryEmpty, filter.filter(userEntry));
102
103    // local scope
104    scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
105    scopes.put(a, HConstants.REPLICATION_SCOPE_LOCAL);
106    userEntry = createEntry(scopes, a, b);
107    assertEquals(userEntryEmpty, filter.filter(userEntry));
108    scopes.put(b, HConstants.REPLICATION_SCOPE_LOCAL);
109    assertEquals(userEntryEmpty, filter.filter(userEntry));
110
111    // only scope a
112    scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
113    scopes.put(a, HConstants.REPLICATION_SCOPE_GLOBAL);
114    userEntry = createEntry(scopes, a, b);
115    assertEquals(userEntryA, filter.filter(userEntry));
116    scopes.put(b, HConstants.REPLICATION_SCOPE_LOCAL);
117    assertEquals(userEntryA, filter.filter(userEntry));
118
119    // only scope b
120    scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
121    scopes.put(b, HConstants.REPLICATION_SCOPE_GLOBAL);
122    userEntry = createEntry(scopes, a, b);
123    assertEquals(userEntryB, filter.filter(userEntry));
124    scopes.put(a, HConstants.REPLICATION_SCOPE_LOCAL);
125    assertEquals(userEntryB, filter.filter(userEntry));
126
127    // scope a and b
128    scopes = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
129    scopes.put(b, HConstants.REPLICATION_SCOPE_GLOBAL);
130    userEntry = createEntry(scopes, a, b);
131    assertEquals(userEntryB, filter.filter(userEntry));
132    scopes.put(a, HConstants.REPLICATION_SCOPE_LOCAL);
133    assertEquals(userEntryB, filter.filter(userEntry));
134  }

136  WALEntryFilter nullFilter = new WALEntryFilter() {
137    @Override
138    public Entry filter(Entry entry) {
139      return null;
140    }

143  WALEntryFilter passFilter = new WALEntryFilter() {

150  @Test
151  public void testChainWALEntryFilter() {
152    Entry userEntry = createEntry(null, a, b, c);
153
154    ChainWALEntryFilter filter = new ChainWALEntryFilter(passFilter);
155    assertEquals(createEntry(null, a,b,c), filter.filter(userEntry));
156
157    filter = new ChainWALEntryFilter(passFilter, passFilter);
158    assertEquals(createEntry(null, a,b,c), filter.filter(userEntry));
159
160    filter = new ChainWALEntryFilter(passFilter, passFilter, passFilter);
161    assertEquals(createEntry(null, a,b,c), filter.filter(userEntry));
162
163    filter = new ChainWALEntryFilter(nullFilter);
164    assertEquals(null, filter.filter(userEntry));
165
166    filter = new ChainWALEntryFilter(nullFilter, passFilter);
167    assertEquals(null, filter.filter(userEntry));
168
169    filter = new ChainWALEntryFilter(passFilter, nullFilter);
170    assertEquals(null, filter.filter(userEntry));
171
172    filter = new ChainWALEntryFilter(nullFilter, passFilter, nullFilter);
173    assertEquals(null, filter.filter(userEntry));
174
175    filter = new ChainWALEntryFilter(nullFilter, nullFilter);
176    assertEquals(null, filter.filter(userEntry));
177
178    // flatten
179    filter =
180        new ChainWALEntryFilter(
181          new ChainWALEntryFilter(passFilter,
182            new ChainWALEntryFilter(passFilter, passFilter),
183          new ChainWALEntryFilter(passFilter),
184          new ChainWALEntryFilter(passFilter)),
185          new ChainWALEntryFilter(passFilter));
186    assertEquals(createEntry(null, a,b,c), filter.filter(userEntry));
187
188
189    filter =
190        new ChainWALEntryFilter(
191          new ChainWALEntryFilter(passFilter,
192            new ChainWALEntryFilter(passFilter,
193              new ChainWALEntryFilter(nullFilter))),
194          new ChainWALEntryFilter(passFilter));
195    assertEquals(null, filter.filter(userEntry));
196  }

198  @Test
199  public void testTableCfWALEntryFilter() {
200    ReplicationPeer peer = mock(ReplicationPeer.class);
201
202    when(peer.getTableCFs()).thenReturn(null);
203    Entry userEntry = createEntry(null, a, b, c);
204    WALEntryFilter filter = new ChainWALEntryFilter(new TableCfWALEntryFilter(peer));
205    assertEquals(createEntry(null, a,b,c), filter.filter(userEntry));
206
207    // empty map
208    userEntry = createEntry(null, a, b, c);
209    Map<TableName, List<String>> tableCfs = new HashMap<TableName, List<String>>();
210    when(peer.getTableCFs()).thenReturn(tableCfs);
211    filter = new ChainWALEntryFilter(new TableCfWALEntryFilter(peer));
212    assertEquals(null, filter.filter(userEntry));
213
214    // table bar
215    userEntry = createEntry(null, a, b, c);
216    tableCfs = new HashMap<TableName, List<String>>();
217    tableCfs.put(TableName.valueOf("bar"), null);
218    when(peer.getTableCFs()).thenReturn(tableCfs);
219    filter = new ChainWALEntryFilter(new TableCfWALEntryFilter(peer));
220    assertEquals(null, filter.filter(userEntry));
221
222    // table foo:a
223    userEntry = createEntry(null, a, b, c);
224    tableCfs = new HashMap<TableName, List<String>>();
225    tableCfs.put(TableName.valueOf("foo"), Lists.newArrayList("a"));
226    when(peer.getTableCFs()).thenReturn(tableCfs);
227    filter = new ChainWALEntryFilter(new TableCfWALEntryFilter(peer));
228    assertEquals(createEntry(null, a), filter.filter(userEntry));
229
230    // table foo:a,c
231    userEntry = createEntry(null, a, b, c, d);
232    tableCfs = new HashMap<TableName, List<String>>();
233    tableCfs.put(TableName.valueOf("foo"), Lists.newArrayList("a", "c"));
234    when(peer.getTableCFs()).thenReturn(tableCfs);
235    filter = new ChainWALEntryFilter(new TableCfWALEntryFilter(peer));
236    assertEquals(createEntry(null, a,c), filter.filter(userEntry));
237  }

239  private Entry createEntry(TreeMap<byte[], Integer> scopes, byte[]... kvs) {
240    WALKey key1 = new WALKey(new byte[] {}, TableName.valueOf("foo"), scopes);
241    WALEdit edit1 = new WALEdit();
242
243    for (byte[] kv : kvs) {
244      edit1.add(new KeyValue(kv, kv, kv));
245    }
246    return new Entry(key1, edit1);
247  }

250  private void assertEquals(Entry e1, Entry e2) {
251    Assert.assertEquals(e1 == null, e2 == null);
252    if (e1 == null) {
253      return;
254    }
255
256    // do not compare WALKeys
257
258    // compare kvs
259    Assert.assertEquals(e1.getEdit() == null, e2.getEdit() == null);
260    if (e1.getEdit() == null) {
261      return;
262    }
263    List<Cell> cells1 = e1.getEdit().getCells();
264    List<Cell> cells2 = e2.getEdit().getCells();
265    Assert.assertEquals(cells1.size(), cells2.size());
266    for (int i = 0; i < cells1.size(); i++) {
267      CellComparator.COMPARATOR.compare(cells1.get(i), cells2.get(i));
268    }
269  }
