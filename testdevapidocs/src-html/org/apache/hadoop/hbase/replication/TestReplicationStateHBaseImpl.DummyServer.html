046@Category({ReplicationTests.class, MediumTests.class})
047public class TestReplicationStateHBaseImpl {
048
049  private static Configuration conf;
050  private static HBaseTestingUtility utility;
051  private static Connection connection;
052  private static ReplicationQueues rqH;
053
054  private final String server1 = ServerName.valueOf("hostname1.example.org", 1234, -1L).toString();
055
056  @BeforeClass
057  public static void setUpBeforeClass() throws Exception {
058    utility = new HBaseTestingUtility();
059    utility.startMiniCluster();
060    conf = utility.getConfiguration();
061    conf.setClass("hbase.region.replica.replication.ReplicationQueuesType",
062        ReplicationQueuesHBaseImpl.class, ReplicationQueues.class);
063    connection = ConnectionFactory.createConnection(conf);
064  }
065
066  @Test
067  public void checkNamingSchema() throws Exception {
068    rqH.init(server1);
069    assertTrue(rqH.isThisOurRegionServer(server1));
070    assertTrue(!rqH.isThisOurRegionServer(server1 + "a"));
071    assertTrue(!rqH.isThisOurRegionServer(null));
072  }
073
074  @Test
075  public void testReplicationStateHBase() {
076    DummyServer ds = new DummyServer(server1);
077    try {
078      rqH = ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, ds, null));
079      rqH.init(server1);
080      // Check that the proper System Tables have been generated
081      Table replicationTable = connection.getTable(
082          ReplicationQueuesHBaseImpl.REPLICATION_TABLE_NAME);
083      assertTrue(replicationTable.getName().isSystemTable());
084
085    } catch (Exception e) {
086      e.printStackTrace();
087      fail("testReplicationStateHBaseConstruction received an Exception");
088    }
089    try {
090      // Test adding in WAL files
091      assertEquals(0, rqH.getAllQueues().size());
092      rqH.addLog("Queue1", "WALLogFile1.1");
093      assertEquals(1, rqH.getAllQueues().size());
094      rqH.addLog("Queue1", "WALLogFile1.2");
095      rqH.addLog("Queue1", "WALLogFile1.3");
096      rqH.addLog("Queue1", "WALLogFile1.4");
097      rqH.addLog("Queue2", "WALLogFile2.1");
098      rqH.addLog("Queue3", "WALLogFile3.1");
099      assertEquals(3, rqH.getAllQueues().size());
100      assertEquals(4, rqH.getLogsInQueue("Queue1").size());
101      assertEquals(1, rqH.getLogsInQueue("Queue2").size());
102      assertEquals(1, rqH.getLogsInQueue("Queue3").size());
103      // Make sure that abortCount is still 0
104      assertEquals(0, ds.getAbortCount());
105      // Make sure that getting a log from a non-existent queue triggers an abort
106      assertNull(rqH.getLogsInQueue("Queue4"));
107      assertEquals(1, ds.getAbortCount());
108    } catch (ReplicationException e) {
109      e.printStackTrace();
110      fail("testAddLog received a ReplicationException");
111    }
112    try {
113
114      // Test updating the log positions
115      assertEquals(0L, rqH.getLogPosition("Queue1", "WALLogFile1.1"));
116      rqH.setLogPosition("Queue1", "WALLogFile1.1", 123L);
117      assertEquals(123L, rqH.getLogPosition("Queue1", "WALLogFile1.1"));
118      rqH.setLogPosition("Queue1", "WALLogFile1.1", 123456789L);
119      assertEquals(123456789L, rqH.getLogPosition("Queue1", "WALLogFile1.1"));
120      rqH.setLogPosition("Queue2", "WALLogFile2.1", 242L);
121      assertEquals(242L, rqH.getLogPosition("Queue2", "WALLogFile2.1"));
122      rqH.setLogPosition("Queue3", "WALLogFile3.1", 243L);
123      assertEquals(243L, rqH.getLogPosition("Queue3", "WALLogFile3.1"));
124
125      // Test that setting log positions in non-existing logs will cause an abort
126      assertEquals(1, ds.getAbortCount());
127      rqH.setLogPosition("NotHereQueue", "WALLogFile3.1", 243L);
128      assertEquals(2, ds.getAbortCount());
129      rqH.setLogPosition("NotHereQueue", "NotHereFile", 243L);
130      assertEquals(3, ds.getAbortCount());
131      rqH.setLogPosition("Queue1", "NotHereFile", 243l);
132      assertEquals(4, ds.getAbortCount());
133
134      // Test reading log positions for non-existent queues and WAL's
135      try {
136        rqH.getLogPosition("Queue1", "NotHereWAL");
137        fail("Replication queue should have thrown a ReplicationException for reading from a " +
138            "non-existent WAL");
139      } catch (ReplicationException e) {
140      }
141      try {
142        rqH.getLogPosition("NotHereQueue", "NotHereWAL");
143        fail("Replication queue should have thrown a ReplicationException for reading from a " +
144            "non-existent queue");
145      } catch (ReplicationException e) {
146      }
147      // Test removing logs
148      rqH.removeLog("Queue1", "WALLogFile1.1");
149      assertEquals(3, rqH.getLogsInQueue("Queue1").size());
150      // Test removing queues
151      rqH.removeQueue("Queue2");
152      assertEquals(2, rqH.getAllQueues().size());
153      assertNull(rqH.getLogsInQueue("Queue2"));
154      // Test that getting logs from a non-existent queue aborts
155      assertEquals(5, ds.getAbortCount());
156      // Test removing all queues for a Region Server
157      rqH.removeAllQueues();
158      assertEquals(0, rqH.getAllQueues().size());
159      assertNull(rqH.getLogsInQueue("Queue1"));
160      // Test that getting logs from a non-existent queue aborts
161      assertEquals(6, ds.getAbortCount());
162    } catch (ReplicationException e) {
163      e.printStackTrace();
164      fail("testAddLog received a ReplicationException");
165    }
166  }
