122  @Test
123  public void testSingleReplicationQueuesHBaseImpl() {
124    try {
125      // Test adding in WAL files
126      assertEquals(0, rq1.getAllQueues().size());
127      rq1.addLog("Queue1", "WALLogFile1.1");
128      assertEquals(1, rq1.getAllQueues().size());
129      rq1.addLog("Queue1", "WALLogFile1.2");
130      rq1.addLog("Queue1", "WALLogFile1.3");
131      rq1.addLog("Queue1", "WALLogFile1.4");
132      rq1.addLog("Queue2", "WALLogFile2.1");
133      rq1.addLog("Queue3", "WALLogFile3.1");
134      assertEquals(3, rq1.getAllQueues().size());
135      assertEquals(4, rq1.getLogsInQueue("Queue1").size());
136      assertEquals(1, rq1.getLogsInQueue("Queue2").size());
137      assertEquals(1, rq1.getLogsInQueue("Queue3").size());
138      // Make sure that abortCount is still 0
139      assertEquals(0, ds1.getAbortCount());
140      // Make sure that getting a log from a non-existent queue triggers an abort
141      assertNull(rq1.getLogsInQueue("Queue4"));
142      assertEquals(1, ds1.getAbortCount());
143    } catch (ReplicationException e) {
144      e.printStackTrace();
145      fail("testAddLog received a ReplicationException");
146    }
147    try {
148
149      // Test updating the log positions
150      assertEquals(0L, rq1.getLogPosition("Queue1", "WALLogFile1.1"));
151      rq1.setLogPosition("Queue1", "WALLogFile1.1", 123L);
152      assertEquals(123L, rq1.getLogPosition("Queue1", "WALLogFile1.1"));
153      rq1.setLogPosition("Queue1", "WALLogFile1.1", 123456789L);
154      assertEquals(123456789L, rq1.getLogPosition("Queue1", "WALLogFile1.1"));
155      rq1.setLogPosition("Queue2", "WALLogFile2.1", 242L);
156      assertEquals(242L, rq1.getLogPosition("Queue2", "WALLogFile2.1"));
157      rq1.setLogPosition("Queue3", "WALLogFile3.1", 243L);
158      assertEquals(243L, rq1.getLogPosition("Queue3", "WALLogFile3.1"));
159
160      // Test that setting log positions in non-existing logs will cause an abort
161      assertEquals(1, ds1.getAbortCount());
162      rq1.setLogPosition("NotHereQueue", "WALLogFile3.1", 243L);
163      assertEquals(2, ds1.getAbortCount());
164      rq1.setLogPosition("NotHereQueue", "NotHereFile", 243L);
165      assertEquals(3, ds1.getAbortCount());
166      rq1.setLogPosition("Queue1", "NotHereFile", 243l);
167      assertEquals(4, ds1.getAbortCount());
168
169      // Test reading log positions for non-existent queues and WAL's
170      try {
171        rq1.getLogPosition("Queue1", "NotHereWAL");
172        fail("Replication queue should have thrown a ReplicationException for reading from a " +
173          "non-existent WAL");
174      } catch (ReplicationException e) {
175      }
176      try {
177        rq1.getLogPosition("NotHereQueue", "NotHereWAL");
178        fail("Replication queue should have thrown a ReplicationException for reading from a " +
179          "non-existent queue");
180      } catch (ReplicationException e) {
181      }
182      // Test removing logs
183      rq1.removeLog("Queue1", "WALLogFile1.1");
184      assertEquals(3, rq1.getLogsInQueue("Queue1").size());
185      // Test removing queues
186      rq1.removeQueue("Queue2");
187      assertEquals(2, rq1.getAllQueues().size());
188      assertNull(rq1.getLogsInQueue("Queue2"));
189      // Test that getting logs from a non-existent queue aborts
190      assertEquals(5, ds1.getAbortCount());
191      // Test removing all queues for a Region Server
192      rq1.removeAllQueues();
193      assertEquals(0, rq1.getAllQueues().size());
194      assertNull(rq1.getLogsInQueue("Queue1"));
195      // Test that getting logs from a non-existent queue aborts
196      assertEquals(6, ds1.getAbortCount());
197      // Test removing a non-existent queue does not cause an abort. This is because we can
198      // attempt to remove a queue that has no corresponding Replication Table row (if we never
199      // registered a WAL for it)
200      rq1.removeQueue("NotHereQueue");
201      assertEquals(6, ds1.getAbortCount());
202    } catch (ReplicationException e) {
203      e.printStackTrace();
204      fail("testAddLog received a ReplicationException");
205    }
206  }

208  @Test
209  public void TestMultipleReplicationQueuesHBaseImpl () {
210    try {
211      rp.registerPeer("Queue1", new ReplicationPeerConfig().setClusterKey("localhost:2818:/bogus1"));
212      rp.registerPeer("Queue2", new ReplicationPeerConfig().setClusterKey("localhost:2818:/bogus2"));
213      rp.registerPeer("Queue3", new ReplicationPeerConfig().setClusterKey("localhost:2818:/bogus3"));
214    } catch (ReplicationException e) {
215      fail("Failed to add peers to ReplicationPeers");
216    }
217    try {
218      // Test adding in WAL files
219      rq1.addLog("Queue1", "WALLogFile1.1");
220      rq1.addLog("Queue1", "WALLogFile1.2");
221      rq1.addLog("Queue1", "WALLogFile1.3");
222      rq1.addLog("Queue1", "WALLogFile1.4");
223      rq1.addLog("Queue2", "WALLogFile2.1");
224      rq1.addLog("Queue3", "WALLogFile3.1");
225      rq2.addLog("Queue1", "WALLogFile1.1");
226      rq2.addLog("Queue1", "WALLogFile1.2");
227      rq2.addLog("Queue2", "WALLogFile2.1");
228      rq3.addLog("Queue1", "WALLogFile1.1");
229      // Test adding logs to replication queues
230      assertEquals(3, rq1.getAllQueues().size());
231      assertEquals(2, rq2.getAllQueues().size());
232      assertEquals(1, rq3.getAllQueues().size());
233      assertEquals(4, rq1.getLogsInQueue("Queue1").size());
234      assertEquals(1, rq1.getLogsInQueue("Queue2").size());
235      assertEquals(1, rq1.getLogsInQueue("Queue3").size());
236      assertEquals(2, rq2.getLogsInQueue("Queue1").size());
237      assertEquals(1, rq2.getLogsInQueue("Queue2").size());
238      assertEquals(1, rq3.getLogsInQueue("Queue1").size());
239    } catch (ReplicationException e) {
240      e.printStackTrace();
241      fail("testAddLogs received a ReplicationException");
242    }
243    try {
244      // Test setting and reading offset in queues
245      rq1.setLogPosition("Queue1", "WALLogFile1.1", 1l);
246      rq1.setLogPosition("Queue1", "WALLogFile1.2", 2l);
247      rq1.setLogPosition("Queue1", "WALLogFile1.3", 3l);
248      rq1.setLogPosition("Queue2", "WALLogFile2.1", 4l);
249      rq1.setLogPosition("Queue2", "WALLogFile2.2", 5l);
250      rq1.setLogPosition("Queue3", "WALLogFile3.1", 6l);
251      rq2.setLogPosition("Queue1", "WALLogFile1.1", 7l);
252      rq2.setLogPosition("Queue2", "WALLogFile2.1", 8l);
253      rq3.setLogPosition("Queue1", "WALLogFile1.1", 9l);
254      assertEquals(1l, rq1.getLogPosition("Queue1", "WALLogFile1.1"));
255      assertEquals(2l, rq1.getLogPosition("Queue1", "WALLogFile1.2"));
256      assertEquals(4l, rq1.getLogPosition("Queue2", "WALLogFile2.1"));
257      assertEquals(6l, rq1.getLogPosition("Queue3", "WALLogFile3.1"));
258      assertEquals(7l, rq2.getLogPosition("Queue1", "WALLogFile1.1"));
259      assertEquals(8l, rq2.getLogPosition("Queue2", "WALLogFile2.1"));
260      assertEquals(9l, rq3.getLogPosition("Queue1", "WALLogFile1.1"));
261      assertEquals(rq1.getListOfReplicators().size(), 3);
262      assertEquals(rq2.getListOfReplicators().size(), 3);
263      assertEquals(rq3.getListOfReplicators().size(), 3);
264    } catch (ReplicationException e) {
265      fail("testAddLogs threw a ReplicationException");
266    }
267    try {
268      // Test claiming queues
269      List<String> claimedQueuesFromRq2 = rq1.getUnClaimedQueueIds(server2);
270      // Check to make sure that list of peers with outstanding queues is decremented by one
271      // after claimQueues
272      // Check to make sure that we claimed the proper number of queues
273      assertEquals(2, claimedQueuesFromRq2.size());
274      assertTrue(claimedQueuesFromRq2.contains("Queue1-" + server2));
275      assertTrue(claimedQueuesFromRq2.contains("Queue2-" + server2));
276      assertEquals(2, rq1.claimQueue(server2, "Queue1-" + server2).getSecond().size());
277      assertEquals(1, rq1.claimQueue(server2, "Queue2-" + server2).getSecond().size());
278      rq1.removeReplicatorIfQueueIsEmpty(server2);
279      assertEquals(rq1.getListOfReplicators().size(), 2);
280      assertEquals(rq2.getListOfReplicators().size(), 2);
281      assertEquals(rq3.getListOfReplicators().size(), 2);
282      assertEquals(5, rq1.getAllQueues().size());
283      // Check that all the logs in the other queue were claimed
284      assertEquals(2, rq1.getLogsInQueue("Queue1-" + server2).size());
285      assertEquals(1, rq1.getLogsInQueue("Queue2-" + server2).size());
286      // Check that the offsets of the claimed queues are the same
287      assertEquals(7l, rq1.getLogPosition("Queue1-" + server2, "WALLogFile1.1"));
288      assertEquals(8l, rq1.getLogPosition("Queue2-" + server2, "WALLogFile2.1"));
289      // Check that the queues were properly removed from rq2
290      assertEquals(0, rq2.getAllQueues().size());
291      assertNull(rq2.getLogsInQueue("Queue1"));
292      assertNull(rq2.getLogsInQueue("Queue2"));
293      // Check that non-existent peer queues are not claimed
294      rq1.addLog("UnclaimableQueue", "WALLogFile1.1");
295      rq1.addLog("UnclaimableQueue", "WALLogFile1.2");
296      assertEquals(6, rq1.getAllQueues().size());
297      List<String> claimedQueuesFromRq1 = rq3.getUnClaimedQueueIds(server1);
298      for(String queue : claimedQueuesFromRq1) {
299        rq3.claimQueue(server1, queue);
300      }
301      rq3.removeReplicatorIfQueueIsEmpty(server1);
302      assertEquals(rq1.getListOfReplicators().size(), 1);
303      assertEquals(rq2.getListOfReplicators().size(), 1);
304      assertEquals(rq3.getListOfReplicators().size(), 1);
305      // Note that we do not pick up the queue: UnclaimableQueue which was not registered in
306      // Replication Peers
307      assertEquals(6, rq3.getAllQueues().size());
308      // Test claiming non-existing queues
309      List<String> noQueues = rq3.getUnClaimedQueueIds("NotARealServer");
310      assertNull(noQueues);
311      assertEquals(6, rq3.getAllQueues().size());
312      // Test claiming own queues
313      noQueues = rq3.getUnClaimedQueueIds(server3);
314      Assert.assertNull(noQueues);
315      assertEquals(6, rq3.getAllQueues().size());
316      // Check that rq3 still remain on list of replicators
317      assertEquals(1, rq3.getListOfReplicators().size());
318    } catch (ReplicationException e) {
319      fail("testClaimQueue threw a ReplicationException");
320    }
321  }

323  @Test
324  public void TestReplicationQueuesClient() throws Exception{
325
326    // Test ReplicationQueuesClient log tracking
327    rq1.addLog("Queue1", "WALLogFile1.1");
328    assertEquals(1, rqc.getLogsInQueue(server1, "Queue1").size());
329    rq1.removeLog("Queue1", "WALLogFile1.1");
330    assertEquals(0, rqc.getLogsInQueue(server1, "Queue1").size());
331    rq2.addLog("Queue2", "WALLogFile2.1");
332    rq2.addLog("Queue2", "WALLogFile2.2");
333    assertEquals(2, rqc.getLogsInQueue(server2, "Queue2").size());
334    rq3.addLog("Queue1", "WALLogFile1.1");
335    rq3.addLog("Queue3", "WALLogFile3.1");
336    rq3.addLog("Queue3", "WALLogFile3.2");
337
338    // Test ReplicationQueueClient log tracking for faulty cases
339    assertEquals(0, ds0.getAbortCount());
340    assertNull(rqc.getLogsInQueue("NotHereServer", "NotHereQueue"));
341    assertNull(rqc.getLogsInQueue(server1, "NotHereQueue"));
342    assertNull(rqc.getLogsInQueue("NotHereServer", "WALLogFile1.1"));
343    assertEquals(3, ds0.getAbortCount());
344    // Test ReplicationQueueClient replicators
345    List<String> replicators = rqc.getListOfReplicators();
346    assertEquals(3, replicators.size());
347    assertTrue(replicators.contains(server1));
348    assertTrue(replicators.contains(server2));
349    rq1.removeQueue("Queue1");
350    assertEquals(2, rqc.getListOfReplicators().size());
351
352    // Test ReplicationQueuesClient queue tracking
353    assertEquals(0, rqc.getAllQueues(server1).size());
354    rq1.addLog("Queue2", "WALLogFile2.1");
355    rq1.addLog("Queue3", "WALLogFile3.1");
356    assertEquals(2, rqc.getAllQueues(server1).size());
357    rq1.removeAllQueues();
358    assertEquals(0, rqc.getAllQueues(server1).size());
359
360    // Test ReplicationQueuesClient queue tracking for faulty cases
361    assertEquals(0, rqc.getAllQueues("NotHereServer").size());
362
363    // Test ReplicationQueuesClient get all WAL's
364    assertEquals(5 , rqc.getAllWALs().size());
365    rq3.removeLog("Queue1", "WALLogFile1.1");
366    assertEquals(4, rqc.getAllWALs().size());
367    rq3.removeAllQueues();
368    assertEquals(2, rqc.getAllWALs().size());
369    rq2.removeAllQueues();
370    assertEquals(0, rqc.getAllWALs().size());
371  }
