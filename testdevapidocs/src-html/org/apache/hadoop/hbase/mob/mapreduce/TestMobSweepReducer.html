068public class TestMobSweepReducer {

070  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
071  private final static String tableName = "testSweepReducer";
072  private final static String row = "row";
073  private final static String family = "family";
074  private final static String qf = "qf";
075  private static BufferedMutator table;
076  private static Admin admin;

091  @Before
092  public void setUp() throws Exception {
093    HTableDescriptor desc = new HTableDescriptor(tableName);
094    HColumnDescriptor hcd = new HColumnDescriptor(family);
095    hcd.setMobEnabled(true);
096    hcd.setMobThreshold(3L);
097    hcd.setMaxVersions(4);
098    desc.addFamily(hcd);
099
100    admin = TEST_UTIL.getHBaseAdmin();
101    admin.createTable(desc);
102    table = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration())
103            .getBufferedMutator(TableName.valueOf(tableName));
104  }

108    admin.disableTable(TableName.valueOf(tableName));

110    admin.close();
111  }

113  private List<String> getKeyFromSequenceFile(FileSystem fs, Path path,
114                                              Configuration conf) throws Exception {
115    List<String> list = new ArrayList<String>();
116    SequenceFile.Reader reader = new SequenceFile.Reader(conf, SequenceFile.Reader.file(path));
117
118    String next = (String) reader.next((String) null);
119    while (next != null) {
120      list.add(next);
121      next = (String) reader.next((String) null);
122    }
123    reader.close();
124    return list;
125  }

127  @Test
128  public void testRun() throws Exception {
129
130    TableName tn = TableName.valueOf(tableName);
131    byte[] mobValueBytes = new byte[100];
132
133    //get the path where mob files lie in
134    Path mobFamilyPath = MobUtils.getMobFamilyPath(TEST_UTIL.getConfiguration(), tn, family);
135
136    Put put = new Put(Bytes.toBytes(row));
137    put.addColumn(Bytes.toBytes(family), Bytes.toBytes(qf), 1, mobValueBytes);
138    Put put2 = new Put(Bytes.toBytes(row + "ignore"));
139    put2.addColumn(Bytes.toBytes(family), Bytes.toBytes(qf), 1, mobValueBytes);
140    table.mutate(put);
141    table.mutate(put2);
142    table.flush();
143    admin.flush(tn);
144
145    FileStatus[] fileStatuses = TEST_UTIL.getTestFileSystem().listStatus(mobFamilyPath);
146    //check the generation of a mob file
147    assertEquals(1, fileStatuses.length);
148
149    String mobFile1 = fileStatuses[0].getPath().getName();
150
151    Configuration configuration = new Configuration(TEST_UTIL.getConfiguration());
152    configuration.setFloat(MobConstants.MOB_SWEEP_TOOL_COMPACTION_RATIO, 0.6f);
153    configuration.setStrings(TableInputFormat.INPUT_TABLE, tableName);
154    configuration.setStrings(TableInputFormat.SCAN_COLUMN_FAMILY, family);
155    configuration.setStrings(SweepJob.WORKING_VISITED_DIR_KEY, "jobWorkingNamesDir");
156    configuration.setStrings(SweepJob.WORKING_FILES_DIR_KEY, "compactionFileDir");
157    configuration.setStrings(CommonConfigurationKeys.IO_SERIALIZATIONS_KEY,
158            JavaSerialization.class.getName());
159    configuration.set(SweepJob.WORKING_VISITED_DIR_KEY, "compactionVisitedDir");
160    configuration.setLong(MobConstants.MOB_SWEEP_TOOL_COMPACTION_START_DATE,
161        System.currentTimeMillis() + 24 * 3600 * 1000);
162
163    ZooKeeperWatcher zkw = new ZooKeeperWatcher(configuration, "1", new DummyMobAbortable());
164    TableName lockName = MobUtils.getTableLockName(tn);
165    String znode = ZKUtil.joinZNode(zkw.znodePaths.tableLockZNode, lockName.getNameAsString());
166    configuration.set(SweepJob.SWEEP_JOB_ID, "1");
167    configuration.set(SweepJob.SWEEP_JOB_TABLE_NODE, znode);
168    ServerName serverName = SweepJob.getCurrentServerName(configuration);
169    configuration.set(SweepJob.SWEEP_JOB_SERVERNAME, serverName.toString());
170
171    TableLockManager tableLockManager = TableLockManager.createTableLockManager(configuration, zkw,
172        serverName);
173    TableLock lock = tableLockManager.writeLock(lockName, "Run sweep tool");
174    lock.acquire();
175    try {
176      // use the same counter when mocking
177      Counter counter = new GenericCounter();
178      Reducer<Text, KeyValue, Writable, Writable>.Context ctx = mock(Reducer.Context.class);
179      when(ctx.getConfiguration()).thenReturn(configuration);
180      when(ctx.getCounter(Matchers.any(SweepCounter.class))).thenReturn(counter);
181      when(ctx.nextKey()).thenReturn(true).thenReturn(false);
182      when(ctx.getCurrentKey()).thenReturn(new Text(mobFile1));
183
184      byte[] refBytes = Bytes.toBytes(mobFile1);
185      long valueLength = refBytes.length;
186      byte[] newValue = Bytes.add(Bytes.toBytes(valueLength), refBytes);
187      KeyValue kv2 = new KeyValue(Bytes.toBytes(row), Bytes.toBytes(family), Bytes.toBytes(qf), 1,
188        KeyValue.Type.Put, newValue);
189      List<KeyValue> list = new ArrayList<KeyValue>();
190      list.add(kv2);
191
192      when(ctx.getValues()).thenReturn(list);
193
194      SweepReducer reducer = new SweepReducer();
195      reducer.run(ctx);
196    } finally {
197      lock.release();
198    }
199    FileStatus[] filsStatuses2 = TEST_UTIL.getTestFileSystem().listStatus(mobFamilyPath);
200    String mobFile2 = filsStatuses2[0].getPath().getName();
201    //new mob file is generated, old one has been archived
202    assertEquals(1, filsStatuses2.length);
203    assertEquals(false, mobFile2.equalsIgnoreCase(mobFile1));
204
205    //test sequence file
206    String workingPath = configuration.get(SweepJob.WORKING_VISITED_DIR_KEY);
207    FileStatus[] statuses = TEST_UTIL.getTestFileSystem().listStatus(new Path(workingPath));
208    Set<String> files = new TreeSet<String>();
209    for (FileStatus st : statuses) {
210      files.addAll(getKeyFromSequenceFile(TEST_UTIL.getTestFileSystem(),
211              st.getPath(), configuration));
212    }
213    assertEquals(1, files.size());
214    assertEquals(true, files.contains(mobFile1));
215  }
