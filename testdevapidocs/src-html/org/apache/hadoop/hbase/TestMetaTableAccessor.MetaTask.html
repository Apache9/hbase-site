088  /**
089   * Does {@link MetaTableAccessor#getRegion(Connection, byte[])} and a write
090   * against hbase:meta while its hosted server is restarted to prove our retrying
091   * works.
092   * @throws IOException
093   * @throws InterruptedException

095  @Test public void testRetrying()
096  throws IOException, InterruptedException {
097    final TableName name =
098        TableName.valueOf("testRetrying");
099    LOG.info("Started " + name);
100    Table t = UTIL.createMultiRegionTable(name, HConstants.CATALOG_FAMILY);
101    int regionCount = -1;
102    try (RegionLocator r = UTIL.getConnection().getRegionLocator(name)) {
103      regionCount = r.getStartKeys().length;
104    }
105    // Test it works getting a region from just made user table.
106    final List<HRegionInfo> regions =
107      testGettingTableRegions(connection, name, regionCount);
108    MetaTask reader = new MetaTask(connection, "reader") {
109      @Override
110      void metaTask() throws Throwable {
111        testGetRegion(connection, regions.get(0));
112        LOG.info("Read " + regions.get(0).getEncodedName());
113      }
114    };
115    MetaTask writer = new MetaTask(connection, "writer") {
116      @Override
117      void metaTask() throws Throwable {
118        MetaTableAccessor.addRegionToMeta(connection, regions.get(0));
119        LOG.info("Wrote " + regions.get(0).getEncodedName());
120      }
121    };
122    reader.start();
123    writer.start();
124
125    // We're gonna check how it takes. If it takes too long, we will consider
126    //  it as a fail. We can't put that in the @Test tag as we want to close
127    //  the threads nicely
128    final long timeOut = 180000;
129    long startTime = System.currentTimeMillis();
130
131    try {
132      // Make sure reader and writer are working.
133      assertTrue(reader.isProgressing());
134      assertTrue(writer.isProgressing());
135
136      // Kill server hosting meta -- twice  . See if our reader/writer ride over the
137      // meta moves.  They'll need to retry.
138      for (int i = 0; i < 2; i++) {
139        LOG.info("Restart=" + i);
140        UTIL.ensureSomeRegionServersAvailable(2);
141        int index = -1;
142        do {
143          index = UTIL.getMiniHBaseCluster().getServerWithMeta();
144        } while (index == -1 &&
145          startTime + timeOut < System.currentTimeMillis());
146
147        if (index != -1){
148          UTIL.getMiniHBaseCluster().abortRegionServer(index);
149          UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
150        }
151      }
152
153      assertTrue("reader: " + reader.toString(), reader.isProgressing());
154      assertTrue("writer: " + writer.toString(), writer.isProgressing());
155    } catch (IOException e) {
156      throw e;
157    } finally {
158      reader.stop = true;
159      writer.stop = true;
160      reader.join();
161      writer.join();
162      t.close();
163    }
164    long exeTime = System.currentTimeMillis() - startTime;
165    assertTrue("Timeout: test took " + exeTime / 1000 + " sec", exeTime < timeOut);
166  }

168  /**
169   * Thread that runs a MetaTableAccessor task until asked stop.
170   */
171  abstract static class MetaTask extends Thread {
172    boolean stop = false;
173    int count = 0;
174    Throwable t = null;
175    final Connection connection;
176
177    MetaTask(final Connection connection, final String name) {
178      super(name);
179      this.connection = connection;
180    }
181
182    @Override
183    public void run() {
184      try {
185        while(!this.stop) {
186          LOG.info("Before " + this.getName()+ ", count=" + this.count);
187          metaTask();
188          this.count += 1;
189          LOG.info("After " + this.getName() + ", count=" + this.count);
190          Thread.sleep(100);
191        }
192      } catch (Throwable t) {
193        LOG.info(this.getName() + " failed", t);
194        this.t = t;
195      }
196    }
197
198    boolean isProgressing() throws InterruptedException {
199      int currentCount = this.count;
200      while(currentCount == this.count) {
201        if (!isAlive()) return false;
202        if (this.t != null) return false;
203        Thread.sleep(10);
204      }
205      return true;
206    }
207
208    @Override
209    public String toString() {
210      return "count=" + this.count + ", t=" +
211        (this.t == null? "null": this.t.toString());
212    }
213
214    abstract void metaTask() throws Throwable;
215  }

364      // check whether the primary is still there

371      // check whether the primary is still there
372      assertMetaLocation(meta, primary.getRegionName(), serverName0, seqNum0, 0, true);
373      // check whether the replica 1 is still there
374      assertMetaLocation(meta, primary.getRegionName(), serverName1, seqNum1, 1, true);

376      assertMetaLocation(meta, primary.getRegionName(), serverName100, seqNum100, 100, true);
377    } finally {
378      meta.close();
379    }

586      // write the serverName column with a big current time, but set the masters time as even
587      // bigger. When region merge deletes the rows for regionA and regionB, the serverName columns
588      // should not be seen by the following get
589      long serverNameTime = EnvironmentEdgeManager.currentTime()   + 100000000;
590      long masterSystemTime = EnvironmentEdgeManager.currentTime() + 123456789;
591
592      // write the serverName columns
593      MetaTableAccessor.updateRegionLocation(connection, regionInfoA, sn, 1, serverNameTime);
594
595      // assert that we have the serverName column with expected ts
596      Get get = new Get(mergedRegionInfo.getRegionName());
597      Result result = meta.get(get);
598      Cell serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
599          MetaTableAccessor.getServerColumn(0));
600      assertNotNull(serverCell);
601      assertEquals(serverNameTime, serverCell.getTimestamp());
602
603      // now merge the regions, effectively deleting the rows for region a and b.
604      MetaTableAccessor.mergeRegions(connection, mergedRegionInfo,
605        regionInfoA, regionInfoB, sn, 1, masterSystemTime);
606
607      result = meta.get(get);
608      serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
609          MetaTableAccessor.getServerColumn(0));
610      Cell startCodeCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
611        MetaTableAccessor.getStartCodeColumn(0));
612      Cell seqNumCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
613        MetaTableAccessor.getSeqNumColumn(0));
614      assertNull(serverCell);
615      assertNull(startCodeCell);
616      assertNull(seqNumCell);
617    } finally {
618      meta.close();
619    }
620  }
