141  /**
142   * Verfy the result of getBalanceSplits() method.
143   * The result are groups of files, used as input list for the "export" mappers.
144   * All the groups should have similar amount of data.
145   *
146   * The input list is a pair of file path and length.
147   * The getBalanceSplits() function sort it by length,
148   * and assign to each group a file, going back and forth through the groups.
149   */
150  @Test
151  public void testBalanceSplit() throws Exception {
152    // Create a list of files
153    List<Pair<SnapshotFileInfo, Long>> files = new ArrayList<Pair<SnapshotFileInfo, Long>>();
154    for (long i = 0; i <= 20; i++) {
155      SnapshotFileInfo fileInfo = SnapshotFileInfo.newBuilder()
156        .setType(SnapshotFileInfo.Type.HFILE)
157        .setHfile("file-" + i)
158        .build();
159      files.add(new Pair<SnapshotFileInfo, Long>(fileInfo, i));
160    }
161
162    // Create 5 groups (total size 210)
163    //    group 0: 20, 11, 10,  1 (total size: 42)
164    //    group 1: 19, 12,  9,  2 (total size: 42)
165    //    group 2: 18, 13,  8,  3 (total size: 42)
166    //    group 3: 17, 12,  7,  4 (total size: 42)
167    //    group 4: 16, 11,  6,  5 (total size: 42)
168    List<List<Pair<SnapshotFileInfo, Long>>> splits = ExportSnapshot.getBalancedSplits(files, 5);
169    assertEquals(5, splits.size());
170
171    String[] split0 = new String[] {"file-20", "file-11", "file-10", "file-1", "file-0"};
172    verifyBalanceSplit(splits.get(0), split0, 42);
173    String[] split1 = new String[] {"file-19", "file-12", "file-9",  "file-2"};
174    verifyBalanceSplit(splits.get(1), split1, 42);
175    String[] split2 = new String[] {"file-18", "file-13", "file-8",  "file-3"};
176    verifyBalanceSplit(splits.get(2), split2, 42);
177    String[] split3 = new String[] {"file-17", "file-14", "file-7",  "file-4"};
178    verifyBalanceSplit(splits.get(3), split3, 42);
179    String[] split4 = new String[] {"file-16", "file-15", "file-6",  "file-5"};
180    verifyBalanceSplit(splits.get(4), split4, 42);
181  }
182
183  private void verifyBalanceSplit(final List<Pair<SnapshotFileInfo, Long>> split,
184      final String[] expected, final long expectedSize) {
185    assertEquals(expected.length, split.size());
186    long totalSize = 0;
187    for (int i = 0; i < expected.length; ++i) {
188      Pair<SnapshotFileInfo, Long> fileInfo = split.get(i);
189      assertEquals(expected[i], fileInfo.getFirst().getHfile());
190      totalSize += fileInfo.getSecond();
191    }
192    assertEquals(expectedSize, totalSize);
193  }

272  /**
273   * Test ExportSnapshot
274   */
275  private void testExportFileSystemState(final TableName tableName, final byte[] snapshotName,
276      final byte[] targetName, int filesExpected, Path copyDir, boolean overwrite)
277      throws Exception {
278    URI hdfsUri = FileSystem.get(TEST_UTIL.getConfiguration()).getUri();
279    FileSystem fs = FileSystem.get(copyDir.toUri(), new Configuration());
280    copyDir = copyDir.makeQualified(fs);
281
282    List<String> opts = new ArrayList<String>();
283    opts.add("-snapshot");
284    opts.add(Bytes.toString(snapshotName));
285    opts.add("-copy-to");
286    opts.add(copyDir.toString());
287    if (targetName != snapshotName) {
288      opts.add("-target");
289      opts.add(Bytes.toString(targetName));
290    }
291    if (overwrite) opts.add("-overwrite");
292
293    // Export Snapshot
294    int res = ExportSnapshot.innerMain(TEST_UTIL.getConfiguration(),
295        opts.toArray(new String[opts.size()]));
296    assertEquals(0, res);
297
298    // Verify File-System state
299    FileStatus[] rootFiles = fs.listStatus(copyDir);
300    assertEquals(filesExpected > 0 ? 2 : 1, rootFiles.length);
301    for (FileStatus fileStatus: rootFiles) {
302      String name = fileStatus.getPath().getName();
303      assertTrue(fileStatus.isDirectory());
304      assertTrue(name.equals(HConstants.SNAPSHOT_DIR_NAME) ||
305                 name.equals(HConstants.HFILE_ARCHIVE_DIRECTORY));
306    }
307
308    // compare the snapshot metadata and verify the hfiles
309    final FileSystem hdfs = FileSystem.get(hdfsUri, TEST_UTIL.getConfiguration());
310    final Path snapshotDir = new Path(HConstants.SNAPSHOT_DIR_NAME, Bytes.toString(snapshotName));
311    final Path targetDir = new Path(HConstants.SNAPSHOT_DIR_NAME, Bytes.toString(targetName));
312    verifySnapshotDir(hdfs, new Path(TEST_UTIL.getDefaultRootDirPath(), snapshotDir),
313        fs, new Path(copyDir, targetDir));
314    Set<String> snapshotFiles = verifySnapshot(fs, copyDir, tableName, Bytes.toString(targetName));
315    assertEquals(filesExpected, snapshotFiles.size());
316  }

334  /*
335   * Execute the ExportSnapshot job injecting failures
336   */
337  private int runExportAndInjectFailures(final byte[] snapshotName, boolean retry)
338      throws Exception {
339    Path copyDir = getLocalDestinationDir();
340    URI hdfsUri = FileSystem.get(TEST_UTIL.getConfiguration()).getUri();
341    FileSystem fs = FileSystem.get(copyDir.toUri(), new Configuration());
342    copyDir = copyDir.makeQualified(fs);
343
344    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
345    conf.setBoolean(ExportSnapshot.CONF_TEST_FAILURE, true);
346    conf.setBoolean(ExportSnapshot.CONF_TEST_RETRY, retry);
347    if (!retry) {
348      conf.setInt("mapreduce.map.maxattempts", 3);
349    }
350    // Export Snapshot
351    Path sourceDir = TEST_UTIL.getHBaseCluster().getMaster().getMasterFileSystem().getRootDir();
352    int res = ExportSnapshot.innerMain(conf, new String[] {
353      "-snapshot", Bytes.toString(snapshotName),
354      "-copy-from", sourceDir.toString(),
355      "-copy-to", copyDir.toString()
356    });
357    return res;
358  }

372  /*
373   * Verify if the files exists
374   */
375  private Set<String> verifySnapshot(final FileSystem fs, final Path rootDir,
376      final TableName tableName, final String snapshotName) throws IOException {
377    final Path exportedSnapshot = new Path(rootDir,
378      new Path(HConstants.SNAPSHOT_DIR_NAME, snapshotName));
379    final Set<String> snapshotFiles = new HashSet<String>();
380    final Path exportedArchive = new Path(rootDir, HConstants.HFILE_ARCHIVE_DIRECTORY);
381    SnapshotReferenceUtil.visitReferencedFiles(TEST_UTIL.getConfiguration(), fs, exportedSnapshot,
382          new SnapshotReferenceUtil.SnapshotVisitor() {
383        @Override
384        public void storeFile(final HRegionInfo regionInfo, final String family,
385            final SnapshotRegionManifest.StoreFile storeFile) throws IOException {
386          if (bypassRegion(regionInfo))
387            return;
388
389          String hfile = storeFile.getName();
390          snapshotFiles.add(hfile);
391          if (storeFile.hasReference()) {
392            // Nothing to do here, we have already the reference embedded
393          } else {
394            verifyNonEmptyFile(new Path(exportedArchive,
395              new Path(FSUtils.getTableDir(new Path("./"), tableName),
396                  new Path(regionInfo.getEncodedName(), new Path(family, hfile)))));
397          }
398        }
399
400        private void verifyNonEmptyFile(final Path path) throws IOException {
401          assertTrue(path + " should exists", fs.exists(path));
402          assertTrue(path + " should not be empty", fs.getFileStatus(path).getLen() > 0);
403        }
404    });
405
406    // Verify Snapshot description
407    SnapshotDescription desc = SnapshotDescriptionUtils.readSnapshotInfo(fs, exportedSnapshot);
408    assertTrue(desc.getName().equals(snapshotName));
409    assertTrue(desc.getTable().equals(tableName.getNameAsString()));
410    return snapshotFiles;
411  }

413  private Set<String> listFiles(final FileSystem fs, final Path root, final Path dir)
414      throws IOException {
415    Set<String> files = new HashSet<String>();
416    int rootPrefix = root.toString().length();
417    FileStatus[] list = FSUtils.listStatus(fs, dir);
418    if (list != null) {
419      for (FileStatus fstat: list) {
420        LOG.debug(fstat.getPath());
421        if (fstat.isDirectory()) {
422          files.addAll(listFiles(fs, root, fstat.getPath()));
423        } else {
424          files.add(fstat.getPath().toString().substring(rootPrefix));
425        }
426      }
427    }
428    return files;
429  }
