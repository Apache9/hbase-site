018package org.apache.hadoop.hbase.snapshot;
019
020import static org.junit.Assert.fail;
021
022import java.io.IOException;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.fs.FSDataOutputStream;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.fs.Path;
030import org.apache.hadoop.hbase.HBaseTestingUtility;
031import org.apache.hadoop.hbase.HColumnDescriptor;
032import org.apache.hadoop.hbase.HRegionInfo;
033import org.apache.hadoop.hbase.TableName;
034import org.apache.hadoop.hbase.shaded.com.google.protobuf.UnsafeByteOperations;
035import org.apache.hadoop.hbase.shaded.protobuf.ProtobufUtil;
036import org.apache.hadoop.hbase.shaded.protobuf.generated.HBaseProtos.SnapshotDescription;
037import org.apache.hadoop.hbase.shaded.protobuf.generated.SnapshotProtos.SnapshotDataManifest;
038import org.apache.hadoop.hbase.shaded.protobuf.generated.SnapshotProtos.SnapshotRegionManifest;
039import org.apache.hadoop.hbase.testclassification.MasterTests;
040import org.apache.hadoop.hbase.testclassification.SmallTests;
041import org.apache.hadoop.hbase.util.Bytes;
042import org.junit.After;
043import org.junit.Before;
044import org.junit.Test;
045import org.junit.experimental.categories.Category;
046
047@Category({MasterTests.class, SmallTests.class})
048public class TestSnapshotManifest {
049  private final Log LOG = LogFactory.getLog(getClass());
050
051  private static final String TABLE_NAME_STR = "testSnapshotManifest";
052  private static final TableName TABLE_NAME = TableName.valueOf(TABLE_NAME_STR);
053  private static final int TEST_NUM_REGIONS = 16000;
054
055  private static HBaseTestingUtility TEST_UTIL;
056  private Configuration conf;
057  private FileSystem fs;
058  private Path rootDir;
059  private Path snapshotDir;
060  private SnapshotDescription snapshotDesc;
061
062  @Before
063  public void setup() throws Exception {
064    TEST_UTIL = HBaseTestingUtility.createLocalHTU();
065
066    rootDir = TEST_UTIL.getDataTestDir(TABLE_NAME_STR);
067    fs = TEST_UTIL.getTestFileSystem();
068    conf = TEST_UTIL.getConfiguration();
069
070    SnapshotTestingUtils.SnapshotMock snapshotMock =
071      new SnapshotTestingUtils.SnapshotMock(conf, fs, rootDir);
072    SnapshotTestingUtils.SnapshotMock.SnapshotBuilder builder =
073      snapshotMock.createSnapshotV2("snapshot", TABLE_NAME_STR, 0);
074    snapshotDir = builder.commit();
075    snapshotDesc = builder.getSnapshotDescription();
076
077    SnapshotDataManifest.Builder dataManifestBuilder =
078      SnapshotDataManifest.newBuilder();
079    byte[] startKey = null;
080    byte[] stopKey = null;
081    for (int i = 1; i <= TEST_NUM_REGIONS; i++) {
082      stopKey = Bytes.toBytes(String.format("%016d", i));
083      HRegionInfo regionInfo = new HRegionInfo(TABLE_NAME, startKey, stopKey, false);
084      SnapshotRegionManifest.Builder dataRegionManifestBuilder =
085        SnapshotRegionManifest.newBuilder();
086
087      for (HColumnDescriptor hcd: builder.getTableDescriptor().getFamilies()) {
088        SnapshotRegionManifest.FamilyFiles.Builder family =
089            SnapshotRegionManifest.FamilyFiles.newBuilder();
090        family.setFamilyName(UnsafeByteOperations.unsafeWrap(hcd.getName()));
091        for (int j = 0; j < 100; ++j) {
092          SnapshotRegionManifest.StoreFile.Builder sfManifest =
093            SnapshotRegionManifest.StoreFile.newBuilder();
094          sfManifest.setName(String.format("%032d", i));
095          sfManifest.setFileSize((1 + i) * (1 + i) * 1024);
096          family.addStoreFiles(sfManifest.build());
097        }
098        dataRegionManifestBuilder.addFamilyFiles(family.build());
099      }
100
101      dataRegionManifestBuilder.setRegionInfo(HRegionInfo.convert(regionInfo));
102      dataManifestBuilder.addRegionManifests(dataRegionManifestBuilder.build());
103
104      startKey = stopKey;
105    }
106
107    dataManifestBuilder
108        .setTableSchema(ProtobufUtil.convertToTableSchema(builder.getTableDescriptor()));
109
110    SnapshotDataManifest dataManifest = dataManifestBuilder.build();
111    writeDataManifest(dataManifest);
112  }
113
114  @After
115  public void tearDown() throws Exception {
116    fs.delete(rootDir,true);
117  }
118
119  @Test
120  public void testReadSnapshotManifest() throws IOException {
121    try {
122      SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);
123      fail("fail to test snapshot manifest because message size is too small.");
124    } catch (CorruptedSnapshotException cse) {
125      try {
126        conf.setInt(SnapshotManifest.SNAPSHOT_MANIFEST_SIZE_LIMIT_CONF_KEY, 128 * 1024 * 1024);
127        SnapshotManifest.open(conf, fs, snapshotDir, snapshotDesc);
128        LOG.info("open snapshot manifest succeed.");
129      } catch (CorruptedSnapshotException cse2) {
130        fail("fail to take snapshot because Manifest proto-message too large.");
131      }
132    }
133  }
134
135  private void writeDataManifest(final SnapshotDataManifest manifest)
136      throws IOException {
137    FSDataOutputStream stream = fs.create(new Path(snapshotDir, SnapshotManifest.DATA_MANIFEST_NAME));
138    try {
139      manifest.writeTo(stream);
140    } finally {
141      stream.close();
142    }
143  }
144}
