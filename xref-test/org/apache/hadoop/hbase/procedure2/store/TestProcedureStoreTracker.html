40    @Test
41    public void testSeqInsertAndDelete() {
42      ProcedureStoreTracker tracker = new ProcedureStoreTracker();
43      assertTrue(tracker.isEmpty());
44
45      final int MIN_PROC = 1;
46      final int MAX_PROC = 1 << 10;
47
48      // sequential insert
49      for (int i = MIN_PROC; i < MAX_PROC; ++i) {
50        tracker.insert(i);
51
52        // All the proc that we inserted should not be deleted
53        for (int j = MIN_PROC; j <= i; ++j) {
54          assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
55        }
56        // All the proc that are not yet inserted should be result as deleted
57        for (int j = i + 1; j < MAX_PROC; ++j) {
58          assertTrue(tracker.isDeleted(j) != ProcedureStoreTracker.DeleteState.NO);
59        }
60      }
61
62      // sequential delete
63      for (int i = MIN_PROC; i < MAX_PROC; ++i) {
64        tracker.delete(i);
65
66        // All the proc that we deleted should be deleted
67        for (int j = MIN_PROC; j <= i; ++j) {
68          assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(j));
69        }
70        // All the proc that are not yet deleted should be result as not deleted
71        for (int j = i + 1; j < MAX_PROC; ++j) {
72          assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(j));
73        }
74      }
75      assertTrue(tracker.isEmpty());
76    }

78    @Test
79    public void testPartialTracker() {
80      ProcedureStoreTracker tracker = new ProcedureStoreTracker();
81      tracker.setPartialFlag(true);
82
83      // nothing in the tracker, the state is unknown
84      assertTrue(tracker.isEmpty());
85      assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(1));
86      assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
87
88      // Mark 1 as deleted, now that is a known state
89      tracker.setDeleted(1, true);
90      tracker.dump();
91      assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
92      assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
93      assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(579));
94
95      // Mark 579 as non-deleted, now that is a known state
96      tracker.setDeleted(579, false);
97      assertEquals(ProcedureStoreTracker.DeleteState.YES, tracker.isDeleted(1));
98      assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(2));
99      assertEquals(ProcedureStoreTracker.DeleteState.NO, tracker.isDeleted(579));
100     assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(577));
101     assertEquals(ProcedureStoreTracker.DeleteState.MAYBE, tracker.isDeleted(580));
102
103     tracker.setDeleted(579, true);
104     tracker.setPartialFlag(false);
105     assertTrue(tracker.isEmpty());
106   }

110     long[][] procIds = new long[][] {{4, 7}, {1024, 1027}, {8192, 8194}};
111     long[][] checkIds = new long[][] {{2, 8}, {1023, 1025}, {8193, 8191}};
112
113     ProcedureStoreTracker tracker = new ProcedureStoreTracker();
114     for (int i = 0; i < procIds.length; ++i) {

116       tracker.insert(seq[0]);
117       tracker.insert(seq[1]);
118     }
119
120     for (int i = 0; i < procIds.length; ++i) {
121       long[] check = checkIds[i];
122       long[] seq = procIds[i];
123       assertTrue(tracker.isTracking(seq[0], seq[1]));
124       assertTrue(tracker.isTracking(check[0], check[1]));
125       tracker.delete(seq[0]);
126       tracker.delete(seq[1]);
127       assertFalse(tracker.isTracking(seq[0], seq[1]));
128       assertFalse(tracker.isTracking(check[0], check[1]));
129     }
130
131     assertTrue(tracker.isEmpty());

251   /**
252    * @param active list of active proc ids. To mark them as non-deleted, since by default a proc
253    *               id is always marked deleted.
254    */
255   ProcedureStoreTracker buildTracker(long[] active, long[] updated, long[] deleted) {
256     ProcedureStoreTracker tracker = new ProcedureStoreTracker();
257     for (long i : active) {
258       tracker.insert(i);
259     }
260     tracker.resetUpdates();
261     for (long i : updated) {
262       tracker.update(i);
263     }
264     for (long i : deleted) {
265       tracker.delete(i);
266     }
267     return tracker;
268   }

270   /**
271    * @param active list of active proc ids. To mark them as non-deleted, since by default a proc
272    *               id is always marked deleted.
273    */
274   BitSetNode buildBitSetNode(long[] active, long[] updated, long[] deleted) {
275     BitSetNode bitSetNode = new BitSetNode(0L, false);
276     for (long i : active) {
277       bitSetNode.update(i);
278     }
279     bitSetNode.resetUpdates();
280     for (long i : updated) {
281       bitSetNode.update(i);
282     }
283     for (long i : deleted) {
284       bitSetNode.delete(i);
285     }
286     return bitSetNode;
287   }

289   @Test
290   public void testBitSetNodeSubtract() {
291     // 1 not updated in n2, nothing to subtract
292     BitSetNode n1 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{ });
293     BitSetNode n2 = buildBitSetNode(new long[]{ 1L }, new long[]{}, new long[]{});
294     assertFalse(n1.subtract(n2));
295
296     // 1 updated in n2, and not deleted in n1, should subtract.
297     n1 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{});
298     n2 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{});
299     assertTrue(n1.subtract(n2));
300
301     // 1 updated in n2, but deleted in n1, should not subtract
302     n1 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{ 1L });
303     n2 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{});
304     assertFalse(n1.subtract(n2));
305
306     // 1 updated in n2, but not deleted in n1, should subtract.
307     n1 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{});
308     n2 = buildBitSetNode(new long[]{ 1L }, new long[]{ 1L }, new long[]{ 1L });
309     assertTrue(n1.subtract(n2));
310
311     // all four cases together.
312     n1 = buildBitSetNode(new long[]{ 0L, 10L, 20L, 30L  }, new long[]{ 0L, 10L, 20L, 30L  },
313         new long[]{ 20L });
314     n2 = buildBitSetNode(new long[]{ 0L, 10L, 20L, 30L  }, new long[]{ 0L, 20L, 30L },
315         new long[]{ 0L });
316     assertTrue(n1.subtract(n2));
317   }
