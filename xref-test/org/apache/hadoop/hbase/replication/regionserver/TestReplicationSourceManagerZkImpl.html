63    @Test
64    public void testNodeFailoverDeadServerParsing() throws Exception {
65      conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
66      final Server server = new DummyServer("ec2-54-234-230-108.compute-1.amazonaws.com");
67      ReplicationQueues repQueues =
68        ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, server,
69          server.getZooKeeper()));
70      repQueues.init(server.getServerName().toString());
71      // populate some znodes in the peer znode
72      files.add("log1");
73      files.add("log2");
74      for (String file : files) {
75        repQueues.addLog("1", file);
76      }
77
78      // create 3 DummyServers
79      Server s1 = new DummyServer("ip-10-8-101-114.ec2.internal");
80      Server s2 = new DummyServer("ec2-107-20-52-47.compute-1.amazonaws.com");
81      Server s3 = new DummyServer("ec2-23-20-187-167.compute-1.amazonaws.com");
82
83      // simulate three servers fail sequentially
84      ReplicationQueues rq1 =
85        ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s1.getConfiguration(), s1,
86          s1.getZooKeeper()));
87      rq1.init(s1.getServerName().toString());
88      String serverName = server.getServerName().getServerName();
89      List<String> unclaimed = rq1.getUnClaimedQueueIds(serverName);
90      rq1.claimQueue(serverName, unclaimed.get(0)).getSecond();
91      rq1.removeReplicatorIfQueueIsEmpty(unclaimed.get(0));
92      ReplicationQueues rq2 =
93        ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s2.getConfiguration(), s2,
94          s2.getZooKeeper()));
95      rq2.init(s2.getServerName().toString());
96      serverName = s1.getServerName().getServerName();
97      unclaimed = rq2.getUnClaimedQueueIds(serverName);
98      rq2.claimQueue(serverName, unclaimed.get(0)).getSecond();
99      rq2.removeReplicatorIfQueueIsEmpty(unclaimed.get(0));
100     ReplicationQueues rq3 =
101       ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s3.getConfiguration(), s3,
102         s3.getZooKeeper()));
103     rq3.init(s3.getServerName().toString());
104     serverName = s2.getServerName().getServerName();
105     unclaimed = rq3.getUnClaimedQueueIds(serverName);
106     String queue3 = rq3.claimQueue(serverName, unclaimed.get(0)).getFirst();
107     rq3.removeReplicatorIfQueueIsEmpty(unclaimed.get(0));
108     ReplicationQueueInfo replicationQueueInfo = new ReplicationQueueInfo(queue3);
109     List<String> result = replicationQueueInfo.getDeadRegionServers();
110     // verify
111     assertTrue(result.contains(server.getServerName().getServerName()));
112     assertTrue(result.contains(s1.getServerName().getServerName()));
113     assertTrue(result.contains(s2.getServerName().getServerName()));
114
115     server.stop("");
116   }

118   @Test
119   public void testFailoverDeadServerCversionChange() throws Exception {
120     conf.setBoolean(HConstants.ZOOKEEPER_USEMULTI, true);
121     final Server s0 = new DummyServer("cversion-change0.example.org");
122     ReplicationQueues repQueues =
123       ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(conf, s0,
124         s0.getZooKeeper()));
125     repQueues.init(s0.getServerName().toString());
126     // populate some znodes in the peer znode
127     files.add("log1");
128     files.add("log2");
129     for (String file : files) {
130       repQueues.addLog("1", file);
131     }
132     // simulate queue transfer
133     Server s1 = new DummyServer("cversion-change1.example.org");
134     ReplicationQueues rq1 =
135       ReplicationFactory.getReplicationQueues(new ReplicationQueuesArguments(s1.getConfiguration(), s1,
136         s1.getZooKeeper()));
137     rq1.init(s1.getServerName().toString());
138
139     ReplicationQueuesClientZKImpl client =
140       (ReplicationQueuesClientZKImpl)ReplicationFactory.getReplicationQueuesClient(
141         new ReplicationQueuesClientArguments(s1.getConfiguration(), s1, s1.getZooKeeper()));
142
143     int v0 = client.getQueuesZNodeCversion();
144     List<String> queues = rq1.getUnClaimedQueueIds(s0.getServerName().getServerName());
145     for(String queue : queues) {
146       rq1.claimQueue(s0.getServerName().getServerName(), queue);
147     }
148     rq1.removeReplicatorIfQueueIsEmpty(s0.getServerName().getServerName());
149     int v1 = client.getQueuesZNodeCversion();
150     // cversion should increase by 1 since a child node is deleted
151     assertEquals(v0 + 1, v1);
152
153     s0.stop("");
154   }
