204       final int minVersions, final int versions, final int ttl, KeepDeletedCells keepDeleted) {
205     HTableDescriptor htd = new HTableDescriptor(TableName.valueOf(name));
206     for (byte[] cfName : new byte[][]{ fam1, fam2, fam3 }) {

208           .setMinVersions(minVersions)

212           .setTimeToLive(ttl)
213       );
214     }

279   /**
280    * Add content to region <code>r</code> on the passed column
281    * <code>column</code>.
282    * Adds data of the from 'aaa', 'aab', etc where key and value are the same.
283    * @return count of what we added.
284    * @throws IOException
285    */
286   public static long addContent(final Table updater,
287                                    final String columnFamily, 
288                                    final String column,
289       final byte [] startKeyBytes, final byte [] endKey, final long ts)
290   throws IOException {
291     long count = 0;
292     // Add rows of three characters.  The first character starts with the
293     // 'a' character and runs up to 'z'.  Per first character, we run the
294     // second character over same range.  And same for the third so rows
295     // (and values) look like this: 'aaa', 'aab', 'aac', etc.
296     char secondCharStart = (char)startKeyBytes[1];
297     char thirdCharStart = (char)startKeyBytes[2];
298     EXIT: for (char c = (char)startKeyBytes[0]; c <= LAST_CHAR; c++) {
299       for (char d = secondCharStart; d <= LAST_CHAR; d++) {
300         for (char e = thirdCharStart; e <= LAST_CHAR; e++) {
301           byte [] t = new byte [] {(byte)c, (byte)d, (byte)e};
302           if (endKey != null && endKey.length > 0
303               && Bytes.compareTo(endKey, t) <= 0) {
304             break EXIT;
305           }
306           try {
307             Put put;
308             if(ts != -1) {
309               put = new Put(t, ts);
310             } else {
311               put = new Put(t);
312             }
313             try {
314               StringBuilder sb = new StringBuilder();
315               if (column != null && column.contains(":")) {
316                 sb.append(column);
317               } else {
318                 if (columnFamily != null) {
319                   sb.append(columnFamily);
320                   if (!columnFamily.endsWith(":")) {
321                     sb.append(":");
322                   }
323                   if (column != null) {
324                     sb.append(column);
325                   }
326                 }
327               }
328               byte[][] split =
329                 KeyValue.parseColumn(Bytes.toBytes(sb.toString()));
330               if(split.length == 1) {
331                 byte[] qualifier = new byte[0];
332                 put.addColumn(split[0], qualifier, t);
333               } else {
334                 put.addColumn(split[0], split[1], t);
335               }
336               put.setDurability(Durability.SKIP_WAL);
337               updater.put(put);
338               count++;
339             } catch (RuntimeException ex) {
340               ex.printStackTrace();
341               throw ex;
342             } catch (IOException ex) {
343               ex.printStackTrace();
344               throw ex;
345             }
346           } catch (RuntimeException ex) {
347             ex.printStackTrace();
348             throw ex;
349           } catch (IOException ex) {
350             ex.printStackTrace();
351             throw ex;
352           }
353         }
354         // Set start character back to FIRST_CHAR after we've done first loop.
355         thirdCharStart = FIRST_CHAR;
356       }
357       secondCharStart = FIRST_CHAR;
358     }
359     return count;
360   }

362   protected void assertResultEquals(final HRegion region, final byte [] row,
363       final byte [] family, final byte [] qualifier, final long timestamp,
364       final byte [] value)
365     throws IOException {
366       Get get = new Get(row);
367       get.setTimeStamp(timestamp);
368       Result res = region.get(get);
369       NavigableMap<byte[], NavigableMap<byte[], NavigableMap<Long, byte[]>>> map =
370         res.getMap();
371       byte [] res_value = map.get(family).get(qualifier).get(timestamp);
372 
373       if (value == null) {
374         assertEquals(Bytes.toString(family) + " " + Bytes.toString(qualifier) +
375             " at timestamp " + timestamp, null, res_value);
376       } else {
377         if (res_value == null) {
378           fail(Bytes.toString(family) + " " + Bytes.toString(qualifier) +
379               " at timestamp " + timestamp + "\" was expected to be \"" +
380               Bytes.toStringBinary(value) + " but was null");
381         }
382         if (res_value != null) {
383           assertEquals(Bytes.toString(family) + " " + Bytes.toString(qualifier) +
384               " at timestamp " +
385               timestamp, value, new String(res_value));
386         }
387       }
388     }

390   /**
391    * Common method to close down a MiniDFSCluster and the associated file system
392    *
393    * @param cluster
394    */
395   public static void shutdownDfs(MiniDFSCluster cluster) {
396     if (cluster != null) {
397       LOG.info("Shutting down Mini DFS ");
398       try {
399         cluster.shutdown();
400       } catch (Exception e) {
401         /// Can get a java.lang.reflect.UndeclaredThrowableException thrown
402         // here because of an InterruptedException. Don't let exceptions in
403         // here be cause of test failure.
404       }
405       try {
406         FileSystem fs = cluster.getFileSystem();
407         if (fs != null) {
408           LOG.info("Shutting down FileSystem");
409           fs.close();
410         }
411         FileSystem.closeAll();
412       } catch (IOException e) {
413         LOG.error("error closing file system", e);
414       }
415     }
416   }
