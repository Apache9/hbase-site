85    @Test
86    public void testLoadWithAck() throws Exception {
87      MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
88      HRegionServer regionServer = cluster.getRegionServer(0);
89      String rsName = regionServer.getServerName().getHostname();
90      int port = regionServer.getServerName().getPort();
91      int noRegions = regionServer.getNumberOfOnlineRegions();
92      String rs = rsName + ":" + Integer.toString(port);
93      RegionMoverBuilder rmBuilder = new RegionMoverBuilder(rs).ack(true).maxthreads(8);
94      RegionMover rm = rmBuilder.build();
95      rm.setConf(TEST_UTIL.getConfiguration());
96      LOG.info("Unloading " + rs);
97      rm.unload();
98      assertEquals(0, regionServer.getNumberOfOnlineRegions());
99      LOG.info("Successfully Unloaded\nNow Loading");
100     rm.load();
101     assertEquals(noRegions, regionServer.getNumberOfOnlineRegions());
102   }

170   /**
171    * To test that we successfully exclude a server from the unloading process We test for the number
172    * of regions on Excluded server and also test that regions are unloaded successfully
173    * @throws Exception
174    */
175   @Test
176   public void testExclude() throws Exception {
177     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
178     File excludeFile = new File(TEST_UTIL.getDataTestDir().toUri().getPath(), "exclude_file");
179     FileWriter fos = new FileWriter(excludeFile);
180     HRegionServer excludeServer = cluster.getRegionServer(1);
181     String excludeHostname = excludeServer.getServerName().getHostname();
182     int excludeServerPort = excludeServer.getServerName().getPort();
183     int regionsExcludeServer = excludeServer.getNumberOfOnlineRegions();
184     String excludeServerName = excludeHostname + ":" + Integer.toString(excludeServerPort);
185     fos.write(excludeServerName);
186     fos.close();
187     HRegionServer regionServer = cluster.getRegionServer(0);
188     String rsName = regionServer.getServerName().getHostname();
189     int port = regionServer.getServerName().getPort();
190     String rs = rsName + ":" + Integer.toString(port);
191     RegionMoverBuilder rmBuilder =
192         new RegionMoverBuilder(rs).ack(true).excludeFile(excludeFile.getCanonicalPath());
193     RegionMover rm = rmBuilder.build();
194     rm.setConf(TEST_UTIL.getConfiguration());
195     rm.unload();
196     LOG.info("Unloading " + rs);
197     assertEquals(0, regionServer.getNumberOfOnlineRegions());
198     assertEquals(regionsExcludeServer, cluster.getRegionServer(1).getNumberOfOnlineRegions());
199     LOG.info("Before:" + regionsExcludeServer + " After:"
200         + cluster.getRegionServer(1).getNumberOfOnlineRegions());
201   }
