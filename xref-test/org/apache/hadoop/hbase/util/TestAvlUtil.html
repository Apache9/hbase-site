46    @Test
47    public void testAvlTreeCrud() {
48      final int MAX_KEY = 99999999;
49      final int NELEM = 10000;
50
51      final TreeMap<Integer, Object> treeMap = new TreeMap<Integer, Object>();
52      TestAvlNode root = null;
53
54      final Random rand = new Random();
55      for (int i = 0; i < NELEM; ++i) {
56        int key = rand.nextInt(MAX_KEY);
57        if (AvlTree.get(root, key, KEY_COMPARATOR) != null) {
58          i--;
59          continue;
60        }
61        root = AvlTree.insert(root, new TestAvlNode(key));
62        treeMap.put(key, null);
63        for (Integer keyX: treeMap.keySet()) {
64          TestAvlNode node = AvlTree.get(root, keyX, KEY_COMPARATOR);
65          assertNotNull(node);
66          assertEquals(keyX.intValue(), node.getKey());
67        }
68      }
69
70      for (int i = 0; i < NELEM; ++i) {
71        int key = rand.nextInt(MAX_KEY);
72        TestAvlNode node = AvlTree.get(root, key, KEY_COMPARATOR);
73        if (!treeMap.containsKey(key)) {
74          assert node == null;
75          continue;
76        }
77        treeMap.remove(key);
78        assertEquals(key, node.getKey());
79        root = AvlTree.remove(root, key, KEY_COMPARATOR);
80        for (Integer keyX: treeMap.keySet()) {
81          node = AvlTree.get(root, keyX, KEY_COMPARATOR);
82          assertNotNull(node);
83          assertEquals(keyX.intValue(), node.getKey());
84        }
85      }

88    @Test
89    public void testAvlTreeVisitor() {
90      final int MIN_KEY = 0;
91      final int MAX_KEY = 50;
92
93      TestAvlNode root = null;
94      for (int i = MAX_KEY; i >= MIN_KEY; --i) {
95        root = AvlTree.insert(root, new TestAvlNode(i));
96      }
97
98      AvlTree.visit(root, new AvlNodeVisitor<TestAvlNode>() {
99        private int prevKey = -1;
100       public boolean visitNode(TestAvlNode node) {
101         assertEquals(prevKey, node.getKey() - 1);
102         assertTrue(node.getKey() >= MIN_KEY);
103         assertTrue(node.getKey() <= MAX_KEY);
104         prevKey = node.getKey();
105         return node.getKey() <= MAX_KEY;
106       }
107     });

131   @Test
132   public void testAvlTreeIterSeekTo() {
133     final int MIN_KEY = 1;
134     final int MAX_KEY = 50;
135
136     TestAvlNode root = null;
137     for (int i = MIN_KEY; i < MAX_KEY; i += 2) {
138       root = AvlTree.insert(root, new TestAvlNode(i));
139     }
140
141     for (int i = MIN_KEY - 1; i < MAX_KEY + 1; ++i) {
142       AvlTreeIterator<TestAvlNode> iter = new AvlTreeIterator<TestAvlNode>(root, i, KEY_COMPARATOR);
143       if (i < MAX_KEY) {
144         assertTrue(iter.hasNext());
145       } else {
146         // searching for something greater than the last node
147         assertFalse(iter.hasNext());
148         break;
149       }
150
151       TestAvlNode node = iter.next();
152       assertEquals((i % 2 == 0) ? i + 1 : i, node.getKey());
153
154       long prevKey = node.getKey();
155       while (iter.hasNext()) {
156         node = iter.next();
157         assertTrue(node.getKey() > prevKey);
158         prevKey = node.getKey();
159       }
160     }

163   @Test
164   public void testAvlIterableListCrud() {
165     final int NITEMS = 10;
166     TestLinkedAvlNode prependHead = null;
167     TestLinkedAvlNode appendHead = null;
168     // prepend()/append()
169     for (int i = 0; i <= NITEMS; ++i) {
170       TestLinkedAvlNode pNode = new TestLinkedAvlNode(i);
171       assertFalse(AvlIterableList.isLinked(pNode));
172       prependHead = AvlIterableList.prepend(prependHead, pNode);
173       assertTrue(AvlIterableList.isLinked(pNode));
174
175       TestLinkedAvlNode aNode = new TestLinkedAvlNode(i);
176       assertFalse(AvlIterableList.isLinked(aNode));
177       appendHead = AvlIterableList.append(appendHead, aNode);
178       assertTrue(AvlIterableList.isLinked(aNode));
179     }
180     // readNext()
181     TestLinkedAvlNode pNode = prependHead;
182     TestLinkedAvlNode aNode = appendHead;
183     for (int i = 0; i <= NITEMS; ++i) {
184       assertEquals(NITEMS - i, pNode.getKey());
185       pNode = AvlIterableList.readNext(pNode);
186
187       assertEquals(i, aNode.getKey());
188       aNode = AvlIterableList.readNext(aNode);
189     }
190     // readPrev()
191     pNode = AvlIterableList.readPrev(prependHead);
192     aNode = AvlIterableList.readPrev(appendHead);
193     for (int i = 0; i <= NITEMS; ++i) {
194       assertEquals(i, pNode.getKey());
195       pNode = AvlIterableList.readPrev(pNode);
196
197       assertEquals(NITEMS - i, aNode.getKey());
198       aNode = AvlIterableList.readPrev(aNode);
199     }
200     // appendList()
201     TestLinkedAvlNode node = AvlIterableList.appendList(prependHead, appendHead);
202     for (int i = NITEMS; i >= 0; --i) {
203       assertEquals(i, node.getKey());
204       node = AvlIterableList.readNext(node);
205     }
206     for (int i = 0; i <= NITEMS; ++i) {
207       assertEquals(i, node.getKey());
208       node = AvlIterableList.readNext(node);
209     }
210   }
