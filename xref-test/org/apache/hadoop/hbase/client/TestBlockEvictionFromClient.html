164   @Test
165   public void testBlockEvictionWithParallelScans() throws Exception {
166     HTable table = null;
167     try {
168       latch = new CountDownLatch(1);
169       TableName tableName = TableName.valueOf("testBlockEvictionWithParallelScans");
170       // Create a table with block size as 1024
171       table = TEST_UTIL.createTable(tableName, FAMILIES_1, 1, 1024,
172           CustomInnerRegionObserver.class.getName());
173       // get the block cache and region
174       RegionLocator locator = table.getRegionLocator();
175       String regionName = locator.getAllRegionLocations().get(0).getRegionInfo().getEncodedName();
176       Region region = TEST_UTIL.getRSForFirstRegionInTable(tableName).getFromOnlineRegions(
177           regionName);
178       Store store = region.getStores().iterator().next();
179       CacheConfig cacheConf = store.getCacheConfig();
180       cacheConf.setCacheDataOnWrite(true);
181       cacheConf.setEvictOnClose(true);
182       BlockCache cache = cacheConf.getBlockCache();
183 
184       // insert data. 2 Rows are added
185       Put put = new Put(ROW);
186       put.addColumn(FAMILY, QUALIFIER, data);
187       table.put(put);
188       put = new Put(ROW1);
189       put.addColumn(FAMILY, QUALIFIER, data);
190       table.put(put);
191       assertTrue(Bytes.equals(table.get(new Get(ROW)).value(), data));
192       // data was in memstore so don't expect any changes
193       // flush the data
194       System.out.println("Flushing cache in problematic area");
195       // Should create one Hfile with 2 blocks
196       region.flush(true);
197       // Load cache
198       // Create three sets of scan
199       ScanThread[] scanThreads = initiateScan(table, false);
200       Thread.sleep(100);
201       checkForBlockEviction(cache, false, false, false);
202       for (ScanThread thread : scanThreads) {
203         thread.join();
204       }
205       // CustomInnerRegionObserver.sleepTime.set(0);
206       Iterator<CachedBlock> iterator = cache.iterator();
207       iterateBlockCache(cache, iterator);
208       // read the data and expect same blocks, one new hit, no misses
209       assertTrue(Bytes.equals(table.get(new Get(ROW)).value(), data));
210       iterator = cache.iterator();
211       iterateBlockCache(cache, iterator);
212       // Check how this miss is happening
213       // insert a second column, read the row, no new blocks, 3 new hits
214       byte[] QUALIFIER2 = Bytes.add(QUALIFIER, QUALIFIER);
215       byte[] data2 = Bytes.add(data, data);
216       put = new Put(ROW);
217       put.addColumn(FAMILY, QUALIFIER2, data2);
218       table.put(put);
219       Result r = table.get(new Get(ROW));
220       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER), data));
221       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER2), data2));
222       iterator = cache.iterator();
223       iterateBlockCache(cache, iterator);
224       // flush, one new block
225       System.out.println("Flushing cache");
226       region.flush(true);
227       iterator = cache.iterator();
228       iterateBlockCache(cache, iterator);
229       // compact, net minus two blocks, two hits, no misses
230       System.out.println("Compacting");
231       assertEquals(2, store.getStorefilesCount());
232       store.triggerMajorCompaction();
233       region.compact(true);
234       waitForStoreFileCount(store, 1, 10000); // wait 10 seconds max
235       assertEquals(1, store.getStorefilesCount());
236       iterator = cache.iterator();
237       iterateBlockCache(cache, iterator);
238       // read the row, this should be a cache miss because we don't cache data
239       // blocks on compaction
240       r = table.get(new Get(ROW));
241       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER), data));
242       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER2), data2));
243       iterator = cache.iterator();
244       iterateBlockCache(cache, iterator);
245     } finally {
246       if (table != null) {
247         table.close();
248       }
249     }
250   }

260       // Create KV that will give you two blocks

267       Region region = TEST_UTIL.getRSForFirstRegionInTable(tableName).getFromOnlineRegions(
268           regionName);
269       Store store = region.getStores().iterator().next();
270       CacheConfig cacheConf = store.getCacheConfig();
271       cacheConf.setCacheDataOnWrite(true);
272       cacheConf.setEvictOnClose(true);
273       BlockCache cache = cacheConf.getBlockCache();
274 
275       insertData(table);
276       // flush the data
277       System.out.println("Flushing cache");
278       // Should create one Hfile with 2 blocks
279       region.flush(true);
280       // Create three sets of scan
281       CustomInnerRegionObserver.waitForGets.set(true);
282       ScanThread[] scanThreads = initiateScan(table, false);
283       // Create three sets of gets
284       GetThread[] getThreads = initiateGet(table, false, false);
285       checkForBlockEviction(cache, false, false, false);
286       CustomInnerRegionObserver.waitForGets.set(false);
287       checkForBlockEviction(cache, false, false, false);
288       for (GetThread thread : getThreads) {
289         thread.join();
290       }
291       // Verify whether the gets have returned the blocks that it had
292       CustomInnerRegionObserver.waitForGets.set(true);
293       // giving some time for the block to be decremented
294       checkForBlockEviction(cache, true, false, false);
295       getLatch.countDown();
296       for (ScanThread thread : scanThreads) {
297         thread.join();
298       }
299       System.out.println("Scans should have returned the bloks");
300       // Check with either true or false
301       CustomInnerRegionObserver.waitForGets.set(false);
302       // The scan should also have released the blocks by now
303       checkForBlockEviction(cache, true, true, false);
304     } finally {
305       if (table != null) {
306         table.close();
307       }
