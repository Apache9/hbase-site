167   public void testBlockEvictionWithParallelScans() throws Exception {
168     HTable table = null;
169     try {
170       latch = new CountDownLatch(1);
171       TableName tableName = TableName.valueOf("testBlockEvictionWithParallelScans");
172       // Create a table with block size as 1024
173       table = TEST_UTIL.createTable(tableName, FAMILIES_1, 1, 1024,
174           CustomInnerRegionObserver.class.getName());
175       // get the block cache and region
176       RegionLocator locator = table.getRegionLocator();
177       String regionName = locator.getAllRegionLocations().get(0).getRegionInfo().getEncodedName();
178       Region region = TEST_UTIL.getRSForFirstRegionInTable(tableName).getFromOnlineRegions(
179           regionName);
180       Store store = region.getStores().iterator().next();
181       CacheConfig cacheConf = store.getCacheConfig();
182       cacheConf.setCacheDataOnWrite(true);
183       cacheConf.setEvictOnClose(true);
184       BlockCache cache = cacheConf.getBlockCache();
185 
186       // insert data. 2 Rows are added
187       Put put = new Put(ROW);
188       put.addColumn(FAMILY, QUALIFIER, data);
189       table.put(put);
190       put = new Put(ROW1);
191       put.addColumn(FAMILY, QUALIFIER, data);
192       table.put(put);
193       assertTrue(Bytes.equals(table.get(new Get(ROW)).value(), data));
194       // data was in memstore so don't expect any changes
195       // flush the data
196       System.out.println("Flushing cache in problematic area");
197       // Should create one Hfile with 2 blocks
198       region.flush(true);
199       // Load cache
200       // Create three sets of scan
201       ScanThread[] scanThreads = initiateScan(table, false);
202       Thread.sleep(100);
203       checkForBlockEviction(cache, false, false, false);
204       for (ScanThread thread : scanThreads) {
205         thread.join();
206       }
207       // CustomInnerRegionObserver.sleepTime.set(0);
208       Iterator<CachedBlock> iterator = cache.iterator();
209       iterateBlockCache(cache, iterator);
210       // read the data and expect same blocks, one new hit, no misses
211       assertTrue(Bytes.equals(table.get(new Get(ROW)).value(), data));
212       iterator = cache.iterator();
213       iterateBlockCache(cache, iterator);
214       // Check how this miss is happening
215       // insert a second column, read the row, no new blocks, 3 new hits
216       byte[] QUALIFIER2 = Bytes.add(QUALIFIER, QUALIFIER);
217       byte[] data2 = Bytes.add(data, data);
218       put = new Put(ROW);
219       put.addColumn(FAMILY, QUALIFIER2, data2);
220       table.put(put);
221       Result r = table.get(new Get(ROW));
222       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER), data));
223       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER2), data2));
224       iterator = cache.iterator();
225       iterateBlockCache(cache, iterator);
226       // flush, one new block
227       System.out.println("Flushing cache");
228       region.flush(true);
229       iterator = cache.iterator();
230       iterateBlockCache(cache, iterator);
231       // compact, net minus two blocks, two hits, no misses
232       System.out.println("Compacting");
233       assertEquals(2, store.getStorefilesCount());
234       store.triggerMajorCompaction();
235       region.compact(true);
236       waitForStoreFileCount(store, 1, 10000); // wait 10 seconds max
237       assertEquals(1, store.getStorefilesCount());
238       iterator = cache.iterator();
239       iterateBlockCache(cache, iterator);
240       // read the row, this should be a cache miss because we don't cache data
241       // blocks on compaction
242       r = table.get(new Get(ROW));
243       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER), data));
244       assertTrue(Bytes.equals(r.getValue(FAMILY, QUALIFIER2), data2));
245       iterator = cache.iterator();
246       iterateBlockCache(cache, iterator);
247     } finally {
248       if (table != null) {
249         table.close();
250       }
251     }
