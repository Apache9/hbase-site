47   * Test is parameterized to run the slow and fast increment code paths. If fast, in the @before, we

51   * Test takes a long time because spin up a cluster between each run -- ugh.

273   @Test
274   public void testIncrement() throws Exception {
275     LOG.info("Starting " + this.name.getMethodName());
276     final TableName TABLENAME =
277         TableName.valueOf(filterStringSoTableNameSafe(this.name.getMethodName()));
278     Table ht = TEST_UTIL.createTable(TABLENAME, FAMILY);
279 
280     byte [][] ROWS = new byte [][] {
281         Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c"),
282         Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
283         Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i")
284     };
285     byte [][] QUALIFIERS = new byte [][] {
286         Bytes.toBytes("a"), Bytes.toBytes("b"), Bytes.toBytes("c"),
287         Bytes.toBytes("d"), Bytes.toBytes("e"), Bytes.toBytes("f"),
288         Bytes.toBytes("g"), Bytes.toBytes("h"), Bytes.toBytes("i")
289     };
290 
291     // Do some simple single-column increments
292 
293     // First with old API
294     ht.incrementColumnValue(ROW, FAMILY, QUALIFIERS[0], 1);
295     ht.incrementColumnValue(ROW, FAMILY, QUALIFIERS[1], 2);
296     ht.incrementColumnValue(ROW, FAMILY, QUALIFIERS[2], 3);
297     ht.incrementColumnValue(ROW, FAMILY, QUALIFIERS[3], 4);
298 
299     // Now increment things incremented with old and do some new
300     Increment inc = new Increment(ROW);
301     inc.addColumn(FAMILY, QUALIFIERS[1], 1);
302     inc.addColumn(FAMILY, QUALIFIERS[3], 1);
303     inc.addColumn(FAMILY, QUALIFIERS[4], 1);
304     ht.increment(inc);
305 
306     // Verify expected results
307     Get get = new Get(ROW);
308     Result r = ht.get(get);
309     Cell [] kvs = r.rawCells();
310     assertEquals(5, kvs.length);
311     assertIncrementKey(kvs[0], ROW, FAMILY, QUALIFIERS[0], 1);
312     assertIncrementKey(kvs[1], ROW, FAMILY, QUALIFIERS[1], 3);
313     assertIncrementKey(kvs[2], ROW, FAMILY, QUALIFIERS[2], 3);
314     assertIncrementKey(kvs[3], ROW, FAMILY, QUALIFIERS[3], 5);
315     assertIncrementKey(kvs[4], ROW, FAMILY, QUALIFIERS[4], 1);
316 
317     // Now try multiple columns by different amounts
318     inc = new Increment(ROWS[0]);
319     for (int i=0;i<QUALIFIERS.length;i++) {
320       inc.addColumn(FAMILY, QUALIFIERS[i], i+1);
321     }
322     ht.increment(inc);
323     // Verify
324     get = new Get(ROWS[0]);
325     r = ht.get(get);
326     kvs = r.rawCells();
327     assertEquals(QUALIFIERS.length, kvs.length);
328     for (int i=0;i<QUALIFIERS.length;i++) {
329       assertIncrementKey(kvs[i], ROWS[0], FAMILY, QUALIFIERS[i], i+1);
330     }
331 
332     // Re-increment them
333     inc = new Increment(ROWS[0]);
334     for (int i=0;i<QUALIFIERS.length;i++) {
335       inc.addColumn(FAMILY, QUALIFIERS[i], i+1);
336     }
337     ht.increment(inc);
338     // Verify
339     r = ht.get(get);
340     kvs = r.rawCells();
341     assertEquals(QUALIFIERS.length, kvs.length);
342     for (int i=0;i<QUALIFIERS.length;i++) {
343       assertIncrementKey(kvs[i], ROWS[0], FAMILY, QUALIFIERS[i], 2*(i+1));
344     }
345 
346     // Verify that an Increment of an amount of zero, returns current count; i.e. same as for above
347     // test, that is: 2 * (i + 1).
348     inc = new Increment(ROWS[0]);
349     for (int i = 0; i < QUALIFIERS.length; i++) {
350       inc.addColumn(FAMILY, QUALIFIERS[i], 0);
351     }
352     ht.increment(inc);
353     r = ht.get(get);
354     kvs = r.rawCells();
355     assertEquals(QUALIFIERS.length, kvs.length);
356     for (int i = 0; i < QUALIFIERS.length; i++) {
357       assertIncrementKey(kvs[i], ROWS[0], FAMILY, QUALIFIERS[i], 2*(i+1));
358     }
359   }
