75    public AbstractTestLogRolling()  {

144     for (int i = 1; i <= 256; i++) {    // 256 writes should cause 8 log rolls

152         }

196     assertLogFileSize(log);
197
198     // flush all regions
199     for (Region r : server.getOnlineRegionsLocalContext()) {
200       r.flush(true);
201     }
202
203     // Now roll the log
204     log.rollWriter();
205
206     int count = AbstractFSWALProvider.getNumRolledLogFiles(log);
207     LOG.info("after flushing all regions and rolling logs there are " + count + " log files");
208     assertTrue(("actual count: " + count), count <= 2);
209     assertLogFileSize(log);

238   /**
239    * Tests that logs are deleted when some region has a compaction
240    * record in WAL and no other records. See HBASE-8597.
241    */
242   @Test
243   public void testCompactionRecordDoesntBlockRolling() throws Exception {
244     Table table = null;
245
246     // When the hbase:meta table can be opened, the region servers are running
247     Table t = TEST_UTIL.getConnection().getTable(TableName.META_TABLE_NAME);
248     try {
249       table = createTestTable(getName());
250
251       server = TEST_UTIL.getRSForFirstRegionInTable(table.getName());
252       Region region = server.getOnlineRegions(table.getName()).get(0);
253       final WAL log = server.getWAL(region.getRegionInfo());
254       Store s = region.getStore(HConstants.CATALOG_FAMILY);
255
256       //have to flush namespace to ensure it doesn't affect wall tests
257       admin.flush(TableName.NAMESPACE_TABLE_NAME);
258
259       // Put some stuff into table, to make sure we have some files to compact.
260       for (int i = 1; i <= 2; ++i) {
261         doPut(table, i);
262         admin.flush(table.getName());
263       }
264       doPut(table, 3); // don't flush yet, or compaction might trigger before we roll WAL
265       assertEquals("Should have no WAL after initial writes", 0,
266         AbstractFSWALProvider.getNumRolledLogFiles(log));
267       assertEquals(2, s.getStorefilesCount());
268
269       // Roll the log and compact table, to have compaction record in the 2nd WAL.
270       log.rollWriter();
271       assertEquals("Should have WAL; one table is not flushed", 1,
272         AbstractFSWALProvider.getNumRolledLogFiles(log));
273       admin.flush(table.getName());
274       region.compact(false);
275       // Wait for compaction in case if flush triggered it before us.
276       Assert.assertNotNull(s);
277       for (int waitTime = 3000; s.getStorefilesCount() > 1 && waitTime > 0; waitTime -= 200) {
278         Threads.sleepWithoutInterrupt(200);
279       }
280       assertEquals("Compaction didn't happen", 1, s.getStorefilesCount());
281
282       // Write some value to the table so the WAL cannot be deleted until table is flushed.
283       doPut(table, 0); // Now 2nd WAL will have both compaction and put record for table.
284       log.rollWriter(); // 1st WAL deleted, 2nd not deleted yet.
285       assertEquals("Should have WAL; one table is not flushed", 1,
286         AbstractFSWALProvider.getNumRolledLogFiles(log));
287
288       // Flush table to make latest WAL obsolete; write another record, and roll again.
289       admin.flush(table.getName());
290       doPut(table, 1);
291       log.rollWriter(); // Now 2nd WAL is deleted and 3rd is added.
292       assertEquals("Should have 1 WALs at the end", 1,
293         AbstractFSWALProvider.getNumRolledLogFiles(log));
294     } finally {
295       if (t != null) t.close();
296       if (table != null) table.close();
297     }
298   }
