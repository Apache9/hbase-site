64    private static final Path DIR = TEST_UTIL.getDataTestDir("TestHRegion");
65    public static final TableName TABLENAME = TableName.valueOf("TestWalAndCompactingMemStoreFlush",
66        "t1");

68    public static final byte[][] FAMILIES = { Bytes.toBytes("f1"), Bytes.toBytes("f2"),
69        Bytes.toBytes("f3"), Bytes.toBytes("f4"), Bytes.toBytes("f5") };

71    public static final byte[] FAMILY1 = FAMILIES[0];
72    public static final byte[] FAMILY2 = FAMILIES[1];
73    public static final byte[] FAMILY3 = FAMILIES[2];

77    private HRegion initHRegion(String callingMethod, Configuration conf) throws IOException {
78      int i=0;
79      HTableDescriptor htd = new HTableDescriptor(TABLENAME);
80      for (byte[] family : FAMILIES) {
81        HColumnDescriptor hcd = new HColumnDescriptor(family);
82        // even column families are going to have compacted memstore
83        if(i%2 == 0) hcd.setInMemoryCompaction(true);
84        htd.addFamily(hcd);
85        i++;
86      }
87
88      HRegionInfo info = new HRegionInfo(TABLENAME, null, null, false);
89      Path path = new Path(DIR, callingMethod);
90      return HBaseTestingUtility.createRegionAndWAL(info, path, conf, htd);
91    }

95    // A helper function to create puts.
96    private Put createPut(int familyNum, int putNum) {
97      byte[] qf  = Bytes.toBytes("q" + familyNum);
98      byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
99      byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
100     Put p = new Put(row);
101     p.addColumn(FAMILIES[familyNum - 1], qf, val);
102     return p;
103   }
104
105
106   // A helper function to create double puts, so something can be compacted later.
107   private Put createDoublePut(int familyNum, int putNum) {
108     byte[] qf  = Bytes.toBytes("q" + familyNum);
109     byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
110     byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
111     Put p = new Put(row);
112     // add twice with different timestamps
113     p.addColumn(FAMILIES[familyNum - 1], qf, 10, val);
114     p.addColumn(FAMILIES[familyNum - 1], qf, 20, val);
115     return p;
116   }
117
118
119   // A helper function to create gets.
120   private Get createGet(int familyNum, int putNum) {
121     byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
122     return new Get(row);
123   }

128   // A helper function to verify edits.
129   void verifyEdit(int familyNum, int putNum, Table table) throws IOException {
130     Result r = table.get(createGet(familyNum, putNum));
131     byte[] family = FAMILIES[familyNum - 1];
132     byte[] qf = Bytes.toBytes("q" + familyNum);
133     byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
134     assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum), r.getFamilyMap(family));
135     assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum),
136       r.getFamilyMap(family).get(qf));
137     assertTrue(("Incorrect value for Put#" + putNum + " for CF# " + familyNum),
138       Arrays.equals(r.getFamilyMap(family).get(qf), val));
139   }

145   @Test(timeout = 180000)
146   public void testSelectiveFlushWhenEnabled() throws IOException {
147
148     // Set up the configuration
149     Configuration conf = HBaseConfiguration.create();
150     conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 600 * 1024);
151     conf.set(FlushPolicyFactory.HBASE_FLUSH_POLICY_KEY, FlushNonSloppyStoresFirstPolicy.class
152         .getName());
153     conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 300 *
154         1024);
155     conf.setDouble(CompactingMemStore.IN_MEMORY_FLUSH_THRESHOLD_FACTOR_KEY, 0.5);
156
157     // Intialize the region
158     Region region = initHRegion("testSelectiveFlushWhenEnabled", conf);
159
160     // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
161     for (int i = 1; i <= 1200; i++) {
162       region.put(createPut(1, i));        // compacted memstore
163
164       if (i <= 100) {
165         region.put(createPut(2, i));
166         if (i <= 50) {
167           region.put(createDoublePut(3, i)); // subject for in-memory compaction
168         }
169       }
170     }
171
172     // Now add more puts for CF2, so that we only flush CF2 to disk
173     for (int i = 100; i < 2000; i++) {
174       region.put(createPut(2, i));
175     }
176
177     long totalMemstoreSize = region.getMemstoreSize();
178
179     // Find the smallest LSNs for edits wrt to each CF.
180     long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
181     long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
182     long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
183
184     // Find the sizes of the memstores of each CF.
185     long cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getMemStoreSize();
186     long cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getMemStoreSize();
187     long cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getMemStoreSize();
188
189     // Get the overall smallest LSN in the region's memstores.
190     long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
191         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
192
193     String s = "\n\n----------------------------------\n"
194         + "Upon initial insert and before any flush, size of CF1 is:"
195         + cf1MemstoreSizePhaseI + ", is CF1 compacted memstore?:"
196         + region.getStore(FAMILY1).getMemStore().isSloppy() + ". Size of CF2 is:"
197         + cf2MemstoreSizePhaseI + ", is CF2 compacted memstore?:"
198         + region.getStore(FAMILY2).getMemStore().isSloppy() + ". Size of CF3 is:"
199         + cf3MemstoreSizePhaseI + ", is CF3 compacted memstore?:"
200         + region.getStore(FAMILY3).getMemStore().isSloppy() + "\n";
201
202     // The overall smallest LSN in the region's memstores should be the same as
203     // the LSN of the smallest edit in CF1
204     assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
205
206     // Some other sanity checks.
207     assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
208     assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
209     assertTrue(cf1MemstoreSizePhaseI > 0);
210     assertTrue(cf2MemstoreSizePhaseI > 0);
211     assertTrue(cf3MemstoreSizePhaseI > 0);
212
213     // The total memstore size should be the same as the sum of the sizes of
214     // memstores of CF1, CF2 and CF3.
215     String msg = "totalMemstoreSize="+totalMemstoreSize +
216         " DefaultMemStore.DEEP_OVERHEAD="+DefaultMemStore.DEEP_OVERHEAD +
217         " DEEP_OVERHEAD_PER_PIPELINE_ITEM="+CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM +
218         " cf1MemstoreSizePhaseI="+cf1MemstoreSizePhaseI +
219         " cf2MemstoreSizePhaseI="+cf2MemstoreSizePhaseI +
220         " cf3MemstoreSizePhaseI="+cf3MemstoreSizePhaseI ;
221     assertEquals(msg,totalMemstoreSize + 3 * DefaultMemStore.DEEP_OVERHEAD,
222         cf1MemstoreSizePhaseI + cf2MemstoreSizePhaseI + cf3MemstoreSizePhaseI);
223
224     // Flush!!!!!!!!!!!!!!!!!!!!!!
225     // We have big compacting memstore CF1 and two small memstores:
226     // CF2 (not compacted) and CF3 (compacting)
227     // All together they are above the flush size lower bound.
228     // Since CF1 and CF3 should be flushed to memory (not to disk),
229     // CF2 is going to be flushed to disk.
230     // CF1 - nothing to compact, CF3 - should be twice compacted
231     ((CompactingMemStore) region.getStore(FAMILY1).getMemStore()).flushInMemory();
232     ((CompactingMemStore) region.getStore(FAMILY3).getMemStore()).flushInMemory();
233     region.flush(false);
234
235     // CF3 should be compacted so wait here to be sure the compaction is done
236     while (((CompactingMemStore) region.getStore(FAMILY3).getMemStore())
237         .isMemStoreFlushingInMemory())
238       Threads.sleep(10);
239
240     // Recalculate everything
241     long cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getMemStoreSize();
242     long cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getMemStoreSize();
243     long cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getMemStoreSize();
244
245     long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
246         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
247     // Find the smallest LSNs for edits wrt to each CF.
248     long smallestSeqCF1PhaseII = region.getOldestSeqIdOfStore(FAMILY1);
249     long smallestSeqCF2PhaseII = region.getOldestSeqIdOfStore(FAMILY2);
250     long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
251
252     s = s + "DefaultMemStore DEEP_OVERHEAD is:" + DefaultMemStore.DEEP_OVERHEAD
253         + ", CompactingMemStore DEEP_OVERHEAD is:" + CompactingMemStore.DEEP_OVERHEAD
254         + ", CompactingMemStore DEEP_OVERHEAD_PER_PIPELINE_ITEM is:" + CompactingMemStore
255         .DEEP_OVERHEAD_PER_PIPELINE_ITEM
256         + "\n----After first flush! CF1 should be flushed to memory, but not compacted.---\n"
257         + "Size of CF1 is:" + cf1MemstoreSizePhaseII + ", size of CF2 is:" + cf2MemstoreSizePhaseII
258         + ", size of CF3 is:" + cf3MemstoreSizePhaseII + "\n";
259
260     // CF1 was flushed to memory, but there is nothing to compact, should
261     // remain the same size plus renewed empty skip-list
262     assertEquals(s, cf1MemstoreSizePhaseII,
263         cf1MemstoreSizePhaseI + CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM);
264
265     // CF2 should become empty
266     assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseII);
267
268     // verify that CF3 was flushed to memory and was compacted (this is approximation check)
269     assertTrue(cf3MemstoreSizePhaseI/2+DefaultMemStore.DEEP_OVERHEAD +
270         CompactingMemStore.DEEP_OVERHEAD_PER_PIPELINE_ITEM >
271         cf3MemstoreSizePhaseII);
272     assertTrue(cf3MemstoreSizePhaseI/2 < cf3MemstoreSizePhaseII);
273
274
275     // Now the smallest LSN in the region should be the same as the smallest
276     // LSN in the memstore of CF1.
277     assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
278
279     // Now add more puts for CF1, so that we also flush CF1 to disk instead of
280     // memory in next flush
281     for (int i = 1200; i < 2000; i++) {
282       region.put(createPut(1, i));
283     }
284
285     s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseII
286         + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseII + ", " +
287         "the smallest sequence in CF2:"
288         + smallestSeqCF2PhaseII +", the smallest sequence in CF3:" + smallestSeqCF3PhaseII + "\n";
289
290     // How much does the CF1 memstore occupy? Will be used later.
291     long cf1MemstoreSizePhaseIII = region.getStore(FAMILY1).getMemStoreSize();
292     long smallestSeqCF1PhaseIII = region.getOldestSeqIdOfStore(FAMILY1);
293
294     s = s + "----After more puts into CF1 its size is:" + cf1MemstoreSizePhaseIII
295         + ", and its sequence is:" + smallestSeqCF1PhaseIII + " ----\n" ;
296
297
298     // Flush!!!!!!!!!!!!!!!!!!!!!!
299     // Flush again, CF1 is flushed to disk
300     // CF2 is flushed to disk, because it is not in-memory compacted memstore
301     // CF3 is flushed empty to memory (actually nothing happens to CF3)
302     region.flush(false);
303
304     // Recalculate everything
305     long cf1MemstoreSizePhaseIV = region.getStore(FAMILY1).getMemStoreSize();
306     long cf2MemstoreSizePhaseIV = region.getStore(FAMILY2).getMemStoreSize();
307     long cf3MemstoreSizePhaseIV = region.getStore(FAMILY3).getMemStoreSize();
308
309     long smallestSeqInRegionCurrentMemstorePhaseIV = getWAL(region)
310         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
311     long smallestSeqCF1PhaseIV = region.getOldestSeqIdOfStore(FAMILY1);
312     long smallestSeqCF2PhaseIV = region.getOldestSeqIdOfStore(FAMILY2);
313     long smallestSeqCF3PhaseIV = region.getOldestSeqIdOfStore(FAMILY3);
314
315     s = s + "----After SECOND FLUSH, CF1 size is:" + cf1MemstoreSizePhaseIV + ", CF2 size is:"
316         + cf2MemstoreSizePhaseIV + " and CF3 size is:" + cf3MemstoreSizePhaseIV
317         + "\n";
318
319     s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseIV
320         + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseIV + ", " +
321         "the smallest sequence in CF2:"
322         + smallestSeqCF2PhaseIV +", the smallest sequence in CF3:" + smallestSeqCF3PhaseIV
323         + "\n";
324
325     // CF1's pipeline component (inserted before first flush) should be flushed to disk
326     // CF2 should be flushed to disk
327     assertEquals(cf1MemstoreSizePhaseIII - cf1MemstoreSizePhaseI + DefaultMemStore.DEEP_OVERHEAD,
328         cf1MemstoreSizePhaseIV);
329     assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseIV);
330
331     // CF3 shouldn't have been touched.
332     assertEquals(cf3MemstoreSizePhaseIV, cf3MemstoreSizePhaseII);
333
334     // the smallest LSN of CF3 shouldn't change
335     assertEquals(smallestSeqCF3PhaseII, smallestSeqCF3PhaseIV);
336
337     // CF3 should be bottleneck for WAL
338      assertEquals(s, smallestSeqInRegionCurrentMemstorePhaseIV, smallestSeqCF3PhaseIV);
339
340     // Flush!!!!!!!!!!!!!!!!!!!!!!
341     // Clearing the existing memstores, CF2 all flushed to disk. The single
342     // memstore segment in the compaction pipeline of CF1 and CF3 should be flushed to disk.
343     // Note that active sets of CF1 and CF3 are empty
344     region.flush(true);
345
346     // Recalculate everything
347     long cf1MemstoreSizePhaseV = region.getStore(FAMILY1).getMemStoreSize();
348     long cf2MemstoreSizePhaseV = region.getStore(FAMILY2).getMemStoreSize();
349     long cf3MemstoreSizePhaseV = region.getStore(FAMILY3).getMemStoreSize();
350     long smallestSeqInRegionCurrentMemstorePhaseV = getWAL(region)
351         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
352
353     assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf1MemstoreSizePhaseV);
354     assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf2MemstoreSizePhaseV);
355     assertEquals(DefaultMemStore.DEEP_OVERHEAD, cf3MemstoreSizePhaseV);
356
357     // Because there is nothing in any memstore the WAL's LSN should be -1
358     assertEquals(smallestSeqInRegionCurrentMemstorePhaseV, HConstants.NO_SEQNUM);
359
360     // What happens when we hit the memstore limit, but we are not able to find
361     // any Column Family above the threshold?
362     // In that case, we should flush all the CFs.
363
364     // The memstore limit is 200*1024 and the column family flush threshold is
365     // around 50*1024. We try to just hit the memstore limit with each CF's
366     // memstore being below the CF flush threshold.
367     for (int i = 1; i <= 300; i++) {
368       region.put(createPut(1, i));
369       region.put(createPut(2, i));
370       region.put(createPut(3, i));
371       region.put(createPut(4, i));
372       region.put(createPut(5, i));
373     }
374
375     region.flush(false);
376
377     s = s + "----AFTER THIRD AND FORTH FLUSH, The smallest sequence in region WAL is: "
378         + smallestSeqInRegionCurrentMemstorePhaseV
379         + ". After additional inserts and last flush, the entire region size is:" + region
380         .getMemstoreSize()
381         + "\n----------------------------------\n";
382
383     // Since we won't find any CF above the threshold, and hence no specific
384     // store to flush, we should flush all the memstores
385     // Also compacted memstores are flushed to disk.
386     assertEquals(0, region.getMemstoreSize());
387     System.out.println(s);
388     HBaseTestingUtility.closeRegionAndWAL(region);
389   }
