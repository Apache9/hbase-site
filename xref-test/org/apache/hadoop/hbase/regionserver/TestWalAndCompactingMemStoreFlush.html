49  /**
50   * This test verifies the correctness of the Per Column Family flushing strategy
51   * when part of the memstores are compacted memstores
52   */
53  @Category({ RegionServerTests.class, LargeTests.class })
54  public class TestWalAndCompactingMemStoreFlush {

56    private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
57    private static final Path DIR = TEST_UTIL.getDataTestDir("TestHRegion");
58    public static final TableName TABLENAME = TableName.valueOf("TestWalAndCompactingMemStoreFlush",
59        "t1");
60
61    public static final byte[][] FAMILIES = { Bytes.toBytes("f1"), Bytes.toBytes("f2"),
62        Bytes.toBytes("f3"), Bytes.toBytes("f4"), Bytes.toBytes("f5") };
63
64    public static final byte[] FAMILY1 = FAMILIES[0];
65    public static final byte[] FAMILY2 = FAMILIES[1];
66    public static final byte[] FAMILY3 = FAMILIES[2];
67
68    private HRegion initHRegion(String callingMethod, Configuration conf) throws IOException {
69      int i=0;
70      HTableDescriptor htd = new HTableDescriptor(TABLENAME);
71      for (byte[] family : FAMILIES) {
72        HColumnDescriptor hcd = new HColumnDescriptor(family);
73        // even column families are going to have compacted memstore
74        if(i%2 == 0) hcd.setInMemoryCompaction(true);
75        htd.addFamily(hcd);
76        i++;
77      }
78
79      HRegionInfo info = new HRegionInfo(TABLENAME, null, null, false);
80      Path path = new Path(DIR, callingMethod);
81      return HBaseTestingUtility.createRegionAndWAL(info, path, conf, htd);
82    }

84    // A helper function to create puts.
85    private Put createPut(int familyNum, int putNum) {
86      byte[] qf  = Bytes.toBytes("q" + familyNum);
87      byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
88      byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
89      Put p = new Put(row);
90      p.addColumn(FAMILIES[familyNum - 1], qf, val);
91      return p;
92    }
93
94    // A helper function to create double puts, so something can be compacted later.
95    private Put createDoublePut(int familyNum, int putNum) {
96      byte[] qf  = Bytes.toBytes("q" + familyNum);
97      byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
98      byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
99      Put p = new Put(row);
100     // add twice with different timestamps
101     p.addColumn(FAMILIES[familyNum - 1], qf, 10, val);
102     p.addColumn(FAMILIES[familyNum - 1], qf, 20, val);
103     return p;
104   }
105
106   // A helper function to create gets.
107   private Get createGet(int familyNum, int putNum) {
108     byte[] row = Bytes.toBytes("row" + familyNum + "-" + putNum);
109     return new Get(row);
110   }
111
112   // A helper function to verify edits.
113   void verifyEdit(int familyNum, int putNum, Table table) throws IOException {
114     Result r = table.get(createGet(familyNum, putNum));
115     byte[] family = FAMILIES[familyNum - 1];
116     byte[] qf = Bytes.toBytes("q" + familyNum);
117     byte[] val = Bytes.toBytes("val" + familyNum + "-" + putNum);
118     assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum), r.getFamilyMap(family));
119     assertNotNull(("Missing Put#" + putNum + " for CF# " + familyNum),
120       r.getFamilyMap(family).get(qf));
121     assertTrue(("Incorrect value for Put#" + putNum + " for CF# " + familyNum),
122       Arrays.equals(r.getFamilyMap(family).get(qf), val));
123   }

125   @Test(timeout = 180000)
126   public void testSelectiveFlushWithDataCompaction() throws IOException {
127
128     // Set up the configuration
129     Configuration conf = HBaseConfiguration.create();
130     conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 300 * 1024);
131     conf.set(FlushPolicyFactory.HBASE_FLUSH_POLICY_KEY,
132         FlushNonSloppyStoresFirstPolicy.class.getName());
133     conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 75 * 1024);
134     conf.setDouble(CompactingMemStore.IN_MEMORY_FLUSH_THRESHOLD_FACTOR_KEY, 0.25);
135     // set memstore to do data compaction
136     conf.set("hbase.hregion.compacting.memstore.type", "data-compaction");
137
138     // Intialize the region
139     Region region = initHRegion("testSelectiveFlushWithDataCompaction", conf);
140
141     // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
142     for (int i = 1; i <= 1200; i++) {
143       region.put(createPut(1, i));        // compacted memstore, all the keys are unique
144
145       if (i <= 100) {
146         region.put(createPut(2, i));
147         if (i <= 50) {
148           // compacted memstore, subject for compaction due to duplications
149           region.put(createDoublePut(3, i));
150         }
151       }
152     }
153
154     // Now add more puts for CF2, so that we only flush CF2 (DefaultMemStore) to disk
155     for (int i = 100; i < 2000; i++) {
156       region.put(createPut(2, i));
157     }
158
159     long totalMemstoreSize = region.getMemstoreSize();
160
161     // Find the smallest LSNs for edits wrt to each CF.
162     long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
163     long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
164     long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
165
166     // Find the sizes of the memstores of each CF.
167     MemstoreSize cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getSizeOfMemStore();
168     MemstoreSize cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getSizeOfMemStore();
169     MemstoreSize cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getSizeOfMemStore();
170
171     // Get the overall smallest LSN in the region's memstores.
172     long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
173         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
174
175     String s = "\n\n----------------------------------\n"
176         + "Upon initial insert and before any flush, size of CF1 is:"
177         + cf1MemstoreSizePhaseI + ", is CF1 compacted memstore?:"
178         + region.getStore(FAMILY1).isSloppyMemstore() + ". Size of CF2 is:"
179         + cf2MemstoreSizePhaseI + ", is CF2 compacted memstore?:"
180         + region.getStore(FAMILY2).isSloppyMemstore() + ". Size of CF3 is:"
181         + cf3MemstoreSizePhaseI + ", is CF3 compacted memstore?:"
182         + region.getStore(FAMILY3).isSloppyMemstore() + "\n";
183
184     // The overall smallest LSN in the region's memstores should be the same as
185     // the LSN of the smallest edit in CF1
186     assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
187
188     // Some other sanity checks.
189     assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
190     assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
191     assertTrue(cf1MemstoreSizePhaseI.getDataSize() > 0);
192     assertTrue(cf2MemstoreSizePhaseI.getDataSize() > 0);
193     assertTrue(cf3MemstoreSizePhaseI.getDataSize() > 0);
194
195     // The total memstore size should be the same as the sum of the sizes of
196     // memstores of CF1, CF2 and CF3.
197     String msg = "totalMemstoreSize="+totalMemstoreSize +
198         " cf1MemstoreSizePhaseI="+cf1MemstoreSizePhaseI +
199         " cf2MemstoreSizePhaseI="+cf2MemstoreSizePhaseI +
200         " cf3MemstoreSizePhaseI="+cf3MemstoreSizePhaseI ;
201     assertEquals(msg, totalMemstoreSize, cf1MemstoreSizePhaseI.getDataSize()
202         + cf2MemstoreSizePhaseI.getDataSize() + cf3MemstoreSizePhaseI.getDataSize());
203
204     // Flush!!!!!!!!!!!!!!!!!!!!!!
205     // We have big compacting memstore CF1 and two small memstores:
206     // CF2 (not compacted) and CF3 (compacting)
207     // All together they are above the flush size lower bound.
208     // Since CF1 and CF3 should be flushed to memory (not to disk),
209     // CF2 is going to be flushed to disk.
210     // CF1 - nothing to compact (but flattening), CF3 - should be twice compacted
211     CompactingMemStore cms1 = (CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore;
212     CompactingMemStore cms3 = (CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore;
213     cms1.flushInMemory();
214     cms3.flushInMemory();
215     region.flush(false);
216
217     // Recalculate everything
218     MemstoreSize cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getSizeOfMemStore();
219     MemstoreSize cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getSizeOfMemStore();
220     MemstoreSize cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getSizeOfMemStore();
221
222     long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
223         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
224     // Find the smallest LSNs for edits wrt to each CF.
225     long smallestSeqCF1PhaseII = region.getOldestSeqIdOfStore(FAMILY1);
226     long smallestSeqCF2PhaseII = region.getOldestSeqIdOfStore(FAMILY2);
227     long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
228
229     s = s + "\n----After first flush! CF1 should be flushed to memory, but not compacted.---\n"
230         + "Size of CF1 is:" + cf1MemstoreSizePhaseII + ", size of CF2 is:" + cf2MemstoreSizePhaseII
231         + ", size of CF3 is:" + cf3MemstoreSizePhaseII + "\n";
232
233     // CF1 was flushed to memory, but there is nothing to compact, and CF1 was flattened
234     assertTrue(cf1MemstoreSizePhaseII.getDataSize() == cf1MemstoreSizePhaseI.getDataSize());
235     assertTrue(cf1MemstoreSizePhaseII.getHeapOverhead() < cf1MemstoreSizePhaseI.getHeapOverhead());
236
237     // CF2 should become empty
238     assertEquals(MemstoreSize.EMPTY_SIZE, cf2MemstoreSizePhaseII);
239
240     // verify that CF3 was flushed to memory and was compacted (this is approximation check)
241     assertTrue(cf3MemstoreSizePhaseI.getDataSize() > cf3MemstoreSizePhaseII.getDataSize());
242     assertTrue(
243         cf3MemstoreSizePhaseI.getHeapOverhead() / 2 > cf3MemstoreSizePhaseII.getHeapOverhead());
244
245     // Now the smallest LSN in the region should be the same as the smallest
246     // LSN in the memstore of CF1.
247     assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
248
249     // Now add more puts for CF1, so that we also flush CF1 to disk instead of
250     // memory in next flush
251     for (int i = 1200; i < 3000; i++) {
252       region.put(createPut(1, i));
253     }
254
255     s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseII
256         + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseII + ", " +
257         "the smallest sequence in CF2:"
258         + smallestSeqCF2PhaseII +", the smallest sequence in CF3:" + smallestSeqCF3PhaseII + "\n";
259
260     // How much does the CF1 memstore occupy? Will be used later.
261     MemstoreSize cf1MemstoreSizePhaseIII = region.getStore(FAMILY1).getSizeOfMemStore();
262     long smallestSeqCF1PhaseIII = region.getOldestSeqIdOfStore(FAMILY1);
263
264     s = s + "----After more puts into CF1 its size is:" + cf1MemstoreSizePhaseIII
265         + ", and its sequence is:" + smallestSeqCF1PhaseIII + " ----\n" ;
266
267
268     // Flush!!!!!!!!!!!!!!!!!!!!!!
269     // Flush again, CF1 is flushed to disk
270     // CF2 is flushed to disk, because it is not in-memory compacted memstore
271     // CF3 is flushed empty to memory (actually nothing happens to CF3)
272     region.flush(false);
273
274     // Recalculate everything
275     MemstoreSize cf1MemstoreSizePhaseIV = region.getStore(FAMILY1).getSizeOfMemStore();
276     MemstoreSize cf2MemstoreSizePhaseIV = region.getStore(FAMILY2).getSizeOfMemStore();
277     MemstoreSize cf3MemstoreSizePhaseIV = region.getStore(FAMILY3).getSizeOfMemStore();
278
279     long smallestSeqInRegionCurrentMemstorePhaseIV = getWAL(region)
280         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
281     long smallestSeqCF1PhaseIV = region.getOldestSeqIdOfStore(FAMILY1);
282     long smallestSeqCF2PhaseIV = region.getOldestSeqIdOfStore(FAMILY2);
283     long smallestSeqCF3PhaseIV = region.getOldestSeqIdOfStore(FAMILY3);
284
285     s = s + "----After SECOND FLUSH, CF1 size is:" + cf1MemstoreSizePhaseIV + ", CF2 size is:"
286         + cf2MemstoreSizePhaseIV + " and CF3 size is:" + cf3MemstoreSizePhaseIV
287         + "\n";
288
289     s = s + "The smallest sequence in region WAL is: " + smallestSeqInRegionCurrentMemstorePhaseIV
290         + ", the smallest sequence in CF1:" + smallestSeqCF1PhaseIV + ", " +
291         "the smallest sequence in CF2:"
292         + smallestSeqCF2PhaseIV +", the smallest sequence in CF3:" + smallestSeqCF3PhaseIV
293         + "\n";
294
295     // CF1's pipeline component (inserted before first flush) should be flushed to disk
296     // CF2 should be flushed to disk
297     assertTrue(cf1MemstoreSizePhaseIII.getDataSize() > cf1MemstoreSizePhaseIV.getDataSize());
298     assertEquals(MemstoreSize.EMPTY_SIZE, cf2MemstoreSizePhaseIV);
299
300     // CF3 shouldn't have been touched.
301     assertEquals(cf3MemstoreSizePhaseIV, cf3MemstoreSizePhaseII);
302
303     // the smallest LSN of CF3 shouldn't change
304     assertEquals(smallestSeqCF3PhaseII, smallestSeqCF3PhaseIV);
305
306     // CF3 should be bottleneck for WAL
307     assertEquals(s, smallestSeqInRegionCurrentMemstorePhaseIV, smallestSeqCF3PhaseIV);
308
309     // Flush!!!!!!!!!!!!!!!!!!!!!!
310     // Trying to clean the existing memstores, CF2 all flushed to disk. The single
311     // memstore segment in the compaction pipeline of CF1 and CF3 should be flushed to disk.
312     region.flush(true);
313
314     // Recalculate everything
315     MemstoreSize cf1MemstoreSizePhaseV = region.getStore(FAMILY1).getSizeOfMemStore();
316     MemstoreSize cf2MemstoreSizePhaseV = region.getStore(FAMILY2).getSizeOfMemStore();
317     MemstoreSize cf3MemstoreSizePhaseV = region.getStore(FAMILY3).getSizeOfMemStore();
318     long smallestSeqInRegionCurrentMemstorePhaseV = getWAL(region)
319         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
320
321     assertEquals(MemstoreSize.EMPTY_SIZE , cf1MemstoreSizePhaseV);
322     assertEquals(MemstoreSize.EMPTY_SIZE, cf2MemstoreSizePhaseV);
323     assertEquals(MemstoreSize.EMPTY_SIZE, cf3MemstoreSizePhaseV);
324
325     // What happens when we hit the memstore limit, but we are not able to find
326     // any Column Family above the threshold?
327     // In that case, we should flush all the CFs.
328
329     // The memstore limit is 100*1024 and the column family flush threshold is
330     // around 25*1024. We try to just hit the memstore limit with each CF's
331     // memstore being below the CF flush threshold.
332     for (int i = 1; i <= 300; i++) {
333       region.put(createPut(1, i));
334       region.put(createPut(2, i));
335       region.put(createPut(3, i));
336       region.put(createPut(4, i));
337       region.put(createPut(5, i));
338     }
339
340     region.flush(false);
341
342     s = s + "----AFTER THIRD AND FORTH FLUSH, The smallest sequence in region WAL is: "
343         + smallestSeqInRegionCurrentMemstorePhaseV
344         + ". After additional inserts and last flush, the entire region size is:" + region
345         .getMemstoreSize()
346         + "\n----------------------------------\n";
347
348     // Since we won't find any CF above the threshold, and hence no specific
349     // store to flush, we should flush all the memstores
350     // Also compacted memstores are flushed to disk.
351     assertEquals(0, region.getMemstoreSize());
352     System.out.println(s);
353     HBaseTestingUtility.closeRegionAndWAL(region);
354   }

358   @Test(timeout = 180000)
359   public void testSelectiveFlushWithIndexCompaction() throws IOException {
360
361     /*------------------------------------------------------------------------------*/
362     /* SETUP */
363     // Set up the configuration
364     Configuration conf = HBaseConfiguration.create();
365     conf.setLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, 300 * 1024);
366     conf.set(FlushPolicyFactory.HBASE_FLUSH_POLICY_KEY,
367         FlushNonSloppyStoresFirstPolicy.class.getName());
368     conf.setLong(FlushLargeStoresPolicy.HREGION_COLUMNFAMILY_FLUSH_SIZE_LOWER_BOUND_MIN, 75 * 1024);
369     conf.setDouble(CompactingMemStore.IN_MEMORY_FLUSH_THRESHOLD_FACTOR_KEY, 0.5);
370     // set memstore to index-compaction
371     conf.set("hbase.hregion.compacting.memstore.type", "index-compaction");
372
373     // Initialize the region
374     Region region = initHRegion("testSelectiveFlushWithIndexCompaction", conf);
375
376     /*------------------------------------------------------------------------------*/
377     /* PHASE I - insertions */
378     // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
379     for (int i = 1; i <= 1200; i++) {
380       region.put(createPut(1, i));        // compacted memstore
381       if (i <= 100) {
382         region.put(createPut(2, i));
383         if (i <= 50) {
384           region.put(createDoublePut(3, i)); // subject for in-memory compaction
385         }
386       }
387     }
388     // Now add more puts for CF2, so that we only flush CF2 to disk
389     for (int i = 100; i < 2000; i++) {
390       region.put(createPut(2, i));
391     }
392
393     /*------------------------------------------------------------------------------*/
394     /*------------------------------------------------------------------------------*/
395     /* PHASE I - collect sizes */
396     long totalMemstoreSizePhaseI = region.getMemstoreSize();
397     // Find the smallest LSNs for edits wrt to each CF.
398     long smallestSeqCF1PhaseI = region.getOldestSeqIdOfStore(FAMILY1);
399     long smallestSeqCF2PhaseI = region.getOldestSeqIdOfStore(FAMILY2);
400     long smallestSeqCF3PhaseI = region.getOldestSeqIdOfStore(FAMILY3);
401     // Find the sizes of the memstores of each CF.
402     MemstoreSize cf1MemstoreSizePhaseI = region.getStore(FAMILY1).getSizeOfMemStore();
403     MemstoreSize cf2MemstoreSizePhaseI = region.getStore(FAMILY2).getSizeOfMemStore();
404     MemstoreSize cf3MemstoreSizePhaseI = region.getStore(FAMILY3).getSizeOfMemStore();
405     // Get the overall smallest LSN in the region's memstores.
406     long smallestSeqInRegionCurrentMemstorePhaseI = getWAL(region)
407         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
408
409     /*------------------------------------------------------------------------------*/
410     /* PHASE I - validation */
411     // The overall smallest LSN in the region's memstores should be the same as
412     // the LSN of the smallest edit in CF1
413     assertEquals(smallestSeqCF1PhaseI, smallestSeqInRegionCurrentMemstorePhaseI);
414     // Some other sanity checks.
415     assertTrue(smallestSeqCF1PhaseI < smallestSeqCF2PhaseI);
416     assertTrue(smallestSeqCF2PhaseI < smallestSeqCF3PhaseI);
417     assertTrue(cf1MemstoreSizePhaseI.getDataSize() > 0);
418     assertTrue(cf2MemstoreSizePhaseI.getDataSize() > 0);
419     assertTrue(cf3MemstoreSizePhaseI.getDataSize() > 0);
420
421     // The total memstore size should be the same as the sum of the sizes of
422     // memstores of CF1, CF2 and CF3.
423     assertEquals(totalMemstoreSizePhaseI, cf1MemstoreSizePhaseI.getDataSize()
424         + cf2MemstoreSizePhaseI.getDataSize() + cf3MemstoreSizePhaseI.getDataSize());
425
426     /*------------------------------------------------------------------------------*/
427     /* PHASE I - Flush */
428     // First Flush in Test!!!!!!!!!!!!!!!!!!!!!!
429     // CF1, CF2, CF3, all together they are above the flush size lower bound.
430     // Since CF1 and CF3 are compacting, CF2 is going to be flushed to disk.
431     // CF1 and CF3 - flushed to memory and flatten explicitly
432     region.flush(false);
433     CompactingMemStore cms1 = (CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore;
434     CompactingMemStore cms3 = (CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore;
435     cms1.flushInMemory();
436     cms3.flushInMemory();
437
438     // CF3/CF1 should be merged so wait here to be sure the compaction is done
439     while (((CompactingMemStore) ((HStore) region.getStore(FAMILY1)).memstore)
440         .isMemStoreFlushingInMemory()) {
441       Threads.sleep(10);
442     }
443     while (((CompactingMemStore) ((HStore) region.getStore(FAMILY3)).memstore)
444         .isMemStoreFlushingInMemory()) {
445       Threads.sleep(10);
446     }
447
448     /*------------------------------------------------------------------------------*/
449     /*------------------------------------------------------------------------------*/
450     /* PHASE II - collect sizes */
451     // Recalculate everything
452     MemstoreSize cf1MemstoreSizePhaseII = region.getStore(FAMILY1).getSizeOfMemStore();
453     MemstoreSize cf2MemstoreSizePhaseII = region.getStore(FAMILY2).getSizeOfMemStore();
454     MemstoreSize cf3MemstoreSizePhaseII = region.getStore(FAMILY3).getSizeOfMemStore();
455     long smallestSeqInRegionCurrentMemstorePhaseII = getWAL(region)
456         .getEarliestMemstoreSeqNum(region.getRegionInfo().getEncodedNameAsBytes());
457     // Find the smallest LSNs for edits wrt to each CF.
458     long smallestSeqCF3PhaseII = region.getOldestSeqIdOfStore(FAMILY3);
459     long totalMemstoreSizePhaseII = region.getMemstoreSize();
460
461     /*------------------------------------------------------------------------------*/
462     /* PHASE II - validation */
463     // CF1 was flushed to memory, should be flattened and take less space
464     assertEquals(cf1MemstoreSizePhaseII.getDataSize() , cf1MemstoreSizePhaseI.getDataSize());
465     assertTrue(cf1MemstoreSizePhaseII.getHeapOverhead() < cf1MemstoreSizePhaseI.getHeapOverhead());
466     // CF2 should become empty
467     assertEquals(MemstoreSize.EMPTY_SIZE, cf2MemstoreSizePhaseII);
468     // verify that CF3 was flushed to memory and was not compacted (this is an approximation check)
469     // if compacted CF# should be at least twice less because its every key was duplicated
470     assertEquals(cf3MemstoreSizePhaseII.getDataSize() , cf3MemstoreSizePhaseI.getDataSize());
471     assertTrue(
472         cf3MemstoreSizePhaseI.getHeapOverhead() / 2 < cf3MemstoreSizePhaseII.getHeapOverhead());
473
474     // Now the smallest LSN in the region should be the same as the smallest
475     // LSN in the memstore of CF1.
476     assertEquals(smallestSeqInRegionCurrentMemstorePhaseII, smallestSeqCF1PhaseI);
477     // The total memstore size should be the same as the sum of the sizes of
478     // memstores of CF1, CF2 and CF3. Counting the empty active segments in CF1/2/3 and pipeline
479     // items in CF1/2
480     assertEquals(totalMemstoreSizePhaseII, cf1MemstoreSizePhaseII.getDataSize()
481         + cf2MemstoreSizePhaseII.getDataSize() + cf3MemstoreSizePhaseII.getDataSize());
482
483     /*------------------------------------------------------------------------------*/
484     /*------------------------------------------------------------------------------*/
485     /* PHASE III - insertions */
486     // Now add more puts for CF1, so that we also flush CF1 to disk instead of
487     // memory in next flush. This is causing the CF! to be flushed to memory twice.
488     for (int i = 1200; i < 8000; i++) {

496     }

499     /* PHASE III - collect sizes */
500     // How much does the CF1 memstore occupy now? Will be used later.
501     MemstoreSize cf1MemstoreSizePhaseIII = region.getStore(FAMILY1).getSizeOfMemStore();

514     // Second Flush in Test!!!!!!!!!!!!!!!!!!!!!!
515     // CF1 is flushed to disk, but not entirely emptied.
516     // CF2 was and remained empty, same way nothing happens to CF3
517     region.flush(false);
518
519     /*------------------------------------------------------------------------------*/
520     /*------------------------------------------------------------------------------*/
521     /* PHASE IV - collect sizes */
522     // Recalculate everything
523     MemstoreSize cf1MemstoreSizePhaseIV = region.getStore(FAMILY1).getSizeOfMemStore();
524     MemstoreSize cf2MemstoreSizePhaseIV = region.getStore(FAMILY2).getSizeOfMemStore();

530     /*------------------------------------------------------------------------------*/
531     /* PHASE IV - validation */
532     // CF1's biggest pipeline component (inserted before first flush) should be flushed to disk
533     // CF2 should remain empty
534     assertTrue(cf1MemstoreSizePhaseIII.getDataSize() > cf1MemstoreSizePhaseIV.getDataSize());
535     assertEquals(MemstoreSize.EMPTY_SIZE, cf2MemstoreSizePhaseIV);
536     // CF3 shouldn't have been touched.
537     assertEquals(cf3MemstoreSizePhaseIV, cf3MemstoreSizePhaseII);
538     // the smallest LSN of CF3 shouldn't change
539     assertEquals(smallestSeqCF3PhaseII, smallestSeqCF3PhaseIV);
540     // CF3 should be bottleneck for WAL
541     assertEquals(smallestSeqInRegionCurrentMemstorePhaseIV, smallestSeqCF3PhaseIV);
542
543     /*------------------------------------------------------------------------------*/
544     /* PHASE IV - Flush */
545     // Third Flush in Test!!!!!!!!!!!!!!!!!!!!!!
546     // Force flush to disk on all memstores (flush parameter true).
547     // CF1/CF3 all flushed to disk. Note that active sets of CF1 and CF3 are empty
548     region.flush(true);

550     /*------------------------------------------------------------------------------*/
551     /*------------------------------------------------------------------------------*/

567     assertEquals(0, totalMemstoreSizePhaseV);

569     assertEquals(smallestSeqInRegionCurrentMemstorePhaseV, HConstants.NO_SEQNUM);

571     // What happens when we hit the memstore limit, but we are not able to find
572     // any Column Family above the threshold?
573     // In that case, we should flush all the CFs.
574
575     /*------------------------------------------------------------------------------*/
576     /*------------------------------------------------------------------------------*/
577     /* PHASE VI - insertions */
578     // The memstore limit is 200*1024 and the column family flush threshold is
579     // around 50*1024. We try to just hit the memstore limit with each CF's

600     // store to flush, we should flush all the memstores
601     // Also compacted memstores are flushed to disk, but not entirely emptied

914   /**
915    * The in-memory-flusher thread performs the flush asynchronously. There is at most one thread per
916    * memstore instance. It takes the updatesLock exclusively, pushes active into the pipeline,
917    * releases updatesLock and compacts the pipeline.
918    */
919   private class ConcurrentPutRunnable implements Runnable {
920     private final HRegion stressedRegion;
921     private final int startNumber;
922
923     ConcurrentPutRunnable(HRegion r, int i) {
924       this.stressedRegion = r;
925       this.startNumber = i;
926     }
927
928     @Override
929     public void run() {
930
931       try {
932         int dummy = startNumber / 10000;
933         System.out.print("Thread " + dummy + " with start number " + startNumber + " starts\n");
934         // Add 1200 entries for CF1, 100 for CF2 and 50 for CF3
935         for (int i = startNumber; i <= startNumber + 3000; i++) {
936           stressedRegion.put(createPut(1, i));
937           if (i <= startNumber + 2000) {
938             stressedRegion.put(createPut(2, i));
939             if (i <= startNumber + 1000) {
940               stressedRegion.put(createPut(3, i));
941             }
942           }
943         }
944         System.out.print("Thread with start number " + startNumber + " continues to more puts\n");
945         // Now add more puts for CF2, so that we only flush CF2 to disk
946         for (int i = startNumber + 3000; i < startNumber + 5000; i++) {
947           stressedRegion.put(createPut(2, i));
948         }
949         // And add more puts for CF1
950         for (int i = startNumber + 5000; i < startNumber + 7000; i++) {
951           stressedRegion.put(createPut(1, i));
952         }
953         System.out.print("Thread with start number " + startNumber + " flushes\n");
954         // flush (IN MEMORY) one of the stores (each thread flushes different store)
955         // and wait till the flush and the following action are done
956         if (startNumber == 0) {
957           ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY1)).memstore)
958               .flushInMemory();
959           while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY1)).memstore)
960               .isMemStoreFlushingInMemory()) {
961             Threads.sleep(10);
962           }
963         }
964         if (startNumber == 10000) {
965           ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY2)).memstore).flushInMemory();
966           while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY2)).memstore)
967               .isMemStoreFlushingInMemory()) {
968             Threads.sleep(10);
969           }
970         }
971         if (startNumber == 20000) {
972           ((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY3)).memstore).flushInMemory();
973           while (((CompactingMemStore) ((HStore) stressedRegion.getStore(FAMILY3)).memstore)
974               .isMemStoreFlushingInMemory()) {
975             Threads.sleep(10);
976           }
977         }
978         System.out.print("Thread with start number " + startNumber + " finishes\n");
979       } catch (IOException e) {
980         assert false;
981       }
982     }
983   }
