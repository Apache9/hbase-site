80    public void testCompaction1Bucket() throws IOException {
81      int counter = 0;
82      String[] keys1 = { "A", "A", "B", "C" }; //A1, A2, B3, C4
83
84      // test 1 bucket
85      addRowsByKeys(memstore, keys1);
86      assertEquals(496, regionServicesForStores.getGlobalMemstoreTotalSize());
87
88      assertEquals(4, memstore.getActive().getCellsCount());
89      long size = memstore.getFlushableSize();
90      ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
91      while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
92        Threads.sleep(10);
93      }
94      assertEquals(0, memstore.getSnapshot().getCellsCount());
95      assertEquals(264, regionServicesForStores.getGlobalMemstoreTotalSize());
96      for ( Segment s : memstore.getSegments()) {
97        counter += s.getCellsCount();
98      }
99      assertEquals(3, counter);
100     size = memstore.getFlushableSize();
101     MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
102     region.addAndGetGlobalMemstoreSize(-size);  // simulate flusher
103     ImmutableSegment s = memstore.getSnapshot();
104     assertEquals(3, s.getCellsCount());
105     assertEquals(0, regionServicesForStores.getGlobalMemstoreTotalSize());
106
107     memstore.clearSnapshot(snapshot.getId());
108   }

180     while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
181       Threads.sleep(10);
182     }
183     assertEquals(0, memstore.getSnapshot().getCellsCount());
184     assertEquals(264, regionServicesForStores.getGlobalMemstoreTotalSize());
185
186     addRowsByKeys(memstore, keys2);
187
188     tstStr += " After adding second part of the keys. Memstore size: " +
189         region.getMemstoreSize() + ", Memstore Total Size: " +
190         regionServicesForStores.getGlobalMemstoreTotalSize() + "\n\n";
191
192     assertEquals(640, regionServicesForStores.getGlobalMemstoreTotalSize());
193
194     ((CompactingMemStore) memstore).disableCompaction();
195     size = memstore.getFlushableSize();
196     ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline without compaction
197     assertEquals(0, memstore.getSnapshot().getCellsCount());
198     assertEquals(640, regionServicesForStores.getGlobalMemstoreTotalSize());
199
200     addRowsByKeys(memstore, keys3);
201     assertEquals(1016, regionServicesForStores.getGlobalMemstoreTotalSize());
202
203     ((CompactingMemStore) memstore).enableCompaction();
204     size = memstore.getFlushableSize();
205     ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
206     while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
207       Threads.sleep(10);
208     }
209     assertEquals(0, memstore.getSnapshot().getCellsCount());
210     assertEquals(384, regionServicesForStores.getGlobalMemstoreTotalSize());
211
212     size = memstore.getFlushableSize();
213     MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
214     region.addAndGetGlobalMemstoreSize(-size);  // simulate flusher
215     ImmutableSegment s = memstore.getSnapshot();
216     assertEquals(4, s.getCellsCount());
217     assertEquals(0, regionServicesForStores.getGlobalMemstoreTotalSize());
218
219     memstore.clearSnapshot(snapshot.getId());

232
233     // set flattening to true
234     memstore.getConfiguration().setBoolean("hbase.hregion.compacting.memstore.flatten", true);
235
236     addRowsByKeys(memstore, keys1);
237
238     ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline should not compact
239
240     while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
241       Threads.sleep(10);
242     }
243     assertEquals(0, memstore.getSnapshot().getCellsCount());
244
245     addRowsByKeys(memstore, keys2); // also should only flatten
246
247     ((CompactingMemStore) memstore).disableCompaction();
248
249     ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline without flattening
250     assertEquals(0, memstore.getSnapshot().getCellsCount());
251
252     addRowsByKeys(memstore, keys3);
253
254     ((CompactingMemStore) memstore).enableCompaction();
255
256
257     ((CompactingMemStore) memstore).flushInMemory(); // push keys to pipeline and compact
258     while (((CompactingMemStore) memstore).isMemStoreFlushingInMemory()) {
259       Threads.sleep(10);
260     }
261     assertEquals(0, memstore.getSnapshot().getCellsCount());
262
263     int counter = 0;
264     for ( Segment s : memstore.getSegments()) {
265       counter += s.getCellsCount();
266     }
267     assertEquals(10,counter);
268
269     MemStoreSnapshot snapshot = memstore.snapshot(); // push keys to snapshot
270     ImmutableSegment s = memstore.getSnapshot();
271     memstore.clearSnapshot(snapshot.getId());

318   private void addRowsByKeysWith50Cols(AbstractMemStore hmc, String[] keys) {
319     byte[] fam = Bytes.toBytes("testfamily");
320     for (int i = 0; i < keys.length; i++) {
321       long timestamp = System.currentTimeMillis();
322       Threads.sleep(1); // to make sure each kv gets a different ts
323       byte[] row = Bytes.toBytes(keys[i]);
324       for(int  j =0 ;j < 50; j++) {
325         byte[] qf = Bytes.toBytes("testqualifier"+j);
326         byte[] val = Bytes.toBytes(keys[i] + j);
327         KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
328         hmc.add(kv);
329       }
330     }

333   private void addRowsByKeys(final AbstractMemStore hmc, String[] keys) {
334     byte[] fam = Bytes.toBytes("testfamily");
335     byte[] qf = Bytes.toBytes("testqualifier");
336     long size = hmc.getActive().getSize();//
337     for (int i = 0; i < keys.length; i++) {
338       long timestamp = System.currentTimeMillis();
339       Threads.sleep(1); // to make sure each kv gets a different ts
340       byte[] row = Bytes.toBytes(keys[i]);
341       byte[] val = Bytes.toBytes(keys[i] + i);
342       KeyValue kv = new KeyValue(row, fam, qf, timestamp, val);
343       hmc.add(kv);
344       LOG.debug("added kv: " + kv.getKeyString() + ", timestamp" + kv.getTimestamp());
345     }
346     regionServicesForStores.addAndGetGlobalMemstoreSize(hmc.getActive().getSize() - size);//
