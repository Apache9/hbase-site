80    /* Generic basic test for immutable CellSet */
81    private void testCellBlocks(CellSet cs) throws Exception {
82      final byte[] oneAndHalf = Bytes.toBytes(20);
83      final byte[] f = Bytes.toBytes("f");
84      final byte[] q = Bytes.toBytes("q");
85      final byte[] v = Bytes.toBytes(4);
86      final KeyValue outerCell = new KeyValue(oneAndHalf, f, q, 10, v);
87
88      assertEquals(NUM_OF_CELLS, cs.size());          // check size
89      assertFalse(cs.contains(outerCell));            // check outer cell
90
91      assertTrue(cs.contains(cells[0]));              // check existence of the first
92      Cell first = cs.first();
93      assertTrue(cells[0].equals(first));
94
95      assertTrue(cs.contains(cells[NUM_OF_CELLS - 1]));  // check last
96      Cell last = cs.last();
97      assertTrue(cells[NUM_OF_CELLS - 1].equals(last));
98
99      SortedSet<Cell> tail = cs.tailSet(cells[1]);    // check tail abd head sizes
100     assertEquals(NUM_OF_CELLS - 1, tail.size());
101     SortedSet<Cell> head = cs.headSet(cells[1]);
102     assertEquals(1, head.size());
103
104     SortedSet<Cell> tailOuter = cs.tailSet(outerCell);  // check tail starting from outer cell
105     assertEquals(NUM_OF_CELLS - 1, tailOuter.size());
106
107     Cell tailFirst = tail.first();
108     assertTrue(cells[1].equals(tailFirst));
109     Cell tailLast = tail.last();
110     assertTrue(cells[NUM_OF_CELLS - 1].equals(tailLast));
111
112     Cell headFirst = head.first();
113     assertTrue(cells[0].equals(headFirst));
114     Cell headLast = head.last();
115     assertTrue(cells[0].equals(headLast));
116   }

118   /* Generic iterators test for immutable CellSet */
119   private void testIterators(CellSet cs) throws Exception {
120
121     // Assert that we have NUM_OF_CELLS values and that they are in order
122     int count = 0;
123     for (Cell kv: cs) {
124       assertEquals("\n\n-------------------------------------------------------------------\n"
125               + "Comparing iteration number " + (count + 1) + " the returned cell: " + kv
126               + ", the first Cell in the CellBlocksMap: " + cells[count]
127               + ", and the same transformed to String: " + cells[count].toString()
128               + "\n-------------------------------------------------------------------\n",
129               cells[count], kv);
130       count++;
131     }
132     assertEquals(NUM_OF_CELLS, count);
133
134     // Test descending iterator
135     count = 0;
136     for (Iterator<Cell> i = cs.descendingIterator(); i.hasNext();) {
137       Cell kv = i.next();
138       assertEquals(cells[NUM_OF_CELLS - (count + 1)], kv);
139       count++;
140     }
141     assertEquals(NUM_OF_CELLS, count);
142   }
