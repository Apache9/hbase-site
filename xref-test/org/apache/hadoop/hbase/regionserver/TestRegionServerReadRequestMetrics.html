115   private static void testReadRequests(long resultCount,
116     long expectedReadRequests, long expectedFilteredReadRequests)
117     throws IOException, InterruptedException {
118     updateMetricsMap();
119     System.out.println("requestsMapPrev = " + requestsMapPrev);
120     System.out.println("requestsMap = " + requestsMap);
121
122     assertEquals(expectedReadRequests,
123       requestsMap.get(Metric.REGION_READ) - requestsMapPrev.get(Metric.REGION_READ));
124     assertEquals(expectedReadRequests,
125       requestsMap.get(Metric.SERVER_READ) - requestsMapPrev.get(Metric.SERVER_READ));
126     assertEquals(expectedFilteredReadRequests,
127       requestsMap.get(Metric.FILTERED_REGION_READ)
128         - requestsMapPrev.get(Metric.FILTERED_REGION_READ));
129     assertEquals(expectedFilteredReadRequests,
130       requestsMap.get(Metric.FILTERED_SERVER_READ)
131         - requestsMapPrev.get(Metric.FILTERED_SERVER_READ));
132     assertEquals(expectedReadRequests, resultCount);
133   }

135   private static void updateMetricsMap() throws IOException, InterruptedException {
136     for (Metric metric : Metric.values()) {
137       requestsMapPrev.put(metric, requestsMap.get(metric));
138     }
139
140     ServerLoad serverLoad = null;
141     RegionLoad regionLoadOuter = null;
142     boolean metricsUpdated = false;
143     for (int i = 0; i < MAX_TRY; i++) {
144       for (ServerName serverName : serverNames) {
145         serverLoad = admin.getClusterStatus().getLoad(serverName);
146
147         Map<byte[], RegionLoad> regionsLoad = serverLoad.getRegionsLoad();
148         for (HRegionInfo tableRegion : tableRegions) {
149           RegionLoad regionLoad = regionsLoad.get(tableRegion.getRegionName());
150           if (regionLoad != null) {
151             regionLoadOuter = regionLoad;
152             for (Metric metric : Metric.values()) {
153               if (getReadRequest(serverLoad, regionLoad, metric) > requestsMapPrev.get(metric)) {
154                 for (Metric metricInner : Metric.values()) {
155                   requestsMap.put(metricInner, getReadRequest(serverLoad, regionLoad, metricInner));
156                 }
157                 metricsUpdated = true;
158                 break;
159               }
160             }
161           }
162         }
163       }
164       if (metricsUpdated) {
165         break;
166       }
167       Thread.sleep(SLEEP_MS);
168     }
169     if (!metricsUpdated) {
170       for (Metric metric : Metric.values()) {
171         requestsMap.put(metric, getReadRequest(serverLoad, regionLoadOuter, metric));
172       }
173     }
174   }

176   private static long getReadRequest(ServerLoad serverLoad, RegionLoad regionLoad, Metric metric) {
177     switch (metric) {
178       case REGION_READ:
179         return regionLoad.getReadRequestsCount();
180       case SERVER_READ:
181         return serverLoad.getReadRequestsCount();
182       case FILTERED_REGION_READ:
183         return regionLoad.getFilteredReadRequestsCount();
184       case FILTERED_SERVER_READ:
185         return serverLoad.getFilteredReadRequestsCount();
186       default:
187         throw new IllegalStateException();
188     }
189   }

235   @Test
236   public void testReadRequestsCountNotFiltered() throws Exception {
237     int resultCount;
238     Scan scan;
239     Append append;
240     Put put;
241     Increment increment;
242     Get get;
243
244     // test for scan
245     scan = new Scan();
246     try (ResultScanner scanner = table.getScanner(scan)) {
247       resultCount = 0;
248       for (Result ignore : scanner) {
249         resultCount++;
250       }
251       testReadRequests(resultCount, 3, 0);
252     }
253
254     // test for scan
255     scan = new Scan(ROW2, ROW3);
256     try (ResultScanner scanner = table.getScanner(scan)) {
257       resultCount = 0;
258       for (Result ignore : scanner) {
259         resultCount++;
260       }
261       testReadRequests(resultCount, 1, 0);
262     }
263
264     // test for get
265     get = new Get(ROW2);
266     Result result = table.get(get);
267     resultCount = result.isEmpty() ? 0 : 1;
268     testReadRequests(resultCount, 1, 0);
269
270     // test for increment
271     increment = new Increment(ROW1);
272     increment.addColumn(CF1, COL3, 1);
273     result = table.increment(increment);
274     resultCount = result.isEmpty() ? 0 : 1;
275     testReadRequests(resultCount, 1, 0);
276
277     // test for checkAndPut
278     put = new Put(ROW1);
279     put.addColumn(CF1, COL2, VAL2);
280     boolean checkAndPut =
281       table.checkAndPut(ROW1, CF1, COL2, CompareFilter.CompareOp.EQUAL, VAL2, put);
282     resultCount = checkAndPut ? 1 : 0;
283     testReadRequests(resultCount, 1, 0);
284
285     // test for append
286     append = new Append(ROW1);
287     append.add(CF1, COL2, VAL2);
288     result = table.append(append);
289     resultCount = result.isEmpty() ? 0 : 1;
290     testReadRequests(resultCount, 1, 0);
291
292     // test for checkAndMutate
293     put = new Put(ROW1);
294     put.addColumn(CF1, COL1, VAL1);
295     RowMutations rm = new RowMutations(ROW1);
296     rm.add(put);
297     boolean checkAndMutate =
298       table.checkAndMutate(ROW1, CF1, COL1, CompareFilter.CompareOp.EQUAL, VAL1, rm);
299     resultCount = checkAndMutate ? 1 : 0;
300     testReadRequests(resultCount, 1, 0);
301   }
