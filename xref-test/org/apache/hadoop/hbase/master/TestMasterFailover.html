65    HRegion createRegion(final HRegionInfo  hri, final Path rootdir, final Configuration c,
66        final HTableDescriptor htd)
67    throws IOException {
68      HRegion r = HBaseTestingUtility.createRegionAndWAL(hri, rootdir, c, htd);
69      // The above call to create a region will create an wal file.  Each
70      // log file create will also create a running thread to do syncing.  We need
71      // to close out this log else we will have a running thread trying to sync
72      // the file system continuously which is ugly when dfs is taken away at the
73      // end of the test.
74      HBaseTestingUtility.closeRegionAndWAL(r);
75      return r;
76    }

85    /**
86     * Simple test of master failover.
87     * <p>
88     * Starts with three masters.  Kills a backup master.  Then kills the active
89     * master.  Ensures the final master becomes active and we can still contact
90     * the cluster.
91     * @throws Exception
92     */
93    @Test (timeout=240000)
94    public void testSimpleMasterFailover() throws Exception {
95  
96      final int NUM_MASTERS = 3;
97      final int NUM_RS = 3;
98  
99      // Start the cluster
100     HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
101 
102     TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
103     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
104 
105     // get all the master threads
106     List<MasterThread> masterThreads = cluster.getMasterThreads();
107 
108     // wait for each to come online
109     for (MasterThread mt : masterThreads) {
110       assertTrue(mt.isAlive());
111     }
112 
113     // verify only one is the active master and we have right number
114     int numActive = 0;
115     int activeIndex = -1;
116     ServerName activeName = null;
117     HMaster active = null;
118     for (int i = 0; i < masterThreads.size(); i++) {
119       if (masterThreads.get(i).getMaster().isActiveMaster()) {
120         numActive++;
121         activeIndex = i;
122         active = masterThreads.get(activeIndex).getMaster();
123         activeName = active.getServerName();
124       }
125     }
126     assertEquals(1, numActive);
127     assertEquals(NUM_MASTERS, masterThreads.size());
128     LOG.info("Active master " + activeName);
129 
130     // Check that ClusterStatus reports the correct active and backup masters
131     assertNotNull(active);
132     ClusterStatus status = active.getClusterStatus();
133     assertTrue(status.getMaster().equals(activeName));
134     assertEquals(2, status.getBackupMastersSize());
135     assertEquals(2, status.getBackupMasters().size());
136 
137     // attempt to stop one of the inactive masters
138     int backupIndex = (activeIndex == 0 ? 1 : activeIndex - 1);
139     HMaster master = cluster.getMaster(backupIndex);
140     LOG.debug("\n\nStopping a backup master: " + master.getServerName() + "\n");
141     cluster.stopMaster(backupIndex, false);
142     cluster.waitOnMaster(backupIndex);
143 
144     // Verify still one active master and it's the same
145     for (int i = 0; i < masterThreads.size(); i++) {
146       if (masterThreads.get(i).getMaster().isActiveMaster()) {
147         assertTrue(activeName.equals(masterThreads.get(i).getMaster().getServerName()));
148         activeIndex = i;
149         active = masterThreads.get(activeIndex).getMaster();
150       }
151     }
152     assertEquals(1, numActive);
153     assertEquals(2, masterThreads.size());
154     int rsCount = masterThreads.get(activeIndex).getMaster().getClusterStatus().getServersSize();
155     LOG.info("Active master " + active.getServerName() + " managing " + rsCount +  " regions servers");
156     assertEquals(4, rsCount);
157 
158     // Check that ClusterStatus reports the correct active and backup masters
159     assertNotNull(active);
160     status = active.getClusterStatus();
161     assertTrue(status.getMaster().equals(activeName));
162     assertEquals(1, status.getBackupMastersSize());
163     assertEquals(1, status.getBackupMasters().size());
164 
165     // kill the active master
166     LOG.debug("\n\nStopping the active master " + active.getServerName() + "\n");
167     cluster.stopMaster(activeIndex, false);
168     cluster.waitOnMaster(activeIndex);
169 
170     // wait for an active master to show up and be ready
171     assertTrue(cluster.waitForActiveAndReadyMaster());
172 
173     LOG.debug("\n\nVerifying backup master is now active\n");
174     // should only have one master now
175     assertEquals(1, masterThreads.size());
176 
177     // and he should be active
178     active = masterThreads.get(0).getMaster();
179     assertNotNull(active);
180     status = active.getClusterStatus();
181     ServerName mastername = status.getMaster();
182     assertTrue(mastername.equals(active.getServerName()));
183     assertTrue(active.isActiveMaster());
184     assertEquals(0, status.getBackupMastersSize());
185     assertEquals(0, status.getBackupMasters().size());
186     int rss = status.getServersSize();
187     LOG.info("Active master " + mastername.getServerName() + " managing " +
188       rss +  " region servers");
189     assertEquals(4, rss);
190 
191     // Stop the cluster
192     TEST_UTIL.shutdownMiniCluster();
193   }

195   /**
196    * Test region in pending_open/close when master failover
197    */
198   @Test (timeout=180000)
199   public void testPendingOpenOrCloseWhenMasterFailover() throws Exception {
200     final int NUM_MASTERS = 1;
201     final int NUM_RS = 1;
202 
203     // Create config to use for this cluster
204     Configuration conf = HBaseConfiguration.create();
205 
206     // Start the cluster
207     HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
208     TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
209     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
210     log("Cluster started");
211 
212     // get all the master threads
213     List<MasterThread> masterThreads = cluster.getMasterThreads();
214     assertEquals(1, masterThreads.size());
215 
216     // only one master thread, let's wait for it to be initialized
217     assertTrue(cluster.waitForActiveAndReadyMaster());
218     HMaster master = masterThreads.get(0).getMaster();
219     assertTrue(master.isActiveMaster());
220     assertTrue(master.isInitialized());
221 
222     // Create a table with a region online
223     Table onlineTable = TEST_UTIL.createTable(TableName.valueOf("onlineTable"), "family");
224     onlineTable.close();
225     // Create a table in META, so it has a region offline
226     HTableDescriptor offlineTable = new HTableDescriptor(
227       TableName.valueOf(Bytes.toBytes("offlineTable")));
228     offlineTable.addFamily(new HColumnDescriptor(Bytes.toBytes("family")));
229 
230     FileSystem filesystem = FileSystem.get(conf);
231     Path rootdir = FSUtils.getRootDir(conf);
232     FSTableDescriptors fstd = new FSTableDescriptors(conf, filesystem, rootdir);
233     fstd.createTableDescriptor(offlineTable);
234 
235     HRegionInfo hriOffline = new HRegionInfo(offlineTable.getTableName(), null, null);
236     createRegion(hriOffline, rootdir, conf, offlineTable);
237     MetaTableAccessor.addRegionToMeta(master.getConnection(), hriOffline);
238 
239     log("Regions in hbase:meta and namespace have been created");
240 
241     // at this point we only expect 3 regions to be assigned out
242     // (catalogs and namespace, + 1 online region)
243     assertEquals(3, cluster.countServedRegions());
244     HRegionInfo hriOnline = null;
245     try (RegionLocator locator =
246         TEST_UTIL.getConnection().getRegionLocator(TableName.valueOf("onlineTable"))) {
247       hriOnline = locator.getRegionLocation(HConstants.EMPTY_START_ROW).getRegionInfo();
248     }
249     RegionStates regionStates = master.getAssignmentManager().getRegionStates();
250     RegionStateStore stateStore = master.getAssignmentManager().getRegionStateStore();
251 
252     // Put the online region in pending_close. It is actually already opened.
253     // This is to simulate that the region close RPC is not sent out before failover
254     RegionState oldState = regionStates.getRegionState(hriOnline);
255     RegionState newState = new RegionState(
256       hriOnline, State.PENDING_CLOSE, oldState.getServerName());
257     stateStore.updateRegionState(HConstants.NO_SEQNUM, newState, oldState);
258 
259     // Put the offline region in pending_open. It is actually not opened yet.
260     // This is to simulate that the region open RPC is not sent out before failover
261     oldState = new RegionState(hriOffline, State.OFFLINE);
262     newState = new RegionState(hriOffline, State.PENDING_OPEN, newState.getServerName());
263     stateStore.updateRegionState(HConstants.NO_SEQNUM, newState, oldState);
264 
265     HRegionInfo failedClose = new HRegionInfo(offlineTable.getTableName(), null, null);
266     createRegion(failedClose, rootdir, conf, offlineTable);
267     MetaTableAccessor.addRegionToMeta(master.getConnection(), failedClose);
268 
269     oldState = new RegionState(failedClose, State.PENDING_CLOSE);
270     newState = new RegionState(failedClose, State.FAILED_CLOSE, newState.getServerName());
271     stateStore.updateRegionState(HConstants.NO_SEQNUM, newState, oldState);
272 
273     HRegionInfo failedOpen = new HRegionInfo(offlineTable.getTableName(), null, null);
274     createRegion(failedOpen, rootdir, conf, offlineTable);
275     MetaTableAccessor.addRegionToMeta(master.getConnection(), failedOpen);
276 
277     // Simulate a region transitioning to failed open when the region server reports the
278     // transition as FAILED_OPEN
279     oldState = new RegionState(failedOpen, State.PENDING_OPEN);
280     newState = new RegionState(failedOpen, State.FAILED_OPEN, newState.getServerName());
281     stateStore.updateRegionState(HConstants.NO_SEQNUM, newState, oldState);
282 
283     HRegionInfo failedOpenNullServer = new HRegionInfo(offlineTable.getTableName(), null, null);
284     LOG.info("Failed open NUll server " + failedOpenNullServer.getEncodedName());
285     createRegion(failedOpenNullServer, rootdir, conf, offlineTable);
286     MetaTableAccessor.addRegionToMeta(master.getConnection(), failedOpenNullServer);
287 
288     // Simulate a region transitioning to failed open when the master couldn't find a plan for
289     // the region
290     oldState = new RegionState(failedOpenNullServer, State.OFFLINE);
291     newState = new RegionState(failedOpenNullServer, State.FAILED_OPEN, null);
292     stateStore.updateRegionState(HConstants.NO_SEQNUM, newState, oldState);
293 
294     // Stop the master
295     log("Aborting master");
296     cluster.abortMaster(0);
297     cluster.waitOnMaster(0);
298     log("Master has aborted");
299 
300     // Start up a new master
301     log("Starting up a new master");
302     master = cluster.startMaster().getMaster();
303     log("Waiting for master to be ready");
304     cluster.waitForActiveAndReadyMaster();
305     log("Master is ready");
306 
307     // Wait till no region in transition any more
308     TEST_UTIL.waitUntilNoRegionsInTransition(60000);
309 
310     // Get new region states since master restarted
311     regionStates = master.getAssignmentManager().getRegionStates();
312 
313     // Both pending_open (RPC sent/not yet) regions should be online
314     assertTrue(regionStates.isRegionOnline(hriOffline));
315     assertTrue(regionStates.isRegionOnline(hriOnline));
316     assertTrue(regionStates.isRegionOnline(failedClose));
317     assertTrue(regionStates.isRegionOnline(failedOpenNullServer));
318     assertTrue(regionStates.isRegionOnline(failedOpen));
319 
320     log("Done with verification, shutting down cluster");
321 
322     // Done, shutdown the cluster
323     TEST_UTIL.shutdownMiniCluster();
324   }

326   /**
327    * Test meta in transition when master failover
328    */
329   @Test(timeout = 180000)
330   public void testMetaInTransitionWhenMasterFailover() throws Exception {
331     final int NUM_MASTERS = 1;
332     final int NUM_RS = 1;
333 
334     // Start the cluster
335     HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
336     TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
337     MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
338     log("Cluster started");
339 
340     log("Moving meta off the master");
341     HMaster activeMaster = cluster.getMaster();
342     HRegionServer rs = cluster.getRegionServer(0);
343     ServerName metaServerName = cluster.getLiveRegionServerThreads()
344       .get(0).getRegionServer().getServerName();
345     activeMaster.move(HRegionInfo.FIRST_META_REGIONINFO.getEncodedNameAsBytes(),
346       Bytes.toBytes(metaServerName.getServerName()));
347     TEST_UTIL.waitUntilNoRegionsInTransition(60000);
348     assertEquals("Meta should be assigned on expected regionserver",
349       metaServerName, activeMaster.getMetaTableLocator()
350         .getMetaRegionLocation(activeMaster.getZooKeeper()));
351 
352     // Now kill master, meta should remain on rs, where we placed it before.
353     log("Aborting master");
354     activeMaster.abort("test-kill");
355     cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
356     log("Master has aborted");
357 
358     // meta should remain where it was
359     RegionState metaState =
360       MetaTableLocator.getMetaRegionState(rs.getZooKeeper());
361     assertEquals("hbase:meta should be onlined on RS",
362       metaState.getServerName(), rs.getServerName());
363     assertEquals("hbase:meta should be onlined on RS",
364       metaState.getState(), State.OPEN);
365 
366     // Start up a new master
367     log("Starting up a new master");
368     activeMaster = cluster.startMaster().getMaster();
369     log("Waiting for master to be ready");
370     cluster.waitForActiveAndReadyMaster();
371     log("Master is ready");
372 
373     // ensure meta is still deployed on RS
374     metaState =
375       MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
376     assertEquals("hbase:meta should be onlined on RS",
377       metaState.getServerName(), rs.getServerName());
378     assertEquals("hbase:meta should be onlined on RS",
379       metaState.getState(), State.OPEN);
380 
381     // Update meta state as PENDING_OPEN, then kill master
382     // that simulates, that RS successfully deployed, but
383     // RPC was lost right before failure.
384     // region server should expire (how it can be verified?)
385     MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
386       rs.getServerName(), State.PENDING_OPEN);
387     Region meta = rs.getFromOnlineRegions(HRegionInfo.FIRST_META_REGIONINFO.getEncodedName());
388     rs.removeFromOnlineRegions(meta, null);
389     ((HRegion)meta).close();
390 
391     log("Aborting master");
392     activeMaster.abort("test-kill");
393     cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
394     log("Master has aborted");
395 
396     // Start up a new master
397     log("Starting up a new master");
398     activeMaster = cluster.startMaster().getMaster();
399     log("Waiting for master to be ready");
400     cluster.waitForActiveAndReadyMaster();
401     log("Master is ready");
402 
403     TEST_UTIL.waitUntilNoRegionsInTransition(60000);
404     log("Meta was assigned");
405 
406     metaState =
407       MetaTableLocator.getMetaRegionState(activeMaster.getZooKeeper());
408     assertEquals("hbase:meta should be onlined on RS",
409       metaState.getServerName(), rs.getServerName());
410     assertEquals("hbase:meta should be onlined on RS",
411       metaState.getState(), State.OPEN);
412 
413     // Update meta state as PENDING_CLOSE, then kill master
414     // that simulates, that RS successfully deployed, but
415     // RPC was lost right before failure.
416     // region server should expire (how it can be verified?)
417     MetaTableLocator.setMetaLocation(activeMaster.getZooKeeper(),
418       rs.getServerName(), State.PENDING_CLOSE);
419 
420     log("Aborting master");
421     activeMaster.abort("test-kill");
422     cluster.waitForMasterToStop(activeMaster.getServerName(), 30000);
423     log("Master has aborted");
424 
425     rs.getRSRpcServices().closeRegion(null, ProtobufUtil.buildCloseRegionRequest(
426       rs.getServerName(), HRegionInfo.FIRST_META_REGIONINFO.getEncodedName()));
427 
428     // Start up a new master
429     log("Starting up a new master");
430     activeMaster = cluster.startMaster().getMaster();
431     log("Waiting for master to be ready");
432     cluster.waitForActiveAndReadyMaster();
433     log("Master is ready");
434 
435     TEST_UTIL.waitUntilNoRegionsInTransition(60000);
436     log("Meta was assigned");
437 
438     // Done, shutdown the cluster
439     TEST_UTIL.shutdownMiniCluster();
440   }
