32  @Category({ IOTests.class, SmallTests.class })
33  public class TestByteBufferListOutputStream {
34
35    @Test
36    public void testWrites() throws Exception {
37      ByteBufferPool pool = new ByteBufferPool(10, 3);
38      ByteBufferListOutputStream bbos = new ByteBufferListOutputStream(pool);
39      bbos.write(2);// Write a byte
40      bbos.writeInt(100);// Write an int
41      byte[] b = Bytes.toBytes("row123");// 6 bytes
42      bbos.write(b);
43      // Just use the 3rd BB from pool so that pabos, on request, wont get one
44      ByteBuffer bb1 = pool.getBuffer();
45      ByteBuffer bb = ByteBuffer.wrap(Bytes.toBytes("row123_cf1_q1"));// 13 bytes
46      bbos.write(bb, 0, bb.capacity());
47      pool.putbackBuffer(bb1);
48      bbos.writeInt(123);
49      bbos.writeInt(124);
50      assertEquals(0, pool.getQueueSize());
51      List<ByteBuffer> allBufs = bbos.getByteBuffers();
52      assertEquals(4, allBufs.size());
53      assertEquals(3, bbos.bufsFromPool.size());
54      ByteBuffer b1 = allBufs.get(0);
55      assertEquals(10, b1.remaining());
56      assertEquals(2, b1.get());
57      assertEquals(100, b1.getInt());
58      byte[] bActual = new byte[b.length];
59      b1.get(bActual, 0, 5);//5 bytes in 1st BB
60      ByteBuffer b2 = allBufs.get(1);
61      assertEquals(10, b2.remaining());
62      b2.get(bActual, 5, 1);// Remaining 1 byte in 2nd BB
63      assertTrue(Bytes.equals(b, bActual));
64      bActual = new byte[bb.capacity()];
65      b2.get(bActual, 0, 9);
66      ByteBuffer b3 = allBufs.get(2);
67      assertEquals(8, b3.remaining());
68      b3.get(bActual, 9, 4);
69      assertTrue(ByteBufferUtils.equals(bb, 0, bb.capacity(), bActual, 0, bActual.length));
70      assertEquals(123, b3.getInt());
71      ByteBuffer b4 = allBufs.get(3);
72      assertEquals(4, b4.remaining());
73      assertEquals(124, b4.getInt());
74      bbos.releaseResources();
75      assertEquals(3, pool.getQueueSize());
76    }
