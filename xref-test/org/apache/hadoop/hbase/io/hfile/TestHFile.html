124   public static void truncateFile(FileSystem fs, Path src, Path dst) throws IOException {
125     FileStatus fst = fs.getFileStatus(src);
126     long len = fst.getLen();
127     len = len / 2 ;
128 
129     // create a truncated hfile
130     FSDataOutputStream fdos = fs.create(dst);
131     byte[] buf = new byte[(int)len];
132     FSDataInputStream fdis = fs.open(src);
133     fdis.read(buf);
134     fdos.write(buf);
135     fdis.close();
136     fdos.close();
137   }

164   // write some records into the hfile
165   // write them twice
166   private int writeSomeRecords(Writer writer, int start, int n, boolean useTags)
167       throws IOException {
168     String value = "value";
169     KeyValue kv;
170     for (int i = start; i < (start + n); i++) {
171       String key = String.format(localFormatter, Integer.valueOf(i));
172       if (useTags) {
173         Tag t = new ArrayBackedTag((byte) 1, "myTag1");
174         Tag[] tags = new Tag[1];
175         tags[0] = t;
176         kv = new KeyValue(Bytes.toBytes(key), Bytes.toBytes("family"), Bytes.toBytes("qual"),
177             HConstants.LATEST_TIMESTAMP, Bytes.toBytes(value + key), tags);
178         writer.append(kv);
179       } else {
180         kv = new KeyValue(Bytes.toBytes(key), Bytes.toBytes("family"), Bytes.toBytes("qual"),
181             Bytes.toBytes(value + key));
182         writer.append(kv);
183       }
184     }
185     return (start + n);
186   }

192   // read the records and check
193   private int readAndCheckbytes(HFileScanner scanner, int start, int n)
194       throws IOException {
195     String value = "value";
196     int i = start;
197     for (; i < (start + n); i++) {
198       ByteBuffer key = ByteBuffer.wrap(((KeyValue)scanner.getKey()).getKey());
199       ByteBuffer val = scanner.getValue();
200       String keyStr = String.format(localFormatter, Integer.valueOf(i));
201       String valStr = value + keyStr;
202       KeyValue kv = new KeyValue(Bytes.toBytes(keyStr), Bytes.toBytes("family"),
203           Bytes.toBytes("qual"), Bytes.toBytes(valStr));
204       byte[] keyBytes = new KeyValue.KeyOnlyKeyValue(Bytes.toBytes(key), 0,
205           Bytes.toBytes(key).length).getKey();
206       assertTrue("bytes for keys do not match " + keyStr + " " +
207         Bytes.toString(Bytes.toBytes(key)),
208           Arrays.equals(kv.getKey(), keyBytes));
209       byte [] valBytes = Bytes.toBytes(val);
210       assertTrue("bytes for vals do not match " + valStr + " " +
211         Bytes.toString(valBytes),
212         Arrays.equals(Bytes.toBytes(valStr), valBytes));
213       if (!scanner.next()) {
214         break;
215       }
216     }
217     assertEquals(i, start + n - 1);
218     return (start + n);
219   }

238   /**
239    * test none codecs
240    * @param useTags
241    */
242   void basicWithSomeCodec(String codec, boolean useTags) throws IOException {
243     if (useTags) {
244       conf.setInt("hfile.format.version", 3);
245     }
246     if (cacheConf == null) cacheConf = new CacheConfig(conf);
247     Path  ncHFile = new Path(ROOT_DIR, "basic.hfile." + codec.toString() + useTags);
248     FSDataOutputStream fout = createFSOutput(ncHFile);
249     HFileContext meta = new HFileContextBuilder()
250                         .withBlockSize(minBlockSize)
251                         .withCompression(HFileWriterImpl.compressionByName(codec))
252                         .build();
253     Writer writer = HFile.getWriterFactory(conf, cacheConf)
254         .withOutputStream(fout)
255         .withFileContext(meta)
256         .withComparator(CellComparator.COMPARATOR)
257         .create();
258     LOG.info(writer);
259     writeRecords(writer, useTags);
260     fout.close();
261     FSDataInputStream fin = fs.open(ncHFile);
262     Reader reader = HFile.createReaderFromStream(ncHFile, fs.open(ncHFile),
263       fs.getFileStatus(ncHFile).getLen(), cacheConf, conf);
264     System.out.println(cacheConf.toString());
265     // Load up the index.
266     reader.loadFileInfo();
267     // Get a scanner that caches and that does not use pread.
268     HFileScanner scanner = reader.getScanner(true, false);
269     // Align scanner at start of the file.
270     scanner.seekTo();
271     readAllRecords(scanner);
272     int seekTo = scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(50)));
273     System.out.println(seekTo);
274     assertTrue("location lookup failed",
275         scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(50))) == 0);
276     // read the key and see if it matches
277     ByteBuffer readKey = ByteBuffer.wrap(((KeyValue)scanner.getKey()).getKey());
278     assertTrue("seeked key does not match", Arrays.equals(getSomeKey(50),
279       Bytes.toBytes(readKey)));
280 
281     scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(0)));
282     ByteBuffer val1 = scanner.getValue();
283     scanner.seekTo(KeyValueUtil.createKeyValueFromKey(getSomeKey(0)));
284     ByteBuffer val2 = scanner.getValue();
285     assertTrue(Arrays.equals(Bytes.toBytes(val1), Bytes.toBytes(val2)));
286 
287     reader.close();
288     fin.close();
289     fs.delete(ncHFile, true);
290   }

303   private void writeNumMetablocks(Writer writer, int n) {
304     for (int i = 0; i < n; i++) {
305       writer.appendMetaBlock("HFileMeta" + i, new Writable() {
306         private int val;
307         public Writable setVal(int val) { this.val = val; return this; }
308 
309         @Override
310         public void write(DataOutput out) throws IOException {
311           out.write(("something to test" + val).getBytes());
312         }
313 
314         @Override
315         public void readFields(DataInput in) throws IOException { }
316       }.setVal(i));
317     }
318   }

324   private void readNumMetablocks(Reader reader, int n) throws IOException {
325     for (int i = 0; i < n; i++) {
326       ByteBuff actual = reader.getMetaBlock("HFileMeta" + i, false).getBufferWithoutHeader();
327       ByteBuffer expected =
328         ByteBuffer.wrap(("something to test" + i).getBytes());
329       assertEquals(
330           "failed to match metadata",
331           Bytes.toStringBinary(expected),
332           Bytes.toStringBinary(actual.array(), actual.arrayOffset() + actual.position(),
333               actual.capacity()));
334     }

341   private void metablocks(final String compress) throws Exception {
342     if (cacheConf == null) cacheConf = new CacheConfig(conf);
343     Path mFile = new Path(ROOT_DIR, "meta.hfile");
344     FSDataOutputStream fout = createFSOutput(mFile);
345     HFileContext meta = new HFileContextBuilder()
346                         .withCompression(HFileWriterImpl.compressionByName(compress))
347                         .withBlockSize(minBlockSize).build();
348     Writer writer = HFile.getWriterFactory(conf, cacheConf)
349         .withOutputStream(fout)
350         .withFileContext(meta)
351         .create();
352     someTestingWithMetaBlock(writer);
353     writer.close();
354     fout.close();
355     FSDataInputStream fin = fs.open(mFile);
356     Reader reader = HFile.createReaderFromStream(mFile, fs.open(mFile),
357         this.fs.getFileStatus(mFile).getLen(), cacheConf, conf);
358     reader.loadFileInfo();
359     // No data -- this should return false.
360     assertFalse(reader.getScanner(false, false).seekTo());
361     someReadingWithMetaBlock(reader);
362     fs.delete(mFile, true);
363     reader.close();
364     fin.close();
365   }
