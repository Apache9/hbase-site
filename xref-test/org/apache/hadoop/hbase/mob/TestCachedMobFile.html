42  @Category(SmallTests.class)
43  public class TestCachedMobFile extends TestCase{
44    static final Log LOG = LogFactory.getLog(TestCachedMobFile.class);
45    private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
46    private Configuration conf = TEST_UTIL.getConfiguration();
47    private CacheConfig cacheConf = new CacheConfig(conf);
48    private static final String FAMILY1 = "familyName1";
49    private static final String FAMILY2 = "familyName2";
50    private static final long EXPECTED_REFERENCE_ZERO = 0;
51    private static final long EXPECTED_REFERENCE_ONE = 1;
52    private static final long EXPECTED_REFERENCE_TWO = 2;

54    @Test
55    public void testOpenClose() throws Exception {
56      String caseName = getName();
57      Path testDir = TEST_UTIL.getDataTestDir();
58      FileSystem fs = testDir.getFileSystem(conf);
59      HFileContext meta = new HFileContextBuilder().withBlockSize(8*1024).build();
60      StoreFileWriter writer = new StoreFileWriter.Builder(conf, cacheConf, fs)
61          .withOutputDir(testDir).withFileContext(meta).build();
62      MobTestUtil.writeStoreFile(writer, caseName);
63      CachedMobFile cachedMobFile = CachedMobFile.create(fs, writer.getPath(), conf, cacheConf);
64      Assert.assertEquals(EXPECTED_REFERENCE_ZERO, cachedMobFile.getReferenceCount());
65      cachedMobFile.open();
66      Assert.assertEquals(EXPECTED_REFERENCE_ONE, cachedMobFile.getReferenceCount());
67      cachedMobFile.open();
68      Assert.assertEquals(EXPECTED_REFERENCE_TWO, cachedMobFile.getReferenceCount());
69      cachedMobFile.close();
70      Assert.assertEquals(EXPECTED_REFERENCE_ONE, cachedMobFile.getReferenceCount());
71      cachedMobFile.close();
72      Assert.assertEquals(EXPECTED_REFERENCE_ZERO, cachedMobFile.getReferenceCount());
73    }

75    @Test
76    public void testCompare() throws Exception {
77      String caseName = getName();
78      Path testDir = TEST_UTIL.getDataTestDir();
79      FileSystem fs = testDir.getFileSystem(conf);
80      Path outputDir1 = new Path(testDir, FAMILY1);
81      HFileContext meta = new HFileContextBuilder().withBlockSize(8 * 1024).build();
82      StoreFileWriter writer1 = new StoreFileWriter.Builder(conf, cacheConf, fs)
83          .withOutputDir(outputDir1).withFileContext(meta).build();
84      MobTestUtil.writeStoreFile(writer1, caseName);
85      CachedMobFile cachedMobFile1 = CachedMobFile.create(fs, writer1.getPath(), conf, cacheConf);
86      Path outputDir2 = new Path(testDir, FAMILY2);
87      StoreFileWriter writer2 = new StoreFileWriter.Builder(conf, cacheConf, fs)
88      .withOutputDir(outputDir2)
89      .withFileContext(meta)
90      .build();
91      MobTestUtil.writeStoreFile(writer2, caseName);
92      CachedMobFile cachedMobFile2 = CachedMobFile.create(fs, writer2.getPath(), conf, cacheConf);
93      cachedMobFile1.access(1);
94      cachedMobFile2.access(2);
95      Assert.assertEquals(cachedMobFile1.compareTo(cachedMobFile2), 1);
96      Assert.assertEquals(cachedMobFile2.compareTo(cachedMobFile1), -1);
97      Assert.assertEquals(cachedMobFile1.compareTo(cachedMobFile1), 0);
98    }

100   @Test
101   public void testReadKeyValue() throws Exception {
102     Path testDir = TEST_UTIL.getDataTestDir();
103     FileSystem fs = testDir.getFileSystem(conf);
104     HFileContext meta = new HFileContextBuilder().withBlockSize(8 * 1024).build();
105     StoreFileWriter writer = new StoreFileWriter.Builder(conf, cacheConf, fs)
106         .withOutputDir(testDir).withFileContext(meta).build();
107     String caseName = getName();
108     MobTestUtil.writeStoreFile(writer, caseName);
109     CachedMobFile cachedMobFile = CachedMobFile.create(fs, writer.getPath(), conf, cacheConf);
110     byte[] family = Bytes.toBytes(caseName);
111     byte[] qualify = Bytes.toBytes(caseName);
112     // Test the start key
113     byte[] startKey = Bytes.toBytes("aa");  // The start key bytes
114     KeyValue expectedKey =
115         new KeyValue(startKey, family, qualify, Long.MAX_VALUE, Type.Put, startKey);
116     KeyValue seekKey = expectedKey.createKeyOnly(false);
117     Cell cell = cachedMobFile.readCell(seekKey, false);
118     MobTestUtil.assertCellEquals(expectedKey, cell);
119 
120     // Test the end key
121     byte[] endKey = Bytes.toBytes("zz");  // The end key bytes
122     expectedKey = new KeyValue(endKey, family, qualify, Long.MAX_VALUE, Type.Put, endKey);
123     seekKey = expectedKey.createKeyOnly(false);
124     cell = cachedMobFile.readCell(seekKey, false);
125     MobTestUtil.assertCellEquals(expectedKey, cell);
126 
127     // Test the random key
128     byte[] randomKey = Bytes.toBytes(MobTestUtil.generateRandomString(2));
129     expectedKey = new KeyValue(randomKey, family, qualify, Long.MAX_VALUE, Type.Put, randomKey);
130     seekKey = expectedKey.createKeyOnly(false);
131     cell = cachedMobFile.readCell(seekKey, false);
132     MobTestUtil.assertCellEquals(expectedKey, cell);
133 
134     // Test the key which is less than the start key
135     byte[] lowerKey = Bytes.toBytes("a1"); // Smaller than "aa"
136     expectedKey = new KeyValue(startKey, family, qualify, Long.MAX_VALUE, Type.Put, startKey);
137     seekKey = new KeyValue(lowerKey, family, qualify, Long.MAX_VALUE, Type.Put, lowerKey);
138     cell = cachedMobFile.readCell(seekKey, false);
139     MobTestUtil.assertCellEquals(expectedKey, cell);
140 
141     // Test the key which is more than the end key
142     byte[] upperKey = Bytes.toBytes("z{"); // Bigger than "zz"
143     seekKey = new KeyValue(upperKey, family, qualify, Long.MAX_VALUE, Type.Put, upperKey);
144     cell = cachedMobFile.readCell(seekKey, false);
145     Assert.assertNull(cell);
146   }
