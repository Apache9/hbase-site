33  @Category({FilterTests.class, SmallTests.class})
34  public class TestBitComparator {

36    private static byte[] zeros = new byte[]{0, 0, 0, 0, 0, 0};
37    private static ByteBuffer zeros_bb = ByteBuffer.wrap(zeros);
38    private static byte[] ones = new byte[]{1, 1, 1, 1, 1, 1};
39    private static ByteBuffer ones_bb = ByteBuffer.wrap(ones);
40    private static byte[] data0 = new byte[]{0, 1, 2, 4, 8, 15};
41    private static byte[] data1 = new byte[]{15, 0, 0, 0, 0, 0};
42    private static ByteBuffer data1_bb = ByteBuffer.wrap(data1);
43    private static byte[] data2 = new byte[]{0, 0, 0, 0, 0, 15};
44    private static ByteBuffer data2_bb = ByteBuffer.wrap(data2);
45    private static byte[] data3 = new byte[]{15, 15, 15, 15, 15};
46    
47    // data for testing compareTo method with offset and length parameters
48    private static byte[] data1_2 = new byte[]{15, 15, 0, 0, 0, 0, 0, 15};
49    private static ByteBuffer data1_2_bb = ByteBuffer.wrap(data1_2);
50    private static byte[] data2_2 = new byte[]{15, 0, 0, 0, 0, 0, 15, 15};
51    private static ByteBuffer data2_2_bb = ByteBuffer.wrap(data2_2);
52    
53    private final int Equal = 0;
54    private final int NotEqual = 1;

56    @Test
57    public void testANDOperation() {
58      testOperation(zeros, ones, BitComparator.BitwiseOp.AND, NotEqual);
59      testOperation(data1, ones, BitComparator.BitwiseOp.AND, Equal);
60      testOperation(data1, data0, BitComparator.BitwiseOp.AND, NotEqual);
61      testOperation(data2, data1, BitComparator.BitwiseOp.AND, NotEqual);
62      testOperation(ones, data0, BitComparator.BitwiseOp.AND, Equal);
63      testOperation(ones, data3, BitComparator.BitwiseOp.AND, NotEqual);
64  
65      testOperation(zeros_bb, ones, BitComparator.BitwiseOp.AND, NotEqual);
66      testOperation(data1_bb, ones, BitComparator.BitwiseOp.AND, Equal);
67      testOperation(data1_bb, data0, BitComparator.BitwiseOp.AND, NotEqual);
68      testOperation(data2_bb, data1, BitComparator.BitwiseOp.AND, NotEqual);
69      testOperation(ones_bb, data0, BitComparator.BitwiseOp.AND, Equal);
70      testOperation(ones_bb, data3, BitComparator.BitwiseOp.AND, NotEqual);
71    }
72  
73    @Test
74    public void testOROperation() {
75      testOperation(ones, zeros, BitComparator.BitwiseOp.OR, Equal);
76      testOperation(zeros, zeros, BitComparator.BitwiseOp.OR, NotEqual);
77      testOperation(data1, zeros, BitComparator.BitwiseOp.OR, Equal);
78      testOperation(data2, data1, BitComparator.BitwiseOp.OR, Equal);
79      testOperation(ones, data3, BitComparator.BitwiseOp.OR, NotEqual);
80  
81      testOperation(ones_bb, zeros, BitComparator.BitwiseOp.OR, Equal);
82      testOperation(zeros_bb, zeros, BitComparator.BitwiseOp.OR, NotEqual);
83      testOperation(data1_bb, zeros, BitComparator.BitwiseOp.OR, Equal);
84      testOperation(data2_bb, data1, BitComparator.BitwiseOp.OR, Equal);
85      testOperation(ones_bb, data3, BitComparator.BitwiseOp.OR, NotEqual);
86    }

88    @Test
89    public void testXOROperation() {
90      testOperation(ones, zeros, BitComparator.BitwiseOp.XOR, Equal);
91      testOperation(zeros, zeros, BitComparator.BitwiseOp.XOR, NotEqual);
92      testOperation(ones, ones, BitComparator.BitwiseOp.XOR, NotEqual);
93      testOperation(data2, data1, BitComparator.BitwiseOp.XOR, Equal);
94      testOperation(ones, data3, BitComparator.BitwiseOp.XOR, NotEqual);
95  
96      testOperation(ones_bb, zeros, BitComparator.BitwiseOp.XOR, Equal);
97      testOperation(zeros_bb, zeros, BitComparator.BitwiseOp.XOR, NotEqual);
98      testOperation(ones_bb, ones, BitComparator.BitwiseOp.XOR, NotEqual);
99      testOperation(data2_bb, data1, BitComparator.BitwiseOp.XOR, Equal);
100     testOperation(ones_bb, data3, BitComparator.BitwiseOp.XOR, NotEqual);
101   }

103   private void testOperation(byte[] data, byte[] comparatorBytes, BitComparator.BitwiseOp operator,
104       int expected) {
105     BitComparator comparator = new BitComparator(comparatorBytes, operator);
106     assertEquals(comparator.compareTo(data), expected);
107   }
108 
109   private void testOperation(ByteBuffer data, byte[] comparatorBytes,
110       BitComparator.BitwiseOp operator, int expected) {
111     BitComparator comparator = new BitComparator(comparatorBytes, operator);
112     assertEquals(comparator.compareTo(data, 0, data.capacity()), expected);
113   }
