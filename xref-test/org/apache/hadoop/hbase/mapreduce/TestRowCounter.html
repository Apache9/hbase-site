158    * Test a case when the timerange is specified with --starttime and --endtime options
159    *
160    * @throws Exception

162   @Test
163   public void testRowCounterTimeRange() throws Exception {
164     final byte[] family = Bytes.toBytes(COL_FAM);
165     final byte[] col1 = Bytes.toBytes(COL1);
166     Put put1 = new Put(Bytes.toBytes("row_timerange_" + 1));
167     Put put2 = new Put(Bytes.toBytes("row_timerange_" + 2));
168     Put put3 = new Put(Bytes.toBytes("row_timerange_" + 3));
169 
170     long ts;
171 
172     // clean up content of TABLE_NAME
173     Table table = TEST_UTIL.createTable(TableName.valueOf(TABLE_NAME_TS_RANGE), Bytes.toBytes(COL_FAM));
174 
175     ts = System.currentTimeMillis();
176     put1.addColumn(family, col1, ts, Bytes.toBytes("val1"));
177     table.put(put1);
178     Thread.sleep(100);
179 
180     ts = System.currentTimeMillis();
181     put2.addColumn(family, col1, ts, Bytes.toBytes("val2"));
182     put3.addColumn(family, col1, ts, Bytes.toBytes("val3"));
183     table.put(put2);
184     table.put(put3);
185     table.close();
186 
187     String[] args = new String[] {
188         TABLE_NAME_TS_RANGE, COL_FAM + ":" + COL1,
189         "--starttime=" + 0,
190         "--endtime=" + ts
191     };
192     runRowCount(args, 1);
193 
194     args = new String[] {
195         TABLE_NAME_TS_RANGE, COL_FAM + ":" + COL1,
196         "--starttime=" + 0,
197         "--endtime=" + (ts - 10)
198     };
199     runRowCount(args, 1);
200 
201     args = new String[] {
202         TABLE_NAME_TS_RANGE, COL_FAM + ":" + COL1,
203         "--starttime=" + ts,
204         "--endtime=" + (ts + 1000)
205     };
206     runRowCount(args, 2);
207 
208     args = new String[] {
209         TABLE_NAME_TS_RANGE, COL_FAM + ":" + COL1,
210         "--starttime=" + (ts - 30 * 1000),
211         "--endtime=" + (ts + 30 * 1000),
212     };

216   /**
217    * Run the RowCounter map reduce job and verify the row count.
218    *
219    * @param args the command line arguments to be used for rowcounter job.
220    * @param expectedCount the expected row count (result of map reduce job).
221    * @throws Exception
222    */
223   private void runRowCount(String[] args, int expectedCount) throws Exception {
224     Job job = RowCounter.createSubmittableJob(TEST_UTIL.getConfiguration(), args);
225     job.waitForCompletion(true);
226     assertTrue(job.isSuccessful());
227     Counter counter = job.getCounters().findCounter(RowCounter.RowCounterMapper.Counters.ROWS);
228     assertEquals(expectedCount, counter.getValue());
229   }

231   /**
232    * Writes TOTAL_ROWS number of distinct rows in to the table. Few rows have
233    * two columns, Few have one.
234    *
235    * @param table
236    * @throws IOException
237    */
238   private static void writeRows(Table table) throws IOException {
239     final byte[] family = Bytes.toBytes(COL_FAM);
240     final byte[] value = Bytes.toBytes("abcd");
241     final byte[] col1 = Bytes.toBytes(COL1);
242     final byte[] col2 = Bytes.toBytes(COL2);
243     final byte[] col3 = Bytes.toBytes(COMPOSITE_COLUMN);
244     ArrayList<Put> rowsUpdate = new ArrayList<Put>();
245     // write few rows with two columns
246     int i = 0;
247     for (; i < TOTAL_ROWS - ROWS_WITH_ONE_COL; i++) {
248       // Use binary rows values to test for HBASE-15287.
249       byte[] row = Bytes.toBytesBinary("\\x00row" + i);
250       Put put = new Put(row);
251       put.addColumn(family, col1, value);
252       put.addColumn(family, col2, value);
253       put.addColumn(family, col3, value);
254       rowsUpdate.add(put);
255     }
256 
257     // write few rows with only one column
258     for (; i < TOTAL_ROWS; i++) {
259       byte[] row = Bytes.toBytes("row" + i);
260       Put put = new Put(row);
261       put.addColumn(family, col2, value);
262       rowsUpdate.add(put);
263     }
264     table.put(rowsUpdate);
265   }

267   /**
268    * test main method. Import should print help and call System.exit
269    */
270   @Test
271   public void testImportMain() throws Exception {
272     PrintStream oldPrintStream = System.err;
273     SecurityManager SECURITY_MANAGER = System.getSecurityManager();
274     LauncherSecurityManager newSecurityManager= new LauncherSecurityManager();
275     System.setSecurityManager(newSecurityManager);
276     ByteArrayOutputStream data = new ByteArrayOutputStream();
277     String[] args = {};
278     System.setErr(new PrintStream(data));
279     try {
280       System.setErr(new PrintStream(data));
281 
282       try {
283         RowCounter.main(args);
284         fail("should be SecurityException");
285       } catch (SecurityException e) {
286         assertEquals(-1, newSecurityManager.getExitCode());
287         assertTrue(data.toString().contains("Wrong number of parameters:"));
288         assertTrue(data.toString().contains(
289             "Usage: RowCounter [options] <tablename> " +
290             "[--starttime=[start] --endtime=[end] " +
291             "[--range=[startKey],[endKey]] " +
292             "[<column1> <column2>...]"));
293         assertTrue(data.toString().contains("-Dhbase.client.scanner.caching=100"));
294         assertTrue(data.toString().contains("-Dmapreduce.map.speculative=false"));
295       }
296       data.reset();
297       try {
298         args = new String[2];
299         args[0] = "table";
300         args[1] = "--range=1";
301         RowCounter.main(args);
302         fail("should be SecurityException");
303       } catch (SecurityException e) {
304         assertEquals(-1, newSecurityManager.getExitCode());
305         assertTrue(data.toString().contains(
306             "Please specify range in such format as \"--range=a,b\" or, with only one boundary," +
307             " \"--range=,b\" or \"--range=a,\""));
308         assertTrue(data.toString().contains(
309             "Usage: RowCounter [options] <tablename> " +
310             "[--starttime=[start] --endtime=[end] " +
311             "[--range=[startKey],[endKey]] " +
312             "[<column1> <column2>...]"));
313       }
314 
315     } finally {
316       System.setErr(oldPrintStream);
317       System.setSecurityManager(SECURITY_MANAGER);
318     }
