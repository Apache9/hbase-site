020package org.apache.hadoop.hbase.master;
021
022import static org.junit.Assert.fail;
023
024import java.net.InetAddress;
025
026import junit.framework.Assert;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.conf.Configuration;
031import org.apache.hadoop.hbase.*;
032import org.apache.hadoop.hbase.catalog.CatalogTracker;
033import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
034import org.junit.Test;
035import org.junit.experimental.categories.Category;
036
037@Category(MediumTests.class)
038public class TestClockSkewDetection {
039  private static final Log LOG =
040    LogFactory.getLog(TestClockSkewDetection.class);
041
042  @Test
043  public void testClockSkewDetection() throws Exception {
044    final Configuration conf = HBaseConfiguration.create();
045    ServerManager sm = new ServerManager(new Server() {
046      @Override
047      public CatalogTracker getCatalogTracker() {
048        return null;
049      }
050
051      @Override
052      public Configuration getConfiguration() {
053        return conf;
054      }
055
056      @Override
057      public ServerName getServerName() {
058        return null;
059      }
060
061      @Override
062      public ZooKeeperWatcher getZooKeeper() {
063        return null;
064      }
065
066      @Override
067      public void abort(String why, Throwable e) {}
068      
069      @Override
070      public boolean isAborted() {
071        return false;
072      }
073
074      @Override
075      public boolean isStopped() {
076        return false;
077      }
078
079      @Override
080      public void stop(String why) {
081      }}, null, false);
082
083    LOG.debug("regionServerStartup 1");
084    InetAddress ia1 = InetAddress.getLocalHost();
085    sm.regionServerStartup(ia1, 1234, -1, System.currentTimeMillis());
086
087    final Configuration c = HBaseConfiguration.create();
088    long maxSkew = c.getLong("hbase.master.maxclockskew", 30000);
089    long warningSkew = c.getLong("hbase.master.warningclockskew", 1000);
090
091    try {
092      // Master Time > Region Server Time
093      LOG.debug("Test: Master Time > Region Server Time");
094      LOG.debug("regionServerStartup 2");
095      InetAddress ia2 = InetAddress.getLocalHost();
096      sm.regionServerStartup(ia2, 1235, -1, System.currentTimeMillis() - maxSkew * 2);
097      fail("HMaster should have thrown a ClockOutOfSyncException but didn't.");
098    } catch (ClockOutOfSyncException e) {
099      // we want an exception
100      LOG.info("Recieved expected exception: " + e);
101    }
102
103    try {
104      // Master Time < Region Server Time
105      LOG.debug("Test: Master Time < Region Server Time");
106      LOG.debug("regionServerStartup 3");
107      InetAddress ia3 = InetAddress.getLocalHost();
108      sm.regionServerStartup(ia3, 1236, -1, System.currentTimeMillis() + maxSkew * 2);
109      fail("HMaster should have thrown a ClockOutOfSyncException but didn't.");
110    } catch (ClockOutOfSyncException e) {
111      // we want an exception
112      LOG.info("Recieved expected exception: " + e);
113    }
114
115    // make sure values above warning threshold but below max threshold don't kill
116    LOG.debug("regionServerStartup 4");
117    InetAddress ia4 = InetAddress.getLocalHost();
118    sm.regionServerStartup(ia4, 1237, -1, System.currentTimeMillis() - warningSkew * 2);
119
120    // make sure values above warning threshold but below max threshold don't kill
121    LOG.debug("regionServerStartup 5");
122    InetAddress ia5 = InetAddress.getLocalHost();
123    sm.regionServerStartup(ia5, 1238, -1, System.currentTimeMillis() + warningSkew * 2);
124    
125  }
126
127  @org.junit.Rule
128  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
129    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
130}
