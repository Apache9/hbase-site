018package org.apache.hadoop.hbase.master.snapshot;
019
020import static org.junit.Assert.assertFalse;
021
022import java.io.IOException;
023
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.fs.FileSystem;
026import org.apache.hadoop.fs.Path;
027import org.apache.hadoop.hbase.HBaseTestingUtility;
028import org.apache.hadoop.hbase.HConstants;
029import org.apache.hadoop.hbase.HRegionInfo;
030import org.apache.hadoop.hbase.SmallTests;
031import org.apache.hadoop.hbase.snapshot.SnapshotDescriptionUtils;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.util.FSUtils;
034import org.junit.AfterClass;
035import org.junit.Test;
036import org.junit.experimental.categories.Category;
037
038/**
039 * Test that the snapshot hfile cleaner finds hfiles referenced in a snapshot
040 */
041@Category(SmallTests.class)
042public class TestSnapshotHFileCleaner {
043
044  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
045
046  @AfterClass
047  public static void cleanup() throws IOException {
048    Configuration conf = TEST_UTIL.getConfiguration();
049    Path rootDir = FSUtils.getRootDir(conf);
050    FileSystem fs = FileSystem.get(conf);
051    // cleanup
052    fs.delete(rootDir, true);
053  }
054
055  @Test
056  public void testFindsSnapshotFilesWhenCleaning() throws IOException {
057    Configuration conf = TEST_UTIL.getConfiguration();
058    FSUtils.setRootDir(conf, TEST_UTIL.getDataTestDir());
059    Path rootDir = FSUtils.getRootDir(conf);
060    Path archivedHfileDir = new Path(TEST_UTIL.getDataTestDir(), HConstants.HFILE_ARCHIVE_DIRECTORY);
061
062    FileSystem fs = FileSystem.get(conf);
063    SnapshotHFileCleaner cleaner = new SnapshotHFileCleaner();
064    cleaner.setConf(conf);
065
066    // write an hfile to the snapshot directory
067    String snapshotName = "snapshot";
068    byte[] snapshot = Bytes.toBytes(snapshotName);
069    String table = "table";
070    byte[] tableName = Bytes.toBytes(table);
071    Path snapshotDir = SnapshotDescriptionUtils.getCompletedSnapshotDir(snapshotName, rootDir);
072    HRegionInfo mockRegion = new HRegionInfo(tableName);
073    Path regionSnapshotDir = new Path(snapshotDir, mockRegion.getEncodedName());
074    Path familyDir = new Path(regionSnapshotDir, "family");
075    // create a reference to a supposedly valid hfile
076    String hfile = "fd1e73e8a96c486090c5cec07b4894c4";
077    Path refFile = new Path(familyDir, hfile);
078
079    // make sure the reference file exists
080    fs.create(refFile);
081
082    // create the hfile in the archive
083    fs.mkdirs(archivedHfileDir);
084    fs.createNewFile(new Path(archivedHfileDir, hfile));
085
086    // make sure that the file isn't deletable
087    assertFalse(cleaner.isFileDeletable(fs.getFileStatus(refFile)));
088  }
089}
