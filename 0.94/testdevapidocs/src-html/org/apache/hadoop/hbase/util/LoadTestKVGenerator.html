017package org.apache.hadoop.hbase.util;
018
019import java.util.Random;
020
021import org.apache.hadoop.hbase.util.Bytes;
022import org.apache.hadoop.hbase.util.MD5Hash;
023
024/**
025 * A generator of random keys and values for load testing. Keys are generated
026 * by converting numeric indexes to strings and prefixing them with an MD5
027 * hash. Values are generated by selecting value size in the configured range
028 * and generating a pseudo-random sequence of bytes seeded by key, column
029 * qualifier, and value size.
030 */
031public class LoadTestKVGenerator {
032
033  /** A random number generator for determining value size */
034  private Random randomForValueSize = new Random();
035
036  private final int minValueSize;
037  private final int maxValueSize;
038
039  public LoadTestKVGenerator(int minValueSize, int maxValueSize) {
040    if (minValueSize <= 0 || maxValueSize <= 0) {
041      throw new IllegalArgumentException("Invalid min/max value sizes: " +
042          minValueSize + ", " + maxValueSize);
043    }
044    this.minValueSize = minValueSize;
045    this.maxValueSize = maxValueSize;
046  }
047
048  /**
049   * Verifies that the given byte array is the same as what would be generated
050   * for the given seed strings (row/cf/column/...). We are assuming that the
051   * value size is correct, and only verify the actual bytes. However, if the
052   * min/max value sizes are set sufficiently high, an accidental match should be
053   * extremely improbable.
054   */
055  public static boolean verify(byte[] value, byte[]... seedStrings) {
056    byte[] expectedData = getValueForRowColumn(value.length, seedStrings);
057    return Bytes.equals(expectedData, value);
058  }
059
060  /**
061   * Converts the given key to string, and prefixes it with the MD5 hash of
062   * the index's string representation.
063   */
064  public static String md5PrefixedKey(long key) {
065    String stringKey = Long.toString(key);
066    String md5hash = MD5Hash.getMD5AsHex(Bytes.toBytes(stringKey));
067
068    // flip the key to randomize
069    return md5hash + "-" + stringKey;
070  }
071
072  /**
073   * Generates a value for the given key index and column qualifier. Size is
074   * selected randomly in the configured range. The generated value depends
075   * only on the combination of the strings passed (key/cf/column/...) and the selected
076   * value size. This allows to verify the actual value bytes when reading, as done
077   * in {#verify(byte[], byte[]...)}
078   * This method is as thread-safe as Random class. It appears that the worst bug ever
079   * found with the latter is that multiple threads will get some duplicate values, which
080   * we don't care about.
081   */
082  public byte[] generateRandomSizeValue(byte[]... seedStrings) {
083    int dataSize = minValueSize;
084    if (minValueSize != maxValueSize) {
085      dataSize = minValueSize + randomForValueSize.nextInt(Math.abs(maxValueSize - minValueSize));
086    }
087    return getValueForRowColumn(dataSize, seedStrings);
088 }
089
090  /**
091   * Generates random bytes of the given size for the given row and column
092   * qualifier. The random seed is fully determined by these parameters.
093   */
094  private static byte[] getValueForRowColumn(int dataSize, byte[]... seedStrings) {
095    long seed = dataSize;
096    for (byte[] str : seedStrings) {
097      seed += Bytes.toString(str).hashCode();
098    }
099    Random seededRandom = new Random(seed);
100    byte[] randomBytes = new byte[dataSize];
101    seededRandom.nextBytes(randomBytes);
102    return randomBytes;
103  }
104
105}
