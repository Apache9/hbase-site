043@Category(SmallTests.class)
044public class TestRegionObserverStacking extends TestCase {
045  private static HBaseTestingUtility TEST_UTIL
046    = new HBaseTestingUtility();
047  static final Path DIR = TEST_UTIL.getDataTestDir();
048
049  public static class ObserverA extends BaseRegionObserver {
050    long id;
051    @Override
052    public void postPut(final ObserverContext<RegionCoprocessorEnvironment> c,
053        final Put put, final WALEdit edit,
054        final boolean writeToWAL)
055        throws IOException {
056      id = System.currentTimeMillis();
057      try {
058        Thread.sleep(10);
059      } catch (InterruptedException ex) {
060      }
061    }
062  }
063
064  public static class ObserverB extends BaseRegionObserver {
065    long id;
066    @Override
067    public void postPut(final ObserverContext<RegionCoprocessorEnvironment> c,
068        final Put put, final WALEdit edit,
069        final boolean writeToWAL)
070        throws IOException {
071      id = System.currentTimeMillis();
072      try {
073        Thread.sleep(10);
074      } catch (InterruptedException ex) {
075      }
076    }
077  }
078
079  public static class ObserverC extends BaseRegionObserver {
080    long id;
081
082    @Override
083    public void postPut(final ObserverContext<RegionCoprocessorEnvironment> c,
084        final Put put, final WALEdit edit,
085        final boolean writeToWAL)
086        throws IOException {
087      id = System.currentTimeMillis();
088      try {

091      }
092    }
093  }
094
095  HRegion initHRegion (byte [] tableName, String callingMethod,
096      Configuration conf, byte [] ... families) throws IOException {
097    HTableDescriptor htd = new HTableDescriptor(tableName);
098    for(byte [] family : families) {
099      htd.addFamily(new HColumnDescriptor(family));
100    }
101    HRegionInfo info = new HRegionInfo(htd.getName(), null, null, false);
102    Path path = new Path(DIR + callingMethod);
103    HRegion r = HRegion.createHRegion(info, path, conf, htd);
104    // this following piece is a hack. currently a coprocessorHost
105    // is secretly loaded at OpenRegionHandler. we don't really
106    // start a region server here, so just manually create cphost
107    // and set it to region.
108    RegionCoprocessorHost host = new RegionCoprocessorHost(r, null, conf);
109    r.setCoprocessorHost(host);
110    return r;
111  }
112
113  public void testRegionObserverStacking() throws Exception {
114    byte[] ROW = Bytes.toBytes("testRow");
115    byte[] TABLE = Bytes.toBytes(getClass().getName());
116    byte[] A = Bytes.toBytes("A");
117    byte[][] FAMILIES = new byte[][] { A } ;
118
119    Configuration conf = HBaseConfiguration.create();
120    HRegion region = initHRegion(TABLE, getClass().getName(),
121      conf, FAMILIES);
122    RegionCoprocessorHost h = region.getCoprocessorHost();
123    h.load(ObserverA.class, Coprocessor.PRIORITY_HIGHEST, conf);
124    h.load(ObserverB.class, Coprocessor.PRIORITY_USER, conf);
125    h.load(ObserverC.class, Coprocessor.PRIORITY_LOWEST, conf);
126
127    Put put = new Put(ROW);
128    put.add(A, A, A);
129    int lockid = region.obtainRowLock(ROW);
130    region.put(put, lockid);
131    region.releaseRowLock(lockid);
132
133    Coprocessor c = h.findCoprocessor(ObserverA.class.getName());
134    long idA = ((ObserverA)c).id;
135    c = h.findCoprocessor(ObserverB.class.getName());
136    long idB = ((ObserverB)c).id;
137    c = h.findCoprocessor(ObserverC.class.getName());
138    long idC = ((ObserverC)c).id;
139
140    assertTrue(idA < idB);
141    assertTrue(idB < idC);
142  }
