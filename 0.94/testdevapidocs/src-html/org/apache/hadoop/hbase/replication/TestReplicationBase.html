019package org.apache.hadoop.hbase.replication;
020
021import org.apache.commons.logging.Log;
022import org.apache.commons.logging.LogFactory;
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.hbase.HBaseConfiguration;
025import org.apache.hadoop.hbase.HBaseTestingUtility;
026import org.apache.hadoop.hbase.HColumnDescriptor;
027import org.apache.hadoop.hbase.HConstants;
028import org.apache.hadoop.hbase.HTableDescriptor;
029import org.apache.hadoop.hbase.client.HBaseAdmin;
030import org.apache.hadoop.hbase.client.HTable;
031import org.apache.hadoop.hbase.client.replication.ReplicationAdmin;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.hbase.zookeeper.MiniZooKeeperCluster;
034import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
035import org.junit.AfterClass;
036import org.junit.BeforeClass;
037
038/**
039 * This class is only a base for other integration-level replication tests.
040 * Do not add tests here.
041 * TestReplicationSmallTests is where tests that don't require bring machines up/down should go
042 * All other tests should have their own classes and extend this one
043 */
044public class TestReplicationBase {
045
046  private static final Log LOG = LogFactory.getLog(TestReplicationBase.class);
047
048  protected static Configuration conf1 = HBaseConfiguration.create();
049  protected static Configuration conf2;
050  protected static Configuration CONF_WITH_LOCALFS;
051
052  protected static ZooKeeperWatcher zkw1;
053  protected static ZooKeeperWatcher zkw2;
054
055  protected static ReplicationAdmin admin;
056
057  protected static HTable htable1;
058  protected static HTable htable2;
059
060  protected static HBaseTestingUtility utility1;
061  protected static HBaseTestingUtility utility2;
062  protected static final int NB_ROWS_IN_BATCH = 100;
063  protected static final int NB_ROWS_IN_BIG_BATCH =
064      NB_ROWS_IN_BATCH * 10;
065  protected static final long SLEEP_TIME = 1000;
066  protected static final int NB_RETRIES = 15;
067  protected static final int NB_RETRIES_FOR_BIG_BATCH = 30;
068
069  protected static final byte[] tableName = Bytes.toBytes("test");
070  protected static final byte[] famName = Bytes.toBytes("f");
071  protected static final byte[] row = Bytes.toBytes("row");
072  protected static final byte[] noRepfamName = Bytes.toBytes("norep");
073
074  /**
075   * @throws java.lang.Exception
076   */
077  @BeforeClass
078  public static void setUpBeforeClass() throws Exception {
079    conf1.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/1");
080    // smaller log roll size to trigger more events
081    conf1.setFloat("hbase.regionserver.logroll.multiplier", 0.0003f);
082    conf1.setInt("replication.source.size.capacity", 10240);
083    conf1.setLong("replication.source.sleepforretries", 100);
084    conf1.setInt("hbase.regionserver.maxlogs", 10);
085    conf1.setLong("hbase.master.logcleaner.ttl", 10);
086    conf1.setInt("zookeeper.recovery.retry", 1);
087    conf1.setInt("zookeeper.recovery.retry.intervalmill", 10);
088    conf1.setBoolean(HConstants.REPLICATION_ENABLE_KEY, true);
089    conf1.setBoolean("dfs.support.append", true);
090    conf1.setLong(HConstants.THREAD_WAKE_FREQUENCY, 100);
091    conf1.setInt("replication.stats.thread.period.seconds", 5);
092
093    utility1 = new HBaseTestingUtility(conf1);
094    utility1.startMiniZKCluster();
095    MiniZooKeeperCluster miniZK = utility1.getZkCluster();
096    // Have to reget conf1 in case zk cluster location different
097    // than default
098    conf1 = utility1.getConfiguration();  
099    zkw1 = new ZooKeeperWatcher(conf1, "cluster1", null, true);
100    admin = new ReplicationAdmin(conf1);
101    LOG.info("Setup first Zk");
102
103    // Base conf2 on conf1 so it gets the right zk cluster.
104    conf2 = HBaseConfiguration.create(conf1);
105    conf2.set(HConstants.ZOOKEEPER_ZNODE_PARENT, "/2");
106    conf2.setInt("hbase.client.retries.number", 6);
107    conf2.setBoolean(HConstants.REPLICATION_ENABLE_KEY, true);
108    conf2.setBoolean("dfs.support.append", true);
109
110    utility2 = new HBaseTestingUtility(conf2);
111    utility2.setZkCluster(miniZK);
112    zkw2 = new ZooKeeperWatcher(conf2, "cluster2", null, true);
113
114    admin.addPeer("2", utility2.getClusterKey());
115    setIsReplication(true);
116
117    LOG.info("Setup second Zk");
118    CONF_WITH_LOCALFS = HBaseConfiguration.create(conf1);
119    utility1.startMiniCluster(2);
120    utility2.startMiniCluster(2);
121
122    HTableDescriptor table = new HTableDescriptor(tableName);
123    HColumnDescriptor fam = new HColumnDescriptor(famName);
124    fam.setScope(HConstants.REPLICATION_SCOPE_GLOBAL);
125    table.addFamily(fam);
126    fam = new HColumnDescriptor(noRepfamName);
127    table.addFamily(fam);
128    HBaseAdmin admin1 = new HBaseAdmin(conf1);
129    HBaseAdmin admin2 = new HBaseAdmin(conf2);
130    admin1.createTable(table, HBaseTestingUtility.KEYS_FOR_HBA_CREATE_TABLE);
131    admin2.createTable(table);
132    htable1 = new HTable(conf1, tableName);
133    htable1.setWriteBufferSize(1024);
134    htable2 = new HTable(conf2, tableName);
135  }
136
137  protected static void setIsReplication(boolean rep) throws Exception {
138    LOG.info("Set rep " + rep);
139    admin.setReplicating(rep);
140    Thread.sleep(SLEEP_TIME);
141  }
142
143  /**
144   * @throws java.lang.Exception
145   */
146  @AfterClass
147  public static void tearDownAfterClass() throws Exception {
148    utility2.shutdownMiniCluster();
149    utility1.shutdownMiniCluster();
150  }
151
152
153}
