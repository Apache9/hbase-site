018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.HRegionInfo;
024import org.apache.hadoop.hbase.HRegionLocation;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.ZooKeeperConnectionException;
027import org.apache.hadoop.hbase.client.HConnectionManager.HConnectionImplementation;
028import org.apache.hadoop.hbase.client.HConnectionManager.HConnectionKey;
029import org.apache.hadoop.hbase.ipc.HRegionInterface;
030import org.mockito.Mockito;
031
032/**
033 * {@link HConnection} testing utility.
034 */
035public class HConnectionTestingUtility {
036  /*
037   * Not part of {@link HBaseTestingUtility} because this class is not
038   * in same package as {@link HConnection}.  Would have to reveal ugly
039   * {@link HConnectionManager} innards to HBaseTestingUtility to give it access.
040   */
041  /**
042   * Get a Mocked {@link HConnection} that goes with the passed <code>conf</code>
043   * configuration instance.  Minimally the mock will return
044   * <code>conf</conf> when {@link HConnection#getConfiguration()} is invoked.
045   * Be sure to shutdown the connection when done by calling
046   * {@link HConnectionManager#deleteConnection(Configuration, boolean)} else it
047   * will stick around; this is probably not what you want.
048   * @param conf configuration
049   * @return HConnection object for <code>conf</code>
050   * @throws ZooKeeperConnectionException
051   */
052  public static HConnection getMockedConnection(final Configuration conf)
053  throws ZooKeeperConnectionException {
054    HConnectionKey connectionKey = new HConnectionKey(conf);
055    synchronized (HConnectionManager.HBASE_INSTANCES) {
056      HConnectionImplementation connection =
057        HConnectionManager.HBASE_INSTANCES.get(connectionKey);
058      if (connection == null) {
059        connection = Mockito.mock(HConnectionImplementation.class);
060        Mockito.when(connection.getConfiguration()).thenReturn(conf);
061        HConnectionManager.HBASE_INSTANCES.put(connectionKey, connection);
062      }
063      return connection;
064    }
065  }
066
067  /**
068   * Calls {@link #getMockedConnection(Configuration)} and then mocks a few
069   * more of the popular {@link HConnection} methods so they do 'normal'
070   * operation (see return doc below for list). Be sure to shutdown the
071   * connection when done by calling
072   * {@link HConnectionManager#deleteConnection(Configuration, boolean)} else it
073   * will stick around; this is probably not what you want.
074   * @param implementation An {@link HRegionInterface} instance; you'll likely
075   * want to pass a mocked HRS; can be null.
076   * 
077   * @param conf Configuration to use
078   * @param implementation An HRegionInterface; can be null but is usually
079   * itself a mock.
080   * @param sn ServerName to include in the region location returned by this
081   * <code>implementation</code>
082   * @param hri HRegionInfo to include in the location returned when
083   * getRegionLocation is called on the mocked connection
084   * @return Mock up a connection that returns a {@link Configuration} when
085   * {@link HConnection#getConfiguration()} is called, a 'location' when
086   * {@link HConnection#getRegionLocation(byte[], byte[], boolean)} is called,
087   * and that returns the passed {@link HRegionInterface} instance when
088   * {@link HConnection#getHRegionConnection(String, int)}
089   * is called (Be sure call
090   * {@link HConnectionManager#deleteConnection(org.apache.hadoop.conf.Configuration, boolean)}
091   * when done with this mocked Connection.
092   * @throws IOException
093   */
094  public static HConnection getMockedConnectionAndDecorate(final Configuration conf,
095      final HRegionInterface implementation, final ServerName sn, final HRegionInfo hri)
096  throws IOException {
097    HConnection c = HConnectionTestingUtility.getMockedConnection(conf);
098    Mockito.doNothing().when(c).close();
099    // Make it so we return a particular location when asked.
100    final HRegionLocation loc = new HRegionLocation(hri, sn.getHostname(), sn.getPort());
101    Mockito.when(c.getRegionLocation((byte[]) Mockito.any(),
102        (byte[]) Mockito.any(), Mockito.anyBoolean())).
103      thenReturn(loc);
104    Mockito.when(c.locateRegion((byte[]) Mockito.any(), (byte[]) Mockito.any())).
105      thenReturn(loc);
106    if (implementation != null) {
107      // If a call to getHRegionConnection, return this implementation.
108      Mockito.when(c.getHRegionConnection(Mockito.anyString(), Mockito.anyInt())).
109        thenReturn(implementation);
110    }
111    return c;
112  }
113
114  /**
115   * Get a Mockito spied-upon {@link HConnection} that goes with the passed
116   * <code>conf</code> configuration instance.
117   * Be sure to shutdown the connection when done by calling
118   * {@link HConnectionManager#deleteConnection(Configuration, boolean)} else it
119   * will stick around; this is probably not what you want.
120   * @param conf configuration
121   * @return HConnection object for <code>conf</code>
122   * @throws ZooKeeperConnectionException
123   * @see http://mockito.googlecode.com/svn/branches/1.6/javadoc/org/mockito/Mockito.html#spy(T)
124   */
125  public static HConnection getSpiedConnection(final Configuration conf)
126  throws ZooKeeperConnectionException {
127    HConnectionKey connectionKey = new HConnectionKey(conf);
128    synchronized (HConnectionManager.HBASE_INSTANCES) {
129      HConnectionImplementation connection =
130        HConnectionManager.HBASE_INSTANCES.get(connectionKey);
131      if (connection == null) {
132        connection = Mockito.spy(new HConnectionImplementation(conf, true, null));
133        HConnectionManager.HBASE_INSTANCES.put(connectionKey, connection);
134      }
135      return connection;
136    }
137  }
138
139  /**
140   * @return Count of extant connection instances
141   */
142  public static int getConnectionCount() {
143    synchronized (HConnectionManager.HBASE_INSTANCES) {
144      return HConnectionManager.HBASE_INSTANCES.size();
145    }
146  }
