041/**
042 * Test for HQuorumPeer.
043 */
044@Category(SmallTests.class)
045public class TestHQuorumPeer {
046  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
047  private static int PORT_NO = 21818;
048  private Path dataDir;
049
050
051  @Before public void setup() throws IOException {
052    // Set it to a non-standard port.
053    TEST_UTIL.getConfiguration().setInt(HConstants.ZOOKEEPER_CLIENT_PORT,
054        PORT_NO);
055    this.dataDir = TEST_UTIL.getDataTestDir(this.getClass().getName());
056    FileSystem fs = FileSystem.get(TEST_UTIL.getConfiguration());
057    if (fs.exists(this.dataDir)) {
058      if (!fs.delete(this.dataDir, true)) {
059        throw new IOException("Failed cleanup of " + this.dataDir);
060      }
061    }
062    if (!fs.mkdirs(this.dataDir)) {
063      throw new IOException("Failed create of " + this.dataDir);
064    }
065  }
066
067  @Test public void testMakeZKProps() {
068    Configuration conf = new Configuration(TEST_UTIL.getConfiguration());
069    conf.set(HConstants.ZOOKEEPER_DATA_DIR, this.dataDir.toString());
070    Properties properties = ZKConfig.makeZKProps(conf);
071    assertEquals(dataDir.toString(), (String)properties.get("dataDir"));
072    assertEquals(Integer.valueOf(PORT_NO),
073      Integer.valueOf(properties.getProperty("clientPort")));
074    assertEquals("localhost:2888:3888", properties.get("server.0"));
075    assertEquals(null, properties.get("server.1"));
076
077    String oldValue = conf.get(HConstants.ZOOKEEPER_QUORUM);
078    conf.set(HConstants.ZOOKEEPER_QUORUM, "a.foo.bar,b.foo.bar,c.foo.bar");
079    properties = ZKConfig.makeZKProps(conf);
080    assertEquals(dataDir.toString(), properties.get("dataDir"));
081    assertEquals(Integer.valueOf(PORT_NO),
082      Integer.valueOf(properties.getProperty("clientPort")));
083    assertEquals("a.foo.bar:2888:3888", properties.get("server.0"));
084    assertEquals("b.foo.bar:2888:3888", properties.get("server.1"));
085    assertEquals("c.foo.bar:2888:3888", properties.get("server.2"));
086    assertEquals(null, properties.get("server.3"));
087    conf.set(HConstants.ZOOKEEPER_QUORUM, oldValue);
088  }
089
090  @Test public void testConfigInjection() throws Exception {
091    String s =
092      "dataDir=" + this.dataDir.toString() + "\n" +
093      "clientPort=2181\n" +
094      "initLimit=2\n" +
095      "syncLimit=2\n" +
096      "server.0=${hbase.master.hostname}:2888:3888\n" +
097      "server.1=server1:2888:3888\n" +
098      "server.2=server2:2888:3888\n";
099
100    System.setProperty("hbase.master.hostname", "localhost");
101    InputStream is = new ByteArrayInputStream(s.getBytes());
102    Configuration conf = TEST_UTIL.getConfiguration();
103    Properties properties = ZKConfig.parseZooCfg(conf, is);
104
105    assertEquals(this.dataDir.toString(), properties.get("dataDir"));
106    assertEquals(Integer.valueOf(2181),
107      Integer.valueOf(properties.getProperty("clientPort")));
108    assertEquals("localhost:2888:3888", properties.get("server.0"));
109
110    HQuorumPeer.writeMyID(properties);
111    QuorumPeerConfig config = new QuorumPeerConfig();
112    config.parseProperties(properties);
113
114    assertEquals(this.dataDir.toString(), config.getDataDir());
115    assertEquals(2181, config.getClientPortAddress().getPort());
116    Map<Long,QuorumServer> servers = config.getServers();
117    assertEquals(3, servers.size());
118    assertTrue(servers.containsKey(Long.valueOf(0)));
119    QuorumServer server = servers.get(Long.valueOf(0));
120    assertEquals("localhost", server.addr.getHostName());
121
122    // Override with system property.
123    System.setProperty("hbase.master.hostname", "foo.bar");
124    is = new ByteArrayInputStream(s.getBytes());
125    properties = ZKConfig.parseZooCfg(conf, is);
126    assertEquals("foo.bar:2888:3888", properties.get("server.0"));
127
128    config.parseProperties(properties);
129
130    servers = config.getServers();
131    server = servers.get(Long.valueOf(0));
132    assertEquals("foo.bar", server.addr.getHostName());
133  }
134
135  @Test public void testShouldAssignDefaultZookeeperClientPort() {
136    Configuration config = HBaseConfiguration.create();
137    config.clear();
138    Properties p = ZKConfig.makeZKProps(config);
139    assertNotNull(p);
140    assertEquals(2181, p.get("clientPort"));
141  }
