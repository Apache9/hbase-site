061  /*
062   * Test utility for building a NavigableSet for scanners.
063   * @param strCols
064   * @return
065   */
066  NavigableSet<byte[]> getCols(String ...strCols) {
067    NavigableSet<byte[]> cols = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
068    for (String col : strCols) {
069      byte[] bytes = Bytes.toBytes(col);
070      cols.add(bytes);
071    }
072    return cols;
073  }

075  public void testScanTimeRange() throws IOException {
076    String r1 = "R1";
077    // returns only 1 of these 2 even though same timestamp
078    KeyValue [] kvs = new KeyValue[] {
079        KeyValueTestUtil.create(r1, CF_STR, "a", 1, KeyValue.Type.Put, "dont-care"),
080        KeyValueTestUtil.create(r1, CF_STR, "a", 2, KeyValue.Type.Put, "dont-care"),
081        KeyValueTestUtil.create(r1, CF_STR, "a", 3, KeyValue.Type.Put, "dont-care"),
082        KeyValueTestUtil.create(r1, CF_STR, "a", 4, KeyValue.Type.Put, "dont-care"),
083        KeyValueTestUtil.create(r1, CF_STR, "a", 5, KeyValue.Type.Put, "dont-care"),
084    };
085    List<KeyValueScanner> scanners = Arrays.<KeyValueScanner>asList(
086        new KeyValueScanner[] {
087            new KeyValueScanFixture(KeyValue.COMPARATOR, kvs)
088    });
089    Scan scanSpec = new Scan(Bytes.toBytes(r1));
090    scanSpec.setTimeRange(0, 6);
091    scanSpec.setMaxVersions();
092    StoreScanner scan = new StoreScanner(scanSpec, scanInfo, scanType,
093        getCols("a"), scanners);
094    List<KeyValue> results = new ArrayList<KeyValue>();
095    assertEquals(true, scan.next(results));
096    assertEquals(5, results.size());
097    assertEquals(kvs[kvs.length - 1], results.get(0));
098    // Scan limited TimeRange
099    scanSpec = new Scan(Bytes.toBytes(r1));
100    scanSpec.setTimeRange(1, 3);
101    scanSpec.setMaxVersions();
102    scan = new StoreScanner(scanSpec, scanInfo, scanType, getCols("a"),
103        scanners);
104    results = new ArrayList<KeyValue>();
105    assertEquals(true, scan.next(results));
106    assertEquals(2, results.size());
107    // Another range.
108    scanSpec = new Scan(Bytes.toBytes(r1));
109    scanSpec.setTimeRange(5, 10);
110    scanSpec.setMaxVersions();
111    scan = new StoreScanner(scanSpec, scanInfo, scanType, getCols("a"),
112        scanners);
113    results = new ArrayList<KeyValue>();
114    assertEquals(true, scan.next(results));
115    assertEquals(1, results.size());
116    // See how TimeRange and Versions interact.
117    // Another range.
118    scanSpec = new Scan(Bytes.toBytes(r1));
119    scanSpec.setTimeRange(0, 10);
120    scanSpec.setMaxVersions(3);
121    scan = new StoreScanner(scanSpec, scanInfo, scanType, getCols("a"),
122        scanners);
123    results = new ArrayList<KeyValue>();
124    assertEquals(true, scan.next(results));
125    assertEquals(3, results.size());
126  }

157  public void testWontNextToNext() throws IOException {

159    KeyValue [] kvs = new KeyValue[] {
160        KeyValueTestUtil.create("R1", "cf", "a", 2, KeyValue.Type.Put, "dont-care"),
161        KeyValueTestUtil.create("R1", "cf", "a", 1, KeyValue.Type.Put, "dont-care"),

164    List<KeyValueScanner> scanners = scanFixture(kvs);
165
166    Scan scanSpec = new Scan(Bytes.toBytes("R1"));
167    // this only uses maxVersions (default=1) and TimeRange (default=all)

169        getCols("a"), scanners);
170
171    List<KeyValue> results = new ArrayList<KeyValue>();
172    scan.next(results);
173    assertEquals(1, results.size());
174    assertEquals(kvs[0], results.get(0));
175    // should be ok...

179    assertEquals(1, results.size());
180    assertEquals(kvs[2], results.get(0));
181
182    results.clear();
183    scan.next(results);
184    assertEquals(0, results.size());
