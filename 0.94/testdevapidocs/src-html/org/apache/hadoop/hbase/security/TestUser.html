032import java.io.IOException;

036@Category(SmallTests.class)
037public class TestUser {
038  private static Log LOG = LogFactory.getLog(TestUser.class);
039
040  @Test
041  public void testBasicAttributes() throws Exception {
042    Configuration conf = HBaseConfiguration.create();
043    User user = User.createUserForTesting(conf, "simple", new String[]{"foo"});
044    assertEquals("Username should match", "simple", user.getName());
045    assertEquals("Short username should match", "simple", user.getShortName());
046    // don't test shortening of kerberos names because regular Hadoop doesn't support them
047  }
048
049  @Test
050  public void testRunAs() throws Exception {
051    Configuration conf = HBaseConfiguration.create();
052    final User user = User.createUserForTesting(conf, "testuser", new String[]{"foo"});
053    final PrivilegedExceptionAction<String> action = new PrivilegedExceptionAction<String>(){
054      public String run() throws IOException {
055          User u = User.getCurrent();
056          return u.getName();
057      }
058    };
059
060    String username = user.runAs(action);
061    assertEquals("Current user within runAs() should match",
062        "testuser", username);
063
064    // ensure the next run is correctly set
065    User user2 = User.createUserForTesting(conf, "testuser2", new String[]{"foo"});
066    String username2 = user2.runAs(action);
067    assertEquals("Second username should match second user",
068        "testuser2", username2);
069
070    // check the exception version
071    username = user.runAs(new PrivilegedExceptionAction<String>(){
072      public String run() throws Exception {
073        return User.getCurrent().getName();
074      }
075    });
076    assertEquals("User name in runAs() should match", "testuser", username);
077
078    // verify that nested contexts work
079    user2.runAs(new PrivilegedExceptionAction(){
080      public Object run() throws IOException, InterruptedException{
081        String nestedName = user.runAs(action);
082        assertEquals("Nest name should match nested user", "testuser", nestedName);
083        assertEquals("Current name should match current user",
084            "testuser2", User.getCurrent().getName());
085        return null;
086      }
087    });
088  }
089
090  /**
091   * Make sure that we're returning a result for the current user.
092   * Previously getCurrent() was returning null if not initialized on
093   * non-secure Hadoop variants.
094   */
095  @Test
096  public void testGetCurrent() throws Exception {
097    User user1 = User.getCurrent();
098    assertNotNull(user1.ugi);
099    LOG.debug("User1 is "+user1.getName());
100
101    for (int i =0 ; i< 100; i++) {
102      User u = User.getCurrent();
103      assertNotNull(u);
104      assertEquals(user1.getName(), u.getName());
105      assertEquals(user1, u);
106      assertEquals(user1.hashCode(), u.hashCode());
107    }
108  }

112    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
113}
