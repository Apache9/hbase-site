018package org.apache.hadoop.hbase.regionserver.wal;
019
020import java.io.IOException;
021
022import org.apache.commons.logging.Log;
023import org.apache.commons.logging.LogFactory;
024import org.apache.hadoop.conf.Configuration;
025import org.apache.hadoop.fs.FileSystem;
026import org.apache.hadoop.fs.Path;
027import org.apache.hadoop.hbase.HBaseFileSystem;
028import org.apache.hadoop.hbase.regionserver.HRegionFileSystem;
029import org.apache.hadoop.hbase.regionserver.wal.HLog.Writer;
030
031/**
032 * Acts as an abstraction between the HLog and the underlying filesystem. This is analogous to the
033 * {@link HRegionFileSystem} class.
034 */
035public class HLogFileSystem extends HBaseFileSystem {
036  public static final Log LOG = LogFactory.getLog(HLogFileSystem.class);
037
038  /**
039   * In order to handle NN connectivity hiccups, one need to retry non-idempotent operation at the
040   * client level.
041   */
042
043  public HLogFileSystem(Configuration conf) {
044    setRetryCounts(conf);
045  }
046
047  /**
048   * Creates writer for the given path.
049   * @param fs
050   * @param conf
051   * @param hlogFile
052   * @return an init'ed writer for the given path.
053   * @throws IOException
054   */
055  public Writer createWriter(FileSystem fs, Configuration conf, Path hlogFile) throws IOException {
056    int i = 0;
057    IOException lastIOE = null;
058    do {
059      try {
060        return HLog.createWriter(fs, hlogFile, conf);
061      } catch (IOException ioe) {
062        lastIOE = ioe;
063        sleepBeforeRetry("Create Writer", i+1);
064      }
065    } while (++i <= hdfsClientRetriesNumber);
066    throw new IOException("Exception in createWriter", lastIOE);
067
068  }
069}
