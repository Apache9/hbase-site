020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.IOException;
023import java.util.SortedSet;
024
025import org.apache.hadoop.hbase.KeyValue;
026import org.apache.hadoop.hbase.client.Scan;
027
028/**
029 * Scanner that returns the next KeyValue.
030 */
031public interface KeyValueScanner {
032  /**
033   * Look at the next KeyValue in this scanner, but do not iterate scanner.
034   * @return the next KeyValue
035   */
036  public KeyValue peek();
037
038  /**
039   * Return the next KeyValue in this scanner, iterating the scanner
040   * @return the next KeyValue
041   */
042  public KeyValue next() throws IOException;
043
044  /**
045   * Seek the scanner at or after the specified KeyValue.
046   * @param key seek value
047   * @return true if scanner has values left, false if end of scanner
048   */
049  public boolean seek(KeyValue key) throws IOException;
050
051  /**
052   * Reseek the scanner at or after the specified KeyValue.
053   * This method is guaranteed to seek at or after the required key only if the
054   * key comes after the current position of the scanner. Should not be used
055   * to seek to a key which may come before the current position.
056   * @param key seek value (should be non-null)
057   * @return true if scanner has values left, false if end of scanner
058   */
059  public boolean reseek(KeyValue key) throws IOException;
060
061  /**
062   * Get the sequence id associated with this KeyValueScanner. This is required
063   * for comparing multiple files to find out which one has the latest data.
064   * The default implementation for this would be to return 0. A file having
065   * lower sequence id will be considered to be the older one.
066   */
067  public long getSequenceID();
068
069  /**
070   * Close the KeyValue scanner.
071   */
072  public void close();
073
074  /**
075   * Allows to filter out scanners (both StoreFile and memstore) that we don't
076   * want to use based on criteria such as Bloom filters and timestamp ranges.
077   * @param scan the scan that we are selecting scanners for
078   * @param columns the set of columns in the current column family, or null if
079   *          not specified by the scan
080   * @param oldestUnexpiredTS the oldest timestamp we are interested in for
081   *          this query, based on TTL
082   * @return true if the scanner should be included in the query
083   */
084  public boolean shouldUseScanner(Scan scan, SortedSet<byte[]> columns,
085      long oldestUnexpiredTS);
086
087  // "Lazy scanner" optimizations
088
089  /**
090   * Similar to {@link #seek} (or {@link #reseek} if forward is true) but only
091   * does a seek operation after checking that it is really necessary for the
092   * row/column combination specified by the kv parameter. This function was
093   * added to avoid unnecessary disk seeks by checking row-column Bloom filters
094   * before a seek on multi-column get/scan queries, and to optimize by looking
095   * up more recent files first.
096   * @param forward do a forward-only "reseek" instead of a random-access seek
097   * @param useBloom whether to enable multi-column Bloom filter optimization
098   */
099  public boolean requestSeek(KeyValue kv, boolean forward, boolean useBloom)
100      throws IOException;
101
102  /**
103   * We optimize our store scanners by checking the most recent store file
104   * first, so we sometimes pretend we have done a seek but delay it until the
105   * store scanner bubbles up to the top of the key-value heap. This method is
106   * then used to ensure the top store file scanner has done a seek operation.
107   */
108  public boolean realSeekDone();
109
110  /**
111   * Does the real seek operation in case it was skipped by
112   * seekToRowCol(KeyValue, boolean) (TODO: Whats this?). Note that this function should
113   * be never called on scanners that always do real seek operations (i.e. most
114   * of the scanners). The easiest way to achieve this is to call
115   * {@link #realSeekDone()} first.
116   */
117  public void enforceSeek() throws IOException;
118
119  /**
120   * @return true if this is a file scanner. Otherwise a memory scanner is
121   *         assumed.
122   */
123  public boolean isFileScanner();  
124}
