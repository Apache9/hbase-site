018package org.apache.hadoop.hbase.regionserver;
019
020import org.apache.hadoop.conf.Configuration;
021import org.apache.hadoop.hbase.HConstants;
022import org.apache.hadoop.hbase.HTableDescriptor;
023
024/**
025 * A {@link RegionSplitPolicy} implementation which splits a region
026 * as soon as any of its store files exceeds a maximum configurable
027 * size.
028 * <p>
029 * This is the default split policy. From 0.94.0 on the default split policy has
030 * changed to {@link IncreasingToUpperBoundRegionSplitPolicy}
031 * </p>
032 */
033public class ConstantSizeRegionSplitPolicy extends RegionSplitPolicy {
034  private long desiredMaxFileSize;
035
036  @Override
037  protected void configureForRegion(HRegion region) {
038    super.configureForRegion(region);
039    Configuration conf = getConf();
040    HTableDescriptor desc = region.getTableDesc();
041    if (desc != null) {
042      this.desiredMaxFileSize = desc.getMaxFileSize();
043    }
044    if (this.desiredMaxFileSize <= 0) {
045      this.desiredMaxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE,
046        HConstants.DEFAULT_MAX_FILE_SIZE);
047    }
048  }
049
050  @Override
051  protected boolean shouldSplit() {
052    boolean force = region.shouldForceSplit();
053    boolean foundABigStore = false;
054
055    for (Store store : region.getStores().values()) {
056      // If any of the stores are unable to split (eg they contain reference files)
057      // then don't split
058      if ((!store.canSplit())) {
059        return false;
060      }
061
062      // Mark if any store is big enough
063      if (store.getSize() > desiredMaxFileSize) {
064        foundABigStore = true;
065      }
066    }
067
068    return foundABigStore || force;
069  }
070
071  long getDesiredMaxFileSize() {
072    return desiredMaxFileSize;
073  }
074}
