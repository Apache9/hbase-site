020package org.apache.hadoop.hbase.regionserver;
021
022import java.io.DataInput;
023import java.io.DataOutput;
024import java.io.IOException;
025
026import org.apache.hadoop.hbase.KeyValue;
027import org.apache.hadoop.hbase.KeyValue.Type;
028import org.apache.hadoop.hbase.io.TimeRange;
029import org.apache.hadoop.hbase.util.Bytes;
030import org.apache.hadoop.io.Writable;
031
032/**
033 * Stores the minimum and maximum timestamp values (both are inclusive).
034 * Can be used to find if any given time range overlaps with its time range
035 * MemStores use this class to track its minimum and maximum timestamps.
036 * When writing StoreFiles, this information is stored in meta blocks and used
037 * at read time to match against the required TimeRange.
038 */
039public class TimeRangeTracker implements Writable {
040
041  long minimumTimestamp = -1;
042  long maximumTimestamp = -1;
043
044  /**
045   * Default constructor.
046   * Initializes TimeRange to be null
047   */
048  public TimeRangeTracker() {
049
050  }
051
052  /**
053   * Copy Constructor
054   * @param trt source TimeRangeTracker
055   */
056  public TimeRangeTracker(final TimeRangeTracker trt) {
057    this.minimumTimestamp = trt.getMinimumTimestamp();
058    this.maximumTimestamp = trt.getMaximumTimestamp();
059  }
060
061  public TimeRangeTracker(long minimumTimestamp, long maximumTimestamp) {
062    this.minimumTimestamp = minimumTimestamp;
063    this.maximumTimestamp = maximumTimestamp;
064  }
065
066  /**
067   * Update the current TimestampRange to include the timestamp from KeyValue
068   * If the Key is of type DeleteColumn or DeleteFamily, it includes the
069   * entire time range from 0 to timestamp of the key.
070   * @param kv the KeyValue to include
071   */
072  public void includeTimestamp(final KeyValue kv) {
073    includeTimestamp(kv.getTimestamp());
074    if (kv.isDeleteColumnOrFamily()) {
075      includeTimestamp(0);
076    }
077  }
078
079  /**
080   * Update the current TimestampRange to include the timestamp from Key.
081   * If the Key is of type DeleteColumn or DeleteFamily, it includes the
082   * entire time range from 0 to timestamp of the key.
083   * @param key
084   */
085  public void includeTimestamp(final byte[] key) {
086    includeTimestamp(Bytes.toLong(key,key.length-KeyValue.TIMESTAMP_TYPE_SIZE));
087    int type = key[key.length - 1];
088    if (type == Type.DeleteColumn.getCode() ||
089        type == Type.DeleteFamily.getCode()) {
090      includeTimestamp(0);
091    }
092  }
093
094  /**
095   * If required, update the current TimestampRange to include timestamp
096   * @param timestamp the timestamp value to include
097   */
098  private synchronized void includeTimestamp(final long timestamp) {
099    if (maximumTimestamp == -1) {
100      minimumTimestamp = timestamp;
101      maximumTimestamp = timestamp;
102    }
103    else if (minimumTimestamp > timestamp) {
104      minimumTimestamp = timestamp;
105    }
106    else if (maximumTimestamp < timestamp) {
107      maximumTimestamp = timestamp;
108    }
109    return;
110  }
111
112  /**
113   * Check if the range has any overlap with TimeRange
114   * @param tr TimeRange
115   * @return True if there is overlap, false otherwise
116   */
117  public synchronized boolean includesTimeRange(final TimeRange tr) {
118    return (this.minimumTimestamp < tr.getMax() &&
119        this.maximumTimestamp >= tr.getMin());
120  }
121
122  /**
123   * @return the minimumTimestamp
124   */
125  public synchronized long getMinimumTimestamp() {
126    return minimumTimestamp;
127  }
128
129  /**
130   * @return the maximumTimestamp
131   */
132  public synchronized long getMaximumTimestamp() {
133    return maximumTimestamp;
134  }
135
136  public synchronized void write(final DataOutput out) throws IOException {
137    out.writeLong(minimumTimestamp);
138    out.writeLong(maximumTimestamp);
139  }
140
141  public synchronized void readFields(final DataInput in) throws IOException {
142    this.minimumTimestamp = in.readLong();
143    this.maximumTimestamp = in.readLong();
144  }
145
146  @Override
147  public synchronized String toString() {
148    return "[" + minimumTimestamp + "," + maximumTimestamp + "]";
149  }
150
151}
