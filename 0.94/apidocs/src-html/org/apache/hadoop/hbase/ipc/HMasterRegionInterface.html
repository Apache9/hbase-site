020package org.apache.hadoop.hbase.ipc;
021
022import java.io.IOException;
023
024import org.apache.hadoop.hbase.HServerLoad;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.security.KerberosInfo;
027import org.apache.hadoop.io.MapWritable;
028import org.apache.hadoop.hbase.ipc.VersionedProtocol;
029
030/**
031 * The Master publishes this Interface for RegionServers to register themselves
032 * on.
033 */
034@KerberosInfo(
035    serverPrincipal = "hbase.master.kerberos.principal",
036    clientPrincipal = "hbase.regionserver.kerberos.principal")
037public interface HMasterRegionInterface extends VersionedProtocol {
038  /**
039   * This Interfaces' version. Version changes when the Interface changes.
040   */
041  // All HBase Interfaces used derive from HBaseRPCProtocolVersion.  It
042  // maintained a single global version number on all HBase Interfaces.  This
043  // meant all HBase RPC was broke though only one of the three RPC Interfaces
044  // had changed.  This has since been undone.
045  public static final long VERSION = 29L;
046
047  /**
048   * Called when a region server first starts.
049   * @param port Port number this regionserver is up on.
050   * @param serverStartcode This servers' startcode.
051   * @param serverCurrentTime The current time of the region server in ms
052   * @throws IOException e
053   * @return Configuration for the regionserver to use: e.g. filesystem,
054   * hbase rootdir, the hostname to use creating the RegionServer ServerName,
055   * etc.
056   */
057  public MapWritable regionServerStartup(final int port,
058    final long serverStartcode, final long serverCurrentTime)
059  throws IOException;
060
061  /**
062   * @param sn {@link ServerName#getVersionedBytes()}
063   * @param hsl Server load.
064   * @throws IOException
065   */
066  public void regionServerReport(byte [] sn, HServerLoad hsl)
067  throws IOException;
068  
069  /**
070   * Called by a region server to report a fatal error that is causing
071   * it to abort.
072   * @param sn {@link ServerName#getVersionedBytes()}
073   * @param errorMessage informative text to expose in the master logs and UI
074   */
075  public void reportRSFatalError(byte [] sn, String errorMessage);
076}
