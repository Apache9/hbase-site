020package org.apache.hadoop.hbase.replication.regionserver;
021import java.io.UnsupportedEncodingException;
022import java.net.URLEncoder;
023
024import org.apache.hadoop.hbase.metrics.MetricsRate;
025import org.apache.hadoop.metrics.MetricsContext;
026import org.apache.hadoop.metrics.MetricsRecord;
027import org.apache.hadoop.metrics.MetricsUtil;
028import org.apache.hadoop.metrics.Updater;
029import org.apache.hadoop.metrics.jvm.JvmMetrics;
030import org.apache.hadoop.metrics.util.MetricsIntValue;
031import org.apache.hadoop.metrics.util.MetricsLongValue;
032import org.apache.hadoop.metrics.util.MetricsRegistry;
033
034/**
035 * This class is for maintaining the various replication statistics
036 * for a source and publishing them through the metrics interfaces.
037 */
038public class ReplicationSourceMetrics implements Updater {
039  private final MetricsRecord metricsRecord;
040  private MetricsRegistry registry = new MetricsRegistry();
041  private ReplicationStatistics replicationStatistics;
042
043  /** Rate of shipped operations by the source */
044  public final MetricsRate shippedOpsRate =
045      new MetricsRate("shippedOpsRate", registry);
046
047  /** Rate of shipped bytes (in KB) by the source */
048  public final MetricsRate shippedKBRate =
049      new MetricsRate("shippedBytesRate", registry);
050
051  /** Rate of shipped batches by the source */
052  public final MetricsRate shippedBatchesRate =
053      new MetricsRate("shippedBatchesRate", registry);
054
055  /** Rate of log entries (can be multiple Puts) read from the logs */
056  public final MetricsRate logEditsReadRate =
057      new MetricsRate("logEditsReadRate", registry);
058
059  /** Rate of log entries filtered by the source */
060  public final MetricsRate logEditsFilteredRate =
061      new MetricsRate("logEditsFilteredRate", registry);
062
063  /** Age of the last operation that was shipped by the source */
064  private final MetricsLongValue ageOfLastShippedOp =
065      new MetricsLongValue("ageOfLastShippedOp", registry);
066
067  /**
068   * Current size of the queue of logs to replicate,
069   * excluding the one being processed at the moment
070   */
071  public final MetricsIntValue sizeOfLogQueue =
072      new MetricsIntValue("sizeOfLogQueue", registry);
073
074  // It's a little dirty to preset the age to now since if we fail
075  // to replicate the very first time then it will show that age instead
076  // of nothing (although that might not be good either).
077  private long lastTimestampForAge = System.currentTimeMillis();
078
079  /**
080   * Constructor used to register the metrics
081   * @param id Name of the source this class is monitoring
082   */
083  public ReplicationSourceMetrics(String id) {
084    MetricsContext context = MetricsUtil.getContext("hbase");
085    String name = Thread.currentThread().getName();
086    metricsRecord = MetricsUtil.createRecord(context, "replication");
087    metricsRecord.setTag("RegionServer", name);
088    context.registerUpdater(this);
089    try {
090      id = URLEncoder.encode(id, "UTF8");
091    } catch (UnsupportedEncodingException e) {
092      id = "CAN'T ENCODE UTF8";
093    }
094    // export for JMX
095    replicationStatistics = new ReplicationStatistics(this.registry, "ReplicationSource for " + id);
096  }
097
098  /**
099   * Set the age of the last edit that was shipped
100   * @param timestamp write time of the edit
101   */
102  public void setAgeOfLastShippedOp(long timestamp) {
103    lastTimestampForAge = timestamp;
104    ageOfLastShippedOp.set(System.currentTimeMillis() - lastTimestampForAge);
105  }
106
107  /**
108   * Convenience method to use the last given timestamp to refresh the age
109   * of the last edit. Used when replication fails and need to keep that
110   * metric accurate.
111   */
112  public void refreshAgeOfLastShippedOp() {
113    setAgeOfLastShippedOp(lastTimestampForAge);
114  }
115
116  @Override
117  public void doUpdates(MetricsContext metricsContext) {
118    synchronized (this) {
119      refreshAgeOfLastShippedOp();
120      this.shippedOpsRate.pushMetric(this.metricsRecord);
121      this.shippedBatchesRate.pushMetric(this.metricsRecord);
122      this.shippedKBRate.pushMetric(this.metricsRecord);
123      this.logEditsReadRate.pushMetric(this.metricsRecord);
124      this.logEditsFilteredRate.pushMetric(this.metricsRecord);
125      this.ageOfLastShippedOp.pushMetric(this.metricsRecord);
126      this.sizeOfLogQueue.pushMetric(this.metricsRecord);
127    }
128    this.metricsRecord.update();
129  }
130  
131  public void stopReportMetrics() {
132    if (this.replicationStatistics != null) {
133      this.replicationStatistics.unRegisterMBean();
134    }
135  }
136}
