020package org.apache.hadoop.hbase.replication;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.List;
025import java.util.concurrent.atomic.AtomicBoolean;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029import org.apache.hadoop.conf.Configuration;
030import org.apache.hadoop.hbase.Abortable;
031import org.apache.hadoop.hbase.ServerName;
032import org.apache.hadoop.hbase.replication.ReplicationZookeeper.PeerState;
033import org.apache.hadoop.hbase.util.Bytes;
034import org.apache.hadoop.hbase.zookeeper.ZKUtil;
035import org.apache.hadoop.hbase.zookeeper.ZooKeeperNodeTracker;
036import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
037import org.apache.zookeeper.KeeperException;
038
039/**
040 * This class acts as a wrapper for all the objects used to identify and
041 * communicate with remote peers and is responsible for answering to expired
042 * sessions and re-establishing the ZK connections.
043 */
044public class ReplicationPeer implements Abortable {
045  private static final Log LOG = LogFactory.getLog(ReplicationPeer.class);
046
047  private final String clusterKey;
048  private final String id;
049  private List<ServerName> regionServers = new ArrayList<ServerName>(0);
050  private final AtomicBoolean peerEnabled = new AtomicBoolean();
051  // Cannot be final since a new object needs to be recreated when session fails
052  private ZooKeeperWatcher zkw;
053  private final Configuration conf;
054
055  private PeerStateTracker peerStateTracker;
056
057  /**
058   * Constructor that takes all the objects required to communicate with the
059   * specified peer, except for the region server addresses.
060   * @param conf configuration object to this peer
061   * @param key cluster key used to locate the peer
062   * @param id string representation of this peer's identifier
063   */
064  public ReplicationPeer(Configuration conf, String key,
065      String id) throws IOException {
066    this.conf = conf;
067    this.clusterKey = key;
068    this.id = id;
069    this.reloadZkWatcher();
070  }
071
072  /**
073   * start a state tracker to check whether this peer is enabled or not
074   *
075   * @param zookeeper zk watcher for the local cluster
076   * @param peerStateNode path to zk node which stores peer state
077   * @throws KeeperException
078   */
079  public void startStateTracker(ZooKeeperWatcher zookeeper, String peerStateNode)
080      throws KeeperException {
081    if (ZKUtil.checkExists(zookeeper, peerStateNode) == -1) {
082      // There is a race b/w PeerWatcher and ReplicationZookeeper#add method to create the
083      // peer-state znode. This happens while adding a peer.
084      // The peer state data is set as "ENABLED" by default.
085      ZKUtil.createNodeIfNotExistsAndWatch(zookeeper, peerStateNode,
086        Bytes.toBytes(PeerState.ENABLED.name()));
087    }
088    this.peerStateTracker = new PeerStateTracker(peerStateNode, zookeeper,
089        this);
090    this.peerStateTracker.start();
091    this.readPeerStateZnode();
092  }
093
094  private void readPeerStateZnode() {
095    String currentState = Bytes.toString(peerStateTracker.getData(false));
096    this.peerEnabled.set(PeerState.ENABLED.equals(PeerState
097        .valueOf(currentState)));
098  }
099
100  /**
101   * Get the cluster key of that peer
102   * @return string consisting of zk ensemble addresses, client port
103   * and root znode
104   */
105  public String getClusterKey() {
106    return clusterKey;
107  }
108
109  /**
110   * Get the state of this peer
111   * @return atomic boolean that holds the status
112   */
113  public AtomicBoolean getPeerEnabled() {
114    return peerEnabled;
115  }
116
117  /**
118   * Get a list of all the addresses of all the region servers
119   * for this peer cluster
120   * @return list of addresses
121   */
122  public List<ServerName> getRegionServers() {
123    return regionServers;
124  }
125
126  /**
127   * Set the list of region servers for that peer
128   * @param regionServers list of addresses for the region servers
129   */
130  public void setRegionServers(List<ServerName> regionServers) {
131    this.regionServers = regionServers;
132  }
133
134  /**
135   * Get the ZK connection to this peer
136   * @return zk connection
137   */
138  public ZooKeeperWatcher getZkw() {
139    return zkw;
140  }
141
142  /**
143   * Get the identifier of this peer
144   * @return string representation of the id (short)
145   */
146  public String getId() {
147    return id;
148  }
149
150  /**
151   * Get the configuration object required to communicate with this peer
152   * @return configuration object
153   */
154  public Configuration getConfiguration() {
155    return conf;
156  }
157
158  @Override
159  public void abort(String why, Throwable e) {
160    LOG.fatal("The ReplicationPeer coresponding to peer " + clusterKey
161        + " was aborted for the following reason(s):" + why, e);
162  }
163
164  /**
165   * Closes the current ZKW (if not null) and creates a new one
166   * @throws IOException If anything goes wrong connecting
167   */
168  public void reloadZkWatcher() throws IOException {
169    if (zkw != null) zkw.close();
170    zkw = new ZooKeeperWatcher(conf,
171        "connection to cluster: " + id, this);    
172  }
173
174  @Override
175  public boolean isAborted() {
176    // Currently the replication peer is never "Aborted", we just log when the
177    // abort method is called.
178    return false;
179  }
180
181  /**
182   * Tracker for state of this peer
183   */
184  public class PeerStateTracker extends ZooKeeperNodeTracker {
185
186    public PeerStateTracker(String peerStateZNode, ZooKeeperWatcher watcher,
187        Abortable abortable) {
188      super(watcher, peerStateZNode, abortable);
189    }
190
191    @Override
192    public synchronized void nodeDataChanged(String path) {
193      if (path.equals(node)) {
194        super.nodeDataChanged(path);
195        readPeerStateZnode();
196      }
197    }
198  }
199}
