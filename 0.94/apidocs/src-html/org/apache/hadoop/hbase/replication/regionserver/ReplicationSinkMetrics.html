020package org.apache.hadoop.hbase.replication.regionserver;
021import org.apache.hadoop.hbase.metrics.MetricsRate;
022import org.apache.hadoop.metrics.MetricsContext;
023import org.apache.hadoop.metrics.MetricsRecord;
024import org.apache.hadoop.metrics.MetricsUtil;
025import org.apache.hadoop.metrics.Updater;
026import org.apache.hadoop.metrics.jvm.JvmMetrics;
027import org.apache.hadoop.metrics.util.MetricsIntValue;
028import org.apache.hadoop.metrics.util.MetricsLongValue;
029import org.apache.hadoop.metrics.util.MetricsRegistry;
030
031/**
032 * This class is for maintaining the various replication statistics
033 * for a sink and publishing them through the metrics interfaces.
034 */
035public class ReplicationSinkMetrics implements Updater {
036  private final MetricsRecord metricsRecord;
037  private MetricsRegistry registry = new MetricsRegistry();
038  private static ReplicationSinkMetrics instance;
039
040  /** Rate of operations applied by the sink */
041  public final MetricsRate appliedOpsRate =
042      new MetricsRate("appliedOpsRate", registry);
043
044  /** Rate of batches (of operations) applied by the sink */
045  public final MetricsRate appliedBatchesRate =
046      new MetricsRate("appliedBatchesRate", registry);
047
048  /** Age of the last operation that was applied by the sink */
049  private final MetricsLongValue ageOfLastAppliedOp =
050      new MetricsLongValue("ageOfLastAppliedOp", registry);
051
052  /**
053   * Constructor used to register the metrics
054   */
055  public ReplicationSinkMetrics() {
056    MetricsContext context = MetricsUtil.getContext("hbase");
057    String name = Thread.currentThread().getName();
058    metricsRecord = MetricsUtil.createRecord(context, "replication");
059    metricsRecord.setTag("RegionServer", name);
060    context.registerUpdater(this);
061    // export for JMX
062    new ReplicationStatistics(this.registry, "ReplicationSink");
063  }
064
065  /**
066   * Set the age of the last edit that was applied
067   * @param timestamp write time of the edit
068   */
069  public void setAgeOfLastAppliedOp(long timestamp) {
070    ageOfLastAppliedOp.set(System.currentTimeMillis() - timestamp);
071  }
072  @Override
073  public void doUpdates(MetricsContext metricsContext) {
074    synchronized (this) {
075      this.appliedOpsRate.pushMetric(this.metricsRecord);
076      this.appliedBatchesRate.pushMetric(this.metricsRecord);
077      this.ageOfLastAppliedOp.pushMetric(this.metricsRecord);
078    }
079    this.metricsRecord.update();
080  }
081}
