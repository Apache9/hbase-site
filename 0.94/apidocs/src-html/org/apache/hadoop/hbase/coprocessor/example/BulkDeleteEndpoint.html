019package org.apache.hadoop.hbase.coprocessor.example;
020
021import java.io.IOException;
022import java.util.ArrayList;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Set;
026import java.util.TreeSet;
027
028import org.apache.commons.logging.Log;
029import org.apache.commons.logging.LogFactory;
030import org.apache.hadoop.hbase.HConstants;
031import org.apache.hadoop.hbase.HConstants.OperationStatusCode;
032import org.apache.hadoop.hbase.KeyValue;
033import org.apache.hadoop.hbase.client.Delete;
034import org.apache.hadoop.hbase.client.Mutation;
035import org.apache.hadoop.hbase.client.Scan;
036import org.apache.hadoop.hbase.coprocessor.BaseEndpointCoprocessor;
037import org.apache.hadoop.hbase.coprocessor.RegionCoprocessorEnvironment;
038import org.apache.hadoop.hbase.filter.FirstKeyOnlyFilter;
039import org.apache.hadoop.hbase.regionserver.HRegion;
040import org.apache.hadoop.hbase.regionserver.OperationStatus;
041import org.apache.hadoop.hbase.regionserver.RegionScanner;
042import org.apache.hadoop.hbase.util.Bytes;
043import org.apache.hadoop.hbase.util.Pair;
044
045public class BulkDeleteEndpoint extends BaseEndpointCoprocessor implements BulkDeleteProtocol {
046  private static final String NO_OF_VERSIONS_TO_DELETE = "noOfVersionsToDelete";
047  private static final Log LOG = LogFactory.getLog(BulkDeleteEndpoint.class);
048  
049  @Override
050  public BulkDeleteResponse delete(Scan scan, byte deleteType, Long timestamp,
051      int rowBatchSize) {
052    long totalRowsDeleted = 0L;
053    long totalVersionsDeleted = 0L;
054    BulkDeleteResponse response = new BulkDeleteResponse();
055    HRegion region = ((RegionCoprocessorEnvironment) getEnvironment()).getRegion();
056    boolean hasMore = true;
057    RegionScanner scanner = null;
058    if (scan.getFilter() == null && deleteType == DeleteType.ROW) {
059      // What we need is just the rowkeys. So only 1st KV from any row is enough.
060      // Only when it is a row delete, we can apply this filter
061      // In other types we rely on the scan to know which all columns to be deleted.
062      scan.setFilter(new FirstKeyOnlyFilter());
063    }
064    // When the delete is based on some conditions so that Filters are available in the scan,
065    // we assume that the scan is perfect having necessary column(s) only.
066    try {
067      scanner = region.getScanner(scan);
068      while (hasMore) {
069        List<List<KeyValue>> deleteRows = new ArrayList<List<KeyValue>>(rowBatchSize);
070        for (int i = 0; i < rowBatchSize; i++) {
071          List<KeyValue> results = new ArrayList<KeyValue>();
072          hasMore = scanner.next(results);
073          if (results.size() > 0) {
074            deleteRows.add(results);
075          }
076          if (!hasMore) {
077            // There are no more rows.
078            break;
079          }
080        }
081        if (deleteRows.size() > 0) {
082          Pair<Mutation, Integer>[] deleteWithLockArr = new Pair[deleteRows.size()];
083          int i = 0;
084          for (List<KeyValue> deleteRow : deleteRows) {
085            Delete delete = createDeleteMutation(deleteRow, deleteType, timestamp);
086            deleteWithLockArr[i++] = new Pair<Mutation, Integer>(delete, null);
087          }
088          OperationStatus[] opStatus = region.batchMutate(deleteWithLockArr);
089          for (i = 0; i < opStatus.length; i++) {
090            if (opStatus[i].getOperationStatusCode() != OperationStatusCode.SUCCESS) {
091              break;
092            }
093            totalRowsDeleted++;
094            if (deleteType == DeleteType.VERSION) {
095              byte[] versionsDeleted = deleteWithLockArr[i].getFirst().getAttribute(
096                  NO_OF_VERSIONS_TO_DELETE);
097              if (versionsDeleted != null) {
098                totalVersionsDeleted += Bytes.toInt(versionsDeleted);
099              }
100            }
101          }
102        } 
103      }
104    } catch (IOException ioe) {
105      LOG.error(ioe);
106      response.setIoException(ioe);
107    } finally {
108      if (scanner != null) {
109        try {
110          scanner.close();
111        } catch (IOException ioe) {
112          LOG.error(ioe);
113        }
114      }
115    }
116    response.setRowsDeleted(totalRowsDeleted);
117    response.setVersionsDeleted(totalVersionsDeleted);
118    return response;
119  }
120
121  private Delete createDeleteMutation(List<KeyValue> deleteRow, byte deleteType, Long timestamp) {
122    long ts;
123    if (timestamp == null) {
124      ts = HConstants.LATEST_TIMESTAMP;
125    } else {
126      ts = timestamp;
127    }
128    // We just need the rowkey. Get it from 1st KV.
129    byte[] row = deleteRow.get(0).getRow();
130    Delete delete = new Delete(row, ts, null);
131    if (deleteType != DeleteType.ROW) {
132      switch (deleteType) {
133      case DeleteType.FAMILY:
134        Set<byte[]> families = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
135        for (KeyValue kv : deleteRow) {
136          if (families.add(kv.getFamily())) {
137            delete.deleteFamily(kv.getFamily(), ts);
138          }
139        }
140        break;
141
142      case DeleteType.COLUMN:
143        Set<Column> columns = new HashSet<Column>();
144        for (KeyValue kv : deleteRow) {
145          Column column = new Column(kv.getFamily(), kv.getQualifier());
146          if (columns.add(column)) {
147            // Making deleteColumns() calls more than once for the same cf:qualifier is not correct
148            // Every call to deleteColumns() will add a new KV to the familymap which will finally
149            // get written to the memstore as part of delete().
150            delete.deleteColumns(column.family, column.qualifier, ts);
151          }
152        }
153        break;
154
155      case DeleteType.VERSION:
156        // When some timestamp was passed to the delete() call only one version of the column (with
157        // given timestamp) will be deleted. If no timestamp passed, it will delete N versions.
158        // How many versions will get deleted depends on the Scan being passed. All the KVs that
159        // the scan fetched will get deleted.
160        int noOfVersionsToDelete = 0;
161        if (timestamp == null) {
162          for (KeyValue kv : deleteRow) {
163            delete.deleteColumn(kv.getFamily(), kv.getQualifier(), kv.getTimestamp());
164            noOfVersionsToDelete++;
165          }
166        } else {
167          columns = new HashSet<Column>();
168          for (KeyValue kv : deleteRow) {
169            Column column = new Column(kv.getFamily(), kv.getQualifier());
170            // Only one version of particular column getting deleted.
171            if (columns.add(column)) {
172              delete.deleteColumn(column.family, column.qualifier, ts);
173              noOfVersionsToDelete++;
174            }
175          }
176        }
177        delete.setAttribute(NO_OF_VERSIONS_TO_DELETE, Bytes.toBytes(noOfVersionsToDelete));
178      }
179    }
180    return delete;
181  }
182  
183  private static class Column {
184    private byte[] family;
185    private byte[] qualifier;
186
187    public Column(byte[] family, byte[] qualifier) {
188      this.family = family;
189      this.qualifier = qualifier;
190    }
191
192    @Override
193    public boolean equals(Object other) {
194      if (!(other instanceof Column)) {
195        return false;
196      }
197      Column column = (Column) other;
198      return Bytes.equals(this.family, column.family)
199          && Bytes.equals(this.qualifier, column.qualifier);
200    }
201
202    @Override
203    public int hashCode() {
204      int h = 31;
205      h = h + 13 * Bytes.hashCode(this.family);
206      h = h + 13 * Bytes.hashCode(this.qualifier);
207      return h;
208    }
209  }
