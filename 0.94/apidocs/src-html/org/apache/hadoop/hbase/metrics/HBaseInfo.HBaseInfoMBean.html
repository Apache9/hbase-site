020package org.apache.hadoop.hbase.metrics;
021
022import org.apache.hadoop.hbase.metrics.MetricsMBeanBase;
023import org.apache.hadoop.metrics.MetricsContext;
024import org.apache.hadoop.metrics.MetricsRecord;
025import org.apache.hadoop.metrics.MetricsUtil;
026import org.apache.hadoop.metrics.util.MBeanUtil;
027import org.apache.hadoop.metrics.util.MetricsRegistry;
028
029import javax.management.ObjectName;
030
031/**
032 * Exports HBase system information as an MBean for JMX observation.
033 */
034public class HBaseInfo {
035  protected static class HBaseInfoMBean extends MetricsMBeanBase {
036    private final ObjectName mbeanName;
037  
038    public HBaseInfoMBean(MetricsRegistry registry, String rsName) {
039      super(registry, "HBase cluster information");
040      // The name seems wrong to me; should include clusterid IMO.
041      // That would make it harder to locate and rare we have
042      // two clusters up on single machine. St.Ack 20120309
043      mbeanName = MBeanUtil.registerMBean("HBase", "Info", this);
044    }
045  
046    public void shutdown() {
047      if (mbeanName != null)
048        MBeanUtil.unregisterMBean(mbeanName);
049    }
050  }
051
052  protected final MetricsRecord mr;
053  protected final HBaseInfoMBean mbean;
054  protected MetricsRegistry registry = new MetricsRegistry();
055
056  private static HBaseInfo theInstance = null;
057  public synchronized static HBaseInfo init() {
058      if (theInstance == null) {
059        theInstance = new HBaseInfo();
060      }
061      return theInstance;
062  }
063  
064  // HBase jar info
065  private MetricsString date = new MetricsString("date", registry,
066      org.apache.hadoop.hbase.util.VersionInfo.getDate());
067  private MetricsString revision = new MetricsString("revision", registry, 
068      org.apache.hadoop.hbase.util.VersionInfo.getRevision());
069  private MetricsString url = new MetricsString("url", registry,
070      org.apache.hadoop.hbase.util.VersionInfo.getUrl());
071  private MetricsString user = new MetricsString("user", registry,
072      org.apache.hadoop.hbase.util.VersionInfo.getUser());
073  private MetricsString version = new MetricsString("version", registry,
074      org.apache.hadoop.hbase.util.VersionInfo.getVersion());
075
076  // Info on the HDFS jar that HBase has (aka: HDFS Client)
077  private MetricsString hdfsDate = new MetricsString("hdfsDate", registry,
078      org.apache.hadoop.util.VersionInfo.getDate());
079  private MetricsString hdfsRev = new MetricsString("hdfsRevision", registry,
080      org.apache.hadoop.util.VersionInfo.getRevision());
081  private MetricsString hdfsUrl = new MetricsString("hdfsUrl", registry,
082      org.apache.hadoop.util.VersionInfo.getUrl());
083  private MetricsString hdfsUser = new MetricsString("hdfsUser", registry,
084      org.apache.hadoop.util.VersionInfo.getUser());
085  private MetricsString hdfsVer = new MetricsString("hdfsVersion", registry,
086      org.apache.hadoop.util.VersionInfo.getVersion());
087
088  protected HBaseInfo() {
089    MetricsContext context = MetricsUtil.getContext("hbase");
090    mr = MetricsUtil.createRecord(context, "info");
091    String name = Thread.currentThread().getName();
092    mr.setTag("Info", name);
093
094    // export for JMX
095    mbean = new HBaseInfoMBean(this.registry, name);
096  }
097
098}
