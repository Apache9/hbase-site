020package org.apache.hadoop.hbase.thrift;
021
022import java.lang.reflect.Method;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.thrift.generated.Hbase;
028import org.apache.hadoop.metrics.MetricsContext;
029import org.apache.hadoop.metrics.MetricsRecord;
030import org.apache.hadoop.metrics.MetricsUtil;
031import org.apache.hadoop.metrics.Updater;
032import org.apache.hadoop.metrics.util.MetricsBase;
033import org.apache.hadoop.metrics.util.MetricsIntValue;
034import org.apache.hadoop.metrics.util.MetricsRegistry;
035import org.apache.hadoop.metrics.util.MetricsTimeVaryingInt;
036import org.apache.hadoop.metrics.util.MetricsTimeVaryingLong;
037import org.apache.hadoop.metrics.util.MetricsTimeVaryingRate;
038
039/**
040 * This class is for maintaining the various statistics of thrift server
041 * and publishing them through the metrics interfaces.
042 */
043public class ThriftMetrics implements Updater {
044  public final static Log LOG = LogFactory.getLog(ThriftMetrics.class);
045  public final static String CONTEXT_NAME = "thriftserver";
046
047  private final MetricsContext context;
048  private final MetricsRecord metricsRecord;
049  private final MetricsRegistry registry = new MetricsRegistry();
050  private final long slowResponseTime;
051  public static final String SLOW_RESPONSE_NANO_SEC =
052    "hbase.thrift.slow.response.nano.second";
053  public static final long DEFAULT_SLOW_RESPONSE_NANO_SEC = 10 * 1000 * 1000;
054
055  private final MetricsIntValue callQueueLen =
056      new MetricsIntValue("callQueueLen", registry);
057  private final MetricsTimeVaryingRate numRowKeysInBatchGet =
058      new MetricsTimeVaryingRate("numRowKeysInBatchGet", registry);
059  private final MetricsTimeVaryingRate numRowKeysInBatchMutate =
060      new MetricsTimeVaryingRate("numRowKeysInBatchMutate", registry);
061  private final MetricsTimeVaryingRate timeInQueue =
062      new MetricsTimeVaryingRate("timeInQueue", registry);
063  private MetricsTimeVaryingRate thriftCall =
064      new MetricsTimeVaryingRate("thriftCall", registry);
065  private MetricsTimeVaryingRate slowThriftCall =
066      new MetricsTimeVaryingRate("slowThriftCall", registry);
067
068  public ThriftMetrics(int port, Configuration conf, Class<?> iface) {
069    slowResponseTime = conf.getLong(
070        SLOW_RESPONSE_NANO_SEC, DEFAULT_SLOW_RESPONSE_NANO_SEC);
071    context = MetricsUtil.getContext(CONTEXT_NAME);
072    metricsRecord = MetricsUtil.createRecord(context, CONTEXT_NAME);
073
074    metricsRecord.setTag("port", port + "");
075
076    LOG.info("Initializing RPC Metrics with port=" + port);
077
078    context.registerUpdater(this);
079
080    createMetricsForMethods(iface);
081  }
082
083  public void incTimeInQueue(long time) {
084    timeInQueue.inc(time);
085  }
086
087  public void setCallQueueLen(int len) {
088    callQueueLen.set(len);
089  }
090
091  public void incNumRowKeysInBatchGet(int diff) {
092    numRowKeysInBatchGet.inc(diff);
093  }
094
095  public void incNumRowKeysInBatchMutate(int diff) {
096    numRowKeysInBatchMutate.inc(diff);
097  }
098
099  public void incMethodTime(String name, int time) {
100    MetricsTimeVaryingRate methodTimeMetrc = getMethodTimeMetrics(name);
101    if (methodTimeMetrc == null) {
102      LOG.warn(
103          "Got incMethodTime() request for method that doesnt exist: " + name);
104      return; // ignore methods that dont exist.
105    }
106
107    // inc method specific processTime
108    methodTimeMetrc.inc(time);
109
110    // inc general processTime
111    thriftCall.inc(time);
112    if (time > slowResponseTime) {
113      slowThriftCall.inc(time);
114    }
115  }
116
117  private void createMetricsForMethods(Class<?> iface) {
118    LOG.debug("Creating metrics for interface " + iface.toString());
119    for (Method m : iface.getDeclaredMethods()) {
120      if (getMethodTimeMetrics(m.getName()) == null)
121        LOG.debug("Creating metrics for method:" + m.getName());
122        createMethodTimeMetrics(m.getName());
123    }
124  }
125
126  private MetricsTimeVaryingRate getMethodTimeMetrics(String key) {
127    return (MetricsTimeVaryingRate) registry.get(key);
128  }
129
130  private MetricsTimeVaryingRate createMethodTimeMetrics(String key) {
131    return new MetricsTimeVaryingRate(key, this.registry);
132  }
133
134  /**
135   * Push the metrics to the monitoring subsystem on doUpdate() call.
136   */
137  public void doUpdates(final MetricsContext context) {
138    // getMetricsList() and pushMetric() are thread safe methods
139    for (MetricsBase m : registry.getMetricsList()) {
140      m.pushMetric(metricsRecord);
141    }
142    metricsRecord.update();
143  }
144}
