020package org.apache.hadoop.hbase.client;
021
022import java.io.IOException;
023import java.lang.InterruptedException;
024import java.util.ArrayList;
025import java.util.HashMap;
026import java.util.List;
027import java.util.Map;
028import org.apache.hadoop.hbase.HRegionLocation;
029import org.apache.hadoop.hbase.client.HTable;
030import org.apache.hadoop.hbase.client.Put;
031import org.apache.hadoop.hbase.client.Row;
032
033/**
034 * Utility class for HTable.
035 * 
036 *
037 */
038public class HTableUtil {
039
040  private static final int INITIAL_LIST_SIZE = 250;
041  
042  /**
043   * Processes a List of Puts and writes them to an HTable instance in RegionServer buckets via the htable.put method. 
044   * This will utilize the writeBuffer, thus the writeBuffer flush frequency may be tuned accordingly via htable.setWriteBufferSize. 
045   * <br><br>
046   * The benefit of submitting Puts in this manner is to minimize the number of RegionServer RPCs in each flush.
047   * <br><br>
048   * Assumption #1:  Regions have been pre-created for the table.  If they haven't, then all of the Puts will go to the same region, 
049   * defeating the purpose of this utility method. See the Apache HBase book for an explanation of how to do this.
050   * <br>
051   * Assumption #2:  Row-keys are not monotonically increasing.  See the Apache HBase book for an explanation of this problem.  
052   * <br>
053   * Assumption #3:  That the input list of Puts is big enough to be useful (in the thousands or more).  The intent of this
054   * method is to process larger chunks of data.
055   * <br>
056   * Assumption #4:  htable.setAutoFlush(false) has been set.  This is a requirement to use the writeBuffer.
057   * <br><br>
058   * @param htable HTable instance for target HBase table
059   * @param puts List of Put instances
060   * @throws IOException if a remote or network exception occurs
061   * 
062   */
063  public static void bucketRsPut(HTable htable, List<Put> puts) throws IOException {
064
065    Map<String, List<Put>> putMap = createRsPutMap(htable, puts);
066    for (List<Put> rsPuts: putMap.values()) {
067      htable.put( rsPuts );
068    }
069    htable.flushCommits();
070  }
071  
072  /**
073   * Processes a List of Rows (Put, Delete) and writes them to an HTable instance in RegionServer buckets via the htable.batch method. 
074   * <br><br>
075   * The benefit of submitting Puts in this manner is to minimize the number of RegionServer RPCs, thus this will
076   * produce one RPC of Puts per RegionServer.
077   * <br><br>
078   * Assumption #1:  Regions have been pre-created for the table.  If they haven't, then all of the Puts will go to the same region, 
079   * defeating the purpose of this utility method. See the Apache HBase book for an explanation of how to do this.
080   * <br>
081   * Assumption #2:  Row-keys are not monotonically increasing.  See the Apache HBase book for an explanation of this problem.  
082   * <br>
083   * Assumption #3:  That the input list of Rows is big enough to be useful (in the thousands or more).  The intent of this
084   * method is to process larger chunks of data.
085   * <br><br>
086   * This method accepts a list of Row objects because the underlying .batch method accepts a list of Row objects.
087   * <br><br>
088   * @param htable HTable instance for target HBase table
089   * @param rows List of Row instances
090   * @throws IOException if a remote or network exception occurs
091   */
092  public static void bucketRsBatch(HTable htable, List<Row> rows) throws IOException {
093
094    try {
095      Map<String, List<Row>> rowMap = createRsRowMap(htable, rows);
096      for (List<Row> rsRows: rowMap.values()) {
097        htable.batch( rsRows );
098      }
099    } catch (InterruptedException e) {
100      throw new IOException(e); 
101    }
102    
103  }
104
105  private static Map<String,List<Put>> createRsPutMap(HTable htable, List<Put> puts) throws IOException {
106
107    Map<String, List<Put>> putMap = new HashMap<String, List<Put>>();
108    for (Put put: puts) {
109      HRegionLocation rl = htable.getRegionLocation( put.getRow() );
110      String hostname = rl.getHostname();
111      List<Put> recs = putMap.get( hostname);
112      if (recs == null) {
113        recs = new ArrayList<Put>(INITIAL_LIST_SIZE);
114        putMap.put( hostname, recs);
115      }
116      recs.add(put);
117    }
118    return putMap;
119  }
120
121  private static Map<String,List<Row>> createRsRowMap(HTable htable, List<Row> rows) throws IOException {
122
123    Map<String, List<Row>> rowMap = new HashMap<String, List<Row>>();
124    for (Row row: rows) {
125      HRegionLocation rl = htable.getRegionLocation( row.getRow() );
126      String hostname = rl.getHostname();
127      List<Row> recs = rowMap.get( hostname);
128      if (recs == null) {
129        recs = new ArrayList<Row>(INITIAL_LIST_SIZE);
130        rowMap.put( hostname, recs);
131      }
132      recs.add(row);
133    }
134    return rowMap;
135  }
136    
137}
