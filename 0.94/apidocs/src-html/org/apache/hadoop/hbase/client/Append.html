018package org.apache.hadoop.hbase.client;
019
020import java.io.DataInput;
021import java.io.DataOutput;
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026import java.util.Map;
027
028import org.apache.hadoop.hbase.KeyValue;
029import org.apache.hadoop.hbase.util.Bytes;
030
031/**
032 * Performs Append operations on a single row.
033 * <p>
034 * Note that this operation does not appear atomic to readers. Appends are done
035 * under a single row lock, so write operations to a row are synchronized, but
036 * readers do not take row locks so get and scan operations can see this
037 * operation partially completed.
038 * <p>
039 * To append to a set of columns of a row, instantiate an Append object with the
040 * row to append to. At least one column to append must be specified using the
041 * {@link #add(byte[], byte[], byte[])} method.
042 */
043public class Append extends Mutation {
044  private static final String RETURN_RESULTS = "_rr_";
045  private static final byte APPEND_VERSION = (byte)1;
046
047  /**
048   * @param returnResults
049   *          True (default) if the append operation should return the results.
050   *          A client that is not interested in the result can save network
051   *          bandwidth setting this to false.
052   */
053  public void setReturnResults(boolean returnResults) {
054    setAttribute(RETURN_RESULTS, Bytes.toBytes(returnResults));
055  }
056
057  /**
058   * @return current setting for returnResults
059   */
060  public boolean isReturnResults() {
061    byte[] v = getAttribute(RETURN_RESULTS);
062    return v == null ? true : Bytes.toBoolean(v);
063  }
064
065  /** Constructor for Writable.  DO NOT USE */
066  public Append() {}
067
068  /**
069   * Create a Append operation for the specified row.
070   * <p>
071   * At least one column must be appended to.
072   * @param row row key
073   */
074  public Append(byte[] row) {
075    this.row = Arrays.copyOf(row, row.length);
076  }
077
078  /**
079   * Add the specified column and value to this Append operation.
080   * @param family family name
081   * @param qualifier column qualifier
082   * @param value value to append to specified column
083   * @return this
084   */
085  public Append add(byte [] family, byte [] qualifier, byte [] value) {
086    List<KeyValue> list = familyMap.get(family);
087    if(list == null) {
088      list = new ArrayList<KeyValue>();
089    }
090    list.add(new KeyValue(
091        this.row, family, qualifier, this.ts, KeyValue.Type.Put, value));
092    familyMap.put(family, list);
093    return this;
094  }
095
096  @Override
097  public void readFields(final DataInput in)
098  throws IOException {
099    int version = in.readByte();
100    if (version > APPEND_VERSION) {
101      throw new IOException("version not supported: "+version);
102    }
103    this.row = Bytes.readByteArray(in);
104    this.ts = in.readLong();
105    this.lockId = in.readLong();
106    this.writeToWAL = in.readBoolean();
107    int numFamilies = in.readInt();
108    if (!this.familyMap.isEmpty()) this.familyMap.clear();
109    for(int i=0;i<numFamilies;i++) {
110      byte [] family = Bytes.readByteArray(in);
111      int numKeys = in.readInt();
112      List<KeyValue> keys = new ArrayList<KeyValue>(numKeys);
113      int totalLen = in.readInt();
114      byte [] buf = new byte[totalLen];
115      int offset = 0;
116      for (int j = 0; j < numKeys; j++) {
117        int keyLength = in.readInt();
118        in.readFully(buf, offset, keyLength);
119        keys.add(new KeyValue(buf, offset, keyLength));
120        offset += keyLength;
121      }
122      this.familyMap.put(family, keys);
123    }
124    readAttributes(in);
125  }
126
127  @Override
128  public void write(final DataOutput out)
129  throws IOException {
130    out.writeByte(APPEND_VERSION);
131    Bytes.writeByteArray(out, this.row);
132    out.writeLong(this.ts);
133    out.writeLong(this.lockId);
134    out.writeBoolean(this.writeToWAL);
135    out.writeInt(familyMap.size());
136    for (Map.Entry<byte [], List<KeyValue>> entry : familyMap.entrySet()) {
137      Bytes.writeByteArray(out, entry.getKey());
138      List<KeyValue> keys = entry.getValue();
139      out.writeInt(keys.size());
140      int totalLen = 0;
141      for(KeyValue kv : keys) {
142        totalLen += kv.getLength();
143      }
144      out.writeInt(totalLen);
145      for(KeyValue kv : keys) {
146        kv.write(out);
147      }
148    }
149    writeAttributes(out);
150  }
151
152  /**
153   * Add the specified {@link KeyValue} to this operation.
154   * @param kv whose value should be to appended to the specified column
155   * @return <tt?this</tt>
156   * @throws IllegalArgumentException if the row or type does not match <tt>this</tt>
157   */
158  public Append add(KeyValue kv) {
159    if(!(kv.getType() == KeyValue.Type.Put.getCode())){
160      throw new IllegalArgumentException("Added type " + KeyValue.Type.codeToType(kv.getType())
161          + ", but appends can only be of type " + KeyValue.Type.Put + ". Rowkey:"
162          + Bytes.toStringBinary(kv.getRow()));
163    }
164
165    if (!kv.matchingRow(row)) {
166      throw new IllegalArgumentException("The row in the recently added KeyValue "
167          + Bytes.toStringBinary(kv.getRow()) + " doesn't match the original one "
168          + Bytes.toStringBinary(this.row));
169    }
170
171    byte[] family = kv.getFamily();
172    List<KeyValue> list = familyMap.get(family);
173    if (list == null) {
174      list = new ArrayList<KeyValue>();
175      familyMap.put(family, list);
176    }
177    list.add(kv);
178    return this;
179  }
180}
