018package org.apache.hadoop.hbase.filter;
019
020import java.io.DataInput;
021import java.io.DataOutput;
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.List;
025import java.util.TreeSet;
026import java.util.ArrayList;
027
028import org.apache.hadoop.hbase.KeyValue;
029import com.google.common.base.Preconditions;
030
031/**
032 * Filter that returns only cells whose timestamp (version) is
033 * in the specified list of timestamps (versions).
034 * <p>
035 * Note: Use of this filter overrides any time range/time stamp
036 * options specified using {@link org.apache.hadoop.hbase.client.Get#setTimeRange(long, long)},
037 * {@link org.apache.hadoop.hbase.client.Scan#setTimeRange(long, long)}, {@link org.apache.hadoop.hbase.client.Get#setTimeStamp(long)},
038 * or {@link org.apache.hadoop.hbase.client.Scan#setTimeStamp(long)}.
039 */
040public class TimestampsFilter extends FilterBase {
041
042  TreeSet<Long> timestamps;
043  private static final int MAX_LOG_TIMESTAMPS = 5;
044
045  // Used during scans to hint the scan to stop early
046  // once the timestamps fall below the minTimeStamp.
047  long minTimeStamp = Long.MAX_VALUE;
048
049  /**
050   * Used during deserialization. Do not use otherwise.
051   */
052  public TimestampsFilter() {
053    super();
054  }
055
056  /**
057   * Constructor for filter that retains only those
058   * cells whose timestamp (version) is in the specified
059   * list of timestamps.
060   *
061   * @param timestamps
062   */
063  public TimestampsFilter(List<Long> timestamps) {
064    for (Long timestamp : timestamps) {
065      Preconditions.checkArgument(timestamp >= 0, "must be positive %s", timestamp);
066    }
067    this.timestamps = new TreeSet<Long>(timestamps);
068    init();
069  }
070
071  /**
072   * @return the list of timestamps
073   */
074  public List<Long> getTimestamps() {
075    List<Long> list = new ArrayList<Long>(timestamps.size());
076    list.addAll(timestamps);
077    return list;
078  }
079
080  private void init() {
081    if (this.timestamps.size() > 0) {
082      minTimeStamp = this.timestamps.first();
083    }
084  }
085
086  /**
087   * Gets the minimum timestamp requested by filter.
088   * @return  minimum timestamp requested by filter.
089   */
090  public long getMin() {
091    return minTimeStamp;
092  }
093
094  @Override
095  public ReturnCode filterKeyValue(KeyValue v) {
096    if (this.timestamps.contains(v.getTimestamp())) {
097      return ReturnCode.INCLUDE;
098    } else if (v.getTimestamp() < minTimeStamp) {
099      // The remaining versions of this column are guaranteed
100      // to be lesser than all of the other values.
101      return ReturnCode.NEXT_COL;
102    }
103    return ReturnCode.SKIP;
104  }
105
106  public static Filter createFilterFromArguments(ArrayList<byte []> filterArguments) {
107    ArrayList<Long> timestamps = new ArrayList<Long>();
108    for (int i = 0; i<filterArguments.size(); i++) {
109      long timestamp = ParseFilter.convertByteArrayToLong(filterArguments.get(i));
110      timestamps.add(timestamp);
111    }
112    return new TimestampsFilter(timestamps);
113  }
114
115  @Override
116  public void readFields(DataInput in) throws IOException {
117    int numTimestamps = in.readInt();
118    this.timestamps = new TreeSet<Long>();
119    for (int idx = 0; idx < numTimestamps; idx++) {
120      this.timestamps.add(in.readLong());
121    }
122    init();
123  }
124
125  @Override
126  public void write(DataOutput out) throws IOException {
127    int numTimestamps = this.timestamps.size();
128    out.writeInt(numTimestamps);
129    for (Long timestamp : this.timestamps) {
130      out.writeLong(timestamp);
131    }
132  }
133
134  @Override
135  public String toString() {
136    return toString(MAX_LOG_TIMESTAMPS);
137  }
138
139  protected String toString(int maxTimestamps) {
140    StringBuilder tsList = new StringBuilder();
141
142    int count = 0;
143    for (Long ts : this.timestamps) {
144      if (count >= maxTimestamps) {
145        break;
146      }
147      ++count;
148      tsList.append(ts.toString());
149      if (count < this.timestamps.size() && count < maxTimestamps) {
150        tsList.append(", ");
151      }
152    }
153
154    return String.format("%s (%d/%d): [%s]", this.getClass().getSimpleName(),
155        count, this.timestamps.size(), tsList.toString());
156  }
157}
