018package org.apache.hadoop.hbase.snapshot;
019
020import java.io.IOException;
021import java.util.Arrays;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.classification.InterfaceAudience;
026import org.apache.hadoop.classification.InterfaceStability;
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.fs.FileStatus;
029import org.apache.hadoop.fs.FileSystem;
030import org.apache.hadoop.fs.Path;
031import org.apache.hadoop.hbase.errorhandling.ForeignException;
032import org.apache.hadoop.hbase.errorhandling.ForeignExceptionDispatcher;
033import org.apache.hadoop.hbase.protobuf.generated.HBaseProtos.SnapshotDescription;
034import org.apache.hadoop.hbase.util.FSUtils;
035
036/**
037 * Reference all the WAL files under a server's WAL directory
038 */
039@InterfaceAudience.Private
040@InterfaceStability.Evolving
041public class ReferenceServerWALsTask extends SnapshotTask {
042  private static final Log LOG = LogFactory.getLog(ReferenceServerWALsTask.class);
043  private final FileSystem fs;
044  private final Configuration conf;
045  private final String serverName;
046  private Path logDir;
047
048  /**
049   * @param snapshot snapshot being run
050   * @param failureListener listener to check for errors while running the operation and to
051   *          propagate errors found while running the task
052   * @param logDir log directory for the server. Name of the directory is taken as the name of the
053   *          server
054   * @param conf {@link Configuration} to extract filesystem information
055   * @param fs filesystem where the log files are stored and should be referenced
056   */
057  public ReferenceServerWALsTask(SnapshotDescription snapshot,
058      ForeignExceptionDispatcher failureListener, final Path logDir, final Configuration conf,
059      final FileSystem fs) {
060    super(snapshot, failureListener);
061    this.fs = fs;
062    this.conf = conf;
063    this.serverName = logDir.getName();
064    this.logDir = logDir;
065  }
066
067  /**
068   * Create reference files (empty files with the same path and file name as original).
069   * @throws IOException exception from hdfs or network problems
070   * @throws ForeignException exception from an external procedure
071   */
072  @Override
073  public Void call() throws IOException, ForeignException {
074    // TODO switch to using a single file to reference all required WAL files
075
076    // Iterate through each of the log files and add a reference to it.
077    // assumes that all the files under the server's logs directory is a log
078    FileStatus[] serverLogs = FSUtils.listStatus(fs, logDir, null);
079    if (serverLogs == null) {
080      LOG.debug("No logs for server directory:" + logDir + ", done referencing files.");
081      return null;
082    }
083
084    if (LOG.isDebugEnabled()) {
085      LOG.debug("Adding references for WAL files:" + Arrays.toString(serverLogs));
086    }
087
088    for (FileStatus file : serverLogs) {
089      this.rethrowException();
090
091      // add the reference to the file. ex: hbase/.snapshots/.logs/<serverName>/<hlog>
092      Path rootDir = FSUtils.getRootDir(conf);
093      Path snapshotDir = SnapshotDescriptionUtils.getWorkingSnapshotDir(this.snapshot, rootDir);
094      Path snapshotLogDir = TakeSnapshotUtils.getSnapshotHLogsDir(snapshotDir, serverName);
095      // actually store the reference on disk (small file)
096      Path ref = new Path(snapshotLogDir, file.getPath().getName());
097      if (!fs.createNewFile(ref)) {
098        if (!fs.exists(ref)) {
099          throw new IOException("Couldn't create reference for:" + file.getPath());
100        }
101      }
102      LOG.debug("Completed WAL referencing for: " + file.getPath() + " to " + ref);
103    }
104
105    LOG.debug("Successfully completed WAL referencing for ALL files");
106    return null;
107  }
108}
