017package org.apache.hadoop.hbase.io.hfile;
018
019import java.io.IOException;
020import java.nio.ByteBuffer;
021
022import org.apache.hadoop.hbase.io.encoding.DataBlockEncoding;
023import org.apache.hadoop.hbase.util.Bytes;
024import org.apache.hadoop.hbase.util.Pair;
025
026/**
027 * Controls what kind of data block encoding is used. If data block encoding is
028 * not set or the given block is not a data block (encoded or not), methods
029 * should just return the unmodified block.
030 */
031public interface HFileDataBlockEncoder {
032  /** Type of encoding used for data blocks in HFile. Stored in file info. */
033  public static final byte[] DATA_BLOCK_ENCODING = Bytes.toBytes("DATA_BLOCK_ENCODING");
034  
035  /**
036   * Converts a block from the on-disk format to the in-cache format. Called in
037   * the following cases:
038   * <ul>
039   * <li>After an encoded or unencoded data block is read from disk, but before
040   * it is put into the cache.</li>
041   * <li>To convert brand-new blocks to the in-cache format when doing
042   * cache-on-write.</li>
043   * </ul>
044   * @param block a block in an on-disk format (read from HFile or freshly
045   *          generated).
046   * @return non null block which is coded according to the settings.
047   */
048  public HFileBlock diskToCacheFormat(HFileBlock block,
049      boolean isCompaction);
050
051  /**
052   * Should be called before an encoded or unencoded data block is written to
053   * disk.
054   * @param in KeyValues next to each other
055   * @param dummyHeader A dummy header to be written as a placeholder
056   * @return a non-null on-heap buffer containing the contents of the
057   *         HFileBlock with unfilled header and block type
058   */
059  public Pair<ByteBuffer, BlockType> beforeWriteToDisk(
060      ByteBuffer in, boolean includesMemstoreTS, byte[] dummyHeader);
061
062  /**
063   * Decides whether we should use a scanner over encoded blocks.
064   * @param isCompaction whether we are in a compaction.
065   * @return Whether to use encoded scanner.
066   */
067  public boolean useEncodedScanner(boolean isCompaction);
068
069  /**
070   * Save metadata in HFile which will be written to disk
071   * @param writer writer for a given HFile
072   * @exception IOException on disk problems
073   */
074  public void saveMetadata(HFile.Writer writer) throws IOException;
075
076  /** @return the on-disk data block encoding */
077  public DataBlockEncoding getEncodingOnDisk();
078
079  /** @return the preferred in-cache data block encoding for normal reads */
080  public DataBlockEncoding getEncodingInCache();
081
082  /**
083   * @return the effective in-cache data block encoding, taking into account
084   *         whether we are doing a compaction.
085   */
086  public DataBlockEncoding getEffectiveEncodingInCache(boolean isCompaction);
087
088}
