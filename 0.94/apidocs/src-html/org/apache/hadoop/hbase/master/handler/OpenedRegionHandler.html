020package org.apache.hadoop.hbase.master.handler;
021
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.hbase.HRegionInfo;
026import org.apache.hadoop.hbase.Server;
027import org.apache.hadoop.hbase.ServerName;
028import org.apache.hadoop.hbase.executor.EventHandler;
029import org.apache.hadoop.hbase.master.AssignmentManager;
030import org.apache.hadoop.hbase.master.AssignmentManager.RegionState;
031import org.apache.hadoop.hbase.zookeeper.ZKAssign;
032import org.apache.zookeeper.KeeperException;
033
034/**
035 * Handles OPENED region event on Master.
036 */
037public class OpenedRegionHandler extends EventHandler implements TotesHRegionInfo {
038  private static final Log LOG = LogFactory.getLog(OpenedRegionHandler.class);
039  private final AssignmentManager assignmentManager;
040  private final HRegionInfo regionInfo;
041  private final ServerName sn;
042  private final OpenedPriority priority;
043  private final int expectedVersion;
044
045  private enum OpenedPriority {
046    ROOT (1),
047    META (2),
048    USER (3);
049
050    private final int value;
051    OpenedPriority(int value) {
052      this.value = value;
053    }
054    public int getValue() {
055      return value;
056    }
057  };
058
059  public OpenedRegionHandler(Server server,
060      AssignmentManager assignmentManager, HRegionInfo regionInfo,
061      ServerName sn, int expectedVersion) {
062    super(server, EventType.RS_ZK_REGION_OPENED);
063    this.assignmentManager = assignmentManager;
064    this.regionInfo = regionInfo;
065    this.sn = sn;
066    this.expectedVersion = expectedVersion;
067    if(regionInfo.isRootRegion()) {
068      priority = OpenedPriority.ROOT;
069    } else if(regionInfo.isMetaRegion()) {
070      priority = OpenedPriority.META;
071    } else {
072      priority = OpenedPriority.USER;
073    }
074  }
075
076  @Override
077  public int getPriority() {
078    return priority.getValue();
079  }
080
081  @Override
082  public HRegionInfo getHRegionInfo() {
083    return this.regionInfo;
084  }
085  
086  @Override
087  public String toString() {
088    String name = "UnknownServerName";
089    if(server != null && server.getServerName() != null) {
090      name = server.getServerName().toString();
091    }
092    return getClass().getSimpleName() + "-" + name + "-" + getSeqid();
093  }
094
095  @Override
096  public void process() {
097    // Code to defend against case where we get SPLIT before region open
098    // processing completes; temporary till we make SPLITs go via zk -- 0.92.
099    RegionState regionState = this.assignmentManager.isRegionInTransition(regionInfo);
100    boolean openedNodeDeleted = false;
101    if (regionState != null
102        && regionState.getState().equals(RegionState.State.OPEN)) {
103      openedNodeDeleted = deleteOpenedNode(expectedVersion);
104      if (!openedNodeDeleted) {
105        LOG.error("The znode of region " + regionInfo.getRegionNameAsString()
106            + " could not be deleted.");
107      }
108    } else {
109      LOG.warn("Skipping the onlining of " + regionInfo.getRegionNameAsString() +
110        " because regions is NOT in RIT -- presuming this is because it SPLIT");
111    }
112    if (!openedNodeDeleted) {
113      if (this.assignmentManager.getZKTable().isDisablingOrDisabledTable(
114          regionInfo.getTableNameAsString())) {
115        debugLog(regionInfo, "Opened region "
116            + regionInfo.getRegionNameAsString() + " but "
117            + "this table is disabled, triggering close of region");
118        assignmentManager.unassign(regionInfo);
119      }
120    }
121  }
122
123  private boolean deleteOpenedNode(int expectedVersion) {
124    debugLog(regionInfo, "Handling OPENED event for " +
125      this.regionInfo.getRegionNameAsString() + " from " + this.sn.toString() +
126      "; deleting unassigned node");
127    try {
128      // delete the opened znode only if the version matches.
129      return ZKAssign.deleteNode(server.getZooKeeper(),
130          regionInfo.getEncodedName(), EventType.RS_ZK_REGION_OPENED, expectedVersion);
131    } catch(KeeperException.NoNodeException e){
132      // Getting no node exception here means that already the region has been opened.
133      LOG.warn("The znode of the region " + regionInfo.getRegionNameAsString() +
134        " would have already been deleted");
135      return false;
136    } catch (KeeperException e) {
137      server.abort("Error deleting OPENED node in ZK (" +
138        regionInfo.getRegionNameAsString() + ")", e);
139    }
140    return false;
141  }
142
143  private void debugLog(HRegionInfo region, String string) {
144    if (region.isMetaTable()) {
145      LOG.info(string);
146    } else {
147      LOG.debug(string);
148    }
149  }
150}
