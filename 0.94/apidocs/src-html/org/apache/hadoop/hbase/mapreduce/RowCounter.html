020package org.apache.hadoop.hbase.mapreduce;
021
022import java.io.IOException;
023import org.apache.commons.lang.StringUtils;
024
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hbase.HBaseConfiguration;
027import org.apache.hadoop.hbase.KeyValue;
028import org.apache.hadoop.hbase.client.Result;
029import org.apache.hadoop.hbase.client.Scan;
030import org.apache.hadoop.hbase.filter.FirstKeyOnlyFilter;
031import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
032import org.apache.hadoop.hbase.util.Bytes;
033import org.apache.hadoop.mapreduce.Job;
034import org.apache.hadoop.mapreduce.lib.output.NullOutputFormat;
035import org.apache.hadoop.util.GenericOptionsParser;
036
037
038/**
039 * A job with a just a map phase to count rows. Map outputs table rows IF the
040 * input row has columns that have content.
041 */
042public class RowCounter {
043
044  /** Name of this 'program'. */
045  static final String NAME = "rowcounter";
046
047  /**
048   * Mapper that runs the count.
049   */
050  static class RowCounterMapper
051  extends TableMapper<ImmutableBytesWritable, Result> {
052
053    /** Counter enumeration to count the actual rows. */
054    public static enum Counters {ROWS}
055
056    /**
057     * Maps the data.
058     *
059     * @param row  The current table row key.
060     * @param values  The columns.
061     * @param context  The current context.
062     * @throws IOException When something is broken with the data.
063     * @see org.apache.hadoop.mapreduce.Mapper#map(KEYIN, VALUEIN,
064     *   org.apache.hadoop.mapreduce.Mapper.Context)
065     */
066    @Override
067    public void map(ImmutableBytesWritable row, Result values,
068      Context context)
069    throws IOException {
070      // Count every row containing data, whether it's in qualifiers or values
071      context.getCounter(Counters.ROWS).increment(1);
072    }
073  }
074
075  /**
076   * Sets up the actual job.
077   *
078   * @param conf  The current configuration.
079   * @param args  The command line parameters.
080   * @return The newly created job.
081   * @throws IOException When setting up the job fails.
082   */
083  public static Job createSubmittableJob(Configuration conf, String[] args)
084  throws IOException {
085    String tableName = args[0];
086    String startKey = null;
087    String endKey = null;
088    StringBuilder sb = new StringBuilder();
089
090    final String rangeSwitch = "--range=";
091
092    // First argument is table name, starting from second
093    for (int i = 1; i < args.length; i++) {
094      if (args[i].startsWith(rangeSwitch)) {
095        String[] startEnd = args[i].substring(rangeSwitch.length()).split(",", 2);
096        if (startEnd.length != 2 || startEnd[1].contains(",")) {
097          printUsage("Please specify range in such format as \"--range=a,b\" " +
098              "or, with only one boundary, \"--range=,b\" or \"--range=a,\"");
099          return null;
100        }
101        startKey = startEnd[0];
102        endKey = startEnd[1];
103      }
104      else {
105        // if no switch, assume column names
106        sb.append(args[i]);
107        sb.append(" ");
108      }
109    }
110
111    Job job = new Job(conf, NAME + "_" + tableName);
112    job.setJarByClass(RowCounter.class);
113    Scan scan = new Scan();
114    scan.setCacheBlocks(false);
115    if (startKey != null && !startKey.equals("")) {
116      scan.setStartRow(Bytes.toBytes(startKey));
117    }
118    if (endKey != null && !endKey.equals("")) {
119      scan.setStopRow(Bytes.toBytes(endKey));
120    }
121    scan.setFilter(new FirstKeyOnlyFilter());
122    if (sb.length() > 0) {
123      for (String columnName : sb.toString().trim().split(" ")) {
124        String family = StringUtils.substringBefore(columnName, ":");
125        String qualifier = StringUtils.substringAfter(columnName, ":");
126        if (StringUtils.isBlank(qualifier)) {
127          scan.addFamily(Bytes.toBytes(family));
128        } else {
129          scan.addColumn(Bytes.toBytes(family), Bytes.toBytes(qualifier));
130        }
131      }
132    }
133    job.setOutputFormatClass(NullOutputFormat.class);
134    TableMapReduceUtil.initTableMapperJob(tableName, scan,
135      RowCounterMapper.class, ImmutableBytesWritable.class, Result.class, job);
136    job.setNumReduceTasks(0);
137    return job;
138  }
139
140  /*
141   * @param errorMessage Can attach a message when error occurs.
142   */
143  private static void printUsage(String errorMessage) {
144    System.err.println("ERROR: " + errorMessage);
145    printUsage();
146  }
147
148  /*
149   * Prints usage without error message
150   */
151  private static void printUsage() {
152    System.err.println("Usage: RowCounter [options] <tablename> " +
153        "[--range=[startKey],[endKey]] [<column1> <column2>...]");
154    System.err.println("For performance consider the following options:\n"
155        + "-Dhbase.client.scanner.caching=100\n"
156        + "-Dmapred.map.tasks.speculative.execution=false");
157  }
158
159  /**
160   * Main entry point.
161   *
162   * @param args  The command line parameters.
163   * @throws Exception When running the job fails.
164   */
165  public static void main(String[] args) throws Exception {
166    Configuration conf = HBaseConfiguration.create();
167    String[] otherArgs = new GenericOptionsParser(conf, args).getRemainingArgs();
168    if (otherArgs.length < 1) {
169      printUsage("Wrong number of parameters: " + args.length);
170      System.exit(-1);
171    }
172    Job job = createSubmittableJob(conf, otherArgs);
173    if (job == null) {
174      System.exit(-1);
175    }
176    System.exit(job.waitForCompletion(true) ? 0 : 1);
177  }
178}
