037/**
038 * Utility methods for reading, parsing, and building zookeeper configuration.
039 */
040public class ZKConfig {
041  private static final Log LOG = LogFactory.getLog(ZKConfig.class);
042
043  private static final String VARIABLE_START = "${";
044  private static final int VARIABLE_START_LENGTH = VARIABLE_START.length();
045  private static final String VARIABLE_END = "}";
046  private static final int VARIABLE_END_LENGTH = VARIABLE_END.length();
047
048  /**
049   * Make a Properties object holding ZooKeeper config equivalent to zoo.cfg.
050   * If there is a zoo.cfg in the classpath, simply read it in. Otherwise parse
051   * the corresponding config options from the HBase XML configs and generate
052   * the appropriate ZooKeeper properties.
053   * @param conf Configuration to read from.
054   * @return Properties holding mappings representing ZooKeeper zoo.cfg file.
055   */
056  public static Properties makeZKProps(Configuration conf) {
057    // First check if there is a zoo.cfg in the CLASSPATH. If so, simply read
058    // it and grab its configuration properties.
059    ClassLoader cl = HQuorumPeer.class.getClassLoader();
060    final InputStream inputStream =
061      cl.getResourceAsStream(HConstants.ZOOKEEPER_CONFIG_NAME);
062    if (inputStream != null) {
063      try {
064        return parseZooCfg(conf, inputStream);
065      } catch (IOException e) {
066        LOG.warn("Cannot read " + HConstants.ZOOKEEPER_CONFIG_NAME +
067                 ", loading from XML files", e);
068      }
069    }
070
071    // Otherwise, use the configuration options from HBase's XML files.
072    Properties zkProperties = new Properties();
073
074    // Directly map all of the hbase.zookeeper.property.KEY properties.
075    for (Entry<String, String> entry : conf) {
076      String key = entry.getKey();
077      if (key.startsWith(HConstants.ZK_CFG_PROPERTY_PREFIX)) {
078        String zkKey = key.substring(HConstants.ZK_CFG_PROPERTY_PREFIX_LEN);
079        String value = entry.getValue();
080        // If the value has variables substitutions, need to do a get.
081        if (value.contains(VARIABLE_START)) {
082          value = conf.get(key);
083        }
084        zkProperties.put(zkKey, value);
085      }
086    }
087
088    // If clientPort is not set, assign the default.
089    if (zkProperties.getProperty(HConstants.CLIENT_PORT_STR) == null) {
090      zkProperties.put(HConstants.CLIENT_PORT_STR,
091          HConstants.DEFAULT_ZOOKEPER_CLIENT_PORT);
092    }
093
094    // Create the server.X properties.
095    int peerPort = conf.getInt("hbase.zookeeper.peerport", 2888);
096    int leaderPort = conf.getInt("hbase.zookeeper.leaderport", 3888);
097
098    final String[] serverHosts = conf.getStrings(HConstants.ZOOKEEPER_QUORUM,
099                                                 HConstants.LOCALHOST);
100    for (int i = 0; i < serverHosts.length; ++i) {
101      String serverHost = serverHosts[i];
102      String address = serverHost + ":" + peerPort + ":" + leaderPort;
103      String key = "server." + i;
104      zkProperties.put(key, address);
105    }
106
107    return zkProperties;
108  }

110  /**
111   * Parse ZooKeeper's zoo.cfg, injecting HBase Configuration variables in.
112   * This method is used for testing so we can pass our own InputStream.
113   * @param conf HBaseConfiguration to use for injecting variables.
114   * @param inputStream InputStream to read from.
115   * @return Properties parsed from config stream with variables substituted.
116   * @throws IOException if anything goes wrong parsing config
117   */
118  public static Properties parseZooCfg(Configuration conf,
119      InputStream inputStream) throws IOException {
120    Properties properties = new Properties();
121    try {
122      properties.load(inputStream);
123    } catch (IOException e) {
124      final String msg = "fail to read properties from "
125        + HConstants.ZOOKEEPER_CONFIG_NAME;
126      LOG.fatal(msg);
127      throw new IOException(msg, e);
128    }
129    for (Entry<Object, Object> entry : properties.entrySet()) {
130      String value = entry.getValue().toString().trim();
131      String key = entry.getKey().toString().trim();
132      StringBuilder newValue = new StringBuilder();
133      int varStart = value.indexOf(VARIABLE_START);
134      int varEnd = 0;
135      while (varStart != -1) {
136        varEnd = value.indexOf(VARIABLE_END, varStart);
137        if (varEnd == -1) {
138          String msg = "variable at " + varStart + " has no end marker";
139          LOG.fatal(msg);
140          throw new IOException(msg);
141        }
142        String variable = value.substring(varStart + VARIABLE_START_LENGTH, varEnd);
143
144        String substituteValue = System.getProperty(variable);
145        if (substituteValue == null) {
146          substituteValue = conf.get(variable);
147        }
148        if (substituteValue == null) {
149          String msg = "variable " + variable + " not set in system property "
150                     + "or hbase configs";
151          LOG.fatal(msg);
152          throw new IOException(msg);
153        }
154
155        newValue.append(substituteValue);
156
157        varEnd += VARIABLE_END_LENGTH;
158        varStart = value.indexOf(VARIABLE_START, varEnd);
159      }
160      // Special case for 'hbase.cluster.distributed' property being 'true'
161      if (key.startsWith("server.")) {
162        boolean mode = conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, HConstants.DEFAULT_CLUSTER_DISTRIBUTED);
163        if (mode == HConstants.CLUSTER_IS_DISTRIBUTED && value.startsWith(HConstants.LOCALHOST)) {
164          String msg = "The server in zoo.cfg cannot be set to localhost " +
165              "in a fully-distributed setup because it won't be reachable. " +
166              "See \"Getting Started\" for more information.";
167          LOG.fatal(msg);
168          throw new IOException(msg);
169        }
170      }
171      newValue.append(value.substring(varEnd));
172      properties.setProperty(key, newValue.toString());
173    }
174    return properties;
175  }

177  /**
178   * Return the ZK Quorum servers string given zk properties returned by
179   * makeZKProps
180   * @param properties
181   * @return Quorum servers String
182   */
183  public static String getZKQuorumServersString(Properties properties) {
184    String clientPort = null;
185    List<String> servers = new ArrayList<String>();
186
187    // The clientPort option may come after the server.X hosts, so we need to
188    // grab everything and then create the final host:port comma separated list.
189    boolean anyValid = false;
190    for (Entry<Object,Object> property : properties.entrySet()) {
191      String key = property.getKey().toString().trim();
192      String value = property.getValue().toString().trim();
193      if (key.equals("clientPort")) {
194        clientPort = value;
195      }
196      else if (key.startsWith("server.")) {
197        String host = value.substring(0, value.indexOf(':'));
198        servers.add(host);
199        anyValid = true;
200      }
201    }
202
203    if (!anyValid) {
204      LOG.error("no valid quorum servers found in " + HConstants.ZOOKEEPER_CONFIG_NAME);
205      return null;
206    }
207
208    if (clientPort == null) {
209      LOG.error("no clientPort found in " + HConstants.ZOOKEEPER_CONFIG_NAME);
210      return null;
211    }
212
213    if (servers.isEmpty()) {
214      LOG.fatal("No server.X lines found in conf/zoo.cfg. HBase must have a " +
215                "ZooKeeper cluster configured for its operation.");
216      return null;
217    }
218
219    StringBuilder hostPortBuilder = new StringBuilder();
220    for (int i = 0; i < servers.size(); ++i) {
221      String host = servers.get(i);
222      if (i > 0) {
223        hostPortBuilder.append(',');
224      }
225      hostPortBuilder.append(host);
226      hostPortBuilder.append(':');
227      hostPortBuilder.append(clientPort);
228    }
229
230    return hostPortBuilder.toString();
231  }
232
233  /**
234   * Return the ZK Quorum servers string given the specified configuration.
235   * @param conf
236   * @return Quorum servers
237   */
238  public static String getZKQuorumServersString(Configuration conf) {
239    return getZKQuorumServersString(makeZKProps(conf));
240  }
