020package org.apache.hadoop.hbase.util;
021
022import java.io.IOException;
023import java.lang.ClassNotFoundException;
024import java.util.zip.Checksum;
025import java.lang.reflect.Constructor;
026
027/**
028 * Utility class that is used to generate a Checksum object.
029 * The Checksum implementation is pluggable and an application
030 * can specify their own class that implements their own
031 * Checksum algorithm.
032 */
033public class ChecksumFactory {
034
035  static private final Class<?>[] EMPTY_ARRAY = new Class[]{};
036
037  /**
038   * Create a new instance of a Checksum object.
039   * @return The newly created Checksum object
040   */
041  static public Checksum newInstance(String className) throws IOException {
042    try {
043      Class<?> clazz = getClassByName(className);
044      return (Checksum)newInstance(clazz);
045    } catch (ClassNotFoundException e) {
046      throw new IOException(e);
047    }
048  }
049
050  /**
051   * Returns a Constructor that can be used to create a Checksum object.
052   * @return The Constructor that can be used to create a 
053   * new Checksum object.
054   * @param theClass classname for which an constructor is created
055   * @return a new Constructor object
056   */
057  static public Constructor<?> newConstructor(String className) 
058    throws IOException {
059    try {
060      Class<?> clazz = getClassByName(className);
061      Constructor<?> ctor = clazz.getDeclaredConstructor(EMPTY_ARRAY);
062      ctor.setAccessible(true);
063      return ctor;
064    } catch (ClassNotFoundException e) {
065      throw new IOException(e);
066    } catch (java.lang.NoSuchMethodException e) {
067      throw new IOException(e);
068    }
069  }
070
071  /** Create an object for the given class and initialize it from conf
072   *
073   * @param theClass class of which an object is created
074   * @return a new object
075   */
076  static private <T> T newInstance(Class<T> theClass) {
077    T result;
078    try {
079      Constructor<T> ctor = theClass.getDeclaredConstructor(EMPTY_ARRAY);
080      ctor.setAccessible(true);
081      result = ctor.newInstance();
082    } catch (Exception e) {
083      throw new RuntimeException(e);
084    }
085    return result;
086  }
087
088  /**
089   * Load a class by name.
090   * @param name the class name.
091   * @return the class object.
092   * @throws ClassNotFoundException if the class is not found.
093   */
094  static private Class<?> getClassByName(String name) 
095    throws ClassNotFoundException {
096    ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
097    return Class.forName(name, true, classLoader);
098  }
099}
