021package org.apache.hadoop.hbase.util;
022
023import java.lang.management.ManagementFactory;
024import java.lang.management.RuntimeMXBean;
025import java.lang.reflect.InvocationTargetException;
026import java.lang.reflect.Method;
027import java.nio.ByteBuffer;
028import java.util.List;
029
030import com.google.common.base.Preconditions;
031
032public class DirectMemoryUtils {
033  /**
034   * @return the setting of -XX:MaxDirectMemorySize as a long. Returns 0 if
035   *         -XX:MaxDirectMemorySize is not set.
036   */
037
038  public static long getDirectMemorySize() {
039    RuntimeMXBean RuntimemxBean = ManagementFactory.getRuntimeMXBean();
040    List<String> arguments = RuntimemxBean.getInputArguments();
041    long multiplier = 1; //for the byte case.
042    for (String s : arguments) {
043      if (s.contains("-XX:MaxDirectMemorySize=")) {
044        String memSize = s.toLowerCase()
045            .replace("-xx:maxdirectmemorysize=", "").trim();
046
047        if (memSize.contains("k")) {
048          multiplier = 1024;
049        }
050
051        else if (memSize.contains("m")) {
052          multiplier = 1048576;
053        }
054
055        else if (memSize.contains("g")) {
056          multiplier = 1073741824;
057        }
058        memSize = memSize.replaceAll("[^\\d]", "");
059
060        long retValue = Long.parseLong(memSize);
061        return retValue * multiplier;
062      }
063
064    }
065    return 0;
066  }
067
068  /**
069   * DirectByteBuffers are garbage collected by using a phantom reference and a
070   * reference queue. Every once a while, the JVM checks the reference queue and
071   * cleans the DirectByteBuffers. However, as this doesn't happen
072   * immediately after discarding all references to a DirectByteBuffer, it's
073   * easy to OutOfMemoryError yourself using DirectByteBuffers. This function
074   * explicitly calls the Cleaner method of a DirectByteBuffer.
075   * 
076   * @param toBeDestroyed
077   *          The DirectByteBuffer that will be "cleaned". Utilizes reflection.
078   *          
079   */
080  public static void destroyDirectByteBuffer(ByteBuffer toBeDestroyed)
081      throws IllegalArgumentException, IllegalAccessException,
082      InvocationTargetException, SecurityException, NoSuchMethodException {
083
084    Preconditions.checkArgument(toBeDestroyed.isDirect(),
085        "toBeDestroyed isn't direct!");
086
087    Method cleanerMethod = toBeDestroyed.getClass().getMethod("cleaner");
088    cleanerMethod.setAccessible(true);
089    Object cleaner = cleanerMethod.invoke(toBeDestroyed);
090    Method cleanMethod = cleaner.getClass().getMethod("clean");
091    cleanMethod.setAccessible(true);
092    cleanMethod.invoke(cleaner);
093
094  }
095}
