019package org.apache.hadoop.hbase.util;
020
021import java.io.IOException;
022import java.lang.reflect.Constructor;
023import java.util.zip.Checksum;
024
025import org.apache.commons.logging.Log;
026import org.apache.commons.logging.LogFactory;
027
028/**
029 * Checksum types. The Checksum type is a one byte number
030 * that stores a representation of the checksum algorithm
031 * used to encode a hfile. The ordinal of these cannot 
032 * change or else you risk breaking all existing HFiles out there.
033 */
034public enum ChecksumType {
035
036  NULL((byte)0) {
037    @Override
038    public String getName() {
039      return "NULL";
040    }
041    @Override
042    public void initialize() {
043      // do nothing
044    }
045    @Override
046    public Checksum getChecksumObject() throws IOException {
047      return null; // checksums not used
048    }
049  },
050
051  CRC32((byte)1) {
052    private volatile Constructor<?> ctor;
053
054    @Override
055    public String getName() {
056      return "CRC32";
057    }
058
059    @Override
060    public void initialize() {
061      final String PURECRC32 = "org.apache.hadoop.util.PureJavaCrc32";
062      final String JDKCRC = "java.util.zip.CRC32";
063      LOG = LogFactory.getLog(ChecksumType.class);
064
065      // check if hadoop library is available
066      try {
067        ctor = ChecksumFactory.newConstructor(PURECRC32);
068        LOG.info("Checksum using " + PURECRC32);
069      } catch (Exception e) {
070        LOG.trace(PURECRC32 + " not available.");
071      }
072      try {
073        // The default checksum class name is java.util.zip.CRC32. 
074        // This is available on all JVMs.
075        if (ctor == null) {
076          ctor = ChecksumFactory.newConstructor(JDKCRC);
077          LOG.info("Checksum can use " + JDKCRC);
078        }
079      } catch (Exception e) {
080        LOG.trace(JDKCRC + " not available.");
081      }
082    }
083
084    @Override
085    public Checksum getChecksumObject() throws IOException {
086      if (ctor == null) {
087        throw new IOException("Bad constructor for " + getName());
088      }
089      try {
090        return (Checksum)ctor.newInstance();
091      } catch (Exception e) {
092        throw new IOException(e);
093      }
094    }
095  },
096
097  CRC32C((byte)2) {
098    private transient Constructor<?> ctor;
099
100    @Override
101    public String getName() {
102      return "CRC32C";
103    }
104
105    @Override
106    public void initialize() {
107      final String PURECRC32C = "org.apache.hadoop.util.PureJavaCrc32C";
108      LOG = LogFactory.getLog(ChecksumType.class);
109      try {
110        ctor = ChecksumFactory.newConstructor(PURECRC32C);
111        LOG.info("Checksum can use " + PURECRC32C);
112      } catch (Exception e) {
113        LOG.trace(PURECRC32C + " not available.");
114      }
115    }
116
117    @Override
118    public Checksum getChecksumObject() throws IOException {
119      if (ctor == null) {
120        throw new IOException("Bad constructor for " + getName());
121      }
122      try {
123        return (Checksum)ctor.newInstance();
124      } catch (Exception e) {
125        throw new IOException(e);
126      }
127    }
128  };
129
130  private final byte code;
131  protected Log LOG;
132
133  /** initializes the relevant checksum class object */
134  abstract void initialize();
135
136  /** returns the name of this checksum type */
137  public abstract String getName();
138
139  private ChecksumType(final byte c) {
140    this.code = c;
141    initialize();
142  }
143
144  /** returns a object that can be used to generate/validate checksums */
145  public abstract Checksum getChecksumObject() throws IOException;
146
147  public byte getCode() {
148    return this.code;
149  }
150
151  /**
152   * Cannot rely on enum ordinals . They change if item is removed or moved.
153   * Do our own codes.
154   * @param b
155   * @return Type associated with passed code.
156   */
157  public static ChecksumType codeToType(final byte b) {
158    for (ChecksumType t : ChecksumType.values()) {
159      if (t.getCode() == b) {
160        return t;
161      }
162    }
163    throw new RuntimeException("Unknown checksum type code " + b);
164  }
165
166  /**
167   * Map a checksum name to a specific type.
168   * Do our own names.
169   * @param b
170   * @return Type associated with passed code.
171   */
172  public static ChecksumType nameToType(final String name) {
173    for (ChecksumType t : ChecksumType.values()) {
174      if (t.getName().equals(name)) {
175        return t;
176      }
177    }
178    throw new RuntimeException("Unknown checksum type name " + name);
179  }
180}
