018package org.apache.hadoop.hbase;
019
020import static org.junit.Assert.*;
021
022import java.io.IOException;
023
024import org.apache.hadoop.hbase.util.Writables;
025import org.junit.Test;
026import org.junit.experimental.categories.Category;
027
028@Category(SmallTests.class)
029public class TestHServerInfo {
030
031  @Test
032  public void testHashCodeAndEquals() {
033    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
034    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
035    HServerInfo hsi2 = new HServerInfo(hsa1, 1L, 5678);
036    HServerInfo hsi3 = new HServerInfo(hsa1, 2L, 5678);
037    HServerInfo hsi4 = new HServerInfo(hsa1, 1L, 5677);
038    HServerAddress hsa2 = new HServerAddress("localhost", 1235);
039    HServerInfo hsi5 = new HServerInfo(hsa2, 1L, 5678);
040    assertEquals(hsi1.hashCode(), hsi2.hashCode());
041    assertTrue(hsi1.equals(hsi2));
042    assertNotSame(hsi1.hashCode(), hsi3.hashCode());
043    assertFalse(hsi1.equals(hsi3));
044    assertNotSame(hsi1.hashCode(), hsi4.hashCode());
045    assertFalse(hsi1.equals(hsi4));
046    assertNotSame(hsi1.hashCode(), hsi5.hashCode());
047    assertFalse(hsi1.equals(hsi5));
048  }
049
050  @Test
051  public void testHServerInfoHServerInfo() {
052    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
053    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
054    HServerInfo hsi2 = new HServerInfo(hsi1);
055    assertEquals(hsi1, hsi2);
056  }
057
058  @Test
059  public void testGetServerAddress() {
060    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
061    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
062    assertEquals(hsi1.getServerAddress(), hsa1);
063  }
064
065  @Test
066  public void testToString() {
067    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
068    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
069    System.out.println(hsi1.toString());
070  }
071
072  @Test
073  public void testReadFields() throws IOException {
074    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
075    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
076    HServerAddress hsa2 = new HServerAddress("localhost", 1235);
077    HServerInfo hsi2 = new HServerInfo(hsa2, 1L, 5678);
078    byte [] bytes = Writables.getBytes(hsi1);
079    HServerInfo deserialized =
080      (HServerInfo)Writables.getWritable(bytes, new HServerInfo());
081    assertEquals(hsi1, deserialized);
082    bytes = Writables.getBytes(hsi2);
083    deserialized = (HServerInfo)Writables.getWritable(bytes, new HServerInfo());
084    assertNotSame(hsa1, deserialized);
085  }
086
087  @Test
088  public void testCompareTo() {
089    HServerAddress hsa1 = new HServerAddress("localhost", 1234);
090    HServerInfo hsi1 = new HServerInfo(hsa1, 1L, 5678);
091    HServerAddress hsa2 = new HServerAddress("localhost", 1235);
092    HServerInfo hsi2 = new HServerInfo(hsa2, 1L, 5678);
093    assertTrue(hsi1.compareTo(hsi1) == 0);
094    assertTrue(hsi2.compareTo(hsi2) == 0);
095    int compare1 = hsi1.compareTo(hsi2);
096    int compare2 = hsi2.compareTo(hsi1);
097    assertTrue((compare1 > 0)? compare2 < 0: compare2 > 0);
098  }

103}
