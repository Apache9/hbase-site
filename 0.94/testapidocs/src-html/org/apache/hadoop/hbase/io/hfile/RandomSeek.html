022import java.io.BufferedReader;
023import java.io.FileReader;
024import java.io.IOException;
025import java.nio.ByteBuffer;
026import java.util.ArrayList;
027import java.util.List;
028import java.util.Random;
029
030import org.apache.hadoop.conf.Configuration;
031import org.apache.hadoop.fs.LocalFileSystem;
032import org.apache.hadoop.fs.Path;
033import org.apache.hadoop.fs.RawLocalFileSystem;
034import org.apache.hadoop.hbase.io.hfile.HFile.Reader;
035import org.apache.hadoop.hbase.util.Bytes;
036
037/**
038 * Random seek test.
039 */
040public class RandomSeek {
041  private static List<String> slurp(String fname) throws IOException {
042    BufferedReader istream = new BufferedReader(new FileReader(fname));
043    String str;
044    List<String> l = new ArrayList<String>();
045    while ( (str=istream.readLine()) != null) {
046      String [] parts = str.split(",");
047      l.add(parts[0] + ":" + parts[1] + ":" + parts[2]);
048    }
049    istream.close();
050    return l;
051  }
052
053  private static String randKey(List<String> keys) {
054    Random r = new Random();
055    //return keys.get(r.nextInt(keys.size()));
056    return "2" + Integer.toString(7+r.nextInt(2)) + Integer.toString(r.nextInt(100));
057    //return new String(r.nextInt(100));
058  }
059
060  public static void main(String [] argv) throws IOException {
061    Configuration conf = new Configuration();
062    conf.setInt("io.file.buffer.size", 64*1024);
063    RawLocalFileSystem rlfs = new RawLocalFileSystem();
064    rlfs.setConf(conf);
065    LocalFileSystem lfs = new LocalFileSystem(rlfs);
066
067    Path path = new Path("/Users/ryan/rfile.big.txt");
068    long start = System.currentTimeMillis();
069    SimpleBlockCache cache = new SimpleBlockCache();
070    CacheConfig cacheConf = new CacheConfig(cache, true, false, false, false,
071        false, false, false);
072
073    Reader reader = HFile.createReader(lfs, path, cacheConf);
074    reader.loadFileInfo();
075    System.out.println(reader.getTrailer());
076    long end = System.currentTimeMillis();
077
078    System.out.println("Index read time: " + (end - start));
079
080    List<String> keys = slurp("/Users/ryan/xaa.50k");
081
082    // Get a scanner that doesn't cache and that uses pread.
083    HFileScanner scanner = reader.getScanner(false, true);
084    int count;
085    long totalBytes = 0;
086    int notFound = 0;
087
088    start = System.nanoTime();
089    for(count = 0; count < 500000; ++count) {
090      String key = randKey(keys);
091      byte [] bkey = Bytes.toBytes(key);
092      int res = scanner.seekTo(bkey);
093      if (res == 0) {
094        ByteBuffer k = scanner.getKey();
095        ByteBuffer v = scanner.getValue();
096        totalBytes += k.limit();
097        totalBytes += v.limit();
098      } else {
099        ++ notFound;
100      }
101      if (res == -1) {
102        scanner.seekTo();
103      }
104      // Scan for another 1000 rows.
105      for (int i = 0; i < 1000; ++i) {
106        if (!scanner.next())
107          break;
108        ByteBuffer k = scanner.getKey();
109        ByteBuffer v = scanner.getValue();
110        totalBytes += k.limit();
111        totalBytes += v.limit();
112      }
113
114      if ( count % 1000 == 0 ) {
115        end = System.nanoTime();
116
117            System.out.println("Cache block count: " + cache.size() + " dumped: "+ cache.dumps);
118            //System.out.println("Cache size: " + cache.heapSize());
119            double msTime = ((end - start) / 1000000.0);
120            System.out.println("Seeked: "+ count + " in " + msTime + " (ms) "
121                + (1000.0 / msTime ) + " seeks/ms "
122                + (msTime / 1000.0) + " ms/seek");
123
124            start = System.nanoTime();
125      }
126    }
127    System.out.println("Total bytes: " + totalBytes + " not found: " + notFound);
128  }
129}
