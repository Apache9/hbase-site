023import static org.junit.Assert.assertTrue;

027import org.apache.commons.logging.Log;

038@Category(MediumTests.class)
039public class TestMasterAddressManager {
040  private static final Log LOG = LogFactory.getLog(TestMasterAddressManager.class);
041
042  private final static HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
043
044  @BeforeClass
045  public static void setUpBeforeClass() throws Exception {
046    TEST_UTIL.startMiniZKCluster();
047  }
048
049  @AfterClass
050  public static void tearDownAfterClass() throws Exception {
051    TEST_UTIL.shutdownMiniZKCluster();
052  }
053  /**
054   * Unit tests that uses ZooKeeper but does not use the master-side methods
055   * but rather acts directly on ZK.
056   * @throws Exception
057   */
058  @Test
059  public void testMasterAddressManagerFromZK() throws Exception {
060
061    ZooKeeperWatcher zk = new ZooKeeperWatcher(TEST_UTIL.getConfiguration(),
062        "testMasterAddressManagerFromZK", null);
063    ZKUtil.createAndFailSilent(zk, zk.baseZNode);
064
065    // Should not have a master yet
066    MasterAddressTracker addressManager = new MasterAddressTracker(zk, null);
067    addressManager.start();
068    assertFalse(addressManager.hasMaster());
069    zk.registerListener(addressManager);
070
071    // Use a listener to capture when the node is actually created
072    NodeCreationListener listener = new NodeCreationListener(zk, zk.masterAddressZNode);
073    zk.registerListener(listener);
074
075    // Create the master node with a dummy address
076    String host = "localhost";
077    int port = 1234;
078    ServerName sn = new ServerName(host, port, System.currentTimeMillis());
079    LOG.info("Creating master node");
080    ZKUtil.createEphemeralNodeAndWatch(zk, zk.masterAddressZNode, sn.getVersionedBytes());
081
082    // Wait for the node to be created
083    LOG.info("Waiting for master address manager to be notified");
084    listener.waitForCreation();
085    LOG.info("Master node created");
086    assertTrue(addressManager.hasMaster());
087    ServerName pulledAddress = addressManager.getMasterAddress();
088    assertTrue(pulledAddress.equals(sn));
089
090  }
091
092  public static class NodeCreationListener extends ZooKeeperListener {
093    private static final Log LOG = LogFactory.getLog(NodeCreationListener.class);
094
095    private Semaphore lock;
096    private String node;
097
098    public NodeCreationListener(ZooKeeperWatcher watcher, String node) {
099      super(watcher);
100      lock = new Semaphore(0);
101      this.node = node;
102    }
103
104    @Override
105    public void nodeCreated(String path) {
106      if(path.equals(node)) {
107        LOG.debug("nodeCreated(" + path + ")");
108        lock.release();
109      }
110    }
111
112    public void waitForCreation() throws InterruptedException {
113      lock.acquire();
114    }
115  }
116
117  @org.junit.Rule
118  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
119    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
120}
