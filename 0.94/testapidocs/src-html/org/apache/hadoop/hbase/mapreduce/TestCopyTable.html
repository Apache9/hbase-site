018package org.apache.hadoop.hbase.mapreduce;

020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023import org.apache.hadoop.hbase.HBaseTestingUtility;
024import org.apache.hadoop.hbase.LargeTests;
025import org.apache.hadoop.hbase.MiniHBaseCluster;
026import org.apache.hadoop.hbase.client.Get;

029import org.apache.hadoop.hbase.client.Result;
030import org.apache.hadoop.hbase.util.Bytes;
031import org.junit.AfterClass;
032import org.junit.BeforeClass;
033import org.junit.Test;
034import org.junit.experimental.categories.Category;
035
036/**
037 * Basic test for the CopyTable M/R tool
038 */
039@Category(LargeTests.class)
040public class TestCopyTable {
041  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
042  private static MiniHBaseCluster cluster;
043
044  @BeforeClass
045  public static void beforeClass() throws Exception {
046    cluster = TEST_UTIL.startMiniCluster(3);
047    TEST_UTIL.startMiniMapReduceCluster();
048  }
049
050  @AfterClass
051  public static void afterClass() throws Exception {
052    TEST_UTIL.shutdownMiniMapReduceCluster();
053    TEST_UTIL.shutdownMiniCluster();
054  }
055
056  /**
057   * Simple end-to-end test
058   * @throws Exception
059   */
060  @Test
061  public void testCopyTable() throws Exception {
062    final byte[] TABLENAME1 = Bytes.toBytes("testCopyTable1");
063    final byte[] TABLENAME2 = Bytes.toBytes("testCopyTable2");
064    final byte[] FAMILY = Bytes.toBytes("family");
065    final byte[] COLUMN1 = Bytes.toBytes("c1");
066
067    HTable t1 = TEST_UTIL.createTable(TABLENAME1, FAMILY);
068    HTable t2 = TEST_UTIL.createTable(TABLENAME2, FAMILY);
069
070    // put rows into the first table
071    for (int i = 0; i < 10; i++) {
072      Put p = new Put(Bytes.toBytes("row" + i));
073      p.add(FAMILY, COLUMN1, COLUMN1);
074      t1.put(p);
075    }
076
077    CopyTable copy = new CopyTable(TEST_UTIL.getConfiguration());
078
079    assertEquals(
080      0,
081      copy.run(new String[] { "--new.name=" + Bytes.toString(TABLENAME2),
082          Bytes.toString(TABLENAME1) }));
083
084    // verify the data was copied into table 2
085    for (int i = 0; i < 10; i++) {
086      Get g = new Get(Bytes.toBytes("row" + i));
087      Result r = t2.get(g);
088      assertEquals(1, r.size());
089      assertTrue(Bytes.equals(COLUMN1, r.raw()[0].getQualifier()));
090    }
091
092    t1.close();
093    t2.close();
094    TEST_UTIL.deleteTable(TABLENAME1);
095    TEST_UTIL.deleteTable(TABLENAME2);
096  }
097
098  @Test
099  public void testStartStopRow() throws Exception {
100    final byte[] TABLENAME1 = Bytes.toBytes("testStartStopRow1");
101    final byte[] TABLENAME2 = Bytes.toBytes("testStartStopRow2");
102    final byte[] FAMILY = Bytes.toBytes("family");
103    final byte[] COLUMN1 = Bytes.toBytes("c1");
104    final byte[] ROW0 = Bytes.toBytes("row0");
105    final byte[] ROW1 = Bytes.toBytes("row1");
106    final byte[] ROW2 = Bytes.toBytes("row2");
107
108    HTable t1 = TEST_UTIL.createTable(TABLENAME1, FAMILY);
109    HTable t2 = TEST_UTIL.createTable(TABLENAME2, FAMILY);
110
111    // put rows into the first table
112    Put p = new Put(ROW0);
113    p.add(FAMILY, COLUMN1, COLUMN1);
114    t1.put(p);
115    p = new Put(ROW1);
116    p.add(FAMILY, COLUMN1, COLUMN1);
117    t1.put(p);
118    p = new Put(ROW2);
119    p.add(FAMILY, COLUMN1, COLUMN1);
120    t1.put(p);
121
122    CopyTable copy = new CopyTable(TEST_UTIL.getConfiguration());
123    assertEquals(
124      0,
125      copy.run(new String[] { "--new.name=" + Bytes.toString(TABLENAME2), "--startrow=row1",
126          "--stoprow=row2", Bytes.toString(TABLENAME1) }));
127
128    // verify the data was copied into table 2
129    // row1 exist, row0, row2 do not exist
130    Get g = new Get(ROW1);
131    Result r = t2.get(g);
132    assertEquals(1, r.size());
133    assertTrue(Bytes.equals(COLUMN1, r.raw()[0].getQualifier()));
134    
135    g = new Get(ROW0);
136    r = t2.get(g);
137    assertEquals(0, r.size());
138    
139    g = new Get(ROW2);
140    r = t2.get(g);
141    assertEquals(0, r.size());
142    
143    t1.close();
144    t2.close();
145    TEST_UTIL.deleteTable(TABLENAME1);
146    TEST_UTIL.deleteTable(TABLENAME2);
147  }
148}
