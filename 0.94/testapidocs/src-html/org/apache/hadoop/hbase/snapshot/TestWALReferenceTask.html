018package org.apache.hadoop.hbase.snapshot;
019
020import java.io.IOException;
021import java.util.HashSet;
022import java.util.Set;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.fs.FileSystem;
028import org.apache.hadoop.fs.Path;
029import org.apache.hadoop.hbase.HBaseTestingUtility;
030import org.apache.hadoop.hbase.HConstants;
031import org.apache.hadoop.hbase.SmallTests;
032import org.apache.hadoop.hbase.errorhandling.ForeignExceptionDispatcher;
033import org.apache.hadoop.hbase.protobuf.generated.HBaseProtos.SnapshotDescription;
034import org.apache.hadoop.hbase.snapshot.ReferenceServerWALsTask;
035import org.apache.hadoop.hbase.snapshot.SnapshotDescriptionUtils;
036import org.apache.hadoop.hbase.snapshot.TakeSnapshotUtils;
037import org.apache.hadoop.hbase.util.FSUtils;
038import org.junit.Test;
039import org.junit.experimental.categories.Category;
040import org.mockito.Mockito;
041
042/**
043 * Test that the WAL reference task works as expected
044 */
045@Category(SmallTests.class)
046public class TestWALReferenceTask {
047
048  private static final Log LOG = LogFactory.getLog(TestWALReferenceTask.class);
049  private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
050
051  @Test
052  public void testRun() throws IOException {
053    Configuration conf = UTIL.getConfiguration();
054    FileSystem fs = UTIL.getTestFileSystem();
055    // setup the log dir
056    Path testDir = UTIL.getDataTestDir();
057    Set<String> servers = new HashSet<String>();
058    Path logDir = new Path(testDir, ".logs");
059    Path server1Dir = new Path(logDir, "Server1");
060    servers.add(server1Dir.getName());
061    Path server2Dir = new Path(logDir, "me.hbase.com,56073,1348618509968");
062    servers.add(server2Dir.getName());
063    // logs under server 1
064    Path log1_1 = new Path(server1Dir, "me.hbase.com%2C56073%2C1348618509968.1348618520536");
065    Path log1_2 = new Path(server1Dir, "me.hbase.com%2C56073%2C1348618509968.1234567890123");
066    // logs under server 2
067    Path log2_1 = new Path(server2Dir, "me.hbase.com%2C56074%2C1348618509998.1348618515589");
068    Path log2_2 = new Path(server2Dir, "me.hbase.com%2C56073%2C1348618509968.1234567890123");
069
070    // create all the log files
071    fs.createNewFile(log1_1);
072    fs.createNewFile(log1_2);
073    fs.createNewFile(log2_1);
074    fs.createNewFile(log2_2);
075
076    FSUtils.logFileSystemState(fs, testDir, LOG);
077    FSUtils.setRootDir(conf, testDir);
078    SnapshotDescription snapshot = SnapshotDescription.newBuilder()
079        .setName("testWALReferenceSnapshot").build();
080    ForeignExceptionDispatcher listener = Mockito.mock(ForeignExceptionDispatcher.class);
081
082    // reference all the files in the first server directory
083    ReferenceServerWALsTask task = new ReferenceServerWALsTask(snapshot, listener, server1Dir,
084        conf, fs);
085    task.call();
086
087    // reference all the files in the first server directory
088    task = new ReferenceServerWALsTask(snapshot, listener, server2Dir, conf, fs);
089    task.call();
090
091    // verify that we got everything
092    FSUtils.logFileSystemState(fs, testDir, LOG);
093    Path workingDir = SnapshotDescriptionUtils.getWorkingSnapshotDir(snapshot, testDir);
094    Path snapshotLogDir = new Path(workingDir, HConstants.HREGION_LOGDIR_NAME);
095
096    // make sure we reference the all the wal files
097    TakeSnapshotUtils.verifyAllLogsGotReferenced(fs, logDir, servers, snapshot, snapshotLogDir);
098
099    // make sure we never got an error
100    Mockito.verify(listener, Mockito.atLeastOnce()).rethrowException();
101    Mockito.verifyNoMoreInteractions(listener);
102  }
103}
