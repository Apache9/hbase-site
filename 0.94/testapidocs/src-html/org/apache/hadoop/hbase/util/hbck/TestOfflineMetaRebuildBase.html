037/**
038 * This builds a table, removes info from meta, and then rebuilds meta.
039 */
040@Category(MediumTests.class)
041public class TestOfflineMetaRebuildBase extends OfflineMetaRebuildTestCore {
042
043  @Test(timeout = 120000)
044  public void testMetaRebuild() throws Exception {
045    wipeOutMeta();
046
047    // is meta really messed up?
048    assertEquals(0, scanMeta());
049    assertErrors(doFsck(conf, false),
050        new ERROR_CODE[] { ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
051            ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
052            ERROR_CODE.NOT_IN_META_OR_DEPLOYED,
053            ERROR_CODE.NOT_IN_META_OR_DEPLOYED, });
054    // Note, would like to check # of tables, but this takes a while to time
055    // out.
056
057    // shutdown the minicluster
058    TEST_UTIL.shutdownMiniHBaseCluster();
059    TEST_UTIL.shutdownMiniZKCluster();
060    HConnectionManager.deleteConnection(conf);
061
062    // rebuild meta table from scratch
063    HBaseFsck fsck = new HBaseFsck(conf);
064    assertTrue(fsck.rebuildMeta(false));
065
066    // bring up the minicluster
067    TEST_UTIL.startMiniZKCluster(); // tables seem enabled by default
068    TEST_UTIL.restartHBaseCluster(3);
069    
070    ZooKeeperWatcher zkw = HBaseTestingUtility.getZooKeeperWatcher(TEST_UTIL);
071    
072    LOG.info("Waiting for no more RIT");
073    ZKAssign.blockUntilNoRIT(zkw);
074    LOG.info("No more RIT in ZK, now doing final test verification");
075
076    // everything is good again.
077    assertEquals(4, scanMeta());
078    HTableDescriptor[] htbls = TEST_UTIL.getHBaseAdmin().listTables();
079    LOG.info("Tables present after restart: " + Arrays.toString(htbls));
080
081    assertEquals(1, htbls.length);
082    assertErrors(doFsck(conf, false), new ERROR_CODE[] {});
083    LOG.info("Table " + table + " has " + tableRowCount(conf, table)
084        + " entries.");
085    assertEquals(16, tableRowCount(conf, table));
086  }
087
088
089  @org.junit.Rule
090  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
091    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
092}
