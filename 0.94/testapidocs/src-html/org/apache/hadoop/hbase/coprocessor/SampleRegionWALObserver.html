021package org.apache.hadoop.hbase.coprocessor;
022
023import java.io.IOException;
024import java.util.List;
025import java.util.Arrays;
026
027import org.apache.commons.logging.Log;
028import org.apache.commons.logging.LogFactory;
029
030import org.apache.hadoop.hbase.HRegionInfo;
031import org.apache.hadoop.hbase.KeyValue;
032import org.apache.hadoop.hbase.regionserver.wal.HLogKey;
033import org.apache.hadoop.hbase.regionserver.wal.WALEdit;
034
035/**
036 * Class for testing WALObserver coprocessor.
037 *
038 * It will monitor WAL writing and restoring, and modify passed-in WALEdit, i.e,
039 * ignore specified columns when writing, or add a KeyValue. On the other
040 * side, it checks whether the ignored column is still in WAL when Restoreed
041 * at region reconstruct.
042 */
043public class SampleRegionWALObserver extends BaseRegionObserver
044implements WALObserver {
045
046  private static final Log LOG = LogFactory.getLog(SampleRegionWALObserver.class);
047
048  private byte[] tableName;
049  private byte[] row;
050  private byte[] ignoredFamily;
051  private byte[] ignoredQualifier;
052  private byte[] addedFamily;
053  private byte[] addedQualifier;
054  private byte[] changedFamily;
055  private byte[] changedQualifier;
056
057  private boolean preWALWriteCalled = false;
058  private boolean postWALWriteCalled = false;
059  private boolean preWALRestoreCalled = false;
060  private boolean postWALRestoreCalled = false;
061
062  /**
063   * Set values: with a table name, a column name which will be ignored, and
064   * a column name which will be added to WAL.
065   */
066  public void setTestValues(byte[] tableName, byte[] row, byte[] igf, byte[] igq,
067      byte[] chf, byte[] chq, byte[] addf, byte[] addq) {
068    this.row = row;
069    this.tableName = tableName;
070    this.ignoredFamily = igf;
071    this.ignoredQualifier = igq;
072    this.addedFamily = addf;
073    this.addedQualifier = addq;
074    this.changedFamily = chf;
075    this.changedQualifier = chq;
076  }
077
078
079  @Override
080  public void postWALWrite(ObserverContext<WALCoprocessorEnvironment> env,
081      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException {
082    postWALWriteCalled = true;
083  }
084
085  @Override
086  public boolean preWALWrite(ObserverContext<WALCoprocessorEnvironment> env,
087      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException {
088    boolean bypass = false;
089    // check table name matches or not.
090    if (!Arrays.equals(HRegionInfo.getTableName(info.getRegionName()), this.tableName)) {
091      return bypass;
092    }
093    preWALWriteCalled = true;
094    // here we're going to remove one keyvalue from the WALEdit, and add
095    // another one to it.
096    List<KeyValue> kvs = logEdit.getKeyValues();
097    KeyValue deletedKV = null;
098    for (KeyValue kv : kvs) {
099      // assume only one kv from the WALEdit matches.
100      byte[] family = kv.getFamily();
101      byte[] qulifier = kv.getQualifier();
102
103      if (Arrays.equals(family, ignoredFamily) &&
104          Arrays.equals(qulifier, ignoredQualifier)) {
105        LOG.debug("Found the KeyValue from WALEdit which should be ignored.");
106        deletedKV = kv;
107      }
108      if (Arrays.equals(family, changedFamily) &&
109          Arrays.equals(qulifier, changedQualifier)) {
110        LOG.debug("Found the KeyValue from WALEdit which should be changed.");
111        kv.getBuffer()[kv.getValueOffset()] += 1;
112      }
113    }
114    kvs.add(new KeyValue(row, addedFamily, addedQualifier));
115    if (deletedKV != null) {
116      LOG.debug("About to delete a KeyValue from WALEdit.");
117      kvs.remove(deletedKV);
118    }
119    return bypass;
120  }
121
122  /**
123   * Triggered before  {@link org.apache.hadoop.hbase.regionserver.HRegion} when WAL is
124   * Restoreed.
125   */
126  @Override
127  public void preWALRestore(ObserverContext<RegionCoprocessorEnvironment> env,
128      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException {
129    preWALRestoreCalled = true;
130  }
131
132  /**
133   * Triggered after {@link org.apache.hadoop.hbase.regionserver.HRegion} when WAL is
134   * Restoreed.
135   */
136  @Override
137  public void postWALRestore(ObserverContext<RegionCoprocessorEnvironment> env,
138      HRegionInfo info, HLogKey logKey, WALEdit logEdit) throws IOException {
139    postWALRestoreCalled = true;
140  }
141
142  public boolean isPreWALWriteCalled() {
143    return preWALWriteCalled;
144  }
145
146  public boolean isPostWALWriteCalled() {
147    return postWALWriteCalled;
148  }
149
150  public boolean isPreWALRestoreCalled() {
151    LOG.debug(SampleRegionWALObserver.class.getName() +
152      ".isPreWALRestoreCalled is called.");
153    return preWALRestoreCalled;
154  }
155
156  public boolean isPostWALRestoreCalled() {
157    LOG.debug(SampleRegionWALObserver.class.getName() +
158      ".isPostWALRestoreCalled is called.");
159    return postWALRestoreCalled;
160  }
161}
