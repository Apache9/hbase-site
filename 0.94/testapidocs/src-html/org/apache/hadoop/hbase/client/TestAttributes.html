023import java.io.ByteArrayInputStream;

037@Category(SmallTests.class)
038public class TestAttributes {
039  @Test
040  public void testAttributesSerialization() throws IOException {
041    Put put = new Put();
042    put.setAttribute("attribute1", Bytes.toBytes("value1"));
043    put.setAttribute("attribute2", Bytes.toBytes("value2"));
044    put.setAttribute("attribute3", Bytes.toBytes("value3"));
045
046    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
047    DataOutput out = new DataOutputStream(byteArrayOutputStream);
048    put.write(out);
049
050    Put put2 = new Put();
051    Assert.assertTrue(put2.getAttributesMap().isEmpty());
052
053    put2.readFields(new DataInputStream(new ByteArrayInputStream(byteArrayOutputStream.toByteArray())));
054
055    Assert.assertNull(put2.getAttribute("absent"));
056    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"), put2.getAttribute("attribute1")));
057    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"), put2.getAttribute("attribute2")));
058    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value3"), put2.getAttribute("attribute3")));
059    Assert.assertEquals(3, put2.getAttributesMap().size());
060  }

062  @Test
063  public void testPutAttributes() {
064    Put put = new Put();
065    Assert.assertTrue(put.getAttributesMap().isEmpty());
066    Assert.assertNull(put.getAttribute("absent"));
067
068    put.setAttribute("absent", null);
069    Assert.assertTrue(put.getAttributesMap().isEmpty());
070    Assert.assertNull(put.getAttribute("absent"));
071
072    // adding attribute
073    put.setAttribute("attribute1", Bytes.toBytes("value1"));
074    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"), put.getAttribute("attribute1")));
075    Assert.assertEquals(1, put.getAttributesMap().size());
076    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value1"), put.getAttributesMap().get("attribute1")));
077
078    // overriding attribute value
079    put.setAttribute("attribute1", Bytes.toBytes("value12"));
080    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value12"), put.getAttribute("attribute1")));
081    Assert.assertEquals(1, put.getAttributesMap().size());
082    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value12"), put.getAttributesMap().get("attribute1")));
083
084    // adding another attribute
085    put.setAttribute("attribute2", Bytes.toBytes("value2"));
086    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"), put.getAttribute("attribute2")));
087    Assert.assertEquals(2, put.getAttributesMap().size());
088    Assert.assertTrue(Arrays.equals(Bytes.toBytes("value2"), put.getAttributesMap().get("attribute2")));
089
090    // removing attribute
091    put.setAttribute("attribute2", null);
092    Assert.assertNull(put.getAttribute("attribute2"));
093    Assert.assertEquals(1, put.getAttributesMap().size());
094    Assert.assertNull(put.getAttributesMap().get("attribute2"));
095
096    // removing non-existed attribute
097    put.setAttribute("attribute2", null);
098    Assert.assertNull(put.getAttribute("attribute2"));
099    Assert.assertEquals(1, put.getAttributesMap().size());
100    Assert.assertNull(put.getAttributesMap().get("attribute2"));
101
102    // removing another attribute
103    put.setAttribute("attribute1", null);
104    Assert.assertNull(put.getAttribute("attribute1"));
105    Assert.assertTrue(put.getAttributesMap().isEmpty());
106    Assert.assertNull(put.getAttributesMap().get("attribute1"));
107  }
