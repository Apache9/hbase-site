021package org.apache.hadoop.hbase.rest.model;
022
023import java.io.StringReader;
024import java.io.StringWriter;
025
026import javax.xml.bind.JAXBContext;
027import javax.xml.bind.JAXBException;
028
029import junit.framework.TestCase;
030import org.apache.hadoop.hbase.SmallTests;
031import org.junit.experimental.categories.Category;
032
033@Category(SmallTests.class)
034public class TestColumnSchemaModel extends TestCase {
035
036  protected static final String COLUMN_NAME = "testcolumn";
037  protected static final boolean BLOCKCACHE = true;
038  protected static final int BLOCKSIZE = 16384;
039  protected static final String BLOOMFILTER = "NONE";
040  protected static final String COMPRESSION = "GZ";
041  protected static final boolean IN_MEMORY = false;
042  protected static final int TTL = 86400;
043  protected static final int VERSIONS = 1;
044
045  protected static final String AS_XML =
046    "<ColumnSchema name=\"testcolumn\"" +
047      " BLOCKSIZE=\"16384\"" +
048      " BLOOMFILTER=\"NONE\"" +
049      " BLOCKCACHE=\"true\"" +
050      " COMPRESSION=\"GZ\"" +
051      " VERSIONS=\"1\"" +
052      " TTL=\"86400\"" +
053      " IN_MEMORY=\"false\"/>";
054
055  private JAXBContext context;
056
057  public TestColumnSchemaModel() throws JAXBException {
058    super();
059    context = JAXBContext.newInstance(ColumnSchemaModel.class);
060  }
061
062  protected static ColumnSchemaModel buildTestModel() {
063    ColumnSchemaModel model = new ColumnSchemaModel();
064    model.setName(COLUMN_NAME);
065    model.__setBlockcache(BLOCKCACHE);
066    model.__setBlocksize(BLOCKSIZE);
067    model.__setBloomfilter(BLOOMFILTER);
068    model.__setCompression(COMPRESSION);
069    model.__setInMemory(IN_MEMORY);
070    model.__setTTL(TTL);
071    model.__setVersions(VERSIONS);
072    return model;
073  }
074
075  @SuppressWarnings("unused")
076  private String toXML(ColumnSchemaModel model) throws JAXBException {
077    StringWriter writer = new StringWriter();
078    context.createMarshaller().marshal(model, writer);
079    return writer.toString();
080  }
081
082  private ColumnSchemaModel fromXML(String xml) throws JAXBException {
083    return (ColumnSchemaModel)
084      context.createUnmarshaller().unmarshal(new StringReader(xml));
085  }
086
087  protected static void checkModel(ColumnSchemaModel model) {
088    assertEquals(model.getName(), COLUMN_NAME);
089    assertEquals(model.__getBlockcache(), BLOCKCACHE);
090    assertEquals(model.__getBlocksize(), BLOCKSIZE);
091    assertEquals(model.__getBloomfilter(), BLOOMFILTER);
092    assertTrue(model.__getCompression().equalsIgnoreCase(COMPRESSION));
093    assertEquals(model.__getInMemory(), IN_MEMORY);
094    assertEquals(model.__getTTL(), TTL);
095    assertEquals(model.__getVersions(), VERSIONS);
096  }
097
098  public void testBuildModel() throws Exception {
099    checkModel(buildTestModel());
100  }
101
102  public void testFromXML() throws Exception {
103    checkModel(fromXML(AS_XML));
104  }
105
106  @org.junit.Rule
107  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
108    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
109}
