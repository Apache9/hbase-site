021package org.apache.hadoop.hbase.rest.model;
022
023import java.io.IOException;
024import java.io.StringReader;
025import java.io.StringWriter;
026import java.util.Iterator;
027
028import javax.xml.bind.JAXBContext;
029import javax.xml.bind.JAXBException;
030
031import org.apache.hadoop.hbase.SmallTests;
032import org.apache.hadoop.hbase.util.Base64;
033
034import junit.framework.TestCase;
035import org.junit.experimental.categories.Category;
036
037@Category(SmallTests.class)
038public class TestTableSchemaModel extends TestCase {
039
040  public static final String TABLE_NAME = "testTable";
041  private static final boolean IS_META = false;
042  private static final boolean IS_ROOT = false;
043  private static final boolean READONLY = false;
044
045  private static final String AS_XML =
046    "<TableSchema name=\"testTable\"" +
047      " IS_META=\"false\"" +
048      " IS_ROOT=\"false\"" +
049      " READONLY=\"false\">" +
050      TestColumnSchemaModel.AS_XML + 
051    "</TableSchema>";
052
053  private static final String AS_PB = 
054    "Cgl0ZXN0VGFibGUSEAoHSVNfTUVUQRIFZmFsc2USEAoHSVNfUk9PVBIFZmFsc2USEQoIUkVBRE9O" +
055    "TFkSBWZhbHNlGpcBCgp0ZXN0Y29sdW1uEhIKCUJMT0NLU0laRRIFMTYzODQSEwoLQkxPT01GSUxU" +
056    "RVISBE5PTkUSEgoKQkxPQ0tDQUNIRRIEdHJ1ZRIRCgtDT01QUkVTU0lPThICR1oSDQoIVkVSU0lP" +
057    "TlMSATESDAoDVFRMEgU4NjQwMBISCglJTl9NRU1PUlkSBWZhbHNlGICjBSABKgJHWigA";
058
059  private JAXBContext context;
060
061  public TestTableSchemaModel() throws JAXBException {
062    super();
063    context = JAXBContext.newInstance(
064      ColumnSchemaModel.class,
065      TableSchemaModel.class);
066  }
067
068  public static TableSchemaModel buildTestModel() {
069    return buildTestModel(TABLE_NAME);
070  }
071
072  public static TableSchemaModel buildTestModel(String name) {
073    TableSchemaModel model = new TableSchemaModel();
074    model.setName(name);
075    model.__setIsMeta(IS_META);
076    model.__setIsRoot(IS_ROOT);
077    model.__setReadOnly(READONLY);
078    model.addColumnFamily(TestColumnSchemaModel.buildTestModel());
079    return model;
080  }
081
082  @SuppressWarnings("unused")
083  private String toXML(TableSchemaModel model) throws JAXBException {
084    StringWriter writer = new StringWriter();
085    context.createMarshaller().marshal(model, writer);
086    return writer.toString();
087  }
088
089  private TableSchemaModel fromXML(String xml) throws JAXBException {
090    return (TableSchemaModel)
091      context.createUnmarshaller().unmarshal(new StringReader(xml));
092  }
093
094  @SuppressWarnings("unused")
095  private byte[] toPB(TableSchemaModel model) {
096    return model.createProtobufOutput();
097  }
098
099  private TableSchemaModel fromPB(String pb) throws IOException {
100    return (TableSchemaModel) 
101      new TableSchemaModel().getObjectFromMessage(Base64.decode(AS_PB));
102  }
103
104  public static void checkModel(TableSchemaModel model) {
105    checkModel(model, TABLE_NAME);
106  }
107
108  public static void checkModel(TableSchemaModel model, String tableName) {
109    assertEquals(model.getName(), tableName);
110    assertEquals(model.__getIsMeta(), IS_META);
111    assertEquals(model.__getIsRoot(), IS_ROOT);
112    assertEquals(model.__getReadOnly(), READONLY);
113    Iterator<ColumnSchemaModel> families = model.getColumns().iterator();
114    assertTrue(families.hasNext());
115    ColumnSchemaModel family = families.next();
116    TestColumnSchemaModel.checkModel(family);
117    assertFalse(families.hasNext());
118  }
119
120  public void testBuildModel() throws Exception {
121    checkModel(buildTestModel());
122  }
123
124  public void testFromXML() throws Exception {
125    checkModel(fromXML(AS_XML));
126  }
127
128  public void testFromPB() throws Exception {
129    checkModel(fromPB(AS_PB));
130  }
131
132  @org.junit.Rule
133  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
134    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
135}
