021package org.apache.hadoop.hbase.rest.model;
022
023import java.io.StringReader;
024import java.io.StringWriter;
025
026import javax.xml.bind.JAXBContext;
027import javax.xml.bind.JAXBException;
028
029import org.apache.hadoop.hbase.*;
030import org.apache.hadoop.hbase.util.Bytes;
031
032import junit.framework.TestCase;
033import org.junit.experimental.categories.Category;
034
035@Category(SmallTests.class)
036public class TestTableRegionModel extends TestCase {
037  private static final String TABLE = "testtable";
038  private static final byte[] START_KEY = Bytes.toBytes("abracadbra");
039  private static final byte[] END_KEY = Bytes.toBytes("zzyzx");
040  private static final long ID = 8731042424L;
041  private static final String LOCATION = "testhost:9876";
042
043  private static final String AS_XML =
044    "<Region location=\"testhost:9876\"" +
045      " endKey=\"enp5eng=\"" +
046      " startKey=\"YWJyYWNhZGJyYQ==\"" +
047      " id=\"8731042424\"" +
048      " name=\"testtable,abracadbra,8731042424\"/>";
049
050  private JAXBContext context;
051
052  public TestTableRegionModel() throws JAXBException {
053    super();
054    context = JAXBContext.newInstance(TableRegionModel.class);
055  }
056
057  private TableRegionModel buildTestModel() {
058    TableRegionModel model =
059      new TableRegionModel(TABLE, ID, START_KEY, END_KEY, LOCATION);
060    return model;
061  }
062
063  @SuppressWarnings("unused")
064  private String toXML(TableRegionModel model) throws JAXBException {
065    StringWriter writer = new StringWriter();
066    context.createMarshaller().marshal(model, writer);
067    return writer.toString();
068  }
069
070  private TableRegionModel fromXML(String xml) throws JAXBException {
071    return (TableRegionModel)
072      context.createUnmarshaller().unmarshal(new StringReader(xml));
073  }
074
075  private void checkModel(TableRegionModel model) {
076    assertTrue(Bytes.equals(model.getStartKey(), START_KEY));
077    assertTrue(Bytes.equals(model.getEndKey(), END_KEY));
078    assertEquals(model.getId(), ID);
079    assertEquals(model.getLocation(), LOCATION);
080    assertEquals(model.getName(), 
081      TABLE + "," + Bytes.toString(START_KEY) + "," + Long.toString(ID) +
082      ".ad9860f031282c46ed431d7af8f94aca.");
083  }
084
085  public void testBuildModel() throws Exception {
086    checkModel(buildTestModel());
087  }
088
089  public void testGetName() {
090    TableRegionModel model = buildTestModel();
091    String modelName = model.getName();
092    HRegionInfo hri = new HRegionInfo(Bytes.toBytes(TABLE),
093      START_KEY, END_KEY, false, ID);
094    assertEquals(modelName, hri.getRegionNameAsString());
095  }
096
097  public void testSetName() {
098    TableRegionModel model = buildTestModel();
099    String name = model.getName();
100    model.setName(name);
101    assertEquals(name, model.getName());
102  }
103
104  public void testFromXML() throws Exception {
105    checkModel(fromXML(AS_XML));
106  }
107
108  @org.junit.Rule
109  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
110    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
111}
