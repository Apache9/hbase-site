018package org.apache.hadoop.hbase.master;
019
020import org.apache.hadoop.hbase.master.AssignmentManager.RegionState;
021
022/**
023 * Package scoped mocking utility.
024 */
025public class Mocking {
026
027  static void waitForRegionPendingOpenInRIT(AssignmentManager am, String encodedName)
028    throws InterruptedException {
029    // We used to do a check like this:
030    //!Mocking.verifyRegionState(this.watcher, REGIONINFO, EventType.M_ZK_REGION_OFFLINE)) {
031    // There is a race condition with this: because we may do the transition to
032    // RS_ZK_REGION_OPENING before the RIT is internally updated. We need to wait for the
033    // RIT to be as we need it to be instead. This cannot happen in a real cluster as we
034    // update the RIT before sending the openRegion request.
035
036    boolean wait = true;
037    while (wait) {
038      RegionState state = am.getRegionsInTransition().get(encodedName);
039      if (state != null && state.isPendingOpen()){
040        wait = false;
041      } else {
042        Thread.sleep(1);
043      }
044    }
045  }
046
047  static void waitForRegionOfflineInRIT(AssignmentManager am, String encodedName)
048      throws InterruptedException {
049    boolean wait = true;
050    while (wait) {
051      RegionState state = am.getRegionsInTransition().get(encodedName);
052      if (state != null && state.isOffline()) {
053        wait = false;
054      } else {
055        Thread.sleep(1);
056      }
057    }
058  }
059}
