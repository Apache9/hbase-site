042@Category(LargeTests.class)
043public class TestRestartCluster {
044  private static final Log LOG = LogFactory.getLog(TestRestartCluster.class);
045  private HBaseTestingUtility UTIL = new HBaseTestingUtility();
046
047  private static final byte[] TABLENAME = Bytes.toBytes("master_transitions");
048  private static final byte [][] FAMILIES = {Bytes.toBytes("a")};
049  private static final byte [][] TABLES = {
050      Bytes.toBytes("restartTableOne"),
051      Bytes.toBytes("restartTableTwo"),
052      Bytes.toBytes("restartTableThree")
053  };
054  private static final byte [] FAMILY = Bytes.toBytes("family");
055
056  @After public void tearDown() throws Exception {
057    UTIL.shutdownMiniCluster();
058  }
059
060  @Test (timeout=300000) public void testRestartClusterAfterKill()
061  throws Exception {
062    UTIL.startMiniZKCluster();
063    ZooKeeperWatcher zooKeeper =
064      new ZooKeeperWatcher(UTIL.getConfiguration(), "cluster1", null, true);
065
066    // create the unassigned region, throw up a region opened state for META
067    String unassignedZNode = zooKeeper.assignmentZNode;
068    ZKUtil.createAndFailSilent(zooKeeper, unassignedZNode);
069
070    ServerName sn = new ServerName(HMaster.MASTER, -1, System.currentTimeMillis());
071
072    ZKAssign.createNodeOffline(zooKeeper, HRegionInfo.ROOT_REGIONINFO, sn);
073
074    ZKAssign.createNodeOffline(zooKeeper, HRegionInfo.FIRST_META_REGIONINFO, sn);
075
076    LOG.debug("Created UNASSIGNED zNode for ROOT and META regions in state " +
077        EventType.M_ZK_REGION_OFFLINE);
078
079    // start the HB cluster
080    LOG.info("Starting HBase cluster...");
081    UTIL.startMiniCluster(2);
082
083    UTIL.createTable(TABLENAME, FAMILIES);
084    LOG.info("Created a table, waiting for table to be available...");
085    UTIL.waitTableAvailable(TABLENAME, 60*1000);
086
087    LOG.info("Master deleted unassigned region and started up successfully.");
088  }
089
090  @Test (timeout=300000)
091  public void testClusterRestart() throws Exception {
092    UTIL.startMiniCluster(3);
093    while (!UTIL.getMiniHBaseCluster().getMaster().isInitialized()) {
094      Threads.sleep(1);
095    }
096    LOG.info("\n\nCreating tables");
097    for(byte [] TABLE : TABLES) {
098      UTIL.createTable(TABLE, FAMILY);
099    }
100    for(byte [] TABLE : TABLES) {
101      UTIL.waitTableAvailable(TABLE, 30000);
102    }
103
104    List<HRegionInfo> allRegions =
105      MetaScanner.listAllRegions(UTIL.getConfiguration(), true);
106    assertEquals(3, allRegions.size());
107
108    LOG.info("\n\nShutting down cluster");
109    UTIL.shutdownMiniHBaseCluster();
110
111    LOG.info("\n\nSleeping a bit");
112    Thread.sleep(2000);
113
114    LOG.info("\n\nStarting cluster the second time");
115    UTIL.restartHBaseCluster(3);
116
117    // Need to use a new 'Configuration' so we make a new HConnection.
118    // Otherwise we're reusing an HConnection that has gone stale because
119    // the shutdown of the cluster also called shut of the connection.
120    allRegions = MetaScanner.
121      listAllRegions(new Configuration(UTIL.getConfiguration()), true);
122    assertEquals(3, allRegions.size());
123
124    LOG.info("\n\nWaiting for tables to be available");
125    for(byte [] TABLE: TABLES) {
126      try {
127        UTIL.createTable(TABLE, FAMILY);
128        assertTrue("Able to create table that should already exist", false);
129      } catch(TableExistsException tee) {
130        LOG.info("Table already exists as expected");
131      }
132      UTIL.waitTableAvailable(TABLE, 30000);
133    }
134  }
