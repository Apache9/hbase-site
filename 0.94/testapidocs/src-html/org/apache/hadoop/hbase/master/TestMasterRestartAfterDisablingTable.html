020package org.apache.hadoop.hbase.master;

022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertTrue;
024
025import java.io.IOException;
026import java.util.List;
027import java.util.NavigableSet;
028import java.util.TreeSet;
029
030import org.apache.commons.logging.Log;
031import org.apache.commons.logging.LogFactory;
032import org.apache.hadoop.conf.Configuration;
033import org.apache.hadoop.hbase.*;
034import org.apache.hadoop.hbase.client.HBaseAdmin;
035import org.apache.hadoop.hbase.client.HTable;
036import org.apache.hadoop.hbase.util.Bytes;
037import org.apache.hadoop.hbase.util.JVMClusterUtil.MasterThread;
038import org.apache.hadoop.hbase.util.JVMClusterUtil.RegionServerThread;
039import org.apache.hadoop.hbase.zookeeper.ZKAssign;
040import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
041import org.apache.zookeeper.KeeperException;
042import org.junit.Test;
043import org.junit.experimental.categories.Category;
044
045@Category(LargeTests.class)
046public class TestMasterRestartAfterDisablingTable {
047
048  private static final Log LOG = LogFactory.getLog(TestMasterRestartAfterDisablingTable.class);
049
050  @Test
051  public void testForCheckingIfEnableAndDisableWorksFineAfterSwitch()
052      throws Exception {
053    final int NUM_MASTERS = 2;
054    final int NUM_RS = 1;
055    final int NUM_REGIONS_TO_CREATE = 4;
056
057    // Start the cluster
058    log("Starting cluster");
059    Configuration conf = HBaseConfiguration.create();
060    conf.setInt("hbase.master.assignment.timeoutmonitor.period", 2000);
061    conf.setInt("hbase.master.assignment.timeoutmonitor.timeout", 5000);
062    HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility(conf);
063    TEST_UTIL.startMiniCluster(NUM_MASTERS, NUM_RS);
064    MiniHBaseCluster cluster = TEST_UTIL.getHBaseCluster();
065    log("Waiting for active/ready master");
066    cluster.waitForActiveAndReadyMaster();
067    ZooKeeperWatcher zkw = new ZooKeeperWatcher(conf, "testmasterRestart", null);
068    HMaster master = cluster.getMaster();
069
070    // Create a table with regions
071    byte[] table = Bytes.toBytes("tableRestart");
072    byte[] family = Bytes.toBytes("family");
073    log("Creating table with " + NUM_REGIONS_TO_CREATE + " regions");
074    HTable ht = TEST_UTIL.createTable(table, family);
075    int numRegions = TEST_UTIL.createMultiRegions(conf, ht, family,
076        NUM_REGIONS_TO_CREATE);
077    numRegions += 2; // catalogs
078    log("Waiting for no more RIT\n");
079    blockUntilNoRIT(zkw, master);
080    log("Disabling table\n");
081    TEST_UTIL.getHBaseAdmin().disableTable(table);
082
083    NavigableSet<String> regions = getAllOnlineRegions(cluster);
084    assertEquals(
085        "The number of regions for the table tableRestart should be 0 and only"
086            + "the catalog tables should be present.", 2, regions.size());
087
088    List<MasterThread> masterThreads = cluster.getMasterThreads();
089    MasterThread activeMaster = null;
090    if (masterThreads.get(0).getMaster().isActiveMaster()) {
091      activeMaster = masterThreads.get(0);
092    } else {
093      activeMaster = masterThreads.get(1);
094    }
095    activeMaster.getMaster().stop(
096        "stopping the active master so that the backup can become active");
097    cluster.hbaseCluster.waitOnMaster(activeMaster);
098    cluster.waitForActiveAndReadyMaster();
099
100    assertTrue("The table should not be in enabled state", cluster.getMaster()
101        .getAssignmentManager().getZKTable().isDisablingOrDisabledTable(
102            "tableRestart"));
103    log("Enabling table\n");
104    // Need a new Admin, the previous one is on the old master
105    HBaseAdmin admin = new HBaseAdmin(TEST_UTIL.getConfiguration());
106    admin.enableTable(table);
107    admin.close();
108    log("Waiting for no more RIT\n");
109    blockUntilNoRIT(zkw, master);
110    log("Verifying there are " + numRegions + " assigned on cluster\n");
111    regions = getAllOnlineRegions(cluster);
112    assertEquals(
113        "The assigned regions were not onlined after master switch except for the catalog tables.",
114        6, regions.size());
115    assertTrue("The table should be in enabled state", cluster.getMaster()
116        .getAssignmentManager().getZKTable().isEnabledTable("tableRestart"));
117    ht.close();
118    TEST_UTIL.shutdownMiniCluster();
119  }
120
121  private void log(String msg) {
122    LOG.debug("\n\nTRR: " + msg + "\n");
123  }
124
125  private void blockUntilNoRIT(ZooKeeperWatcher zkw, HMaster master)
126      throws KeeperException, InterruptedException {
127    ZKAssign.blockUntilNoRIT(zkw);
128    master.assignmentManager.waitUntilNoRegionsInTransition(60000);
129  }
130
131  private NavigableSet<String> getAllOnlineRegions(MiniHBaseCluster cluster)
132      throws IOException {
133    NavigableSet<String> online = new TreeSet<String>();
134    for (RegionServerThread rst : cluster.getLiveRegionServerThreads()) {
135      for (HRegionInfo region : rst.getRegionServer().getOnlineRegions()) {
136        online.add(region.getRegionNameAsString());
137      }
138    }
139    return online;
140  }
141
142  @org.junit.Rule
143  public org.apache.hadoop.hbase.ResourceCheckerJUnitRule cu =
144    new org.apache.hadoop.hbase.ResourceCheckerJUnitRule();
145}
