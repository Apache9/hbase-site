018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021import java.net.InetSocketAddress;
022
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.hbase.CoprocessorEnvironment;
025import org.apache.hadoop.hbase.ServerName;
026import org.apache.hadoop.hbase.client.HConnectionManager.HConnectionImplementation;
027import org.apache.hadoop.hbase.coprocessor.RegionCoprocessorEnvironment;
028import org.apache.hadoop.hbase.ipc.HRegionInterface;
029import org.apache.hadoop.hbase.regionserver.HRegionServer;
030import org.apache.hadoop.hbase.regionserver.RegionServerServices;
031
032/**
033 * Connection to an HTable from within a Coprocessor. Can do some nice tricks since we know we are
034 * on a regionserver.
035 * <p>
036 * This shouldn't be used by usual HBase clients - its merely in this package to maintain visibility
037 * considerations for the {@link HConnectionImplementation}.
038 */
039@SuppressWarnings("javadoc")
040public class CoprocessorHConnection extends HConnectionManager.HConnectionImplementation {
041
042  /**
043   * Create an unmanaged {@link HConnection} based on the environment in which we are running the
044   * coprocessor. The {@link HConnection} must be externally cleaned up (we bypass the usual HTable
045   * cleanup mechanisms since we own everything).
046   * @param env environment hosting the {@link HConnection}
047   * @return an unmanaged {@link HConnection}.
048   * @throws IOException if we cannot create the basic connection
049   */
050  @SuppressWarnings("resource")
051  public static HConnection getConnectionForEnvironment(CoprocessorEnvironment env)
052      throws IOException {
053    Configuration conf = env.getConfiguration();
054    HConnection connection = null;
055    // this bit is a little hacky - we need to reach kind far into the internals. However, since we
056    // are in a coprocessor (which is part of the internals), this is more ok.
057    if (env instanceof RegionCoprocessorEnvironment) {
058      RegionCoprocessorEnvironment e = (RegionCoprocessorEnvironment) env;
059      RegionServerServices services = e.getRegionServerServices();
060      if (services instanceof HRegionServer) {
061        connection = new CoprocessorHConnection(conf, (HRegionServer) services);
062      }
063    }
064    // didn't create the custom HConnection, so just create the usual connection. Saves us some conf
065    // lookups, but no network accesses or anything else with excessive overhead.
066    if (connection == null) {
067      connection = HConnectionManager.createConnection(conf);
068    }
069    return connection;
070  }
071
072  private ServerName serverName;
073  private HRegionServer server;
074
075  CoprocessorHConnection(Configuration conf, HRegionServer server) throws IOException {
076    super(conf, false, null);
077    this.server = server;
078    this.serverName = server.getServerName();
079  }
080
081  @Override
082  HRegionInterface getHRegionConnection(final String hostname, final int port,
083      final InetSocketAddress isa, final boolean master) throws IOException {
084    // check to see where the server is running
085    // need this isa stuff here since its what the HConnectionManager is doing too
086    boolean isRemote = false;
087    if (isa != null) {
088      isRemote = checkRemote(isa.getHostName(), isa.getPort());
089    } else {
090      isRemote = checkRemote(hostname, port);
091    }
092    // if we aren't talking to the local HRegionServer, then do the usual thing
093    if (isRemote) {
094      return super.getHRegionConnection(hostname, port, isa, master);
095    }
096
097    // local access, so just pass the actual server, rather than a proxy
098    return this.server;
099  }
100
101  /**
102   * Check that the hostname and port map the the server on which we are currently running
103   * @param hostName hostname to check
104   * @param port port to check
105   * @return <tt>true</tt> the connection is <b>not</b> currently running on the given host and port
106   */
107  private boolean checkRemote(String hostName, int port) {
108    return !(this.serverName.getHostname().equals(hostName) && this.serverName.getPort() == port);
109  }
110}
