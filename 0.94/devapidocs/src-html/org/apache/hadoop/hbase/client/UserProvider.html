018package org.apache.hadoop.hbase.client;
019
020import java.io.IOException;
021
022import org.apache.hadoop.conf.Configuration;
023import org.apache.hadoop.hbase.BaseConfigurable;
024import org.apache.hadoop.hbase.security.User;
025import org.apache.hadoop.security.UserGroupInformation;
026import org.apache.hadoop.util.ReflectionUtils;
027
028/**
029 * Provide an instance of a user. Allows custom {@link User} creation.
030 */
031public class UserProvider extends BaseConfigurable {
032
033  private static final String USER_PROVIDER_CONF_KEY = "hbase.client.userprovider.class";
034
035  /**
036   * Instantiate the {@link UserProvider} specified in the configuration and set the passed
037   * configuration via {@link UserProvider#setConf(Configuration)}
038   * @param conf to read and set on the created {@link UserProvider}
039   * @return a {@link UserProvider} ready for use.
040   */
041  public static UserProvider instantiate(Configuration conf){
042    Class<? extends UserProvider> clazz =
043        conf.getClass(USER_PROVIDER_CONF_KEY, UserProvider.class, UserProvider.class);
044    return ReflectionUtils.newInstance(clazz, conf);
045  }
046
047  /**
048   * Set the {@link UserProvider} in the given configuration that should be instantiated
049   * @param conf to update
050   * @param provider class of the provider to set
051   */
052  public static void setUserProviderForTesting(Configuration conf,
053      Class<? extends UserProvider> provider) {
054    conf.set(USER_PROVIDER_CONF_KEY, provider.getName());
055  }
056
057  /**
058   * @return the userName for the current logged-in user.
059   * @throws IOException if the underlying user cannot be obtained
060   */
061  public String getCurrentUserName() throws IOException {
062    User user = getCurrent();
063    return user == null ? null : user.getName();
064  }
065
066  /**
067   * @return <tt>true</tt> if security is enabled, <tt>false</tt> otherwise
068   */
069  public boolean isHBaseSecurityEnabled() {
070    return User.isHBaseSecurityEnabled(this.getConf());
071  }
072
073  /**
074   * @return the current user within the current execution context
075   * @throws IOException if the user cannot be loaded
076   */
077  public User getCurrent() throws IOException {
078    return User.getCurrent();
079  }
080
081  public User create(UserGroupInformation ugi) {
082    return User.create(ugi);
083  }
084
085  /**
086   * Log in the current process using the given configuration keys for the credential file and login
087   * principal.
088   * <p>
089   * <strong>This is only applicable when running on secure Hadoop</strong> -- see
090   * org.apache.hadoop.security.SecurityUtil#login(Configuration,String,String,String). On regular
091   * Hadoop (without security features), this will safely be ignored.
092   * </p>
093   * @param conf The configuration data to use
094   * @param fileConfKey Property key used to configure path to the credential file
095   * @param principalConfKey Property key used to configure login principal
096   * @param localhost Current hostname to use in any credentials
097   * @throws IOException underlying exception from SecurityUtil.login() call
098   */
099  public void login(String fileConfKey, String principalConfKey, String localhost)
100      throws IOException {
101    User.login(getConf(), fileConfKey, principalConfKey, localhost);
102  }
103
104  /**
105   * @return whether or not Kerberos authentication is configured for Hadoop. For non-secure Hadoop,
106   *         this always returns <code>false</code>. For secure Hadoop, it will return the value
107   *         from {@code UserGroupInformation.isSecurityEnabled()}.
108   */
109  public boolean isHadoopSecurityEnabled() {
110    return User.isSecurityEnabled();
111  }
112}
