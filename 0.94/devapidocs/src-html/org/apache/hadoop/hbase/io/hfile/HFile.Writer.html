063/**
064 * File format for hbase.
065 * A file of sorted key/value pairs. Both keys and values are byte arrays.
066 * <p>
067 * The memory footprint of a HFile includes the following (below is taken from the
068 * <a
069 * href=https://issues.apache.org/jira/browse/HADOOP-3315>TFile</a> documentation
070 * but applies also to HFile):
071 * <ul>
072 * <li>Some constant overhead of reading or writing a compressed block.
073 * <ul>
074 * <li>Each compressed block requires one compression/decompression codec for
075 * I/O.
076 * <li>Temporary space to buffer the key.
077 * <li>Temporary space to buffer the value.
078 * </ul>
079 * <li>HFile index, which is proportional to the total number of Data Blocks.
080 * The total amount of memory needed to hold the index can be estimated as
081 * (56+AvgKeySize)*NumBlocks.
082 * </ul>
083 * Suggestions on performance optimization.
084 * <ul>
085 * <li>Minimum block size. We recommend a setting of minimum block size between
086 * 8KB to 1MB for general usage. Larger block size is preferred if files are
087 * primarily for sequential access. However, it would lead to inefficient random
088 * access (because there are more data to decompress). Smaller blocks are good
089 * for random access, but require more memory to hold the block index, and may
090 * be slower to create (because we must flush the compressor stream at the
091 * conclusion of each data block, which leads to an FS I/O flush). Further, due
092 * to the internal caching in Compression codec, the smallest possible block
093 * size would be around 20KB-30KB.
094 * <li>The current implementation does not offer true multi-threading for
095 * reading. The implementation uses FSDataInputStream seek()+read(), which is
096 * shown to be much faster than positioned-read call in single thread mode.
097 * However, it also means that if multiple threads attempt to access the same
098 * HFile (using multiple scanners) simultaneously, the actual I/O is carried out
099 * sequentially even if they access different DFS blocks (Reexamine! pread seems
100 * to be 10% faster than seek+read in my testing -- stack).
101 * <li>Compression codec. Use "none" if the data is not very compressable (by
102 * compressable, I mean a compression ratio at least 2:1). Generally, use "lzo"
103 * as the starting point for experimenting. "gz" overs slightly better
104 * compression ratio over "lzo" but requires 4x CPU to compress and 2x CPU to
105 * decompress, comparing to "lzo".
106 * </ul>
107 *
108 * For more on the background behind HFile, see <a
109 * href=https://issues.apache.org/jira/browse/HBASE-61>HBASE-61</a>.
110 * <p>
111 * File is made of data blocks followed by meta data blocks (if any), a fileinfo
112 * block, data block index, meta data block index, and a fixed size trailer
113 * which records the offsets at which file changes content type.
114 * <pre>&lt;data blocks>&lt;meta blocks>&lt;fileinfo>&lt;data index>&lt;meta index>&lt;trailer></pre>
115 * Each block has a bit of magic at its start.  Block are comprised of
116 * key/values.  In data blocks, they are both byte arrays.  Metadata blocks are
117 * a String key and a byte array value.  An empty file looks like this:
118 * <pre>&lt;fileinfo>&lt;trailer></pre>.  That is, there are not data nor meta
119 * blocks present.
120 * <p>
121 * TODO: Do scanners need to be able to take a start and end row?
122 * TODO: Should BlockIndex know the name of its file?  Should it have a Path
123 * that points at its file say for the case where an index lives apart from
124 * an HFile instance?
125 */

182  // For getting more detailed stats on FS latencies
183  // If, for some reason, the metrics subsystem stops polling for latencies,
184  // I don't want data to pile up in a memory leak
185  // so, after LATENCY_BUFFER_SIZE items have been enqueued for processing,
186  // fs latency stats will be dropped (and this behavior will be logged)
187  private static final int LATENCY_BUFFER_SIZE = 5000;
188  private static final BlockingQueue<Long> fsReadLatenciesNanos =
189      new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
190  private static final BlockingQueue<Long> fsWriteLatenciesNanos =
191      new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
192  private static final BlockingQueue<Long> fsPreadLatenciesNanos =
193      new ArrayBlockingQueue<Long>(LATENCY_BUFFER_SIZE);
194
195  public static final void offerReadLatency(long latencyNanos, boolean pread) {
196    if (pread) {
197      fsPreadLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
198      preadOps.incrementAndGet();
199      preadTimeNano.addAndGet(latencyNanos);
200    } else {
201      fsReadLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
202      readTimeNano.addAndGet(latencyNanos);
203      readOps.incrementAndGet();
204    }
205  }
206
207  public static final void offerWriteLatency(long latencyNanos) {
208    fsWriteLatenciesNanos.offer(latencyNanos); // might be silently dropped, if the queue is full
209
210    writeTimeNano.addAndGet(latencyNanos);
211    writeOps.incrementAndGet();
212  }

214  public static final Collection<Long> getReadLatenciesNanos() {
215    final List<Long> latencies =
216        Lists.newArrayListWithCapacity(fsReadLatenciesNanos.size());
217    fsReadLatenciesNanos.drainTo(latencies);
218    return latencies;
219  }

392    public Writer create() throws IOException {

529  /**
530   * Method returns the reader given the specified arguments.
531   * TODO This is a bad abstraction.  See HBASE-6635.
532   *
533   * @param path hfile's path
534   * @param fsdis an open checksummed stream of path's file
535   * @param fsdisNoFsChecksum an open unchecksummed stream of path's file
536   * @param size max size of the trailer.
537   * @param closeIStream boolean for closing file after the getting the reader version.
538   * @param cacheConf Cache configuation values, cannot be null.
539   * @param preferredEncodingInCache
540   * @param hfs
541   * @return an appropriate instance of HFileReader
542   * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
543   */
544  private static Reader pickReaderVersion(Path path, FSDataInputStream fsdis,
545      FSDataInputStream fsdisNoFsChecksum,
546      long size, boolean closeIStream, CacheConfig cacheConf,
547      DataBlockEncoding preferredEncodingInCache, HFileSystem hfs)
548      throws IOException {
549    FixedFileTrailer trailer = null;
550    try {
551      trailer = FixedFileTrailer.readFromStream(fsdis, size);
552      switch (trailer.getMajorVersion()) {
553      case 1:
554        return new HFileReaderV1(path, trailer, fsdis, size, closeIStream,
555            cacheConf);
556      case 2:
557        return new HFileReaderV2(path, trailer, fsdis, fsdisNoFsChecksum,
558            size, closeIStream,
559            cacheConf, preferredEncodingInCache, hfs);
560      default:
561        throw new IllegalArgumentException("Invalid HFile version " + trailer.getMajorVersion());
562      }
563    } catch (Throwable t) {
564      if (closeIStream) {
565        try {
566          if (fsdis != fsdisNoFsChecksum && fsdisNoFsChecksum != null) {
567            fsdisNoFsChecksum.close();
568            fsdisNoFsChecksum = null;
569          }
570        } catch (Throwable t2) {
571          LOG.warn("Error closing fsdisNoFsChecksum FSDataInputStream", t2);
572        }
573        try {
574          if (fsdis != null) {
575            fsdis.close();
576            fsdis = null;
577          }
578        } catch (Throwable t2) {
579          LOG.warn("Error closing fsdis FSDataInputStream", t2);
580        }
581      }
582      throw new CorruptHFileException("Problem reading HFile Trailer from file " + path, t);
583    }
584  }

586  /**
587   * @param fs A file system
588   * @param path Path to HFile
589   * @param cacheConf Cache configuration for hfile's contents
590   * @param preferredEncodingInCache Preferred in-cache data encoding algorithm.
591   * @return A version specific Hfile Reader
592   * @throws IOException If file is invalid, will throw CorruptHFileException flavored IOException
593   */
594  public static Reader createReaderWithEncoding(
595      FileSystem fs, Path path, CacheConfig cacheConf,
596      DataBlockEncoding preferredEncodingInCache) throws IOException {
597    final boolean closeIStream = true;
598    HFileSystem hfs = null;
599    FSDataInputStream fsdis = fs.open(path);
600    FSDataInputStream fsdisNoFsChecksum = fsdis;
601    // If the fs is not an instance of HFileSystem, then create an
602    // instance of HFileSystem that wraps over the specified fs.
603    // In this case, we will not be able to avoid checksumming inside
604    // the filesystem.
605    if (!(fs instanceof HFileSystem)) {
606      hfs = new HFileSystem(fs);
607    } else {
608      hfs = (HFileSystem)fs;
609      // open a stream to read data without checksum verification in
610      // the filesystem
611      if (hfs != null) {
612        fsdisNoFsChecksum = hfs.getNoChecksumFs().open(path);
613      }
614    }
615    return pickReaderVersion(path, fsdis, fsdisNoFsChecksum,
616        fs.getFileStatus(path).getLen(), closeIStream, cacheConf,
617        preferredEncodingInCache, hfs);
618  }

677  /*
678   * Metadata for this file.  Conjured by the writer.  Read in by the reader.
679   */
680  static class FileInfo extends HbaseMapWritable<byte [], byte []> {
681    static final String RESERVED_PREFIX = "hfile.";
682    static final byte[] RESERVED_PREFIX_BYTES = Bytes.toBytes(RESERVED_PREFIX);
683    static final byte [] LASTKEY = Bytes.toBytes(RESERVED_PREFIX + "LASTKEY");
684    static final byte [] AVG_KEY_LEN =
685      Bytes.toBytes(RESERVED_PREFIX + "AVG_KEY_LEN");
686    static final byte [] AVG_VALUE_LEN =
687      Bytes.toBytes(RESERVED_PREFIX + "AVG_VALUE_LEN");
688    static final byte [] COMPARATOR =
689      Bytes.toBytes(RESERVED_PREFIX + "COMPARATOR");
690
691    /**
692     * Append the given key/value pair to the file info, optionally checking the
693     * key prefix.
694     *
695     * @param k key to add
696     * @param v value to add
697     * @param checkPrefix whether to check that the provided key does not start
698     *          with the reserved prefix
699     * @return this file info object
700     * @throws IOException if the key or value is invalid
701     */
702    public FileInfo append(final byte[] k, final byte[] v,
703        final boolean checkPrefix) throws IOException {
704      if (k == null || v == null) {
705        throw new NullPointerException("Key nor value may be null");
706      }
707      if (checkPrefix && isReservedFileInfoKey(k)) {
708        throw new IOException("Keys with a " + FileInfo.RESERVED_PREFIX
709            + " are reserved");
710      }
711      put(k, v);
712      return this;
713    }
714
715  }
716
717  /** Return true if the given file info key is reserved for internal use. */
718  public static boolean isReservedFileInfoKey(byte[] key) {
719    return Bytes.startsWith(key, FileInfo.RESERVED_PREFIX_BYTES);
720  }

743  static int longToInt(final long l) {
744    // Expecting the size() of a block not exceeding 4GB. Assuming the
745    // size() will wrap to negative integer if it exceeds 2GB (From tfile).
746    return (int)(l & 0x00000000ffffffffL);
747  }

749  /**
750   * Returns all files belonging to the given region directory. Could return an
751   * empty list.
752   *
753   * @param fs  The file system reference.
754   * @param regionDir  The region directory to scan.
755   * @return The list of files found.
756   * @throws IOException When scanning the files fails.
757   */
758  static List<Path> getStoreFiles(FileSystem fs, Path regionDir)
759      throws IOException {
760    List<Path> res = new ArrayList<Path>();
761    PathFilter dirFilter = new FSUtils.DirFilter(fs);
762    FileStatus[] familyDirs = fs.listStatus(regionDir, dirFilter);
763    for(FileStatus dir : familyDirs) {
764      FileStatus[] files = fs.listStatus(dir.getPath());
765      for (FileStatus file : files) {
766        if (!file.isDir()) {
767          res.add(file.getPath());
768        }
769      }
770    }
771    return res;
772  }

779  /**
780   * Checks the given {@link HFile} format version, and throws an exception if
781   * invalid. Note that if the version number comes from an input file and has
782   * not been verified, the caller needs to re-throw an {@link IOException} to
783   * indicate that this is not a software error, but corrupted input.
784   *
785   * @param version an HFile version
786   * @throws IllegalArgumentException if the version is invalid
787   */
788  public static void checkFormatVersion(int version)
789      throws IllegalArgumentException {
790    if (version < MIN_FORMAT_VERSION || version > MAX_FORMAT_VERSION) {
791      throw new IllegalArgumentException("Invalid HFile version: " + version
792          + " (expected to be " + "between " + MIN_FORMAT_VERSION + " and "
793          + MAX_FORMAT_VERSION + ")");
794    }
795  }
