017package org.apache.hadoop.hbase.io.encoding;
018
019import java.io.DataInputStream;
020import java.io.DataOutputStream;
021import java.io.IOException;
022import java.nio.ByteBuffer;
023
024import org.apache.hadoop.hbase.KeyValue;
025import org.apache.hadoop.io.RawComparator;
026
027/**
028 * Encoding of KeyValue. It aims to be fast and efficient using assumptions:
029 * <ul>
030 * <li>the KeyValues are stored sorted by key</li>
031 * <li>we know the structure of KeyValue</li>
032 * <li>the values are always iterated forward from beginning of block</li>
033 * <li>knowledge of Key Value format</li>
034 * </ul>
035 * It is designed to work fast enough to be feasible as in memory compression.
036 */
037public interface DataBlockEncoder {
038  /**
039   * Compress KeyValues and write them to output buffer.
040   * @param out Where to write compressed data.
041   * @param in Source of KeyValue for compression.
042   * @param includesMemstoreTS true if including memstore timestamp after every
043   *          key-value pair
044   * @throws IOException If there is an error writing to output stream.
045   */
046  public void compressKeyValues(DataOutputStream out,
047      ByteBuffer in, boolean includesMemstoreTS) throws IOException;
048
049  /**
050   * Uncompress.
051   * @param source Compressed stream of KeyValues.
052   * @param includesMemstoreTS true if including memstore timestamp after every
053   *          key-value pair
054   * @return Uncompressed block of KeyValues.
055   * @throws IOException If there is an error in source.
056   */
057  public ByteBuffer uncompressKeyValues(DataInputStream source,
058      boolean includesMemstoreTS) throws IOException;
059
060  /**
061   * Uncompress.
062   * @param source Compressed stream of KeyValues.
063   * @param allocateHeaderLength allocate this many bytes for the header.
064   * @param skipLastBytes Do not copy n last bytes.
065   * @param includesMemstoreTS true if including memstore timestamp after every
066   *          key-value pair
067   * @return Uncompressed block of KeyValues.
068   * @throws IOException If there is an error in source.
069   */
070  public ByteBuffer uncompressKeyValues(DataInputStream source,
071      int allocateHeaderLength, int skipLastBytes, boolean includesMemstoreTS)
072      throws IOException;
073
074  /**
075   * Return first key in block. Useful for indexing. Typically does not make
076   * a deep copy but returns a buffer wrapping a segment of the actual block's
077   * byte array. This is because the first key in block is usually stored
078   * unencoded.
079   * @param block encoded block we want index, the position will not change
080   * @return First key in block.
081   */
082  public ByteBuffer getFirstKeyInBlock(ByteBuffer block);
083
084  /**
085   * Create a HFileBlock seeker which find KeyValues within a block.
086   * @param comparator what kind of comparison should be used
087   * @param includesMemstoreTS true if including memstore timestamp after every
088   *          key-value pair
089   * @return A newly created seeker.
090   */
091  public EncodedSeeker createSeeker(RawComparator<byte[]> comparator,
092      boolean includesMemstoreTS);
093
094  /**
095   * An interface which enable to seek while underlying data is encoded.
096   *
097   * It works on one HFileBlock, but it is reusable. See
098   * {@link #setCurrentBuffer(ByteBuffer)}.
099   */
100  public static interface EncodedSeeker {
101    /**
102     * Set on which buffer there will be done seeking.
103     * @param buffer Used for seeking.
104     */
105    public void setCurrentBuffer(ByteBuffer buffer);
106
107    /**
108     * Does a deep copy of the key at the current position. A deep copy is
109     * necessary because buffers are reused in the decoder.
110     * @return key at current position
111     */
112    public ByteBuffer getKeyDeepCopy();
113
114    /**
115     * Does a shallow copy of the value at the current position. A shallow
116     * copy is possible because the returned buffer refers to the backing array
117     * of the original encoded buffer.
118     * @return value at current position
119     */
120    public ByteBuffer getValueShallowCopy();
121
122    /** @return key value at current position. */
123    public ByteBuffer getKeyValueBuffer();
124
125    /**
126     * @return the KeyValue object at the current position. Includes memstore
127     *         timestamp.
128     */
129    public KeyValue getKeyValue();
130
131    /** Set position to beginning of given block */
132    public void rewind();
133
134    /**
135     * Move to next position
136     * @return true on success, false if there is no more positions.
137     */
138    public boolean next();
139
140    /**
141     * Moves the seeker position within the current block to:
142     * <ul>
143     * <li>the last key that that is less than or equal to the given key if
144     * <code>seekBefore</code> is false</li>
145     * <li>the last key that is strictly less than the given key if <code>
146     * seekBefore</code> is true. The caller is responsible for loading the
147     * previous block if the requested key turns out to be the first key of the
148     * current block.</li>
149     * </ul>
150     * @param key byte array containing the key
151     * @param offset key position the array
152     * @param length key length in bytes
153     * @param seekBefore find the key strictly less than the given key in case
154     *          of an exact match. Does not matter in case of an inexact match.
155     * @return 0 on exact match, 1 on inexact match.
156     */
157    public int seekToKeyInBlock(byte[] key, int offset, int length,
158        boolean seekBefore);
159
160    /**
161     * Compare the given key against the current key
162     * @param comparator
163     * @param key
164     * @param offset
165     * @param length
166     * @return -1 is the passed key is smaller than the current key, 0 if equal and 1 if greater
167     */
168    public int compareKey(RawComparator<byte []> comparator, byte[] key, int offset, int length);
169  }
170}
