021package org.apache.hadoop.hbase.util;
022
023import java.io.FileNotFoundException;
024import java.io.IOException;
025import java.lang.reflect.Method;
026import java.net.URL;
027import java.util.Map;
028
029import org.apache.hadoop.conf.Configuration;
030import org.apache.hadoop.hbase.HBaseConfiguration;
031import org.apache.hadoop.http.HttpServer;
032import org.mortbay.jetty.handler.ContextHandlerCollection;
033import org.mortbay.jetty.servlet.Context;
034import org.mortbay.jetty.servlet.DefaultServlet;
035
036/**
037 * Create a Jetty embedded server to answer http requests. The primary goal
038 * is to serve up status information for the server.
039 * There are three contexts:
040 *   "/stacks/" -> points to stack trace
041 *   "/static/" -> points to common static files (src/hbase-webapps/static)
042 *   "/" -> the jsp server code from (src/hbase-webapps/<name>)
043 */
044public class InfoServer extends HttpServer {
045  private final Configuration config;
046
047  /**
048   * Create a status server on the given port.
049   * The jsp scripts are taken from src/hbase-webapps/<code>name<code>.
050   * @param name The name of the server
051   * @param bindAddress address to bind to
052   * @param port The port to use on the server
053   * @param findPort whether the server should start at the given port and
054   * increment by 1 until it finds a free port.
055   * @throws IOException e
056   */
057  public InfoServer(String name, String bindAddress, int port, boolean findPort,
058      final Configuration c)
059  throws IOException {
060    super(name, bindAddress, port, findPort, c);
061    this.config = c;
062    fixupLogsServletLocation();
063  }
064
065  /**
066   * Fixup where the logs app points, make it point at hbase logs rather than
067   * hadoop logs.
068   */
069  private void fixupLogsServletLocation() {
070    // Must be same as up in hadoop.
071    final String logsContextPath = "/logs";
072    // Now, put my logs in place of hadoops... disable old one first.
073    Context oldLogsContext = null;
074    for (Map.Entry<Context, Boolean> e : defaultContexts.entrySet()) {
075      if (e.getKey().getContextPath().equals(logsContextPath)) {
076        oldLogsContext = e.getKey();
077        break;
078      }
079    }
080    if (oldLogsContext != null) {
081      this.defaultContexts.put(oldLogsContext, Boolean.FALSE);
082    }
083    // Now do my logs.
084    // Set up the context for "/logs/" if "hbase.log.dir" property is defined.
085    String logDir = System.getProperty("hbase.log.dir");
086    if (logDir != null) {
087      // This is a little presumptious but seems to work.
088      Context logContext =
089        new Context((ContextHandlerCollection)this.webServer.getHandler(),
090          logsContextPath);
091      logContext.setResourceBase(logDir);
092      logContext.addServlet(DefaultServlet.class, "/");
093      HttpServerUtil.constrainHttpMethods(logContext);
094      defaultContexts.put(logContext, true);
095    }
096  }
097
098  /**
099   * Get the pathname to the webapps files.
100   * @param appName eg "secondary" or "datanode"
101   * @return the pathname as a URL
102   * @throws FileNotFoundException if 'webapps' directory cannot be found on CLASSPATH.
103   */
104  protected String getWebAppsPath(String appName) throws FileNotFoundException {
105    // Copied from the super-class.
106    String resourceName = "hbase-webapps/" + appName;
107    URL url = getClass().getClassLoader().getResource(resourceName);
108    if (url == null)
109      throw new FileNotFoundException(resourceName + " not found in CLASSPATH");
110    String urlString = url.toString();
111    return urlString.substring(0, urlString.lastIndexOf('/'));
112  }
113
114  /**
115   * Get the pathname to the <code>path</code> files.
116   * @return the pathname as a URL
117   */
118  protected String getWebAppsPath() throws IOException {
119    // Hack: webapps is not a unique enough element to find in CLASSPATH
120    // We'll more than likely find the hadoop webapps dir.  So, instead
121    // look for the 'master' webapp in the webapps subdir.  That should
122    // get us the hbase context.  Presumption is that place where the
123    // master webapp resides is where we want this InfoServer picking up
124    // web applications.
125    final String master = "master";
126    String p = getWebAppsPath(master);
127    // Now strip master off the end if it is present
128    if(p.endsWith(master)) {
129      return p.substring(0, p.lastIndexOf(master));
130    }
131    return p;
132  }
133}
