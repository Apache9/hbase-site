021package org.apache.hadoop.hbase.util;
022
023import org.apache.hadoop.conf.Configuration;
024import org.apache.hadoop.hbase.HBaseConfiguration;
025import org.apache.hadoop.hbase.filter.Filter;
026import org.apache.hadoop.io.WritableFactories;
027
028/**
029 * Utilities for class manipulation.
030 */
031public class Classes {
032
033  /**
034   * Dynamic class loader to load filter/comparators
035   */
036  private final static ClassLoader CLASS_LOADER;
037
038  static {
039    ClassLoader parent = Classes.class.getClassLoader();
040    Configuration conf = HBaseConfiguration.create();
041    CLASS_LOADER = new DynamicClassLoader(conf, parent);
042  }
043
044  /**
045   * Equivalent of {@link Class#forName(String)} which also returns classes for
046   * primitives like <code>boolean</code>, etc.
047   * 
048   * @param className
049   *          The name of the class to retrieve. Can be either a normal class or
050   *          a primitive class.
051   * @return The class specified by <code>className</code>
052   * @throws ClassNotFoundException
053   *           If the requested class can not be found.
054   */
055  public static Class<?> extendedForName(String className)
056      throws ClassNotFoundException {
057    Class<?> valueType;
058    if (className.equals("boolean")) {
059      valueType = boolean.class;
060    } else if (className.equals("byte")) {
061      valueType = byte.class;
062    } else if (className.equals("short")) {
063      valueType = short.class;
064    } else if (className.equals("int")) {
065      valueType = int.class;
066    } else if (className.equals("long")) {
067      valueType = long.class;
068    } else if (className.equals("float")) {
069      valueType = float.class;
070    } else if (className.equals("double")) {
071      valueType = double.class;
072    } else if (className.equals("char")) {
073      valueType = char.class;
074    } else {
075      valueType = Class.forName(className);
076    }
077    return valueType;
078  }
079
080  @SuppressWarnings("rawtypes")
081  public static String stringify(Class[] classes) {
082    StringBuilder buf = new StringBuilder();
083    if (classes != null) {
084      for (Class c : classes) {
085        if (buf.length() > 0) {
086          buf.append(",");
087        }
088        buf.append(c.getName());
089      }
090    } else {
091      buf.append("NULL");
092    }
093    return buf.toString();
094  }
095
096  /**
097   * Used to dynamically load a filter class.
098   *
099   * @param className the filter class name
100   * @return a filter class
101   * @throws ClassNotFoundException if couldn't find the class
102   */
103  @SuppressWarnings("unchecked")
104  public static Class<? extends Filter> getFilterClassByName(
105      String className) throws ClassNotFoundException {
106    return (Class<? extends Filter>) Class.forName(className, true, CLASS_LOADER);
107  }
108
109  /**
110   * Used to dynamically load a filter class, and create a Writable filter.
111   * This filter class most likely extends Configurable.
112   *
113   * @param className the filter class name.
114   * @return a filter
115   */
116  public static Filter createWritableForName(String className) {
117    try {
118      Class<? extends Filter> clazz = getFilterClassByName(className);
119      return (Filter)WritableFactories.newInstance(clazz, new Configuration());
120    } catch (ClassNotFoundException e) {
121      throw new RuntimeException("Can't find class " + className);
122    }
123  }
124
125  /**
126   * This method is almost the same as #createWritableForName, except
127   * that this one doesn't expect the filter class to extends Configurable.
128   *
129   * @param className the filter class name.
130   * @return a filter
131   */
132  public static Filter createForName(String className) {
133    try {
134      Class<? extends Filter> clazz = getFilterClassByName(className);
135      return (Filter)clazz.newInstance();
136    } catch (ClassNotFoundException e) {
137      throw new RuntimeException("Can't find class " + className);
138    } catch (InstantiationException e) {
139      throw new RuntimeException("Couldn't instantiate " + className, e);
140    } catch (IllegalAccessException e) {
141      throw new RuntimeException("No access to " + className, e);
142    }
143  }
144}
