021package org.apache.hadoop.hbase.coprocessor;
022
023import java.io.IOException;
024import java.util.List;
025
026import org.apache.hadoop.hbase.client.Scan;
027import org.apache.hadoop.hbase.client.coprocessor.AggregationClient;
028import org.apache.hadoop.hbase.ipc.CoprocessorProtocol;
029import org.apache.hadoop.hbase.util.Pair;
030
031/**
032 * Defines the aggregation functions that are to be supported in this
033 * Coprocessor. For each method, it takes a Scan object and a columnInterpreter.
034 * The scan object should have a column family (else an exception will be
035 * thrown), and an optional column qualifier. In the current implementation
036 * {@link AggregateImplementation}, only one column family and column qualifier
037 * combination is served. In case there are more than one, only first one will
038 * be picked. Refer to {@link AggregationClient} for some general conditions on
039 * input parameters.
040 */
041public interface AggregateProtocol extends CoprocessorProtocol {
042  public static final long VERSION = 1L;
043
044  /**
045   * Gives the maximum for a given combination of column qualifier and column
046   * family, in the given row range as defined in the Scan object. In its
047   * current implementation, it takes one column family and one column qualifier
048   * (if provided). In case of null column qualifier, maximum value for the
049   * entire column family will be returned.
050   * @param ci
051   * @param scan
052   * @return max value as mentioned above
053   * @throws IOException
054   */
055  <T, S> T getMax(ColumnInterpreter<T, S> ci, Scan scan) throws IOException;
056
057  /**
058   * Gives the minimum for a given combination of column qualifier and column
059   * family, in the given row range as defined in the Scan object. In its
060   * current implementation, it takes one column family and one column qualifier
061   * (if provided). In case of null column qualifier, minimum value for the
062   * entire column family will be returned.
063   * @param ci
064   * @param scan
065   * @return min as mentioned above
066   * @throws IOException
067   */
068  <T, S> T getMin(ColumnInterpreter<T, S> ci, Scan scan) throws IOException;
069
070  /**
071   * Gives the sum for a given combination of column qualifier and column
072   * family, in the given row range as defined in the Scan object. In its
073   * current implementation, it takes one column family and one column qualifier
074   * (if provided). In case of null column qualifier, sum for the entire column
075   * family will be returned.
076   * @param ci
077   * @param scan
078   * @return sum of values as defined by the column interpreter
079   * @throws IOException
080   */
081  <T, S> S getSum(ColumnInterpreter<T, S> ci, Scan scan) throws IOException;
082
083  /**
084   * @param ci
085   * @param scan
086   * @return Row count for the given column family and column qualifier, in
087   * the given row range as defined in the Scan object.
088   * @throws IOException
089   */
090  <T, S> long getRowNum(ColumnInterpreter<T, S> ci, Scan scan)
091      throws IOException;
092
093  /**
094   * Gives a Pair with first object as Sum and second object as row count,
095   * computed for a given combination of column qualifier and column family in
096   * the given row range as defined in the Scan object. In its current
097   * implementation, it takes one column family and one column qualifier (if
098   * provided). In case of null column qualifier, an aggregate sum over all the
099   * entire column family will be returned.
100   * <p>
101   * The average is computed in
102   * {@link AggregationClient#avg(byte[], ColumnInterpreter, Scan)} by
103   * processing results from all regions, so its "ok" to pass sum and a Long
104   * type.
105   * @param ci
106   * @param scan
107   * @return Average
108   * @throws IOException
109   */
110  <T, S> Pair<S, Long> getAvg(ColumnInterpreter<T, S> ci, Scan scan)
111      throws IOException;
112
113  /**
114   * Gives a Pair with first object a List containing Sum and sum of squares,
115   * and the second object as row count. It is computed for a given combination of
116   * column qualifier and column family in the given row range as defined in the
117   * Scan object. In its current implementation, it takes one column family and
118   * one column qualifier (if provided). The idea is get the value of variance first:
119   * the average of the squares less the square of the average a standard
120   * deviation is square root of variance.
121   * @param ci
122   * @param scan
123   * @return STD
124   * @throws IOException
125   */
126  <T, S> Pair<List<S>, Long> getStd(ColumnInterpreter<T, S> ci, Scan scan)
127      throws IOException;
128
129  /**
130   * Gives a List containing sum of values and sum of weights.
131   * It is computed for the combination of column
132   * family and column qualifier(s) in the given row range as defined in the
133   * Scan object. In its current implementation, it takes one column family and
134   * two column qualifiers. The first qualifier is for values column and 
135   * the second qualifier (optional) is for weight column.
136   * @param ci
137   * @param scan
138   * @return Pair
139   * @throws IOException
140   */
141  <T, S> List<S> getMedian(ColumnInterpreter<T, S> ci, Scan scan)
142      throws IOException;
143
144}
