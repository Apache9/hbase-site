021package org.apache.hadoop.hbase.coprocessor;
022
023import org.apache.hadoop.hbase.CoprocessorEnvironment;
024
025/**
026 * Carries the execution state for a given invocation of an Observer coprocessor
027 * ({@link RegionObserver}, {@link MasterObserver}, or {@link WALObserver})
028 * method.  The same ObserverContext instance is passed sequentially to all loaded
029 * coprocessors for a given Observer method trigger, with the
030 * <code>CoprocessorEnvironment</code> reference swapped out for each
031 * coprocessor.
032 * @param <E> The {@link CoprocessorEnvironment} subclass applicable to the
033 *     revelant Observer interface.
034 */
035public class ObserverContext<E extends CoprocessorEnvironment> {
036  private E env;
037  private boolean bypass;
038  private boolean complete;
039
040  public ObserverContext() {
041  }
042
043  public E getEnvironment() {
044    return env;
045  }
046
047  public void prepare(E env) {
048    this.env = env;
049  }
050
051  /**
052   * Call to indicate that the current coprocessor's return value should be
053   * used in place of the normal HBase obtained value.
054   */
055  public void bypass() {
056    bypass = true;
057  }
058
059  /**
060   * Call to indicate that additional coprocessors further down the execution
061   * chain do not need to be invoked.  Implies that this coprocessor's response
062   * is definitive.
063   */
064  public void complete() {
065    complete = true;
066  }
067
068  /**
069   * For use by the coprocessor framework.
070   * @return <code>true</code> if {@link ObserverContext#bypass()}
071   *     was called by one of the loaded coprocessors, <code>false</code> otherwise.
072   */
073  public boolean shouldBypass() {
074    boolean current = bypass;
075    bypass = false;
076    return current;
077  }
078
079  /**
080   * For use by the coprocessor framework.
081   * @return <code>true</code> if {@link ObserverContext#complete()}
082   *     was called by one of the loaded coprocessors, <code>false</code> otherwise.
083   */
084  public boolean shouldComplete() {
085    boolean current = complete;
086    complete = false;
087    return current;
088  }
089
090  /**
091   * Instantiates a new ObserverContext instance if the passed reference is
092   * <code>null</code> and sets the environment in the new or existing instance.
093   * This allows deferring the instantiation of a ObserverContext until it is
094   * actually needed.
095   *
096   * @param env The coprocessor environment to set
097   * @param context An existing ObserverContext instance to use, or <code>null</code>
098   *     to create a new instance
099   * @param <T> The environment type for the context
100   * @return An instance of <code>ObserverContext</code> with the environment set
101   */
102  public static <T extends CoprocessorEnvironment> ObserverContext<T> createAndPrepare(
103      T env, ObserverContext<T> context) {
104    if (context == null) {
105      context = new ObserverContext<T>();
106    }
107    context.prepare(env);
108    return context;
109  }
110}
