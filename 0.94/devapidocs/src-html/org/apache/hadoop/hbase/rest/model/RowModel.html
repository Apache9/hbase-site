021package org.apache.hadoop.hbase.rest.model;
022
023import java.io.IOException;
024import java.io.Serializable;
025import java.util.ArrayList;
026import java.util.List;
027
028import javax.xml.bind.annotation.XmlAttribute;
029import javax.xml.bind.annotation.XmlElement;
030import javax.xml.bind.annotation.XmlRootElement;
031
032import org.apache.hadoop.hbase.rest.ProtobufMessageHandler;
033
034/**
035 * Representation of a row. A row is a related set of cells, grouped by common
036 * row key. RowModels do not appear in results by themselves. They are always
037 * encapsulated within CellSetModels.
038 * 
039 * <pre>
040 * &lt;complexType name="Row"&gt;
041 *   &lt;sequence&gt;
042 *     &lt;element name="key" type="base64Binary"&gt;&lt;/element&gt;
043 *     &lt;element name="cell" type="tns:Cell" 
044 *       maxOccurs="unbounded" minOccurs="1"&gt;&lt;/element&gt;
045 *   &lt;/sequence&gt;
046 * &lt;/complexType&gt;
047 * </pre>
048 */
049@XmlRootElement(name="Row")
050public class RowModel implements ProtobufMessageHandler, Serializable {
051  private static final long serialVersionUID = 1L;
052
053  private byte[] key;
054  private List<CellModel> cells = new ArrayList<CellModel>();
055
056  /**
057   * Default constructor
058   */
059  public RowModel() { }
060
061  /**
062   * Constructor
063   * @param key the row key
064   */
065  public RowModel(final String key) {
066    this(key.getBytes());
067  }
068  
069  /**
070   * Constructor
071   * @param key the row key
072   */
073  public RowModel(final byte[] key) {
074    this.key = key;
075    cells = new ArrayList<CellModel>();
076  }
077
078  /**
079   * Constructor
080   * @param key the row key
081   * @param cells the cells
082   */
083  public RowModel(final String key, final List<CellModel> cells) {
084    this(key.getBytes(), cells);
085  }
086  
087  /**
088   * Constructor
089   * @param key the row key
090   * @param cells the cells
091   */
092  public RowModel(final byte[] key, final List<CellModel> cells) {
093    this.key = key;
094    this.cells = cells;
095  }
096  
097  /**
098   * Adds a cell to the list of cells for this row
099   * @param cell the cell
100   */
101  public void addCell(CellModel cell) {
102    cells.add(cell);
103  }
104
105  /**
106   * @return the row key
107   */
108  @XmlAttribute
109  public byte[] getKey() {
110    return key;
111  }
112
113  /**
114   * @param key the row key
115   */
116  public void setKey(byte[] key) {
117    this.key = key;
118  }
119
120  /**
121   * @return the cells
122   */
123  @XmlElement(name="Cell")
124  public List<CellModel> getCells() {
125    return cells;
126  }
127
128  @Override
129  public byte[] createProtobufOutput() {
130    // there is no standalone row protobuf message
131    throw new UnsupportedOperationException(
132        "no protobuf equivalent to RowModel");
133  }
134
135  @Override
136  public ProtobufMessageHandler getObjectFromMessage(byte[] message)
137      throws IOException {
138    // there is no standalone row protobuf message
139    throw new UnsupportedOperationException(
140        "no protobuf equivalent to RowModel");
141  }
142}
