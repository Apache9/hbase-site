019package org.apache.hadoop.hbase.regionserver.wal;
020
021import java.util.HashMap;
022
023import org.apache.hadoop.hbase.util.Bytes;
024
025import com.google.common.base.Preconditions;
026
027/**
028 * WALDictionary using an LRU eviction algorithm. Uses a linked list running
029 * through a hashtable.  Currently has max of 2^15 entries.  Will start
030 * evicting if exceeds this number  The maximum memory we expect this dictionary
031 * to take in the worst case is about:
032 * <code>(2 ^ 15) * 5 (Regionname, Row key, CF, Column qual, table) * 100 bytes (these are some big names) = ~16MB</code>.
033 * If you want to get silly, even at 1kb entries, it maxes out at 160 megabytes.
034 */
035public class LRUDictionary implements Dictionary {
036  private final BidirectionalLRUMap backingStore = new BidirectionalLRUMap();
037
038  @Override
039  public byte[] getEntry(short idx) {
040    return backingStore.get(idx);
041  }
042
043  @Override
044  public short findEntry(byte[] data, int offset, int length) {
045    short ret = backingStore.findIdx(data, offset, length);
046    if (ret == NOT_IN_DICTIONARY) {
047      addEntry(data, offset, length);
048    }
049    return ret;
050  }
051
052  @Override
053  public short addEntry(byte[] data, int offset, int length) {
054    if (length <= 0) return NOT_IN_DICTIONARY;
055    return backingStore.put(data, offset, length);
056  }
057
058  @Override
059  public void clear() {
060    backingStore.clear();
061  }
062
063  /*
064   * Internal class used to implement LRU eviction and dual lookup (by key and
065   * value).
066   * 
067   * This is not thread safe. Don't use in multi-threaded applications.
068   */
069  static class BidirectionalLRUMap {
070    static final int MAX_SIZE = Short.MAX_VALUE;
071    private int currSize = 0;
072
073    // Head and tail of the LRU list.
074    private Node head;
075    private Node tail;
076
077    private HashMap<Node, Short> nodeToIndex = new HashMap<Node, Short>();
078    private Node[] indexToNode = new Node[MAX_SIZE];
079
080    public BidirectionalLRUMap() {
081      for (int i = 0; i < MAX_SIZE; i++) {
082        indexToNode[i] = new Node();
083      }
084    }
085
086    private short put(byte[] array, int offset, int length) {
087      // We copy the bytes we want, otherwise we might be holding references to
088      // massive arrays in our dictionary (or those arrays might change)
089      byte[] stored = new byte[length];
090      Bytes.putBytes(stored, 0, array, offset, length);
091
092      if (currSize < MAX_SIZE) {
093        // There is space to add without evicting.
094        indexToNode[currSize].setContents(stored, 0, stored.length);
095        setHead(indexToNode[currSize]);
096        short ret = (short) currSize++;
097        nodeToIndex.put(indexToNode[ret], ret);
098        return ret;
099      } else {
100        short s = nodeToIndex.remove(tail);
101        tail.setContents(stored, 0, stored.length);
102        // we need to rehash this.
103        nodeToIndex.put(tail, s);
104        moveToHead(tail);
105        return s;
106      }
107    }
108
109    private short findIdx(byte[] array, int offset, int length) {
110      Short s;
111      final Node comparisonNode = new Node();
112      comparisonNode.setContents(array, offset, length);
113      if ((s = nodeToIndex.get(comparisonNode)) != null) {
114        moveToHead(indexToNode[s]);
115        return s;
116      } else {
117        return -1;
118      }
119    }
120
121    private byte[] get(short idx) {
122      Preconditions.checkElementIndex(idx, currSize);
123      moveToHead(indexToNode[idx]);
124      return indexToNode[idx].container;
125    }
126
127    private void moveToHead(Node n) {
128      if (head == n) {
129        // no-op -- it's already the head.
130        return;
131      }
132      // At this point we definitely have prev, since it's not the head.
133      assert n.prev != null;
134      // Unlink prev.
135      n.prev.next = n.next;
136
137      // Unlink next
138      if (n.next != null) {
139        n.next.prev = n.prev;
140      } else {
141        assert n == tail;
142        tail = n.prev;
143      }
144      // Node is now removed from the list. Re-add it at the head.
145      setHead(n);
146    }
147    
148    private void setHead(Node n) {
149      // assume it's already unlinked from the list at this point.
150      n.prev = null;
151      n.next = head;
152      if (head != null) {
153        assert head.prev == null;
154        head.prev = n;
155      }
156
157      head = n;
158
159      // First entry
160      if (tail == null) {
161        tail = n;
162      }
163    }
164
165    private void clear() {
166      currSize = 0;
167      nodeToIndex.clear();
168      tail = null;
169      head = null;
170
171      for (Node n : indexToNode) {
172        n.container = null;
173      }
174
175      for (int i = 0; i < MAX_SIZE; i++) {
176        indexToNode[i].next = null;
177        indexToNode[i].prev = null;
178      }
179    }
180
181    private static class Node {
182      byte[] container;
183      int offset;
184      int length;
185      Node next; // link towards the tail
186      Node prev; // link towards the head
187
188      public Node() {
189      }
190
191      private void setContents(byte[] container, int offset, int length) {
192        this.container = container;
193        this.offset = offset;
194        this.length = length;
195      }
196
197      @Override
198      public int hashCode() {
199        return Bytes.hashCode(container, offset, length);
200      }
201
202      @Override
203      public boolean equals(Object other) {
204        if (!(other instanceof Node)) {
205          return false;
206        }
207
208        Node casted = (Node) other;
209        return Bytes.equals(container, offset, length, casted.container,
210            casted.offset, casted.length);
211      }
212    }
213  }
214}
