021package org.apache.hadoop.hbase.regionserver;
022
023import org.apache.hadoop.hbase.KeyValue;
024import org.apache.hadoop.hbase.util.Bytes;
025
026/**
027 * This class is responsible for the tracking and enforcement of Deletes
028 * during the course of a Scan operation.
029 *
030 * It only has to enforce Delete and DeleteColumn, since the
031 * DeleteFamily is handled at a higher level.
032 *
033 * <p>
034 * This class is utilized through three methods:
035 * <ul><li>{@link #add} when encountering a Delete or DeleteColumn
036 * <li>{@link #isDeleted} when checking if a Put KeyValue has been deleted
037 * <li>{@link #update} when reaching the end of a StoreFile or row for scans
038 * <p>
039 * This class is NOT thread-safe as queries are never multi-threaded
040 */
041public class ScanDeleteTracker implements DeleteTracker {
042
043  private boolean hasFamilyStamp = false;
044  private long familyStamp = 0L;
045  private byte [] deleteBuffer = null;
046  private int deleteOffset = 0;
047  private int deleteLength = 0;
048  private byte deleteType = 0;
049  private long deleteTimestamp = 0L;
050
051  /**
052   * Constructor for ScanDeleteTracker
053   */
054  public ScanDeleteTracker() {
055    super();
056  }
057
058  /**
059   * Add the specified KeyValue to the list of deletes to check against for
060   * this row operation.
061   * <p>
062   * This is called when a Delete is encountered in a StoreFile.
063   * @param buffer KeyValue buffer
064   * @param qualifierOffset column qualifier offset
065   * @param qualifierLength column qualifier length
066   * @param timestamp timestamp
067   * @param type delete type as byte
068   */
069  @Override
070  public void add(byte[] buffer, int qualifierOffset, int qualifierLength,
071      long timestamp, byte type) {
072    if (!hasFamilyStamp || timestamp > familyStamp) {
073      if (type == KeyValue.Type.DeleteFamily.getCode()) {
074        hasFamilyStamp = true;
075        familyStamp = timestamp;
076        return;
077      }
078
079      if (deleteBuffer != null && type < deleteType) {
080        // same column, so ignore less specific delete
081        if (Bytes.equals(deleteBuffer, deleteOffset, deleteLength,
082            buffer, qualifierOffset, qualifierLength)){
083          return;
084        }
085      }
086      // new column, or more general delete type
087      deleteBuffer = buffer;
088      deleteOffset = qualifierOffset;
089      deleteLength = qualifierLength;
090      deleteType = type;
091      deleteTimestamp = timestamp;
092    }
093    // missing else is never called.
094  }
095
096  /**
097   * Check if the specified KeyValue buffer has been deleted by a previously
098   * seen delete.
099   *
100   * @param buffer KeyValue buffer
101   * @param qualifierOffset column qualifier offset
102   * @param qualifierLength column qualifier length
103   * @param timestamp timestamp
104   * @return deleteResult
105   */
106  @Override
107  public DeleteResult isDeleted(byte [] buffer, int qualifierOffset,
108      int qualifierLength, long timestamp) {
109    if (hasFamilyStamp && timestamp <= familyStamp) {
110      return DeleteResult.FAMILY_DELETED;
111    }
112
113    if (deleteBuffer != null) {
114      int ret = Bytes.compareTo(deleteBuffer, deleteOffset, deleteLength,
115          buffer, qualifierOffset, qualifierLength);
116
117      if (ret == 0) {
118        if (deleteType == KeyValue.Type.DeleteColumn.getCode()) {
119          return DeleteResult.COLUMN_DELETED;
120        }
121        // Delete (aka DeleteVersion)
122        // If the timestamp is the same, keep this one
123        if (timestamp == deleteTimestamp) {
124          return DeleteResult.VERSION_DELETED;
125        }
126        // use assert or not?
127        assert timestamp < deleteTimestamp;
128
129        // different timestamp, let's clear the buffer.
130        deleteBuffer = null;
131      } else if(ret < 0){
132        // Next column case.
133        deleteBuffer = null;
134      } else {
135        throw new IllegalStateException("isDelete failed: deleteBuffer="
136            + Bytes.toStringBinary(deleteBuffer, deleteOffset, deleteLength)
137            + ", qualifier="
138            + Bytes.toStringBinary(buffer, qualifierOffset, qualifierLength)
139            + ", timestamp=" + timestamp + ", comparison result: " + ret);
140      }
141    }
142
143    return DeleteResult.NOT_DELETED;
144  }
145
146  @Override
147  public boolean isEmpty() {
148    return deleteBuffer == null && !hasFamilyStamp;
149  }
150
151  @Override
152  // called between every row.
153  public void reset() {
154    hasFamilyStamp = false;
155    familyStamp = 0L;
156    deleteBuffer = null;
157  }
158
159  @Override
160  // should not be called at all even (!)
161  public void update() {
162    this.reset();
163  }
164}
