020package org.apache.hadoop.hbase.regionserver;
021
022/**
023 * This interface is used for the tracking and enforcement of Deletes
024 * during the course of a Get or Scan operation.
025 * <p>
026 * This class is utilized through three methods:
027 * <ul><li>{@link #add} when encountering a Delete
028 * <li>{@link #isDeleted} when checking if a Put KeyValue has been deleted
029 * <li>{@link #update} when reaching the end of a StoreFile
030 */
031public interface DeleteTracker {
032
033  /**
034   * Add the specified KeyValue to the list of deletes to check against for
035   * this row operation.
036   * <p>
037   * This is called when a Delete is encountered in a StoreFile.
038   * @param buffer KeyValue buffer
039   * @param qualifierOffset column qualifier offset
040   * @param qualifierLength column qualifier length
041   * @param timestamp timestamp
042   * @param type delete type as byte
043   */
044  public void add(byte [] buffer, int qualifierOffset, int qualifierLength,
045      long timestamp, byte type);
046
047  /**
048   * Check if the specified KeyValue buffer has been deleted by a previously
049   * seen delete.
050   * @param buffer KeyValue buffer
051   * @param qualifierOffset column qualifier offset
052   * @param qualifierLength column qualifier length
053   * @param timestamp timestamp
054   * @return deleteResult The result tells whether the KeyValue is deleted and why
055   */
056  public DeleteResult isDeleted(byte [] buffer, int qualifierOffset,
057      int qualifierLength, long timestamp);
058
059  /**
060   * @return true if there are no current delete, false otherwise
061   */
062  public boolean isEmpty();
063
064  /**
065   * Called at the end of every StoreFile.
066   * <p>
067   * Many optimized implementations of Trackers will require an update at
068   * when the end of each StoreFile is reached.
069   */
070  public void update();
071
072  /**
073   * Called between rows.
074   * <p>
075   * This clears everything as if a new DeleteTracker was instantiated.
076   */
077  public void reset();
078
079
080  /**
081   * Return codes for comparison of two Deletes.
082   * <p>
083   * The codes tell the merging function what to do.
084   * <p>
085   * INCLUDE means add the specified Delete to the merged list.
086   * NEXT means move to the next element in the specified list(s).
087   */
088  enum DeleteCompare {
089    INCLUDE_OLD_NEXT_OLD,
090    INCLUDE_OLD_NEXT_BOTH,
091    INCLUDE_NEW_NEXT_NEW,
092    INCLUDE_NEW_NEXT_BOTH,
093    NEXT_OLD,
094    NEXT_NEW
095  }
096
097  /**
098   * Returns codes for delete result.
099   * The codes tell the ScanQueryMatcher whether the kv is deleted and why.
100   * Based on the delete result, the ScanQueryMatcher will decide the next
101   * operation
102   */
103  public static enum DeleteResult {
104    FAMILY_DELETED, // The KeyValue is deleted by a delete family.
105    COLUMN_DELETED, // The KeyValue is deleted by a delete column.
106    VERSION_DELETED, // The KeyValue is deleted by a version delete.
107    NOT_DELETED
108  }
109
110}
