020package org.apache.hadoop.hbase;
021
022import java.util.Map.Entry;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026import org.apache.hadoop.conf.Configuration;
027import org.apache.hadoop.hbase.util.VersionInfo;
028
029/**
030 * Adds HBase configuration files to a Configuration
031 */
032public class HBaseConfiguration extends Configuration {
033
034  private static final Log LOG = LogFactory.getLog(HBaseConfiguration.class);
035
036  // a constant to convert a fraction to a percentage
037  private static final int CONVERT_TO_PERCENTAGE = 100;
038
039  /**
040   * Instantinating HBaseConfiguration() is deprecated. Please use
041   * HBaseConfiguration#create() to construct a plain Configuration
042   */
043  @Deprecated
044  public HBaseConfiguration() {
045    //TODO:replace with private constructor, HBaseConfiguration should not extend Configuration
046    super();
047    addHbaseResources(this);
048    LOG.warn("instantiating HBaseConfiguration() is deprecated. Please use" +
049        " HBaseConfiguration#create() to construct a plain Configuration");
050  }
051
052  /**
053   * Instantiating HBaseConfiguration() is deprecated. Please use
054   * HBaseConfiguration#create(conf) to construct a plain Configuration
055   */
056  @Deprecated
057  public HBaseConfiguration(final Configuration c) {
058    //TODO:replace with private constructor
059    this();
060    merge(this, c);
061  }
062
063  private static void checkDefaultsVersion(Configuration conf) {
064    if (conf.getBoolean("hbase.defaults.for.version.skip", Boolean.FALSE)) return;
065    String defaultsVersion = conf.get("hbase.defaults.for.version");
066    String thisVersion = VersionInfo.getVersion();
067    if (!thisVersion.equals(defaultsVersion)) {
068      throw new RuntimeException(
069        "hbase-default.xml file seems to be for and old version of HBase (" +
070        defaultsVersion + "), this version is " + thisVersion);
071    }
072  }
073
074  private static void checkForClusterFreeMemoryLimit(Configuration conf) {
075      float globalMemstoreLimit = conf.getFloat("hbase.regionserver.global.memstore.upperLimit", 0.4f);
076      int gml = (int)(globalMemstoreLimit * CONVERT_TO_PERCENTAGE);
077      float blockCacheUpperLimit =
078        conf.getFloat(HConstants.HFILE_BLOCK_CACHE_SIZE_KEY,
079          HConstants.HFILE_BLOCK_CACHE_SIZE_DEFAULT);
080      int bcul = (int)(blockCacheUpperLimit * CONVERT_TO_PERCENTAGE);
081      if (CONVERT_TO_PERCENTAGE - (gml + bcul)
082              < (int)(CONVERT_TO_PERCENTAGE * 
083                      HConstants.HBASE_CLUSTER_MINIMUM_MEMORY_THRESHOLD)) {
084          throw new RuntimeException(
085            "Current heap configuration for MemStore and BlockCache exceeds " +
086            "the threshold required for successful cluster operation. " +
087            "The combined value cannot exceed 0.8. Please check " +
088            "the settings for hbase.regionserver.global.memstore.upperLimit and " +
089            "hfile.block.cache.size in your configuration. " +
090            "hbase.regionserver.global.memstore.upperLimit is " + 
091            globalMemstoreLimit +
092            " hfile.block.cache.size is " + blockCacheUpperLimit);
093      }
094  }
095
096  public static Configuration addHbaseResources(Configuration conf) {
097    conf.addResource("hbase-default.xml");
098    conf.addResource("hbase-site.xml");
099
100    checkDefaultsVersion(conf);
101    checkForClusterFreeMemoryLimit(conf);
102    return conf;
103  }
104
105  /**
106   * Creates a Configuration with HBase resources
107   * @return a Configuration with HBase resources
108   */
109  public static Configuration create() {
110    Configuration conf = new Configuration();
111    return addHbaseResources(conf);
112  }
113
114  /**
115   * @param that Configuration to clone.
116   * @return a Configuration created with the hbase-*.xml files plus
117   * the given configuration.
118   */
119  public static Configuration create(final Configuration that) {
120    Configuration conf = create();
121    merge(conf, that);
122    return conf;
123  }
124
125  /**
126   * Merge two configurations.
127   * @param destConf the configuration that will be overwritten with items
128   *                 from the srcConf
129   * @param srcConf the source configuration
130   **/
131  public static void merge(Configuration destConf, Configuration srcConf) {
132    for (Entry<String, String> e : srcConf) {
133      destConf.set(e.getKey(), e.getValue());
134    }
135  }
136  
137  /**
138   * 
139   * @return whether to show HBase Configuration in servlet
140   */
141  public static boolean isShowConfInServlet() {
142    boolean isShowConf = false;
143    try {
144      if (Class.forName("org.apache.hadoop.conf.ConfServlet") != null) {
145        isShowConf = true;  
146      }
147    } catch (Exception e) {
148      
149    }
150    return isShowConf;
151  }
152
153  /** For debugging.  Dump configurations to system output as xml format.
154   * Master and RS configurations can also be dumped using
155   * http services. e.g. "curl http://master:60010/dump"
156   */
157  public static void main(String[] args) throws Exception {
158    HBaseConfiguration.create().writeXml(System.out);
159  }
160}
