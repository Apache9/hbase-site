018package org.apache.hadoop.hbase.snapshot;
019
020import java.io.IOException;
021import java.util.NavigableSet;
022
023import org.apache.commons.logging.Log;
024import org.apache.commons.logging.LogFactory;
025import org.apache.hadoop.classification.InterfaceAudience;
026import org.apache.hadoop.classification.InterfaceStability;
027import org.apache.hadoop.fs.FileStatus;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.fs.FileUtil;
030import org.apache.hadoop.fs.Path;
031import org.apache.hadoop.hbase.errorhandling.ForeignExceptionDispatcher;
032import org.apache.hadoop.hbase.protobuf.generated.HBaseProtos.SnapshotDescription;
033import org.apache.hadoop.hbase.regionserver.wal.HLog;
034
035/**
036 * Copy over each of the files in a region's recovered.edits directory to the region's snapshot
037 * directory.
038 * <p>
039 * This is a serial operation over each of the files in the recovered.edits directory and also
040 * streams all the bytes to the client and then back to the filesystem, so the files being copied
041 * should be <b>small</b> or it will (a) suck up a lot of bandwidth, and (b) take a long time.
042 */
043@InterfaceAudience.Private
044@InterfaceStability.Evolving
045public class CopyRecoveredEditsTask extends SnapshotTask {
046
047  private static final Log LOG = LogFactory.getLog(CopyRecoveredEditsTask.class);
048  private final FileSystem fs;
049  private final Path regiondir;
050  private final Path outputDir;
051
052  /**
053   * @param snapshot Snapshot being taken
054   * @param monitor error monitor for the snapshot
055   * @param fs {@link FileSystem} where the snapshot is being taken
056   * @param regionDir directory for the region to examine for edits
057   * @param snapshotRegionDir directory for the region in the snapshot
058   */
059  public CopyRecoveredEditsTask(SnapshotDescription snapshot, ForeignExceptionDispatcher monitor,
060      FileSystem fs, Path regionDir, Path snapshotRegionDir) {
061    super(snapshot, monitor);
062    this.fs = fs;
063    this.regiondir = regionDir;
064    this.outputDir = HLog.getRegionDirRecoveredEditsDir(snapshotRegionDir);
065  }
066
067  @Override
068  public Void call() throws IOException {
069    NavigableSet<Path> files = HLog.getSplitEditFilesSorted(this.fs, regiondir);
070    if (files == null || files.size() == 0) return null;
071
072    // copy over each file.
073    // this is really inefficient (could be trivially parallelized), but is
074    // really simple to reason about.
075    for (Path source : files) {
076      // check to see if the file is zero length, in which case we can skip it
077      FileStatus stat = fs.getFileStatus(source);
078      if (stat.getLen() <= 0) continue;
079
080      // its not zero length, so copy over the file
081      Path out = new Path(outputDir, source.getName());
082      LOG.debug("Copying " + source + " to " + out);
083      FileUtil.copy(fs, source, fs, out, true, fs.getConf());
084
085      // check for errors to the running operation after each file
086      this.rethrowException();
087    }
088    return null;
089  }
090}
