018package org.apache.hadoop.hbase.metrics;
019
020import org.apache.commons.logging.Log;
021import org.apache.commons.logging.LogFactory;
022import org.apache.hadoop.metrics.MetricsRecord;
023import org.apache.hadoop.metrics.util.MetricsRegistry;
024import org.apache.hadoop.metrics.util.MetricsTimeVaryingRate;
025import org.apache.hadoop.util.StringUtils;
026
027/**
028 * This class extends MetricsTimeVaryingRate to let the metrics
029 * persist past a pushMetric() call
030 */
031public class PersistentMetricsTimeVaryingRate extends MetricsTimeVaryingRate {
032  protected static final Log LOG =
033    LogFactory.getLog("org.apache.hadoop.hbase.metrics");
034
035  protected boolean reset = false;
036  protected long lastOper = 0;
037  protected long totalOps = 0;
038
039  /**
040   * Constructor - create a new metric
041   * @param nam the name of the metrics to be used to publish the metric
042   * @param registry - where the metrics object will be registered
043   * @param description metrics description
044   */
045  public PersistentMetricsTimeVaryingRate(final String nam, 
046      final MetricsRegistry registry, 
047      final String description) {
048    super(nam, registry, description);
049  }
050
051  /**
052   * Constructor - create a new metric
053   * @param nam the name of the metrics to be used to publish the metric
054   * @param registry - where the metrics object will be registered
055   */
056  public PersistentMetricsTimeVaryingRate(final String nam, 
057      MetricsRegistry registry) {
058    this(nam, registry, NO_DESCRIPTION);
059  }
060
061  /**
062   * Push updated metrics to the mr.
063   * 
064   * Note this does NOT push to JMX
065   * (JMX gets the info via {@link #getPreviousIntervalAverageTime()} and
066   * {@link #getPreviousIntervalNumOps()}
067   *
068   * @param mr owner of this metric
069   */
070  @Override
071  public synchronized void pushMetric(final MetricsRecord mr) {
072    // this will reset the currentInterval & num_ops += prevInterval()
073    super.pushMetric(mr);
074    // since we're retaining prevInterval(), we don't want to do the incr
075    // instead, we want to set that value because we have absolute ops
076    try {
077      mr.setMetric(getName() + "_num_ops", totalOps);
078    } catch (Exception e) {
079      LOG.info("pushMetric failed for " + getName() + "\n" +
080          StringUtils.stringifyException(e));
081    }
082    if (reset) {
083      // use the previous avg as our starting min/max/avg
084      super.inc(getPreviousIntervalAverageTime());
085      reset = false;
086    } else {
087      // maintain the stats that pushMetric() cleared
088      maintainStats();
089    }
090  }
091  
092  /**
093   * Increment the metrics for numOps operations
094   * @param numOps - number of operations
095   * @param time - time for numOps operations
096   */
097  @Override
098  public synchronized void inc(final int numOps, final long time) {
099    super.inc(numOps, time);
100    totalOps += numOps;
101  }
102  
103  /**
104   * Increment the metrics for numOps operations
105   * @param time - time for numOps operations
106   */
107  @Override
108  public synchronized void inc(final long time) {
109    super.inc(time);
110    ++totalOps;
111  }
112  
113  /**
114   * Rollover to a new interval
115   * NOTE: does not reset numOps.  this is an absolute value
116   */
117  public synchronized void resetMinMaxAvg() {
118    reset = true;
119  }
120
121  /* MetricsTimeVaryingRate will reset every time pushMetric() is called
122   * This is annoying for long-running stats that might not get a single 
123   * operation in the polling period.  This function ensures that values
124   * for those stat entries don't get reset.
125   */
126  protected void maintainStats() {
127    int curOps = this.getPreviousIntervalNumOps();
128    if (curOps > 0) {
129      long curTime = this.getPreviousIntervalAverageTime();
130      long totalTime = curTime * curOps;
131      if (curTime == 0 || totalTime / curTime == curOps) {
132        super.inc(curOps, totalTime);
133      } else {
134        LOG.info("Stats for " + this.getName() + " overflowed! resetting");
135      }
136    }
137  }
138}
