020package org.apache.hadoop.hbase.master.handler;
021
022import org.apache.commons.logging.Log;
023import org.apache.commons.logging.LogFactory;
024import org.apache.hadoop.hbase.HRegionInfo;
025import org.apache.hadoop.hbase.Server;
026import org.apache.hadoop.hbase.executor.EventHandler;
027import org.apache.hadoop.hbase.master.AssignmentManager;
028
029/**
030 * Handles CLOSED region event on Master.
031 * <p>
032 * If table is being disabled, deletes ZK unassigned node and removes from
033 * regions in transition.
034 * <p>
035 * Otherwise, assigns the region to another server.
036 */
037public class ClosedRegionHandler extends EventHandler implements TotesHRegionInfo {
038  private static final Log LOG = LogFactory.getLog(ClosedRegionHandler.class);
039  private final AssignmentManager assignmentManager;
040  private final HRegionInfo regionInfo;
041  private final ClosedPriority priority;
042
043  private enum ClosedPriority {
044    ROOT (1),
045    META (2),
046    USER (3);
047
048    private final int value;
049    ClosedPriority(int value) {
050      this.value = value;
051    }
052    public int getValue() {
053      return value;
054    }
055  };
056
057  public ClosedRegionHandler(Server server, AssignmentManager assignmentManager,
058      HRegionInfo regionInfo) {
059    super(server, EventType.RS_ZK_REGION_CLOSED);
060    this.assignmentManager = assignmentManager;
061    this.regionInfo = regionInfo;
062    if(regionInfo.isRootRegion()) {
063      priority = ClosedPriority.ROOT;
064    } else if(regionInfo.isMetaRegion()) {
065      priority = ClosedPriority.META;
066    } else {
067      priority = ClosedPriority.USER;
068    }
069  }
070
071  @Override
072  public int getPriority() {
073    return priority.getValue();
074  }
075
076  @Override
077  public HRegionInfo getHRegionInfo() {
078    return this.regionInfo;
079  }
080
081  @Override
082  public String toString() {
083    String name = "UnknownServerName";
084    if(server != null && server.getServerName() != null) {
085      name = server.getServerName().toString();
086    }
087    return getClass().getSimpleName() + "-" + name + "-" + getSeqid();
088  }
089  
090  @Override
091  public void process() {
092    LOG.debug("Handling CLOSED event for " + regionInfo.getEncodedName());
093    // Check if this table is being disabled or not
094    if (this.assignmentManager.getZKTable().
095        isDisablingOrDisabledTable(this.regionInfo.getTableNameAsString())) {
096      assignmentManager.offlineDisabledRegion(regionInfo);
097      return;
098    }
099    // ZK Node is in CLOSED state, assign it.
100    // TODO: Should we remove the region from RIT too?  We don't?  Makes for
101    // a 'forcing' log message when we go to update state from CLOSED to OFFLINE
102    assignmentManager.setOffline(regionInfo);
103    // This below has to do w/ online enable/disable of a table
104    assignmentManager.removeClosedRegion(regionInfo);
105    assignmentManager.assign(regionInfo, true);
106  }
107}
