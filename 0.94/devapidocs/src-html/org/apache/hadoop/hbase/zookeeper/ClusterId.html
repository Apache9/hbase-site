021package org.apache.hadoop.hbase.zookeeper;
022
023import org.apache.hadoop.hbase.Abortable;
024import org.apache.hadoop.hbase.util.Bytes;
025import org.apache.zookeeper.KeeperException;
026
027/**
028 * Publishes and synchronizes a unique identifier specific to a given HBase
029 * cluster.  The stored identifier is read from the file system by the active
030 * master on startup, and is subsequently available to all watchers (including
031 * clients).
032 */
033public class ClusterId {
034  private ZooKeeperWatcher watcher;
035  private Abortable abortable;
036  private String id;
037
038  public ClusterId(ZooKeeperWatcher watcher, Abortable abortable) {
039    this.watcher = watcher;
040    this.abortable = abortable;
041  }
042
043  public boolean hasId() {
044    return getId() != null;
045  }
046
047  public String getId() {
048    try {
049      if (id == null) {
050        id = readClusterIdZNode(watcher);
051      }
052    } catch (KeeperException ke) {
053      abortable.abort("Unexpected exception from ZooKeeper reading cluster ID",
054          ke);
055    }
056    return id;
057  }
058
059  public static String readClusterIdZNode(ZooKeeperWatcher watcher)
060      throws KeeperException {
061    if (ZKUtil.checkExists(watcher, watcher.clusterIdZNode) != -1) {
062      byte[] data = ZKUtil.getData(watcher, watcher.clusterIdZNode);
063      if (data != null) {
064        return Bytes.toString(data);
065      }
066    }
067    return null;
068  }
069
070  public static void setClusterId(ZooKeeperWatcher watcher, String id)
071      throws KeeperException {
072    ZKUtil.createSetData(watcher, watcher.clusterIdZNode, Bytes.toBytes(id));
073  }
074}
