020package org.apache.hadoop.hbase;
021
022import java.io.IOException;
023import java.lang.reflect.Constructor;
024import java.lang.reflect.InvocationTargetException;
025
026import org.apache.hadoop.ipc.RemoteException;
027
028/**
029 * An immutable class which contains a static method for handling
030 * org.apache.hadoop.ipc.RemoteException exceptions.
031 */
032public class RemoteExceptionHandler {
033  /* Not instantiable */
034  private RemoteExceptionHandler() {super();}
035
036  /**
037   * Examine passed Throwable.  See if its carrying a RemoteException. If so,
038   * run {@link #decodeRemoteException(RemoteException)} on it.  Otherwise,
039   * pass back <code>t</code> unaltered.
040   * @param t Throwable to examine.
041   * @return Decoded RemoteException carried by <code>t</code> or
042   * <code>t</code> unaltered.
043   */
044  public static Throwable checkThrowable(final Throwable t) {
045    Throwable result = t;
046    if (t instanceof RemoteException) {
047      try {
048        result =
049          RemoteExceptionHandler.decodeRemoteException((RemoteException)t);
050      } catch (Throwable tt) {
051        result = tt;
052      }
053    }
054    return result;
055  }
056
057  /**
058   * Examine passed IOException.  See if its carrying a RemoteException. If so,
059   * run {@link #decodeRemoteException(RemoteException)} on it.  Otherwise,
060   * pass back <code>e</code> unaltered.
061   * @param e Exception to examine.
062   * @return Decoded RemoteException carried by <code>e</code> or
063   * <code>e</code> unaltered.
064   */
065  public static IOException checkIOException(final IOException e) {
066    Throwable t = checkThrowable(e);
067    return t instanceof IOException? (IOException)t: new IOException(t);
068  }
069
070  /**
071   * Converts org.apache.hadoop.ipc.RemoteException into original exception,
072   * if possible. If the original exception is an Error or a RuntimeException,
073   * throws the original exception.
074   *
075   * @param re original exception
076   * @return decoded RemoteException if it is an instance of or a subclass of
077   *         IOException, or the original RemoteException if it cannot be decoded.
078   *
079   * @throws IOException indicating a server error ocurred if the decoded
080   *         exception is not an IOException. The decoded exception is set as
081   *         the cause.
082   * @deprecated Use {@link RemoteException#unwrapRemoteException()} instead.
083   * In fact we should look into deprecating this whole class - St.Ack 2010929
084   */
085  public static IOException decodeRemoteException(final RemoteException re)
086  throws IOException {
087    IOException i = re;
088
089    try {
090      Class<?> c = Class.forName(re.getClassName());
091
092      Class<?>[] parameterTypes = { String.class };
093      Constructor<?> ctor = c.getConstructor(parameterTypes);
094
095      Object[] arguments = { re.getMessage() };
096      Throwable t = (Throwable) ctor.newInstance(arguments);
097
098      if (t instanceof IOException) {
099        i = (IOException) t;
100
101      } else {
102        i = new IOException("server error");
103        i.initCause(t);
104        throw i;
105      }
106
107    } catch (ClassNotFoundException x) {
108      // continue
109    } catch (NoSuchMethodException x) {
110      // continue
111    } catch (IllegalAccessException x) {
112      // continue
113    } catch (InvocationTargetException x) {
114      // continue
115    } catch (InstantiationException x) {
116      // continue
117    }
118    return i;
119  }
120}
